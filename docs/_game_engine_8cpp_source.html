135void DeepCRCSanityCheck::reset(void)
136{
137 static Int timesThrough = 0;
138 static UnsignedInt lastCRC = 0;
139
140 AsciiString fname;
141 fname.format("%sCRCAfter%dMaps.dat", TheGlobalData->getPath_UserData().str(), timesThrough);
142 UnsignedInt thisCRC = TheGameLogic->getCRC( CRC_RECALC, fname );
143
144 DEBUG_LOG(("DeepCRCSanityCheck: CRC is %X\n", thisCRC));
145 DEBUG_ASSERTCRASH(timesThrough == 0 || thisCRC == lastCRC,
146 ("CRC after reset did not match beginning CRC!\nNetwork games won't work after this.\nOld: 0x%8.8X, New: 0x%8.8X",
147 lastCRC, thisCRC));
148 lastCRC = thisCRC;
149
150 timesThrough++;
151}
152#endif // DEBUG_CRC

937#define CONVERT_EXEC1 "..\\Build\\nvdxt -list buildDDS.txt -dxt5 -full -outdir Art\\Textures > buildDDS.out"
938
939void updateTGAtoDDS()
940{
941 // Here's the scoop. We're going to traverse through all of the files in the Art\Textures folder
942 // and determine if there are any .tga files that are newer than associated .dds files. If there
943 // are, then we will re-run the compression tool on them.
944
945 File *fp = TheLocalFileSystem->openFile("buildDDS.txt", File::WRITE | File::CREATE | File::TRUNCATE | File::TEXT);
946 if (!fp) {
947 return;
948 }
949
950 FilenameList files;
951 TheLocalFileSystem->getFileListInDirectory("Art\\Textures\\", "", "*.tga", files, TRUE);
952 FilenameList::iterator it;
953 for (it = files.begin(); it != files.end(); ++it) {
954 AsciiString filenameTGA = *it;
955 AsciiString filenameDDS = *it;
956 FileInfo infoTGA;
957 TheLocalFileSystem->getFileInfo(filenameTGA, &infoTGA);
958
959 // skip the water textures, since they need to be NOT compressed
960 filenameTGA.toLower();
961 if (strstr(filenameTGA.str(), "caust"))
962 {
963 continue;
964 }
965 // and the recolored stuff.
966 if (strstr(filenameTGA.str(), "zhca"))
967 {
968 continue;
969 }
970
971 // replace tga with dds
972 filenameDDS.removeLastChar(); // a
973 filenameDDS.removeLastChar(); // g
974 filenameDDS.removeLastChar(); // t
975 filenameDDS.concat("dds");
976
977 Bool needsToBeUpdated = FALSE;
978 FileInfo infoDDS;
979 if (TheFileSystem->doesFileExist(filenameDDS.str())) {
980 TheFileSystem->getFileInfo(filenameDDS, &infoDDS);
981 if (infoTGA.timestampHigh > infoDDS.timestampHigh ||
982 (infoTGA.timestampHigh == infoDDS.timestampHigh &&
983 infoTGA.timestampLow > infoDDS.timestampLow)) {
984 needsToBeUpdated = TRUE;
985 }
986 } else {
987 needsToBeUpdated = TRUE;
988 }
989
990 if (!needsToBeUpdated) {
991 continue;
992 }
993
994 filenameTGA.concat("\n");
995 fp->write(filenameTGA.str(), filenameTGA.getLength());
996 }
997
998 fp->close();
999
1000 system(CONVERT_EXEC1);
1001}
