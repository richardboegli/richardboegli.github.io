55inline char* skipSeps(char* p, const char* seps)
56{
57 while (*p && strchr(seps, *p) != NULL)

120void AsciiString::ensureUniqueBufferOfSize(int numCharsNeeded, Bool preserveData, const char* strToCopy, const char* strToCat)
121{
122 validate();
123
124 if (m_data &&
125 m_data->m_refCount == 1 &&
126 m_data->m_numCharsAllocated >= numCharsNeeded)
127 {
128 // no buffer manhandling is needed (it's already large enough, and unique to us)
129 if (strToCopy)
130 strcpy(m_data->peek(), strToCopy);
131 if (strToCat)
132 strcat(m_data->peek(), strToCat);
133 return;
134 }
135
136 int minBytes = sizeof(AsciiStringData) + numCharsNeeded*sizeof(char);
137 if (minBytes > MAX_LEN)
138 throw ERROR_OUT_OF_MEMORY;
139
140 int actualBytes = TheDynamicMemoryAllocator->getActualAllocationSize(minBytes);
141 AsciiStringData* newData = (AsciiStringData*)TheDynamicMemoryAllocator->allocateBytesDoNotZero(actualBytes, "STR_AsciiString::ensureUniqueBufferOfSize");
142 newData->m_refCount = 1;
143 newData->m_numCharsAllocated = (actualBytes - sizeof(AsciiStringData))/sizeof(char);
144#if defined(_DEBUG) || defined(_INTERNAL)
145 newData->m_debugptr = newData->peek(); // just makes it easier to read in the debugger
146#endif
147
148 if (m_data && preserveData)
149 strcpy(newData->peek(), m_data->peek());
150 else
151 newData->peek()[0] = 0;
152
153 // do these BEFORE releasing the old buffer, so that self-copies
154 // or self-cats will work correctly.
155 if (strToCopy)
156 strcpy(newData->peek(), strToCopy);
157 if (strToCat)
158 strcat(newData->peek(), strToCat);
159
160 releaseBuffer();
161 m_data = newData;
162
163 validate();
164}

189// -----------------------------------------------------
190void AsciiString::trim()
191{
192 validate();
193
194 if (m_data)
195 {
196 char *c = peek();
197
198 // Strip leading white space from the string.
199 c = skipWhitespace(c);
200 if (c != peek())
201 {
202 set(c);
203 }
204
205 if (m_data) // another check, because the previous set() could erase m_data
206 {
207 // Clip trailing white space from the string.
208 int len = strlen(peek());
209 for (int index = len-1; index >= 0; index--)
210 {
211 if (isspace(getCharAt(index)))
212 {
213 removeLastChar();
214 }
215 else
216 {
217 break;
218 }
219 }
220 }
221 }
222 validate();
223}

226void AsciiString::toLower()
227{
228 validate();
229 if (m_data)
230 {
231 char buf[MAX_FORMAT_BUF_LEN];
232 strcpy(buf, peek());
233
234 char *c = buf;
235 while (c && *c)
236 {
237 *c = tolower(*c);
238 c++;
239 }
240 set(buf);
241 }
242 validate();
243}

246void AsciiString::removeLastChar()
247{
248 validate();
249 if (m_data)
250 {
251 int len = strlen(peek());
252 if (len > 0)
253 {
254 ensureUniqueBufferOfSize(len+1, true, NULL, NULL);
255 peek()[len - 1] = 0;
256 }
257 }
258 validate();
259}

306Bool AsciiString::startsWith(const char* p) const
307{
308 if (*p == 0)
309 return true; // everything starts with the empty string
310
311 int lenThis = getLength();
312 int lenThat = strlen(p);
313 if (lenThis < lenThat)
314 return false; // that must be smaller than this
315
316 return strncmp(peek(), p, lenThat) == 0;

368Bool AsciiString::nextToken(AsciiString* tok, const char* seps)
369{
370 if (this->isEmpty() || tok == this)
371 return false;
372
373 if (seps == NULL)
374 seps = " \n\r\t";
375
376 char* start = skipSeps(peek(), seps);
377 char* end = skipNonSeps(start, seps);
378
379 if (end > start)
380 {
381 Int len = end - start;
382 char* tmp = tok->getBufferForRead(len + 1);
383 memcpy(tmp, start, len);
384 tmp[len] = 0;
385
386 this->set(end);
387
388 return true;
389 }
390 else
391 {
392 this->clear();
393 tok->clear();
394 return false;
395 }
396}
