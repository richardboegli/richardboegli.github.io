83// drawButtonText =============================================================
85//=============================================================================
86static void drawButtonText( GameWindow *window, WinInstanceData *instData )
87{
88 ICoord2D origin, size, textPos;
89 Int width, height;
90 Color textColor, dropColor;
91 DisplayString *text = instData->getTextDisplayString();
92
93 // sanity
94 if( text == NULL || text->getTextLength() == 0 )
95 return;
96
97 // get window position and size
98 window->winGetScreenPosition( &origin.x, &origin.y );
99 window->winGetSize( &size.x, &size.y );
100
101 // set whether or not we center the wrapped text
102 text->setWordWrapCentered( BitTest( instData->getStatus(), WIN_STATUS_WRAP_CENTERED ));
103 text->setWordWrap(size.x);
104 // get the right text color
105 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
106 {
107 textColor = window->winGetDisabledTextColor();
108 dropColor = window->winGetDisabledTextBorderColor();
109 } // end if, disabled
110 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
111 {
112 textColor = window->winGetHiliteTextColor();
113 dropColor = window->winGetHiliteTextBorderColor();
114 } // end else if, hilited
115 else
116 {
117 textColor = window->winGetEnabledTextColor();
118 dropColor = window->winGetEnabledTextBorderColor();
119 } // end enabled only
120
121 // set our font to that of our parent if not the same
122 if( text->getFont() != window->winGetFont() )
123 text->setFont( window->winGetFont() );
124
125 // get text size
126 text->getSize( &width, &height );
127
128 // where to draw
129 if( BitTest( window->winGetStatus(), WIN_STATUS_SHORTCUT_BUTTON ) )
130 {
131 // Oh god... this is a total hack for shortcut buttons to handle rendering text top left corner...
132 textPos.x = origin.x + 2;
133 textPos.y = origin.y + 0;
134 }
135 else
136 {
137 textPos.x = origin.x + (size.x / 2) - (width / 2);
138 textPos.y = origin.y + (size.y / 2) - (height / 2);
139 }
140
141 // draw it
142 text->draw( textPos.x, textPos.y, textColor, dropColor );
143
144} // end drawButtonText

150// W3DGadgetPushButtonDraw ====================================================
152//=============================================================================
153void W3DGadgetPushButtonDraw( GameWindow *window, WinInstanceData *instData )
154{
155 Color color, border;
156 ICoord2D origin, size, start, end;
157
158 // get window position and size
159 window->winGetScreenPosition( &origin.x, &origin.y );
160 window->winGetSize( &size.x, &size.y );
161
162 //
163 // get pointer to image we want to draw depending on our state,
164 // see GadgetPushButton.h for info
165 //
166 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
167 {
168
169 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
170 {
171 color = GadgetButtonGetDisabledSelectedColor( window );
172 border = GadgetButtonGetDisabledSelectedBorderColor( window );
173 }
174 else
175 {
176 color = GadgetButtonGetDisabledColor( window );
177 border = GadgetButtonGetDisabledBorderColor( window );
178 }
179
180 } // end if, disabled
181 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
182 {
183
184 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
185 {
186 color = GadgetButtonGetHiliteSelectedColor( window );
187 border = GadgetButtonGetHiliteSelectedBorderColor( window );
188 }
189 else
190 {
191 color = GadgetButtonGetHiliteColor( window );
192 border = GadgetButtonGetHiliteBorderColor( window );
193 }
194
195 } // end else if, hilited and enabled
196 else
197 {
198
199 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
200 {
201 color = GadgetButtonGetEnabledSelectedColor( window );
202 border = GadgetButtonGetEnabledSelectedBorderColor( window );
203 }
204 else
205 {
206 color = GadgetButtonGetEnabledColor( window );
207 border = GadgetButtonGetEnabledBorderColor( window );
208 }
209
210 } // end else, enabled only
211
212 // compute draw position
213 start.x = origin.x;
214 start.y = origin.y;
215 end.x = start.x + size.x;
216 end.y = start.y + size.y;
217
218 // box and border
219 if( border != WIN_COLOR_UNDEFINED )
220 {
221
222 TheWindowManager->winOpenRect( border, WIN_DRAW_LINE_WIDTH,
223 start.x, start.y, end.x, end.y );
224
225 } // end if
226
227 if( color != WIN_COLOR_UNDEFINED )
228 {
229
230 // draw inside border
231 start.x++;
232 start.y++;
233 end.x--;
234 end.y--;
235 TheWindowManager->winFillRect( color, WIN_DRAW_LINE_WIDTH,
236 start.x, start.y, end.x, end.y );
237
238 } // end if
239
240 // draw the button text
241 if( instData->getTextLength() )
242 drawButtonText( window, instData );
243
244 // if we have a video buffer, draw the video buffer
245 if ( instData->m_videoBuffer )
246 {
247 TheDisplay->drawVideoBuffer( instData->m_videoBuffer, origin.x, origin.y, origin.x + size.x, origin.y + size.y );
248 }
249
250 PushButtonData *pData = (PushButtonData *)window->winGetUserData();
251 if( pData )
252 {
253 if( pData->overlayImage )
254 {
255 //Render the overlay image now.
256 TheDisplay->drawImage( pData->overlayImage, origin.x, origin.y, origin.x + size.x, origin.y + size.y );
257 }
258
259 if( pData->drawClock )
260 {
261 if( pData->drawClock == NORMAL_CLOCK )
262 {
263 TheDisplay->drawRectClock(origin.x, origin.y, size.x, size.y, pData->percentClock,pData->colorClock);
264 }
265 else if( pData->drawClock == INVERSE_CLOCK )
266 {
267 TheDisplay->drawRemainingRectClock( origin.x, origin.y, size.x, size.y, pData->percentClock,pData->colorClock );
268 }
269 pData->drawClock = NO_CLOCK;
270 window->winSetUserData(pData);
271 }
272
273 if( pData->drawBorder && pData->colorBorder != GAME_COLOR_UNDEFINED )
274 {
275 TheDisplay->drawOpenRect(origin.x -1, origin.y - 1, size.x + 2, size.y + 2,1 , pData->colorBorder);
276 }
277 }
278
279} // end W3DGadgetPushButtonDraw

318void W3DGadgetPushButtonImageDrawOne( GameWindow *window,
319 WinInstanceData *instData )
320{
321 const Image *image = NULL;
322 ICoord2D size, start, end;
323
324 //
325 // get pointer to image we want to draw depending on our state,
326 // see GadgetPushButton.h for info
327 //
328 image = GadgetButtonGetEnabledImage( window );
329
330 if( !BitTest( window->winGetStatus(), WIN_STATUS_USE_OVERLAY_STATES ) )
331 {
332 //Certain buttons have the option to specify specific images for
333 //altered states. If they do, then we won't render the auto-overlay versions.
334 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
335 {
336
337 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
338 image = GadgetButtonGetDisabledSelectedImage( window );
339 else
340 image = GadgetButtonGetDisabledImage( window );
341
342 } // end if, disabled
343 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
344 {
345
346 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
347 image = GadgetButtonGetHiliteSelectedImage( window );
348 else
349 image = GadgetButtonGetHiliteImage( window );
350
351 } // end else if, hilited and enabled
352 else
353 {
354
355 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
356 image = GadgetButtonGetHiliteSelectedImage( window );
357 } // end else, enabled only
358 }
359
360
361 // draw the image
362 if( image )
363 {
364
365 // get window position
366 window->winGetScreenPosition( &start.x, &start.y );
367 window->winGetSize( &size.x, &size.y );
368
369
370 // offset position by image offset
371 start.x += instData->m_imageOffset.x;
372 start.y += instData->m_imageOffset.y;
373
374 // find end point
375 end.x = start.x + size.x;
376 end.y = start.y + size.y;
377
378 Display::DrawImageMode drawMode=Display::DRAW_IMAGE_ALPHA;
379 Int colorMultiplier = 0xffffffff;
380
381 if(BitTest( window->winGetStatus(), WIN_STATUS_USE_OVERLAY_STATES ) )
382 {
383 //we're using a new drawing system which does "grayscale" disabled buttons using original color artwork.
384 if( !BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) )
385 {
386 if( !BitTest( window->winGetStatus(), WIN_STATUS_NOT_READY ) )
387 {
388 //The button is disabled -- but if the button isn't "ready", we don't want to do this because
389 //we want to show the button in color with just the clock overlay.
390 if( !BitTest( window->winGetStatus(), WIN_STATUS_ALWAYS_COLOR ) )
391 {
392 drawMode=Display::DRAW_IMAGE_GRAYSCALE;
393 }
394 else
395 {
396 colorMultiplier = 0xff909090; //RGB values are 144/255 (90) -- Alpha is opaque (ff) --> ff909090;
397 }
398 }
399 }
400 }
401 TheDisplay->drawImage( image, start.x, start.y, end.x, end.y, colorMultiplier, drawMode );
402 } // end if
403
404 // draw the button text
405 if( instData->getTextLength() )
406 drawButtonText( window, instData );
407
408 // get window position
409 window->winGetScreenPosition( &start.x, &start.y );
410 window->winGetSize( &size.x, &size.y );
411
412
413 // if we have a video buffer, draw the video buffer
414 if ( instData->m_videoBuffer )
415 {
416 TheDisplay->drawVideoBuffer( instData->m_videoBuffer, start.x, start.y, start.x + size.x, start.y + size.y );
417 }
418 PushButtonData *pData = (PushButtonData *)window->winGetUserData();
419
420 if( pData )
421 {
422 if( pData->overlayImage )
423 {
424 //Render the overlay image now.
425 TheDisplay->drawImage( pData->overlayImage, start.x, start.y, start.x + size.x, start.y + size.y );
426 }
427
428 if( pData->drawClock )
429 {
430 if( pData->drawClock == NORMAL_CLOCK )
431 {
432 TheDisplay->drawRectClock(start.x, start.y, size.x, size.y, pData->percentClock,pData->colorClock);
433 }
434 else if( pData->drawClock == INVERSE_CLOCK )
435 {
436 TheDisplay->drawRemainingRectClock( start.x, start.y, size.x, size.y, pData->percentClock,pData->colorClock );
437 }
438 pData->drawClock = NO_CLOCK;
439 window->winSetUserData(pData);
440 }
441
442 if( pData->drawBorder && pData->colorBorder != GAME_COLOR_UNDEFINED )
443 {
444
445 TheDisplay->drawOpenRect(start.x - 1, start.y - 1, size.x + 2, size.y + 2, 1, pData->colorBorder);
446
447 }
448 }
449
450 //Now render overlays that pertain to the correct state.
451
452 if( BitTest( window->winGetStatus(), WIN_STATUS_FLASHING ) )
453 {
454 //Handle cameo flashing (let the flashing stack with overlay states)
455 static const Image *hilitedOverlayIcon = TheMappedImageCollection->findImageByName( "Cameo_push" );
456 TheDisplay->drawImage( hilitedOverlayIcon, start.x, start.y, start.x + size.x, start.y + size.y );
457 }
458
459 if( BitTest( window->winGetStatus(), WIN_STATUS_USE_OVERLAY_STATES ) )
460 {
461 image = NULL;
462 static const Image *pushedOverlayIcon = TheMappedImageCollection->findImageByName( "Cameo_push" );
463 static const Image *hilitedOverlayIcon = TheMappedImageCollection->findImageByName( "Cameo_hilited" );
464 if( pushedOverlayIcon && hilitedOverlayIcon )
465 {
466 if(BitTest(window->winGetStatus(), WIN_STATUS_ENABLED))
467 {
468 if (BitTest( instData->getState(), WIN_STATE_HILITED ))
469 {
470 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
471 {
472 //The button is hilited and pushed
473 TheDisplay->drawImage( pushedOverlayIcon, start.x, start.y, start.x + size.x, start.y + size.y );
474 }
475 else
476 {
477 //The button is hilited
478 TheDisplay->drawImage( hilitedOverlayIcon, start.x, start.y, start.x + size.x, start.y + size.y );
479 }
480 }
481 else if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
482 {
483 //The button appears to be pushed -- CHECK_LIKE buttons that are on.
484 TheDisplay->drawImage( pushedOverlayIcon, start.x, start.y, start.x + size.x, start.y + size.y );
485 }
486 }
487 }
488 }
489} // end W3DGadgetPushButtonImageDraw

492void W3DGadgetPushButtonImageDrawThree(GameWindow *window, WinInstanceData *instData )
493{
494
495 const Image *leftImage, *rightImage, *centerImage;
496 ICoord2D origin, size, start, end;
497 Int xOffset, yOffset;
498 Int i;
499
500 // get screen position and size
501 window->winGetScreenPosition( &origin.x, &origin.y );
502 window->winGetSize( &size.x, &size.y );
503
504 // get image offset
505 xOffset = instData->m_imageOffset.x;
506 yOffset = instData->m_imageOffset.y;
507
508
509 //
510 // get pointer to image we want to draw depending on our state,
511 // see GadgetPushButton.h for info
512 //
513 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
514 {
515
516 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
517 {
518 leftImage = GadgetButtonGetLeftDisabledSelectedImage( window );
519 rightImage = GadgetButtonGetRightDisabledSelectedImage( window );
520 centerImage = GadgetButtonGetMiddleDisabledSelectedImage( window );
521 }
522 else
523 {
524
525 leftImage = GadgetButtonGetLeftDisabledImage( window );
526 rightImage = GadgetButtonGetRightDisabledImage( window );
527 centerImage = GadgetButtonGetMiddleDisabledImage( window );
528
529 }
530
531 } // end if, disabled
532 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
533 {
534
535 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
536 {
537 leftImage = GadgetButtonGetLeftHiliteSelectedImage( window );
538 rightImage = GadgetButtonGetRightHiliteSelectedImage( window );
539 centerImage = GadgetButtonGetMiddleHiliteSelectedImage( window );
540 }
541 else
542 {
543
544 leftImage = GadgetButtonGetLeftHiliteImage( window );
545 rightImage = GadgetButtonGetRightHiliteImage( window );
546 centerImage = GadgetButtonGetMiddleHiliteImage( window );
547
548 }
549
550 } // end else if, hilited and enabled
551 else
552 {
553
554 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
555 {
556 leftImage = GadgetButtonGetLeftEnabledSelectedImage( window );
557 rightImage = GadgetButtonGetRightEnabledSelectedImage( window );
558 centerImage = GadgetButtonGetMiddleEnabledSelectedImage( window );
559 }
560 else
561 {
562
563 leftImage = GadgetButtonGetLeftEnabledImage( window );
564 rightImage = GadgetButtonGetRightEnabledImage( window );
565 centerImage = GadgetButtonGetMiddleEnabledImage( window );
566
567 }
568
569 } // end else, enabled only
570
571 // sanity, we need to have these images to make it look right
572 if( leftImage == NULL || rightImage == NULL ||
573 centerImage == NULL )
574 return;
575
576 // get image sizes for the ends
577 ICoord2D leftSize, rightSize;
578 leftSize.x = leftImage->getImageWidth();
579 leftSize.y = leftImage->getImageHeight();
580 rightSize.x = rightImage->getImageWidth();
581 rightSize.y = rightImage->getImageHeight();
582
583 // get two key points used in the end drawing
584 ICoord2D leftEnd, rightStart;
585 leftEnd.x = origin.x + leftSize.x + xOffset;
586 leftEnd.y = origin.y + size.y + yOffset;
587 rightStart.x = origin.x + size.x - rightSize.x + xOffset;
588 rightStart.y = origin.y + yOffset;
589
590 // draw the center repeating bar
591 Int centerWidth, pieces;
592
593 // get width we have to draw our repeating center in
594 centerWidth = rightStart.x - leftEnd.x;
595
596 if( centerWidth <= 0)
597 {
598 // draw left end
599 start.x = origin.x + xOffset;
600 start.y = origin.y + yOffset;
601 end.y = leftEnd.y;
602 end.x = origin.x + xOffset + size.x/2;
603 TheWindowManager->winDrawImage(leftImage, start.x, start.y, end.x, end.y);
604
605 // draw right end
606 start.y = rightStart.y;
607 start.x = end.x;
608 end.x = origin.x + size.x;
609 end.y = start.y + size.y;
610 TheWindowManager->winDrawImage(rightImage, start.x, start.y, end.x, end.y);
611 }
612 else
613 {
614
615 // how many whole repeating pieces will fit in that width
616 pieces = centerWidth / centerImage->getImageWidth();
617
618 // draw the pieces
619 start.x = leftEnd.x;
620 start.y = origin.y + yOffset;
621 end.y = start.y + size.y + yOffset; //centerImage->getImageHeight() + yOffset;
622 for( i = 0; i < pieces; i++ )
623 {
624
625 end.x = start.x + centerImage->getImageWidth();
626 TheWindowManager->winDrawImage( centerImage,
627 start.x, start.y,
628 end.x, end.y );
629 start.x += centerImage->getImageWidth();
630
631 } // end for i
632
633 // we will draw the image but clip the parts we don't want to show
634 IRegion2D reg;
635 reg.lo.x = start.x;
636 reg.lo.y = start.y;
637 reg.hi.x = rightStart.x;
638 reg.hi.y = end.y;
639 centerWidth = rightStart.x - start.x;
640 if( centerWidth > 0)
641 {
642 TheDisplay->setClipRegion(&reg);
643 end.x = start.x + centerImage->getImageWidth();
644 TheWindowManager->winDrawImage( centerImage,
645 start.x, start.y,
646 end.x, end.y );
647 TheDisplay->enableClipping(FALSE);
648 }
649
650 // draw left end
651 start.x = origin.x + xOffset;
652 start.y = origin.y + yOffset;
653 end = leftEnd;
654 TheWindowManager->winDrawImage(leftImage, start.x, start.y, end.x, end.y);
655
656 // draw right end
657 start = rightStart;
658 end.x = start.x + rightSize.x;
659 end.y = start.y + size.y;
660 TheWindowManager->winDrawImage(rightImage, start.x, start.y, end.x, end.y);
661 }
662
663 // draw the button text
664 if( instData->getTextLength() )
665 drawButtonText( window, instData );
666
667 // get window position
668 window->winGetScreenPosition( &start.x, &start.y );
669 window->winGetSize( &size.x, &size.y );
670
671
672 // if we have a video buffer, draw the video buffer
673 if ( instData->m_videoBuffer )
674 {
675 TheDisplay->drawVideoBuffer( instData->m_videoBuffer, start.x, start.y, start.x + size.x, start.y + size.y );
676 }
677 PushButtonData *pData = (PushButtonData *)window->winGetUserData();
678
679 if( pData )
680 {
681 if( pData->overlayImage )
682 {
