79WindowMsgHandledType GadgetHorizontalSliderInput( GameWindow *window, UnsignedInt msg,
80 WindowMsgData mData1, WindowMsgData mData2 )
81{
82 SliderData *s = (SliderData *)window->winGetUserData();
83 WinInstanceData *instData = window->winGetInstanceData();
84 ICoord2D size, childSize, childCenter;
85 window->winGetSize( &size.x, &size.y );
86 switch( msg )
87 {
88 // ------------------------------------------------------------------------
89 case GWM_MOUSE_ENTERING:
90 {
91
92 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
93 {
94
95 BitSet( instData->m_state, WIN_STATE_HILITED );
96 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
97 GBM_MOUSE_ENTERING,
98 (WindowMsgData)window,
99 0 );
100 //TheWindowManager->winSetFocus( window );
101
102 } // end if
103
104 if(window->winGetChild() && BitTest(window->winGetChild()->winGetStyle(),GWS_PUSH_BUTTON) )
105 {
106 WinInstanceData *instDataChild = window->winGetChild()->winGetInstanceData();
107 BitSet(instDataChild->m_state, WIN_STATE_HILITED);
108 }
109
110 break;
111
112 } // end mouse entering
113
114 // ------------------------------------------------------------------------
115 case GWM_MOUSE_LEAVING:
116 {
117
118 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ))
119 {
120
121 BitClear( instData->m_state, WIN_STATE_HILITED );
122 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
123 GBM_MOUSE_LEAVING,
124 (WindowMsgData)window,
125 0 );
126 } // end if
127 if(window->winGetChild() && BitTest(window->winGetChild()->winGetStyle(),GWS_PUSH_BUTTON) )
128 {
129 WinInstanceData *instDataChild = window->winGetChild()->winGetInstanceData();
130 BitClear(instDataChild->m_state, WIN_STATE_HILITED);
131 }
132
133 break;
134
135 } // end mouse leaving
136
137 // ------------------------------------------------------------------------
138 case GWM_LEFT_DRAG:
139
140 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
141 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
142 GGM_LEFT_DRAG,
143 (WindowMsgData)window,
144 mData1 );
145 break;
146
147 // ------------------------------------------------------------------------
148 case GWM_LEFT_DOWN:
149 break;
150
151 // ------------------------------------------------------------------------
152 case GWM_LEFT_UP:
153 {
154 Int x, y;
155 Int mousex = mData1 & 0xFFFF;
156// Int mousey = mData1 >> 16;
157
158 GameWindow *child = window->winGetChild();
159 Int pageClickSize, clickPos;
160
161 window->winGetScreenPosition( &x, &y );
162
163 child->winGetSize( &childSize.x, &childSize.y );
164 child->winGetPosition( &childCenter.x, &childCenter.y );
165 childCenter.x += childSize.x / 2;
166 childCenter.y += childSize.y / 2;
167
168 //
169 // when you click on the slider, but not the button, we will jump
170 // the slider position up/down by this much
171 //
172 pageClickSize = size.x / 5;
173
174 clickPos = mousex - x;
175 if( clickPos >= childCenter.x )
176 {
177
178 clickPos = childCenter.x + pageClickSize;
179 if( clickPos > mousex - x )
180 clickPos = mousex - x;
181
182 } // end if
183 else
184 {
185
186 clickPos = childCenter.x - pageClickSize;
187 if( clickPos < mousex - x )
188 clickPos = mousex - x;
189
190 } // end else
191
192 // keep it all valid to the window
193 if( clickPos > x + size.x - childSize.x / 2 )
194 clickPos = x + size.y - childSize.x / 2;
195 if( clickPos < childSize.x / 2 )
196 clickPos = childSize.x / 2;
197
198 child->winSetPosition( clickPos - childSize.x / 2, HORIZONTAL_SLIDER_THUMB_POSITION);
199 TheWindowManager->winSendSystemMsg( window, GGM_LEFT_DRAG, 0, mData1 );
200 break;
201
202 } // end left up, left click
203
204 // ------------------------------------------------------------------------
205 case GWM_CHAR:
206 {
207
208 switch( mData1 )
209 {
210
211 // --------------------------------------------------------------------
212 case KEY_RIGHT:
213 if( BitTest( mData2, KEY_STATE_DOWN ) )
214 {
215
216 if( s->position > s->minVal + 1 )
217 {
218 GameWindow *child = window->winGetChild();
219
220 s->position -= 2;
221 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
222 GSM_SLIDER_TRACK,
223 (WindowMsgData)window,
224 s->position );
225
226 // Translate to window coords
227 child->winSetPosition( (Int)((s->position - s->minVal) * s->numTicks), HORIZONTAL_SLIDER_THUMB_POSITION );
228
229 } // end if
230
231 } // if key down
232
233 break;
234
235 // --------------------------------------------------------------------
236 case KEY_LEFT:
237
238 if( BitTest( mData2, KEY_STATE_DOWN ) )
239 {
240
241 if( s->position < s->maxVal - 1 )
242 {
243 GameWindow *child = window->winGetChild();
244
245 s->position += 2;
246 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
247 GSM_SLIDER_TRACK,
248 (WindowMsgData)window,
249 s->position );
250
251 // Translate to window coords
252 child->winSetPosition( (Int)((s->position - s->minVal) * s->numTicks),HORIZONTAL_SLIDER_THUMB_POSITION );
253
254 }
255
256 } // end if key down
257
258 break;
259
260 // --------------------------------------------------------------------
261 case KEY_DOWN:
262 case KEY_TAB:
263
264 if( BitTest( mData2, KEY_STATE_DOWN ) )
265 window->winNextTab();
266 break;
267
268 // --------------------------------------------------------------------
269 case KEY_UP:
270
271 if( BitTest( mData2, KEY_STATE_DOWN ) )
272 window->winPrevTab();
273 break;
274
275 // --------------------------------------------------------------------
276 default:
277 return MSG_IGNORED;
278
279 } // end switch( mData1 )
280
281 break;
282
283 } // end char
284
285 // ------------------------------------------------------------------------
286 default:
287 return MSG_IGNORED;
288
289 }
290
291 return MSG_HANDLED;
292
293} // end GadgetHorizontalSliderInput

298WindowMsgHandledType GadgetHorizontalSliderSystem( GameWindow *window, UnsignedInt msg,
299 WindowMsgData mData1, WindowMsgData mData2 )
300{
301 SliderData *s = (SliderData *)window->winGetUserData();
302 WinInstanceData *instData = window->winGetInstanceData();
303 ICoord2D size, childSize, childCenter,childRelativePos;
304 window->winGetSize( &size.x, &size.y );
305 switch( msg )
306 {
307 // ------------------------------------------------------------------------
308 case GGM_LEFT_DRAG:
309 {
310 Int mousex = mData2 & 0xFFFF;
311// Int mousey = mData2 >> 16;
312 Int x, y, delta;
313 GameWindow *child = window->winGetChild();
314
315
316 window->winGetScreenPosition( &x, &y );
317
318 child->winGetSize( &childSize.x, &childSize.y );
319 child->winGetScreenPosition( &childCenter.x, &childCenter.y );
320 child->winGetPosition(&childRelativePos.x, &childRelativePos.y);
321 childCenter.x += childSize.x / 2;
322 childCenter.y += childSize.y / 2;
323
324 //
325 // ignore drag attempts when the mouse is right or left of slider totally
326 // and put the dragging thumb back at the slider pos
327 //
328 if( mousex > x + size.x -HORIZONTAL_SLIDER_THUMB_WIDTH/2 )
329 {
330
331 TheWindowManager->winSendSystemMsg( window, GSM_SET_SLIDER,
332 s->maxVal, 0 );
333 // tell owner i moved
334 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
335 GSM_SLIDER_TRACK,
336 (WindowMsgData)window,
337 s->position );
338 break;
339
340 } // end if
341 else if( mousex < x + HORIZONTAL_SLIDER_THUMB_WIDTH/2)
342 {
343
344 TheWindowManager->winSendSystemMsg( window, GSM_SET_SLIDER,
345 s->minVal, 0 );
346 // tell owner i moved
347 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
348 GSM_SLIDER_TRACK,
349 (WindowMsgData)window,
350 s->position );
351 break;
352
353 } // end else if
354
355 if( childCenter.x < x + childSize.x / 2 )
356 {
357 child->winSetPosition( 0,HORIZONTAL_SLIDER_THUMB_POSITION );
358 s->position = s->minVal;
359
360 }
361 else if( childCenter.x >= x + size.x - childSize.x / 2 )
362 {
363 child->winSetPosition( (Int)((s->maxVal - s->minVal) * s->numTicks) -HORIZONTAL_SLIDER_THUMB_WIDTH/2 , HORIZONTAL_SLIDER_THUMB_POSITION );
364 s->position = s->maxVal;
365
366 }
367 else
368 {
369 delta = childCenter.x - x -HORIZONTAL_SLIDER_THUMB_WIDTH/2;
370
371 // Calc slider position
372 s->position = (Int)((delta) / s->numTicks)+ s->minVal ;
373
374 /*
375 s->position += s->minVal;
376 */
377
378 if( s->position > s->maxVal )
379 s->position = s->maxVal;
380 if( s->position < s->minVal)
381 s->position = s->minVal;
382
383 child->winSetPosition( childRelativePos.x, HORIZONTAL_SLIDER_THUMB_POSITION );
384 }
385
386 // tell owner i moved
387 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
388 GSM_SLIDER_TRACK,
389 (WindowMsgData)window,
390 s->position );
391
392 break;
393
394 } // end left drag
395
396 // ------------------------------------------------------------------------
397 case GSM_SET_SLIDER:
398 {
399 Int newPos = (Int)mData1;
400 GameWindow *child = window->winGetChild();
401
402 if( newPos < s->minVal || newPos > s->maxVal )
403 break;
404
405 s->position = newPos;
406
407 // Translate to window coords
408 newPos = (Int)((newPos - s->minVal) * s->numTicks);
409
410 child->winSetPosition( newPos , HORIZONTAL_SLIDER_THUMB_POSITION );
411 break;
412
413 } // end set slider
414
415 // ------------------------------------------------------------------------
416 case GSM_SET_MIN_MAX:
417 {
418 ICoord2D size;
419 GameWindow *child = window->winGetChild();
420
421 window->winGetSize( &size.x, &size.y );
422
423 s->minVal = (Int)mData1;
424 s->maxVal = (Int)mData2;
425 s->numTicks = (Real)(size.x - HORIZONTAL_SLIDER_THUMB_WIDTH)/(Real)(s->maxVal - s->minVal);
426 s->position = s->minVal;
427
428 child->winSetPosition( 0, HORIZONTAL_SLIDER_THUMB_POSITION );
429 break;
430
431 } // end set min max
432
433 // ------------------------------------------------------------------------
434 case GWM_CREATE:
435 break;
436
437 // ------------------------------------------------------------------------
438 case GWM_DESTROY:
439 delete ( (SliderData *)window->winGetUserData() );
440 break;
441
442 // ------------------------------------------------------------------------
443 case GWM_INPUT_FOCUS:
444 {
445
446 // If we're losing focus
447 if( mData1 == FALSE )
448 BitClear( instData->m_state, WIN_STATE_HILITED );
449 else
450 BitSet( instData->m_state, WIN_STATE_HILITED );
451
452 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
453 GGM_FOCUS_CHANGE,
454 mData1,
455 window->winGetWindowId() );
456
457 *(Bool*)mData2 = TRUE;
458 break;
459
460 } // end focus msg
461
462 // ------------------------------------------------------------------------
463 case GGM_RESIZED:
464 {
465// Int width = (Int)mData1;
466 Int height = (Int)mData2;
467 GameWindow *thumb = window->winGetChild();
468
469 if( thumb )
470 thumb->winSetSize( GADGET_SIZE, height );
471
472 break;
473
474 } // end resized
475
476 default:
477 return MSG_IGNORED;
478
479 } // end switch( msg )
480
481 return MSG_HANDLED;
482
483} // end GadgetHorizontalSliderSystem
