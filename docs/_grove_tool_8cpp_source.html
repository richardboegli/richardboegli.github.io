52Bool localIsUnderwater( Real x, Real y)
53{
54 ICoord3D iLoc;
55 iLoc.x = (floor(x+0.5f));
56 iLoc.y = (floor(y+0.5f));
57 iLoc.z = 0;
58 // Look for water areas.
59 for (PolygonTrigger *pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
60 if (!pTrig->isWaterArea()) {
61 continue;
62 }
63 // See if point is in a water area
64 if (pTrig->pointInTrigger(iLoc)) {
65 Real wZ = pTrig->getPoint(0)->z;
66 // See if the ground height is less than the water level.
67 Real curHeight = TheTerrainRenderObject->getHeightMapHeight(x, y, NULL);
68 return (curHeight<wZ);
69 }
70 }
71 return false;
72}

76// plant a tree
77void GroveTool::plantTree( Coord3D *pos )
78{
79 AsciiString treeName;
80 int totalValue = TheGroveOptions->getTotalTreePerc();
81 int randVal = GameLogicRandomValue(0, totalValue - 1);
82
83
84 int runningCum = 0;
85 for (int i = 1; i <= 5; ++i) {
86 runningCum += TheGroveOptions->getNumType(i);
87 if (randVal < runningCum) {
88 if (TheGroveOptions->getTypeName(i).isEmpty()) {
89 return;
90 }
91
92 treeName = TheGroveOptions->getTypeName(i).str();
93 break;
94 }
95 }
96
97 addObj(pos, treeName);
98}

114void GroveTool::_plantGroveInBox(CPoint tl, CPoint br, WbView* pView)
115{
116
117 int numTotalTrees = TheGroveOptions->getNumTrees();
118
119 Coord3D tl3d, tr3d, bl3d;
120 pView->viewToDocCoords(tl, &tl3d);
121 pView->viewToDocCoords(CPoint(tl.x, br.y), &bl3d);
122 pView->viewToDocCoords(CPoint(br.x, tl.y), &tr3d);
123
124 tl3d.z = 0;
125
126 bl3d.x -= tl3d.x;
127 bl3d.y -= tl3d.y;
128 bl3d.z = 0;
129
130 tr3d.x -= tl3d.x;
131 tr3d.y -= tl3d.y;
132 tr3d.z = 0;
133
134 for (int i = 0; i < numTotalTrees; ++i) {
135 Real trModifier = GameLogicRandomValueReal(0.0f, 1.0f);
136 Real blModifier = GameLogicRandomValueReal(0.0f, 1.0f);
137
138 Vector3 tlVec(tl3d.x, tl3d.y, tl3d.z);
139 Vector3 trVec(tr3d.x * trModifier, tr3d.y * trModifier, tr3d.z);
140 Vector3 blVec(bl3d.x * blModifier, bl3d.y * blModifier, bl3d.z);
141
142 tlVec = tlVec + trVec;
143 tlVec = tlVec + blVec;
144
145 Coord3D position;
146
147 position.x = tlVec.X;
148 position.y = tlVec.Y;
149 position.z = 0;
150
151 // (maybe) don't put trees on steep slopes
152 // (maybe) tree must not be in the water
153 if (!TheGroveOptions->getCanPlaceInWater() && localIsUnderwater(position.x, position.y)) {
154 continue;
155 }
156
157 if (!TheGroveOptions->getCanPlaceOnCliffs() && _positionIsTooCliffyForTrees(position)) {
158 continue;
159 }
160 // We've satisfied our conditions. Plant the little bugger
161 plantTree(&position);
162 }
163}

169void GroveTool::plantGrove( Coord3D pos, Coord3D prevDir, Real baseHeight, Int level, CPoint bounds )
170{
171 Coord3D childPos, normal, dir;
172 Real childChance = 1.0f;
173 Real heightTolerance = 1.5f*MAP_XY_FACTOR; // was 4
174 Real angle, spread;
175 Int numChildren = 2, retry, maxTries = 5;
176
177 if (level == 1)
178 {
179 // plant a shrub around the outskirts of the grove
180 plantShrub( &pos );
181 }
182 else
183 {
184 // plant a tree here
185 plantTree( &pos );
186 }
187
188 // if reached base level, stop
189 if (level == 0)
190 return;
191
192
193 // spawn child trees
194 for( Int i=0; i<numChildren; i++ )
195 {
196 if (GameLogicRandomValueReal( 0.0f, 1.0f ) < childChance)
197 {
198 for( retry=0; retry<maxTries; retry++ )
199 {
200 angle = GameLogicRandomValueReal( 0.0f, 2.0f * PI );
201 spread = GameLogicRandomValueReal( 3.0f*MAP_XY_FACTOR, 6.0f*MAP_XY_FACTOR );
202 childPos.x = pos.x + spread * (Real)cos( angle );
203 childPos.y = pos.y + spread * (Real)sin( angle );
204 childPos.z = 0;
205 if (TheTerrainRenderObject) {
206 TheTerrainRenderObject->getHeightMapHeight( childPos.x, childPos.y, &normal );
207 }
208
209 dir.x = childPos.x - pos.x;
210 dir.y = childPos.y - pos.y;
211
212 // don't select a spot too much towards where we came from
213 // (note that if prevDir the zero vector, it passes this test (for the 1st tree))
214 // don't put trees on steep slopes
215 // dont' plant if height changed too much
216 // tree must be on map
217 // tree must not be in the water
218 if (dir.x * prevDir.x + dir.y * prevDir.y >= 0.0f &&
219 normal.z > flatTolerance &&
220 fabs(childPos.z - baseHeight) < heightTolerance &&
221 childPos.x > 0 && childPos.y > 0 &&
222 childPos.x < bounds.x && childPos.y < bounds.y &&
223 (!localIsUnderwater(childPos.x, childPos.y)))
224 break;
225 }
226
227 if (retry < maxTries)
228 plantGrove( childPos, dir, baseHeight, level-1, bounds );
229 }
230 }
231}

261void GroveTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
262{
263 if (m != TRACK_L) return;
264
265 if (pView && m_dragging) {
266 CRect box;
267 box.left = viewPt.x;
268 box.top = viewPt.y;
269 box.right = m_downPt.x;
270 box.bottom = m_downPt.y;
271 box.NormalizeRect();
272 pView->doRectFeedback(false, box);
273 pView->Invalidate();
274
275 _plantGroveInBox(m_downPt, viewPt, pView);
276 if (m_headMapObj != NULL) {
277 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, m_headMapObj);
278 pDoc->AddAndDoUndoable(pUndo);
279 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
280 m_headMapObj = NULL; // undoable owns it now.
281 }
282 return;
283 }
284
285 Coord3D loc;
286
287 pView->viewToDocCoords(m_downPt, &loc);
288
289 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
290 CPoint bounds;
291 bounds.x = pMap->getXExtent()*MAP_XY_FACTOR;
292 bounds.y = pMap->getYExtent()*MAP_XY_FACTOR;
293// Real angle = 0;
294 Int depth = 3;
295 Coord3D zeroDir;
296
297 zeroDir.x = 0.0f;
298 zeroDir.y = 0.0f;
299 zeroDir.z = 0.0f;
300 loc.z = TheTerrainRenderObject ? TheTerrainRenderObject->getHeightMapHeight( loc.x, loc.y, NULL ) : 0;
301
302 // grow tree grove out from here
303 plantGrove( loc, zeroDir, loc.z, depth, bounds );
304 if (m_headMapObj != NULL) {
305 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, m_headMapObj);
306 pDoc->AddAndDoUndoable(pUndo);
307 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
308 m_headMapObj = NULL; // undoable owns it now.
309 }
310}

312void GroveTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
313{
314 if (m != TRACK_L) {
315 return;
316 }
317
318 if (abs(viewPt.x - m_downPt.x) > DRAG_THRESHOLD || abs(viewPt.y - m_downPt.y) > DRAG_THRESHOLD) {
319 m_dragging = true;
320 }
321
322 if (pView && m_dragging) {
323 CRect box;
324 box.left = viewPt.x;
325 box.bottom = viewPt.y;
326 box.top = m_downPt.y;
327 box.right = m_downPt.x;
328 box.NormalizeRect();
329 pView->doRectFeedback(true, box);
330 pView->Invalidate();
331 return;
332
333 }
