116UpdateSleepTime HeightDieUpdate::update( void )
117{
118 UnsignedInt now = TheGameLogic->getFrame();
119 if( m_earliestDeathFrame == UINT_MAX )
120 m_earliestDeathFrame = now + getHeightDieUpdateModuleData()->m_initialDelay;
121
122 // If at least a one frame delay has been set, then stop for a while
123 if( m_earliestDeathFrame > now )
124 return UPDATE_SLEEP_NONE;
125
126 // do nothing if we're contained within other objects ... like a transport
127 if( getObject()->getContainedBy() != NULL )
128 {
129
130 // keep track of our last position even though we're not doing anything yet
131 m_lastPosition = *getObject()->getPosition();
132
133 // get outta here
134 return UPDATE_SLEEP_NONE;
135
136 } // end if
137
138 // get the module data
139 const HeightDieUpdateModuleData *modData = getHeightDieUpdateModuleData();
140
141 // get our current position
142 const Coord3D *pos = getObject()->getPosition();
143
144 Bool directionOK = TRUE;
145 if( m_hasDied == FALSE )
146 {
147
148 if( modData->m_onlyWhenMovingDown )
149 {
150
151 if( pos->z >= m_lastPosition.z )
152 directionOK = FALSE;
153
154 } // end fi
155
156 // get the terrain height
157 Real terrainHeightAtPos = TheTerrainLogic->getGroundHeight( pos->x, pos->y );
158
159 // if including structures, check for bridges
160 if (modData->m_targetHeightIncludesStructures)
161 {
162 PathfindLayerEnum layer = TheTerrainLogic->getHighestLayerForDestination(pos);
163 if (layer != LAYER_GROUND)
164 {
165 Real layerHeight = TheTerrainLogic->getLayerHeight(pos->x, pos->y, layer);
166 if (layerHeight > terrainHeightAtPos)
167 terrainHeightAtPos = layerHeight;
168 }
169 }
170
171 //
172 // our target height to die at is by default the height specified in the INI entry above
173 // the terrain ... we may change our target height if we care about dying above
174 // objects under us (see below)
175 //
176 Real targetHeight = terrainHeightAtPos + modData->m_targetHeightAboveTerrain;
177
178 //
179 // if we consider objects under us ... we will die when we are the specified distance above
180 // those objects
181 //
182 if( modData->m_targetHeightIncludesStructures == TRUE )
183 {
184
185 // scan all objects in the radius of our extent and find the tallest height among them
186 PartitionFilterAcceptByKindOf filter1( MAKE_KINDOF_MASK( KINDOF_STRUCTURE ),KINDOFMASK_NONE );
187 PartitionFilter *filters[] = { &filter1, NULL };
188 Real range = getObject()->getGeometryInfo().getBoundingCircleRadius();
189 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( getObject(),
190 range,
191 FROM_BOUNDINGSPHERE_3D,
192 filters );
193 MemoryPoolObjectHolder hold( iter );
194 Object *obj;
195
196 Real tallestHeight = 0.0f;
197 Real thisHeight;
198 for( obj = iter->first(); obj; obj = iter->next() )
199 {
200
201 // ignore ourselves
202 if( obj == getObject() )
203 continue;
204
205 // store the height of the tallest object under us
206 thisHeight = obj->getGeometryInfo().getMaxHeightAbovePosition();
207
208 if( thisHeight > tallestHeight )
209 tallestHeight = thisHeight;
210
211 } // end for obj
212
213 //
214 // our target height is either the height above the terrain as specified by the INI
215 // entry for the object that has this update ... or it is the building height of the
216 // tallest thing under us
217 //
218 if( tallestHeight > modData->m_targetHeightAboveTerrain )
219 targetHeight = tallestHeight + terrainHeightAtPos;
220
221 } // end if
222
223 // if we are below the target height ... DIE!
224 if( pos->z < targetHeight && directionOK )
225 {
226
227 // if we're supposed to snap us to the ground on death do so
228 // AND: even if we're not snapping to ground, be sure we don't go BELOW ground
229 if( modData->m_snapToGroundOnDeath || pos->z < terrainHeightAtPos )
230 {
231 Coord3D ground;
232
233 ground.x = pos->x;
234 ground.y = pos->y;
235 ground.z = terrainHeightAtPos;
236 getObject()->setPosition( &ground );
237
238 }
239
240 // kill the object
241 getObject()->kill();
242
243 // we have died ... don't do this again
244 m_hasDied = TRUE;
245
246 } // end if
247
248 } // end if
249
250 //
251 // if our height is below the destroy attached particles height above the terrain, clean
252 // them up from the particle system
253 //
254 if( m_particlesDestroyed == FALSE && pos->z < modData->m_destroyAttachedParticlesAtHeight && (m_hasDied || directionOK) )
255 {
256
257 // destroy them
258 TheParticleSystemManager->destroyAttachedSystems( getObject() );
259
260 // don't do this again
261 m_particlesDestroyed = TRUE;
262
263 } // end if
264
265 // save our current position as the last position we monitored
266 m_lastPosition = *pos;
267
268 return UPDATE_SLEEP_NONE;
269
270} // end update
