67enum StateReturnType
68{
69 // note that all positive values are reserved for STATE_SLEEP!
70
71 STATE_CONTINUE = 0,
72 STATE_SUCCESS = -1,
73 STATE_FAILURE = -2,
74};
75
76#define STATE_SLEEP(numFrames) ((StateReturnType)(numFrames))
77#define IS_STATE_SLEEP(ret) ((Int)(ret) > 0)
78#define GET_STATE_SLEEP_FRAMES(ret) ((UnsignedInt)(ret))
79
80// (we use 0x3fffffff so that we can add offsets and not overflow...
81// at 30fps that's around ~414 days!)
82#define STATE_SLEEP_FOREVER STATE_SLEEP(0x3fffffff)
83
84// this is mainly useful for states that enclose other state machines...
85// where, even if the enclosed machine is sleeping, the encloser still needs
86// to run every frame.
87inline StateReturnType CONVERT_SLEEP_TO_CONTINUE(StateReturnType s)
88{
89 return IS_STATE_SLEEP(s) ? STATE_CONTINUE : s;
90}
91
92// this is mainly useful for states that enclose other state machines...
93// where the encloser and enclosee both might sleep. we need to choose the min
94// sleep time that satisfies both.
95inline StateReturnType MIN_SLEEP(UnsignedInt encloserSleep, StateReturnType encloseeResult)
96{
97 if (IS_STATE_SLEEP(encloseeResult))
98 {
99 UnsignedInt encloseeSleep = GET_STATE_SLEEP_FRAMES(encloseeResult);
100 return STATE_SLEEP(min(encloserSleep, encloseeSleep));
101 }
102 else
103 {
104 // if enclosee needs to stay awake, we better do so, regardless
105 return encloseeResult;
106 }
107}
