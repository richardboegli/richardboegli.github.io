120static Int getListboxEntryBasedOnCoord(GameWindow *window, Int x, Int y, Int &row, Int &column)
121{
122 Int pos;
123 Int winx, winy, i;
124 WinInstanceData *instData = window->winGetInstanceData();
125 ListboxData *list = (ListboxData *)window->winGetUserData();
126
127 window->winGetScreenPosition( &winx, &winy );
128
129 // Adjust for title if present
130 if( instData->getTextLength() )
131 winy += TheWindowManager->winFontHeight( instData->getFont() ) + 1;
132
133 pos = -2;
134
135 for( i=0; ; i++ )
136 {
137 if( i > 0 )
138 if( list->listData[ i - 1 ].listHeight >
139 (list->displayPos + list->displayHeight ) )
140 {
141 pos = -1;
142 break;
143 }
144
145 if( i == list->endPos )
146 {
147 pos = -1;
148 break;
149 }
150
151 if( list->listData[i].listHeight > (y - winy + list->displayPos) )
152 break;
153 }
154
155 column = -1;
156 if( pos == -2 )
157 {
158 pos = i;
159 Int total = 0;
160 for( i = 0; i < list->columns ;i++)
161 {
162 total += list->columnWidth[i];
163 if(x - winx < total)
164 {
165 column = i;
166 break;
167 }
168 }
169 }
170 row = pos;
171 return pos;
172}

182static Int getListboxTopEntry( ListboxData *list )
183{
184 Int entry;
185
186 // determin which entry is at the top of the display area
187 for( entry=0; ; entry++ )
188 {
189 if( list->listData[entry].listHeight > list->displayPos )
190 return entry;
191
192 if( entry >= list->endPos )
193 return 0;
194 }
195
196 return 0;
197}

199// getListboxTopEntry =========================================================
200//=============================================================================
201static Int getListboxBottomEntry( ListboxData *list )
202{
203 Int entry;
204
205 // determin which entry is at the top of the display area
206 for( entry=list->endPos - 1; ; entry-- )
207 {
208 if( list->listData[entry].listHeight == list->displayPos + list->displayHeight )
209 return entry;
210 if( list->listData[entry].listHeight < list->displayPos + list->displayHeight && entry != list->endPos - 1)
211 return entry + 1;
212 if( list->listData[entry].listHeight < list->displayPos + list->displayHeight)
213 return entry;
214 if( entry < 0 )
215 return 0;
216 }
217
218 return 0;
219}

242 ListboxData *list = (ListboxData *)window->winGetUserData();
243

250 entry = list->endPos - 1;
251
252 if( updateSlider )
253 {
254 if( entry > 0 )
255 list->displayPos = list->listData[(entry - 1)].listHeight + 1;
256 else
257 list->displayPos = 0;
258 }

261 {
262 ICoord2D sliderSize, sliderChildSize;
263 GameWindow *child;
264
265 sData = (SliderData *)list->slider->winGetUserData();
266 list->slider->winGetSize( &sliderSize.x, &sliderSize.y );
267 // Take into account that there is a line-drawn outline surrounding listbox
268 sData->maxVal = list->totalHeight - ( list->displayHeight - TOTAL_OUTLINE_HEIGHT ) + 1;
269

271 {
272 sData->maxVal = 0;
273 }
274
275 child = list->slider->winGetChild();
276 child->winGetSize( &sliderChildSize.x, &sliderChildSize.y );
277 sData->numTicks = (float)((sliderSize.y - sliderChildSize.y) / (float)sData->maxVal);
278
279 if( updateSlider )
280 TheWindowManager->winSendSystemMsg( list->slider,
281 GSM_SET_SLIDER,
282 (sData->maxVal - list->displayPos),
283 0 );
284 }

291static void computeTotalHeight( GameWindow *window )
292{
293 Int i, height = 0;
294 Int tempHeight;
295 ListboxData *list = (ListboxData *)window->winGetUserData();
296 WinInstanceData *instData = window->winGetInstanceData();
297
298 for( i=0; i<list->endPos; i++ )
299 {
300
301 if(!list->listData[i].cell)
302 continue;
303 tempHeight = 0;
304
305 for (Int j = 0; j < list->columns; j++)
306 {
307 Int cellHeight = 0;
308 if(list->listData[i].cell[j].cellType == LISTBOX_TEXT)
309 {
310 if( BitTest( window->winGetStatus(), WIN_STATUS_ONE_LINE ) == TRUE )
311 {
312 cellHeight = TheWindowManager->winFontHeight( instData->getFont() );
313 }
314 else
315 {
316 DisplayString *displayString = (DisplayString *)list->listData[i].cell[j].data;
317 if(displayString)
318 displayString->getSize( NULL, &cellHeight );
319 }//else
320 }//if
321 else if(list->listData[i].cell[j].cellType == LISTBOX_IMAGE)
322 {
323 if(list->listData[i].cell[j].height > 0)
324 cellHeight = list->listData[i].cell[j].height + 1;
325 else
326 cellHeight = TheWindowManager->winFontHeight( instData->getFont() );
327 }
328 if(cellHeight > tempHeight)
329 tempHeight = cellHeight;
330 }//for
331 list->listData[i].height = tempHeight;
332 height += (list->listData[i].height + 1);
333 list->listData[i].listHeight = height;
334 }
335
336 list->totalHeight = height;
337
338 adjustDisplay( window, 0, TRUE );
339}

393// startingRow will get moved to startingRow+1, etc. This assumes there is space!!!!!
394static Int moveRowsDown(ListboxData *list, Int startingRow)
395{
396 //
397 // copy the cells down
398 //
399 Int copyLen = (list->endPos - startingRow) * sizeof(ListEntryRow);
400 char *buf = NEW char[copyLen];
401 memcpy(buf, list->listData + startingRow, copyLen);
402 memcpy(list->listData + startingRow + 1, buf, copyLen );
403 delete buf;
404
405 list->endPos ++;
406 list->insertPos = list->endPos;
407
408 //
409 // remove the display or links to images after the shift
410 //
411 list->listData[startingRow].cell = NULL;
412 list->listData[startingRow].height = 0;
413 list->listData[startingRow].listHeight = 0;
414
415 if( list->multiSelect )
416 {
417 Int i = 0;
418
419 while( list->selections[i] >= 0 )
420 {
421 if( startingRow <= list->selections[i] )
422 list->selections[i]++;
423 i++;
424 }
425 }
426 else
427 {
428 if( list->selectPos >= startingRow )
429 list->selectPos++;
430 }
431
432 /*
433 if( list->displayPos > 0 )
434 adjustDisplay( window, (-1 * mData1), TRUE );
435
436 computeTotalHeight( window );
437 */
438
439 return 1;
440}

445static Int addEntry( UnicodeString *string, Int color, Int row, Int column, GameWindow *window, Bool overwrite )
446{
447// WinInstanceData *instData = window->winGetInstanceData();
448 ListboxData *list = (ListboxData *)window->winGetUserData();
449 Int width;
450 DisplayString *displayString;
451
452 // make sure our params are good
453 if( column >= list->columns || row >= list->listLength )
454 {
455 DEBUG_ASSERTCRASH(false, ("Tried to add text to Listbox at invalid position"));
456 return -1;
457 }
458
459 // If we want to just add an entry to the bottom, set the defaults
460 if (row == -1)
461 {
462 row = list->insertPos;
463 list->insertPos++;
464 list->endPos++;
465 }
466 if( column == -1 )
467 column = 0;
468
469 width = list->columnWidth[column] - TEXT_WIDTH_OFFSET;
470
471 Int rowsAdded = 0;
472
473 ListEntryRow *listRow = &list->listData[row];
474 // Here I've decided to just overright what's in the row, if that's not what we want, change it here
475 // Check and see if we have allocated cells for that row yet, if not, allocate them
476 if(!listRow->cell)
477 {
478 listRow->cell = NEW ListEntryCell[list->columns];
479 memset(listRow->cell,0,list->columns * sizeof(ListEntryCell));
480 rowsAdded = 1;
481 }
482 else if (!overwrite)
483 {
484 // Shove things down
485 moveRowsDown(list, row);
486 listRow->cell = NEW ListEntryCell[list->columns];
487 memset(listRow->cell,0,list->columns * sizeof(ListEntryCell));
488 rowsAdded = 1;
489 }
490
491 //add Image to selected row/cell
492 listRow->cell[column].cellType = LISTBOX_TEXT;
493
494 // assign the color to the list data element
495 listRow->cell[column].color = color;
496
497 // copy text
498 if( !listRow->cell[column].data )
499 listRow->cell[column].data = (void *) TheDisplayStringManager->newDisplayString();
500 displayString = (DisplayString *) listRow->cell[column].data;
501 if ( BitTest( window->winGetStatus(), WIN_STATUS_ONE_LINE ) == FALSE )
502 displayString->setWordWrap( width );
503 displayString->setText( *string );
504
507 displayString->setFont( window->winGetFont() );
508
509 if (overwrite)
510 {
511 Int oldRowHeight = listRow->height;
512 Int oldTotalHeight = listRow->listHeight;
513 Int rowHeight;
514 Int totalHeight;
515
516 if (!oldTotalHeight && row)
517 {
518 oldTotalHeight = list->listData[row-1].listHeight;
519 }
520
521 displayString->getSize( NULL, &rowHeight );
522 if (rowHeight > oldRowHeight)
523 {
524 totalHeight = oldTotalHeight + (rowHeight - oldRowHeight);
525 listRow->height = rowHeight;
526 listRow->listHeight = totalHeight + rowsAdded;
527 list->totalHeight += (rowHeight - oldRowHeight) + rowsAdded;
528 adjustDisplay( window, 0, TRUE );
529 }
530 }
531 else
532 {
533 computeTotalHeight( window );
534 }
535
536 return (row);
537} // end addEntry

549// GadgetListBoxInput =========================================================
551//=============================================================================
552WindowMsgHandledType GadgetListBoxInput( GameWindow *window, UnsignedInt msg,
553 WindowMsgData mData1, WindowMsgData mData2 )
554{
555 ListboxData *list = (ListboxData *)window->winGetUserData();
556 WinInstanceData *instData = window->winGetInstanceData();
557
558 switch (msg)
559 {
560
561 // ------------------------------------------------------------------------
562 case GWM_CHAR:
563 {
564
565 switch (mData1)
566 {
567
568 // --------------------------------------------------------------------
569 case KEY_ENTER:
570 case KEY_SPACE:
571 {
572
573 if( BitTest( mData2, KEY_STATE_UP ) )
574 {
575 doAudioFeedback(window);
576
577 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
578 GLM_DOUBLE_CLICKED,
579 (WindowMsgData)window,
580 list->selectPos );
581 } // end if
582
583 break;
584
585 } // end enter or space
586
587 // --------------------------------------------------------------------
588 case KEY_DOWN:
589 {
590
591 if( BitTest( mData2, KEY_STATE_DOWN ) )
592 {
593
594 if( list->selectPos == -1 )
595 {
596 list->selectPos = 0;
597 adjustDisplay( window, 0, TRUE );
598 }
599 else if( list->selectPos < list->endPos - 1 )
600 {
601
602 list->selectPos++;
603
604 while (1)
605 {
606 Int cellBottom = list->listData[list->selectPos].listHeight;
607 Int cellTop = cellBottom - list->listData[list->selectPos].height;
608 Int displayTop = list->displayPos;
609 Int displayBottom = list->displayPos + list->displayHeight - 1; // account for the border
610
611 if ( cellTop < displayTop )
612 {
613 adjustDisplay(window, -1, TRUE );
614 }
615 else if ( cellBottom < displayBottom )
616 {
617 adjustDisplay(window, 0, TRUE );
618 break;
619 }
620 else
621 {
622 adjustDisplay(window, 1, TRUE );
623 }
624 }
625
626 }
627
628 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
629 GLM_SELECTED,
630 (WindowMsgData)window,
631 list->selectPos );
632 } // end if
633
634 break;
635
636 } // end key down
637
638 // --------------------------------------------------------------------
639 case KEY_UP:
640 {
641
642 if( BitTest( mData2, KEY_STATE_DOWN ) )
643 {
644
645 if( list->selectPos == -1 )
646 {
647 list->selectPos = 0;
648 adjustDisplay( window, 0, TRUE );
649 }
650 else if( list->selectPos > 0 )
651 {
652
653 list->selectPos--;
654
655 while (1)
656 {
657 if( list->listData[list->selectPos].listHeight - list->listData[list->selectPos].height < list->displayPos )
658 {
659 list->displayPos = list->listData[list->selectPos].listHeight +1;
660 adjustDisplay( window, -1, TRUE );
661 }
662 else if( list->listData[list->selectPos].listHeight > list->displayPos + list->displayHeight)
663 {
664 list->displayPos = list->listData[list->selectPos].listHeight - list->displayHeight;
665 adjustDisplay(window, 1, TRUE);
666 }
667 else
668 {
669 adjustDisplay(window, 0, TRUE);
670 break;
671 }
672 }
673 }
674
675 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
676 GLM_SELECTED,
677 (WindowMsgData)window,
678 list->selectPos );
679 }
680
681 break;
682
683 } // end key up
684
685 // --------------------------------------------------------------------
686 case KEY_RIGHT:
687 case KEY_TAB:
688
689 if( BitTest( mData2, KEY_STATE_DOWN ) )
690 TheWindowManager->winNextTab(window);
691 break;
692
693 // --------------------------------------------------------------------
694 case KEY_LEFT:
695
696 if( BitTest( mData2, KEY_STATE_DOWN ) )
697 TheWindowManager->winPrevTab(window);
698 break;
699
700 // --------------------------------------------------------------------
701 default:
702 {
703 Bool foundIt = false;
704 if( BitTest( mData2, KEY_STATE_DOWN ) )
705 {
706 // set the position to start looking for the line of text with this character
707 Int position = list->selectPos;
708 // only search the max number of times so that we're not looping over and over
709 for(Int i = 0; i < list->endPos; ++i)
710 {
711 // start at the next position
712 ++position;
713 // if we've reached the end of the list, start at the beginning
714 if( position >= list->endPos)
715 position = 0;
716
717 ListEntryCell *cell = NULL;
718 // go through the columns until we find a column with text
719 for(Int j = 0; j < list->columns; ++j)
720 {
721 cell = &list->listData[position].cell[j];
722 if(cell && cell->cellType == LISTBOX_TEXT && cell->data)
723 {
724 break;
725 }
726 }
727 if(!cell || cell->cellType != LISTBOX_TEXT)
728 continue;
729 DisplayString *dString = (DisplayString *)cell->data;
730 if(!dString)
731 continue;
732 for(j = 0; j < TheKeyboard->MAX_KEY_STATES; ++j)
733 {
734 if(dString->getText().getCharAt(0) == TheKeyboard->getPrintableKey(mData1, j))
735 {
736 list->selectPos = position;
737 Int prevPos = getListboxTopEntry(list);
738 //list->displayPos = list->listData[position].listHeight - list->displayHeight;
739 adjustDisplay(window, position - prevPos, TRUE);
740 foundIt = TRUE;
741 break;
742 }
743 }
744 if(foundIt)
745 {
746 doAudioFeedback(window);
747 break;
748 }
749
750 }
751
752
753 }
754
755 if (!foundIt)
756 return MSG_IGNORED;
757 }
758
759 } // end switch( mData1 )
760
761 break;
762
763 } // end case char
764
765 // ------------------------------------------------------------------------
766 case GWM_WHEEL_DOWN:
767 {
768 if( list->endPos <= 0)
769 break;
770
771 if (list->listData[list->endPos - 1].listHeight > list->displayHeight + list->displayPos)
772 adjustDisplay( window, 1, TRUE );
773
774 break;
775
776 } // end wheel down
777
778 // ------------------------------------------------------------------------
779 case GWM_WHEEL_UP:
780 {
781 if( list->endPos <= 0)
782 break;
783
784 adjustDisplay( window, -1, TRUE );
785 break;
786
787 } // end wheel up
788
789 // ------------------------------------------------------------------------
790 case GWM_LEFT_UP:
791 {
792 TheWindowManager->winSetFocus( window );
793// Int mousex = mData1 & 0xFFFF;
794 Int mousey = mData1 >> 16;
795 Int x, y, i;
796 Int oldPos = list->selectPos;
797
798 window->winGetScreenPosition( &x, &y );
799
800 // Adjust for title if present
801 if( instData->getTextLength() )
802 y += TheWindowManager->winFontHeight( instData->getFont() ) + 1;
803
804 list->selectPos = -2;
805
806 for( i=0; ; i++ )
807 {
808
809 if( i > 0 )
810 if( list->listData[ i - 1 ].listHeight >
811 (list->displayPos + list->displayHeight) )
812 {
813 list->selectPos = -1;
814 break;
815 }
816
817 if( i == list->endPos )
818 {
819 list->selectPos = -1;
820 break;
821 }
822
823 if( list->listData[i].listHeight > (mousey - y + list->displayPos) )
824 break;
825 }
826
827 //Bool dblClicked = FALSE;
828 if( list->doubleClickTime + doubleClickTime > timeGetTime() &&
829 (i == oldPos || (oldPos == -1 && ( i>=0 && i<list->endPos ) )) )
830 {
831 int temp;
832 list->doubleClickTime = 0;
833 if( oldPos == -1 )
834 temp = i;
835 else
836 temp = oldPos;
837 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
838 GLM_DOUBLE_CLICKED,
839 (WindowMsgData)window,
840 temp );
841 //break;
842 }
843 if( (i == oldPos) && (list->forceSelect == FALSE) )
844 {
845 list->selectPos = -1;
846 }
847
848 if( (list->selectPos == -2) && (i < list->endPos) )
849 {
850 list->selectPos = i;
851 }
852
853 if( (list->selectPos < 0) && (list->forceSelect) )
854 {
855 list->selectPos = oldPos;
856 }
857 list->doubleClickTime = timeGetTime();
858 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
859 GLM_SELECTED,
860 (WindowMsgData)window,
861 list->selectPos );
862
863 break;
864
865 } // end left click, left up
866
867 // ------------------------------------------------------------------------
868 case GWM_RIGHT_DOWN:
869 doAudioFeedback(window);
870 break; // if we're in game, we want to eat this message because we're right clicking on a listbox
