109NetGameCommandMsg::NetGameCommandMsg(GameMessage *msg) : NetCommandMsg() {

112 m_type = msg->getType();

134void NetGameCommandMsg::addArgument(const GameMessageArgumentDataType type, GameMessageArgumentType arg)
135{
136 if (m_argTail == NULL) {
137 m_argList = newInstance(GameMessageArgument);
138 m_argTail = m_argList;
139 m_argList->m_data = arg;
140 m_argList->m_type = type;
141 m_argList->m_next = NULL;
142 return;
143 }
144
145 GameMessageArgument *newArg = newInstance(GameMessageArgument);
146 newArg->m_data = arg;
147 newArg->m_type = type;
148 newArg->m_next = NULL;
149 m_argTail->m_next = newArg;
150 m_argTail = newArg;
151}

153// here's where we figure out which slot corresponds to which player
154static Int indexFromMask(UnsignedInt mask)
155{
156 Player *player = NULL;
157 Int i;
158
159 for( i = 0; i < MAX_PLAYER_COUNT; i++ )
160 {
161 player = ThePlayerList->getNthPlayer( i );
162 if( player && player->getPlayerMask() == mask )
163 return i;
164 } // end for i
165
166 return -1;
167}

172GameMessage *NetGameCommandMsg::constructGameMessage()
173{
174 GameMessage *retval = newInstance(GameMessage)(m_type);
175
176 AsciiString name;
177 name.format("player%d", getPlayerID());
178 retval->friend_setPlayerIndex( ThePlayerList->findPlayerWithNameKey(TheNameKeyGenerator->nameToKey(name))->getPlayerIndex());
179// retval->friend_setPlayerIndex(indexFromMask(ThePlayerList->findPlayerWithNameKey(TheNameKeyGenerator->nameToKey(name))->getPlayerMask()));
180
181 GameMessageArgument *arg = m_argList;
182 while (arg != NULL) {
183// retval->appendGenericArgument(arg->m_data);
184 if (arg->m_type == ARGUMENTDATATYPE_INTEGER) {
185 retval->appendIntegerArgument(arg->m_data.integer);
186 } else if (arg->m_type == ARGUMENTDATATYPE_REAL) {
187 retval->appendRealArgument(arg->m_data.real);
188 } else if (arg->m_type == ARGUMENTDATATYPE_BOOLEAN) {
189 retval->appendBooleanArgument(arg->m_data.boolean);
190 } else if (arg->m_type == ARGUMENTDATATYPE_OBJECTID) {
191 retval->appendObjectIDArgument(arg->m_data.objectID);
192 } else if (arg->m_type == ARGUMENTDATATYPE_DRAWABLEID) {
193 retval->appendDrawableIDArgument(arg->m_data.drawableID);
194 } else if (arg->m_type == ARGUMENTDATATYPE_TEAMID) {
195 retval->appendTeamIDArgument(arg->m_data.teamID);
196 } else if (arg->m_type == ARGUMENTDATATYPE_LOCATION) {
197 retval->appendLocationArgument(arg->m_data.location);
198 } else if (arg->m_type == ARGUMENTDATATYPE_PIXEL) {
199 retval->appendPixelArgument(arg->m_data.pixel);
200 } else if (arg->m_type == ARGUMENTDATATYPE_PIXELREGION) {
201 retval->appendPixelRegionArgument(arg->m_data.pixelRegion);
202 } else if (arg->m_type == ARGUMENTDATATYPE_TIMESTAMP) {
203 retval->appendTimestampArgument(arg->m_data.timestamp);
204 } else if (arg->m_type == ARGUMENTDATATYPE_WIDECHAR) {
205 retval->appendWideCharArgument(arg->m_data.wChar);
206 }
207 arg = arg->m_next;
208 }
209 return retval;
210}
