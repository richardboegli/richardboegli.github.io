113HRESULT DownloadManager::downloadNextQueuedFile( void )
114{
115 QueuedDownload q;
116 std::list<QueuedDownload>::iterator it = m_queuedDownloads.begin();
117 if (it != m_queuedDownloads.end())
118 {
119 q = *it;
120 m_queuedDownloads.pop_front();
121 m_wasError = m_sawEnd = false;
122 return downloadFile( q.server, q.userName, q.password, q.file, q.localFile, q.regKey, q.tryResume );
123 }
124 else
125 {
126 DEBUG_CRASH(("Starting non-existent download!"));
127 return S_OK;
128 }
129}
