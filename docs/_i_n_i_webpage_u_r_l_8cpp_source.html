51AsciiString encodeURL(AsciiString source)
52{
53 if (source.isEmpty())
54 {
55 return AsciiString::TheEmptyString;
56 }
57
58 AsciiString target;
59 AsciiString allowedChars = "$-_.+!*'(),\\";
60 const char *ptr = source.str();
61 while (*ptr)
62 {
63 if (isalnum(*ptr) || allowedChars.find(*ptr))
64 {
65 target.concat(*ptr);
66 }
67 else
68 {
69 AsciiString tmp;
70 target.concat('%');
71 tmp.format("%2.2x", ((int)*ptr));
72 target.concat(tmp);
73 }
74 ++ptr;
75 }
76
77 return target;
78}
