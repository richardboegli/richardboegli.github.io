77Bool W3DSnowManager::ReAcquireResources(void)
78{
79 ReleaseResources();
80
81 if (!TheWeatherSetting->m_snowEnabled)
82 return TRUE; //no need for resources if snow is disabled.
83
84 if (TheWeatherSetting->m_usePointSprites && DX8Wrapper::Get_Current_Caps()->Support_PointSprites())
85 {
86 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
87
88 DEBUG_ASSERTCRASH(m_pDev, ("Trying to ReAquireResources on W3DSnowManager without device"));
89
90 if (m_VertexBufferD3D == NULL)
91 { // Create vertex buffer
92
93 if (FAILED(m_pDev->CreateVertexBuffer
94 (
95 SNOW_BUFFER_SIZE*sizeof(POINTVERTEX),
96 D3DUSAGE_WRITEONLY|D3DUSAGE_DYNAMIC|D3DUSAGE_POINTS,
97 D3DFVF_POINTVERTEX,
98 D3DPOOL_DEFAULT,
99 &m_VertexBufferD3D
100 )))
101 return FALSE;
102 }
103 }
104 else
105 {
106 m_indexBuffer=NEW_REF(DX8IndexBufferClass,(SNOW_BATCH_SIZE *6)); //allocate 2 triangles per flake, each with 3 indices.
107
108 // Fill up the IB with static vertex indices that will be used for all smudges.
109 {
110 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
111 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
112 //quad of 4 triangles:
113 // 0-----3
114 // |\ /|
115 // | X |
116 // |/ \|
117 // 1-----2
118 Int vbCount=0;
119 for (Int i=0; i<SNOW_BATCH_SIZE; i++)
120 {
121 //Top
122 ib[0]=vbCount+3;
123 ib[1]=vbCount;
124 ib[2]=vbCount+2;
125 //Bottom
126 ib[3]=vbCount+2;
127 ib[4]=vbCount;
128 ib[5]=vbCount+1;
129
130 vbCount += 4;
131 ib+=6;
132 }
133 }
134 }
135
136 m_snowTexture = WW3DAssetManager::Get_Instance()->Get_Texture(TheWeatherSetting->m_snowTexture.str());
137
138 m_dwBase = SNOW_BUFFER_SIZE;
139 m_dwDiscard = SNOW_BUFFER_SIZE;
140 m_dwFlush = SNOW_BATCH_SIZE;
141
142 return TRUE;
143}

171#define MAXIMUM_CAMERA_DISTANCE 100000 //maximum distance of camera position from world origin.
172#define ISPOW2(x) (x && (x & (x-1)) == 0) //is a number a power of 2?
173#define MODPOW2(x,y) ((x) & (y-1)) //mod '%' operator for powers of 2.

176inline DWORD FtoDW( FLOAT f ) { return *((DWORD*)&f); }

178/*Recursively subdivide the large snow box enclosing the camera until we reach some predefined leaf size. This
179method is used so that very few off-screen particles end up getting rendered. Culling them individually would
180be too expensive since we're dealing with 1000's for this effect.*/
181void W3DSnowManager::renderSubBox(RenderInfoClass &rinfo, Int originX, Int originY, Int cubeDimX, Int cubeDimY )
182{
183 //check if this box is too large and needs subdivision
184 Int boxDimX=cubeDimX - originX;
185 Int boxDimY=cubeDimY - originY;
186 Int halfX=REAL_TO_INT_CEIL(boxDimX*0.5f);
187 Int halfY=REAL_TO_INT_CEIL(boxDimY*0.5f);
188
189 CameraClass &camera=rinfo.Camera;
190 MinMaxAABoxClass mmbox;
191
192 if (boxDimX > m_leafDim)
193 { //subdivide the box
194 if (boxDimY > m_leafDim)
195 { //subdivide in both directions
196 //Upper left
197 mmbox.MinCorner.Set(originX*m_emitterSpacing-m_cullOverscan, (originY + halfY)*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
198 mmbox.MaxCorner.Set((originX + halfX)*m_emitterSpacing+m_cullOverscan, cubeDimY*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
199 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
200 renderSubBox(rinfo, originX, originY + halfY, originX + halfX, cubeDimY);
201 //Upper right
202 mmbox.MinCorner.Set((originX + halfX)*m_emitterSpacing-m_cullOverscan, (originY + halfY)*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
203 mmbox.MaxCorner.Set(cubeDimX*m_emitterSpacing+m_cullOverscan, cubeDimY*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
204 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
205 renderSubBox(rinfo, originX + halfX, originY + halfY,cubeDimX, cubeDimY);
206 //Lower left
207 mmbox.MinCorner.Set(originX*m_emitterSpacing-m_cullOverscan, originY*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
208 mmbox.MaxCorner.Set((originX + halfX)*m_emitterSpacing+m_cullOverscan, (originY + halfY)*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
209 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
210 renderSubBox(rinfo, originX,originY,originX + halfX, originY + halfY);
211 //Lower right
212 mmbox.MinCorner.Set((originX + halfX)*m_emitterSpacing-m_cullOverscan, originY*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
213 mmbox.MaxCorner.Set(cubeDimX*m_emitterSpacing+m_cullOverscan, (originY + halfY)*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
214 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
215 renderSubBox(rinfo, originX + halfX, originY, cubeDimX, originY + halfY);
216 return;
217 }
218 else
219 { //only subdivide in x direction.
220 //Left
221 mmbox.MinCorner.Set(originX*m_emitterSpacing-m_cullOverscan, originY*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
222 mmbox.MaxCorner.Set((originX + halfX)*m_emitterSpacing+m_cullOverscan, cubeDimY*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
223 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
224 renderSubBox(rinfo, originX, originY, originX + halfX, cubeDimY);
225 //Right
226 mmbox.MinCorner.Set((originX + halfX)*m_emitterSpacing-m_cullOverscan, originY*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
227 mmbox.MaxCorner.Set(cubeDimX*m_emitterSpacing+m_cullOverscan, cubeDimY*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
228 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
229 renderSubBox(rinfo, originX + halfX, originY, cubeDimX, cubeDimY);
230 return;
231 }
232 }
233 else
234 if (boxDimY > m_leafDim)
235 { //only subdivide in y direction
236 //Top
237 mmbox.MinCorner.Set(originX*m_emitterSpacing-m_cullOverscan, (originY+halfY)*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
238 mmbox.MaxCorner.Set(cubeDimX*m_emitterSpacing+m_cullOverscan, cubeDimY*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
239 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
240 renderSubBox(rinfo, originX, originY+halfY,cubeDimX, cubeDimY);
241 //Bottom
242 mmbox.MinCorner.Set(originX*m_emitterSpacing-m_cullOverscan, originY*m_emitterSpacing-m_cullOverscan, m_snowCeiling-m_boxDimensions);
243 mmbox.MaxCorner.Set(cubeDimX*m_emitterSpacing+m_cullOverscan, (originY + halfY)*m_emitterSpacing+m_cullOverscan, m_snowCeiling);
244 if (CollisionMath::Overlap_Test(camera.Get_Frustum(),mmbox) != CollisionMath::OUTSIDE)
245 renderSubBox(rinfo, originX, originY, cubeDimX, originY + halfY);
246 return;
247 }
248
249 //Box too small to subdivide so render it.
250
251 //Find total number of particles that need rendering.
252 Int totalPart=(cubeDimY-originY)*(cubeDimX-originX);
253
254 if (!totalPart)
255 return; //nothing to render.
256
257 Int y=originY; //loop counter.
258 Int cubeOriginXRemainder = originX; //loop counter - adjusted when not all particles fit into render buffer.
259 Vector3 snowCenter;
260
261 m_totalRendered += totalPart;
262
263 while (totalPart)
264 {
265 Int batchSize=totalPart;
266
267 if (batchSize > m_dwFlush)
268 batchSize = m_dwFlush;
269
270 if((m_dwBase + batchSize) > m_dwDiscard)
271 m_dwBase = 0;
272
273 POINTVERTEX* verts;
274
275 if(m_VertexBufferD3D->Lock(m_dwBase * sizeof(POINTVERTEX), batchSize * sizeof(POINTVERTEX),
276 (unsigned char **) &verts, m_dwBase ? D3DLOCK_NOOVERWRITE : D3DLOCK_DISCARD) != D3D_OK )
277 return; //couldn't lock buffer.
278
279 Int numberInBatch=0;
280
281 for (;y<cubeDimY; y++)
282 {
283 for (Int x=cubeOriginXRemainder; x<cubeDimX; x++)
284 {
285 if (numberInBatch >= batchSize)
286 { cubeOriginXRemainder = x;
287 goto flush_particles;
288 }
289
290 //Get initial height from noise table. We add a large value to make sure it's positive. Then
291 //modulate by table dimensions to find a value.
292 Int noiseOffset=MODPOW2(x+MAXIMUM_CAMERA_DISTANCE,SNOW_NOISE_X)+MODPOW2(y+MAXIMUM_CAMERA_DISTANCE,SNOW_NOISE_Y)*SNOW_NOISE_X;
293 if (noiseOffset > (SNOW_NOISE_X * SNOW_NOISE_Y))
294 noiseOffset = 0; //this should never happen but check to prevent buffer over/under flow.
295
296 //find current height
297 Real h0=m_snowCeiling-fmod(m_heightTraveled+m_startingHeights[noiseOffset],m_boxDimensions);
298
299 //find world-space position of snow flake
300 snowCenter.Set(x*m_emitterSpacing,y*m_emitterSpacing,h0);
301
302 //Adjust position so snow flakes don't fall straight down.
303 snowCenter.X += m_amplitude * WWMath::Fast_Sin( h0 * m_frequencyScaleX + (Real)x);
304 snowCenter.Y += m_amplitude * WWMath::Fast_Sin( h0 * m_frequencyScaleY + (Real)y);
305
306 *(Vector3 *)verts=snowCenter;
307 verts++;
308
309 numberInBatch++;
310 }
311 //getting here means we did not overflow the render buffer, so reset x origin to normal.
312 cubeOriginXRemainder = originX; //reset to normal amount
313 }
314
315flush_particles:
316 m_VertexBufferD3D->Unlock();
317 //Render any particles that may be queued up.
318 if (numberInBatch)
319 {
320 Debug_Statistics::Record_DX8_Polys_And_Vertices(numberInBatch*2,numberInBatch*4,ShaderClass::_PresetOpaqueShader);
321 DX8Wrapper::_Get_D3D_Device8()->DrawPrimitive( D3DPT_POINTLIST, m_dwBase, numberInBatch);
322 totalPart -= numberInBatch;
323 m_dwBase += numberInBatch;
324 }
325 }
326}

328void W3DSnowManager::render(RenderInfoClass &rinfo)
329{
330 if (!TheWeatherSetting->m_snowEnabled || !m_isVisible)
331 return;
332
333 Int usePointSprites = DX8Wrapper::Get_Current_Caps()->Support_PointSprites() && TheWeatherSetting->m_usePointSprites;
334
335 //make sure the noise table is powers of 2 in dimensions.
336 WWASSERT(ISPOW2(SNOW_NOISE_X) && ISPOW2(SNOW_NOISE_Y));
337
338 //CameraClass &camera=rinfo.Camera;
339
340 const Coord3D &cPos=TheTacticalView->get3DCameraPosition();
341 Vector3 camPos(cPos.x,cPos.y,cPos.z);
342
343 //Number of emitters from cube center to edge of visible extent.
344 Int mumEmittersInHalf=(Int)floor(m_boxDimensions / m_emitterSpacing * 0.5f);
345
346 //Find origin of visible cube surrounding camera.
347 Int cubeCenterX=(Int)floor(camPos.X/m_emitterSpacing);
348 Int cubeCenterY=(Int)floor(camPos.Y/m_emitterSpacing);
349
350 //Find extents of visible cube surrounding camera.
351 Int cubeOriginX=cubeCenterX - mumEmittersInHalf; //top/left extents.
352 Int cubeOriginY=cubeCenterY - mumEmittersInHalf;
353 Int cubeDimX=cubeCenterX + mumEmittersInHalf; //bottom/right extents.
354 Int cubeDimY=cubeCenterY + mumEmittersInHalf;
355
356 const FrustumClass & frustum = rinfo.Camera.Get_Frustum();
357 AABoxClass bbox;
358
359 //Get a bounding box around our visible universe. Bounded by terrain and the sky
360 //so much tighter fitting volume than what's actually visible. This will cull
361 //particles falling under the ground.
362
363 TheTerrainRenderObject->getMaximumVisibleBox(frustum, &bbox, TRUE);
364
365 //Particles move outside the visible box as a result of local sine movement
366 //so adjust bounding box to include them.
367 bbox.Extent.X += m_amplitude+m_quadSize;
368 bbox.Extent.Y += m_amplitude+m_quadSize;
369
370 //Clip our visible snow rendering box
371 if ((cubeOriginX * m_emitterSpacing ) < (bbox.Center.X - bbox.Extent.X))
372 cubeOriginX = (Int)floor ((bbox.Center.X - bbox.Extent.X)/m_emitterSpacing);
373
374 if ((cubeOriginY * m_emitterSpacing ) < (bbox.Center.Y - bbox.Extent.Y))
375 cubeOriginY = (Int)floor ((bbox.Center.Y - bbox.Extent.Y)/m_emitterSpacing);
376
377 if ((cubeDimX * m_emitterSpacing ) > (bbox.Center.X + bbox.Extent.X))
378 cubeDimX = (Int)floor ((bbox.Center.X + bbox.Extent.X)/m_emitterSpacing);
379
380 if ((cubeDimY * m_emitterSpacing ) > (bbox.Center.Y + bbox.Extent.Y))
381 cubeDimY = (Int)floor ((bbox.Center.Y + bbox.Extent.Y)/m_emitterSpacing);
382
383 if ((cubeDimY - cubeOriginY) < 0 || (cubeDimX-cubeOriginX) < 0)
384 return; //entire snow box is culled by either x or y screen boundary.
385
386 //Find total number of particles that need rendering.
387 Int totalPart=(cubeDimY-cubeOriginY)*(cubeDimX-cubeOriginX);
388
389 if (totalPart <= 0)
390 return; //nothing to render.
391
392 //Height at the top of the cube with camera at center.
393 m_snowCeiling = camPos.Z + m_boxDimensions/2.0f;
394
395 //Offset to allow cube extents to move with camera.
396 Real cameraOffset = fmod (camPos.Z,m_boxDimensions);
397 m_heightTraveled=m_time*m_velocity+cameraOffset; //height that snow flake traveled this frame.
398
399 Matrix4x4 identity(true);
400 DX8Wrapper::Set_Transform(D3DTS_WORLD,identity);
401
402 DX8Wrapper::Set_Shader(ShaderClass::_PresetAlphaShader);
403
404 VertexMaterialClass *vmat=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
405 DX8Wrapper::Set_Material(vmat);
406 REF_PTR_RELEASE(vmat);
407
408 //make sure we have all the resources we need
409 if (usePointSprites && !m_VertexBufferD3D)
410 ReAcquireResources();
411
412 if (!usePointSprites && !m_indexBuffer)
413 ReAcquireResources();
414
415 DX8Wrapper::Set_Texture(0,m_snowTexture);
416
417 if (!usePointSprites)
418 {
419 renderAsQuads(rinfo,cubeOriginX,cubeOriginY,cubeDimX,cubeDimY);
420 return;
421 }
422
423 Vector3 snowCenter;
424
425 DX8Wrapper::Apply_Render_State_Changes();
426
427 // Set the render states for using point sprites
428 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSPRITEENABLE, TRUE );
429 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSCALEENABLE, TRUE );
430 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSIZE, FtoDW(m_pointSize) );
431 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSIZE_MIN, FtoDW(m_minPointSize) );
432 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSIZE_MAX, FtoDW(m_maxPointSize) );
433 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSCALE_A, FtoDW(0.00f) );
434 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSCALE_B, FtoDW(0.00f) );
435 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSCALE_C, FtoDW(1.00f) );
436
437 DX8Wrapper::_Get_D3D_Device8()->SetStreamSource( 0, m_VertexBufferD3D, sizeof(POINTVERTEX) );
438 DX8Wrapper::_Get_D3D_Device8()->SetVertexShader( D3DFVF_POINTVERTEX );
439 m_dwBase = SNOW_BUFFER_SIZE; //start with a new vertex buffer each frame.
440
441 m_leafDim = 45; //cull boxes that are 20x20 emitters in size. Making them much smaller will result in too many draw calls.
442 m_totalRendered = 0; //keep track of how many particles were rendered.
443
444 //Particle centers can deviate from center by by amplitude of sine offset. They also have radius m_quadSize.
445 //Enlarge culling bounds to compensate.
446 m_cullOverscan = m_amplitude+m_quadSize;
447 renderSubBox(rinfo,cubeOriginX,cubeOriginY,cubeDimX,cubeDimY);
448
449 // Reset render states
450 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSPRITEENABLE, FALSE );
451 DX8Wrapper::Set_DX8_Render_State( D3DRS_POINTSCALEENABLE, FALSE );
452
453}

456void W3DSnowManager::renderAsQuads(RenderInfoClass &rinfo, Int cubeOriginX, Int cubeOriginY, Int cubeDimX, Int cubeDimY)
457{
458
459 Matrix4x4 proj;
460 Matrix3D view;
461 Vector3 snowCenter;
462 Vector3 snowCenterVS;
463
464 CameraClass &camera=rinfo.Camera;
465
466 camera.Get_View_Matrix(&view);
467 camera.Get_Projection_Matrix(&proj);
468
469 Vector3 vertex_offsets[4] = {
470 Vector3(-0.5f, 0.5f, 0.0f),
471 Vector3(-0.5f, -0.5f, 0.0f),
472 Vector3(0.5f, -0.5f, 0.0f),
473 Vector3(0.5f, 0.5f, 0.0f)
474 };
475
476 Vector2 quad_uvs[4] = {
477 Vector2(0.0f, 0.0f),
478 Vector2(0.0f, 1.0f),
479 Vector2(1.0f, 1.0f),
480 Vector2(1.0f, 0.0f)
481 };
482
483
484 //pre-multiple the offsets by particle size
485 for (Int i=0; i<4; i++)
486 {
487 vertex_offsets[i] *= m_quadSize;
488 }
489
490 Matrix4x4 identity(true);
491 DX8Wrapper::Set_Transform(D3DTS_VIEW,identity);
492
493 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
494
495 Int y=cubeOriginY; //loop counter.
496 Int cubeOriginXRemainder = cubeOriginX; //loop counter - adjusted when not all particles fit into render buffer.
497
498 //Find total number of particles that need rendering.
499 Int totalPart=(cubeDimY-cubeOriginY)*(cubeDimX-cubeOriginX);
500
501 m_totalRendered += totalPart;
502
503 while (totalPart)
504 {
505 Int batchSize=totalPart;
506
507 if (batchSize > SNOW_BATCH_SIZE)
508 batchSize = SNOW_BATCH_SIZE;
509
510 Int numberInBatch=0;
511
512 DynamicVBAccessClass vb_access(BUFFER_TYPE_DYNAMIC_DX8,dynamic_fvf_type,batchSize*4); //allocate 4 verts per flake
513 {
514 DynamicVBAccessClass::WriteLockClass lock(&vb_access);
515 VertexFormatXYZNDUV2* verts=lock.Get_Formatted_Vertex_Array();
516
517 for (;y<cubeDimY; y++)
518 {
519 for (Int x=cubeOriginXRemainder; x<cubeDimX; x++)
520 {
521 if (numberInBatch >= batchSize)
522 { cubeOriginXRemainder = x;
523 goto flush_particles;
524 }
525
526 //Get initial height from noise table. We add a large value to make sure it's positive. Then
527 //modulate by table dimensions to find a value.
528 Int noiseOffset=MODPOW2(x+MAXIMUM_CAMERA_DISTANCE,SNOW_NOISE_X)+MODPOW2(y+MAXIMUM_CAMERA_DISTANCE,SNOW_NOISE_Y)*SNOW_NOISE_X;
529 if (noiseOffset > (SNOW_NOISE_X * SNOW_NOISE_Y))
530 noiseOffset = 0; //this should never happen but check to prevent buffer over/under flow.
531
532 //find current height
533 Real h0=m_snowCeiling-fmod(m_heightTraveled+m_startingHeights[noiseOffset],m_boxDimensions);
534
535 //find world-space position of snow flake
536 snowCenter.Set(x*m_emitterSpacing,y*m_emitterSpacing,h0);
537
538 //Get view-space position
539 Matrix3D::Transform_Vector(view,snowCenter,&snowCenterVS);
540
541 //Adjust position so snow flakes don't fall straight down.
542 snowCenterVS.X += m_amplitude * WWMath::Fast_Sin( h0 * m_frequencyScaleX + (Real)x);
543 snowCenterVS.Y += m_amplitude * WWMath::Fast_Sin( h0 * m_frequencyScaleY + (Real)y);
544
545 for (Int i=0; i<4; i++)
546 {
547 *(Vector3 *)verts=snowCenterVS + vertex_offsets[i];
548 verts->nx=0; //keep AGP write-combining active
549 verts->ny=0;
550 verts->nz=0;
