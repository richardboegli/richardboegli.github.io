45void W3DDynamicLight::On_Frame_Update(void)
46{
47 if (!m_enabled) {
48 return;
49 }
50 Real factor = 1.0f;
51 if (m_curIncreaseFrameCount>0 && m_increaseFrameCount>0) {
52 // increasing
53 m_curIncreaseFrameCount--;
54 factor = (m_increaseFrameCount-m_curIncreaseFrameCount)/(Real)m_increaseFrameCount;
55
56 } else if (m_decayFrameCount==0) {
57 factor = 1.0; // never decays,
58 } else {
59 m_curDecayFrameCount--;
60 if (m_curDecayFrameCount == 0) {
61 m_enabled = false;
62 return;
63 }
64 factor = m_curDecayFrameCount/(Real)m_decayFrameCount;
65 }
66 if (m_decayRange) {
67 this->FarAttenEnd = factor*m_targetRange;
68 if (FarAttenEnd < FarAttenStart) {
69 FarAttenEnd = FarAttenStart;
70 }
71 }
72 if (m_decayColor) {
73 this->Ambient = m_targetAmbient*factor;
74 this->Diffuse = m_targetDiffuse*factor;
75 }
76}
