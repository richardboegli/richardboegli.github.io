135Bool PolygonTrigger::ParsePolygonTriggersDataChunk(DataChunkInput &file, DataChunkInfo *info, void *userData)
136{
137 Int count;
138 Int numPoints;
139 Int triggerID;
140 Int maxTriggerId = 0;
141 Bool isWater;
142 Bool isRiver;
143 Int riverStart;
144 AsciiString triggerName;
145 AsciiString layerName;
146 // Remove any existing polygon triggers, if any.
147 PolygonTrigger::deleteTriggers(); // just in case.
148 PolygonTrigger *pPrevTrig = NULL;
149 ICoord3D loc;
150 count = file.readInt();
151 while (count>0) {
152 count--;
153 triggerName = file.readAsciiString();
154 if (info->version >= K_TRIGGERS_VERSION_4) {
155 layerName = file.readAsciiString();
156 }
157 triggerID = file.readInt();
158 isWater = false;
159 if (info->version >= K_TRIGGERS_VERSION_2) {
160 isWater = file.readByte();
161 }
162 isRiver = false;
163 riverStart = 0;
164 if (info->version >= K_TRIGGERS_VERSION_3) {
165 isRiver = file.readByte();
166 riverStart = file.readInt();
167 }
168
169 numPoints = file.readInt();
170 PolygonTrigger *pTrig = newInstance(PolygonTrigger)(numPoints+1);
171 pTrig->setTriggerName(triggerName);
172 if (info->version >= K_TRIGGERS_VERSION_4) {
173 pTrig->setLayerName(layerName);
174 }
175 pTrig->setWaterArea(isWater);
176 pTrig->setRiver(isRiver);
177 pTrig->setRiverStart(riverStart);
178 pTrig->m_triggerID = triggerID;
179 if (triggerID > maxTriggerId) {
180 maxTriggerId = triggerID;
181 }
182 Int i;
183 for (i=0; i<numPoints; i++) {
184 loc.x = file.readInt();
185 loc.y = file.readInt();
186 loc.z = file.readInt();
187 pTrig->addPoint(loc);
188 }
189 if (numPoints<2) {
190 DEBUG_LOG(("Deleting polygon trigger '%s' with %d points.\n",
191 pTrig->getTriggerName().str(), numPoints));
192 pTrig->deleteInstance();
193 continue;
194 }
195 if (pPrevTrig) {
196 pPrevTrig->setNextPoly(pTrig);
197 } else {
198 PolygonTrigger::addPolygonTrigger(pTrig);
199 }
200 pPrevTrig = pTrig;
201 }
202 if (info->version == K_TRIGGERS_VERSION_1)
203 {
204 // before water areas existed, so create a default one.
205 PolygonTrigger *pTrig = newInstance(PolygonTrigger)(4);
206 pTrig->setWaterArea(true);
207#ifdef _DEBUG
208 pTrig->setTriggerName("AutoAddedWaterAreaTrigger");
209#endif
210 pTrig->m_triggerID = maxTriggerId++;
211 loc.x = -30*MAP_XY_FACTOR;
212 loc.y = -30*MAP_XY_FACTOR;
213 loc.z = 7; // The old water position.
214 pTrig->addPoint(loc);
215 loc.x = 30*MAP_XY_FACTOR + TheGlobalData->m_waterExtentX;
216 pTrig->addPoint(loc);
217 loc.y = 30*MAP_XY_FACTOR + TheGlobalData->m_waterExtentY;
218 pTrig->addPoint(loc);
219 loc.x = -30*MAP_XY_FACTOR;
220 pTrig->addPoint(loc);
221 if (pPrevTrig) {
222 pPrevTrig->setNextPoly(pTrig);
223 } else {
224 PolygonTrigger::addPolygonTrigger(pTrig);
225 }
226 pPrevTrig = pTrig;
227 }
228 s_currentID = maxTriggerId+1;
229 DEBUG_ASSERTCRASH(file.atEndOfChunk(), ("Incorrect data file length."));
230 return true;
231}

240void PolygonTrigger::WritePolygonTriggersDataChunk(DataChunkOutput &chunkWriter)
241{
242 chunkWriter.openDataChunk("PolygonTriggers", K_TRIGGERS_VERSION_4);
243
244 PolygonTrigger *pTrig;
245 Int count = 0;
246 for (pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
247 count++;
248 }
249 chunkWriter.writeInt(count);
250 for (pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
251 chunkWriter.writeAsciiString(pTrig->getTriggerName());
252 chunkWriter.writeAsciiString(pTrig->getLayerName());
253 chunkWriter.writeInt(pTrig->getID());
254 chunkWriter.writeByte(pTrig->isWaterArea());
255 chunkWriter.writeByte(pTrig->isRiver());
256 chunkWriter.writeInt(pTrig->getRiverStart());
257 chunkWriter.writeInt(pTrig->getNumPoints());
258 Int i;
259 for (i=0; i<pTrig->getNumPoints(); i++) {
260 ICoord3D loc = *pTrig->getPoint(i);
261 chunkWriter.writeInt( loc.x);
262 chunkWriter.writeInt( loc.y);
263 chunkWriter.writeInt( loc.z);
264 }
265 }
266
267 chunkWriter.closeDataChunk();
268}

273void PolygonTrigger::updateBounds(void) const
274{
275 const Int BIG_INT=0x7ffff0;
276 m_bounds.lo.x = m_bounds.lo.y = BIG_INT;
277 m_bounds.hi.x = m_bounds.hi.y = -BIG_INT;
278 Int i;
279 for (i=0; i<m_numPoints; i++) {
280 if (m_points[i].x < m_bounds.lo.x) m_bounds.lo.x = m_points[i].x;
281 if (m_points[i].y < m_bounds.lo.y) m_bounds.lo.y = m_points[i].y;
282 if (m_points[i].x > m_bounds.hi.x) m_bounds.hi.x = m_points[i].x;
283 if (m_points[i].y > m_bounds.hi.y) m_bounds.hi.y = m_points[i].y;
284 }
285 m_boundsNeedsUpdate = 0;
286 Real halfWidth = (m_bounds.hi.x - m_bounds.lo.x) / 2.0f;
287 Real halfHeight = (m_bounds.hi.y + m_bounds.lo.y) / 2.0f;
288
289 m_radius = sqrt(halfHeight*halfHeight + halfWidth*halfWidth);
290}

310void PolygonTrigger::removePolygonTrigger(PolygonTrigger *pTrigger)
311{
312 PolygonTrigger *pPrev = NULL;
313 for (PolygonTrigger *pTrig=getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
314 if (pTrig==pTrigger) break;
315 pPrev = pTrig;
316 }
317 DEBUG_ASSERTCRASH(pTrig, ("Attempting to remove a polygon not in the list."));
318 if (pTrig) {
319 if (pPrev) {
320 DEBUG_ASSERTCRASH(pTrigger==pPrev->m_nextPolygonTrigger, ("Logic errror. jba."));
321 pPrev->m_nextPolygonTrigger = pTrig->m_nextPolygonTrigger;
322 } else {
323 DEBUG_ASSERTCRASH(pTrigger==ThePolygonTriggerListPtr, ("Logic errror. jba."));
324 ThePolygonTriggerListPtr = pTrig->m_nextPolygonTrigger;
325 }
326 }
327 pTrigger->m_nextPolygonTrigger = NULL;
328}

419void PolygonTrigger::getCenterPoint(Coord3D* pOutCoord) const
420{
421 DEBUG_ASSERTCRASH(pOutCoord != NULL, ("pOutCoord was null. Non-Fatal, but shouldn't happen."));
422 if (!pOutCoord) {
423 return;
424 }
425
426 if (m_boundsNeedsUpdate) {
427 updateBounds();
428 }
429 (*pOutCoord).x = (m_bounds.lo.x + m_bounds.hi.x) / 2.0f;
430 (*pOutCoord).y = (m_bounds.lo.y + m_bounds.hi.y) / 2.0f;
431
432 (*pOutCoord).z = TheTerrainLogic->getGroundHeight(pOutCoord->x, pOutCoord->y);
433}

447Bool PolygonTrigger::pointInTrigger(ICoord3D &point) const
448{
449 if (m_boundsNeedsUpdate) {
450 updateBounds();
451 }
452 if (point.x < m_bounds.lo.x) return false;
453 if (point.y < m_bounds.lo.y) return false;
454 if (point.x > m_bounds.hi.x) return false;
455 if (point.y > m_bounds.hi.y) return false;
456
457 Bool inside = false;
458 Int i;
459 for (i=0; i<m_numPoints; i++) {
460 ICoord3D pt1 = m_points[i];
461 ICoord3D pt2;
462 if (i==m_numPoints-1) {
463 pt2 = m_points[0];
464 } else {
465 pt2 = m_points[i+1];
466 }
467 if (pt1.y == pt2.y) {
468 continue; // ignore horizontal lines.
469 }
470 if (pt1.y < point.y && pt2.y < point.y) continue;
471 if (pt1.y >= point.y && pt2.y >= point.y) continue;
472 if (pt1.x<point.x && pt2.x < point.x) continue;
473 // Line segment crosses ray from point x->infinity.
474 Int dy = pt2.y-pt1.y;
475 Int dx = pt2.x-pt1.x;
476
477 Real intersectionX = pt1.x + (dx * (point.y-pt1.y)) / ((Real)dy);
478 if (intersectionX >= point.x) {
479 inside = !inside;
480 }
481 }
482 return inside;
483}

518void PolygonTrigger::xfer( Xfer *xfer )
519{
520
521 // version
522 XferVersion currentVersion = 1;
523 XferVersion version = currentVersion;
524 xfer->xferVersion( &version, currentVersion );
525
526 // number of data points
527 xfer->xferInt( &m_numPoints );
528
529 // xfer all data points
530 ICoord3D *point;
531 for( Int i = 0; i < m_numPoints; ++i )
532 {
533
534 // get this point
535 point = &m_points[ i ];
536
537 // xfer point
538 xfer->xferICoord3D( point );
539
540 } // end for, i
541
542 // bounds
543 xfer->xferIRegion2D( &m_bounds );
544
545 // radius
546 xfer->xferReal( &m_radius );
547
548 // bounds need update
549 xfer->xferBool( &m_boundsNeedsUpdate );
550
551} // end xfer
