26// PolygonTrigger.h
27// Class to encapsulate polygon triggers for maps.
28// Note - Polygons are used for two reasons - one is area triggers for
29// scripts, so units can be tested for entering or exiting areas, and
30// second to specify areas that are filled with water in the map.
31// See the m_isWaterArea to differentiate.
