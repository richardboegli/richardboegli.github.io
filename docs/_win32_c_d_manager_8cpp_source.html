129void Win32CDDrive::refreshInfo( void )
130{
131 Bool mayRequireUpdate = (m_disk != CD::NO_DISK);
132 Char volName[1024];
133 // read the volume info
134 if ( GetVolumeInformation( m_drivePath.str(), volName, sizeof(volName) -1, NULL, NULL, NULL, NULL, 0 ))
135 {
136 m_diskName = volName;
137 m_disk = CD::UNKNOWN_DISK;
138 }
139 else
140 {
141 m_diskName.clear();
142 m_disk = CD::NO_DISK;
143
144 if (mayRequireUpdate)
145 TheFileSystem->unloadMusicFilesFromCD();
146 }
147
148 // This is an override, not an extension of CDDrive
149}

172
173void Win32CDManager::init( void )
174{
175 CDManager::init(); // init base classes
176
177 destroyAllDrives();
178
179 // detect CD Drives
180 for ( Char driveLetter = 'a'; driveLetter <= 'z'; driveLetter++ )
181 {
182 AsciiString drivePath;
183 drivePath.format( "%c:\\", driveLetter );
184
185 if ( GetDriveType( drivePath.str() ) == DRIVE_CDROM )
186 {
187 newDrive( drivePath.str() );
188 }
189 }
190
191 refreshDrives();
192}
