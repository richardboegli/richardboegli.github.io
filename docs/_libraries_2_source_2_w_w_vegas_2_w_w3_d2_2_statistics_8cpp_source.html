168static bool Find_Record_Texture(TextureClass* t)
169{
170 for (int a=0;a<texture_count;++a) {
171 if (texture_statistics[a].tex==t) {
172 if (record_texture_mode==Debug_Statistics::RECORD_TEXTURE_DETAILS) {
173 texture_statistics[a].usage_count++;
174 if (t!=latest_texture) {
175 texture_statistics[a].change_count++;
176 }
177 }
178
179 return true;
180 }
181 }
182 return false;
183}
184
185static void Add_Record_Texture(TextureClass* t)
186{
187 TextureStatisticsStruct tss;
188 tss.tex=t;
189 tss.usage_count=1;
190 tss.change_count=1;
191 texture_statistics.Add(tss);
192
193 texture_count++;
194 if (t->Is_Lightmap()) lightmap_texture_count++;
195 if (t->Is_Procedural()) procedural_texture_count++;
196}

198void Debug_Statistics::Record_Texture(TextureClass* t)
199{
200 record_count++;
201 if (t!=latest_texture) {
202 texture_change_count++;
203 }
204
205 if (record_texture_mode==RECORD_TEXTURE_NONE) {
206 latest_texture=t;
207 return;
208 }
209 if (!t) {
210 latest_texture=t;
211 return;
212 }
213
214 if (Find_Record_Texture(t)) {
215 latest_texture=t;
216 return;
217 }
218 Add_Record_Texture(t);
219
220 texture_memory+=t->Get_Texture_Memory_Usage();
221 latest_texture=t;
222 if (t->Is_Lightmap()) lightmap_texture_memory+=t->Get_Texture_Memory_Usage();
223 if (t->Is_Procedural()) procedural_texture_memory+=t->Get_Texture_Memory_Usage();
224
225}
