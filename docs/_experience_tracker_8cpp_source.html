89// Set Level to AT LEAST this... if we are already >= this level, do nothing.
90void ExperienceTracker::setMinVeterancyLevel( VeterancyLevel newLevel )
91{
92 // This does not check for IsTrainable, because this function is for explicit setting,
93 // so the setter is assumed to know what they are doing. The game function
94 // of addExperiencePoints cares about Trainability.
95 if (m_currentLevel < newLevel)
96 {
97 VeterancyLevel oldLevel = m_currentLevel;
98 m_currentLevel = newLevel;
99 m_currentExperience = m_parent->getTemplate()->getExperienceRequired(m_currentLevel); //Minimum for this level
100 if (m_parent)
101 m_parent->onVeterancyLevelChanged( oldLevel, newLevel );
102 }
103}

122Bool ExperienceTracker::gainExpForLevel(Int levelsToGain, Bool canScaleForBonus)
123{
124 Int newLevel = (Int)m_currentLevel + levelsToGain;
125 if (newLevel > LEVEL_LAST)
126 newLevel = LEVEL_LAST;
127 // gain what levels we can, even if we can't use 'em all
128 if (newLevel > m_currentLevel)
129 {
130 Int experienceNeeded = m_parent->getTemplate()->getExperienceRequired(newLevel) - m_currentExperience;
131 addExperiencePoints( experienceNeeded, canScaleForBonus );
132 return true;
133 }
134 return false;
135}

148void ExperienceTracker::addExperiencePoints( Int experienceGain, Bool canScaleForBonus)
149{
150 if( m_experienceSink != INVALID_ID )
151 {
152 // I have been set up to give my experience to someone else
153 Object *sinkPointer = TheGameLogic->findObjectByID( m_experienceSink );
154 if( sinkPointer )
155 {
156 // Not a fatal failure if not valid, he died when I was in the air.
157 sinkPointer->getExperienceTracker()->addExperiencePoints( experienceGain * m_experienceScalar, canScaleForBonus );
158 return;
159 }
160 }
161
162 if( !isTrainable() )
163 return; //safety
164
165 VeterancyLevel oldLevel = m_currentLevel;
166
167 Int amountToGain = experienceGain;
168 if ( canScaleForBonus )
169 amountToGain *= m_experienceScalar;
170
171
172 m_currentExperience += amountToGain;
173
174 Int levelIndex = 0;
175 while( ( (levelIndex + 1) < LEVEL_COUNT)
176 && m_currentExperience >= m_parent->getTemplate()->getExperienceRequired(levelIndex + 1)
177 )
178 {
179 // If there is a higher level to qualify for, and I qualify for it, advance the index
180 levelIndex++;
181 }
182
183 m_currentLevel = (VeterancyLevel)levelIndex;
184
185 if( oldLevel != m_currentLevel )
186 {
187 // Edge trigger special level gain effects.
188 m_parent->onVeterancyLevelChanged( oldLevel, m_currentLevel );
189 }
190
191}
