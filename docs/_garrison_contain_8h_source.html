83 static void parseInitialRoster( INI* ini, void *instance, void *store, const void* )
84 {
85 GarrisonContainModuleData* self = (GarrisonContainModuleData*)instance;
86 const char* name = ini->getNextToken();
87 const char* countStr = ini->getNextTokenOrNull();
88 Int count = countStr ? INI::scanInt(countStr) : 1;
89
90 self->m_initialRoster.templateName.set(name);
91 self->m_initialRoster.count = count;
92 };
