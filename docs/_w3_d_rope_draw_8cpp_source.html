79void W3DRopeDraw::buildSegments()
80{
81 DEBUG_ASSERTCRASH(m_segments.empty(), ("Hmmn, not empty"));
82 m_segments.clear();
83
84 Int numSegs = ceil(m_maxLen / m_wobbleLen);
85 Real eachLen = m_maxLen / (Real)numSegs;
86 Coord3D pos = *getDrawable()->getPosition();
87 for (int i = 0; i < numSegs; ++i, pos.z += eachLen)
88 {
89 SegInfo info;
90
91 Real axis = GameClientRandomValueReal(0, 2*PI);
92 info.wobbleAxisX = Cos(axis);
93 info.wobbleAxisY = Sin(axis);
94 info.line = NEW Line3DClass( Vector3(pos.x,pos.y,pos.z),
95 Vector3(pos.x,pos.y,pos.z+eachLen),
96 m_width * 0.5f, // width
97 m_color.red, // red
98 m_color.green, // green
99 m_color.blue, // blue
100 1.0f ); // transparency
101
102 info.softLine = NEW Line3DClass( Vector3(pos.x,pos.y,pos.z),
103 Vector3(pos.x,pos.y,pos.z+eachLen),
104 m_width, // width
105 m_color.red, // red
106 m_color.green, // green
107 m_color.blue, // blue
108 0.5f ); // transparency
109
110 W3DDisplay::m_3DScene->Add_Render_Object( info.line );
111 W3DDisplay::m_3DScene->Add_Render_Object( info.softLine );
112 m_segments.push_back(info);
113 }
114}

118void W3DRopeDraw::tossSegments()
119{
120 // remove tracer from the scene and delete
121 for (std::vector<SegInfo>::iterator it = m_segments.begin(); it != m_segments.end(); ++it)
122 {
123 if (it->line)
124 {
125 W3DDisplay::m_3DScene->Remove_Render_Object(it->line);
126 REF_PTR_RELEASE((it->line));
127 }
128 if (it->softLine)
129 {
130 W3DDisplay::m_3DScene->Remove_Render_Object(it->softLine);
131 REF_PTR_RELEASE((it->softLine));
132 }
133 }
134 m_segments.clear();
135}

180void W3DRopeDraw::doDrawModule(const Matrix3D* transformMtx)
181{
182 if (m_segments.empty())
183 {
184 buildSegments();
185 }
186
187 if (!m_segments.empty())
188 {
189 Real deflection = Sin(m_curWobblePhase) * m_wobbleAmp;
190 const Coord3D* pos = getDrawable()->getPosition();
191 Vector3 start(pos->x, pos->y, pos->z + m_curZOffset);
192 Real eachLen = m_curLen / m_segments.size();
193 for (std::vector<SegInfo>::iterator it = m_segments.begin(); it != m_segments.end(); ++it)
194 {
195 Vector3 end(pos->x + deflection*it->wobbleAxisX, pos->y + deflection*it->wobbleAxisY, start.Z - eachLen);
196 if (it->line)
197 (it->line)->Reset(start, end);
198 if (it->softLine)
199 (it->softLine)->Reset(start, end);
200 start = end;
201 }
202 }
203
204 m_curWobblePhase += m_wobbleRate;
205 if (m_curWobblePhase > 2*PI)
206 m_curWobblePhase -= 2*PI;
207
208 m_curZOffset += m_curSpeed;
209 m_curSpeed += m_accel;
210 if (m_curSpeed > m_maxSpeed)
211 m_curSpeed = m_maxSpeed;
212 else if (m_curSpeed < -m_maxSpeed)
213 m_curSpeed = -m_maxSpeed;
214}
