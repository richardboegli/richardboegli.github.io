80static LRESULT CALLBACK genericPropertiesCallback( HWND hWndDialog,
81 UINT message,
82 WPARAM wParam,
83 LPARAM lParam )
84{
85 Int returnCode;
86
87 //
88 // handle any common messages between all property dialogs cause they
89 // are designed to have controls doing the same functionality
90 // and names
91 //
92 if( HandleCommonDialogMessages( hWndDialog, message,
93 wParam, lParam, &returnCode ) == TRUE )
94 return returnCode;
95
96 switch( message )
97 {
98
99 // ------------------------------------------------------------------------
100 case WM_DRAWITEM:
101 {
102 UINT controlID = (UINT)wParam; // control identifier
103 LPDRAWITEMSTRUCT drawItem = (LPDRAWITEMSTRUCT)lParam; // item drawing
104 RGBColorInt *color = GetControlColor( controlID );
105
106 // we only care about color button controls
107 if( color )
108 {
109 HBRUSH hBrushNew, hBrushOld;
110 RECT rect;
111 HWND hWndControl = GetDlgItem( hWndDialog, controlID );
112
113 // if this control is disabled just let windows handle drawing
114 if( IsWindowEnabled( hWndControl ) == FALSE )
115 return FALSE;
116
117 // Get the area we have to draw in
118 GetClientRect( hWndControl, &rect );
119
120 // create a new brush and select it into DC
121 hBrushNew = CreateSolidBrush (RGB ((BYTE)color->red,
122 (BYTE)color->green,
123 (BYTE)color->blue));
124 hBrushOld = (HBRUSH)SelectObject( drawItem->hDC, hBrushNew );
125
126 // draw the rectangle
127 Rectangle( drawItem->hDC, rect.left, rect.top, rect.right, rect.bottom );
128
129 // put the old brush back and delete the new one
130 SelectObject( drawItem->hDC, hBrushOld );
131 DeleteObject( hBrushNew );
132
133 // validate this new area
134 ValidateRect( hWndControl, NULL );
135
136 // we have taken care of it
137 return TRUE;
138
139 } // end if
140
141 return FALSE;
142
143 } // end draw item
144
145 // ------------------------------------------------------------------------
146 case WM_COMMAND:
147 {
148// Int notifyCode = HIWORD( wParam ); // notification code
149 Int controlID = LOWORD( wParam ); // control ID
150 HWND hWndControl = (HWND)lParam; // control window handle
151
152 switch( controlID )
153 {
154
155 // --------------------------------------------------------------------
156 case BUTTON_ENABLED_COLOR:
157 case BUTTON_ENABLED_BORDER_COLOR:
158 case BUTTON_DISABLED_COLOR:
159 case BUTTON_DISABLED_BORDER_COLOR:
160 case BUTTON_HILITE_COLOR:
161 case BUTTON_HILITE_BORDER_COLOR:
162 {
163 RGBColorInt *currColor = GetControlColor( controlID );
164
165 // bring up color selector for this color control at the mouse
166 if( currColor )
167 {
168 RGBColorInt *newColor;
169 POINT mouse;
170
171 GetCursorPos( &mouse );
172 newColor = SelectColor( currColor->red, currColor->green,
173 currColor->blue, currColor->alpha,
174 mouse.x, mouse.y );
175
176 if( newColor )
177 {
178 Color newGameColor = GameMakeColor( newColor->red,
179 newColor->green,
180 newColor->blue,
181 newColor->alpha );
182 SetControlColor( controlID, newGameColor );
183 InvalidateRect( hWndControl, NULL, TRUE );
184
185 } // end if
186
187 } // end if
188
189 break;
190
191 } // end color buttons
192
193 // --------------------------------------------------------------------
194 case IDOK:
195 {
196 GameWindow *window = TheEditor->getPropertyTarget();
197
198 // sanity
199 if( window )
200 {
201 const Image *image;
202 RGBColorInt *rgbColor;
203 Color color;
204
205 // save the common properties
206 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
207 break;
208
209 // save callbacks
210 SaveCallbacks( window, hWndDialog );
211
212 // save the enabled colors/images
213 image = ComboBoxSelectionToImage( GetDlgItem( hWndDialog, COMBO_ENABLED_IMAGE ) );
214 window->winSetEnabledImage( 0, image );
215 rgbColor = GetControlColor( BUTTON_ENABLED_COLOR );
216 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
217 window->winSetEnabledColor( 0, color );
218 rgbColor = GetControlColor( BUTTON_ENABLED_BORDER_COLOR );
219 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
220 window->winSetEnabledBorderColor( 0, color );
221
222 // save the disabled colors/images
223 image = ComboBoxSelectionToImage( GetDlgItem( hWndDialog, COMBO_DISABLED_IMAGE ) );
224 window->winSetDisabledImage( 0, image );
225 rgbColor = GetControlColor( BUTTON_DISABLED_COLOR );
226 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
227 window->winSetDisabledColor( 0, color );
228 rgbColor = GetControlColor( BUTTON_DISABLED_BORDER_COLOR );
229 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
230 window->winSetDisabledBorderColor( 0, color );
231
232 // save the hilite colors/images
233 image = ComboBoxSelectionToImage( GetDlgItem( hWndDialog, COMBO_HILITE_IMAGE ) );
234 window->winSetHiliteImage( 0, image );
235 rgbColor = GetControlColor( BUTTON_HILITE_COLOR );
236 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
237 window->winSetHiliteColor( 0, color );
238 rgbColor = GetControlColor( BUTTON_HILITE_BORDER_COLOR );
239 color = GameMakeColor( rgbColor->red, rgbColor->green, rgbColor->blue, rgbColor->alpha );
240 window->winSetHiliteBorderColor( 0, color );
241
242 } // end if
243
244 DestroyWindow( hWndDialog );
245 break;
246
247 } // end OK
248
249 // --------------------------------------------------------------------
250 case IDCANCEL:
251 {
252
253 DestroyWindow( hWndDialog );
254 break;
255
256 } // end cancel
257
258 } // end switch( LOWORD( wParam ) )
259
260 return 0;
261
262 } // end of WM_COMMAND
263
264 // ------------------------------------------------------------------------
265 case WM_CLOSE:
266 {
267
268 DestroyWindow( hWndDialog );
269 return 0;
270
271 } // end close
272
273 // ------------------------------------------------------------------------
274 default:
275 return 0;
276
277 } // end of switch
278
279} // end genericPropertiesCallback

426// InitUserWinPropertiesDialog ================================================
428//=============================================================================
429HWND InitUserWinPropertiesDialog( GameWindow *window )
430{
431 HWND dialog;
432
433 // create the dialog box
434 dialog = CreateDialog( TheEditor->getInstance(),
435 (LPCTSTR)GENERIC_PROPERTIES_DIALOG,
436 TheEditor->getWindowHandle(),
437 (DLGPROC)genericPropertiesCallback );
438 if( dialog == NULL )
439 return NULL;
440
441 // do the common initialization
442 CommonDialogInitialize( window, dialog );
443
444 //
445 // initialize the dialog with values from the window
446 //
447
448 // fill out the image combo boxes
449 LoadImageListComboBox( GetDlgItem( dialog, COMBO_ENABLED_IMAGE ) );
450 LoadImageListComboBox( GetDlgItem( dialog, COMBO_DISABLED_IMAGE ) );
451 LoadImageListComboBox( GetDlgItem( dialog, COMBO_HILITE_IMAGE ) );
452
453 // select any images in the combo boxes
454 const Image *image;
455
456 image = window->winGetEnabledImage( 0 );
457 if( image )
458 SendDlgItemMessage( dialog, COMBO_ENABLED_IMAGE, CB_SELECTSTRING,
459 -1, (LPARAM)image->getName().str() );
460 image = window->winGetDisabledImage( 0 );
461 if( image )
462 SendDlgItemMessage( dialog, COMBO_DISABLED_IMAGE, CB_SELECTSTRING,
463 -1, (LPARAM)image->getName().str() );
464 image = window->winGetHiliteImage( 0 );
465 if( image )
466 SendDlgItemMessage( dialog, COMBO_HILITE_IMAGE, CB_SELECTSTRING,
467 -1, (LPARAM)image->getName().str() );
468
469 // initialize the color buttons
470 SetControlColor( BUTTON_ENABLED_COLOR, window->winGetEnabledColor( 0 ) );
471 SetControlColor( BUTTON_ENABLED_BORDER_COLOR, window->winGetEnabledBorderColor( 0 ) );
472 SetControlColor( BUTTON_DISABLED_COLOR, window->winGetDisabledColor( 0 ) );
473 SetControlColor( BUTTON_DISABLED_BORDER_COLOR, window->winGetDisabledBorderColor( 0 ) );
474 SetControlColor( BUTTON_HILITE_COLOR, window->winGetHiliteColor( 0 ) );
475 SetControlColor( BUTTON_HILITE_BORDER_COLOR, window->winGetHiliteBorderColor( 0 ) );
476
477 // load the combo boxes with the callbacks the user can use
478 InitCallbackCombos( dialog, window );
479
480 return dialog;
481
482} // end InitUserWinPropertiesDialog
