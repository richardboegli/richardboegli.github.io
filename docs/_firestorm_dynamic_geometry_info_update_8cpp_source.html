119UpdateSleepTime FirestormDynamicGeometryInfoUpdate::update( void )
120{
121
122 // extend functionality
123 DynamicGeometryInfoUpdate::update();
124
125 // get out of here if we haven't fired yet
126 if( m_startingDelayCountdown > 0 )
127 return UPDATE_SLEEP_NONE;
128
129 // get the module data
130 const FirestormDynamicGeometryInfoUpdateModuleData *modData = getFirestormDynamicGeometryInfoUpdateModuleData();
131
132 // start effects for the first time
133 if( m_effectsFired == FALSE )
134 {
135 ParticleSystem *sys;
136
137 //
138 // construct a position to play the particle effects ... that is on the ground, but
139 // note that we add a small amount to it to avoid particles from the system (especially
140 // flat XY particles) from popping up through the terrain
141 //
142 Coord3D pos;
143 pos.x = getObject()->getPosition()->x;
144 pos.y = getObject()->getPosition()->y;
145 pos.z = modData->m_particleOffsetZ + TheTerrainLogic->getGroundHeight( pos.x, pos.y );
146
147 // do the particle systems
148 for( Int i = 0; i < MAX_FIRESTORM_SYSTEMS; i++ )
149 {
150
151 sys = TheParticleSystemManager->createParticleSystem( modData->m_particleSystem[ i ] );
152 if( sys )
153 {
154
155 sys->setPosition( &pos );
156 m_myParticleSystemID[ i ] = sys->getSystemID();
157
158 } // end if
159
160 } // end for i
161
162 // do the FX list
163 FXList::doFXObj( modData->m_fxList, getObject() );
164
165 // effects have been fired
166 m_effectsFired = TRUE;
167
168 getObject()->getControllingPlayer()->getAcademyStats()->recordFirestormCreated();
169
170 }
171
172 //
173 // Parent goes first, and we use part of what he did to alter our bound particle system
174 // get and update our running particle system if we have one
175 //
176 for( Int i = 0; i < MAX_FIRESTORM_SYSTEMS; i++ )
177 {
178
179 if( m_myParticleSystemID[ i ] )
180 {
181 ParticleSystem *sys = TheParticleSystemManager->findParticleSystem( m_myParticleSystemID[ i ] );
182
183 if( sys )
184 {
185 ParticleSystemInfo::EmissionVolumeType type = sys->getEmisionVolumeType();
186
187 if( type == ParticleSystemInfo::EmissionVolumeType::SPHERE )
188 sys->setEmissionVolumeSphereRadius( getObject()->getGeometryInfo().getMajorRadius() );
189 else if( type == ParticleSystemInfo::EmissionVolumeType::CYLINDER )
190 sys->setEmissionVolumeCylinderRadius( getObject()->getGeometryInfo().getMajorRadius() );
191
192 } // end if
193 else
194 {
195
196 // this system not found (it probably died)... stop trying to find it in the future
197 m_myParticleSystemID[ i ] = INVALID_PARTICLE_SYSTEM_ID;
198
199 } // end else
200
201 } // end if
202
203 } // end for, i
204
205 // when we first start running backward ... make a scorch mark
206 if( m_switchedDirections == TRUE && m_scorchPlaced == FALSE )
207 {
208
209 TheGameClient->addScorch( getObject()->getPosition(), modData->m_scorchSize, SCORCH_1 );
210 m_scorchPlaced = TRUE;
211
212 } // end if
213
214 // scan and do some damage every once in a while
215 if( TheGameLogic->getFrame() - m_lastDamageFrame >= modData->m_delayBetweenDamageFrames )
216 {
217
218 doDamageScan();
219 m_lastDamageFrame = TheGameLogic->getFrame();
220
221 } // end if
222
223 return UPDATE_SLEEP_NONE;
224}

228void FirestormDynamicGeometryInfoUpdate::doDamageScan( void )
229{
230
231 // get the module data
232 const FirestormDynamicGeometryInfoUpdateModuleData *modData = getFirestormDynamicGeometryInfoUpdateModuleData();
233
234 // get the object and position
235 Object *firestorm = getObject();
236 const Coord3D *firestormPos = firestorm->getPosition();
237
238 // setup a damage info structure to do some damage
239 DamageInfo damageInfo;
240 damageInfo.in.m_damageType = DAMAGE_FLAME;
241 damageInfo.in.m_deathType = DEATH_BURNED;
242 damageInfo.in.m_sourceID = firestorm->getID();
243 damageInfo.in.m_amount = modData->m_damageAmount;
244
245 // get the current bounding circle size for the firestorm
246 Real boundingCircle = firestorm->getGeometryInfo().getBoundingCircleRadius();
247
248 // scan objects around us and do damage to objects we have "passed over"
249 if( boundingCircle )
250 {
251 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( firestormPos,
252 boundingCircle,
253 FROM_BOUNDINGSPHERE_2D,
254 NULL );
255 MemoryPoolObjectHolder hold( iter );
256 Object *other;
257 for( other = iter->first(); other; other = iter->next() )
258 {
259 // it's too high above us. skip it.
260 if (other->getPosition()->z > firestorm->getPosition()->z + modData->m_maxHeightForDamage)
261 continue;
262
263 // do damage
264 other->attemptDamage( &damageInfo );
265
266 } // end for, other
267
268 } // end if, an boundingCircle radius exists
269
270} // end doDamageScan
