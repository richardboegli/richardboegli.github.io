217void RailroadBehavior::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
218{
219 Object *obj = getObject();
220 if ( ! obj)
221 return; //sanity
222 if ( ! other)
223 return; //sanity
224 if ( ! loc)
225 return; //sanity
226 if ( ! normal)
227 return; //sanity
228
229 if (m_waitingInWings)
230 return ;
231 if (m_endOfLine)
232 return ;
233
234 for (BehaviorModule** m = other->getBehaviorModules(); *m; ++m)
235 {
236 CollideModuleInterface* collide = (*m)->getCollide();
237 if (!collide)
238 continue;
239
240 if( collide->isRailroad())
241 {
242
243 if ( m_isLocomotive )//yikes! I just rear ended a stranded train! Oh well...
244 {
245 other->kill();
246 }
247 else if ( m_isLeadCarraige ) //yikes! I just coasted into some other carriage
248 {
249 other->kill();
250 obj->kill();//and I am dead, too
251 }
252
253 return ;
254 }
255
256 }
257
258 if (other->isKindOf( KINDOF_STRUCTURE ) )// now be careful here, we ignore buildings, except for hostile, nasty ones that can blow us up
259 {
260 //If it is a civilian building like a tunnel or a train station, let it be
261 //but if it is a faction building, kill it!
262 if (other->isKindOf( KINDOF_FS_POWER ) ||
263 other->isKindOf( KINDOF_FS_FACTORY ) ||
264 other->isKindOf( KINDOF_FS_BASE_DEFENSE ) ||
265 other->isKindOf( KINDOF_FS_TECHNOLOGY ) ||
266 other->isKindOf( KINDOF_REBUILD_HOLE ) )
267 {
268 playImpactSound(other, other->getPosition());
269 other->kill();
270 return;
271 }
272
273 static NameKeyType key_DemoTrapUpdate = NAMEKEY("DemoTrapUpdate");
274 DemoTrapUpdate *dtu = (DemoTrapUpdate*)other->findUpdateModule(key_DemoTrapUpdate);
275 if( dtu )
276 {
277 if( !other->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
278 obj->kill(); // it can only detonate on me if it is ready
279
280 playImpactSound(other, other->getPosition());
281 other->kill();
282 return;
283 }
284 }
285
286
287 PhysicsBehavior *theirPhys = other->getPhysics();
288 if( ! theirPhys )
289 return;
290
291 // maybe we don't want to trample this unit?
292 const RailroadBehaviorModuleData *modData = getRailroadBehaviorModuleData();
293 if ( m_conductorState == WAIT_AT_STATION && (m_pullInfo.speed < modData->m_runningGarrisonSpeedMax) ) // they can grab on safely
294 {
295 AIUpdateInterface *ai = other->getAI();
296 if (ai && ai->getEnterTarget() == obj) // other intends to garrison me.
297 {
298 return;
299 }
300 }
301
302 if (other->getContainedBy() == getObject() )//I dont mash my own sweet passengers
303 return;
304
305
306 //if ( other->getModelConditionFlags().test( MODELCONDITION_RUBBLE ) == TRUE )
307 // return; // I don't mess with rubble
308
309 Bool victimIsInfantry = other->isKindOf( KINDOF_INFANTRY );
310
311 const Coord3D *myDir = obj->getUnitDirectionVector2D();
312 const Coord3D *myLoc = obj->getPosition();
313 const Coord3D *theirLoc = other->getPosition();
314
315
316 //---------------------------
317 // if we made it this far, it is something we dont want to share space with
318
319
320 Coord3D dlt;
321 dlt.x = theirLoc->x - myLoc->x;
322 dlt.y = theirLoc->y - myLoc->y;
323 dlt.z = theirLoc->z - myLoc->z;
324
325 //Alert all the players of recent disaster
326 if ( ! m_whistleSound.isCurrentlyPlaying())
327 m_whistleSound.setPlayingHandle(TheAudio->addAudioEvent( &m_whistleSound ));
328
329
330 Real dist = (Real)sqrtf( dlt.x*dlt.x + dlt.y*dlt.y + dlt.z*dlt.z);
331 Real usRadius = obj->getGeometryInfo().getMajorRadius();
332 Real themRadius = other->getGeometryInfo().getMajorRadius();
333 Real overlap = ((usRadius + themRadius) - dist) + 1;// the plus 1 makes them go just outside of me.
334 dlt.normalize();
335 if ( ! victimIsInfantry )
336 {
337 overlap/= 4;
338
339 dlt.scale( overlap);
340 Coord3D newPos;
341 newPos.x = theirLoc->x + dlt.x;
342 newPos.y = theirLoc->y + dlt.y;
343 newPos.z = theirLoc->z + dlt.z;
344 other->setPosition( &newPos );
345 }
346
347 if ( m_conductorState == WAIT_AT_STATION || (m_conductorState == COAST && m_pullInfo.speed < modData->m_runningGarrisonSpeedMax) || !m_isLocomotive )
348 {
349// AIUpdateInterface *ai = other->getAI();
350// if ( ai )
351// ai->aiIdle( CMD_FROM_AI );// this eliminates yadda by telling them to stop driving into me
352
353 return;//let those trying to board pass through unhindered
354 }
355
356
357
358
359
360 //figure out the relative slope between them and me
361 Coord3D delta = *theirLoc;
362 delta.sub( myLoc );
363 delta.normalize();
364 Real dot = delta.x * myDir->x + delta.y * myDir->y + delta.z * myDir->z;
365
366 if (other->isEffectivelyDead())
367 {// we just run over debris, instead of shoving it around
368 delta.scale( MIN(0.3f,m_pullInfo.speed * 0.66f) );
369 }
370 else
371 {
372 delta.scale( MIN(1.4f,m_pullInfo.speed * 0.66f) );// the faster I go, the harder I slam!
373
374 //Absolute death to be hit by a train, no survival
375 if ( m_pullInfo.speed >= modData->m_killSpeedMin ) // they can grab on safely
376 {
377 other->kill();
378 theirPhys->setPitchRate(GameLogicRandomValueReal(-0.03f, 0.03f));
379 theirPhys->setRollRate(GameLogicRandomValueReal(-0.03f, 0.03f));
380 }
381 else
382 {
383 playImpactSound(other, loc); //Play a bounce sound
384
385 DamageInfo damageInfo;
386 damageInfo.in.m_damageType = DAMAGE_CRUSH;
387 damageInfo.in.m_deathType = DEATH_CRUSHED;
388 damageInfo.in.m_sourceID = getObject()->getID();
389 damageInfo.in.m_amount = m_pullInfo.speed * 10; // hurt!
390 other->attemptDamage( &damageInfo );
391 }
392 }
393
394 Coord3D heft = *theirLoc;
395 heft.z = MAX(heft.z, TheTerrainLogic->getGroundHeight(heft.x, heft.y) + 2); // lift them off the ground
396 other->setPosition(&heft);
397
398 delta.z = GameLogicRandomValueReal(0.05f, m_pullInfo.speed/10); // for some fake heft
399 delta.scale(dot);
400
401
402 // This is a special check so that it wont hurl infantry clear across the map!
403 if( ! ( victimIsInfantry && theirPhys->getVelocityMagnitude() > 5.0f ) )
404 theirPhys->addVelocityTo( &delta );
405
406
407
408 theirPhys->setAllowToFall( TRUE );
409 theirPhys->setAllowBouncing( TRUE );
410 theirPhys->setAllowAirborneFriction( TRUE );
411
412
413 const Coord3D up = {0,0,1};
414 Coord3D cross;
415 myDir->crossProduct( myDir, &up, &cross );
416
417 delta.normalize();
418 Real deviationCOG = cross.x * delta.x + cross.y * delta.y + cross.z * delta.z;
419
420 if (dot > 0)
421 theirPhys->setYawRate( deviationCOG * -0.06f * m_pullInfo.speed);
422
423
424
425
426} // end RailroadBehavior:: on collide

432void RailroadBehavior::playImpactSound(Object *victim, const Coord3D *impactPosition)
433{
434 AudioEventRTS impact;
435 //AudioEventRTS *pImpact = &impact;
436 PhysicsBehavior *theirPhys = victim->getPhysics();
437
438 Bool hasBounceSound = FALSE;
439
440 if ( theirPhys )
441 {
442 const AudioEventRTS* bsound = theirPhys->getBounceSound();
443 if (bsound)
444 {
445 impact = *bsound;//copy theirs
446 hasBounceSound = TRUE;
447 }
448 }
449
450 if ( ! hasBounceSound )
451 {
452 const RailroadBehaviorModuleData *modData = getRailroadBehaviorModuleData();
453 if (victim->isKindOf( KINDOF_INFANTRY ))
454 {
455 impact = modData->m_meatyImpactDefaultSound;//copy
456 }
457 else if (victim->isKindOf( KINDOF_HUGE_VEHICLE ) || victim->isKindOf( KINDOF_STRUCTURE ))
458 {
459 impact = modData->m_bigMetalImpactDefaultSound;//copy
460 }
461 else if (victim->isKindOf( KINDOF_VEHICLE ) )
462 {
463 impact = modData->m_smallMetalImpactDefaultSound;//copy
464 }
465 }
466
467
468 if (impact.getEventName().isEmpty())
469 return;
470
471 Real vel = NORMAL_VEL_Z; //the max
472 Real mass = NORMAL_MASS; //the max
473
474 impact.setPosition(impactPosition);
475 if ( theirPhys )
476 {
477 vel += fabs(theirPhys->getVelocity()->length());
478 mass += fabs(theirPhys->getMass());
479
480 vel /= 2;
481 mass /= 2;//average of him and me
482 impact.setPosition(victim->getPosition());
483 }
484
485 vel = MIN(NORMAL_VEL_Z, MAX(0, vel));
486 mass = MIN(NORMAL_MASS, MAX(0, mass));
487
488 Real volAdjust = NormalizeToRange(MuLaw(vel, NORMAL_VEL_Z, 500), -1, 1, 0.25, 1.0);
489 volAdjust *= NormalizeToRange(MuLaw(mass, NORMAL_MASS, 500), -1, 1, 0.25, 1.0);
490 impact.setVolume( volAdjust );
491
492 // This sound really should only be played at the position where the collision occurs.
493 // Therefore, set the player index of the sound to be the player index of the victim object.
494 impact.setPlayerIndex( victim->getControllingPlayer()->getPlayerIndex() );
495
496 TheAudio->addAudioEvent( &impact );
497}

504void RailroadBehavior::loadTrackData( void )
505{
506
507 if ( m_track != NULL )
508 return;// lets do this only once!
509
510
511 //First we scan the map for the nearest waypoint
512 Object *obj = getObject();
513 const Coord3D *myPos = obj->getPosition();
514 Coord3D delta;
515 Real closestDistance = 99999.9f;
516
517
518
519 // m_anchorWaypointID COULD HAVE BEEN RECORDED IN XFER, WHICH MEANS I LOADED MY TRACK DATA IN A PRIOR LIFE,
520 // SO LETS JUST RE_INIT THE TRACK BASED ON THAT POINT, AUTOMAGICALLY
521 Waypoint *anchorWaypoint = NULL;
522 if ( m_anchorWaypointID == INVALID_WAYPOINT_ID )
523 {
524 Waypoint *anyWaypoint = TheTerrainLogic->getFirstWaypoint();
525
526 while ( anyWaypoint )
527 {
528 //measure the distance from me to the waypoint
529 delta.x = myPos->x - anyWaypoint->getLocation()->x;
530 delta.y = myPos->y - anyWaypoint->getLocation()->y;
531 delta.z = myPos->z - anyWaypoint->getLocation()->z;
532 if (closestDistance > delta.length() )
533 {
534 closestDistance = delta.length();
535 anchorWaypoint = anyWaypoint;
536 m_anchorWaypointID = anchorWaypoint->getID();// save for XFER, later
537 }
538
539 anyWaypoint = anyWaypoint->getNext();
540 }
541 }
542 else
543 {
544 anchorWaypoint = TheTerrainLogic->getWaypointByID( m_anchorWaypointID );
545 }
546
547
548
549 DEBUG_ASSERTCRASH( anchorWaypoint, ( "Railroad... couldn't find a waypoint close enough to be the anchor.\n You should put your train engine right on a waypoint,\n and make sure the path forms a valid track.") );
550 if ( ! anchorWaypoint )
551 return;
552
553
554
555
556
557
558 m_track = NEW( TrainTrack );// this constructor inc's the refcount to 1
559
560 // From now until the next carriage is added, this track is writable using getWritablePointList();
561 // This method will return NULL when refcount is 2 or more
562 // getPointList returns the list as const to any caller
563 // each carriage must increment the reference when this pointer is passed to it,
564 // any subsequent carriage (destructor) will delete this memory here if it releases refcount to zero
565
566 Coord3D fromPos, toPos, fromToDelta;
567 m_track->m_length = 0.0f;
568 Waypoint *scanner = anchorWaypoint;
569 Real distFromTo = 0.0f;
570
571
572 //Let's start buliding our own track data from the waypoint data we find
573 TrackPointList* track = m_track->getWritablePointList();
574 TrackPoint trackPoint; // local workspace
575
576 if ( scanner && track)
577 {
578 trackPoint.m_distanceFromPrev = 0;
579 trackPoint.m_distanceFromFirst = 0;
580 trackPoint.m_isFirstPoint = TRUE;
581 trackPoint.m_isLastPoint = FALSE;
582 trackPoint.m_isTunnelOrBridge = scanner->getName().endsWith("Tunnel");
583 trackPoint.m_isStation = scanner->getName().endsWith("Station");
584 trackPoint.m_isDisembark = scanner->getName().endsWith("Disembark");
585 trackPoint.m_isPingPong = FALSE;
586 trackPoint.m_position.set( scanner->getLocation() );
587 trackPoint.m_handle = scanner->getID();
588 track->push_back( trackPoint );
589 }
590
591 while ( scanner )
592 {
593 trackPoint.clear();// clean up the local workspace to make debugging more fun
594
595 if ( scanner->getNumLinks() )
596 {
597 Waypoint *anotherWaypoint = scanner->getLink( 0 );
598
599 // if scanner's link is valid, we'll add it to the track. now
600 if ( anotherWaypoint != NULL )
601 {
602
603 //measure the track while we are at it
604 fromPos = *scanner->getLocation();
605 toPos = *anotherWaypoint->getLocation();
606 fromToDelta.x = fromPos.x - toPos.x;
607 fromToDelta.y = fromPos.y - toPos.y;
608 fromToDelta.z = fromPos.z - toPos.z;
609 distFromTo = fromToDelta.length();
610 m_track->m_length += distFromTo;
611
612 trackPoint.m_distanceFromPrev = distFromTo;
613 trackPoint.m_distanceFromFirst = m_track->m_length;
614 trackPoint.m_isFirstPoint = FALSE;
615 trackPoint.m_isLastPoint = anotherWaypoint->getLink( 0 ) == NULL;
616 trackPoint.m_isTunnelOrBridge = anotherWaypoint->getName().endsWith("Tunnel");
617 trackPoint.m_isStation = anotherWaypoint->getName().endsWith("Station");
618 trackPoint.m_isPingPong = scanner->getName().endsWith("PingPong");
619 trackPoint.m_isDisembark = scanner->getName().endsWith("Disembark");
620 trackPoint.m_position.set( anotherWaypoint->getLocation() );
621 trackPoint.m_handle = scanner->getID();
622 track->push_back( trackPoint );
623
624 }
625
626 scanner = anotherWaypoint;
627
628 }
629 else
630 break; // since we have reached a non-linked waypoint
631
632
633 if ( scanner == anchorWaypoint )
634 {
635 m_track->m_isLooping = TRUE;
636 break; // it must be a looping track. Cool.
637 }
638 }
639
640} // end loadTrackData

675// %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
676UpdateSleepTime RailroadBehavior::update( void )
677{
678
679
680 // load the waypoint data if not loaded
681 if( m_trackDataLoaded == FALSE && m_isLocomotive )
682 {
683
684 loadTrackData();
685
686 if ( m_track )
687 createCarriages();
688
689 m_trackDataLoaded = TRUE;
690 }
691
692
693
694
695 if ( ! m_track )
696 {
697 return UPDATE_SLEEP_NONE;
698 }
699
700
701 //Object *us = getObject();
702 const RailroadBehaviorModuleData *modData = getRailroadBehaviorModuleData();
703
704 if ( m_isLocomotive )
705 {
706
707 if ( m_conductorState == APPLY_BRAKES )
708 {
709 conductorPullInfo.speed *= modData->m_braking;
710 if (fabs(conductorPullInfo.speed) < 0.1f)
711 {
712 conductorPullInfo.speed = 0;
714
715 m_waitAtStationTimer = modData->m_waitAtStationTime;
716 m_conductorState = WAIT_AT_STATION;
717
718
719 makeAWallOutOfThisTrain( TRUE );
720
721
722 if ( m_disembark )
723 {
724 disembark();
725 m_disembark = FALSE;
726 }
727 }
728 }
729 else if ( m_conductorState == WAIT_AT_STATION)
730 {
731 --m_waitAtStationTimer;
732 if ( m_waitAtStationTimer <= 0 && (!m_held) )
733 {
734 m_conductorState = ACCELERATE;
735 conductorPullInfo.speed = 0.05f * conductorPullInfo.m_direction;
736
737 m_runningSound.setPlayingHandle(TheAudio->addAudioEvent( &m_runningSound ));
738
739 makeAWallOutOfThisTrain( FALSE );
740
741
742 }
743 else if ( m_waitAtStationTimer == (modData->m_waitAtStationTime/4) )
744 {
745 m_whistleSound.setPlayingHandle(TheAudio->addAudioEvent( &m_whistleSound ));
746 }
747
748 }
749 else if ( m_conductorState == ACCELERATE )
750 {
751 conductorPullInfo.speed += 0.02f * conductorPullInfo.m_direction; // push start multiplier
752 conductorPullInfo.speed *= modData->m_acceleration;
753 if ( conductorPullInfo.speed > modData->m_speedMax)
754 {
755 conductorPullInfo.speed = modData->m_speedMax;
756 }
757 else if ( conductorPullInfo.speed < -modData->m_speedMax)
758 {
759 conductorPullInfo.speed = -modData->m_speedMax;
760 }
761
762
763 if ( ! m_runningSound.isCurrentlyPlaying() )
764 m_runningSound.setPlayingHandle(TheAudio->addAudioEvent( &m_runningSound ));

1236 x.z = 0.0f;
1237 if (z.z != 0.0f)

1246 y.crossProduct( &z, &x, &y );
1247 y.normalize();
1248

1250 x.y, y.y, z.y, pos.y,
1251 x.z, y.z, z.z, pos.z );

1379void RailroadBehavior::FindPosByPathDistance( Coord3D *pos, const Real dist, const Real length, Bool setState )
1380{
1381
1382 if ( ! m_track )
1383 return;
1384
1385 m_waitingInWings = FALSE;
1386
1387 Real actualDistance = dist;// WRAP TO NORMALIZE
1388
1389 if ( m_track->m_isLooping )
1390 {
1391 while ( actualDistance < 0.0f )
1392 {
1393 actualDistance += length;
1394 }
1395 while ( actualDistance > length )
1396 {
1397 actualDistance -= length;
1398 }
1399 }
1400 else
1401 {
1402
1403 if ( dist < 0 )
1404 {
1405 actualDistance = 0;
1406 m_waitingInWings = TRUE;
1407 }
1408 else if ( dist >= length )
1409 {
1410 actualDistance = length;
1411 m_endOfLine = TRUE;
1412 }
1413
1414 actualDistance = MAX(MIN(length,dist),0); // CLAMP TO NORMALIZE
1415 }
1416
1417 pos->set( 0.0f, 0.0f, 0.0f );
1418
1419 const TrackPointList *pointList = m_track->getPointList();//read-only
1420 if ( ! pointList )
1421 return;
1422
1423 std::list<TrackPoint>::const_iterator pointIter = pointList->begin();
1424
1425
1426 while ( pointIter != pointList->end() )
1427 {
1428 const TrackPoint *thisPoint = &(*pointIter);
1429 ++pointIter;// next pointIter in this list, so then...
1430 const TrackPoint *nextPoint = &(*pointIter);
1431
1432
1433 if (thisPoint && thisPoint->m_distanceFromFirst < actualDistance)// I am after this point, and
1434 {
1435 Coord3D thisPointPos = thisPoint->m_position;
1436
1437 if (nextPoint && nextPoint->m_distanceFromFirst > actualDistance)
1438 {
1439 //I am between this point and the next
1440 const Int handleFound = ((TrackPoint*)thisPoint)->getHandle();
1441 Bool edge = (m_currentPointHandle != handleFound);
1442 if ( setState )
1443 {
1444 m_inTunnel = thisPoint->m_isTunnelOrBridge;
1445
1446 if ( m_isLocomotive )// since only locomotives care about such things
1447 {
1448 // THE LOCOMOTIVE SNIFFS THE TRACK TO SMELL FOR THE NEXT STATION AND TUNNEL
1449 if ( edge )
1450 {
1451 m_currentPointHandle = handleFound;
1452 if ( thisPoint->m_isStation )
1453 {
1454 m_conductorState = APPLY_BRAKES;
1455 m_disembark = FALSE;
1456 TheAudio->removeAudioEvent(m_runningSound.getPlayingHandle());
1457 }
1458 else if ( thisPoint->m_isDisembark )
1459 {
1460 m_conductorState = APPLY_BRAKES;
1461 m_disembark = TRUE;
1462 TheAudio->removeAudioEvent(m_runningSound.getPlayingHandle());
1463 }
1464 else if ( thisPoint->m_isPingPong && conductorPullInfo.m_mostRecentSpecialPointHandle != handleFound )
1465 {
1466 conductorPullInfo.m_mostRecentSpecialPointHandle = handleFound;
1467 m_conductorState = APPLY_BRAKES;
1468 m_disembark = FALSE;
1469 TheAudio->removeAudioEvent(m_runningSound.getPlayingHandle());
1470 conductorPullInfo.m_direction = -conductorPullInfo.m_direction;
1471 }
1472 }
1473 }
1474
1475 if ( edge && ! m_inTunnel )
1476 {//play my clickety clack sound, `cause I just rode over a join IN the tracks
1477 TheAudio->addAudioEvent( &m_clicketyClackSound );
1478 m_clicketyClackSound.setPosition( getObject()->getPosition() );
1479 m_clicketyClackSound.setVolume( (Real)conductorPullInfo.speed / 10.0f );//assumed max speed
1480 }
1481 }
1482
1483
1484
1485 Real difference = actualDistance - thisPoint->m_distanceFromFirst;
1486
1487 Coord3D delta = nextPoint->m_position;
1488
1489 delta.sub( &thisPointPos );
1490 delta.normalize();
1491 delta.scale( difference );
1492 thisPointPos.add( &delta );
1493
1494 *pos = thisPointPos;//copy out
1495 return;
1496 }
1497 else
1498 *pos = thisPointPos;//copy out
1499
1500 }
1501
1502 }
