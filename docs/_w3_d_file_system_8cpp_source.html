140char const * GameFileClass::Set_Name( char const *filename )
141{
142
143 if( Is_Open() )
144 Close();
145
146 // save the filename
147 strncpy( m_filename, filename, _MAX_PATH );
148
149 char name[_MAX_PATH];
150 const Int EXT_LEN = 32;
151 char extension[EXT_LEN];
152 extension[0] = 0;
153 strcpy(name, filename);
154 Int i = strlen(name);
155 i--;
156 Int extLen = 1;
157 while(i>0 && extLen < EXT_LEN) {
158 if (name[i] == '.') {
159 strcpy(extension, name+i);
160 name[i] = 0;
161 break;
162 }
163 i--;
164 extLen++;
165 }
166 Int j = 0;
167 // Strip out spaces.
168 for (i=0; name[i]; i++) {
169 if (name[i] != ' ') {
170 name[j] = name[i];
171 j++;
172 }
173 }
174 name[j] = 0;
175
176 // test the extension to recognize a few key file types
177 GameFileType fileType = FILE_TYPE_COMPLETELY_UNKNOWN; // MBL FILE_TYPE_UNKNOWN change due to compile error
178 if( stricmp( extension, ".w3d" ) == 0 )
179 fileType = FILE_TYPE_W3D;
180 else if( stricmp( extension, ".tga" ) == 0 )
181 fileType = FILE_TYPE_TGA;
182 else if( stricmp( extension, ".dds" ) == 0 )
183 fileType = FILE_TYPE_DDS;
184
185
186
187 // We need to be able to grab w3d's from a localization dir, since Germany hates exploding people units.
188 if( fileType == FILE_TYPE_W3D )
189 {
190 static const char *localizedPathFormat = "Data/%s/Art/W3D/";
191 sprintf(m_filePath,localizedPathFormat, GetRegistryLanguage().str());
192 strcat( m_filePath, filename );
193
194 } // end if
195
196 // We need to be able to grab images from a localization dir, because Art has a fetish for baked-in text. Munkee.
197 if( isImageFileType(fileType) )
198 {
199 static const char *localizedPathFormat = "Data/%s/Art/Textures/";
200 sprintf(m_filePath,localizedPathFormat, GetRegistryLanguage().str());
201 strcat( m_filePath, filename );
202
203 } // end else if
204
205 // see if the file exists
206 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
207
208
209
210 // Now try the main lookup of hitting local files and big files
211 if( m_fileExists == FALSE )
212 {
213 // all .w3d files are in W3D_DIR_PATH, all .tga files are in TGA_DIR_PATH
214 if( fileType == FILE_TYPE_W3D )
215 {
216
217 strcpy( m_filePath, W3D_DIR_PATH );
218 strcat( m_filePath, filename );
219
220 } // end if
221 else if( isImageFileType(fileType) )
222 {
223
224 strcpy( m_filePath, TGA_DIR_PATH );
225 strcat( m_filePath, filename );
226
227 } // end else if
228 else
229 strcpy( m_filePath, filename );
230
231 // see if the file exists
232 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
233 }
234
235
236
237 // maintain legacy compatibility directories for now
238 #ifdef MAINTAIN_LEGACY_FILES
239 if( m_fileExists == FALSE )
240 {
241
242 if( fileType == FILE_TYPE_W3D )
243 {
244
245 strcpy( m_filePath, LEGACY_W3D_DIR_PATH );
246 strcat( m_filePath, filename );
247
248 } // end if
249 else if( isImageFileType(fileType) )
250 {
251
252 strcpy( m_filePath, LEGACY_TGA_DIR_PATH );
253 strcat( m_filePath, filename );
254
255 } // end else if
256
257 // see if the file exists
258 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
259
260 } // end if
261 #endif
262
263
264
265 // if file is still not found, try the test art folders
266 #ifdef LOAD_TEST_ASSETS
267 if( m_fileExists == FALSE )
268 {
269
270 if( fileType == FILE_TYPE_W3D )
271 {
272
273 strcpy( m_filePath, TEST_W3D_DIR_PATH );
274 strcat( m_filePath, filename );
275
276 } // end if
277 else if( isImageFileType(fileType) )
278 {
279
280 strcpy( m_filePath, TEST_TGA_DIR_PATH );
281 strcat( m_filePath, filename );
282
283 } // end else if
284
285 // see if the file exists
286 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
287
288 } // end if
289 #endif
290
291 // We allow the user to load their own images for various assets (like the control bar)
292 if( m_fileExists == FALSE && TheGlobalData)
293 {
294 if( fileType == FILE_TYPE_W3D )
295 {
296 sprintf(m_filePath,USER_W3D_DIR_PATH, TheGlobalData->getPath_UserData().str());
297 //strcpy( m_filePath, USER_W3D_DIR_PATH );
298 strcat( m_filePath, filename );
299
300 } // end if
301 if( isImageFileType(fileType) )
302 {
303 sprintf(m_filePath,USER_TGA_DIR_PATH, TheGlobalData->getPath_UserData().str());
304 //strcpy( m_filePath, USER_TGA_DIR_PATH );
305 strcat( m_filePath, filename );
306
307 } // end else if
308
309 // see if the file exists
310 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
311
312 } // end if
313
314
315 // We Need to be able to "temporarily copy over the map preview for whichever directory it came from
316 if( m_fileExists == FALSE && TheGlobalData)
317 {
318 if( fileType == FILE_TYPE_TGA ) // just TGA, since we don't dds previews
319 {
320 sprintf(m_filePath,MAP_PREVIEW_DIR_PATH, TheGlobalData->getPath_UserData().str());
321 //strcpy( m_filePath, USER_TGA_DIR_PATH );
322 strcat( m_filePath, filename );
323
324 } // end else if
325
326 // see if the file exists
327 m_fileExists = TheFileSystem->doesFileExist( m_filePath );
328
329 } // end if
330
331 return m_filename;
332
333}

397int GameFileClass::Seek(int pos, int dir)
398{
399 File::seekMode mode = File::CURRENT;
400 switch (dir) {
401 default:
402 case SEEK_CUR: mode = File::CURRENT; break;
403 case SEEK_SET: mode = File::START; break;
404 case SEEK_END: mode = File::END; break;
405 }
406 if (m_theFile) {
407 return m_theFile->seek(pos, mode);
408 }
409 return 0xFFFFFFFF;
410}
