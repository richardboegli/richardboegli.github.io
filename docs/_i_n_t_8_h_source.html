203template<class T>
204struct RemainderTable
205{
206 RemainderTable(const T & p) : HasZeroEntry(false)
207 {
208 int primesize = XMP_Fetch_Prime_Size();
209 unsigned short const * primetable = XMP_Fetch_Prime_Table();
210 for (int i = 0; i < primesize; i++) {
211 table[i] = p % primetable[i];
212 }
213 }
214 bool HasZero() const {return(HasZeroEntry);}
215 void Increment(unsigned short increment = 1)
216 {
217 int primesize = XMP_Fetch_Prime_Size();
218 unsigned short const * primetable = XMP_Fetch_Prime_Table();
219 HasZeroEntry = false;
220 for (int i = 0; i < primesize; i++) {
221 table[i] += increment;
222 while (table[i] >= primetable[i]) {
223 table[i] -= primetable[i];
224 }
225 HasZeroEntry = (HasZeroEntry || !table[i]);
226 }
227 }
228 void Increment(const RemainderTable & rtQ)
229 {
230 HasZeroEntry = false;
231 int primesize = XMP_Fetch_Prime_Size();
232 unsigned short const * primetable = XMP_Fetch_Prime_Table();
233 for (int i = 0; i < primesize; i++) {
234 table[i] += rtQ.table[i];
235 if (table[i] >= primetable[i]) {
236 table[i] -= primetable[i];
237 }
238 HasZeroEntry = (HasZeroEntry || !table[i]);
239 }
240 }
241 bool HasZeroEntry;
242 unsigned short table[3511];
243};

247template<class T>
248T Gcd(const T & a, const T & n)
249{
250 T g[3]={n, a, 0UL};
251
252 unsigned int i = 1;
253 while (!!g[i%3]) {
254 g[(i+1)%3] = g[(i-1)%3] % g[i%3];
255 i++;
256 }
257 return g[(i-1)%3];
258}

268template<class T>
269T Generate_Prime(Straw & rng, int pbits, T const *)
270{
271 T minQ = (T(1UL) << (unsigned short)(pbits-(unsigned short)2));
272 T maxQ = ((T(1UL) << (unsigned short)(pbits-(unsigned short)1)) - (unsigned short)1);
273
274 T q;
275 T p;
276
277 do {
278 q.Randomize(rng, minQ, maxQ);
279 p = (q*2) + (unsigned short)1;
280
281 RemainderTable<T> rtQ(q);
282 RemainderTable<T> rtP(p);
283
284 while (rtQ.HasZero() || rtP.HasZero() || !q.IsPrime() || !p.IsPrime()) {
285 q += 2;
286 p += 4;
287 if (q > maxQ) break;
288
289 rtQ.Increment(2);
290 rtP.Increment(4);
291 }
292 } while (q > maxQ);
293
294 return(p);
295}
