40BOOL CFixTeamOwnerDialog::OnInitDialog()
41{
42 AsciiString teamName = "No Name";
43
44 Bool exists;
45 AsciiString temp = m_ti->getDict()->getAsciiString(TheKey_teamName, &exists);
46 if (exists) {
47 teamName = temp;
48 }
49
50 CString loadStr;
51 loadStr.Format(IDS_REPLACEOWNER, teamName.str());
52 CWnd *pWnd = GetDlgItem(IDC_REPLACETHISTEXT);
53 pWnd->SetWindowText(loadStr);
54
55 // now load all of the things with the other things
56 Int numSides = m_sl->getNumSides();
57 CListBox *pList = (CListBox*) GetDlgItem(IDC_VALIDTEAMLIST);
58
59 for (Int i = 0; i < numSides; ++i) {
60 SidesInfo *si = m_sl->getSideInfo(i);
61 if (!si) {
62 continue;
63 }
64
65 Bool displayExists;
66 AsciiString displayName = si->getDict()->getAsciiString(TheKey_playerDisplayName, &displayExists);
67 if (displayExists) {
68 if (displayName.isEmpty()) {
69 displayName = NEUTRAL_NAME_STR;
70 }
71 pList->InsertString(-1, displayName.str());
72 } else {
73 AsciiString internalName = si->getDict()->getAsciiString(TheKey_playerName, &displayExists);
74 if (internalName.isEmpty()) {
75 internalName = NEUTRAL_NAME_STR;
76 }
77 pList->InsertString(-1, internalName.str());
78 }
79 }
80
81 return FALSE;
82}

84void CFixTeamOwnerDialog::OnOK()
85{
86 CDialog::OnOK();
87
88 CListBox *pList = (CListBox*) GetDlgItem(IDC_VALIDTEAMLIST);
89 int curSel = pList->GetCurSel();
90
91 if (curSel < 0) {
92 return;
93 }
94
95 SidesInfo *si = m_sl->getSideInfo(curSel);
96 if (!si) {
97 return;
98 }
99
100 m_pickedValidTeam = true;
101 Bool exists;
102 m_selectedOwner = si->getDict()->getAsciiString(TheKey_playerName, &exists);
103}
