111Int CashHackSpecialPower::findAmountToSteal() const
112{
113 const CashHackSpecialPowerModuleData* d = getCashHackSpecialPowerModuleData();
114 const Player* controller = getObject()->getControllingPlayer();
115 if (controller != NULL)
116 {
117 for (std::vector<CashHackSpecialPowerModuleData::Upgrades>::const_iterator it = d->m_upgrades.begin();
118 it != d->m_upgrades.end();
119 ++it)
120 {
121 if (controller->hasScience(it->m_science))
122 return it->m_amountToSteal;
123 }
124 }
125 return d->m_defaultAmountToSteal;
126}

128// ------------------------------------------------------------------------------------------------
129// ------------------------------------------------------------------------------------------------
130void CashHackSpecialPower::doSpecialPowerAtObject( Object *victim, UnsignedInt commandOptions )
131{
132 if (getObject()->isDisabled())
133 return;
134
135 // sanity
136 if (!victim)
137 return;
138
139 // call the base class action cause we are *EXTENDING* functionality
140 SpecialPowerModule::doSpecialPowerAtObject( victim, commandOptions );
141
142 // get our module data
143 Object *self = getObject();
144
145 //Steal a thousand cash from the other team!
146 Money *targetMoney = victim->getControllingPlayer()->getMoney();
147 Money *selfMoney = self->getControllingPlayer()->getMoney();
148 if( targetMoney && selfMoney )
149 {
150 UnsignedInt cash = targetMoney->countMoney();
151 UnsignedInt desiredAmount = findAmountToSteal();
152 //Check to see if they have 1000 cash, otherwise, take the remainder!
153 cash = min( desiredAmount, cash );
154 if( cash > 0 )
155 {
156 //Steal the cash
157 targetMoney->withdraw( cash );
158 selfMoney->deposit( cash );
159 self->getControllingPlayer()->getScoreKeeper()->addMoneyEarned( cash );
160
161 //Display cash income floating over the blacklotus
162 UnicodeString moneyString;
163 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), cash );
164 Coord3D pos;
165 pos.zero();
166 pos.add( self->getPosition() );
167 pos.z += 20.0f; //add a little z to make it show up above the unit.
168 TheInGameUI->addFloatingText( moneyString, &pos, GameMakeColor( 0, 255, 0, 255 ) );
169
170 //Display cash lost floating over the target
171 moneyString.format( TheGameText->fetch( "GUI:LoseCash" ), cash );
172 pos.zero();
173 pos.add( victim->getPosition() );
174 pos.z += 30.0f; //add a little z to make it show up above the unit.
175 TheInGameUI->addFloatingText( moneyString, &pos, GameMakeColor( 255, 0, 0, 255 ) );
176 }
177 }
178
179}
