64void Image::parseImageCoords( INI* ini, void *instance, void *store, const void* /*userData*/ )
65{
66 Int left = INI::scanInt(ini->getNextSubToken("Left"));
67 Int top = INI::scanInt(ini->getNextSubToken("Top"));
68 Int right = INI::scanInt(ini->getNextSubToken("Right"));
69 Int bottom = INI::scanInt(ini->getNextSubToken("Bottom"));
70
71 // get the image we're storing in
72 Image *theImage = (Image *)instance;
73
74 //
75 // store the UV coords based on what we've read in and the texture size
76 // defined for this image
77 //
78 Region2D uvCoords;
79
80 uvCoords.lo.x = (Real)left;
81 uvCoords.lo.y = (Real)top;
82 uvCoords.hi.x = (Real)right;
83 uvCoords.hi.y = (Real)bottom;
84
85 // adjust the coords by texture size
86 const ICoord2D *textureSize = theImage->getTextureSize();
87 if( textureSize->x )
88 {
89 uvCoords.lo.x /= (Real)textureSize->x;
90 uvCoords.hi.x /= (Real)textureSize->x;
91 } // end if
92 if( textureSize->y )
93 {
94 uvCoords.lo.y /= (Real)textureSize->y;
95 uvCoords.hi.y /= (Real)textureSize->y;
96 } // end if
97
98 // store the uv coords
99 theImage->setUV( &uvCoords );
100
101 // compute the image size based on the coords we read and store
102 ICoord2D imageSize;
103 imageSize.x = right - left;
104 imageSize.y = bottom - top;
105 theImage->setImageSize( &imageSize );
106
107} // end parseImageCoord

112void Image::parseImageStatus( INI* ini, void *instance, void *store, const void* /*userData*/)
113{
114 // use existing INI parsing for the bit strings
115 INI::parseBitString32(ini, instance, store, imageStatusNames);
116
117 //
118 // if we are rotated 90 degrees clockwise we need to swap our width and height as
119 // they were computed from the page location rect, which was for the rotated image
120 // (see ImagePacker tool for more details)
121 //
122 UnsignedInt *theStatusBits = (UnsignedInt *)store;
123 if( BitTest( *theStatusBits, IMAGE_STATUS_ROTATED_90_CLOCKWISE ) )
124 {
125 Image *theImage = (Image *)instance;
126 ICoord2D imageSize;
127
128 imageSize.x = theImage->getImageHeight(); // note it's height not width
129 imageSize.y = theImage->getImageWidth(); // note it's width not height
130 theImage->setImageSize( &imageSize );
131
132 } // end if
133
134} // end parseImageStatus

232void ImageCollection::load( Int textureSize )
233{
234 char buffer[ _MAX_PATH ];
235 INI ini;
236 // first load in the user created mapped image files if we have them.
237 WIN32_FIND_DATA findData;
238 AsciiString userDataPath;
239 if(TheGlobalData)
240 {
241 userDataPath.format("%sINI\\MappedImages\\*.ini",TheGlobalData->getPath_UserData().str());
242 if(FindFirstFile(userDataPath.str(), &findData) !=INVALID_HANDLE_VALUE)
243 {
244 userDataPath.format("%sINI\\MappedImages",TheGlobalData->getPath_UserData().str());
245 ini.loadDirectory(userDataPath, TRUE, INI_LOAD_OVERWRITE, NULL );
246 }
247 }
248
249 // construct path to the mapped images folder of the correct texture size
250 sprintf( buffer, "Data\\INI\\MappedImages\\TextureSize_%d", textureSize );
251
252 // load all the ine files in that directory
253
254 ini.loadDirectory( AsciiString( buffer ), TRUE, INI_LOAD_OVERWRITE, NULL );
255
256 ini.loadDirectory("Data\\INI\\MappedImages\\HandCreated", TRUE, INI_LOAD_OVERWRITE, NULL );
257

259} // end load
