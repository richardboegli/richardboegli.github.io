34/*
35 In order for something to live in an OVERRIDE<> object, it must be derived from Overridable
36 (publicly).
37
38 This is useful for things like templates, where we want to override the template and make sure
39 that all instances get the updated values (for instance, via map.ini)
40*/
41
42class Overridable : public MemoryPoolObject
43{
44 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE( Overridable, "Overridable" )
45
46 private:
47 Overridable *m_nextOverride;
48 Bool m_isOverride;
49
50 public:
51 Overridable() : m_nextOverride(NULL), m_isOverride(false) {}
52
53 // return a constant version of m_nextOverride, which can be NULL if there is no
54 // override
55 const Overridable *getNextOverride( void ) const
56 {
57 return m_nextOverride;
58 }
59
60 // recursively ask if there is a next override, and if not, return this.
61 const Overridable *getFinalOverride( void ) const
62 {
63 if (m_nextOverride)
64 return m_nextOverride->getFinalOverride();
65 return this;
66 }
67
68 // set the next override on this object. This currently makes no attempt to prevent leaks.
69 // it probably shouldn't because doing so could cause infinite loops
70 void setNextOverride(Overridable *nextOverridable)
71 {
72 m_nextOverride = nextOverridable;
73 }
74
75 // useful for the LocomotorStore to cleanup overrides.
76 Overridable *friend_getNextOverride( void )
77 {
78 return m_nextOverride;
79 }
80
81 // useful for the LocomotorStore to create an override dangling off the final override.
82 Overridable *friend_getFinalOverride( void )
83 {
84 if (m_nextOverride)
85 return m_nextOverride->friend_getFinalOverride();
86 return this;
87 }
88
89 // useful for the LocomotorStore to create an override dangling off the final override.
90 const Overridable *friend_getFinalOverride( void ) const
91 {
92 if (m_nextOverride)
93 return m_nextOverride->friend_getFinalOverride();
94 return this;
95 }
96
97 // used by ini-parsing functions to mark specific Overridables as overrides
98 void markAsOverride( void )
99 {
100 m_isOverride = true;
101 }
102
103 // used in factory reset() calls at the end of a game to clean up overrides. Can return NULL
104 // if the first Overridable is itself an override
105 Overridable *deleteOverrides( void )
106 {
107 if ( m_isOverride )
108 {
109 deleteInstance();
110 return NULL;
111 }
112 else if ( m_nextOverride )
113 {
114 m_nextOverride = m_nextOverride->deleteOverrides();
115 }
116 return this;
117 }
118};
119
120// cleans up and dangling overrides.
121__inline Overridable::~Overridable()
122{
123 if (m_nextOverride)
124 m_nextOverride->deleteInstance();
125}
