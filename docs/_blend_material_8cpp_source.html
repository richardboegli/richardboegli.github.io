81Bool BlendMaterial::setTerrainTreeViewSelection(HTREEITEM parent, Int selection)
82{
83 TVITEM item;
84 char buffer[_MAX_PATH];
85 ::memset(&item, 0, sizeof(item));
86 HTREEITEM child = m_terrainTreeView.GetChildItem(parent);
87 while (child != NULL) {
88 item.mask = TVIF_HANDLE|TVIF_PARAM;
89 item.hItem = child;
90 item.pszText = buffer;
91 item.cchTextMax = sizeof(buffer)-2;
92 m_terrainTreeView.GetItem(&item);
93 if (item.lParam == selection) {
94 m_terrainTreeView.SelectItem(child);
95 return(true);
96 }
97 if (setTerrainTreeViewSelection(child, selection)) {
98 return(true);
99 }
100 child = m_terrainTreeView.GetNextSiblingItem(child);
101 }
102 return(false);
103}

141HTREEITEM BlendMaterial::findOrAdd(HTREEITEM parent, char *pLabel)
142{
143 TVINSERTSTRUCT ins;
144 char buffer[_MAX_PATH];
145 ::memset(&ins, 0, sizeof(ins));
146 HTREEITEM child = m_terrainTreeView.GetChildItem(parent);
147 while (child != NULL) {
148 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
149 ins.item.hItem = child;
150 ins.item.pszText = buffer;
151 ins.item.cchTextMax = sizeof(buffer)-2;
152 m_terrainTreeView.GetItem(&ins.item);
153 if (strcmp(buffer, pLabel) == 0) {
154 return(child);
155 }
156 child = m_terrainTreeView.GetNextSiblingItem(child);
157 }
158
159 // not found, so add it.
160 ::memset(&ins, 0, sizeof(ins));
161 ins.hParent = parent;
162 ins.hInsertAfter = TVI_LAST;
163 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
164 ins.item.lParam = -1;
165 ins.item.pszText = pLabel;
166 ins.item.cchTextMax = strlen(pLabel);
167 child = m_terrainTreeView.InsertItem(&ins);
168 return(child);
169}

172void BlendMaterial::addTerrain(const char *pPath, Int terrainNdx, HTREEITEM parent)
173{
174 AsciiString className;
175 if (terrainNdx >= 0) {
176 className = WorldHeightMapEdit::getTexClassName( terrainNdx );
177 }
178 TerrainType *terrain = TheTerrainTypes->findTerrain( className );
179 Bool doAdd = FALSE;
180 char buffer[_MAX_PATH];
181 //
182 // if we have a 'terrain' entry, it means that our terrain index was properly defined
183 // in an INI file, otherwise it was from eval textures. We will sort all of
184 // the eval texture entries in a tree leaf all their own while the others are
185 // sorted according to a field specified in INI
186 //
187 if( terrain )
188 {
189 if (!terrain->isBlendEdge()) {
190 return; // Only do blend edges to the blend material list.
191 }
192
193 // set the name in the tree view to that of the entry
194 strcpy( buffer, terrain->getName().str() );
195
196 doAdd = TRUE;
197 } else if (terrainNdx==-1) {
198 strcpy(buffer, pPath);
199 doAdd = true;
200 } else if (WorldHeightMapEdit::getTexClassIsBlendEdge(terrainNdx)) {
201 parent = findOrAdd( parent, "**EVAL**" );
202 strcpy(buffer, pPath);
203 doAdd = true;
204 } // end if
205
206// Int tilesPerRow = TEXTURE_WIDTH/(2*TILE_PIXEL_EXTENT+TILE_OFFSET);
207// Int availableTiles = 4 * tilesPerRow * tilesPerRow;
208// Int percent = (WorldHeightMapEdit::getTexClassNumTiles(terrainNdx)*100 + availableTiles/2) / availableTiles;
209
210 char label[_MAX_PATH];
211 sprintf(label, "%s", buffer);
212
213
214 if( doAdd )
215 {
216 TVINSERTSTRUCT ins;
217
218 ::memset(&ins, 0, sizeof(ins));
219 ins.hParent = parent;
220 ins.hInsertAfter = TVI_LAST;
221 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
222 ins.item.lParam = terrainNdx;
223 ins.item.pszText = label;
224 ins.item.cchTextMax = strlen(label)+2;
225 m_terrainTreeView.InsertItem(&ins);
226 }
227
228}

230//* Create the tree view of textures from the textures in pMap. */
231void BlendMaterial::updateTextures(void)
232{
233 m_updating = true;
234 m_terrainTreeView.DeleteAllItems();
235 Int i;
236 CString label;
237 label.LoadString(IDS_ALPHA_BLEND);
238 addTerrain(label, -1, TVI_ROOT);
239 for (i=WorldHeightMapEdit::getNumTexClasses()-1; i>=0; i--) {
240 char path[_MAX_PATH];
241 AsciiString uiName = WorldHeightMapEdit::getTexClassUiName(i);
242 strncpy(path, uiName.str(), _MAX_PATH-2);
243 addTerrain(path, i, TVI_ROOT);
244 }
245 m_updating = false;
246 m_currentBlendTexture = defaultMaterialIndex;
247 setTerrainTreeViewSelection(TVI_ROOT, m_currentBlendTexture);
248}

253BOOL BlendMaterial::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
254{
255 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
256 if (pHdr->hdr.hwndFrom == m_terrainTreeView.m_hWnd) {
257 if (pHdr->hdr.code == TVN_SELCHANGED) {
258 HTREEITEM hItem = m_terrainTreeView.GetSelectedItem();
259 TVITEM item;
260 ::memset(&item, 0, sizeof(item));
261 item.mask = TVIF_HANDLE|TVIF_PARAM;
262 item.hItem = hItem;
263 m_terrainTreeView.GetItem(&item);
264 if (item.lParam >= -1) {
265 Int texClass = item.lParam;
266 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
267 if (!pDoc) return 0;
268
269 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
270 if (!pMap) return 0;
271 if (m_updating) return 0;
272 m_currentBlendTexture = texClass;
273 } else if (!(item.state & TVIS_EXPANDEDONCE) ) {
274 HTREEITEM child = m_terrainTreeView.GetChildItem(hItem);
275 while (child != NULL) {
276 hItem = child;
277 child = m_terrainTreeView.GetChildItem(hItem);
278 }
279 if (hItem != m_terrainTreeView.GetSelectedItem()) {
280 m_terrainTreeView.SelectItem(hItem);
281 }
282 }
283 }
284 }
285
286 return CDialog::OnNotify(wParam, lParam, pResult);
