92UpdateSleepTime HealContain::update( void )
93{
94
95 // extending functionality
96 /*UpdateSleepTime result =*/ OpenContain::update();
97
98 // get the module data
99 const HealContainModuleData *modData = getHealContainModuleData();
100
101 //
102 // for each of our objects, we give them a little health each frame so that when the
103 // the TimeTillHealed is up, the object will exit and be fully healed
104 //
105 Bool doneHealing;
106 Object *obj;
107 ContainedItemsList::const_iterator it = getContainList().begin();
108 while( it != getContainList().end() )
109 {
110
111 // get the object
112 obj = *it;
113
114 // increment the iterator, which allows failure to not cause an infinite loop
115 ++it;
116
117 // do the healing on this object
118 doneHealing = doHeal( obj, modData->m_framesForFullHeal );
119
120 // if we're done healing, we need to remove us from the healing container
121 if( doneHealing == TRUE )
122 {
123 ExitDoorType exitDoor = reserveDoorForExit(obj->getTemplate(), obj);
124 if (exitDoor != DOOR_NONE_AVAILABLE)
125 exitObjectViaDoor( obj, exitDoor );
126 } // end if
127
128 } // end for, it
129
130 return UPDATE_SLEEP_NONE;
131
132} // end update

137Bool HealContain::doHeal( Object *obj, UnsignedInt framesForFullHeal )
138{
139 Bool doneHealing = FALSE;
140
141 // setup the healing damageInfo structure with all but the amount
142 DamageInfo healInfo;
143 healInfo.in.m_damageType = DAMAGE_HEALING;
144 healInfo.in.m_deathType = DEATH_NONE;
145 healInfo.in.m_sourceID = getObject()->getID();
146
147 // get body module of the thing to heal
148 BodyModuleInterface *body = obj->getBodyModule();
149
150 // if we've been in here long enough ... set our health to max
151 if( TheGameLogic->getFrame() - obj->getContainedByFrame() >= framesForFullHeal )
152 {
153
154 // set the amount to max just to be sure we're at the top
155 healInfo.in.m_amount = body->getMaxHealth();
156
157 // set max health
158 body->attemptHealing( &healInfo );
159
160 // we're done healing
161 doneHealing = TRUE;
162
163 } // end if
164 else
165 {
166
167 //
168 // given the *whole* time it would take to heal this object, lets pretend that the
169 // object is at zero health ... and give it a sliver of health as if it were at 0 health
170 // and would be fully healed at 'framesForFullHeal'
171 //
172 healInfo.in.m_amount = body->getMaxHealth() / (Real)framesForFullHeal;
173
174 // do the healing
175 body->attemptHealing( &healInfo );
176
177 } // end else
178
179 // return if we're done healing
180 return doneHealing;
181
182} // end doHeal
