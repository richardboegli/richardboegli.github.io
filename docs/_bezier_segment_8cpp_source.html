99void BezierSegment::evaluateBezSegmentAtT(Real tValue, Coord3D *outResult) const
100
101{
102 if (!outResult)
103 return;
104
105 D3DXVECTOR4 tVec(tValue * tValue * tValue, tValue * tValue, tValue, 1);
106
107 D3DXVECTOR4 xCoords(m_controlPoints[0].x, m_controlPoints[1].x, m_controlPoints[2].x, m_controlPoints[3].x);
108 D3DXVECTOR4 yCoords(m_controlPoints[0].y, m_controlPoints[1].y, m_controlPoints[2].y, m_controlPoints[3].y);
109 D3DXVECTOR4 zCoords(m_controlPoints[0].z, m_controlPoints[1].z, m_controlPoints[2].z, m_controlPoints[3].z);
110
111 D3DXVECTOR4 tResult;
112 D3DXVec4Transform(&tResult, &tVec, &BezierSegment::s_bezBasisMatrix);
113
114 outResult->x = D3DXVec4Dot(&xCoords, &tResult);
115 outResult->y = D3DXVec4Dot(&yCoords, &tResult);
116 outResult->z = D3DXVec4Dot(&zCoords, &tResult);
117}

120void BezierSegment::getSegmentPoints(Int numSegments, VecCoord3D *outResult) const
121{
122 if (!outResult) {
123 return;
124 }
125
126 outResult->clear();
127 outResult->resize(numSegments);
128
129 BezFwdIterator iter(numSegments, this);
130 iter.start();
131 Int i = 0;
132 while (!iter.done()) {
133 (*outResult)[i] = iter.getCurrent();
134 ++i;
135 iter.next();
136 }
137}

140// This function isn't terribly fast. There are alternatives, and if this is too slow, we can
141// take a look at the other approximations.
142// There is no known close-form solution to this problem.
143Real BezierSegment::getApproximateLength(Real withinTolerance) const
144{
145 /*
146 How this works:
147 We can determine the approximate length of a bezier segment by
148 L0 = |(P0,P1)| + |(P1,P2)| + |(P2,P3)|
149 L1 = |(P0,P3)|
150
151 The length of the segment is approximately 1/2 L0 + 1/2 L1
152
153 P1__P2
154 / \
155 P0----P3
156
157 The error in this is L1 - L0. If the error is too much, then we subdivide the curve and
158 try again.
159 */
160
161 Coord3D p0p1 = { m_controlPoints[1].x - m_controlPoints[0].x,
162 m_controlPoints[1].y - m_controlPoints[0].y,
163 m_controlPoints[1].z - m_controlPoints[0].z, };
164
165 Coord3D p1p2 = { m_controlPoints[2].x - m_controlPoints[1].x,
166 m_controlPoints[2].y - m_controlPoints[1].y,
167 m_controlPoints[2].z - m_controlPoints[1].z, };
168
169 Coord3D p2p3 = { m_controlPoints[3].x - m_controlPoints[2].x,
170 m_controlPoints[3].y - m_controlPoints[2].y,
171 m_controlPoints[3].z - m_controlPoints[2].z, };
172
173 Coord3D p0p3 = { m_controlPoints[3].x - m_controlPoints[0].x,
174 m_controlPoints[3].y - m_controlPoints[0].y,
175 m_controlPoints[3].z - m_controlPoints[0].z, };
176
177 Real length0 = p0p3.length();
178 Real length1 = p0p1.length() + p1p2.length() + p2p3.length();
179
180 if ((length1 - length0) > withinTolerance) {
181 BezierSegment seg1, seg2;
182 splitSegmentAtT(0.5f, seg1, seg2);
183 return (seg1.getApproximateLength(withinTolerance) + seg2.getApproximateLength(withinTolerance));
184 }
185
186 return (INT_TO_REAL(length0 + length1) / 2.0f);
187}

190void BezierSegment::splitSegmentAtT(Real tValue, BezierSegment &outSeg1, BezierSegment &outSeg2) const
191{
192 // I think there are faster ways to do this. Could someone clue me in?
193
194 Coord3D p0p1 = { m_controlPoints[1].x - m_controlPoints[0].x,
195 m_controlPoints[1].y - m_controlPoints[0].y,
196 m_controlPoints[1].z - m_controlPoints[0].z, };
197
198 Coord3D p1p2 = { m_controlPoints[2].x - m_controlPoints[1].x,
199 m_controlPoints[2].y - m_controlPoints[1].y,
200 m_controlPoints[2].z - m_controlPoints[1].z, };
201
202 Coord3D p2p3 = { m_controlPoints[3].x - m_controlPoints[2].x,
203 m_controlPoints[3].y - m_controlPoints[2].y,
204 m_controlPoints[3].z - m_controlPoints[2].z, };
205
206 p0p1.scale(tValue);
207 p1p2.scale(tValue);
208 p2p3.scale(tValue);
209
210 p0p1.add(&m_controlPoints[0]);
211 p1p2.add(&m_controlPoints[1]);
212 p2p3.add(&m_controlPoints[2]);
213
214 Coord3D triLeft = { p1p2.x - p0p1.x,
215 p1p2.y - p0p1.y,
216 p1p2.z - p0p1.z, };
217
218 Coord3D triRight = { p2p3.x - p1p2.x,
219 p2p3.y - p1p2.y,
220 p2p3.z - p1p2.z, };
221
222 triLeft.scale(tValue);
223 triRight.scale(tValue);
224
225 triLeft.add(&p0p1);
226 triRight.add(&p1p2);
227
228 outSeg1.m_controlPoints[0] = m_controlPoints[0];
229 outSeg1.m_controlPoints[1] = p0p1;
230 outSeg1.m_controlPoints[2] = triLeft;
231 evaluateBezSegmentAtT(tValue, &outSeg1.m_controlPoints[3]);
232
233 outSeg2.m_controlPoints[0] = outSeg1.m_controlPoints[3];
234 outSeg2.m_controlPoints[1] = triRight;
235 outSeg2.m_controlPoints[2] = p2p3;
236 outSeg2.m_controlPoints[3] = m_controlPoints[3];
237}
