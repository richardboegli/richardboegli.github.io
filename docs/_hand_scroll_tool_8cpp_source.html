70void HandScrollTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
71{
72 if (m == TRACK_NONE)
73 return;
74
75 if (m == TRACK_M) {
76
77 // camera rotation
78 const Real factor = 0.01f;
79 Real rot = factor * (viewPt.x - m_prevPt2d.x);
80 /*
81 if (pView->isDoingPitch())
82 pView->pitchCamera(rot);
83 else
84 */
85 pView->rotateCamera(rot);
86 m_prevPt2d = viewPt;
87
88 } else if (m == TRACK_L || m == TRACK_R) {
89 if (!m_scrolling) {
90 // see if we moved enough to start scrolling.
91 if (abs(viewPt.x - m_downPt2d.x) > HYSTERESIS) m_scrolling = true;
92 if (abs(viewPt.y - m_downPt2d.y) > HYSTERESIS) m_scrolling = true;
93 }
94 if (!m_scrolling) {
95 return;
96 }
97 // Scroll dynamically.
98 Coord3D prev, cur;
99 if (pView->viewToDocCoords(m_prevPt2d, &prev, false) &&
100 pView->viewToDocCoords(viewPt, &cur, false))
101 {
102
103 Real dx = cur.x - prev.x;
104 Real dy = cur.y - prev.y;
105 if (IABS(dx)>MAX_SCROLL) {
106 dx = 0;
107 }
108 if (IABS(dy)>MAX_SCROLL) {
109 dy = 0;
110 }
111 dx /= MAP_XY_FACTOR;
112 dy /= MAP_XY_FACTOR;
113 pView->scrollInView(dx, dy, false);
114
115 m_prevPt2d = viewPt;
116 }
117 }
118}

120void HandScrollTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
121{
122 if (m == TRACK_M) {
123 // if haven't moved, reset view
124 Bool moved = false;
125 if (abs(viewPt.x - m_downPt2d.x) > HYSTERESIS) moved = true;
126 if (abs(viewPt.y - m_downPt2d.y) > HYSTERESIS) moved = true;
127
128 if (!moved && GetTickCount() - m_mouseDownTime < ::GetDoubleClickTime())
129 {
130 pView->setDefaultCamera();
131 }
132 } else if (m == TRACK_L || m == TRACK_R) {
133 if (m_scrolling) {
134 // Tell the view we are done scrolling.
135 pView->scrollInView(0,0,true);
136 } else if (m==TRACK_R) {
137 // Clicked right. Deselect & go to pointer.
138 WbApp()->selectPointerTool();
139 }
140 }
