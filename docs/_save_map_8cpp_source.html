121void SaveMap::populateMapListbox( Bool systemMaps )
122{
123 m_pInfo->usingSystemDir = m_usingSystemDir = systemMaps;
124#if defined(_DEBUG) || defined(_INTERNAL)
125 ::AfxGetApp()->WriteProfileInt(MAP_OPENSAVE_PANEL_SECTION, "UseSystemDir", m_usingSystemDir);
126#endif
127
128 HANDLE hFindFile = 0;
129 WIN32_FIND_DATA findData;
130 char dirBuf[_MAX_PATH];
131 char findBuf[_MAX_PATH];
132 char fileBuf[_MAX_PATH];
133
134 if (systemMaps)
135 strcpy(dirBuf, ".\\Maps\\");
136 else
137 sprintf(dirBuf, "%sMaps\\", TheGlobalData->getPath_UserData().str());
138 int len = strlen(dirBuf);
139
140 if (len > 0 && dirBuf[len - 1] != '\\') {
141 dirBuf[len++] = '\\';
142 dirBuf[len] = 0;
143 }
144 CListBox *pList = (CListBox *)this->GetDlgItem(IDC_SAVE_LIST);
145 if (pList == NULL) return;
146 pList->ResetContent();
147 strcpy(findBuf, dirBuf);
148 strcat(findBuf, "*.*");
149
150 hFindFile = FindFirstFile(findBuf, &findData);
151 if (hFindFile != INVALID_HANDLE_VALUE) {
152 do {
153 if (strcmp(findData.cFileName, ".") == 0 || strcmp(findData.cFileName, "..") == 0)
154 continue;
155 if ((findData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0) {
156 continue;
157 }
158 strcpy(fileBuf, dirBuf);
159 strcat(fileBuf, findData.cFileName);
160 strcat(fileBuf, "\\");
161 strcat(fileBuf, findData.cFileName);
162 strcat(fileBuf, ".map");
163 try {
164 CFileStatus status;
165 if (CFile::GetStatus(fileBuf, status)) {
166 if (!(status.m_attribute & CFile::directory)) {
167 pList->AddString(findData.cFileName);
168 };
169 }
170 } catch(...) {}
171
172 } while (FindNextFile(hFindFile, &findData));
173
174 if (hFindFile) FindClose(hFindFile);
175 }
176 CEdit *pEdit = (CEdit*)GetDlgItem(IDC_SAVE_MAP_EDIT);
177 if (pEdit != NULL) {
178 strcpy(fileBuf, m_pInfo->filename);
179 Int len = strlen(fileBuf);
180 if (len>4 && stricmp(".map", fileBuf+(len-4)) == 0) {
181 // strip of the .map
182 fileBuf[len-4] = 0;
183 }
184 while (len>0) {
185 if (fileBuf[len] == '\\') {
186 len++;
187 break;
188 }
189 len--;
190 }
191 pEdit->SetWindowText(&fileBuf[len]);
192 pEdit->SetSel(0, 1000, true);
193 pEdit->SetFocus();
194 }
195}

197void SaveMap::OnSelchangeSaveList()
198{

208 }
209 CWnd *pEdit = GetDlgItem(IDC_SAVE_MAP_EDIT);
210 if (pEdit == NULL) {
211 return;
