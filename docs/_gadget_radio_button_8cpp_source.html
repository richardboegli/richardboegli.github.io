68// doRadioUnselect ============================================================
70//=============================================================================
71static void doRadioUnselect( GameWindow *window, Int group, Int screen,
72 GameWindow *except )
73{
74
75 //
76 // if this is a radio button we have something to consider, but we
77 // will ignore the except window
78 //
79 if( window != except && BitTest( window->winGetStyle(), GWS_RADIO_BUTTON ) )
80 {
81 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
82
83 if( radioData->group == group && radioData->screen == screen )
84 {
85 WinInstanceData *instData = window->winGetInstanceData();
86
87 BitClear( instData->m_state, WIN_STATE_SELECTED );
88
89 } // end if
90
91 } // end if
92
93 // recursively call on all my children
94 GameWindow *child;
95
96 for( child = window->winGetChild(); child; child = child->winGetNext() )
97 doRadioUnselect( child, group, screen, except );
98
99} // end doRadioUnselect

105//=============================================================================
106static void unselectOtherRadioOfGroup( Int group, Int screen,

109 GameWindow *window = TheWindowManager->winGetWindowList();
110
111 for( window = TheWindowManager->winGetWindowList();
112 window;
113 window = window->winGetNext() )

115
116} // end unselectOtherRadioOfGroup

122//=============================================================================
123WindowMsgHandledType GadgetRadioButtonInput( GameWindow *window, UnsignedInt msg,
124 WindowMsgData mData1, WindowMsgData mData2 )
125{
126 WinInstanceData *instData = window->winGetInstanceData();
127
128 switch( msg )
129 {
130
131 // ------------------------------------------------------------------------
132 case GWM_MOUSE_ENTERING:
133 {
134
135 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
136 {
137
138 BitSet( instData->m_state, WIN_STATE_HILITED );
139 TheWindowManager->winSendSystemMsg( instData->getOwner(),
140 GBM_MOUSE_ENTERING,
141 (WindowMsgData)window,
142 mData1 );
143 //TheWindowManager->winSetFocus( window );
144
145 } // end if
146
147 break;
148
149 } // end mouse enter
150
151 // ------------------------------------------------------------------------
152 case GWM_MOUSE_LEAVING:
153 {
154
155 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
156 {
157
158 BitClear( instData->m_state, WIN_STATE_HILITED );
159 TheWindowManager->winSendSystemMsg( instData->getOwner(),
160 GBM_MOUSE_LEAVING,
161 (WindowMsgData)window,
162 mData1 );
163 } // end if
164
165 break;
166
167 } // end mouse leaving
168
169 // ------------------------------------------------------------------------
170 case GWM_LEFT_DRAG:
171 {
172
173 TheWindowManager->winSendSystemMsg( instData->getOwner(), GGM_LEFT_DRAG,
174 (WindowMsgData)window, mData1 );
175 break;
176
177 } // end left drag
178
179 // ------------------------------------------------------------------------
180 case GWM_LEFT_DOWN:
181 {
182
183 break;
184
185 } // end down
186
187 // ------------------------------------------------------------------------
188 case GWM_LEFT_UP:
189 {
190
191 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) == FALSE )
192 {
193 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
194
195 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
196 GBM_SELECTED,
197 (WindowMsgData)window,
198 mData1 );
199
200 //
201 // unselect any windows in the system (including children) that
202 // are radio buttons with this same group and screen ID
203 //
204 if( radioData->group != 0 )
205 unselectOtherRadioOfGroup(radioData->group, radioData->screen, window );
206
207 // this button is now selected
208 BitSet( instData->m_state, WIN_STATE_SELECTED );
209
210 } // end if, not selected
211 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) == FALSE )
212 {
213
214 // this up click was not meant for this button
215 return MSG_IGNORED;
216
217 } // end else if
218
219 break;
220
221 } // end left up or click
222
223 // ------------------------------------------------------------------------
224 case GWM_CHAR:
225 {
226
227 switch( mData1 )
228 {
229
230 // --------------------------------------------------------------------
231 case KEY_ENTER:
232 case KEY_SPACE:
233 if( BitTest( mData2, KEY_STATE_DOWN ) )
234 {
235
236 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) == FALSE )
237 {
238 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
239
240 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
241 GBM_SELECTED,
242 (WindowMsgData)window,
243 mData1 );
244
245 //
246 // unselect any windows in the system (including children) that
247 // are radio buttons with this same group and screen ID
248 //
249 if( radioData->group != 0 )
250 unselectOtherRadioOfGroup(radioData->group, radioData->screen, window );
251
252 // this button is now selected
253 BitSet( instData->m_state, WIN_STATE_SELECTED );
254
255 } // end if, not selected
256
257 } // end key down
258
259 break;
260
261 // --------------------------------------------------------------------
262 case KEY_DOWN:
263 case KEY_RIGHT:
264 case KEY_TAB:
265 {
266
267 if( BitTest( mData2, KEY_STATE_DOWN ) )
268 window->winNextTab();
269 break;
270
271 } // end down, right, or tab
272
273 // --------------------------------------------------------------------
274 case KEY_UP:
275 case KEY_LEFT:
276 {
277
278 if( BitTest( mData2, KEY_STATE_DOWN ) )
279 window->winPrevTab();
280 break;
281
282 } // end up, left
283
284 // --------------------------------------------------------------------
285 default:
286 {
287
288 return MSG_IGNORED;
289
290 } // end default
291
292 } // end switch( mData1 )
293
294 break;
295
296 } // end char messsage
297
298 // ------------------------------------------------------------------------
299 default:
300 {
301
302 return MSG_IGNORED;
303
304 } // end default
305
306 } // end switch( msg )
307
308 return MSG_HANDLED;
309
310} // end GadgetRadioButtonInput

312// GadgetRadioButtonSystem ====================================================
314//=============================================================================
315WindowMsgHandledType GadgetRadioButtonSystem( GameWindow *window, UnsignedInt msg,
316 WindowMsgData mData1, WindowMsgData mData2 )
317{
318 WinInstanceData *instData = window->winGetInstanceData();
319
320 switch( msg )
321 {
322
323 // ------------------------------------------------------------------------
324 case GBM_SET_SELECTION:
325 {
326
327 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) == FALSE )
328 {
329
330 // do we want to send a selected message?
331 if( (Bool)mData1 == TRUE )
332 {
333
334 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
335 GBM_SELECTED,
336 (WindowMsgData)window,
337 0 );
338 } // end if
339
340 //
341 // unselect any windows in the system (including children) that
342 // are radio buttons with this same group and screen ID
343 //
344 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
345 if( radioData->group != 0 )
346 unselectOtherRadioOfGroup(radioData->group, radioData->screen, window );
347
348 // this button is now selected
349 BitSet( instData->m_state, WIN_STATE_SELECTED );
350
351 } // end if
352
353 break;
354
355 } // end set selection
356
357 // ------------------------------------------------------------------------
358 case GGM_SET_LABEL:
359 {
360
361 window->winSetText( *(UnicodeString*)mData1 );
362 break;
363
364 } // end set label
365
366 // ------------------------------------------------------------------------
367 case GWM_CREATE:
368 break;
369
370 // ------------------------------------------------------------------------
371 case GWM_DESTROY:
372 {
373 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
374
375 // free radio button user data
376 delete radioData;
377
378 break;
379
380 } // end destroy
381
382 // ------------------------------------------------------------------------
383 case GWM_INPUT_FOCUS:
384 {
385
386 if( mData1 == FALSE )
387 BitClear( instData->m_state, WIN_STATE_HILITED );
388
389 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
390 GGM_FOCUS_CHANGE,
391 mData1,
392 window->winGetWindowId() );
393
394 *(Bool*)mData2 = TRUE;
395 break;
396
397 } // end focus
398
399 default:
400 return MSG_IGNORED;
401
402 } // end switch( msg )
403
404 return MSG_HANDLED;
405
406} // end GadgetRadioButtonSystem
