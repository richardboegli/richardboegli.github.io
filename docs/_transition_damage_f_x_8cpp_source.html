88static void parseFXLocInfo( INI *ini, void *instance, FXLocInfo *locInfo )
89{
90 const char *token = ini->getNextToken( ini->getSepsColon() );
91
92 if( stricmp( token, "bone" ) == 0 )
93 {
94
95 // save bone name and location type
96 locInfo->boneName = AsciiString( ini->getNextToken() );
97 locInfo->locType = FX_DAMAGE_LOC_TYPE_BONE;
98
99 //
100 // bones are followed by RandomBone:<Yes|No>, if random bone is yes, the bone name is
101 // assumed to be a "base bone name" and we will find all the bones with that prefix
102 // when picking an effect position. If it's no, the bone name is assumed to be explicit
103 //
104 token = ini->getNextToken( ini->getSepsColon() );
105 if( stricmp( token, "randombone" ) != 0 )
106 {
107
108 DEBUG_CRASH(( "parseFXLocInfo: Bone name not followed by RandomBone specifier\nPress IGNORE to see which INI file and line # is incorrect." ));
109 throw INI_INVALID_DATA;
110
111 } // end if
112
113 // parse the Bool definition
114 ini->parseBool( ini, instance, &locInfo->randomBone, NULL );
115
116 } // end if
117 else if( stricmp( token, "loc" ) == 0 )
118 {
119
120 // save location and location type
121 locInfo->loc.x = ini->scanReal( ini->getNextSubToken("X") );
122 locInfo->loc.y = ini->scanReal( ini->getNextSubToken("Y") );
123 locInfo->loc.z = ini->scanReal( ini->getNextSubToken("Z") );
124 locInfo->locType = FX_DAMAGE_LOC_TYPE_COORD;
125
126 } // end else
127 else
128 {
129
130 // error
131 throw INI_INVALID_DATA;
132
133 } // end else
134
135} // end parseFXLocInfo

259static Coord3D getLocalEffectPos( const FXLocInfo *locInfo, Drawable *draw )
260{
261
262 DEBUG_ASSERTCRASH( locInfo, ("getLocalEffectPos: locInfo is NULL\n") );
263
264 if( locInfo->locType == FX_DAMAGE_LOC_TYPE_BONE && draw )
265 {
266
267 if( locInfo->randomBone == FALSE )
268 {
269 Coord3D pos;
270
271 // get the bone position
272 Int count = draw->getPristineBonePositions( locInfo->boneName.str(), 0, &pos, NULL, 1 );
273
274 // sanity, if bone not found revert back to location defined in struct (which is 0,0,0)
275 if( count == 0 )
276 return locInfo->loc;
277
278 // return the position retrieved
279 return pos;
280
281 } // end if
282 else
283 {
284 const Int MAX_BONES = 32;
285 Coord3D positions[ MAX_BONES ];
286
287 // get the bone positions
288 Int boneCount;
289 boneCount = draw->getPristineBonePositions( locInfo->boneName.str(), 1, positions, NULL, MAX_BONES );
290
291 // sanity, if bone not found revert back to location defined in struct (which is 0,0,0)
292 if( boneCount == 0 )
293 return locInfo->loc;
294
295 // pick one of the bone positions
296 Int pick = GameLogicRandomValue( 0, boneCount - 1 );
297 return positions[ pick ];
298
299 } // end else
300
301 } // end if
302 else
303 return locInfo->loc;
304
305} // end getLocalEffectPos

310void TransitionDamageFX::onBodyDamageStateChange( const DamageInfo* damageInfo,
311 BodyDamageType oldState,
312 BodyDamageType newState )
313{
314 Object *damageSource = NULL;
315 Int i;
316 Drawable *draw = getObject()->getDrawable();
317 const TransitionDamageFXModuleData *modData = getTransitionDamageFXModuleData();
318
319 // get the source of the damage if present
320 if( damageInfo )
321 damageSource = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
322
323 // remove any particle systems that might be emitting from our old state
324 for( i = 0; i < DAMAGE_MODULE_MAX_FX; i++ )
325 {
326
327 if( m_particleSystemID[ oldState ][ i ] != INVALID_PARTICLE_SYSTEM_ID )
328 {
329
330 TheParticleSystemManager->destroyParticleSystemByID( m_particleSystemID[ oldState ][ i ] );
331 m_particleSystemID[ oldState ][ i ] = INVALID_PARTICLE_SYSTEM_ID;
332
333 } // end if
334
335 } // end for i
336
337 //
338 // when we are transitioning to a "worse" state we will play a set of effects for that
339 // new state to make the transition
340 //
341 if( IS_CONDITION_WORSE( newState, oldState ) )
342 {
343 const ParticleSystemTemplate *pSystemT;
344 Coord3D pos;
345
346 // if we are restricted by the damage type executing effect, bail out of here
347 const DamageInfo *lastDamageInfo = getObject()->getBodyModule()->getLastDamageInfo();
348
349 for( i = 0; i < DAMAGE_MODULE_MAX_FX; i++ )
350 {
351
352 // play fx list for our new state
353 if( modData->m_fxList[ newState ][ i ].fx )
354 {
355
356 if( lastDamageInfo == NULL ||
357 getDamageTypeFlag( modData->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
358 {
359
360 pos = getLocalEffectPos( &modData->m_fxList[ newState ][ i ].locInfo, draw );
361 getObject()->convertBonePosToWorldPos( &pos, NULL, &pos, NULL );
362 FXList::doFXPos( modData->m_fxList[ newState ][ i ].fx, &pos );
363
364 } // end if
365
366 } // end if
367
368 // do any object creation list for our new state
369 if( modData->m_OCL[ newState ][ i ].ocl )
370 {
371
372 if( lastDamageInfo == NULL ||
373 getDamageTypeFlag( modData->m_damageOCLTypes, lastDamageInfo->in.m_damageType ) )
374 {
375
376 pos = getLocalEffectPos( &modData->m_OCL[ newState ][ i ].locInfo, draw );
377 getObject()->convertBonePosToWorldPos( &pos, NULL, &pos, NULL );
378 ObjectCreationList::create( modData->m_OCL[ newState ][ i ].ocl,
379 getObject(), &pos, damageSource->getPosition(), INVALID_ANGLE );
380
381 } // end if
382
383 } // end if
384
385 // get the template of the system to create
386 pSystemT = modData->m_particleSystem[ newState ][ i ].particleSysTemplate;
387 if( pSystemT )
388 {
389
390 if( lastDamageInfo == NULL ||
391 getDamageTypeFlag( modData->m_damageParticleTypes, lastDamageInfo->in.m_damageType ) )
392 {
393
394 // create a new particle system based on the template provided
395 ParticleSystem* pSystem = TheParticleSystemManager->createParticleSystem( pSystemT );
396 if( pSystem )
397 {
398
399 // get the what is the position we're going to playe the effect at
400 pos = getLocalEffectPos( &modData->m_particleSystem[ newState ][ i ].locInfo, draw );
401
402 //
403 // set position on system given any bone position provided, the bone position is
404 // local to the object and that's what we want for the particle system ... the
405 // transormation into world space using the object position is taken care of in
406 // the particle system attachToObject method
407 //
408 pSystem->setPosition( &pos );
409
410 // attach to object
411 pSystem->attachToObject( getObject() );
412
413 // save the id of this particle system so we can remove it later if it still exists
414 m_particleSystemID[ newState ][ i ] = pSystem->getSystemID();
415
416 } // end if
417
418 } // end if
419
420 } // end if
421
422 } // end for i
423
424 } // end if
425
426} // end onBodyDamageStateChange
