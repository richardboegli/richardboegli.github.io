100void WindowLayout::addWindow( GameWindow *window )
101{
102 GameWindow *win = findWindow( window );
103
104 // only add window if window is not in this layout already
105 if( win == NULL )
106 {
107
108 DEBUG_ASSERTCRASH( window->winGetNextInLayout() == NULL,
109 ("NextInLayout should be NULL before adding") );
110 DEBUG_ASSERTCRASH( window->winGetPrevInLayout() == NULL,
111 ("PrevInLayout should be NULL before adding") );
112
113 window->winSetPrevInLayout( NULL );
114 window->winSetNextInLayout( m_windowList );
115 if( m_windowList )
116 m_windowList->winSetPrevInLayout( window );
117 m_windowList = window;
118
119 // set layout into window
120 window->winSetLayout( this );
121
122 // if no tail pointer, this is it
123 if( m_windowTail == NULL )
124 m_windowTail = window;
125
126 // we gots another window now
127 m_windowCount++;
128
129 } // end if
130
131} // end addWindow

136void WindowLayout::removeWindow( GameWindow *window )
137{
138 GameWindow *win = findWindow( window );
139
140 // can't remove window unless it's really part of this layout
141 if( win )
142 {
143 GameWindow *prev, *next;
144
145 prev = win->winGetPrevInLayout();
146 next = win->winGetNextInLayout();
147
148 if( next )
149 next->winSetPrevInLayout( prev );
150 if( prev )
151 prev->winSetNextInLayout( next );
152 else
153 m_windowList = next;
154
155 // set window as having no layout info
156 win->winSetLayout( NULL );
157 win->winSetNextInLayout( NULL );
158 win->winSetPrevInLayout( NULL );
159
160 // if we removed the tail, set the new tail
161 if( m_windowTail == win )
162 m_windowTail = prev;
163
164 // we lost one sir!
165 m_windowCount--;
166
167 } // end if
168
169} // end removeWindow

199Bool WindowLayout::load( AsciiString filename )
200{
201
202 // sanity
203 if( filename.isEmpty() )
204 return FALSE;
205
206 //
207 // when we create new windows they are always placed on the top
208 // of the window stack. The load layout from script will return the
209 // FIRST window loaded from the script, if we traverse from the head of
210 // the window list till we see that window returned from the layout
211 // loader, that will be all the root windows loaded from the .wnd file
212 //
213 GameWindow *target;
214 WindowLayoutInfo info;
215
216 target = TheWindowManager->winCreateFromScript( filename, &info );
217 if( target == NULL )
218 {
219
220 DEBUG_ASSERTCRASH( target, ("WindowLayout::load - Failed to load layout") );
221 DEBUG_LOG(( "WindowLayout::load - Unable to load layout file '%s'\n", filename.str() ));
222 return FALSE;
223
224 } // end if
225
226 //
227 // add windows loaded from .wnd file to the layout, via info.windows.
228 //
229 std::list<GameWindow *>::iterator it;
230 for (it = info.windows.begin(); it != info.windows.end(); ++it)
231 {
232 // add window to this layout
233 addWindow( *it );
234 }
235
236 /* MDC - can't do this, as modal windows will be at the head...
237 //
238 // add windows loaded from .wnd file to the layout, note we start at
239 // target and go backwards as the layout->addWindow() puts at top
240 // of list and will therefore reverse the order
241 //
242 GameWindow *window;
243 for( window = target; window; window = window->winGetPrev() )
244 {
245
246 // add window to this layout
247 addWindow( window );
248
249 } // end for window
250 */
251
252 // copy filename
253 m_filenameString = filename;
254
255 // assign script info to the layout
256 setInit( info.init );
257 setUpdate( info.update );
258 setShutdown( info.shutdown );
259
260 return TRUE; // success
261
262} // end load

267void WindowLayout::bringForward( void )
268{
269
270 //
271 // loop through all our windows and bring each of them to the top of
272 // the window stack, note that we are getting a prev pointer because the
273 // action of bringing a window to the top of the stack does detach it
274 // from the window list and the layout window list ... it is then
275 // re-attached at the top. Also note, to preserve the ordering of
276 // the windows we pull from the tail and add to the top
277 //
278 GameWindow *window, *prev;
279 Int countLeft = m_windowCount;
280
281 for( window = m_windowTail; countLeft; window = prev )
282 {
283
284 DEBUG_ASSERTCRASH( window, ("Must have window: m_windowCount is off") );
285 prev = window->winGetPrevInLayout();
286 window->winBringToTop();
287 countLeft--;
288
289 } // end for window
290
291} // end bringForward

304 return win;
305
306 return NULL; // window not found
307
308} // end findWindow
