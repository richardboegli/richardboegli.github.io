39static RailedTransportDockUpdateInterface *getRailedTransportDockUpdateInterface( Object *obj )
40{
41
42 // sanity
43 if( obj == NULL )
44 return NULL;
45
46 // find us our dock interface
47 RailedTransportDockUpdateInterface *rtdui = NULL;
48 for( BehaviorModule **u = obj->getBehaviorModules(); *u; ++u )
49 if( (rtdui = (*u)->getRailedTransportDockUpdateInterface()) != NULL )
50 break;
51
52 return rtdui;
53
54} // end getRailedTransportDockUpdateInterface

92Bool RailedTransportContain::isSpecificRiderFreeToExit( Object *obj )
93{
94 Object *us = getObject();
95
96 // if we are in transit we cannot exit, when we're in transit, the dock is closed
97 DockUpdateInterface *dui = us->getDockUpdateInterface();
98 if( dui && dui->isDockOpen() == FALSE )
99 return FALSE;
100
101 // we can now exit, note we're not extending the base class cause *we* handle it all
102 return TRUE;
103
104} // end isSpecificRiderFreeToExit
