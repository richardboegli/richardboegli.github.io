402GameMessageDisposition MetaEventTranslator::translateGameMessage(const GameMessage *msg)
403{
404 GameMessageDisposition disp = KEEP_MESSAGE;
405 GameMessage::Type t = msg->getType();
406
407 if (t == GameMessage::MSG_RAW_KEY_DOWN || t == GameMessage::MSG_RAW_KEY_UP)
408 {
409 MappableKeyType key = (MappableKeyType)msg->getArgument(0)->integer;
410 Int keyState = msg->getArgument(1)->integer;
411
412 // for our purposes here, we don't care to distinguish between right and left keys,
413 // so just fudge a little to simplify things.
414 Int newModState = 0;
415
416 if( keyState & KEY_STATE_CONTROL )
417 {
418 newModState |= CTRL;
419 }
420
421 if( keyState & KEY_STATE_SHIFT )
422 {
423 newModState |= SHIFT;
424 }
425
426 if( keyState & KEY_STATE_ALT )
427 {
428 newModState |= ALT;
429 }
430
431
432 for (const MetaMapRec *map = TheMetaMap->getFirstMetaMapRec(); map; map = map->m_next)
433 {
434 DEBUG_ASSERTCRASH(map->m_meta > GameMessage::MSG_BEGIN_META_MESSAGES &&
435 map->m_meta < GameMessage::MSG_END_META_MESSAGES, ("hmm, expected only meta-msgs here"));
436
437 //
438 // if this command is *only* usable in the game, we will ignore it if the game client
439 // has not yet incremented to frame 1 (keeps us from doing in-game commands during
440 // a map load, which throws the input system into wack because there isn't a
441 // client frame for the input event, and in the case of a command that pauses the
442 // game, like the quit menu, the client frame will never get beyond 0 and we
443 // lose the ability to process any input
444 //
445 if( map->m_usableIn == COMMANDUSABLE_GAME && TheGameClient->getFrame() < 1 )
446 continue;
447
448 // if the shell is active, and this command is not usable in shell, continue
449 if (TheShell && TheShell->isShellActive() && !(map->m_usableIn & COMMANDUSABLE_SHELL) )
450 continue;
451
452 // if the shell is not active and this command is not usable in the game, continue
453 if (TheShell && !TheShell->isShellActive() && !(map->m_usableIn & COMMANDUSABLE_GAME) )
454 continue;
455
456
457
458
459
460 // check for the special case of mods-only-changed.
461 if (
462 map->m_key == MK_NONE &&
463 newModState != m_lastModState &&
464 (
465 (map->m_transition == UP && map->m_modState == m_lastModState) ||
466 (map->m_transition == DOWN && map->m_modState == newModState)
467 )
468 )
469 {
470 //DEBUG_LOG(("Frame %d: MetaEventTranslator::translateGameMessage() Mods-only change: %s\n", TheGameLogic->getFrame(), findGameMessageNameByType(map->m_meta)));
471 /*GameMessage *metaMsg =*/ TheMessageStream->appendMessage(map->m_meta);
472 disp = DESTROY_MESSAGE;
473 break;
474 }
475
476 // ok, now check for "normal" key transitions.
477 if (
478 map->m_key == key &&
479 map->m_modState == newModState &&
480 (
481 (map->m_transition == UP && (keyState & KEY_STATE_UP)) ||
482 (map->m_transition == DOWN && (keyState & KEY_STATE_DOWN)) //||
483 //(map->m_transition == DOUBLEDOWN && (keyState & KEY_STATE_DOWN) && m_lastKeyDown == key)
484 )
485 )
486 {
487
488 if( keyState & KEY_STATE_AUTOREPEAT )
489 {
490 // if it's an autorepeat of a "known" key, don't generate the meta-event,
491 // but DO eat the keystroke so no one else can mess with it
492 //DEBUG_LOG(("Frame %d: MetaEventTranslator::translateGameMessage() auto-repeat: %s\n", TheGameLogic->getFrame(), findGameMessageNameByType(map->m_meta)));
493 }
494 else
495 {
496
497 // THIS IS A GREASY HACK... MESSAGE SHOULD BE HANDLED IN A TRANSLATOR, BUT DURING CINEMATICS THE TRANSLATOR IS DISABLED
498 if( map->m_meta == GameMessage::MSG_META_TOGGLE_FAST_FORWARD_REPLAY)
499 {
500 #if defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)//may be defined in GameCommon.h
501 if( TheGlobalData )
502 #else
503 if( TheGlobalData && TheGameLogic->isInReplayGame())
504 #endif
505 {
506 if ( TheWritableGlobalData )
507 TheWritableGlobalData->m_TiVOFastMode = 1 - TheGlobalData->m_TiVOFastMode;
508
509 if ( TheInGameUI )
510 TheInGameUI->message( TheGlobalData->m_TiVOFastMode ? TheGameText->fetch("GUI:FF_ON") : TheGameText->fetch("GUI:FF_OFF") );
511 }
512 disp = KEEP_MESSAGE; // cause for goodness sake, this key gets used a lot by non-replay hotkeys
513 break;
514 }
515
516
517 /*GameMessage *metaMsg =*/ TheMessageStream->appendMessage(map->m_meta);
518 //DEBUG_LOG(("Frame %d: MetaEventTranslator::translateGameMessage() normal: %s\n", TheGameLogic->getFrame(), findGameMessageNameByType(map->m_meta)));
519 }
520 disp = DESTROY_MESSAGE;
521 break;
522 }
523 }
524
525
526
527 if (t == GameMessage::MSG_RAW_KEY_DOWN)
528 {
529 m_lastKeyDown = key;
530
531
532#ifdef DUMP_ALL_KEYS_TO_LOG
533
534 WideChar Wkey = TheKeyboard->getPrintableKey(key, 0);
535 UnicodeString uKey;
536 uKey.set(&Wkey);
537 AsciiString aKey;
538 aKey.translate(uKey);
539 DEBUG_LOG(("^%s ", aKey.str()));
540#endif
541
542 }
543
544
545
546 m_lastModState = newModState;
547 }
548
549
550 if (t > GameMessage::MSG_RAW_MOUSE_BEGIN && t < GameMessage::MSG_RAW_MOUSE_END )
551 {
552 Int index = 0;
553 switch (t)
554 {
555 case GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_DOWN:
556 case GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_DOWN:
557 case GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_DOWN:
558 {
559 // Fill out which the current mouse down position
560 if (t == GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_DOWN)
561 index = 1;
562 else if (t == GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_DOWN)
563 index = 2;
564 // else index == 0
565 m_mouseDownPosition[index] = msg->getArgument(0)->pixel;
566 m_nextUpShouldCreateDoubleClick[index] = FALSE;
567
568 break;
569 }
570
571 case GameMessage::MSG_RAW_MOUSE_LEFT_DOUBLE_CLICK:
572 case GameMessage::MSG_RAW_MOUSE_MIDDLE_DOUBLE_CLICK:
573 case GameMessage::MSG_RAW_MOUSE_RIGHT_DOUBLE_CLICK:
574 {
575 if (t == GameMessage::MSG_RAW_MOUSE_MIDDLE_DOUBLE_CLICK)
576 index = 1;
577 else if (t == GameMessage::MSG_RAW_MOUSE_RIGHT_DOUBLE_CLICK)
578 index = 2;
579 // else index == 0
580
581 m_nextUpShouldCreateDoubleClick[index] = TRUE;
582 break;
583 }
584
585 case GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_UP:
586 case GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_UP:
587 case GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_UP:
588 {
589 ICoord2D location = msg->getArgument(0)->pixel;
590
591 // Fill out which the current mouse down position
592 if (t == GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_UP)
593 index = 1;
594 else if (t == GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_UP)
595 index = 2;
596 // else index == 0
597
598 GameMessage *newMessage = NULL;
599 if (t == GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_UP)
600 {
601 if (m_nextUpShouldCreateDoubleClick[index])
602 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_LEFT_DOUBLE_CLICK, const_cast<GameMessage*>(msg));
603 else
604 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_LEFT_CLICK, const_cast<GameMessage*>(msg));
605 m_nextUpShouldCreateDoubleClick[index] = FALSE;
606 }
607 else if (t == GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_UP)
608 {
609 if (m_nextUpShouldCreateDoubleClick[index])
610 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_MIDDLE_DOUBLE_CLICK, const_cast<GameMessage*>(msg));
611 else
612 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_MIDDLE_CLICK, const_cast<GameMessage*>(msg));
613 m_nextUpShouldCreateDoubleClick[index] = FALSE;
614 }
615 else if (t == GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_UP)
616 {
617 if (m_nextUpShouldCreateDoubleClick[index])
618 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_RIGHT_DOUBLE_CLICK, const_cast<GameMessage*>(msg));
619 else
620 newMessage = TheMessageStream->insertMessage(GameMessage::MSG_MOUSE_RIGHT_CLICK, const_cast<GameMessage*>(msg));
621 m_nextUpShouldCreateDoubleClick[index] = FALSE;
622 }
623
624 IRegion2D pixelRegion;
625 buildRegion( &m_mouseDownPosition[index], &location, &pixelRegion );
626 if (abs(pixelRegion.hi.x - pixelRegion.lo.x) < TheMouse->m_dragTolerance &&
627 abs(pixelRegion.hi.y - pixelRegion.lo.y) < TheMouse->m_dragTolerance)
628 {
629 pixelRegion.hi.x = pixelRegion.lo.x;
630 pixelRegion.hi.y = pixelRegion.lo.y;
631 }
632
633 newMessage->appendPixelRegionArgument( pixelRegion );
634
635 // append the modifier keys to the message.
636 newMessage->appendIntegerArgument( msg->getArgument(1)->integer );
637 break;
638 }
639
640 }
641
642 }
643
644 return disp;
645}
