80void VeterancyGainCreate::onCreate( void )
81{
82
83 // When produced normally, this Object will ask the Player if the correct Science is known for it
84 // to set its level to the given level
85
86 const VeterancyGainCreateModuleData *md = getVeterancyGainCreateModuleData();
87 Player *myPlayer = getObject()->getControllingPlayer();
88 if( myPlayer && (md->m_scienceRequired == SCIENCE_INVALID ||
89 myPlayer->hasScience( md->m_scienceRequired )) )
90 {
91 ExperienceTracker* myExp = getObject()->getExperienceTracker();
92 if( myExp && myExp->isTrainable() )
93 {
94 // srj sez: use "setMin" here so that we never lose levels
95 myExp->setMinVeterancyLevel( md->m_startingLevel );// sVL can override isTrainable, but this module should not.
96 }
97 }
98
99}
