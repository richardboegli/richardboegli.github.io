107void ControlBarPopupDescriptionUpdateFunc( WindowLayout *layout, void *param )
108{
109 if(TheScriptEngine->isGameEnding())
110 TheControlBar->hideBuildTooltipLayout();
111
112 if(theAnimateWindowManager && !TheControlBar->getShowBuildTooltipLayout() && !theAnimateWindowManager->isReversed())
113 theAnimateWindowManager->reverseAnimateWindow();
114 else if(!TheControlBar->getShowBuildTooltipLayout() && (!TheGlobalData->m_animateWindows || !useAnimation))
115 TheControlBar->deleteBuildTooltipLayout();
116
117
118 if ( useAnimation && theAnimateWindowManager && TheGlobalData->m_animateWindows)
119 {
120 Bool wasFinished = theAnimateWindowManager->isFinished();
121 theAnimateWindowManager->update();
122 if (theAnimateWindowManager && theAnimateWindowManager->isFinished() && !wasFinished && theAnimateWindowManager->isReversed())
123 {
124 delete theAnimateWindowManager;
125 theAnimateWindowManager = NULL;
126 TheControlBar->deleteBuildTooltipLayout();
127 }
128 }

133void ControlBar::showBuildTooltipLayout( GameWindow *cmdButton )
134{
135 if (TheInGameUI->areTooltipsDisabled() || TheScriptEngine->isGameEnding())
136 {
137 return;
138 }
139
140 Bool passedWaitTime = FALSE;
141 static Bool isInitialized = FALSE;
142 static UnsignedInt beginWaitTime;
143 if(prevWindow == cmdButton)
144 {
145 m_showBuildToolTipLayout = TRUE;
146 if(!isInitialized && beginWaitTime + cmdButton->getTooltipDelay() < timeGetTime())
147 {
148 //DEBUG_LOG(("%d beginwaittime, %d tooltipdelay, %dtimegettime\n", beginWaitTime, cmdButton->getTooltipDelay(), timeGetTime()));
149 passedWaitTime = TRUE;
150 }
151
152 if(!passedWaitTime)
153 return;
154 }
155 else if( !m_buildToolTipLayout->isHidden() )
156 {
157 if(useAnimation && TheGlobalData->m_animateWindows && !theAnimateWindowManager->isReversed())
158 theAnimateWindowManager->reverseAnimateWindow();
159 else if( useAnimation && TheGlobalData->m_animateWindows && theAnimateWindowManager->isReversed())
160 {
161 return;
162 }
163 else
164 {
165// m_buildToolTipLayout->destroyWindows();
166// m_buildToolTipLayout->deleteInstance();
167// m_buildToolTipLayout = NULL;
168 m_buildToolTipLayout->hide(TRUE);
169 prevWindow = NULL;
170 }
171 return;
172 }
173
174
175 // will only get here the firsttime through the function through this window
176 if(!passedWaitTime)
177 {
178 prevWindow = cmdButton;
179 beginWaitTime = timeGetTime();
180 isInitialized = FALSE;
181 return;
182 }
183 isInitialized = TRUE;
184
185 if(!cmdButton)
186 return;
187 if(BitTest(cmdButton->winGetStyle(), GWS_PUSH_BUTTON))
188 {
189 const CommandButton *commandButton = (const CommandButton *)GadgetButtonGetData(cmdButton);
190
191 if(!commandButton)
192 return;
193
194 // note that, in this branch, ENABLE_SOLO_PLAY is ***NEVER*** defined...
195 // this is so that we have a multiplayer build that cannot possibly be hacked
196 // to work as a solo game!
197 if (TheGameLogic->isInReplayGame())
198 return;
199
200 if (TheInGameUI->isQuitMenuVisible())
201 return;
202
203 if (TheDisconnectMenu && TheDisconnectMenu->isScreenVisible())
204 return;
205
206 // if (m_buildToolTipLayout)
207 // {
208 // m_buildToolTipLayout->destroyWindows();
209 // m_buildToolTipLayout->deleteInstance();
210 //
211 // }
212
213 m_showBuildToolTipLayout = TRUE;
214 // m_buildToolTipLayout = TheWindowManager->winCreateLayout( "ControlBarPopupDescription.wnd" );
215 // m_buildToolTipLayout->setUpdate(ControlBarPopupDescriptionUpdateFunc);
216
217 populateBuildTooltipLayout(commandButton);
218 }
219 else
220 {
221 // we're a generic window
222 if(!BitTest(cmdButton->winGetStyle(), GWS_USER_WINDOW) && !BitTest(cmdButton->winGetStyle(), GWS_STATIC_TEXT))
223 return;
224 populateBuildTooltipLayout(NULL, cmdButton);
225 }
226 m_buildToolTipLayout->hide(FALSE);
227
228 if (useAnimation && TheGlobalData->m_animateWindows)
229 {
230 theAnimateWindowManager = NEW AnimateWindowManager;
231 theAnimateWindowManager->reset();
232 theAnimateWindowManager->registerGameWindow( m_buildToolTipLayout->getFirstWindow(), WIN_ANIMATION_SLIDE_RIGHT_FAST, TRUE, 200 );
233 }
234
235
236}

244 return;

246 populateBuildTooltipLayout(commandButton);
247}

249void ControlBar::populateBuildTooltipLayout( const CommandButton *commandButton, GameWindow *tooltipWin)
250{
251 if(!m_buildToolTipLayout)
252 return;
253
254 Player *player = ThePlayerList->getLocalPlayer();
255 UnicodeString name, cost, descrip;
256 UnicodeString requires = UnicodeString::TheEmptyString, requiresList;
257 Bool firstRequirement = true;
258 const ProductionPrerequisite *prereq;
259 Bool fireScienceButton = false;
260 UnsignedInt costToBuild = 0;
261
262 if(commandButton)
263 {
264 const ThingTemplate *thingTemplate = commandButton->getThingTemplate();
265 const UpgradeTemplate *upgradeTemplate = commandButton->getUpgradeTemplate();
266
267 ScienceType st = SCIENCE_INVALID;
268 if( commandButton->getCommandType() != GUI_COMMAND_PLAYER_UPGRADE &&
269 commandButton->getCommandType() != GUI_COMMAND_OBJECT_UPGRADE )
270 {
271 if( commandButton->getScienceVec().size() > 1 )
272 {
273 for(Int j = 0; j < commandButton->getScienceVec().size(); ++j)
274 {
275 st = commandButton->getScienceVec()[ j ];
276
277 if( commandButton->getCommandType() != GUI_COMMAND_PURCHASE_SCIENCE )
278 {
279 if( !player->hasScience( st ) && j > 0 )
280 {
281 //If we're not looking at a command button that purchases a science, then
282 //it means we are looking at a command button that can USE the science. This
283 //means we want to get the description for the previous science -- the one
284 //we can use, not purchase!
285 st = commandButton->getScienceVec()[ j - 1 ];
286 }
287
288 //Now that we got the science for the button that executes the science, we need
289 //to generate a simpler help text!
290 fireScienceButton = TRUE;
291
292 break;
293 }
294 else if( !player->hasScience( st ) )
295 {
296 //Purchase science case. The first science we run into that we don't have, that's the
297 //one we'll want to show!
298 break;
299 }
300 }
301 }
302 else if(commandButton->getScienceVec().size() == 1 )
303 {
304 st = commandButton->getScienceVec()[ 0 ];
305 if( commandButton->getCommandType() != GUI_COMMAND_PURCHASE_SCIENCE )
306 {
307 //Now that we got the science for the button that executes the science, we need
308 //to generate a simpler help text!
309 fireScienceButton = TRUE;
310 }
311 }
312 }
313
314 if( commandButton->getDescriptionLabel().isNotEmpty() )
315 {
316 descrip = TheGameText->fetch(commandButton->getDescriptionLabel());
317
318 Drawable *draw = TheInGameUI->getFirstSelectedDrawable();
319 Object *selectedObject = draw ? draw->getObject() : NULL;
320 if( selectedObject )
321 {
322 //Special case: Append status of overcharge on China power plant.
323 if( commandButton->getCommandType() == GUI_COMMAND_TOGGLE_OVERCHARGE )
324 {
325 {
326 OverchargeBehaviorInterface *obi;
327 for( BehaviorModule **bmi = selectedObject->getBehaviorModules(); *bmi; ++bmi )
328 {
329 obi = (*bmi)->getOverchargeBehaviorInterface();
330 if( obi )
331 {
332 descrip.concat( L"\n" );
333 if( obi->isOverchargeActive() )
334 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipNukeReactorOverChargeIsOn" ) );
335 else
336 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipNukeReactorOverChargeIsOff" ) );
337 }
338 }
339 }
340 } //End overcharge special case
341
342 //Special case: When building units & buildings, the CanMakeType determines reasons for not being able to buy stuff.
343 else if( thingTemplate )
344 {
345 CanMakeType makeType = TheBuildAssistant->canMakeUnit( selectedObject, commandButton->getThingTemplate() );
346 switch( makeType )
347 {
348 case CANMAKE_NO_MONEY:
349 descrip.concat( L"\n\n" );
350 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipNotEnoughMoneyToBuild" ) );
351 break;
352 case CANMAKE_QUEUE_FULL:
353 descrip.concat( L"\n\n" );
354 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotPurchaseBecauseQueueFull" ) );
355 break;
356 case CANMAKE_PARKING_PLACES_FULL:
357 descrip.concat( L"\n\n" );
358 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotBuildUnitBecauseParkingFull" ) );
359 break;
360 case CANMAKE_MAXED_OUT_FOR_PLAYER:
361 descrip.concat( L"\n\n" );
362 if ( thingTemplate->isKindOf( KINDOF_STRUCTURE ) )
363 {
364 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotBuildBuildingBecauseMaximumNumber" ) );
365 }
366 else
367 {
368 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotBuildUnitBecauseMaximumNumber" ) );
369 }
370 break;
371 //case CANMAKE_NO_PREREQ:
372 // descrip.concat( L"\n\n" );
373 // descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotBuildDueToPrerequisites" ) );
374 // break;
375 }
376 }
377
378 //Special case: When building upgrades
379 else if( upgradeTemplate && !player->hasUpgradeInProduction( upgradeTemplate ) )
380 {
381 if( commandButton->getCommandType() == GUI_COMMAND_PLAYER_UPGRADE ||
382 commandButton->getCommandType() == GUI_COMMAND_OBJECT_UPGRADE )
383 {
384 ProductionUpdateInterface *pui = selectedObject->getProductionUpdateInterface();
385 if( pui && pui->getProductionCount() == MAX_BUILD_QUEUE_BUTTONS )
386 {
387 descrip.concat( L"\n\n" );
388 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipCannotPurchaseBecauseQueueFull" ) );
389 }
390 else if( !TheUpgradeCenter->canAffordUpgrade( ThePlayerList->getLocalPlayer(), upgradeTemplate, FALSE ) )
391 {
392 descrip.concat( L"\n\n" );
393 descrip.concat( TheGameText->fetch( "TOOLTIP:TooltipNotEnoughMoneyToBuild" ) );
394 }
395 }
396 }
397
398 }
399
400 }
401
402 name = TheGameText->fetch(commandButton->getTextLabel().str());
403
404 if( thingTemplate && commandButton->getCommandType() != GUI_COMMAND_PURCHASE_SCIENCE )
405 {
406 //We are either looking at building a unit or a structure that may or may not have any
407 //prerequisites.
408
409 //Format the cost only when we have to pay for it.
410 costToBuild = thingTemplate->calcCostToBuild( player );
411 if( costToBuild > 0 )
412 {
413 cost.format( TheGameText->fetch("TOOLTIP:Cost"), costToBuild );
414 }
415
416 // ask each prerequisite to give us a list of the non satisfied prerequisites
417 for( Int i=0; i<thingTemplate->getPrereqCount(); i++ )
418 {
419 prereq = thingTemplate->getNthPrereq(i);
420 requiresList = prereq->getRequiresList(player);
421
422 if( requiresList != UnicodeString::TheEmptyString )
423 {
424 // make sure to put in 'returns' to space things correctly
425 if (firstRequirement)
426 firstRequirement = false;
427 else
428 requires.concat(L", ");
429 }
430 requires.concat(requiresList);
431 }
432 if( !requires.isEmpty() )
433 {
434 UnicodeString requireFormat = TheGameText->fetch("CONTROLBAR:Requirements");
435 requires.format(requireFormat.str(), requires.str());
436 if(!descrip.isEmpty())
437 descrip.concat(L"\n");
438 descrip.concat(requires);
439
440 }
441 }
442 else if( upgradeTemplate )
443 {
444 //We are looking at an upgrade purchase icon. Maybe we already purchased it?
445
446 Bool hasUpgradeAlready = player->hasUpgradeComplete( upgradeTemplate );
447 Bool hasConflictingUpgrade = FALSE;
448 Bool missingScience = FALSE;
449 Bool playerUpgradeButton = commandButton->getCommandType() == GUI_COMMAND_PLAYER_UPGRADE;
450 Bool objectUpgradeButton = commandButton->getCommandType() == GUI_COMMAND_OBJECT_UPGRADE;
451
452 if( !hasUpgradeAlready )
453 {
454 //Check if the first selected object has the specified upgrade.
455 Drawable *draw = TheInGameUI->getFirstSelectedDrawable();
456 if( draw )
457 {
458 Object *object = draw->getObject();
459 if( object )
460 {
461 hasUpgradeAlready = object->hasUpgrade( upgradeTemplate );
462 if( objectUpgradeButton )
463 {
464 hasConflictingUpgrade = !object->affectedByUpgrade( upgradeTemplate );
465 }
466 }
467 }
468 }
469 if( hasConflictingUpgrade && !hasUpgradeAlready )
470 {
471 if( commandButton->getConflictingLabel().isNotEmpty() )
472 {
473 descrip = TheGameText->fetch( commandButton->getConflictingLabel() );
474 }
475 else
476 {
477 descrip = TheGameText->fetch( "TOOLTIP:HasConflictingUpgradeDefault" );
478 }
479 }
480 else if( hasUpgradeAlready && ( playerUpgradeButton || objectUpgradeButton ) )
481 {
482 //See if we can fetch the "already upgraded" text for this upgrade. If not.... use the default "fill me in".
483 if( commandButton->getPurchasedLabel().isNotEmpty() )
484 {
485 descrip = TheGameText->fetch( commandButton->getPurchasedLabel() );
486 }
487 else
488 {
489 descrip = TheGameText->fetch( "TOOLTIP:AlreadyUpgradedDefault" );
490 }
491 }
492 else if( !hasUpgradeAlready )
493 {
494
495 //Do we have a prerequisite science?
496 for( Int i = 0; i < commandButton->getScienceVec().size(); i++ )
497 {
498 ScienceType st = commandButton->getScienceVec()[ i ];
499 if( !player->hasScience( st ) )
500 {
501 missingScience = TRUE;
502 break;
503 }
504 }
505
506 //Determine the cost of the upgrade.
507 costToBuild = upgradeTemplate->calcCostToBuild( player );
508 if( costToBuild > 0 )
509 {
510 cost.format( TheGameText->fetch("TOOLTIP:Cost"), costToBuild );
511 }
512
513 if( missingScience )
514 {
515 if( !descrip.isEmpty() )
516 descrip.concat(L"\n");
517 requires.format( TheGameText->fetch( "CONTROLBAR:Requirements" ).str(), TheGameText->fetch( "CONTROLBAR:GeneralsPromotion" ).str() );
518 descrip.concat( requires );
519 }
520 }
521 }
522 else if( st != SCIENCE_INVALID && !fireScienceButton )
523 {
524 TheScienceStore->getNameAndDescription(st, name, descrip);
525
526 costToBuild = TheScienceStore->getSciencePurchaseCost( st );
527 if( costToBuild > 0 )
528 {
529 cost.format( TheGameText->fetch("TOOLTIP:ScienceCost"), costToBuild );
530 }
531
532 // ask each prerequisite to give us a list of the non satisfied prerequisites
533 if( thingTemplate )
534 {
535 for( Int i=0; i<thingTemplate->getPrereqCount(); i++ )
536 {
537 prereq = thingTemplate->getNthPrereq(i);
538 requiresList = prereq->getRequiresList(player);
539
540 if( requiresList != UnicodeString::TheEmptyString )
541 {
542 // make sure to put in 'returns' to space things correctly
543 if (firstRequirement)
544 firstRequirement = false;
545 else
546 requires.concat(L", ");
547 }
548 requires.concat(requiresList);
549 }
550 if( !requires.isEmpty() )
551 {
552 UnicodeString requireFormat = TheGameText->fetch("CONTROLBAR:Requirements");
553 requires.format(requireFormat.str(), requires.str());
554 if(!descrip.isEmpty())
555 descrip.concat(L"\n");
556 descrip.concat(requires);
557 }
558 }
559
560 }
561 }
562 else if(tooltipWin)
563 {
564
565 if( tooltipWin == TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBar.wnd:MoneyDisplay")))
566 {
567 name = TheGameText->fetch("CONTROLBAR:Money");
568 descrip = TheGameText->fetch("CONTROLBAR:MoneyDescription");
569 }
570 else if(tooltipWin == TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBar.wnd:PowerWindow")) )
571 {
572 name = TheGameText->fetch("CONTROLBAR:Power");
573 descrip = TheGameText->fetch("CONTROLBAR:PowerDescription");
574
575 Player *playerToDisplay = NULL;
576 if(TheControlBar->isObserverControlBarOn())
577 playerToDisplay = TheControlBar->getObserverLookAtPlayer();
578 else
579 playerToDisplay = ThePlayerList->getLocalPlayer();
580
581 if( playerToDisplay && playerToDisplay->getEnergy() )
582 {
583 Energy *energy = playerToDisplay->getEnergy();
584 descrip.format(descrip, energy->getProduction(), energy->getConsumption());
585 }
586 else
587 {
588 descrip.format(descrip, 0, 0);
589 }
590 }
591 else if(tooltipWin == TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBar.wnd:GeneralsExp")) )
592 {
593 name = TheGameText->fetch("CONTROLBAR:GeneralsExp");

595 }
596 else
597 {
598 DEBUG_ASSERTCRASH(FALSE, ("ControlBar::populateBuildTooltipLayout We attempted to call the popup tooltip on a game window that has yet to be hand coded in as this fuction was/is designed for only buttons but has been hacked to work with GameWindows."));
599 return;
600 }
601
602 }
603 GameWindow *win = TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBarPopupDescription.wnd:StaticTextName"));
604 if(win)
605 {
606 GadgetStaticTextSetText(win, name);
607 }
608
609 win = TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBarPopupDescription.wnd:StaticTextCost"));
610 if(win)
611 {
612 if( costToBuild > 0 )

614 win->winHide( FALSE );

619 win->winHide( TRUE );

623 win = TheWindowManager->winGetWindowFromId(m_buildToolTipLayout->getFirstWindow(), TheNameKeyGenerator->nameToKey("ControlBarPopupDescription.wnd:StaticTextDescription"));

652 parent->winGetPosition(&pos.x, &pos.y);
653// if(size.y + diffSize < 102)
654// {
655//
656// parent->winSetPosition(pos.x, pos.y - (102 - (newSize.y + size.y + diffSize) ));
657// }
658// else
659
660// heightChange = controlBarPos.y - m_defaultControlBarPosition.y;
661
662 GameWindow *marker = TheWindowManager->winGetWindowFromId(NULL,winNamekey);

665 {
666 return;
667 }
668 TheControlBar->getBackgroundMarkerPos(&basePos.x, &basePos.y);
669 ICoord2D curPos, offset;
670 marker->winGetScreenPosition(&curPos.x,&curPos.y);
671
672 offset.x = curPos.x - basePos.x;
673 offset.y = curPos.y - basePos.y;
674
675 parent->winSetPosition(pos.x, (pos.y - diffSize) + (offset.y - lastOffset.y));
676
677 lastOffset.x = offset.x;
678 lastOffset.y = offset.y;
679
680 win->winGetSize(&size.x, &size.y);
681 win->winSetSize(size.x, size.y + diffSize);
682
683 GadgetStaticTextSetText(win, descrip);
684 }
685 m_buildToolTipLayout->hide(FALSE);
