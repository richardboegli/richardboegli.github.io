266void
267AudibleSoundClass::Set_Buffer (SoundBufferClass *buffer)
268{
269 //
270 // Delay the release of the buffer (fixes a sync bug
271 // with Miles internals).
272 //
273 if (m_Buffer != NULL) {
274 WWAudioThreadsClass::Add_Delayed_Release_Object (m_Buffer);

276 }

278
279 // Stop playing if necessary
280 bool resume = false;

283 }
284

286 if (m_Buffer != NULL) {
287 m_Length = m_Buffer->Get_Duration ();
288 }
289
290 // Reinitialize the handle with this new data
291 Initialize_Miles_Handle ();
292
293 // Resume playing if necessary
294 if (resume) {
295 Play ();
296 }
297
298 return ;
299}

331//
332// Play
333//
335bool
336AudibleSoundClass::Play (bool alloc_handle)
337{
338 MMSLockClass lock;
339
340 // If we don't have a valid handle already, try to get one from miles
341 if (alloc_handle && (m_pConvertedFormat == NULL)) {
342 Allocate_Miles_Handle ();
343 }
344
345 // Let the audio system know this sound is playing
346 if (m_State != STATE_PLAYING) {
347 WWAudioClass::Get_Instance ()->Add_To_Playlist (this);
348 m_State = STATE_PLAYING;
349 m_Timestamp = ::GetTickCount ();
350 m_LoopsLeft = m_LoopCount;
351
352 // If we have a valid handle, then start playing the sample
353 if (m_SoundHandle != NULL) {
354 m_SoundHandle->Start_Sample ();
355 }
356
357 m_CurrentPosition = m_StartOffset * m_Length;
358 if (m_CurrentPosition > 0) {
359 Seek (m_CurrentPosition);
360 }
361
362
363 // Fire an event
364 On_Event (AudioCallbackClass::EVENT_SOUND_STARTED);
365
366 //
367 // Create the associate logical sound (if necessary)
368 //
369 if (m_LogicalSound == NULL && m_Definition != NULL) {
370 m_LogicalSound = m_Definition->Create_Logical ();
371 }
372
373 //
374 // Add this logical sound to the scene
375 //
376 if (m_LogicalSound != NULL) {
377 m_LogicalSound->Set_User_Data (m_UserObj, m_UserData);
378 m_LogicalSound->Set_Transform (m_Transform);
379 m_LogicalSound->Add_To_Scene ();
380 }
381
382 //
383 // Should we send off the text notification?
384 //
385 if (m_IsCulled == false && m_Definition != NULL) {
386 const StringClass &text = m_Definition->Get_Display_Text ();
387 WWAudioClass::Get_Instance ()->Fire_Text_Callback (this, text);
388 }
389 }
390
391 return true;
392}

456//
457// Stop
458//
460bool
461AudibleSoundClass::Stop (bool remove_from_playlist)
462{
463 MMSLockClass lock;
464
465 // Assume failure
466 bool retval = false;
467
468 if ((m_State == STATE_PAUSED) ||
469 (m_State == STATE_PLAYING)) {
470
471 // Actually stop the sample from playing
472 if (m_SoundHandle != NULL) {
473 m_SoundHandle->Stop_Sample ();
474 }
475
476 // Free up the handle we have been using
477 Free_Miles_Handle ();
478 m_State = STATE_STOPPED;
479 retval = true;
480
481 // Reset some of the playing attributes
482 m_Timestamp = 0;
483 m_CurrentPosition = 0;
484
485 if (remove_from_playlist) {
486 WWAudioClass::Get_Instance ()->Remove_From_Playlist (this);
487 }
488
489 //
490 // Stop the logical portion of the sound
491 //
492 if (m_LogicalSound != NULL && m_LogicalSound->Is_Single_Shot () == false) {
493 m_LogicalSound->Remove_From_Scene ();
494 }
495 }
496
497 // Return the true/false result code
498 return retval;
499}

504// Seek
505//
507void
508AudibleSoundClass::Seek (unsigned long milliseconds)
509{
510 MMSLockClass lock;
511
512 if ((milliseconds >= 0) && (milliseconds < m_Length)) {
513
514 // Record our new position and recalculate the 'starting' timestamp
515 // from this information
516 m_CurrentPosition = milliseconds;
517 if (m_State == STATE_PLAYING) {
518 m_Timestamp = ::GetTickCount () - m_CurrentPosition;
519 }
520
521 // Update the actual sound data if we are playing the sound
522 if (m_SoundHandle != NULL) {
523 m_SoundHandle->Set_Sample_MS_Position (m_CurrentPosition);
524 }
525 }
526
527 return ;
528}

533// Set_Miles_Handle
534//
536void
537AudibleSoundClass::Set_Miles_Handle (MILES_HANDLE handle)
538{
539 //
540 // Start fresh
541 //
542 Free_Miles_Handle ();
543
544 //
545 // Is our data valid?
546 //
547 if (handle != INVALID_MILES_HANDLE && m_Buffer != NULL) {
548
549 //
550 // Determine which type of sound handle to create, streaming or standard 2D
551 //
552 if (m_Buffer->Is_Streaming ()) {
553 m_SoundHandle = W3DNEW SoundStreamHandleClass;
554 } else {
555 m_SoundHandle = W3DNEW Sound2DHandleClass;
556 }
557
558 //
559 // Configure the sound handle
560 //
561 m_SoundHandle->Set_Miles_Handle (handle);
562
563 //
564 // Use this new handle
565 //
566 Initialize_Miles_Handle ();
567 }
568
569 return ;

574//
575// Initialize_Miles_Handle
576//
578void
579AudibleSoundClass::Initialize_Miles_Handle (void)
580{
581 MMSLockClass lock;
582
583 // If this sound is already playing, then update its
584 // playing position to make sure we really should
585 // be playing it... (it will free the miles handle if not)
586 if (m_State == STATE_PLAYING) {
587 Update_Play_Position ();
588 }
589
590 // Do we have a valid sample handle from miles?
591 if (m_SoundHandle != NULL) {
592
593 //
594 // Initialize the handle
595 //
596 m_SoundHandle->Initialize (m_Buffer);
597
598 //
599 // Record the total length of the sample in milliseconds...
600 //
601 m_SoundHandle->Get_Sample_MS_Position ((S32 *)&m_Length, NULL);
602
603 //
604 // Pass our cached settings onto miles
605 //
606 m_SoundHandle->Set_Sample_Volume (0);
607 m_SoundHandle->Set_Sample_Pan (int(m_Pan * 127.0F));
608 m_SoundHandle->Set_Sample_Loop_Count (m_LoopCount);
609
610 //
611 // Apply the pitch factor (if necessary)
612 //
613 if (m_PitchFactor != 1.0F) {
614 Set_Pitch_Factor (m_PitchFactor);
615 }
616
617 // If this sound is already playing (and just now got a handle)
618 // then make sure we start it.
619 if (m_State == STATE_PLAYING) {
620 m_SoundHandle->Start_Sample ();
621
622 // Update the loop count based on the number of loops left
623 m_SoundHandle->Set_Sample_Loop_Count (m_LoopsLeft);
624 }
625
626 // Seek to the position of the sound where we last left off.
627 // For example, this sound could have gotten bumped due to a low priority,
628 // but is now back and ready to resume at the position it would have been
629 // at if it was never bumped.
630 Seek (m_CurrentPosition);
631
632 //
633 // Pass the 'real' volume onto miles
634 //
635 float real_volume = Determine_Real_Volume ();
636 m_SoundHandle->Set_Sample_Volume (int(real_volume * 127.0F));
637
638 //
639 // Associate this object instance with the handle
640 //
641 m_SoundHandle->Set_Sample_User_Data (INFO_OBJECT_PTR, (S32)this);
642 }
643
644 return ;
645}

935//
936// Update_Play_Position
937//
939void
940AudibleSoundClass::Update_Play_Position (void)
941{
942 // Determine the current offset from the beginning of the sound buffer.
943 unsigned long play_time = ::GetTickCount () - m_Timestamp;
944 m_CurrentPosition = play_time;
945
946 // Have we gone past the end of a sounds play-time?
947 if ((m_CurrentPosition > m_Length) && (m_Length > 0)) {
948
949 // Normalize our position and timestamp information
950 m_CurrentPosition = m_CurrentPosition % m_Length;
951 m_Timestamp = ::GetTickCount () - m_CurrentPosition;
952
953 // Decrement our count of remaining loops (if necessary)
954 if (m_LoopCount != INFINITE_LOOPS) {
955 m_LoopsLeft -= (play_time / m_Length);
956 }
957
958 // Trigger the 'end loop' event
959 On_Loop_End ();
960 }
961
962 return ;
963}

988//
990void
991AudibleSoundClass::On_Loop_End (void)
992{
993 // Determine if the sound is actually finished or still looping
994 if ((m_LoopCount != INFINITE_LOOPS) && (m_LoopsLeft < 1)) {
