71void MissileLauncherBuildingUpdate::switchToState(DoorStateType dst)
72{
73 if (m_doorState == dst)
74 return;
75
76 const MissileLauncherBuildingUpdateModuleData* d = getMissileLauncherBuildingUpdateModuleData();
77 ModelConditionFlags clr, set;
78 UnsignedInt now = TheGameLogic->getFrame();
79 switch (dst)
80 {
81 case DOOR_CLOSED:
82 DEBUG_LOG(("switch to state DOOR_CLOSED at %d\n",now));
84 clr.set(MODELCONDITION_DOOR_1_WAITING_TO_CLOSE);
85 clr.set(MODELCONDITION_DOOR_1_CLOSING);
86 clr.set(MODELCONDITION_DOOR_1_OPENING);
87 clr.set(MODELCONDITION_DOOR_1_WAITING_OPEN);
88 m_timeoutFrame = 0;
89 m_timeoutState = DOOR_CLOSED;
90 if (d->m_closedFX)
91 {
92 const Coord3D *pos = getObject()->getPosition();
93 FXList::doFXPos(d->m_closedFX, pos);
94 }
95 if (m_openIdleAudio.isCurrentlyPlaying())
96 {
97 TheAudio->removeAudioEvent(m_openIdleAudio.getPlayingHandle());
98 }
99 break;
100
101 case DOOR_OPENING:
102 DEBUG_LOG(("switch to state DOOR_OPENING at %d\n",now));
104 clr.set(MODELCONDITION_DOOR_1_WAITING_TO_CLOSE);
105 clr.set(MODELCONDITION_DOOR_1_CLOSING);
106 clr.set(MODELCONDITION_DOOR_1_WAITING_OPEN);
107 set.set(MODELCONDITION_DOOR_1_OPENING);
108 // we want this to be done BEFORE the power is ready, so
109 // end it one frame ahead.
110 m_timeoutFrame = m_specialPowerModule->getReadyFrame() - 1;
111 m_timeoutState = DOOR_OPEN;
112 if (d->m_openingFX)
113 {
114 const Coord3D *pos = getObject()->getPosition();
115 FXList::doFXPos(d->m_openingFX, pos);
116 }
117 if (m_openIdleAudio.isCurrentlyPlaying())
118 {
119 TheAudio->removeAudioEvent(m_openIdleAudio.getPlayingHandle());
120 }
121 break;
122
123 case DOOR_OPEN:
124 DEBUG_LOG(("switch to state DOOR_OPEN at %d\n",now));
126 clr.set(MODELCONDITION_DOOR_1_WAITING_TO_CLOSE);
127 clr.set(MODELCONDITION_DOOR_1_CLOSING);
128 clr.set(MODELCONDITION_DOOR_1_OPENING);
129 set.set(MODELCONDITION_DOOR_1_WAITING_OPEN);
130 m_timeoutFrame = 0;
131 m_timeoutState = DOOR_OPEN;
132 if (d->m_openFX)
133 {
134 const Coord3D *pos = getObject()->getPosition();
135 FXList::doFXPos(d->m_openFX, pos);
136 }
137 if (!m_openIdleAudio.isCurrentlyPlaying())
138 {
139 m_openIdleAudio.setPlayingHandle(TheAudio->addAudioEvent(&m_openIdleAudio));
140 }
141 break;
142
143 case DOOR_WAITING_TO_CLOSE:
144 DEBUG_LOG(("switch to state DOOR_WAITING_TO_CLOSE at %d\n",now));
146 clr.set(MODELCONDITION_DOOR_1_CLOSING);
147 clr.set(MODELCONDITION_DOOR_1_OPENING);
148 clr.set(MODELCONDITION_DOOR_1_WAITING_OPEN);
149 set.set(MODELCONDITION_DOOR_1_WAITING_TO_CLOSE);
150 m_timeoutFrame = now + d->m_doorWaitOpenTime;
151 m_timeoutState = DOOR_CLOSING;
152 if (d->m_waitingToCloseFX)
153 {
154 const Coord3D *pos = getObject()->getPosition();
155 FXList::doFXPos(d->m_waitingToCloseFX, pos);
156 }
157 if (m_openIdleAudio.isCurrentlyPlaying())
158 {
159 TheAudio->removeAudioEvent(m_openIdleAudio.getPlayingHandle());
160 }
161 break;
162
163 case DOOR_CLOSING:
164 DEBUG_LOG(("switch to state DOOR_CLOSING at %d\n",now));
166 clr.set(MODELCONDITION_DOOR_1_WAITING_TO_CLOSE);
167 clr.set(MODELCONDITION_DOOR_1_WAITING_OPEN);
168 clr.set(MODELCONDITION_DOOR_1_OPENING);
169 set.set(MODELCONDITION_DOOR_1_CLOSING);
170 m_timeoutFrame = now + d->m_doorClosingTime;
171
172 {
173 Int delta = m_specialPowerModule->getReadyFrame() - now;
174 if (m_timeoutFrame > now + delta/2)
175 m_timeoutFrame = now + delta/2;
176 }
177
178 m_timeoutState = DOOR_CLOSED;
179 if (d->m_closingFX)
180 {
181 const Coord3D *pos = getObject()->getPosition();
182 FXList::doFXPos(d->m_closingFX, pos);
183 }
184 if (m_openIdleAudio.isCurrentlyPlaying())
185 {
186 TheAudio->removeAudioEvent(m_openIdleAudio.getPlayingHandle());
187 }
188 break;
189#ifdef _DEBUG
190 default:
191 DEBUG_CRASH(("unknown state"));
192 break;
193#endif
194 }
195 m_doorState = dst;
196 getObject()->clearAndSetModelConditionFlags(clr, set);
197
198 if (m_timeoutFrame > 0)
199 {
200 Drawable* draw = getObject()->getDrawable();
201 if (draw)
202 draw->setAnimationLoopDuration(m_timeoutFrame - now);
203 }
204}

229UpdateSleepTime MissileLauncherBuildingUpdate::update( void )
230{
231 const MissileLauncherBuildingUpdateModuleData* d = getMissileLauncherBuildingUpdateModuleData();
232
233 UnsignedInt now = TheGameLogic->getFrame();
234
235 // If we are under construction, any decision we make about door status could be wrong.
236 // Our special power module is randomly going to be initialized or not (which would result
237 // in him reporting a 0 frame ready, which means we will start open).
238 if( getObject()->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) )
239 return UPDATE_SLEEP_NONE;
240
241 if (!m_specialPowerModule)
242 {
243 m_specialPowerModule = getObject()->getSpecialPowerModule(d->m_specialPowerTemplate);
244 DEBUG_ASSERTCRASH(m_specialPowerModule, ("Missing special power"));
245 }
246
247 if (m_specialPowerModule)
248 {
249 UnsignedInt readyFrame = m_specialPowerModule->getReadyFrame();
250 UnsignedInt whenToStartOpening = (readyFrame >= d->m_doorOpenTime) ? (readyFrame - d->m_doorOpenTime) : 0;
251
252 if (m_timeoutFrame != 0 && now > m_timeoutFrame)
253 {
254 switchToState(m_timeoutState);
255 }
256
257 DEBUG_ASSERTCRASH(!TheGlobalData->m_specialPowerUsesDelay || !(m_specialPowerModule->isReady() && m_doorState != DOOR_OPEN), ("door is not fully open when specialpower is ready!"));
258
259 if (m_doorState != DOOR_OPEN && m_specialPowerModule->isReady())
260 {
261 DEBUG_LOG(("*** had to POP the door open!\n"));
262 switchToState(DOOR_OPEN);
263 }
264 else if (m_doorState == DOOR_CLOSED && now >= whenToStartOpening)
265 {
266 switchToState(DOOR_OPENING);
267 }
268 }
269 return UPDATE_SLEEP_NONE;
270}
