393{

396 {
397 return TRUE;
398 }
399
400 m_device = TheAudio->device();
401
402 if( m_device )
403 {
404 AudioDeviceAttribsAdd( m_device, m_masterAttribs );
405 AudioDeviceAttribsAdd( m_device, m_masterFadeAttribs );
406 }
407
408 for (int i = 0; i < NUM_DLG_PRIORITIES; ++i) {
409 createSpeaker(dlgPriorityNames[i], (i + 1), 3000, 300);
410 }
411
412 m_on = TRUE;
413

913void SpeechManager::addDialogEvent(const AudioEventRTS *eventRTS, Speech* speech, AudioEventRTS *returnEvent)
914{
915 if (!eventRTS) {
916 return;
917 }
918
919 Speech *useSpeech = speech;
920 if (!useSpeech) {
921 useSpeech = getSpeech(eventRTS->m_eventName.str());
922 }
923
924 if (!useSpeech) {
925 return;
926 }
927
928 if (1 || (eventRTS->m_priorityBoost == 0 &&
929 eventRTS->m_overrideLoopingBehavior == 0 &&
930 eventRTS->m_timeOfDay == useSpeech->info.m_timeOfDay)) {
931 // We can use the default sound, so no need to create and add a new one.
932 say(useSpeech);
933 if (returnEvent) {
934 (*returnEvent) = (*eventRTS);
935 (*returnEvent).m_speechToPlay = useSpeech;
936 (*returnEvent).m_isCurrentlyPlaying = true;
937 }
938 return;
939 }
940
941 Speech tmpSpeech = (*useSpeech);
942 tmpSpeech.info.m_priority += eventRTS->m_priorityBoost;
943 // Can't go less than 0 priority or > NUM_DLG_PRIORITIES
944 if (tmpSpeech.info.m_priority < 0) {
945 tmpSpeech.info.m_priority = 0;
946 } else if (tmpSpeech.info.m_priority > NUM_DLG_PRIORITIES) {
947 tmpSpeech.info.m_priority = NUM_DLG_PRIORITIES;
948 }
949
950 tmpSpeech.info.m_timeOfDay = eventRTS->m_timeOfDay;
951#if 0
952 // CRASHING BUG IN HERE. NEED TO FIX.
953 Speech* playSpeech = addTemporaryDialog(tmpSpeech);
954 if (returnEvent) {
955 (*returnEvent) = (*eventRTS);
956 returnEvent->m_speechToPlay = NULL;
957 }
958
959 say(playSpeech);
960#endif
961}

963//============================================================================
964// SpeechManager::removeDialogEvent
965//============================================================================
966
967void SpeechManager::removeDialogEvent(const AudioEventRTS *eventRTS, AudioEventRTS* eventToUse)
968{
969 if (!eventRTS) {
970 return;
971 }
972
973 Speech *useSpeech = eventRTS->m_speechToPlay;
974
975 if (!useSpeech) {
976 useSpeech = getSpeech(eventRTS->m_eventName.str());
977 }
978
979 if (!useSpeech) {
980 return;
981 }
982
983 if (eventToUse) {
984 (*eventToUse) = (*eventRTS);
985 eventToUse->m_speechToPlay = useSpeech;
986 }
987
988 Bool isPlaying = true;
989 LListNode *node = m_speakers.firstNode();
990 while (node && isPlaying) {
991 Speaker *pCurrSpeaker = (Speaker*) node->item();
992 if (pCurrSpeaker->saying() == useSpeech || pCurrSpeaker->isGoingToSay(useSpeech)) {
993 pCurrSpeaker->cancel(useSpeech);
994 isPlaying = false;
995 }
996 }
997
998 if (eventToUse) {
999 (*eventToUse).m_isCurrentlyPlaying = isPlaying;
1000 }
1001}

1021//============================================================================
1022// SpeechManager::getFilenameForPlay
1023//============================================================================
1024AsciiString SpeechManager::getFilenameForPlay( Speech *speech )
1025{
1026 if (!speech) {
1027 return AsciiString::TheEmptyString;
1028 }
1029
1030 SpeechInfo* speechInfo = getSpeechInfo(speech);
1031
1032 if (!speechInfo) {
1033 return AsciiString::TheEmptyString;
1034 }
1035
1036 Int regularSamples = speechInfo->m_dialogFiles.size();
1037 Int eveningSamples = speechInfo->m_dialogFilesEvening.size();
1038 Int morningSamples = speechInfo->m_dialogFilesMorning.size();
1039 Int nightSamples = speechInfo->m_dialogFilesNight.size();
1040 Int numSamples = regularSamples + eveningSamples + morningSamples + nightSamples;
1041
1042 // using a random number generator, select a random sample from all the samples
1043 // that correspond to this sound name
1044 Int soundToPlay = GameClientRandomValue(0, numSamples - 1);
1045 char name[_MAX_PATH];
1046
1047 if (soundToPlay < regularSamples) {
1048 strcpy(name, speechInfo->m_dialogFiles[soundToPlay].str());
1049 } else if (soundToPlay < (regularSamples + eveningSamples)) {
1050 strcpy(name, speechInfo->m_dialogFilesEvening[soundToPlay - regularSamples].str());
1051 } else if (soundToPlay < (regularSamples + eveningSamples + morningSamples)) {
1052 strcpy(name, speechInfo->m_dialogFilesMorning[soundToPlay - regularSamples - eveningSamples].str());
1053 } else if (soundToPlay < numSamples) {
1054 strcpy(name, speechInfo->m_dialogFilesNight[soundToPlay - regularSamples - eveningSamples - morningSamples].str());
1055 } else {
1056 return AsciiString::TheEmptyString;
1057 }
1058
1059
1060
1061 // @todo this should all go into a function "generateLocalizedSoundName"
1062 Bool localized = FALSE;
1063 if ( name[0] == '$' )
1064 {
1065 localized = TRUE;
1066 char nameTemp[_MAX_PATH];
1067 strcpy(nameTemp, name + 1);
1068 strcpy(name, nameTemp);
1069 }
1070
1071 if ( name[0] == 0 )
1072 {
1073 return NULL;
1074 }
1075
1076 AsciiString path;
1077 char *localDir = "";
1078
1079 if ( localized )
1080 {
1081 localDir = "english\\";
1082 }
1083
1084 path.format( "data\\audio\\sounds\\%s%s.wav", localDir, name);
1085 return path;
