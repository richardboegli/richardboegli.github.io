77void INI::parseWindowTransitions( INI* ini )
78{
79 AsciiString name;
80 TransitionGroup *g;
81
82 // read the name
83 const char* c = ini->getNextToken();
84 name.set( c );
85
86 // find existing item if present
87
88 DEBUG_ASSERTCRASH( TheTransitionHandler, ("parseWindowTransitions: TheTransitionHandler doesn't exist yet\n") );
89 if( !TheTransitionHandler )
90 return;
91
92 // If we have a previously allocated control bar, this will return a cleared out pointer to it so we
93 // can overwrite it
94 g = TheTransitionHandler->getNewGroup( name );
95
96 // sanity
97 DEBUG_ASSERTCRASH( g, ("parseWindowTransitions: Unable to allocate group '%s'\n", name.str()) );
98
99 // parse the ini definition
100 ini->initFromINI( g, TheTransitionHandler->getFieldParse() );
101
102
103}

107Transition *getTransitionForStyle( Int style )
108{
109 switch (style) {

112 case BUTTON_TRANSITION_FLASH:
113 return NEW ButtonFlashTransition;
114 case WIN_FADE_TRANSITION:
115 return NEW FadeTransition;
116 case WIN_SCALE_UP_TRANSITION:

124 case COUNT_UP_TRANSITION:
125 return NEW CountUpTransition;
126 case FULL_FADE_TRANSITION:
127 return NEW FullFadeTransition;
128 case TEXT_ON_FRAME_TRANSITION:
129 return NEW TextOnFrameTransition;
130 case REVERSE_SOUND_TRANSITION:
131 return NEW ReverseSoundTransition;
132
133 case MAINMENU_MEDIUM_SCALE_UP_TRANSITION:

290void TransitionGroup::reverse( void )
291{
292 Int totalFrames =0;
293 m_directionMultiplier = -1;
294
295 TransitionWindowList::iterator it = m_transitionWindowList.begin();
296 while (it != m_transitionWindowList.end())
297 {
298 TransitionWindow *tWin = *it;
299 Int winFrames = tWin->getTotalFrames();
300 if(winFrames > totalFrames)
301 totalFrames = winFrames;
302 it++;
303 }
304 it = m_transitionWindowList.begin();
305 while (it != m_transitionWindowList.end())
306 {
307 TransitionWindow *tWin = *it;
308 tWin->reverse(totalFrames);
309 it++;
310 }
311 m_currentFrame = totalFrames;
312// m_currentFrame ++;
313}

403void GameWindowTransitionsHandler::update( void )
404{
405 if(m_drawGroup != m_currentGroup)
406 m_secondaryDrawGroup = m_drawGroup;
407 else
408 m_secondaryDrawGroup = NULL;
409
410 m_drawGroup = m_currentGroup;
411 if(m_currentGroup && !m_currentGroup->isFinished())
412 m_currentGroup->update();
413
414 if(m_currentGroup && m_currentGroup->isFinished() && m_currentGroup->isFireOnce())
415 {
416 m_currentGroup = NULL;
417 }
418
419 if(m_currentGroup && m_pendingGroup && m_currentGroup->isFinished())
420 {
421 m_currentGroup = m_pendingGroup;
422 m_pendingGroup = NULL;
423 }
424
425 if(!m_currentGroup && m_pendingGroup)
426 {
427 m_currentGroup = m_pendingGroup;
428 m_pendingGroup = NULL;
429 }
430
431 if(m_currentGroup && m_currentGroup->isFinished() && m_currentGroup->isReversed())
432 m_currentGroup = NULL;
433}

445void GameWindowTransitionsHandler::setGroup(AsciiString groupName, Bool immidiate )
446{
447 if(groupName.isEmpty() && immidiate)
448 m_currentGroup = NULL;
449 if(immidiate && m_currentGroup)
450 {
451 m_currentGroup->skip();
452 m_currentGroup = findGroup(groupName);
453 if(m_currentGroup)
454 m_currentGroup->init();
455 return;
456 }
457
458 if(m_currentGroup)
459 {
460 if(!m_currentGroup->isFireOnce() && !m_currentGroup->isReversed())
461 m_currentGroup->reverse();
462 m_pendingGroup = findGroup(groupName);
463 if(m_pendingGroup)
464 m_pendingGroup->init();
465 return;
466 }
467
468 m_currentGroup = findGroup(groupName);
469 if(m_currentGroup)
470 m_currentGroup->init();
471
472
473
474}

476void GameWindowTransitionsHandler::reverse( AsciiString groupName )
477{
478 TransitionGroup *g = findGroup(groupName);
479 if( m_currentGroup == g )
480 {
481 m_currentGroup->reverse();
482 return;
483 }
484 if( m_pendingGroup == g)
485 {
486 m_pendingGroup = NULL;
487 return;
488 }
489 if(m_currentGroup)
490 m_currentGroup->skip();
491 if(m_pendingGroup)
492 m_pendingGroup->skip();
493
494 m_currentGroup = g;
495 m_currentGroup->init();
496 m_currentGroup->skip();
497 m_currentGroup->reverse();
498 m_pendingGroup = NULL;
499}

501void GameWindowTransitionsHandler::remove( AsciiString groupName, Bool skipPending )
502{
503 TransitionGroup *g = findGroup(groupName);
504 if(m_pendingGroup == g)
505 {
506 if(skipPending)
507 m_pendingGroup->skip();
508
509 m_pendingGroup = NULL;
510 }
511 if(m_currentGroup == g)
512 {
513 m_currentGroup->skip();
514 m_currentGroup = NULL;
515 if(m_pendingGroup)
516 m_currentGroup = m_pendingGroup;
517 }
518}
