61// StackAllocator
62//
63// Introduction:
64// This templated class allows you to allocate a dynamically sized temporary
65// block of memory local to a function in a way that causes it to use some
66// stack space instead of using the (very slow) malloc function. You don't
67// want to call malloc/new to allocate temporary blocks of memory when speed
68// is important. This class is only practical if you are allocating arrays of
69// objects (especially smaller objects). If you have a class whose size is
70// 40K, you aren't going to be able to put even one of them on the stack.
71// If you don't need an array but instead just need a single object, then you
72// can simply create your object on the stack and be done with it. However,
73// there are some times when you need a temporary array of objects (or just
74// an array of chars or pointers) but don't know the size beforehand. That's
75// where this class is useful.
76//
77// Parameters:
78// The class takes three templated parameters: T, nStackCount, bConstruct.
79// T -- The type of object being allocated. Examples include char,
80// cRZString, int*, cRZRect*.
81// nStackCount -- How many items to allocate on the stack before switching
82// to a heap allocation. Note that if you want to write
83// portable code, you need to use no more than about 2K
84// of stack space. So make sure nStackCount*sizeof(T) < 2048.
85// bConstruct -- This is normally 1, but setting it to 0 is a hint to the
86// compiler that you are constructing a type that has no
87// constructor. Testing showed that the VC++ compiler wasn't
88// completely able to optimize on its own. In particular, it
89// was smart enough to remove the while loop, but not smart
90// enough to remore the pTArrayEnd assignment.
91//
92// Benchmarks:
93// Testing one one machine showed that allocating and freeing a block of
94// 2048 bytes via malloc/free took 8700 and 8400 clock ticks respectively.
95// Using this stack allocator to do the same thing tooko 450 and 375 clock
96// ticks respectively.
97//
98// Usage and examples:
99// Since we are using a local allocator and not overloading global or class
100// operator new, you have to directory call StackAllocator::New instead
101// of operator new. So instead of saying:
102// SomeStruct* pStructArray = new SomeStruct[13];
103// you say:
104// SomeStruct* pStructArray = stackAllocator.New(13);
105//
106// void Example(int nSize){
107// StackAllocator<char, 512> stackAllocator; //Create an instance
108// char* pArray = stackAllocator.New(nSize); //Allocate memory, it will auto-freed.
109// memset(pArray, 0, nSize*sizeof(char)); //Do something with the memory.
110// }
111//
112// void Example(int nSize){
113// StackAllocator<int*, 512, 0> stackAllocator; //Create an instance. We use the 'construct' hint feature here.
114// int** pArray = stackAllocator.New(nSize); //Allocate memory.
115// memset(pArray, 0, nSize*sizeof(int*)); //Do something with the memory.
116// stackAllocator.Delete(pArray); //In this example, we explicity free the memory.
117// }
118//
119// void Example(int nSize){
120// StackAllocator<cRZRect, 200> stackAllocator; //Create an instance
121// cRZRect* pRectArray = stackAllocator.New(nSize); //Allocate memory, it will auto-freed.
122// pRectArray[0].SetRect(0,0,1,1); //Do something with the memory.
123// }
124//
125// void Example(int nSize){
126// StackAllocator<char, 200> stackAllocator; //Create an instance
127// char* pArray = stackAllocator.New(nSize); //Allocate memory.
128// char* pArray2 = stackAllocator.New(nSize); //Allocate some additional memory.
129// memset(pArray, 0, nSize*sizeof(char)); //Do something with the memory.
130// stackAllocator.Delete(pArray); //Delete the memory
131// pArray = stackAllocator.New(nSize*2); //Allocate some new memory. We'll let the allocator free it.
132// memset(pArray, 0, nSize*2*sizeof(char)); //Do something with the new memory.
133// stackAllocator.Delete(pArray2); //Delete the additional memory.
134// }
135//
136template<class T, int nStackCount, int bConstruct=1>
137class StackAllocator{
138public:
139 StackAllocator() : mnAllocCount(-1), mpTHeap(NULL){}
140 ~StackAllocator(){
141 if(mnAllocCount != -1){ //If there is anything to do...
142 if(mpTHeap)
143 delete mpTHeap;
144 else{
145 if(bConstruct){ //Since this constant, the comparison gets optimized away.
146 T* pTArray = (T*)mTArray;
147 const T* const pTArrayEnd = pTArray + mnAllocCount;
148 while(pTArray < pTArrayEnd){
149 pTArray->~T(); //Call the destructor on the object directly.
150 ++pTArray;
151 }
152 }
153 }
154 }
155 }
156
157 T* New(unsigned nCount){
158 if(mnAllocCount == -1){
159 mnAllocCount = nCount;
160 if(nCount < nStackCount){ //If the request is small enough to come from the stack...
161 //We call the constructors of all the objects here.
162 if(bConstruct){ //Since this constant, the comparison gets optimized away.
163 T* pTArray = (T*)mTArray;
164 const T* const pTArrayEnd = pTArray + nCount;
165 while(pTArray < pTArrayEnd){
166 new(pTArray)T; //Use the placement operator new. This simply calls the constructor
167 ++pTArray; //of T with 'this' set to the input address. Note that we don't put
168 } //a '()' after the T this is because () causes trivial types like int
169 } //and class* to be assigned zero/NULL. We don't want that.
170 return (T*)mTArray;
171 } //Else the request is too big. So let's use (the slower) operator new.
172 return (mpTHeap = new T[nCount]); //The compiler will call the constructors here.
173 } //Else we are being used. Let's be nice and allocate something anyway.
174 return new T[nCount];
175 }
176
177 void Delete(T* pT){
178 if(pT == (T*)mTArray){ //If the allocation came from our stack...
179 if(bConstruct){ //Since this constant, the comparison gets optimized away.
180 T* pTArray = (T*)mTArray;
181 const T* const pTArrayEnd = pTArray + mnAllocCount;
182 while(pTArray < pTArrayEnd){
183 pTArray->~T(); //Call the destructor on the object directly.
184 ++pTArray;
185 }
186 }
187 mnAllocCount = -1;
188 }
189 else if(pT == mpTHeap){ //If the allocation came from our heap...
190 delete[] mpTHeap; //The compiler will call the destructors here.
191 mpTHeap = NULL; //We clear these out so that we can possibly
192 mnAllocCount = -1; // use the allocator again.
193 }
194 else //Else the allocation came from the external heap.
195 delete[] pT;
196 }
197
198protected:
199 int mnAllocCount; //Count of objects allocated. -1 means that nothing is allocated. We don't use zero because zero is a legal allocation count in C++.
200 T* mpTHeap; //This is normally NULL, but gets used of the allocation request is too high.
201 char mTArray[nStackCount*sizeof(T)]; //This is our stack memory.
202};

328WWINLINE void FastFixedAllocator::grow()
329{
330 Chunk* n = new Chunk;
331 n->next = chunks;
332 chunks = n;
333 TotalHeapSize+=sizeof(Chunk);
334
335 const int nelem = Chunk::size/esize;
336 char* start = n->mem;
337 char* last = &start[(nelem-1)*esize];
338 for(char* p = start; p<last; p+=esize)
339 reinterpret_cast<Link*>(p)->next = reinterpret_cast<Link*>(p+esize);
340 reinterpret_cast<Link*>(last)->next = 0;
341 head = reinterpret_cast<Link*>(start);
342}

424WWINLINE void* FastAllocatorGeneral::Alloc(unsigned int n)
425{
426 void* pMemory;
427 static int re_entrancy=0;
428 re_entrancy++;
429
430 //We actually allocate n+4 bytes. We store the # allocated
431 //in the first 4 bytes, and return the ptr to the rest back
432 //to the user.
433 n += sizeof(unsigned int);
434#ifdef MEMORY_OVERWRITE_TEST
435 n+=sizeof(unsigned int);
436#endif
437
438 if (re_entrancy==1) {
439 ActualMemoryUsage+=n;
440 }
441 if (n<MAX_ALLOC_SIZE) {
442 int index=(n)/ALLOC_STEP;
443 {
444 FastCriticalSectionClass::LockClass lock(CriticalSections[index]);
445 pMemory = allocators[index].Alloc();
446 }
447 }
448 else {
449 if (re_entrancy==1) {
450 AllocatedWithMalloc+=n;
451 AllocatedWithMallocCount++;
452 }
453 pMemory = ::malloc(n);
454 }
455#ifdef MEMORY_OVERWRITE_TEST
456 *((unsigned int*)((char*)pMemory+n)-1)=0xabbac0de;
457#endif
458
459 re_entrancy--;
460 *((unsigned int*)pMemory) = n; //Write modified (augmented by 4) count into first four bytes.
461 return ((unsigned int*)pMemory)+1; //return ptr to bytes after it back to user.
462}

470WWINLINE void FastAllocatorGeneral::Free(void* pAlloc)
471{
472 if (pAlloc) {
473 unsigned int* n = ((unsigned int*)pAlloc)-1; //Subtract four bytes and the count is stored there.
474
475#ifdef MEMORY_OVERWRITE_TEST
476 WWASSERT(*((unsigned int*)((char*)n+*n)-1)==0xabbac0de);
477#endif
478
479 unsigned size=*n;
480 ActualMemoryUsage-=size;
481
482 if (size<MAX_ALLOC_SIZE) {
483 int index=size/ALLOC_STEP;
484 FastCriticalSectionClass::LockClass lock(CriticalSections[index]);
485 allocators[index].Free(n);
486 }
487 else {
488 AllocatedWithMallocCount--;
489 AllocatedWithMalloc-=size;
490 ::free(n);
491 }
492 }
493}

495//ANSI C requires:
496// (1) realloc(NULL, newsize) is equivalent to malloc(newsize).
497// (2) realloc(pblock, 0) is equivalent to free(pblock) (except that NULL is returned).
498// (3) if the realloc() fails, the object pointed to by pblock is left unchanged.
499//
500WWINLINE void* FastAllocatorGeneral::Realloc(void* pAlloc, unsigned int n){
501 if(n){
502 void* const pNewAlloc = Alloc(n); //Allocate the new memory. This never fails.
503 if(pAlloc){
504 n = *(((unsigned int*)pAlloc)-1); //Subtract four bytes and the count is stored there.
505 ::memcpy(pNewAlloc, pAlloc, n); //Copy the old memory into the new memory.
506 Free(pAlloc); //Delete the old memory.
507 }
508 return pNewAlloc;
509 }
510 Free(pAlloc);
511 return NULL;
512}

521// FastSTLAllocator
522//
523// An STL allocator based on a simple fixed size allocator is not going to
524// buy you the performance you really want. This is because STL containers
525// allocate not just objects of the size of the container, but other objects
526// as well. For example, the STL "list" class generally allocates item nodes
527// and not items themselves. The STL vector class usually allocates in chunks
528// of contiguous items. So your allocator will probably want to have a bucket
529// system whereby it maintains buckets for integral sizes.
530//
531
532#ifdef _MSC_VER
533 //VC++ continues to be the one compiler that lacks the ability to compile
534 //standard C++. So we define a version of the STL allocator specifically
535 //for VC++, and let other compilers use a standard allocator template.
536 template <class T>
537 struct FastSTLAllocator{
538 typedef size_t size_type; //basically, "unsigned int"
539 typedef ptrdiff_t difference_type; //basically, "int"
540 typedef T* pointer;
541 typedef const T* const_pointer;
542 typedef T& reference;
543 typedef const T& const_reference;
544 typedef T value_type;
545
546 T* address(T& t) const { return (&t); } //These two are slightly strange but
547 const T* address(const T& t) const { return (&t); } //required functions. Just do it.
548 static T* allocate(size_t n, const void* =NULL) { return (T*)FastAllocatorGeneral::Get_Allocator()->Alloc(n*sizeof(T)); }
549 static void construct(T* ptr, const T& value) { new(ptr) T(value); }
550 static void deallocate(void* ptr, size_t /*n*/) { FastAllocatorGeneral::Get_Allocator()->Free(ptr); }
551 static void destroy(T* ptr) { ptr->~T(); }
552 static size_t max_size() { return (size_t)-1; }
553
554 //This _Charalloc is required by VC++5 since it VC++5 predates
555 //the language standardization. Allocator behaviour is one of the
556 //last things to have been hammered out. Important note: If you
557 //decide to write your own fast allocator, containers will allocate
558 //random objects through this function but delete them through
559 //the above delallocate() function. So your version of deallocate
560 //should *not* assume that it will only be given T objects to delete.
561 char* _Charalloc(size_t n){ return (char*)FastAllocatorGeneral::Get_Allocator()->Alloc(n*sizeof(char)); }
562 };
563#else
564 //This is a C++ language standard allocator. Most C++ compilers after 1999
565 //other than Microsoft C++ compile this fine. Otherwise. you might be able
566 //to use the same allocator as VC++ uses above.

572 typedef T* pointer;
573 typedef const T* const_pointer;
574 typedef T& reference;
