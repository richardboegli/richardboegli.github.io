125void FeatherOptions::OnChangeSizeEdit()
126{
127 if (m_updating) return;
128 CWnd *pEdit = m_staticThis->GetDlgItem(IDC_SIZE_EDIT);
129 char buffer[_MAX_PATH];
130 if (pEdit) {
131 pEdit->GetWindowText(buffer, sizeof(buffer));
132 Int width;
133 m_updating = true;
134 if (1==sscanf(buffer, "%d", &width)) {
135 m_currentFeather = width;
136 FeatherTool::setFeather(m_currentFeather);
137 sprintf(buffer, "%.1f FEET.", m_currentFeather*MAP_XY_FACTOR);
138 pEdit = m_staticThis->GetDlgItem(IDC_WIDTH_LABEL);
139 if (pEdit) pEdit->SetWindowText(buffer);
140 }
141 m_updating = false;
142 }
143}

178void FeatherOptions::PopSliderChanged(const long sliderID, long theVal)
179{
180 CString str;
181 CWnd *pEdit;
182 switch (sliderID) {
183
184 case IDC_SIZE_POPUP:
185 m_currentFeather = theVal;
186 str.Format("%d",m_currentFeather);
187 pEdit = m_staticThis->GetDlgItem(IDC_SIZE_EDIT);
188 if (pEdit) pEdit->SetWindowText(str);
189 FeatherTool::setFeather(m_currentFeather);
190 break;
191
192 case IDC_RADIUS_POPUP:
193 m_currentRadius = theVal;
194 str.Format("%d",m_currentRadius);
195 pEdit = m_staticThis->GetDlgItem(IDC_RADIUS_EDIT);
196 if (pEdit) pEdit->SetWindowText(str);
197 FeatherTool::setRadius(m_currentRadius);
198 break;
199
200 case IDC_RATE_POPUP:
201 m_currentRate = theVal;
202 str.Format("%d",m_currentRate);
203 pEdit = m_staticThis->GetDlgItem(IDC_RATE_EDIT);
204 if (pEdit) pEdit->SetWindowText(str);
205 FeatherTool::setRate(m_currentRate);
206 break;
207
