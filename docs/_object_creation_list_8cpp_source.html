280 virtual Object* create(const Object* primaryObj, const Coord3D *primary, const Coord3D* secondary, Bool createOwner, UnsignedInt lifetimeFrames = 0 ) const
281 {
282 if (!primaryObj || !primary || !secondary)
283 {
284 DEBUG_CRASH(("You must have a primary and secondary source for this effect"));
285 return NULL;
286 }
287
288 Team* owner = primaryObj ? primaryObj->getControllingPlayer()->getDefaultTeam() : NULL;
289
290
291 //What I'm doing for the purposes of the formations is to calculate the relative positions of
292 //each member of the formation. To do so, we take the vector from the target location to the
293 //lead plane location, normalize it, then rotate it 90 degrees (CW and CCW). When we add the
294 //resultant vectors to the initial vectors, we can calculate the delta positions for each plane.
295 Real CCWx = 0.0f, CCWy = 0.0f, CWx = 0.0f, CWy = 0.0f;
296
297 if( m_formationSize > 1 )
298 {
299 //Get the delta x and y values from the target to the origin.
300 Real dx = primary->x - secondary->x;
301 Real dy = primary->y - secondary->y;
302
303 //Calc length
304 Real length = sqrt( dx*dx + dy*dy );
305
306 //Normalize length
307 dx /= length;
308 dy /= length;
309
310 //Rotate 90 degrees CCW.
311 Real radians = 90.0f * PI / 180.0f;
312 Real s = Sin( radians );
313 Real c = Cos( radians );
314 CCWx = dx * c + dy * -s + dx;
315 CCWy = dx * s + dy * c + dy;
316
317 //Rotate 90 degrees CW
318 s = Sin( -radians );
319 c = Cos( -radians );
320 CWx = dx * c + dy * -s + dx;
321 CWy = dx * s + dy * c + dy;
322 }
323
324 Object *firstTransport = NULL;
325 for( Int formationIndex = 0; formationIndex < m_formationSize; formationIndex++ )
326 {
327 Coord3D offset;
328 offset.zero();
329
330 Int offsetMultiplier = ( formationIndex + 1 ) / 2 * m_formationSpacing;
331
332 if( formationIndex % 2 )
333 {
334 //Formation index is odd -- use the CCW deltas
335 offset.x = CCWx * offsetMultiplier;
336 offset.y = CCWy * offsetMultiplier;
337 }
338 else
339 {
340 //Formation index is even -- use the CW deltas
341 offset.x = CWx * offsetMultiplier;
342 offset.y = CWy * offsetMultiplier;
343 }
344
345 Coord3D startPos = *primary;
346 Coord3D moveToPos = *secondary;
347 startPos.add( &offset );
348 //Also give our moveToPos the same offset to maintain perfect formation.
349 moveToPos.add( &offset );
350
351 Coord3D targetPos = *secondary;
352
353
354 //Our target position only applies when using fireweapon and when we have multiple planes,
355 //as is the case with the napalm strike. The target position either be somewhere between the
356 //moveToPos of the lead plane and that of the relative offset -- determined by the convergenceFactor.
357 targetPos.x += offset.x * (1.0f - m_convergenceFactor);
358 targetPos.y += offset.y * (1.0f - m_convergenceFactor);
359
360
361 // first guy in each formation is always spot-on (to keep targeting cursor well-matched)
362 if ( m_errorRadius > 1.0f && formationIndex > 0 )
363 {
364 Real randomRadius = GameLogicRandomValueReal(0, m_errorRadius );
365 Real randomAngle = GameLogicRandomValueReal(0, PI*2 );
366 targetPos.x += randomRadius * Cos( randomAngle );
367 targetPos.y += randomRadius * Sin( randomAngle );
368 }
369
370
371 Real orient = atan2( moveToPos.y - startPos.y, moveToPos.x - startPos.x);
372 if( m_data.m_distToTarget > 0 )
373 {
374 const Real SLOP = 1.5f;
375 startPos.x -= Cos(orient) * m_data.m_distToTarget * SLOP;
376 startPos.y -= Sin(orient) * m_data.m_distToTarget * SLOP;
377 }
378
379 Object *transport;
380
381 if( createOwner )
382 {
383 const ThingTemplate* ttn = TheThingFactory->findTemplate(m_transportName);
384 transport = TheThingFactory->newObject( ttn, owner );
385 if( !transport )
386 {
387 return NULL;
388 }
389 if( !firstTransport )
390 {
391 firstTransport = transport;
392 }
393 transport->setPosition(&startPos);
394 transport->setOrientation(orient);
395 transport->setProducer(primaryObj);
396 //Adding this nifty flag allows enemy players to target it manually with weapons :)
397 transport->setScriptStatus( OBJECT_STATUS_SCRIPT_TARGETABLE );
398
399 if ( m_delayDeliveryFramesMax > 0 )
400 {
401 transport->setDisabledUntil( DISABLED_DEFAULT, TheGameLogic->getFrame() + GameLogicRandomValue(0, m_delayDeliveryFramesMax) );
402 }
403 }
404 else
405 {
406 transport = (Object*)primaryObj;
407 }
408
409 // Notify special power tracking
410 SpecialPowerCompletionDie *die = transport->findSpecialPowerCompletionDie();
411 if (die)
412 {
413 if (formationIndex == 0)
414 die->setCreator(primaryObj->getID());
415 else
416 die->setCreator(INVALID_ID);
417 }
418
419 static NameKeyType key_DeliverPayloadAIUpdate = NAMEKEY("DeliverPayloadAIUpdate");
420 DeliverPayloadAIUpdate *ai = (DeliverPayloadAIUpdate*)transport->findUpdateModule(key_DeliverPayloadAIUpdate);
421 if( ai )
422 {
423 if( m_startAtMaxSpeed && createOwner )
424 {
425 PhysicsBehavior* physics = transport->getPhysics();
426 if (physics)
427 {
428 Coord3D startingForce = *transport->getUnitDirectionVector2D();
429 Real maxSpeed = ai->getCurLocomotor()->getMaxSpeedForCondition(transport->getBodyModule()->getDamageState());
430 Real factor = maxSpeed * physics->getMass();
431 startingForce.x *= factor;
432 startingForce.y *= factor;
433 startingForce.z *= factor;
434 physics->applyMotiveForce( &startingForce );
435 }
436 }
437
438 // only the first guy in each formation gets a delivery decal
439 DeliverPayloadData data = m_data;
440 if (formationIndex != 0)
441 data.m_deliveryDecalRadius = 0;
442 ai->deliverPayload( &moveToPos, &targetPos, &data );
443 if( m_startAtPreferredHeight && createOwner )
444 {
445 startPos.z = TheTerrainLogic->getGroundHeight(startPos.x, startPos.y) + ai->getCurLocomotor()->getPreferredHeight();
446 transport->setPosition(&startPos);
447 }
448
449 const ThingTemplate* putInContainerTmpl = m_putInContainerName.isEmpty() ? NULL : TheThingFactory->findTemplate(m_putInContainerName);
450 for (std::vector<Payload>::const_iterator it = m_payload.begin(); it != m_payload.end(); ++it)
451 {
452 const ThingTemplate* payloadTmpl = TheThingFactory->findTemplate(it->m_payloadName);
453 if( !payloadTmpl )
454 {
455 DEBUG_CRASH( ("DeliverPayloadNugget::create() -- %s couldn't create %s (template not found).",
456 transport->getTemplate()->getName().str(), it->m_payloadName.str() ) );
457 return NULL;
458 }
459 for (int i = 0; i < it->m_payloadCount; ++i)
460 {
461 Object* payload = TheThingFactory->newObject( payloadTmpl, owner );
462 payload->setPosition(&startPos);
463 payload->setProducer(transport);
464
465 // Notify special power tracking
466 SpecialPowerCompletionDie *die = payload->findSpecialPowerCompletionDie();
467 if (die)
468 {
469 if (formationIndex == 0 && i == 0)
470 die->setCreator(primaryObj->getID());
471 else
472 die->setCreator(INVALID_ID);
473 }
474
475 if (putInContainerTmpl)
476 {
477 Object* container = TheThingFactory->newObject( putInContainerTmpl, owner );
478 container->setPosition(&startPos);
479 container->setProducer(transport);
480
481 // Notify special power tracking
482 SpecialPowerCompletionDie *die = container->findSpecialPowerCompletionDie();
483 if (die)
484 {
485 if (formationIndex == 0 && i == 0)
486 die->setCreator(primaryObj->getID());
487 else
488 die->setCreator(INVALID_ID);
489 }
490
491 if (container->getContain() && container->getContain()->isValidContainerFor(payload, true))
492 {
493 container->getContain()->addToContain(payload);
494 payload = container;
495 }
496 else
497 {
498 DEBUG_CRASH(("DeliverPayload: PutInContainer %s is full, or not valid for the payload %s!",m_putInContainerName.str(),it->m_payloadName.str()));
499 }
500 }
501
502 if (transport->getContain() && transport->getContain()->isValidContainerFor(payload, true))
503 {
504 transport->getContain()->addToContain(payload);
505 }
506 else
507 {
508 DEBUG_CRASH(("DeliverPayload: transport %s is full, or not valid for the payload %s!",m_transportName.str(),it->m_payloadName.str()));
509 }
510 }
511 }
512 }
513 else
514 {
515 DEBUG_CRASH(("You should really have a DeliverPayloadAIUpdate here"));
516 }
517 }
518 return firstTransport;
519 }

599static void calcRandomForce(Real minMag, Real maxMag, Real minPitch, Real maxPitch, Coord3D* force)
600{
601 Real angle = GameLogicRandomValueReal(0, 2*PI);
602 Real pitch = GameLogicRandomValueReal(minPitch, maxPitch);
603 Real mag = GameLogicRandomValueReal(minMag, maxMag);
604
605 Matrix3D mtx(1);
606 mtx.Scale(mag);
607 mtx.Rotate_Z(angle);
608 mtx.Rotate_Y(-pitch);
609
610 Vector3 v = mtx.Get_X_Vector();
611
612 force->x = v.X;
613 force->y = v.Y;
614 force->z = v.Z;
615}
