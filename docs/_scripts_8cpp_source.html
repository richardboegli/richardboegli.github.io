249void ScriptList::xfer( Xfer *xfer )
250{
251 UnsignedShort countVerify;
252
253 // version
254 XferVersion currentVersion = 1;
255 XferVersion version = currentVersion;
256 xfer->xferVersion( &version, currentVersion );
257
258 // count of scripts here
259 Script *script;
260 UnsignedShort scriptCount = 0;
261 for( script = getScript(); script; script = script->getNext() )
262 scriptCount++;
263 countVerify = scriptCount;
264 xfer->xferUnsignedShort( &scriptCount );
265 if( countVerify != scriptCount )
266 {
267
268 DEBUG_CRASH(( "ScriptList::xfer - Script list count has changed, attempting to recover."));
269 // throw SC_INVALID_DATA; try to recover. jba.
270
271 } // end if
272
273 // all script data here
274 for( script = getScript(); script; script = script->getNext() ) {
275 xfer->xferSnapshot( script );
276 scriptCount--;
277 if (scriptCount==0) break;
278 }
279 if (scriptCount>0) {
280 DEBUG_CRASH(("Stripping out extra scripts - Bad..."));
281 if (s_mtScript==NULL) s_mtScript = newInstance(Script); // Yes it leaks, but this is unusual recovery only. jba.
282 while (scriptCount) {
283 xfer->xferSnapshot(s_mtScript);
284 scriptCount--;
285 }
286 }
287
288 // count of script groups
289 ScriptGroup *scriptGroup;
290 UnsignedShort scriptGroupCount = 0;
291 for( scriptGroup = getScriptGroup(); scriptGroup; scriptGroup = scriptGroup->getNext() )
292 scriptGroupCount++;
293 countVerify = scriptGroupCount;
294 xfer->xferUnsignedShort( &scriptGroupCount );
295 if( countVerify != scriptGroupCount )
296 {
297
298 DEBUG_CRASH(( "ScriptList::xfer - Script group count has changed, attempting to recover."));
299
300 } // end if
301
302 // all script group data
303 for( scriptGroup = getScriptGroup(); scriptGroup; scriptGroup = scriptGroup->getNext() ) {
304 xfer->xferSnapshot( scriptGroup );
305 scriptGroupCount--;
306 if (scriptGroupCount==0) break;
307 }
308 if (scriptGroupCount>0) {
309 DEBUG_CRASH(("Stripping out extra groups. - Bad..."));
310 if (s_mtGroup == NULL) s_mtGroup = newInstance(ScriptGroup); // Yes it leaks, but this is only for recovery.
311 while (scriptGroupCount) {
312 xfer->xferSnapshot(s_mtGroup);
313 scriptGroupCount--;
314 }
315 }

430void ScriptList::addGroup(ScriptGroup *pGrp, Int ndx)
431{
432 ScriptGroup *pPrev = NULL;
433 ScriptGroup *pCur = m_firstGroup;
434 DEBUG_ASSERTCRASH(pGrp->getNext()==NULL, ("Adding already linked group."));
435 while (ndx && pCur) {
436 pPrev = pCur;
437 pCur = pCur->getNext();
438 ndx--;
439 }
440 if (pPrev) {
441 // Weave into prev link.
442 pGrp->setNextGroup(pPrev->getNext());
443 pPrev->setNextGroup(pGrp);
444 } else {
445 // Goes at head of list.
446 pGrp->setNextGroup(m_firstGroup);
447 m_firstGroup = pGrp;
448 }

531Bool ScriptList::ParseScriptsDataChunk(DataChunkInput &file, DataChunkInfo *info, void *userData)
532{
533 Int i;
534 file.registerParser( AsciiString("ScriptList"), info->label, ScriptList::ParseScriptListDataChunk );
535 DEBUG_ASSERTCRASH(s_numInReadList==0, ("Leftover scripts floating aroung."));
536 for (i=0; i<s_numInReadList; i++) {
537 if (s_readLists[i]) {
538 s_readLists[i]->deleteInstance();
539 s_readLists[i] = NULL;
540 }
541 }
542 TScriptListReadInfo readInfo;
543 for (i=0; i<MAX_PLAYER_COUNT; i++) {
544 readInfo.readLists[i] = 0;
545 }
546 readInfo.numLists = 0;
547 if (file.parse(&readInfo)) {
548 DEBUG_ASSERTCRASH(readInfo.numLists<MAX_PLAYER_COUNT, ("Read too many, overrun buffer."));
549 s_numInReadList = readInfo.numLists;
550 for (i=0; i<s_numInReadList; i++) {
551 s_readLists[i] = readInfo.readLists[i];
552 }
553 return true;
554 }
555 return false;
556}
