380Int ScreenBWFilter::set(enum FilterModes mode)
381{
382 HRESULT hr;
383
384 if (mode > FM_NULL_MODE)
385 { //rendering a quad with redirected rendering surface tinted by pixel shader
386
387 if (m_fadeDirection > 0)
388 { //turning effect on
389 m_curFadeFrame++;
390 Int fade = m_curFadeFrame;
391
392 if (fade<m_fadeFrames)
393 {
394 m_curFadeValue = (Real)fade/(Real)m_fadeFrames;
395 }
396 else
397 {
398 m_curFadeFrame = 0;
399 m_curFadeValue = 1.0f;
400 m_fadeDirection = 0;
401 }
402 }
403 else
404 if (m_fadeDirection < 0)
405 { //turning effect off
406 m_curFadeFrame++;
407 Int fade = m_curFadeFrame;
408 if (fade<m_fadeFrames)
409 {
410 m_curFadeValue = 1.0f - (Real)fade/(Real)m_fadeFrames;
411 }
412 else
413 { m_curFadeValue = 0.0f;
414 TheTacticalView->setViewFilterMode(FM_NULL_MODE);
415 TheTacticalView->setViewFilter(FT_NULL_FILTER);
416 m_curFadeFrame = 0;
417 m_fadeDirection = 0;
418 }
419 }
420
421 VertexMaterialClass *vmat=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
422 DX8Wrapper::Set_Material(vmat);
423 REF_PTR_RELEASE(vmat); //no need to keep a reference since it's a preset.
424 DX8Wrapper::Set_Shader(ShaderClass::_PresetOpaqueShader);
425 DX8Wrapper::Set_Texture(0,NULL);
426 DX8Wrapper::Apply_Render_State_Changes(); //force update of view and projection matrices
427
428 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZFUNC,D3DCMP_ALWAYS);
429 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZWRITEENABLE,FALSE);
430 DX8Wrapper::Apply_Render_State_Changes(); //force update of view and projection matrices
431
432 hr=DX8Wrapper::_Get_D3D_Device8()->SetPixelShader(m_dwBWPixelShader);
433 DX8Wrapper::_Get_D3D_Device8()->SetPixelShaderConstant(0, D3DXVECTOR4(0.3f, 0.59f, 0.11f, 1.0f), 1);
434
435 D3DXVECTOR4 color(1.0f,1.0f,1.0f,1.0f); //multiply color
436
437 if (mode == FM_VIEW_BW_BLACK_AND_WHITE)
438 { //back & white mode
439 color.x=1.0f;
440 color.y=1.0f;
441 color.z=1.0f;
442 }
443 if (mode == FM_VIEW_BW_RED_AND_WHITE)
444 { //red is on
445 color.x = 1.0f;
446 color.y = 0.0f;
447 color.z = 0.0f;
448 //inverse red is on
449 //red is on
450// color.x = 0.0f;
451// color.y = 1.0f;
452// color.z = 1.0f;
453 }
454 if (mode == FM_VIEW_BW_GREEN_AND_WHITE)
455 {
456 color.x = 0.0f;
457 color.y = 1.0f;
458 color.z = 0.0f;
459 }

3153{
3154 float ztot, yran, ymult, ymod, x, y, z, pi, prod;
3155 long int low, ixran, itot, j, iprod;
3156
3157 __int64 endTime64,freq64,startTime64;
3158 QueryPerformanceFrequency((LARGE_INTEGER *)&freq64);
3159 QueryPerformanceCounter((LARGE_INTEGER *)&startTime64);
3160
3161 ztot = 0.0;
3162 low = 1;
3163 ixran = 1907;
3164 yran = 5813.0;
3165 ymult = 1307.0;
3166 ymod = 5471.0;
3167 itot = 560000; //total iterations. This value ends up running at ~30 fps on our P4-2.2Ghz.
3168
3169 for(j=1; j<=itot; j++)
3170 {
3171 iprod = 27611 * ixran;
3172 ixran = iprod - 74383*(long int)(iprod/74383);
3173 x = (float)ixran / 74383.0;
3174 prod = ymult * yran;
3175 yran = (prod - ymod*(long int)(prod/ymod));
3176 y = yran / ymod;
3177 z = x*x + y*y;
3178 add(&ztot,&z);
3179 if ( z <= 1.0 )
3180 {
3181 low = low + 1;
3182 }
3183 }
3184 pi = 4.0 * (float)low/(float)itot;
3185
3186 QueryPerformanceCounter((LARGE_INTEGER *)&endTime64);
3187 return ((double)(endTime64-startTime64)/(double)(freq64));
