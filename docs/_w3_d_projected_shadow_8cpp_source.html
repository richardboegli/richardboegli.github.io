352Int W3DProjectedShadowManager::renderProjectedTerrainShadow(W3DProjectedShadow *shadow, AABoxClass &box)
353{
354 static Matrix4x4 mWorld(true); //initialize to identity matrix
355 struct SHADOW_VOLUME_VERTEX //vertex structure passed to D3D
356 {
357 float x,y,z;
358 };
359
360 Int i,j,k;
361 UnsignedByte alpha[4];
362 float UA[4], VA[4];
363 Bool flipForBlend;
364
365
366 #define SHADOW_VOLUME_FVF D3DFVF_XYZ
367
368 if (TheTerrainRenderObject)
369 {
370 WorldHeightMap *hmap=TheTerrainRenderObject->getMap();
371
372 //Find size of heightmap sub-rectangle affected by shadow
373 Real cx=box.Center.X;
374 Real cy=box.Center.Y;
375 Real dx=box.Extent.X;
376 Real dy=box.Extent.Y;
377 Real mapScaleInv=1.0f/MAP_XY_FACTOR;
378 SHADOW_VOLUME_VERTEX* pvVertices;
379 UnsignedShort *pvIndices;
380 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
381
382 if (!m_pDev) return 0;
383
384 //Get terrain cell index for area with shadow
385 Int startX=REAL_TO_INT_FLOOR(((cx - dx)*mapScaleInv));
386 Int endX=REAL_TO_INT_CEIL(((cx + dx)*mapScaleInv));
387 Int startY=REAL_TO_INT_FLOOR(((cy - dy)*mapScaleInv));
388 Int endY=REAL_TO_INT_CEIL(((cy + dy)*mapScaleInv));
389
390 //clip bounds to extents of heightmap
391 startX = __max(startX,0);
392 endX = __min(endX,hmap->getXExtent()-1);
393 startY = __max(startY,0);
394 endY = __min(endY,hmap->getYExtent()-1);
395
396 Int vertsPerRow=endX - startX+1; //number of cells +1
397 Int vertsPerColumn=endY-startY+1; //number of cells +1
398
399 if (vertsPerRow == 1 || vertsPerColumn == 1)
400 return 0; //nothing to render
401
402 Int numVerts = vertsPerRow *vertsPerColumn; //number of terrain vertices
403
404 if (nShadowVertsInBuf > (SHADOW_VERTEX_SIZE-numVerts)) //check if room for model verts
405 { //flush the buffer by drawing the contents and re-locking again
406 if (shadowVertexBufferD3D->Lock(0,numVerts*sizeof(SHADOW_VOLUME_VERTEX),(unsigned char**)&pvVertices,D3DLOCK_DISCARD) != D3D_OK)
407 return 0;
408 nShadowVertsInBuf=0;
409 nShadowStartBatchVertex=0;
410 }
411 else
412 { if (shadowVertexBufferD3D->Lock(nShadowVertsInBuf*sizeof(SHADOW_VOLUME_VERTEX),numVerts*sizeof(SHADOW_VOLUME_VERTEX), (unsigned char**)&pvVertices,D3DLOCK_NOOVERWRITE) != D3D_OK)
413 return 0;
414 }
415
416 if(pvVertices)
417 {
418 //insert each cell's bottom/left edge vertex
419 for (j=startY; j <= endY; j++)
420 {
421 float ycoord = (float)j * MAP_XY_FACTOR;
422
423 for (i=startX; i <= endX; i++)
424 {
425 pvVertices->x=(float)i*MAP_XY_FACTOR;
426 pvVertices->y=ycoord;
427 pvVertices->z=(float)hmap->getHeight(i,j)*MAP_HEIGHT_SCALE;
428 pvVertices++;
429 }
430 }
431 }
432
433 shadowVertexBufferD3D->Unlock();
434
435 Int numIndex=(endX - startX) * (endY-startY)*6; //6 indices per terrain cell (2 triangles).
436
437 if (nShadowIndicesInBuf > (SHADOW_INDEX_SIZE-numIndex)) //check if room for model verts
438 { //flush the buffer by drawing the contents and re-locking again
439 if (shadowIndexBufferD3D->Lock(0,numIndex*sizeof(short),(unsigned char**)&pvIndices,D3DLOCK_DISCARD) != D3D_OK)
440 return 0;
441 nShadowIndicesInBuf=0;
442 nShadowStartBatchIndex=0;
443 }
444 else
445 { if (shadowIndexBufferD3D->Lock(nShadowIndicesInBuf*sizeof(short),numIndex*sizeof(short), (unsigned char**)&pvIndices,D3DLOCK_NOOVERWRITE) != D3D_OK)
446 return 0;
447 }
448
449 if(pvIndices)
450 { //fill each cell's vertex indices
451 Int rowStart;
452 for (j=startY,rowStart=0; j<endY; j++,rowStart+=vertsPerRow)
453 {
454 for (i=rowStart,k=startX; k<endX; i++,k++)
455 {
456 hmap->getAlphaUVData(k, j, UA, VA, alpha, &flipForBlend, false);
457/* if (flipForBlend)
458 { pvIndices[0]=i;
459 pvIndices[1]=i+1;
460 pvIndices[2]=i+vertsPerRow;
461 pvIndices[3]=i+vertsPerRow;
462 pvIndices[4]=i+1;
463 pvIndices[5]=i+vertsPerRow+1;
464 }
465 else
466 { pvIndices[0]=i+vertsPerRow;
467 pvIndices[4]=pvIndices[1]=i;
468 pvIndices[3]=pvIndices[2]=i+vertsPerRow+1;
469 pvIndices[5]=i+1;
470 }*/
472 if (flipForBlend)
473 { pvIndices[0]=i+1;
474 pvIndices[1]=i+vertsPerRow;
475 pvIndices[2]=i;
476 pvIndices[3]=i+1;
477 pvIndices[4]=i+1+vertsPerRow;
478 pvIndices[5]=i+vertsPerRow;
479 }
480 else
481 { pvIndices[0]=i;
482 pvIndices[1]=i+1+vertsPerRow;
483 pvIndices[2]=i+vertsPerRow;
484 pvIndices[3]=i;
485 pvIndices[4]=i+1;
486 pvIndices[5]=i+1+vertsPerRow;
487 }
488 pvIndices += 6;
489 }
490 }
491 }
492
493 shadowIndexBufferD3D->Unlock();
494
495 m_pDev->SetIndices(shadowIndexBufferD3D,nShadowStartBatchVertex);
496
497 m_pDev->SetTransform(D3DTS_WORLD,(_D3DMATRIX *)&mWorld);
498
499 m_pDev->SetStreamSource(0,shadowVertexBufferD3D,sizeof(SHADOW_VOLUME_VERTEX));
500 m_pDev->SetVertexShader(SHADOW_VOLUME_FVF);
501
502 Int numPolys = (endX - startX)*(endY - startY)*2; //2 triangles per cell
503
504 m_pDev->SetRenderState(D3DRS_ALPHATESTENABLE, TRUE); //should reject background pixels
505 m_pDev->SetRenderState( D3DRS_STENCILENABLE, TRUE );
506 m_pDev->SetRenderState( D3DRS_STENCILFUNC, D3DCMP_ALWAYS );
507 m_pDev->SetRenderState( D3DRS_STENCILREF, 0x1 );
508 m_pDev->SetRenderState( D3DRS_STENCILMASK, 0xffffffff );
509 m_pDev->SetRenderState( D3DRS_STENCILWRITEMASK,0xffffffff );
510 m_pDev->SetRenderState( D3DRS_STENCILZFAIL, D3DSTENCILOP_KEEP );
511 m_pDev->SetRenderState( D3DRS_STENCILFAIL, D3DSTENCILOP_KEEP );
512 m_pDev->SetRenderState( D3DRS_STENCILPASS, D3DSTENCILOP_INCR );
513
514// m_pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, FALSE ); //useful to see bounds
515 m_pDev->SetRenderState( D3DRS_LIGHTING, FALSE);
516 m_pDev->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_DESTCOLOR);
517 m_pDev->SetRenderState( D3DRS_DESTBLEND, D3DBLEND_ZERO );
518
519
520 if (DX8Wrapper::_Is_Triangle_Draw_Enabled())
521 {
522 Debug_Statistics::Record_DX8_Polys_And_Vertices(numPolys,numVerts,ShaderClass::_PresetOpaqueShader);
523 m_pDev->DrawIndexedPrimitive(D3DPT_TRIANGLELIST,0,numVerts,nShadowStartBatchIndex,numPolys);
524 }
525
526 m_pDev->SetRenderState(D3DRS_ALPHATESTENABLE, FALSE); //should reject background pixels
527 m_pDev->SetRenderState( D3DRS_STENCILENABLE, FALSE );
528// m_pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, TRUE );
529 m_pDev->SetRenderState( D3DRS_LIGHTING, TRUE);
530
531 nShadowVertsInBuf += numVerts;
532 nShadowStartBatchVertex=nShadowVertsInBuf;
533
534 nShadowIndicesInBuf += numIndex;
535 nShadowStartBatchIndex=nShadowIndicesInBuf;
536 return 1;
537 }
538 return 0;
539}

579static void RenderVBTile(TextureClass *text, Real ox, Real oy, Real ou, Real ov, BlendDirection bd=B_A)
580{
581 DynamicVBAccessClass vb_access(BUFFER_TYPE_DYNAMIC_DX8,DX8_FVF_XYZNDUV2,4);
582 DynamicIBAccessClass ib_access(BUFFER_TYPE_DYNAMIC_DX8,6);
583
584 DynamicVBAccessClass::WriteLockClass lock(&vb_access);
585 VertexFormatXYZNDUV2* vb= lock.Get_Formatted_Vertex_Array();
586 DynamicIBAccessClass::WriteLockClass lockib(&ib_access);
587 if (!vb) return;
588
589 UnsignedShort *ib=lockib.Get_Index_Array();
590
591 vb->x=ox;
592 vb->y=oy;
593 vb->z=TILE_HEIGHT;
594 vb->diffuse=TILE_DIFFUSE|BDToVA[bd][0];
595 vb->u1=ou;
596 vb->v1=ov;
597 vb++;
598
599 vb->x=ox;
600 vb->y=oy-10.0f;
601 vb->z=TILE_HEIGHT;
602 vb->diffuse=TILE_DIFFUSE|BDToVA[bd][1];
603 vb->u1=ou;
604 vb->v1=ov+0.25f;
605 vb++;
606
607 vb->x=ox+10.0f;
608 vb->y=oy-10.0f;
609 vb->z=TILE_HEIGHT;
610 vb->diffuse=TILE_DIFFUSE|BDToVA[bd][2];
611 vb->u1=ou+0.25f;
612 vb->v1=ov+0.25f;
613 vb++;
614
615 vb->x=ox+10.0f;
616 vb->y=oy;
617 vb->z=TILE_HEIGHT;
618 vb->diffuse=TILE_DIFFUSE|BDToVA[bd][3];
619 vb->u1=ou+0.25f;
620 vb->v1=ov;
621 vb++;
622
623 ib[0]=0;
624 ib[1]=1;
625 ib[2]=3;
626 ib[3]=3;
627 ib[4]=1;
628 ib[5]=2;
629
630 if (bd == B_TR || bd == B_BL)
631 { //need to flip triangles so alpha gradient doesn't follow diagonal edge
632 ib[2]=2;
633 ib[4]=0;
634 }
635
636 DX8Wrapper::Set_Index_Buffer(ib_access,0);

1154void W3DProjectedShadowManager::queueSimpleDecal(W3DProjectedShadow *shadow)
1155{
1156 Vector3 objPos;
1157 Matrix3D objXform;
1158 Vector3 uVector,vVector;
1159 Coord3D normal;
1160
1161 if (TheTerrainRenderObject)
1162 {
1163 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
1164
1165 if (!m_pDev) return; //no D3D Device to render
1166
1167 objPos=shadow->m_robj->Get_Position();
1168 objXform=shadow->m_robj->Get_Transform();
1169 Real groundHeight=TheTerrainRenderObject->getHeightMapHeight(objPos.X, objPos.Y, &normal);
1170 Vector3 groundNormal(normal.x,normal.y,normal.z);
1171
1172 //Find new tu vector parallel to terrain by projecting existing x_vector onto
1173 //terrain normal and subtracting the result.
1174 uVector=objXform.Get_X_Vector();
1175 Real uVectorAlongNormal = Vector3::Dot_Product(uVector,groundNormal);
1176 uVector -= uVectorAlongNormal * groundNormal;
1177 uVector.Normalize();
1178 //Find new tv vector parallel to terrain by crossing new tu vector with terrain normal.
1179 Vector3::Cross_Product(uVector,groundNormal,&vVector);
1180
1181 Int numVerts = 4; //number of decal vertices
1182 Int numIndex=6; //(2 triangles).
1183
1184 SHADOW_DECAL_VERTEX* pvVertices;
1185 UnsignedShort *pvIndices;
1186
1187 if (nShadowDecalVertsInBuf > (SHADOW_DECAL_VERTEX_SIZE-numVerts)) //check if room for model verts
1188 { //flush the buffer by drawing the contents and re-locking again
1189 flushDecals(shadow->m_shadowTexture[0], shadow->m_type);
1190 if (shadowDecalVertexBufferD3D->Lock(0,numVerts*sizeof(SHADOW_DECAL_VERTEX),(unsigned char**)&pvVertices,D3DLOCK_DISCARD) != D3D_OK)
1191 return;
1192
1193 nShadowDecalStartBatchVertex=0;
1194 nShadowDecalPolysInBatch=0; //reset number of polys in texture batch
1195 nShadowDecalVertsInBatch=0;
1196 nShadowDecalVertsInBuf=0;
1197 }
1198 else
1199 { if (shadowDecalVertexBufferD3D->Lock(nShadowDecalVertsInBuf*sizeof(SHADOW_DECAL_VERTEX),numVerts*sizeof(SHADOW_DECAL_VERTEX), (unsigned char**)&pvVertices,D3DLOCK_NOOVERWRITE) != D3D_OK)
1200 return;
1201 }
1202
1203 objPos.Z=groundHeight; //force decal to ground level
1204 objPos += groundNormal * 1.0f; //offset decal slightly above terrain to reduce z-fighting.
1205 Vector3 vertex;
1206
1207 if(pvVertices)
1208 {
1209 //Top-left
1210 vertex = objPos + vVector * shadow->m_decalSizeY * -0.5f - uVector * shadow->m_decalSizeX * 0.5f;
1211 pvVertices->x=vertex.X;
1212 pvVertices->y=vertex.Y;
1213 pvVertices->z=vertex.Z;
1214 pvVertices->u=0.0f;
1215 pvVertices->v=0.0f;
1216 pvVertices++;
1217
1218 //Bottom-left
1219 vertex += vVector * shadow->m_decalSizeY;
1220 pvVertices->x=vertex.X;
1221 pvVertices->y=vertex.Y;
1222 pvVertices->z=vertex.Z;
1223 pvVertices->u=0.0f;
1224 pvVertices->v=1.0f;
1225 pvVertices++;
1226
1227 //Bottom-right
1228 vertex += uVector * shadow->m_decalSizeX;
1229 pvVertices->x=vertex.X;
1230 pvVertices->y=vertex.Y;
1231 pvVertices->z=vertex.Z;
1232 pvVertices->u=1.0f;
1233 pvVertices->v=1.0f;
1234 pvVertices++;
1235
1236 //Top-right
1237 vertex -= vVector * shadow->m_decalSizeY;
1238 pvVertices->x=vertex.X;
1239 pvVertices->y=vertex.Y;
1240 pvVertices->z=vertex.Z;
1241 pvVertices->u=1.0f;
1242 pvVertices->v=0.0f;
1243 pvVertices++;
1244 }
1245
1246 shadowDecalVertexBufferD3D->Unlock();
1247
1248 if (nShadowDecalIndicesInBuf > (SHADOW_DECAL_INDEX_SIZE-numIndex)) //check if room for model verts
1249 { //flush the buffer by drawing the contents and re-locking again
1250 flushDecals(shadow->m_shadowTexture[0],shadow->m_type);
1251
1252 if (shadowDecalIndexBufferD3D->Lock(0,numIndex*sizeof(short),(unsigned char**)&pvIndices,D3DLOCK_DISCARD) != D3D_OK)
1253 return;
1254
1255 nShadowDecalStartBatchIndex=0;
1256 nShadowDecalPolysInBatch=0; //reset number of polys in texture batch
1257 nShadowDecalVertsInBatch=0;
1258 nShadowDecalIndicesInBuf=0;
1259 }
1260 else
1261 { if (shadowDecalIndexBufferD3D->Lock(nShadowDecalIndicesInBuf*sizeof(short),numIndex*sizeof(short), (unsigned char**)&pvIndices,D3DLOCK_NOOVERWRITE) != D3D_OK)
1262 return;
1263 }
1264
1265 try {
1266 if(pvIndices)
1267 { pvIndices[0]=nShadowDecalVertsInBatch;
1268 pvIndices[1]=nShadowDecalVertsInBatch+1;
1269 pvIndices[2]=nShadowDecalVertsInBatch+2;
1270 pvIndices[3]=nShadowDecalVertsInBatch;
1271 pvIndices[4]=nShadowDecalVertsInBatch+2;
1272 pvIndices[5]=nShadowDecalVertsInBatch+3;
1273 pvIndices += 6;
1274 }
1275 IndexBufferExceptionFunc();
1276 } catch(...) {
1277 IndexBufferExceptionFunc();
1278 }
1279
1280 shadowDecalIndexBufferD3D->Unlock();
1281
1282 Int numPolys = 2; //2 triangles per decal
1283 nShadowDecalPolysInBatch += numPolys;
1284
1285 nShadowDecalVertsInBuf += numVerts;
1286 nShadowDecalVertsInBatch += numVerts;
1287
1288 nShadowDecalIndicesInBuf += numIndex;
1289 return;
1290 }

1294Int W3DProjectedShadowManager::renderShadows(RenderInfoClass & rinfo)
1295{
1297 W3DProjectedShadow *shadow;
1298 static AABoxClass aaBox;
1299 static SphereClass sphere;
1300 Int projectionCount=0;
1301
1302 if (!m_shadowList && !m_decalList)
1303 return projectionCount; //there are no shadows to render.
1304
1305 //Find extents of visible terrain
1306 if (TheTerrainRenderObject)
1307 {
1308 WorldHeightMap *hmap=TheTerrainRenderObject->getMap();
1309
1310 drawEdgeY=hmap->getDrawOrgY()+hmap->getDrawHeight()-1;
1311 drawEdgeX=hmap->getDrawOrgX()+hmap->getDrawWidth()-1;
1312 if (drawEdgeX > (hmap->getXExtent()-1))
1313 drawEdgeX = hmap->getXExtent()-1;
1314 if (drawEdgeY > (hmap->getYExtent()-1))
1315 drawEdgeY = hmap->getYExtent()-1;
1316 drawStartX=hmap->getDrawOrgX();
1317 drawStartY=hmap->getDrawOrgY();
1318 }
1319 else
1320 return projectionCount;
1321
1322 //According to Nvidia there's a D3D bug that happens if you don't start with a
1323 //new dynamic VB each frame - so we force a DISCARD by overflowing the counter.
1324 nShadowDecalVertsInBuf = 0xffff;
1325 nShadowDecalIndicesInBuf = 0xffff;
1326
1327 if (TheGlobalData->m_useShadowDecals)
1328 {
1329 // Render the object
1330 TheDX8MeshRenderer.Set_Camera(&rinfo.Camera);
1331
1332 //keep track of active decal texture so we can render all decals at once.
1333 W3DShadowTexture *lastShadowDecalTexture=NULL;
1334 ShadowType lastShadowType = SHADOW_NONE;
1335
1336 for( shadow = m_shadowList; shadow; shadow = shadow->m_next )
1337 {
1338 if (shadow->m_isEnabled && !shadow->m_isInvisibleEnabled)
1339 {
1340 if (shadow->m_type & SHADOW_DECAL)
1341 {
1342 if (lastShadowDecalTexture == NULL)
1343 lastShadowDecalTexture=m_shadowList->m_shadowTexture[0];
1344 if (lastShadowType == SHADOW_NONE)
1345 lastShadowType = m_shadowList->m_type;
1346
1347 if (shadow->m_shadowTexture[0] != lastShadowDecalTexture ||
1348 shadow->m_type != lastShadowType)
1349 { flushDecals(lastShadowDecalTexture,lastShadowType); //switched to a new texture, need to render polys using last texture.
1350 lastShadowDecalTexture=shadow->m_shadowTexture[0];
1351 lastShadowType=shadow->m_type;
1352 }
1354 if (shadow->m_robj->Is_Really_Visible())
1355 { //queueSimpleDecal(shadow);
1356 queueDecal(shadow); //only draw shadow if casting object is visible
1357 projectionCount++;
1358 }
1359 continue;
1360 }
1361
1362 //First test if shadow is visible on screen
1363 sphere=shadow->m_shadowTexture[0]->getBoundingSphere();
1364 sphere.Center += shadow->m_robj->Get_Position();
1365
1366 CollisionMath::OverlapType result=CollisionMath::Overlap_Test(*shadowCameraFrustum,sphere);
1367 if (result == CollisionMath::OVERLAPPED)
1368 { //do a more accurate test against bounding box.
1369 aaBox=shadow->m_shadowTexture[0]->getBoundingBox();
1370 aaBox.Translate(shadow->m_robj->Get_Position()); //translate bounding box to world space.
1371 if (CollisionMath::Overlap_Test(*shadowCameraFrustum,aaBox) == CollisionMath::OUTSIDE)
1372 continue;
1373 }
1374 else
1375 if (result == CollisionMath::OUTSIDE)
1376 continue;
1377
1378 //Shadow is visible on screen. Figure out which visible objects it may affect.
1379
1380 //Check if bounding sphere was inside so bounding box never initialized
1381 if (result == CollisionMath::INSIDE)
1382 { aaBox=shadow->m_shadowTexture[0]->getBoundingBox();
1383 aaBox.Translate(shadow->m_robj->Get_Position()); //translate bounding box to world space.
1384 }
1385
1386 if (shadow->m_type == SHADOW_PROJECTION)
1387 {
1388 //build inverse camera/view transforms needed for projection
1389 shadow->updateProjectionParameters(rinfo.Camera.Get_Transform());
1390 TexProjectClass *projector=shadow->getShadowProjector();
1391
1392 //terrain is always visible and affected by all shadows so must render
1393 projector->Peek_Material_Pass()->Install_Materials();
1394 DX8Wrapper::Apply_Render_State_Changes(); //force update of view and projection matrices
1395 if (renderProjectedTerrainShadow(shadow, aaBox))
1396 projectionCount++;
1397 projector->Peek_Material_Pass()->UnInstall_Materials();
1398
1399 SimpleObjectIterator *iter;
1400 Object *obj;
1401
1402 iter = ThePartitionManager->iterateObjectsInRange((const Coord3D*)&sphere.Center,sphere.Radius, FROM_CENTER_3D);
1403 MemoryPoolObjectHolder hold( iter );
1404
1405 AABoxIntersectionTestClass boxtest(aaBox,COLL_TYPE_ALL);
1406
1407 for( obj = iter->first(); obj; obj = iter->next() )
1408 {
1409 Drawable *draw = obj->getDrawable();
1410
1411 for (DrawModule ** dm = draw->getDrawModules(); *dm; ++dm)
1412 {
1413 const ObjectDrawInterface* di = (*dm)->getObjectDrawInterface();
1414 if (di)
1415 {
1416 W3DModelDraw *w3dDraw= (W3DModelDraw *)di;
1417 RenderObjClass *robj=NULL;
1418
1420 if ((robj=w3dDraw->getRenderObject()) != 0 && (!robj->Is_Alpha() || !obj->isKindOf(KINDOF_IMMOBILE)) && robj != shadow->m_robj && robj->Is_Really_Visible())
1421 {
1422 //do a more accurate test against W3D render bounding boxes.
1423 if (robj->Intersect_AABox(boxtest))
1424 {
1425 //Shadow reached a visible object so it needs to be rendered with shadow applied.
1426 rinfo.Push_Material_Pass(projector->Peek_Material_Pass());
1427 rinfo.Push_Override_Flags(RenderInfoClass::RINFO_OVERRIDE_ADDITIONAL_PASSES_ONLY);
1428 robj->Render(rinfo); //WW3D::Render(*robj,rinfo);
1429 rinfo.Pop_Override_Flags();
1430 rinfo.Pop_Material_Pass();
1431 projectionCount++; //keep track of number of shadow projections
1432 }
1433 }//robj
1434 }//di
1435 } // end for drawmodule
1436 } // end for obj
1437 }
1438 }//shadow is enabled
1439 }
1440
1441 flushDecals(lastShadowDecalTexture,lastShadowType); //make sure there are not any unrendered decals left over.
1442 TheDX8MeshRenderer.Flush(); //draw all the shadow receiving objects
1443 }//rendering shadows
1444 if (m_decalList)
1445 {
1446 //keep track of active decal texture so we can render all decals at once.
1447 W3DShadowTexture *lastShadowDecalTexture=NULL;
1448 ShadowType lastShadowType = SHADOW_NONE;
1449
1450 for( shadow = m_decalList; shadow; shadow = shadow->m_next )
1451 {
1452 if (shadow->m_isEnabled && !shadow->m_isInvisibleEnabled)
1453 {
1454 if (lastShadowDecalTexture == NULL)
1455 lastShadowDecalTexture=m_decalList->m_shadowTexture[0];
1456 if (lastShadowType == SHADOW_NONE)
1457 lastShadowType = m_decalList->m_type;
1458
1459 if (shadow->m_shadowTexture[0] != lastShadowDecalTexture ||
1460 shadow->m_type != lastShadowType)
1461 { flushDecals(lastShadowDecalTexture,lastShadowType); //switched to a new texture, need to render polys using last texture.
1462 lastShadowDecalTexture=shadow->m_shadowTexture[0];
1463 lastShadowType=shadow->m_type;
1464 }
1466 if (!(shadow->m_robj && !shadow->m_robj->Is_Really_Visible()))
1467 { //queueSimpleDecal(shadow);
1468 queueDecal(shadow); //only draw shadow if casting object is visible
1469 projectionCount++;
1470 }
1471 }//shadow is enabled
1472 }
1473
1474 flushDecals(lastShadowDecalTexture,lastShadowType); //make sure there are not any unrendered decals left over.
1475 }
1476 return projectionCount;
1477}
