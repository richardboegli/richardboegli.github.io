562 inline void prependToDirtyModules(PartitionData** pListHead)
563 {
564 DEBUG_ASSERTCRASH((m_dirtyStatus != NOT_DIRTY), ("dirty flag mismatch"));
565 m_nextDirty = *pListHead;
566 if (*pListHead)
567 (*pListHead)->m_prevDirty = this;
568 *pListHead = this;
569 }
570 void removeFromDirtyModules(PartitionData** pListHead)
571 {
572 m_dirtyStatus = NOT_DIRTY;
573 if (m_nextDirty)
574 m_nextDirty->m_prevDirty = m_prevDirty;
575 if (m_prevDirty)
576 m_prevDirty->m_nextDirty = m_nextDirty;
577 else
578 *pListHead = m_nextDirty;
579 m_prevDirty = 0;
580 m_nextDirty = 0;
581 }
