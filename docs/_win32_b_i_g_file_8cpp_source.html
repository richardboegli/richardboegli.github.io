59File* Win32BIGFile::openFile( const Char *filename, Int access )
60{
61 const ArchivedFileInfo *fileInfo = getArchivedFileInfo(AsciiString(filename));
62
63 if (fileInfo == NULL) {
64 return NULL;
65 }
66
67 RAMFile *ramFile = NULL;
68
69 if (BitTest(access, File::STREAMING))
70 ramFile = newInstance( StreamingArchiveFile );
71 else
72 ramFile = newInstance( RAMFile );
73
74 ramFile->deleteOnClose();
75 if (ramFile->openFromArchive(m_file, fileInfo->m_filename, fileInfo->m_offset, fileInfo->m_size) == FALSE) {
76 ramFile->close();
77 ramFile = NULL;
78 return NULL;
79 }
80
81 if ((access & File::WRITE) == 0) {
82 // requesting read only access. Just return the RAM file.
83 return ramFile;
84 }
85
86 // whoever is opening this file wants write access, so copy the file to the local disk
87 // and return that file pointer.
88
89 File *localFile = TheLocalFileSystem->openFile(filename, access);
90 if (localFile != NULL) {
91 ramFile->copyDataToFile(localFile);
92 }
93
94 ramFile->close();
95 ramFile = NULL;
96
97 return localFile;
98}

149Bool Win32BIGFile::getFileInfo(const AsciiString& filename, FileInfo *fileInfo) const
150{
151 const ArchivedFileInfo *tempFileInfo = getArchivedFileInfo(filename);
152
153 if (tempFileInfo == NULL) {
154 return FALSE;
155 }
156
157 TheLocalFileSystem->getFileInfo(AsciiString(m_file->getName()), fileInfo);
158
159 // fill in the size info. Since the size can't be bigger than a JUNK file, the high Int will always be 0.
160 fileInfo->sizeHigh = 0;
161 fileInfo->sizeLow = tempFileInfo->m_size;
162
163 return TRUE;
164}
