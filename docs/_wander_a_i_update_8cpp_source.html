53UpdateSleepTime WanderAIUpdate::update( void )
54{
55 // If I'm standing still, move somewhere
56 if (isIdle())
57 {
58 Coord3D dest = *(getObject()->getPosition());
59 dest.x += GameLogicRandomValue( 5, 50 );
60 dest.y += GameLogicRandomValue( 5, 50 );

62 }
63
