111ResizerWindow *ControlBarResizer::findResizerWindow( AsciiString name )
112{
113 ResizerWindowList::iterator it = m_resizerWindowsList.begin();
114
115 while (it != m_resizerWindowsList.end())
116 {
117 ResizerWindow *rWin = *it;
118 if( !rWin )
119 {
120 DEBUG_ASSERTCRASH(FALSE,("There's no resizerWindow in ControlBarResizer::findResizerWindow"));
121 it++;
122 continue;
123 }
124 // find the scheme that best matches our resolution
125 if(rWin->m_name.compare(name) == 0)
126 {
127 return rWin;
128 }
129 it ++;
130 }
131 return NULL;
132}

134ResizerWindow *ControlBarResizer::newResizerWindow( AsciiString name )
135{
136 ResizerWindow *newRwin = NEW ResizerWindow;
137 if(!newRwin)
138 return NULL;
139
140 newRwin->m_name = name;
141 GameWindow *win = NULL;
142 win = TheWindowManager->winGetWindowFromId(NULL, TheNameKeyGenerator->nameToKey(name));
143 if( !win )
144 {
145 DEBUG_ASSERTCRASH(win,("ControlBarResizer::newResizerWindow could not find window %s Are you sure that window is loaded yet?", name.str()) );
146 delete newRwin;
147 return NULL;
148 }
149 win->winGetPosition(&newRwin->m_defaultPos.x,&newRwin->m_defaultPos.y);
150 win->winGetSize(&newRwin->m_defaultSize.x,&newRwin->m_defaultSize.y);
151 m_resizerWindowsList.push_back(newRwin);
152 return newRwin;
153}

179void ControlBarResizer::sizeWindowsAlt( void )
180{
181 ResizerWindowList::iterator it = m_resizerWindowsList.begin();
182 GameWindow *win = NULL;
183 Real x = (Real)TheDisplay->getWidth() / 800;
184 Real y = (Real)TheDisplay->getHeight() / 600;
185 while (it != m_resizerWindowsList.end())
186 {
187 ResizerWindow *rWin = *it;
188 if( !rWin )
189 {
190 DEBUG_ASSERTCRASH(FALSE,("There's no resizerWindow in ControlBarResizer::sizeWindowsDefault"));
191 it++;
192 continue;
193 }
194 win = TheWindowManager->winGetWindowFromId(NULL, TheNameKeyGenerator->nameToKey(rWin->m_name));
195 if(!win)
196 {
197 it++;
198 continue;
199 }
200
201 win->winSetPosition(rWin->m_altPos.x * x, rWin->m_altPos.y * y);
202 if(rWin->m_altSize.x >0 || rWin->m_altSize.y > 0)
203 win->winSetSize(rWin->m_altSize.x *x, rWin->m_altSize.y *y);
204 DEBUG_LOG(("sizeWindowsAlt:%s pos X:%d pos Y: %d size X:%d sizeY: %d",rWin->m_name.str(), rWin->m_altPos.x*x, rWin->m_altPos.y*y,rWin->m_altSize.x*x, rWin->m_altSize.y *y));
205 it ++;
206 }
207}
