103void DemoTrapUpdate::onObjectCreated()
104{
105
106 const DemoTrapUpdateModuleData *data = getDemoTrapUpdateModuleData();
107
108 if( data->m_detonationWeaponSlot == data->m_proximityModeWeaponSlot ||
109 data->m_detonationWeaponSlot == data->m_manualModeWeaponSlot ||
110 data->m_proximityModeWeaponSlot == data->m_manualModeWeaponSlot )
111 {
112 DEBUG_CRASH( ("The demo trap requires three weaponslots: One for each of the detonation mode, proximity mode, and manual mode.") );
113 }
114
115 getObject()->setWeaponSetFlag( WEAPONSET_VETERAN );
116 if( data->m_defaultsToProximityMode )
117 {
118 // lock it just till the weapon is empty or the attack is "done"
119 getObject()->setWeaponLock( data->m_proximityModeWeaponSlot, LOCKED_TEMPORARILY );
120 }
121 else
122 {
123 // lock it just till the weapon is empty or the attack is "done"
124 getObject()->setWeaponLock( data->m_manualModeWeaponSlot, LOCKED_TEMPORARILY );
125 }
126}

131UpdateSleepTime DemoTrapUpdate::update()
132{
134 const DemoTrapUpdateModuleData *data = getDemoTrapUpdateModuleData();
135
136 if( m_detonated )
137 {
138 return UPDATE_SLEEP_NONE;
139 }
140
141 Object *me = getObject();
142
143 if( me->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) || me->testStatus(OBJECT_STATUS_SOLD) )
144 {
145 return UPDATE_SLEEP_NONE;
146 }
147
148 if( me->isEffectivelyDead() )
149 {
150 if( data->m_detonateWhenKilled )
151 {
152 detonate();
153 }
154 return UPDATE_SLEEP_NONE;
155 }
156
157 //Get the current weapon slot -- this determines what mode we're in.
158 WeaponSlotType weaponSlot = getObject()->getCurrentWeapon()->getWeaponSlot();
159
160 if( weaponSlot == data->m_detonationWeaponSlot )
161 {
162 //We've been externally triggered by the press of a command button.
163 detonate();
164 return UPDATE_SLEEP_NONE;
165 }
166
167 //Don't scan every frame for performance reasons.
168 if( m_nextScanFrames > 0 )
169 {
170 m_nextScanFrames--;
171 return UPDATE_SLEEP_NONE;
172 }
173
174
175 if( weaponSlot == data->m_manualModeWeaponSlot )
176 {
177 //Don't scan!
178 return UPDATE_SLEEP_NONE;
179 }
180
181 //Reset timer here -- because if we are in manual mode, and switch, we want instant
182 //gratification (if possible).
183 m_nextScanFrames = data->m_scanFrames;
184
185 //Scan for a valid enemy in proximity range.
186
187 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( me->getPosition(), data->m_triggerDetonationRange, FROM_CENTER_2D );
188 MemoryPoolObjectHolder hold(iter);
189
190 Bool shallDetonate = false;
191
192 //Now iterate through each object in range and check to see if it should detonate us!
193 for( Object *other = iter->first(); other; other = iter->next() )
194 {
195 if( other->isAnyKindOf( data->m_ignoreKindOf ) )
196 {
197 //Skip specified types to ignore.
198 continue;
199 }
200 if( other->isEffectivelyDead() )
201 {
202 continue;
203 }
204
205 if( other->isKindOf( KINDOF_DOZER ) )
206 {
207 //If we're dealing with a dozer... check if it's trying to disarm me. If so, don't blow up!
208 Weapon *weapon = other->getCurrentWeapon();
209 if( weapon && weapon->getDamageType() == DAMAGE_DISARM )
210 {
211 //Also check if it's attacking, because it seems to stay in disarm mode.
212 if( other->testStatus( OBJECT_STATUS_IS_ATTACKING ) )
213 {
214 continue;
215 }
216 }
217 }
218
219 // order matters: we want to know if I consider it to be an enemy, not vice versa
220 if( getObject()->getRelationship( other ) != ENEMIES )
221 {
222 if( !data->m_friendlyDetonation )
223 {
224 //Not allowed to proximity detonate with friends nearby
225 return UPDATE_SLEEP_NONE;
226 }
227 //Don't shoot our friends!
228 continue;
229 }
230
231 if( other->isAboveTerrain() )
232 {
233 //Don't detonate on anything airborne.
234 continue;
235 }
236
237 //Anyone close enough?
238 Real fDist = ThePartitionManager->getDistanceSquared( me, other, FROM_CENTER_2D );
239 if( fDist <= data->m_triggerDetonationRange * data->m_triggerDetonationRange )
240 {
241 //Yeehaw!
242 shallDetonate = true;
243
244 if( data->m_friendlyDetonation )
245 {
246 //Okay, no need to look for friends because we don't care. All we care
247 //about is the fact that there is an enemy nearby!
248 break;
249 }
250 }
251 }
252
253 if( shallDetonate )
254 {
255 //Enemy in proximity and we are in proximity detonation mode, so trigger the explosion
256 //and kill them!!! Muwahahaha!
257 detonate();
258 }
259 return UPDATE_SLEEP_NONE;
260}

264void DemoTrapUpdate::detonate()
265{
266 const DemoTrapUpdateModuleData *data = getDemoTrapUpdateModuleData();
267 Object *me = getObject();
268
269 // Only shoot the weapon if not being built or sold.
270 if( !me->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) && !me->testStatus(OBJECT_STATUS_SOLD) )
271 TheWeaponStore->createAndFireTempWeapon( data->m_detonationWeaponTemplate, me, me->getPosition() );
272
273 me->kill();
274 m_detonated = true;
275}
