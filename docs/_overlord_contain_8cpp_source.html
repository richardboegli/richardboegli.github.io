119void OverlordContain::createPayload()
120{
121 OverlordContainModuleData* self = (OverlordContainModuleData*)getOverlordContainModuleData();
122
123
124 // Any number of different passengers can be loaded here at init time
125 Object* object = getObject();
126 ContainModuleInterface *contain = object->getContain();
127 if( contain )
128 {
129 contain->enableLoadSounds( FALSE );
130
131 TemplateNameList list = self->m_payloadTemplateNameData;
132 TemplateNameIterator iter = list.begin();
133 while ( iter != list.end() )
134 {
135 const ThingTemplate* temp = TheThingFactory->findTemplate( *iter );
136 if (temp)
137 {
138 Object* payload = TheThingFactory->newObject( temp, object->getTeam() );
139
140 if( contain->isValidContainerFor( payload, true ) )
141 {
142 contain->addToContain( payload );
143 }
144 else
145 {
146 DEBUG_CRASH( ( "OverlordContain::createPayload: %s is full, or not valid for the payload %s!", object->getName().str(), self->m_initialPayload.name.str() ) );
147 }
148
149 }
150
151 ++iter;
152 }
153
154 contain->enableLoadSounds( TRUE );
155
156 } // endif contain
157
158 m_payloadCreated = TRUE;
159
160}

164void OverlordContain::onBodyDamageStateChange( const DamageInfo* damageInfo,
165 BodyDamageType oldState,
166 BodyDamageType newState)
167{
168 // I can't use any convienience functions, as they will all get routed to the bunker I may carry.
169 // I want just me.
170 // Oh, and I don't want this function trying to do death. That is more complicated and will be handled
171 // on my death.
172 if( newState != BODY_RUBBLE && m_containListSize == 1 )
173 {
174 Object *myGuy = m_containList.front();
175 myGuy->getBodyModule()->setDamageState( newState );
176 }
177}

181ContainModuleInterface *OverlordContain::getRedirectedContain() const
182{
183 // Naturally, I can not use a redirectible convienience function
184 // to answer if I am redirecting yet.
185
186 // If I am empty, say no.
187 if( m_containListSize < 1 )
188 return NULL;
189
190 if( !m_redirectionActivated )
191 return NULL;// Shut off early to allow death to happen without my bunker having
192 // trouble finding me to say goodbye as messages get sucked up the pipe to him.
193
194 Object *myGuy = m_containList.front();
195 if( myGuy )
196 return myGuy->getContain();
197
198 return NULL;// Or say no if they have no contain.
199}

209void OverlordContain::onDie( const DamageInfo *damageInfo )
210{
211 // Do you mean me the Overlord, or my behavior of passing stuff on to my passengers?
212 if( getRedirectedContain() == NULL )
213 {
214 TransportContain::onDie( damageInfo );
215 return;
216 }
217 //Everything is fine if I am empty or carrying a regular guy. If I have a redirected contain
218 // set up, then I need to handle the order of death explicitly, or things will become confused
219 // when I stop redirecting in the middle of the process. Or I will get confused as my commands
220 // get sucked up the pipe.
221
222 // So this is an extend that lets me control the order of death.
223
224 deactivateRedirectedContain();
225 Object *myGuy = m_containList.front();
226 myGuy->kill();
227
228 TransportContain::onDie( damageInfo );
229}

365void OverlordContain::onContaining( Object *obj, Bool wasSelected )
366{
367 // Do you mean me the Overlord, or my behavior of passing stuff on to my passengers?
368 if( getRedirectedContain() == NULL )
369 {
370 TransportContain::onContaining( obj, wasSelected );
371
372
373 if ( obj->isKindOf( KINDOF_PORTABLE_STRUCTURE ) )
374 {
375 activateRedirectedContain();//Am now carrying something
376
377 // And this contain style explicitly sucks XP from our little friend.
378 if( getOverlordContainModuleData()->m_experienceSinkForRider && obj->getExperienceTracker() )
379 obj->getExperienceTracker()->setExperienceSink(getObject()->getID());
380
381
382 if ( obj->isKindOf( KINDOF_PORTABLE_STRUCTURE ) && getObject()->testStatus( OBJECT_STATUS_STEALTHED ) )
383 {
384 StealthUpdate *myStealth = obj->getStealth();
385 if ( myStealth )
386 {
387 myStealth->receiveGrant( true );
388 // note to anyone... once stealth is granted to this gattlingcannon ( or such )
389 // let its own stealthupdate govern the allowedtostealth cases
390 // a portable structure never gets removed, so...
391 }
392 }

400 return;
401 }
