86WindowMsgHandledType GadgetTextEntryInput( GameWindow *window, UnsignedInt msg,
87 WindowMsgData mData1, WindowMsgData mData2 )
88{
89 EntryData *e = (EntryData *)window->winGetUserData();
90 WinInstanceData *instData = window->winGetInstanceData();
91
92 if ( TheIMEManager && TheIMEManager->isAttachedTo( window) && TheIMEManager->isComposing())
93 {
94 // ignore input while IME has focus
95 return MSG_HANDLED;
96 }
97
98 switch( msg )
99 {
100 // ------------------------------------------------------------------------
101 case GWM_IME_CHAR:
102 {
103 WideChar ch = (WideChar) mData1;
104
105 // --------------------------------------------------------------------
106 if ( ch == VK_RETURN )
107 {
108 // Done with this edit
109 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
110 GEM_EDIT_DONE,
111 (WindowMsgData)window,
112 0 );
113 return MSG_HANDLED;
114 };
115
116 if( ch )
117 {
118 // Constrain keys based on rules for entry box.
119 if( e->numericalOnly )
120 {
121 if( TheWindowManager->winIsDigit( ch ) == 0 )
122 return MSG_HANDLED;
123 }
124 if( e->alphaNumericalOnly )
125 {
126 if( TheWindowManager->winIsAlNum( ch ) == 0 )
127 return MSG_HANDLED;
128 }
129 if ( e->aSCIIOnly )
130 {
131 if ( TheWindowManager->winIsAscii( ch ) == 0 )
132 {
133 return MSG_HANDLED;
134 }
135 }
136
137 if( e->charPos < e->maxTextLen-1 )
138 {
139 e->text->appendChar( ch );
140 e->sText->appendChar( L'*' );
141 e->charPos++;
142 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
143 GEM_UPDATE_TEXT,
144 (WindowMsgData)window,
145 0 );
146 }
147 }
148 break;
149 }
150 // ------------------------------------------------------------------------
151 case GWM_CHAR:
152 if ( BitTest( mData2, KEY_STATE_DOWN ) && BitTest( mData2, KEY_STATE_ALT | KEY_STATE_CONTROL ) )
153 {
154 return MSG_IGNORED; // text extries shouldn't care about CTRL+* or ALT+*
155 }
156
157 switch( mData1 )
158 {
159 /*
160 // --------------------------------------------------------------------
161 case KEY_KPENTER:
162 case KEY_ENTER:
163 // Done with this edit
164 if( BitTest( mData2, KEY_STATE_DOWN ) )
165 {
166 if( e->receivedUnichar == FALSE )
167 {
168 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
169 GEM_EDIT_DONE,
170 (WindowMsgData)window,
171 0 );
172 }
173 }
174
175 break;
176 */
177 // --------------------------------------------------------------------
178 // Don't process these keys
179 case KEY_ESC:
180 case KEY_PGUP:
181 case KEY_PGDN:
182 case KEY_HOME:
183 case KEY_END:
184 case KEY_F1:
185 case KEY_F2:
186 case KEY_F3:
187 case KEY_F4:
188 case KEY_F5:
189 case KEY_F6:
190 case KEY_F7:
191 case KEY_F8:
192 case KEY_F9:
193 case KEY_F10:
194 case KEY_F11:
195 case KEY_F12:
196 case KEY_CAPS:
197 case KEY_DEL:
198 return MSG_IGNORED;
199
200 // --------------------------------------------------------------------
201 case KEY_DOWN:
202 case KEY_RIGHT:
203 case KEY_TAB:
204
205 if( BitTest( mData2, KEY_STATE_DOWN ) )
206 {
207 GameWindow *parent;
208 parent = window->winGetParent();
209 if(parent && !BitTest(parent->winGetStyle(), GWS_COMBO_BOX))
210 parent = NULL;
211 if(parent)
212 TheWindowManager->winNextTab(parent);
213 else
214 TheWindowManager->winNextTab(window);
215 }
216
217 break;
218
219 // --------------------------------------------------------------------
220 case KEY_UP:
221 case KEY_LEFT:
222
223 if( BitTest( mData2, KEY_STATE_DOWN ) )
224 {
225 GameWindow *parent;
226 parent = window->winGetParent();
227 if(parent && !BitTest(parent->winGetStyle(), GWS_COMBO_BOX))
228 parent = NULL;
229 if(parent)
230 TheWindowManager->winPrevTab(parent);
231 else
232 TheWindowManager->winPrevTab(window);
233 }
234 break;
235
236 // --------------------------------------------------------------------
237 case KEY_BACKSPACE:
238
239 if( BitTest( mData2, KEY_STATE_DOWN ) )
240 {
241 // if conCharPos != 0 this will fall through to next case.
242 // it should be noted that conCharPos can only != 0 in Jap & Kor
243 if( e->conCharPos == 0 )
244 {
245 if( e->charPos > 0 )
246 {
247
248 e->text->removeLastChar();
249 e->sText->removeLastChar();
250 e->charPos--;
251 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
252 GEM_UPDATE_TEXT,
253 (WindowMsgData)window,
254 0 );
255 } // end if
256 }
257 }
258 break;
259
260 } // end switch( mData1 )
261
262 break;
263
264 // ------------------------------------------------------------------------
265 case GWM_LEFT_DOWN:
266 BitSet( instData->m_state, WIN_STATE_HILITED );
267 TheWindowManager->winSetFocus( window );
268 break;
269
270 // ------------------------------------------------------------------------
271 case GWM_MOUSE_ENTERING:
272
273 if (BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
274 {
275
276 BitSet( instData->m_state, WIN_STATE_HILITED );
277 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
278 GBM_MOUSE_ENTERING,
279 (WindowMsgData)window, 0 );
280 //TheWindowManager->winSetFocus( window );
281 }
282
283 break;
284
285 // ------------------------------------------------------------------------
286 case GWM_MOUSE_LEAVING:
287
288 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
289 {
290
291 BitClear( instData->m_state, WIN_STATE_HILITED );
292 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
293 GBM_MOUSE_LEAVING,
294 (WindowMsgData)window, 0 );
295 }
296 break;
297
298 // ------------------------------------------------------------------------
299 case GWM_LEFT_DRAG:
300
301 if( BitTest( instData->getStyle(), GWS_MOUSE_TRACK ) )
302 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
303 GGM_LEFT_DRAG,
304 (WindowMsgData)window, 0 );
305 break;
306
307 // ------------------------------------------------------------------------
308 default:
309 return MSG_IGNORED;
310
311 } // end switch( msg )
312
313 return MSG_HANDLED;
314
315} // end GadgetTextEntryInput

317// GadgetTextEntrySystem ======================================================

320WindowMsgHandledType GadgetTextEntrySystem( GameWindow *window, UnsignedInt msg,
321 WindowMsgData mData1, WindowMsgData mData2 )
322{
323 EntryData *e = (EntryData *)window->winGetUserData();
324 WinInstanceData *instData = window->winGetInstanceData();
325
326 switch( msg )
327 {
328
329 // ------------------------------------------------------------------------
330 case GEM_GET_TEXT:
331 *(UnicodeString*)mData2 = e->text->getText();
332 break;
333
334 // ------------------------------------------------------------------------
335 case GEM_SET_TEXT:
336 {
337
338 const UnicodeString* ustr = (const UnicodeString*)mData1;
339 e->text->setText( *ustr );
340 e->charPos = ustr->getLength();
341 e->constructText->setText( UnicodeString::TheEmptyString );
342 e->conCharPos = 0;
343
344 // set our secret text string to be filled with '*' the same length
345 e->sText->setText( UnicodeString::TheEmptyString );
346 Int len = ustr->getLength();
347 for( Int i = 0; i < len; i++ )
348 e->sText->appendChar( L'*' );
349
350 break;
351
352 } // end set text
353
354 // ------------------------------------------------------------------------
355 case GWM_CREATE:
356
357 break;
358
359 // ------------------------------------------------------------------------
360 case GWM_DESTROY:
361
362 // delete the edit display string
363 TheDisplayStringManager->freeDisplayString( e->text );
364 TheDisplayStringManager->freeDisplayString( e->sText );
365 TheDisplayStringManager->freeDisplayString( e->constructText );
366
367 // delete construct list
368 if( e->constructList )
369 TheWindowManager->winDestroy( e->constructList );
370
371 // free all edit data
372 delete( (EntryData *)window->winGetUserData() );
373 break;
374
375 // ------------------------------------------------------------------------
376 case GWM_INPUT_FOCUS:
377 if( mData1 == FALSE )
378 {
379 // If we're losing focus
381 // ourIME->UnActivate();
382 curWindow = NULL;
383 BitClear( instData->m_state, WIN_STATE_SELECTED );
384 BitClear( instData->m_state, WIN_STATE_HILITED );
385
386 if( e->constructList )
387 e->constructList->winHide( TRUE );
388 e->constructText->setText( UnicodeString::TheEmptyString );
389 e->conCharPos = 0;
390 if(TheIMEManager && TheIMEManager->isAttachedTo(window))
391 TheIMEManager->attach(NULL);
392 //TheIMEManager->detatch();
393 }
394 else
395 {
396 curWindow = window;
398 if (TheIMEManager)
399 TheIMEManager->attach( window );
400 // ourIME->Activate( (void *)ApplicationHWnd );
401 BitSet( instData->m_state, WIN_STATE_SELECTED );
402 BitSet( instData->m_state, WIN_STATE_HILITED );
403 }
404
405 TheWindowManager->winSendSystemMsg( window->winGetOwner(),
406 GGM_FOCUS_CHANGE,
407 mData1,
408 window->winGetWindowId() );
409
410 *(Bool*)mData2 = TRUE;
411 break;
412
413 default:
414 return MSG_IGNORED;
415
416 } // end switch( msg )
417
418 return MSG_HANDLED;
419
420} // end GadgetTextEntrySystem
