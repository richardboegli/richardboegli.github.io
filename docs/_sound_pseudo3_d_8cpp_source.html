131//
132// Update_Pseudo_Volume
133//
135void
136SoundPseudo3DClass::Update_Pseudo_Volume (float distance)
137{
138 MMSLockClass lock;
139
140 //
141 // Only do this if the sound is really playing
142 //
143 if (m_SoundHandle != NULL) {
144
145 float volume_mod = Determine_Real_Volume ();
146 float max_distance = Get_DropOff_Radius ();
147 float min_distance = Get_Max_Vol_Radius ();
148 float delta = max_distance - min_distance;
149
150 // Determine a normalized volume from the position
151 float volume = 1.0F;
152 if (distance > min_distance) {
153 volume = 1.0F - ((distance - min_distance) / delta);
154 volume = min (volume, 1.0F);
155 volume = max (volume, 0.0F);
156 }
157
158 // Multiply the 'max' volume with the calculated volume
159 volume = volume * volume_mod;
160
161 //
162 // Pass the volume on
163 //
164 m_SoundHandle->Set_Sample_Volume (int(volume * 127.0F));
165 }
166
167 return ;
168}

172//
173// Update_Pseudo_Volume
174//
176void
177SoundPseudo3DClass::Update_Pseudo_Volume (void)
178{
179 MMSLockClass lock;
180
181 // Only do this if the sound is really playing
182 if (m_SoundHandle != NULL) {
183
184 //
185 // Find the difference in the sound position and its listener's position
186 //
187 Vector3 sound_pos = m_ListenerTransform.Get_Translation () - m_Transform.Get_Translation ();
188 float distance = sound_pos.Quick_Length ();
189
190 //
191 // Determine a normalized volume from the position
192 //
193 Update_Pseudo_Volume (distance);
194 }
195
196 return ;
197}

201//
202// Update_Pseudo_Pan
203//
205void
206SoundPseudo3DClass::Update_Pseudo_Pan (void)
207{
208 MMSLockClass lock;
209
210 //
211 // Only do this if the sound is really playing
212 //
213 if (m_SoundHandle != NULL) {
214
215 //
216 // Transform the sound's position into 'listener-space'
217 //
218 Vector3 sound_pos = m_Transform.Get_Translation ();
219 Vector3 rel_sound_pos;
220 Matrix3D::Inverse_Transform_Vector (m_ListenerTransform, sound_pos, &rel_sound_pos);
221
222 //
223 // Calculate a normalized pan from 0 (hard left) to 1.0F (hard right)
224 //
225 float angle = WWMath::Atan2 (rel_sound_pos.Y, rel_sound_pos.X);
226 float pan = -WWMath::Fast_Sin (angle);
227 pan = (pan / 2.0F) + 0.5F;
228
229 //
230 // Pass the pan on
231 //
232 m_SoundHandle->Set_Sample_Pan (S32(pan * 127.0F));
233 }
234
235 return ;
236}
