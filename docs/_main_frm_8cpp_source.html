315void CMainFrame::adjustWindowSize(void)
316{
317 HWND hDesk = ::GetDesktopWindow();
318 CRect top;
319 ::GetWindowRect(hDesk, &top);
320 top.right -= 2*::GetSystemMetrics(SM_CYCAPTION);
321 top.bottom -= 3*::GetSystemMetrics(SM_CYCAPTION);
322
323 CRect client, window;
324 Int borderX = ::GetSystemMetrics(SM_CXEDGE);
325// Int borderY = ::GetSystemMetrics(SM_CYEDGE);
326 Int viewWidth = ::AfxGetApp()->GetProfileInt(MAIN_FRAME_SECTION, "Width", THREE_D_VIEW_WIDTH);
327 Int viewHeight = ::AfxGetApp()->GetProfileInt(MAIN_FRAME_SECTION, "Height", THREE_D_VIEW_HEIGHT);
328 WbView3d * pView = CWorldBuilderDoc::GetActive3DView();
329 if (pView) {
330 pView->GetClientRect(&client);
331 } else {
332 GetClientRect(&client);
333 client.right -= 2*borderX;
334 }
335 int widthDelta = client.Width() - (viewWidth);
336 int heightDelta = client.Height() - (viewHeight);
337 this->GetWindowRect(window);
338 Int newWidth = window.Width()-widthDelta;
339 Int newHeight = window.Height()-heightDelta;
340 this->SetWindowPos(NULL, 0,
341 0, newWidth, newHeight,
342 SWP_NOMOVE|SWP_NOZORDER); // MainFrm.cpp sets the top and left.
343 if (pView) {
344 pView->reset3dEngineDisplaySize(viewWidth, viewHeight);
345 }
346 m_3dViewWidth = viewWidth;
347}

379void CMainFrame::showOptionsDialog(Int dialogID)
380{
381 CWnd *newOptions = NULL;
382 switch(dialogID) {
383 case IDD_BRUSH_OPTIONS : newOptions = &m_brushOptions; break;
384 case IDD_TERRAIN_MATERIAL: newOptions = &m_terrainMaterial; break;
385 case IDD_BLEND_MATERIAL: newOptions = &m_blendMaterial; break;
386 case IDD_OBJECT_OPTIONS: newOptions = &m_objectOptions; break;
387 case IDD_FENCE_OPTIONS: newOptions = &m_fenceOptions; break;
388 case IDD_MAPOBJECT_PROPS: newOptions = &m_mapObjectProps; break;
389 case IDD_ROAD_OPTIONS:newOptions = &m_roadOptions; break;
390 case IDD_MOUND_OPTIONS:newOptions = &m_moundOptions; break;
391 case IDD_RULER_OPTIONS:newOptions = &m_rulerOptions; break;
392 case IDD_FEATHER_OPTIONS:newOptions = &m_featherOptions; break;
393 case IDD_MESHMOLD_OPTIONS:newOptions = &m_meshMoldOptions; break;
394 case IDD_WAYPOINT_OPTIONS:newOptions = &m_waypointOptions; break;
395 case IDD_WATER_OPTIONS:newOptions = &m_waterOptions; break;
396 case IDD_LIGHT_OPTIONS:newOptions = &m_lightOptions; break;
397 case IDD_BUILD_LIST_PANEL:newOptions = &m_buildListOptions; break;
398 case IDD_GROVE_OPTIONS:newOptions = &m_groveOptions; break;
399 case IDD_RAMP_OPTIONS:newOptions = &m_rampOptions; break;
400 case IDD_SCORCH_OPTIONS:newOptions = &m_scorchOptions; break;
401 case IDD_NO_OPTIONS:newOptions = &m_noOptions; break;
402 default : break;
403 }
404 CRect frameRect;
405 if (newOptions && newOptions != m_curOptions) {
406 newOptions->GetWindowRect(&frameRect);
407 if (m_curOptions) {
408 m_curOptions->GetWindowRect(&frameRect);
409 }
410 newOptions->SetWindowPos(m_curOptions, frameRect.left, frameRect.top,
411 m_optionsPanelWidth, m_optionsPanelHeight,
412 SWP_NOZORDER | SWP_NOACTIVATE );
413 ::AfxGetApp()->WriteProfileInt(OPTIONS_PANEL_SECTION, "Top", frameRect.top);
414 ::AfxGetApp()->WriteProfileInt(OPTIONS_PANEL_SECTION, "Left", frameRect.left);
415 newOptions->ShowWindow(SW_SHOWNA);
416 if (m_curOptions) {
417 m_curOptions->ShowWindow(SW_HIDE);
418 }
419 m_curOptions = newOptions;
420 }
