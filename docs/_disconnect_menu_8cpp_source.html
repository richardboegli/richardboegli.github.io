119void DisconnectMenu::setPlayerName(Int playerNum, UnicodeString name) {
120 NameKeyType id = TheNameKeyGenerator->nameToKey(m_playerNameTextControlNames[playerNum]);
121 GameWindow *control = TheWindowManager->winGetWindowFromId(NULL, id);
122
123 if (control != NULL) {
124 if (name.getLength() > 0) {
125 GadgetStaticTextSetText(control, name);
126// showPlayerControls(playerNum);

128 }
129
130 id = TheNameKeyGenerator->nameToKey(m_playerTimeoutTextControlNames[playerNum]);
131 control = TheWindowManager->winGetWindowFromId(NULL, id);
132

137 }
