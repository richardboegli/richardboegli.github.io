145void Connection::sendNetCommandMsg(NetCommandMsg *msg, UnsignedByte relay) {
146 static NetPacket *packet = NULL;
147
148 // this is done so we don't have to allocate and delete a packet every time we send a message.
149 if (packet == NULL) {
150 packet = newInstance(NetPacket);
151 }
152
153
154 if (m_isQuitting)
155 return;
156
157 if (m_netCommandList != NULL) {
158 // check to see if this command will fit in a packet. If not, we need to split it up.
159 // we are splitting up the command here so that the retry logic will not try to
160 // resend the ENTIRE command (i.e. multiple packets work of data) and only do the retry
161 // one wrapper command at a time.
162 packet->reset();
163
164 NetCommandRef *tempref = NEW_NETCOMMANDREF(msg);
165
166 Bool msgFits = packet->addCommand(tempref);
167 tempref->deleteInstance(); // delete the temporary reference.
168 tempref = NULL;
169
170 if (!msgFits) {
171 NetCommandRef *origref = NEW_NETCOMMANDREF(msg);
172 origref->setRelay(relay);
173 // the message doesn't fit in a single packet, need to split it up.
174 NetPacketList packetList = NetPacket::ConstructBigCommandPacketList(origref);
175 NetPacketListIter tempPacketPtr = packetList.begin();
176
177 while (tempPacketPtr != packetList.end()) {
178 NetPacket *tempPacket = (*tempPacketPtr);
179
180 NetCommandList *list = tempPacket->getCommandList();
181 NetCommandRef *ref1 = list->getFirstMessage();
182 while (ref1 != NULL) {
183 NetCommandRef *ref2 = m_netCommandList->addMessage(ref1->getCommand());
184 ref2->setRelay(relay);
185
186 ref1 = ref1->getNext();
187 }
188
189 tempPacket->deleteInstance();
190 tempPacket = NULL;
191 ++tempPacketPtr;
192
193 list->deleteInstance();
194 list = NULL;
195 }
196
197 origref->deleteInstance();
198 origref = NULL;
199
200 return;
201 }
202
203 // the message fits in a packet, add to the command list normally.
204 NetCommandRef *ref = m_netCommandList->addMessage(msg);
205
206 if (ref != NULL) {
207
208/*
209#if ((defined(_DEBUG)) || (defined(_INTERNAL)))
210 if (msg->getNetCommandType() == NETCOMMANDTYPE_GAMECOMMAND) {
211 DEBUG_LOG(("Connection::sendNetCommandMsg - added game command %d to net command list for frame %d.\n",
212 msg->getID(), msg->getExecutionFrame()));
213 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_FRAMEINFO) {
214 DEBUG_LOG(("Connection::sendNetCommandMsg - added frame info for frame %d\n", msg->getExecutionFrame()));
215 }
216#endif // _DEBUG || _INTERNAL
217*/
218
219 ref->setRelay(relay);
220 }
221 }
222}

262UnsignedInt Connection::doSend() {
263 Int numpackets = 0;
264 time_t curtime = timeGetTime();
265 Bool couldQueue = TRUE;
266
267 // Do this check first, since it's an important fail-safe
268 if (m_isQuitting && curtime > m_quitTime + MaxQuitFlushTime)
269 {
270 DEBUG_LOG(("Timed out a quitting connection. Deleting all %d messages\n", m_netCommandList->length()));
271 m_netCommandList->reset();
272 return 0;
273 }
274
275 if ((curtime - m_lastTimeSent) < m_frameGrouping) {
276// DEBUG_LOG(("not sending packet, time = %d, m_lastFrameSent = %d, m_frameGrouping = %d\n", curtime, m_lastTimeSent, m_frameGrouping));
277 return 0;
278 }
279
280 // iterate through all the messages and put them into a packet(s).
281 NetCommandRef *msg = m_netCommandList->getFirstMessage();
282
283 while ((msg != NULL) && couldQueue) {
284 NetPacket *packet = newInstance(NetPacket);
285 packet->init();
286 packet->setAddress(m_user->GetIPAddr(), m_user->GetPort());
287
288 Bool notDone = TRUE;
289
290 // add the command messages until either we run out of messages or the packet is full.
291 while ((msg != NULL) && notDone) {
292 NetCommandRef *next = msg->getNext(); // Need this since msg could be deleted
293
294 time_t timeLastSent = msg->getTimeLastSent();
295
296 if (((curtime - timeLastSent) > m_retryTime) || (timeLastSent == -1)) {
297 notDone = packet->addCommand(msg);
298 if (notDone) {
299 // the msg command was added to the packet.
300 if (CommandRequiresAck(msg->getCommand())) {
301 if (timeLastSent != -1) {
302 ++m_numRetries;
303 }
304 doRetryMetrics();
305 msg->setTimeLastSent(curtime);
306 } else {
307 m_netCommandList->removeMessage(msg);
308 msg->deleteInstance();
309 }
310 }
311 }
312 msg = next;
313 }
314
315 if (msg != NULL) {
316 DEBUG_LOG(("didn't finish sending all commands in connection\n"));
317 }
318
319 ++numpackets;
320
322 if (packet->getNumCommands() > 0) {
323 // If the packet actually has any information to give, give it to the transport object
324 // for transmission.
325 couldQueue = m_transport->queueSend(packet->getAddr(), packet->getPort(), packet->getData(), packet->getLength());
326 m_lastTimeSent = curtime;
327 }
328 if (packet != NULL) {
329 packet->deleteInstance(); // delete the packet now that we're done with it.
330 }
331 }
332
333 return numpackets;
334}

362NetCommandRef * Connection::processAck(UnsignedShort commandID, UnsignedByte originalPlayerID) {
363 NetCommandRef *temp = m_netCommandList->getFirstMessage();
364 while ((temp != NULL) && ((temp->getCommand()->getID() != commandID) || (temp->getCommand()->getPlayerID() != originalPlayerID))) {
365
366 // cycle through the commands till we find the one we need to remove.
367 // Need to check for both the command ID and the player ID.
368 temp = temp->getNext();
369 }
370 if (temp == NULL) {
371 return NULL;
372 }
373
374#if defined(_DEBUG) || defined(_INTERNAL)
375 Bool doDebug = FALSE;
376 if (temp->getCommand()->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTFRAME) {
377 doDebug = TRUE;
378 }
379#endif
380
381 Int index = temp->getCommand()->getID() % CONNECTION_LATENCY_HISTORY_LENGTH;
382 m_averageLatency -= ((Real)(m_latencies[index])) / CONNECTION_LATENCY_HISTORY_LENGTH;
383 Real lat = timeGetTime() - temp->getTimeLastSent();
384 m_averageLatency += lat / CONNECTION_LATENCY_HISTORY_LENGTH;
385 m_latencies[index] = lat;
386
387#if defined(_DEBUG) || defined(_INTERNAL)
388 if (doDebug == TRUE) {
389 DEBUG_LOG(("Connection::processAck - disconnect frame command %d found, removing from command list.\n", commandID));
390 }
391#endif
392 m_netCommandList->removeMessage(temp);
393 return temp;
394}
