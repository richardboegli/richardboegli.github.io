189Bool SpectreGunshipUpdate::initiateIntentToDoSpecialPower(const SpecialPowerTemplate *specialPowerTemplate, const Object *targetObj, const Coord3D *targetPos, const Waypoint *way, UnsignedInt commandOptions )
190{
191 const SpectreGunshipUpdateModuleData *data = getSpectreGunshipUpdateModuleData();
192
193 if( m_specialPowerModule->getSpecialPowerTemplate() != specialPowerTemplate )
194 {
195 //Check to make sure our modules are connected.
196 return FALSE;
197 }
198
199 if( !BitTest( commandOptions, COMMAND_FIRED_BY_SCRIPT ) )
200 {
201 m_initialTargetPosition.set( targetPos );
202 m_overrideTargetDestination.set( targetPos );
203 m_gattlingTargetPosition.set( targetPos );
204 }
205 else
206 {
207 UnsignedInt now = TheGameLogic->getFrame();
208 m_specialPowerModule->setReadyFrame( now );
209 m_initialTargetPosition.set( targetPos );
210 setLogicalStatus( GUNSHIP_STATUS_INSERTING );
211 }
212
213
214 Object * gunShip = getObject();
215
216
217 if ( gunShip )
218 {
219 // MAKE TWEAKS TO AI SO SHIP MOVES PRETTY
220 AIUpdateInterface *shipAI = gunShip->getAIUpdateInterface();
221 if ( shipAI)
222 {
223 shipAI->chooseLocomotorSet( LOCOMOTORSET_PANIC );
224 shipAI->getCurLocomotor()->setAllowInvalidPosition(TRUE);
225 shipAI->getCurLocomotor()->setUltraAccurate(TRUE); // set ultra-accurate just so AI won't try to adjust our dest
226 }
227
228 Drawable *draw = gunShip->getDrawable();
229 if ( draw )
230 draw->clearAndSetModelConditionState( MODELCONDITION_DOOR_1_OPENING, MODELCONDITION_DOOR_1_CLOSING );
231 friend_enableAfterburners(TRUE);
232
233 setLogicalStatus( GUNSHIP_STATUS_INSERTING ); // The gunship is en route to the tharget area, from map-edge
234
235
236 // TELL THE GUNNERS ABOARD THE GUNSHIP TO HOLD THEIR FIRE UNTIL ORBIT INSERTION
237 ContainModuleInterface *shipContain = gunShip->getContain();
238 if ( shipContain )
239 {
240
241 Object *newGattling = TheGameLogic->findObjectByID( m_gattlingID );
242 const ThingTemplate *gattlingTemplate = TheThingFactory->findTemplate( data->m_gattlingTemplateName );
243 if( newGattling != NULL )
244 {
245 m_gattlingID = INVALID_ID;
246 newGattling = NULL;
247 }
248 if ( gattlingTemplate )
249 {
250 newGattling = TheThingFactory->newObject( gattlingTemplate, getObject()->getTeam() );
251 DEBUG_ASSERTCRASH( gunShip, ("SpecterGunshipUpdate failed to find or create a GATTLING object"));
252 shipContain->addToContain( newGattling );
253 m_gattlingID = newGattling->getID();
254 }
255
256
257 Object *gattling = TheGameLogic->findObjectByID( m_gattlingID );
258 if ( gattling )
259 gattling->setDisabled ( DISABLED_PARALYZED );
260
261 }
262
263 }
264
265
266 data->m_attackAreaDecalTemplate.createRadiusDecal( *getObject()->getPosition(), data->m_attackAreaRadius, getObject()->getControllingPlayer(), m_attackAreaDecal);
267 data->m_targetingReticleDecalTemplate.createRadiusDecal( *getObject()->getPosition(), data->m_targetingReticleRadius, getObject()->getControllingPlayer(), m_targetingReticleDecal);
268
269
270#if defined TRACKERS
271 data->m_targetingReticleDecalTemplate.createRadiusDecal( *getObject()->getPosition(), data->m_targetingReticleRadius, getObject()->getControllingPlayer(), m_howitzerTrackerDecal);
272#endif
273
274
275 SpecialPowerModuleInterface *spmInterface = getObject()->getSpecialPowerModule( specialPowerTemplate );
276 if( spmInterface )
277 {
278 SpecialPowerModule *spModule = (SpecialPowerModule*)spmInterface;
279 spModule->markSpecialPowerTriggered( &m_initialTargetPosition );
280 }
281
282 return TRUE;
283}

323class PartitionFilterLiveMapEnemies : public PartitionFilter
324{
325private:
326 const Object *m_obj;
327public:
328 PartitionFilterLiveMapEnemies(const Object *obj) : m_obj(obj) { }
329
330 virtual Bool allow(Object *objOther)
331 {
332 // this is way fast (bit test) so do it first.
333 if (objOther->isEffectivelyDead())
334 return false;
335
336 // this is also way fast (bit test) so do it next.
337 if (objOther->isOffMap() != m_obj->isOffMap())
338 return false;
339
340 Relationship r = m_obj->getRelationship(objOther);
341 if (r != ENEMIES)
342 return false;
343
344 return true;
345 }
346
347#if defined(_DEBUG) || defined(_INTERNAL)
348 virtual const char* debugGetName() { return "PartitionFilterLiveMapEnemies"; }
349#endif
350};

360UpdateSleepTime SpectreGunshipUpdate::update()
361{
362 const SpectreGunshipUpdateModuleData *data = getSpectreGunshipUpdateModuleData();
363
364 Object *gunship = getObject();
365 if ( gunship )
366 {
367
368
369 if ( gunship->isEffectivelyDead() )
370 return UPDATE_SLEEP_FOREVER;
371
372
373
374
375 m_attackAreaDecal.update();
376 m_targetingReticleDecal.update();
377#if defined TRACKERS
378 m_howitzerTrackerDecal.update();
379#endif
380
381 AIUpdateInterface *shipAI = gunship->getAIUpdateInterface();
382 AIUpdateInterface *gattlingAI = NULL;
383
384 Object *gattling = TheGameLogic->findObjectByID( m_gattlingID );
385 if ( gattling )
386 {
387 gattlingAI = gattling->getAIUpdateInterface();
388 }
389
390
391
392 if ( m_status == GUNSHIP_STATUS_INSERTING || m_status == GUNSHIP_STATUS_ORBITING )
393 {
394
395
396 // init'l apogee
397 // target *<-------->*
398 // posi ^\ /
399 // | \ /
400 // perigee | \ /
401 // | */ declination
402 // | / \
403 // v/ * m_satelliteposition
404 // *
405 // |
406 // |
407 // |
408 // |
409 // |
410 // * gunship->getPosition()
411
412 //perigee is the point in the orbital arc nearest the satellite being captured
413 Coord3D perigee = *gunship->getPosition();
414 perigee.sub( &m_initialTargetPosition );
415 perigee.z = zero;
416 Real distanceToTarget = perigee.length();
417 perigee.normalize();
418
419 //apogee is the anteclockwise point fathest from the perigee line
420 Coord3D apogee;
421 apogee.z = zero;
422 apogee.x = -perigee.y;
423 apogee.y = perigee.x;
424
425 // declination intersects line [p][a], given an attack slope of n1/n2
426 Coord3D declination;
427 Real n1 = min( ORBIT_INSERTION_SLOPE_MAX, max(ORBIT_INSERTION_SLOPE_MIN, data->m_orbitInsertionSlope) );
428 Real n2 = ONE - n1;
429 declination.z = zero;
430 declination.x = ( perigee.x * n1 ) + ( apogee.x * n2 );
431 declination.y = ( perigee.y * n1 ) + ( apogee.y * n2 );
432
433 //scale out to the orbital radius
434 Real orbitalRadius = data->m_gunshipOrbitRadius;
435 declination.x *= orbitalRadius;
436 declination.y *= orbitalRadius;
437
438 m_satellitePosition.x = m_initialTargetPosition.x + declination.x;
439 m_satellitePosition.y = m_initialTargetPosition.y + declination.y;
440
441 if ( shipAI)
442 {
443 shipAI->aiMoveToPosition( &m_satellitePosition, CMD_FROM_AI );
444 }
445
446 Real constraintRadius = data->m_attackAreaRadius - data->m_targetingReticleRadius;
447
448 //Constrain Target Override to the targeting radius
449 Coord3D overrideTargetDelta = m_initialTargetPosition;
450 overrideTargetDelta.sub( &m_overrideTargetDestination );
451 if ( overrideTargetDelta.length() > constraintRadius )
452 {
453 overrideTargetDelta.normalize();
454 overrideTargetDelta.x *= constraintRadius;
455 overrideTargetDelta.y *= constraintRadius;
456
457 m_overrideTargetDestination.x = m_initialTargetPosition.x - overrideTargetDelta.x;
458 m_overrideTargetDestination.y = m_initialTargetPosition.y - overrideTargetDelta.y;
459
460 }
461
462 m_attackAreaDecal.setPosition( m_initialTargetPosition );
463 m_targetingReticleDecal.setPosition( m_overrideTargetDestination );
464
465
466#if defined TRACKERS
467 m_howitzerTrackerDecal.setPosition( m_gattlingTargetPosition );
468#endif
469
470
471 if ( (m_status == GUNSHIP_STATUS_INSERTING) && (distanceToTarget < orbitalRadius ) )// close enough to shoot
472 {
473 setLogicalStatus( GUNSHIP_STATUS_ORBITING );
474 m_orbitEscapeFrame = TheGameLogic->getFrame() + data->m_orbitFrames;
475
476 Object *gattling = TheGameLogic->findObjectByID( m_gattlingID );
477 if ( gattling )
478 gattling->clearDisabled ( DISABLED_PARALYZED );
479
480 AIUpdateInterface *shipAI = gunship->getAIUpdateInterface();
481 if ( shipAI)
482 {
483 shipAI->chooseLocomotorSet( LOCOMOTORSET_NORMAL );
484 shipAI->getCurLocomotor()->setAllowInvalidPosition(TRUE);
485 shipAI->getCurLocomotor()->setUltraAccurate(TRUE); // set ultra-accurate just so AI won't try to adjust our dest
486 }
487
488 Drawable *draw = gunship->getDrawable();
489 if ( draw )
490 draw->clearAndSetModelConditionState( MODELCONDITION_DOOR_1_CLOSING, MODELCONDITION_DOOR_1_OPENING );
491 friend_enableAfterburners(FALSE);
492
493 }
494
495 } // endif status == ORBITING || INSERTING
496
497
498 if ( m_status == GUNSHIP_STATUS_ORBITING )
499 {
500 Object *validTargetObject = NULL;
501
502
503 if ( TheGameLogic->getFrame() >= m_orbitEscapeFrame )
504 {
505 cleanUp();
506 setLogicalStatus( GUNSHIP_STATUS_DEPARTING );
507
508 // CEASE FIRE, RETURN TO BASE
509 disengageAndDepartAO( gunship );
510
511
512 }//endif escapeframe
513 else
514 {
515
516 // ONLY EVERY FEW FRAMES DO WE RE_EVALUATE THE TARGET OBJECT
517 if ( TheGameLogic->getFrame() %data->m_howitzerFiringRate < ONE )
518 {
519
520 m_positionToShootAt = m_overrideTargetDestination; // unless we get a hit, below
521
522 PartitionFilterLiveMapEnemies filterObvious( gunship );
523 PartitionFilterStealthedAndUndetected filterStealth( gunship, false );
524 PartitionFilterPossibleToAttack filterAttack(ATTACK_NEW_TARGET, gunship, CMD_FROM_AI);
525 PartitionFilterFreeOfFog filterFogged( gunship->getControllingPlayer()->getPlayerIndex() );
526 PartitionFilter *filters[6];
527 Int numFilters = 0;
528 filters[numFilters++] = &filterObvious;
529 filters[numFilters++] = &filterStealth;
530 filters[numFilters++] = &filterAttack;
531 filters[numFilters++] = &filterFogged;
532 filters[numFilters] = NULL;
533
534
535
536 // THIS WILL FIND A VALID TARGET WITHIN THE TARGETING RETICLE
537 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange(&m_overrideTargetDestination,
538 data->m_targetingReticleRadius,
539 FROM_BOUNDINGSPHERE_2D,
540 filters,
541 ITER_SORTED_NEAR_TO_FAR);
542 MemoryPoolObjectHolder holder(iter);
543 for (Object *theEnemy = iter->first(); theEnemy; theEnemy = iter->next())
544 {
545 if ( theEnemy && isFairDistanceFromShip( theEnemy ) )
546 {
547 validTargetObject = theEnemy;
548 break;
549 }
550 }
551
552
553
554 // WE WANT THE WIDE_RANGE AUTOACQUIRE POWER DISABLED FOR HUMAN PLAYERS
555 // SO THAT THE SPECTREGUNSHIP REQUIRES BABYSITTING AT ALL TIMES
556 if (gunship->getControllingPlayer()->getPlayerType() != PLAYER_HUMAN )
557 {
558 if ( ! validTargetObject )
559 {
560 // set a flag to start the targeting decal fading, since there is nothing to kill there
561 // THIS WILL FIND A VALID TARGET ANYWHERE INSIDE THE TARGETING AREA (THE BIG CIRCLE)
562 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange(&m_initialTargetPosition,
563 data->m_attackAreaRadius,
564 FROM_BOUNDINGSPHERE_2D,
565 filters,
566 ITER_SORTED_NEAR_TO_FAR);
567 MemoryPoolObjectHolder holder(iter);
568 for (Object *theEnemy = iter->first(); theEnemy; theEnemy = iter->next())
569 {
570 if ( theEnemy && isFairDistanceFromShip( theEnemy ) )
571 {
572 // WE GOT A HIT!!!! SHOOT HIM!
573 validTargetObject = theEnemy;
574 m_positionToShootAt = *validTargetObject->getPosition();
575
576 break;
577 }
578 }
579 }
580 }
581
582
583
584 //lets keep a constant barrage of gattling bullets on the current aim location
585 if( gattlingAI )
586 {
587 if ( validTargetObject )// either in the reticle or the targeting area
588 gattlingAI->aiAttackObject( validTargetObject, LOTS_OF_SHOTS, CMD_FROM_AI );
589 else
590 gattlingAI->aiAttackPosition( &m_gattlingTargetPosition, LOTS_OF_SHOTS, CMD_FROM_AI );
591 }
592
593
594
595
596 // IF THE GATTLING GUN HAS BEEN PLINKING THE TARGET LONG ENOUGH, LETS FOLLOW WITH A VOLLEY OF HOWITZER FIRE
597 if ( m_okToFireHowitzerCounter > data->m_howitzerFollowLag )
598 {
599 WeaponTemplate *wt = data->m_howitzerWeaponTemplate;
600 if( wt )
601 {
602 Coord3D attackPositionWithRandomOffset;
603 Real offs = data->m_randomOffsetForHowitzer;
604 attackPositionWithRandomOffset.x = m_gattlingTargetPosition.x + GameLogicRandomValue( -offs, offs );
605 attackPositionWithRandomOffset.y = m_gattlingTargetPosition.y + GameLogicRandomValue( -offs, offs );
606 attackPositionWithRandomOffset.z = m_gattlingTargetPosition.z;
607 TheWeaponStore->createAndFireTempWeapon( wt, gunship, &attackPositionWithRandomOffset );
608
609 m_howitzerFireSound.setObjectID(gunship->getID());
610 TheAudio->addAudioEvent( &m_howitzerFireSound );
611
612 }
613 }
614
615
616 }//endif frame modulator
617
618
619
620
621
622 // GATTLING TARGETING LOGIC------------------------------------------
623 const ParticleSystemTemplate *tmp = data->m_gattlingStrafeFXParticleSystem;
624 if (tmp && gattling && gattling->testStatus( OBJECT_STATUS_IS_FIRING_WEAPON) )
625 {
626
627
628
629 // I am going to wind my gattling gun toward the next good spot,
630 //whether an object's position or a cursor position
631
632
633 Coord3D delta = m_positionToShootAt;
634 delta.sub( &m_gattlingTargetPosition );
635 Real dist = delta.length();
636 if ( dist < data->m_strafingIncrement )
637 {
638 m_gattlingTargetPosition = m_positionToShootAt;
639 ++m_okToFireHowitzerCounter;
640 }
641 else
642 {
643 m_okToFireHowitzerCounter = ZERO;
644 delta.normalize();
645 delta.scale( data->m_strafingIncrement );
646 m_gattlingTargetPosition.add( &delta );
647 }
648
649
650 const Player *localPlayer = ThePlayerList->getLocalPlayer();
651
652 //Make sure the gunship is visible to the player before drawing effects.
653 if ( gunship->getShroudedStatus( localPlayer->getPlayerIndex() ) <= OBJECTSHROUD_PARTIAL_CLEAR )
654 {
655
656 // This makes the client smoke effects of the gattling cannon strafing the ground toward the attack position
657 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem(tmp);
658 if (sys)
659 {
660 Coord3D impactPosition;
661 impactPosition.x = m_gattlingTargetPosition.x + GameClientRandomValueReal( -5.0f, 5.0f );
662 impactPosition.y = m_gattlingTargetPosition.y + GameClientRandomValueReal( -5.0f, 5.0f );
663 impactPosition.z = TheTerrainLogic->getGroundHeight( impactPosition.x, impactPosition.y );
664 sys->setPosition( &impactPosition );
665
666 }
667 }
668
669 }
670 }// end else
671
672
673 }//not orbiting
674 else if ( m_status == GUNSHIP_STATUS_DEPARTING )
675 {
676 if ( isPointOffMap( *gunship->getPosition() ) )
677 {
678
679 TheGameLogic->destroyObject( gunship );
680 setLogicalStatus( GUNSHIP_STATUS_IDLE );
