219static UnsignedByte grabUByte(const char *s)
220{
221 char tmp[5] = "0xff";
222 tmp[2] = s[0];
223 tmp[3] = s[1];
224 UnsignedByte b = strtol(tmp, NULL, 16);
225 return b;
226}

228static void updateNumPlayersOnline(void)
229{
230 GameWindow *playersOnlineWindow = TheWindowManager->winGetWindowFromId(
231 NULL, NAMEKEY("WOLWelcomeMenu.wnd:StaticTextNumPlayersOnline") );
232
233 if (playersOnlineWindow)
234 {
235 UnicodeString valStr;
236 valStr.format(TheGameText->fetch("GUI:NumPlayersOnline"), lastNumPlayersOnline);
237 GadgetStaticTextSetText(playersOnlineWindow, valStr);
238 }
239
240 if (listboxInfo && TheGameSpyInfo)
241 {
242 GadgetListBoxReset(listboxInfo);
243 AsciiString aLine;
244 UnicodeString line;
245 AsciiString aMotd = TheGameSpyInfo->getMOTD();
246 UnicodeString headingStr;
247 //Kris: Patch 1.01 - November 12, 2003
248 //Removed number of players from string, and removed the argument. The number is incorrect anyways...
249 //This was a Harvard initiated fix.
250 headingStr.format(TheGameText->fetch("MOTD:NumPlayersHeading"));
251
252 while (headingStr.nextToken(&line, UnicodeString(L"\n")))
253 {
254 if (line.getCharAt(line.getLength()-1) == '\r')
255 line.removeLastChar(); // there is a trailing '\r'
256
257 line.trim();
258
259 if (line.isEmpty())
260 {
261 line = UnicodeString(L" ");
262 }
263
264 GadgetListBoxAddEntryText(listboxInfo, line, GameSpyColor[GSCOLOR_MOTD_HEADING], -1, -1);
265 }
266 GadgetListBoxAddEntryText(listboxInfo, UnicodeString(L" "), GameSpyColor[GSCOLOR_MOTD_HEADING], -1, -1);
267
268 while (aMotd.nextToken(&aLine, "\n"))
269 {
270 if (aLine.getCharAt(aLine.getLength()-1) == '\r')
271 aLine.removeLastChar(); // there is a trailing '\r'
272
273 aLine.trim();
274
275 if (aLine.isEmpty())
276 {
277 aLine = " ";
278 }
279
280 Color c = GameSpyColor[GSCOLOR_MOTD];
281 if (aLine.startsWith("\\\\"))
282 {
283 aLine = aLine.str()+1;
284 }
285 else if (aLine.startsWith("\\") && aLine.getLength() > 9)
286 {
287 // take out the hex value from strings starting as "\ffffffffText"
288 UnsignedByte a, r, g, b;
289 a = grabUByte(aLine.str()+1);
290 r = grabUByte(aLine.str()+3);
291 g = grabUByte(aLine.str()+5);
292 b = grabUByte(aLine.str()+7);
293 c = GameMakeColor(r, g, b, a);
294 DEBUG_LOG(("MOTD line '%s' has color %X\n", aLine.str(), c));
295 aLine = aLine.str() + 9;
296 }
297 line = UnicodeString(MultiByteToWideCharSingleLine(aLine.str()).c_str());
298
299 GadgetListBoxAddEntryText(listboxInfo, line, c, -1, -1);
300 }
301 }
302}

318static const char* FindNextNumber( const char* pStart )
319{
320 char* pNum = strchr( pStart, '\n' ); //go to next line
321 if( !pNum )
322 return pStart; //error
323
324 while( !isdigit(*pNum) )
325 ++pNum; //go to next number
326 return pNum;
327}

330//parse win/loss stats received from GameSpy
331void HandleOverallStats( const char* szHTTPStats, unsigned len )
332{
333//x DEBUG_LOG(("Parsing win percent stats:\n%s\n", szHTTPStats));
334 //find today's stats
335 const char* pToday = strstr( szHTTPStats, "Today" );
336 if( !pToday )
337 { //error
338 DEBUG_LOG(( "Unable to parse win/loss stats. Could not find 'Today' in:\n%s\n", szHTTPStats ));
339 return;
340 }
341 s_winStats.clear();
342 s_totalWinPercent = 0;
343
344 //find win/loss for each side
345 for( int i = 0; i < ThePlayerTemplateStore->getPlayerTemplateCount(); i++)
346 { //get current side (USA, GLA, etc.)
347 const PlayerTemplate* pTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(i);
348 if( !pTemplate->isPlayableSide() || pTemplate->getSide().compare("Boss") == 0 )
349 continue; //skip non-players
350
351 AsciiString side = pTemplate->getSide();
352 if( side == "America" )
353 side = "USA"; //conform to GameSpy name
354
355 //find this side
356 const char* pSide = strstr( pToday, side.str() );
357 if( pSide == NULL )
358 { //error, skip this side
359 DEBUG_LOG(( "Unable to parse win/loss stats for %s in:\n%s\n", side.str(), szHTTPStats ));
360 continue;
361 }
362
363 //Note: win% reported by GameSpy = team's wins / #games played by team
364 // we want win% = team's wins / total # games played by all teams
365 const char* pTotal = FindNextNumber(pSide);
366 const char* pWins = FindNextNumber(pTotal);
367 float percent = atof(pWins) / max(1,atof(pTotal)); //max prevents divide by zero
368 s_totalWinPercent += percent;
369
370 s_winStats.insert(std::make_pair( side, percent ));
371//x DEBUG_LOG(("Added win percent: %s, %d\n", side.str(), percent));
372 } //for i
373} //HandleOverallStats

376//called only from WOLWelcomeMenuInit to set %win stats
377static void updateOverallStats(void)
378{
379 UnicodeString percStr;
380 AsciiString wndName;
381 GameWindow* pWin;
382
383 if( s_totalWinPercent <= 0 )
384 s_totalWinPercent = 1; //prevent divide by zero
385
386 std::map<AsciiString,float>::iterator it;
387 for( it = s_winStats.begin(); it != s_winStats.end(); ++it )
388 {
389 int percent = REAL_TO_INT(100.0f * (it->second / s_totalWinPercent));
390 percStr.format( TheGameText->fetch("GUI:WinPercent"), percent );
391 wndName.format( "WOLWelcomeMenu.wnd:Percent%s", it->first.str() );
392 pWin = TheWindowManager->winGetWindowFromId( NULL, NAMEKEY(wndName) );
393 GadgetCheckBoxSetText( pWin, percStr );
394//x DEBUG_LOG(("Initialized win percent: %s -> %s %f=%s\n", wndName.str(), it->first.str(), it->second, percStr.str() ));
395 } //for
396} //updateOverallStats

606void WOLWelcomeMenuUpdate( WindowLayout * layout, void *userData)
607{
608 // We'll only be successful if we've requested to
609 if(isShuttingDown && TheShell->isAnimFinished() && TheTransitionHandler->isFinished())
610 shutdownComplete(layout);
611
612 if (raiseMessageBoxes)
613 {
614 RaiseGSMessageBox();
615 raiseMessageBoxes = FALSE;
616 }
617
618 if (TheFirewallHelper != NULL)
619 {
620 if (TheFirewallHelper->behaviorDetectionUpdate())
621 {
622 TheWritableGlobalData->m_firewallBehavior = TheFirewallHelper->getFirewallBehavior();
623
624 TheFirewallHelper->writeFirewallBehavior();
625
626 TheFirewallHelper->flagNeedToRefresh(FALSE); // 2/19/03 BGC, we're done, so we don't need to refresh the NAT anymore.
627
628 // we are now done with the firewall helper
629 delete TheFirewallHelper;
630 TheFirewallHelper = NULL;
631 }
632 }
633
634 if (TheShell->isAnimFinished() && !buttonPushed && TheGameSpyPeerMessageQueue)
635 {
636 HandleBuddyResponses();
637 HandlePersistentStorageResponses();
638
639 Int allowedMessages = TheGameSpyInfo->getMaxMessagesPerUpdate();
640 Bool sawImportantMessage = FALSE;
641 PeerResponse resp;
642 while (allowedMessages-- && !sawImportantMessage && TheGameSpyPeerMessageQueue->getResponse( resp ))
643 {
644 switch (resp.peerResponseType)
645 {
646 case PeerResponse::PEERRESPONSE_GROUPROOM:
647 {
648 GameSpyGroupRoom room;
649 room.m_groupID = resp.groupRoom.id;
650 room.m_maxWaiting = resp.groupRoom.maxWaiting;
651 room.m_name = resp.groupRoomName.c_str();
652 room.m_translatedName = UnicodeString(L"TEST");
653 room.m_numGames = resp.groupRoom.numGames;
654 room.m_numPlaying = resp.groupRoom.numPlaying;
655 room.m_numWaiting = resp.groupRoom.numWaiting;
656 TheGameSpyInfo->addGroupRoom( room );
657 if (room.m_groupID == 0)
658 {
659 enableControls( TRUE );
660 }
661 }
662 break;
663 case PeerResponse::PEERRESPONSE_JOINGROUPROOM:
664 {
665 sawImportantMessage = TRUE;
666 enableControls( TRUE );
667 if (resp.joinGroupRoom.ok)
668 {
