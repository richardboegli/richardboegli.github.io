81void SwayClientUpdate::updateSway()
82{
83 const BreezeInfo& info = TheScriptEngine->getBreezeInfo();
84 if (info.m_randomness == 0.0f)
85 {
86 m_curValue = 0;
87 }
88 Real delta = info.m_randomness * 0.5f;
89 m_curAngleLimit = info.m_intensity * GameClientRandomValueReal(1.0f-delta, 1.0f+delta);
90 m_curDelta = 2*PI/info.m_breezePeriod * GameClientRandomValueReal(1.0f-delta, 1.0f+delta);
91 m_leanAngle = info.m_lean * GameClientRandomValueReal(1.0f-delta, 1.0f+delta);
92 m_curVersion = info.m_breezeVersion;
93}

101void SwayClientUpdate::clientUpdate( void )
102{
103 if( !m_swaying )
104 return;
105
106 Drawable *draw = getDrawable();
107
108 // if breeze changes, always process the full update, even if not visible,
109 // so that things offscreen won't 'pop' when first viewed
110 const BreezeInfo& info = TheScriptEngine->getBreezeInfo();
111 if (info.m_breezeVersion != m_curVersion)
112 {
113 updateSway();
114 }
115 else
116 {
117 // Otherwise, only update visible drawables
118 if (!draw || !draw->isVisible())
119 return;
120 }
121
122 m_curValue += m_curDelta;
123 if (m_curValue > 2*PI)
124 m_curValue -= 2*PI;
125 Real cosine = Cos(m_curValue);
126
127 Real targetAngle = cosine * m_curAngleLimit + m_leanAngle;
128 Real deltaAngle = targetAngle - m_curAngle;
129
130 Matrix3D xfrm = *draw->getInstanceMatrix();
131 xfrm.In_Place_Pre_Rotate_X(-deltaAngle * info.m_directionVec.x);
132 xfrm.In_Place_Pre_Rotate_Y(deltaAngle * info.m_directionVec.y);
133 draw->setInstanceMatrix(&xfrm);
134
135 m_curAngle = targetAngle;
136
137 // burned things don't sway.
138 Object* obj = draw->getObject();
139 if( obj && obj->getStatusBits().test( OBJECT_STATUS_BURNED ) )
140 stopSway();
141
142}
