70// Pick a waypoint.
71MapObject *WaypointTool::pickWaypoint(Coord3D loc){
72 // Tight check first.
73 MapObject *pObj;
74 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
75 if (!pObj->isWaypoint()) {
76 continue;
77 }
78 Coord3D cloc = *pObj->getLocation();
79 // Check and see if we are within 1/2 cell size of the center.
80 Coord3D cpt = loc;
81 cpt.x -= cloc.x;
82 cpt.y -= cloc.y;
83 cpt.z = 0;
84 if (cpt.length() < 0.5f*MAP_XY_FACTOR) {
85 return pObj;
86 }
87 }
88 // Loose check
89 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
90 if (!pObj->isWaypoint()) {
91 continue;
92 }
93 Coord3D cloc = *pObj->getLocation();
94 // Check and see if we are within 1 & 1/2 cell size of the center.
95 Coord3D cpt = loc;
96 cpt.x -= cloc.x;
97 cpt.y -= cloc.y;
98 cpt.z = 0;
99 if (cpt.length() < 1.5f*MAP_XY_FACTOR) {
100 return pObj;
101 }
102 }
103 return NULL;
104}

108void WaypointTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
109{
110 if (m != TRACK_L) return;
111 m_downWaypointID = 0;
112 Coord3D docPt;
113 pView->viewToDocCoords(viewPt, &docPt);
114 MapObject *pObj = pickWaypoint(docPt);
115 if (pObj) {
116 pObj->setSelected(true);
117 m_downWaypointID = pObj->getWaypointID();
118 docPt = *pObj->getLocation();
119 WaypointOptions::update();
120 } else {
121 pView->snapPoint(&docPt);
122 MapObject *pNew = newInstance( MapObject)(docPt, AsciiString("*Waypoints/Waypoint"), 0, 0, NULL, NULL );
123 Int id = pDoc->getNextWaypointID();
124 AsciiString name = WaypointOptions::GenerateUniqueName(id);
125 pNew->setSelected(true);
126 pNew->setIsWaypoint();
127 pNew->setWaypointID(id);
128 pNew->setWaypointName(name);
129 pNew->getProperties()->setAsciiString(TheKey_originalOwner, AsciiString("team"));
130 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew);
131 pDoc->AddAndDoUndoable(pUndo);
132 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
133 pNew = NULL; // undoable owns it now.
134 m_downWaypointID = id;
135 WaypointOptions::update();
136 }
137 m_mouseDownPt = docPt;
138}

157void WaypointTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
158{
159 DrawObject::stopWaypointDragFeedback();
160 if (m != TRACK_L) return;
161
162 Coord3D docPt;
163 pView->viewToDocCoords(viewPt, &docPt);
164 MapObject *pObj;
165 PointerTool::clearSelection();
166 pObj = pickWaypoint(docPt);
167 if (pObj == NULL) {
168 pView->snapPoint(&docPt);
169 MapObject *pNew = newInstance( MapObject)(docPt, AsciiString("*Waypoints/Waypoint"), 0, 0, NULL, NULL );
170 Int id = pDoc->getNextWaypointID();
171 AsciiString name;
172 name.format("Waypoint %d", id);
173 pNew->setSelected(true);
174 pNew->setIsWaypoint();
175 pNew->setWaypointID(id);
176 pNew->setWaypointName(name);
177 pNew->getProperties()->setAsciiString(TheKey_originalOwner, AsciiString("team"));
178 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew);
179 pDoc->AddAndDoUndoable(pUndo);
180 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
181 pObj = pNew;
182 pNew = NULL; // undoable owns it now.
183 }
184 if (pObj)
185 {
186 pObj->setSelected(true);
187 if (m_downWaypointID && (pObj->getWaypointID() != m_downWaypointID))
188 {
189 Int waypointID = pObj->getWaypointID();
190 // If a link
191 if (pDoc->waypointLinkExists(m_downWaypointID, waypointID))
192 {
193 pDoc->removeWaypointLink(m_downWaypointID, waypointID);
194 }
195 else if (m_downWaypointID != waypointID)
196 {
197 pDoc->addWaypointLink(m_downWaypointID, waypointID);
198 }
199 MapObject *pDown = pDoc->getWaypointByID(m_downWaypointID);
200 if (pDown) {
201 pDoc->updateLinkedWaypointLabels(pDown);
202 }
203 }
204 WaypointOptions::update();
205 }
