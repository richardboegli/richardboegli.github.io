197Bool W3DBridge::load(enum BodyDamageType curDamageState)
198{
199 REF_PTR_RELEASE(m_bridgeTexture);
200 REF_PTR_RELEASE(m_leftMesh);
201 REF_PTR_RELEASE(m_sectionMesh);
202 REF_PTR_RELEASE(m_rightMesh);
203
204 Real scale, width, length;
205 char textureFile[_MAX_PATH] = "No Texture";
206 char modelName[_MAX_PATH] = "BRIDGESECTIONAL";
207
209 scale = 0.7f;
210 width = 34;
211 length = 170;
212
213 // try to find bridge in INI
214 TerrainRoadType *bridge = TheTerrainRoads->findBridge( m_templateName );
215 if (!bridge) return false;
216
217 scale = bridge->getBridgeScale();
218 switch (curDamageState) {
219 default: return false;
220
221 case BODY_PRISTINE:
222 strcpy( textureFile, bridge->getTexture().str() );
223 strcpy( modelName, bridge->getBridgeModel().str() );
224 break;
225 case BODY_DAMAGED:
226 strcpy( textureFile, bridge->getTextureDamaged().str() );
227 strcpy( modelName, bridge->getBridgeModelNameDamaged().str() );
228 break;
229 case BODY_REALLYDAMAGED:
230 strcpy( textureFile, bridge->getTextureReallyDamaged().str() );
231 strcpy( modelName, bridge->getBridgeModelNameReallyDamaged().str() );
232 break;
233 case BODY_RUBBLE:
234 strcpy( textureFile, bridge->getTextureBroken().str() );
235 strcpy( modelName, bridge->getBridgeModelNameBroken().str() );
236 break;
237 }
238
239 WW3DAssetManager *pMgr = W3DAssetManager::Get_Instance();
240 char left[_MAX_PATH];
241 char section[_MAX_PATH];
242 char right[_MAX_PATH];
243
244 strcpy(left, modelName);
245 strcat(left, ".BRIDGE_LEFT");
246 strcpy(section, modelName);
247 strcat(section, ".BRIDGE_SPAN");
248 strcpy(right, modelName);
249 strcat(right, ".BRIDGE_RIGHT");
250
251 m_bridgeTexture = pMgr->Get_Texture(textureFile, MIP_LEVELS_3);
252 m_leftMtx.Make_Identity();
253 m_rightMtx.Make_Identity();
254 m_sectionMtx.Make_Identity();
255
256 RenderObjClass *pObj = pMgr->Create_Render_Obj(modelName );
257 if (!pObj) return false;
258 Int i;
259 for (i=0; i<pObj->Get_Num_Sub_Objects(); i++) {
260 RenderObjClass *pSub = pObj->Get_Sub_Object(i);
261 Matrix3D mtx = pSub->Get_Transform();
262 if (0==strnicmp(left, pSub->Get_Name(), strlen(left))) {
263 m_leftMtx = mtx;
264 strcpy(left, pSub->Get_Name());
265 }
266 if (0==strnicmp(section, pSub->Get_Name(), strlen(section))) {
267 m_sectionMtx = mtx;
268 strcpy(section, pSub->Get_Name());
269 }
270 if (0==strnicmp(right, pSub->Get_Name(), strlen(right))) {
271 m_rightMtx = mtx;
272 strcpy(right, pSub->Get_Name());
273 }
274 REF_PTR_RELEASE(pSub);
275 //DEBUG_LOG(("Sub obj name %s\n", pSub->Get_Name()));
276 }
277
278 REF_PTR_RELEASE(pObj);
279
280 m_leftMesh = (MeshClass*)pMgr->Create_Render_Obj(left );
281 m_sectionMesh = (MeshClass*)pMgr->Create_Render_Obj(section);
282 m_rightMesh = (MeshClass*)pMgr->Create_Render_Obj(right);
283 m_scale = scale;
284
285
286 if (m_leftMesh == NULL) {
287 clearBridge();
288 return(false);
289 }
290 m_bridgeType = SECTIONAL_BRIDGE;
291
292 if (m_rightMesh == NULL || m_sectionMesh == NULL) {
293 m_bridgeType = FIXED_BRIDGE;
294 }
295
296 Int numVertex = m_leftMesh->Peek_Model()->Get_Vertex_Count();
297 Vector3 *pVert = m_leftMesh->Peek_Model()->Get_Vertex_Array();
298 m_leftMinX = FLT_MAX;
299 m_leftMaxX = -FLT_MAX;
300 m_minY = FLT_MAX;
301 m_maxY = -FLT_MAX;
302 for (i=0; i<numVertex; i++) {
303 Vector3 vert;
304 Matrix3D::Transform_Vector(m_leftMtx, pVert[i], &vert);
305 if (m_leftMinX > vert.X) m_leftMinX = vert.X;
306 if (m_minY > vert.Y) m_minY = vert.Y;
307 if (vert.X > m_leftMaxX) m_leftMaxX = vert.X;
308 if (vert.Y > m_maxY) m_maxY = vert.Y; // Note - we assume all sections are the same width, so we only do maxY for first section.
309 }
310 if (m_bridgeType == SECTIONAL_BRIDGE) {
311 numVertex = m_sectionMesh->Peek_Model()->Get_Vertex_Count();
312 pVert = m_sectionMesh->Peek_Model()->Get_Vertex_Array();
313 m_sectionMinX = FLT_MAX;
314 m_sectionMaxX = -FLT_MAX;
315 for (i=0; i<numVertex; i++) {
316 Vector3 vert;
317 Matrix3D::Transform_Vector(m_sectionMtx, pVert[i], &vert);
318 if (m_sectionMinX > vert.X) m_sectionMinX = vert.X;
319 if (vert.X > m_sectionMaxX) m_sectionMaxX = vert.X;
320 }
321
322 numVertex = m_rightMesh->Peek_Model()->Get_Vertex_Count();
323 pVert = m_rightMesh->Peek_Model()->Get_Vertex_Array();
324 m_rightMinX = FLT_MAX;
325 m_rightMaxX = -FLT_MAX;
326 for (i=0; i<numVertex; i++) {
327 Vector3 vert;
328 Matrix3D::Transform_Vector(m_rightMtx, pVert[i], &vert);
329 if (m_rightMinX > vert.X) m_rightMinX = vert.X;
330 if (vert.X > m_rightMaxX) m_rightMaxX = vert.X;
331 }
332 } else {
333 m_sectionMinX = m_leftMaxX;
334 m_sectionMaxX = m_leftMaxX;
335 m_rightMinX = m_leftMaxX;
336 m_rightMaxX = m_leftMaxX;
337 }
338 length = m_rightMaxX - m_leftMinX;
339 if (length < 1) length = 1;
340 m_length = length;
341 if (m_bridgeType == SECTIONAL_BRIDGE) {
342 Real allowableError = 0.05f*length;
343 // make sure the sections align.
344
345 if (m_leftMaxX>m_sectionMinX+allowableError) {
346 m_bridgeType = FIXED_BRIDGE;
347 }
348
349 if (m_rightMinX<m_sectionMaxX-allowableError) {
350 m_bridgeType = FIXED_BRIDGE;
351 }
352
353 }
354 return(true);
355}

363void W3DBridge::getBridgeInfo(BridgeInfo *pInfo)
364{
365
366 pInfo->from.x = m_start.X;
367 pInfo->from.y = m_start.Y;
368 pInfo->from.z = m_start.Z;
369 pInfo->to.x = m_end.X;
370 pInfo->to.y = m_end.Y;
371 pInfo->to.z = m_end.Z;
372 pInfo->bridgeWidth = (m_maxY - m_minY) *m_scale;
373
374 Vector3 vec = m_end-m_start;
375 Vector3 vecNormal(-vec.Y, vec.X, 0);
376 vecNormal.Normalize();
377
378 // From left = from + vecNormal*maxY*scale
379 pInfo->fromLeft.x = m_start.X + vecNormal.X * m_maxY * m_scale;
380 pInfo->fromLeft.y = m_start.Y + vecNormal.Y * m_maxY * m_scale;
381 pInfo->fromLeft.z = m_start.Z + vecNormal.Z * m_maxY * m_scale;
382
383 // From right = from + vecNormal*minY*scale
384 pInfo->fromRight.x = m_start.X + vecNormal.X * m_minY * m_scale;
385 pInfo->fromRight.y = m_start.Y + vecNormal.Y * m_minY * m_scale;
386 pInfo->fromRight.z = m_start.Z + vecNormal.Z * m_minY * m_scale;
387
388 // to left = to + vecNormal*maxY*scale
389 pInfo->toLeft.x = m_end.X + vecNormal.X * m_maxY * m_scale;
390 pInfo->toLeft.y = m_end.Y + vecNormal.Y * m_maxY * m_scale;
391 pInfo->toLeft.z = m_end.Z + vecNormal.Z * m_maxY * m_scale;
392
393 // to right = to + vecNormal*minY*scale
394 pInfo->toRight.x = m_end.X + vecNormal.X * m_minY * m_scale;
395 pInfo->toRight.y = m_end.Y + vecNormal.Y * m_minY * m_scale;
396 pInfo->toRight.z = m_end.Z + vecNormal.Z * m_minY * m_scale;
397
398}

402//=============================================================================
403// W3DBridge::getModelVertices
404//=============================================================================
406//=============================================================================
407Int W3DBridge::getModelVertices(VertexFormatXYZNDUV1 *destination_vb, Int curVertex, Real xOffset,
408 Vector3 &vec, Vector3 &vecNormal, Vector3 &vecZ, Vector3 &offset,
409 const Matrix3D &mtx,
410 MeshClass *pMesh, RefRenderObjListIterator *pLightsIterator)
411{
412 if (pMesh == NULL)
413 return(0);
414
415 Int i;
416 Int numVertex = pMesh->Peek_Model()->Get_Vertex_Count();
417 Vector3 *pVert = pMesh->Peek_Model()->Get_Vertex_Array();
418
419 const Vector3 *pNormal = pMesh->Peek_Model()->Get_Vertex_Normal_Array();
420
421 // If we happen to have too many bridges, stop.
422 if (curVertex+numVertex+2>= W3DBridgeBuffer::MAX_BRIDGE_VERTEX) {
423 return(0);
424 }
425
426 Vector3 lightRay[MAX_GLOBAL_LIGHTS];
427 const Coord3D *lightPos;
428
429 for (Int lightIndex=0; lightIndex < TheGlobalData->m_numGlobalLights; lightIndex++)
430 {
431 lightPos=&TheGlobalData->m_terrainLightPos[lightIndex];
432 lightRay[lightIndex].Set(-lightPos->x,-lightPos->y, -lightPos->z);
433// __asm {int 3}; //see if it really needs normalization!!
434 lightRay[lightIndex].Normalize();
435 }
436
437 const Vector2*uvs=pMesh->Peek_Model()->Get_UV_Array_By_Index(0);
438 VertexFormatXYZNDUV1 *curVb = destination_vb+curVertex;
439
440 for (i=0; i<numVertex; i++) {
441 Vector3 vLoc;
442 Vector3 vertex;
443 Matrix3D::Transform_Vector(mtx, pVert[i], &vertex);
444 vLoc = (vertex.X+xOffset) * vec + vertex.Y*vecNormal + vertex.Z*vecZ;
445
446 vLoc.X += m_start.X;
447 vLoc.Y += m_start.Y;
448 vLoc.Z += m_start.Z;
449
450 curVb->x = vLoc.X;
451 curVb->y = vLoc.Y;
452 curVb->z = vLoc.Z;
453
454 VERTEX_FORMAT vb;
455 vb.x = vLoc.X;
456 vb.y = vLoc.Y;
457 vb.z = vLoc.Z;
458
459 Vector3 normal;
460 Matrix3D::Rotate_Vector(mtx, pNormal[i], &normal);
461#ifdef USE_BRIDGE_NORMALS
462 curVb->nx = normal.X;
463 curVb->ny = normal.Y;
464 curVb->nz = normal.Z;
465 curVb->diffuse = 0xFF000000;
466#else
467 normal = (normal.X) * vec + normal.Y*vecNormal + normal.Z*vecZ;
468 normal.Normalize();
469 TheTerrainRenderObject->doTheLight(&vb, lightRay, &normal, NULL, 1.0f);
470 curVb->nx = 0; //will these to keep AGP write buffer happy.
471 curVb->ny = 0;
472 curVb->nz = 1;
473 curVb->diffuse = vb.diffuse | 0xFF000000;
474#endif
475 curVb->u1 = uvs[i].U;
476 curVb->v1 = uvs[i].V;
477 curVb++;
478 }
479 return(numVertex);
480}

482//=============================================================================
483// W3DBridge::getModelVerticesFixed
484//=============================================================================
486//=============================================================================
487Int W3DBridge::getModelVerticesFixed(VertexFormatXYZNDUV1 *destination_vb, Int curVertex,
488 const Matrix3D &mtx, MeshClass *pMesh, RefRenderObjListIterator *pLightsIterator)
489{
490 if (pMesh == NULL)
491 return(0);
492
493 Vector3 vec = m_end - m_start;
494 if (vec.Length2() < 1.0f) {
495 vec.Normalize();
496 }
497 Vector3 vecNormal(-vec.Y, vec.X, 0);
498 vecNormal.Normalize();
499 Real deltaZ = m_end.Z - m_start.Z;
500 deltaZ /= vec.Length();
501 Real deltaX = sqrt(1.0 - deltaZ*deltaZ);
502 Vector3 vecZ(-deltaZ, 0, deltaX);
503 vec /= m_length;
504 vecNormal *= m_scale;
505 vecZ *= m_scale;
506 Real xOffset = -m_leftMinX;
507 return(getModelVertices(destination_vb, curVertex, xOffset, vec, vecNormal, vecZ, m_start, mtx, pMesh, pLightsIterator));
508}

511// W3DBridge::getIndicesNVertices
512//=============================================================================
514//=============================================================================
515void W3DBridge::getIndicesNVertices(UnsignedShort *destination_ib, VertexFormatXYZNDUV1 *destination_vb,
516 Int *curIndexP, Int *curVertexP, RefRenderObjListIterator *pLightsIterator)
517{
518 Int numI;
519 Int numV;
520 m_firstVertex = *curVertexP;
521 m_firstIndex = *curIndexP;
522 m_numVertex = 0;
523 m_numPolygons = 0;
524 if (m_sectionMesh == NULL) {
525 numV = getModelVerticesFixed(destination_vb, *curVertexP, m_leftMtx, m_leftMesh, pLightsIterator);
526 if (!numV)
527 { //not enough room for vertices
528 DEBUG_ASSERTCRASH( numV, ("W3DBridge::GetIndicesNVertices(). Vertex overflow.\n") );
529 return;
530 }
531 numI = getModelIndices( destination_ib, *curIndexP, *curVertexP, m_leftMesh);
532 if (!numI)
533 { //not enough room for indices
534 DEBUG_ASSERTCRASH( numI, ("W3DBridge::GetIndicesNVertices(). Index overflow.\n") );
535 return;
536 }
537 *curIndexP += numI;
538 *curVertexP += numV;
539 m_numVertex += numV;
540 m_numPolygons += numI/3;
541 return;
542 }
543
544 Vector3 vec = m_end - m_start;
545 if (vec.Length2() < 1.0f) {
546 vec.Normalize();
547 }
548
549 Vector3 vecNormal(-vec.Y, vec.X, 0);
550 vecNormal.Normalize();
551 vecNormal *= m_scale;
552
553 // Rotate along the y axis to get the appropriate Z height adjustment.
554 Real deltaZ = m_end.Z - m_start.Z;
555 Real desiredLength = vec.Length();
556 deltaZ /= desiredLength;
557 Real deltaX = sqrt(1.0 - deltaZ*deltaZ);
558 Vector3 vecZ(-deltaZ, 0, deltaX);
559 vecZ *= m_scale;
560
561 Real spanLength = m_rightMinX - m_leftMaxX;
562 Int numSpans = 1;
563 if (m_bridgeType != FIXED_BRIDGE) {
564 Real spannable = desiredLength - (m_length-spanLength);
565 numSpans = REAL_TO_INT_FLOOR( (spannable + spanLength/2)/spanLength);
566 if (numSpans<0) numSpans = 0;
567 }
568
569 Real bridgeLength = m_length + (numSpans-1)*spanLength;
570 Real xOffset = -m_leftMinX;
571
572 // Draw the left end.
573 vec /= bridgeLength;
574 numV = getModelVertices(destination_vb, *curVertexP, xOffset, vec, vecNormal, vecZ, m_start,
575 m_leftMtx, m_leftMesh, pLightsIterator);
576 if (!numV)
577 { //not enough room for vertices
578 DEBUG_ASSERTCRASH( numV, ("W3DBridge::GetIndicesNVertices(). Vertex overflow.\n") );
579 return;
580 }
581 numI = getModelIndices( destination_ib, *curIndexP, *curVertexP, m_leftMesh);
582 if (!numI)
583 { //not enough room for indices
584 DEBUG_ASSERTCRASH( numI, ("W3DBridge::GetIndicesNVertices(). Index overflow.\n") );
585 return;
586 }
587 *curIndexP += numI;
588 *curVertexP += numV;
589 m_numVertex += numV;
590 m_numPolygons += numI/3;
591
592 Int i;
593 // draw the spans.
594 for (i=0; i<numSpans; i++) {
595 numV = getModelVertices(destination_vb, *curVertexP, xOffset+i*spanLength, vec, vecNormal, vecZ, m_start,
596 m_sectionMtx, m_sectionMesh, pLightsIterator);
597 if (!numV)
598 { //not enough room for vertices
599 DEBUG_ASSERTCRASH( numV, ("W3DBridge::GetIndicesNVertices(). Vertex overflow.\n") );
600 return;
601 }
602 numI = getModelIndices( destination_ib, *curIndexP, *curVertexP, m_sectionMesh);
603 if (!numI)
604 { //not enough room for indices
605 DEBUG_ASSERTCRASH( numI, ("W3DBridge::GetIndicesNVertices(). Index overflow.\n") );
606 return;
607 }
608 *curIndexP += numI;
609 *curVertexP += numV;
610 m_numVertex += numV;
611 m_numPolygons += numI/3;
612 }
613
614 // Draw the right end.
615 numV = getModelVertices(destination_vb, *curVertexP, xOffset+(numSpans-1)*spanLength, vec, vecNormal, vecZ, m_start,
616 m_rightMtx, m_rightMesh, pLightsIterator);
617 if (!numV)
618 { //not enough room for vertices
619 DEBUG_ASSERTCRASH( numV, ("W3DBridge::GetIndicesNVertices(). Vertex overflow.\n") );
620 return;
621 }
622 numI = getModelIndices( destination_ib, *curIndexP, *curVertexP, m_rightMesh);
623 if (!numI)
624 { //not enough room for indices
625 DEBUG_ASSERTCRASH( numI, ("W3DBridge::GetIndicesNVertices(). Index overflow.\n") );
626 return;
627 }
628 *curIndexP += numI;
629 *curVertexP += numV;
630 m_numVertex += numV;
631 m_numPolygons += numI/3;
632 return;
633}

640Int W3DBridge::getModelIndices(UnsignedShort *destination_ib, Int curIndex, Int vertexOffset, MeshClass *pMesh)
641{
642 if (pMesh == NULL)
643 return(0);
644 Int numPoly = pMesh->Peek_Model()->Get_Polygon_Count();
645 const TriIndex *pPoly =pMesh->Peek_Model()->Get_Polygon_Array();
646 if (curIndex+3*numPoly+6 >= W3DBridgeBuffer::MAX_BRIDGE_INDEX) {
647 return(0);
648 }
649 UnsignedShort *curIb = destination_ib+curIndex;

928static void updateTowerPos( RenderObjClass* tower,
929 BridgeTowerType type,
930 BridgeInfo* bridgeInfo )
931{
932
933 // sanity
934 if( tower == NULL || type < 0 || type >= BRIDGE_MAX_TOWERS || bridgeInfo == NULL )
935 return;
936
937 //
938 // compute the angle of the bridge ... we consider the angle of the bridge to be
939 // from 'from' to 'to' in the bridge info ... and so does the game
940 //
941 Coord2D v;
942 v.x = bridgeInfo->toLeft.x - bridgeInfo->fromLeft.x;
943 v.y = bridgeInfo->toLeft.y - bridgeInfo->fromLeft.y;
944 Real angle = v.toAngle();
945
946 //
947 // given the type of tower (corner position) find the appropriate spot to put the tower
948 // NOTE that we're also adjusting the angle for the from side to point the
949 // opposite way the "bridge is pointing"
950 //
951 Coord3D towerPos;
952 switch( type )
953 {
954
955 case BRIDGE_TOWER_FROM_LEFT: towerPos = bridgeInfo->fromLeft; angle += PI; break;
956 case BRIDGE_TOWER_FROM_RIGHT: towerPos = bridgeInfo->fromRight; angle += PI; break;
957 case BRIDGE_TOWER_TO_LEFT: towerPos = bridgeInfo->toLeft; break;
958 case BRIDGE_TOWER_TO_RIGHT: towerPos = bridgeInfo->toRight; break;
959 default: return;
960
961 } // end switch
962
963 // set the position of the tower render object to the position in the world
964 Matrix3D transform;
965 transform.Make_Identity();
966 transform.Set_X_Translation( towerPos.x );
967 transform.Set_Y_Translation( towerPos.y );
968 transform.Set_Z_Translation( towerPos.z );
969 transform.Rotate_Z( angle );
970 tower->Set_Transform( transform );
971
972 // set the angle for the tower
973// tower->setAngle( angle );
