108UpdateSleepTime PoisonedBehavior::update()
109{
110 const PoisonedBehaviorModuleData* d = getPoisonedBehaviorModuleData();
111 UnsignedInt now = TheGameLogic->getFrame();
112
113 if( m_poisonOverallStopFrame == 0 )
114 {
115 DEBUG_CRASH(("hmm, this should not happen"));
116 return UPDATE_SLEEP_FOREVER;
117 //we aren't poisoned, so nevermind
118 }
119
120 if (m_poisonDamageFrame != 0 && now >= m_poisonDamageFrame)
121 {
122 // If it is time to do damage, then do it and reset the damage timer
123 DamageInfo damage;
124 damage.in.m_amount = m_poisonDamageAmount;
125 damage.in.m_sourceID = INVALID_ID;
126 damage.in.m_damageType = DAMAGE_UNRESISTABLE; // Not poison, as that will infect us again
127 damage.in.m_damageFXOverride = DAMAGE_POISON; // but this will ensure that the right effect is played
128 damage.in.m_deathType = m_deathType;
129 getObject()->attemptDamage( &damage );
130
131 m_poisonDamageFrame = now + d->m_poisonDamageIntervalData;
132 }
133
134 // If we are now at zero we need to turn off our special effects...
135 // unless the poison killed us, then we continue to be a pulsating toxic pus ball
136 if( m_poisonOverallStopFrame != 0 &&
137 now >= m_poisonOverallStopFrame &&
138 !getObject()->isEffectivelyDead())
139 {
140 stopPoisonedEffects();
141 }
142
143 return calcSleepTime();
144}

148UpdateSleepTime PoisonedBehavior::calcSleepTime()
149{
150 // UPDATE_SLEEP requires a count-of-frames, not an absolute-frame, so subtract 'now'
151 UnsignedInt now = TheGameLogic->getFrame();
152 if (m_poisonOverallStopFrame == 0 || m_poisonOverallStopFrame == now)
153 return UPDATE_SLEEP_FOREVER;
154 return frameToSleepTime(m_poisonDamageFrame, m_poisonOverallStopFrame);
155}

159void PoisonedBehavior::startPoisonedEffects( const DamageInfo *damageInfo )
160{
161 const PoisonedBehaviorModuleData* d = getPoisonedBehaviorModuleData();
162 UnsignedInt now = TheGameLogic->getFrame();
163
164 // We are going to take the damage dealt by the original poisoner every so often for a while.
165 m_poisonDamageAmount = damageInfo->out.m_actualDamageDealt;
166
167 m_poisonOverallStopFrame = now + d->m_poisonDurationData;
168
169 // If we are getting re-poisoned, don't reset the damage counter if running, but do set it if unset
170 if( m_poisonDamageFrame != 0 )
171 m_poisonDamageFrame = min( m_poisonDamageFrame, now + d->m_poisonDamageIntervalData );
172 else
173 m_poisonDamageFrame = now + d->m_poisonDamageIntervalData;
174
175 m_deathType = damageInfo->in.m_deathType;
176
177 Drawable *myDrawable = getObject()->getDrawable();
178 if( myDrawable )
179 myDrawable->setTintStatus( TINT_STATUS_POISONED );// Graham, It has changed, see UpdateDrawable()
180
181 setWakeFrame(getObject(), calcSleepTime());
182}

213void PoisonedBehavior::xfer( Xfer *xfer )
