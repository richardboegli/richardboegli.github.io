70void TempWeaponBonusHelper::clearTempWeaponBonus()
71{
72 if( m_currentBonus != WEAPONBONUSCONDITION_INVALID )
73 {
74 getObject()->clearWeaponBonusCondition(m_currentBonus);
75 m_currentBonus = WEAPONBONUSCONDITION_INVALID;
76 m_frameToRemove = 0;
77
78 if( getObject()->getDrawable() )
79 {
80 getObject()->getDrawable()->clearTintStatus(TINT_STATUS_FRENZY);
81// if (getObject()->isKindOf(KINDOF_INFANTRY))
82// getObject()->getDrawable()->setSecondMaterialPassOpacity( 0.0f );
83 }
84 }
85}

88// ------------------------------------------------------------------------------------------------
89void TempWeaponBonusHelper::doTempWeaponBonus( WeaponBonusConditionType status, UnsignedInt duration )
90{
91 // Clear any different status we may have. Re-getting the same status will just reset the timer
92 if( m_currentBonus != status )
93 clearTempWeaponBonus();
94
95 getObject()->setWeaponBonusCondition(status);
96 m_currentBonus = status;
97 m_frameToRemove = TheGameLogic->getFrame() + duration;
98
99 if( getObject()->getDrawable() )
100 {
101 getObject()->getDrawable()->setTintStatus(TINT_STATUS_FRENZY);
102// if (getObject()->isKindOf(KINDOF_INFANTRY))
103// getObject()->getDrawable()->setSecondMaterialPassOpacity( 1.0f );
104 }
105
106 setWakeFrame( getObject(), UPDATE_SLEEP(duration) );
107}
