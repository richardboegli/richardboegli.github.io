175/***********************************************************************************************
176 * CDControlClass::Open_Removable_Volume -- Fetch a handle to a removable drive *
177 * *
178 * INPUT: Nothing *
179 * *
180 * OUTPUT: Nothing *
181 * *
182 * WARNINGS: None *
183 * *
184 * HISTORY: *
185 * 2/16/99 11:25PM ST : Created *
186 *=============================================================================================*/
187HANDLE CDControlClass::Open_Removable_Volume( char drive )
188{
189 assert (WinVersion.Is_WinNT());
190
191 HANDLE volume;
192 unsigned drivetype;
193 char volume_name[8];
194 char rootname[5];
195 unsigned long access_flags;
196
197 /*----------------------------------------------------------------------------------------
198 ** Get the drive type to ensure that this is a removable volume.
199 */
200 _stprintf (rootname, _TEXT( "%c:\\" ), drive + 'A');
201 drivetype = GetDriveType( rootname );
202
203 switch( drivetype ) {
204
205 case DRIVE_REMOVABLE:
206 access_flags = GENERIC_READ | GENERIC_WRITE;
207 break;
208
209 case DRIVE_CDROM:
210 access_flags = GENERIC_READ;
211 break;
212
213 default:
214// DebugString ("Attempt to open non-removable volume for locking or ejection\n");
215 Msg( __LINE__, TEXT(__FILE__), TEXT("Attempt to open non-removable volume for locking or ejection" ));
216 return( INVALID_HANDLE_VALUE );
217 }
218
219 /*----------------------------------------------------------------------------------------
220 ** Get a handle to the volume.
221 */
222 _stprintf( volume_name, _TEXT( "\\\\.\\%c:" ), drive + 'A' );
223 volume = CreateFile( volume_name, access_flags, FILE_SHARE_READ | FILE_SHARE_WRITE, NULL, OPEN_EXISTING, 0, NULL );
224
225// assert (volume != INVALID_HANDLE_VALUE);
226
227 if ( volume == INVALID_HANDLE_VALUE ) {
228// DebugString ("Unable to open drive %c: for ejection\n", drive + 'A');
229 Msg( __LINE__, TEXT(__FILE__), TEXT("Unable to open drive %c: for ejection"), drive + 'A' - 1 );
230 }
231 return( volume );
232}

264bool CDControlClass::Lock_Volume( HANDLE volume )
265{
266 assert( WinVersion.Is_WinNT( ));
267
268 unsigned long bytes_returned = 0;
269 unsigned long sleep_amount = LOCK_TIMEOUT / LOCK_RETRIES;
270
271 /*
272 ** Do this in a loop until a timeout period has expired
273 */
274 for ( int trycount = 0; trycount < LOCK_RETRIES; trycount++ ) {
275
276 if ( DeviceIoControl( volume, FSCTL_LOCK_VOLUME, NULL, 0, NULL, 0, &bytes_returned, NULL )) {
277 return( true );
278 }
279// Msg( __LINE__, TEXT(__FILE__), TEXT("DeviceIoControl failed to lock volume. Error %d - %s"), GetLastError(), Last_Error_Text());
280 Last_Error_Text( _TEXT( "DeviceIoControl failed to lock volume." ), GetLastError());
281// Sleep( sleep_amount );
282 Sleep( LOCK_TIMEOUT );
283 }
284 return( false );
285}

406/***********************************************************************************************
407 * CDControlClass::Eject_CD -- Force the CD drive to eject *
408 * *
409 * INPUT: drive number *
410 * *
411 * OUTPUT: true if ejected *
412 * *
413 * WARNINGS: None *
414 * *
415 * HISTORY: *
416 * 2/16/99 11:35PM ST : Created *
417 *=============================================================================================*/
418bool CDControlClass::Eject_CD(char drive)
419{
420 assert (WinVersion.Is_WinNT());
421 HANDLE volume;
422 bool ejected = false;
423
424 volume = Open_Removable_Volume(drive);
425
426 if (volume == INVALID_HANDLE_VALUE) return (false);
427
428 /*
429 ** Lock and dismount the volume.
430 */
431 if (Lock_Volume(volume) && Dismount_Volume(volume)) {
432
433 /*
434 ** Set prevent removal to false and eject the volume.
435 */
436 if (Prevent_Removal_Of_Volume(volume, false) && Auto_Eject_Volume(volume)) {
437 ejected = true;
438 }
439 }
440
441 /*
442 ** Close the volume so other processes can use the drive.
443 */
444 Close_Removable_Volume(volume);
445
446 return (ejected);
447}

537/***********************************************************************************************
538 * CDControlClass::Eject_CD_Win95 -- Eject the cd in the given drive *
539 * *
540 * INPUT: drive. 0=a etc. *
541 * *
542 * OUTPUT: Nothing *
543 * *
544 * WARNINGS: None *
545 * *
546 * HISTORY: *
547 * 2/17/99 1:04AM ST : Created *
548 *=============================================================================================*/
549void CDControlClass::Eject_CD_Win95 (char drive)
550{
551 assert (WinVersion.Is_Win9x());
552 HANDLE vwin32 = INVALID_HANDLE_VALUE;
553
554 drive++;
555 vwin32 = Open_VWin32 ();
556 assert (vwin32 != INVALID_HANDLE_VALUE);
557
558 /*
559 ** Make sure no other applications are using the drive.
560 */
561 bool drive_locked = Lock_Logical_Volume (vwin32, drive, 0, 0);
562// assert (drive_locked);
563 if (!drive_locked) {
564// DebugString("Unable to lock volume %c:\n", 'A' + drive - 1);
565 Msg( __LINE__, TEXT(__FILE__), TEXT("Unable to lock volume %c: "), 'A' + drive - 1 );
566 return;
567 }
568
569 /*
570 ** Make sure there is no software lock keeping the media in the drive.
571 */
572 if (!Unlock_Volume_95 (vwin32, drive)) {
573// DebugString("Could not unlock media from drive %c:\n", 'A' + drive - 1);
574 Msg( __LINE__, TEXT(__FILE__), TEXT("Could not unlock media from drive %c: "), 'A' + drive - 1 );
575 Unlock_Logical_Volume (vwin32, drive);
576 return;
577 }
578
579 /*
580 ** Eject!
581 */
582 if (!Auto_Eject_Volume_95 (vwin32, drive)) {
583// DebugString("Could not eject media from drive %c:\n", 'A' + drive - 1);
584 Msg( __LINE__, TEXT(__FILE__), TEXT("Could not eject media from drive %c: "), 'A' + drive - 1 );
585 }
586
587 Unlock_Logical_Volume (vwin32, drive);
588 Close_VWin32 (vwin32);
589}

726/***********************************************************************************************
727 * CDControlClass::Unlock_Volume_95 -- Unlocks removable media so that it can be ejected *
728 * *
729 * INPUT: Handle to VWIN32 *
730 * drive to unlock (DOS format) *
731 * *
732 * OUTPUT: true if unlocked *
733 * *
734 * WARNINGS: None *
735 * *
736 * HISTORY: *
737 * 2/17/99 0:19AM ST : Created *
738 *=============================================================================================*/
739bool CDControlClass::Unlock_Volume_95 (HANDLE vwin32, char drive)
740{
741 assert (WinVersion.Is_Win9x());
742 DIOC_REGISTERS regs = {0};
743 PARAMBLOCK unlock_params = {0};
744 bool result;
745 unsigned long cb;
746
747 /*
748 ** First, check the lock status. This way, we'll know the number of pending locks we must unlock.
749 */
750 unlock_params.bOperation = 2; // return lock/unlock status
751
752 regs.reg_EAX = 0x440D;
753 regs.reg_EBX = drive;
754 regs.reg_ECX = MAKEWORD(0x48, 0x08);
755 regs.reg_EDX = (unsigned long)&unlock_params;
756
757 result = (DeviceIoControl (vwin32, VWIN32_DIOC_DOS_IOCTL, &regs, sizeof(regs), &regs, sizeof(regs), &cb, 0)) ? true : false;
758
759 if (result) {
760
761 /*
762 ** DeviceIoControl succeeded. Now see if the unlock succeeded. It
763 ** succeeded if the carry flag is not set, or if the carry flag is
764 ** set but EAX is 0x01 or 0xB0.
765 **
766 ** It failed if the carry flag is set and EAX is not 0x01 or 0xB0.
767 **
768 ** If the carry flag is clear, then unlock succeeded. However, you
769 ** don't need to set fResult because it is already TRUE when you get
770 ** in here.
771 */
772 if (regs.reg_Flags & CARRY_FLAG) {
773 result = (regs.reg_EAX == 0xB0) || (regs.reg_EAX == 0x01);
774 }
775
776 } else {
777// DebugString ("DeviceIoControl failed to perform DOS IO control function. Error %d - %s\n", GetLastError(), Last_Error_Text());
778// Msg( __LINE__, __FILE__, "DeviceIoControl failed to perform DOS IO control function. Error %d - %s.", GetLastError(), Last_Error_Text() );
779 Last_Error_Text( TEXT("DeviceIoControl failed to perform DOS IO control function.."), GetLastError());
780 }
781
782 if (!result) return (false);
783
784 /*
785 ** Now we have to unlock the media for every time it was locked. This gets us a lock count of
786 ** 0 and totally unlocked media.
787 */
788 for (int i = 0; i < unlock_params.bNumLocks; ++i) {
789 unlock_params.bOperation = 1; // unlock the media
790 regs.reg_EAX = 0x440D;
791 regs.reg_EBX = drive;
792 regs.reg_ECX = MAKEWORD(0x48, 0x08); // LOCK/UNLOCK
793 regs.reg_EDX = (unsigned long)&unlock_params;
794
795 result = (DeviceIoControl (vwin32, VWIN32_DIOC_DOS_IOCTL, &regs, sizeof(regs), &regs, sizeof(regs), &cb, 0)) ? true : false;
796 if (result == false) {
797// DebugString ("DeviceIoControl failed to perform DOS IO control function. Error %d - %s\n", GetLastError(), Last_Error_Text());
798// Msg( __LINE__, __FILE__, "DeviceIoControl failed to perform DOS IO control function. Error %d - %s.", GetLastError(), Last_Error_Text() );
799 Last_Error_Text( TEXT("DeviceIoControl failed to perform DOS IO control function.."), GetLastError());
800 }
801
802 /*
803 ** See if DeviceIoControl and the lock succeeded
804 */
805 result = result && !(regs.reg_Flags & CARRY_FLAG);
806 if (!result) break;
807 }
808 return (result);
809}

952bool WINAPI CDControlClass::Lock_Logical_Volume (HANDLE vwin32, char drive, char lock_level, short permissions)
953{
954 assert (WinVersion.Is_Win9x());

956 DIOC_REGISTERS regs = {0};
957 char device_cat; // can be either 0x48 or 0x08

961 ** lock_level

963 ** be taken when there are no open files on the specified drive.
964 ** Levels 1 through 3 form a hierarchy where 1 must be taken before
965 ** 2, which must be taken before 3.
966 **
967 ** permissions
968 ** Specifies how the lock will affect file operations when lock levels
969 ** 1 through 3 are taken. Also specifies whether a formatting lock
970 ** should be taken after a level 0 lock.
971 */
972
973 /*
974 ** Try first with device category 0x48 for FAT32 volumes. If it
975 ** doesn't work, try again with device category 0x08. If that
976 ** doesn't work, then the lock failed.
977 */
978 device_cat = 0;
979
980 do {
981 if (device_cat == 0) {
982 device_cat = 0x48;
983 }else{
984 device_cat = 0x08;
985 }
986
987 /*
988 ** Set up the parameters for the call.
989 */
990 regs.reg_EAX = 0x440D;
991 regs.reg_EBX = MAKEWORD(drive, lock_level);
992 regs.reg_ECX = MAKEWORD(0x4A, device_cat);
993 regs.reg_EDX = permissions;
994
995 result = (DeviceIoControl (vwin32, VWIN32_DIOC_DOS_IOCTL, &regs, sizeof(regs), &regs, sizeof(regs), &cb, 0)) ? true : false;

999 Last_Error_Text( TEXT("DeviceIoControl failed to perform DOS IO control function."), GetLastError());
1000 }
1001 result = result && !(regs.reg_Flags & CARRY_FLAG);
1002
1003 } while (result == false && device_cat != 0x08);
1004
