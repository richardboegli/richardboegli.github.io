58MapObject* RoadTool::findSegment(const Coord3D *pLoc, Coord3D *outLoc)
59{
60 for (MapObject* pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext())
61 {
62 if (pMapObj->getFlag(FLAG_ROAD_POINT1))
63 {
64 MapObject* pMapObj2 = pMapObj->getNext();
65 if (pMapObj2==NULL)
66 break;
67 if (!pMapObj2->getFlag(FLAG_ROAD_POINT2))
68 continue;
69 Coord2D start, end, loc, snapLoc;
70 start.x = pMapObj->getLocation()->x;
71 start.y = pMapObj->getLocation()->y;
72 end.x = pMapObj2->getLocation()->x;
73 end.y = pMapObj2->getLocation()->y;
74 loc.x = pLoc->x;
75 loc.y = pLoc->y;
76 Real dist;
77 Real u;
78
79 ShortestDistancePointToSegment2D(&start, &end, &loc, NULL, &snapLoc, &u);
80 if (u < 0 || u > 1) {
81 continue;
82 }
83 Coord2D segment;
84 segment.x = loc.x - snapLoc.x;
85 segment.y = loc.y - snapLoc.y;
86 dist = segment.length();
87
88 if (dist < ROAD_SNAP_DISTANCE*MAP_XY_FACTOR)
89 {
90 outLoc->x = snapLoc.x;
91 outLoc->y = snapLoc.y;
92 outLoc->z = MAGIC_GROUND_Z;
93 return(pMapObj);
94 }
95 }
96 }
97 return NULL;
98}

105Bool RoadTool::snap(Coord3D *pLoc, Bool skipFirst)
106{
107 MapObject *pMapObj;
108 MapObject *pMapObj2;
109 Real snapDist = ROAD_SNAP_DISTANCE*MAP_XY_FACTOR;
110 Coord3D newLoc = *pLoc;
111 Bool snapped = false;
112
113 for (pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext()) {
114 if (skipFirst) {
115 skipFirst = false;
116 continue;
117 }
118 if (pMapObj->getFlag(FLAG_ROAD_POINT1)) {
119 pMapObj2 = pMapObj->getNext();
120 if (pMapObj2==NULL) break;
121 if (!pMapObj2->getFlag(FLAG_ROAD_POINT2)) continue;
122 Vector2 dist;
123 if (!pMapObj->isSelected()) {
124 dist.Set(pMapObj->getLocation()->x - pLoc->x, pMapObj->getLocation()->y - pLoc->y);
125 if (dist.Length() < snapDist) {
126 newLoc = *pMapObj->getLocation();
127 snapDist = dist.Length();
128 snapped = true;
129 }
130 }
131 if (!pMapObj2->isSelected()) {
132 dist.Set(pMapObj2->getLocation()->x - pLoc->x, pMapObj2->getLocation()->y - pLoc->y);
133 if (dist.Length() < snapDist) {
134 newLoc = *pMapObj2->getLocation();
135 snapDist = dist.Length();
136 snapped = true;
137 }
138 }
139 }
140 }
141 newLoc.z = MAGIC_GROUND_Z; // roads always snap to terrain.
142 if (snapped) {
143 *pLoc = newLoc;
144 }
145 return snapped;
146}

159void RoadTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
160{
161 if (m != TRACK_L) return;
162
163 Coord3D cpt;
164 pView->viewToDocCoords(viewPt, &cpt);
165
166 Coord3D loc1, loc2, loc3;
167 loc1 = cpt;
168 if (!pDoc->getCellPositionFromCoord(cpt, &loc1))
169 return;
170
171 Bool isBridge = RoadOptions::isBridge();
172
173 if (isBridge) {
174 Bool isLandmark = false;
175 const ThingTemplate *tt = TheThingFactory->findTemplate(RoadOptions::getCurRoadName());
176 if (tt) {
177 isLandmark = tt->isBridge();
178 }
179 if (isLandmark) {
180 MapObject *pNew1 = newInstance(MapObject)(loc1, RoadOptions::getCurRoadName(), 0.0f, 0, NULL, tt );
181 pNew1->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
182 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew1);
183 pNew1->setSelected(true);
184 pDoc->AddAndDoUndoable(pUndo);
185 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
186 m_mapObj = NULL;
187 return;
188 }
189 }
190
191
192 Bool snapped = false;
193 Bool divideSegment = false;
194 MapObject* pickedSegment = NULL;
195 if (!isBridge) {
196 snapped = snap(&loc1, false);
197 if (!snapped) {
198 pickedSegment = findSegment(&loc1, &loc2);
199 if (pickedSegment) {
200 snapped = true;
201 divideSegment = true;
202 loc1 = *pickedSegment->getLocation();
203 loc3 = *pickedSegment->getNext()->getLocation();
204 }
205 }
206 }
207 if (!snapped) {
208 pView->snapPoint(&loc1);
209 }
210
211 if (!divideSegment) {
212 loc2 = loc1;
213 }
214
215 loc1.z = MAGIC_GROUND_Z; // roads stick to terrain anyway.
216 loc2.z = MAGIC_GROUND_Z;
217 loc3.z = MAGIC_GROUND_Z;
218 AsciiString roadName;
219 if (divideSegment) {
220 roadName = pickedSegment->getName();
221 } else {
222 roadName = RoadOptions::getCurRoadName();
223 }
224
225 MapObject *pNew1 = newInstance(MapObject)(loc1, roadName, 0.0f, 0, NULL, NULL );
226 MapObject *pNew2 = newInstance(MapObject)(loc2, roadName, 0.0f, 0, NULL, NULL );
227 MapObject *pNew3 = NULL;
228 MapObject *pNew4 = NULL;
229 if (divideSegment) {
230 pNew3 = newInstance(MapObject)(loc2, roadName, 0.0f, 0, NULL, NULL );
231 pNew4 = newInstance(MapObject)(loc3, roadName, 0.0f, 0, NULL, NULL );
232 }
233
234 pNew1->setColor(RGB(255,255,0)); // make road endpoints yellow.
235 pNew2->setColor(RGB(255,255,0)); // make road endpoints yellow.
236 if (divideSegment) {
237 pNew3->setColor(RGB(255,255,0)); // make road endpoints yellow.
238 pNew4->setColor(RGB(255,255,0)); // make road endpoints yellow.
239 }
240
241 if (RoadOptions::isBridge()) {
242 pNew1->setFlag(FLAG_BRIDGE_POINT1);
243 pNew2->setFlag(FLAG_BRIDGE_POINT2);
244 } else {
245 pNew1->setFlag(FLAG_ROAD_POINT1);
246 pNew2->setFlag(FLAG_ROAD_POINT2);
247 if (divideSegment) {
248 pNew3->setFlag(FLAG_ROAD_POINT1);
249 pNew4->setFlag(FLAG_ROAD_POINT2);
250 }
251 }
252
253 if (divideSegment) {
254 pNew1->setFlag(pickedSegment->getFlags());
255 pNew4->setFlag(pickedSegment->getNext()->getFlags());
256 } else {
257 if (RoadOptions::isAngled()) {
258 pNew1->setFlag(FLAG_ROAD_CORNER_ANGLED);
259 pNew2->setFlag(FLAG_ROAD_CORNER_ANGLED);
260 } else if (RoadOptions::isTightCurve()) {
261 pNew1->setFlag(FLAG_ROAD_CORNER_TIGHT);
262 pNew2->setFlag(FLAG_ROAD_CORNER_TIGHT);
263 }
264 }
265
266 // Roads belong to the neutral player. :)
267 pNew1->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
268 pNew2->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
269 pNew1->setNextMap(pNew2);
270 m_mapObj = pNew2;
271
272 if (divideSegment) {
273 // Roads belong to the neutral player. :)
274 pNew3->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
275 pNew4->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
276 pNew3->setNextMap(pNew4);
277 }
278
279 if (divideSegment) {
280 PointerTool::clearSelection();
281 pickedSegment->setSelected(true);
282 DeleteObjectUndoable *pDeleteUndo = new DeleteObjectUndoable(pDoc);
283 pDoc->AddAndDoUndoable(pDeleteUndo);
284 REF_PTR_RELEASE(pDeleteUndo); // belongs to pDoc now.
285 }
286 // Clear the selection.
287 PointerTool::clearSelection();
288
289 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew1);
290 pDoc->AddAndDoUndoable(pUndo);
291 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
292 m_mapObj->setSelected(true);
293
294 if (divideSegment) {
295 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew3);
296 pDoc->AddAndDoUndoable(pUndo);
297 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
298 m_mapObj = NULL;
299 PointerTool::clearSelection();
300 pNew2->setSelected(true);
301 pNew3->setSelected(true);
302 }

306void RoadTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
307{
308 if (m != TRACK_L) return;
309
310 Coord3D loc1 ;
311 if (m_mapObj == NULL) {
312 return;
313 }
314
315 Coord3D cpt;
316 pView->viewToDocCoords(viewPt, &cpt);
317
318 loc1.x=loc1.y=loc1.z=0;
319 if (!pDoc->getCellPositionFromCoord(cpt, &loc1))
320 return;
321
322 Bool isBridge = RoadOptions::isBridge();
323 loc1.z = MAGIC_GROUND_Z;
324 Bool snapped = false;
325 if (!isBridge) {
326 snapped = snap(&loc1, false);
327 }
328 if (!snapped) {
329 pView->snapPoint(&loc1);
330 }
331 pDoc->invalObject(m_mapObj);
332 m_mapObj->setLocation(&loc1);
333 pDoc->invalObject(m_mapObj);
334}
