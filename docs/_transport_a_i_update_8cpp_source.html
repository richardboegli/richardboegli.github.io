67void TransportAIUpdate::privateAttackObject( Object *victim, Int maxShotsToFire, CommandSourceType cmdSource )
68{
69 ContainModuleInterface* contain = getObject()->getContain();
70 if( contain != NULL && contain->isPassengerAllowedToFire() )
71 {
72 // As an extension of the normal attack, I may want to tell my passengers to attack
73 // too, but only if this is a direct command. (As opposed to a passive aquire)
74 if( cmdSource == CMD_FROM_PLAYER || cmdSource == CMD_FROM_SCRIPT )
75 {
76 const ContainedItemsList *passengerList = contain->getContainedItemsList();
77 ContainedItemsList::const_iterator passengerIterator;
78 passengerIterator = passengerList->begin();
79
80 while( passengerIterator != passengerList->end() )
81 {
82 Object *passenger = *passengerIterator;
83 //Advance to the next iterator
84 passengerIterator++;
85
86 // If I am an overlord with a gattling upgrade, I do not tell it to fire if it is disabled
87 if ( passenger->isKindOf( KINDOF_PORTABLE_STRUCTURE ) )
88 {
89 if( passenger->isDisabledByType( DISABLED_HACKED )
90 || passenger->isDisabledByType( DISABLED_EMP )
91 || passenger->isDisabledByType( DISABLED_SUBDUED )
92 || passenger->isDisabledByType( DISABLED_PARALYZED) )
93 continue;
94 }
95
96 AIUpdateInterface *passengerAI = passenger->getAIUpdateInterface();
97 if( passengerAI )
98 {
99 passengerAI->aiAttackObject( victim, maxShotsToFire, cmdSource );
100 }
101 }
102 }
103 }
104
105 AIUpdateInterface::privateAttackObject( victim, maxShotsToFire, cmdSource );
106}
