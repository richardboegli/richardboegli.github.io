59void BorderTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
60{
61 if (m != TRACK_L) {
62 return;
63 }
64
65 if (m_addingNewBorder) {
66 Int count = pDoc->getNumBoundaries();
67 ICoord2D current;
68 pDoc->getBoundary(count - 1, &current);
69 Coord3D new3DPoint;
70 pView->viewToDocCoords(viewPt, &new3DPoint, false);
71
72 if (current.x < 0) {
73 current.x = 0;
74 }
75
76 if (current.y < 0) {
77 current.y = 0;
78 }
79
80 current.x = REAL_TO_INT((new3DPoint.x / MAP_XY_FACTOR) + 0.5f);
81 current.y = REAL_TO_INT((new3DPoint.y / MAP_XY_FACTOR) + 0.5f);
82 pDoc->changeBoundary(count - 1, &current);
83 return;
84 }
85
86 if (m_modifyBorderNdx >= 0) {
87 ICoord2D currentBorder;
88 pDoc->getBoundary(m_modifyBorderNdx, &currentBorder);
89
90 Coord3D new3DPoint;
91 pView->viewToDocCoords(viewPt, &new3DPoint, false);
92
93 switch (m_modificationType)
94 {
95 case MOD_TYPE_INVALID: m_modifyBorderNdx = -1; return;
96 case MOD_TYPE_UP:
97 currentBorder.y = REAL_TO_INT((new3DPoint.y / MAP_XY_FACTOR) + 0.5f);
98 break;
99 case MOD_TYPE_RIGHT:
100 currentBorder.x = REAL_TO_INT((new3DPoint.x / MAP_XY_FACTOR) + 0.5f);
101 break;
102 case MOD_TYPE_FREE:
103 currentBorder.x = REAL_TO_INT((new3DPoint.x / MAP_XY_FACTOR) + 0.5f);
104 currentBorder.y = REAL_TO_INT((new3DPoint.y / MAP_XY_FACTOR) + 0.5f);
105 break;
106 }
107
108 if (currentBorder.x < 0) {
109 currentBorder.x = 0;
110 }
111
112 if (currentBorder.y < 0) {
113 currentBorder.y = 0;
114 }
115
116
117 pDoc->changeBoundary(m_modifyBorderNdx, &currentBorder);
118 }
119}

121void BorderTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
122{
123 if (m != TRACK_L) {
124 return;
125 }
126
127 //static Coord3D zero = {0.0f, 0.0f, 0.0f};
128
129 Coord3D groundPt;
130 pView->viewToDocCoords(viewPt, &groundPt);
131 if (groundPt.length() < BOUNDARY_PICK_DISTANCE) {
132 m_addingNewBorder = true;
133
134 ICoord2D initialBoundary = { 1, 1 };
135 pDoc->addBoundary(&initialBoundary);
136 return;
137 }
138
139 Int motion;
140 pDoc->findBoundaryNear(&groundPt, BOUNDARY_PICK_DISTANCE, &m_modifyBorderNdx, &motion);
141
142 // if bottom left boundary grabbed
143 if (motion == 0)
144 {
145 // modifying the bottom left is not allowed.
146 m_modifyBorderNdx = -1;
147 }
148 // else if no boundary is near
149 else if (motion == -1)
150 {
151 // add a boundary
152 m_addingNewBorder = true;
153
154 ICoord2D initialBoundary = { 1, 1 };
155 pDoc->addBoundary(&initialBoundary);
156 }
157 else
158 {
159 m_modificationType = (ModificationType) motion;
160 }
161}
