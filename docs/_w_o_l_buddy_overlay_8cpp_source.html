203{
204 if(!TheGameSpyInfo || TheGameSpyInfo->getLocalProfileID() == 0 || !buddyControls.isInit)
205 {
206 return MSG_IGNORED;
207 }
208
209 switch( msg )
210 {
211 case GLM_RIGHT_CLICKED:
212 {
213 GameWindow *control = (GameWindow *)mData1;
214 Int controlID = control->winGetWindowId();
215
216 if( controlID == buddyControls.listboxBuddiesID )
217 {
218 RightClickStruct *rc = (RightClickStruct *)mData2;
219 WindowLayout *rcLayout;
220 if(rc->pos < 0)
221 break;
222
223 GPProfile profileID = (GPProfile)GadgetListBoxGetItemData(control, rc->pos, 0);
224 RCItemType itemType = (RCItemType)(Int)GadgetListBoxGetItemData(control, rc->pos, 1);
225 UnicodeString nick = GadgetListBoxGetText(control, rc->pos);
226
227 GadgetListBoxSetSelected(control, rc->pos);
228 if (itemType == ITEM_BUDDY)
229 rcLayout = TheWindowManager->winCreateLayout(AsciiString("Menus/RCBuddiesMenu.wnd"));
230 else if (itemType == ITEM_REQUEST)
231 rcLayout = TheWindowManager->winCreateLayout(AsciiString("Menus/RCBuddyRequestMenu.wnd"));
232 else
233 rcLayout = TheWindowManager->winCreateLayout(AsciiString("Menus/RCNonBuddiesMenu.wnd"));
234 rcMenu = rcLayout->getFirstWindow();
235 rcMenu->winGetLayout()->runInit();
236 rcMenu->winBringToTop();
237 rcMenu->winHide(FALSE);
238
239
240 ICoord2D rcSize, rcPos;
241 rcMenu->winGetSize(&rcSize.x, &rcSize.y);
242 rcPos.x = rc->mouseX;
243 rcPos.y = rc->mouseY;
244 if(rc->mouseX + rcSize.x > TheDisplay->getWidth())
245 rcPos.x = TheDisplay->getWidth() - rcSize.x;
246 if(rc->mouseY + rcSize.y > TheDisplay->getHeight())
247 rcPos.y = TheDisplay->getHeight() - rcSize.y;
248 rcMenu->winSetPosition(rcPos.x, rcPos.y);
249
250
251 GameSpyRCMenuData *rcData = NEW GameSpyRCMenuData;
252 rcData->m_id = profileID;
253 rcData->m_nick.translate(nick);
254 rcData->m_itemType = itemType;
255 setUnignoreText(rcLayout, rcData->m_nick, rcData->m_id);
256 rcMenu->winSetUserData((void *)rcData);
257 TheWindowManager->winSetLoneWindow(rcMenu);
258 }
259 else
260 return MSG_IGNORED;
261 break;
262 }
263 case GEM_EDIT_DONE:
264 {
265 GameWindow *control = (GameWindow *)mData1;
266 Int controlID = control->winGetWindowId();
267 if(controlID != buddyControls.textEntryEditID)
268 return MSG_IGNORED;
269
270 // see if someone's selected
271 Int selected = -1;
272 GadgetListBoxGetSelected(buddyControls.listboxBuddies, &selected);
273 if (selected >= 0)
274 {
275 GPProfile selectedProfile = (GPProfile)GadgetListBoxGetItemData(buddyControls.listboxBuddies, selected);
276 BuddyInfoMap *m = TheGameSpyInfo->getBuddyMap();
277 BuddyInfoMap::iterator recipIt = m->find(selectedProfile);
278 if (recipIt == m->end())
279 break;
280
281 DEBUG_LOG(("Trying to send a buddy message to %d.\n", selectedProfile));
282 if (TheGameSpyGame && TheGameSpyGame->isInGame() && TheGameSpyGame->isGameInProgress() &&
283 !ThePlayerList->getLocalPlayer()->isPlayerActive())
284 {
285 DEBUG_LOG(("I'm dead - gotta look for cheats.\n"));
286 for (Int i=0; i<MAX_SLOTS; ++i)
287 {
288 DEBUG_LOG(("Slot[%d] profile is %d\n", i, TheGameSpyGame->getGameSpySlot(i)->getProfileID()));
289 if (TheGameSpyGame->getGameSpySlot(i)->getProfileID() == selectedProfile)
290 {
291 // can't send to someone in our game if we're dead/observing. security breach and all that. no seances for you.
292 if (buddyControls.listboxChat)
293 {
294 GadgetListBoxAddEntryText( buddyControls.listboxChat, TheGameText->fetch("Buddy:CantTalkToIngameBuddy"),
295 GameSpyColor[GSCOLOR_DEFAULT], -1, -1 );
296 }
297 return MSG_HANDLED;
298 }
299 }
300 }
301
302 // read the user's input and clear the entry box
303 UnicodeString txtInput;
304 txtInput.set(GadgetTextEntryGetText( buddyControls.textEntryEdit ));
305 GadgetTextEntrySetText(buddyControls.textEntryEdit, UnicodeString::TheEmptyString);
306 txtInput.trim();
307 if (!txtInput.isEmpty())
308 {
309 // Send the message
310 BuddyRequest req;
311 req.buddyRequestType = BuddyRequest::BUDDYREQUEST_MESSAGE;
312 wcsncpy(req.arg.message.text, txtInput.str(), MAX_BUDDY_CHAT_LEN);
313 req.arg.message.text[MAX_BUDDY_CHAT_LEN-1] = 0;
314 req.arg.message.recipient = selectedProfile;
315 TheGameSpyBuddyMessageQueue->addRequest(req);
316
317 // save message for future incarnations of the buddy window
318 BuddyMessageList *messages = TheGameSpyInfo->getBuddyMessages();
319 BuddyMessage message;
320 message.m_timestamp = time(NULL);
321 message.m_senderID = TheGameSpyInfo->getLocalProfileID();
322 message.m_senderNick = TheGameSpyInfo->getLocalBaseName();
323 message.m_recipientID = selectedProfile;
324 message.m_recipientNick = recipIt->second.m_name;
325 message.m_message = UnicodeString(req.arg.message.text);
326 messages->push_back(message);
327
328 // put message on screen
329 insertChat(message);
330 }
331 }
332 else
333 {
334 // nobody selected. Prompt the user.
335 if (buddyControls.listboxChat)
336 {
337 GadgetListBoxAddEntryText( buddyControls.listboxChat, TheGameText->fetch("Buddy:SelectBuddyToChat"),
338 GameSpyColor[GSCOLOR_DEFAULT], -1, -1 );
339 }
340 }
341 break;
342 }
343 default:
344 return MSG_IGNORED;
345 }
346 return MSG_HANDLED;
347}

385void updateBuddyInfo( void )
386{
387 if (!TheGameSpyBuddyMessageQueue->isConnected())
388 {
389 GadgetListBoxReset(buddyControls.listboxBuddies);
390 return;
391 }
392
393 if (!buddyControls.isInit)
394 return;
395
396 int selected;
397 GPProfile selectedProfile = 0;
398 int visiblePos = GadgetListBoxGetTopVisibleEntry(buddyControls.listboxBuddies);
399
400 GadgetListBoxGetSelected(buddyControls.listboxBuddies, &selected);
401 if (selected >= 0)
402 selectedProfile = (GPProfile)GadgetListBoxGetItemData(buddyControls.listboxBuddies, selected);
403
404 selected = -1;
405 GadgetListBoxReset(buddyControls.listboxBuddies);
406
407 // Add buddies
408 BuddyInfoMap *buddies = TheGameSpyInfo->getBuddyMap();
409 BuddyInfoMap::iterator bIt;
410 for (bIt = buddies->begin(); bIt != buddies->end(); ++bIt)
411 {
412 BuddyInfo info = bIt->second;
413 GPProfile profileID = bIt->first;
414
415 // insert name into box
416 UnicodeString formatStr;
417 formatStr.translate(info.m_name.str());//, info.m_status, info.m_statusString.str(), info.m_locationString.str());
418 Color nameColor = (TheGameSpyInfo->isSavedIgnored(profileID)) ?
419 GameSpyColor[GSCOLOR_PLAYER_IGNORED] : GameSpyColor[GSCOLOR_PLAYER_BUDDY];
420 int index = GadgetListBoxAddEntryText(buddyControls.listboxBuddies, formatStr, nameColor, -1, -1);
421
422 // insert status into box
423 AsciiString marker;
424 marker.format("Buddy:%ls", info.m_statusString.str());
425 if (!info.m_statusString.compareNoCase(L"Offline") ||
426 !info.m_statusString.compareNoCase(L"Online") ||
427 !info.m_statusString.compareNoCase(L"Matching"))
428 {
429 formatStr = TheGameText->fetch(marker);
430 }
431 else if (!info.m_statusString.compareNoCase(L"Staging") ||
432 !info.m_statusString.compareNoCase(L"Loading") ||
433 !info.m_statusString.compareNoCase(L"Playing"))
434 {
435 formatStr.format(TheGameText->fetch(marker), info.m_locationString.str());
436 }
437 else if (!info.m_statusString.compareNoCase(L"Chatting"))
438 {
439 UnicodeString roomName;
440 GroupRoomMap::iterator gIt = TheGameSpyInfo->getGroupRoomList()->find( _wtoi(info.m_locationString.str()) );
441 if (gIt != TheGameSpyInfo->getGroupRoomList()->end())
442 {
443 AsciiString s;
444 s.format("GUI:%s", gIt->second.m_name.str());
445 roomName = TheGameText->fetch(s);
446 }
447 formatStr.format(TheGameText->fetch(marker), roomName.str());
448 }
449 else
450 {
451 formatStr = info.m_statusString;
452 }
453 GadgetListBoxAddEntryText(buddyControls.listboxBuddies, formatStr, GameSpyColor[GSCOLOR_DEFAULT], index, 1);
454 GadgetListBoxSetItemData(buddyControls.listboxBuddies, (void *)(profileID), index, 0 );
455 GadgetListBoxSetItemData(buddyControls.listboxBuddies, (void *)(ITEM_BUDDY), index, 1 );
456
457 if (profileID == selectedProfile)
458 selected = index;
459 }
460
461 // add requests
462 buddies = TheGameSpyInfo->getBuddyRequestMap();
463 for (bIt = buddies->begin(); bIt != buddies->end(); ++bIt)
464 {
465 BuddyInfo info = bIt->second;
466 GPProfile profileID = bIt->first;
467
468 // insert name into box
469 UnicodeString formatStr;
470 formatStr.translate(info.m_name.str());
471 int index = GadgetListBoxAddEntryText(buddyControls.listboxBuddies, formatStr, GameSpyColor[GSCOLOR_DEFAULT], -1, -1);
472 GadgetListBoxSetItemData(buddyControls.listboxBuddies, (void *)(profileID), index, 0 );
473
474 // insert status into box
475 formatStr = TheGameText->fetch("GUI:BuddyAddReq");
476 GadgetListBoxAddEntryText(buddyControls.listboxBuddies, formatStr, GameSpyColor[GSCOLOR_DEFAULT], index, 1);
477 GadgetListBoxSetItemData(buddyControls.listboxBuddies, (void *)(ITEM_REQUEST), index, 1 );
478
479 if (profileID == selectedProfile)
480 selected = index;
481 }
482
483
484 // select the same guy
485 if (selected >= 0)
486 {
487 GadgetListBoxSetSelected(buddyControls.listboxBuddies, selected);
488 }
489
490 // view the same spot
491 GadgetListBoxSetTopVisibleEntry(buddyControls.listboxBuddies, visiblePos);
492}

494void HandleBuddyResponses( void )
495{
496 if (TheGameSpyBuddyMessageQueue)
497 {
498 BuddyResponse resp;
499 if (TheGameSpyBuddyMessageQueue->getResponse( resp ))
500 {
501 switch (resp.buddyResponseType)
502 {
503 case BuddyResponse::BUDDYRESPONSE_LOGIN:
504 {
505 deleteNotificationBox();
506 }
507 break;
508 case BuddyResponse::BUDDYRESPONSE_DISCONNECT:
509 {
510 lastNotificationWasStatus = FALSE;
511 numOnlineInNotification = 0;
512 showNotificationBox(AsciiString::TheEmptyString, TheGameText->fetch("Buddy:MessageDisconnected"));
513 }
514 break;
515 case BuddyResponse::BUDDYRESPONSE_MESSAGE:
516 {
517 if ( !wcscmp(resp.arg.message.text, L"I have authorized your request to add me to your list") )
518 break;
519
520 if (TheGameSpyInfo->isSavedIgnored(resp.profile))
521 {
522 //DEBUG_CRASH(("Player is ignored!\n"));
523 break; // no buddy messages from ignored people
524 }
525
526 // save message for future incarnations of the buddy window
527 BuddyMessageList *messages = TheGameSpyInfo->getBuddyMessages();
528 BuddyMessage message;
529 message.m_timestamp = resp.arg.message.date;
530 message.m_senderID = resp.profile;
531 message.m_recipientID = TheGameSpyInfo->getLocalProfileID();
532 message.m_recipientNick = TheGameSpyInfo->getLocalBaseName();
533 message.m_message = resp.arg.message.text;
534 // insert status into box
535 BuddyInfoMap *m = TheGameSpyInfo->getBuddyMap();
536 BuddyInfoMap::iterator senderIt = m->find(message.m_senderID);
537 AsciiString nick;
538 if (senderIt != m->end())
539 nick = senderIt->second.m_name.str();
540 else
541 nick = resp.arg.message.nick;
542 message.m_senderNick = nick;
543 messages->push_back(message);
544
545 DEBUG_LOG(("Inserting buddy chat from '%s'/'%s'\n", nick.str(), resp.arg.message.nick));
546
547 // put message on screen
548 insertChat(message);
549
550 // play audio notification
551 AudioEventRTS buddyMsgAudio("GUIMessageReceived");
552 if( TheAudio )
553 {
554 TheAudio->addAudioEvent( &buddyMsgAudio );
555 } // end if
556
557 UnicodeString snippet = message.m_message;
558 while (snippet.getLength() > 11)
559 {
560 snippet.removeLastChar();
561 }
562 UnicodeString s;
563 s.format(TheGameText->fetch("Buddy:MessageNotification"), nick.str(), snippet.str());
564 lastNotificationWasStatus = FALSE;
565 numOnlineInNotification = 0;
566 showNotificationBox(AsciiString::TheEmptyString, s);
567 }
568 break;
569 case BuddyResponse::BUDDYRESPONSE_REQUEST:
570 {
571 // save request for future incarnations of the buddy window
572 BuddyInfoMap *m = TheGameSpyInfo->getBuddyRequestMap();
573 BuddyInfo info;
574 info.m_countryCode = resp.arg.request.countrycode;
575 info.m_email = resp.arg.request.email;
576 info.m_name = resp.arg.request.nick;
577 info.m_id = resp.profile;
578 info.m_status = (GPEnum)0;
579 info.m_statusString = resp.arg.request.text;
580 (*m)[resp.profile] = info;
581
582 // TODO: put request on screen
583 updateBuddyInfo();
584 // insert status into box
585 lastNotificationWasStatus = FALSE;
586 numOnlineInNotification = 0;
587 showNotificationBox(info.m_name, TheGameText->fetch("Buddy:AddNotification"));
588 }
589 break;
590 case BuddyResponse::BUDDYRESPONSE_STATUS:
591 {
592 BuddyInfoMap *m = TheGameSpyInfo->getBuddyMap();
593 BuddyInfoMap::const_iterator bit = m->find(resp.profile);
594 Bool seenPreviously = FALSE;
595 GPEnum oldStatus = GP_OFFLINE;
596 GPEnum newStatus = resp.arg.status.status;
597 if (bit != m->end())
598 {
599 seenPreviously = TRUE;
600 oldStatus = (*m)[resp.profile].m_status;
601 }
602 BuddyInfo info;
603 info.m_countryCode = resp.arg.status.countrycode;
604 info.m_email = resp.arg.status.email;
605 info.m_name = resp.arg.status.nick;
606 info.m_id = resp.profile;
607 info.m_status = newStatus;
608 info.m_statusString = UnicodeString(MultiByteToWideCharSingleLine(resp.arg.status.statusString).c_str());
609 info.m_locationString = UnicodeString(MultiByteToWideCharSingleLine(resp.arg.status.location).c_str());
610 (*m)[resp.profile] = info;
611
612 updateBuddyInfo();
