254//============================================================================
255// CDManager::refreshDrives
256//============================================================================
257
258void CDManager::refreshDrives( void )
259{
260 LListNode *node = m_drives.firstNode();
261
262 while ( node )
263 {
264 CDDriveInterface *drive = (CDDriveInterface *) node->item();
265 if ( drive )
266 {
267 drive->refreshInfo();
268 }
269
270 node = node->next();
271 }
272}

278
279void CDManager::destroyAllDrives( void )
280{
281 LListNode *node;
282
283 while ( (node = m_drives.firstNode() ) != NULL )
284 {
285 node->remove();
286 CDDriveInterface *drive = (CDDriveInterface *) node->item();
287 if ( drive )
288 {
289 delete drive;
290 }
291 }
