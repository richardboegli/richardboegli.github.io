62Bool PrisonDockUpdate::action( Object *docker, Object *drone )
63{
64
65 // sanity
66 if( docker == NULL )
67 return FALSE;
68
69 // if docker has no contents, do nothing and stop the dock process
70 ContainModuleInterface *contain = docker->getContain();
71 if( contain && contain->getContainCount() == 0 )
72 return FALSE;
73
74 // unload the prisoners from the docker into us
75 AIUpdateInterface *ai = docker->getAIUpdateInterface();
76 DEBUG_ASSERTCRASH( ai, ("'%s' docking with prison has no AI\n",
77 docker->getTemplate()->getName().str()) );
78 POWTruckAIUpdateInterface *powAI = ai->getPOWTruckAIUpdateInterface();
79 DEBUG_ASSERTCRASH( powAI, ("'s' docking with prison has no POW Truck AI\n",
80 docker->getTemplate()->getName().str()) );
81
82 powAI->unloadPrisonersToPrison( getObject() );
83
84 // end docking
85 return FALSE;
86
87} // end action
