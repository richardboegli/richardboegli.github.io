47/******************************************************************************
48*
49* NAME
50* DebugPrint(String, ArgList...)
51*
52* DESCRIPTION
53* Ouput debug print messages to the debugger and log file.
54*
55* INPUTS
56* String - String to output.
57* ArgList - Argument list
58*
59* RESULT
60* NONE
61*
62******************************************************************************/

64void __cdecl DebugPrint(const char* string, ...)
65 {
66 static char _buffer[1024];
67 static char _filename[512] = "";
68
69 if (string != NULL)
70 {
71 // Format string
72 va_list va;
73 va_start(va, string);
74 vsprintf(&_buffer[0], string, va);
75 va_end(va);
76
77 // Open log file
78 HANDLE file = INVALID_HANDLE_VALUE;
79
80 if (strlen(_filename) == 0)
81 {
82 char path[_MAX_PATH];
83 char drive[_MAX_DRIVE];
84 char dir[_MAX_DIR];
85
86 GetModuleFileName(GetModuleHandle(NULL), &path[0], sizeof(path));
87 _splitpath(path, drive, dir, NULL, NULL);
88 _makepath(_filename, drive, dir, debugLogName, "txt");
89
90 OutputDebugString("Creating ");
91 OutputDebugString(_filename);
92 OutputDebugString("\n");
93
94 file = CreateFile(_filename, GENERIC_WRITE, 0, NULL, CREATE_ALWAYS,
95 FILE_ATTRIBUTE_NORMAL, NULL);
96 }
97 else
98 {
99 file = CreateFile(_filename, GENERIC_WRITE, 0, NULL, OPEN_ALWAYS,
100 FILE_ATTRIBUTE_NORMAL, NULL);
101 }
102
103 // Send string to debugger
104 OutputDebugString(_buffer);
105
106 // Insert carriage return after newlines
107 int i = 0;
108
109 while (_buffer[i] != '\0')
110 {
111 if (_buffer[i] == '\n')
112 {
113 int end = strlen(_buffer);
114 assert((end + 1) <= sizeof(_buffer));
115
116 while (end >= i)
117 {
118 _buffer[end + 1] = _buffer[end];
119 end--;
120 }
121
122 _buffer[i] = '\r';
123 i++;
124 }
125
126 i++;
127 }
128
129 // Send string to log file
130 assert(file != INVALID_HANDLE_VALUE);
131
132 if (file != INVALID_HANDLE_VALUE)
133 {
134 SetFilePointer(file, 0, NULL, FILE_END);
135
136 DWORD written;
137 WriteFile(file, &_buffer[0], strlen(_buffer), &written, NULL);
138
139 CloseHandle(file);
140 }
141 }
142 }

148* PrintWin32Error
149*
150* DESCRIPTION
151* Display Win32 error message (Error retrieved from GetLastError())
152*
153* INPUTS
154* Message string
155*
156* RESULT
157* NONE
158*
159******************************************************************************/
160
161void __cdecl PrintWin32Error(const char* string, ...)
162 {
163 static char _buffer[1024];
164
165 if (string != NULL)
166 {
167 // Format string
168 va_list va;
169 va_start(va, string);
170 vsprintf(&_buffer[0], string, va);
171 va_end(va);
172
173 LPVOID lpMsgBuf;
174 FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM |
175 FORMAT_MESSAGE_IGNORE_INSERTS, NULL, GetLastError(),
176 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), (LPTSTR)&lpMsgBuf, 0, NULL);
177
178 DebugPrint("***** Win32 Error: %s\n", _buffer);
179 DebugPrint(" Reason: %s\n", (char*)lpMsgBuf);
180
181 LocalFree(lpMsgBuf);
182 }
183 }
