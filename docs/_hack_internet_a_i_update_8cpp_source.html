107{

113 {

128//-------------------------------------------------------------------------------------------------
129void HackInternetAIUpdate::aiDoCommand(const AICommandParms* parms)
130{
131 if (!isAllowedToRespondToAiCommands(parms))
132 return;
133
134 //If our hacker is currently packing up his gear, we need to prevent him
135 //from moving until completed. In order to accomplish this, we'll detect,
136 //then
137 if( getStateMachine()->getCurrentStateID() == HACK_INTERNET || getStateMachine()->getCurrentStateID() == PACKING )
138 {
139 // nuke any existing pending cmd
140 m_pendingCommand.store(*parms);
141 m_hasPendingCommand = true;
142
143 if( getStateMachine()->getCurrentStateID() == HACK_INTERNET )
144 {
145 getStateMachine()->clear();
146 setLastCommandSource( CMD_FROM_AI );
147 getStateMachine()->setState( PACKING );
148 }
149 return;
150 }
151
152 m_hasPendingCommand = false;
153 AIUpdateInterface::aiDoCommand(parms);
154}

286{

299 sound.setObjectID( owner->getID() );

301
302 Real variationFactor = ai->getPackUnpackVariationFactor();
303 Real variation = GameLogicRandomValueReal( 1.0f - variationFactor, 1.0f + variationFactor );
304 m_framesRemaining = ai->getUnpackTime() * variation; //In frames
305 owner->getDrawable()->setAnimationLoopDuration( m_framesRemaining );
306
307 return STATE_CONTINUE;

450StateReturnType HackInternetState::onEnter()
451{
452 //Go into the hack internet stance.
453 Object *owner = getMachineOwner();
454 HackInternetAIUpdate *ai = (HackInternetAIUpdate*)owner->getAIUpdateInterface();
455 if( !ai )
456 {
457 return STATE_FAILURE;
458 }

468//-------------------------------------------------------------------------------------------------
469StateReturnType HackInternetState::update()
470{
471 Object *owner = getMachineOwner();
472 HackInternetAIUpdate *ai = (HackInternetAIUpdate*)owner->getAIUpdateInterface();
473 if( !ai )
474 {
475 return STATE_FAILURE;
476 }
477
478 if( owner->isDisabledByType( DISABLED_HACKED ) )
479 {
480 //Don't hack while hacked, hehe.
481 return STATE_CONTINUE;
482 }
483
484 if( m_framesRemaining > 0 )
485 {
486 //Decrement frame counter.
487 m_framesRemaining--;
488 }
489 else
490 {
491 //We have waited the full amount of the delay, so hack some cash from the heavens!
492
493 //Add cash
494 Money *money = owner->getControllingPlayer()->getMoney();
495 if( money )
496 {
497 ExperienceTracker *xp = owner->getExperienceTracker();
498 if( xp )
499 {
500 UnsignedInt amount = 0;
501 switch( xp->getVeterancyLevel() )
502 {
503 case LEVEL_HEROIC:
504 amount = ai->getHeroicCashAmount();
505 if( amount )
506 {
507 break;
508 }
509 //If entry missing, fall through!
510 case LEVEL_ELITE:
511 amount = ai->getEliteCashAmount();
512 if( amount )
513 {
514 break;
515 }
516 //If entry missing, fall through!
517 case LEVEL_VETERAN:
518 amount = ai->getVeteranCashAmount();
519 if( amount )
520 {
521 break;
522 }
523 //If entry missing, fall through!
524 case LEVEL_REGULAR:
525 amount = ai->getRegularCashAmount();
526 if( amount )
527 {
528 break;
529 }
530 //If entry missing, fall through!
531 default:
532 amount = 1;
533 break;
534 }
535 money->deposit( amount );
536 owner->getControllingPlayer()->getScoreKeeper()->addMoneyEarned( amount );
537
538 //Grant the unit some experience for a successful hack.
539 xp->addExperiencePoints( ai->getXpPerCashUpdate() );
540
541 Bool displayMoney = TRUE;
542 if( owner->testStatus(OBJECT_STATUS_STEALTHED) )
543 {
544 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
545 if( !owner->isLocallyControlled() && !owner->testStatus(OBJECT_STATUS_DETECTED) )
546 {
547 displayMoney = FALSE;
548 }
549 }
550 if( owner->getContainedBy() && owner->getContainedBy()->testStatus(OBJECT_STATUS_STEALTHED) )
551 {
552 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
553 if( !owner->getContainedBy()->isLocallyControlled() && !owner->getContainedBy()->testStatus(OBJECT_STATUS_DETECTED) )
554 {
555 displayMoney = FALSE;
556 }
557 }
558
559 if( displayMoney )
560 {
561 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
562 //Display cash income floating over the hacker.
563 UnicodeString moneyString;
564 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), amount );
565 Coord3D pos;
566 pos.zero();
567 pos.add( owner->getPosition() );
568 pos.z += 20.0f; //add a little z to make it show up above the unit.
569
570
571 Object *internetCenter = owner->getContainedBy();
572 if ( internetCenter )
573 {
574 Real width = internetCenter->getGeometryInfo().getMajorRadius() * 0.3f;
575 Real depth = internetCenter->getGeometryInfo().getMinorRadius() * 0.3f;
576 pos.x += GameClientRandomValue(-width,width);
577 pos.y += GameClientRandomValue(-depth,depth);
578 }
579
580
581 TheInGameUI->addFloatingText( moneyString, &pos, GameMakeColor( 0, 255, 0, 255 ) );
582 }
583
584 AudioEventRTS sound = *(owner->getTemplate()->getPerUnitSound( "UnitCashPing" ));
585 sound.setObjectID( owner->getID() );
586 TheAudio->addAudioEvent( &sound );
587 }
588 }
589
590
591 //Reset timer and start a new cycle.
592 m_framesRemaining = ai->getCashUpdateDelay();
593
594 }
595
596 //This is a persistent state until told otherwise.
597 return STATE_CONTINUE;
598}
