169void TensileFormationUpdate::initLinks( void )
170{
171
172 m_linksInited = TRUE;
173
174 Object *obj = getObject();
175 if ( !obj )
176 return;
177
178 PartitionFilterTensileFormationMember tfmFilter( getObject() );
179 PartitionFilter *filters[] = { &tfmFilter, NULL };
180 SimpleObjectIterator *iter = NULL;
181 iter = ThePartitionManager->iterateObjectsInRange(getObject(), 1000.0f, FROM_BOUNDINGSPHERE_3D, filters);
182 MemoryPoolObjectHolder hold(iter);
183
184 Real closestDistance = 99999.9f;
185 Real thisDistance = 99999.9f;
186 const Coord3D *myPos = obj->getPosition();
187
188 for (Object* other = iter->first(); other; other = iter->next())
189 {
190 const Coord3D *theirPos = other->getPosition();
191 Coord3D delta = { theirPos->x - myPos->x, theirPos->y - myPos->y, theirPos->z - myPos->z };
192
193 thisDistance = delta.length();
194 if ( closestDistance > thisDistance )
195 {
196 closestDistance = thisDistance;
197
198 //push a new link
199 for ( int t = 3; t > 0; --t )
200 m_links[ t ] = m_links[ t-1 ];
201
202 m_links[ 0 ].id = other->getID();
203 m_links[ 0 ].tensor = delta;
204 }
205 }
206
207 //ICoord2D gridPos;
208 //gridPos.x = REAL_TO_INT_FLOOR(getObject()->getPosition()->x/MAP_XY_FACTOR);
209 //gridPos.y = REAL_TO_INT_FLOOR(getObject()->getPosition()->y/MAP_XY_FACTOR);
210 //TheTerrainVisual->setRawMapHeight(&gridPos, 500);
211
212 getObject()->setOrientation(GameLogicRandomValueReal(-PI,PI));
213
214}

223UpdateSleepTime TensileFormationUpdate::update( void )
224{
225
226 if ( ! m_linksInited )
227 initLinks();
228
229
230
231 if( m_enabled == FALSE )
232 {
233 //We are all going to sit here idle and do not much of anything until one of us gets hurt
234 BodyModuleInterface *body = getObject()->getBodyModule();
235 BodyDamageType bdt = body->getDamageState();
236
237 if ( bdt >= BODY_DAMAGED )
238 {
239 //THen the hurt one (maybe me) will tell every other member of the formation to enable
240 //So we will start moving
241 setEnabled( TRUE );
242
243 TheAI->pathfinder()->removeWallFromMyFootprint( getObject() ); // Undo createAWallFromMyFootprint.
244
245
246 //ALert players of disaster happening
247 if ( ! m_crackSound.isCurrentlyPlaying())
248 m_crackSound.setPlayingHandle(TheAudio->addAudioEvent( &m_crackSound ));
249
250 }
251 else
252 return UPDATE_SLEEP(30);
253 }
254
255
256 Drawable *draw = getObject()->getDrawable();
257 if ( ! draw )
258 return UPDATE_SLEEP_NONE;
259
260 if (++m_life > 300)
261 {
262 draw->clearModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_MOVING));
263 draw->clearModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_FREEFALL));
264 draw->clearModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_POST_COLLAPSE));
265
266 BodyModuleInterface *body = getObject()->getBodyModule();
267 body->setDamageState( BODY_RUBBLE );
268
269
270 TheAI->pathfinder()->createAWallFromMyFootprint( getObject() ); // Temporarily treat this object as an obstacle.
271
272 return UPDATE_SLEEP_FOREVER;
273 }
274
275
276 if ( m_life%30 == 29 )
277 propagateDislodgement( TRUE );
278
279
280 //APPLY PHYSICS===========================
281 const Coord3D *pos = getObject()->getPosition();
282
283 Coord3D normal = { 0.0f, 0.0f, 1.0f };
284 TheTerrainLogic->getGroundHeight(pos->x, pos->y, &normal); // which way does the ground slope?
285
286 Coord3D slope = { normal.x, normal.y, 0.0f};
287 Real steepness = 1.0f - normal.z;
288 slope.scale( 0.3f + steepness);
289
290 m_inertia.add( &slope );
291
292 Real friction = 0.95f;
293 m_inertia.scale( friction );
294
295
296
297 Coord3D newPos;
298 newPos.x = pos->x + m_inertia.x;//flow down the slope
299 newPos.y = pos->y + m_inertia.y;
300 newPos.z = TheTerrainLogic->getGroundHeight(newPos.x, newPos.y);//rest on surface here
301
302
303
304 Object *tree = ThePartitionManager->getClosestObject( &newPos, getObject()->getGeometryInfo().getMajorRadius(), FROM_CENTER_2D );
305 if (tree->isKindOf( KINDOF_SHRUBBERY ))
306 tree->topple( &m_inertia, m_inertia.length(), 1 );//No Bounce
307
308
309
310 //APPLY TENSORS===========================
311 Coord3D tensorSum = { 0, 0, 0 };
312 for ( int t = 0; t < 4; ++t )
313 {
314 Object *other = TheGameLogic->findObjectByID( m_links[ t ].id );
315
316 if ( other )
317 {
318 Coord3D desiredPos = *other->getPosition();
319
320 Coord3D tensor = m_links[ t ].tensor;
321
322 desiredPos.sub( &tensor );
323
324 //Coord3D desiredPos = { theirPos->x - m_links[ t ].tensor.x, theirPos->y - m_links[ t ].tensor.y, theirPos->z - m_links[ t ].tensor.z };
325
326 newPos.x = newPos.x*0.93f + desiredPos.x*0.07f;
327 newPos.y = newPos.y*0.93f + desiredPos.y*0.07f;
328 newPos.z = MIN( m_lowestSlideElevation, TheTerrainLogic->getGroundHeight(newPos.x, newPos.y) );//rest on surface here
329
330 tensor.normalize();
331 tensorSum.add( &tensor );
332
333 }
334
335 }
336
337 tensorSum.normalize();
338
339 Coord3D inertiaNormal = m_inertia;
340 inertiaNormal.normalize();
341
342
343
344
345
346
347
348
349
350 draw->setModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_POST_COLLAPSE));
351
352 if ( m_life < 200 )
353 draw->setModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_MOVING));
354 else
355 draw->clearModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_MOVING));
356
357 if ( fabs( pos->z - newPos.z ) > 0.2f && m_life < 100)
358 draw->setModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_FREEFALL));
359 else
360 draw->clearModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_FREEFALL));
361
362// {
363// RGBColor cyan = {0,1,1};
364// draw->flashAsSelected(&cyan);
365// }
366
367
368
369 //else
370 //m_motionlessCounter = 0;
371
372
373 //if ( newPos.z > 80)
374 // newPos.z = 80 + ( ( newPos.z - 80 ) * 2);
375
376
377 m_lowestSlideElevation = newPos.z;
378 getObject()->setPosition( &newPos );
379
380
381
382 return UPDATE_SLEEP_NONE;
383} // end update

400 {
401

406 if ( body )

411
