133WindowMsgHandledType ControlBarObserverSystem( GameWindow *window, UnsignedInt msg,
134 WindowMsgData mData1, WindowMsgData mData2 )
135{
136 static NameKeyType buttonCommunicator = NAMEKEY_INVALID;
137
138 switch( msg )
139 {
140 // --------------------------------------------------------------------------------------------
141 case GWM_CREATE:
142 {
143 break;
144
145 } // end create
146
147 //---------------------------------------------------------------------------------------------
148 case GBM_MOUSE_ENTERING:
149 case GBM_MOUSE_LEAVING:
150 {
151 break;
152 }
153
154 //---------------------------------------------------------------------------------------------
155 case GBM_SELECTED:
156 case GBM_SELECTED_RIGHT:
157 {
158 GameWindow *control = (GameWindow *)mData1;
159
160 Int controlID = control->winGetWindowId();
161 if( controlID == buttonCancelID)
162 {
163 TheControlBar->setObserverLookAtPlayer(NULL);
164 ObserverPlayerInfoWindow->winHide(TRUE);
165 ObserverPlayerListWindow->winHide(FALSE);
166 TheControlBar->populateObserverList();
167
168 }
169 for(Int i = 0; i <MAX_BUTTONS; ++i)
170 {
171 if( controlID == buttonPlayerID[i])
172 {
173 ObserverPlayerInfoWindow->winHide(FALSE);
174 ObserverPlayerListWindow->winHide(TRUE);
175 TheControlBar->setObserverLookAtPlayer((Player *) GadgetButtonGetData( buttonPlayer[i]));
176 if(TheControlBar->getObserverLookAtPlayer())
177 TheControlBar->populateObserverInfoWindow();
178 return MSG_HANDLED;
179 }
180 }
181
182 // if( controlID == buttonCommunicator && TheGameLogic->getGameMode() == GAME_INTERNET )
183 /*
184 {
185 popupCommunicatorLayout = TheWindowManager->winCreateLayout( AsciiString( "Menus/PopupCommunicator.wnd" ) );
186 popupCommunicatorLayout->runInit();
187 popupCommunicatorLayout->hide( FALSE );
188 popupCommunicatorLayout->bringForward();
189 }
190*/
191
192 break;
193
194 } // end button selected
195
196 //---------------------------------------------------------------------------------------------
197 default:
198 return MSG_IGNORED;
199
200 } // end switch( msg )
201
202 return MSG_HANDLED;
203
204} // end ControlBarSystem

210void ControlBar::populateObserverList( void )
211{
212 Int currentButton = 0, i;
213 if(TheRecorder->isMultiplayer())
214 {
215
216 for (i = 0; i < MAX_SLOTS; ++i)
217 {
218 AsciiString name;
219 name.format("player%d", i);
220 Player *p = ThePlayerList->findPlayerWithNameKey(TheNameKeyGenerator->nameToKey(name));
221 if(p)
222 {
223 if(p->isPlayerObserver())
224 continue;
225 DEBUG_ASSERTCRASH(currentButton < MAX_BUTTONS, ("ControlBar::populateObserverList trying to populate more buttons then we have"));
226 GadgetButtonSetData(buttonPlayer[currentButton], (void *)p);
227 GadgetButtonSetEnabledImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getEnabledImage() );
228 //GadgetButtonSetHiliteImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getHiliteImage() );
229 //GadgetButtonSetHiliteSelectedImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getPushedImage() );
230 //GadgetButtonSetDisabledImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getDisabledImage() );
231 buttonPlayer[currentButton]->winSetTooltip(p->getPlayerDisplayName());
232 buttonPlayer[currentButton]->winHide(FALSE);
233 buttonPlayer[currentButton]->winSetStatus( WIN_STATUS_USE_OVERLAY_STATES );
234
235 const GameSlot *slot = TheGameInfo->getConstSlot(currentButton);
236 Color playerColor = p->getPlayerColor();
237 Color backColor = GameMakeColor(0, 0, 0, 255);
238 staticTextPlayer[currentButton]->winSetEnabledTextColors( playerColor, backColor );
239 staticTextPlayer[currentButton]->winHide(FALSE);
240 AsciiString teamStr;
241 teamStr.format("Team:%d", slot->getTeamNumber() + 1);
242 if (slot->isAI() && slot->getTeamNumber() == -1)
243 teamStr = "Team:AI";
244
245 UnicodeString text;
246 text.format(TheGameText->fetch("CONTROLBAR:ObsPlayerLabel"), p->getPlayerDisplayName().str(),
247 TheGameText->fetch(teamStr).str());
248
249 GadgetStaticTextSetText(staticTextPlayer[currentButton], text );
250
251 ++currentButton;
252 }
253 }
254 for(currentButton; currentButton<MAX_BUTTONS; ++currentButton)
255 {
256 buttonPlayer[currentButton]->winHide(TRUE);
257 staticTextPlayer[currentButton]->winHide(TRUE);
258 }
259 }
260 else
261 {
262 for(i =0; i < MAX_PLAYER_COUNT; ++i)
263 {
264 Player *p = ThePlayerList->getNthPlayer(i);
265 if(p && !p->isPlayerObserver() && p->getPlayerType() == PLAYER_HUMAN)
266 {
267 DEBUG_ASSERTCRASH(currentButton < MAX_BUTTONS, ("ControlBar::populateObserverList trying to populate more buttons then we have"));
268 GadgetButtonSetData(buttonPlayer[currentButton], (void *)p);
269 GadgetButtonSetEnabledImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getEnabledImage() );
270 //GadgetButtonSetHiliteImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getHiliteImage() );
271 //GadgetButtonSetHiliteSelectedImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getPushedImage() );
272 //GadgetButtonSetDisabledImage( buttonPlayer[currentButton], p->getPlayerTemplate()->getDisabledImage() );
273 buttonPlayer[currentButton]->winSetTooltip(p->getPlayerDisplayName());
274 buttonPlayer[currentButton]->winHide(FALSE);
275 buttonPlayer[currentButton]->winSetStatus( WIN_STATUS_USE_OVERLAY_STATES );
276
277 Color playerColor = p->getPlayerColor();
278 Color backColor = GameMakeColor(0, 0, 0, 255);
279 staticTextPlayer[currentButton]->winSetEnabledTextColors( playerColor, backColor );
280 staticTextPlayer[currentButton]->winHide(FALSE);
281 GadgetStaticTextSetText(staticTextPlayer[currentButton], p->getPlayerDisplayName());
282
283 ++currentButton;
284 break;
285 }
286 }
287 for(currentButton; currentButton<MAX_BUTTONS; ++currentButton)
288 {
289 buttonPlayer[currentButton]->winHide(TRUE);
290 staticTextPlayer[currentButton]->winHide(TRUE);
291 }
292 }

295void ControlBar::populateObserverInfoWindow ( void )
296{
297 if(ObserverPlayerInfoWindow->winIsHidden())
298 return;
299
300 if( !m_observerLookAtPlayer )
301 {
302 ObserverPlayerInfoWindow->winHide(TRUE);
303 ObserverPlayerListWindow->winHide(FALSE);
304 populateObserverList();
305 return;
306 }
307
308 UnicodeString uString;
309 KindOfMaskType mask,clearmask;
310 mask.set(KINDOF_SCORE);
311 clearmask.set(KINDOF_STRUCTURE);
312
313 uString.format(L"%d",m_observerLookAtPlayer->countObjects(mask,clearmask));
314 GadgetStaticTextSetText(staticTextNumberOfUnits, uString);
315
316 Int numBuildings = 0;
317 mask.clear();
318 mask.set(KINDOF_SCORE);
319 mask.set(KINDOF_STRUCTURE);
320 clearmask.clear();
321 numBuildings = m_observerLookAtPlayer->countObjects(mask,clearmask);
322 mask.clear();
323 mask.set(KINDOF_SCORE_CREATE);
324 mask.set(KINDOF_STRUCTURE);
325 numBuildings += m_observerLookAtPlayer->countObjects(mask,clearmask);
326 mask.clear();
327 mask.set(KINDOF_SCORE_DESTROY);
328 mask.set(KINDOF_STRUCTURE);
329 numBuildings += m_observerLookAtPlayer->countObjects(mask,clearmask);
330 uString.format(L"%d",numBuildings);
331 GadgetStaticTextSetText(staticTextNumberOfBuildings, uString);
332 uString.format(L"%d",m_observerLookAtPlayer->getScoreKeeper()->getTotalUnitsDestroyed());
333 GadgetStaticTextSetText(staticTextNumberOfUnitsKilled, uString);
334 uString.format(L"%d",m_observerLookAtPlayer->getScoreKeeper()->getTotalUnitsLost());
335 GadgetStaticTextSetText(staticTextNumberOfUnitsLost, uString);
336 GadgetStaticTextSetText(staticTextPlayerName, m_observerLookAtPlayer->getPlayerDisplayName());
337 Color color = m_observerLookAtPlayer->getPlayerColor();
338 staticTextPlayerName->winSetEnabledTextColors(color, GameMakeColor(0,0,0,255));
339 winFlag->winSetEnabledImage(0, m_observerLookAtPlayer->getPlayerTemplate()->getFlagWaterMarkImage());
340 winGeneralPortrait->winHide(FALSE);
341}
