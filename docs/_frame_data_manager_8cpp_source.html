122void FrameDataManager::resetFrame(UnsignedInt frame, Bool isAdvancing) {
123 UnsignedInt frameindex = frame % FRAME_DATA_LENGTH;
124
125 m_frameData[frameindex].reset();
126
127 if (isAdvancing) {
128 m_frameData[frameindex].setFrame(frame + MAX_FRAMES_AHEAD);
129 }
130
131 if (m_isLocal) {
132 m_frameData[frameindex].setFrameCommandCount(m_frameData[frameindex].getCommandCount());
133 }
134
135 DEBUG_ASSERTCRASH(m_frameData[frameindex].getCommandCount() == 0, ("we just reset the frame data and the command count is not zero, huh?"));
136}

168void FrameDataManager::zeroFrames(UnsignedInt startingFrame, UnsignedInt numFrames) {
169 UnsignedInt frameIndex = startingFrame % FRAME_DATA_LENGTH;
170 for (UnsignedInt i = 0; i < numFrames; ++i) {
171 //DEBUG_LOG(("Calling zeroFrame for frame index %d\n", frameIndex));
172 m_frameData[frameIndex].zeroFrame();
173 ++frameIndex;
174 frameIndex = frameIndex % FRAME_DATA_LENGTH;
175 }
176}
