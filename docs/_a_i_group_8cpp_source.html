100 {
101 member->leaveGroup();
102 i = m_memberList.begin(); // jump back to the beginning, cause ai->leaveGroup will remove this element.

170void AIGroup::add( Object *obj )
171{
172// DEBUG_LOG(("***AIGROUP %x is adding Object %x (%s).\n", this, obj, obj->getTemplate()->getName().str()));
173 DEBUG_ASSERTCRASH(obj != NULL, ("trying to add null obj to AIGroup"));
174 if (obj == NULL)
175 return;
176
177 AIUpdateInterface *ai = obj->getAIUpdateInterface();
178
179 //If this object doesn't have an AIUpdateInterface, then
180 //don't add it to the group UNLESS it is a structure! Structures
181 //with AIUpdateInterfaces also issue similar commands, but those
182 //commands don't need AI updates... they are instant commands like
183 //evacuate or triggering certain special powers...
184 KindOfMaskType validNonAIKindofs;
185 validNonAIKindofs.set(KINDOF_STRUCTURE);
186 validNonAIKindofs.set(KINDOF_ALWAYS_SELECTABLE);
187 if( ai == NULL && !obj->isAnyKindOf( validNonAIKindofs ) )
188 {
189 return;
190 }
191
192 // add to group's list of objects
193 m_memberList.push_back( obj );
194 ++m_memberListSize;
195// DEBUG_LOG(("***AIGROUP %x has size %u now.\n", this, m_memberListSize));
196
197 obj->enterGroup( this );
198
199 // list has changed, properties need recomputation
200 m_dirty = true;
201}

206Bool AIGroup::remove( Object *obj )
207{
208// DEBUG_LOG(("***AIGROUP %x is removing Object %x (%s).\n", this, obj, obj->getTemplate()->getName().str()));
209 std::list<Object *>::iterator i = std::find( m_memberList.begin(), m_memberList.end(), obj );
210
211 // make sure object is actually in the group
212 if (i == m_memberList.end())
213 return FALSE;
214
215 // remove it
216 m_memberList.erase( i );
217 --m_memberListSize;
218// DEBUG_LOG(("***AIGROUP %x has size %u now.\n", this, m_memberListSize));
219
220 // tell object to forget about group
221 obj->leaveGroup();
222
223 // list has changed, properties need recomputation
224 m_dirty = true;
225
226 // if the group is empty, no-one is using it any longer, so destroy it
227 if (isEmpty()) {
228 TheAI->destroyGroup( this );
229 return TRUE;
230 }
231
232 return FALSE;
233}

289Bool AIGroup::getCenter( Coord3D *center )
290{
291 Int count = 0;
292 center->x = 0.0f;
293 center->y = 0.0f;
294 center->z = 0.0f;
295
296 std::list<Object *>::iterator i;
297 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
298 {
299 if( (*i)->isDisabledByType( DISABLED_HELD) )
300 {
301 continue; // don't bother counting riders in the center calculation.
302 }
303 AIUpdateInterface *ai = (*i)->getAIUpdateInterface();
304 if (ai)
305 {
306 const Coord3D *objPos = (*i)->getPosition();
307 center->x += objPos->x;
308 center->y += objPos->y;
309 center->z += objPos->z;
310 ++count;
311 }
312 }
313
314 if (count == 0 && !m_memberList.empty())
315 {
316 /*
317 if there are no AIs (eg, the team consists of a faction bldg), we can get here.
318
319 This was originally used to offset the centers of objects moving (still used for that) and non-ais can't move.
320 So if you have a mix of ai's & not ai's, you want just the ais.
321 But it seems reasonable that if there are no ai's, it returns the center of the other stuff. Cause they won't be moving anyway.
322 */
323 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
324 {
325 if( (*i)->isDisabledByType( DISABLED_HELD) )
326 {
327 continue; // don't bother counting riders in the center calculation.
328 }
329 const Coord3D *objPos = (*i)->getPosition();
330 center->x += objPos->x;
331 center->y += objPos->y;
332 center->z += objPos->z;
333 ++count;
334 }
335 }
336
337 center->x /= count;
338 center->y /= count;
339 center->z /= count;
340
341 return count > 0;
342}

344Bool AIGroup::getMinMaxAndCenter( Coord2D *min, Coord2D *max, Coord3D *center )
345{
346 Int count = 0;
347 min->x = 1e10f;
348 max->x = -1e10f;
349 min->y = 1e10f;
350 max->y = -1e10f;
351 center->x = 0.0f;
352 center->y = 0.0f;
353 center->z = 0.0f;
354
355 std::list<Object *>::iterator i;
356 FormationID id= NO_FORMATION_ID;
357 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
358 {
359 if( (*i)->isDisabledByType( DISABLED_HELD) )
360 {
361 continue; // don't bother counting riders in the center calculation.
362 }
363 AIUpdateInterface *ai = (*i)->getAIUpdateInterface();
364 if (ai)
365 {
366 const Coord3D *objPos = (*i)->getPosition();
367 center->x += objPos->x;
368 center->y += objPos->y;
369 center->z += objPos->z;
370
371 //Calculate the bounding coordinates of all units
372 min->x = min->x > objPos->x ? objPos->x : min->x;
373 max->x = max->x < objPos->x ? objPos->x : max->x;
374 min->y = min->y > objPos->y ? objPos->y : min->y;
375 max->y = max->y < objPos->y ? objPos->y : max->y;
376 FormationID curID = (*i)->getFormationID() ;
377 if (count==0) {
378 id = curID;
379 } else {
380 if (id == NO_FORMATION_ID) {
381 id = NO_FORMATION_ID;
382 }
383 }
384
385 count++;
386 }
387 }
388
389 center->x /= count;
390 center->y /= count;
391 center->z /= count;
392 Bool isFormation = (id!=NO_FORMATION_ID);
393 if (count<2) isFormation = false;
394 return isFormation;
395}

402void AIGroup::recompute( void )
403{
404 Real closeDist = 999999999.9f;
405 Real dx, dy, dist;
406 const Coord3D *objPos;
407 Coord3D center;
408
409 getCenter( &center );
410
411 if (m_groundPath) {
412 m_groundPath->deleteInstance();
413 m_groundPath = NULL;
414 }
415
416 m_speed = 9999999999.9f;
417
418 std::list<Object *>::iterator i;
419 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
420 {
421 // don't consider immobile things for leadership
422 if ((*i)->isKindOf(KINDOF_IMMOBILE))
423 continue;
424
425 if( (*i)->isDisabledByType( DISABLED_HELD) )
426 {
427 continue; // don't bother counting riders in the max speed calculation.
428 }
429 Object *obj = (*i);
430 AIUpdateInterface *ai = obj->getAIUpdateInterface();
431 if (ai)
432 {
433
434 //
435 // speed is slowest speed ... we won't consider slow speeds from objects that
436 // are being penalized by their damage state, if we did the whole group would slow
437 // down to that slowest penalized speed, instead only those objects that are penalized
438 // will fall out of "formation" ... bummer for them!
439 //
440 Real maxSpeed = ai->getCurLocomotorSpeed();
441 if( m_speed > maxSpeed &&
442 IS_CONDITION_BETTER( obj->getBodyModule()->getDamageState(), TheGlobalData->m_movementPenaltyDamageState ) )
443 m_speed = maxSpeed;
444
445 // leader is closest to the group's center
446 objPos = obj->getPosition();
447 dx = objPos->x - center.x;
448 dy = objPos->y - center.y;
449 dist = dx*dx + dy*dy;
450 if (dist < closeDist)
451 {
452 closeDist = dist;
453 }
454 }
455 }
456 // clear "dirty bit" - data is up-to-date
457 m_dirty = false;
458}

480void AIGroup::computeIndividualDestination( Coord3D *dest, const Coord3D *groupDest,
481 Object *obj, const Coord3D *center, Bool isFormation )
482{
483 Coord2D v;
484
485 // compute vector from "group center" to self
486 const Coord3D *pos = obj->getPosition();
487 if (isFormation) {
488 obj->getFormationOffset(&v);
489 } else {
490 v.x = pos->x - center->x;
491 v.y = pos->y - center->y;
492 }
493 Real length = v.length();
494 if (length > 6*obj->getGeometryInfo().getBoundingCircleRadius()) {
495 length = 6*obj->getGeometryInfo().getBoundingCircleRadius();
496 }
497 v.normalize();
498 v.x *= length;
499 v.y *= length;
500 PathfindLayerEnum layer = TheTerrainLogic->getLayerForDestination(groupDest);
501
502 // move to same offset at destination
504 dest->x = groupDest->x + v.x;
505 dest->y = groupDest->y + v.y;
506 dest->z = TheTerrainLogic->getLayerHeight( dest->x, dest->y, layer );
507 AIUpdateInterface *ai = obj->getAIUpdateInterface();
508 if (ai && ai->isDoingGroundMovement()) {
509 if (isFormation) {
510 TheAI->pathfinder()->adjustDestination(obj, ai->getLocomotorSet(), dest, NULL);
511 } else {
512 TheAI->pathfinder()->adjustDestination(obj, ai->getLocomotorSet(), dest, groupDest);
513 }
514 TheAI->pathfinder()->updateGoal(obj, dest, LAYER_GROUND);
515 }

528Bool AIGroup::friend_computeGroundPath( const Coord3D *pos, CommandSourceType cmdSource )
529
530{
531
532 if (m_dirty)
533 recompute();
534
535 std::list<Object *>::iterator i;
536 // compute current centroid of the team
537 Coord3D center;
538 Coord2D min;
539 Coord2D max;
540 Real dx, dy;
541
542 if (TheGlobalData->m_debugAI==AI_DEBUG_TERRAIN) return false;
543
544 Bool closeEnough = false;
545 getMinMaxAndCenter( &min, &max, &center );
546 Real distSqr = 4*sqr(TheAI->getAiData()->m_distanceRequiresGroup);
547
548 Int numInfantry = 0;
549 Int numVehicles = 0;
550 Object *centerVehicle = NULL;
551 Real distSqrCenterVeh = distSqr*10;
552 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
553 {
554 Object *obj = (*i);
555 TheAI->pathfinder()->removeGoal(obj);
556 if (obj->isDisabledByType( DISABLED_HELD ) )
557 {
558 continue; // don't bother telling the occupants to move.
559 }
560 if( obj->getAI()==NULL )
561 {
562 continue;
563 }
564 if( obj->isKindOf( KINDOF_INFANTRY ) )
565 {
566 numInfantry++;
567 } else if (obj->isKindOf( KINDOF_VEHICLE)) {
568 if (obj->isKindOf(KINDOF_AIRCRAFT)) {
569 continue;
570 }
571 numVehicles++;
572 } else {
573 continue;
574 }
575 // Note - we are getting the closest of ANY type of unit for later testing intentionally. jba.
576 Coord3D unitPos = *((*i)->getPosition());
577
578 dx = unitPos.x-pos->x;
579 dy = unitPos.y-pos->y;
580 if (dx*dx+dy*dy<distSqr) {
581 distSqr = dx*dx+dy*dy;
582 }
583
584 // find object closest to the center.
585 dx = unitPos.x-center.x;
586 dy = unitPos.y-center.y;
587 if (centerVehicle==NULL || dx*dx+dy*dy<distSqrCenterVeh) {
588 centerVehicle = (*i);
589 distSqrCenterVeh = dx*dx+dy*dy;
590 }
591 }
592
593 if(centerVehicle==NULL) return false;
594 center = *centerVehicle->getPosition();
595
596 dx = max.x - min.x;
597 dy = max.y - min.y;
598 if (dx*dx + dy*dy > sqr(TheAI->getAiData()->m_distanceRequiresGroup)) {
599 distSqr = dx*dx+dy*dy;
600 }
601 if (distSqr < sqr(TheAI->getAiData()->m_minDistanceForGroup)) {
602 return false;
603 }
604 if (distSqr>sqr(TheAI->getAiData()->m_distanceRequiresGroup)) {
605 closeEnough = true;
606 }
607 if (numInfantry>6) {
608 closeEnough = true;
609 }
610 if (numVehicles>4) {
611 closeEnough = true;
612 }
613
614 if (!closeEnough) {
615 Bool isPassable = true;
616 // see if all units have an unobstructed path to the center.
617 // If so, then they are close enough.
618 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
619 {
620 Object *obj = (*i);
621 if (!obj->isKindOf(KINDOF_INFANTRY)) {
622 continue;
623 }
624 AIUpdateInterface *ai = (*i)->getAIUpdateInterface();
625 if (ai)
626 {
627 if (!TheAI->pathfinder()->isLinePassable(obj,
628 ai->getLocomotorSet().getValidSurfaces(), obj->getLayer(), *obj->getPosition(),
629 center, false, true)) {
630 isPassable = false;
631 }
632 }
633 }
634 if (isPassable) closeEnough = true;
635 }
636 if (!closeEnough) return false;
637
638 m_groundPath = TheAI->pathfinder()->findGroundPath(&center, pos, PATH_DIAMETER_IN_CELLS, false);
639 return m_groundPath!=NULL;
640
641}

644// Clamps to the player's current visible map area. jba. [8/28/2003]

683Bool AIGroup::friend_moveInfantryToPos( const Coord3D *pos, CommandSourceType cmdSource )
684
685{
686 if (m_groundPath==NULL) return false;
687
688 Int numColumns = 3;
689 Int halfNumColumns = numColumns/2;
690 Real dx, dy;
691 Coord3D center;
692 if (!getCenter( &center )) return false;
693
694 // Get the start & end vectors for the path.
695 Coord3D startPoint = *m_groundPath->getFirstNode()->getPosition();
696 Real farEnoughSqr = sqr(PATH_DIAMETER_IN_CELLS*PATHFIND_CELL_SIZE_F);
697 PathNode *startNode = NULL;
698 PathNode *node;
699 for (node = m_groundPath->getFirstNode(); node; node=node->getNextOptimized()) {
700 dx = node->getPosition()->x - startPoint.x;
701 dy = node->getPosition()->y - startPoint.y;
702 if (dx*dx+dy*dy>farEnoughSqr) {
703 startNode = node;
704 break;
705 }
706 }
707 Coord3D endPoint = *m_groundPath->getLastNode()->getPosition();
708 PathNode *endNode = NULL;
709 for (node = m_groundPath->getFirstNode(); node; node=node->getNextOptimized()) {
710 Real dx = node->getPosition()->x - endPoint.x;
711 Real dy = node->getPosition()->y - endPoint.y;
712 if (dx*dx+dy*dy>farEnoughSqr) {
713 endNode = node;
714 }
715 }
716 if (startNode==NULL || endNode==NULL) {
717 m_groundPath->deleteInstance();
718 m_groundPath = NULL;
719 return false;
720 }
721
722 Coord2D startVector;
723 startVector.x = startNode->getPosition()->x - startPoint.x;
724 startVector.y = startNode->getPosition()->y - startPoint.y;
725 startVector.normalize();
726
727 Coord2D endVector;
728 endVector.x = endPoint.x - endNode->getPosition()->x;
729 endVector.y = endPoint.y - endNode->getPosition()->y;
730 endVector.normalize();
731
732 Coord2D startVectorNormal;
733 startVectorNormal.x = -startVector.y;
734 startVectorNormal.y = startVector.x;
735 startVectorNormal.normalize();
736
737 Coord2D endVectorNormal;
738 endVectorNormal.x = -endVector.y;
739 endVectorNormal.y = endVector.x;
740 endVectorNormal.normalize();
741
742 Bool useEndVector = false;
743 Int unitsToPath = 0;
744 // Move.
745 MemoryPoolObjectHolder iterHolder;
746 SimpleObjectIterator *iter = newInstance(SimpleObjectIterator);
747 iterHolder.hold(iter);
748 MemoryPoolObjectHolder iterHolder2;
749 SimpleObjectIterator *iter2 = newInstance(SimpleObjectIterator);
750 iterHolder2.hold(iter2);
751 std::list<Object *>::iterator i;
752 PlayerType controllingPlayerType = PLAYER_COMPUTER;
753 for( i = m_memberList.begin(); i != m_memberList.end(); ++i )
754 {
755 if ((*i)->isDisabledByType( DISABLED_HELD ) )
756 {
757 continue; // don't bother telling the occupants to move.
758 }
759 if( !(*i)->isKindOf( KINDOF_INFANTRY ) )
760 {
761 continue;
762 }
763 if( (*i)->getAI()==NULL )
764 {
765 continue;
766 }
767 if ( (*i)->isKindOf( KINDOF_MOB_NEXUS ) )
768 {
769 return FALSE;// means I did NOT do a column group pathfind,
770 //so the nexus will have a far-away goal position for the mobsters to aim at
771 }
772 if ((*i)->getControllingPlayer()) {
773 controllingPlayerType = (*i)->getControllingPlayer()->getPlayerType();
774 }
775 Coord3D unitPos = *((*i)->getPosition());
776 TheAI->pathfinder()->removeGoal(*i);
777 dx = unitPos.x - center.x;
778 dy = unitPos.y - center.y;
779 // Sort by the dot product of normal.
780 iter->insert((*i), dx*startVectorNormal.x+dy*startVectorNormal.y);
781 unitsToPath++;
782
783 // If units are closer to the end vector than the start vector, use the end vector.
784 Real distToEndSqr;
785 Real distToStartSqr;
786 dx = unitPos.x - endPoint.x;
787 dy = unitPos.y - endPoint.y;
788 distToEndSqr = dx*dx + dy*dy;
789 dx = unitPos.x - startPoint.x;
790 dy = unitPos.y - startPoint.y;
791 distToStartSqr = dx*dx + dy*dy;
792 if (distToStartSqr>distToEndSqr) {
793 useEndVector = true;
794 }
795
796 }
797 if (unitsToPath<TheAI->getAiData()->m_minInfantryForGroup) {
798 return false;
799 }
800
801 Object *theUnit;
802 if (useEndVector) {

1437 dest.y += offset * columnDelta * cornerVectorNormal.y;
1438 if (factor&1) {

1440 dest.y += 0.5f*PATHFIND_CELL_SIZE_F * cornerVectorNormal.y;

1447 curVector.x = dest.x-prevPos.x;
1448 curVector.y = dest.y-prevPos.y;

1451 if (cornerVector.x*curVector.x + cornerVector.y*curVector.y > 0) {

1456 node=node->getNextOptimized();

1468 if (threeColumnDelta<-3) threeColumnDelta=-3;

1475 dest.y += offset * threeColumnDelta * endVectorNormal.y;
1476 if (factor&1) {
1477 dest.x += PATHFIND_CELL_SIZE_F * endVectorNormal.x;
1478 dest.y += PATHFIND_CELL_SIZE_F * endVectorNormal.y;
1479 }
1480#if 0

1489 dest.x -= factor*offset*endVector.x;
1490 dest.y -= factor*offset*endVector.y;
1491 dest.z = TheTerrainLogic->getLayerHeight( dest.x, dest.y, layer );

1493 while (path.size()>0) {

1495 prevPos = path[path.size()-1];

1497 curVector.y = dest.y-prevPos.y;

1508 TheAI->pathfinder()->updateGoal(theUnit, &dest, LAYER_GROUND);
1509 path.push_back(dest);
1510 ai->aiFollowPath( &path, NULL, cmdSource );
1511 }
