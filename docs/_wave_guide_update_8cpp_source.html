156Bool WaveGuideUpdate::startMoving( void )
157{
158 Object *waveGuide = getObject();
159 Waypoint *waypoint;
160 const WaveGuideUpdateModuleData *data = getWaveGuideUpdateModuleData();
161
162 AudioEventRTS loopingSound = data->m_loopingSound;
163 loopingSound.setObjectID( getObject()->getID() );
164 TheAudio->addAudioEvent( &loopingSound );
165
166 waypoint = TheTerrainLogic->getWaypointByName( "WaveGuide1" );
167 if( waypoint )
168 {
169
170 //
171 // traverse the entire waypoint path, there must be one link in the path, and
172 // at any of the links there cannot be multiple choices of where to go next
173 //
174 Waypoint *verify = waypoint;
175 while( verify )
176 {
177
178 // can't have more than one link
179 if( verify->getNumLinks() > 1 )
180 {
181
182 DEBUG_CRASH(( "WaveGuideUpdate::startMoving - The waypoint path cannot have multiple link choices at any node\n" ));
183 return FALSE;
184
185 } // end if
186
187 // set our destination location to this waypoint position
188 m_finalDestination = *verify->getLocation();
189
190 // on to the next
191 verify = verify->getLink( 0 );
192
193 } // end while
194
195 // there must be at least one link
196 Waypoint *next = waypoint->getLink( 0 );
197 if( next == NULL )
198 {
199
200 DEBUG_CRASH(( "WaveGuideUpdate:startMoving - There must be a linked waypoint path to follow\n" ));
201 return FALSE;
202
203 } // end if
204
205 // get vector from next waypoint to first waypoint
206 Coord2D v;
207 v.x = next->getLocation()->x - waypoint->getLocation()->x;
208 v.y = next->getLocation()->y - waypoint->getLocation()->y;
209
210 // turn vector into angle
211 Real angle = v.toAngle();
212
213 // orient the waveguide the same direction
214 waveGuide->setOrientation( angle );
215
216 // get the ai update interface for the waveguide
217 AIUpdateInterface *ai = waveGuide->getAIUpdateInterface();
218 if( ai )
219 {
220
221 // instantly move to waypoint location on the terrain
222 Coord3D pos;
223 pos.x = waypoint->getLocation()->x;
224 pos.y = waypoint->getLocation()->y;
225 pos.z = TheTerrainLogic->getGroundHeight( pos.x, pos.y );
226 waveGuide->setPosition( &pos );
227
228 // follow the waypoint path here
229 ai->aiFollowWaypointPath( waypoint, CMD_FROM_AI );
230
231 //
232 // make sure it doesn't stop or slow down at each waypoint along the path if the path
233 // we're following consists of multiple nodes
234 //
235 ai->setPathExtraDistance( PATH_EXTRA_DISTANCE );
236
237 } // end if
238
239 } // endif
240
241 return TRUE; // all is well
242
243} // end startMoving

313void WaveGuideUpdate::computeWaveShapePoints( void )
314{
315
316 // get the module data
317 const WaveGuideUpdateModuleData *modData = getWaveGuideUpdateModuleData();
318
319 //
320 // build an array of points that will represent the wavefront. these points are
321 // created in local object space and can be put through the transform matrix of the
322 // object to get the world position given the waveguides' current position and orientation
323 //
324 Int halfY = modData->m_ySize / 2.0f;
325
326 for( Int y = -halfY; y < halfY; y += modData->m_linearWaveSpacing )
327 {
328
329 // sanity
330 if( m_shapePointCount >= MAX_WAVEGUIDE_SHAPE_POINTS )
331 break; // exit for
332
333 //
334 // if there is a bend magnitude we will use a form of "y = x^2" to make a parabola
335 // that forms the wavefront ... otherwise we just make a straight line
336 //
337 if( modData->m_waveBendMagnitude )
338 m_shapePoints[ m_shapePointCount ].x = -(y * y) / modData->m_waveBendMagnitude;
339 else
340 m_shapePoints[ m_shapePointCount ].x = 0.0f;
341 m_shapePoints[ m_shapePointCount ].y = y;
342 m_shapePoints[ m_shapePointCount ].z = 0.0f;
343 m_shapePointCount++;
344
345 } // end for width
346
347} // end computeWaveShapePoints

375void WaveGuideUpdate::doShapeEffects( void )
376{
377 const WaveGuideUpdateModuleData *modData = getWaveGuideUpdateModuleData();
378
379 //
380 // the particle systems that make up the wave shape need to maintain a position that is
381 // just above the ground
382 //
383 Int i, j;
384 ParticleSystem *particleSys;
385 Coord3D pos;
386 for( i = 0; i < m_shapePointCount; i++ )
387 {
388
389 for( j = 0; j < MAX_SHAPE_EFFECTS; j++ )
390 {
391
392 // get the particle system here
393 particleSys = TheParticleSystemManager->findParticleSystem( m_shapeEffects[ i ][ j ] );
394 if( particleSys )
395 {
396
397 //
398 // we have the height of the terrain at this shape point in the transformed shape
399 // point array ... just set the LOCAL position for the particle system to reflect
400 // this Z *IF* that Z will not take us above the target water height
401 //
402 if( m_transformedShapePoints[ i ].z < modData->m_preferredHeight )
403 {
404
405 particleSys->getPosition( &pos );
406 pos.z = m_transformedShapePoints[ i ].z;
407 particleSys->setPosition( &pos );
408
409 } // end if
410
411 } // end if
412
413 } // end for j
414
415 } // end for i
416
417} // end doShapeEffects

449// ------------------------------------------------------------------------------------------------
450void WaveGuideUpdate::doShoreEffects( void )
451{
452
453 // this is expensive to do *all* the time
454 if( TheGameLogic->getFrame() & 0x1 )
455 return;
456
457 Object *waveGuide = getObject();
458 Int i;
459
460 // get module data
461 const WaveGuideUpdateModuleData *modData = getWaveGuideUpdateModuleData();
462
463 //
464 // setup an array of points just behind the front of the wave where the first wave
465 // "crest" is rising up ... we will use these points to make collide with shore effects
466 //
467 Coord3D effectPoints[ MAX_WAVEGUIDE_SHAPE_POINTS ];
468 for( i = 0; i < m_shapePointCount; i++ )
469 {
470
471 // setup point to be a distance "behind" the wave shape points
472 effectPoints[ i ].x = m_shapePoints[ i ].x - modData->m_shorelineEffectDistance;
473 effectPoints[ i ].y = m_shapePoints[ i ].y;
474 effectPoints[ i ].z = m_shapePoints[ i ].z;
475
476 // transform the point
477 waveGuide->transformPoint( &effectPoints[ i ], &effectPoints[ i ] );
478
479 } // end for i
480
481 //
482 // go across the shape of our wave ... when we detect a transition from underground to
483 // above ground or vice verse, that is a shoreline ... play an effect between those points
484 //
485 static const ParticleSystemTemplate *left = TheParticleSystemManager->findTemplate( "WaveSplashLeft01" );
486 static const ParticleSystemTemplate *right = TheParticleSystemManager->findTemplate( "WaveSplashRight01" );
487 ParticleSystem *particleSystem;
488 Real terrainZ;
489 Bool underWater = TRUE;
490 Coord3D *point;
491 for( i = 0; i < m_shapePointCount; i++ )
492 {
493
494 // get the point in question
495 point = &effectPoints[ i ];
496
497 // get terrain height at this point
498 terrainZ = TheTerrainLogic->getGroundHeight( point->x, point->y );
499
500 // is the terrain at this point below or above the preferred water height
501 if( terrainZ > modData->m_preferredHeight )
502 {
503
504 //
505 // if we were previously 'under' we are not any longer, this is a shore
506 // note that we're ignoring the first point i = 0
507 //
508 if( underWater == TRUE && i != 0 )
509 {
510 Coord3D *prevPoint = &effectPoints[ i - 1 ]; // the prev point is actuall on the water so we'll use it
511
512 particleSystem = TheParticleSystemManager->createParticleSystem( right );
513 if( particleSystem )
514 particleSystem->setPosition( prevPoint );
515
516 } // end if
517
518 // we are now 'above'
519 underWater = FALSE;
520
521 } // end if
522 else
523 {
524
525 //
526 // if we were previously 'above' we are not any longer, this is a shore
527 // note that we're ignoring the first point i = 0
528 //
529 if( underWater == FALSE && i != 0 )
530 {
531
532 particleSystem = TheParticleSystemManager->createParticleSystem( left );
533 if( particleSystem )
534 particleSystem->setPosition( point );
535
536 } // end if
537
538 // we are now 'under'
539 underWater = TRUE;
540
541 } // end else
542
543 } // end for i
544
545} // end doShoreEffects

549// ------------------------------------------------------------------------------------------------
550void WaveGuideUpdate::doDamage( void )
551{
552 Object *waveGuide = getObject();
553 const WaveGuideUpdateModuleData *modData = getWaveGuideUpdateModuleData();
554
555 // get our position forward unit direction vector
556// const Coord3D *unitForward = waveGuide->getUnitDirectionVector2D();
557
558 // iterate over all our sample points and kill stuff around us
559 for( Int i = 0; i < m_shapePointCount; i++ )
560 {
561
562 // scan objects around us and do damage to objects we have "passed over" and are behind us
563 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( &m_transformedShapePoints[ i ],
564 modData->m_damageRadius,
565 FROM_CENTER_2D,
566 NULL );
567 MemoryPoolObjectHolder hold( iter );
568 Object *obj;
569 const Coord3D *objPos;
570 Coord3D v;
571 Real angle;
572 for( obj = iter->first(); obj; obj = iter->next() )
573 {
574
575 // ignore all waveguide types
576 if( obj->isKindOf( KINDOF_WAVEGUIDE ) )
577 continue;
578
579 //
580 // ignore bridge towers ... water waves will destroy bridges by colliding with
581 // the actual bridge object in the *center* of the bridge, not the towers
582 //
583 if( obj->isKindOf( KINDOF_BRIDGE_TOWER ) )
584 continue;
585
586 // get other object position
587 objPos = obj->getPosition();
588
589 //
590 // only damage objects that are below the preferred height of the wave doing the damage,
591 // bridges are an exception as their object is raised above the ground, but we'll
592 // say the water is destroying the foundation of it
593 //
594 if( objPos->z > modData->m_preferredHeight && obj->isKindOf( KINDOF_BRIDGE ) == FALSE )
595 continue;
596
597 // get the vector from us to the object
598 v.x = objPos->x - m_transformedShapePoints[ i ].x;
599 v.y = objPos->y - m_transformedShapePoints[ i ].y;
600 v.z = 0.0f; // forget Z, this is really a top down 2D calculation
601 v.normalize();
602
603 //
604 // get the cosine of the angle between the our forward direction and the vector to the obj
605 // otherwise known as a dot product
606 //
607 angle = v.x * m_transformedShapePoints[ i ].x +
608 v.y * m_transformedShapePoints[ i ].y +
609 v.z * m_transformedShapePoints[ i ].z;
610
611 //
612 // we will only do damage if the object is "behind" the wave ... that is, behind
613 // the 180 degrees mark behind us which we figure out by seeing if the cosine of
614 // the angle between us (dot product) is less than zero
615 //
616 if( angle < 0 )
617 {
618
619 // if object was not wet before we kill it and play effects
620 if( !obj->getStatusBits().test( OBJECT_STATUS_WET ) )
621 {
622 static const ParticleSystemTemplate *splash = TheParticleSystemManager->findTemplate( "WaveHit01" );
623 ParticleSystem *particleSystem;
624
625 // create particle system at position
626 particleSystem = TheParticleSystemManager->createParticleSystem( splash );
627 if( particleSystem )
628 {
629 Coord3D pos;
630
631 //
632 // we will use the position of the shape point (in local space) with the
633 // Z of the object we're killin
634 //
635 pos.x = m_shapePoints[ i ].x;
636 pos.y = m_shapePoints[ i ].y;
637 pos.z = obj->getPosition()->z;
638 particleSystem->setPosition( &pos );
639 particleSystem->attachToObject( waveGuide );
640
641 } // end if
642
643 // this object is now wet
644 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_WET ) );
645
646 // some things can be toppled ... ooo, xtra special of us!
647 Coord3D toppleVector;
648 toppleVector.x = obj->getPosition()->x - m_transformedShapePoints[ i ].x;
649 toppleVector.y = obj->getPosition()->y - m_transformedShapePoints[ i ].y;
650 toppleVector.z = 0;
651 obj->topple( &toppleVector, modData->m_toppleForce, TOPPLE_OPTIONS_NO_BOUNCE |
652 TOPPLE_OPTIONS_NO_FX );
653
654 // do a lot of water damage
655 DamageInfo damageInfo;
656 damageInfo.in.m_damageType = DAMAGE_WATER;
657 damageInfo.in.m_deathType = DEATH_FLOODED;
658 damageInfo.in.m_sourceID = waveGuide->getID();
659 damageInfo.in.m_amount = modData->m_damageAmount;
660 obj->attemptDamage( &damageInfo );
661
662 //
663 // set flooded condition for model and turn off shadows cause we're inside water,
664 // note that we want to do this after we damage the object so that we switch
665 // off the shadows on the new model data loaded after damage is dealt
666 //
667 Drawable *draw = obj->getDrawable();
668 if( draw )
669 {
670
671 draw->setModelConditionState( MODELCONDITION_FLOODED );
672 draw->setShadowsEnabled( FALSE );
673
674 } // end if
675
676 //
677 // Temp demo hack, replace bridges destroyed with the special destroyed bridge art
678 //
679 if( obj->isKindOf( KINDOF_BRIDGE ) )
680 {
681 const ThingTemplate* ttn = TheThingFactory->findTemplate("WaterWaveBridge");
682 Object *newBridge = TheThingFactory->newObject( ttn, NULL );
683
684 if( newBridge )
685 {
686 Real angle = 0.0f;
687
688 // get the bridge represented by the object we're killing
689 Bridge *oldBridge = TheTerrainLogic->findBridgeAt( obj->getPosition() );
690 if( oldBridge )
691 {
692 BridgeInfo bridgeInfo;
693
694 // get the bridge info
695 oldBridge->getBridgeInfo( &bridgeInfo );
696
697 // compute the angle of the object based on the angle of the bridge
698 Coord2D v;
699 v.x = bridgeInfo.to.x - bridgeInfo.from.x;
700 v.y = bridgeInfo.to.y - bridgeInfo.from.y;
701 angle = v.toAngle();
702
703 } // end if
704
705 // put new bridge looking object in the world
706 newBridge->setPosition( obj->getPosition() );
707 newBridge->setOrientation( angle );
708
709 // create and attach a wave hit bridge particle system to the new object
710 ParticleSystem *particleSystem = TheParticleSystemManager->createParticleSystem( modData->m_bridgeParticle );
711 if( particleSystem )
712 {
713 Coord3D u, x, y, z;
714 Matrix3D transform;
715
716 z.x = 0.0f;
717 z.y = 0.0f;
718 z.z = 1.0f;
719
720 //
721 // angle is rotated, becuase we computed from 'from' and 'to' points of
722 // the bridge going *across* the valley, not pointing *down* it
723 //
724 u.x = Cos( angle + modData->m_bridgeParticleAngleFudge );
725 u.y = Sin( angle + modData->m_bridgeParticleAngleFudge );
726 u.z = 0.0f;
727
728 y.crossProduct( &z, &u, &y );
729 x.crossProduct( &y, &z, &x );
730
731 transform.Set( x.x, y.x, z.x, obj->getPosition()->x,
732 x.y, y.y, z.y, obj->getPosition()->y,
733 x.z, y.z, z.z, obj->getPosition()->z );
734
735 particleSystem->setLocalTransform( &transform );
736
737 } // end if
738
739 // destroy the old bridge and bridge object
740 TheTerrainLogic->deleteBridge( oldBridge );
741
742 } // end if
743
744 } // end if
745
746 } // end if
747
748 } // end if
749
750 } // end for obj
751
752 } // end for i
753
754} // end doDamage

758UpdateSleepTime WaveGuideUpdate::update( void )
759{
761 Object *waveGuide = getObject();
762
763 if (m_needDisable)
764 {
765 m_needDisable = false;
766 getObject()->setDisabled( DISABLED_DEFAULT );
767 return UPDATE_SLEEP_NONE;
768 }
769
770 // do nothing if we're disabled
771 if( waveGuide->isDisabled() )
772 return UPDATE_SLEEP_NONE;
773
774 // get module data
775 const WaveGuideUpdateModuleData *modData = getWaveGuideUpdateModuleData();
776
777 // set the frame we became active on if not set
778 if( m_activeFrame == 0 )
779 m_activeFrame = TheGameLogic->getFrame();
780
781 // if we're waiting for a delay check it and get out of here
782 if( TheGameLogic->getFrame() - m_activeFrame < modData->m_waveDelay )
783 return UPDATE_SLEEP_NONE;
784
785 // get our position
786// const Coord3D *waveGuidePos = waveGuide->getPosition();
787
788 // when we become enabled we will start particle effects for our wave
789 if( m_initialized == FALSE )
790 {
791
792 // start the effects we need
793 if( initWaveGuide() == FALSE )
794 {
795
796 // initialization failed, destroy object and get out of here
797 TheGameLogic->destroyObject( getObject() );
798 return UPDATE_SLEEP_NONE;
799
800 } // end if
801
802 // we are now in motion
803 m_initialized = TRUE;
804
805 } // end if
806
807 // every half second we try to play a random spash sound
808 if( TheGameLogic->getFrame() - m_splashSoundFrame > LOGICFRAMES_PER_SECOND / 2.0f )
809 {
810
811 // mark that we've had the opportunity to play
812 m_splashSoundFrame = TheGameLogic->getFrame();
813
814 // pick a random number and play according to frequency
815 if( GameLogicRandomValue( 1, 100 ) > modData->m_randomSplashSoundFrequency )
816 {
817 AudioEventRTS randomSplash(modData->m_randomSplashSound);
818 randomSplash.setObjectID(waveGuide->getID());
819 TheAudio->addAudioEvent(&randomSplash);
820 } // end if
821 } // end if
822
823 //
824 // transform the wave shape points once for the current position ... we have this array
825 // because we access this data several times and don't want to recompute it every time
826 //
827 transformWaveShape();
828
829 // see if we are close enough to the end of our journey on the waypoint path
830 const Coord3D *currentPos = waveGuide->getPosition();
831 Real distSquared = PATH_EXTRA_DISTANCE * PATH_EXTRA_DISTANCE;
832 Coord2D v;
833 v.x = m_finalDestination.x - currentPos->x;
834 v.y = m_finalDestination.y - currentPos->y;
835 if( v.x * v.x + v.y * v.y <= distSquared )
836 {
837 static const ParticleSystemTemplate *waveSplash = TheParticleSystemManager->findTemplate( "WaveSplash01" );
838 ParticleSystem *particleSys;
839
840 // create spash effect
841 particleSys = TheParticleSystemManager->createParticleSystem( waveSplash );
842 if( particleSys )
843 particleSys->setLocalTransform( waveGuide->getTransformMatrix() );
844
845 // destroy object
846 TheGameLogic->destroyObject( waveGuide );
847
848 // update the radar with the final new water levels
849 TheRadar->refreshTerrain( TheTerrainLogic );
850
851 return UPDATE_SLEEP_NONE;
852
853 } // end if
854
855 // do wavefront effects
856 doShapeEffects();
857
858 // do water effects
859 doWaterMotion();
860
861 // do shore effects
862 doShoreEffects();
863
864 // do damage to things in our way
865 doDamage();
866
867 return UPDATE_SLEEP_NONE;
868
869} // end update
