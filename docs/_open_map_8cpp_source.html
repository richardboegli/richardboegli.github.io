85void OpenMap::OnOK()
86{
87 CListBox *pList = (CListBox *)this->GetDlgItem(IDC_OPEN_LIST);
88 if (pList == NULL) {
89 OnCancel();
90 return;
91 }
92
93 Int sel = pList->GetCurSel();
94 if (sel == LB_ERR) {
95 m_pInfo->browse = true;
96 } else {
97 CString newName;
98 pList->GetText(sel, newName );
99 if (m_usingSystemDir)
100 m_pInfo->filename = ".\\Maps\\" + newName + "\\" + newName + ".map";
101 else
102 {
103 m_pInfo->filename = TheGlobalData->getPath_UserData().str();
104 m_pInfo->filename = m_pInfo->filename + "Maps\\" + newName + "\\" + newName + ".map";
105 }
106 }
107 CDialog::OnOK();
108}

110void OpenMap::populateMapListbox( Bool systemMaps )
111{
112 m_usingSystemDir = systemMaps;
113#if defined(_DEBUG) || defined(_INTERNAL)
114 ::AfxGetApp()->WriteProfileInt(MAP_OPENSAVE_PANEL_SECTION, "UseSystemDir", m_usingSystemDir);
115#endif
116
117 HANDLE hFindFile = 0;
118 WIN32_FIND_DATA findData;
119 char dirBuf[_MAX_PATH];
120 char findBuf[_MAX_PATH];
121 char fileBuf[_MAX_PATH];
122
123 if (systemMaps)
124 strcpy(dirBuf, "Maps\\");
125 else
126 {
127 strcpy(dirBuf, TheGlobalData->getPath_UserData().str());
128 strcat(dirBuf, "Maps\\");
129 }
130
131 int len = strlen(dirBuf);
132
133 if (len > 0 && dirBuf[len - 1] != '\\') {
134 dirBuf[len++] = '\\';
135 dirBuf[len] = 0;
136 }
137 CListBox *pList = (CListBox *)this->GetDlgItem(IDC_OPEN_LIST);
138 if (pList == NULL) return;
139 pList->ResetContent();
140 strcpy(findBuf, dirBuf);
141 strcat(findBuf, "*.*");
142
143 Bool found = false;
144
145 hFindFile = FindFirstFile(findBuf, &findData);
146 if (hFindFile != INVALID_HANDLE_VALUE) {
147 do {
148 if (strcmp(findData.cFileName, ".") == 0 || strcmp(findData.cFileName, "..") == 0)
149 continue;
150 if ((findData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0) {
151 continue;
152 }
153
154 strcpy(fileBuf, dirBuf);
155 strcat(fileBuf, findData.cFileName);
156 strcat(fileBuf, "\\");
157 strcat(fileBuf, findData.cFileName);
158 strcat(fileBuf, ".map");
159 try {
160 CFileStatus status;
161 if (CFile::GetStatus(fileBuf, status)) {
162 if (!(status.m_attribute & CFile::directory)) {
163 pList->AddString(findData.cFileName);
164 found = true;
165 };
166 }
167 } catch(...) {}
168
169 } while (FindNextFile(hFindFile, &findData));
170
171 if (hFindFile) FindClose(hFindFile);
172 }
173 if (found) {
174 pList->SetCurSel(0);
175 } else {
176 CWnd *pOk = GetDlgItem(IDOK);
177 if (pOk) pOk->EnableWindow(false);
178 }
179}

181BOOL OpenMap::OnInitDialog()
182{
183 CDialog::OnInitDialog();
184
185 CButton *pSystemMaps = (CButton *)this->GetDlgItem(IDC_SYSTEMMAPS);
186 if (pSystemMaps != NULL)
187 pSystemMaps->SetCheck( m_usingSystemDir );
188
189 CButton *pUserMaps = (CButton *)this->GetDlgItem(IDC_USERMAPS);
190 if (pUserMaps != NULL)
191 pUserMaps->SetCheck( !m_usingSystemDir );
192
193#if !defined(_DEBUG) && !defined(_INTERNAL)
194 if (pSystemMaps)
195 pSystemMaps->ShowWindow( FALSE );
196 if (pUserMaps)
197 pUserMaps->ShowWindow( FALSE );
198#endif
199
200 populateMapListbox( m_usingSystemDir );
201
202 return TRUE; // return TRUE unless you set the focus to a control
203 // EXCEPTION: OCX Property Pages should return FALSE
204}
