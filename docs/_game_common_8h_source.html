70enum
71{
72 LOGICFRAMES_PER_SECOND = 30,
73 MSEC_PER_SECOND = 1000
74};
75const Real LOGICFRAMES_PER_MSEC_REAL = (((Real)LOGICFRAMES_PER_SECOND) / ((Real)MSEC_PER_SECOND));
76const Real MSEC_PER_LOGICFRAME_REAL = (((Real)MSEC_PER_SECOND) / ((Real)LOGICFRAMES_PER_SECOND));
77const Real LOGICFRAMES_PER_SECONDS_REAL = (Real)LOGICFRAMES_PER_SECOND;
78const Real SECONDS_PER_LOGICFRAME_REAL = 1.0f / LOGICFRAMES_PER_SECONDS_REAL;
79
80// ----------------------------------------------------------------------------------------------
81// note that this returns a REAL value, not an int... most callers will want to
82// call ceil() on the result, so that partial frames get converted to full frames!
83inline Real ConvertDurationFromMsecsToFrames(Real msec)
84{
85 return (msec * LOGICFRAMES_PER_MSEC_REAL);
86}
87
88// ----------------------------------------------------------------------------------------------
89inline Real ConvertVelocityInSecsToFrames(Real distPerMsec)
90{
91 // this looks wrong, but is the correct conversion factor.
92 return (distPerMsec * SECONDS_PER_LOGICFRAME_REAL);
93}
94
95// ----------------------------------------------------------------------------------------------
96inline Real ConvertAccelerationInSecsToFrames(Real distPerSec2)
97{
98 // this looks wrong, but is the correct conversion factor.
99 const Real SEC_PER_LOGICFRAME_SQR = (SECONDS_PER_LOGICFRAME_REAL * SECONDS_PER_LOGICFRAME_REAL);
100 return (distPerSec2 * SEC_PER_LOGICFRAME_SQR);
101}
102
103// ----------------------------------------------------------------------------------------------
104inline Real ConvertAngularVelocityInDegreesPerSecToRadsPerFrame(Real degPerSec)
105{
106 const Real RADS_PER_DEGREE = PI / 180.0f;
107 return (degPerSec * (SECONDS_PER_LOGICFRAME_REAL * RADS_PER_DEGREE));
108}

440// ------------------------------------------------------------------------
441template<class OBJCLASS>
442class DLINK_ITERATOR
443{
444public:
445 // this is the weird C++ syntax for "pointer-to-member-function"
446 typedef OBJCLASS* (OBJCLASS::*GetNextFunc)() const;
447private:
448 OBJCLASS* m_cur;
449 GetNextFunc m_getNextFunc; // this is the weird C++ syntax for "pointer-to-member-function"
450public:
451 DLINK_ITERATOR(OBJCLASS* cur, GetNextFunc getNextFunc) : m_cur(cur), m_getNextFunc(getNextFunc)
452 {
453 }
454
455 void advance()
456 {
457 if (m_cur)
458 m_cur = callMemberFunction(*m_cur, m_getNextFunc)();
459 }
460
461 Bool done() const
462 {
463 return m_cur == NULL;
464 }
465
466 OBJCLASS* cur() const
467 {
468 return m_cur;
469 }
470
471};
