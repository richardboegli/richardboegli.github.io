99UpdateSleepTime FloatUpdate::update( void )
100{
102
103 // if we're not enabled, do nothing
104 if( m_enabled == TRUE )
105 {
106 // get object position
107 const Coord3D *pos = getObject()->getPosition();
108
109 // get the height of the water here
110 Real waterZ;
111 TheTerrainLogic->isUnderwater( pos->x, pos->y, &waterZ );
112
113 // snap to the water surface
114 Coord3D newPos;
115 newPos.x = pos->x;
116 newPos.y = pos->y;
117 newPos.z = waterZ;
118 getObject()->setPosition( &newPos );
119
120 }
121
122 Drawable *draw = getObject()->getDrawable();
123 if (draw)
124 {
125
126 Real angle = INT_TO_REAL(TheGameLogic->getFrame());
127 Real yaw = sin(angle * 0.0291f) * 0.05f;
128 Real pitch = sin(angle * 0.0515f) * 0.05f;
129
130 Matrix3D mx = *draw->getInstanceMatrix();
131
132 Real zRot = mx.Get_Z_Rotation();
133 mx.Make_Identity();
134 mx.Rotate_Z(zRot);
135 mx.Rotate_Y(yaw);
136 mx.Rotate_X(pitch);
137
138 draw->setInstanceMatrix(&mx);
139 }
140
141 return UPDATE_SLEEP_NONE;
142} // end update
