71UpdateSleepTime HijackerUpdate::update( void )
72{
74
75 if ( ! m_update) // have not flagged for updating
76 {
77 return UPDATE_SLEEP_NONE;
78 }
79
80 if (m_isInVehicle)
81 {
82 Object *obj = getObject();
83
84 // If hijacker has hijacked a vehicle, he needs to move along with it...
85 //Continually reset position of hijacker to match the position of the target.
86 Object *target = getTargetObject();
87 if( target )
88 {
89 // @todo I think we should test for ! IsEffectivelyDead() as well, here
90 obj->setPosition( target->getPosition() );
91 m_wasTargetAirborne = target->isSignificantlyAboveTerrain();
92 m_ejectPos = *target->getPosition();
93
94 // So, if while I am driving this American war vehicle, I gain skill points, I get to keep them when I wreck the vehicle
95 ExperienceTracker *targetExp = target->getExperienceTracker();
96 ExperienceTracker *jackerExp = obj->getExperienceTracker();
97 if ( targetExp && jackerExp )
98 {
99 VeterancyLevel highestLevel = MAX(targetExp->getVeterancyLevel(),jackerExp->getVeterancyLevel());
100 jackerExp->setVeterancyLevel( highestLevel );
101 targetExp->setVeterancyLevel( highestLevel );
102 }
103
104 }
105 else // the car we have been "driving" is dead now, and has safely ejected us
106 {
107 {
108
109
110 //THIS BLOCK RESTORES HIJACKER TO PARTITION MANAGER AND UNHIDES HIM
111 ThePartitionManager->registerObject( obj );
112
113 if( obj->getDrawable() )
114 {
115 // so it is time to unhide ourselves and be a pedestrian hijacker again
116 obj->getDrawable()->setDrawableHidden( false );
117 }
118
119 // We won't come back here until and unless we have hijacked another vehicle
120 obj->clearStatus( MAKE_OBJECT_STATUS_MASK3( OBJECT_STATUS_NO_COLLISIONS, OBJECT_STATUS_MASKED, OBJECT_STATUS_UNSELECTABLE ) );
121
122 AIUpdateInterface* ai = obj->getAIUpdateInterface();
123 if ( ai )
124 {
125 ai->aiIdle( CMD_FROM_AI );
126 }
127
128 if (m_wasTargetAirborne)
129 {
130 const ThingTemplate* putInContainerTmpl = TheThingFactory->findTemplate(getHijackerUpdateModuleData()->m_parachuteName);
131 DEBUG_ASSERTCRASH(putInContainerTmpl,("DeliverPayload: PutInContainer %s not found!",getHijackerUpdateModuleData()->m_parachuteName.str()));
132 if (putInContainerTmpl)
133 {
134 Object* container = TheThingFactory->newObject( putInContainerTmpl, obj->getTeam() );
135 container->setPosition(&m_ejectPos);
136 if (container->getContain()->isValidContainerFor(obj, true))
137 {
138 container->getContain()->addToContain(obj);
139 }
140 else
141 {
142 DEBUG_CRASH(("DeliverPayload: PutInContainer %s is full, or not valid for the payload!",getHijackerUpdateModuleData()->m_parachuteName.str()));
143 }
144 }
145
146 }
147
148
149 }// end if (! hostVehicleHasEjection)
150
151 setTargetObject( NULL );
152 setIsInVehicle( FALSE );
153 setUpdate( FALSE );
154 m_wasTargetAirborne = false;
155
156 }// end if( target )
157
158 }
159 else // not in vehicle
160 {
161 m_wasTargetAirborne = false;
162 }
163
164 return UPDATE_SLEEP_NONE;
165}

220void HijackerUpdate::xfer( Xfer *xfer )
