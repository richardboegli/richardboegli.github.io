81unsigned long ggetm(void *src, int bytes)
82{
83 unsigned char *s = (unsigned char *) src;
84 unsigned long value;
85
86 value = 0L;
87 while (bytes--)
88 {
89 value = (value<<8) + ((*s++));
90 }
91 return(value);
92}

94/* get intel memory */

96unsigned long ggeti(void *src, int bytes)
97{
98 unsigned char *s = (unsigned char *) src;
99 int i = 0;
100 unsigned long value;
101
102 value = 0L;
103 while (bytes--)
104 {
105 value += ((*s++)) << (i);
106 i += 8;
107 }
108 return(value);
109}

111/* put motorolla memory */
112
113void gputm(void *dst, unsigned long data, int bytes)
114{
115 unsigned char *d = (unsigned char *) dst;
116 unsigned long pval;
117
118 data <<= (4-bytes)*8;
119 while (bytes)
120 {
121 pval = data >> 24;
122 *d++ = (unsigned char) pval;
123 data <<= 8;
124 --bytes;
125 }
126}

146GSTREAM * GCALL gopen(const char *filename)
147{
148 FILE *handle;
149
150 Msg( __LINE__, __FILE__, "gopen:: %s.", filename );
151
152 handle = fopen( filename, "r+b" );
153
154 Msg( __LINE__, __FILE__, "gopen:: handle = %d", (( handle != NULL )? 1 : 0 ));
155
156 if ( !handle ) {
157
158 handle = fopen( filename, "rb" );
159
160 Msg( __LINE__, __FILE__, "gopen:: handle = %d", (( handle != NULL )? 1 : 0 ));
161 }
162 return((GSTREAM *) handle);
163}

166{
167 FILE *handle;
