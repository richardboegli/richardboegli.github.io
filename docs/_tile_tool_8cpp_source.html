63void TileTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
64{
65 if (m == TRACK_L)
66 m_textureClassToDraw = TerrainMaterial::getFgTexClass();
67 else if (m == TRACK_R)
68 m_textureClassToDraw = TerrainMaterial::getBgTexClass();
69 else
70 return;
71
72// WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
73 // just in case, release it.
74 REF_PTR_RELEASE(m_htMapEditCopy);
75 m_htMapEditCopy = pDoc->GetHeightMap()->duplicate();
76 m_prevXIndex = -1;
77 m_prevYIndex = -1;
78 m_prevViewPt = viewPt;
79 mouseMoved(m, viewPt, pView, pDoc);
80}

83void TileTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
84{
85 if (m != TRACK_L && m != TRACK_R) return;
86#define DONT_DO_FULL_UPDATE
87#ifdef DO_FULL_UPDATE
88 m_htMapEditCopy->optimizeTiles(); // force to optimize tileset
89 IRegion2D partialRange = {0,0,0,0};
90 pDoc->updateHeightMap(m_htMapEditCopy, false, partialRange);
91#endif
92 WBDocUndoable *pUndo = new WBDocUndoable(pDoc, m_htMapEditCopy);
93 pDoc->AddAndDoUndoable(pUndo);
94 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
95 REF_PTR_RELEASE(m_htMapEditCopy);
96}

99void TileTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
100{
101 Bool didAnything = false;
102 Bool fullUpdate = false;
103 Coord3D cpt;
104
105 pView->viewToDocCoords(viewPt, &cpt);
106 DrawObject::setFeedbackPos(cpt);
107 if (m != TRACK_L && m != TRACK_R) return;
108 Int dx = m_prevViewPt.x - viewPt.x;
109 Int dy = m_prevViewPt.y - viewPt.y;
110 Int count = sqrt(dx*dx + dy*dy);
111 Int k;
112
113 Int totalMinX = m_htMapEditCopy->getXExtent();
114 Int totalMinY = m_htMapEditCopy->getYExtent();
115 Int totalMaxX = 0;
116 Int totalMaxY = 0;
117
118 count /= 2;
119 if (count<1) count = 1;
120 for (k=0; k<=count; k++) {
121 CPoint curViewPt;
122 curViewPt.x = viewPt.x + ((count-k)*dx)/count;
123 curViewPt.y = viewPt.y + ((count-k)*dy)/count;
124
125 if (k==0) {
126 DEBUG_ASSERTCRASH(curViewPt.x == m_prevViewPt.x, ("Bad x"));
127 DEBUG_ASSERTCRASH(curViewPt.y == m_prevViewPt.y, ("Bad y"));
128 }
129 if (k==count) {
130 DEBUG_ASSERTCRASH(curViewPt.x == viewPt.x, ("Bad x"));
131 DEBUG_ASSERTCRASH(curViewPt.y == viewPt.y, ("Bad y"));
132 }
133 CPoint ndx;
134 Int width = getWidth();
135 pView->viewToDocCoords(curViewPt, &cpt);
136 getCenterIndex(&cpt, width, &ndx, pDoc);
137 if (m_prevXIndex == ndx.x && m_prevYIndex == ndx.y) continue;
138
139 m_prevXIndex = ndx.x;
140 m_prevYIndex = ndx.y;
141
142 Int i, j;
143 Int minX = ndx.x - (width)/2;
144 Int minY = ndx.y - (width)/2;
145 for (i=minX; i<minX+width; i++) {
146 if (i<0 || i>=m_htMapEditCopy->getXExtent()) continue;
147 for (j=minY; j<minY+width; j++) {
148 if (j<0 || j>=m_htMapEditCopy->getYExtent()) continue;
149 if (TerrainMaterial::isPaintingPathingInfo()) {
150 m_htMapEditCopy->setCliff(i, j, !TerrainMaterial::isPaintingPassable());
151 } else {
152 if (m_htMapEditCopy->setTileNdx(i, j, m_textureClassToDraw, width==1)) {
153 fullUpdate = true;
154 }
155 }
156 didAnything = true;
157 if (totalMinX>i) totalMinX = i;
158 if (totalMinY>j) totalMinY = j;
159 if (totalMaxX<i) totalMaxX = i;
160 if (totalMaxY<j) totalMaxY = j;
161 }
162 }
163 }
164 if (didAnything) {
165 IRegion2D partialRange;
166 partialRange.lo.x = totalMinX;
167 partialRange.hi.x = totalMaxX+1;
168 partialRange.lo.y = totalMinY;
169 partialRange.hi.y = totalMaxY+1;
170 if (fullUpdate) {
171 m_htMapEditCopy->optimizeTiles(); // force to optimize tileset
172 }
173 pDoc->updateHeightMap(m_htMapEditCopy, !fullUpdate, partialRange);
174 }
175 pView->UpdateWindow();
176 m_prevViewPt = viewPt;
177}
