116void BuildList::loadSides(void)
117{
118 CComboBox *pCombo = (CComboBox*)GetDlgItem(IDC_SIDES_COMBO);
119 if (!pCombo) {
120 DEBUG_LOG(("*** BuildList::loadSides Missing resource!!!\n"));
121 return;
122 }
123 pCombo->ResetContent();
124 Int i;
125 for (i=0; i<TheSidesList->getNumSides(); i++) {
126 Dict *d = TheSidesList->getSideInfo(i)->getDict();
127 AsciiString name = d->getAsciiString(TheKey_playerName);
128 UnicodeString uni = d->getUnicodeString(TheKey_playerDisplayName);
129 AsciiString fmt;
130 if (name.isEmpty())
131 fmt = "(neutral player, cannot be edited)";
132 else
133 fmt.format("%s=\"%ls\"",name.str(),uni.str());
134 pCombo->AddString(fmt.str());
135 }
136 updateCurSide();
137}

140void BuildList::updateCurSide(void)
141{
142 if (TheSidesList->getNumSides() < 1)
143 return;
144
145 CComboBox *pCombo = (CComboBox*)GetDlgItem(IDC_SIDES_COMBO);
146 if (!pCombo) {
147 DEBUG_LOG(("*** BuildList::updateCurSide Missing resource!!!\n"));
148 return;
149 }
150 if (m_curSide<0 || m_curSide >= TheSidesList->getNumSides()) {
151 m_curSide = 0;
152 }
153 if (pCombo->GetCurSel() != m_curSide) {
154 pCombo->SetCurSel(m_curSide);
155 }
156 SidesInfo *pSide = TheSidesList->getSideInfo(m_curSide);
157
158 CListBox *pList = (CListBox*)GetDlgItem(IDC_BUILD_LIST);
159 if (!pList) {
160 DEBUG_LOG(("*** BuildList::updateCurSide Missing resource!!! IDC_BUILD_LIST\n"));
161 return;
162 }
163 pList->ResetContent();
164
165 BuildListInfo *pBuild = pSide->getBuildList();
166 while (pBuild) {
167 const char *pName = pBuild->getTemplateName().str();
168 pList->AddString(pName);
169 pBuild = pBuild->getNext();
170 }
171 OnSelchangeBuildList();
172}

197void BuildList::OnMoveUp()
198{
199 CListBox *pList = (CListBox*)GetDlgItem(IDC_BUILD_LIST);

210 Int count = m_curBuildList;
211
212 BuildListInfo *pBuildInfo = pSide->getBuildList();
213 while (count) {
214 count--;
215 pBuildInfo = pBuildInfo->getNext();
216 if (pBuildInfo == NULL) return;
217 }
218 Int newSel = m_curBuildList-1;
219 pSide->reorderInBuildList(pBuildInfo, newSel);
220 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
221 SidesListUndoable *pUndo = new SidesListUndoable(sides, pDoc);
222 pDoc->AddAndDoUndoable(pUndo);
223 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
224
225 updateCurSide();
226 pList->SetCurSel(newSel);
227 OnSelchangeBuildList();

297void BuildList::setSelectedBuildList(BuildListInfo *pInfo)
298{
299 if (!m_staticThis) {
300 return;
301 }
302 Int i;
303 for (i=0; i<TheSidesList->getNumSides(); i++) {
304 SidesInfo *pSide = TheSidesList->getSideInfo(i);
305 Int listSel = 0;
306 for (BuildListInfo *pBuild = pSide->getBuildList(); pBuild; pBuild = pBuild->getNext()) {
307 if (pInfo == pBuild) {
308 pBuild->setSelected(true);
309// CComboBox *pCombo = (CComboBox*)m_staticThis->GetDlgItem(IDC_SIDES_COMBO);
310 if (m_staticThis->m_curSide != i) {
311 m_staticThis->m_curSide = i;
312 m_staticThis->updateCurSide();
313 }
314 CListBox *pList = (CListBox*)m_staticThis->GetDlgItem(IDC_BUILD_LIST);
315 pList->SetCurSel(listSel);
316 m_staticThis->OnSelchangeBuildList();
317 } else {
318 pBuild->setSelected(false);
319 }
320 listSel++;
321 }
322 }

326void BuildList::OnSelchangeBuildList()
327{
328 if (TheSidesList->getNumSides() < 1)
329 return;
330
331 CListBox *pList = (CListBox*)GetDlgItem(IDC_BUILD_LIST);
332 if (!pList) {
333 DEBUG_LOG(("*** BuildList::updateCurSide Missing resource!!! IDC_BUILD_LIST\n"));
334 return;
335 }
336 m_curBuildList = pList->GetCurSel();
337 Int numBL = pList->GetCount();
338
339 SidesInfo *pSide = NULL;
340 if (TheSidesList) {
341 pSide = TheSidesList->getSideInfo(m_curSide);
342 }
343 Int count = m_curBuildList;
344 BuildListInfo *pBuildInfo = NULL;
345 if (pSide) {
346 pBuildInfo = pSide->getBuildList();
347 }
348 if (count<0) pBuildInfo = NULL;
349 while (count && pBuildInfo) {
350 count--;
351 pBuildInfo = pBuildInfo->getNext();
352 }
353
354 Bool enableUp=false;
355 Bool enableDown=false;
356 Bool enableAttrs = true;
357
358 Int energyConsumption = 0;
359 Int energyProduction = 0;
360 Real energyUsed = 0.0f;
361 const Dict *playerDict = pSide->getDict();
362 AsciiString playerName = playerDict->getAsciiString(TheKey_playerName);
363
364 // add up energy consumed/produced by objects on the map
365 const ThingTemplate *thingTemplate;
366 for (MapObject *pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext())
367 {
368 // get thing template based from map object name
369 thingTemplate = pMapObj->getThingTemplate();
370 if (!thingTemplate)
371 continue;
372 Dict *d = pMapObj->getProperties();
373 Bool exists;
374 AsciiString objectTeamName = d->getAsciiString(TheKey_originalOwner, &exists);
375 TeamsInfo *teamInfo = TheSidesList->findTeamInfo(objectTeamName);
376 Dict *teamDict = (teamInfo)?teamInfo->getDict():NULL;
377 AsciiString objectOwnerName = (teamDict)?teamDict->getAsciiString(TheKey_teamOwner):AsciiString::TheEmptyString;
378
379 Int energy = 0;
380 if (objectOwnerName.compareNoCase(playerName) == 0)
381 energy = thingTemplate->getEnergyProduction();
382
383 if (energy > 0)
384 energyProduction += energy;
385 else
386 energyConsumption -= energy; // keep it positive
387 }
388
389 // add up energy consumed/produced by objects in the build list
390 {
391 count = m_curBuildList;
392 BuildListInfo *pBuildInfo = pSide->getBuildList();
393 if (count<0) pBuildInfo = NULL;
394 while (count>=0 && pBuildInfo) {
395 AsciiString tName = pBuildInfo->getTemplateName();
396 const ThingTemplate *templ = TheThingFactory->findTemplate(tName);
397 if (!templ)
398 break;
399
400 Int energy = templ->getEnergyProduction();
401
402 if (energy > 0)
403 energyProduction += energy;
404 else
405 energyConsumption -= energy; // keep it positive
406
407 count--;
408 pBuildInfo = pBuildInfo->getNext();
409 }
410 }
411
412 if (energyProduction)
413 {
414 energyUsed = (Real)energyConsumption/(Real)energyProduction;
415 energyUsed = min(1.0f, energyUsed);
416 energyUsed = max(0.0f, energyUsed);
417 }
418 else if (energyConsumption)
419 {
420 energyUsed = 1.0f;
421 }
422 //DEBUG_LOG(("Energy: %d/%d - %g\n", energyConsumption, energyProduction, energyUsed));
423 CProgressCtrl *progressWnd = (CProgressCtrl *)GetDlgItem(IDC_POWER);
424 if (progressWnd)
425 {
426 progressWnd->EnableWindow(true);
427 progressWnd->SetPos((Int)((1.0f-energyUsed)*100));
428 }
429
430 if (pBuildInfo==NULL) {
431 enableAttrs = false;
432 }
433 if (m_curBuildList > 0) {
434 enableUp = true;
435 }
436 if (m_curBuildList >= 0 && m_curBuildList < numBL-1) {
437 enableDown = true;
438 }
439 CWnd *pWnd = GetDlgItem(IDC_MOVE_UP);
440 if (pWnd) pWnd->EnableWindow(enableUp);
441 pWnd = GetDlgItem(IDC_MOVE_DOWN);
442 if (pWnd) pWnd->EnableWindow(enableDown);
443 pWnd = GetDlgItem(IDC_ALREADY_BUILD);
444 if (pWnd) pWnd->EnableWindow(enableAttrs);
445 pWnd = GetDlgItem(IDC_REBUILDS);
446 if (pWnd) pWnd->EnableWindow(enableAttrs);
447 pWnd = GetDlgItem(IDC_DELETE_BUILDING);
448 if (pWnd) pWnd->EnableWindow(enableAttrs);
449 PointerTool::clearSelection(); // unselect other stuff.
450 if (pBuildInfo) {
451 CWnd *edit;
452 static char buff[12];
453 pBuildInfo->setSelected(true);
454
455 m_angle = pBuildInfo->getAngle() * 180/PI;
456 sprintf(buff, "%0.2f", m_angle);
457 edit = GetDlgItem(IDC_MAPOBJECT_Angle);
458 edit->SetWindowText(buff);
459
460 m_height = pBuildInfo->getLocation()->z;
461 sprintf(buff, "%0.2f", m_height);
462 edit = GetDlgItem(IDC_MAPOBJECT_ZOffset);
463 edit->SetWindowText(buff);
464
465 CButton *pBtn = (CButton *)GetDlgItem(IDC_ALREADY_BUILD);
466 if (pBtn) pBtn->SetCheck(pBuildInfo->isInitiallyBuilt()?1:0);
467 CComboBox *pCombo = (CComboBox *)GetDlgItem(IDC_REBUILDS);
468 if (pCombo==NULL) return;
469 UnsignedInt nr = pBuildInfo->getNumRebuilds();
470 if (nr == BuildListInfo::UNLIMITED_REBUILDS) {
471 pCombo->SetCurSel(6);
472 } else if (nr<6) {
473 pCombo->SetCurSel(nr);
474 } else {
475 CString str;
476 str.Format("%d", nr);
477 pCombo->SetWindowText(str);
478 }
479 }

680void BuildList::OnChangeZOffset()
681{
682 Real value = 0.0f;
683 CWnd* edit = GetDlgItem(IDC_MAPOBJECT_ZOffset);
684 CString cstr;
685 edit->GetWindowText(cstr);
686 if (!cstr.IsEmpty()) {
687 value = atof(cstr);
688 }
689 m_height = value;
690
691 SidesInfo *pSide = TheSidesList->getSideInfo(m_curSide);
692 CListBox *pList = (CListBox*)GetDlgItem(IDC_BUILD_LIST);
693 m_curBuildList = pList->GetCurSel();
694 if (m_curBuildList < 0) return;
695 Int count = m_curBuildList;
696 BuildListInfo *pBuildInfo = pSide->getBuildList();
697 while (count) {
698 count--;
699 pBuildInfo = pBuildInfo->getNext();
700 if (pBuildInfo == NULL) return;
