181// ------------------------------------------------------------------------------------------------
182UpdateSleepTime PropagandaTowerBehavior::update( void )
183{
185 const PropagandaTowerBehaviorModuleData *modData = getPropagandaTowerBehaviorModuleData();
186
187 //Sep 27, 2002 (Kris): Added this code to prevent the tower from working while under construction.
188 Object *self = getObject();
189 if( self->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
190 return UPDATE_SLEEP_NONE;
191
192 if( self->testStatus(OBJECT_STATUS_SOLD) )
193 {
194 removeAllInfluence();
195 return UPDATE_SLEEP_FOREVER;
196 }
197
198 if( self->isEffectivelyDead() )
199 return UPDATE_SLEEP_FOREVER;
200
201 if( self->isDisabled() )
202 {
203 // We need to let go of everyone if we are EMPd or underpowered or yadda, but not if we are only held
204
205 DisabledMaskType allButHeld = MAKE_DISABLED_MASK( DISABLED_HELD );
206 FLIP_DISABLEDMASK(allButHeld);
207
208 if( TEST_DISABLEDMASK_ANY(self->getDisabledFlags(), allButHeld) )
209 {
210 removeAllInfluence();
211 return UPDATE_SLEEP_NONE;
212 }
213 }
214
215 if( self->getContainedBy() && self->getContainedBy()->getContainedBy() )
216 {
217 // If our container is contained, we turn the heck off. Seems like a weird specific check, but all of
218 // attacking is guarded by the same check in isPassengersAllowedToFire. We similarly work in a container,
219 // but not in a double container.
220 removeAllInfluence();
221 return UPDATE_SLEEP_NONE;
222 }
223
224 // if it's not time to scan, nothing to do
225 UnsignedInt currentFrame = TheGameLogic->getFrame();
226 if( currentFrame - m_lastScanFrame >= modData->m_scanDelayInFrames )
227 {
228
229 // do a scan
230 doScan();
231 m_lastScanFrame = currentFrame;
232
233 } // end if
234
235 // go through any objects in our area of influence and do the effect logic on them
236 Object *obj;
237 ObjectTracker *curr = NULL, *prev = NULL, *next = NULL;
238 for( curr = m_insideList; curr; curr = next )
239 {
240
241 // get the next link
242 next = curr->next;
243
244 // find this object
245 obj = TheGameLogic->findObjectByID( curr->objectID );
246 if ((obj) &&
247 (obj->isKindOf(KINDOF_SCORE) || obj->isKindOf(KINDOF_SCORE_CREATE) || obj->isKindOf(KINDOF_SCORE_DESTROY) || obj->isKindOf(KINDOF_MP_COUNT_FOR_VICTORY)))
248 {
249
250 // give any bonus to this object
251 effectLogic( obj, TRUE, getPropagandaTowerBehaviorModuleData() );
252
253 // record this element as the previous one found in the list
254 prev = curr;
255
256 } // end if
257 else
258 {
259
260 //
261 // actual object wasn't found, remove this entry from our inside list so we don't
262 // have to search through it again
263 //
264 if( prev )
265 prev->next = curr->next;
266 else
267 m_insideList = curr->next;
268 curr->deleteInstance();
269
270 } // end else
271
272 } // end for, curr
273
274 return UPDATE_SLEEP_NONE;
275
276} // end update

296void PropagandaTowerBehavior::effectLogic( Object *obj, Bool giving,
297 const PropagandaTowerBehaviorModuleData *modData )
298{
299 Bool effectUpgraded = getObject()->getControllingPlayer()->hasUpgradeComplete( m_upgradeRequired );
300
301 // if giving the effect
302 if( giving )
303 {
304 if ( obj->hasAnyDamageWeapon() == TRUE )
305 {
306 if( obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_ENTHUSIASTIC ) == FALSE )
307 obj->setWeaponBonusCondition( WEAPONBONUSCONDITION_ENTHUSIASTIC );
308
309
310 if (effectUpgraded)
311 {
312 if (obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_SUBLIMINAL ) == FALSE)
313 obj->setWeaponBonusCondition( WEAPONBONUSCONDITION_SUBLIMINAL );
314 }
315
316 } // hasdamageweapon
317
318
319 // grant health to this object as well
320 BodyModuleInterface *body = obj->getBodyModule();
321 if( body )
322 {
323 Real healthPercent;
324 if(effectUpgraded)
325 healthPercent = modData->m_upgradedAutoHealPercentPerSecond;
326 else
327 healthPercent = modData->m_autoHealPercentPerSecond;
328
329 Real amount = healthPercent / LOGICFRAMES_PER_SECOND * body->getMaxHealth();
330
331 // Dustin wants the healing effect not to stack from multiple propaganda towers...
332 // To accomplish this, I'll give every object a single healing-sender (ID)
333 // Any given healing recipient (object) can only receive healing from one particular healing sender
334 // and cannot change healing senders until the previous one expires (its scandelay)
335
336// obj->attemptHealing(amount, getObject()); // the regular way to give healing...
337 obj->attemptHealingFromSoleBenefactor( amount, getObject(), modData->m_scanDelayInFrames );//the non-stacking way
338
339 } // end if
340
341 } // end if
342 else
343 {
344
345 // taking effect away
346 obj->clearWeaponBonusCondition( WEAPONBONUSCONDITION_ENTHUSIASTIC );
347 obj->clearWeaponBonusCondition( WEAPONBONUSCONDITION_SUBLIMINAL );
348
349 } // end else
350
351} // end effectLogic

356void PropagandaTowerBehavior::removeAllInfluence( void )
357{
358 ObjectTracker *o;
359
360 // go through all objects we've given bonuses to and remove them
361 Object *obj;
362 for( o = m_insideList; o; o = o->next )
363 {
364
365 obj = TheGameLogic->findObjectByID( o->objectID );
366 if( obj )
367 effectLogic( obj, FALSE, getPropagandaTowerBehaviorModuleData() );
368
369 } // end for
370
371 // delete the list of objects under our influence
372 while( m_insideList )
373 {
374
375 o = m_insideList->next;
376 m_insideList->deleteInstance();
377 m_insideList = o;
378
379 } // end while
380
381} // end removeAllInfluence

386void PropagandaTowerBehavior::doScan( void )
387{
388 const PropagandaTowerBehaviorModuleData *modData = getPropagandaTowerBehaviorModuleData();
389 Object *us = getObject();
390 ObjectTracker *newInsideList = NULL;
391
392 // The act of scanning is when we play our effect
393 Bool upgradePresent = FALSE;
394 if( m_upgradeRequired )
395 {
396
397 // see if we have the upgrade
398 switch( m_upgradeRequired->getUpgradeType() )
399 {
400
401 // ------------------------------------------------------------------------------------------
402 case UPGRADE_TYPE_PLAYER:
403 {
404 Player *player = us->getControllingPlayer();
405
406 upgradePresent = player->hasUpgradeComplete( m_upgradeRequired );
407 break;
408
409 } // end player upgrade
410
411 // ------------------------------------------------------------------------------------------
412 case UPGRADE_TYPE_OBJECT:
413 {
414
415 upgradePresent = us->hasUpgrade( m_upgradeRequired );
416 break;
417
418 } // end object upgrade
419
420 // ------------------------------------------------------------------------------------------
421 default:
422 {
423
424 DEBUG_CRASH(( "PropagandaTowerBehavior::doScan - Unknown upgrade type '%d'\n",
425 m_upgradeRequired->getUpgradeType() ));
426 break;
427
428 } // end default
429
430 } // end switch
431
432 } // end if
433
434
435 Bool doFX = TRUE;
436 // if it is us, or if it is he and we do see him
437 Object *overlord = us->getContainedBy();
438 if ( overlord )
439 {
440 if ( us->getControllingPlayer() != ThePlayerList->getLocalPlayer() )// daling with someone else's tower
441 {
442 if ( overlord->testStatus( OBJECT_STATUS_STEALTHED ) && !overlord->testStatus( OBJECT_STATUS_DETECTED ) )
443 doFX = FALSE;// so they don't give up their position
444
445 }
446
447 // Sorry, this is a lot of hard coded mishmash, but it must be done... ship it!
448 // Propaganda towers (proper) never get contained, so they don't care about this code
449 // The PropTowers that ride on the backs of overlord tanks want to suppress their fx if the overlord is stealthed to the local player
450 // The PropTowers that ride on the bellies of Helixes also suppress their fx when hiding stealthed (never happens?)
451 // The PropTowers must do nothing, scanning-or-FX-wise when they are inside something... there are two cases
452 // 1) When the prop tower is on an Overlord that is in a helicopter (likely a Helix II)
453 // 2) When the prop tower is the EmperorTank, which is in a helicopter (likely a Helix II).
454
455 if ( us->isKindOf( KINDOF_VEHICLE ) && !us->isKindOf( KINDOF_PORTABLE_STRUCTURE ) )//craptacular!
456 // oh my dear Lord... I am not a propaganda tower, I am an emperorTank! that means the overlord is a helix or something!
457 return; // proptowers that are riding IN things cannot do their scan!
458
459 //okay this is wacky, but this proptower may be on an overlord thank that is riding in a helix... oy!
460 Object *helix = overlord->getContainedBy();
461 if ( helix )
462 return; // proptowers that are riding ON things that are in-turn riding IN things cannot do their scan!
463
464 }
465
466 if ( us->getControllingPlayer() != ThePlayerList->getLocalPlayer() )// daling with someone else's tower
467 {
468 if ( us->testStatus( OBJECT_STATUS_STEALTHED ) && !us->testStatus( OBJECT_STATUS_DETECTED ) )
469 {
470 doFX = FALSE;// Certainly don't play if we ourselves are stelthed.
471 }
472 }
473
474 if ( doFX )
475 {
476 // play the right pulse
477 if( upgradePresent == TRUE )
478 FXList::doFXObj( modData->m_upgradedPulseFX, us );
479 else
480 FXList::doFXObj( modData->m_pulseFX, us );
481 }
482
483 // setup scan filters
484 PartitionFilterRelationship relationship( us, PartitionFilterRelationship::ALLOW_ALLIES );
485 PartitionFilterAlive filterAlive;
486 PartitionFilterSameMapStatus filterMapStatus(us);
487 PartitionFilterAcceptByKindOf filterOutBuildings(KINDOFMASK_NONE, MAKE_KINDOF_MASK(KINDOF_STRUCTURE));
488 PartitionFilter *filters[] = { &relationship,
489 &filterAlive,
490 &filterMapStatus,
491 &filterOutBuildings,
492 NULL
493 };
494
495 // scan objects in our region
496 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( us->getPosition(),
497 modData->m_scanRadius,
498 FROM_CENTER_2D,
499 filters );
500 MemoryPoolObjectHolder hold( iter );
501 Object *obj;
502 ObjectTracker *newEntry;
503 for( obj = iter->first(); obj; obj = iter->next() )
504 {
505
506 // ignore ourselves, unless Design wants us to affect ourselves
507 if( obj == us && !modData->m_affectsSelf)
508 continue;
509
510 // record this object as being in the new "in list"
511 newEntry = newInstance(ObjectTracker);
512 newEntry->objectID = obj->getID();
513 newEntry->next = newInsideList;
514 newInsideList = newEntry;
515
516 } // end for obj
517
518 //
519 // now that we have a list of objects that are in our area of influence, look through
520 // the objects that were last recorded as in our area of influence and remove any
521 // bonus we've given them (if they are within the area of effect of another tower it's
522 // OK, they'll get the bonus back again when that tower does a scan which won't be too long
523 //
524 for( ObjectTracker *curr = m_insideList; curr; curr = curr->next )
525 {
526
527 // find this entry in the new list
528 ObjectTracker *o = NULL;
529 for( o = newInsideList; o; o = o->next )
530 if( o->objectID == curr->objectID )
531 break;
532
533 // if entry wasn't there, remove the bonus from this object
534 if( o == NULL )
535 {
536
537 obj = TheGameLogic->findObjectByID( curr->objectID );
538 if( obj )
539 effectLogic( obj, FALSE, modData );
540
541 } // end if
542
543 } // end for
544
545 // delete the inside list we have recoreded
546 ObjectTracker *next;
547 while( m_insideList )
548 {
549
550 next = m_insideList->next;
551 m_insideList->deleteInstance();
552 m_insideList = next;
553
554 } // end while
555
556 // set the new inside list to the one we're recording
557 m_insideList = newInsideList;
558
559} // end doScan

572// ------------------------------------------------------------------------------------------------

577void PropagandaTowerBehavior::xfer( Xfer *xfer )
578{
579
580 // version
581 XferVersion currentVersion = 1;
582 XferVersion version = currentVersion;
583 xfer->xferVersion( &version, currentVersion );
584
585 // extend base class
586 UpdateModule::xfer( xfer );
587
588 // last scan frame
589 xfer->xferUnsignedInt( &m_lastScanFrame );
590
591 // inside list tracking
592 ObjectTracker *trackerEntry;
593 UnsignedShort insideCount = 0;
594 for( trackerEntry = m_insideList; trackerEntry; trackerEntry = trackerEntry->next )
595 insideCount++;
596 xfer->xferUnsignedShort( &insideCount );
597 if( xfer->getXferMode() == XFER_SAVE )
598 {
599
600 // write all entries
601 for( trackerEntry = m_insideList; trackerEntry; trackerEntry = trackerEntry->next )
602 {
603
604 // object id
605 xfer->xferObjectID( &trackerEntry->objectID );
606
607 } // end for
608
609 } // end if, save
610 else
611 {
612
613 // sanity
614 if( m_insideList != NULL )
615 {
616
617 DEBUG_CRASH(( "PropagandaTowerBehavior::xfer - m_insideList should be empty but is not\n" ));
618 throw SC_INVALID_DATA;
619
620 } // end if
621
622 // read all entries
623 for( UnsignedShort i = 0; i < insideCount; ++i )
624 {
625
626 // allocate new tracker entry and tie to list
627 trackerEntry = newInstance(ObjectTracker);
628 trackerEntry->next = m_insideList;
629 m_insideList = trackerEntry;
630
631 // read object id
632 xfer->xferObjectID( &trackerEntry->objectID );
633
634 } // end for i
635
636 } // end else, load
637
638} // end xfer
