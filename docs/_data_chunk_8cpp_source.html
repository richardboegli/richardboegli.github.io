278void DataChunkOutput::openDataChunk( char *name, DataChunkVersionType ver )
279{
280 // allocate (or get existing) ID from the table of m_contents
281 UnsignedInt id = m_contents.allocateID( AsciiString(name) );
282
283 // allocate a new chunk and place it on top of the chunk stack
284 OutputChunk *c = newInstance(OutputChunk);
285 c->next = m_chunkStack;
286 m_chunkStack = c;
287 m_chunkStack->id = id;
288
289 // store the chunk ID
290 ::fwrite( (const char *)&id, sizeof(UnsignedInt), 1, m_tmp_file );
291
292 // store the chunk version number
293 ::fwrite( (const char *)&ver, sizeof(DataChunkVersionType), 1, m_tmp_file );
294
295 // remember this m_tmp_file position so we can write the real data size later
296 c->filepos = ::ftell(m_tmp_file);
297#ifdef VERBOSE
298 DEBUG_LOG(("Writing chunk %s at %d (%x)\n", name, ::ftell(m_tmp_file), ::ftell(m_tmp_file)));
299#endif
300 // store a placeholder for the data size
301 Int dummy = 0xffff;
302 ::fwrite( (const char *)&dummy, sizeof(Int), 1, m_tmp_file );
303}
304
305void DataChunkOutput::closeDataChunk( void )
306{
307 if (m_chunkStack == NULL)
308 {
309 // TODO: Throw exception
310 return;
311 }
312
313 // remember where we are
314 Int here = ::ftell(m_tmp_file);
315
316 // rewind to store the data size
317 ::fseek(m_tmp_file, m_chunkStack->filepos , SEEK_SET);
318
319 // compute data size (not including the actual data size itself)
320 Int size = here - m_chunkStack->filepos - sizeof(Int);
321
322 // store the data size
323 ::fwrite( (const char *)&size, sizeof(Int) , 1, m_tmp_file );
324
325 // go back to where we were
326 ::fseek(m_tmp_file, here , SEEK_SET);
327
328 // pop the chunk off the stack
329 OutputChunk *c = m_chunkStack;
330#ifdef VERBOSE
331 DEBUG_LOG(("Closing chunk %s at %d (%x)\n", m_contents.getName(c->id).str(), here, here));
332#endif
333 m_chunkStack = m_chunkStack->next;
334 c->deleteInstance();
335}

381void DataChunkOutput::writeDict( const Dict& d )
382{
383 UnsignedShort len = d.getPairCount();
384 ::fwrite( (const char *)&len, sizeof(UnsignedShort) , 1, m_tmp_file );
385 for (int i = 0; i < len; i++)
386 {
387 NameKeyType k = d.getNthKey(i);
388 AsciiString kname = TheNameKeyGenerator->keyToName(k);
389
390 Int keyAndType = m_contents.allocateID(kname);
391 keyAndType <<= 8;
392 Dict::DataType t = d.getNthType(i);
393 keyAndType |= (t & 0xff);
394 writeInt(keyAndType);
395
396 switch(t)
397 {
398 case Dict::DICT_BOOL:
399 writeByte(d.getNthBool(i)?1:0);
400 break;
401 case Dict::DICT_INT:
402 writeInt(d.getNthInt(i));
403 break;
404 case Dict::DICT_REAL:
405 writeReal(d.getNthReal(i));
406 break;
407 case Dict::DICT_ASCIISTRING:
408 writeAsciiString(d.getNthAsciiString(i));
409 break;
410 case Dict::DICT_UNICODESTRING:
411 writeUnicodeString(d.getNthUnicodeString(i));
412 break;
413 default:
414 DEBUG_CRASH(("impossible"));
415 break;
416 }
417 }

506// output the table of m_contents to a binary m_tmp_file stream
507void DataChunkTableOfContents::write( OutputStream &s )
508{
509 Mapping *m;
510 unsigned char len;
511
512 Byte tag[4]={'C','k', 'M', 'p'}; // Chunky height map. jba.
513 s.write(tag,sizeof(tag));
514
515 // output number of elements in the table
516 s.write( (void *)&this->m_listLength, sizeof(Int) );
517
518 // output symbol table
519 for( m=this->m_list; m; m=m->next )
520 {
521 len = m->name.getLength();
522 s.write( (char *)&len, sizeof(unsigned char) );
523 s.write( (char *)m->name.str(), len);
524 s.write( (char *)&m->id, sizeof(UnsignedInt) );
525 }
526}

528// read the table of m_contents from a binary m_tmp_file stream
529// TODO: Should this reset the symbol table?
530// Append symbols to table
531void DataChunkTableOfContents::read( ChunkInputStream &s)
532{
533 Int count, i;
534 UnsignedInt maxID = 0;
535 unsigned char len;
536 Mapping *m;
537
538 Byte tag[4]={'x','x', 'x', 'x'}; // Chunky height map. jba.
539 s.read(tag,sizeof(tag));
540 if (tag[0] != 'C' || tag[1] != 'k' || tag[2] != 'M' || tag[3] != 'p') {
541 return; // Don't throw, may happen with legacy files.
542 }
543
544 // get number of symbols in table
545 s.read( (char *)&count, sizeof(Int) );
546
547 for( i=0; i<count; i++ )
548 {
549 // allocate new id mapping
550 m = newInstance(Mapping);
551
552 // read string length
553 s.read( (char *)&len, sizeof(unsigned char) );
554
555 // allocate and read in string
556 if (len>0) {
557 char *str = m->name.getBufferForRead(len);
558 s.read( str, len );
559 str[len] = '\000';
560 }
561
562 // read id
563 s.read( (char *)&m->id, sizeof(UnsignedInt) );
564
565 // prepend to list
566 m->next = this->m_list;
567 this->m_list = m;
568
569 this->m_listLength++;
570
571 // track max ID used
572 if (m->id > maxID)
573 maxID = m->id;
574 }
575 m_headerOpened = count > 0 && !s.eof();
576
577 // adjust next ID so no ID's are reused
578 this->m_nextID = max( this->m_nextID, maxID+1 );
579}

625// parse the chunk stream using registered parsers
626// it is assumed that the file position is at the start of a data chunk
627// (it can be inside a parent chunk) when parse is called.
628Bool DataChunkInput::parse( void *userData )
629{
630 AsciiString label;
631 AsciiString parentLabel;
632 DataChunkVersionType ver;
633 UserParser *parser;
634 Bool scopeOK;
635 DataChunkInfo info;
636
637 // If the header wasn't a chunk table of contents, we can't parse.
638 if (!m_contents.isOpenedForRead()) {
639 return false;
640 }
641
642 // if we are inside a data chunk right now, get its name
643 if (m_chunkStack)
644 parentLabel = m_contents.getName( m_chunkStack->id );
645
646 while( atEndOfFile() == false )
647 {
648 if (m_chunkStack) { // If we are parsing chunks in a chunk, check current length.
649 if (m_chunkStack->dataLeft < CHUNK_HEADER_BYTES) {
650 DEBUG_ASSERTCRASH( m_chunkStack->dataLeft==0, ("Unexpected extra data in chunk."));
651 break;
652 }
653 }
654 // open the chunk
655 label = openDataChunk( &ver );
656 if (atEndOfFile()) { // FILE * returns eof after you read past end of file, so check.
657 break;
658 }
659
660 // find a registered parser for this chunk
661 for( parser=m_parserList; parser; parser=parser->next )
662 {
663 // chunk labels must match
664 if ( parser->label == label )
665 {
666 // make sure parent name (scope) also matches
667 scopeOK = true;
668
669 if (parentLabel != parser->parentLabel)
670 scopeOK = false;
671
672 if (scopeOK)
673 {
674 // m_tmp_file out the chunk info and call the user parser
675 info.label = label;
676 info.parentLabel = parentLabel;
677 info.version = ver;
678 info.dataSize = getChunkDataSize();
679
680 if (parser->parser( *this, &info, userData ) == false)
681 return false;
682 break;
683 }
684 }
685 }
686
687 // close chunk (and skip to end if need be)
688 closeDataChunk();
689 }
690
691 return true;
692}

721AsciiString DataChunkInput::openDataChunk(DataChunkVersionType *ver )
722{
723 // allocate a new chunk and place it on top of the chunk stack
724 InputChunk *c = newInstance(InputChunk);
725 c->id = 0;
726 c->version = 0;
727 c->dataSize = 0;
728 //DEBUG_LOG(("Opening data chunk at offset %d (%x)\n", m_file->tell(), m_file->tell()));
729 // read the chunk ID
730 m_file->read( (char *)&c->id, sizeof(UnsignedInt) );
731 decrementDataLeft( sizeof(UnsignedInt) );
732
733 // read the chunk version number
734 m_file->read( (char *)&c->version, sizeof(DataChunkVersionType) );
735 decrementDataLeft( sizeof(DataChunkVersionType) );
736
737 // read the chunk data size
738 m_file->read( (char *)&c->dataSize, sizeof(Int) );
739 decrementDataLeft( sizeof(Int) );
740
741 // all of the data remains to be read
742 c->dataLeft = c->dataSize;
743 c->chunkStart = m_file->tell();
744
745 *ver = c->version;
746
747 c->next = m_chunkStack;
748 m_chunkStack = c;
749 if (this->atEndOfFile()) {
750 return (AsciiString(""));
751 }
752 return m_contents.getName( c->id );
753}

756void DataChunkInput::closeDataChunk( void )
757{
758 if (m_chunkStack == NULL)
759 {
760 // TODO: Throw exception

763
764 if (m_chunkStack->dataLeft > 0)
765 {
766 // skip past the remainder of this chunk
767 m_file->absoluteSeek( m_file->tell()+m_chunkStack->dataLeft );
768 decrementDataLeft( m_chunkStack->dataLeft );
769
770 }
771
772 // pop the chunk off the stack
773 InputChunk *c = m_chunkStack;
774 m_chunkStack = m_chunkStack->next;
775 c->deleteInstance();
776}

832Bool DataChunkInput::atEndOfChunk( void )
833{
834 if (m_chunkStack)

907Dict DataChunkInput::readDict()
908{
909 UnsignedShort len;
910 DEBUG_ASSERTCRASH(m_chunkStack->dataLeft>=sizeof(UnsignedShort), ("Read past end of chunk."));
911 m_file->read( &len, sizeof(UnsignedShort) );
912 decrementDataLeft( sizeof(UnsignedShort) );
913 DEBUG_ASSERTCRASH(m_chunkStack->dataLeft>=len, ("Read past end of chunk."));
914
915 Dict d(len);
916
917 for (int i = 0; i < len; i++)
918 {
919 Int keyAndType = readInt();
920 Dict::DataType t = (Dict::DataType)(keyAndType & 0xff);
921 keyAndType >>= 8;
922
923 AsciiString kname = m_contents.getName(keyAndType);
924 NameKeyType k = TheNameKeyGenerator->nameToKey(kname);
925
926 switch(t)
927 {
928 case Dict::DICT_BOOL:
929 d.setBool(k, readByte() ? true : false);
930 break;
931 case Dict::DICT_INT:
932 d.setInt(k, readInt());
933 break;
934 case Dict::DICT_REAL:
935 d.setReal(k, readReal());
936 break;
937 case Dict::DICT_ASCIISTRING:
938 d.setAsciiString(k, readAsciiString());
939 break;
940 case Dict::DICT_UNICODESTRING:
941 d.setUnicodeString(k, readUnicodeString());
942 break;
943 default:
944 throw ERROR_CORRUPT_FILE_FORMAT;
945 break;
946 }
947 }
948
949 return d;
950}
