118void ObjectOptions::updateLabel()
119{
120 CWnd *pLabel = GetDlgItem(IDC_OBJECT_NAME);
121 if (pLabel) {
122 pLabel->SetWindowText(m_currentObjectName);
123 }
124
125 CComboBox *list = (CComboBox*)GetDlgItem(IDC_OWNINGTEAM);
126 list->ResetContent();
127
128 Bool found = false;
129 AsciiString defTeamName;
130 MapObject *pCur = getCurMapObject();
131 if (!pCur) {
132 // No valid selection. Just return.
133 return;
134 }
135 if (pCur)
136 {
137 const ThingTemplate* tt = pCur->getThingTemplate();
138 if (tt)
139 {
140 AsciiString defPlayerSide = tt->getDefaultOwningSide();
141 Int i = findSideListEntryWithPlayerOfSide(defPlayerSide);
142 if (i >= 0)
143 {
144 defTeamName.set("team");
145 defTeamName.concat(TheSidesList->getSideInfo(i)->getDict()->getAsciiString(TheKey_playerName));
146 found = true;
147 }
148 }
149 else
150 {
151 // put it on neutral
152 defTeamName.set("team");
153 found = true;
154 }
155 }
156
157 // flag the preview for redraw also
158 if (pCur && pCur->getThingTemplate())
159 {
160 m_objectPreview.SetThingTemplate(pCur->getThingTemplate());
161 }
162 else
163 {
164 m_objectPreview.SetThingTemplate(NULL);
165 }
166 m_objectPreview.Invalidate();
167
168 Int neutral = -1;
169 Int sel = -1;
170 for (int i = 0; i < TheSidesList->getNumTeams(); i++)
171 {
172 Dict *d = TheSidesList->getTeamInfo(i)->getDict();
173 AsciiString name = d->getAsciiString(TheKey_teamName);
174 DEBUG_ASSERTCRASH(!name.isEmpty(),("bad"));
175
176 if (name == defTeamName)
177 sel = i;
178
179 if (name == "team")
180 {
181 name = "(neutral)"; // aka, neutral
182 neutral = i;
183 }
184 list->AddString(name.str());
185 }
186 DEBUG_ASSERTCRASH(TheSidesList->getNumTeams() == 0 || neutral != -1, ("must have a neutral"));
187 if (sel == -1)
188 {
189 DEBUG_ASSERTCRASH(defTeamName.isEmpty(), ("owning team not found, using neutral"));
190 sel = neutral;
191 }
192 if (sel == -1)
193 {
194 DEBUG_ASSERTCRASH(!pCur || TheSidesList->getNumTeams()==0,("hmm, should not happen"));
195 m_curOwnerName.clear();
196 }
197 else
198 {
199 DEBUG_ASSERTCRASH(pCur,("hmm, should not happen"));
200 m_curOwnerName = TheSidesList->getTeamInfo(sel)->getDict()->getAsciiString(TheKey_teamName);
201 }
202 list->SetCurSel(sel);
203}

379HTREEITEM ObjectOptions::findOrAdd(HTREEITEM parent, const char *pLabel)
380{
381 TVINSERTSTRUCT ins;
382 char buffer[_MAX_PATH];
383 ::memset(&ins, 0, sizeof(ins));
384 HTREEITEM child = m_objectTreeView.GetChildItem(parent);
385 while (child != NULL) {
386 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
387 ins.item.hItem = child;
388 ins.item.pszText = buffer;
389 ins.item.cchTextMax = sizeof(buffer)-2;
390 m_objectTreeView.GetItem(&ins.item);
391 if (strcmp(buffer, pLabel) == 0) {
392 return(child);
393 }
394 child = m_objectTreeView.GetNextSiblingItem(child);
395 }
396
397 // not found, so add it.
398 ::memset(&ins, 0, sizeof(ins));
399 ins.hParent = parent;
400 ins.hInsertAfter = TVI_SORT;
401 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
402 ins.item.lParam = -1;
403 ins.item.pszText = (char*)pLabel;
404 ins.item.cchTextMax = strlen(pLabel);
405 child = m_objectTreeView.InsertItem(&ins);
406 return(child);
407}

414HTREEITEM ObjectOptions::_FindOrDont(const char* pLabel, HTREEITEM startPoint)
415{
416 std::list<HTREEITEM> itemsToEx;
417 itemsToEx.push_back(startPoint);
418
419 while (itemsToEx.front()) {
420 char buffer[_MAX_PATH];
421 HTREEITEM hItem = itemsToEx.front();
422 itemsToEx.pop_front();
423
424 if (!m_objectTreeView.ItemHasChildren(hItem)) {
425 TVITEM item;
426 memset(&item, 0, sizeof(item));
427
428 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT;
429 item.hItem = hItem;
430 item.pszText = buffer;
431 item.cchTextMax = sizeof(buffer)-2;
432 m_objectTreeView.GetItem(&item);
433
434 char* strToTest = strrchr(pLabel, '/');
435// if (strstr((strToTest ? strToTest : pLabel), buffer))
436 if (strcmp((strToTest ? strToTest : pLabel), buffer) == 0)
437 {
438 return hItem;
439 }
440 } else {
441 // add the first child, the others will be caught by the adding of the siblings
442 itemsToEx.push_back(m_objectTreeView.GetChildItem(hItem));
443 } // Always add the first sibling, if it exists
444
445 if (m_objectTreeView.GetNextSiblingItem(hItem)) {
446 itemsToEx.push_back(m_objectTreeView.GetNextSiblingItem(hItem));
447 }
448 }
449 return NULL;
450}

456void ObjectOptions::addObject( MapObject *mapObject, const char *pPath,
457 Int terrainNdx, HTREEITEM parent )
458{
459 char buffer[ _MAX_PATH ];
460 const char *leafName = NULL;
461
462 // sanity
463 if( mapObject == NULL )
464 return;
465
466 //
467 // if we have an thing template in mapObject, we've read it from the new INI database,
468 // we will sort those items into the tree based on properties of the template that
469 // make it easier for us to browse when building levels
470 //
471 // Feel free to reorganize how this tree is constructed from the template
472 // data at will, whatever makes it easier for design
473 //
474 const ThingTemplate *thingTemplate = mapObject->getThingTemplate();
475 if( thingTemplate )
476 {
477
478 // first check for test sorted objects
479 if( thingTemplate->getEditorSorting() == ES_TEST )
480 parent = findOrAdd( parent, "TEST" );
481
482 // first sort by side, either create or find the tree item with matching side name
483 AsciiString side = thingTemplate->getDefaultOwningSide();
484 DEBUG_ASSERTCRASH( !side.isEmpty(), ("NULL default side in template\n") );
485 strcpy( buffer, side.str() );
486 parent = findOrAdd( parent, buffer );
487
488 // next tier uses the editor sorting that design can specify in the INI
489 for( EditorSortingType i = ES_FIRST;
490 i < ES_NUM_SORTING_TYPES;
491 i = (EditorSortingType)(i + 1) )
492 {
493
494 if( thingTemplate->getEditorSorting() == i )
495 {
496
497 parent = findOrAdd( parent, EditorSortingNames[ i ] );
498 break; // exit for
499
500 } // end if
501
502 } // end for i
503
504 if( i == ES_NUM_SORTING_TYPES )
505 parent = findOrAdd( parent, "UNSORTED" );
506
507 // the leaf name is the name of the template
508 leafName = thingTemplate->getName().str();
509
510 } // end if
511 else
512 {
513
514 // all these old entries we will put in a tree for legacy GDF items
515 parent = findOrAdd( parent, "**TEST MODELS" );
516
517 Int i=0;
518 leafName = pPath;
519 while (pPath[i] && i<sizeof(buffer)) {
520 if (pPath[i] == 0) {
521 return;
522 }
523 if (pPath[i] == '/') {
524 pPath+= i+1;
525 i = 0;
526 }
527 buffer[i] = pPath[i];
528 i++;
529 }
530
531 if( i > 0 )
532 {
533 buffer[ i ] = 0;
534 leafName = buffer;
535
536 } // end if
537
538 } // end else if
539
540 // add to the tree view
541 if( leafName )
542 {
543 TVINSERTSTRUCT ins;
544
545 ::memset(&ins, 0, sizeof(ins));
546 ins.hParent = parent;
547 ins.hInsertAfter = TVI_SORT;
548 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
549 ins.item.lParam = terrainNdx;
550 ins.item.pszText = (char*)leafName;
551 ins.item.cchTextMax = strlen(leafName)+2;
552 m_objectTreeView.InsertItem(&ins);
553
554 }
555
556}

559Bool ObjectOptions::setObjectTreeViewSelection(HTREEITEM parent, Int selection)
560{
561 TVITEM item;
562 char buffer[NAME_MAX_LEN];
563 ::memset(&item, 0, sizeof(item));
564 HTREEITEM child = m_objectTreeView.GetChildItem(parent);
565 while (child != NULL) {
566 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT;
567 item.hItem = child;
568 item.pszText = buffer;
569 item.cchTextMax = sizeof(buffer)-2;
570 m_objectTreeView.GetItem(&item);
571 if (item.lParam == selection) {
572 m_objectTreeView.SelectItem(child);
573 return(true);
574 }
575 if (setObjectTreeViewSelection(child, selection))
576 {
577 strcpy(m_currentObjectName, buffer);
578 updateLabel();
579 return(true);
580 }
581 child = m_objectTreeView.GetNextSiblingItem(child);
582 }
583 return(false);
584}

587BOOL ObjectOptions::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
588{
589 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
590 if (pHdr->hdr.hwndFrom == m_objectTreeView.m_hWnd) {
591
592 if (pHdr->hdr.code == TVN_ITEMEXPANDED) {
593 if (pHdr->action == TVE_COLLAPSE) {
594 TVITEM item;
595 ::memset(&item, 0, sizeof(item));
596 item.mask = TVIF_STATE;
597 item.hItem = pHdr->itemOld.hItem;
598 m_objectTreeView.GetItem(&item);
599 item.state &= (~TVIS_EXPANDEDONCE);
600 item.mask = TVIF_STATE;
601 m_objectTreeView.SetItem(&item);
602 }
603 }
604 if (pHdr->hdr.code == TVN_SELCHANGED) {
605 char buffer[NAME_MAX_LEN];
606 HTREEITEM hItem = m_objectTreeView.GetSelectedItem();
607 TVITEM item;
608 ::memset(&item, 0, sizeof(item));
609 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT|TVIF_STATE;
610 item.hItem = hItem;
611 item.pszText = buffer;
612 item.cchTextMax = sizeof(buffer)-2;
613 m_objectTreeView.GetItem(&item);
614 if (item.lParam >= 0) {
615 m_currentObjectIndex = item.lParam;
616 strcpy(m_currentObjectName, buffer);
617 } else if (m_objectTreeView.ItemHasChildren(item.hItem)) {
618 strcpy(m_currentObjectName, "No Selection");
619 m_currentObjectIndex = -1;
620 }
621 updateLabel();
622 }
623 }
624
625 return CDialog::OnNotify(wParam, lParam, pResult);
626}

654MapObject *ObjectOptions::duplicateCurMapObjectForPlace(const Coord3D* loc, Real angle, Bool checkPlayers)
655{
656 MapObject *pCur = getCurMapObject();
657 if (pCur)
658 {
659
660 Bool found = false;
661 const ThingTemplate* tt = pCur->getThingTemplate();
662 if (checkPlayers)
663 {
664 AsciiString defaultTeam("team");
665
666 AsciiString objectTeamName = m_curOwnerName;
667 if (objectTeamName != defaultTeam) {
668 TeamsInfo *teamInfo = TheSidesList->findTeamInfo(objectTeamName);
669 if (teamInfo) {
670 AsciiString teamOwner = teamInfo->getDict()->getAsciiString(TheKey_teamOwner);
671 SidesInfo* pSide = TheSidesList->findSideInfo(teamOwner);
672 if (pSide) {
673 found = true;
674 }
675 }
676 }
677 if (!found && tt)
678 {
679 AsciiString defPlayerSide = tt->getDefaultOwningSide();
680 Int si = findSideListEntryWithPlayerOfSide(defPlayerSide);
681
682 found = (si >= 0);
683 if (!found)
684 {
685 AddPlayerDialog addPlyr(pCur->getThingTemplate()->getDefaultOwningSide());
686 if (addPlyr.DoModal() == IDOK)
687 {
688 for (int i = 0; i < TheSidesList->getNumSides(); i++)
689 {
690 AsciiString playerTmplName = TheSidesList->getSideInfo(i)->getDict()->getAsciiString(TheKey_playerFaction);
691 if (playerTmplName == addPlyr.getAddedSide())
692 {
693 m_curOwnerName.set("team");
694 m_curOwnerName.concat(TheSidesList->getSideInfo(i)->getDict()->getAsciiString(TheKey_playerName));
695 found = true;
696 break;
697 }
698 }
699 }
700 }
701 }
702 else
703 {
704 if (!found) {
705 // neutral
706 m_curOwnerName.set("team");
707 found = true;
708 }
709 }
710 } else {
711 found = true;
712 }
713 if (found)
714 {
715 MapObject *pNew = newInstance(MapObject)( *loc, pCur->getName(), angle,
716 pCur->getFlags(), pCur->getProperties(),
717 pCur->getThingTemplate() );
718 pNew->getProperties()->setAsciiString(TheKey_originalOwner, m_curOwnerName);
719 pNew->setColor(pCur->getColor());
720 return pNew;
721 }
722 }
723 AfxMessageBox("Unable to add object.");
724 return(NULL);
725}
