88UpdateSleepTime TechBuildingBehavior::update( void )
89{
90 Object *us = getObject();
91 const TechBuildingBehaviorModuleData* d = getTechBuildingBehaviorModuleData();
92 Bool captured = false;
93
94 // update our model condition for the captured status
95 Player *player = us->getControllingPlayer();
96 if( player && player->isPlayableSide() )
97 {
98 us->setModelConditionState( MODELCONDITION_CAPTURED );
99 captured = true;
100 }
101 else
102 {
103 us->clearModelConditionState( MODELCONDITION_CAPTURED );
104 captured = false;
105 }
106
107 // if we have a pulse fx, and are owned, sleep only a little while, otherwise sleep forever
108 if (d->m_pulseFX != NULL && d->m_pulseFXRate > 0 && captured)
109 {
110 FXList::doFXObj( d->m_pulseFX, us );
111 return UPDATE_SLEEP(d->m_pulseFXRate);
112 }
113 else
114 {
115 // now sleep forever my dear
116 return UPDATE_SLEEP_FOREVER;
117 }
118}
