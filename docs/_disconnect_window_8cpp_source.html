189WindowMsgHandledType DisconnectControlSystem( GameWindow *window, UnsignedInt msg,
190 WindowMsgData mData1, WindowMsgData mData2 )
191{
192 switch( msg )
193 {
194
195 //---------------------------------------------------------------------------------------------
196 case GBM_SELECTED:
197 {
198
199 GameWindow *control = (GameWindow *) mData1;
200 Int controlID = control->winGetWindowId();
201
202 if (controlID == buttonQuitID) {
203 TheDisconnectMenu->quitGame();
204 buttonQuitWindow->winEnable(FALSE);
205 } else if (controlID == buttonVotePlayer1ID) {
206 TheDisconnectMenu->voteForPlayer(0);
207 buttonVotePlayer1Window->winEnable(FALSE);
208 } else if (controlID == buttonVotePlayer2ID) {
209 TheDisconnectMenu->voteForPlayer(1);
210 buttonVotePlayer2Window->winEnable(FALSE);
211 } else if (controlID == buttonVotePlayer3ID) {
212 TheDisconnectMenu->voteForPlayer(2);
213 buttonVotePlayer3Window->winEnable(FALSE);
214 } else if (controlID == buttonVotePlayer4ID) {
215 TheDisconnectMenu->voteForPlayer(3);
216 buttonVotePlayer4Window->winEnable(FALSE);
217 } else if (controlID == buttonVotePlayer5ID) {
218 TheDisconnectMenu->voteForPlayer(4);
219 buttonVotePlayer5Window->winEnable(FALSE);
220 } else if (controlID == buttonVotePlayer6ID) {
221 TheDisconnectMenu->voteForPlayer(5);
222 buttonVotePlayer6Window->winEnable(FALSE);
223 } else if (controlID == buttonVotePlayer7ID) {
224 TheDisconnectMenu->voteForPlayer(6);
225 buttonVotePlayer7Window->winEnable(FALSE);
226 }
227
228 break;
229
230 } // end button selected
231
232 case GEM_EDIT_DONE:
233 {
234// DEBUG_LOG(("DisconnectControlSystem - got GEM_EDIT_DONE.\n"));
235 GameWindow *control = (GameWindow *)mData1;
236 Int controlID = control->winGetWindowId();
237
238 // Take the user's input and echo it into the chat window as well as
239 // send it to the other clients on the lan
240 if ( controlID == textEntryID )
241 {
242 UnicodeString txtInput;
243
244// DEBUG_LOG(("DisconnectControlSystem - GEM_EDIT_DONE was from the text entry control.\n"));
245
246 // read the user's input
247 txtInput.set(GadgetTextEntryGetText( textEntryWindow ));
248 // Clear the text entry line
249 GadgetTextEntrySetText(textEntryWindow, UnicodeString::TheEmptyString);
250 // Clean up the text (remove leading/trailing chars, etc)
251 txtInput.trim();
252 // Echo the user's input to the chat window
253 if (!txtInput.isEmpty()) {
254// DEBUG_LOG(("DisconnectControlSystem - sending string %ls\n", txtInput.str()));
255 TheDisconnectMenu->sendChat(txtInput);
256 }
257
258 }// if ( controlID == textEntryChatID )
259 break;
260 }
261
262 //---------------------------------------------------------------------------------------------
263 default:
264 return MSG_IGNORED;
265
266 } // end switch( msg )
267
268 return MSG_HANDLED;
269
270} // end ControlBarSystem
