64// HashListClass<> and HashNodeClass<>:
65// The purpose of these templates is to allow a user to insert records into
66// a hash table that also contains a double linked list that includes every
67// record in the hash table. The functionality of the hash list mirrors
68// very closely that of the DataNode<> and List<> templates. The differences
69// between them are:
70// -User calls Add(?) to add to list instead of Add_Head() or others.
71// -A key must be set to hash off of.
72//
73// Additional features of the HashList<> over List<> are:
74// -Quick searching as long as key's are well hashed..
75// -Node can inherit user class so additional data can be put in node.
76// For example a timestamp to be set by user. Default is
77// HashDefaluatUserClass that has no data.
78// -If class T is actually a class (not a pointer), user may get a ptr
79// to class with Get_Ptr() instead of Get() that returns an INSTANCE
80// of the class.
81// -Flags to tell user if record was just created or record was just put
82// into a list (user must clear flags if desired.)
83// -HashListClass<> can create and destroy nodes so that the nodes do not
84// need to be put as members in classes. This allows an object to be
85// put into an infinite (subject to memory constraints) lists.
86// -User may specify number of hash node slots (default is A_LARGE_PRIME_NUMBER).
87
88// You can typedef these templates with the the following:
89// typedef HashListClass<TestHashClass *> HashTableClass;
90// typedef HashNodeClass<TestHashClass *> HashTableNodeClass;
91// or if using a UserClass
92// typedef HashListClass<TestHashClass *, 4, UserClass> HashTableClass;
93// typedef HashNodeClass<TestHashClass *, UserClass> HashTableNodeClass;
94
95
96template<class T, class U> class HashNodeClass;
97template<class T, class U> class HashNodeFriendClass;
98template<class T, class U, int NumHashValues> class HashListClass;

105// to allow the user to override it with his/her own class so that additional
106// data may be stored in the node about the object.
107// Since the object might be in various lists, this data is specific to
108// the list that it in.

115template<class T, class U = class HashDefaultUserClass>
116class HashNodeClass : public DataNode<HashNodeClass<T,U> *>, public U
117{
118 public:
119 // Creation when key and record are known.
120 HashNodeClass(T record, unsigned key) :
121 Flags(0),
122 Record(record),
123 Key(key)
124 {
125 DataNode<HashNodeClass<T,U> *>::Set(this);
126 Flag.NewRecord = true;
127 Flag.KeySet = true;
128 }
129
130 // Creation when key and record are to be set later.
131 // node may not be put into list until the key is set.
132 HashNodeClass() :
133 Flags(0),
134 Key(-1)
135 {
136 DataNode<HashNodeClass<T,U> *>::Set(this);
137 Flag.NewRecord = true;
138 }
139
140 virtual ~HashNodeClass() {
141 // Assert is raised when user tries to delete HashNodeClass still in a list
142 // NOTE: This might be raised because user expected node to clean
143 // itself up when deleted. Currently, this feature is not available
144 // because it would require an additional pointer to the HashListClass
145 // that the node is a member of.
146 assert(!Flag.InList);
147 }
148
149 // Get next and prev record given a node.
150 // A node may be retrieved from: HashListClass::Find();
151 HashNodeClass<T,U> *Next() {
152 return((HashNodeClass<T,U> *) DataNode<HashNodeClass<T,U> *>::Next());
153 }
154 HashNodeClass<T,U> *Prev() {
155 return((HashNodeClass<T,U> *) DataNode<HashNodeClass<T,U> *>::Prev());
156 }
157
158 // User may use these if he desires not to have to check for valid.
159 HashNodeClass<T,U> *Next_Valid() {
160 HashNodeClass<T,U> *next = Next();
161 if (next && next->Is_Valid()) {
162 return(next);
163 }
164 return(NULL);
165 }
166 HashNodeClass<T,U> *Prev_Valid() {
167 HashNodeClass<T,U> *prev = Prev();
168 if (prev && prev->Is_Valid()) {
169 return(prev);
170 }
171 return(NULL);
172 }
173
174 // Get record that is in hash table.
175 T Get_Record() {return(Record);}
176 T Get_Record_Ptr() {return(&Record);}
177
178 // Overload DataNode::Get() to get the record we are pointing to so
179 // that it behaves how people expect List to operate.
180 // To Get HashNodeClass - it is this.
181 T Get() {return(Record);}
182
183 // Get a pointer to Record to avoid a copy construction of Record if
184 // it is an instance of a class that does not want to be copied all
185 // over the place.
186 T *Get_Ptr() {return(&Record);}
187
188 // To mirror usage of DataNode().
189 void Set(T record) {Record = record;}
190
191 void Set_Key(unsigned key) {
192 // Can't change key once in a list, we would never find it.
193 assert(!Flag.InList);
194 Flag.KeySet = true;
195 Key = key;
196 }
197
198 // Get key of record for has table.
199 unsigned Get_Key() {
200 assert(Flag.KeySet);
201 return (Key);
202 }
203
204 // Enable user to know if a record has just been added to the list.
205 // It stays set until user clears it - not required.
206 int Is_New_Record() {return(Flag.NewRecord);}
207 void Clear_New_Record() {Flag.NewRecord = false;}
208
209 // Each time object gets put into a list NewInList is set.
210 // Stays set until user clears - not required.
211 int Is_New_In_List() {return(Flag.NewInList);}
212 void Clear_New_In_List() {Flag.NewInList = false;}
213
214 bool Is_In_List() {return(Flag.InList);}
215 bool Is_Key_Set() {return(Flag.KeySet);}
216
217
218 protected:
219 // Record that we are keeping in hash table.
220 // This is commonly a pointer to the actual record.
221 T Record;
222
223 // Key that user gives to identify record.
224 // WARNING: Duplicate keys are undefined behavior.
225 unsigned Key;
226
227 // Unioned flags so Flags can be initialized to 0.
228 union {
229 struct {
230 // When record collides in hash list, these ?InTable
231 // tell when to stop looking for object in hash entry.
232 unsigned FirstInTable:1;
233 unsigned LastInTable:1;
234
235 // This is set when record is first created. It is up to user
236 // to clear if he uses flag.
237 unsigned NewRecord:1;
238
239 // This is set when record is first put in list. It is up to user
240 // to clear if he uses flag.
241 unsigned NewInList:1;
242
243 // Certain functions may not happen if node is InList (in hash table)
244 // These include delete, Set_Key(), and possibly others.
245 unsigned InList:1;
246
247 // This is set so HashListClass knows if it should delete the
248 // node when removed from it's list.
249 unsigned ListCreated:1;
250
251 // Make sure someone does not insert a node into a list
252 // without setting it's key.
253 unsigned KeySet:1;
254 } Flag;
255 unsigned Flags;
256 };
257
258 private:
259 // Not something the casual user can call.
260 void Unlink(void) {
261 DataNode<HashNodeClass<T,U> *>::Unlink();
262 }
263
264 friend class HashNodeFriendClass<T,U>;
265};

304// The sole purpose of using NumHashValues as a template is to make it easier to
305// view in a debugger. If it were an allocated array of pointers, the debugger does
306// not easily show each element in the array.
307template<class T, class U = class HashDefaultUserClass, int NumHashValues = A_LARGE_PRIME_NUMBER>
308class HashListClass : protected HashNodeFriendClass<T,U>
309{
310 public:
311 // Creation and destruction.
312 HashListClass() :
313 List(),NumRecords(0), UsedValues(0)
314 {
315 memset(HashTable, 0, sizeof(HashTable));
316 }
317 ~HashListClass() {Delete();}
318
319 // Get first node in list so user can itterate through it.
320 // CAUTION: Do not destroy pointer returned or itterated though -
321 // you must call the Remove function on it.
322 // An assert() will come up if you try.
323 HashNodeClass<T,U> *First() {
324 return((HashNodeClass<T,U> *)List.First());
325 }
326 HashNodeClass<T,U> *First_Valid() {
327 return((HashNodeClass<T,U> *)List.First_Valid());
328 }
329 HashNodeClass<T,U> *Last() {
330 return((HashNodeClass<T,U> *)List.Last());
331 }
332 HashNodeClass<T,U> *Last_Valid() {
333 return((HashNodeClass<T,U> *)List.Last_Valid());
334 }
335
336
337 // Add a record to hash creating new HashNodeClass.
338 HashNodeClass<T,U> *Add(T record, unsigned key);
339
340 // Add an existing node not in a list to this list.
341 // This allows closer behavior to List class.
342 HashNodeClass<T,U> *Add(HashNodeClass<T,U> *node);
343
344 // Search for record based off key in hash table.
345 HashNodeClass<T,U> *Find(unsigned key);
346
347 // Use this remove if you know the node * already. It is quicker.
348 void Remove(HashNodeClass<T,U> *node);
349
350 // Use this remove if you need to do a find to get the node.
351 // OK if key does not exist in table.
352 bool Remove(unsigned key);
353
354 // Removes all nodes in list.
355 void Delete(void) {while (First()->Is_Valid()) Remove(First());}
356
357 // Move contents of one list to another (this becomes empty).
358 void Move_To(HashListClass<T,U> *newlist);
359
360 // So we always do it the same way.
361 unsigned Hash_Value(unsigned key) {
362 return(unsigned)(key % NumHashValues);
363 }
364
365 // Total number of records in hash table.
366 int Num_Records() {
367 return(NumRecords);
368 }
369 // how many hash values are used.
370 // The closer this number is to NumRecords, the
371 // better hashing going on.
372 int Num_Used_Values() {
373 return(UsedValues);
374 }
375
376 protected:
377 // This list can be walked from start to end of all objects in the list.
378 // User can use Get_First() to get first (valid) record in linked list.
379 List<DataNode<T> *> List;
380
381 // Points to first record in List that has the right 'value'.
382 // Must check LaastInTable to see if end of records in has list.
383 HashNodeClass<T,U> *HashTable[NumHashValues];
384
385 // Number of records we have in class.
386 int NumRecords;
387 int UsedValues;
388
389};

395/***********************************************************************************************
396 * HashListClass::Add -- Add record to hash list. *
397 * *
398 * INPUT: *
399 * *
400 * OUTPUT: *
401 * *
402 * WARNINGS: *
403 * *
404 * HISTORY: *
405 * 05/26/1999 SKB : Created. *
406 *=============================================================================================*/
407template <class T, class U, int NumHashValues>
408HashNodeClass<T,U> *HashListClass<T, U, NumHashValues>::Add(HashNodeClass<T,U> *node)
409{
410 // Get our index into the hash table.
411 int hashidx = Hash_Value(node->Get_Key());
412 assert(hashidx < NumHashValues);
413
414 // Tell system we are being put in the list (and make sure we weren't already in one).
415 assert(!node->Is_In_List());
416 Set_In_List(node);
417
418 // Get first hit in table.
419 HashNodeClass<T,U> *first = HashTable[hashidx];
420 if (first) {
421 // This will add the new node right after the first node in the list.
422 first->Link(node);
423
424 // If we are the first one to collide, then we will be at end of list.
425 // otherwise, we are at the end.
426 if (Is_Last(first)) {
427 Clear_Last(first);
428 Set_Last(node);
429 }
430 } else {
431 // We are first hit at hash index. Put ourselves at start of
432 List.Add_Head(node);
433 Set_First(node);
434 Set_Last(node);
435 HashTable[hashidx] = node;
436 UsedValues++;
437 }
438 NumRecords++;
439 return (node);
440}

467/***********************************************************************************************
468 * HashListClass::Find -- Find record in hash table. *
469 * *
470 * INPUT: *
471 * *
472 * OUTPUT: *
473 * *
474 * WARNINGS: *
475 * *
476 * HISTORY: *
477 * 05/26/1999 SKB : Created. *
478 *=============================================================================================*/
479template <class T, class U, int NumHashValues>
480HashNodeClass<T,U> *HashListClass<T, U, NumHashValues>::Find(unsigned key)
481{
482 int hashidx = Hash_Value(key);
483 assert(hashidx < NumHashValues);
484
485 HashNodeClass<T,U> *cur = HashTable[hashidx];
486 if (cur) for(;;) {
487 // Have we found our match?
488 if (cur->Get_Key() == key) {
489 return(cur);
490 }
491
492 // We have to find record before the end of the list.
493 if (Is_Last(cur)) {
494 break;
495 }
496
497 // Continue on because we appear to have some collisions.
498 cur = cur->Next();
499
500 // The end of a list should always be marked with Is_Last().
501 assert(cur);
502 assert(cur->Is_Valid());
503 }
504 return(NULL);
505
506}

508/***********************************************************************************************
509 * HashListClass::Remove_Node -- Remove node from hash table. *
510 * Protected function for usage by Remove() routines. *
511 * *
512 * INPUT: *
513 * HashNodeClass<T,U> *node - node to remove. *
514 * *
515 * OUTPUT: *
516 * *
517 * WARNINGS: *
518 * *
519 * HISTORY: *
520 * 06/02/1999 SKB : Created. *
521 * 01/19/2000 SKB : Remove the node with Unlink. *
522 *=============================================================================================*/
523template <class T, class U, int NumHashValues>
524void HashListClass<T, U, NumHashValues>::Remove(HashNodeClass<T,U> *node)
525{
526 assert(node);
527 assert(node->Is_In_List());
528
529 // See if entry is first in list for hash value.
530 if (Is_First(node)) {
531
532 int hashidx = Hash_Value(node->Get_Key());
533 assert(HashTable[hashidx]);
534
535 // SKB: 2/20/01 - clear incase inserted in new list later.
536 Clear_First(node);
537
538 // is it only one in hash index?
539 if (Is_Last(node)) {
540 // SKB: 2/20/01 - clear incase inserted in new list later.
541 Clear_Last(node);
542 HashTable[hashidx] = NULL;
543 UsedValues--;
544 } else {
545 HashTable[hashidx] = node->Next_Valid();
546 Set_First(HashTable[hashidx]);
547 }
548
549 } else if (Is_Last(node)) {
550 // SKB: 2/20/01 - clear incase inserted in new list later.
551 Clear_Last(node);
552
553 // There is one before us because it failed above.
554 Set_Last(node->Prev());
555 }
556
557
558 // All done with you. Set flag to avoid assert.
559 Clear_In_List(node);
560
561 // Delete node if it was created by the this class instead of by the user.
562 if (Is_List_Created(node)) {
563 delete node;
564 } else {
565 // Remove it from the link list at least so it can be put in another.
566 Unlink(node);
567 }
568 NumRecords--;
569}

571/***********************************************************************************************
572 * HashListClass::Remove -- Remove record from hash table. *
573 * *
574 * INPUT: *
575 * unsigned key - key to find node by. *
576 * *
577 * OUTPUT: *
578 * *
579 * WARNINGS: *
580 * *
581 * HISTORY: *
582 * 05/26/1999 SKB : Created. *
583 *=============================================================================================*/
584template <class T, class U, int NumHashValues>
585bool HashListClass<T, U, NumHashValues>::Remove(unsigned key)
586{
587 int hashidx = Hash_Value(key);
588 assert(hashidx < NumHashValues);
589
590 HashNodeClass<T,U> *node = Find(key);
591 if (node) {
592 Remove(node);
593 return(true);
594 }
595 return(false);
596}

598/***********************************************************************************************
599 * HashListClass::Move_To -- Move nodes from one list to another. *
600 * *
601 * INPUT: *
602 * *
603 * OUTPUT: *
604 * *
605 * WARNINGS: *
606 * *
607 * HISTORY: *
608 * 06/04/1999 SKB : Created. *
609 *=============================================================================================*/
610template <class T, class U, int NumHashValues>
611void HashListClass<T, U, NumHashValues>::Move_To(HashListClass<T,U> *newlist)
612{
613 HashNodeClass<T,U> *node = First_Valid();
614 while (node) {
615 HashNodeClass<T,U> *next = node->Next_Valid();
616
617 // If list created node, clear that fact for a moment so it
618 // will not get deleted in the remove.
619 if (Is_List_Created(node)) {
620 Clear_List_Created(node);
621 Remove(node);
622 newlist->Add(node);
623 Set_List_Created(node);
624 } else {
625 Remove(node);
626 newlist->Add(node);
627 }
628 node = next;
629 }
630}
