137void StatsCollector::collectMsgStats( const GameMessage *msg )
138{
139 // We only care about our own messages.
140 if(ThePlayerList->getLocalPlayer()->getPlayerIndex() != msg->getPlayerIndex())
141 return;
142
143 switch (msg->getType())
144 {
145 case GameMessage::MSG_QUEUE_UNIT_CREATE:
146 case GameMessage::MSG_DOZER_CONSTRUCT:
147 case GameMessage::MSG_DOZER_CONSTRUCT_LINE:
148 {
149 ++m_buildCommands;
150 break;
151 }
152 }
153
154}

158void StatsCollector::collectUnitCountStats( void )
159{
160
161 for(Object *obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject())
162 {
163
164 if((!(obj->isKindOf(KINDOF_INFANTRY) || obj->isKindOf(KINDOF_VEHICLE))) || ( obj->isNeutralControlled()) ||(obj->getControllingPlayer()->getSide().compare("Civilian") == 0))
165 continue;
166
167 if(obj->getControllingPlayer()->isLocalPlayer())
168 {
169 ++m_playerUnits;
170 }
171 else
172 {
173 ++m_AIUnits;
174 }
175 }

179// call every frame and only do stuff when our time is up
180//=============================================================================
181void StatsCollector::update( void )
182{
183 if(m_lastUpdate + (TheGlobalData->m_playStats * LOGICFRAMES_PER_SECOND) > TheGameLogic->getFrame())
184 return;
185
186 collectUnitCountStats();
187
188 if(m_isScrolling)
189 {
190 m_scrollTime += TheGameLogic->getFrame() - m_scrollBeginTime;
191 m_scrollBeginTime = TheGameLogic->getFrame();
192 }
193
194 m_timeCount += TheGlobalData->m_playStats;
195 writeStatInfo();
196
197 zeroOutStats();
198
199 m_lastUpdate = TheGameLogic->getFrame(); //timeGetTime();
200
201}

222void StatsCollector::writeFileEnd( void )
223{
224 //open the file
225 FILE *f = fopen(m_statsFileName.str(), "a");
226 if(!f)
227 {
228 DEBUG_ASSERTCRASH(f, ("Unable to open file %s to write", m_statsFileName.str()));
229 return;
230 }
231
232 m_timeCount += (TheGameLogic->getFrame() - m_lastUpdate) / LOGICFRAMES_PER_SECOND;
233 writeStatInfo();
234 fprintf(f, "---------------------------------------------------\n");
235
236 // Time
237 struct tm *newTime;
238 time_t aclock;
239 time( &aclock );
240 newTime = localtime( &aclock );
241 fprintf(f, "End Time:\t%s\n",asctime(newTime) );
242
243 fprintf(f, "=KEY===============================================\n");
244 fprintf(f, "Time* = The Time Interval\n");
245 fprintf(f, "BC = Build Commands\n");
246 fprintf(f, "MC = Move Commands\n");
247 fprintf(f, "AC = Attack Commands\n");
248 fprintf(f, "SMC = Scroll Map Commands\n");
249 fprintf(f, "ST* = Scroll Time in Seconds\n");
250 fprintf(f, "OC = Other Commands (N/A)\n");
251 fprintf(f, "$$$ = Local Player's Cash Amount\n");
252 fprintf(f, "#PU = # of Player's Units\n");
253 fprintf(f, "#AIU = # of AI's Units\n");
254 fprintf(f, "===================================================\n");
255 fprintf(f, "* Times are in Game Seconds which are based off of frames. Current fps is set to %d\n", LOGICFRAMES_PER_SECOND);
256
257#if defined(_DEBUG) || defined(_INTERNAL)
258 if (TheGlobalData->m_benchmarkTimer > 0)
259 {
260 fprintf(f, "\n*** BENCHMARK MODE STATS ***\n");
261 fprintf(f, " Frames = %d\n", TheGameLogic->getFrame()-m_startFrame);
262 fprintf(f, "Seconds = %d\n", TheGlobalData->m_benchmarkTimer);
263 fprintf(f, " FPS = %.2f\n", ((Real)TheGameLogic->getFrame()-(Real)m_startFrame)/(Real)TheGlobalData->m_benchmarkTimer);
264 }
265#endif
266
267 fclose(f);
268
269}

303// create the filename based off of map time and date
304//=============================================================================
305void StatsCollector::createFileName( void )
306{
307 m_statsFileName.clear();
308 // Date and Time
309 char datestr[256] = "";
310 time_t longTime;
311 struct tm *curtime;
312 time(&longTime);
313 curtime = localtime(&longTime);
314 strftime(datestr, 256, "_%b%d_%I%M%p", curtime);
315// const MapMetaData *m = TheMapCache->findMap(TheGlobalData->m_mapName);
316 AsciiString name = TheGlobalData->m_mapName;
317 const char *fname = name.reverseFind('\\');
318 if (fname)
319 name = fname+1;
320 name.removeLastChar(); // p
321 name.removeLastChar(); // a
322 name.removeLastChar(); // m
323 name.removeLastChar(); // .
324 m_statsFileName.clear();
325#if defined(_DEBUG) || defined(_INTERNAL)
326 if (TheGlobalData->m_saveStats)
327 {
328 m_statsFileName.set(TheGlobalData->m_baseStatsDir);
329 m_statsFileName.concat(statsDir);
330
331 if (TheNetwork)
332 {
333 if (TheLAN)
334 {
335 GameInfo *game = TheLAN->GetMyGame();
336 AsciiString players;
337 AsciiString full;
338 AsciiString fullPlusNum;
339 for (Int i=0; i<MAX_SLOTS; ++i)
340 {
341 GameSlot *slot = game->getSlot(i);
342 if (slot && slot->isHuman())
343 {
344 AsciiString player;
345 player.format("%ls_", slot->getName().str());
346 players.concat(player);
347 }
348 }
349 full.format("%s%s_%d_%d", players.str(), name.str(), game->getSeed(), game->getLocalSlotNum());
350 AsciiString testString;
351 testString.format("%s%s.txt", m_statsFileName.str(), full.str());
352 m_statsFileName = testString;
353 }
354 }
355 else
356 {
357 m_statsFileName.format("%s%s%s.txt",statsDir, name.str(),datestr);
358 }
359 }
360 else
361#endif
362 {
363 m_statsFileName.format("%s%s%s.txt",statsDir, name.str(),datestr);
364 }
365}
