71Real Tool::calcRoundBlendFactor(CPoint center, Int x, Int y, Int brushWidth, Int featherWidth)
72{
73 Real offset = 0;
74 if (brushWidth&1) offset = 0.5;
75 const Real CLOSE_ENOUGH = 0.05f; // We are working on an integer grid.
76
77 Real dx = abs(center.x+offset-x);
78 Real dy = abs(center.y+offset-y);
79
80 Real dist = (Real)sqrt(dx*dx+dy*dy);
81
82 if (dist <= (brushWidth/2.0f)+CLOSE_ENOUGH) return (1.0f);
83
84 dist -= (brushWidth/2.0f);
85
86 if (featherWidth < 1) {
87 return(0);
88 }
89
90 if (dist <= featherWidth) {
91 return (featherWidth-dist)/featherWidth;
92 }
93
94 return(0);
95}

100Real Tool::calcSquareBlendFactor(CPoint center, Int x, Int y, Int brushWidth, Int featherWidth)
101{
102 Real offset = 0;
103 if (brushWidth&1) offset = 0.5;
104
105 Real dx = abs(center.x+offset-x);
106 Real dy = abs(center.y+offset-y);
107
108 Real dist = dx;
109 if (dy>dist) dist = dy;
110
111 if (dist <= (brushWidth/2.0f)) return (1.0f);
112
113 dist -= (brushWidth/2.0f);
114
115 if (featherWidth < 1) {
116 return(0);
117 }
118
119 if (dist <= featherWidth) {
120 return (featherWidth-dist)/featherWidth;
121 }
122
123 return(0);
124}
