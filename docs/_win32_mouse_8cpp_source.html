57UnsignedByte Win32Mouse::getMouseEvent( MouseIO *result, Bool flush )
58{
59
60 // if there is nothing here there is no event data to do
61 if( m_eventBuffer[ m_nextGetIndex ].msg == 0 )
62 return MOUSE_NONE;
63
64 // translate the win32 mouse message to our own system
65 translateEvent( m_nextGetIndex, result );
66
67 // remove this event from the buffer by setting msg to zero
68 m_eventBuffer[ m_nextGetIndex ].msg = 0;
69
70 //
71 // our next get index will now be advanced to the next index, wrapping at
72 // the mad
73 //
74 m_nextGetIndex++;
75 if( m_nextGetIndex >= Mouse::NUM_MOUSE_EVENTS )
76 m_nextGetIndex = 0;
77
78 // got event OK and all done with this one
79 return MOUSE_OK;
80
81} // end getMouseEvent

86void Win32Mouse::translateEvent( UnsignedInt eventIndex, MouseIO *result )
87{
88 UINT msg = m_eventBuffer[ eventIndex ].msg;
89 WPARAM wParam = m_eventBuffer[ eventIndex ].wParam;
90 LPARAM lParam = m_eventBuffer[ eventIndex ].lParam;
91 UnsignedInt frame;
92
93 //
94 // get the current input frame from the client, if we don't have
95 // a client (like in the GUI editor) we just use frame 1 so it
96 // registers with the system
97 //
98 if( TheGameClient )
99 frame = TheGameClient->getFrame();
100 else
101 frame = 1;
102
103 // set these to defaults
104 result->leftState = result->middleState = result->rightState = MBS_Up;
105 result->leftFrame = result->middleFrame = result->rightFrame = 0;
106 result->pos.x = result->pos.y = result->wheelPos = 0;
107
108 // Time is the same for all events
109 result->time = m_eventBuffer[ eventIndex ].time;
110
111 switch( msg )
112 {
113
114 // ------------------------------------------------------------------------
115 case WM_LBUTTONDOWN:
116 {
117
118 result->leftState = MBS_Down;
119 result->leftFrame = frame;
120 result->pos.x = LOWORD( lParam );
121 result->pos.y = HIWORD( lParam );
122 break;
123
124 } // end left button down
125
126 // ------------------------------------------------------------------------
127 case WM_LBUTTONUP:
128 {
129
130 result->leftState = MBS_Up;
131 result->leftFrame = frame;
132 result->pos.x = LOWORD( lParam );
133 result->pos.y = HIWORD( lParam );
134 break;
135
136 } // end left button up
137
138 // ------------------------------------------------------------------------
139 case WM_LBUTTONDBLCLK:
140 {
141
142 result->leftState = MBS_DoubleClick;
143 result->leftFrame = frame;
144 result->pos.x = LOWORD( lParam );
145 result->pos.y = HIWORD( lParam );
146 break;
147
148 } // end left button double click
149
150 // ------------------------------------------------------------------------
151 case WM_MBUTTONDOWN:
152 {
153
154 result->middleState = MBS_Down;
155 result->middleFrame = frame;
156 result->pos.x = LOWORD( lParam );
157 result->pos.y = HIWORD( lParam );
158 break;
159
160 } // end middle button down
161
162 // ------------------------------------------------------------------------
163 case WM_MBUTTONUP:
164 {
165
166 result->middleState = MBS_Up;
167 result->middleFrame = frame;
168 result->pos.x = LOWORD( lParam );
169 result->pos.y = HIWORD( lParam );
170 break;
171
172 } // end middle button up
173
174 // ------------------------------------------------------------------------
175 case WM_MBUTTONDBLCLK:
176 {
177
178 result->middleState = MBS_DoubleClick;
179 result->middleFrame = frame;
180 result->pos.x = LOWORD( lParam );
181 result->pos.y = HIWORD( lParam );
182 break;
183
184 } // end middle button double click
185
186 // ------------------------------------------------------------------------
187 case WM_RBUTTONDOWN:
188 {
189
190 result->rightState = MBS_Down;
191 result->rightFrame = frame;
192 result->pos.x = LOWORD( lParam );
193 result->pos.y = HIWORD( lParam );
194 break;
195
196 } // end right button down
197
198 // ------------------------------------------------------------------------
199 case WM_RBUTTONUP:
200 {
201
202 result->rightState = MBS_Up;
203 result->rightFrame = frame;
204 result->pos.x = LOWORD( lParam );
205 result->pos.y = HIWORD( lParam );
206 break;
207
208 } // end right button up
209
210 // ------------------------------------------------------------------------
211 case WM_RBUTTONDBLCLK:
212 {
213
214 result->rightState = MBS_DoubleClick;
215 result->rightFrame = frame;
216 result->pos.x = LOWORD( lParam );
217 result->pos.y = HIWORD( lParam );
218 break;
219
220 } // end right button double click
221
222 // ------------------------------------------------------------------------
223 case WM_MOUSEMOVE:
224 {
225
226 result->pos.x = LOWORD( lParam );
227 result->pos.y = HIWORD( lParam );
228 break;
229
230 } // end mouse move
231
232 // ------------------------------------------------------------------------
233 case 0x020A: // WM_MOUSEWHEEL
234 {
235 POINT p;
236
237 // translate the screen mouse position to be relative to the application window
238 p.x = LOWORD( lParam );
239 p.y = HIWORD( lParam );
240 ScreenToClient( ApplicationHWnd, &p );
241
242 // note the short cast here to keep signed information in tact
243 result->wheelPos = (Short)HIWORD( wParam );
244 result->pos.x = p.x;
245 result->pos.y = p.y;
246 break;
247
248 } // end mouse wheel
249
250 // ------------------------------------------------------------------------
251 default:
252 {
253
254 DEBUG_CRASH(( "translateEvent: Unknown Win32 mouse event [%d,%d,%d]\n",
255 msg, wParam, lParam ));
256 return;
257
258 } // end default
259
260 } // end switch on message at event index in buffer
261
262} // end translateEvent

340void Win32Mouse::addWin32Event( UINT msg, WPARAM wParam, LPARAM lParam, DWORD time )
341{
342
343 //
344 // we can only add this event if our next free index does not already
345 // have an event in it, if it does ... our buffer is full and this input
346 // event will be lost
347 //
348 if( m_eventBuffer[ m_nextFreeIndex ].msg != 0 )
349 return;
350
351 // add to this index
352 m_eventBuffer[ m_nextFreeIndex ].msg = msg;
353 m_eventBuffer[ m_nextFreeIndex ].wParam = wParam;
354 m_eventBuffer[ m_nextFreeIndex ].lParam = lParam;
355 m_eventBuffer[ m_nextFreeIndex ].time = time;
356
357 // wrap index at max
358 m_nextFreeIndex++;
359 if( m_nextFreeIndex >= Mouse::NUM_MOUSE_EVENTS )
360 m_nextFreeIndex = 0;
361
362} // end addWin32Event

376void Win32Mouse::initCursorResources(void)
377{
378 for (Int cursor=FIRST_CURSOR; cursor<NUM_MOUSE_CURSORS; cursor++)
379 {
380 for (Int direction=0; direction<m_cursorInfo[cursor].numDirections; direction++)
381 { if (!cursorResources[cursor][direction] && !m_cursorInfo[cursor].textureName.isEmpty())
382 { //this cursor has never been loaded before.
383 char resourcePath[256];
384 //Check if this is a directional cursor
385 if (m_cursorInfo[cursor].numDirections > 1)
386 sprintf(resourcePath,"data\\cursors\\%s%d.ANI",m_cursorInfo[cursor].textureName.str(),direction);
387 else
388 sprintf(resourcePath,"data\\cursors\\%s.ANI",m_cursorInfo[cursor].textureName.str());
389
390 cursorResources[cursor][direction]=LoadCursorFromFile(resourcePath);
391 DEBUG_ASSERTCRASH(cursorResources[cursor][direction], ("MissingCursor %s\n",resourcePath));
392 }
393 }
394// SetCursor(cursorResources[cursor][m_directionFrame]);
395 }
396}
