75// radioButtonPropertiesCallback ==============================================
77//=============================================================================
78static LRESULT CALLBACK radioButtonPropertiesCallback( HWND hWndDialog,
79 UINT message,
80 WPARAM wParam,
81 LPARAM lParam )
82{
83 Int returnCode;
84
85 //
86 // handle any common messages between all property dialogs cause they
87 // are designed to have controls doing the same functionality
88 // and names
89 //
90 if( HandleCommonDialogMessages( hWndDialog, message,
91 wParam, lParam, &returnCode ) == TRUE )
92 return returnCode;
93
94 switch( message )
95 {
96
97 // ------------------------------------------------------------------------
98 case WM_COMMAND:
99 {
100// Int notifyCode = HIWORD( wParam ); // notification code
101 Int controlID = LOWORD( wParam ); // control ID
102// HWND hWndControl = (HWND)lParam; // control window handle
103
104 switch( controlID )
105 {
106
107 // --------------------------------------------------------------------
108 case BUTTON_CLEAR_GROUP:
109 {
110
111 SetDlgItemInt( hWndDialog, COMBO_GROUP, 0, FALSE );
112 break;
113
114 } // end clear group
115
116 // --------------------------------------------------------------------
117 case IDOK:
118 {
119 GameWindow *window = TheEditor->getPropertyTarget();
120
121 // sanity
122 if( window )
123 {
124 ImageAndColorInfo *info;
125
126 // save the common properties
127 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
128 break;
129
130 // save the image and color data
131 // ----------------------------------------------------------------
132 info = GetStateInfo( RADIO_ENABLED );
133 GadgetRadioSetEnabledImage( window, info->image );
134 GadgetRadioSetEnabledColor( window, info->color );
135 GadgetRadioSetEnabledBorderColor( window, info->borderColor );
136
137 info = GetStateInfo( RADIO_ENABLED_UNCHECKED_BOX );
138 GadgetRadioSetEnabledUncheckedBoxImage( window, info->image );
139 GadgetRadioSetEnabledUncheckedBoxColor( window, info->color );
140 GadgetRadioSetEnabledUncheckedBoxBorderColor( window, info->borderColor );
141
142 info = GetStateInfo( RADIO_ENABLED_CHECKED_BOX );
143 GadgetRadioSetEnabledCheckedBoxImage( window, info->image );
144 GadgetRadioSetEnabledCheckedBoxColor( window, info->color );
145 GadgetRadioSetEnabledCheckedBoxBorderColor( window, info->borderColor );
146
147 // ----------------------------------------------------------------
148 info = GetStateInfo( RADIO_DISABLED );
149 GadgetRadioSetDisabledImage( window, info->image );
150 GadgetRadioSetDisabledColor( window, info->color );
151 GadgetRadioSetDisabledBorderColor( window, info->borderColor );
152
153 info = GetStateInfo( RADIO_DISABLED_UNCHECKED_BOX );
154 GadgetRadioSetDisabledUncheckedBoxImage( window, info->image );
155 GadgetRadioSetDisabledUncheckedBoxColor( window, info->color );
156 GadgetRadioSetDisabledUncheckedBoxBorderColor( window, info->borderColor );
157
158 info = GetStateInfo( RADIO_DISABLED_CHECKED_BOX );
159 GadgetRadioSetDisabledCheckedBoxImage( window, info->image );
160 GadgetRadioSetDisabledCheckedBoxColor( window, info->color );
161 GadgetRadioSetDisabledCheckedBoxBorderColor( window, info->borderColor );
162
163 // ----------------------------------------------------------------
164 info = GetStateInfo( RADIO_HILITE );
165 GadgetRadioSetHiliteImage( window, info->image );
166 GadgetRadioSetHiliteColor( window, info->color );
167 GadgetRadioSetHiliteBorderColor( window, info->borderColor );
168
169 info = GetStateInfo( RADIO_HILITE_UNCHECKED_BOX );
170 GadgetRadioSetHiliteUncheckedBoxImage( window, info->image );
171 GadgetRadioSetHiliteUncheckedBoxColor( window, info->color );
172 GadgetRadioSetHiliteUncheckedBoxBorderColor( window, info->borderColor );
173
174 info = GetStateInfo( RADIO_HILITE_CHECKED_BOX );
175 GadgetRadioSetHiliteCheckedBoxImage( window, info->image );
176 GadgetRadioSetHiliteCheckedBoxColor( window, info->color );
177 GadgetRadioSetHiliteCheckedBoxBorderColor( window, info->borderColor );
178
179 // save group
180 Int group = GetDlgItemInt( hWndDialog, COMBO_GROUP, NULL, FALSE );
181 Int screen = TheNameKeyGenerator->nameToKey( AsciiString(TheEditor->getSaveFilename()) );
182 GadgetRadioSetGroup( window, group, screen );
183
184 } // end if
185
186 DestroyWindow( hWndDialog );
187 break;
188
189 } // end OK
190
191 // --------------------------------------------------------------------
192 case IDCANCEL:
193 {
194
195 DestroyWindow( hWndDialog );
196 break;
197
198 } // end cancel
199
200 } // end switch( LOWORD( wParam ) )
201
202 return 0;
203
204 } // end of WM_COMMAND
205
206 // ------------------------------------------------------------------------
207 case WM_CLOSE:
208 {
209
210 DestroyWindow( hWndDialog );
211 return 0;
212
213 } // end close
214
215 // ------------------------------------------------------------------------
216 default:
217 return 0;
218
219 } // end of switch
220
221} // end radioButtonPropertiesCallback

228// loadExistingGroupsCombo ====================================================
230//=============================================================================
231static void loadExistingGroupsCombo( HWND combo, GameWindow *window )
232{
233
234 // sanity
235 if( combo == NULL )
236 return;
237
238 // end of recursion
239 if( window == NULL )
240 return;
241
242 // if this is a radio button get the group
243 if( BitTest( window->winGetStyle(), GWS_RADIO_BUTTON ) )
244 {
245 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
246 char buffer[ 64 ];
247
248 // convert to string
249 sprintf( buffer, "%d", radioData->group );
250
251 // only add it not already there
252 if( SendMessage( combo, CB_FINDSTRINGEXACT, -1, (LPARAM)buffer ) == CB_ERR )
253 SendMessage( combo, CB_ADDSTRING, 0, (LPARAM)buffer );
254
255 } // end if
256
257 // search our children
258 GameWindow *child;
259 for( child = window->winGetChild(); child; child = child->winGetNext() )
260 loadExistingGroupsCombo( combo, child );
261
262 // search the next in line
263 loadExistingGroupsCombo( combo, window->winGetNext() );
264
265} // end loadExistingGroupsCombo

267// InitRadioButtonPropertiesDialog ============================================
269//=============================================================================
270HWND InitRadioButtonPropertiesDialog( GameWindow *window )
271{
272 HWND dialog;
273
274 // create the dialog box
275 dialog = CreateDialog( TheEditor->getInstance(),
276 (LPCTSTR)RADIO_BUTTON_PROPERTIES_DIALOG,
277 TheEditor->getWindowHandle(),
278 (DLGPROC)radioButtonPropertiesCallback );
279 if( dialog == NULL )
280 return NULL;
281
282 // do the common initialization
283 CommonDialogInitialize( window, dialog );
284
285 //
286 // store in the image and color table the values for this putton
287 //
288 const Image *image;
289 Color color, borderColor;
290
291 // --------------------------------------------------------------------------
292 image = GadgetRadioGetEnabledImage( window );
293 color = GadgetRadioGetEnabledColor( window );
294 borderColor = GadgetRadioGetEnabledBorderColor( window );
295 StoreImageAndColor( RADIO_ENABLED, image, color, borderColor );
296
297 image = GadgetRadioGetEnabledUncheckedBoxImage( window );
298 color = GadgetRadioGetEnabledUncheckedBoxColor( window );
299 borderColor = GadgetRadioGetEnabledUncheckedBoxBorderColor( window );
300 StoreImageAndColor( RADIO_ENABLED_UNCHECKED_BOX, image, color, borderColor );
301
302 image = GadgetRadioGetEnabledCheckedBoxImage( window );
303 color = GadgetRadioGetEnabledCheckedBoxColor( window );
304 borderColor = GadgetRadioGetEnabledCheckedBoxBorderColor( window );
305 StoreImageAndColor( RADIO_ENABLED_CHECKED_BOX, image, color, borderColor );
306
307 // --------------------------------------------------------------------------
308 image = GadgetRadioGetDisabledImage( window );
309 color = GadgetRadioGetDisabledColor( window );
310 borderColor = GadgetRadioGetDisabledBorderColor( window );
311 StoreImageAndColor( RADIO_DISABLED, image, color, borderColor );
312
313 image = GadgetRadioGetDisabledUncheckedBoxImage( window );
314 color = GadgetRadioGetDisabledUncheckedBoxColor( window );
315 borderColor = GadgetRadioGetDisabledUncheckedBoxBorderColor( window );
316 StoreImageAndColor( RADIO_DISABLED_UNCHECKED_BOX, image, color, borderColor );
317
318 image = GadgetRadioGetDisabledCheckedBoxImage( window );
319 color = GadgetRadioGetDisabledCheckedBoxColor( window );
320 borderColor = GadgetRadioGetDisabledCheckedBoxBorderColor( window );
321 StoreImageAndColor( RADIO_DISABLED_CHECKED_BOX, image, color, borderColor );
322
323 // --------------------------------------------------------------------------
324 image = GadgetRadioGetHiliteImage( window );
325 color = GadgetRadioGetHiliteColor( window );
326 borderColor = GadgetRadioGetHiliteBorderColor( window );
327 StoreImageAndColor( RADIO_HILITE, image, color, borderColor );

336 borderColor = GadgetRadioGetHiliteCheckedBoxBorderColor( window );
337 StoreImageAndColor( RADIO_HILITE_CHECKED_BOX, image, color, borderColor );
338
339 // radio data
340 RadioButtonData *radioData = (RadioButtonData *)window->winGetUserData();
341
342 // fill the group combo box with all the other groups in the screen
343 loadExistingGroupsCombo( GetDlgItem( dialog, COMBO_GROUP ),
344 TheWindowManager->winGetWindowList() );
345
346 // set the group for this radio button
347 SetDlgItemInt( dialog, COMBO_GROUP, radioData->group, FALSE );
348
349 // select the button enabled state for display
350 SwitchToState( RADIO_ENABLED, dialog );
351
352 //
353 // initialize the dialog with values from the window
354 //
355
356 return dialog;
357
358} // end InitRadioButtonPropertiesDialog
