84void FiringTracker::shotFired(const Weapon* weaponFired, ObjectID victimID)
85{
86 UnsignedInt now = TheGameLogic->getFrame();
87 Object *me = getObject();
88 const Object *victim = TheGameLogic->findObjectByID(victimID); // May be null for ground shot
89
90 if( victim && victim->testStatus(OBJECT_STATUS_FAERIE_FIRE) )
91 {
92 if( !me->testWeaponBonusCondition(WEAPONBONUSCONDITION_TARGET_FAERIE_FIRE) )
93 {
94 // We shoot faster at guys marked thusly
95 me->setWeaponBonusCondition(WEAPONBONUSCONDITION_TARGET_FAERIE_FIRE);
96 }
97 }
98 else
99 {
100 // A ground shot or the lack of the status on the target will clear this
101 if( me->testWeaponBonusCondition(WEAPONBONUSCONDITION_TARGET_FAERIE_FIRE) )
102 {
103 me->clearWeaponBonusCondition(WEAPONBONUSCONDITION_TARGET_FAERIE_FIRE);
104 }
105 }
106
107 if( victimID == m_victimID )
108 {
109 // Shooting at the same guy
110 ++m_consecutiveShots;
111 }
112 else if( now < m_frameToStartCooldown )
113 {
114 // Switching targets within the coast time is valid, and we will not spin down
115 ++m_consecutiveShots;
116 m_victimID = victimID;
117 }
118 else
119 {
120 // Start the count over for the new guy
121 m_consecutiveShots = 1;
122 m_victimID = victimID;
123 }
124
125 // Push back the time that we should force a reload with each shot
126 UnsignedInt autoReloadDelay = weaponFired->getAutoReloadWhenIdleFrames();
127 if( autoReloadDelay > 0 )
128 m_frameToForceReload = now + autoReloadDelay;
129
130 UnsignedInt coast = weaponFired->getContinuousFireCoastFrames();
131 if (coast)
132 m_frameToStartCooldown = weaponFired->getPossibleNextShotFrame() + coast;
133 else
134 m_frameToStartCooldown = 0;
135
136 Int shotsNeededOne = weaponFired->getContinuousFireOneShotsNeeded();
137 Int shotsNeededTwo = weaponFired->getContinuousFireTwoShotsNeeded();
138
139 if( me->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN ) )
140 {
141 // Can either go up or down from here.
142 if( m_consecutiveShots < shotsNeededOne )
143 coolDown();
144 else if( m_consecutiveShots > shotsNeededTwo )
145 speedUp();
146 }
147 else if( me->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST ) )
148 {
149 // Only place I can go here from here is all the way down
150 if( m_consecutiveShots < shotsNeededTwo )
151 {
152 coolDown();
153 }
154 }
155 else
156 {
157 // Check to go up
158 if( m_consecutiveShots > shotsNeededOne )
159 speedUp();
160 }
161
162 UnsignedInt fireSoundLoopTime = weaponFired->getFireSoundLoopTime();
163 if (fireSoundLoopTime != 0)
164 {
165 // If the sound has stopped playing, then we need to re-add it.
166 if (m_frameToStopLoopingSound == 0 || !TheAudio->isCurrentlyPlaying(m_audioHandle))
167 {
168 AudioEventRTS audio = weaponFired->getFireSound();
169 audio.setObjectID(getObject()->getID());
170 m_audioHandle = TheAudio->addAudioEvent( &audio );
171 }
172 m_frameToStopLoopingSound = now + fireSoundLoopTime;
173 }
174 else
175 {
176 AudioEventRTS fireAndForgetSound = weaponFired->getFireSound();
177 fireAndForgetSound.setObjectID(getObject()->getID());
178 TheAudio->addAudioEvent(&fireAndForgetSound);
179 m_frameToStopLoopingSound = 0;
180 }
181
182
183 setWakeFrame(me, calcTimeToSleep());
184}

187UpdateSleepTime FiringTracker::update()
188{
189 //DEBUG_ASSERTCRASH(m_frameToStartCooldown != 0 || m_frameToStopLoopingSound != 0, ("hmm, should be asleep"));
190
191 UnsignedInt now = TheGameLogic->getFrame();
192
193 // I have been idle long enough that I should reload, so I do not hang around with a near empty clip forever.
194 if( m_frameToForceReload != 0 && now >= m_frameToForceReload )
195 {
196 getObject()->reloadAllAmmo(TRUE);
197 m_frameToForceReload = 0;
198 }
199
200 // If it has been too long since I fired. I have to start over.
201 // (don't call if we don't need to cool down... it's expensive!)
202
203 if (m_frameToStopLoopingSound != 0)
204 {
205 if (now >= m_frameToStopLoopingSound)
206 {
207 TheAudio->removeAudioEvent( m_audioHandle );
208 m_audioHandle = AHSV_NoSound;
209 m_frameToStopLoopingSound = 0;
210 }
211 }
212
213 if( m_frameToStartCooldown != 0 && now > m_frameToStartCooldown )
214 {
215 m_frameToStartCooldown = now + LOGICFRAMES_PER_SECOND;
216 coolDown();// if this is the coolest call to cooldown, it will set m_frameToStartCooldown to zero
217 return UPDATE_SLEEP(LOGICFRAMES_PER_SECOND);
218 }
219
220 UpdateSleepTime sleepTime = calcTimeToSleep();
221
222 return sleepTime;
223}

225//-------------------------------------------------------------------------------------------------
226UpdateSleepTime FiringTracker::calcTimeToSleep()
227{
228 // Figure out the longest amount of time we can sleep as unneeded
229
230 // If all the timers are off, then we aren't needed at all
231 if (m_frameToStopLoopingSound == 0 && m_frameToStartCooldown == 0 && m_frameToForceReload == 0)
232 return UPDATE_SLEEP_FOREVER;
233
234 // Otherwise, we need to wake up to service the shortest timer
235 UnsignedInt now = TheGameLogic->getFrame();
236 UnsignedInt sleepTime = UPDATE_SLEEP_FOREVER;
237 if( m_frameToStopLoopingSound != 0 )
238 {
239 if( m_frameToStopLoopingSound <= now )
240 sleepTime = UPDATE_SLEEP_NONE;
241 else if( (m_frameToStopLoopingSound - now) < sleepTime )
242 sleepTime = m_frameToStopLoopingSound - now;
243 }
244 if( m_frameToStartCooldown != 0 )
245 {
246 if( m_frameToStartCooldown <= now )
247 sleepTime = UPDATE_SLEEP_NONE;
248 else if( (m_frameToStartCooldown - now) < sleepTime )
249 sleepTime = m_frameToStartCooldown - now;
250 }
251 if( m_frameToForceReload != 0 )
252 {
253 if( m_frameToForceReload <= now )
254 sleepTime = UPDATE_SLEEP_NONE;
255 else if( (m_frameToForceReload - now) < sleepTime )
256 sleepTime = m_frameToForceReload - now;
257 }
258
259 return UPDATE_SLEEP(sleepTime);
260}

263void FiringTracker::speedUp()
264{
265 ModelConditionFlags clr, set;
266 Object *self = getObject();
267
268 if( self->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST ) )
269 {
270 //self->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN );
271 //self->clearModelConditionState( MODELCONDITION_CONTINUOUS_FIRE_MEAN );
272 //self->clearModelConditionState( MODELCONDITION_CONTINUOUS_FIRE_SLOW );
273 }
274 else if(self->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN ) )
275 {
276 const AudioEventRTS *soundToPlayPtr = self->getTemplate()->getPerUnitSound( "VoiceRapidFire" );
277 AudioEventRTS soundToPlay = *soundToPlayPtr;
278 soundToPlay.setObjectID( self->getID() );
279 TheAudio->addAudioEvent( &soundToPlay );
280
281 // These flags are exclusive, not cumulative
282 self->setWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST );
283 set.set(MODELCONDITION_CONTINUOUS_FIRE_FAST);
284
285 // These flags are exclusive, not cumulative
286 self->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN );
287 clr.set(MODELCONDITION_CONTINUOUS_FIRE_MEAN);
288 clr.set(MODELCONDITION_CONTINUOUS_FIRE_SLOW);
289
290
291 }
292 else
293 {
294
295 self->setWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN );
296 set.set(MODELCONDITION_CONTINUOUS_FIRE_MEAN);
297
298 // These flags are exclusive, not cumulative
299 self->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST );
300 clr.set(MODELCONDITION_CONTINUOUS_FIRE_FAST);
301 clr.set(MODELCONDITION_CONTINUOUS_FIRE_SLOW);
302
303 }
304
305 self->clearAndSetModelConditionFlags(clr, set);
306
307}

310void FiringTracker::coolDown()
311{
312 ModelConditionFlags clr, set;
313
314 if( getObject()->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST )
315 || getObject()->testWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN ))
316 {
317
318 // Straight to zero from wherever it is
319 set.set(MODELCONDITION_CONTINUOUS_FIRE_SLOW);
320
321 getObject()->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST );
322 getObject()->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN );
323 clr.set(MODELCONDITION_CONTINUOUS_FIRE_FAST);
324 clr.set(MODELCONDITION_CONTINUOUS_FIRE_MEAN);
325
326
327 }
328 else // we stop, now
329 {
330
331 getObject()->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_FAST );
332 getObject()->clearWeaponBonusCondition( WEAPONBONUSCONDITION_CONTINUOUS_FIRE_MEAN );
333 // Just chillin, nothing to change
334 clr.set(MODELCONDITION_CONTINUOUS_FIRE_FAST);
335 clr.set(MODELCONDITION_CONTINUOUS_FIRE_MEAN);
336 clr.set(MODELCONDITION_CONTINUOUS_FIRE_SLOW);
337
338 m_frameToStartCooldown = 0;
339 }
340
341 getObject()->clearAndSetModelConditionFlags(clr, set);
342
343 // Start everything over
344 m_consecutiveShots = 0;
345 m_victimID = INVALID_ID;
346}
