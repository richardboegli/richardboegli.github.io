35/*********************************************************************************************
36 * IGROptions::Init -- Class initializer. Reads from the registry *
37 * *
38 * INPUT: None *
39 * *
40 * OUTPUT: bool; Did we read everything OK? *
41 * *
42 * WARNINGS: none *
43 * *
44 * HISTORY: *
45 * 07/05/00 JeffB: Initial coding *
46 *===========================================================================================*/
47bool IGROptionsClass::Init( void )
48{
49 int size;
50 int returnValue;
51 HKEY handle;
52 char key[128];
53 unsigned long type;
54
55 valid = false;
56
57 // Load the options from the registry
58 size = sizeof( int );
59
60 // Setup the key
61 strcpy( key, WOLAPI_REG_KEY_BOTTOM );
62
63 // Get a handle to the WOLAPI entry
64 if ( RegOpenKeyEx( HKEY_LOCAL_MACHINE, key, 0, KEY_ALL_ACCESS, &handle ) == ERROR_SUCCESS ) {
65
66 // If successful, get the options
67 IGROptionsType ReadOptions = 0;
68
69 returnValue = RegQueryValueEx(handle, WOLAPI_REG_KEY_OPTIONS, NULL,
70 (unsigned long *) &type, (unsigned char *) &ReadOptions, (unsigned long *)&size);
71
72 if (returnValue == ERROR_SUCCESS) {
73
74 // If successful, and we got a DWORD, store options and set the valid flage
75 if (type == REG_DWORD) {
76 options = ReadOptions;
77 valid = true;
78 }
79 }
80
81 // Clean up
82 RegCloseKey( handle );
83 }
84 return ( valid );
85}

138/*********************************************************************************************
139 * IGROptions::Set_Options -- Set the passed options in the registry *
140 * *
141 * INPUT: Options to set *
142 * *
143 * OUTPUT: bool; Did we set the options successfully *
144 * *
145 * WARNINGS: none *
146 * *
147 * HISTORY: *
148 * 07/05/00 JeffB: Initial coding *
149 *===========================================================================================*/
150bool IGROptionsClass::Set_Options( IGROptionsType options )
151{
152 bool ReturnValue = false;
153 HKEY handle;
154 int disp;
155 char key[ 128 ];
156
157 // We don't care if it's valid, we'll MAKE it valid.
158 strcpy( key, WOLAPI_REG_KEY_BOTTOM );
159
160 // Do they have the WOLAPI key?
161 if( RegOpenKeyEx( HKEY_LOCAL_MACHINE, key, 0, KEY_ALL_ACCESS, &handle ) != ERROR_SUCCESS ) {
162
163 // If not, make the WOLAPI key
164 if( RegCreateKeyEx( HKEY_LOCAL_MACHINE, key, 0, NULL, REG_OPTION_NON_VOLATILE, KEY_ALL_ACCESS,
165 NULL, &handle, (unsigned long *)&disp ) != ERROR_SUCCESS )
166 return false;
167 }
168
169 if( RegSetValueEx( handle, WOLAPI_REG_KEY_OPTIONS, 0, REG_DWORD, (unsigned char *)&options, sizeof(int))
170 == ERROR_SUCCESS ) {
171 ReturnValue = true;
172 }
173 RegCloseKey( handle );
174
175 // Reinit the class to make sure we have these settings for later queries.
176 Init();
177
178 assert( valid == TRUE );
179
180 return ReturnValue;
181}
