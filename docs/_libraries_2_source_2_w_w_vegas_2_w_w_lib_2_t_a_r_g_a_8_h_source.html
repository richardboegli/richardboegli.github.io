53/* TGAHeader - Targa Image File header.
54 *
55 * IDLength - Size of Image ID field
56 * ColorMapType - Color map type.
57 * ImageType - Image type code.
58 * CMapStart - Color map origin.
59 * CMapLength - Color map length.
60 * CMapDepth - Depth of color map entries.
61 * XOffset - X origin of image.
62 * YOffset - Y origin of image.
63 * Width - Width of image.
64 * Height - Height of image.
65 * PixelDepth - Image pixel size
66 * ImageDescriptor - Image descriptor byte.
67 */
68typedef struct _TGAHeader
69 {
70 char IDLength;
71 char ColorMapType;
72 char ImageType;
73 short CMapStart;
74 short CMapLength;
75 char CMapDepth;
76 short XOffset;
77 short YOffset;
78 short Width;
79 short Height;
80 char PixelDepth;
81 char ImageDescriptor;
82 } TGAHeader;

84/* ImageType definiton */
85#define TGA_NOIMAGE 0 /* No image data included in file */
86#define TGA_CMAPPED 1 /* Color-mapped image data */
87#define TGA_TRUECOLOR 2 /* Truecolor image data */
88#define TGA_MONO 3 /* Monochrome image data */
89#define TGA_CMAPPED_ENCODED 9 /* Color-mapped image data (Encoded) */
90#define TGA_TRUECOLOR_ENCODED 10 /* Truecolor image data (Encoded) */
91#define TGA_MONO_ENCODED 11 /* Monochrome image data (Encoded) */

95#define TGAIDF_XORIGIN (1<<4)

126/* TGA2Footer - Targa 2.0 footer
127 *
128 * Extension - Offset to the Extension area from start of file.
129 * Developer - Offset to the Developer area from start of file.
130 * Signature - 16 byte Targa 2.0 signature "TRUEVISION-XFILE"
131 * RsvdChar - Reserved character, must be ASCII "." (period)
132 * BZST - Binary Zero String Terminator.
133 */
134typedef struct _TGA2Footer
135 {
136 long Extension;
137 long Developer;
138 char Signature[16];
139 char RsvdChar;
140 char BZST;
141 _TGA2Footer() {}
142 } TGA2Footer;
143
144/* TGA2DateStamp - A series of 3 WORD values which define the integer value
145 * for the date the image was saved.
146 *
147 * Month - Month number (1 - 12)
148 * Day - Day number (1 - 31)
149 * Year - Year number (4 digit, ie. 1989)
150 */
151typedef struct _TGA2DateStamp
152 {
153 short Month;
154 short Day;
155 short Year;
156 } TGA2DateStamp;
157
158/* TGA2TimeStamp - A series of 3 WORD values which define the integer value
159 * for the time the image was saved.
160 *
161 * Hour - Hour number, military time (0 - 23)
162 * Minute - Minute number (0 - 59)
163 * Second - Second number (0 - 59)
164 */
165typedef struct _TGA2TimeStamp
166 {
167 short Hour;
168 short Minute;
169 short Second;
170 } TGA2TimeStamp;
171
172/* TGA2SoftVer - Define the version of the software used to generate file.
173 *
174 * Number - Version number * 100
175 * Letter - Version letter
176 */
177typedef struct _TGA2SoftVer
178 {
179 short Number;
180 char Letter;
181 } TGA2SoftVer;
182
183/* TGA2Ratio - Numerator and denominator which when taken together specify
184 * a ratio.
185 *
186 * Numer - Numerator
187 * Denom - Denominator (a value of zero indicates no ratio specified)
188 */
189typedef struct _TGA2Ratio
190 {
191 short Numer;
192 short Denom;
193 } TGA2Ratio;
194
195/* TGA2Extension - Extension area, provided for additional file information.
196 * This data is pointed to by the Extension offset in the
197 * TGA2Footer.
198 *
199 * ExtSize - Extension area size. (495 bytes for 2.0)
200 * AuthName - Name of the person who created image (NULL terminated ASCII)
201 * AuthComment - Comments of the author (NULL terminated ASCII)
202 * DateStamp - Date the file was created. (See TGA2DateStamp)
203 * TimeStamp - Time the file was created. (See TGA2TimeStamp)
204 * JobName - Name of job image belongs to (NULL terminated ASCII)
205 * JobTime - Elapsed time of the job.
206 * SoftID - ID of software used to create image (NULL terminated ASCII)
207 * SoftVer - Version number of software used.
208 * KeyColor - Tranparent color value.
209 * Aspect - Pixel aspect ratio.
210 * Gamma - Fractional gamma value.
211 * ColorCor - Color correction table offset.
212 * PostStamp - Postage stamp image offset.
213 * ScanLine - Scan line table offset.
214 * Attributes - Alpha channel attributes. (Set defines below)
215 */
216typedef struct _TGA2Extension
217 {
218 short ExtSize;
219 char AuthName[41];
220 char AuthComment[324];
221 TGA2DateStamp Date;
222 TGA2TimeStamp Time;
223 char JobName[41];
224 TGA2TimeStamp JobTime;
225 char SoftID[41];
226 TGA2SoftVer SoftVer;
227 long KeyColor;
228 TGA2Ratio Aspect;
229 TGA2Ratio Gamma;
230 long ColorCor;
231 long PostStamp;
232 long ScanLine;
233 char Attributes;
234 } TGA2Extension;
235
236/* Alpha channel attributes (Extension Area) */
237#define EXTA_NOALPHA 0 /* No alpha data included */
238#define EXTA_IGNORE 1 /* Undefined alpha data, can ignore */
239#define EXTA_RETAIN 2 /* Undefined alpha data, should retain */
240#define EXTA_USEFUL 3 /* Useful alpha channel */
241#define EXTA_PREMULT 4 /* Pre-Multiplied alpha data */
