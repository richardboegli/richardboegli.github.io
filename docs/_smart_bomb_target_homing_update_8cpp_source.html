57void SmartBombTargetHomingUpdate::SetTargetPosition( const Coord3D& target )
58{
59
60 // Ensure that we have been passed a real-world location
61 DEBUG_ASSERTCRASH( target.length() > 0.0f, ("SmartBombTargetHomingUpdate::SetTargetPosition() received a zero coord") );
62 if ( ! (target.length() > 0.0f) )
63 return;
64
65 m_target.x = target.x;
66 m_target.y = target.y;
67 m_target.z = target.z;
68
69 m_targetReceived = TRUE;
70
71}

76UpdateSleepTime SmartBombTargetHomingUpdate::update( void )
77{
78
79 if ( ! m_targetReceived )
80 return UPDATE_SLEEP_NONE;
81
82 const SmartBombTargetHomingUpdateModuleData* d = getSmartBombTargetHomingUpdateModuleData();
83 if ( ! d )
84 return UPDATE_SLEEP_NONE;
85
86 Object *self = getObject();
87 if ( ! self )
88 return UPDATE_SLEEP_NONE;
89
90
91 if ( ! self->isSignificantlyAboveTerrain() )
92 return UPDATE_SLEEP_NONE;
93
94 const Coord3D *currentPos = self->getPosition();
95
96 Coord3D pos;
97 pos.zero();
98
99 Real statusCoeff = MAX( 0.0f, MIN( 1.0f, d->m_courseCorrectionScalar));
100 Real targetCoeff = 1.0f - statusCoeff;
101
102
103 pos.x = m_target.x * targetCoeff + currentPos->x * statusCoeff;
104 pos.y = m_target.y * targetCoeff + currentPos->y * statusCoeff;
105 pos.z = currentPos->z;
106
107 self->setPosition( &pos );
108
109 return UPDATE_SLEEP_NONE;
110
111}
