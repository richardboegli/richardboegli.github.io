194// FlatHeightMapRenderObjClass::ReAcquireResources
195//=============================================================================

198void FlatHeightMapRenderObjClass::ReAcquireResources(void)
199{
200 if (m_map) {
201 Int width = (m_map->getXExtent()+CELLS_PER_TILE-2)/CELLS_PER_TILE;
202 Int height = (m_map->getYExtent()+CELLS_PER_TILE-2)/CELLS_PER_TILE;
203 Int i, j;
204 Int numTiles = width*height;
205 m_tiles = new W3DTerrainBackground[numTiles];
206 m_numTiles = numTiles;
207 m_tilesWidth = width;
208 m_tilesHeight = height;
209 for (i=0; i<m_tilesWidth; i++) {
210 for (j=0; j<m_tilesHeight; j++) {
211 W3DTerrainBackground *tile = m_tiles+j*m_tilesWidth+i;
212 tile->allocateTerrainBuffers(m_map, i*CELLS_PER_TILE, j*CELLS_PER_TILE, CELLS_PER_TILE);
213 tile->setFlip(m_map);
214 }
215 }
216 IRegion2D range;
217 range.lo.x = 0;
218 range.lo.y = 0;
219 range.hi.x = m_map->getXExtent();
220 range.hi.y = m_map->getYExtent();
221 for (i=0; i<m_tilesWidth; i++) {
222 for (j=0; j<m_tilesHeight; j++) {
223 W3DTerrainBackground *tile = m_tiles+j*m_tilesWidth+i;
224 tile->doPartialUpdate(range, m_map, true);
225 }
226 }
227 }
228 BaseHeightMapRenderObjClass::ReAcquireResources();
229
230}

321 releaseTiles();
322 m_tiles = new W3DTerrainBackground[numTiles];
323 m_numTiles = numTiles;
324 m_tilesWidth = width;
325 m_tilesHeight = height;
326 for (i=0; i<m_tilesWidth; i++) {

328 W3DTerrainBackground *tile = m_tiles+j*m_tilesWidth+i;

330 tile->setFlip(pMap);
331 }
332 }
333 }
334 IRegion2D range;
335 range.lo.x = 0;
336 range.lo.y = 0;
337 range.hi.x = pMap->getXExtent();

416void FlatHeightMapRenderObjClass::updateCenter(CameraClass *camera , RefRenderObjListIterator *pLightsIterator)
417{
418#ifdef DO_UNIT_TIMINGS
419#pragma MESSAGE("*** WARNING *** DOING DO_UNIT_TIMINGS!!!!")
420 return;
421#endif
422 BaseHeightMapRenderObjClass::updateCenter(camera, pLightsIterator);
423 m_needFullUpdate = false;
424 Int i, j;
425 Int culled = 0;
426 static Int prevCulled = 0;
427 Int t2X = 0;
428 static Int prevT2X = 0;
429 Int t4X = 0;
430 static Int prevT4X = 0;
431 for (i=0; i<m_tilesWidth; i++) {
432 for (j=0; j<m_tilesHeight; j++) {
433 W3DTerrainBackground *tile = m_tiles+j*m_tilesWidth+i;
434 tile->updateCenter(camera);
435 if (tile->isCulled()) {
436 culled++;
437 }
438 Int tx = tile->getTexMultiplier();
439 if (tx==4) {
440 t4X++;
441 } else if (tx==2) {
442 t2X++;
443 }
444 }
445 }
446 if (culled!=prevCulled || t4X!=prevT4X || t2X!=prevT2X) {
447 DEBUG_LOG(("%d of %d culled, %d 4X, %d 2X.\n", culled, m_numTiles, t4X, t2X));
448 prevCulled = culled;
449 prevT2X = t2X;
450 prevT4X = t4X;
451 }
452 m_updateState = STATE_MOVING;
453}

461
462void FlatHeightMapRenderObjClass::Render(RenderInfoClass & rinfo)
463{
464 //USE_PERF_TIMER(Terrain_Render)
465
466 Int devicePasses;
467 W3DShaderManager::ShaderTypes st;
468 Bool doCloud = TheGlobalData->m_useCloudMap;
469
470 Matrix3D tm(Transform);
471 // If there are trees, tell them to draw at the transparent time to draw.
472 if (m_treeBuffer) {
473 m_treeBuffer->setIsTerrain();
474 }
475
476
477#ifdef DO_UNIT_TIMINGS
478#pragma MESSAGE("*** WARNING *** DOING DO_UNIT_TIMINGS!!!!")
479 return;
480#endif
481
482#ifdef EXTENDED_STATS
483 if (DX8Wrapper::stats.m_disableTerrain) {
484 return;
485 }
486#endif
487
488 DX8Wrapper::Set_Light_Environment(rinfo.light_environment);
489
490 // Force shaders to update.
491 m_stageTwoTexture->restore();
492 DX8Wrapper::Set_Texture(0,NULL);
493 DX8Wrapper::Set_Texture(1,NULL);
494 ShaderClass::Invalidate();
495
496 // tm.Scale(ObjSpaceExtent);
497 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
498
499
500 DX8Wrapper::Set_Material(m_vertexMaterialClass);
501 DX8Wrapper::Set_Shader(m_shaderClass);
502
503 if (TheGlobalData->m_timeOfDay == TIME_OF_DAY_NIGHT) {
504 doCloud = false;
505 }
506
507 st=W3DShaderManager::ST_FLAT_TERRAIN_BASE; //set default shader
508
509 //set correct shader based on current settings
510 if (TheGlobalData->m_useLightMap && doCloud)
511 { st=W3DShaderManager::ST_FLAT_TERRAIN_BASE_NOISE12;
512 }
513 else
514 if (TheGlobalData->m_useLightMap)
515 { //lightmap only
516 st=W3DShaderManager::ST_FLAT_TERRAIN_BASE_NOISE2;
517 }
518 else
519 if (doCloud)
520 { //cloudmap only
521 st=W3DShaderManager::ST_FLAT_TERRAIN_BASE_NOISE1;
522 }
523
524
525
526 //Find number of passes required to render current shader
527 devicePasses=W3DShaderManager::getShaderPasses(st);
528
529 if (m_disableTextures)
530 devicePasses=1; //force to 1 lighting-only pass
531
532 //Specify all textures that this shader may need.
533 W3DShaderManager::setTexture(0,m_stageZeroTexture);
534 if (m_shroud && rinfo.Additional_Pass_Count() && !m_disableTextures)
535 {
536 W3DShaderManager::setTexture(0,TheTerrainRenderObject->getShroud()->getShroudTexture());
537 }
538
539 W3DShaderManager::setTexture(1,NULL); // Set by the tile later. [3/31/2003]
540 W3DShaderManager::setTexture(2,m_stageTwoTexture); //cloud
541 W3DShaderManager::setTexture(3,m_stageThreeTexture);//noise
542 //Disable writes to destination alpha channel (if there is one)
543 if (DX8Wrapper::getBackBufferFormat() == WW3D_FORMAT_A8R8G8B8) {
544 DX8Wrapper::Set_DX8_Render_State(D3DRS_COLORWRITEENABLE,D3DCOLORWRITEENABLE_BLUE|D3DCOLORWRITEENABLE_GREEN|D3DCOLORWRITEENABLE_RED);
545 }
546
547 Int pass;
548 Int yCoordMax = 0;
549 Int yCoordMin = m_map->getXExtent();
550 Int xCoordMax = 0;
551 Int xCoordMin = m_map->getYExtent();
552 for (pass=0; pass<devicePasses; pass++) {
553 Bool disableTex = m_disableTextures;
554 if (m_disableTextures ) {
555 DX8Wrapper::Set_Shader(ShaderClass::_PresetOpaque2DShader);
556 DX8Wrapper::Set_Texture(0,NULL);
557 } else {
558 W3DShaderManager::setShader(st, pass);
559 }
560
561 Int i, j;
562 for (i=0; i<m_tilesWidth; i++) {
563 for (j=0; j<m_tilesHeight; j++) {
564 W3DTerrainBackground *tile = m_tiles+j*m_tilesWidth+i;
565 if (pass>0) {
566 disableTex = TRUE; // doing cloud/noise
567 }
568 if (!tile->isCulled()) {
569 tile->drawVisiblePolys(rinfo, disableTex);
570 if (i*CELLS_PER_TILE<xCoordMin) {
571 xCoordMin = i*CELLS_PER_TILE;
572 }
573 if (j*CELLS_PER_TILE<yCoordMin) {
574 yCoordMin = j*CELLS_PER_TILE;
575 }
576 if ((i+1)*CELLS_PER_TILE>xCoordMax) {
577 xCoordMax = (i+1)*CELLS_PER_TILE;
578 }
579 if ((j+1)*CELLS_PER_TILE>yCoordMax) {
580 yCoordMax = (j+1)*CELLS_PER_TILE;
581 }
582 }
583 }
584 }
585 }
586
587 if (pass) //shader was applied at least once?
588 W3DShaderManager::resetShader(st);
589#if 1
590
591 //Draw feathered shorelines
592 renderShoreLines(&rinfo.Camera);
593
594#ifdef DO_ROADS
595 DX8Wrapper::Set_Texture(0,NULL);
596 DX8Wrapper::Set_Texture(1,NULL);
597 m_stageTwoTexture->restore();
598
599 ShaderClass::Invalidate();
600 if (!ShaderClass::Is_Backface_Culling_Inverted()) {
601 DX8Wrapper::Set_Material(m_vertexMaterialClass);
602 if (Scene) {
603 RTS3DScene *pMyScene = (RTS3DScene *)Scene;
604 RefRenderObjListIterator pDynamicLightsIterator(pMyScene->getDynamicLights());
605 m_roadBuffer->drawRoads(&rinfo.Camera, doCloud?m_stageTwoTexture:NULL, TheGlobalData->m_useLightMap?m_stageThreeTexture:NULL,
606 m_disableTextures,xCoordMin-m_map->getBorderSizeInline(), xCoordMax-m_map->getBorderSizeInline(), yCoordMin-m_map->getBorderSizeInline(), yCoordMax-m_map->getBorderSizeInline(), &pDynamicLightsIterator);
607 }
608 }
609#endif
610
611#ifdef DO_SCORCH
612 DX8Wrapper::Set_Texture(0,NULL);
613 DX8Wrapper::Set_Texture(1,NULL);
614 m_stageTwoTexture->restore();
615
616 ShaderClass::Invalidate();
617 if (!ShaderClass::Is_Backface_Culling_Inverted()) {
618 drawScorches();
619 }
620#endif
621 DX8Wrapper::Set_Texture(0,NULL);
622 DX8Wrapper::Set_Texture(1,NULL);
623 m_stageTwoTexture->restore();
624 ShaderClass::Invalidate();
625 DX8Wrapper::Apply_Render_State_Changes();
626
627 m_bridgeBuffer->drawBridges(&rinfo.Camera, m_disableTextures, m_stageTwoTexture);
628
629 if (TheTerrainTracksRenderObjClassSystem)
630 TheTerrainTracksRenderObjClassSystem->flush();
631
632 ShaderClass::Invalidate();
633 DX8Wrapper::Apply_Render_State_Changes();
634
635 m_waypointBuffer->drawWaypoints(rinfo);
636
637 m_bibBuffer->renderBibs();
638#endif
639 // We do some custom blending, so tell the shader class to reset everything.
640 DX8Wrapper::Set_Texture(0,NULL);
641 DX8Wrapper::Set_Texture(1,NULL);
642 m_stageTwoTexture->restore();
643 ShaderClass::Invalidate();
644 DX8Wrapper::Set_Material(NULL);
645
646}
