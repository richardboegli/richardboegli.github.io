75void RampTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
76{
77 if (!((m == TRACK_L) && mIsMouseDown)) {
78 if (TheRampOptions->shouldApplyTheRamp()) {
79 // Call me now for your free ramp application!
80 applyRamp(pDoc);
81 return;
82 }
83 } else if (m == TRACK_L) {
84 Coord3D docPt;
85 pView->viewToDocCoords(viewPt, &docPt);
86 docPt.z = TheTerrainRenderObject->getHeightMapHeight(docPt.x, docPt.y, NULL);
87 mEndPoint = docPt;
88 }
89
90 drawFeedback(&mEndPoint);
91}

128void RampTool::applyRamp(CWorldBuilderDoc* pDoc)
129{
130 Coord3D bl, tl, br, tr;
131 VecHeightMapIndexes indices;
132
133 WorldHeightMapEdit *worldHeightDup = pDoc->GetHeightMap()->duplicate();
134
135 Real width = TheRampOptions->getRampWidth();
136
137 BuildRectFromSegmentAndWidth(&mStartPoint, &mEndPoint, width,
138 &bl, &tl, &br, &tr);
139
140 if (bl == br || bl == tl) {
141 return;
142 }
143
144 getAllIndexesIn(&bl, &br, &tl, &tr, 0, pDoc, &indices);
145 int indiceCount = indices.size();
146 if (indiceCount == 0) {
147 return;
148 }
149
150 /*
151 This part is pretty straightforward. Determine the U value for the shortest segment from the
152 index's actual location to the segments from mStartPoint to mEndPoint, and then apply a
153 linear gradient factor according to the height from the beginning to the end of mStartPoint
154 and mEndPoint.
155 */
156
157 for (int i = 0; i < indiceCount; ++i) {
158 Coord3D pt;
159 pDoc->getCoordFromCellIndex(indices[i], &pt);
160
161 Real uVal;
162 Coord2D start = { mStartPoint.x, mStartPoint.y };
163 Coord2D end = { mEndPoint.x, mEndPoint.y };
164 Coord2D pt2D = { pt.x, pt.y };
165
166 ShortestDistancePointToSegment2D(&start, &end, &pt2D, NULL, NULL, &uVal);
167 Real height = mStartPoint.z + uVal * (mEndPoint.z - mStartPoint.z);
168
169 worldHeightDup->setHeight(indices[i].x, indices[i].y, (UnsignedByte) (height / MAP_HEIGHT_SCALE));
170 }
171
172 IRegion2D partialRange = {0,0,0,0};
173 pDoc->updateHeightMap(worldHeightDup, false, partialRange);
174
175 WBDocUndoable *pUndo = new WBDocUndoable(pDoc, worldHeightDup);
176 pDoc->AddAndDoUndoable(pUndo);
177 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
178 REF_PTR_RELEASE(worldHeightDup);
179
180
181 // Once we've applied the ramp, its no longer a mutable thing, so blow away the feedback
182 DrawObject::setDoRampFeedback(false);
183}
