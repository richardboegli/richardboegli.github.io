64MapObject *ScorchOptions::getSingleSelectedScorch(void)
65{
66 MapObject *theMapObj = NULL;
67// Bool found = false;
68 Int selCount=0;
69 MapObject *pMapObj;
70 for (pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext()) {
71 if (pMapObj->isSelected()) {
72 if (pMapObj->isScorch()) {
73 theMapObj = pMapObj;
74 }
75 selCount++;
76 }
77 }
78 if (selCount==1 && theMapObj) {
79 return theMapObj;
80 }
81
82 return(NULL);
83}

85void ScorchOptions::updateTheUI(void)
86{
87 m_updating = true;
88 MapObject *theMapObj = getSingleSelectedScorch();
89 CString str;
90 CWnd *pEdit;
91 if (theMapObj) {
92 m_scorchtype = (Scorches) theMapObj->getProperties()->getInt(TheKey_scorchType);
93 m_scorchsize = theMapObj->getProperties()->getReal(TheKey_objectRadius);
94 }
95 CComboBox *scorch = (CComboBox*)GetDlgItem(IDC_SCORCHTYPE);
96
97 scorch->SetCurSel((int)m_scorchtype);
98 str.Format("%f",m_scorchsize);
99 pEdit = GetDlgItem(IDC_SIZE_EDIT);
100 if (pEdit)
101 pEdit->SetWindowText(str);
102 m_updating = false;
103}

219void ScorchOptions::changeScorch(void)
220{
221 getAllSelectedDicts();
222
223 Dict newDict;
224 newDict.setInt(TheKey_scorchType, (Int)m_scorchtype);
225 DictItemUndoable *pUndo = new DictItemUndoable(m_allSelectedDicts.begin(), newDict, newDict.getNthKey(0), m_allSelectedDicts.size());
226 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
227 pDoc->AddAndDoUndoable(pUndo);
228 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
229
230 WbView3d *pView = CWorldBuilderDoc::GetActive3DView();
231 pView->Invalidate();
232}

249void ScorchOptions::getAllSelectedDicts(void)
250{
251 m_allSelectedDicts.clear();
252
253 for (MapObject *pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext()) {
254 if (!pMapObj->isSelected() || !pMapObj->isScorch()) {
255 continue;
256 }
257 m_allSelectedDicts.push_back(pMapObj->getProperties());
258 }
