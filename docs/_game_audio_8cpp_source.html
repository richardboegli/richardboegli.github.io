312//-------------------------------------------------------------------------------------------------
313void AudioManager::update()
314{
315 Coord3D groundPos, microphonePos;
316 TheTacticalView->getPosition( &groundPos );
317 Real angle = TheTacticalView->getAngle();
318 Matrix3D rot = Matrix3D::Identity;
319 rot.Rotate_Z( angle );
320 Vector3 forward( 0, 1, 0 );
321 rot.mulVector3( forward );
322
323 Real desiredHeight = TheAudio->getAudioSettings()->m_microphoneDesiredHeightAboveTerrain;
324 Real maxPercentage = TheAudio->getAudioSettings()->m_microphoneMaxPercentageBetweenGroundAndCamera;
325
326 Coord3D lookTo;
327 lookTo.set(forward.X, forward.Y, forward.Z);
328
329 //Kris: At this point, the microphone is calculated to be at the ground position where the camera is looking at.
330 //Instead we want to move the microphone towards the camera. Hopefully, it'll be a desired altitude, but if it
331 //gets too close to the camera (or even past it), that would be undesirable. Therefore, we have a backup method
332 //of making sure we only go a certain percentage towards the camera or the desired height, whichever occurs first.
333 Coord3D cameraPos = TheTacticalView->get3DCameraPosition();
334 Coord3D groundToCameraVector;
335 groundToCameraVector.set( &cameraPos );
336 groundToCameraVector.sub( &groundPos );
337 Real bestScaleFactor;
338
339 if( cameraPos.z <= desiredHeight || groundToCameraVector.z <= 0.0f )
340 {
341 //Use the percentage calculation!
342 bestScaleFactor = maxPercentage;
343 }
344 else
345 {
346 //Calculate the stopping position of the groundToCameraVector when we force z to be m_microphoneDesiredHeightAboveTerrain
347 Real zScale = desiredHeight / groundToCameraVector.z;
348
349 //Use the smallest of the two scale calculations
350 bestScaleFactor = MIN( maxPercentage, zScale );
351 }
352
353 //Now apply the best scalar to the ground-to-camera vector.
354 groundToCameraVector.scale( bestScaleFactor );
355
356 //Set the microphone to be the ground position adjusted for terrain plus the vector we just calculated.
357 groundPos.z = TheTerrainLogic->getGroundHeight( groundPos.x, groundPos.y );
358 microphonePos.set( &groundPos );
359 microphonePos.add( &groundToCameraVector );
360
361 //Viola! A properly placed microphone.
362 setListenerPosition( &microphonePos, &lookTo );
363
364
365 //Now determine if we would like to boost the volume based on the camera being close to the microphone!
366 Real maxBoostScalar = TheAudio->getAudioSettings()->m_zoomSoundVolumePercentageAmount;
367 Real minDist = TheAudio->getAudioSettings()->m_zoomMinDistance;
368 Real maxDist = TheAudio->getAudioSettings()->m_zoomMaxDistance;
369
370 //We can't boost a sound above 100%, instead reduce the normal sound level.
371 m_zoomVolume = 1.0f - maxBoostScalar;
372
373 //Are we even using a boost?
374 if( maxBoostScalar > 0.0f )
375 {
376 //How far away is the camera from the microphone?
377 Coord3D vector = cameraPos;
378 vector.sub( &microphonePos );
379 Real dist = vector.length();
380
381 if( dist < minDist )
382 {
383 //Max volume!
384 m_zoomVolume = 1.0f;
385 }
386 else if( dist < maxDist )
387 {
388 //Determine what the boost amount will be.
389 Real scalar = (dist - minDist) / (maxDist - minDist);
390 m_zoomVolume = 1.0f - scalar * maxBoostScalar;
391 }
392 }
393
394 set3DVolumeAdjustment( m_zoomVolume );
395
396}

413//-------------------------------------------------------------------------------------------------
414AudioHandle AudioManager::addAudioEvent(const AudioEventRTS *eventToAdd)
415{
416 if (eventToAdd->getEventName().isEmpty() || eventToAdd->getEventName() == AsciiString("NoSound")) {
417 return AHSV_NoSound;
418 }
419
420#ifdef INTENSIVE_AUDIO_DEBUG
421 DEBUG_LOG(("AUDIO (%d): Received addAudioEvent('%s')", TheGameLogic->getFrame(), eventToAdd->getEventName().str()));
422#endif
423 if (!eventToAdd->getAudioEventInfo()) {
424 getInfoForAudioEvent(eventToAdd);
425 if (!eventToAdd->getAudioEventInfo()) {
426 DEBUG_CRASH(("No info for requested audio event '%s'\n", eventToAdd->getEventName().str()));
427 return AHSV_Error;
428 }
429 }
430
431 switch (eventToAdd->getAudioEventInfo()->m_soundType)
432 {
433 case AT_Music:
434 if (!isOn(AudioAffect_Music))
435 return AHSV_NoSound;
436 break;
437 case AT_SoundEffect:
438 if (!isOn(AudioAffect_Sound) || !isOn(AudioAffect_Sound3D))
439 return AHSV_NoSound;
440 break;
441 case AT_Streaming:
442 if (!isOn(AudioAffect_Speech))
443 return AHSV_NoSound;
444 break;
445 }
446
447 // if we're currently playing uninterruptable speech, then disallow the addition of this sample
448 if (getDisallowSpeech() && eventToAdd->getAudioEventInfo()->m_soundType == AT_Streaming) {
449 return AHSV_NoSound;
450 }
451
452
453 AudioEventRTS *audioEvent = MSGNEW("AudioEventRTS") AudioEventRTS(*eventToAdd); // poolify
454 audioEvent->setPlayingHandle( allocateNewHandle() );
455 audioEvent->generateFilename(); // which file are we actually going to play?
456 ((AudioEventRTS*)eventToAdd)->setPlayingAudioIndex( audioEvent->getPlayingAudioIndex() );
457 audioEvent->generatePlayInfo(); // generate pitch shift and volume shift now as well
458
459 std::list<std::pair<AsciiString, Real> >::iterator it;
460 for (it = m_adjustedVolumes.begin(); it != m_adjustedVolumes.end(); ++it) {
461 if (it->first == audioEvent->getEventName()) {
462 audioEvent->setVolume(it->second);
463 break;
464 }
465 }
466
467 if (!audioEvent->getUninterruptable()) {
468 if (!shouldPlayLocally(audioEvent)) {
469 releaseAudioEventRTS(audioEvent);
470 return AHSV_NotForLocal;
471 }
472 }
473
474 // cull muted audio
475 if (audioEvent->getVolume() < TheAudio->getAudioSettings()->m_minVolume) {
476#ifdef INTENSIVE_AUDIO_DEBUG
477 DEBUG_LOG((" - culled due to muting (%d).\n", audioEvent->getVolume()));
478#endif
479 releaseAudioEventRTS(audioEvent);
480 return AHSV_Muted;
481 }
482
483 AudioType type = eventToAdd->getAudioEventInfo()->m_soundType;
484 if (type == AT_Music)
485 {
486 m_music->addAudioEvent(audioEvent);
487 }
488 else
489 {
490 //Possible to nuke audioEvent inside.
491 m_sound->addAudioEvent(audioEvent);
492 }
493
494 if( audioEvent )
495 {
496 return audioEvent->getPlayingHandle();
497 }
498 return AHSV_NoSound;
499}

604//-------------------------------------------------------------------------------------------------
605void AudioManager::setAudioEventVolumeOverride( AsciiString eventToAffect, Real newVolume )
606{
607 if (eventToAffect == AsciiString::TheEmptyString) {
608 m_adjustedVolumes.clear();
609 return;
610 }
611
612 // Find any playing audio events and adjust their volume accordingly.
613 if (newVolume != -1.0f) {
614 adjustVolumeOfPlayingAudio(eventToAffect, newVolume);
615 }
616
617 std::list<std::pair<AsciiString, Real> >::iterator it;
618 for (it = m_adjustedVolumes.begin(); it != m_adjustedVolumes.end(); ++it) {
619 if (it->first == eventToAffect) {
620 if (newVolume == -1.0f) {
621 m_adjustedVolumes.erase(it);
622 return;
623 } else {
624 it->second = newVolume;
625 return;
626 }
627 }
628 }
629
630 if (newVolume != -1.0f) {
631 std::pair<AsciiString, Real> newPair;
632 newPair.first = eventToAffect;
633 newPair.second = newVolume;
634 m_adjustedVolumes.push_front(newPair);
635 }
636}

1020Bool AudioManager::shouldPlayLocally(const AudioEventRTS *audioEvent)
1021{
1022 Player *localPlayer = ThePlayerList->getLocalPlayer();
1023 if( !localPlayer->isPlayerActive() )
1024 {
1025 //We are dead, thus are observing. Get the player we are observing. It's
1026 //possible that we're not looking at any player, therefore it can be NULL.
1027 localPlayer = TheControlBar->getObserverLookAtPlayer();
1028 }
1029
1030 const AudioEventInfo *ei = audioEvent->getAudioEventInfo();
1031
1032 // Music should always play locally.
1033 if (ei->m_soundType == AT_Music) {
1034 return TRUE;
1035 }
1036
1037 if (!BitTest(ei->m_type, (ST_PLAYER | ST_ALLIES | ST_ENEMIES | ST_EVERYONE))) {
1038 DEBUG_CRASH(("No player restrictions specified for '%s'. Using Everyone.\n", ei->m_audioName.str()));
1039 return TRUE;
1040 }
1041
1042 if (BitTest(ei->m_type, ST_EVERYONE)) {
1043 return TRUE;
1044 }
1045
1046 Player *owningPlayer = ThePlayerList->getNthPlayer(audioEvent->getPlayerIndex());
1047
1048 if (BitTest(ei->m_type, ST_PLAYER) && BitTest(ei->m_type, ST_UI) && owningPlayer == NULL) {
1049 DEBUG_ASSERTCRASH(!TheGameLogic->isInGameLogicUpdate(), ("Playing %s sound -- player-based UI sound without specifying a player.\n"));
1050 return TRUE;
1051 }
1052
1053 if (owningPlayer == NULL) {
1054 DEBUG_CRASH(("Sound '%s' expects an owning player, but the audio event that created it didn't specify one.\n", ei->m_audioName.str()));
1055 return FALSE;
1056 }
1057
1058 if( !localPlayer )
1059 {
1060 return FALSE;
1061 }
1062
1063 const Team *localTeam = localPlayer->getDefaultTeam();
1064 if (localTeam == NULL) {
1065 return FALSE;
1066 }
1067
1068 if (BitTest(ei->m_type, ST_PLAYER)) {
1069 return owningPlayer == localPlayer;
1070 }
1071
1072 if (BitTest(ei->m_type, ST_ALLIES)) {
1073 // We have to also check that the owning player isn't the local player, because PLAYER
1074 // wasn't specified, or we wouldn't have gotten here.
1075 return (owningPlayer != localPlayer) && owningPlayer->getRelationship(localTeam) == ALLIES;
1076 }
1077
1078 if (BitTest(ei->m_type, ST_ENEMIES)) {
1079 return owningPlayer->getRelationship(localTeam) == ENEMIES;
1080 }
1081
1082 return FALSE;
1083}
