43void INI::parseMappedImageDefinition( INI* ini )
44{
45 AsciiString name;
46
47 // read the name
48 const char* c = ini->getNextToken();
49 name.set( c );
50
51 //
52 // find existing item if present, note that we do not support overrides
53 // in the images like we do in systems that are more "design" oriented, images
54 // are assets as they are
55 //
56 if( !TheMappedImageCollection )
57 {
58 //We don't need it if we're in the builder... which doesn't have this.
59 return;
60 }
61 Image *image = const_cast<Image*>(TheMappedImageCollection->findImageByName( name ));
62 if(image)
63 DEBUG_ASSERTCRASH(!image->getRawTextureData(), ("We are trying to parse over an existing image that contains a non-null rawTextureData, you should fix that"));
64
65 if( image == NULL )
66 {
67
68 // image not found, create a new one
69 image = newInstance(Image);
70 image->setName( name );
71 TheMappedImageCollection->addImage(image);
72 DEBUG_ASSERTCRASH( image, ("parseMappedImage: unable to allocate image for '%s'\n",
73 name.str()) );
74
75 } // end if
76
77 // parse the ini definition
78 ini->initFromINI( image, image->getFieldParse());
79
80} // end parseMappedImage
