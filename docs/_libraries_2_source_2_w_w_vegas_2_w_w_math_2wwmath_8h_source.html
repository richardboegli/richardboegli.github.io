281WWINLINE bool WWMath::Is_Valid_Float(float x)
282{
283 unsigned long * plong = (unsigned long *)(&x);
284 unsigned long exponent = ((*plong) & 0x7F800000) >> (32-9);
285

388WWINLINE float WWMath::Fast_Sin(float val)
389{
390 val*=float(SIN_TABLE_SIZE) / (2.0f * WWMATH_PI);
391
392 int idx0=Float_To_Int_Floor(val);
393 int idx1=idx0+1;
394 float frac=val-(float)idx0;
395
396 idx0 = ((unsigned)idx0) & (SIN_TABLE_SIZE-1);
397 idx1 = ((unsigned)idx1) & (SIN_TABLE_SIZE-1);
398
399 return (1.0f - frac) * _FastSinTable[idx0] + frac * _FastSinTable[idx1];
400}

478// ----------------------------------------------------------------------------
479// Fast, table based arc cos
480// ----------------------------------------------------------------------------
481
482WWINLINE float WWMath::Fast_Acos(float val)
483{
484 // Near -1 and +1, the table becomes too inaccurate
485 if (WWMath::Fabs(val) > 0.975f) {
486 return WWMath::Acos(val);
487 }
488
489 val*=float(ARC_TABLE_SIZE/2);
490
491 int idx0=Float_To_Int_Floor(val);
492 int idx1=idx0+1;
493 float frac=val-(float)idx0;
494
495 idx0+=ARC_TABLE_SIZE/2;
496 idx1+=ARC_TABLE_SIZE/2;
497
498 // we dont even get close to the edge of the table...
499 assert((idx0 >= 0) && (idx0 < ARC_TABLE_SIZE));
500 assert((idx1 >= 0) && (idx1 < ARC_TABLE_SIZE));
501
502 // compute and return the interpolated value
503 return (1.0f - frac) * _FastAcosTable[idx0] + frac * _FastAcosTable[idx1];
504}

574WWINLINE int WWMath::Float_To_Int_Chop(const float& f)
575{
576 int a = *reinterpret_cast<const int*>(&f); // take bit pattern of float into a register
577 int sign = (a>>31); // sign = 0xFFFFFFFF if original value is negative, 0 if positive
578 int mantissa = (a&((1<<23)-1))|(1<<23); // extract mantissa and add the hidden bit
579 int exponent = ((a&0x7fffffff)>>23)-127; // extract the exponent
580 int r = ((unsigned int)(mantissa)<<8)>>(31-exponent); // ((1<<exponent)*mantissa)>>24 -- (we know that mantissa > (1<<24))
581 return ((r ^ (sign)) - sign ) &~ (exponent>>31); // add original sign. If exponent was negative, make return value 0.
582}
583
584WWINLINE int WWMath::Float_To_Int_Floor (const float& f)
585{
586 int a = *reinterpret_cast<const int*>(&f); // take bit pattern of float into a register
587 int sign = (a>>31); // sign = 0xFFFFFFFF if original value is negative, 0 if positive
588 a&=0x7fffffff; // we don't need the sign any more
589
590 int exponent = (a>>23)-127; // extract the exponent
591 int expsign = ~(exponent>>31); // 0xFFFFFFFF if exponent is positive, 0 otherwise
592 int imask = ( (1<<(31-(exponent))))-1; // mask for true integer values
593 int mantissa = (a&((1<<23)-1)); // extract mantissa (without the hidden bit)
594 int r = ((unsigned int)(mantissa|(1<<23))<<8)>>(31-exponent); // ((1<<exponent)*(mantissa|hidden bit))>>24 -- (we know that mantissa > (1<<24))
595
596 r = ((r & expsign) ^ (sign)) + ((!((mantissa<<8)&imask)&(expsign^((a-1)>>31)))&sign); // if (fabs(value)<1.0) value = 0; copy sign; if (value < 0 && value==(int)(value)) value++;
597 return r;
598}

604#if defined(_MSC_VER) && defined(_M_IX86)
605WWINLINE __declspec(naked) float __fastcall WWMath::Inv_Sqrt(float a)
606{
607 __asm {
608 mov eax, 0be6eb508h
609 mov DWORD PTR [esp-12],03fc00000h ; 1.5 on the stack
610 sub eax, DWORD PTR [esp+4]; a
611 sub DWORD PTR [esp+4], 800000h ; a/2 a=Y0
612 shr eax, 1 ; firs approx in eax=R0
613 mov DWORD PTR [esp-8], eax
614
615 fld DWORD PTR [esp-8] ;r
616 fmul st, st ;r*r
617 fld DWORD PTR [esp-8] ;r
618 fxch st(1)
619 fmul DWORD PTR [esp+4];a ;r*r*y0
620 fld DWORD PTR [esp-12];load 1.5
621 fld st(0)
622 fsub st,st(2) ;r1 = 1.5 - y1
623 ;x1 = st(3)
624 ;y1 = st(2)
625 ;1.5 = st(1)
626 ;r1 = st(0)
627
628 fld st(1)
629 fxch st(1)
630 fmul st(3),st ; y2=y1*r1*...
631 fmul st(3),st ; y2=y1*r1*r1
632 fmulp st(4),st ; x2=x1*r1
633 fsub st,st(2) ; r2=1.5-y2
634 ;x2=st(3)
635 ;y2=st(2)
636 ;1.5=st(1)
637 ;r2 = st(0)
638
639 fmul st(2),st ;y3=y2*r2*...
640 fmul st(3),st ;x3=x2*r2
641 fmulp st(2),st ;y3=y2*r2*r2
642 fxch st(1)
643 fsubp st(1),st ;r3= 1.5 - y3
644 ;x3 = st(1)
645 ;r3 = st(0)
646 fmulp st(1), st
647 ret 4
648 }
649}
