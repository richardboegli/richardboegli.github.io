80void UnicodeString::ensureUniqueBufferOfSize(int numCharsNeeded, Bool preserveData, const WideChar* strToCopy, const WideChar* strToCat)
81{
82 validate();
83
84 if (m_data &&
85 m_data->m_refCount == 1 &&
86 m_data->m_numCharsAllocated >= numCharsNeeded)
87 {
88 // no buffer manhandling is needed (it's already large enough, and unique to us)
89 if (strToCopy)
90 wcscpy(m_data->peek(), strToCopy);
91 if (strToCat)
92 wcscat(m_data->peek(), strToCat);
93 return;
94 }
95
96 int minBytes = sizeof(UnicodeStringData) + numCharsNeeded*sizeof(WideChar);
97 if (minBytes > MAX_LEN)
98 throw ERROR_OUT_OF_MEMORY;
99
100 int actualBytes = TheDynamicMemoryAllocator->getActualAllocationSize(minBytes);
101 UnicodeStringData* newData = (UnicodeStringData*)TheDynamicMemoryAllocator->allocateBytesDoNotZero(actualBytes, "STR_UnicodeString::ensureUniqueBufferOfSize");
102 newData->m_refCount = 1;
103 newData->m_numCharsAllocated = (actualBytes - sizeof(UnicodeStringData))/sizeof(WideChar);
104#if defined(_DEBUG) || defined(_INTERNAL)
105 newData->m_debugptr = newData->peek(); // just makes it easier to read in the debugger
106#endif
107
108 if (m_data && preserveData)
109 wcscpy(newData->peek(), m_data->peek());
110 else
111 newData->peek()[0] = 0;
112
113 // do these BEFORE releasing the old buffer, so that self-copies
114 // or self-cats will work correctly.
115 if (strToCopy)
116 wcscpy(newData->peek(), strToCopy);
117 if (strToCat)
118 wcscat(newData->peek(), strToCat);
119
120 releaseBuffer();
121 m_data = newData;
122
123 validate();
124}

230// -----------------------------------------------------
231void UnicodeString::trim()
232{
233 validate();
234
235 if (m_data)
236 {
237 const WideChar *c = peek();
238
239 // Strip leading white space from the string.
240 while (c && iswspace(*c))
241 {
242 c++;
243 }
244 if (c != peek())
245 {
246 set(c);
247 }
248
249 if (m_data) // another check, because the previous set() could erase m_data
250 {
251 // Clip trailing white space from the string.
252 int len = wcslen(peek());
253 for (int index = len-1; index >= 0; index--)
254 {
255 if (iswspace(getCharAt(index)))
256 {
257 removeLastChar();
258 }
259 else
260 {
261 break;
262 }
263 }
264 }
265 }
266 validate();
267}

330Bool UnicodeString::nextToken(UnicodeString* tok, UnicodeString delimiters)
331{
332 if (this->isEmpty() || tok == this)
333 return false;
334
335 if (delimiters.isEmpty())
336 delimiters = UnicodeString(L" \t\n\r");
337
338 Int offset;
339
340 offset = wcsspn(peek(), delimiters.str());
341 WideChar* start = peek() + offset;
342
343 offset = wcscspn(start, delimiters.str());
344 WideChar* end = start + offset;
345
346 if (end > start)
347 {
348 Int len = end - start;
349 WideChar* tmp = tok->getBufferForRead(len + 1);
350 memcpy(tmp, start, len*2);
351 tmp[len] = 0;
352
353 this->set(end);
354
355 return true;
356 }
357 else
358 {
359 this->clear();
360 tok->clear();
361 return false;
362 }
