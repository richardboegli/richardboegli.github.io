90static Object *validUnderCursor( const ICoord2D *mouse, const CommandButton *command, PickType pickType )
91{
92 Object *pickObj = NULL;
93
94 // pick a drawable at the mouse location
95 Drawable *pick = TheTacticalView->pickDrawable( mouse, FALSE, pickType );
96
97 // only continue if there is something there
98 if( pick && pick->getObject() )
99 {
100 Player *player = ThePlayerList->getLocalPlayer();
101
102 // get object we picked
103 pickObj = pick->getObject();
104
105 if (!command->isValidObjectTarget(player, pickObj))
106 pickObj = NULL;
107
108 } // end if
109
110
111 return pickObj;
112
113} // end validUnderCursor

117static CommandStatus doFireWeaponCommand( const CommandButton *command, const ICoord2D *mouse )
118{
119
120 // sanity
121 if( command == NULL || mouse == NULL )
122 return COMMAND_COMPLETE;
123
124 //
125 // for single object selections get the source ID and sanity check for illegal object and
126 // bail along the way
127 //
128 ObjectID sourceID = INVALID_ID;
129 if( TheInGameUI->getSelectCount() == 1 )
130 {
131 Drawable *draw = TheInGameUI->getFirstSelectedDrawable();
132
133 // sanity
134 if( draw == NULL || draw->getObject() == NULL )
135 return COMMAND_COMPLETE;
136
137 // get object id
138 sourceID = draw->getObject()->getID();
139
140 } // end if
141
142 // create message and send to the logic
143 GameMessage *msg;
144 if( BitTest( command->getOptions(), NEED_TARGET_POS ) )
145 {
146 Coord3D world;
147
148 // translate the mouse location into world coords
149 TheTacticalView->screenToTerrain( mouse, &world );
150
151 // create the message and append arguments
152 msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_WEAPON_AT_LOCATION );
153 msg->appendIntegerArgument( command->getWeaponSlot() );
154 msg->appendLocationArgument( world );
155 msg->appendIntegerArgument( command->getMaxShotsToFire() );
156
157 //Also append the object ID (incase weapon doesn't like obstacles on land).
158 Object *target = validUnderCursor( mouse, command, PICK_TYPE_SELECTABLE );
159 ObjectID targetID = target ? target->getID() : INVALID_ID;
160 msg->appendObjectIDArgument( targetID );
161
162
163 } // end if
164 else if( BitTest( command->getOptions(), COMMAND_OPTION_NEED_OBJECT_TARGET ) )
165 {
166
167 // setup the pick type ... some commands allow us to target shrubbery
168 PickType pickType = PICK_TYPE_SELECTABLE;
169
170 if( BitTest( command->getOptions(), ALLOW_SHRUBBERY_TARGET ) == TRUE )
171 pickType = (PickType)((Int)pickType | (Int)PICK_TYPE_SHRUBBERY);
172
173 if( BitTest( command->getOptions(), ALLOW_MINE_TARGET ) == TRUE )
174 pickType = (PickType)((Int)pickType | (Int)PICK_TYPE_MINES);
175
176 // get the target object under the cursor
177 Object *target = validUnderCursor( mouse, command, pickType );
178
179 // only continue if the object meets all the command criteria
180 if( target )
181 {
182
183 msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_WEAPON_AT_OBJECT );
184 msg->appendIntegerArgument( command->getWeaponSlot() );
185 msg->appendObjectIDArgument( target->getID() );
186 msg->appendIntegerArgument( command->getMaxShotsToFire() );
187
188 } // end if
189
190 } // end else
191 else
192 {
193 msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_WEAPON );
194 msg->appendIntegerArgument( command->getWeaponSlot() );
195 msg->appendIntegerArgument( command->getMaxShotsToFire() );
196
197 //This could be legit now -- think of firing a self destruct weapon
198 //-----------------------------------------------------------------
199 //DEBUG_ASSERTCRASH( 0, ("doFireWeaponCommand: Command options say it doesn't need additional user input '%s'\n",
200 // command->m_name.str()) );
201 //return COMMAND_COMPLETE;
202
203 } // end else
204
205 return COMMAND_COMPLETE;
206
207} // end fire weapon

211static CommandStatus doGuardCommand( const CommandButton *command, GuardMode guardMode, const ICoord2D *mouse )
212{
213 // sanity
214 if( command == NULL || mouse == NULL )
215 return COMMAND_COMPLETE;
216
217 if( TheInGameUI->getSelectCount() == 0 )
218 return COMMAND_COMPLETE;
219
220 GameMessage *msg = NULL;
221
222 if ( msg == NULL && BitTest( command->getOptions(), COMMAND_OPTION_NEED_OBJECT_TARGET ) )
223 {
224 // get the target object under the cursor
225 Object* target = validUnderCursor( mouse, command, PICK_TYPE_SELECTABLE );
226 if( target )
227 {
228 msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_GUARD_OBJECT );
229 msg->appendObjectIDArgument( target->getID() );
230 msg->appendIntegerArgument(guardMode);
231 pickAndPlayUnitVoiceResponse(TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_DO_GUARD_OBJECT);
232 }
233 }
234
235 if( msg == NULL )
236 {
237 Coord3D world;
238 if (BitTest( command->getOptions(), NEED_TARGET_POS ))
239 {
240 // translate the mouse location into world coords
241 TheTacticalView->screenToTerrain( mouse, &world );
242 }
243 else
244 {
245 Drawable *draw = TheInGameUI->getFirstSelectedDrawable();
246 if( draw == NULL || draw->getObject() == NULL )
247 return COMMAND_COMPLETE;
248 world = *draw->getObject()->getPosition();
249 }
250
251 // create the message and append arguments
252 msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_GUARD_POSITION );
253 msg->appendLocationArgument(world);
254 msg->appendIntegerArgument(guardMode);
255 pickAndPlayUnitVoiceResponse(TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_DO_GUARD_POSITION);
256 }
257
258 return COMMAND_COMPLETE;
259
260}

265static CommandStatus doAttackMoveCommand( const CommandButton *command, const ICoord2D *mouse )
266{
267
268 // sanity
269 if( command == NULL || mouse == NULL )
270 return COMMAND_COMPLETE;
271
272 //
273 // we can only set rally points for structures ... and we never multiple select structures
274 // so we must be sure there is only one thing selected (that thing we will set the point on)
275 //
276 Drawable *draw = TheInGameUI->getFirstSelectedDrawable();
277 DEBUG_ASSERTCRASH( draw, ("doAttackMoveCommand: No selected object(s)\n") );
278
279 // sanity
280 if( draw == NULL || draw->getObject() == NULL )
281 return COMMAND_COMPLETE;
282
283 // convert mouse point to world coords
284 Coord3D world;
285 TheTacticalView->screenToTerrain( mouse, &world );
286
287 // send the message to set the rally point
288 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_ATTACKMOVETO );
289 msg->appendLocationArgument( world );
290
291 // Play the unit voice response
292 pickAndPlayUnitVoiceResponse(TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_DO_ATTACKMOVETO);
293
294 return COMMAND_COMPLETE;
295
296}

359GameMessageDisposition GUICommandTranslator::translateGameMessage(const GameMessage *msg)
360{
361 GameMessageDisposition disp = KEEP_MESSAGE;
362
363 // only pay attention to clicks in this translator if there is a pending GUI command
364 const CommandButton *command = TheInGameUI->getGUICommand();
365 if( command == NULL )
366 return disp;
367
368 switch( msg->getType() )
369 {
370
371 //---------------------------------------------------------------------------------------------
372 case GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_DOWN:
373 {
374
375 //
376 //
377 // it is necessary to use this input when there is a pending gui command, we don't wan't
378 // it to fall through to the rest of the system when we're in pending gui command "mode"
379 // because things like selection rectangles will start when we want to stay totally
380 // within the gui command "mode" here
381 //
382 disp = DESTROY_MESSAGE;
383
384 break;
385
386 } // end left mouse down
387
388 //---------------------------------------------------------------------------------------------
389 case GameMessage::MSG_MOUSE_LEFT_DOUBLE_CLICK:
390 case GameMessage::MSG_MOUSE_LEFT_CLICK:
391 {
392 CommandStatus commandStatus = COMMAND_COMPLETE;
393 ICoord2D mouse = msg->getArgument(0)->pixelRegion.hi;
394
395 // do the command action
396 if( command && !command->isContextCommand() )
397 {
398 switch( command->getCommandType() )
399 {
400
401 //---------------------------------------------------------------------------------------
402 case GUI_COMMAND_FIRE_WEAPON:
403 {
404 commandStatus = doFireWeaponCommand( command, &mouse );
405
406 PickAndPlayInfo info;
407 WeaponSlotType slot = command->getWeaponSlot();
408 info.m_weaponSlot = &slot;
409
410 pickAndPlayUnitVoiceResponse( TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_DO_WEAPON_AT_LOCATION, &info );
411 break;
412
413 } // end fire weapon command
414
415
416 //---------------------------------------------------------------------------------------
417 case GUI_COMMAND_EVACUATE:
418 {
419 if (BitTest(command->getOptions(), NEED_TARGET_POS)) {
420 Coord3D worldPos;
421
422 TheTacticalView->screenToTerrain(&mouse, &worldPos);
423
424 GameMessage *msg = TheMessageStream->appendMessage(GameMessage::MSG_EVACUATE);
425 msg->appendLocationArgument(worldPos);
426
427 pickAndPlayUnitVoiceResponse( TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_EVACUATE );
428
429 commandStatus = COMMAND_COMPLETE;
430 }
431
432 break;
433 }
434
435 //---------------------------------------------------------------------------------------
436 case GUI_COMMAND_GUARD:
437 {
438 commandStatus = doGuardCommand( command, GUARDMODE_NORMAL, &mouse );
439 break;
440 }
441
442 //---------------------------------------------------------------------------------------
443 case GUI_COMMAND_GUARD_WITHOUT_PURSUIT:
444 {
445 commandStatus = doGuardCommand( command, GUARDMODE_GUARD_WITHOUT_PURSUIT, &mouse );
446 break;
447 }
448
449 //---------------------------------------------------------------------------------------
450 case GUI_COMMAND_GUARD_FLYING_UNITS_ONLY:
451 {
452 commandStatus = doGuardCommand( command, GUARDMODE_GUARD_FLYING_UNITS_ONLY, &mouse );
453 break;
454 }
455
456 //Special weapons are now always context commands...
457 //---------------------------------------------------------------------------------------
458 case GUI_COMMAND_SPECIAL_POWER:
459 case GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT:
460 {
461 return KEEP_MESSAGE;
462 break;
463
464 } // end special power
465
466 case GUI_COMMAND_ATTACK_MOVE:
467 {
468 commandStatus = doAttackMoveCommand( command, &mouse );
469 break;
470 }
471
472 //---------------------------------------------------------------------------------------
473 case GUI_COMMAND_SET_RALLY_POINT:
474 {
475 commandStatus = doSetRallyPointCommand( command, &mouse );
476 break;
477
478 } // end set rally point
479
480 //---------------------------------------------------------------------------------------
481 case GUICOMMANDMODE_PLACE_BEACON:
482 {
483 commandStatus = doPlaceBeacon( command, &mouse );
484 break;
485
486 } // end set rally point
487
488 } // end switch
489
490 // used the input
491 disp = DESTROY_MESSAGE;
492
493 // get out of GUI command mode if we completed the command one way or another
494 if( commandStatus == COMMAND_COMPLETE )
495 {
496 TheInGameUI->setPreventLeftClickDeselectionInAlternateMouseModeForOneClick( TRUE );
497 TheInGameUI->setGUICommand( NULL );
498 }
499 } // end if
500
501 break;
502
503 } // end left mouse up
504
505 } // end switch
506
507 // If we're destroying the message, it means we used it. Therefore, destroy the current
508 // attack move instruction as well.
509 if (disp == DESTROY_MESSAGE)
510 TheInGameUI->clearAttackMoveToMode();
511
512
513 return disp;
514
515} // end translateMessage
