92void BridgeTowerBehavior::onDamage( DamageInfo *damageInfo )
93{
94 Object *bridge = TheGameLogic->findObjectByID( getBridgeID() );
95
96 // sanity
97 if( bridge == NULL )
98 return;
99
100 //
101 // get our body info so we now how much damage percent is being done to us ... we need this
102 // so that we can propagate the same damage percentage amont the towers and the bridge
103 //
104 BodyModuleInterface *body = getObject()->getBodyModule();
105 Real damagePercentage = damageInfo->in.m_amount / body->getMaxHealth();
106
107 // get the bridge behavior module for our bridge
108 BehaviorModule **bmi;
109 BridgeBehaviorInterface *bridgeInterface = NULL;
110 for( bmi = bridge->getBehaviorModules(); *bmi; ++bmi )
111 {
112
113 bridgeInterface = (*bmi)->getBridgeBehaviorInterface();
114 if( bridgeInterface )
115 break;
116
117 } // end for bmi
118 DEBUG_ASSERTCRASH( bridgeInterface != NULL, ("BridgeTowerBehavior::onDamage - no 'BridgeBehaviorInterface' found\n") );
119 if( bridgeInterface )
120 {
121
122 //
123 // damage each of the other towers if the source of this damage isn't from the bridge
124 // or other towers
125 //
126 Object *source = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
127 if( source == NULL ||
128 (source->isKindOf( KINDOF_BRIDGE ) == FALSE &&
129 source->isKindOf( KINDOF_BRIDGE_TOWER ) == FALSE) )
130 {
131
132 for( Int i = 0; i < BRIDGE_MAX_TOWERS; ++i )
133 {
134 Object *tower;
135
136 tower = TheGameLogic->findObjectByID( bridgeInterface->getTowerID( (BridgeTowerType)i ) );
137 if( tower && tower != getObject() )
138 {
139 BodyModuleInterface *towerBody = tower->getBodyModule();
140 DamageInfo towerDamage;
141
142 towerDamage.in.m_amount = damagePercentage * towerBody->getMaxHealth();
143 towerDamage.in.m_sourceID = getObject()->getID(); // we're now the source
144 towerDamage.in.m_damageType = damageInfo->in.m_damageType;
145 towerDamage.in.m_deathType = damageInfo->in.m_deathType;
146 tower->attemptDamage( &towerDamage );
147
148 } // end if
149
150 } // end for i
151
152 //
153 // damage bridge object, but make sure it's done through the bridge interface
154 // so that it doesn't automatically propagate that damage to the towers
155 //
156 BodyModuleInterface *bridgeBody = bridge->getBodyModule();
157 DamageInfo bridgeDamage;
158
159 bridgeDamage.in.m_amount = damagePercentage * bridgeBody->getMaxHealth();
160 bridgeDamage.in.m_sourceID = getObject()->getID(); // we're now the source
161 bridgeDamage.in.m_damageType = damageInfo->in.m_damageType;
162 bridgeDamage.in.m_deathType = damageInfo->in.m_deathType;
163 bridge->attemptDamage( &bridgeDamage );
164
165 } // end if
166
167 } // end if
168
169} // end onDamage
