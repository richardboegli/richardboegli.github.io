114//-------------------------------------------------------------------------------------------------
115UpdateSleepTime SlavedUpdate::update( void )
116{
118 if( m_framesToWait > 0 )
119 {
120 m_framesToWait--;
121 }
122 if( m_repairState == REPAIRSTATE_NONE )
123 {
124 if( m_framesToWait > 0 )
125 {
126 return UPDATE_SLEEP_NONE;
127 }
128 m_framesToWait = SLAVED_UPDATE_RATE;
129 }
130
131 if( m_slaver == INVALID_ID )
132 return UPDATE_SLEEP_NONE;
133
134 const SlavedUpdateModuleData* data = getSlavedUpdateModuleData();
135 Object *me = getObject();
136 if( !me )
137 {
138 return UPDATE_SLEEP_NONE;
139 }
140 AIUpdateInterface *myAI = me->getAIUpdateInterface();
141 if( !myAI )
142 {
143 return UPDATE_SLEEP_NONE;
144 }
145 Locomotor *locomotor = myAI->getCurLocomotor();
146 if( !locomotor )
147 {
148 return UPDATE_SLEEP_NONE;
149 }
150
151 Object *master = TheGameLogic->findObjectByID( m_slaver );
152 if( !master || master->isEffectivelyDead() || master->isDisabledByType( DISABLED_UNMANNED ) )
153 {
154 stopSlavedEffects();
155
156 //Killing is lame.
157 //me->kill();
158
159 //Let's disable the drone so it crashes instead!
160 //Added special case code in physics falling to ensure death.
161 me->setDisabled( DISABLED_UNMANNED );
162 if ( me->getAI() )
163 me->getAI()->aiIdle( CMD_FROM_AI);
164
165 return UPDATE_SLEEP_NONE;
166 }
167 else
168 {
169 Team *masterTeam = master->getTeam();
170 Team *myTeam = me->getTeam();
171 if ( masterTeam->getRelationship( myTeam ) != ALLIES )
172 {//slaver must have been hijacked or something.. // we will join his team
173 me->defect( masterTeam, 0 );
174 }
175
176
177
178
179 }
180
181 if (data->m_stayOnSameLayerAsMaster)
182 me->setLayer(master->getLayer());
183
184 //Clear the drone spotting bonus to the master. Up to the drone
185 //to satisfy the conditions to set it again for the next update.
186 master->clearWeaponBonusCondition( WEAPONBONUSCONDITION_DRONE_SPOTTING );
187
188 //Get my master's AI. If he is attacking something, grant him a range bonus,
189 //and I'll fly over the target.
190 Object *target = NULL;
191 AIUpdateInterface *masterAI = master->getAIUpdateInterface();
192 if( masterAI )
193 {
194 target = masterAI->getCurrentVictim();
195 }
196
197 //Calculate the health percentage of the master -- there are two places we care.
198 //NOTE: Health percentage will always be 100 should the drone be incapable of
199 //repairing.
200 Int healthPercentage = 100;
201 if( data->m_repairRatePerSecond > 0.0f )
202 {
203 BodyModuleInterface *body = master->getBodyModule();
204 if( body )
205 {
206 Real health = body->getHealth();
207 Real maxHealth = body->getMaxHealth();
208 healthPercentage = (Int)(health / maxHealth * 100.0f);
209 }
210 }
211
212 //Determine whether or not we need to go back to the master to repair him
213 if( healthPercentage <= data->m_repairWhenHealthBelowPercentage )
214 {
215 //1ST PRIORITY: Go to the master's position to repair him because he needs it.
216 doRepairLogic();
217 return UPDATE_SLEEP_NONE;
218 }
219
220 if( data->m_attackRange )
221 {
222 //2ND PRIORITY: Go to the master's current victim (as close as wander distance allows)
223 if( target )
224 {
225 //At this point, we officially are in an attack mode! Now, simply
226 endRepair();
227 doAttackLogic( target );
228 return UPDATE_SLEEP_NONE;
229 }
230 }
231
232 if( data->m_scoutRange )
233 {
234 //3RD PRIORITY: Hover above master's current move destination (as close as wander distance
235 //allows).
236 if( masterAI->getPath() )
237 {
238 const Coord3D *masterDest = masterAI->getPath()->getLastNode()->getPosition();
239
240 //Check to see if master is close to the goal position.
241 Real distSqr = ThePartitionManager->getDistanceSquared( master, masterDest, FROM_BOUNDINGSPHERE_2D );
242 if( distSqr > (data->m_guardMaxRange * 0.5f) * (data->m_guardMaxRange * 0.5f) )
243 {
244 //If the master's distance to destination is more than half of the guarding range of the slave,
245 //then order the slave to scout it.
246 endRepair();
247 doScoutLogic( masterDest );
248 return UPDATE_SLEEP_NONE;
249 }
250 }
251 }
252
253 //NOTE: Health percentage will always be 100 should the drone be incapable of
254 //repairing.
255 if( healthPercentage < 100 )
256 {
257 //3RD PRIORITY: Go to the master's position to repair him (because we're idle)
258 doRepairLogic();
259 return UPDATE_SLEEP_NONE;
260 }
261
262 // update our "pinned" location based on where our master is
263 Coord3D pinnedPosition = *master->getPosition();
264 pinnedPosition.x += m_guardPointOffset.x;
265 pinnedPosition.y += m_guardPointOffset.y;
266 m_guardPointOffset.z = TheTerrainLogic->getGroundHeight( pinnedPosition.x, pinnedPosition.y );
267
268 if( data->m_guardMaxRange )
269 {
270 //3RD PRIORITY: Guard the master's area.
271 if( myAI->isIdle() && ThePartitionManager->getDistanceSquared(me, &pinnedPosition, FROM_CENTER_3D) > CLOSE_ENOUGH_SQR )
272 {
273 //I'm idle and too far away.
274 endRepair();
275 doGuardLogic( &pinnedPosition );
276 }
277 else if( ThePartitionManager->getDistanceSquared( me, master, FROM_CENTER_3D ) > sqr(STRAY_MULTIPLIER * data->m_guardMaxRange ) )
278 {
279 //I'm too far away, no matter what I'm doing.
280 endRepair();
281 doGuardLogic( &pinnedPosition );
282 }
283 }
284 return UPDATE_SLEEP_NONE;
285}

288// We are ordered to attempt to get as close as possible to my master's target.
289//-------------------------------------------------------------------------------------------------
290void SlavedUpdate::doAttackLogic( const Object *target )
291{
292 const SlavedUpdateModuleData* data = getSlavedUpdateModuleData();
293 Object *me = getObject();
294 Object *master = TheGameLogic->findObjectByID( m_slaver );
295 Coord3D attackPosition;
296
297 //First, determine the attack position. If the target is too far away, then we'll
298 //calculate the closest allowable position.
299 const Coord3D *targetPos = target->getPosition();
300 Real dist = ThePartitionManager->getDistanceSquared( me, targetPos, FROM_BOUNDINGSPHERE_2D );
301 if( dist > sqr( data->m_attackRange ) )
302 {
303 //The distance is too far, so calculate the best allowable position.
304 Coord3D vector;
305 vector.set( targetPos );
306 vector.sub( master->getPosition() );
307 vector.normalize();
308 vector.scale( data->m_attackRange );
309
310 //Now that we have calculated the vector relative to me, add it to my position to get my goal.
311 attackPosition.set( master->getPosition() );
312 attackPosition.add( &vector );
313 }
314 else
315 {
316 //We are close enough, so use the target position -- easy!
317 attackPosition.set( targetPos );
318 }
319
320 //Finally, if we have a wander distance, then randomly select a point within
321 //the wander range radius of the pinned position, and we'll go there.
322 if( data->m_attackWanderRange )
323 {
324 //Allow me to wander away from the pinnedPosition.
325 Real randomDirection = GameLogicRandomValue( 0, 2*PI );
326 m_guardPointOffset.zero();
327 m_guardPointOffset.x += data->m_attackWanderRange * Cos( randomDirection );
328 m_guardPointOffset.y += data->m_attackWanderRange * Sin( randomDirection );
329
330 //Offset our pinned position by our random offset.
331 attackPosition.x += m_guardPointOffset.x;
332 attackPosition.y += m_guardPointOffset.y;
333 m_guardPointOffset.z = TheTerrainLogic->getGroundHeight( attackPosition.x, attackPosition.y );
334 }
335
336 //Move to the updated position!
337 AIUpdateInterface *ai = me->getAIUpdateInterface();
338 if( ai )
339 {
340 ai->aiMoveToPosition( &attackPosition, CMD_FROM_AI );
341 }
342
343 if( dist < sqr( data->m_distToTargetToGrantRangeBonus ) )
344 {
345 //Finally, seeing we are close enough to the target, grant our
346 //master extended weapon range!
347 master->setWeaponBonusCondition( WEAPONBONUSCONDITION_DRONE_SPOTTING );
348 }
349}

435// We are ordered to repair our master
436//-------------------------------------------------------------------------------------------------
437void SlavedUpdate::doRepairLogic()
438{
439 Object *me = getObject();
440 Object *master = TheGameLogic->findObjectByID( m_slaver );
441 const SlavedUpdateModuleData* data = getSlavedUpdateModuleData();
442 AIUpdateInterface *ai = me->getAIUpdateInterface();
443 if( !ai )
444 {
445 return;
446 }
447
448 //There are two major things... either move closer or repair.
449 Real distSqr = ThePartitionManager->getDistanceSquared( me, master, FROM_BOUNDINGSPHERE_2D );
450 Bool closeEnough = distSqr < 12.0f * 12.0f;
451
452 //We're going to do different things based on the repair state.
453 if( closeEnough )
454 {
455 //Now drive the repairing states.
456 switch( m_repairState )
457 {
458 case REPAIRSTATE_NONE:
459 setRepairState( REPAIRSTATE_READY );
460 break;
461 case REPAIRSTATE_READY:
462 case REPAIRSTATE_EXTENDING:
463 if( m_framesToWait == 0 )
464 {
465 setRepairState( REPAIRSTATE_WELDING );
466 }
467 break;
468 case REPAIRSTATE_UNPACKING:
469 case REPAIRSTATE_WELDING:
470 case REPAIRSTATE_RETRACTING:
471 if( m_framesToWait == 0 )
472 {
473 setRepairState( REPAIRSTATE_READY );
474 }
475 break;
476 }
477 }
478 else
479 {
480 m_repairing = false;
481
482 Bool closeEnoughForZPrecision = distSqr < sqr(master->getGeometryInfo().getBoundingSphereRadius() * 2);
483
484 //We're too far away to repair, so get closer.
485 Locomotor *locomotor = ai->getCurLocomotor();
486 if( locomotor )
487 {
488 locomotor->setUsePreciseZPos( closeEnoughForZPrecision );
489 }
490 Coord3D pos;
491 pos.set( master->getPosition() );
492 Real altitude = GameLogicRandomValueReal( data->m_repairMinAltitude, data->m_repairMaxAltitude );
493 pos.z += altitude;
494 ai->aiMoveToPosition( &pos, CMD_FROM_AI );
495
496 //Also speed things up by retracting the repair arm (so viewers
497 //will see it's intentions)
498 if( m_framesToWait == 0 )
499 {
500 setRepairState( REPAIRSTATE_READY );
501 }
502 }
503
504 if( closeEnough && m_repairing )
505 {
506 //We're close enough to repair.
507 BodyModuleInterface *body = master->getBodyModule();
508 if( body )
509 {
510 //Calculate the repair rate per frame.
511 Real repairAmount = data->m_repairRatePerSecond / LOGICFRAMES_PER_SECOND;
512
513 DamageInfo healingInfo;
514 healingInfo.in.m_amount = repairAmount;
515 healingInfo.in.m_damageType = DAMAGE_HEALING;
516 healingInfo.in.m_deathType = DEATH_NONE;
517 body->attemptHealing( &healingInfo );
518 }
519 }
520}

522//-------------------------------------------------------------------------------------------------
523void SlavedUpdate::endRepair()
524{
525 if( m_repairState != REPAIRSTATE_NONE )
526 {
527 //Remove the state so we can start using the update timer again.
528 m_repairState = REPAIRSTATE_NONE;
529 m_framesToWait = SLAVED_UPDATE_RATE;
530 m_repairing = false;
531
532 //Set the packing state and clear any other states (to force the arm to fully retract)
533 setRepairModelConditionStates( MODELCONDITION_PACKING );
534 }
535
536 AIUpdateInterface* ai = getObject()->getAIUpdateInterface();
537 if( ai )
538 {
539 ai->chooseLocomotorSet(LOCOMOTORSET_NORMAL);
540 Locomotor* locomotor = ai->getCurLocomotor();
541 if (locomotor)
542 {
543 locomotor->setUltraAccurate(false);
544 locomotor->setUsePreciseZPos(false);
545 }
546 }
547}

564//-------------------------------------------------------------------------------------------------
565void SlavedUpdate::setRepairState( RepairStates repairState )
566{
567 Object *obj = getObject();
568 Drawable *draw = obj->getDrawable();
569 //Object *master = TheGameLogic->findObjectByID( m_slaver );
570 const SlavedUpdateModuleData* data = getSlavedUpdateModuleData();
571
572 if( repairState == m_repairState )
573 {
574 return;
575 }
576
577 switch( repairState )
578 {
579 case REPAIRSTATE_UNPACKING:
580 setRepairModelConditionStates( MODELCONDITION_UNPACKING );
581 m_framesToWait = 15;
582 break;
583 case REPAIRSTATE_PACKING:
584 setRepairModelConditionStates( MODELCONDITION_PACKING );
585 m_framesToWait = 15;
586 break;
587 case REPAIRSTATE_READY:
588 {
589 switch( m_repairState )
590 {
591 case REPAIRSTATE_NONE:
592 //We're not in a repair state -- so assume we need to unpack
593 setRepairModelConditionStates( MODELCONDITION_UNPACKING );
594 m_repairState = REPAIRSTATE_UNPACKING;
595 m_framesToWait = 15;
596 break;
597 case REPAIRSTATE_WELDING:
598 //We are welding so retract before going into ready state.
599 m_repairState = REPAIRSTATE_RETRACTING;
600 m_framesToWait = 5;
601 setRepairModelConditionStates( MODELCONDITION_FIRING_C );
602 moveToNewRepairSpot();
603 break;
604 default:
605 m_repairState = REPAIRSTATE_READY;
606 m_framesToWait = GameLogicRandomValue( data->m_minReadyFrames, data->m_maxReadyFrames );
607 break;
608 }
609 break;
610 }
611 case REPAIRSTATE_WELDING:
612 {
613 if( m_repairState == REPAIRSTATE_READY )
614 {
615 m_repairState = REPAIRSTATE_EXTENDING;
616 m_framesToWait = 5;
617 setRepairModelConditionStates( MODELCONDITION_FIRING_B );
618 break;
619 }
620 else
621 {
622 m_repairState = REPAIRSTATE_WELDING;
623 m_framesToWait = GameLogicRandomValue( data->m_minWeldFrames, data->m_maxWeldFrames );
624
625 //Make sparks!
626 if( !data->m_weldingSysName.isEmpty() )
627 {
628 const ParticleSystemTemplate *tmp = TheParticleSystemManager->findTemplate( data->m_weldingSysName );
629 if( tmp )
630 {
631 ParticleSystem *weldingSys = TheParticleSystemManager->createParticleSystem(tmp);
632 if( weldingSys )
633 {
634 Coord3D pos;
635 //Get the bone position
636 if( draw->getPristineBonePositions( data->m_weldingFXBone.str(), 0, &pos, NULL, 1 ) )
637 {
638 pos.add( obj->getPosition() );
639 }
640 else
641 {
642 pos.set( obj->getPosition() );
643 }
644
645 weldingSys->setPosition( &pos );
646 Real time = (Real)(m_framesToWait * LOGICFRAMES_PER_SECOND);
647 weldingSys->setLifetimeRange( time, time );
648
649 AudioEventRTS soundToPlay = TheAudio->getMiscAudio()->m_repairSparks;
650 soundToPlay.setPosition( &pos );
651 TheAudio->addAudioEvent( &soundToPlay );
652 }
653 }
654 }
655
656 if( !m_repairing )
657 {
658 //Don't actually start healing until we fire our first sparks
659 m_repairing = true;
660 AIUpdateInterface *ai = obj->getAIUpdateInterface();
661 if( ai )
662 {
663 //ai->aiFaceObject( master, CMD_FROM_AI );
664 }
665 }
666 break;
667 }
668 break;
669 }
670 }
671}

674void SlavedUpdate::moveToNewRepairSpot()
675{
676 const SlavedUpdateModuleData* data = getSlavedUpdateModuleData();
677 Object *me = getObject();
678 Object *master = TheGameLogic->findObjectByID( m_slaver );
679
680 //Finally, if we have a wander distance, then randomly select a point within
681 //the wander range radius of the pinned position, and we'll go there.
682 if( data->m_repairRange )
683 {
684 //Allow me to wander away from the pinnedPosition.
685 Real randomDirection = GameLogicRandomValue( 0, 2*PI );
686 m_guardPointOffset.set( master->getPosition() );
687 m_guardPointOffset.x += data->m_repairRange * Cos( randomDirection );
688 m_guardPointOffset.y += data->m_repairRange * Sin( randomDirection );
689 m_guardPointOffset.z = TheTerrainLogic->getGroundHeight( m_guardPointOffset.x, m_guardPointOffset.y );
690 Real altitude = GameLogicRandomValueReal( data->m_repairMinAltitude, data->m_repairMaxAltitude );
691 m_guardPointOffset.z += altitude;
692
693 AIUpdateInterface *ai = me->getAIUpdateInterface();
694 if( ai )
695 {
696 ai->chooseLocomotorSet( LOCOMOTORSET_PANIC );
697 ai->getCurLocomotor()->setUltraAccurate( true );
698 ai->aiMoveToPosition( &m_guardPointOffset, CMD_FROM_AI );
699
700 Locomotor *locomotor = ai->getCurLocomotor();
701 if( locomotor )
702 {
703 locomotor->setUsePreciseZPos( true );
704 }
705 }
706 }

709//-------------------------------------------------------------------------------------------------
710void SlavedUpdate::startSlavedEffects( const Object *slaver )
711{
712 if( slaver == NULL )
713 return;
714
715 m_slaver = slaver->getID();
