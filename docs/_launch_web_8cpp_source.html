58bool LaunchWebBrowser(const char* url)
59 {
60 // Just return if no URL specified
61 if (!url || (strlen(url) == 0))
62 {
63 return false;
64 }
65
66 // Create a temporary file with HTML content
67 char tempPath[MAX_PATH];
68 GetWindowsDirectory(tempPath, MAX_PATH);
69
70 char filename[MAX_PATH];
71 GetTempFileName(tempPath, "WWS", 0, filename);
72
73 char* extPtr = strrchr(filename, '.');
74 strcpy(extPtr, ".html");
75
76 HANDLE file = CreateFile(filename, GENERIC_WRITE, 0, NULL, CREATE_ALWAYS,
77 FILE_ATTRIBUTE_NORMAL, NULL);
78
79 assert(INVALID_HANDLE_VALUE != file && "Failed to create temporary HTML file.");
80
81 if (INVALID_HANDLE_VALUE == file)
82 {
83 return false;
84 }
85
86 // Write generic contents
87 const char* contents = "<title>ViewHTML</title>";
88 DWORD written;
89 WriteFile(file, contents, strlen(contents), &written, NULL);
90 CloseHandle(file);
91
92 // Find the executable that can launch this file
93 char exeName[MAX_PATH];
94 HINSTANCE hInst = FindExecutable(filename, NULL, exeName);
95 assert(((int)hInst > 32) && "Unable to find executable that will display HTML files.");
96
97 // Delete temporary file
98 DeleteFile(filename);
99
100 if ((int)hInst <= 32)
101 {
102 return false;
103 }
104
105 // Launch browser with specified URL
106 char commandLine[MAX_PATH];
107 sprintf(commandLine, "[open] %s", url);
108
109 STARTUPINFO startupInfo;
110 memset(&startupInfo, 0, sizeof(startupInfo));
111 startupInfo.cb = sizeof(startupInfo);
112
113 PROCESS_INFORMATION processInfo;
114 BOOL createSuccess = CreateProcess(exeName, commandLine, NULL, NULL, FALSE,
115 0, NULL, NULL, &startupInfo, &processInfo);
116
117 assert(createSuccess && "Failed to launch default WebBrowser.");
118
119 return (TRUE == createSuccess);
120 }
