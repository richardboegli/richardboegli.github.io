60Bool ConvertToCarBombCrateCollide::isValidToExecute( const Object *other ) const
61{
62 if( !CrateCollide::isValidToExecute(other) )
63 {
64 return FALSE;
65 }
66
67 if( other->isEffectivelyDead() )
68 {
69 return FALSE;
70 }
71
72 if( other->isKindOf( KINDOF_AIRCRAFT ) || other->isKindOf( KINDOF_BOAT ) )
73 {
74 //Can't make carbombs out of planes and boats!
75 return FALSE;
76 }
77
78 if( other->getStatusBits().test( OBJECT_STATUS_IS_CARBOMB ) )
79 {
80 return FALSE;// oops, sorry, I'll convert the next one.
81 }
82
83 // Check to see if this other object has a carbomb weapon set that isn't in use.
84 WeaponSetFlags flags;
85 flags.set( WEAPONSET_CARBOMB );
86 const WeaponTemplateSet* set = other->getTemplate()->findWeaponTemplateSet( flags );
87 if( !set )
88 {
89 //This unit has no weapon set!
90 return FALSE;
91 }
92 if( !set->testWeaponSetFlag( WEAPONSET_CARBOMB ) )
93 {
94 //This unit has a weaponset, but the best match code above chose a different
95 //weaponset.
96 return FALSE;
97 }
98
99 // Also make sure that the car isn't already a carbomb!
100 if( other->testWeaponSetFlag( WEAPONSET_CARBOMB ) )
101 {
102 return FALSE;
103 }
104
105 return TRUE;
106}

110Bool ConvertToCarBombCrateCollide::executeCrateBehavior( Object *other )
111{
112 //Check to make sure that the other object is also the goal object in the AIUpdateInterface
113 //in order to prevent an unintentional conversion simply by having the terrorist walk too close
114 //to it.
115 //Assume ai is valid because CrateCollide::isValidToExecute(other) checks it.
116 Object *obj = getObject();
117 AIUpdateInterface* ai = obj->getAIUpdateInterface();
118 if (ai && ai->getGoalObject() != other)
119 return false;
120
121 if( other && other->checkAndDetonateBoobyTrap(obj) )
122 {
123 // Whoops, it was mined. Cancel if it or us is now dead.
124 if( other->isEffectivelyDead() || getObject()->isEffectivelyDead() )
125 {
126 return false;
127 }
128 }
129
130 other->setWeaponSetFlag( WEAPONSET_CARBOMB );
131
132 FXList::doFXObj( getConvertToCarBombCrateCollideModuleData()->m_fxList, other );
133
134 other->defect( getObject()->getControllingPlayer()->getDefaultTeam(), 0);
135
136 //In order to make things easier for the designers, we are going to transfer the terrorist name
137 //to the car... so the designer can control the car with their scripts.
138 TheScriptEngine->transferObjectName( getObject()->getName(), other );
139
140 //This is kinda special... we will endow our new ride with our vision and shroud range, since we are driving
141 other->setVisionRange(getObject()->getVisionRange());
142 other->setShroudClearingRange(getObject()->getShroudClearingRange());
143 other->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_IS_CARBOMB ) );
144
145 ExperienceTracker *exp = other->getExperienceTracker();
146 if (exp)
147 {
148 exp->setVeterancyLevel(obj->getExperienceTracker()->getVeterancyLevel());
149 }
150
151
152
153 TheRadar->removeObject( other );
154 TheRadar->addObject( other );
155
156
157
158
159 return TRUE;
160}
