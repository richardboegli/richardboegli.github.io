115/***********************************************************************************************
116 * CRC_Memory -- calculates a CRC for a block of memory *
117 * *
118 * INPUT: *
119 * *
120 * OUTPUT: *
121 * *
122 * WARNINGS: *
123 * *
124 * HISTORY: *
125 *=============================================================================================*/
126unsigned long CRC_Memory( const unsigned char *data, unsigned long length, unsigned long crc )
127{
128 crc ^= 0xFFFFFFFF; // invert previous CRC
129 while ( length-- ) {
130 crc = CRC32( *data++, crc ); // calc crc for each byte
131 }
132 return (crc ^ 0xFFFFFFFF); // invert new CRC and return it
133}

168unsigned long CRC_Stringi( const char *string, unsigned long crc )
169{
170 crc ^= 0xFFFFFFFF; // invert previous CRC
171 while ( *string ) {
172 char c = (char)toupper(*string++);
