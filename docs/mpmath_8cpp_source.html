137/***********************************************************************************************
138 * _Byte_Precision -- Determines the number of bytes significant in long integer. *
139 * *
140 * This utility routine will determine the number of precision bytes exist in the long *
141 * integer specified. There are some optimizations that can occur if the byte precision *
142 * is known. *
143 * *
144 * INPUT: value -- The value of the long integer that the byte precision will be calculated *
145 * for. *
146 * *
147 * OUTPUT: Returns with the number of bytes that the long integer requires (at a minimum) *
148 * to cover the precision of the integer. The minimum value will be 1, the maximum *
149 * will be 4. *
150 * *
151 * WARNINGS: none *
152 * *
153 * HISTORY: *
154 * 07/01/1996 JLB : Created. *
155 *=============================================================================================*/
156static int _Byte_Precision(unsigned long value)
157{
158 int byte_count;
159 for (byte_count = sizeof(value); byte_count; byte_count--) {
160 if (value >> ((byte_count-1)*8)) break;
161 }
162 return(byte_count);
163}

166/***********************************************************************************************
167 * XMP_DER_Length_Encode -- Output the length of a DER block. *
168 * *
169 * This routine will output the length of the block using Distinguished Encoding Rules. *
170 * The rest of the block must be filled in as appropriate. For data blocks that are less *
171 * than 128 bytes long, the header consists of only one byte. Longer buffers lengths *
172 * can consume up to 5 bytes (depends on magnitude of the length value). *
173 * *
174 * INPUT: length -- The length of the data block to be output. *
175 * *
176 * output -- Pointer to the memory block that will be set up. *
177 * *
178 * OUTPUT: Returns with the number of bytes (header) that was used to store the length *
179 * value. Subsequent data must be placed after the header. *
180 * *
181 * WARNINGS: none *
182 * *
183 * HISTORY: *
184 * 07/01/1996 JLB : Created. *
185 *=============================================================================================*/
186int MPEXPORT XMP_DER_Length_Encode(unsigned long length, unsigned char * output)
187{
188 assert(output != NULL);
189
190 int header_length = 0;
191
192 if (length <= SCHAR_MAX) {
193 output[header_length++] = (unsigned char)length;
194 } else {
195 output[header_length++] = (unsigned char)(_Byte_Precision(length) | 0x80);
196 for (int byte_counter = _Byte_Precision(length); byte_counter; --byte_counter) {
197 output[header_length++] = (unsigned char)(length >> ((byte_counter-1)*8));
198 }
199 }
200 return(header_length);
201}

204/***********************************************************************************************
205 * XMP_DER_Encode -- Encode a number into a buffer using DER. *
206 * *
207 * This routine is used to encode a number into a buffer using Distinguished Encoding *
208 * Rules. The number of bytes used will be, typically, two bytes more than the number of *
209 * precision bytes in the number. *
210 * *
211 * INPUT: from -- Pointer to the multi-precision number. *
212 * *
213 * output -- Pointer to the buffer that will hold the DER encoded number. *
214 * *
215 * precision-- The precision of the multi-precision number. *
216 * *
217 * OUTPUT: Returns with the number of bytes used in the output buffer. *
218 * *
219 * WARNINGS: Make sure the buffer is big enough to hold the DER encoded number. For safety *
220 * make sure it is precision+6 bytes long. *
221 * *
222 * HISTORY: *
223 * 07/01/1996 JLB : Created. *
224 *=============================================================================================*/
225int MPEXPORT XMP_DER_Encode(digit const * from, unsigned char * output, int precision)
226{
227 assert(from != NULL);
228 assert(output != NULL);
229 assert(precision > 0);
230
231 unsigned char buffer[MAX_UNIT_PRECISION*sizeof(digit)+1];
232 int header_count = 0;
233
234 unsigned number_count = XMP_Encode(buffer, from, precision);
235
236 output[header_count++] = 0x02;
237 header_count += XMP_DER_Length_Encode(number_count, &output[header_count]);
238 memcpy(&output[header_count], buffer, number_count);
239
240 return(header_count+number_count);
241}

244/***********************************************************************************************
245 * XMP_DER_Decode -- Decode a DER number into an MP number. *
246 * *
247 * Use this routine to decode a Distinguished Encoding Rules number into a multi-precision *
248 * number. This is the counterpart function to the XMP_DER_Encode() function. *
249 * *
250 * INPUT: result -- The buffer the hold the result MP number. *
251 * *
252 * input -- Pointer to the DER encoded number. *
253 * *
254 * precision -- The precision of the MP number. This is the maximum precision the *
255 * DER number can be. *
256 * *
257 * OUTPUT: none *
258 * *
259 * WARNINGS: none *
260 * *
261 * HISTORY: *
262 * 07/01/1996 JLB : Created. *
263 *=============================================================================================*/
264void MPEXPORT XMP_DER_Decode(digit * result, unsigned char const * input, int precision)
265{
266 assert(result != NULL);
267 assert(input != NULL);
268 assert(precision > 0);
269
270 if (*input++ == 0x02) {
271 unsigned byte_count;
272
273 if ((*input & 0x80) == 0) {
274 byte_count = *input++;
275 } else {
276 int length = *input++ & 0x7f;
277 if (length > 2) return;
278 byte_count = *input++;
279 if (length > 1) byte_count = (byte_count << 8) | *input++;
280 }
281 if (byte_count <= (precision * sizeof(digit))) {
282 XMP_Signed_Decode(result, input, byte_count, precision);
283 }
284 }
285}

288/***********************************************************************************************
289 * XMP_Encode_Bounded -- Encode MP number into buffer. *
290 * *
291 * This routine will encode an multi-precision number into a buffer of specified length. *
292 * The number of stored in "big endian" format with appropriate sign extension. *
293 * *
294 * INPUT: to -- Pointer to the buffer to place the number. *
295 * *
296 * tobytes -- The number of bytes to use in the destination buffer. *
297 * *
298 * from -- Pointer to the MP number to be encoded. *
299 * *
300 * precision-- The precision of the MP number. *
301 * *
302 * OUTPUT: Returns with the number of bytes placed into the destination buffer. *
303 * *
304 * WARNINGS: none *
305 * *
306 * HISTORY: *
307 * 07/01/1996 JLB : Created. *
308 *=============================================================================================*/
309unsigned MPEXPORT XMP_Encode_Bounded(unsigned char * to, unsigned tobytes, digit const * from, int precision)
310{
311 assert(to != NULL);
312 assert(from != NULL);
313 assert(tobytes > 0);
314 assert(precision > 0);
315
316 unsigned frombytes = precision * sizeof(digit);
317 unsigned char filler = (unsigned char)(XMP_Is_Negative(from, precision) ? 0xff : 0);
318
319 unsigned index;
320 for (index = 0; index < (tobytes-frombytes); index++) {
321 *to++ = filler;
322 }
323
324 const unsigned char * fptr = ((const unsigned char *)from) + min(tobytes, frombytes);
325 for (index = 0; index < min(tobytes, frombytes); index++) {
326 *to++ = *--fptr;
327 }
328
329 return(tobytes);
330}

333/***********************************************************************************************
334 * XMP_Encode -- Encode MP number into buffer as compactly as possible. *
335 * *
336 * This routine will encode the MP number into the specified buffer. The number will be *
337 * encoded using the least number of bytes possible. *
338 * *
339 * INPUT: to -- The buffer to encode the MP number into. *
340 * *
341 * from -- Pointer to the MP number to be encoded. *
342 * *
343 * precision-- The precision of the MP number. *
344 * *
345 * OUTPUT: Returns with the number of bytes used in the destination buffer to hold the *
346 * encoded number. *
347 * *
348 * WARNINGS: Be sure the destination buffer is big enough to hold the encoded MP number. *
349 * A safe size would be the precision plus one. *
350 * *
351 * HISTORY: *
352 * 07/01/1996 JLB : Created. *
353 *=============================================================================================*/
354#ifdef __WATCOMC__
355#pragma warning 364 9
356#endif
357unsigned MPEXPORT XMP_Encode(unsigned char * to, digit const * from, int precision)
358{
359 assert(to != NULL);
360 assert(from != NULL);
361 assert(precision > 0);
362
363 bool is_negative = XMP_Is_Negative(from, precision);
364 unsigned char filler = (unsigned char)(is_negative ? 0xff : 0);
365 unsigned char * number_ptr;
366
367 unsigned char * const end = (unsigned char *)from;
368 for (number_ptr = (unsigned char *)end + precision - 1; number_ptr > (unsigned char *)end; number_ptr--) {
369 if (*number_ptr != filler) break;
370 }
371
372 unsigned index = 0;
373 if (((*number_ptr & 0x80) && !is_negative) || (!(*number_ptr & 0x80) && is_negative)) {
374 to[index++] = filler;
375 }
376
377 to[index++] = *number_ptr;
378
379 while (number_ptr != end) {
380 to[index++] = *--number_ptr;
381 }
382 return(index);
383}

386/***********************************************************************************************
387 * XMP_Signed_Decode -- Decode a number as if it were signed. *
388 * *
389 * Use this routine to convert a coded number back into an MP number. The coded number *
390 * is presumed to be signed. *
391 * *
392 * INPUT: result -- Pointer to the buffer that will hold the decoded MP number. *
393 * *
394 * from -- Pointer to the encoded MP number. *
395 * *
396 * frombytes-- The number of bytes consumed by the encoded MP number. *
397 * *
398 * precision -- The precision of the MP number (maximum) of the result. *
399 * *
400 * OUTPUT: none *
401 * *
402 * WARNINGS: Be sure that the precision is sufficient to hold the decoded MP number. *
403 * Otherwise, the result is undefined. *
404 * *
405 * HISTORY: *
406 * 07/01/1996 JLB : Created. *
407 *=============================================================================================*/
408void MPEXPORT XMP_Signed_Decode(digit * result, const unsigned char * from, int frombytes, int precision)
409{
410 assert(result != NULL);
411 assert(from != NULL);
412 assert(frombytes > 0);
413 assert(precision > 0);
414
415 unsigned char filler = (unsigned char)((*from & 0x80) ? 0xff : 0);
416
417 int fillcount = precision * sizeof(digit) - frombytes;
418 unsigned char * dest = (unsigned char *)&result[precision];
419
420 /*
421 ** Fill in any excess significant bytes.
422 */
423 int index;
424 for (index = 0; index < fillcount; index++) {
425 *--dest = filler;
426 }
427
428 /*
429 ** Move in the remaining bytes.
430 */
431 for (index = 0; index < frombytes; index++) {
432 *--dest = *from++;
433 }
434}

486 * XMP_Significance -- Fetch the precision (bytes) of the MP number. *
487 * *
488 * This routine will return with the precision of the MP number expressed as bytes. The *
489 * MP number is presumed unsigned. *
490 * *
491 * INPUT: number -- Pointer to the MP number to examine. *
492 * *
493 * precision-- The precision of the MP number to examine. *
494 * *
495 * OUTPUT: Returns with the minimum number of bytes consumed by this MP number. *
496 * *
497 * WARNINGS: Passing a signed MP number to this routine will return an artificially greater *
498 * precision than it really is. *
499 * *
500 * HISTORY: *
501 * 07/01/1996 JLB : Created. *
502 *=============================================================================================*/
503int MPEXPORT XMP_Significance(const digit * number, int precision)
504{
505 assert(number != NULL);
506 assert(precision > 0);
507
508 number += precision;
509 do {
510 number--;
511 if (*number) break;
512 precision--;
513 } while (precision);
514 return(precision);
515}

633/***********************************************************************************************
634 * XMP_Shift_Right_Bits -- Shift the MP number right by specified bit count. *
635 * *
636 * Use this routine to perform a right shift of the MP number by the number of bits *
637 * specified. *
638 * *
639 * INPUT: number -- Pointer to the MP number to perform the shift upon. *
640 * *
641 * bits -- The number of bits to shift. *
642 * *
643 * precision-- The precision of the MP number. *
644 * *
645 * OUTPUT: none *
646 * *
647 * WARNINGS: This is an unsigned shift. *
648 * *
649 * HISTORY: *
650 * 07/01/1996 JLB : Created. *
651 *=============================================================================================*/
652void MPEXPORT XMP_Shift_Right_Bits(digit * number, int bits, int precision)
653{
654 assert(number != NULL);
655 assert(bits >= 0);
656 assert(precision > 0);
657
658 if (bits == 0) return; /* shift zero bits is a no-op */
659
660 /*
661 ** If the shift is by whole bytes, then the shift operation can
662 ** be performed very quickly.
663 */
664 if (bits == UNITSIZE) {
665 number += precision;
666 digit carry = 0;
667 while (precision--) {
668 number--;
669 digit temp = *number;
670 *number = carry;
671 carry = temp;
672 }
673 return;
674 }
675
676 /*
677 ** If the number of bits to shift is less than one byte, then the
678 ** shift operation is a relatively simple "ripple" effect through
679 ** the MP number buffer.
680 */
681 if (bits < UNITSIZE) {
682 number += precision;
683 digit carry = 0;
684 digit bitmask = (1L << bits) - 1;
685 int unbits = UNITSIZE - bits;
686
687 while (precision--) {
688 number--;
689 digit temp = *number & bitmask;
690 *number >>= bits;
691 *number |= carry << unbits;
692 carry = temp;
693 }
694 return;
695 }
696
697 /*
698 ** General purpose slow right.
699 */
700 int digits_to_shift = bits / UNITSIZE;
701 int bits_to_shift = bits % UNITSIZE;
702
703 int index;
704 for (index = digits_to_shift; index < (precision-1); index++) {
705 *number = (*(number + digits_to_shift) >> bits_to_shift) | (*(number + (digits_to_shift + 1)) << (UNITSIZE - bits_to_shift));
706 number++;
707 }
708
709 if (digits_to_shift < precision) {
710 *number = (*(number + digits_to_shift) >> bits_to_shift);
711 number++;
712 }
713
714 for (index= 0; index < min(digits_to_shift, precision); index++) {
715 *number++ = 0;
716 }
717}

736 * 07/01/1996 JLB : Created. *
737 *=============================================================================================*/
738void MPEXPORT XMP_Shift_Left_Bits(digit * number, int bits, int precision)
739{
740 assert(number != NULL);
741 assert(bits >= 0);
742 assert(precision > 0);
743
744 if (bits == 0) return; /* shift zero bits is a no-op */
745
746 /*
747 ** If the shift is by whole bytes, then the shift operation can
748 ** be performed very quickly.
749 */
750 if (bits == UNITSIZE) {
751 digit carry = 0;
752 while (precision--) {
753 digit temp = *number;
754 *number = carry;
755 carry = temp;
756 number++;
757 }
758 return;
759 }
760
761 /*
762 ** If the number of bits to shift is less than one byte, then the
763 ** shift operation is a relatively simple "ripple" effect through
764 ** the MP number buffer.
765 */
766 if (bits < UNITSIZE) {
767 digit carry = 0;
768 digit bitmask = ~(((digit)-1) >> bits);
769 int unbits = UNITSIZE - bits; /* shift bits must be <= UNITSIZE */
770
771 while (precision--) {
772 digit temp = *number & bitmask;
773 *number = (*number << bits) | (carry >> unbits);
774 carry = temp;
775 number++;
776 }
777 return;
778 }
779
780 /*
781 ** General purpose slow left;
782 */
783 int digits_to_shift = bits / UNITSIZE;
784 int bits_to_shift = bits % UNITSIZE;
785
786 int index;
787 number += precision-1;
788 for (index = digits_to_shift; index < (precision-1); index++) {
789 *number = (*(number - digits_to_shift) << bits_to_shift) | (*(number - (digits_to_shift + 1)) >> (UNITSIZE - bits_to_shift));
790 number--;
791 }
792
793 if (digits_to_shift < precision) {
794 *number = (*(number - digits_to_shift) << bits_to_shift);
795 number--;
796 }
797
798 for (index = 0; index < min(digits_to_shift, precision); index++) {
799 *number-- = 0;
800 }
801}

1433 if ((*dividend_ptr) & high_bit_mask) remainder++;
1434
1435 if (remainder >= divisor) {
1436 remainder -= divisor;
1437 *quotient_ptr |= high_bit_mask;
1438 }
1439
1440 high_bit_mask >>= 1;
1441 if (!high_bit_mask) {
1442 high_bit_mask = UPPER_MOST_BIT;
1443 --dividend_ptr;
1444 --quotient_ptr;
1445 }
1446 }
1447
1448 return(remainder);
1449}

1454 * *

1456 * slowness of this routine, both the quotient and the remainder are available as a *
1457 * result of the operation. *
1458 * *
1459 * INPUT: remainder -- Pointer to the MP buffer that will hold the remainder of the divide.*
1460 * *
1461 * quotient -- Pointer to the MP buffer that will hold the quotient of the divide. *
1462 * *

1464 * *
1465 * divisor -- The MP divisor (denominator) number. *
1466 * *

1468 * *

1470 * *
1471 * WARNINGS: This is very slow. *
1472 * *
1473 * HISTORY: *
1474 * 07/02/1996 JLB : Created. *
1475 *=============================================================================================*/

1479 if (XMP_Test_Eq_Int(divisor, 0, precision)) return(-1);
1480

1482 XMP_Init(quotient, 0, precision);
1483

1500 XMP_Sub(remainder, remainder, divisor, 0, precision);
1501 *quotient_ptr |= high_bit_mask;
1502 }
1503
1504 high_bit_mask >>= 1;
1505 if (!high_bit_mask) {
1506 high_bit_mask = UPPER_MOST_BIT;
1507 dividend_ptr--;
1508 quotient_ptr--;
1509 }
1510
1511 }
1512 return 0;

1625 * *
1626 * Use this routine to determine the inverse of the specified MP number. The inverse is *
1627 * defined as 1/number. *
1628 * *
1629 * INPUT: result -- Pointer to the result MP number buffer. *
1630 * *

1632 * *

1636 * *
1637 * WARNINGS: none *
1638 * *
1639 * HISTORY: *
1640 * 07/02/1996 JLB : Created. *
1641 *=============================================================================================*/

1643{

1645
1646 if (XMP_Test_Eq_Int(divisor, 0, precision)) return -1; /* zero divisor means divide error */

1699{
1700 /*
1701 ** Initialize the multiprecision number to zero. From this point
1702 ** onward, this object can be manipulated as a regular number.
1703 ** This is, in fact, what is done as the ascii string is parsed
1704 ** into a working number.
1705 */

1709 ** No string or zero length is considered '0'.
1710 */

1720

1723 break;
1724
1725 case 'O':

1727 break;

1778/***********************************************************************************************
1779 * XMP_Hybrid_Mul -- Special hybrid short multiply (with carry). *
1780 * *
1781 * Multiply the single-word multiplier times the multiprecision integer *
1782 * in multiplicand, accumulating result in prod. The resulting *
1783 * multiprecision prod will be 1 word longer than the multiplicand. *
1784 * multiplicand is double precision words long. We add into prod, so caller *
1785 * should zero it out first. For best results, this time-critical *
1786 * function should be implemented in assembly. *
1787 * NOTE: Unlike other functions in the multiprecision arithmetic *
1788 * library, both multiplicand and prod are pointing at the LSB, *
1789 * regardless of byte order of the machine. On an 80x86, this makes *
1790 * no difference. But if this assembly function is implemented *
1791 * on a 680x0, it becomes important. *
1792 * *
1793 * Note that this has been modified from the previous version to allow *
1794 * better support for Smith's modmult: *
1795 * The final carry bit is added to the existing product *
1796 * array, rather than simply stored. *
1797 * *
1798 * INPUT: prod -- Pointer to the product MP number buffer. *
1799 * *
1800 * multiplicand -- Pointer to the multiplicand MP number. *
1801 * *
1802 * multiplier -- The short integer used as the multiplier. *
1803 * *
1804 * precision -- The precision of the MP number used. *
1805 * *
1806 * OUTPUT: none *
1807 * *
1808 * WARNINGS: The carry (if any) is added into the integer one beyond the end of the *
1809 * product buffer. *
1810 * *
1811 * HISTORY: *
1812 * 07/02/1996 JLB : Created. *
1813 *=============================================================================================*/
1814void XMP_Hybrid_Mul(unsigned short * prod, unsigned short * multiplicand, unsigned short multiplier, int precision)
1815{
1816 unsigned long carry = 0;
1817 for (int i = 0; i < precision; ++i) {
1818 unsigned long p = (unsigned long)multiplier * *multiplicand++;
1819 p += *prod + carry;
1820 *prod++ = (unsigned short) p;
1821 carry = p >> 16;
1822 }
1823
1824 /* Add carry to the next higher word of product / dividend */
1825 *prod += (unsigned short) carry;
1826}

1885/***********************************************************************************************
1886 * XMP_Prepare_Modulus -- Prepare globals for modulus operation. *
1887 * *
1888 * Calculate the reciprocal of modulus with a precision of two MULTUNITs. *
1889 * Assumes that precision has already been adjusted to the *
1890 * size of the modulus, plus SLOP_BITS. *
1891 * *
1892 * Note: This routine was designed to work with large values and *
1893 * doesn't have the necessary testing or handling to work with a *
1894 * modulus having less than three significant digits. For such cases, *
1895 * the separate multiply and modulus routines can be used. *
1896 * *
1897 * INPUT: modulus -- Pointer to the modulus number. *
1898 * *
1899 * precision-- The precision of the modulus number. *
1900 * *
1901 * OUTPUT: none *
1902 * *
1903 * WARNINGS: none *
1904 * *
1905 * HISTORY: *
1906 * 07/02/1996 JLB : Created. *
1907 *=============================================================================================*/
1908int XMP_Prepare_Modulus(const digit * n_modulus, int precision)
1909{
1910 XMP_Move(_scratch_modulus, n_modulus, precision);
1911
1912 _modulus_bit_count = XMP_Count_Bits(_scratch_modulus, precision);
1913 _modulus_sub_precision = (_modulus_bit_count + 16 - 1) / 16;
1914
1915 /*
1916 ** Keep 2*16 bits in _mod_divisor.
1917 ** This will (normally) result in a reciprocal of 2*16+1 bits.
1918 */
1919 int sub_precision = XMP_Significance(_scratch_modulus, precision); // significant digits in modulus
1920 XMP_Move(_mod_divisor, &_scratch_modulus[sub_precision-2], 2);
1921 _modulus_shift = XMP_Count_Bits(_mod_divisor, 2) - 2 * 16;
1922 XMP_Shift_Right_Bits(_mod_divisor, _modulus_shift, 2);
1923
1924 XMP_Reciprocal(_mod_quotient, _mod_divisor, 2);
1925 XMP_Shift_Right_Bits(_mod_quotient, 1, 2);
1926
1927 /* Reduce to: 0 < _modulus_shift <= 16 */
1928 _modulus_shift = ((_modulus_shift + (16 - 1)) % 16) + 1;
1929
1930 /* round up */
1931 XMP_Inc(_mod_quotient, 2);
1932 if (XMP_Count_Bits(_mod_quotient, 2) > 2 * 16) {
1933 XMP_Shift_Right_Bits(_mod_quotient, 1, 2);
1934 _modulus_shift--; /* now 0 <= _modulus_shift <= 16 */
1935 }
1936 unsigned short * mpm = (unsigned short *) _mod_quotient;
1937 _reciprical_low_digit = *mpm++;
1938 _reciprical_high_digit = *mpm;
1939
1940 return 0;
1941}

1944/***********************************************************************************************
1945 * XMP_Mod_Mult -- Perform a multiply - modulus operation. *
1946 * *
1947 * This routine will combine a multiply and a modulus operation. This takes advantage of *
1948 * a tremendous speed advantage possible if these two processes are combined rather than *
1949 * being performed separately. *
1950 * *
1951 * INPUT: prod -- Pointer to the MP buffer that will hold the result. *
1952 * *
1953 * multiplicand-- The number to multiply. *
1954 * *
1955 * multiplier -- The number to multiply by. *
1956 * *
1957 * precision -- The precision of the MP numbers involved. *
1958 * *
1959 * OUTPUT: none *
1960 * *
1961 * WARNINGS: The modulus must already have been prepared by the routine XMP_Prepare_Modulus. *
1962 * *
1963 * HISTORY: *
1964 * 07/02/1996 JLB : Created. *
1965 *=============================================================================================*/
1966int MPEXPORT XMP_Mod_Mult(digit * prod, const digit * multiplicand, const digit * multiplier, int precision)
1967{
1968 XMP_Double_Mul(_double_staging_number, multiplicand, multiplier, precision);
1969
1970 int double_precision = precision * 2 + 1;
1971
1972 _double_staging_number[double_precision - 1] = 0; /* leading 0 digit */
1973
1974 /*
1975 ** We now start working with MULTUNITs.
1976 ** Determine the most significant MULTUNIT of the product so we don't
1977 ** have to process leading zeros in our divide loop.
1978 */
1979 int dmi = XMP_Significance(_double_staging_number, double_precision) * 2; // number of significant MULTUNITs in product
1980
1981 if (dmi >= _modulus_sub_precision) {
1982
1983 /* Make dividend negative. This allows the use of mp_single_mul to
1984 ** "subtract" the product of the modulus and the trial divisor
1985 ** by actually adding to a negative dividend.
1986 ** The one's complement of the dividend is used, since it causes
1987 ** a zero value to be represented as all ones. This facilitates
1988 ** testing the result for possible overflow, since a sign bit
1989 ** indicates that no adjustment is necessary, and we should not
1990 ** attempt to adjust if the result of the addition is zero.
1991 */
1992 XMP_Inc(_double_staging_number, double_precision);
1993 XMP_Neg(_double_staging_number, double_precision);
1994
1995 int nqd = dmi + 1 - _modulus_sub_precision; // number of quotient digits remaining to be generated
1996
1997 /* Set msb, lsb, and normal ptrs of dividend */
1998 unsigned short * dmph = ((unsigned short *)_double_staging_number) + dmi + 1; // points to one higher than precision would indicate
1999 unsigned short * dmpl = dmph - _modulus_sub_precision;
2000
2001 /*
2002 ** Divide loop.
2003 ** Each iteration computes the next quotient MULTUNIT digit, then
2004 ** multiplies the divisor (modulus) by the quotient digit and adds
2005 ** it to the one's complement of the dividend (equivalent to
2006 ** subtracting). If the product was greater than the remaining dividend,
2007 ** we get a non-negative result, in which case we subtract off the
2008 ** modulus to get the proper negative remainder.
2009 */
2010 for (; nqd; nqd--) {
2011 --dmph;
2012 --dmpl;
2013
2014 unsigned short q = mp_quo_digit(dmph); // trial quotient digit
2015 if (q > 0) {
2016 XMP_Hybrid_Mul(dmpl, (unsigned short *)_scratch_modulus, q, precision*2);
2017
2018 /* Perform correction if q too large.
2019 ** This rarely occurs.
2020 */
2021 if (!(*dmph & SEMI_UPPER_MOST_BIT)) {
2022 unsigned short * dmp = dmpl;
2023 if (XMP_Sub((unsigned long *)dmp, (unsigned long *)dmp, _scratch_modulus, false, precision)) {
2024 (*dmph)--;
2025 }
2026 }
2027 }
2028 }
2029
2030 /* d contains the one's complement of the remainder. */
2031 XMP_Neg(_double_staging_number, precision);
2032 XMP_Dec(_double_staging_number, precision);
2033 }
2034
2035 XMP_Move(prod, _double_staging_number, precision);
2036 return (0);
2037}

2069/*
2070** The function mp_quo_digit is the heart of Smith's modulo reduction,
2071** which uses a form of long division. It computes a trial quotient
2072** "digit" (MULTUNIT-sized digit) by multiplying the three most
2073** significant MULTUNITs of the dividend by the two most significant
2074** MULTUNITs of the reciprocal of the modulus. Note that this function
2075** requires that 16 * 2 <= sizeof(unsigned long).
2076**
2077** An important part of this technique is that the quotient never be
2078** too small, although it may occasionally be too large. This was
2079** done to eliminate the need to check and correct for a remainder
2080** exceeding the divisor. It is easier to check for a negative
2081** remainder. The following technique rarely needs correction for
2082** MULTUNITs of at least 16 bits.
2083**
2084** The following routine has two implementations:
2085**
2086** Parameter: dividend - points to the most significant MULTUNIT
2087** of the dividend. Note that dividend actually contains the
2088** one's complement of the actual dividend value (see comments for
2089** XMP_Mod_Mult).
2090**
2091** Return: the trial quotient digit resulting from dividing the first
2092** three MULTUNITs at dividend by the upper two MULTUNITs of the
2093** modulus.
2094*/
2095unsigned short mp_quo_digit(unsigned short * dividend)
2096{
2097 unsigned long q, q0, q1, q2;
2098
2099 /*
2100 * Compute the least significant product group.
2101 * The last terms of q1 and q2 perform upward rounding, which is
2102 * needed to guarantee that the result not be too small.
2103 */
2104 q1 = (dividend[-2] ^ SEMI_MASK) * (unsigned long) _reciprical_high_digit + _reciprical_high_digit;
2105 q2 = (dividend[-1] ^ SEMI_MASK) * (unsigned long) _reciprical_low_digit + (1L << 16);
2106 q0 = (q1 >> 1) + (q2 >> 1) + 1;
2107
2108 /* Compute the middle significant product group. */
2109 q1 = (dividend[-1] ^ SEMI_MASK) * (unsigned long) _reciprical_high_digit;
2110 q2 = (dividend[0] ^ SEMI_MASK) * (unsigned long) _reciprical_low_digit;
2111 q = (q0 >> 16) + (q1 >> 1) + (q2 >> 1) + 1;
2112
2113 /* Compute the most significant term and add in the others */
2114 q = (q >> (16 - 2)) + (((dividend[0] ^ SEMI_MASK) * (unsigned long) _reciprical_high_digit) << 1);
2115 q >>= _modulus_shift;
2116
2117 /* Prevent overflow and then wipe out the intermediate results. */
2118 return (unsigned short) min(q, (unsigned long)(1L << 16) - 1);
2119}

2353 * XMP_Rabin_Miller_Test -- Performs the Rabin Miller test for primality. *
2354 * *
2355 * This test for primality is even more expensive the Fermat's Little Theorem. It doesn't *
2356 * prove that a number is prime, but it can prove that it is not prime. *
2357 * *
2358 * INPUT: rng -- Reference to to a random number generator. *
2359 * *
2360 * candidate-- Pointer to the candidate MP number that is to be tested. *
2361 * *
2362 * rounds -- The number of test rounds to perform. *
2363 * *
2364 * precision-- The precision of the MP number specified. *
2365 * *
2366 * OUTPUT: bool; Was the number not proven to be not prime? A FALSE means that the number is *
2367 * not prime. A TRUE means that it might be. *
2368 * *
2369 * WARNINGS: This routine takes a long time. Use as few rounds as possible. *
2370 * *
2371 * HISTORY: *
2372 * 07/02/1996 JLB : Created. *
2373 *=============================================================================================*/
2374bool MPEXPORT XMP_Rabin_Miller_Test(Straw & rng, digit const * w, int rounds, int precision)
2375{
2376 digit wminus1[MAX_UNIT_PRECISION];
2377 XMP_Sub_Int(wminus1, w, 1, 0, precision);
2378
2379 unsigned maxbitprecision = precision * sizeof(digit) * 8;
2380 unsigned a;
2381 for (a = 0; a < maxbitprecision; a++) {
2382 if (XMP_Test_Bit(wminus1, a)) {
2383 break;
2384 }
2385 }
2386
2387 digit m[MAX_UNIT_PRECISION];
2388 XMP_Move(m, wminus1, precision);
2389 XMP_Shift_Right_Bits(wminus1, a, precision);
2390
2391 for (int i = 0; i < rounds; i++) {
2392 digit b[MAX_UNIT_PRECISION];
2393 digit temp[MAX_UNIT_PRECISION];
2394 XMP_Init(temp, 2, precision);
2395 XMP_Randomize_Bounded(b, rng, temp, wminus1, precision);
2396
2397 digit z[MAX_UNIT_PRECISION];
2398 XMP_Exponent_Mod(z, b, m, w, precision);
2399
2400 if (XMP_Test_Eq_Int(z, 1, precision) || XMP_Compare(z, wminus1, precision) == 0) {
2401 continue; // passes this round
2402 }
2403
2404 unsigned j;
2405 for (j = 1; j < a; j++) {
2406 digit t2[MAX_UNIT_PRECISION];
2407 XMP_Exponent_Mod(t2, z, temp, w, precision);
2408
2409 if (XMP_Compare(t2, wminus1, precision) == 0) {
2410 break; // passed this round
2411 }
2412 if (XMP_Test_Eq_Int(z, 1, precision)) {
2413 return false;
2414 }
2415 }
2416 if (j == a) {
2417 return false;
2418 }
2419 }
2420 return true;
2421}

2520 ** Even numbers are ALWAYS not prime.
2521 */
2522 if (!(*prime & 0x01)) return(false);
2523
2524 /*
2525 ** Compare the prime number against the exhaustive list of prime
