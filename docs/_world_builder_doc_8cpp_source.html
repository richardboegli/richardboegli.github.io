162class MFCFileOutputStream : public OutputStream

168 virtual Int write(const void *pData, Int numBytes) {

178struct CachedChunk
179{
180 UnsignedByte *pData;
181 Int size;
182};
183
184class CachedMFCFileOutputStream : public OutputStream
185{
186protected:
187 CFile *m_file;
188 std::list<CachedChunk> m_cachedChunks;
189 Int m_totalBytes;
190public:
191 CachedMFCFileOutputStream(CFile *pFile):m_file(pFile), m_totalBytes(0) {};
192 virtual Int write(const void *pData, Int numBytes) {
193 UnsignedByte *tmp = new UnsignedByte[numBytes];
194 memcpy(tmp, pData, numBytes);
195 CachedChunk c;
196 c.pData = tmp;
197 c.size = numBytes;
198 m_cachedChunks.push_back(c);
199 DEBUG_LOG(("Caching %d bytes in chunk %d\n", numBytes, m_cachedChunks.size()));
200 m_totalBytes += numBytes;
201 return(numBytes);
202 };
203 virtual void flush(void) {
204 while (m_cachedChunks.size() != 0)
205 {
206 CachedChunk c = m_cachedChunks.front();
207 m_cachedChunks.pop_front();
208 try {
209 DEBUG_LOG(("Flushing %d bytes\n", c.size));
210 m_file->Write(c.pData, c.size);
211 } catch(...) {}
212 delete[] c.pData;
213 m_totalBytes -= c.size;
214 }
215 }
216};

218class CompressedCachedMFCFileOutputStream : public OutputStream
219{
220protected:
221 CFile *m_file;
222 std::list<CachedChunk> m_cachedChunks;
223 Int m_totalBytes;
224public:
225 CompressedCachedMFCFileOutputStream(CFile *pFile):m_file(pFile), m_totalBytes(0) {};
226 virtual Int write(const void *pData, Int numBytes) {
227 UnsignedByte *tmp = new UnsignedByte[numBytes];
228 memcpy(tmp, pData, numBytes);
229 CachedChunk c;
230 c.pData = tmp;
231 c.size = numBytes;
232 m_cachedChunks.push_back(c);
233 //DEBUG_LOG(("Caching %d bytes in chunk %d\n", numBytes, m_cachedChunks.size()));
234 m_totalBytes += numBytes;
235 return(numBytes);
236 };
237 virtual void flush(void) {
238 if (!m_totalBytes)
239 return;
240 UnsignedByte *srcBuffer = NEW UnsignedByte[m_totalBytes];
241 UnsignedByte *insertPos = srcBuffer;
242 while (m_cachedChunks.size() != 0)
243 {
244 CachedChunk c = m_cachedChunks.front();
245 m_cachedChunks.pop_front();
246 try {
247 //DEBUG_LOG(("Flushing %d bytes\n", c.size));
248 memcpy(insertPos, c.pData, c.size);
249 insertPos += c.size;
250 } catch(...) {}
251 delete[] c.pData;
252 }
253 CompressionType compressionToUse = CompressionManager::getPreferredCompression();
254 Dict *worldDict = MapObject::getWorldDict();
255 if (worldDict)
256 {
257 Bool exists = FALSE;
258 compressionToUse = (CompressionType)worldDict->getInt(TheKey_compression, &exists);
259 if (!exists || compressionToUse > COMPRESSION_MAX || compressionToUse < COMPRESSION_MIN)
260 compressionToUse = CompressionManager::getPreferredCompression();
261 }
262
263 Int compressedLen = CompressionManager::getMaxCompressedSize( m_totalBytes, compressionToUse );
264 UnsignedByte *destBuffer = NEW UnsignedByte[compressedLen];
265 compressedLen = CompressionManager::compressData( compressionToUse, srcBuffer, m_totalBytes, destBuffer, compressedLen );
266 DEBUG_LOG(("Compressed %d bytes to %d bytes - compression of %g%%\n", m_totalBytes, compressedLen,
267 compressedLen/(Real)m_totalBytes*100.0f));
268 DEBUG_ASSERTCRASH(compressedLen, ("Failed to compress!\n"));
269 if (compressedLen)
270 {
271 m_file->Write(destBuffer, compressedLen);
272 }
273 else
274 {
275 m_file->Write(srcBuffer, m_totalBytes);
276 }
277 delete[] srcBuffer;
278 srcBuffer = NULL;
279 delete[] destBuffer;
280 destBuffer = NULL;
281 }
282};

285void CWorldBuilderDoc::Serialize(CArchive& ar)
286{
287 ar.Flush();
288 m_waypointTableNeedsUpdate = true;
289 if (ar.IsStoring() && m_heightMap)
290 {
291 try {
292 Int i;
293 MapPreview mPreview;
294 mPreview.save(ar.GetFile()->GetFilePath());
295
296 CompressedCachedMFCFileOutputStream theStream(ar.GetFile());
297 DataChunkOutput *chunkWriter = new DataChunkOutput(&theStream);
298
299
300 m_heightMap->saveToFile(*chunkWriter);
301 /***************WAYPOINTS DATA ***************/
302 chunkWriter->openDataChunk("WaypointsList", K_WAYPOINTS_VERSION_1);
303 chunkWriter->writeInt(this->m_numWaypointLinks);
304 for (i=0; i<m_numWaypointLinks; i++) {
305 chunkWriter->writeInt(this->m_waypointLinks[i].waypoint1);
306 chunkWriter->writeInt(this->m_waypointLinks[i].waypoint2);
307 }
308 chunkWriter->closeDataChunk();
309
310 delete chunkWriter;
311 chunkWriter = NULL;
312 theStream.flush();
313 } catch(...) {
314 char *msg = "WorldHeightMapEdit::WorldHeightMapEdit height map file write failed: ";
315 AfxMessageBox(msg);
316 return;
317 }
318 }
319 else
320 {
321 WorldHeightMapEdit *pOldHeightMap = m_heightMap;
322 CString pth = ar.GetFile()->GetFilePath();
323 CachedFileInputStream theInputStream;
324 if (theInputStream.open(AsciiString((const char *)pth)))
325 try {
326
327 WbApp()->selectPointerTool();
328 PolygonTrigger::deleteTriggers();
329 ChunkInputStream *pStrm = &theInputStream;
330
331 // Read the logical data (map objects, waypoints, etc.)
332 WorldHeightMap *terrainHeightMap = new WorldHeightMap(pStrm, true);
333 REF_PTR_RELEASE(terrainHeightMap);
334 pStrm->absoluteSeek(0);
335 // Read & keep the graphical data.
336 m_heightMap = NEW_REF(WorldHeightMapEdit, (pStrm));
337 pStrm->absoluteSeek(0);
338 try {
339 DataChunkInput file( pStrm );
340 if (file.isValidFileType()) { // Backwards compatible files aren't valid data chunk files.
341 // Read the waypoints.
342 file.registerParser( AsciiString("WaypointsList"), AsciiString::TheEmptyString, ParseWaypointDataChunk );
343 if (!file.parse(this)) {
344 throw(ERROR_CORRUPT_FILE_FORMAT);
345 }
346 }
347 } catch(...) {
348 // just eat the error - legacy files aren't chunk format.
349 }
350 theInputStream.close();
351
352 validate();
353
354 compressWaypointIds(); // remove any unused waypoint ids.
355 WbView3d * p3View = Get3DView();
356 if (p3View) {
357 p3View->resetRenderObjects();
358 }
359 m_heightMap->optimizeTiles(); // force to optimize tileset
360 SetHeightMap(m_heightMap, true);
361 Coord3D center;
362 center.x = MAP_XY_FACTOR*m_heightMap->getXExtent()/2;
363 center.y = MAP_XY_FACTOR*m_heightMap->getYExtent()/2;
364 center.x -= m_heightMap->getBorderSize();
365 center.y -= m_heightMap->getBorderSize();
366 /* update objects. */
367 AsciiString startingCamName = TheNameKeyGenerator->keyToName(TheKey_InitialCameraPosition);
368
369 TheLayersList->resetLayers();
370 AsciiString layerName;
371 Bool exists;
372
373 // always assign unique IDs. The things will still live in the correct layers, so this isn't
374 // an especially big deal.
375 MapObject::fastAssignAllUniqueIDs();
376
377 TheLayersList->disableUpdates();
378 MapObject *pMapObj = MapObject::getFirstMapObject();
379 while (pMapObj) {
380
381 // Then, add it to the Layers List
382 layerName = pMapObj->getProperties()->getAsciiString(TheKey_objectLayer, &exists);
383 if (exists) {
384 TheLayersList->addMapObjectToLayersList(pMapObj, layerName);
385 } else {
386 TheLayersList->addMapObjectToLayersList(pMapObj);
387 }
388
389 MapObject *pTemplateObj = ObjectOptions::getObjectNamed(pMapObj->getName());
390 if (pTemplateObj) {
391 pMapObj->setColor(pTemplateObj->getColor());
392 }
393 if (pMapObj->isWaypoint()) {
394 if (pMapObj->getWaypointID() >= m_curWaypointID) {
395 m_curWaypointID = pMapObj->getWaypointID();
396 }
397 if (startingCamName == pMapObj->getWaypointName()) {
398 center = *pMapObj->getLocation();
399 }
400 }
401 pMapObj = pMapObj->getNext();
402 }
403
404 PolygonTrigger* polyTrigger = PolygonTrigger::getFirstPolygonTrigger();
405 // Add the triggers to the layers list.
406 while (polyTrigger) {
407 layerName = polyTrigger->getLayerName();
408 TheLayersList->addPolygonTriggerToLayersList(polyTrigger, layerName);
409
410 polyTrigger = polyTrigger->getNext();
411 }
412
413 TheLayersList->enableUpdates();
414
415 TerrainMaterial::updateTextures(m_heightMap);
416
417 REF_PTR_RELEASE(m_undoList);
418 m_curRedo = 0;
419 POSITION pos = GetFirstViewPosition();
420 while (pos != NULL)
421 {
422 CView* pView = GetNextView(pos);
423 WbView* pWView = (WbView *)pView;
424 ASSERT_VALID(pWView);
425 pWView->setCenterInView(center.x/MAP_XY_FACTOR, center.y/MAP_XY_FACTOR);
426 }
427 REF_PTR_RELEASE(pOldHeightMap);
428 if (p3View) {
429 p3View->setDefaultCamera();
430 }
431
432 } catch(...) {
433 m_heightMap = pOldHeightMap;
434 }
435
436 // note - mHeight map has ref count of 1.
437 }
438}

491void CWorldBuilderDoc::validate(void)
492{
493 DEBUG_LOG(("Validating\n"));
494
495 Dict swapDict;
496 Bool changed = false;
497 AsciiString swapName;
498
499 Bool needToFixTeams = false;
500
501 // verify/fix the build lists
502 for (int side=0; side<TheSidesList->getNumSides(); side++) {
503 SidesInfo *pSide = TheSidesList->getSideInfo(side);
504
505 AsciiString tmplname = pSide->getDict()->getAsciiString(TheKey_playerFaction);
506 AsciiString playername = pSide->getDict()->getAsciiString(TheKey_playerName);
507 if (tmplname.isEmpty()) {
508 continue; // Neutral player has empty template. jba. [8/8/2003]
509 }
510 const PlayerTemplate* pt = ThePlayerTemplateStore->findPlayerTemplate(NAMEKEY(tmplname));
511 if (!pt) {
512 DEBUG_LOG(("Player '%s' Faction '%s' could not be found in sides list!\n", playername.str(), tmplname.str()));
513 if (tmplname.startsWith("FactionFundamentalist")) {
514 swapName = ConvertFaction(tmplname);
515 if (swapName != AsciiString::TheEmptyString) {
516 DEBUG_LOG(("Changing Faction from %s to %s\n", tmplname.str(), swapName.str()));
517 pSide->getDict()->setAsciiString(TheKey_playerFaction, swapName);
518 }
519 }
520 }
521
522 BuildListInfo *pBuild = pSide->getBuildList();
523 while (pBuild) {
524 AsciiString name = pBuild->getTemplateName();
525 if (name.startsWith("Fundamentalist")) {
526 swapName = ConvertName(name);
527 if (swapName != AsciiString::TheEmptyString) {
528 DEBUG_LOG(("Changing BuildList from %s to %s\n", name.str(), swapName.str()));
529 pBuild->setTemplateName(swapName);
530 }
531 }
532 pBuild = pBuild->getNext();
533 }
534 }
535
536
537#define FIX_TEAM(key) \
538 type = teamDict->getAsciiString(key, &exists); \
539 if (exists) { \
540 if (type.startsWith("Fundamentalist")) { \
541 swapName = ConvertName(type); \
542 if (swapName != AsciiString::TheEmptyString) { \
543 DEBUG_LOG(("Changing Team Ref from %s to %s\n", type.str(), swapName.str())); \
544 teamDict->setAsciiString(key, swapName); \
545 } \
546 } \
547 } \
548
549 // verify/fix the team definitions
550 Int numTeams = TheSidesList->getNumTeams();
551 for (Int team=0; team<numTeams; team++)
552 {
553 TeamsInfo *ti = TheSidesList->getTeamInfo(team);
554 Dict* teamDict = ti->getDict();
555 AsciiString type;
556 Bool exists;
557 FIX_TEAM(TheKey_teamUnitType1)
558 FIX_TEAM(TheKey_teamUnitType2)
559 FIX_TEAM(TheKey_teamUnitType3)
560 FIX_TEAM(TheKey_teamUnitType4)
561 FIX_TEAM(TheKey_teamUnitType5)
562 FIX_TEAM(TheKey_teamUnitType6)
563 FIX_TEAM(TheKey_teamUnitType7)
564 }
565
566 MapObject *pMapObj;
567 for (pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext())
568 {
569 // there is no validation code for these items as of yet.
570 if (pMapObj->isScorch() || pMapObj->isWaypoint() || pMapObj->isLight() || pMapObj->getFlag(FLAG_ROAD_FLAGS) || pMapObj->getFlag(FLAG_BRIDGE_FLAGS))
571 {
572 continue;
573 }
574
575 // at this point, only objects with models and teams should be left to process
576
577 // start by verifying the ThingTemplate for the object.
578 // swapDict contains a 'history' of missing model swaps done this load, so all objects with a
579 // particular name are replaced with the exact same model.
580 AsciiString name = pMapObj->getName();
581 if (pMapObj->getThingTemplate() == NULL)
582 {
583 Bool exists = false;
584 swapName = swapDict.getAsciiString(NAMEKEY(name), &exists);
585
586 // quick hack to make loading models with "Fundamentalist" switch to "GLA"
587 if (name.startsWith("Fundamentalist")) {
588 swapName = ConvertName(name);
589 if (swapName != AsciiString::TheEmptyString) {
590 swapDict.setAsciiString(NAMEKEY(name), swapName);
591 exists = true;
592 }
593 }
594
595 // quick hack to remove "GC_" objects from Generals mission disk maps.
596 if (name.startsWith("GC_")) {
597 swapName = ConvertToNonGCName(name);
598 if (swapName != AsciiString::TheEmptyString) {
599 swapDict.setAsciiString(NAMEKEY(name), swapName);
600 exists = true;
601 }
602 }
603
604 if (!exists) {
605 ReplaceUnitDialog dlg;
606 dlg.setMissing(name);
607 for (int i = ES_FIRST; i<ES_NUM_SORTING_TYPES; i++) {
608 dlg.SetAllowableType((EditorSortingType)i);
609 }
610 dlg.SetFactionOnly(false);
611 if (dlg.DoModal() == IDOK) {
612 const ThingTemplate* thing = dlg.getPickedThing();
613 if (thing) {
614 swapName = thing->getName();
615 swapDict.setAsciiString(NAMEKEY(name), swapName);
616 }
617 }
618 }
619 swapName = swapDict.getAsciiString(NAMEKEY(name), &exists);
620 if (exists)
621 {
622 const ThingTemplate *tt = TheThingFactory->findTemplate(swapName);
623 if (tt) {
624 changed = true;
625 pMapObj->setName(swapName);
626 pMapObj->setThingTemplate(tt);
627 DEBUG_LOG(("Changing Map Object from %s to %s\n", name.str(), swapName.str()));
628 }
629 }
630 }
631
632
633 // the following code verifies and fixes the team name, player name, and faction linkages
634 Bool exists;
635 AsciiString teamName = pMapObj->getProperties()->getAsciiString(TheKey_originalOwner, &exists);
636 if (exists) {
637 TeamsInfo *teamInfo = TheSidesList->findTeamInfo(teamName);
638 if (teamInfo) {
639 AsciiString teamOwner = teamInfo->getDict()->getAsciiString(TheKey_teamOwner);
640 SidesInfo* pSide = TheSidesList->findSideInfo(teamOwner);
641 if (pSide) {
642// Bool hasColor = false;
643 AsciiString tmplname = pSide->getDict()->getAsciiString(TheKey_playerFaction);
644 AsciiString playername = pSide->getDict()->getAsciiString(TheKey_playerName);
645 if (tmplname.isEmpty()) {
646 continue; // Neutral player has empty template. jba. [8/8/2003]
647 }
648 const PlayerTemplate* pt = ThePlayerTemplateStore->findPlayerTemplate(NAMEKEY(tmplname));
649 if (!pt) {
650 DEBUG_LOG(("Player '%s' Faction '%s' could not be found in sides list!\n", playername.str(), tmplname.str()));
651 if (tmplname.startsWith("FactionFundamentalist")) {
652 swapName = ConvertFaction(tmplname);
653 if (swapName != AsciiString::TheEmptyString) {
654 DEBUG_LOG(("Changing Faction from %s to %s\n", tmplname.str(), swapName.str()));
655 pSide->getDict()->setAsciiString(TheKey_playerFaction, swapName);
656 }
657 }
658 }
659 } else {
660 needToFixTeams = true;
661 DEBUG_LOG(("Side '%s' could not be found in sides list!\n", teamOwner.str()));
662 }

721BOOL CWorldBuilderDoc::DoSave(LPCTSTR lpszPathName, BOOL bReplace)
722 // Save the document data to a file
723 // lpszPathName = path name where to save document file
724 // if lpszPathName is NULL then the user will be prompted (SaveAs)
725 // note: lpszPathName can be different than 'm_strPathName'
726 // if 'bReplace' is TRUE will change file name if successful (SaveAs)
727 // if 'bReplace' is FALSE will not change path name (SaveCopyAs)
728{
729 CString newName = lpszPathName;
730 if (newName.IsEmpty())
731 {
732 CDocTemplate* pTemplate = GetDocTemplate();
733 ASSERT(pTemplate != NULL);
734
735 newName = m_strPathName;
736 if (bReplace && newName.IsEmpty())
737 {
738 newName = m_strTitle;
739 // check for dubious filename
740 int iBad = newName.FindOneOf(_T(" #%;/\\"));
741 if (iBad != -1)
742 newName.ReleaseBuffer(iBad);
743
744 // append the default suffix if there is one
745 CString strExt;
746 if (pTemplate->GetDocString(strExt, CDocTemplate::filterExt) &&
747 !strExt.IsEmpty())
748 {
749 ASSERT(strExt[0] == '.');
750 newName += strExt;
751 }
752 }
753
754 TSaveMapInfo info;
755 info.filename = newName;
756 SaveMap saveDlg(&info);
757 if (saveDlg.DoModal() == IDCANCEL) {
758 return FALSE;
759 }
760 if (info.browse) {
761 if (!AfxGetApp()->DoPromptFileName(newName,
762 bReplace ? AFX_IDS_SAVEFILE : AFX_IDS_SAVEFILECOPY,
763 OFN_HIDEREADONLY | OFN_PATHMUSTEXIST, FALSE, pTemplate))
764 return FALSE; // don't even attempt to save
765 } else {
766 // Construct file name of .\Maps\mapname\mapname.map
767 if (info.usingSystemDir)
768 newName = ".\\Maps\\";
769 else
770 {
771 newName = TheGlobalData->getPath_UserData().str();
772 newName = newName + "Maps\\";
773 }
774 newName += info.filename;
775 // Create directory.
776 CFileStatus status;
777 if (CFile::GetStatus(newName, status)) {
778 if (!(status.m_attribute&CFile::directory)) {
779 CString error = "Error: file '" + newName + "' exists, and is not a directory.";
780 ::AfxMessageBox(error);
781 return FALSE;
782 }
783 } else {
784 Int status = ::_mkdir(newName);
785 if (status != 0) {
786 CString error = "Error: could not create directory '" + newName + "'.";
787 ::AfxMessageBox(error);
788 return FALSE;
789 }
790 }
791 newName += "\\";
792 newName += info.filename;
793 newName += ".map";
794 }
795 }
796
797 CWaitCursor wait;
798
799 if (!OnSaveDocument(newName))
800 {
801 if (lpszPathName == NULL)
802 {
803 // be sure to delete the file
804 try
805 {
806 CFile::Remove(newName);
807 }
808 catch(...)
809 {
810 TRACE0("Warning: failed to delete file after failed SaveAs.\n");
811 }
812 }
813 return FALSE;
814 }
815
816 // reset the title and change the document name
817 if (bReplace)
818 SetPathName(newName);
819
820 return TRUE; // success
821}

1479 inMap = false;
1480 yIndex = pMap->getYExtent()-1;

1509 const Coord3D* tl, const Coord3D* tr,
1510 Int widthOutside, VecHeightMapIndexes* allIndices)
1511{
1512 // given the four corners of this rectangle, find all indices that are within
1513 // widthOutside of the rect and place them into allIndices.

1515 return false;
1516 }

1519
1520 allIndices->clear();
1521
1522 CPoint ndx;
1523
1524 FindIndexNearest(this, &center, &ndx, PREFER_CENTER);
1525 AddUniqueAndNeighbors(this, bl, br, tl, tr, ndx, allIndices);
1526
1527 FindIndexNearest(this, &center, &ndx, PREFER_LEFT);
1528 AddUniqueAndNeighbors(this, bl, br, tl, tr, ndx, allIndices);
1529
1530 FindIndexNearest(this, &center, &ndx, PREFER_TOP);
1531 AddUniqueAndNeighbors(this, bl, br, tl, tr, ndx, allIndices);
1532
1533 FindIndexNearest(this, &center, &ndx, PREFER_RIGHT);
1534 AddUniqueAndNeighbors(this, bl, br, tl, tr, ndx, allIndices);
1535
1536 FindIndexNearest(this, &center, &ndx, PREFER_BOTTOM);
1537 AddUniqueAndNeighbors(this, bl, br, tl, tr, ndx, allIndices);

1838void CWorldBuilderDoc::updateLWL(MapObject *pWay, MapObject *pSrcWay)

1841
1842 Bool exists;
1843 AsciiString label;
1844 label = pSrcWay->getProperties()->getAsciiString(TheKey_waypointPathLabel1, &exists);
1845 if (exists) {
1846 pWay->getProperties()->setAsciiString(TheKey_waypointPathLabel1, label);
1847 } else if (pWay->getProperties()->known(TheKey_waypointPathLabel1, Dict::DICT_ASCIISTRING)) {
1848 pWay->getProperties()->remove(TheKey_waypointPathLabel1);
1849 }
1850 label = pSrcWay->getProperties()->getAsciiString(TheKey_waypointPathLabel2, &exists);
1851 if (exists) {
1852 pWay->getProperties()->setAsciiString(TheKey_waypointPathLabel2, label);
1853 } else if (pWay->getProperties()->known(TheKey_waypointPathLabel2, Dict::DICT_ASCIISTRING)){
1854 pWay->getProperties()->remove(TheKey_waypointPathLabel2);
1855 }
1856 label = pSrcWay->getProperties()->getAsciiString(TheKey_waypointPathLabel3, &exists);
1857 if (exists) {
1858 pWay->getProperties()->setAsciiString(TheKey_waypointPathLabel3, label);
1859 } else if (pWay->getProperties()->known(TheKey_waypointPathLabel3, Dict::DICT_ASCIISTRING)) {
1860 pWay->getProperties()->remove(TheKey_waypointPathLabel3);
1861 }
1862
1863 Bool biDirectional;
1864 biDirectional = pSrcWay->getProperties()->getBool(TheKey_waypointPathBiDirectional, &exists);
1865 if (exists) {
1866 pWay->getProperties()->setBool(TheKey_waypointPathBiDirectional, biDirectional);
1867 }
1868
1869 MapObject *pCurWay = pWay;
1870 pWay = NULL;
1871 Int i;
1872
1873 for (i=0; i<m_numWaypointLinks; i++) {
1874 if (m_waypointLinks[i].processedFlag) continue;
1875 Bool process = false;
1876 MapObject *pNewWay = NULL;

1878 Int waypointID2 = m_waypointLinks[i].waypoint2;

1880 DEBUG_ASSERTCRASH(waypointID2>=0 && waypointID2<MAX_WAYPOINTS, ("Invalid id."));

1891 }
1892 }
1893 if (process) {
1894 m_waypointLinks[i].processedFlag = true;
1895 if (pWay == NULL) {
1896 pWay = pNewWay;

1898 updateLWL(pNewWay, pSrcWay);

1901 }
