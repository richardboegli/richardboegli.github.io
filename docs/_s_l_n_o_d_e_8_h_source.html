54//
55// The node class is responsible for maintaining the links between
56// data in a linked list. It works with the Single List Class to
57// manage a singularly linked list of objects.
58//
59
60class GenericSLNode : public AutoPoolClass<GenericSLNode, 256>
61{
62 protected:
63 void* Internal_Get_Next(void) { return NodeNext; };
64 void Internal_Set_Next(void* n) { NodeNext=n; };
65 void *Internal_Get_Data(void) { return NodeData; };
66 void Internal_Get_Data(void* d) { NodeData=d; };
67
68 //
69 // Note that their is only one constructor for this class and it
70 // requires you to provide an object. Furthermore it can be
71 // created from anything but a friend or parent class.
72 //
73 GenericSLNode(void *obj)
74 {NodeData = obj; NodeNext = 0; };
75
76 //
77 // You cannot declare a node class without giving it a data object.
78 // Defining this type of constructor as private to the class insures
79 // that it cannot be used.
80 //
81 private:
82 GenericSLNode(void) {};
83
84 void *NodeNext; // Next Node in the list chain
85 void *NodeData; // Current Node in the list chain
86};

89template <class T>
90class SLNode : public GenericSLNode
91{
92 public:
93 //
94 // Since the list class manages the Node Class it must be able to
95 // access its private data.
96 //
97 friend class SList<T>;
98
99 SLNode<T>* Next(void) { return reinterpret_cast<SLNode<T>*>(Internal_Get_Next()); }
100 T *Data(void) { return reinterpret_cast<T*>(Internal_Get_Data()); }
101
102 void Set_Next(SLNode<T>* n) { Internal_Set_Next(reinterpret_cast<void*>(n)); }
103
104 protected:
105 //
