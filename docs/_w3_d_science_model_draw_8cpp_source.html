74// All this does is stop the call path if we haven't been cleared to draw yet
75void W3DScienceModelDraw::doDrawModule(const Matrix3D* transformMtx)
76{
77 ScienceType science = getW3DScienceModelDrawModuleData()->m_requiredScience;
78 if( science == SCIENCE_INVALID )
79 {
80 DEBUG_ASSERTCRASH(science != SCIENCE_INVALID, ("ScienceModelDraw has invalid science as condition.") );
81 setHidden( TRUE );
82 return;
83 }
84
85 if( !ThePlayerList->getLocalPlayer()->hasScience(science)
86 && ThePlayerList->getLocalPlayer()->isPlayerActive()
87 )
88 {
89 // We just don't draw for people without our science except for Observers
90 setHidden( TRUE );
91 return;
92 }
93
94 W3DModelDraw::doDrawModule(transformMtx);
95}
