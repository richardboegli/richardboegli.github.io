495void MilesAudioManager::stopAudio( AudioAffect which )
496{
497 // All we really need to do is:
498 // 1) Remove the EOS callback.
499 // 2) Stop the sample, (so that when we later unload it, bad stuff doesn't happen)
500 // 3) Set the status to stopped, so that when we next process the playing list, we will
501 // correctly clean up the sample.
502
503
504 std::list<PlayingAudio *>::iterator it;
505
506 PlayingAudio *playing = NULL;
507 if (BitTest(which, AudioAffect_Sound)) {
508 for (it = m_playingSounds.begin(); it != m_playingSounds.end(); ++it) {
509 playing = *it;
510 if (playing) {
511 AIL_register_EOS_callback(playing->m_sample, NULL);
512 AIL_stop_sample(playing->m_sample);
513 playing->m_status = PS_Stopped;
514 }
515 }
516 }
517
518 if (BitTest(which, AudioAffect_Sound3D)) {
519 for (it = m_playing3DSounds.begin(); it != m_playing3DSounds.end(); ++it) {
520 playing = *it;
521 if (playing) {
522 AIL_register_3D_EOS_callback(playing->m_3DSample, NULL);
523 AIL_stop_3D_sample(playing->m_3DSample);
524 playing->m_status = PS_Stopped;
525 }
526 }
527 }
528
529 if (BitTest(which, AudioAffect_Speech | AudioAffect_Music)) {
530 for (it = m_playingStreams.begin(); it != m_playingStreams.end(); ++it) {
531 playing = *it;
532 if (playing) {
533 if (playing->m_audioEventRTS->getAudioEventInfo()->m_soundType == AT_Music) {
534 if (!BitTest(which, AudioAffect_Music)) {
535 continue;
536 }
537 } else {
538 if (!BitTest(which, AudioAffect_Speech)) {
539 continue;
540 }
541 }
542 AIL_register_stream_callback(playing->m_stream, NULL);
543 AIL_pause_stream(playing->m_stream, 1);
544 playing->m_status = PS_Stopped;
545 }
546 }
547 }
548}

567 playing = *it;
568 if (playing) {
569 AIL_stop_3D_sample(playing->m_3DSample);
570 }
571 }
572 }
573
574 if (BitTest(which, AudioAffect_Speech | AudioAffect_Music)) {
575 for (it = m_playingStreams.begin(); it != m_playingStreams.end(); ++it) {
576 playing = *it;
577 if (playing) {
578 if (playing->m_audioEventRTS->getAudioEventInfo()->m_soundType == AT_Music) {
579 if (!BitTest(which, AudioAffect_Music)) {
580 continue;
581 }
582 } else {
583 if (!BitTest(which, AudioAffect_Speech)) {
584 continue;
585 }
586 }
587
588 AIL_pause_stream(playing->m_stream, 1);
589 }
590 }
591 }
592
593 //Get rid of PLAY audio requests when pausing audio.
594 std::list<AudioRequest*>::iterator ait;
595 for (ait = m_audioRequests.begin(); ait != m_audioRequests.end(); /* empty */)
596 {
597 AudioRequest *req = (*ait);
598 if( req && req->m_request == AR_Play )
599 {
600 req->deleteInstance();
601 ait = m_audioRequests.erase(ait);
602 }
603 else
604 {
605 ait++;
606 }
607 }
608}

661void MilesAudioManager::playAudioEvent( AudioEventRTS *event )
662{
663#ifdef INTENSIVE_AUDIO_DEBUG
664 DEBUG_LOG(("MILES (%d) - Processing play request: %d (%s)", TheGameLogic->getFrame(), event->getPlayingHandle(), event->getEventName().str()));
665#endif
666 const AudioEventInfo *info = event->getAudioEventInfo();
667 if (!info) {
668 return;
669 }
670
671 std::list<PlayingAudio *>::iterator it;
672 PlayingAudio *playing = NULL;
673
674 AudioHandle handleToKill = event->getHandleToKill();
675
676 AsciiString fileToPlay = event->getFilename();
677 PlayingAudio *audio = allocatePlayingAudio();
678 switch(info->m_soundType)
679 {
680 case AT_Music:
681 case AT_Streaming:
682 {
683 #ifdef INTENSIVE_AUDIO_DEBUG
684 DEBUG_LOG(("- Stream\n"));
685 #endif
686
687 if ((info->m_soundType == AT_Streaming) && event->getUninterruptable()) {
688 stopAllSpeech();
689 }
690
691 Real curVolume = 1.0;
692 if (info->m_soundType == AT_Music) {
693 curVolume = m_musicVolume;
694 } else {
695 curVolume = m_speechVolume;
696 }
697 curVolume *= event->getVolume();
698
699 Bool foundSoundToReplace = false;
700 if (handleToKill) {
701 for (it = m_playingStreams.begin(); it != m_playingStreams.end(); ++it) {
702 playing = (*it);
703 if (!playing) {
704 continue;
705 }
706
707 if (playing->m_audioEventRTS && playing->m_audioEventRTS->getPlayingHandle() == handleToKill)
708 {
709 //Release this streaming channel immediately because we are going to play another sound in it's place.
710 releasePlayingAudio(playing);
711 m_playingStreams.erase(it);
712 foundSoundToReplace = true;
713 break;
714 }
715 }
716 }
717
718 HSTREAM stream;
719 if (!handleToKill || foundSoundToReplace) {
720 stream = AIL_open_stream(m_digitalHandle, fileToPlay.str(), 0);
721 } else {
722 stream = NULL;
723 }
724
725 // Put this on here, so that the audio event RTS will be cleaned up regardless.
726 audio->m_audioEventRTS = event;
727 audio->m_stream = stream;
728 audio->m_type = PAT_Stream;
729
730 if (stream) {
731 if ((info->m_soundType == AT_Streaming) && event->getUninterruptable()) {
732 setDisallowSpeech(TRUE);
733 }
734 AIL_set_stream_volume_pan(stream, curVolume, 0.5f);
735 playStream(event, stream);
736 m_playingStreams.push_back(audio);
737 audio = NULL;
738 }
739 break;
740 }
741
742 case AT_SoundEffect:
743 {
744 #ifdef INTENSIVE_AUDIO_DEBUG
745 DEBUG_LOG(("- Sound"));
746 #endif
747
748
749 if (event->isPositionalAudio()) {
750 // Sounds that are non-global are positional 3-D sounds. Deal with them accordingly
751 #ifdef INTENSIVE_AUDIO_DEBUG
752 DEBUG_LOG((" Positional"));
753 #endif
754 Bool foundSoundToReplace = false;
755 if (handleToKill)
756 {
757 for (it = m_playing3DSounds.begin(); it != m_playing3DSounds.end(); ++it) {
758 playing = (*it);
759 if (!playing) {
760 continue;
761 }
762
763 if( playing->m_audioEventRTS && playing->m_audioEventRTS->getPlayingHandle() == handleToKill )
764 {
765 //Release this 3D sound channel immediately because we are going to play another sound in it's place.
766 releasePlayingAudio(playing);
767 m_playing3DSounds.erase(it);
768 foundSoundToReplace = true;
769 break;
770 }
771 }
772 }
773
774 H3DSAMPLE sample3D;
775 if( !handleToKill || foundSoundToReplace )
776 {
777 sample3D = getFirst3DSample( event );
778 if( !sample3D )
779 {
780 //If we don't have an available sample, kill the lowest priority assuming we have one that is lower
781 //than the sound we are trying to add. One possibility for strangeness is when an interrupt sound
782 //that wants to kill a handle to replace it, it's possible that another request already killed it,
783 //in which case we need to attempt to find another sound to kill.
784 if( killLowestPrioritySoundImmediately( event ) )
785 {
786 sample3D = getFirst3DSample( event );
787 }
788 }
789 }
790 else
791 {
792 sample3D = NULL;
793 }
794 // Push it onto the list of playing things
795 audio->m_audioEventRTS = event;
796 audio->m_3DSample = sample3D;
797 audio->m_file = NULL;
798 audio->m_type = PAT_3DSample;
799 m_playing3DSounds.push_back(audio);
800
801 if (sample3D) {
802 audio->m_file = playSample3D(event, sample3D);
803 m_sound->notifyOf3DSampleStart();
804 }
805
806 if( !audio->m_file )
807 {
808 m_playing3DSounds.pop_back();
809 #ifdef INTENSIVE_AUDIO_DEBUG
810 DEBUG_LOG((" Killed (no handles available)\n"));
811 #endif
812 }
813 else
814 {
815 audio = NULL;
816 #ifdef INTENSIVE_AUDIO_DEBUG
817 DEBUG_LOG((" Playing.\n"));
818 #endif
819 }
820 }
821 else
822 {
823 // UI sounds are always 2-D. All other sounds should be Positional
824 // Unit acknowledgement, etc, falls into the UI category of sound.
825 Bool foundSoundToReplace = false;
826 if (handleToKill) {
827 for (it = m_playingSounds.begin(); it != m_playingSounds.end(); ++it) {
828 playing = (*it);
829 if (!playing) {
830 continue;
831 }
832
833 if (playing->m_audioEventRTS && playing->m_audioEventRTS->getPlayingHandle() == handleToKill)
834 {
835 //Release this 2D sound channel immediately because we are going to play another sound in it's place.
836 releasePlayingAudio(playing);
837 m_playingSounds.erase(it);
838 foundSoundToReplace = true;
839 break;
840 }
841 }
842 }
843
844 HSAMPLE sample;
845 if( !handleToKill || foundSoundToReplace )
846 {
847 sample = getFirst2DSample(event);
848 if( !sample )
849 {
850 //If we don't have an available sample, kill the lowest priority assuming we have one that is lower
851 //than the sound we are trying to add. One possibility for strangeness is when an interrupt sound
852 //that wants to kill a handle to replace it, it's possible that another request already killed it,
853 //in which case we need to attempt to find another sound to kill.
854 if( killLowestPrioritySoundImmediately( event ) )
855 {
856 sample = getFirst2DSample( event );
857 }
858 }
859 }
860 else
861 {
862 sample = NULL;
863 }
864
865 // Push it onto the list of playing things
866 audio->m_audioEventRTS = event;
867 audio->m_sample = sample;
868 audio->m_file = NULL;
869 audio->m_type = PAT_Sample;
870 m_playingSounds.push_back(audio);
871
872 if (sample) {
873 audio->m_file = playSample(event, sample);
874 m_sound->notifyOf2DSampleStart();
875 }
876
877 if (!audio->m_file) {
878 #ifdef INTENSIVE_AUDIO_DEBUG
879 DEBUG_LOG((" Killed (no handles available)\n"));
880 #endif
881 m_playingSounds.pop_back();
882 } else {
883 audio = NULL;
884 }
885
886 #ifdef INTENSIVE_AUDIO_DEBUG
887 DEBUG_LOG((" Playing.\n"));
888 #endif
889 }
890 break;
891 }
892 }
893
894 // If we were able to successfully play audio, then we set it to NULL above. (And it will be freed
895 // later. However, if audio is non-NULL at this point, then it must be freed.
896 if (audio) {
897 releasePlayingAudio(audio);
898 }
899}

1533 PlayingAudio *playing = findPlayingAudioFrom(audioCompleted, flags);
1534 if (!playing) {
1535 DEBUG_CRASH(("Audio has completed playing, but we can't seem to find it. - jkmcd"));
1536 return;
1537 }
1538
1539 if (getDisallowSpeech() && playing->m_audioEventRTS->getAudioEventInfo()->m_soundType == AT_Streaming) {
1540 setDisallowSpeech(FALSE);
1541 }
1542
1543 if (playing->m_audioEventRTS->getAudioEventInfo()->m_control & AC_LOOP) {

1545 playing->m_audioEventRTS->setNextPlayPortion(PP_Sound);

1547 if (playing->m_audioEventRTS->getNextPlayPortion() == PP_Sound) {
1548 // First, decrease the loop count.

1550
1551 // Now, try to start the next loop
1552 if (startNextLoop(playing)) {
1553 return;
1554 }
1555 }
1556 }
1557
1558 playing->m_audioEventRTS->advanceNextPlayPortion();
1559 if (playing->m_audioEventRTS->getNextPlayPortion() != PP_Done) {
1560 if (playing->m_type == PAT_Sample) {
1561 closeFile(playing->m_file); // close it so as not to leak it.
1562 playing->m_file = playSample(playing->m_audioEventRTS, playing->m_sample);
1563
1564 // If we don't have a file now, then we should drop to the stopped status so that
1565 // We correctly close this handle.
1566 if (playing->m_file) {
1567 return;
1568 }
1569 } else if (playing->m_type == PAT_3DSample) {
1570 closeFile(playing->m_file); // close it so as not to leak it.
1571 playing->m_file = playSample3D(playing->m_audioEventRTS, playing->m_3DSample);
1572
1573 // If we don't have a file now, then we should drop to the stopped status so that
1574 // We correctly close this handle.
1575 if (playing->m_file) {
1576 return;
1577 }
1578 }
1579 }
1580
1581 if (playing->m_type == PAT_Stream) {
1582 if (playing->m_audioEventRTS->getAudioEventInfo()->m_soundType == AT_Music) {
1583 playStream(playing->m_audioEventRTS, playing->m_stream);
1584
1585 return;
1586 }
1587 }

1589 playing->m_status = PS_Stopped; // it will be cleaned up on the next frame update

1826Bool MilesAudioManager::doesViolateLimit( AudioEventRTS *event ) const
1827{
1828 Int limit = event->getAudioEventInfo()->m_limit;
1829 if (limit == 0) {
1830 return false;
1831 }
1832
1833 Int totalCount = 0;
1834 Int totalRequestCount = 0;
1835
1836 std::list<PlayingAudio *>::const_iterator it;
1837 if (!event->isPositionalAudio()) {
1838 // 2-D
1839 for ( it = m_playingSounds.begin(); it != m_playingSounds.end(); ++it ) {
1840 if ((*it)->m_audioEventRTS->getEventName() == event->getEventName()) {
1841 if (totalCount == 0) {
1842 // This is the oldest audio of this type playing.
1843 event->setHandleToKill((*it)->m_audioEventRTS->getPlayingHandle());
1844 }
1845 ++totalCount;
1846 }
1847 }
1848 } else {
1849 // 3-D
1850 for ( it = m_playing3DSounds.begin(); it != m_playing3DSounds.end(); ++it ) {
1851 if ((*it)->m_audioEventRTS->getEventName() == event->getEventName()) {
1852 if (totalCount == 0) {
1853 // This is the oldest audio of this type playing.
1854 event->setHandleToKill((*it)->m_audioEventRTS->getPlayingHandle());
1855 }
1856 ++totalCount;
1857 }
1858 }
1859 }
1860
1861 // Also check the request list in case we've requested to play this sound.
1862 std::list<AudioRequest*>::const_iterator arIt;
1863 for (arIt = m_audioRequests.begin(); arIt != m_audioRequests.end(); ++arIt) {
1864 AudioRequest *req = (*arIt);
1865 if (req == NULL) {
1866 continue;
1867 }
1868 if( req->m_usePendingEvent )
1869 {
1870 if( req->m_pendingEvent->getEventName() == event->getEventName() )
1871 {
1872 totalRequestCount++;
1873 totalCount++;
1874 }
1875 }
1876 }
1877
1878 //If our event is an interrupting type, then normally we would always add it. The exception is when we have requested
1879 //multiple sounds in the same frame and those requests violate the limit. Because we don't have any "old" sounds to
1880 //remove in the case of an interrupt, we need to catch it early and prevent the sound from being added if we already
1881 //reached the limit
1882 if( event->getAudioEventInfo()->m_control & AC_INTERRUPT )
1883 {
1884 if( totalRequestCount < limit )
1885 {
1886 Int totalPlayingCount = totalCount - totalRequestCount;
1887 if( totalRequestCount + totalPlayingCount < limit )
1888 {
1889 //We aren't exceeding the actual limit, then clear the kill handle.
1890 event->setHandleToKill(0);
1891 return false;
1892 }
1893
1894 //We are exceeding the limit - the kill handle will kill the
1895 //oldest playing sound to enforce the actual limit.
1896 return false;
1897 }
1898 }
1899
1900 if( totalCount < limit )
1901 {
1902 event->setHandleToKill(0);
1903 return false;
1904 }
1905
1906 return true;
1907}
