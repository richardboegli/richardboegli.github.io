102void FlammableUpdate::onDamage( DamageInfo *damageInfo )
103{
104 if( damageInfo->in.m_damageType == DAMAGE_FLAME || damageInfo->in.m_damageType == DAMAGE_PARTICLE_BEAM )
105 {
106 UnsignedInt now = TheGameLogic->getFrame();
107 if( now - getFlammableUpdateModuleData()->m_flameDamageExpirationDelay > m_lastFlameDamageDealt )
108 {
109 // If it has been a long time since our last flame damage, reset the threshold
110 m_flameDamageLimit = getFlammableUpdateModuleData()->m_flameDamageLimitData;
111 }
112 m_lastFlameDamageDealt = now;
113
114 Object *me = getObject();
115 if( !me->getStatusBits().test( OBJECT_STATUS_AFLAME ) && !me->getStatusBits().test( OBJECT_STATUS_BURNED ) )
116 {
117 // If I'm not on fire, and I haven't burned up, see if I should try to catch fire.
118 m_flameDamageLimit -= damageInfo->out.m_actualDamageDealt;
119 if( m_flameDamageLimit <= 0 )
120 {
121 tryToIgnite();
122 }
123 }
124 }
125}

129UpdateSleepTime FlammableUpdate::update( void )
130{
131 Object *me = getObject();
132 DEBUG_ASSERTCRASH(m_status == FS_AFLAME, ("hmm, should be aflame"));
133
134 UnsignedInt now = TheGameLogic->getFrame();
135 const FlammableUpdateModuleData *data = getFlammableUpdateModuleData();
136
137 if( m_damageEndFrame != 0 && now >= m_damageEndFrame )
138 {
139 m_damageEndFrame = now + data->m_aflameDamageDelay;
140 doAflameDamage();
141 }
142
143 if( m_burnedEndFrame != 0 && now >= m_burnedEndFrame )
144 {
145 // So this status is set, but I am still aflame on an independent timer.
146 me->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_BURNED ) );
147 me->setModelConditionState( MODELCONDITION_SMOLDERING );
148 }
149
150 if( m_aflameEndFrame != 0 && now >= m_aflameEndFrame )
151 {
152 // This is the important one. I am no longer on fire.
153 if( me->getStatusBits().test( OBJECT_STATUS_BURNED ) )
154 {
155 // If I am burned, then I will never catch fire again.
156 m_status = FS_BURNED;
157 }
158 else
159 {
160 // otherwise I am free to burn again
161 m_status = FS_NORMAL;
162 }
163 stopBurningSound();
164 me->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_AFLAME ) );
165 me->getBodyModule()->setAflame( FALSE );
166 me->clearModelConditionState( MODELCONDITION_AFLAME );
167 }
168
169 return calcSleepTime();
170}

174UpdateSleepTime FlammableUpdate::calcSleepTime()
175{
176 UnsignedInt now = TheGameLogic->getFrame();
177 if (m_status == FS_AFLAME && m_aflameEndFrame != 0 && m_aflameEndFrame > now)
178 {
179 UnsignedInt soonest = m_aflameEndFrame;
180 if (m_burnedEndFrame != 0 && m_burnedEndFrame < soonest && m_burnedEndFrame > now) soonest = m_burnedEndFrame;
181 if (m_damageEndFrame != 0 && m_damageEndFrame < soonest && m_damageEndFrame > now) soonest = m_damageEndFrame;
182 DEBUG_ASSERTCRASH(soonest - now > 0, ("hmm"));
183 // UPDATE_SLEEP requires a count-of-frames, not an absolute-frame, so subtract 'now'
184 return UPDATE_SLEEP(soonest - now);
185 }
186 else
187 {
188 return UPDATE_SLEEP_FOREVER;
189 }
190}

194void FlammableUpdate::tryToIgnite()
195{
196 if( m_status == FS_NORMAL )
197 {
198 Object *me = getObject();
199 me->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_AFLAME ) );
200 me->getBodyModule()->setAflame( TRUE );
201 me->setModelConditionState( MODELCONDITION_AFLAME );
202 startBurningSound();
203
204 // bleah. this sucks. (srj)
205 static const NameKeyType key_FireSpreadUpdate = NAMEKEY("FireSpreadUpdate");
206 FireSpreadUpdate* fu = (FireSpreadUpdate*)getObject()->findUpdateModule(key_FireSpreadUpdate);
207 if (fu != NULL)
208 {
209 fu->startFireSpreading();
210 }
211
212 m_status = FS_AFLAME;
213
214 const FlammableUpdateModuleData *data = getFlammableUpdateModuleData();
215 UnsignedInt now = TheGameLogic->getFrame();
216 m_aflameEndFrame = now + data->m_aflameDuration;
217 m_burnedEndFrame = data->m_burnedDelay ? now + data->m_burnedDelay : 0;
218 m_damageEndFrame = data->m_aflameDamageDelay ? now + data->m_aflameDamageDelay : 0;
219
220 setWakeFrame(getObject(), calcSleepTime());
221 }
222}

226void FlammableUpdate::doAflameDamage()
227{
228 const FlammableUpdateModuleData *data = getFlammableUpdateModuleData();
229
230 DamageInfo info;
231 info.in.m_amount = data->m_aflameDamageAmount;
232 info.in.m_sourceID = getObject()->getID();
233 info.in.m_damageType = DAMAGE_FLAME;
234 info.in.m_deathType = DEATH_BURNED;
235
236 getObject()->attemptDamage( &info );
237}
