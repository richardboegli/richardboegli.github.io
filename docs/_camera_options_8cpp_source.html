82//WST 10/7/2002 - Drop waypoint button for Adam Isgreen -----------
83void CameraOptions::OnDropWaypointButton()
84{
85 //The following code is taken from waypointTool.cpp. On mouse down.
86 WbView3d * p3View = CWorldBuilderDoc::GetActive3DView();
87 Coord3D docPt;
88 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
89
90 Vector3 camTarget = p3View->getCameraTarget();
91 docPt.x = camTarget.X;
92 docPt.y = camTarget.Y;
93 docPt.z = MAGIC_GROUND_Z;
94
95 //
96 // MBL CNC3 INCURSION 10.29.2002 - Fix compile error w/ 10-15-2002 Drop
97 //
98 // MapObject *pNew = new MapObject(docPt, AsciiString("*Waypoints/Waypoint"), 0, 0, NULL, NULL );
99 MapObject *pNew = newInstance(MapObject)(docPt, AsciiString("*Waypoints/Waypoint"), 0, 0, NULL, NULL );
100
101 Int id = pDoc->getNextWaypointID();
102 AsciiString name = WaypointOptions::GenerateUniqueName(id);
103 pNew->setSelected(true);
104 pNew->setIsWaypoint();
105 pNew->setWaypointID(id);
106 pNew->setWaypointName(name);
107 pNew->getProperties()->setAsciiString(TheKey_originalOwner, AsciiString("team"));
108 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew);
109 pDoc->AddAndDoUndoable(pUndo);
110 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
111 pNew = NULL; // undoable owns it now.
112}

114//WST 11/25/2002 - New Center Camera button for Designers -----------
115void CameraOptions::OnCenterOnSelectedButton()
116{
117 // Center camera on the selected map object
118
119 int count = 0;
120 const Coord3D *objectPosition;
121
122 MapObject *mapObject = MapObject::getFirstMapObject();
123 while (mapObject) {
124 if (mapObject->isSelected()) {
125 objectPosition = mapObject->getLocation();
126 count++;
127 }
128 mapObject = mapObject->getNext();
129 }
130
131 if (count==1) {
132 // Only center if there is only one object selected on map
133 WbView3d * p3View = CWorldBuilderDoc::GetActive3DView();
134 if (p3View) {
135 p3View->setCenterInView(objectPosition->x/MAP_XY_FACTOR,objectPosition->y/MAP_XY_FACTOR);
136 }
137 }
138}

199void CameraOptions::stuffValuesIntoFields( void )
200{
201 WbView3d * p3View = CWorldBuilderDoc::GetActive3DView();
202 if (p3View)
203 {
204 m_updating = true;
205 putReal(IDC_PITCH_EDIT, p3View->getCameraPitch());
206 m_updating = false;
207
208 Real height = p3View->getHeightAboveGround();
209 //WST 10/11/2002 This is inaccurate Real zoom = height/TheGlobalData->m_maxCameraHeight;
210 Real zoom = p3View->getCurrentZoom(); //WST 10.11.2002
211
212 putReal(IDC_ZOOMTEXT, zoom);
213 putReal(IDC_HEIGHTTEXT, height);
214
215 Vector3 source = p3View->getCameraSource();
216 Vector3 target = p3View->getCameraTarget();
217// Real angle = p3View->getCameraAngle();
218
219 AsciiString s;
220 s.format("(%g, %g)", target.X, target.Y);
221 putAsciiString(IDC_POSTEXT, s);
222 s.format("(%g, %g)", target.X - 1.0f*(source.X-target.X), target.Y - 1.0f*(source.Y-target.Y));
223 putAsciiString(IDC_TARGETTEXT, s);
224 }
225}

243void CameraOptions::GetPopSliderInfo(const long sliderID, long *pMin, long *pMax, long *pLineSize, long *pInitial)
244{
245 switch (sliderID) {
246
247 case IDC_PITCH_POPUP:
248 *pMin = 0;
249 *pMax = 100;
250 *pInitial = 100;
251 *pLineSize = 1;
252 {
253 WbView3d * p3View = CWorldBuilderDoc::GetActive3DView();

258 }
259 break;
260
261 default:
262 // uh-oh!
263 DEBUG_CRASH(("Slider message from unknown control"));
264 break;
265 } // switch
