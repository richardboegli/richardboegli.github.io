52void Dict::DictPair::copyFrom(DictPair* that)
53{
54 Dict::DataType curType = this->getType();
55 Dict::DataType newType = that->getType();
56 if (curType != newType)
57 {
58 clear();
59 }
60
61 switch(newType)
62 {
63 case DICT_BOOL:
64 case DICT_INT:
65 case DICT_REAL:
66 *this = *that;
67 break;
68 case DICT_ASCIISTRING:
69 this->m_key = that->m_key;
70 *this->asAsciiString() = *that->asAsciiString();
71 break;
72 case DICT_UNICODESTRING:
73 this->m_key = that->m_key;
74 *this->asUnicodeString() = *that->asUnicodeString();
75 break;
76 }
77}

124Dict::DictPair* Dict::findPairByKey(NameKeyType key) const
125{
126 DEBUG_ASSERTCRASH(key != NAMEKEY_INVALID, ("invalid namekey!"));
127 DEBUG_ASSERTCRASH((UnsignedInt)key < (1L<<23), ("namekey too large!"));
128 if (!m_data)
129 return NULL;
130 DictPair* base = m_data->peek();
131 Int minIdx = 0;
132 Int maxIdx = m_data->m_numPairsUsed;
133 while (minIdx < maxIdx)
134 {
135 Int midIdx = (((minIdx + maxIdx) - 1) >> 1);
136 DictPair* mid = base + midIdx;
137 NameKeyType midKey = mid->getName();
138 if (key > midKey)
139 minIdx = midIdx + 1;
140 else if (key < midKey)
141 maxIdx = midIdx;
142 else
143 return mid;
144 }
145
146 return NULL;
147}

150Dict::DictPair *Dict::ensureUnique(int numPairsNeeded, Bool preserveData, DictPair *pairToTranslate)
151{
152 if (m_data &&
153 m_data->m_refCount == 1 &&
154 m_data->m_numPairsAllocated >= numPairsNeeded)
155 {
156 // no buffer manhandling is needed (it's already large enough, and unique to us)
157 return pairToTranslate;
158 }
159
160 if (numPairsNeeded > MAX_LEN)
161 throw ERROR_OUT_OF_MEMORY;
162
163 Dict::DictPairData* newData = NULL;
164 if (numPairsNeeded > 0)
165 {
166 int minBytes = sizeof(Dict::DictPairData) + numPairsNeeded*sizeof(Dict::DictPair);
167 int actualBytes = TheDynamicMemoryAllocator->getActualAllocationSize(minBytes);
168 // note: be certain to alloc with zero; we'll take advantage of the fact that all-zero
169 // is a bit-pattern that happens to init all our pairs to legal values:
170 // type BOOL, key INVALID, value FALSE.
171 newData = (Dict::DictPairData*)TheDynamicMemoryAllocator->allocateBytes(actualBytes, "Dict::ensureUnique");
172 newData->m_refCount = 1;
173 newData->m_numPairsAllocated = (actualBytes - sizeof(Dict::DictPairData))/sizeof(Dict::DictPair);
174 newData->m_numPairsUsed = 0;
175
176 if (preserveData && m_data)
177 {
178 Dict::DictPair* src = m_data->peek();
179 Dict::DictPair* dst = newData->peek();
180 for (Int i = 0; i < m_data->m_numPairsUsed; ++i, ++src, ++dst)
181 dst->copyFrom(src);
182 newData->m_numPairsUsed = m_data->m_numPairsUsed;
183 }
184 }
185
186 Int delta;
187 if (pairToTranslate && m_data)
188 delta = pairToTranslate - m_data->peek();
189
190 releaseData();
191 m_data = newData;
192
193 if (pairToTranslate && m_data)
194 pairToTranslate = m_data->peek() + delta;
195
196 return pairToTranslate;
197}

208void Dict::releaseData()
209{
210 if (m_data)
211 {
212 if (--m_data->m_refCount == 0)
213 {
214 Dict::DictPair* src = m_data->peek();
215 for (Int i = 0; i < m_data->m_numPairsUsed; ++i, ++src)
216 src->clear();
217 TheDynamicMemoryAllocator->freeBytes(m_data);
218 }
219 m_data = 0;
220 }
221}

225{
226

228 This class plays some skanky games, in the name of memory and code
229 efficiency; it assumes all the data types will fit into a pointer.
230 This is currently true, but if that assumption ever changes, all hell
231 will break loose. So we do a quick check to assure this...
232 */
233 DEBUG_ASSERTCRASH(sizeof(Bool) <= sizeof(void*) &&
234 sizeof(Int) <= sizeof(void*) &&
235 sizeof(Real) <= sizeof(void*) &&
236 sizeof(AsciiString) <= sizeof(void*) &&
237 sizeof(UnicodeString) <= sizeof(void*), ("oops, this code needs attention"));
238
239 if (numPairsToPreAllocate)
240 ensureUnique(numPairsToPreAllocate, false, NULL); // will throw on error

419Dict::DictPair *Dict::setPrep(NameKeyType key, Dict::DataType type)
420{
421 DictPair* pair = findPairByKey(key);
422 Int pairsNeeded = getPairCount();
423 if (!pair)
424 ++pairsNeeded;
425 pair = ensureUnique(pairsNeeded, true, pair);
426 if (!pair)
427 {
428 pair = &m_data->peek()[m_data->m_numPairsUsed++];
429 }
430 pair->setNameAndType(key, type);
431 DEBUG_ASSERTCRASH(pair, ("pair must not be null here"));
432 return pair;
433}

435// -----------------------------------------------------
436void Dict::sortPairs()
437{
438 if (!m_data)
439 return;
440
441 // yer basic shellsort.
442 for (Int gap = m_data->m_numPairsUsed >> 1; gap > 0; gap >>= 1)
443 {
444 for (Int i = gap; i < m_data->m_numPairsUsed; i++)
445 {
446 for (Int j = i - gap; j >= 0; j -= gap)
447 {
448 DictPair* a = m_data->peek() + j;
449 DictPair* b = m_data->peek() + j + gap;
450 if (a->getName() > b->getName())
451 {
452 DictPair tmp = *a;
453 *a = *b;
454 *b = tmp;
455 }
456 else
457 {
458 break;
459 }
460 }
461 }
462 }

517{
518 validate();
519 DictPair* pair = findPairByKey(key);
520 if (pair)
521 {
522 pair = ensureUnique(m_data->m_numPairsUsed, true, pair);
523 pair->setNameAndType((NameKeyType)0x7fffffff, DICT_BOOL);
524 sortPairs();
525 --m_data->m_numPairsUsed;
526 validate();
527 return true;
528 }
529 DEBUG_CRASH(("dict key missing in remove\n"));
530 return false;

535{
536 this->validate();
537 DictPair* thatPair = that.findPairByKey(key);
538 if (thatPair)
539 {
540 DictPair* thisPair = this->setPrep(key, thatPair->getType());
541 thisPair->copyFrom(thatPair);
542 this->sortPairs();
543 }
544 else

547 this->remove(key);
