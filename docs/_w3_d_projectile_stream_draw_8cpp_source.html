100void W3DProjectileStreamDraw::setFullyObscuredByShroud(Bool fullyObscured)
101{
102 if (fullyObscured)
103 { //we need to remove all our lines from the scene because they are hidden
104 for( Int lineIndex = 0; lineIndex < m_linesValid; lineIndex++ )
105 {
106 SegmentedLineClass *deadLine = m_allLines[lineIndex];
107 if (deadLine && deadLine->Peek_Scene())
108 deadLine->Remove();
109 }
110 }
111 else
112 { //we need to restore lines into scene
113 for( Int lineIndex = 0; lineIndex < m_linesValid; lineIndex++ )
114 {
115 SegmentedLineClass *deadLine = m_allLines[lineIndex];
116 if (deadLine && !deadLine->Peek_Scene())
117 W3DDisplay::m_3DScene->Add_Render_Object(deadLine);
118 }
119 }
120}

125void W3DProjectileStreamDraw::doDrawModule(const Matrix3D* )
126{
127 // get object from logic
128 Object *me = getDrawable()->getObject();
129 if (me == NULL)
130 return;
131
132 static NameKeyType key_ProjectileStreamUpdate = NAMEKEY("ProjectileStreamUpdate");
133 ProjectileStreamUpdate* update = (ProjectileStreamUpdate*)me->findUpdateModule(key_ProjectileStreamUpdate);
134
135 const W3DProjectileStreamDrawModuleData *data = getW3DProjectileStreamDrawModuleData();
136
137 Vector3 allPoints[MAX_PROJECTILE_STREAM];
138 Int pointsUsed;
139
140 update->getAllPoints( allPoints, &pointsUsed );
141
142 Vector3 stagingPoints[MAX_PROJECTILE_STREAM];
143 Vector3 zeroVector(0, 0, 0);
144
145 Int linesMade = 0;
146 Int currentMasterPoint = 0;
147 UnsignedInt currentStagingPoint = 0;
148
149 if( data->m_maxSegments )
150 {
151 // If I have a drawing cap, I need to increase the start point in the array. The furthest (oldest)
152 // point from the tank is in spot zero.
153 currentMasterPoint = pointsUsed - data->m_maxSegments;
154 currentMasterPoint = max( 0, currentMasterPoint ); // (but if they say to draw more than exists, draw all)
155 }
156
157 // Okay. I have an array of ordered points that may have blanks in it. I need to copy to the staging area
158 // until I hit a blank or the end. Then if I have a line made, I'll overwrite it, otherwise I'll make a new one.
159 // I'll keep doing this until I run out of valid points.
160 while( currentMasterPoint < pointsUsed )
161 {
162 while( currentMasterPoint < pointsUsed && allPoints[currentMasterPoint] != zeroVector )
163 {
164 // While I am not looking at a bad point (off edge or zero)
165 stagingPoints[currentStagingPoint] = allPoints[currentMasterPoint];// copy to the staging
166 currentStagingPoint++;// increment how many I have
167 currentMasterPoint++;// increment what I am looking at
168 }
169 // Use or reuse a line
170 if( currentStagingPoint > 1 )
171 {
172 // Don't waste a line on a double hole (0) or a one point line (1)
173 makeOrUpdateLine( stagingPoints, currentStagingPoint, linesMade );
174 linesMade++;// keep track of how many are real this frame
175 }
176 currentMasterPoint++;//I am either pointed off the edge anyway, or I am pointed at a zero I want to skip
177 currentStagingPoint = 0;//start over in the staging area
178 }
179
180 Int oldLinesValid = m_linesValid;
181 for( Int lineIndex = linesMade; lineIndex < oldLinesValid; lineIndex++ )
182 {
183 // Delete any line we aren't using anymore.
184 SegmentedLineClass *deadLine = m_allLines[lineIndex];
185 if (deadLine->Peek_Scene())
186 W3DDisplay::m_3DScene->Remove_Render_Object( deadLine );
187 REF_PTR_RELEASE( deadLine );
188
189 m_allLines[lineIndex] = NULL;
190 m_linesValid--;
191 }
192}

194void W3DProjectileStreamDraw::makeOrUpdateLine( Vector3 *points, UnsignedInt pointCount, Int lineIndex )
195{
196 Bool newLine = FALSE;
197
198 if( m_allLines[lineIndex] == NULL )
199 {
200 //Need a new one if this is blank, otherwise I'll reset the existing one
201 m_allLines[lineIndex] = NEW SegmentedLineClass;
202 m_linesValid++;
203 newLine = TRUE;
204 }
205
206 SegmentedLineClass *line = m_allLines[lineIndex];
207
208 line->Set_Points(pointCount, points); //tell the line which points to use
209
210 if( newLine )
211 {
212 // This is one time stuff we only need to do if this is a new and not a change
213 const W3DProjectileStreamDrawModuleData *data = getW3DProjectileStreamDrawModuleData();
214 line->Set_Texture(m_texture); //set the texture
215 line->Set_Shader(ShaderClass::_PresetAdditiveSpriteShader); //pick the alpha blending mode you want - see shader.h for others.
216 line->Set_Width(data->m_width); //set line width in world units
217 line->Set_Texture_Mapping_Mode(SegLineRendererClass::TILED_TEXTURE_MAP); //this tiles the texture across the line
218 line->Set_Texture_Tile_Factor(data->m_tileFactor); //number of times to tile texture across each segment
219 line->Set_UV_Offset_Rate(Vector2(0.0f,data->m_scrollRate)); //amount to scroll texture on each draw
220 W3DDisplay::m_3DScene->Add_Render_Object( line); //add it to our scene so it gets rendered with other objects.
221 }
222}
