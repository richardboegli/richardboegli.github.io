149// strtrim ====================================================================
151//=============================================================================
152static char* strtrim(char* buffer)
153{
154 if (buffer != NULL) {
155 // Strip leading white space from the string.
156 char * source = buffer;
157 while ((*source != 0) && ((unsigned char)*source <= 32))
158 {
159 source++;
160 }
161
162 if (source != buffer)
163 {
164 strcpy(buffer, source);
165 }
166
167 // Clip trailing white space from the string.
168 for (int index = strlen(buffer)-1; index >= 0; index--)
169 {
170 if ((*source != 0) && ((unsigned char)buffer[index] <= 32))
171 {
172 buffer[index] = '\0';
173 }
174 else
175 {
176 break;
177 }
178 }
179 }
180
181 return buffer;
182}
183
184static char *nextParam(char *newSource, char *seps)
185{
186 static char *source = NULL;
187 if (newSource)
188 {
189 source = newSource;
190 }
191 if (!source)
192 {
193 return NULL;
194 }
195
196 // find first separator
197 char *first = source;//strpbrk(source, seps);
198 if (first)
199 {
200 // go past initial spaces
201 char *firstNonSpace = first;
202 while (*firstNonSpace == ' ')
203 ++firstNonSpace;
204 first = firstNonSpace;
205
206 // go past separator
207 char *firstSep = strpbrk(first, seps);
208 char firstChar[2] = {0,0};
209 if (firstSep == first)
210 {
211 firstChar[0] = *first;
212 while (*first == firstChar[0]) first++;
213 }
214
215 // find end
216 char *end;
217 if (firstChar[0])
218 end = strpbrk(first, firstChar);
219 else
220 end = strpbrk(first, seps);
221
222 // trim string & save next start pos
223 if (end)
224 {
225 source = end+1;
226 *end = 0;
227
228 if (!*source)
229 source = NULL;
230 }
231 else
232 {
233 source = NULL;
234 }
235
236 if (first && !*first)
237 first = NULL;
238 }
239
240 return first;
241}
