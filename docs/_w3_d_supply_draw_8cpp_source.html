71void W3DSupplyDraw::updateDrawModuleSupplyStatus( Int maxSupply, Int currentSupply )
72{
73 W3DModelDraw::updateDrawModuleSupplyStatus( maxSupply, currentSupply );
74
75 AsciiString boneName = getW3DSupplyDrawModuleData()->m_supplyBonePrefix;
76 if( m_totalBones == -1 )
77 {
78 m_totalBones = getDrawable()->getPristineBonePositions( boneName.str(), 1, NULL, NULL, INT_MAX );// The last arg is to guard the size of the arrays. I am not passing any in, I am just counting bones.
79 m_lastNumberShown = m_totalBones;
80 }
81
82 // Figure the % of our bones we should show, and if it is a different % than last time
83 // start showing and hiding them.
84 Int bonesToShow = ceil(m_totalBones * ( currentSupply / (float)maxSupply ));
85 bonesToShow = min( bonesToShow, m_totalBones );
86
87 if( bonesToShow != m_lastNumberShown )
88 {
89 // Show/hide the bones that are now different, the indices between last and now (low, high].
90 Int lowIndex = min( m_lastNumberShown, bonesToShow );
91 Int highIndex = max( m_lastNumberShown, bonesToShow );
92 Bool hide = bonesToShow < m_lastNumberShown;
93 Int currentIndex = lowIndex + 1;
94
95 std::vector<ModelConditionInfo::HideShowSubObjInfo> boneVector;
96 while( currentIndex <= highIndex )
97 {
98 char buffer[16];
99 sprintf( buffer, "%s%02d", boneName.str(), currentIndex );
100 ModelConditionInfo::HideShowSubObjInfo info;
101 info.hide = hide;
102 info.subObjName = buffer;
103 boneVector.push_back(info);
104
105 ++currentIndex;
106 }
107 doHideShowSubObjs(&boneVector);
108
109 m_lastNumberShown = bonesToShow;
110 }
111}
