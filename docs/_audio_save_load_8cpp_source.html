192//
193// Save
194//
196bool
197DynamicAudioSaveLoadClass::Save (ChunkSaveClass &csave)
198{
199 bool retval = true;
200
201 //
202 // Save the static sounds
203 //
204 SoundSceneClass *scene = WWAudioClass::Get_Instance ()->Get_Sound_Scene ();
205 if (scene != NULL) {
206
207 csave.Begin_Chunk (CHUNKID_DYNAMIC_VARIABLES);
208 float global_scale = LogicalListenerClass::Get_Global_Scale ();
209 WRITE_MICRO_CHUNK (csave, VARID_LOGICAL_LISTENER_GLOBAL_SCALE, global_scale);
210 csave.End_Chunk ();
211
212 csave.Begin_Chunk (CHUNKID_DYNAMIC_SCENE);
213 scene->Save_Dynamic (csave);
214 csave.End_Chunk ();
215 }
216
217 return retval;
218}

222//
223// Load
224//
226bool
227DynamicAudioSaveLoadClass::Load (ChunkLoadClass &cload)
228{
229 bool retval = true;
230 while (cload.Open_Chunk ()) {
231 switch (cload.Cur_Chunk_ID ()) {
232
233 case CHUNKID_DYNAMIC_VARIABLES:
234 {
235 //
236 // Read all the variables from their micro-chunks
237 //
238 while (cload.Open_Micro_Chunk ()) {
239 switch (cload.Cur_Micro_Chunk_ID ()) {
240
241 //
242 // Load the global scale for logical listeners
243 //
244 case VARID_LOGICAL_LISTENER_GLOBAL_SCALE:
245 {
246 float global_scale = 1.0F;
247 LOAD_MICRO_CHUNK (cload, global_scale);
248 LogicalListenerClass::Set_Global_Scale (global_scale);
249 break;
250 }
251 }
252
253 cload.Close_Micro_Chunk ();
254 }
255 }
256 break;
257
258 //
259 // Load the static scene information
260 //
261 case CHUNKID_DYNAMIC_SCENE:
262 {
263 SoundSceneClass *scene = WWAudioClass::Get_Instance ()->Get_Sound_Scene ();
264 if (scene != NULL) {
265 scene->Load_Dynamic (cload);
266 }
267 }
268 break;
269 }
270
271 cload.Close_Chunk ();
272 }
273
274 return retval;
275}
