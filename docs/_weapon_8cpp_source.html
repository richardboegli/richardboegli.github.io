376/*static*/ void WeaponTemplate::parseShotDelay( INI* ini, void *instance, void * /*store*/, const void* /*userData*/ )
377{
378 // This smart parser allows both a single number for traditional delay, and a labeled pair of numbers for a delay range
379 WeaponTemplate* self = (WeaponTemplate*)instance;
380 static const char *MIN_LABEL = "Min";
381 static const char *MAX_LABEL = "Max";
382
383 const char* token = ini->getNextTokenOrNull(ini->getSepsColon());
384
385 if( stricmp(token, MIN_LABEL) == 0 )
386 {
387 // Two entry min/max
388 self->m_minDelayBetweenShots = INI::scanInt(ini->getNextToken(ini->getSepsColon()));
389 token = ini->getNextTokenOrNull(ini->getSepsColon());
390 if( stricmp(token, MAX_LABEL) != 0 )
391 {
392 // Messed up double entry
393 self->m_maxDelayBetweenShots = self->m_minDelayBetweenShots;
394 }
395 else
396 self->m_maxDelayBetweenShots = INI::scanInt(ini->getNextToken(ini->getSepsColon()));
397 }
398 else
399 {
400 // single entry, as in no label so the first token is just a number
401 self->m_minDelayBetweenShots = INI::scanInt(token);
402 self->m_maxDelayBetweenShots = self->m_minDelayBetweenShots;
403 }
404
405 // No matter what we have now, we want to convert it to frames from msec.
406 // ShotDelay used to use parseDurationUnsignedInt, and we are expanding on that.
407 self->m_minDelayBetweenShots = ceilf(ConvertDurationFromMsecsToFrames((Real)self->m_minDelayBetweenShots));
408 self->m_maxDelayBetweenShots = ceilf(ConvertDurationFromMsecsToFrames((Real)self->m_maxDelayBetweenShots));
409
410}

413void WeaponTemplate::postProcessLoad()
414{
415 if (!TheThingFactory)
416 {
417 DEBUG_CRASH(("you must call this after TheThingFactory is inited"));
418 return;
419 }
420
421 if (m_projectileName.isEmpty())
422 {
423 m_projectileTmpl = NULL;
424 }
425 else
426 {
427 m_projectileTmpl = TheThingFactory->findTemplate(m_projectileName);
428 DEBUG_ASSERTCRASH(m_projectileTmpl, ("projectile %s not found!",m_projectileName.str()));
429 }
430
431 for (Int i = LEVEL_FIRST; i <= LEVEL_LAST; ++i)
432 {
433 // And the OCL if there is one
434 if (m_fireOCLNames[i].isEmpty())
435 {
436 m_fireOCLs[i] = NULL;
437 }
438 else
439 {
440 m_fireOCLs[i] = TheObjectCreationListStore->findObjectCreationList(m_fireOCLNames[i].str() );
441 DEBUG_ASSERTCRASH(m_fireOCLs[i], ("OCL %s not found in a weapon!",m_fireOCLNames[i].str()));
442 }
443 m_fireOCLNames[i].clear();
444
445 // And the other OCL if there is one
446 if (m_projectileDetonationOCLNames[i].isEmpty() )
447 {
448 m_projectileDetonationOCLs[i] = NULL;
449 }
450 else
451 {
452 m_projectileDetonationOCLs[i] = TheObjectCreationListStore->findObjectCreationList(m_projectileDetonationOCLNames[i].str() );
453 DEBUG_ASSERTCRASH(m_projectileDetonationOCLs[i], ("OCL %s not found in a weapon!",m_projectileDetonationOCLNames[i].str()));
454 }
455 m_projectileDetonationOCLNames[i].clear();
456 }
457
458} // end postProcessLoad

461Real WeaponTemplate::getAttackRange(const WeaponBonus& bonus) const
462{
463#ifdef RATIONALIZE_ATTACK_RANGE
464 // Note - undersize by 1/4 of a pathfind cell, so that the goal is not teetering on the edge
465 // of firing range. jba.
466 const Real UNDERSIZE = PATHFIND_CELL_SIZE_F*0.25f;
467 Real r = m_attackRange * bonus.getField(WeaponBonus::RANGE) - UNDERSIZE;
468 if (r < 0.0f) r = 0.0f;
469 return r;
470#else
471// fudge this a little to account for pathfinding roundoff & such
472 const Real ATTACK_RANGE_FUDGE = 1.05f;
473 return m_attackRange * bonus.getField(WeaponBonus::RANGE) * ATTACK_RANGE_FUDGE;
474#endif
475}

570Real WeaponTemplate::estimateWeaponTemplateDamage(
571 const Object *sourceObj,
572 const Object *victimObj,
573 const Coord3D* victimPos,
574 const WeaponBonus& bonus
575) const
576{
577 if (sourceObj == NULL || (victimObj == NULL && victimPos == NULL))
578 {
579 DEBUG_CRASH(("bad args to estimate"));
580 return 0.0f;
581 }
582
583 DamageType damageType = getDamageType();
584 DeathType deathType = getDeathType();
585
586 if (victimObj && victimObj->isKindOf(KINDOF_SHRUBBERY))
587 {
588 if (deathType == DEATH_BURNED)
589 {
590 // this is just a nonzero value, to ensure we can target shrubbery with flame weapons, regardless...
591 return 1.0f;
592 }
593 else
594 {
595 return 0.0f;
596 }
597 }
598
599
600 // hmm.. must be shooting a firebase or such, if there is noone home to take the bullet, return 0!
601 if ( victimObj->isKindOf( KINDOF_STRUCTURE) && damageType == DAMAGE_SNIPER )
602 {
603 if ( victimObj->getContain() )
604 {
605 if ( victimObj->getContain()->getContainCount() == 0 )
606 return 0.0f;
607 }
608 }
609
610
611
612
613 if (damageType == DAMAGE_SURRENDER || m_allowAttackGarrisonedBldgs)
614 {
615 ContainModuleInterface* contain = victimObj->getContain();
616 if( contain && contain->getContainCount() > 0 && contain->isGarrisonable() && !contain->isImmuneToClearBuildingAttacks() )
617 {
618 // this is just a nonzero value, to ensure we can target garrisoned things with surrender weapons, regardless...
619 return 1.0f;
620 }
621 }
622
623 if( victimObj )
624 {
625 if( damageType == DAMAGE_DISARM )
626 {
627 if( victimObj->isKindOf( KINDOF_MINE ) || victimObj->isKindOf( KINDOF_BOOBY_TRAP ) || victimObj->isKindOf( KINDOF_DEMOTRAP ) )
628 {
629 // this is just a nonzero value, to ensure we can target mines with disarm weapons, regardless...
630 return 1.0f;
631 }
632 return 0.0f;
633 }
634 if( damageType == DAMAGE_DEPLOY && !victimObj->isAirborneTarget() )
635 {
636 return 1.0f;
637 }
638 }
639
640 //@todo Kris need to examine the DAMAGE_HACK type for damage estimation purposes.
641 //Likely this damage type will have threat implications that won't properly be dealt with until resolved.
642
643// const Coord3D* sourcePos = sourceObj->getPosition();
644 if (victimPos == NULL)
645 {
646 victimPos = victimObj->getPosition();
647 }
648
649 Real damageAmount = getPrimaryDamage(bonus);
650 if (victimObj == NULL)
651 {
652 return damageAmount;
653 }
654 else
655 {
656 DamageInfoInput damageInfo;
657 damageInfo.m_damageType = damageType;
658 damageInfo.m_deathType = deathType;
659 damageInfo.m_sourceID = sourceObj->getID();
660 damageInfo.m_amount = damageAmount;
661 return victimObj->estimateDamage(damageInfo);
662 }

665//-------------------------------------------------------------------------------------------------
666Bool WeaponTemplate::shouldProjectileCollideWith(
667 const Object* projectileLauncher,
668 const Object* projectile,
669 const Object* thingWeCollidedWith,
670 ObjectID intendedVictimID // could be INVALID_ID for a position-shot
671) const
672{
673 if (!projectile || !thingWeCollidedWith)
674 return false;
675
676 // if it's our intended victim, we want to collide with it, regardless of any other considerations.
677 if (intendedVictimID == thingWeCollidedWith->getID())
678 return true;
679
680 if (projectileLauncher != NULL)
681 {
682
683 // Don't hit your own launcher, ever.
684 if (projectileLauncher == thingWeCollidedWith)
685 return false;
686
687 // If our launcher is inside something, and that something is 'thingWeCollidedWith' we won't collide
688 const Object *launcherContainedBy = projectileLauncher->getContainedBy();
689 if( launcherContainedBy == thingWeCollidedWith )
690 return false;
691
692 }
693
694 // never bother burning already-burned things. (srj)
695 if (getDamageType() == DAMAGE_FLAME || getDamageType() == DAMAGE_PARTICLE_BEAM)
696 {
697 if (thingWeCollidedWith->testStatus(OBJECT_STATUS_BURNED))
698 {
699 return false;
700 }
701 }
702
703 // horrible special case for airplanes sitting on airfields: the projectile might
704 // "collide" with the airfield's (invisible) collision geometry when a resting plane
705 // is targeted. we don't want this. special case it:
706 if (thingWeCollidedWith->isKindOf(KINDOF_FS_AIRFIELD))
707 {
708 //
709 // ok, so if we are an airfield, and our intended victim has a reserved space
710 // with us, it "belongs" to us and collisions intended for it should not detonate
711 // as a result of colliding with us.
712 //
713 // notes:
714 // -- we have already established that "thingWeCollidedWith" is not the intended victim (above)
715 // -- this does NOT verify that the plane is actually parked at the airfield; it might be elsewhere
716 // (but if it is, it's highly unlikely that this sort of collision could occur)
717 //
718 for (BehaviorModule** i = thingWeCollidedWith->getBehaviorModules(); *i; ++i)
719 {
720 ParkingPlaceBehaviorInterface* pp = (*i)->getParkingPlaceBehaviorInterface();
721 if (pp != NULL && pp->hasReservedSpace(intendedVictimID))
722 return false;
723 }
724 }
725
726 // if something has a Sneaky Target offset, it is momentarily immune to being hit...
727 // normally this shouldn't happen, but occasionally can by accident. so avoid it. (srj)
728 const AIUpdateInterface* ai = thingWeCollidedWith->getAI();
729 if (ai != NULL && ai->getSneakyTargetingOffset(NULL))
730 {
731 return false;
732 }
733
734 Int requiredMask = 0;
735
736 Relationship r = projectile->getRelationship(thingWeCollidedWith);
737 if (r == ALLIES) requiredMask = WEAPON_COLLIDE_ALLIES;
738 else if (r == ENEMIES) requiredMask = WEAPON_COLLIDE_ENEMIES;
739
740 if (thingWeCollidedWith->isKindOf(KINDOF_STRUCTURE))
741 {
742 if (thingWeCollidedWith->getControllingPlayer() == projectile->getControllingPlayer())
743 requiredMask |= WEAPON_COLLIDE_CONTROLLED_STRUCTURES;
744 else
745 requiredMask |= WEAPON_COLLIDE_STRUCTURES;
746 }
747 if (thingWeCollidedWith->isKindOf(KINDOF_SHRUBBERY)) requiredMask |= WEAPON_COLLIDE_SHRUBBERY;
748 if (thingWeCollidedWith->isKindOf(KINDOF_PROJECTILE)) requiredMask |= WEAPON_COLLIDE_PROJECTILE;
749 if (thingWeCollidedWith->getTemplate()->getFenceWidth() > 0) requiredMask |= WEAPON_COLLIDE_WALLS;
750 if (thingWeCollidedWith->isKindOf(KINDOF_SMALL_MISSILE)) requiredMask |= WEAPON_COLLIDE_SMALL_MISSILES; //All missiles are also projectiles!
751 if (thingWeCollidedWith->isKindOf(KINDOF_BALLISTIC_MISSILE)) requiredMask |= WEAPON_COLLIDE_BALLISTIC_MISSILES; //All missiles are also projectiles!
752
753 // if any in requiredMask are present in collidemask, do the collision. (srj)
754 if ((getProjectileCollideMask() & requiredMask) != 0)
755 return true;
756
757 //DEBUG_LOG(("Rejecting projectile collision between %s and %s!\n",projectile->getTemplate()->getName().str(),thingWeCollidedWith->getTemplate()->getName().str()));
758 return false;
759}

761//-------------------------------------------------------------------------------------------------
762UnsignedInt WeaponTemplate::fireWeaponTemplate
763(
764 const Object *sourceObj,
765 WeaponSlotType wslot,
766 Int specificBarrelToUse,
767 Object *victimObj,
768 const Coord3D* victimPos,
769 const WeaponBonus& bonus,
770 Bool isProjectileDetonation,
771 Bool ignoreRanges,
772 Weapon *firingWeapon,
773 ObjectID* projectileID,
774 Bool inflictDamage
775) const
776{
777
778 //-extraLogging
779 #if (defined(_DEBUG) || defined(_INTERNAL))
780 AsciiString targetStr;
781 if( TheGlobalData->m_extraLogging )
782 {
783 if( victimObj )
784 targetStr.format( "%s", victimObj->getTemplate()->getName().str() );
785 else if( victimPos )
786 targetStr.format( "%d,%d,%d", victimPos->x, victimPos->y, victimPos->z );
787 else
788 targetStr.format( "SELF." );
789
790 DEBUG_LOG( ("%d - WeaponTemplate::fireWeaponTemplate() begin - %s attacking %s - ",
791 TheGameLogic->getFrame(), sourceObj->getTemplate()->getName().str(), targetStr.str() ) );
792 }
793 #endif
794 //end -extraLogging
795
796 //CRCDEBUG_LOG(("WeaponTemplate::fireWeaponTemplate() from %s\n", DescribeObject(sourceObj).str()));
797 DEBUG_ASSERTCRASH(specificBarrelToUse >= 0, ("specificBarrelToUse should no longer be -1\n"));
798
799 if (sourceObj == NULL || (victimObj == NULL && victimPos == NULL))
800 {
801 //-extraLogging
802 #if (defined(_DEBUG) || defined(_INTERNAL))
803 if( TheGlobalData->m_extraLogging )
804 DEBUG_LOG( ("FAIL 1 (sourceObj %d == NULL || (victimObj %d == NULL && victimPos %d == NULL)\n", sourceObj != 0, victimObj != 0, victimPos != 0) );
805 #endif
806 //end -extraLogging
807
808 return 0;
809 }
810
811 DEBUG_ASSERTCRASH((m_primaryDamage > 0) || (victimObj == NULL), ("You can't really shoot a zero damage weapon at an Object.") );
812
813 ObjectID sourceID = sourceObj->getID();
814 const Coord3D* sourcePos = sourceObj->getPosition();
815
816 Real distSqr;
817 ObjectID victimID;
818 TBridgeAttackInfo info;
819 Coord3D victimPosStorage;
820 if (victimObj)
821 {
822 DEBUG_ASSERTLOG(sourceObj != victimObj, ("*** firing weapon at self -- is this really what you want?\n"));
823 victimPos = victimObj->getPosition();
824 victimID = victimObj->getID();
825
826 Coord3D sneakyOffset;
827 const AIUpdateInterface* ai = victimObj->getAI();
828 if (ai != NULL && ai->getSneakyTargetingOffset(&sneakyOffset))
829 {
830 victimPosStorage = *victimPos;
831 victimPosStorage.x += sneakyOffset.x;
832 victimPosStorage.y += sneakyOffset.y;
833 victimPosStorage.z += sneakyOffset.z;
834
835 victimPos = &victimPosStorage;
836 // for a sneaky offset, we always target a position rather than an object
837 victimObj = NULL;
838 victimID = INVALID_ID;
839 distSqr = ThePartitionManager->getDistanceSquared(sourceObj, victimPos, ATTACK_RANGE_CALC_TYPE);
840 }
841 else
842 {
843 if (victimObj->isKindOf(KINDOF_BRIDGE))
844 {
845 // Bridges are kind of oddball - they have 2 target points at either end.
846 TheTerrainLogic->getBridgeAttackPoints(victimObj, &info);
847 distSqr = ThePartitionManager->getDistanceSquared( sourceObj, &info.attackPoint1, ATTACK_RANGE_CALC_TYPE );
848 victimPos = &info.attackPoint1;
849 Real distSqr2 = ThePartitionManager->getDistanceSquared( sourceObj, &info.attackPoint2, ATTACK_RANGE_CALC_TYPE );
850 if (distSqr > distSqr2)
851 {
852 // Try the other one.
853 distSqr = distSqr2;
854 victimPos = &info.attackPoint2;
855 }
856 }
857 else
858 {
859 distSqr = ThePartitionManager->getDistanceSquared(sourceObj, victimObj, ATTACK_RANGE_CALC_TYPE);
860 }
861 }
862 }
863 else
864 {
865 victimID = INVALID_ID;
866 distSqr = ThePartitionManager->getDistanceSquared(sourceObj, victimPos, ATTACK_RANGE_CALC_TYPE);
867 }
868
869// DEBUG_LOG(("WeaponTemplate::fireWeaponTemplate: firing weapon %s (source=%s, victim=%s)\n",
870// m_name.str(),sourceObj->getTemplate()->getName().str(),victimObj?victimObj->getTemplate()->getName().str():"NULL"));
871
872 //Only perform this check if the weapon isn't a leech range weapon (which can have unlimited range!)
873 if( !ignoreRanges && !isLeechRangeWeapon() )
874 {
875 Real attackRangeSqr = sqr(getAttackRange(bonus));
876 if (distSqr > attackRangeSqr)
877 {
878 //DEBUG_ASSERTCRASH(distSqr < 5*5 || distSqr < attackRangeSqr*1.2f, ("*** victim is out of range (%f vs %f) of this weapon -- why did we attempt to fire?\n",sqrtf(distSqr),sqrtf(attackRangeSqr)));
879
880 //-extraLogging
881 #if (defined(_DEBUG) || defined(_INTERNAL))
882 if( TheGlobalData->m_extraLogging )
883 DEBUG_LOG( ("FAIL 2 (distSqr %.2f > attackRangeSqr %.2f)\n", distSqr, attackRangeSqr ) );
884 #endif
885 //end -extraLogging
886
887 return 0;
888 }
889 }
890
891 if (!ignoreRanges)
892 {
893 Real minAttackRangeSqr = sqr(getMinimumAttackRange());
894#ifdef RATIONALIZE_ATTACK_RANGE
895 if (distSqr < minAttackRangeSqr && !isProjectileDetonation)
896#else
897 if (distSqr < minAttackRangeSqr-0.5f && !isProjectileDetonation)
898#endif
899 {
900 DEBUG_ASSERTCRASH(distSqr > minAttackRangeSqr*0.8f, ("*** victim is closer than min attack range (%f vs %f) of this weapon -- why did we attempt to fire?\n",sqrtf(distSqr),sqrtf(minAttackRangeSqr)));
901
902 //-extraLogging
903 #if (defined(_DEBUG) || defined(_INTERNAL))
904 if( TheGlobalData->m_extraLogging )
905 DEBUG_LOG( ("FAIL 3 (distSqr %.2f< minAttackRangeSqr %.2f - 0.5f && !isProjectileDetonation %d)\n", distSqr, minAttackRangeSqr, isProjectileDetonation ) );
906 #endif
907 //end -extraLogging
908
909 return 0;
910 }
911 }
912
913 // call this even if FXList is null, because this also handles stuff like Gun Barrel Recoil
914 if (sourceObj && sourceObj->getDrawable())
915 {
916 Coord3D targetPos;
917 if( victimObj )
918 {
919 victimObj->getGeometryInfo().getCenterPosition( *victimObj->getPosition(), targetPos );
920 }
921 else
922 {
923 targetPos.set( victimPos );
924 }
925 Real reAngle = getWeaponRecoilAmount();
926 Real reDir = reAngle != 0.0f ? (atan2(victimPos->y - sourcePos->y, victimPos->x - sourcePos->x)) : 0.0f;
927 VeterancyLevel v = sourceObj->getVeterancyLevel();
928 const FXList* fx = isProjectileDetonation ? getProjectileDetonateFX(v) : getFireFX(v);
929
930 if ( TheGameLogic->getFrame() < firingWeapon->getSuspendFXFrame() )
931 fx = NULL;
932
933 Bool handled;
934
935 if(!sourceObj->isLocallyControlled() // if user watching is not controller and
936 && sourceObj->testStatus(OBJECT_STATUS_STEALTHED) // if unit is stealthed (like a Pathfinder)
937 && !sourceObj->testStatus(OBJECT_STATUS_DETECTED) // but not detected...
938 && !sourceObj->testStatus(OBJECT_STATUS_DISGUISED) // and not disguised...
939 && !sourceObj->isKindOf(KINDOF_MINE) // and not a mine (which always do the FX, even if hidden)...
940 && !isPlayFXWhenStealthed() // and not a weapon marked to playwhenstealthed
941 )
942 {
943 handled = TRUE; // then let's just pretend like we did the fx by returning true
944 }
945 else
946 {
947 handled = sourceObj->getDrawable()->handleWeaponFireFX(wslot,
948 specificBarrelToUse,
949 fx,
950 getWeaponSpeed(),
951 reAngle,
952 reDir,
953 &targetPos,
954 getPrimaryDamageRadius(bonus)
955 );
956 }
957
958 if (handled == false && fx != NULL)
959 {
960 // bah. just play it at the drawable's pos.
961 //DEBUG_LOG(("*** WeaponFireFX not fully handled by the client\n"));
962 const Coord3D* where = isContactWeapon() ? &targetPos : sourceObj->getDrawable()->getPosition();
963 FXList::doFXPos(fx, where, sourceObj->getDrawable()->getTransformMatrix(), getWeaponSpeed(), &targetPos, getPrimaryDamageRadius(bonus));
964 }
965 }
966
967 // Now do the FireOCL if there is one
968 if( sourceObj )
969 {
970 VeterancyLevel v = sourceObj->getVeterancyLevel();
971 const ObjectCreationList *oclToUse = isProjectileDetonation ? getProjectileDetonationOCL(v) : getFireOCL(v);
972 if( oclToUse )
973 ObjectCreationList::create( oclToUse, sourceObj, NULL );
974 }
975
976 Coord3D projectileDestination = *victimPos; //Need to copy this, as we have a pointer to their actual position
977 Real scatterRadius = 0.0f;
978 if( m_scatterRadius > 0.0f || m_infantryInaccuracyDist > 0.0f && victimObj && victimObj->isKindOf( KINDOF_INFANTRY ) )
979 {
980 // This weapon scatters, so clear the victimObj, as we are no longer shooting it directly,
981 // and find a random point within the radius to shoot at as victimPos
982 scatterRadius = m_scatterRadius;
983
984 // if it's an object, aim at the center, not the ground part (srj)
985 PathfindLayerEnum targetLayer = LAYER_GROUND;
986 if( victimObj )
987 {
988 if( victimObj->isKindOf( KINDOF_STRUCTURE ) )
989 {
990 victimObj->getGeometryInfo().getCenterPosition(*victimObj->getPosition(), projectileDestination);
991 }
992 if( m_infantryInaccuracyDist > 0.0f && victimObj->isKindOf( KINDOF_INFANTRY ) )
993 {
994 //If we are firing a weapon that is considered inaccurate against infantry, then add it to
995 //the scatter radius!
996 scatterRadius += m_infantryInaccuracyDist;
997 }
998 targetLayer = victimObj->getLayer();
999 }
1000
1001 //victimObj = NULL; // his position is already in victimPos, if he existed
1002
1003 //Randomize the scatter radius (sometimes it can be more accurate than others)
1004 scatterRadius = GameLogicRandomValueReal( 0, scatterRadius );
1005 Real scatterAngleRadian = GameLogicRandomValueReal( 0, 2*PI );
1006
1007 Coord3D firingOffset;
1008 firingOffset.zero();
1009 firingOffset.x = scatterRadius * Cos( scatterAngleRadian );
1010 firingOffset.y = scatterRadius * Sin( scatterAngleRadian );
1011
1012 projectileDestination.x += firingOffset.x;
1013 projectileDestination.y += firingOffset.y;
1014
1015 //What's suddenly become crucially important is to FIRE at the ground at this location!!!
1016 //If we aim for the center point of our target and miss, the shot will go much farther than
1017 //we expect!
1018 // srj sez: we should actually fire at the layer the victim is on, if possible, in case it is on a bridge...
1019 projectileDestination.z = TheTerrainLogic->getLayerHeight( projectileDestination.x, projectileDestination.y, targetLayer );
1020 }
1021
1022 if (getProjectileTemplate() == NULL || isProjectileDetonation)
1023 {
1024 // see if we need to be called back at a later point to deal the damage.
1025 Coord3D v;
1026 v.x = victimPos->x - sourcePos->x;
1027 v.y = victimPos->y - sourcePos->y;
1028 v.z = victimPos->z - sourcePos->z;
1029 // don't round the result; we WANT a fractional-frame-delay in this case.
1030 Real delayInFrames = (v.length() / getWeaponSpeed());
1031
1032 ObjectID damageID = getDamageDealtAtSelfPosition() ? INVALID_ID : victimID;
1033
1034 if( firingWeapon->isLaser() )
1035 {
1036 if( scatterRadius <= getPrimaryDamageRadius( bonus ) || scatterRadius <= getSecondaryDamageRadius( bonus ) )
1037 {
1038 //The laser is close enough to damage the object, so just hit it directly. Some victim objects
1039 //adjust the laser's position to prevent it from hitting the ground.
1040 if( victimObj )
1041 {
1042 projectileDestination.set( victimObj->getPosition() );
1043 }
1044 firingWeapon->createLaser( sourceObj, victimObj, &projectileDestination );
1045 }
1046 else
1047 {
1048 //We are missing our intended target, so now we want to aim at the ground at the projectile offset.
1049 damageID = INVALID_ID;
1050 firingWeapon->createLaser( sourceObj, NULL, &projectileDestination );
1051 }
1052 if( inflictDamage )
1053 {
1054 dealDamageInternal( sourceID, damageID, &projectileDestination, bonus, isProjectileDetonation );
1055 }
1056 return TheGameLogic->getFrame();
1057 }
1058
1059 const Coord3D* damagePos = getDamageDealtAtSelfPosition() ? sourcePos : victimPos;
1060 if (delayInFrames < 1.0f)
1061 {
1062 // go ahead and do it now
1063 //DEBUG_LOG(("WeaponTemplate::fireWeaponTemplate: firing weapon immediately!\n"));
1064 if( inflictDamage )
1065 {
1066 dealDamageInternal(sourceID, damageID, damagePos, bonus, isProjectileDetonation);
1067 }
1068
1069 //-extraLogging
1070 #if (defined(_DEBUG) || defined(_INTERNAL))
1071 if( TheGlobalData->m_extraLogging )
1072 DEBUG_LOG( ("EARLY 4 (delayed damage applied now)\n") );
1073 #endif
1074 //end -extraLogging
1075
1076
1077 return TheGameLogic->getFrame();
1078 }
1079 else
1080 {
1081 UnsignedInt when = 0;
1082 if( TheWeaponStore && inflictDamage )
1083 {
1084 UnsignedInt delayInWholeFrames = REAL_TO_INT_CEIL(delayInFrames);
1085 when = TheGameLogic->getFrame() + delayInWholeFrames;
1086 //DEBUG_LOG(("WeaponTemplate::fireWeaponTemplate: firing weapon in %d frames (= %d)!\n", delayInWholeFrames,when));
1087 TheWeaponStore->setDelayedDamage(this, damagePos, when, sourceID, damageID, bonus);
1088 }
1089
1090 //-extraLogging
1091 #if (defined(_DEBUG) || defined(_INTERNAL))
1092 if( TheGlobalData->m_extraLogging )
1093 DEBUG_LOG( ("EARLY 5 (delaying damage applied until frame %d)\n", when ) );
1094 #endif
1095 //end -extraLogging
1096
1097
1098 return when;
1099 }
1100 }
1101 else // must be a projectile
1102 {
1103 Player *owningPlayer = sourceObj->getControllingPlayer(); //Need to know so missiles don't collide with firer
1104 Object *projectile = TheThingFactory->newObject( getProjectileTemplate(), owningPlayer->getDefaultTeam() );
1105 projectile->setProducer(sourceObj);
1106
1107 //If the player has battle plans (America Strategy Center), then apply those bonuses
1108 //to this object if applicable. Internally it validates certain kinds of objects.
1109 //When projectiles are created, weapon bonuses such as damage may get applied.
1110 if( owningPlayer->getNumBattlePlansActive() > 0 )
1111 {
1112 owningPlayer->applyBattlePlanBonusesForObject( projectile );
1113 }
1114
1115
1116 //Store the project ID in the object as the last projectile fired!
1117 if (projectileID)
1118 *projectileID = projectile->getID();
1119
1120 // Notify special power tracking
1121 SpecialPowerCompletionDie *die = sourceObj->findSpecialPowerCompletionDie();
1122 if (die)

1221void WeaponTemplate::dealDamageInternal(ObjectID sourceID, ObjectID victimID, const Coord3D *pos, const WeaponBonus& bonus, Bool isProjectileDetonation) const

1224 return;
1225
1226 if (victimID == 0 && pos == NULL) // must have some sort of destination
1227 return;
1228
1229 Object *source = TheGameLogic->findObjectByID(sourceID); // might be null...
1230
1231 //
1236 //
1237
1238 trimOldHistoricDamage();
1239
1240 if( m_historicBonusCount > 0 && m_historicBonusWeapon != this )
1241 {
1242 Real radSqr = m_historicBonusRadius * m_historicBonusRadius;
1243 Int count = 0;
1244 UnsignedInt frameNow = TheGameLogic->getFrame();
1245 UnsignedInt oldestThatWillCount = frameNow - m_historicBonusTime; // Anything before this frame is "more than two seconds ago" eg
1246 for( HistoricWeaponDamageList::const_iterator it = m_historicDamage.begin(); it != m_historicDamage.end(); ++it )
1247 {

1249 is2DDistSquaredLessThan( *pos, it->location, radSqr ) )
1250 {
1251 // This one is close enough in time and distance, so count it. This is tracked by template since it applies
1252 // across units, so don't try to clear historicDamage on success in here.
1253 ++count;
1254 }
1255 }
1256
1257 if( count >= m_historicBonusCount - 1 ) // minus 1 since we include ourselves implicitly
1258 {
1259 TheWeaponStore->createAndFireTempWeapon(m_historicBonusWeapon, source, pos);
1260
1264 m_historicDamage.clear();
1265
1266 }
1267 else
1268 {
1269
1270 // add AFTER checking for historic stuff
1271 m_historicDamage.push_back( HistoricWeaponDamageInfo(frameNow, *pos) );
1272
1273 } // end else
1274
1275 } // if historic bonuses
1276
1277//DEBUG_LOG(("WeaponTemplate::dealDamageInternal: dealing damage %s at frame %d\n",m_name.str(),TheGameLogic->getFrame()));
1278
1279 // if there's a specific victim, use it's pos (overriding the value passed in)
1280 Object *primaryVictim = victimID ? TheGameLogic->findObjectByID(victimID) : NULL; // might be null...
1281 if (primaryVictim)
1282 {
1283 pos = primaryVictim->getPosition();
1284 }
1285
1286 DamageType damageType = getDamageType();
1287 DeathType deathType = getDeathType();
1288 ObjectStatusTypes damageStatusType = getDamageStatusType();
1289 if (getProjectileTemplate() == NULL || isProjectileDetonation)
1290 {
1291 SimpleObjectIterator *iter;
1292 Object *curVictim;
1293 Real curVictimDistSqr;
1294
1295 Real primaryRadius = getPrimaryDamageRadius(bonus);
1296 Real secondaryRadius = getSecondaryDamageRadius(bonus);
1297 Real primaryDamage = getPrimaryDamage(bonus);
1298 Real secondaryDamage = getSecondaryDamage(bonus);
1299 Int affects = getAffectsMask();
1300
1301 DEBUG_ASSERTCRASH(secondaryRadius >= primaryRadius || secondaryRadius == 0.0f, ("secondary radius should be >= primary radius (or zero)\n"));
1302
1303 Real primaryRadiusSqr = sqr(primaryRadius);
1304 Real radius = max(primaryRadius, secondaryRadius);
1305 if (radius > 0.0f)
1306 {
1307 iter = ThePartitionManager->iterateObjectsInRange(pos, radius, DAMAGE_RANGE_CALC_TYPE);
1308 curVictim = iter->firstWithNumeric(&curVictimDistSqr);
1309 }
1310 else
1311 {
1312 //DEBUG_ASSERTCRASH(primaryVictim != NULL, ("weapons without radii should always pass in specific victims"));
1313 // check against victimID rather than primaryVictim, since we may have targeted a legitimate victim
1314 // that got killed before the damage was dealt... (srj)
1315 //DEBUG_ASSERTCRASH(victimID != 0, ("weapons without radii should always pass in specific victims"));
1316 iter = NULL;
1317 curVictim = primaryVictim;
1318 curVictimDistSqr = 0.0f;
1319
1320 if( affects & WEAPON_KILLS_SELF )
1321 {
1322 DamageInfo damageInfo;
1323 damageInfo.in.m_damageType = damageType;
1324 damageInfo.in.m_deathType = deathType;
1325 damageInfo.in.m_sourceID = sourceID;
1326 damageInfo.in.m_sourcePlayerMask = 0;
1327 damageInfo.in.m_damageStatusType = damageStatusType;
1328 damageInfo.in.m_amount = HUGE_DAMAGE_AMOUNT;
1329 source->attemptDamage( &damageInfo );
1330 return;
1331 }
1332 }
1333 MemoryPoolObjectHolder hold(iter);
1334
1335 for (; curVictim != NULL; curVictim = iter ? iter->nextWithNumeric(&curVictimDistSqr) : NULL)
1336 {
1337 Bool killSelf = false;
1338 if (source != NULL)
1339 {
1340 // anytime something is designated as the "primary victim" (ie, the direct target
1341 // of the weapon), we ignore all the "affects" flags.
1342 if (curVictim != primaryVictim)
1343 {
1344
1345 if( (affects & WEAPON_KILLS_SELF) && source == curVictim )
1346 {
1347 killSelf = true;
1348 }
1349 else
1350 {
1351
1352 // should object ever be allowed to damage themselves? methinks not...
1353 // exception: a few weapons allow this (eg, for suicide bombers).
1354 if( (affects & WEAPON_AFFECTS_SELF) == 0 )
1355 {
1356 // Remember that source is a missile for some units, and they don't want to injure them'selves' either
1357 if( source == curVictim || source->getProducerID() == curVictim->getID() )
1358 {
1359 //DEBUG_LOG(("skipping damage done to SELF...\n"));
1360 continue;
1361 }
1362 }
1363
1364 if( affects & WEAPON_DOESNT_AFFECT_SIMILAR )
1365 {
1366 //This means we probably are affecting allies, but don't want to kill nearby members that are the same type as us.
1367 //A good example are a group of terrorists blowing themselves up. We don't want to cause a domino effect that kills
1368 //all of them.
1369 if( source->getTemplate()->isEquivalentTo(curVictim->getTemplate()) && source->getRelationship( curVictim ) == ALLIES )
1370 {
1371 continue;
1372 }
1373 }
1374
1375 if ((affects & WEAPON_DOESNT_AFFECT_AIRBORNE) != 0 && curVictim->isSignificantlyAboveTerrain())
1376 {
1377 continue;
1378 }
1379
1380 /*
1381 The idea here is: if its our ally(/enemies), AND it's not the direct target, AND the weapon doesn't
1382 do radius-damage to allies(/enemies)... skip it.
1383 */
1384 Relationship r = curVictim->getRelationship(source);
1385 Int requiredMask;
1386 if (r == ALLIES)
1387 requiredMask = WEAPON_AFFECTS_ALLIES;
1388 else if (r == ENEMIES)
1389 requiredMask = WEAPON_AFFECTS_ENEMIES;
1390 else /* r == NEUTRAL */
1391 requiredMask = WEAPON_AFFECTS_NEUTRALS;
1392
1393 if( !(affects & requiredMask) )
1394 {
1395 //Skip if we aren't affected by this weapon.
1396 continue;
1397 }
1398 }
1399 }
1400 }
1401
1402 DamageInfo damageInfo;
1403 damageInfo.in.m_damageType = damageType;
1404 damageInfo.in.m_deathType = deathType;
1405 damageInfo.in.m_sourceID = sourceID;
1406 damageInfo.in.m_sourcePlayerMask = 0;
1407 damageInfo.in.m_damageStatusType = damageStatusType;
1408
1409 Coord3D damageDirection;
1410 damageDirection.zero();
1411 if( curVictim && source )
1412 {
1413 damageDirection.set( curVictim->getPosition() );
1414 damageDirection.sub( source->getPosition() );
1415 }
1416
1417 Real allowedAngle = getRadiusDamageAngle();
1418 if( allowedAngle < PI )
1419 {
1420 if( curVictim == NULL || source == NULL )
1421 continue; // We are directional damage, but can't figure out our direction. Just bail.
1422
1423 // People can only be hit in a cone oriented as the firer is oriented
1424 Vector3 sourceVector = source->getTransformMatrix()->Get_X_Vector();
1425 Vector3 damageVector(damageDirection.x, damageDirection.y, damageDirection.z);
1426 sourceVector.Normalize();
1427 damageVector.Normalize();
1428
1429 // These are now normalized, so the dot productis actually the Cos of the angle they form
1430 // A smaller Cos would mean a more obtuse angle
1431 if( Vector3::Dot_Product(sourceVector, damageVector) < Cos(allowedAngle) )
1432 continue;// Too far to the side, can't hurt them.
1433 }
1434
1435 // Grab the vector between the source object causing the damage and the victim in order that we can
1436 // simulate a shockwave pushing objects around
1437 damageInfo.in.m_shockWaveAmount = m_shockWaveAmount;
1438 if (damageInfo.in.m_shockWaveAmount > 0.0f)
1439 {
1440 // Calculate the vector of the shockwave
1441 Coord3D shockWaveVector = damageDirection;
1442
1443 // Guard against zero vector. Make vector stright up if that is the case
1444 if (fabs(shockWaveVector.x) < WWMATH_EPSILON &&
1445 fabs(shockWaveVector.y) < WWMATH_EPSILON &&
1446 fabs(shockWaveVector.z) < WWMATH_EPSILON)
1447 {
1448 shockWaveVector.z = 1.0f;
1449 }
1450
1451 // Populate the damge information with the shockwave information
1452 damageInfo.in.m_shockWaveVector = shockWaveVector;
1453 damageInfo.in.m_shockWaveRadius = m_shockWaveRadius;
1454 damageInfo.in.m_shockWaveTaperOff = m_shockWaveTaperOff;
1455 }
1456
1457 if (source && source->getControllingPlayer()) {
1458 damageInfo.in.m_sourcePlayerMask = source->getControllingPlayer()->getPlayerMask();
1459 }
1460 // note, don't bother with damage multipliers here...
1461 // that's handled internally by the attemptDamage() method.
1462 damageInfo.in.m_amount = (curVictimDistSqr <= primaryRadiusSqr) ? primaryDamage : secondaryDamage;
1463
1464 if( killSelf )
1465 {
1466 //Deal enough damage to kill yourself. I thought about getting the current health and applying
1467 //enough unresistable damage to die... however it's possible that we have different types of
1468 //deaths based on damage type and/or the possibility to resist certain damage types and
1469 //surviving -- so instead, I'm blindly inflicting a very high value of the intended damage type.
1470 damageInfo.in.m_amount = HUGE_DAMAGE_AMOUNT;
1471 //BodyModuleInterface* body = curVictim->getBodyModule();
1472 //if( body )
1473 //{
1474 // Real curVictimHealth = curVictim->getBodyModule()->getHealth();
1475 // damageInfo.in.m_amount = __max( damageInfo.in.m_amount, curVictimHealth );
1476 //}
1477 }
1478
1479 // if the damage-dealer is a projectile, designate the damage as done by its launcher, not the projectile.
1480 // this is much more useful for the AI...
1481 if (source && source->isKindOf(KINDOF_PROJECTILE))
1482 {
1483 for (BehaviorModule** u = source->getBehaviorModules(); *u; ++u)
1484 {
1485 ProjectileUpdateInterface* pui = (*u)->getProjectileUpdateInterface();
1486 if (pui != NULL)
1487 {
1488 damageInfo.in.m_sourceID = pui->projectileGetLauncherID();
1489 break;
1490 }
1491 }
1492 }
1493
1494 curVictim->attemptDamage(&damageInfo);
1495 //DEBUG_ASSERTLOG(damageInfo.out.m_noEffect, ("WeaponTemplate::dealDamageInternal: dealt to %s %08lx: attempted %f, actual %f (%f)\n",
1496 // curVictim->getTemplate()->getName().str(),curVictim,
1497 // damageInfo.in.m_amount, damageInfo.out.m_actualDamageDealt, damageInfo.out.m_actualDamageClipped));
1498 }
1499 }
1500 else
1501 {
1502 DEBUG_CRASH(("projectile weapons should never get dealDamage called directly\n"));
1503 }

2001Bool Weapon::computeApproachTarget(const Object *source, const Object *target, const Coord3D *pos, Real angleOffset, Coord3D& approachTargetPos) const
2002{
2003 // compute unit direction vector from us to our victim
2004 const Coord3D *targetPos;
2005 Coord3D dir;
2006 if (target)
2007 {
2008 targetPos = target->getPosition();
2009 ThePartitionManager->getVectorTo( target, source, ATTACK_RANGE_CALC_TYPE, dir );
2010 }
2011 else if (pos)
2012 {
2013 targetPos = pos;
2014 ThePartitionManager->getVectorTo( source, pos, ATTACK_RANGE_CALC_TYPE, dir );
2015 // Flip the vector to get from source to pos.
2016 dir.x = -dir.x;
2017 dir.y = -dir.y;
2018 dir.z = -dir.z;
2019 }
2020 else
2021 {
2022 DEBUG_CRASH(("error"));
2023 approachTargetPos.zero();
2024 return false;
2025 }
2026
2027 Real dist = dir.length();
2028 Real minAttackRange = m_template->getMinimumAttackRange();
2029 if (minAttackRange > PATHFIND_CELL_SIZE_F && dist < minAttackRange)
2030 {
2031 // We aret too close, so move away from the target.
2032 DEBUG_ASSERTCRASH((minAttackRange<0.9f*getAttackRange(source)), ("Min attack range is too near attack range.\n"));
2033 // Recompute dir, cause if the bounding spheres touch, it will be 0.
2034 Coord3D srcPos = *source->getPosition();
2035 dir.x = srcPos.x-targetPos->x;
2036 dir.y = srcPos.y-targetPos->y;
2037#ifdef ATTACK_RANGE_IS_2D
2038 dir.z = 0.0f;
2039#else
2040 dir.z = srcPos.z-targetPos->z;
2041#endif
2042 dir.normalize();
2043
2044 // if we're airborne and too close, just head for the opposite side.
2045 if (source->isAboveTerrain())
2046 {
2047 // Don't do a 180 degree turn.
2048 Real angle = atan2(-dir.y, -dir.x);
2049 Real relAngle = source->getOrientation()- angle;
2050 if (relAngle>2*PI) relAngle -= 2*PI;
2051 if (relAngle<-2*PI) relAngle += 2*PI;
2052 if (fabs(relAngle)<PI/2) {
2053 dir.x = -dir.x;
2054 dir.y = -dir.y;
2055 dir.z = -dir.z;
2056 }
2057 }
2058
2059 if (angleOffset != 0.0f)
2060 {
2061 Real angle = atan2(dir.y, dir.x);
2062 dir.x = (Real)Cos(angle + angleOffset);
2063 dir.y = (Real)Sin(angle + angleOffset);
2064 }
2065
2066 // select a spot along the line between us, halfway between the min & max range.
2067 Real attackRange = (getAttackRange(source) + minAttackRange)/2.0f;
2068#ifdef ATTACK_RANGE_IS_2D
2069 if (target)
2070 attackRange += target->getGeometryInfo().getBoundingCircleRadius();
2071 attackRange += source->getGeometryInfo().getBoundingCircleRadius();
2072#else
2073 if (target)
2074 attackRange += target->getGeometryInfo().getBoundingSphereRadius();
2075 attackRange += source->getGeometryInfo().getBoundingSphereRadius();
2076#endif
2077 approachTargetPos.x = attackRange * dir.x + targetPos->x;
2078 approachTargetPos.y = attackRange * dir.y + targetPos->y;
2079 approachTargetPos.z = attackRange * dir.z + targetPos->z;
2081 clipToTerrainExtent(approachTargetPos);
2082 return false;
2083 }
2084
2085 const Real FUDGE = 0.001f;
2086 if (dist < FUDGE)
2087 {
2088 // we're close enough!
2089 approachTargetPos = *source->getPosition();
2090 return true;
2091 }
2092 else
2093 {
2094 if (isContactWeapon())
2095 {
2096 // Weapon is basically a contact weapon, like a car bomb. The approach target logic
2097 // has been modified to let it approach the object, so just return the target position. jba.
2098 approachTargetPos = *targetPos;
2099 return false;
2100 }
2101
2102 dir.x /= dist;
2103 dir.y /= dist;
2104 dir.z /= dist;
2105
2106 if (angleOffset != 0.0f)
2107 {
2108 Real angle = atan2(dir.y, dir.x);
2109 dir.x = (Real)Cos(angle + angleOffset);
2110 dir.y = (Real)Sin(angle + angleOffset);
2111 }
2112
2113 // select a spot along the line between us, in range of our weapon
2114 const Real ATTACK_RANGE_APPROACH_FUDGE = 0.9f;
2115 Real attackRange = getAttackRange(source) * ATTACK_RANGE_APPROACH_FUDGE;
2116 approachTargetPos.x = attackRange * dir.x + targetPos->x;
2117 approachTargetPos.y = attackRange * dir.y + targetPos->y;
2118 approachTargetPos.z = attackRange * dir.z + targetPos->z;
2119
2120 if (source->getAI() && source->getAI()->isAircraftThatAdjustsDestination()) {
2121 // Adjust the target so that we are not stacked atop another aircraft.
2122 TheAI->pathfinder()->adjustTargetDestination(source, target, pos, this, &approachTargetPos);
2123 }
2124
2125 return false;
2126 }
2127}

2937/*static*/ void Weapon::calcProjectileLaunchPosition(
2938 const Object* launcher,
2939 WeaponSlotType wslot,
2940 Int specificBarrelToUse,
2941 Matrix3D& worldTransform,
2942 Coord3D& worldPos
2943)
2944{
2945 if( launcher->getContainedBy() )
2946 {
2947 // If we are in an enclosing container, our launch position is our actual position. Yes, I am putting
2948 // a minor case and an oft used function, but the major case is huge and full of math.
2949 if(launcher->getContainedBy()->getContain()->isEnclosingContainerFor(launcher))
2950 {
2951 worldTransform = *launcher->getTransformMatrix();
2952 Vector3 tmp = worldTransform.Get_Translation();
2953 worldPos.x = tmp.X;
2954 worldPos.y = tmp.Y;
2955 worldPos.z = tmp.Z;
2956 return;
2957 }
2958 }
2959
2960 Real turretAngle = 0.0f;
2961 Real turretPitch = 0.0f;
2962 const AIUpdateInterface* ai = launcher->getAIUpdateInterface();
2963 WhichTurretType tur = ai ? ai->getWhichTurretForWeaponSlot(wslot, &turretAngle, &turretPitch) : TURRET_INVALID;
2964 //CRCDEBUG_LOG(("calcProjectileLaunchPosition(): Turret %d, slot %d, barrel %d for %s\n", tur, wslot, specificBarrelToUse, DescribeObject(launcher).str()));
2965
2966 Matrix3D attachTransform(true);
2967 Coord3D turretRotPos = {0.0f, 0.0f, 0.0f};
2968 Coord3D turretPitchPos = {0.0f, 0.0f, 0.0f};
2969 const Drawable* draw = launcher->getDrawable();
2970 //CRCDEBUG_LOG(("Do we have a drawable? %d\n", (draw != NULL)));
2971 if (!draw || !draw->getProjectileLaunchOffset(wslot, specificBarrelToUse, &attachTransform, tur, &turretRotPos, &turretPitchPos))
2972 {
2973 //CRCDEBUG_LOG(("ProjectileLaunchPos %d %d not found!\n",wslot, specificBarrelToUse));
2974 DEBUG_CRASH(("ProjectileLaunchPos %d %d not found!\n",wslot, specificBarrelToUse));
2975 attachTransform.Make_Identity();
2976 turretRotPos.zero();
2977 turretPitchPos.zero();
2978 }
2979 if (tur != TURRET_INVALID)
2980 {
2981 // The attach transform is the pristine front and center position of the fire point
2982 // We can't read from the client, so we need to reproduce the actual point that
2983 // takes turn and pitch into account.
2984 Matrix3D turnAdjustment(1);
2985 Matrix3D pitchAdjustment(1);
2986
2987 // To rotate about a point, move that point to 0,0, rotate, then move it back.
2988 // Pre rotate will keep the first twist from screwing the angle of the second pitch
2989 pitchAdjustment.Translate( turretPitchPos.x, turretPitchPos.y, turretPitchPos.z );
2990 pitchAdjustment.In_Place_Pre_Rotate_Y(-turretPitch);
2991 pitchAdjustment.Translate( -turretPitchPos.x, -turretPitchPos.y, -turretPitchPos.z );
2992
2993 turnAdjustment.Translate( turretRotPos.x, turretRotPos.y, turretRotPos.z );
2994 turnAdjustment.In_Place_Pre_Rotate_Z(turretAngle);
2995 turnAdjustment.Translate( -turretRotPos.x, -turretRotPos.y, -turretRotPos.z );
2996
2997#ifdef ALLOW_TEMPORARIES
2998 attachTransform = turnAdjustment * pitchAdjustment * attachTransform;
2999#else
3000 Matrix3D tmp = attachTransform;
3001 attachTransform.mul(turnAdjustment, pitchAdjustment);
3002 attachTransform.postMul(tmp);
3003#endif
3004 }
3005
