85Bool AssistedTargetingUpdate::isFreeToAssist() const
86{
87 // The reload times of my two weapons are tied together, so Ready is indicitive of either.
88 const Object *me = getObject();
89 if( !me->isAbleToAttack() )
90 return FALSE;// This will cover under construction among other things
91
92 Bool ready = me->getCurrentWeapon() && me->getCurrentWeapon()->getStatus() == READY_TO_FIRE;
93 return ready;
94}

98void AssistedTargetingUpdate::assistAttack( const Object *requestingObject, Object *victimObject )
99{
100 const AssistedTargetingUpdateModuleData *md = getAssistedTargetingUpdateModuleData();
101 Object *me = getObject();
102 if( !me->getAI() )
103 return;
104
105 // lock it just till the weapon is empty or the attack is "done"
106 me->setWeaponLock( md->m_weaponSlot, LOCKED_TEMPORARILY );
107 me->getAI()->aiAttackObject( victimObject, md->m_clipSize, CMD_FROM_AI );
108
109
110 if( m_laserFromAssisted )
111 makeFeedbackLaser( m_laserFromAssisted, requestingObject, me );
112 if( m_laserToTarget )
113 makeFeedbackLaser( m_laserToTarget, me, victimObject );
114}

118void AssistedTargetingUpdate::makeFeedbackLaser( const ThingTemplate *laserTemplate, const Object *from, const Object *to )
119{
120 if( !getObject()->getControllingPlayer() )
121 return;
122
123 Team *laserTeam = getObject()->getControllingPlayer()->getDefaultTeam();
124 Object *laser = TheThingFactory->newObject( laserTemplate, laserTeam );
125 if( !laser )
126 return;
127
128 // Give it a good basis in reality to ensure it can draw when on screen.
129 laser->setPosition(from->getPosition());
130
131 Drawable *draw = laser->getDrawable();
132 static const NameKeyType key_LaserUpdate = NAMEKEY( "LaserUpdate" );
133 LaserUpdate *update = (LaserUpdate*)draw->findClientUpdateModule( key_LaserUpdate );
134 if( !update )
135 {
136 TheGameLogic->destroyObject( laser );
137 return;
138 }
139
140 update->initLaser( getObject(), to, from->getPosition(), to->getPosition(), "" );
141}
