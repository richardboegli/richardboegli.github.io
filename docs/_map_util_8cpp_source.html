89static UnsignedInt calcCRC( AsciiString dirName, AsciiString fname )
90{
91 CRC theCRC;
92 theCRC.clear();
93
94 // Try the official map dir
95 AsciiString asciiFile;
96 char tempBuf[_MAX_PATH];
97 char filenameBuf[_MAX_PATH];
98 int length = 0;
99 strcpy(tempBuf, fname.str());
100 length = strlen( tempBuf );
101 if( length >= 4 )
102 {
103 memset( filenameBuf, '\0', _MAX_PATH);
104 strncpy( filenameBuf, tempBuf, length - 4);
105 }
106
107 File *fp;
108 asciiFile = fname;
109 fp = TheFileSystem->openFile(asciiFile.str(), File::READ);
110 if( !fp )
111 {
112 DEBUG_CRASH(("Couldn't open '%s'\n", fname.str()));
113 return 0;
114 }
115
116 UnsignedByte buf[4096];
117 Int num;
118 while ( (num=fp->read(buf, 4096)) > 0 )
119 {
120 theCRC.computeCRC(buf, num);
121 }
122
123 fp->close();
124 fp = NULL;
125
126 return theCRC.get();
127}

129static Bool ParseObjectDataChunk(DataChunkInput &file, DataChunkInfo *info, void *userData)
130{
131 Bool readDict = info->version >= K_OBJECTS_VERSION_2;
132
133 Coord3D loc;
134 loc.x = file.readReal();
135 loc.y = file.readReal();
136 loc.z = file.readReal();
137 if (info->version <= K_OBJECTS_VERSION_2)
138 {
139 loc.z = 0;
140 }
141
142 Real angle = file.readReal();
143 Int flags = file.readInt();
144 AsciiString name = file.readAsciiString();
145 Dict d;
146 if (readDict)
147 {
148 d = file.readDict();
149 }
150 MapObject *pThisOne;
151
152 // create the map object
153 pThisOne = newInstance( MapObject )( loc, name, angle, flags, &d,
154 TheThingFactory->findTemplate( name, FALSE ) );
155
156//DEBUG_LOG(("obj %s owner %s\n",name.str(),d.getAsciiString(TheKey_originalOwner).str()));
157
158 if (pThisOne->getProperties()->getType(TheKey_waypointID) == Dict::DICT_INT)
159 {
160 pThisOne->setIsWaypoint();
161
162 // grab useful info
163 (*m_waypoints)[pThisOne->getWaypointName()] = loc;
164 }
165 else if (pThisOne->getThingTemplate() && pThisOne->getThingTemplate()->isKindOf(KINDOF_TECH_BUILDING))
166 {
167 m_techPositions.push_back(loc);
168 }
169 else if (pThisOne->getThingTemplate() && pThisOne->getThingTemplate()->isKindOf(KINDOF_SUPPLY_SOURCE_ON_PREVIEW))
170 {
171 m_supplyPositions.push_back(loc);
172 }
173
174 pThisOne->deleteInstance();
175 return TRUE;
176}

191static Bool ParseSizeOnly(DataChunkInput &file, DataChunkInfo *info, void *userData)
192{
193 m_width = file.readInt();
194 m_height = file.readInt();
195 if (info->version >= K_HEIGHT_MAP_VERSION_3) {
196 m_borderSize = file.readInt();
197 } else {
198 m_borderSize = 0;
199 }
200
201 if (info->version >= K_HEIGHT_MAP_VERSION_4) {
202 Int numBorders = file.readInt();
203 m_boundaries.resize(numBorders);
204 for (int i = 0; i < numBorders; ++i) {
205 m_boundaries[i].x = file.readInt();
206 m_boundaries[i].y = file.readInt();
207 }
208 }
209 return true;
210
211 m_dataSize = file.readInt();
212 m_data = NEW UnsignedByte[m_dataSize]; // pool[]ify
213 if (m_dataSize <= 0 || (m_dataSize != (m_width*m_height))) {
214 throw ERROR_CORRUPT_FILE_FORMAT ;
215 }
216 file.readArrayOfBytes((char *)m_data, m_dataSize);
217 // Resize me.
218 if (info->version == K_HEIGHT_MAP_VERSION_1) {
219 Int newWidth = (m_width+1)/2;
220 Int newHeight = (m_height+1)/2;
221 Int i, j;
222 for (i=0; i<newHeight; i++) {
223 for (j=0; j<newWidth; j++) {
224 m_data[i*newWidth+j] = m_data[2*i*m_width+2*j];
225 }
226 }
227 m_width = newWidth;
228 m_height = newHeight;
229 }
230 return true;
231}

238static Bool loadMap( AsciiString filename )
239{
240 char tempBuf[_MAX_PATH];
241 char filenameBuf[_MAX_PATH];
242 AsciiString asciiFile;
243 int length = 0;
244
245 strcpy(tempBuf, filename.str());
246
247 length = strlen( tempBuf );
248 if( length >= 4 )
249 {
250 memset( filenameBuf, '\0', _MAX_PATH);
251 strncpy( filenameBuf, tempBuf, length - 4);
252 }
253
254 CachedFileInputStream fileStrm;
255
256 asciiFile = filename;
257 if( !fileStrm.open(asciiFile) )
258 {
259 return FALSE;
260 }
261
262 ChunkInputStream *pStrm = &fileStrm;
263
264 DataChunkInput file( pStrm );
265
266 m_waypoints = NEW WaypointMap;
267
268 file.registerParser( AsciiString("HeightMapData"), AsciiString::TheEmptyString, ParseSizeOnlyInChunk );
269 file.registerParser( AsciiString("WorldInfo"), AsciiString::TheEmptyString, ParseWorldDictDataChunk );
270 file.registerParser( AsciiString("ObjectsList"), AsciiString::TheEmptyString, ParseObjectsDataChunk );
271 if (!file.parse(NULL)) {
272 throw(ERROR_CORRUPT_FILE_FORMAT);
273 }
274
275 m_mapDX = m_width - 2*m_borderSize;
276 m_mapDY = m_height - 2*m_borderSize;
277
278 return TRUE;
279}

315void WaypointMap::update( void )
316{
317 if (!m_waypoints)
318 {
319 m_numStartSpots = 1;
320 return;
321 }
322
323 this->clear();
324
325 AsciiString startingCamName = TheNameKeyGenerator->keyToName(TheKey_InitialCameraPosition);
326 WaypointMap::const_iterator it;
327
328 it = m_waypoints->find(startingCamName);
329 if (it != m_waypoints->end())
330 {
331 (*this)[startingCamName] = it->second;
332 }
333
334 m_numStartSpots = 0;
335 for (Int i=0; i<MAX_SLOTS; ++i)
336 {
337 startingCamName.format("Player_%d_Start", i+1); // start pos waypoints are 1-based
338 it = m_waypoints->find(startingCamName);
339 if (it != m_waypoints->end())
340 {
341 (*this)[startingCamName] = it->second;
342 ++m_numStartSpots;
343 }
344 else
345 {
346 break;
347 }
348 }
349
350 m_numStartSpots = max(1, m_numStartSpots);
351}

522Bool MapCache::loadUserMaps()
523{
524 // Read in map list from disk
525 AsciiString mapDir;
526 if (TheGlobalData->m_buildMapCache)
527 {
528 mapDir = getMapDir();
529 }
530 else
531 {
532 mapDir = getUserMapDir();
533
534 INI ini;
535 AsciiString fname;
536 fname.format("%s\\%s", mapDir.str(), m_mapCacheName);
537 File *fp = TheFileSystem->openFile(fname.str(), File::READ);
538 if (fp)
539 {
540 fp->close();
541 ini.load( fname, INI_LOAD_OVERWRITE, NULL );
542 }
543
544 }
545
546 // mark all as unseen
547 m_seen.clear();
548 MapCache::iterator it = begin();
549 while (it != end())
550 {
551 m_seen[it->first] = FALSE;
552 ++it;
553 }
554
555 FilenameList filenameList;
556 FilenameListIter iter;
557 AsciiString toplevelPattern;
558 toplevelPattern.format("%s\\", mapDir.str());
559 Bool parsedAMap = FALSE;
560 AsciiString filenamepattern;
561 filenamepattern.format("*.%s", getMapExtension().str());
562
563 TheFileSystem->getFileListInDirectory(toplevelPattern, filenamepattern, filenameList, TRUE);
564
565 iter = filenameList.begin();
566
567 while (iter != filenameList.end()) {
568 FileInfo fileInfo;
569 AsciiString tempfilename;
570 tempfilename = (*iter);
571 tempfilename.toLower();
572
573 const char *s = tempfilename.reverseFind('\\');
574 if (!s)
575 {
576 DEBUG_CRASH(("Couldn't find \\ in map name!"));
577 }
578 else
579 {
580 AsciiString endingStr;
581 AsciiString fname = s+1;
582 for (Int i=0; i<strlen(mapExtension); ++i)
583 fname.removeLastChar();
584
585 endingStr.format("%s\\%s%s", fname.str(), fname.str(), mapExtension);
586
587 Bool skipMap = FALSE;
588 if (TheGlobalData->m_buildMapCache)
589 {
590 std::set<AsciiString>::const_iterator sit = m_allowedMaps.find(fname);
591 if (m_allowedMaps.size() != 0 && sit == m_allowedMaps.end())
592 {
593 //DEBUG_LOG(("Skipping map: '%s'\n", fname.str()));
594 skipMap = TRUE;
595 }
596 else
597 {
598 //DEBUG_LOG(("Parsing map: '%s'\n", fname.str()));
599 }
600 }
601
602 if (!skipMap)
603 {
604 if (!tempfilename.endsWithNoCase(endingStr.str()))
605 {
606 DEBUG_CRASH(("Found map '%s' in wrong spot (%s)", fname.str(), tempfilename.str()));
607 }
608 else
609 {
610 if (TheFileSystem->getFileInfo(tempfilename, &fileInfo)) {
611 char funk[_MAX_PATH];
612 strcpy(funk, tempfilename.str());
613 char *filenameptr = funk;
614 char *tempchar = funk;
615 while (*tempchar != 0) {
616 if ((*tempchar == '\\') || (*tempchar == '/')) {
617 filenameptr = tempchar+1;
618 }
619 ++tempchar;
620 }
621
622 m_seen[tempfilename] = TRUE;
623 parsedAMap |= addMap(mapDir, *iter, &fileInfo, TheGlobalData->m_buildMapCache);
624 } else {
625 DEBUG_CRASH(("Could not get file info for map %s", (*iter).str()));
626 }
627 }
628 }
629 }
630 iter++;
631 }
632
633 // clean out unseen maps
634 if (clearUnseenMaps(mapDir))
635 return TRUE;
636
637 return parsedAMap;
638}

641Bool MapCache::addMap( AsciiString dirName, AsciiString fname, FileInfo *fileInfo, Bool isOfficial)
642{
643 if (fileInfo == NULL) {
644 return FALSE;
645 }
646
647 AsciiString lowerFname;
648 lowerFname = fname;
649 lowerFname.toLower();
650 MapCache::iterator it = find(lowerFname);
651
652 MapMetaData md;
653 UnsignedInt filesize = fileInfo->sizeLow;
654
655 if (it != end())
656 {
657 // Found the map in our cache. Check to see if it has changed.
658 md = it->second;
659
660 if ((md.m_filesize == filesize) &&
661 (md.m_CRC != 0))
662 {
663 // Force a lookup so that we don't display the English localization in all builds.
664 if (md.m_nameLookupTag.isEmpty())
665 {
666 // unofficial maps or maps without names
667 AsciiString tempdisplayname;
668 tempdisplayname = fname.reverseFind('\\') + 1;
669 (*this)[lowerFname].m_displayName.translate(tempdisplayname);
670 if (md.m_numPlayers >= 2)
671 {
672 UnicodeString extension;
673 extension.format(L" (%d)", md.m_numPlayers);
674 (*this)[lowerFname].m_displayName.concat(extension);
675 }
676 }
677 else
678 {
679 // official maps with name tags
680 (*this)[lowerFname].m_displayName = TheGameText->fetch(md.m_nameLookupTag);
681 if (md.m_numPlayers >= 2)
682 {
683 UnicodeString extension;
684 extension.format(L" (%d)", md.m_numPlayers);
685 (*this)[lowerFname].m_displayName.concat(extension);
686 }
687 }
688// DEBUG_LOG(("MapCache::addMap - found match for map %s\n", lowerFname.str()));
689 return FALSE; // OK, it checks out.
690 }
691 DEBUG_LOG(("%s didn't match file in MapCache\n", fname.str()));
692 DEBUG_LOG(("size: %d / %d\n", filesize, md.m_filesize));
693 DEBUG_LOG(("time1: %d / %d\n", fileInfo->timestampHigh, md.m_timestamp.m_highTimeStamp));
694 DEBUG_LOG(("time2: %d / %d\n", fileInfo->timestampLow, md.m_timestamp.m_lowTimeStamp));
695// DEBUG_LOG(("size: %d / %d\n", filesize, md.m_filesize));
696// DEBUG_LOG(("time1: %d / %d\n", timestamp.m_highTimeStamp, md.m_timestamp.m_highTimeStamp));
697// DEBUG_LOG(("time2: %d / %d\n", timestamp.m_lowTimeStamp, md.m_timestamp.m_lowTimeStamp));
698 }
699
700 DEBUG_LOG(("MapCache::addMap(): caching '%s' because '%s' was not found\n", fname.str(), lowerFname.str()));
701
702 loadMap(fname); // Just load for querying the data, since we aren't playing this map.
703
704 // The map is now loaded. Pick out what we need.
705 md.m_fileName = lowerFname;
706 md.m_filesize = filesize;
707 md.m_isOfficial = isOfficial;
708 md.m_waypoints.update();
709 md.m_numPlayers = md.m_waypoints.m_numStartSpots;
710 md.m_isMultiplayer = (md.m_numPlayers >= 2);
711 md.m_timestamp.m_highTimeStamp = fileInfo->timestampHigh;
712 md.m_timestamp.m_lowTimeStamp = fileInfo->timestampLow;
713 md.m_supplyPositions = m_supplyPositions;
714 md.m_techPositions = m_techPositions;
715 md.m_CRC = calcCRC(dirName, fname);
716
717 Bool exists = false;
718 AsciiString munkee = worldDict.getAsciiString(TheKey_mapName, &exists);
719 md.m_nameLookupTag = munkee;
720 if (!exists || munkee.isEmpty())
721 {
722 DEBUG_LOG(("Missing TheKey_mapName!\n"));
723 AsciiString tempdisplayname;
724 tempdisplayname = fname.reverseFind('\\') + 1;
725 md.m_displayName.translate(tempdisplayname);
726 if (md.m_numPlayers >= 2)
727 {
728 UnicodeString extension;
729 extension.format(L" (%d)", md.m_numPlayers);
730 md.m_displayName.concat(extension);
731 }
732 TheGameText->reset();
733 }
