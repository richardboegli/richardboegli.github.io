78UpdateSleepTime ProneUpdate::update( void )
79{
81 if( m_proneFrames > 0 )
82 {
83 m_proneFrames--;
84 if( m_proneFrames == 0 )
85 stopProneEffects();
86 }
87 return UPDATE_SLEEP_NONE;
88}

91//-------------------------------------------------------------------------------------------------
92void ProneUpdate::goProne( const DamageInfo *damageInfo )
93{
94 //add to the prone time
95 Bool wasProne = (m_proneFrames > 0);
96 Int damageTaken = damageInfo->out.m_actualDamageDealt;
97 m_proneFrames += damageTaken * getProneUpdateModuleData()->m_damageToFramesRatio;
98
99 if( !wasProne && (m_proneFrames > 0) )
100 startProneEffects();
101}
