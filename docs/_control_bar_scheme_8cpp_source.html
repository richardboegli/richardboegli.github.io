216void ControlBarScheme::reset(void)
217{
218 for (Int i = 0; i < MAX_CONTROL_BAR_SCHEME_IMAGE_LAYERS; i++)
219 {
220 ControlBarSchemeImageList::iterator it = m_layer[i].begin();
221
222 while (it != m_layer[i].end())
223 {
224 ControlBarSchemeImage *im = *it;
225 if( im )
226 delete im;
227 im = NULL;
228 it ++;
229 }
230 m_layer[i].clear();
231
232 }
233
234
235 ControlBarSchemeAnimationList::iterator it = m_animations.begin();
236
237 while (it != m_animations.end())
238 {
239 ControlBarSchemeAnimation *anim = *it;
240 if( anim )
241 {
242 anim->m_animImage = NULL;

244 }
245 anim = NULL;
