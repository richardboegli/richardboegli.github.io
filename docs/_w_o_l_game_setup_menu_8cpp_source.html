104void SendStatsToOtherPlayers(const GameInfo *game)
105{
106 PeerRequest req;
107 req.peerRequestType = PeerRequest::PEERREQUEST_UTMPLAYER;
108 req.UTM.isStagingRoom = TRUE;
109 req.id = "STATS/";
110 AsciiString fullStr;
111 PSPlayerStats fullStats = TheGameSpyPSMessageQueue->findPlayerStatsByID(TheGameSpyInfo->getLocalProfileID());
112 PSPlayerStats subStats;
113 subStats.id = fullStats.id;
114 subStats.wins = fullStats.wins;
115 subStats.losses = fullStats.losses;
116 subStats.discons = fullStats.discons;
117 subStats.desyncs = fullStats.desyncs;
118 subStats.games = fullStats.games;
119 subStats.locale = fullStats.locale;
120 subStats.gamesAsRandom = fullStats.gamesAsRandom;
121 GetAdditionalDisconnectsFromUserFile(&subStats);
122 fullStr.format("%d %s", TheGameSpyInfo->getLocalProfileID(), TheGameSpyPSMessageQueue->formatPlayerKVPairs( subStats ));
123 req.options = fullStr.str();
124
125 Int localIndex = game->getLocalSlotNum();
126 for (Int i=0; i<MAX_SLOTS; ++i)
127 {
128 const GameSlot *slot = game->getConstSlot(i);
129 if (slot->isHuman() && i != localIndex)
130 {
131 AsciiString hostName;
132 hostName.translate(slot->getName());
133 req.nick = hostName.str();
134 DEBUG_LOG(("SendStatsToOtherPlayers() - sending to '%s', data of\n\t'%s'\n", hostName.str(), req.options.c_str()));
135 TheGameSpyPeerMessageQueue->addRequest(req);
136 }
137 }
138}

345static void playerTooltip(GameWindow *window,
346 WinInstanceData *instData,
347 UnsignedInt mouse)
348{
349 Int slotIdx = -1;
350 for (Int i=0; i<MAX_SLOTS; ++i)
351 {
352 if (window == comboBoxPlayer[i] || window == staticTextPlayer[i])
353 {
354 slotIdx = i;
355 break;
356 }
357 }
358 if (slotIdx < 0)
359 {
360 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString, -1, NULL, 1.5f );
361 return;
362 }
363
364 GameSpyStagingRoom *game = TheGameSpyInfo->getCurrentStagingRoom();
365 if (!game)
366 {
367 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString, -1, NULL, 1.5f );
368 return;
369 }
370
371 GameSpyGameSlot *slot = game->getGameSpySlot(slotIdx);
372 if (!slot || !slot->isHuman())
373 {
374 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString, -1, NULL, 1.5f );
375 return;
376 }
377
378 // for tooltip, we want:
379 // * player name
380 // * ping
381 // * locale
382 // * win/loss history
383 // * discons/desyncs as one var
384 // * favorite army
385 // in that order. got it? good.
386
387 UnicodeString uName = slot->getName();
388
389 AsciiString aName;
390 aName.translate(uName);
391 PlayerInfoMap::iterator pmIt = TheGameSpyInfo->getPlayerInfoMap()->find(aName);
392 if (pmIt == TheGameSpyInfo->getPlayerInfoMap()->end())
393 {
394 TheMouse->setCursorTooltip( uName, -1, NULL, 1.5f );
395 return;
396 }
397 Int profileID = pmIt->second.m_profileID;
398
399 PSPlayerStats stats = TheGameSpyPSMessageQueue->findPlayerStatsByID(profileID);
400 if (stats.id == 0)
401 {
402 TheMouse->setCursorTooltip( uName, -1, NULL, 1.5f );
403 return;
404 }
405
406 Bool isLocalPlayer = slot == game->getGameSpySlot(game->getLocalSlotNum());
407
408 AsciiString localeIdentifier;
409 localeIdentifier.format("WOL:Locale%2.2d", stats.locale);
410 UnicodeString playerInfo;
411 Int totalWins = 0, totalLosses = 0, totalDiscons = 0;
412 PerGeneralMap::iterator it;
413
414 for (it = stats.wins.begin(); it != stats.wins.end(); ++it)
415 {
416 totalWins += it->second;
417 }
418 for (it = stats.losses.begin(); it != stats.losses.end(); ++it)
419 {
420 totalLosses += it->second;
421 }
422 for (it = stats.discons.begin(); it != stats.discons.end(); ++it)
423 {
424 totalDiscons += it->second;
425 }
426 for (it = stats.desyncs.begin(); it != stats.desyncs.end(); ++it)
427 {
428 totalDiscons += it->second;
429 }
430 UnicodeString favoriteSide;
431 Int numGames = 0;
432 Int favorite = 0;
433 for(it = stats.games.begin(); it != stats.games.end(); ++it)
434 {
435 if(it->second >= numGames)
436 {
437 numGames = it->second;
438 favorite = it->first;
439 }
440 }
441 if(numGames == 0)
442 favoriteSide = TheGameText->fetch("GUI:None");
443 else if( stats.gamesAsRandom >= numGames )
444 favoriteSide = TheGameText->fetch("GUI:Random");
445 else
446 {
447 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(favorite);
448 if (fac)
449 {
450 AsciiString side;
451 side.format("SIDE:%s", fac->getSide().str());
452
453 favoriteSide = TheGameText->fetch(side);
454 }
455 }
456
457 playerInfo.format(TheGameText->fetch("TOOLTIP:StagingPlayerInfo"),
458 TheGameText->fetch(localeIdentifier).str(),
459 slot->getPingAsInt(),
460 totalWins, totalLosses, totalDiscons,
461 favoriteSide.str());
462
463 UnicodeString tooltip = UnicodeString::TheEmptyString;
464 if (isLocalPlayer)
465 {
466 tooltip.format(TheGameText->fetch("TOOLTIP:LocalPlayer"), uName.str());
467 }
468 else
469 {
470 // not us
471 if (TheGameSpyInfo->getBuddyMap()->find(profileID) != TheGameSpyInfo->getBuddyMap()->end())
472 {
473 // buddy
474 tooltip.format(TheGameText->fetch("TOOLTIP:BuddyPlayer"), uName.str());
475 }
476 else
477 {
478 if (profileID)
479 {
480 // non-buddy profiled player
481 tooltip.format(TheGameText->fetch("TOOLTIP:ProfiledPlayer"), uName.str());
482 }
483 else
484 {
485 // non-profiled player
486 tooltip.format(TheGameText->fetch("TOOLTIP:GenericPlayer"), uName.str());
487 }
488 }
489 }
490
491 tooltip.concat(playerInfo);
492
493 TheMouse->setCursorTooltip( tooltip, -1, NULL, 1.5f ); // the text and width are the only params used. the others are the default values.
494}

659static void handleStartPositionSelection(Int player, int startPos)
660{
661 GameSpyStagingRoom *myGame = TheGameSpyInfo->getCurrentStagingRoom();
662
663 if (myGame)
664 {
665 GameSpyGameSlot * slot = myGame->getGameSpySlot(player);
666 if (!slot)
667 return;
668
669 if (startPos == slot->getStartPos())
670 return;
671 Bool skip = FALSE;
672 if (startPos < 0)
673 {
674 skip = TRUE;
675 }
676
677 if(!skip)
678 {
679 Bool isAvailable = TRUE;
680 for(Int i = 0; i < MAX_SLOTS; ++i)
681 {
682 if(i != player && myGame->getSlot(i)->getStartPos() == startPos)
683 {
684 isAvailable = FALSE;
685 break;
686 }
687 }
688 if( !isAvailable )
689 return;
690 }
691 slot->setStartPos(startPos);
692
693 if (myGame->amIHost())
694 {
695 // send around a new slotlist
696 myGame->resetAccepted();
697 TheGameSpyInfo->setGameOptions();
698 WOLDisplaySlotList();
699 }
700 else
701 {
702 // request the color from the host
703 if (AreSlotListUpdatesEnabled())
704 {
705 // request the playerTemplate from the host
706 AsciiString options;
707 options.format("StartPos=%d", slot->getStartPos());
708 AsciiString hostName;
709 hostName.translate(myGame->getSlot(0)->getName());
710 PeerRequest req;
711 req.peerRequestType = PeerRequest::PEERREQUEST_UTMPLAYER;
712 req.UTM.isStagingRoom = TRUE;
713 req.id = "REQ/";
714 req.nick = hostName.str();
715 req.options = options.str();

817static void StartPressed(void)
818{
819 Bool isReady = TRUE;
820 Bool allHaveMap = TRUE;
821 Int playerCount = 0;
822 Int humanCount = 0;
823 GameSpyStagingRoom *myGame = TheGameSpyInfo->getCurrentStagingRoom();
824 if (!myGame)
825 return;
826
827 // see if everyone's accepted and count the number of players in the game
828 UnicodeString mapDisplayName;
829 const MapMetaData *mapData = TheMapCache->findMap( myGame->getMap() );
830 Bool willTransfer = TRUE;
831 if (mapData)
832 {
833 mapDisplayName.format(L"%ls", mapData->m_displayName.str());
834 willTransfer = !mapData->m_isOfficial;
835 }
836 else
837 {
838 mapDisplayName.format(L"%hs", myGame->getMap().str());
839 willTransfer = WouldMapTransfer(myGame->getMap());
840 }
841 for( int i = 0; i < MAX_SLOTS; i++ )
842 {
843 if ((myGame->getSlot(i)->isAccepted() == FALSE) && (myGame->getSlot(i)->isHuman() == TRUE))
844 {
845 isReady = FALSE;
846 if (!myGame->getSlot(i)->hasMap() && !willTransfer)
847 {
848 UnicodeString msg;
849 msg.format(TheGameText->fetch("GUI:PlayerNoMap"), myGame->getSlot(i)->getName().str(), mapDisplayName.str());
850 TheGameSpyInfo->addText(msg, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
851 allHaveMap = FALSE;
852 }
853 }
854 if(myGame->getSlot(i)->isOccupied() && myGame->getSlot(i)->getPlayerTemplate() != PLAYERTEMPLATE_OBSERVER)
855 {
856 if (myGame->getSlot(i)->isHuman())
857 humanCount++;
858 playerCount++;
859 }
860 }
861
862 // Check for too many players
863 const MapMetaData *md = TheMapCache->findMap( myGame->getMap() );
864 if (!md || md->m_numPlayers < playerCount)
865 {
866 if (myGame->amIHost())
867 {
868 UnicodeString text;
869 text.format(TheGameText->fetch("LAN:TooManyPlayers"), (md)?md->m_numPlayers:0);
870 TheGameSpyInfo->addText(text, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
871 }
872 return;
873 }
874
875 // Check for observer + AI players
876 if (TheGlobalData->m_netMinPlayers && !humanCount)
877 {
878 if (myGame->amIHost())
879 {
880 UnicodeString text = TheGameText->fetch("GUI:NeedHumanPlayers");
881 TheGameSpyInfo->addText(text, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
882 }
883 return;
884 }
885
886 // Check for too few players
887 if (playerCount < TheGlobalData->m_netMinPlayers)
888 {
889 if (myGame->amIHost())
890 {
891 UnicodeString text;
892 text.format(TheGameText->fetch("LAN:NeedMorePlayers"),playerCount);
893 TheGameSpyInfo->addText(text, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
894 }
895 return;
896 }
897
898 // Check for too few teams
899 int numRandom = 0;
900 std::set<Int> teams;
901 for (i=0; i<MAX_SLOTS; ++i)
902 {
903 GameSlot *slot = myGame->getSlot(i);
904 if (slot && slot->isOccupied() && slot->getPlayerTemplate() != PLAYERTEMPLATE_OBSERVER)
905 {
906 if (slot->getTeamNumber() >= 0)
907 {
908 teams.insert(slot->getTeamNumber());
909 }
910 else
911 {
912 ++numRandom;
913 }
914 }
915 }
916 if (numRandom + teams.size() < TheGlobalData->m_netMinPlayers)
917 {
918 if (myGame->amIHost())
919 {
920 UnicodeString text;
921 text.format(TheGameText->fetch("LAN:NeedMoreTeams"));
922 TheGameSpyInfo->addText(text, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
923 }
924 return;
925 }
926
927 if (numRandom + teams.size() < 2)
928 {
929 UnicodeString text;
930 text.format(TheGameText->fetch("GUI:SandboxMode"));
931 TheGameSpyInfo->addText(text, GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
932 }
933
934 if(isReady)
935 {
936 PeerRequest req;
937 req.peerRequestType = PeerRequest::PEERREQUEST_STARTGAME;
938 TheGameSpyPeerMessageQueue->addRequest(req);
939
940 SendStatsToOtherPlayers(myGame);
941
942 // we've started, there's no going back
943 // i.e. disable the back button.
944 buttonBack->winEnable(FALSE);
945 GameWindow *buttonBuddy = TheWindowManager->winGetWindowFromId(NULL, NAMEKEY("GameSpyGameOptionsMenu.wnd:ButtonCommunicator"));
946 if (buttonBuddy)
947 buttonBuddy->winEnable(FALSE);
948 GameSpyCloseOverlay(GSOVERLAY_BUDDY);
949
950 *TheGameSpyGame = *myGame;
951 TheGameSpyGame->startGame(0);
952 }
953 else if (allHaveMap)
954 {
955 TheGameSpyInfo->addText(TheGameText->fetch("GUI:NotifiedStartIntent"), GameSpyColor[GSCOLOR_DEFAULT], listboxGameSetupChat);
956 PeerRequest req;
957 req.peerRequestType = PeerRequest::PEERREQUEST_UTMROOM;
958 req.UTM.isStagingRoom = TRUE;
959 req.id = "HWS/";
960 req.options = "true";
961 TheGameSpyPeerMessageQueue->addRequest(req);
962 }

980 if (md && localSlot && localSlot->hasMap())

1023 GadgetCheckBoxSetChecked( checkBoxLimitSuperweapons, limitSuperweapons );

1030 {
1031 // Note: must check if combobox is already correct to avoid infinite recursion

1035 GadgetComboBoxSetSelectedPos(comboBoxStartingCash, index, TRUE);
1036

1038 }
1039 }
1040
1041 DEBUG_ASSERTCRASH( index < itemCount, ("Could not find new starting cash amount %d in list", theGame->getStartingCash().countMoney() ) );
