202/***********************************************************************************************
203 * Matrix3D::Set_Rotation -- Sets the rotation part of the matrix *
204 * *
205 * INPUT: *
206 * *
207 * OUTPUT: *
208 * *
209 * WARNINGS: *
210 * *
211 * HISTORY: *
212 * 5/11/98 GTH : Created. *
213 *=============================================================================================*/
214void Matrix3D::Set_Rotation(const Quaternion & q)
215{
216 Row[0][0] = (float)(1.0 - 2.0 * (q[1] * q[1] + q[2] * q[2]));
217 Row[0][1] = (float)(2.0 * (q[0] * q[1] - q[2] * q[3]));
218 Row[0][2] = (float)(2.0 * (q[2] * q[0] + q[1] * q[3]));
219
220 Row[1][0] = (float)(2.0 * (q[0] * q[1] + q[2] * q[3]));
221 Row[1][1] = (float)(1.0 - 2.0f * (q[2] * q[2] + q[0] * q[0]));
222 Row[1][2] = (float)(2.0 * (q[1] * q[2] - q[0] * q[3]));
223
224 Row[2][0] = (float)(2.0 * (q[2] * q[0] - q[1] * q[3]));
225 Row[2][1] = (float)(2.0 * (q[1] * q[2] + q[0] * q[3]));
226 Row[2][2] =(float)(1.0 - 2.0 * (q[1] * q[1] + q[0] * q[0]));
227}

326/***********************************************************************************************
327 * M3DC::Look_At -- Creates a "look at" transformation. *
328 * *
329 * Builds a transformation matrix which positions the origin at p, *
330 * points the negative z-axis towards a target t, and rolls about the z-axis *
331 * by the angle specified by roll. *
332 * *
333 * This can be useful for creating a camera matrix, just invert *
334 * the matrix after initializing it with this function... *
335 * *
336 * INPUT: *
337 * p - position of the coordinate system *
338 * t - target of the coordinate system *
339 * roll - roll angle (in radians) *
340 * *
341 * OUTPUT: *
342 * *
343 * WARNINGS: *
344 * This function is written assuming the convention that the "ground" is the X-Y plane and *
345 * Z is altitude. *
346 * *
347 * HISTORY: *
348 *=============================================================================================*/
349void Matrix3D::Look_At(const Vector3 &p,const Vector3 &t,float roll)
350{
351 float dx,dy,dz; //vector from p to t
352 float len1,len2;
353 float sinp,cosp; //sine and cosine of the pitch ("up-down" tilt about x)
354 float siny,cosy; //sine and cosine of the yaw ("left-right"tilt about z)
355
356 dx = (t[0] - p[0]);
357 dy = (t[1] - p[1]);
358 dz = (t[2] - p[2]);
359
360 len1 = (float)WWMath::Sqrt(dx*dx + dy*dy + dz*dz);
361 len2 = (float)WWMath::Sqrt(dx*dx + dy*dy);
362
363 if (len1 != 0.0f) {
364 sinp = dz/len1;
365 cosp = len2/len1;
366 } else {
367 sinp = 0.0f;
368 cosp = 1.0f;
369 }
370
371 if (len2 != 0.0f) {
372 siny = dy/len2;
373 cosy = dx/len2;
374 } else {
375 siny = 0.0f;
376 cosy = 1.0f;
377 }
378
379 // init the matrix with position p and -z pointing down +x and +y up
380 Row[0].X = 0.0f; Row[0].Y = 0.0f; Row[0].Z = -1.0f;
381 Row[1].X = -1.0f; Row[1].Y = 0.0f; Row[1].Z = 0.0f;
382 Row[2].X = 0.0f; Row[2].Y = 1.0f; Row[2].Z = 0.0f;
383
384 Row[0].W = p.X;
385 Row[1].W = p.Y;
386 Row[2].W = p.Z;
387
388 // Yaw rotation to make the matrix look at the projection of the target
389 // into the x-y plane
390 Rotate_Y(siny,cosy);
391
392 // rotate about local x axis to pitch up to the targets position
393 Rotate_X(sinp,cosp);
394
395 // roll about the local z axis (negate since we look down -z)
396 Rotate_Z(-roll);
397}

400/***********************************************************************************************
401 * M3DC::Obj_Look_At -- Commando Object "look at" transformation. *
402 * *
403 * Builds a transformation matrix which positions the origin at p, *
404 * points the positive X axis towards a target t, and rolls about the X axis *
405 * by the angle specified by roll. *
406 * *
407 * The object convention used by Commando and G is Forward = +X, Left = +Y, Up = +Z. The *
408 * world is basically the x-y plane with z as altitude and +x is the default "forward". *
409 * *
410 * INPUT: *
411 * p - position of the coordinate system *
412 * t - target of the coordinate system *
413 * roll - roll angle (in radians) *
414 * *
415 * OUTPUT: *
416 * *
417 * WARNINGS: *
418 * *
419 * HISTORY: *
420 *=============================================================================================*/
421void Matrix3D::Obj_Look_At(const Vector3 &p,const Vector3 &t,float roll)
422{
423 float dx,dy,dz; //vector from p to t
424 float len1,len2;
425 float sinp,cosp; //sine and cosine of the pitch ("up-down" tilt about y)
426 float siny,cosy; //sine and cosine of the yaw ("left-right"tilt about z)
427
428 dx = (t[0] - p[0]);
429 dy = (t[1] - p[1]);
430 dz = (t[2] - p[2]);
431
432 len1 = (float)sqrt(dx*dx + dy*dy + dz*dz);
433 len2 = (float)sqrt(dx*dx + dy*dy);
434
435 if (len1 != 0.0f) {
436 sinp = dz/len1;
437 cosp = len2/len1;
438 } else {
439 sinp = 0.0f;
440 cosp = 1.0f;
441 }
442
443 if (len2 != 0.0f) {
444 siny = dy/len2;
445 cosy = dx/len2;
446 } else {
447 siny = 0.0f;
448 cosy = 1.0f;
449 }
450
451 Make_Identity();
452 Translate(p);
453
454 // Yaw rotation to projection of target in x-y plane
455 Rotate_Z(siny,cosy);
456
457 // Pitch rotation
458 Rotate_Y(-sinp,cosp);
459
460 // Roll rotation
461 Rotate_X(roll);
462}

483/***********************************************************************************************
484 * Matrix3D::Get_Orthogonal_Inverse -- Returns the inverse of the matrix *
485 * *
486 * NOTE!!! This only works if the matrix is really ORTHOGONAL!!! *
487 * *
488 ***********************************************************************************************
489 * Inverting an orthogonal Matrix3D *
490 * *
491 * M is the original transform, *
492 * R is rotation submatrix, *
493 * T is translation vector in M. *
494 * *
495 * To build MINV *
496 * *
497 * R' = transpose of R (inverse of orthogonal 3x3 matrix is transpose) *
498 * T' = -R'T *
499 * *
500 * Build MINV with R'and T' *
501 * MINV is the inverse of M *
502 * *
503 ***********************************************************************************************
504 * INPUT: *
505 * *
506 * OUTPUT: *
507 * *
508 * WARNINGS: *
509 * *
510 * HISTORY: *
511 *=============================================================================================*/
512void Matrix3D::Get_Orthogonal_Inverse(Matrix3D & inv) const
513{
514 // Transposing the rotation submatrix
515 inv.Row[0][0] = Row[0][0];
516 inv.Row[0][1] = Row[1][0];
517 inv.Row[0][2] = Row[2][0];
518
519 inv.Row[1][0] = Row[0][1];
520 inv.Row[1][1] = Row[1][1];
521 inv.Row[1][2] = Row[2][1];
522
523 inv.Row[2][0] = Row[0][2];
524 inv.Row[2][1] = Row[1][2];
525 inv.Row[2][2] = Row[2][2];
526
527 // Now, calculate translation portion of matrix:
528 // T' = -R'T
529 Vector3 trans = Get_Translation();
530 trans = inv.Rotate_Vector(trans);
531 trans = -trans;
532
533 inv.Row[0][3] = trans[0];
534 inv.Row[1][3] = trans[1];
535 inv.Row[2][3] = trans[2];
536}

568/***********************************************************************************************
569 * Matrix3D::Multiply -- matrix multiplication without temporaries. *
570 * *
571 * INPUT: *
572 * *
573 * OUTPUT: *
574 * *
575 * WARNINGS: *
576 * *
577 * HISTORY: *
578 * 4/22/98 GTH : Created. *
579 *=============================================================================================*/
580//void print_matrix(const Matrix3D & m);
581void Matrix3D::Multiply(const Matrix3D & A,const Matrix3D & B,Matrix3D * set_res)
582{
583 assert(set_res != NULL);
584
585 Matrix3D tmp;
586 Matrix3D * Aptr;
587 float tmp1,tmp2,tmp3;
588
589 // Check for aliased parameters, copy the 'A' matrix into a temporary if the
590 // result is going into 'A'. (in this case, this function is no better than
591 // the overloaded C++ operator...)
592 if (set_res == &A) {
593 tmp = A;
594 Aptr = &tmp;
595 } else {
596 Aptr = (Matrix3D *)&A;
597 }
598
599 tmp1 = B[0][0];
600 tmp2 = B[1][0];
601 tmp3 = B[2][0];
602
603 (*set_res)[0][0] = (*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3;
604 (*set_res)[1][0] = (*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3;
605 (*set_res)[2][0] = (*Aptr)[2][0]*tmp1 + (*Aptr)[2][1]*tmp2 + (*Aptr)[2][2]*tmp3;
606
607 tmp1 = B[0][1];
608 tmp2 = B[1][1];
609 tmp3 = B[2][1];
610
611 (*set_res)[0][1] = (*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3;
612 (*set_res)[1][1] = (*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3;
613 (*set_res)[2][1] = (*Aptr)[2][0]*tmp1 + (*Aptr)[2][1]*tmp2 + (*Aptr)[2][2]*tmp3;
614
615 tmp1 = B[0][2];
616 tmp2 = B[1][2];
617 tmp3 = B[2][2];
618
619 (*set_res)[0][2] = (*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3;
620 (*set_res)[1][2] = (*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3;
621 (*set_res)[2][2] = (*Aptr)[2][0]*tmp1 + (*Aptr)[2][1]*tmp2 + (*Aptr)[2][2]*tmp3;
622
623 tmp1 = B[0][3];
624 tmp2 = B[1][3];
625 tmp3 = B[2][3];
626
627 (*set_res)[0][3] = (*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3 + (*Aptr)[0][3];
628 (*set_res)[1][3] = (*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3 + (*Aptr)[1][3];
629 (*set_res)[2][3] = (*Aptr)[2][0]*tmp1 + (*Aptr)[2][1]*tmp2 + (*Aptr)[2][2]*tmp3 + (*Aptr)[2][3];
630
631}

634/***********************************************************************************************
635 * Matrix3D::Transform_Min_Max_AABox -- compute transformed axis-aligned box *
636 * *
637 * INPUT: *
638 * *
639 * OUTPUT: *
640 * *
641 * WARNINGS: *
642 * *
643 * HISTORY: *
644 * 7/17/98 GTH : Created. *
645 *=============================================================================================*/
646void Matrix3D::Transform_Min_Max_AABox
647(
648 const Vector3 & min,
649 const Vector3 & max,
650 Vector3 * set_min,
651 Vector3 * set_max
652) const
653{
654 assert(set_min != &min);
655 assert(set_max != &max);
656
657 float tmp0,tmp1;
658
659 // init the min and max to the translation of the transform
660 set_min->X = set_max->X = Row[0][3];
661 set_min->Y = set_max->Y = Row[1][3];
662 set_min->Z = set_max->Z = Row[2][3];
663
664 // now push them both out by the projections of the original intervals
665 for (int i=0; i<3; i++) {
666
667 for (int j=0; j<3; j++) {
668
669 tmp0 = Row[i][j] * min[j];
670 tmp1 = Row[i][j] * max[j];
671
672 if (tmp0 < tmp1) {
673

760 * *
761 * INPUT: *
762 * *
763 * OUTPUT: *
764 * *

768 * *
769 * HISTORY: *
770 * 9/16/98 GTH : Created. *
771 *=============================================================================================*/
772void Matrix3D::Re_Orthogonalize(void)
773{
774 Vector3 x(Row[0][0],Row[0][1],Row[0][2]);
775 Vector3 y(Row[1][0],Row[1][1],Row[1][2]);
776 Vector3 z;
777
778 Vector3::Cross_Product(x,y,&z);

780

782 if (len < WWMATH_EPSILON) {
783 Make_Identity();
784 return;
785 } else {
786 x *= 1.0f/len;
787 }
788
789 len = y.Length();
790 if (len < WWMATH_EPSILON) {
791 Make_Identity();
792 return;
793 } else {
794 y *= 1.0f/len;
795 }
796
797 len = z.Length();
798 if (len < WWMATH_EPSILON) {
799 Make_Identity();
800 return;
801 } else {
802 z *= 1.0f/len;
803 }
804
805 Row[0][0] = x.X;
806 Row[0][1] = x.Y;
807 Row[0][2] = x.Z;
808
809 Row[1][0] = y.X;
810 Row[1][1] = y.Y;
811 Row[1][2] = y.Z;
812
813 Row[2][0] = z.X;
814 Row[2][1] = z.Y;
815 Row[2][2] = z.Z;
