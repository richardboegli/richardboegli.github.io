85static void parseFX( INI* ini, void *instance, void * /*store*/, const void* /*userData*/ )
86{
87 StructureCollapseUpdateModuleData* self = (StructureCollapseUpdateModuleData*)instance;
88 StructureCollapsePhaseType scphase = (StructureCollapsePhaseType)INI::scanIndexList(ini->getNextToken(), TheStructureCollapsePhaseNames);

132void StructureCollapseUpdate::beginStructureCollapse(const DamageInfo *damageInfo)
133{
134 const StructureCollapseUpdateModuleData *d = getStructureCollapseUpdateModuleData();
135
136
137 Object *building = getObject();
138 UnsignedInt now = TheGameLogic->getFrame();
139 // This has to use a game logic random value since the bursts can spawn debris, and debris is sync'd.
140 m_collapseFrame = now + GameLogicRandomValue(d->m_minCollapseDelay, d->m_maxCollapseDelay);
141
142 doPhaseStuff(SCPHASE_INITIAL, building->getPosition());
143
144 m_collapseState = COLLAPSESTATE_WAITINGFORCOLLAPSESTART;
145 m_currentHeight = 0.0f;
146
147 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
148}

152void StructureCollapseUpdate::onDie( const DamageInfo *damageInfo )
153{
154 const StructureCollapseUpdateModuleData* d = getStructureCollapseUpdateModuleData();
155 if (!d->m_dieMuxData.isDieApplicable(getObject(), damageInfo))
156 return;
157
158 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
159 if (ai)
160 ai->markAsDead();
161
162 // deselect this object for all players.
163 TheGameLogic->deselectObject(getObject(), PLAYERMASK_ALL, TRUE);
164
165 beginStructureCollapse(damageInfo);
166}

170UpdateSleepTime StructureCollapseUpdate::update( void )
171{
172 static const Real COLLAPSE_ACCELERATION_FACTOR = 0.02f;
173 const StructureCollapseUpdateModuleData *d = getStructureCollapseUpdateModuleData();
174
175 if (m_collapseState == COLLAPSESTATE_STANDING)
176 {
177 DEBUG_CRASH(("hmm, what?"));
178 return UPDATE_SLEEP_FOREVER;
179 }
180
181 // We are in the dramatic pause between when the building has lost all its hit points and
182 // when it starts toppling over.
183 if (m_collapseState == COLLAPSESTATE_WAITINGFORCOLLAPSESTART)
184 {
185 UnsignedInt now = TheGameLogic->getFrame();
186 Object *building = getObject();
187
188 const Coord3D *currentPosition = building->getPosition();
189 Vector3 shudder;
190 shudder.Set(GameClientRandomValueReal(-(d->m_maxShudder), d->m_maxShudder), GameClientRandomValueReal(-(d->m_maxShudder), d->m_maxShudder), 0);
191
192 const Matrix3D *instMatrix = building->getDrawable()->getInstanceMatrix();
193 Matrix3D newInstMatrix;
194 newInstMatrix = *instMatrix;
195 newInstMatrix.Set_Translation(shudder);
196
197 building->getDrawable()->setInstanceMatrix(&newInstMatrix);
198
199 if (now >= m_collapseFrame)
200 {
201 m_collapseState = COLLAPSESTATE_COLLAPSING;
202 doPhaseStuff(SCPHASE_BURST, currentPosition);
203 // This has to use a game logic random value since the bursts can spawn debris, and debris is sync'd.
204 m_burstFrame = now + GameLogicRandomValue(d->m_minBurstDelay, d->m_maxBurstDelay);
205 }
206 }
207
208 // The building is in the process of falling over.
209 if (m_collapseState == COLLAPSESTATE_COLLAPSING)
210 {
211 Object *building = getObject();
212 UnsignedInt now = TheGameLogic->getFrame();
213 m_currentHeight -= m_collapseVelocity;
214 m_collapseVelocity -= TheGlobalData->m_gravity * (1.0 - d->m_collapseDamping);
215
216 const Coord3D *currentPosition = building->getPosition();
217 Vector3 shudder;
218 shudder.Set(GameClientRandomValueReal(-(d->m_maxShudder), d->m_maxShudder), GameClientRandomValueReal(-(d->m_maxShudder), d->m_maxShudder), m_currentHeight);
219 const Matrix3D *instMatrix = building->getDrawable()->getInstanceMatrix();
220 Matrix3D newInstMatrix;
221 newInstMatrix = *instMatrix;
222 newInstMatrix.Set_Translation(shudder);
223
224 building->getDrawable()->setInstanceMatrix(&newInstMatrix);
225
226 if (now >= m_burstFrame)
227 {
228 if (GameLogicRandomValue(1, d->m_bigBurstFrequency) == 1)
229 {
230 doPhaseStuff(SCPHASE_BURST, currentPosition);
231 }
232 else
233 {
234 doPhaseStuff(SCPHASE_DELAY, currentPosition);
235 }
236 // This has to use a game logic random value since the bursts can spawn debris, and debris is sync'd.
237 m_burstFrame += GameLogicRandomValue(d->m_minBurstDelay, d->m_maxBurstDelay);
238 }
239
240// if ((m_currentHeight + building->getGeometryInfo().getMaxHeightAbovePosition()) <= 0)
241 if ((m_currentHeight + building->getTemplate()->getTemplateGeometryInfo().getMaxHeightAbovePosition()) <= 0)
242 {
243 m_collapseState = COLLAPSESTATE_DONE;
244 doPhaseStuff(SCPHASE_FINAL, building->getPosition());
245 Drawable *drawable = building->getDrawable();
246
247 doCollapseDoneStuff();
248
249 drawable->clearModelConditionState(MODELCONDITION_RUBBLE);
250 drawable->setModelConditionState(MODELCONDITION_POST_COLLAPSE);
251 building->setOrientation(building->getOrientation());
252
253
254 // Need to update body particle systems, now
255 BodyModuleInterface *body = building->getBodyModule();
256 body->updateBodyParticleSystems();
257
258
259 Vector3 shudder;
260 shudder.Set(0, 0, 0);
261 const Matrix3D *instMatrix = building->getDrawable()->getInstanceMatrix();
262 Matrix3D newInstMatrix;
263 newInstMatrix = *instMatrix;
264 newInstMatrix.Set_Translation(shudder);
265 building->getDrawable()->setInstanceMatrix(&newInstMatrix);
266
267 return UPDATE_SLEEP_FOREVER;
268 }
269 }
270
271 return UPDATE_SLEEP_NONE;
272}

289{

304void StructureCollapseUpdate::doPhaseStuff(StructureCollapsePhaseType scphase, const Coord3D *target)
305{
306 DEBUG_LOG(("Firing phase %d on frame %d\n", scphase, TheGameLogic->getFrame()));
307
308 const StructureCollapseUpdateModuleData* d = getStructureCollapseUpdateModuleData();
309 Int i, idx, count, listSize;
310 Int idxList[MAX_IDX];
311
312 listSize = d->m_fxs[scphase].size();
313 if (listSize > 0)
314 {
315 count = d->m_fxCount[scphase];
316 buildNonDupRandomIndexList(listSize, count, idxList);
317 for (i = 0; i < count; ++i)
318 {
319 idx = idxList[i];
320 const FXVec& v = d->m_fxs[scphase];
321 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
322 const FXList* fxl = v[idx];
323 FXList::doFXPos(fxl, target);
324 }
325 }
326
327 listSize = d->m_ocls[scphase].size();
328 if (listSize > 0)
329 {
330 count = d->m_oclCount[scphase];
331 buildNonDupRandomIndexList(listSize, count, idxList);
332 for (i = 0; i < count; ++i)
333 {
334 idx = idxList[i];
335 const OCLVec& v = d->m_ocls[scphase];
336 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
337 const ObjectCreationList* ocl = v[idx];
338 ObjectCreationList::create(ocl, getObject(), target, NULL, getObject()->getOrientation() );
339 }
340 }
