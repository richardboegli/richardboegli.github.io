218//
219// Load

222bool
223LogicalSoundClass::Load (ChunkLoadClass &cload)
224{
225 while (cload.Open_Chunk ()) {
226 switch (cload.Cur_Chunk_ID ()) {
227
228 case CHUNKID_BASE_CLASS:
229 PersistClass::Load (cload);
230 break;
231
232 case CHUNKID_VARIABLES:
233 {
234 //
235 // Read all the variables from their micro-chunks
236 //
237 while (cload.Open_Micro_Chunk ()) {
238 switch (cload.Cur_Micro_Chunk_ID ()) {
239
240 READ_MICRO_CHUNK (cload, VARID_DROP_OFF_RADIUS, m_DropOffRadius);
241 READ_MICRO_CHUNK (cload, VARID_IS_SINGLE_SHOT, m_IsSingleShot);
242 READ_MICRO_CHUNK (cload, VARID_TYPE_MASK, m_TypeMask);
243 READ_MICRO_CHUNK (cload, VARID_POSITION, m_Position);
244 READ_MICRO_CHUNK (cload, VARID_MAX_LISTENERS, m_MaxListeners);
245 READ_MICRO_CHUNK (cload, VARID_NOTIFY_DELAY, m_NotifyDelayInMS);
246 READ_MICRO_CHUNK (cload, VARID_LAST_NOTIFY, m_LastNotification);
247 }
248
249 cload.Close_Micro_Chunk ();
250 }
251 }
252 break;
253 }
254
255 cload.Close_Chunk ();
256 }
257
258 return true;
259}
