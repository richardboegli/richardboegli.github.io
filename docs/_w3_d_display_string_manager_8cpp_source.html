74void W3DDisplayStringManager::postProcessLoad( void )
75{
76 // Get the font.
77 GameFont *font = TheFontLibrary->getFont(
78 TheDrawGroupInfo->m_fontName,
79 TheDrawGroupInfo->m_fontSize,
80 TheDrawGroupInfo->m_fontIsBold );
81
82 for (Int i = 0; i < MAX_GROUPS; ++i)
83 {
84 m_groupNumeralStrings[i] = newDisplayString();
85 m_groupNumeralStrings[i]->setFont(font);
86
87#ifdef KRIS_BRUTAL_HACK_FOR_AIRCRAFT_CARRIER_DEBUGGING
88 UnicodeString displayNumber;
89 displayNumber.format( L"%d", i);
90 m_groupNumeralStrings[i]->setText( displayNumber );
91#else
92 AsciiString displayNumber;
93 displayNumber.format("NUMBER:%d", i);
94 m_groupNumeralStrings[i]->setText(TheGameText->fetch(displayNumber));
95#endif
96 }
97
98 m_formationLetterDisplayString = newDisplayString();
99 m_formationLetterDisplayString->setFont(font);
100 AsciiString displayLetter;
101 displayLetter.format("LABEL:FORMATION");
102 m_formationLetterDisplayString->setText(TheGameText->fetch(displayLetter));
103
104
105}

111DisplayString *W3DDisplayStringManager::newDisplayString( void )
112{
113 DisplayString *newString = newInstance(W3DDisplayString);
114
115 // sanity
116 if( newString == NULL )
117 {
118
119 DEBUG_LOG(( "newDisplayString: Could not allcoate new W3D display string\n" ));
120 assert( 0 );
121 return NULL;
122
123 } // end if
124
125 // assign a default font
126 if (TheGlobalLanguageData && TheGlobalLanguageData->m_defaultDisplayStringFont.name.isNotEmpty())
127 {
128 newString->setFont(TheFontLibrary->getFont(
129 TheGlobalLanguageData->m_defaultDisplayStringFont.name,
130 TheGlobalLanguageData->m_defaultDisplayStringFont.size,
131 TheGlobalLanguageData->m_defaultDisplayStringFont.bold) );
132 }
133 else
134 newString->setFont( TheFontLibrary->getFont( AsciiString("Times New Roman"), 12, FALSE ) );
135
136 // link string to list
137 link( newString );
138
139 // return our new string
140 return newString;
141
142} // end newDisplayString

147void W3DDisplayStringManager::freeDisplayString( DisplayString *string )
148{
149
150 // sanity
151 if( string == NULL )
152 return;
153
154 // unlink
155 unLink( string );
156
157 // if the string happens to fall where our current checkpoint was, set the checkpoint to null
158 if ( m_currentCheckpoint == string) {
159 m_currentCheckpoint = NULL;
160 }
161
162 // free data
163 string->deleteInstance();
164
165} // end freeDisplayString

173//-------------------------------------------------------------------------------------------------
174void W3DDisplayStringManager::update( void )
175{
176 // call base in case we add something later
177 DisplayStringManager::update();
178
179 W3DDisplayString *string = static_cast<W3DDisplayString *>(m_stringList);
180
181 // if the m_currentCheckpoint is valid, use it for the starting point for the search
182 if (m_currentCheckpoint) {
183 string = static_cast<W3DDisplayString *>(m_currentCheckpoint);
184 }
185
186 UnsignedInt currFrame = TheGameClient->getFrame();
187 const UnsignedInt w3dCleanupTime = 60;
190
191 int numStrings = 10;
192 // looping through all the strings eats up a lot of ambient time. Instead,
193 // loop through 10 (arbitrarily chosen) or till the end is hit.
194 while ( numStrings-- && string)
195 {
196
197 //
198 // has this string "expired" in terms of using resources, a string
199 // with a resource frame of zero isn't using any resources at all
200 //
201 if( string->m_lastResourceFrame != 0 &&
202 currFrame - string->m_lastResourceFrame > w3dCleanupTime )
203 {
204
205 // free the resources
206 string->m_textRenderer.Reset();
207 string->m_textRendererHotKey.Reset();
208 //
209 // mark data in the string as changed so that if it needs to
210 // be drawn again it will know to reconstruct the render data
211 //
212 string->m_textChanged = TRUE;
213
214 //
215 // set the last resource frame to zero, this allows us to ignore it
216 // in future cleanup passes of this update routine
217 //
218 string->m_lastResourceFrame = 0;
219
220 } // end if
221
222 // move to next string
223 string = static_cast<W3DDisplayString *>(string->next());
224
225 } // end while
226
227 // reset the starting point for our next search
228 m_currentCheckpoint = string;
229} // end update
