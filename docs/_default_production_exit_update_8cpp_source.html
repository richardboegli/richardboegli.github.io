65void DefaultProductionExitUpdate::exitObjectViaDoor( Object *newObj, ExitDoorType exitDoor )
66{
67 DEBUG_ASSERTCRASH(exitDoor == DOOR_1, ("multiple exit doors not supported here"));
68
69 Object *creationObject = getObject();
70 if (creationObject)
71 {
72 const DefaultProductionExitUpdateModuleData* md = getDefaultProductionExitUpdateModuleData();
73
74 Real exitAngle = creationObject->getOrientation();
75 const Matrix3D *transform = creationObject->getTransformMatrix();
76 Vector3 loc;
77 Coord3D createPoint;
78
79 //
80 // calculate the position to create the object at, we take the coord specified
81 // in INI which is in model space, rotate it to match the building angle
82 // and translate for building location via a transform call
83 //
84 loc.Set( md->m_unitCreatePoint.x, md->m_unitCreatePoint.y, md->m_unitCreatePoint.z );
85 transform->Transform_Vector( *transform, loc, &loc );
86
87 // make sure the point is on the terrain
88 loc.Z = TheTerrainLogic ? TheTerrainLogic->getLayerHeight( loc.X, loc.Y, creationObject->getLayer() ) : 0.0f;
89
90 // we need it in Coord3D form
91 createPoint.x = loc.X;
92 createPoint.y = loc.Y;
93 createPoint.z = loc.Z;
94
95 newObj->setPosition( &createPoint );
96 newObj->setOrientation( exitAngle );
97 newObj->setLayer( creationObject->getLayer() );
98
101 // tell the AI about it
102 TheAI->pathfinder()->addObjectToPathfindMap( newObj );
103 Coord3D tmp;
104 getNaturalRallyPoint(tmp);
105 std::vector<Coord3D> exitPath;
106 exitPath.push_back(tmp);
107
108 AIUpdateInterface *ai = newObj->getAIUpdateInterface();
109 if (m_rallyPointExists)
110 {
111 tmp = m_rallyPoint;
112 if (ai && ai->isDoingGroundMovement())
113 {
114 if (TheAI->pathfinder()->adjustDestination(newObj, ai->getLocomotorSet(), &tmp))
115 exitPath.push_back(tmp);
116
117 }
118 }
119 if (ai) {
120 ai->aiFollowExitProductionPath( &exitPath, creationObject, CMD_FROM_AI );
121 }
122 }
123
124}

128Bool DefaultProductionExitUpdate::getExitPosition( Coord3D& exitPosition ) const
129{
130 const Object *obj = getObject();
131 if (!obj)
132 return FALSE;
133
134 const Matrix3D *transform = obj->getTransformMatrix();
135
136 const DefaultProductionExitUpdateModuleData *md = getDefaultProductionExitUpdateModuleData();
137
138 Vector3 loc;
139 loc.Set( md->m_unitCreatePoint.x, md->m_unitCreatePoint.y, md->m_unitCreatePoint.z );
140 transform->Transform_Vector( *transform, loc, &loc );
141
142 exitPosition.x = loc.X;
143 exitPosition.y = loc.Y;
144 exitPosition.z = loc.Z;
145
146 return TRUE;
147
148}

151Bool DefaultProductionExitUpdate::getNaturalRallyPoint( Coord3D& rallyPoint, Bool offset ) const
152{
153 const DefaultProductionExitUpdateModuleData *data = getDefaultProductionExitUpdateModuleData();
154 Vector3 p;
155
156 //
157 // get the natural rally point from the INI definition, this coord is in model space relative
158 // to the model (0,0,0)
159 //
160 p.X = data->m_naturalRallyPoint.x;
161 p.Y = data->m_naturalRallyPoint.y;
162 p.Z = data->m_naturalRallyPoint.z;
163
164 if ( offset )
165 {
166 Vector3 offset = p;
167 offset.Normalize();
168 offset *= (2*PATHFIND_CELL_SIZE_F);
169 p+=offset;
170 }
171
172 // transform the point into world space
173 const Matrix3D *transform = getObject()->getTransformMatrix();
174 transform->Transform_Vector( *transform, p, &p );
175
176 rallyPoint.x = p.X; rallyPoint.y = p.Y; rallyPoint.z = p.Z;
177 return TRUE;
178}
