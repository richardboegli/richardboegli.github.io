82{
83 for (Int i = 0; i < m_prereqUnits.size(); i++)
84 {
85
86 //
87 // note that this will find the template at the "top most" level (not override
88 // sub-temlates), which is what we want ... we conceptually only have one
89 // template for any given thing, it's only the *data* that is overridden
90 //
91 if( m_prereqUnits[ i ].name.isNotEmpty() )
92 {
93 m_prereqUnits[i].unit = TheThingFactory->findTemplate(m_prereqUnits[i].name); // might be null

98 DEBUG_ASSERTCRASH(m_prereqUnits[i].unit,("could not find prereq %s\n",m_prereqUnits[i].name.str()));
99
100 m_prereqUnits[i].name.clear(); // we're done with it
101 }

108Int ProductionPrerequisite::calcNumPrereqUnitsOwned(const Player *player, Int counts[MAX_PREREQ]) const
109{
110 const ThingTemplate *tmpls[MAX_PREREQ];
111 Int cnt = m_prereqUnits.size();
112 if (cnt > MAX_PREREQ)
113 cnt = MAX_PREREQ;
114 for (int i = 0; i < cnt; i++)
115 tmpls[i] = m_prereqUnits[i].unit;
116 player->countObjectsByThingTemplate(cnt, tmpls, false, counts);
117 return cnt;

120//-----------------------------------------------------------------------------
121Int ProductionPrerequisite::getAllPossibleBuildFacilityTemplates(const ThingTemplate* tmpls[], Int maxtmpls) const
122{
123 Int count = 0;
124 for (int i = 0; i < m_prereqUnits.size(); i++)
125 {
126 if (i > 0 && !(m_prereqUnits[i].flags & UNIT_OR_WITH_PREV))
127 break;
128 if (count >= maxtmpls)
129 break;
130 tmpls[count++] = m_prereqUnits[i].unit;
131 }
132 return count;
133}

136const ThingTemplate *ProductionPrerequisite::getExistingBuildFacilityTemplate( const Player *player ) const
137{
138 DEBUG_ASSERTCRASH(player, ("player may not be null"));
139 if (m_prereqUnits.size())
140 {
141 Int ownCount[MAX_PREREQ];
142 Int cnt = calcNumPrereqUnitsOwned(player, ownCount);
143 for (int i = 0; i < cnt; i++)
144 {
145 if (i > 0 && !(m_prereqUnits[i].flags & UNIT_OR_WITH_PREV))
146 break;
147 if (ownCount[i])
148 return m_prereqUnits[i].unit;
149 }
150 }
151 return NULL;
152}

154//-----------------------------------------------------------------------------
155Bool ProductionPrerequisite::isSatisfied(const Player *player) const
156{
157 Int i;
158
159 if (!player)
160 return false;
161
162 // gotta have all the prereq sciences.
163 for (i = 0; i < m_prereqSciences.size(); i++)
164 {
165 if (!player->hasScience(m_prereqSciences[i]))
166 return false;
167 }
168
169 // the player must have at least one instance of each prereq unit.
170 Int ownCount[MAX_PREREQ];
171 Int cnt = calcNumPrereqUnitsOwned(player, ownCount);
172
173 // fix up the "or" cases. (start at 1!)
174 for (i = 1; i < cnt; i++)
175 {
176 if (m_prereqUnits[i].flags & UNIT_OR_WITH_PREV)
177 {
178 ownCount[i] += ownCount[i-1]; // lump 'em together for prereq purposes
179 ownCount[i-1] = -1; // flag for "ignore me"
180 }
181 }
182
183 for (i = 0; i < cnt; i++)
184 {
185 if (ownCount[i] == -1) // the magic "ignore me" flag
186 continue;
187 if (ownCount[i] == 0) // everything not ignored, is required
188 return false;
189 }
190
191 return true;
192}

230UnicodeString ProductionPrerequisite::getRequiresList(const Player *player) const
231{
232
233 // if player is invalid, return empty string
234 if (!player)
235 return UnicodeString::TheEmptyString;
236
237 UnicodeString requiresList = UnicodeString::TheEmptyString;
238
239 // check the prerequired units
240 Int ownCount[MAX_PREREQ];
241 Int cnt = calcNumPrereqUnitsOwned(player, ownCount);
242 Int i;
243
244 Bool orRequirements[MAX_PREREQ];
245 //Added for fix below in getRequiresList
246 //By Sadullah Nader
247 //Initializes the OR_WITH_PREV structures
248 for (i = 0; i < MAX_PREREQ; i++)
249 {
250 orRequirements[i] = FALSE;
251 }
252 //
253 // account for the "or" unit cases, start for loop at 1
254 for (i = 1; i < cnt; i++)
255 {
256 if (m_prereqUnits[i].flags & UNIT_OR_WITH_PREV)
257 {
258 orRequirements[i] = TRUE; // set the flag for this unit to be "ored" with previous
259 ownCount[i] += ownCount[i-1]; // lump 'em together for prereq purposes
260 ownCount[i-1] = -1; // flag for "ignore me"
261 }
262 }
263
264 // check to see if anything is required
265 const ThingTemplate *unit;
266 UnicodeString unitName;
267 Bool firstRequirement = true;
268 for (i = 0; i < cnt; i++)
269 {
270 // we have an unfulfilled requirement
271 if (ownCount[i] == 0) {
272
273 if(orRequirements[i])
274 {
275 unit = m_prereqUnits[i-1].unit;
276 unitName = unit->getDisplayName();
277 unitName.concat( L" " );
278 unitName.concat(TheGameText->fetch("CONTROLBAR:OrRequirement", NULL));
279 unitName.concat( L" " );
280 requiresList.concat(unitName);
281 }
282
283 // get the requirement and then its name
284 unit = m_prereqUnits[i].unit;
285 unitName = unit->getDisplayName();
286
287 // gets command button, and then modifies unitName
288 //CommandButton *cmdButton = TheControlBar->findCommandButton(unit->getName());

291
292 // format name appropriately with 'returns' if necessary

295 else
296 unitName.concat(L"\n");
297
298 // add it to the list
299 requiresList.concat(unitName);
300 }
301 }
302
303 Bool hasSciences = TRUE;
304 // gotta have all the prereq sciences.
305 for (i = 0; i < m_prereqSciences.size(); i++)
306 {
307 if (!player->hasScience(m_prereqSciences[i]))
308 hasSciences = FALSE;
309 }
310
311 if (hasSciences == FALSE) {
312 if (firstRequirement) {
313 firstRequirement = false;
314 } else {
315 unitName.concat(L"\n");
316 }
317 requiresList.concat(TheGameText->fetch("CONTROLBAR:GeneralsPromotion", NULL));
318 }
319
320 // return final list
321 return requiresList;
322}
