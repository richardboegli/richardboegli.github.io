52MapObject *LightOptions::getSingleSelectedLight(void)
53{
54 MapObject *pMapObj;
55 MapObject *theMapObj = NULL;
56// Bool found = false;
57 Int selCount=0;
58 for (pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext()) {
59 if (pMapObj->isSelected()) {
60 if (pMapObj->isLight()) {
61 theMapObj = pMapObj;
62 }
63 selCount++;
64 }
65 }
66 if (selCount==1 && theMapObj) {
67 return theMapObj;
68 }
69 return(NULL);
70}

154void LightOptions::OnChangeLightEdit()
155{
156 MapObject *theMapObj = getSingleSelectedLight();
157 if (!theMapObj) return;
158 if (!theMapObj->isLight()) return;
159
160 Real lightHeightAboveTerrain, lightInnerRadius, lightOuterRadius;
161 RGBColor lightAmbientColor, lightDiffuseColor;
162
163 Dict *props = theMapObj->getProperties();
164 lightHeightAboveTerrain = props->getReal(TheKey_lightHeightAboveTerrain);
165 lightInnerRadius = props->getReal(TheKey_lightInnerRadius);
166 lightOuterRadius = props->getReal(TheKey_lightOuterRadius);
167 lightAmbientColor.setFromInt(props->getInt(TheKey_lightAmbientColor));
168 lightDiffuseColor.setFromInt(props->getInt(TheKey_lightDiffuseColor));
169
170 Real clr;
171 char buffer[_MAX_PATH];
172
173 CWnd *pEdit = m_staticThis->GetDlgItem(IDC_RA_EDIT);
174 if (pEdit) {
175 pEdit->GetWindowText(buffer, sizeof(buffer));
176 if (1==sscanf(buffer, "%f", &clr)) lightAmbientColor.red = clr;
177 }
178 pEdit = m_staticThis->GetDlgItem(IDC_GA_EDIT);
179 if (pEdit) {
180 pEdit->GetWindowText(buffer, sizeof(buffer));
181 if (1==sscanf(buffer, "%f", &clr)) lightAmbientColor.green = clr;
182 }
183 pEdit = m_staticThis->GetDlgItem(IDC_BA_EDIT);
184 if (pEdit) {
185 pEdit->GetWindowText(buffer, sizeof(buffer));
186 if (1==sscanf(buffer, "%f", &clr)) lightAmbientColor.blue = clr;
187 }
188 pEdit = m_staticThis->GetDlgItem(IDC_HEIGHT_EDIT);
189 Real r;
190 if (pEdit) {
191 pEdit->GetWindowText(buffer, sizeof(buffer));
192 if (1==sscanf(buffer, "%f", &r)) lightHeightAboveTerrain = r;
193 }
194 pEdit = m_staticThis->GetDlgItem(IDC_RADIUS_EDIT);
195 if (pEdit) {
196 pEdit->GetWindowText(buffer, sizeof(buffer));
197 if (1==sscanf(buffer, "%f", &r)) lightOuterRadius = r;
198 }
199 lightDiffuseColor.red = 0;
200 lightDiffuseColor.green = 0;
201 lightDiffuseColor.blue = 0;
202 lightInnerRadius = lightOuterRadius/2;
203
204 props->setReal(TheKey_lightHeightAboveTerrain, lightHeightAboveTerrain);
205 props->setReal(TheKey_lightInnerRadius, lightInnerRadius);
206 props->setReal(TheKey_lightOuterRadius, lightOuterRadius);
207 props->setInt(TheKey_lightAmbientColor, lightAmbientColor.getAsInt());
208 props->setInt(TheKey_lightDiffuseColor, lightDiffuseColor.getAsInt());
209
210 WbView3d * pView = CWorldBuilderDoc::GetActive3DView();
211 if (pView) {
212 pView->invalObjectInView(theMapObj);
213 }
214}
