120Bool GameSpyInfo::sendChat( UnicodeString message, Bool isAction, GameWindow *playerListbox )
121{
122 static UnicodeString s_prevMsg = UnicodeString::TheEmptyString; //stop spam before it happens
123
124 RoomType roomType = StagingRoom;
125 if (getCurrentGroupRoom())
126 roomType = GroupRoom;
127
128 PeerRequest req;
129 req.text = message.str();
130
131 message.trim();
132 // Echo the user's input to the chat window
133 if (!message.isEmpty())
134 {
135 if (!playerListbox)
136 { // Public message
137 if( isAction || message.compare(s_prevMsg) != 0 ) //don't send duplicate messages
138 {
139 req.message.isAction = isAction;
140 req.peerRequestType = PeerRequest::PEERREQUEST_MESSAGEROOM;
141 TheGameSpyPeerMessageQueue->addRequest(req);
142 s_prevMsg = message;
143 }
144 return false;
145 }
146
147 // Get the selections (is this a private message?)
148 Int maxSel = GadgetListBoxGetListLength(playerListbox);
149 Int *selections;
150 GadgetListBoxGetSelected(playerListbox, (Int *)&selections);
151
152 if (selections[0] == -1)
153 { // Public message
154 if( isAction || message.compare(s_prevMsg) != 0 ) //don't send duplicate messages
155 {
156 req.message.isAction = isAction;
157 req.peerRequestType = PeerRequest::PEERREQUEST_MESSAGEROOM;
158 TheGameSpyPeerMessageQueue->addRequest(req);
159 s_prevMsg = message;
160 }
161 return false;
162 }
163 else
164 {
165 // Private message
166
167 // Construct a list
168 AsciiString names = AsciiString::TheEmptyString;
169 AsciiString tmp = AsciiString::TheEmptyString;
170 AsciiString aStr; // AsciiString buf for translating Unicode entries
171 names.format("%s", TheGameSpyInfo->getLocalName().str());
172 for (int i=0; i<maxSel; i++)
173 {
174 if (selections[i] != -1)
175 {
176 aStr.translate(GadgetListBoxGetText(playerListbox, selections[i], GadgetListBoxGetNumColumns(playerListbox)-1));
177 if (aStr.compareNoCase(TheGameSpyInfo->getLocalName()))
178 {
179 tmp.format(",%s", aStr.str());
180 names.concat(tmp);
181 }
182 }
183 else
184 {
185 break;
186 }
187 }
188
189 if (!names.isEmpty())
190 {
191 req.nick = names.str();
192 req.message.isAction = isAction;
193 req.peerRequestType = PeerRequest::PEERREQUEST_MESSAGEPLAYER;
194 TheGameSpyPeerMessageQueue->addRequest(req);
195 }
196 s_prevMsg = message;
197 return true;
198 }
199 }
200 s_prevMsg = message;
201 return false;
202}

216void GameSpyInfo::addChat( PlayerInfo p, UnicodeString msg, Bool isPublic, Bool isAction, GameWindow *win )
217{
218 Int style;
219 if(isSavedIgnored(p.m_profileID) || isIgnored(p.m_name))
220 return;
221
222 Bool isOwner = p.m_flags & PEER_FLAG_OP;
223 Bool isBuddy = getBuddyMap()->find(p.m_profileID) != getBuddyMap()->end();
224
225 Bool isMe = p.m_name.compare(TheGameSpyInfo->getLocalName()) == 0;
226
227 if(!isMe)
228 {
229 if(m_disallowAsainText)
230 {
231 const WideChar *buff = msg.str();
232 Int length = msg.getLength();
233 for(Int i = 0; i < length; ++i)
234 {
235 if(buff[i] >= 256)
236 return;
237 }
238 }
239 else if(m_disallowNonAsianText)
240 {
241 const WideChar *buff = msg.str();
242 Int length = msg.getLength();
243 Bool hasUnicode = FALSE;
244 for(Int i = 0; i < length; ++i)
245 {
246 if(buff[i] >= 256)
247 {
248 hasUnicode = TRUE;
249 break;
250 }
251 }
252 if(!hasUnicode)
253 return;
254 }
255
256 if (!isPublic)
257 {
258 AudioEventRTS privMsgAudio("GUIMessageReceived");
259
260 if( TheAudio )
261 {
262 TheAudio->addAudioEvent( &privMsgAudio );
263 } // end if
264 }
265 }
266
267
268 if (isBuddy)
269 {
270 style = GSCOLOR_CHAT_BUDDY;
271 }
272 else if (isPublic && isAction)
273 {
274 style = (isOwner)?GSCOLOR_CHAT_OWNER_EMOTE:GSCOLOR_CHAT_EMOTE;
275 }
276 else if (isPublic)
277 {
278 style = (isOwner)?GSCOLOR_CHAT_OWNER:GSCOLOR_CHAT_NORMAL;
279 }
280 else if (isAction)
281 {
282 style = (isOwner)?GSCOLOR_CHAT_PRIVATE_OWNER_EMOTE:GSCOLOR_CHAT_PRIVATE_EMOTE;
283 }
284 else
285 {
286 style = (isOwner)?GSCOLOR_CHAT_PRIVATE_OWNER:GSCOLOR_CHAT_PRIVATE;
287 }
288
289 UnicodeString name;
290 name.translate(p.m_name);
291
292 // filters language
293// if( TheGlobalData->m_languageFilterPref )
294// {
295 TheLanguageFilter->filterLine(msg);
296// }
297
298 UnicodeString fullMsg;
299 if (isAction)
300 {
301 fullMsg.format( L"%ls %ls", name.str(), msg.str() );
302 }
303 else
304 {
305 fullMsg.format( L"[%ls] %ls", name.str(), msg.str() );
306 }
307
308 Int index = addText(fullMsg, GameSpyColor[style], win);
309 if (index >= 0)
310 {
311 GadgetListBoxSetItemData(win, (void *)p.m_profileID, index);
312 }
313}

315Int GameSpyInfo::addText( UnicodeString message, Color c, GameWindow *win )
316{
317 if (TheGameSpyGame && TheGameSpyGame->isInGame() && TheGameSpyGame->isGameInProgress())
318 {
319 static AudioEventRTS messageFromChatSound("GUIMessageReceived");
320 TheAudio->addAudioEvent(&messageFromChatSound);
321
322 TheInGameUI->message(message);
323 }
324
325 if (!win)
326 {
327 // try to pick up a registered text window
328 if (m_textWindows.empty())
329 return -1;
330
331 win = *(m_textWindows.begin());
332 }
333 Int index = GadgetListBoxAddEntryText(win, message, c, -1, -1);
334 GadgetListBoxSetItemData(win, (void *)-1, index);
335
336 return index;
337}
