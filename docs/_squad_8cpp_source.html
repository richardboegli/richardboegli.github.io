89// getAllObjects //////////////////////////////////////////////////////////////////////////////////
90const VecObjectPtr& Squad::getAllObjects(void) // Not a const function cause we clear away dead object here too
91{
92 // prunes all NULL objects
93 m_objectsCached.clear();
94 for (VecObjectIDIt it = m_objectIDs.begin(); it != m_objectIDs.end(); ) {
95 Object *obj = TheGameLogic->findObjectByID(*it);
96 if (obj) {
97 m_objectsCached.push_back(obj);
98 ++it;
99 } else {
100 it = m_objectIDs.erase(it);
101 }
102 }
103
104 return m_objectsCached;
105}

152{
153 if (!fromTeam) {
154 return;
155 }
156
157 if (clearSquadFirst) {
158 m_objectIDs.clear();
159 }
160
161 for (DLINK_ITERATOR<Object> iter = fromTeam->iterate_TeamMemberList(); !iter.done(); iter.advance()) {

208void Squad::xfer( Xfer *xfer )
209{
210
211 // version
212 XferVersion currentVersion = 1;
213 XferVersion version = currentVersion;
214 xfer->xferVersion( &version, currentVersion );
215
216 // length of object ID list
217 UnsignedShort objectCount = m_objectIDs.size();
218 xfer->xferUnsignedShort( &objectCount );
219
220 // object id elements
221 ObjectID objectID;
222 if( xfer->getXferMode() == XFER_SAVE )
223 {
224
225 // save each object id
226 VecObjectIDIt it;
227 for( it = m_objectIDs.begin(); it != m_objectIDs.end(); ++it )
228 {
229
230 // save object ID
231 objectID = *it;
232 xfer->xferObjectID( &objectID );
233
234 } // end for, it
235
236 } // end if, save
237 else
238 {
239
240 // the cached objects list should be empty
241 if( m_objectsCached.size() != 0 )
242 {
243
244 DEBUG_CRASH(( "Squad::xfer - m_objectsCached should be emtpy, but is not\n" ));
245 throw SC_INVALID_DATA;
246
247 } // end of
248
249 // read all items
250 for( UnsignedShort i = 0; i < objectCount; ++i )
251 {
252
253 // read id
254 xfer->xferObjectID( &objectID );
255
256 // put on list
257 m_objectIDs.push_back( objectID );
258
259 } // end for, i
260
261 } // end else, load
262
263} // end xfer
