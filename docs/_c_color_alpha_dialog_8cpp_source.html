101void CColorAlphaDialog::performUpdate(IN Bool toUI)
102{
103 static char buff[ARBITRARY_BUFF_SIZE];
104 DebugWindowDialog *pParent = (DebugWindowDialog*) GetParent();
105 if (!pParent) {
106 return;
107 }
108
109 { // update the colors
110 for (int i = 0; i < MAX_KEYFRAMES; ++i) {
111 // update the color swatch
112 CButtonShowColor *pSwatch = (CButtonShowColor*) GetDlgItem(colorControls[i][0]);
113 CWnd *pFrame = GetDlgItem(colorControls[i][1]);
114 RGBColorKeyframe colorFrame;
115
116 if (pSwatch && pFrame) {
117 if (toUI) {
118 pParent->getColorValueFromSystem(i, colorFrame);
119
120 pSwatch->setColor(colorFrame.color);
121 pSwatch->Invalidate(TRUE);
122
123 sprintf(buff, "%u", colorFrame.frame);
124 pFrame->SetWindowText(buff);
125 } else {
126 colorFrame.color = pSwatch->getColor();
127
128 pFrame->GetWindowText(buff, ARBITRARY_BUFF_SIZE - 1);
129 colorFrame.frame = atoi(buff);
130 pParent->updateColorValueToSystem(i, colorFrame);
131 }
132 }
133 }
134 }
135
136 { // update the values
137 for (int i = 0; i < MAX_KEYFRAMES; ++i) {
138 ParticleSystemInfo::RandomKeyframe keyFrame;
139
140 pParent->getAlphaRangeFromSystem(i, keyFrame);
141
142 { // Minimum first
143 CWnd *pMin = GetDlgItem(alphaControls[i][0]);
144 if (pMin) {
145 if (toUI) {
146 sprintf(buff, FORMAT_STRING, keyFrame.var.getMinimumValue());
147 pMin->SetWindowText(buff);
148 } else {
149 pMin->GetWindowText(buff, ARBITRARY_BUFF_SIZE - 1);
150 keyFrame.var.m_low = atof(buff);
151 }
152 }
153 }
154
155 { // then maximum
156 CWnd *pMax = GetDlgItem(alphaControls[i][1]);
157 if (pMax) {
158 if (toUI) {
159 sprintf(buff, FORMAT_STRING, keyFrame.var.getMaximumValue());
160 pMax->SetWindowText(buff);
161 } else {
162 pMax->GetWindowText(buff, ARBITRARY_BUFF_SIZE - 1);
163 keyFrame.var.m_high = atof(buff);
164 }
165 }
166 }
167
168 { // then the frame
169 CWnd *pFrame = GetDlgItem(alphaControls[i][2]);
170 if (pFrame) {
171 if (toUI) {
172 // Unsigned int
173 sprintf(buff, "%u", keyFrame.frame);
174 pFrame->SetWindowText(buff);
175 } else {
176 pFrame->GetWindowText(buff, ARBITRARY_BUFF_SIZE - 1);
177 keyFrame.frame = (unsigned) atoi(buff);
178 }
179 }
180 }
181
182 if (toUI) {
183 // We're all done.
184 } else {
185 pParent->updateAlphaRangeToSystem(i, keyFrame);
186 }
187 }
188 }

201BOOL CColorAlphaDialog::OnInitDialog()
202{
203 // replace all the buttons with our buttons.
204 for (int i = 0; i < MAX_KEYFRAMES; ++i) {
205 CRect rect;
206 CWnd *item = GetDlgItem(colorControls[i][0]);
207 if (!item) {
208 continue;
209 }
210
211 item->GetWindowRect(&rect);
212 ScreenToClient(&rect);
213 DWORD style = item->GetStyle();
214 m_colorButton[i].Create("", style, rect, this, colorControls[i][0]);
215
216 item->DestroyWindow();
217 }
218
219 return CDialog::OnInitDialog();
220}
