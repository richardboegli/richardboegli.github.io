307const PlayerTemplate* PlayerTemplateStore::findPlayerTemplate(NameKeyType namekey) const
308{
309// begin ugly, hokey code to quietly load old maps...
310 static NameKeyType a0 = NAMEKEY("FactionAmerica");
311 static NameKeyType a1 = NAMEKEY("FactionAmericaChooseAGeneral");
312 static NameKeyType a2 = NAMEKEY("FactionAmericaTankCommand");
313 static NameKeyType a3 = NAMEKEY("FactionAmericaSpecialForces");
314 static NameKeyType a4 = NAMEKEY("FactionAmericaAirForce");
315
316 static NameKeyType c0 = NAMEKEY("FactionChina");
317 static NameKeyType c1 = NAMEKEY("FactionChinaChooseAGeneral");
318 static NameKeyType c2 = NAMEKEY("FactionChinaRedArmy");
319 static NameKeyType c3 = NAMEKEY("FactionChinaSpecialWeapons");
320 static NameKeyType c4 = NAMEKEY("FactionChinaSecretPolice");
321
322 static NameKeyType g0 = NAMEKEY("FactionGLA");
323 static NameKeyType g1 = NAMEKEY("FactionGLAChooseAGeneral");
324 static NameKeyType g2 = NAMEKEY("FactionGLATerrorCell");
325 static NameKeyType g3 = NAMEKEY("FactionGLABiowarCommand");
326 static NameKeyType g4 = NAMEKEY("FactionGLAWarlordCommand");
327
328 if (namekey == a1 || namekey == a2 || namekey == a3 || namekey == a4)
329 namekey = a0;
330 else if (namekey == c1 || namekey == c2 || namekey == c3 || namekey == c4)
331 namekey = c0;
332 else if (namekey == g1 || namekey == g2 || namekey == g3 || namekey == g4)
333 namekey = g0;
334// end ugly, hokey code to quietly load old maps...
335
336 #ifdef _DEBUG
337 AsciiString nn = KEYNAME(namekey);
338 #endif
339 for (PlayerTemplateVector::const_iterator it = m_playerTemplates.begin(); it != m_playerTemplates.end(); ++it)
340 {
341 #ifdef _DEBUG
342 AsciiString n = KEYNAME((*it).getNameKey());
343 #endif
344 if ((*it).getNameKey() == namekey)
345 return &(*it);
346 }
347 return NULL;
348}

360// @todo: PERF_EVALUATE Get a perf timer on this.
361// If this function is called frequently, there are some relatively trivial changes we could make to
362// have it run a lot faster.
363void PlayerTemplateStore::getAllSideStrings(AsciiStringList *outStringList)
364{
365 if (!outStringList)
366 return;
367
368 // should outStringList be cleared first? If so, that would go here
369 AsciiStringList tmpList;
370
371 Int numTemplates = getPlayerTemplateCount();
372 for ( Int i = 0; i < numTemplates; ++i )
373 {
374 const PlayerTemplate *pt = getNthPlayerTemplate(i);
375 // Sanity
376 if (!pt)
377 continue;
378
379 if (std::find(tmpList.begin(), tmpList.end(), pt->getSide()) == tmpList.end())
380 tmpList.push_back(pt->getSide());
381 }
382 // tmpList is now filled with all unique sides found in the player templates.
383
384 // splice is a constant-time function which takes all elements from tmpList and
385 // inserts them before outStringList->end(), and also removes them from tmpList
386 outStringList->splice(outStringList->end(), tmpList);
387
388 // all done
389}

392/*static*/ void PlayerTemplateStore::parsePlayerTemplateDefinition( INI* ini )
393{
394 const char* c = ini->getNextToken();
395 NameKeyType namekey = NAMEKEY(c);
396
397 PlayerTemplate* pt = const_cast<PlayerTemplate*>(ThePlayerTemplateStore->findPlayerTemplate(namekey));
398 if (pt)
399 {
400 ini->initFromINI(pt, pt->getFieldParse() );
401 pt->setNameKey(namekey);
402 }
403 else
404 {
405 PlayerTemplate npt;
406 ini->initFromINI( &npt, npt.getFieldParse() );
407 npt.setNameKey(namekey);
408 ThePlayerTemplateStore->m_playerTemplates.push_back(npt);
409 }
410
411}
