48/***********************************************************************************************
49 * HSVClass::Adjust -- Adjust an HSV color toward specified color. *
50 * *
51 * This routine will adjust the HSV color object toward the color of the specified HSV *
52 * object. Typical users of this would be palette morphing or fading routines. *
53 * *
54 * INPUT: ratio -- The ratio to move the HSV object toward the color specified. A value *
55 * of zero means no movement at all. A value of 255 means move completely *
56 * toward the specified color (changed completely). *
57 * *
58 * hsv -- A reference to the color that the current HSV object is to change *
59 * toward. *
60 * *
61 * OUTPUT: none *
62 * *
63 * WARNINGS: none *
64 * *
65 * HISTORY: *
66 * 02/20/1996 JLB : Created. *
67 *=============================================================================================*/
68void HSVClass::Adjust(int ratio, HSVClass const & hsv)
69{
70 /*
71 ** Ratio conversion is limited to 0 through 100%. This is
72 ** the range of 0 to 255.
73 */
74 ratio &= 0x00FF;
75
76 /*
77 ** Adjust the color guns by the ratio specified toward the
78 ** destination color.
79 */
80 int value = hsv.Get_Value() - Get_Value();
81 Value = (unsigned char)(Get_Value() + (value * ratio) / 256);
82
83 int saturation = hsv.Get_Saturation() - Get_Saturation();
84 Saturation = (unsigned char)(Get_Saturation() + (saturation * ratio) / 256);
85
86 int hue = hsv.Get_Hue() - Get_Hue();
87 Hue = (unsigned char)(Get_Hue() + (hue * ratio) / 256);
88}

91/***********************************************************************************************
92 * HSVClass::Difference -- Finds the difference between two HSV color objects. *
93 * *
94 * This routine will determine a color difference between two HSV objects. The difference *
95 * has no particular meaning other that larger numbers meaning greater difference. *
96 * *
97 * INPUT: hsv -- The other HSV object to compare this HSV object to. *
98 * *
99 * OUTPUT: Returns with a relative distance (in arbitrary units) between this HSV object and *
100 * the HSV object supplied. *
101 * *
102 * WARNINGS: none *
103 * *
104 * HISTORY: *
105 * 02/20/1996 JLB : Created. *
106 *=============================================================================================*/
107int HSVClass::Difference(HSVClass const & hsv) const
108{
109 int hue = (int)Hue - (int)hsv.Hue;
110 if (hue < 0) hue = -hue;
111
112 int saturation = (int)Saturation - (int)hsv.Saturation;
113 if (saturation < 0) saturation = -saturation;
114
115 int value = (int)Value - (int)hsv.Value;
116 if (value < 0) value = -value;
117
118 return(hue*hue + saturation*saturation + value*value);
119}
