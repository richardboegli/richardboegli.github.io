365/***********************************************************************************************
366 * Matrix4x4::Init_Ortho -- Initialize to an orthographic projection matrix *
367 * *
368 * You can find the formulas for this in the appendix of the OpenGL programming guide. Also *
369 * this happens to be the same convention used by Surrender. *
370 * *
371 * The result of this projection will be that points inside the volume will have all coords *
372 * between -1 and +1. A point at znear will project to z=-1. A point at zfar will project *
373 * to z=+1... *
374 * *
375 * INPUT: *
376 * *
377 * OUTPUT: *
378 * *
379 * WARNINGS: *
380 * Note that the znear and zfar parameters are positive distances to the clipping planes *
381 * even though in the camera coordinate system, the clipping planes are at negative z *
382 * coordinates. This holds for all of the projection initializations and is consistent *
383 * with OpenGL's convention. *
384 * *
385 * HISTORY: *
386 * 11/5/99 gth : Created. *
387 *=============================================================================================*/
388WWINLINE void Matrix4x4::Init_Ortho
389(
390 float left,
391 float right,
392 float bottom,
393 float top,
394 float znear,
395 float zfar
396)
397{
398 assert(znear >= 0.0f);
399 assert(zfar > znear);
400
401 Make_Identity();
402 Row[0][0] = 2.0f / (right - left);
403 Row[0][3] = -(right + left) / (right - left);
404 Row[1][1] = 2.0f / (top - bottom);
405 Row[1][3] = -(top + bottom) / (top - bottom);
406 Row[2][2] = -2.0f / (zfar - znear);
407 Row[2][3] = -(zfar + znear) / (zfar - znear);
408}

411/***********************************************************************************************
412 * Matrix4x4::Init_Perspective -- Initialize to a perspective projection matrix *
413 * *
414 * You can find the formulas for this matrix in the appendix of the OpenGL programming guide. *
415 * Also, this happens to be the same convention used by Surrender. *
416 * *
417 * The result of this projection will be that points inside the volume will have all coords *
418 * between -1 and +1. A point at znear will project to z=-1. A point at zfar will project *
419 * to z=+1... *
420 * *
421 * INPUT: *
422 * hfov - horizontal field of view (in radians) *
423 * vfov - vertical field of view (in radians) *
424 * znear - distance to near z clipping plane (positive) *
425 * zfar - distance to the far z clipping plane (positive) *
426 * *
427 * OUTPUT: *
428 * *
429 * WARNINGS: *
430 * Note that the znear and zfar parameters are positive distances to the clipping planes *
431 * even though in the camera coordinate system, the clipping planes are at negative z *
432 * coordinates. This holds for all of the projection initializations and is consistent *
433 * with OpenGL's convention. *
434 * *
435 * HISTORY: *
436 * 11/5/99 gth : Created. *
437 *=============================================================================================*/
438WWINLINE void Matrix4x4::Init_Perspective(float hfov,float vfov,float znear,float zfar)
439{
440 assert(znear > 0.0f);
441 assert(zfar > znear);
442
443 Make_Identity();
444 Row[0][0] = static_cast<float>(1.0 / tan(hfov*0.5));
445 Row[1][1] = static_cast<float>(1.0 / tan(vfov*0.5));
446 Row[2][2] = -(zfar + znear) / (zfar - znear);
447 Row[2][3] = static_cast<float>(-(2.0*zfar*znear) / (zfar - znear));
448 Row[3][2] = -1.0f;
449 Row[3][3] = 0.0f;
450}

453/***********************************************************************************************
454 * Matrix4x4::Init_Perspective -- Initialize to a perspective projection matrix *
455 * *
456 * You can find the formulas for this matrix in the appendix of the OpenGL programming guide. *
457 * Also, this happens to be the same convention used by Surrender. *
458 * *
459 * The result of this projection will be that points inside the volume will have all coords *
460 * between -1 and +1. A point at znear will project to z=-1. A point at zfar will project *
461 * to z=+1... *
462 * *
463 * INPUT: *
464 * *
465 * left - min x coordinate of near clip plane *
466 * right - max x coordinate of near clip plane *
467 * bottom - min y coordinate of near clip plane *
468 * top - max y coordinate of near clip plane *
469 * znear - distance to near Z clipping plane *
470 * zfar - distance to far Z clipping plane *
471 * *
472 * OUTPUT: *
473 * *
474 * WARNINGS: *
475 * Note that the znear and zfar parameters are positive distances to the clipping planes *
476 * even though in the camera coordinate system, the clipping planes are at negative z *
477 * coordinates. This holds for all of the projection initializations and is consistent *
478 * with OpenGL's convention. *
479 * *
480 * HISTORY: *
481 * 11/5/99 gth : Created. *
482 *=============================================================================================*/
483WWINLINE void Matrix4x4::Init_Perspective
484(
485 float left,
486 float right,
487 float bottom,
488 float top,
489 float znear,
490 float zfar
491)
492{
493 assert(znear > 0.0f);
494 assert(zfar > 0.0f);
495
496 Make_Identity();
497 Row[0][0] = static_cast<float>(2.0*znear / (right - left));
498 Row[0][2] = (right + left) / (right - left);
499 Row[1][1] = static_cast<float>(2.0*znear / (top - bottom));
500 Row[1][2] = (top + bottom) / (top - bottom);
501 Row[2][2] = -(zfar + znear) / (zfar - znear);
502 Row[2][3] = static_cast<float>(-(2.0*zfar*znear) / (zfar - znear));
503 Row[3][2] = -1.0f;
504 Row[3][3] = 0.0f;
505}

529/***********************************************************************************************
530 * Matrix4x4::Inverse -- returns the inverse of the matrix *
531 * *
532 * INPUT: *
533 * *
534 * OUTPUT: *
535 * *
536 * WARNINGS: *
537 * *
538 * HISTORY: *
539 * 06/02/1997 GH : Created. *
540 *=============================================================================================*/
541WWINLINE Matrix4x4 Matrix4x4::Inverse() const // Gauss-Jordan elimination with partial pivoting
542{
543 WWASSERT_PRINT(0,"Matrix4x4::Inverse does not work, re-implement!");
544
545 Matrix4x4 a(*this); // As a evolves from original mat into identity
546 Matrix4x4 b(true); // b evolves from identity into inverse(a)
547 int i, j, i1;
548
549 // Loop over cols of a from left to right, eliminating above and below diagonal
550 for (j=0; j<4; j++) {
551
552 // Find largest pivot in column j among rows j..3
553 i1 = j;
554 for (i=j+1; i<4; i++) {
555 if (WWMath::Fabs(a[i][j]) > WWMath::Fabs(a[i1][j])) {
556 i1 = i;
557 }
558 }
559
560 // Swap rows i1 and j in a and b to put pivot on diagonal
561 Swap(a.Row[i1], a.Row[j]);
562 Swap(b.Row[i1], b.Row[j]);
563
564 // Scale row j to have a unit diagonal
565 if (a[j][j]==0.) {
566 //ALGEBRA_ERROR("Matrix4x4::inverse: singular matrix; can't invert\n");
567 }
568 b.Row[j] /= a.Row[j][j];
569 a.Row[j] /= a.Row[j][j];
570
571 // Eliminate off-diagonal elems in col j of a, doing identical ops to b
572 for (i=0; i<4; i++) {
573 if (i != j) {
574 b.Row[i] -= a[i][j] * b.Row[j];
575 a.Row[i] -= a[i][j] * a.Row[j];
576 }
577 }
578 }
579 return b;
580}
