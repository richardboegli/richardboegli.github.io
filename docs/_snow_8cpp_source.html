59void SnowManager::updateIniSettings(void)
60{
61 Real *dst=m_startingHeights;
62 //initialize a table of random starting positions for each particle.
63 Int boxDimensions = (Int)TheWeatherSetting->m_snowBoxDimensions;
64 for (Int y=0; y<SNOW_NOISE_Y; y++)
65 {
66 for (Int x=0; x<SNOW_NOISE_X; x++)
67 {
68 *dst=(Real)(rand()%(boxDimensions));
69 dst++;
70 }
71 }
72
73 m_velocity = TheWeatherSetting->m_snowVelocity;
74 m_frequencyScaleX = TheWeatherSetting->m_snowFrequencyScaleX;
75 m_frequencyScaleY = TheWeatherSetting->m_snowFrequencyScaleY;
76 m_amplitude = TheWeatherSetting->m_snowAmplitude;
77 m_pointSize = TheWeatherSetting->m_snowPointSize;
78 m_quadSize = TheWeatherSetting->m_snowQuadSize;
79 m_boxDimensions = TheWeatherSetting->m_snowBoxDimensions;
80 m_emitterSpacing = 1.0f/TheWeatherSetting->m_snowBoxDensity;
81 m_maxPointSize = TheWeatherSetting->m_snowMaxPointSize;
82 m_minPointSize = TheWeatherSetting->m_snowMinPointSize;
83
84 //Time for snow flake to make it from top to bottom of rendered cube around camera.
85 m_fullTimePeriod = m_boxDimensions/m_velocity;
86}
