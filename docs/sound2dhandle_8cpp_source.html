70Sound2DHandleClass::Initialize (SoundBufferClass *buffer)
71{
72 SoundHandleClass::Initialize (buffer);
73
74 if (SampleHandle != (HSAMPLE)INVALID_MILES_HANDLE) {
75
76 //
