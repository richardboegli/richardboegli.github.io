75void INI::parseShellMenuSchemeDefinition( INI *ini )
76{
77 AsciiString name;
78 ShellMenuSchemeManager *SMSchemeManager;
79 ShellMenuScheme *SMScheme;
80
81 // read the name
82 const char* c = ini->getNextToken();
83 name.set( c );
84
85 // find existing item if present
86 SMSchemeManager = TheShell->getShellMenuSchemeManager();
87 DEBUG_ASSERTCRASH( SMSchemeManager, ("parseShellMenuSchemeDefinition: Unable to Get SMSchemeManager\n") );
88 if( !SMSchemeManager )
89 return;
90
91 // If we have a previously allocated control bar, this will return a cleared out pointer to it so we
92 // can overwrite it
93 SMScheme = SMSchemeManager->newShellMenuScheme( name );
94
95 // sanity
96 DEBUG_ASSERTCRASH( SMScheme, ("parseControlBarSchemeDefinition: Unable to allocate Scheme '%s'\n", name.str()) );
97
98 // parse the ini definition
99 ini->initFromINI( SMScheme, SMSchemeManager->getFieldParse() );
100
101} // end parseCommandButtonDefinition

172void ShellMenuScheme::draw( void )
173{
174
175 ShellMenuSchemeImageListIt imageIt = m_imageList.begin();
176 while(imageIt != m_imageList.end())
177 {
178 ShellMenuSchemeImage *image = *imageIt;
179 if(image && image->m_image)
180 {
181 TheDisplay->drawImage(image->m_image, image->m_position.x, image->m_position.y,
182 image->m_position.x + image->m_size.x , image->m_position.y + image->m_size.y);
183 }
184 ++imageIt;
185 }
186
187 ShellMenuSchemeLineListIt it = m_lineList.begin();
188 while(it != m_lineList.end())
189 {
190 ShellMenuSchemeLine *line = *it;
191
192 if(line)
193 {
194 TheDisplay->drawLine(line->m_startPos.x, line->m_startPos.y, line->m_endPos.x,
195 line->m_endPos.y,line->m_width, line->m_color);
196 }
197 ++it;
198 }

258ShellMenuScheme *ShellMenuSchemeManager::newShellMenuScheme(AsciiString name)
259{
260 ShellMenuSchemeListIt it;
261 it = m_schemeList.begin();
262 name.toLower();
263 while(it != m_schemeList.end())
264 {
265 ShellMenuScheme *scheme = *it;
266 if(scheme->m_name.compare(name) == 0)
267 {
268 m_schemeList.erase( it );
269 delete scheme;
270 break;
271 }
272 else
273 ++it;
274 }
275 ShellMenuScheme *newScheme = NEW ShellMenuScheme;
276 newScheme->m_name.set(name);
277 m_schemeList.push_back(newScheme);
278 return newScheme;
279}
