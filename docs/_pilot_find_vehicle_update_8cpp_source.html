101UpdateSleepTime PilotFindVehicleUpdate::update()
102{
103
104 Object *obj = getObject();
105 if (obj->getControllingPlayer()->getPlayerType() == PLAYER_HUMAN) {
106 // This is an ai only behavior.
107 return UPDATE_SLEEP_FOREVER;
108 }
109 const PilotFindVehicleUpdateModuleData *data = getPilotFindVehicleUpdateModuleData();
110
111 AIUpdateInterface *ai = obj->getAI();
112 if (ai==NULL) return UPDATE_SLEEP_FOREVER;
113
114 if( !ai->isIdle() )
115 {
116 return UPDATE_SLEEP(data->m_scanFrames);
117 }
118
119 //Periodic scanning (expensive)
120 Object *upgradeUnit = scanClosestTarget();
121 if(upgradeUnit)
122 {
123 ai->aiEnter(upgradeUnit, CMD_FROM_AI);
124 m_didMoveToBase = false;
125 } else {
126 Coord3D pos;
127 // Try moving to base 1 time.
128 if (!m_didMoveToBase && obj->getControllingPlayer()->getAiBaseCenter(&pos)) {
129 ai->aiMoveToPosition(&pos, CMD_FROM_AI);
130 m_didMoveToBase = true;
131 }
132 }
133 return UPDATE_SLEEP(data->m_scanFrames);
134}

138Object* PilotFindVehicleUpdate::scanClosestTarget()
139{
140 const PilotFindVehicleUpdateModuleData *data = getPilotFindVehicleUpdateModuleData();
141 Object *me = getObject();
142
143 PartitionFilterAcceptByKindOf kindFilter(MAKE_KINDOF_MASK(KINDOF_VEHICLE), KINDOFMASK_NONE);
144 PartitionFilterAlive aliveFilter;
145 PartitionFilterPlayer playerFilter(me->getControllingPlayer(), true);
146 PartitionFilterSameMapStatus filterMapStatus(getObject());
147 PartitionFilter *filters[5];
148 filters[0] = &kindFilter;
149 filters[1] = &aliveFilter;
150 filters[2] = &playerFilter;
151 filters[3] = &filterMapStatus;
152 filters[4] = NULL;
153
154 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( me->getPosition(), data->m_scanRange,
155 FROM_CENTER_2D, filters, ITER_SORTED_NEAR_TO_FAR );
156 MemoryPoolObjectHolder hold(iter);
157
158 for( Object *other = iter->first(); other; other = iter->next() )
159 {
160 // Check health.
161 BodyModuleInterface *body = other->getBodyModule();
162 if (!body) continue;
163 // If we're real healthy, don't bother looking for healing.
164 if (body->getHealth() < body->getMaxHealth()*data->m_minHealth)
165 {
166 continue;
167 }
168 // first, see if we'd like to collide with 'other'
169 for (BehaviorModule** m = me->getBehaviorModules(); *m; ++m)
170 {
171 CollideModuleInterface* collide = (*m)->getCollide();
172 if (!collide)
173 continue;
174
175 if( collide->wouldLikeToCollideWith( other ) )
176 {
177 return other;
178 }
179 }
180 }
181 return NULL;
182}
