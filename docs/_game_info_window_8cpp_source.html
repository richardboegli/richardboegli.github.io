102void RefreshGameInfoWindow(GameInfo *gameInfo, UnicodeString gameName)
103{
104 static const Image *randomIcon = TheMappedImageCollection->findImageByName("GameinfoRANDOM");
105 static const Image *observerIcon = TheMappedImageCollection->findImageByName("GameinfoOBSRVR");
106 if(!gameInfoWindowLayout || !gameInfo )
107 return;
108
109 parent->winHide( FALSE );
110 parent->winBringToTop();
111
112 // Set the game name
113 GadgetStaticTextSetText(staticTextGameName, ((LANGameInfo *)gameInfo)->getPlayerName(0));
114 // set the map name
115 UnicodeString map;
116 AsciiString asciiMap = gameInfo->getMap();
117 asciiMap.toLower();
118 std::map<AsciiString, MapMetaData>::iterator it = TheMapCache->find(asciiMap);
119 if (it != TheMapCache->end())
120 {
121 map = it->second.m_displayName;
122 }
123 else
124 {
125 // can happen if the map will have to be transferred... so use the leaf name (srj)
126 const char *noPath = gameInfo->getMap().reverseFind('\\');
127 if (noPath)
128 {
129 ++noPath;
130 }
131 else
132 {
133 noPath = gameInfo->getMap().str();
134 }
135 map.translate(noPath);
136 }
137 GadgetStaticTextSetText(staticTextMapName,map);
138
139 // fill in the player list
140
141 GadgetListBoxReset(listBoxPlayers);
142
143 Int numColors = TheMultiplayerSettings->getNumColors();
144 Color white = GameMakeColor(255,255,255,255);
145// Color grey = GameMakeColor(188,188,188,255);
146 for (Int i = 0; i < MAX_SLOTS; i ++)
147 {
148 Color playerColor = white;
149 Int color = -1;
150 Int addedRow;
151 GameSlot *slot = gameInfo->getSlot(i);
152 if(!slot || (slot->isOccupied() == FALSE))
153 continue;
154 color = slot->getColor();
155 if(color > -1 && color < numColors)
156 {
157 MultiplayerColorDefinition *def = TheMultiplayerSettings->getColor(color);
158 playerColor = def->getColor();
159 }
160 if(slot->isAI())
161 {
162 switch(slot->getState())
163 {
164 case SLOT_EASY_AI:
165 {
166 addedRow = GadgetListBoxAddEntryText(listBoxPlayers,TheGameText->fetch("GUI:EasyAI"),playerColor,-1, 1);
167 break;
168 }
169 case SLOT_MED_AI:
170 {
171 addedRow = GadgetListBoxAddEntryText(listBoxPlayers,TheGameText->fetch("GUI:MediumAI"),playerColor,-1, 1);
172 break;
173 }
174 case SLOT_BRUTAL_AI:
175 {
176 addedRow = GadgetListBoxAddEntryText(listBoxPlayers,TheGameText->fetch("GUI:HardAI"),playerColor,-1, 1);
177 break;
178 }
179 default:
180 break;
181 }
182 }
183 else if(slot->isHuman())
184 {
185 addedRow = GadgetListBoxAddEntryText(listBoxPlayers, slot->getName(),playerColor,-1,1);
186 }
187 Int playerTemplate = slot->getPlayerTemplate();
188 if(playerTemplate == PLAYERTEMPLATE_OBSERVER)
189 {
190 GadgetListBoxAddEntryImage(listBoxPlayers, observerIcon,addedRow, 0, 22,25);
191 }
192 else if(playerTemplate < 0 || playerTemplate >= ThePlayerTemplateStore->getPlayerTemplateCount())
193 {
195 GadgetListBoxAddEntryImage(listBoxPlayers, randomIcon,addedRow, 0, 22,25);
196 //GadgetListBoxAddEntryText(listBoxPlayers,TheGameText->fetch("GUI:???"),playerColor,addedRow, 0);
197 }
198 else
199 {
200 const PlayerTemplate *fact = ThePlayerTemplateStore->getNthPlayerTemplate(playerTemplate);
201 GadgetListBoxAddEntryImage(listBoxPlayers, fact->getSideIconImage(),addedRow, 0, 22,25);
202 //GadgetListBoxAddEntryText(listBoxPlayers,fact->getDisplayName(),playerColor,addedRow, 0);
203 }
204
205 }
206}
