100SpecialPowerModule::SpecialPowerModule( Thing *thing, const ModuleData *moduleData )
101 : BehaviorModule( thing, moduleData )
102{
103
104 m_availableOnFrame = 0;
105 m_pausedCount = 0;
106 m_pausedOnFrame = 0;
107 m_pausedPercent = 0.0f;
108
109 // we won't be able to use the power for X number of frames now
110
111 // if we're pre-built, start counting down
112 if( !getObject()->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
113 {
114 //A sharedNSync special only startPowerRecharges when first scienced or when executed,
115 //Since a new modue with same SPTemplates may construct at any time.
116 if ( getSpecialPowerTemplate()->isSharedNSync() == FALSE )
117 startPowerRecharge();
118 }
119 // WE USED TO DO THE POLL-EVERYBODY-AND-VOTE-ON-WHO-TO-SYNC-TO THING HERE,
120 // BUT NO MORE, NOW IT IS HANDLED IN PLAYER
121
122 // Some Special powers need to be activated by an Upgrade, so prevent the timer from going until then
123 const SpecialPowerModuleData *md = (const SpecialPowerModuleData *)moduleData;
124 if( md->m_startsPaused )
125 pauseCountdown( TRUE );
126
127 resolveSpecialPower();
128
129 // Now, if we find that we have just come into being,
130 // but there is already a science granted for our shared superweapon,
131 // lets make sure TheIngameUI knows about our public timer
132 // add this weapon to the UI if it has a public timer for all to see
133 if( m_pausedCount == 0 &&
134 getSpecialPowerTemplate()->isSharedNSync() == TRUE &&
135 getSpecialPowerTemplate()->hasPublicTimer() == TRUE &&
136 getObject()->getControllingPlayer() &&
137 getObject()->isKindOf( KINDOF_STRUCTURE ) )
138 {
139 TheInGameUI->addSuperweapon( getObject()->getControllingPlayer()->getPlayerIndex(),
140 getPowerName(),
141 getObject()->getID(),
142 getSpecialPowerModuleData()->m_specialPowerTemplate );
143 }
144
145
146} // end SpecialPowerModule

202void SpecialPowerModule::onSpecialPowerCreation( void )
203{
204 // THIS gets called by addScience(), that is, when the General has purchased a new special power,
205 // and this module is thus activated.
206
207 // start a power recharge going
208 startPowerRecharge();
209
210 // Dustin wants these special powers to start ready to fire,
211 // so here (and only here) we will expressly set them to ready-now.
212 if ( getSpecialPowerTemplate()->isSharedNSync())
213 {
214 Player *player = getObject()->getControllingPlayer();
215 if ( player )
216 {
217 player->expressSpecialPowerReadyFrame( getSpecialPowerTemplate(), TheGameLogic->getFrame() );
218 m_availableOnFrame = player->getOrStartSpecialPowerReadyFrame( getSpecialPowerTemplate() );
219 }
220 }
221
222 // Some Special powers need to be activated by an Upgrade, so prevent the timer from going until then
223 const SpecialPowerModuleData *md = getSpecialPowerModuleData();
224 if( md->m_startsPaused )
225 pauseCountdown( TRUE );
226
227 // add this weapon to the UI if it has a public timer for all to see
228 if( getSpecialPowerModuleData()->m_specialPowerTemplate->hasPublicTimer() == TRUE &&
229 getObject()->getControllingPlayer() &&
230 getObject()->isKindOf( KINDOF_STRUCTURE ) )
231 {
232 TheInGameUI->addSuperweapon( getObject()->getControllingPlayer()->getPlayerIndex(),
233 getPowerName(),
234 getObject()->getID(),
235 getSpecialPowerModuleData()->m_specialPowerTemplate );
236 }
237}

289Bool SpecialPowerModule::isReady() const
290{
291#if defined(_DEBUG) || defined(_INTERNAL) || defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
292 // this is a cheat ... remove this for release!
293 if( TheGlobalData->m_specialPowerUsesDelay == FALSE )
294 return TRUE;
295#endif
296
297 const Object* obj = getObject();
298 const SpecialPowerModuleData *modData = getSpecialPowerModuleData();
299
300 if ( obj && modData )
301 {
302 Player *player = getObject()->getControllingPlayer();
303 if ( player )
304 {
305 if ( modData->m_specialPowerTemplate->isSharedNSync())
306 return (TheGameLogic->getFrame() >= player->getOrStartSpecialPowerReadyFrame( modData->m_specialPowerTemplate ) );
307 }
308 }
309
310 return (m_pausedCount == 0) && (TheGameLogic->getFrame() >= m_availableOnFrame);
311
312} // end isReady

320//-------------------------------------------------------------------------------------------------
321Real SpecialPowerModule::getPercentReady() const
322{
323 if( m_pausedCount > 0 && m_pausedPercent == 1.0f )
324 {
325 //Don't consider it ready if paused.
326 return 0.99999f;
327 }
328
329#if defined(_DEBUG) || defined(_INTERNAL) || defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
330 if( TheGlobalData->m_specialPowerUsesDelay == FALSE )
331 return 1.0f;
332#endif
333
334 // easy case ... is ready
335 if( isReady() )
336 return 1.0f;
337
338 if( m_pausedCount > 0 )
339 {
340 return m_pausedPercent;
341 }
342
343 // get the module data
344 const SpecialPowerModuleData *modData = getSpecialPowerModuleData();
345
346 // sanity
347 if( modData->m_specialPowerTemplate == NULL )
348 return 0.0f;
349
350 UnsignedInt readyFrame = m_availableOnFrame;
351
352 //unless
353 const Object* obj = getObject();
354 if ( obj )
355 {
356 Player *player = getObject()->getControllingPlayer();
357 if ( player )
358 {
359 if ( modData->m_specialPowerTemplate->isSharedNSync())
360 {
361 readyFrame = player->getOrStartSpecialPowerReadyFrame( getSpecialPowerTemplate() );
362 }
363 }
364 }
365
366 // calculate the percent
367 Real percent = 1.0f - ((readyFrame - TheGameLogic->getFrame()) /
368 (Real)modData->m_specialPowerTemplate->getReloadTime());
369
370 return percent;
371}

385//-------------------------------------------------------------------------------------------------
388//-------------------------------------------------------------------------------------------------
389void SpecialPowerModule::startPowerRecharge()
390{
391#if defined(_DEBUG) || defined(_INTERNAL) || defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
392 // this is a cheat ... remove this for release!
393 if( TheGlobalData->m_specialPowerUsesDelay == FALSE )
394 return;
395#endif
396
397 const SpecialPowerModuleData *modData = getSpecialPowerModuleData();
398
399 // sanity
400 if( modData->m_specialPowerTemplate == NULL )
401 {
402 DEBUG_CRASH(("special power not found"));
403 return;
404 }
405
406 Object* obj = getObject();
407 if (!obj)
408 return;
409
410 Player* player = getObject()->getControllingPlayer();
411 if (!player)
412 return;
413
414 //Here, we make sure that general specials work as one between command centers
415 // only factory type faction buildings should do this, and only with generals powers (in general)
416 // but there are no restrictions on the use of SharedNSync on specialPowerTemplates at large
417 if ( modData->m_specialPowerTemplate->isSharedNSync() )
418 {
419 player->resetOrStartSpecialPowerReadyFrame( modData->m_specialPowerTemplate );
420 }
421 else
422 {
423 // set the frame we will be 100% available on now
424 m_availableOnFrame = TheGameLogic->getFrame() + getSpecialPowerTemplate()->getReloadTime();
425 }
426}

430Bool SpecialPowerModule::initiateIntentToDoSpecialPower( const Object *targetObj, const Coord3D *targetPos, const Waypoint *way, UnsignedInt commandOptions )
431{
432 Bool valid = false;
433 // tell our update modules that we intend to do this special power.
434 for( BehaviorModule** u = getObject()->getBehaviorModules(); *u; ++u )
435 {
436 SpecialPowerUpdateInterface* spu = (*u)->getSpecialPowerUpdateInterface();
437 if( spu )
438 {
439 //Validate that we are calling the correct module!
440 if( isModuleForPower( getSpecialPowerModuleData()->m_specialPowerTemplate ) )
441 {
442 if( spu->doesSpecialPowerUpdatePassScienceTest() )
443 {
444 if( spu->initiateIntentToDoSpecialPower( getSpecialPowerModuleData()->m_specialPowerTemplate, targetObj, targetPos, way, commandOptions ) )
445 {
446 //Kris: Aug 2003
447 //We have executed the special power, so don't try to execute any more. This logic
448 //was changed for multi-level spectres. Before, multiple modules would get launched
449 //causing 2 or 3 spectres to be created.
450 valid = true;
451 break;
452 }
453 }
454 }
455 }
456 }
457
458 getObject()->getControllingPlayer()->getAcademyStats()->recordSpecialPowerUsed( getSpecialPowerModuleData()->m_specialPowerTemplate );
459
460 //If we depend on our update module to trigger the special power, make sure we have the
461 //appropriate update module!
462 if( !valid && getSpecialPowerModuleData()->m_updateModuleStartsAttack )
463 {
464 DEBUG_CRASH( ("Object does not contain a special power module to execute. Did you forget to add it to the object INI?\n"));
465 //DEBUG_CRASH(( "Object does not contain special power module (%s) to execute. Did you forget to add it to the object INI?\n",
466 // command->m_specialPower->getName().str() ));
467 }
468
469 return valid;
470}

485//-------------------------------------------------------------------------------------------------
486void SpecialPowerModule::createViewObject( const Coord3D *location )
487{
488 const SpecialPowerModuleData *modData = getSpecialPowerModuleData();
489 const SpecialPowerTemplate *powerTemplate = modData->m_specialPowerTemplate;
490
491 if( modData == NULL || powerTemplate == NULL )
492 return;
493
494 Real visionRange = powerTemplate->getViewObjectRange();
495 UnsignedInt visionDuration = powerTemplate->getViewObjectDuration();
496
497 if( visionRange == 0 || visionDuration == 0 )
498 return; // We don't want a view object at all.
499
500 AsciiString objectName = TheGlobalData->m_specialPowerViewObjectName;
501 if( objectName.isEmpty() )
502 return;
503
504 const ThingTemplate *viewObjectTemplate = TheThingFactory->findTemplate( objectName );
505 if( viewObjectTemplate == NULL )
506 return;
507
508 Object *viewObject = TheThingFactory->newObject( viewObjectTemplate, getObject()->getControllingPlayer()->getDefaultTeam() );
509
510 if( viewObject == NULL )
511 return;
512
513 viewObject->setPosition( location );
514 viewObject->setShroudClearingRange( visionRange );
515
516 static NameKeyType key_DeletionUpdate = NAMEKEY("DeletionUpdate");
517 DeletionUpdate* dup = (DeletionUpdate*)viewObject->findUpdateModule(key_DeletionUpdate);
518 if( dup )
519 {
520 dup->setLifetimeRange( visionDuration, visionDuration );
521 }
522}

658//-------------------------------------------------------------------------------------------------
659//By default, special powers are not triggered by it's update module -- in which case
660//it triggers it and resets its timer immediately. When the update module triggers it,
661//then all we do is initiate the special power, and trust that the update module will
662//do the rest.
663//-------------------------------------------------------------------------------------------------
664void SpecialPowerModule::doSpecialPower( UnsignedInt commandOptions )
665{
666 if (m_pausedCount > 0 || getObject()->isDisabled()) {
667 return;
668 }
669
670 //This tells the update module that we want to do our special power. The update modules
671 //will then start processing each frame.
672 initiateIntentToDoSpecialPower( NULL, NULL, NULL, commandOptions );
673
674 //Only trigger the special power immediately if the updatemodule doesn't start the attack.
675 //An example of a case that wouldn't trigger immediately is for a unit that needs to
676 //close to range before firing the special attack. A case that would trigger immediately
677 //is the napalm strike. If we don't call this now, it's up to the update module to do so.
678 if( !getSpecialPowerModuleData()->m_updateModuleStartsAttack )
679 {
680 triggerSpecialPower( NULL );// Location-less trigger
681 }
682}

752void SpecialPowerModule::pauseCountdown( Bool pause )
753{
754 if (pause)// If pausing
755 {
756 if( m_pausedCount == 0 )
757 {
758 // Only record this with the first pausing, otherwise upon final unpausing you would get credited the time
759 // between pauses as time served.
760 m_pausedOnFrame = TheGameLogic->getFrame();
761 m_pausedPercent = getPercentReady();
762 }
763
764 ++m_pausedCount;
765 }
766 else if( m_pausedCount > 0 )//Else if unpausing, but only if I am in fact paused, so multiple unpauses don't break.
767 {
768 --m_pausedCount;
769
770 // And only update the ready time if we are fully unpaused now.
771 if( m_pausedCount == 0 )
772 {
773 m_availableOnFrame += (TheGameLogic->getFrame() - m_pausedOnFrame);
774 }
775 }
776} // end pauseCountdown
