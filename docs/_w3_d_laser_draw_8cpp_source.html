116W3DLaserDraw::W3DLaserDraw( Thing *thing, const ModuleData* moduleData ) :
117 DrawModule( thing, moduleData ),
118 m_line3D(NULL),
119 m_texture(NULL),
120 m_textureAspectRatio(1.0f),
121 m_selfDirty(TRUE)
122{
123 Vector3 dummyPos1( 0.0f, 0.0f, 0.0f );
124 Vector3 dummyPos2( 1.0f, 1.0f, 1.0f );
125 Int i;
126
127 const W3DLaserDrawModuleData *data = getW3DLaserDrawModuleData();
128
129 m_texture = WW3DAssetManager::Get_Instance()->Get_Texture( data->m_textureName.str() );
130 if (m_texture)
131 {
132 if (!m_texture->Is_Initialized())
133 m_texture->Init(); //make sure texture is actually loaded before accessing surface.
134
135 SurfaceClass::SurfaceDescription surfaceDesc;
136 m_texture->Get_Level_Description(surfaceDesc);
137 m_textureAspectRatio = (Real)surfaceDesc.Width/(Real)surfaceDesc.Height;
138 }
139
140 //Get the color components for calculation purposes.
141 Real innerRed, innerGreen, innerBlue, innerAlpha, outerRed, outerGreen, outerBlue, outerAlpha;
142 GameGetColorComponentsReal( data->m_innerColor, &innerRed, &innerGreen, &innerBlue, &innerAlpha );
143 GameGetColorComponentsReal( data->m_outerColor, &outerRed, &outerGreen, &outerBlue, &outerAlpha );
144
145 //Make sure our beams range between 1 and the maximum cap.
146#ifdef I_WANT_TO_BE_FIRED
147// srj sez: this data is const for a reason. casting away the constness because we don't like the values
148// isn't an acceptable solution. if you need to constrain the values, do so at parsing time, when
149// it's still legal to modify these values. (In point of fact, there's not even really any reason to limit
150// the numBeams or segments anymore.)
151 data->m_numBeams = __min( __max( 1, data->m_numBeams ), MAX_LASER_LINES );
152 data->m_segments = __min( __max( 1, data->m_segments ), MAX_SEGMENTS );
153 data->m_tilingScalar = __max( 0.01f, data->m_tilingScalar );
154#endif
155
156 //Allocate an array of lines equal to the number of beams * segments
157 m_line3D = NEW SegmentedLineClass *[ data->m_numBeams * data->m_segments ];
158
159 for( int segment = 0; segment < data->m_segments; segment++ )
160 {
161 //We don't care about segment positioning yet until we actually set the position
162
163 // create all the lines we need at the right transparency level
164 for( i = data->m_numBeams - 1; i >= 0; i-- )
165 {
166 int index = segment * data->m_numBeams + i;
167
168 Real red, green, blue, alpha, width;
169
170 if( data->m_numBeams == 1 )
171 {
172 width = data->m_innerBeamWidth;
173 alpha = innerAlpha;
174 red = innerRed * innerAlpha;
175 green = innerGreen * innerAlpha;
176 blue = innerBlue * innerAlpha;
177 }
178 else
179 {
180 //Calculate the scale between min and max values
181 //0 means use min value, 1 means use max value
182 //0.2 means min value + 20% of the diff between min and max
183 Real scale = i / ( data->m_numBeams - 1.0f);
184
185 width = data->m_innerBeamWidth + scale * (data->m_outerBeamWidth - data->m_innerBeamWidth);
186 alpha = innerAlpha + scale * (outerAlpha - innerAlpha);
187 red = innerRed + scale * (outerRed - innerRed) * innerAlpha;
188 green = innerGreen + scale * (outerGreen - innerGreen) * innerAlpha;
189 blue = innerBlue + scale * (outerBlue - innerBlue) * innerAlpha;
190 }
191
192 m_line3D[ index ] = NEW SegmentedLineClass;
193
194 SegmentedLineClass *line = m_line3D[ index ];
195 if( line )
196 {
197 line->Set_Texture( m_texture );
198 line->Set_Shader( ShaderClass::_PresetAdditiveShader ); //pick the alpha blending mode you want - see shader.h for others.
199 line->Set_Width( width );
200 line->Set_Color( Vector3( red, green, blue ) );
201 line->Set_UV_Offset_Rate( Vector2(0.0f, data->m_scrollRate) ); //amount to scroll texture on each draw
202 if( m_texture )
203 {
204 line->Set_Texture_Mapping_Mode(SegLineRendererClass::TILED_TEXTURE_MAP); //this tiles the texture across the line
205 }
206
207 // add to scene
208 W3DDisplay::m_3DScene->Add_Render_Object( line ); //add it to our scene so it gets rendered with other objects.
209
210 // hide the render object until the first time we come to draw it and
211 // set the correct position
212 line->Set_Visible( 0 );
213 }
214
215
216 } // end for i
217
218 } //end segment loop
219
220}

252void W3DLaserDraw::doDrawModule(const Matrix3D* transformMtx)
253{
254 //UnsignedInt currentFrame = TheGameClient->getFrame();
255 const W3DLaserDrawModuleData *data = getW3DLaserDrawModuleData();
256
257 //Get the updatemodule that drives it...
258 Drawable *draw = getDrawable();
259 static NameKeyType key_LaserUpdate = NAMEKEY( "LaserUpdate" );
260 LaserUpdate *update = (LaserUpdate*)draw->findClientUpdateModule( key_LaserUpdate );
261 if( !update )
262 {
263 DEBUG_ASSERTCRASH( 0, ("W3DLaserDraw::doDrawModule() expects its owner drawable %s to have a ClientUpdate = LaserUpdate module.", draw->getTemplate()->getName().str() ));
264 return;
265 }
266
267 //If the update has moved the laser, it requires a reset of the laser.
268 if (update->isDirty() || m_selfDirty)
269 {
270 update->setDirty(false);
271 m_selfDirty = false;
272
273 Vector3 laserPoints[ 2 ];
274
275 for( int segment = 0; segment < data->m_segments; segment++ )
276 {
277 if( data->m_arcHeight > 0.0f && data->m_segments > 1 )
278 {
279 //CALCULATE A CURVED LINE BASED ON TOTAL LENGTH AND DESIRED HEIGHT INCREASE
280 //To do this we will use a portion of the cos wave ranging between -0.25PI
281 //and +0.25PI. 0PI is 1.0 and 0.25PI is 0.70 -- resulting in a somewhat
282 //gentle curve depending on the line height and length. We also have to make
283 //the line *level* for this phase of the calculations.
284
285 //Get the desired direct line
286 Coord3D lineStart, lineEnd, lineVector;
287 lineStart.set( update->getStartPos() );
288 lineEnd.set( update->getEndPos() );
289 //This is critical -- in the case we have sloped lines (at the end, we'll fix it)
290// lineEnd.z = lineStart.z;
291
292 //Get the length of the line
293 lineVector.set( &lineEnd );
294 lineVector.sub( &lineStart );
295 Real lineLength = lineVector.length();
296
297 //Get the middle point (we'll use this to determine how far we are from
298 //that to calculate our height -- middle point is the highest).
299 Coord3D lineMiddle;
300 lineMiddle.set( &lineStart );
301 lineMiddle.add( &lineEnd );
302 lineMiddle.scale( 0.5 );
303
304 //The half length is used to scale with the distance from middle to
305 //get our cos( 0 to 0.25 PI) cos value
306 Real halfLength = lineLength * 0.5f;
307
308 //Now calculate which segment we will use.
309 Real startSegmentRatio = segment / ((Real)data->m_segments);
310 Real endSegmentRatio = (segment + 1.0f) / ((Real)data->m_segments);
311
312 //Offset the segment ever-so-slightly to minimize overlap -- only apply
313 //to segments that are not the start/end point
314 if( segment > 0 )
315 {
316 startSegmentRatio -= data->m_segmentOverlapRatio;
317 }
318 if( segment < data->m_segments - 1 )
319 {
320 endSegmentRatio += data->m_segmentOverlapRatio;
321 }
322
323 //Calculate our start segment position on the *ground*.
324 Coord3D segmentStart, segmentEnd, vector;
325 vector.set( &lineVector );
326 vector.scale( startSegmentRatio );
327 segmentStart.set( &lineStart );
328 segmentStart.add( &vector );
329
330 //Calculate our end segment position on the *ground*.
331 vector.set( &lineVector );
332 vector.scale( endSegmentRatio );
333 segmentEnd.set( &lineStart );
334 segmentEnd.add( &vector );
335
336 //--------------------------------------------------------------------------------
337 //Now at this point, we have our segment line in the level positions that we want.
338 //Calculate the raised height for the start/end segment positions using cosine.
339 //--------------------------------------------------------------------------------
340
341 //Calculate the distance from midpoint for the start positions.
342 vector.set( &lineMiddle );
343 vector.sub( &segmentStart );
344 Real dist = vector.length();
345 Real scaledRadians = dist / halfLength * PI * 0.5f;
346 Real height = cos( scaledRadians );
347 height *= data->m_arcHeight;
348 segmentStart.z += height;
349
350 //Now do the same thing for the end position.
351 vector.set( &lineMiddle );
352 vector.sub( &segmentEnd );
353 dist = vector.length();
354 scaledRadians = dist / halfLength * PI * 0.5f;
355 height = cos( scaledRadians );
356 height *= data->m_arcHeight;
357 segmentEnd.z += height;
358
359 //This makes the laser skim the ground rather than penetrate it!
360 laserPoints[ 0 ].Set( segmentStart.x, segmentStart.y,
361 MAX( segmentStart.z, 2.0f + TheTerrainLogic->getGroundHeight(segmentStart.x, segmentStart.y) ) );
362 laserPoints[ 1 ].Set( segmentEnd.x, segmentEnd.y,
363 MAX( segmentEnd.z, 2.0f + TheTerrainLogic->getGroundHeight(segmentEnd.x, segmentEnd.y) ) );
364
365 }
366 else
367 {
368 //No arc -- way simpler!
369 laserPoints[ 0 ].Set( update->getStartPos()->x, update->getStartPos()->y, update->getStartPos()->z );
370 laserPoints[ 1 ].Set( update->getEndPos()->x, update->getEndPos()->y, update->getEndPos()->z );
371 }
372
373 //Get the color components for calculation purposes.
374 Real innerRed, innerGreen, innerBlue, innerAlpha, outerRed, outerGreen, outerBlue, outerAlpha;
375 GameGetColorComponentsReal( data->m_innerColor, &innerRed, &innerGreen, &innerBlue, &innerAlpha );
376 GameGetColorComponentsReal( data->m_outerColor, &outerRed, &outerGreen, &outerBlue, &outerAlpha );
377
378 for( Int i = data->m_numBeams - 1; i >= 0; i-- )
379 {
380
381 Real alpha, width;
382 int index = segment * data->m_numBeams + i;
383
384 if( data->m_numBeams == 1 )
385 {
386 width = data->m_innerBeamWidth * update->getWidthScale();
387 alpha = innerAlpha;
388 }
389 else
390 {
391 //Calculate the scale between min and max values
392 //0 means use min value, 1 means use max value
393 //0.2 means min value + 20% of the diff between min and max
394 Real scale = i / ( data->m_numBeams - 1.0f);
395 Real ultimateScale = update->getWidthScale();
396 width = (data->m_innerBeamWidth + scale * (data->m_outerBeamWidth - data->m_innerBeamWidth));
397 width *= ultimateScale;
398 alpha = innerAlpha + scale * (outerAlpha - innerAlpha);
399 }
400
401
402 //Calculate the number of times to tile the line based on the height of the texture used.
403 if( m_texture && data->m_tile )
404 {
405 //Calculate the length of the line.
406 Vector3 lineVector;
407 Vector3::Subtract( laserPoints[1], laserPoints[0], &lineVector );
408 Real length = lineVector.Length();
409
410 //Adjust tile factor so texture is NOT stretched but tiled equally in both width and length.
411 Real tileFactor = length/width*m_textureAspectRatio*data->m_tilingScalar;
412
413 //Set the tile factor
414 m_line3D[ index ]->Set_Texture_Tile_Factor( tileFactor ); //number of times to tile texture across each segment
415 }
416
417 m_line3D[ index ]->Set_Width( width );
418 m_line3D[ index ]->Set_Points( 2, &laserPoints[0] );
419 }
420 }
421 }
422
423 return;
424}
