107 m_doneForeverFrame = TheGameLogic->getFrame() + m_stateCountDown;// a failsafe to force a shutdown on schedule

111
112 m_decalsCreated = FALSE;
113 m_nativeClearingRange = 200;// a sensible default
114 m_currentClearingRange = 0;
115
116 m_state = DSCRU_NOT_STARTED_YET;
117// m_shrinkFinished = FALSE;
118// m_shrinkStarted = FALSE;
119
120 Object* obj = getObject();
121 if ( ! obj )
122 return;

124 if ( ! controller )
125 return;
126
127 m_nativeClearingRange = obj->getShroudClearingRange();//capture this before we start monkeying with it
128
129 for (int d = 0; d < GRID_FX_DECAL_COUNT; ++d)
130 {

160void DynamicShroudClearingRangeUpdate::animateGridDecals( void )
161{
162
163 const Coord3D *ctr = getObject()->getPosition();
164 Coord3D pos;
165 pos.z = 0;
166
167 Real radius = m_currentClearingRange + ((m_totalFrames - m_stateCountDown) * 2);
168 Real angle = 0.0f;
169 Real angleInc = (PI*2.0f) / (Real)GRID_FX_DECAL_COUNT;
170 Real opacity = 1.0f - (m_currentClearingRange / m_nativeClearingRange);
171
172 for (int d = 0; d < GRID_FX_DECAL_COUNT; ++d)
173 {
174 pos.x = ctr->x + (sinf(angle) * radius);
175 pos.y = ctr->y + (cosf(angle) * radius);
176
177 pos.x -= ((Int)pos.x)%23;
178 pos.y -= ((Int)pos.y)%23;
179
180 m_gridDecal[d].setPosition(pos);
181 m_gridDecal[d].setOpacity( opacity );
182 angle += angleInc;
183
184 }
185
186
187}

206{
207 if (m_state == DSCRU_SLEEPING) {
208 return UPDATE_SLEEP_NONE;
209 }
210 //Housekeeping-----------------------------------------
211 Object *me = getObject();
212 const DynamicShroudClearingRangeUpdateModuleData *md = getDynamicShroudClearingRangeUpdateModuleData();
213 UnsignedInt currentFrame = TheGameLogic->getFrame();
214
215 if ( ! m_decalsCreated )

218 m_decalsCreated = TRUE;

220 //-----------------------------------------------------

222
223 if( m_stateCountDown <= 0 || currentFrame > m_doneForeverFrame )
224 m_state = DSCRU_DONE_FOREVER;
225 else if ( m_stateCountDown <= m_shrinkStartDeadline )
226 m_state = DSCRU_SHRINKING;
227 else if ( m_stateCountDown <= m_sustainDeadline )
228 m_state = DSCRU_SUSTAINING;
229 else if ( m_stateCountDown <= m_growStartDeadline )
230 m_state = DSCRU_GROWING;
231
232
233 switch (m_state)
234 {
235
236 case DSCRU_NOT_STARTED_YET :
237 {
238 animateGridDecals();
239
240 break;
241 }
242 case DSCRU_GROWING :
243 {
244 animateGridDecals();
245
246 m_currentClearingRange += m_nativeClearingRange / max(1.0f, (Real)md->m_growTime);
247 if (m_currentClearingRange >= m_nativeClearingRange)
248 m_state = DSCRU_SUSTAINING;
249 break;
250 }
251 case DSCRU_SUSTAINING :
252 {
253 m_currentClearingRange = m_nativeClearingRange;
254 killGridDecals();
255 break;
256 }

258 {
259 m_currentClearingRange -= (m_nativeClearingRange-md->m_finalVision) / max(1.0f, (Real)md->m_shrinkTime);
