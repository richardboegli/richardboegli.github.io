44enum DisabledType
45{
46 DISABLED_DEFAULT, //Typical disable -- like systems, things that don't need to run.
47 DISABLED_HACKED, //This unit has been hacked
48 DISABLED_EMP, //This unit has been disabled via electro-magnetic-pulse.
49 DISABLED_HELD, //Special case -- held means it can fire and isHeld checks to make sure ONLY held is set!
50 DISABLED_PARALYZED, //Battle plans have changed, and unit is confused/paralyzed
51 DISABLED_UNMANNED, //Vehicle is unmanned
52 DISABLED_UNDERPOWERED,//Seperate from ScriptUnderpowered, the owning player has insufficient power. Energy status controls this
53 DISABLED_FREEFALL, //This unit has been disabled via being in free fall
54
55 DISABLED_AWESTRUCK,
56 DISABLED_BRAINWASHED,
57 DISABLED_SUBDUED,
58 //These ones are specificially for scripts to enable/reenable!
59 DISABLED_SCRIPT_DISABLED,
60 DISABLED_SCRIPT_UNDERPOWERED,
61
62 DISABLED_COUNT,
63
64 DISABLED_ANY = 65535
65};
