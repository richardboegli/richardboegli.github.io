209Int Pinger::getPing( AsciiString hostname )
210{
211 MutexClass::LockClass m(m_pingMapMutex, 0);
212 if (m.Failed())
213 return false;
214
215 std::map<std::string, Int>::const_iterator it = m_pingMap.find(hostname.str());
216 if (it != m_pingMap.end())
217 return it->second;
218
219 return -1;

228AsciiString Pinger::getPingString( Int timeout )
229{
230 MutexClass::LockClass m(m_pingMapMutex);
231
232 AsciiString pingString;
233 AsciiString tmp;
234 for (std::map<std::string, Int>::const_iterator it = m_pingMap.begin(); it != m_pingMap.end(); ++it)
235 {
236 Int ping = it->second;
237 if (ping < 0 || ping > timeout)
238 ping = timeout;
239 ping = ping * 255 / timeout;
240 tmp.format("%2.2X", ping);
241 pingString.concat(tmp);
242 }
243 return pingString;
244}

248void PingThreadClass::Thread_Function()
249{
250 try {
251 _set_se_translator( DumpExceptionInfo ); // Hook that allows stack trace.
252 PingRequest req;
253
254 WSADATA wsaData;
255
256 // Fire up winsock (prob already done, but doesn't matter)
257 WORD wVersionRequested = MAKEWORD(1, 1);
258 WSAStartup( wVersionRequested, &wsaData );
259
260 while ( running )
261 {
262 // deal with requests
263 if (ThePinger->getRequest(req))
264 {
265 // resolve the hostname
266 const char *hostnameBuffer = req.hostname.c_str();
267 UnsignedInt IP = 0xFFFFFFFF;
268 if (isdigit(hostnameBuffer[0]))
269 {
270 IP = inet_addr(hostnameBuffer);
271 in_addr hostNode;
272 hostNode.s_addr = IP;
273 DEBUG_LOG(("pinging %s - IP = %s\n", hostnameBuffer, inet_ntoa(hostNode) ));
274 }
275 else
276 {
277 HOSTENT *hostStruct;
278 in_addr *hostNode;
279 hostStruct = gethostbyname(hostnameBuffer);
280 if (hostStruct == NULL)
281 {
282 DEBUG_LOG(("pinging %s - host lookup failed\n", hostnameBuffer));
283
284 // Even though this failed to resolve IP, still need to send a
285 // callback.
286 IP = 0xFFFFFFFF; // flag for IP resolve failed
287 }
288 hostNode = (in_addr *) hostStruct->h_addr;
289 IP = hostNode->s_addr;
290 DEBUG_LOG(("pinging %s IP = %s\n", hostnameBuffer, inet_ntoa(*hostNode) ));
291 }
292
293 // do ping
294 Int totalPing = 0;
295 Int goodReps = 0;
296 Int reps = req.repetitions;
297 while (reps-- && running && IP != 0xFFFFFFFF)
298 {
299 Int ping = doPing(IP, req.timeout);
300 if (ping >= 0)
301 {
302 totalPing += ping;
303 ++goodReps;
304 }
305
306 // end our timeslice
307 Switch_Thread();
308 }
309 if (!goodReps)
310 totalPing = -1;
311 else
312 totalPing = totalPing / goodReps;
313
314 PingResponse resp;
315 resp.hostname = req.hostname;
316 resp.avgPing = totalPing;
317 resp.repetitions = goodReps;
318 ThePinger->addResponse(resp);
319 }
320
321 // end our timeslice
322 Switch_Thread();
323 }
324
325 WSACleanup();
326 } catch ( ... ) {
327 DEBUG_CRASH(("Exception in ping thread!"));
328 }
329}

414#define BUFSIZE 8192
415#define DEFAULT_LEN 32
416#define LOOPLIMIT 4
417#define DEFAULT_TTL 64
418
419Int PingThreadClass::doPing(UnsignedInt IP, Int timeout)
420{
421 /*
422 * Initialize default settings
423 */
424
425 IPINFO stIPInfo, *lpstIPInfo;
426 HANDLE hICMP, hICMP_DLL;
427 int i, j, nDataLen, nLoopLimit, nTimeOut, nTTL, nTOS;
428 DWORD dwReplyCount;
430 BOOL fRet, fDontStop;
432
433 nDataLen = DEFAULT_LEN;
434 nLoopLimit = LOOPLIMIT;
435 nTimeOut = timeout;
436 fDontStop = FALSE;
437 lpstIPInfo = NULL;
438 nTTL = DEFAULT_TTL;
439 nTOS = 0;
440
441 Int pingTime = -1; // in case of error
442
443 char achReqData[BUFSIZE];
444 char achRepData[sizeof(ICMPECHO) + BUFSIZE];
445
446
447 HANDLE ( WINAPI *lpfnIcmpCreateFile )( VOID ) = NULL;
448 BOOL ( WINAPI *lpfnIcmpCloseHandle )( HANDLE ) = NULL;
449 DWORD (WINAPI *lpfnIcmpSendEcho)(HANDLE, DWORD, LPVOID, WORD, LPVOID,
450 LPVOID, DWORD, DWORD) = NULL;
451
452
453 /*
454 * Load the ICMP.DLL
455 */
456 hICMP_DLL = LoadLibrary("ICMP.DLL");
457 if (hICMP_DLL == 0)
458 {
459 DEBUG_LOG(("LoadLibrary() failed: Unable to locate ICMP.DLL!\n"));
460 goto cleanup;
461 }
462
463 /*
464 * Get pointers to ICMP.DLL functions
465 */
466 lpfnIcmpCreateFile = (void * (__stdcall *)(void))GetProcAddress( (HINSTANCE)hICMP_DLL, "IcmpCreateFile");
467 lpfnIcmpCloseHandle = (int (__stdcall *)(void *))GetProcAddress( (HINSTANCE)hICMP_DLL, "IcmpCloseHandle");
468 lpfnIcmpSendEcho = (unsigned long (__stdcall *)(void *, unsigned long, void *, unsigned short,
469 void *, void *, unsigned long, unsigned long))GetProcAddress( (HINSTANCE)hICMP_DLL, "IcmpSendEcho" );
470
471 if ((!lpfnIcmpCreateFile) ||
472 (!lpfnIcmpCloseHandle) ||
473 (!lpfnIcmpSendEcho))
474 {
475 DEBUG_LOG(("GetProcAddr() failed for at least one function.\n"));
476 goto cleanup;
477 }
478
479
480 /*
481 * IcmpCreateFile() - Open the ping service
482 */
483 hICMP = (HANDLE) lpfnIcmpCreateFile();
484 if (hICMP == INVALID_HANDLE_VALUE)
485 {
486 DEBUG_LOG(("IcmpCreateFile() failed"));
487 goto cleanup;
488 }
489
490 /*
491 * Init data buffer printable ASCII
492 * 32 (space) through 126 (tilde)
493 */
494 for (j = 0, i = 32; j < nDataLen; j++, i++)
495 {
496 if (i >= 126)
497 i = 32;
498 achReqData[j] = i;
499 }
500
501 /*
502 * Init IPInfo structure
503 */
504 lpstIPInfo = &stIPInfo;
505 stIPInfo.Ttl = nTTL;
506 stIPInfo.Tos = nTOS;
507 stIPInfo.Flags = 0;
508 stIPInfo.OptionsSize = 0;
509 stIPInfo.OptionsData = NULL;
510
511
512
513 /*
514 * IcmpSendEcho() - Send the ICMP Echo Request
515 * and read the Reply
516 */
517 dwReplyCount = lpfnIcmpSendEcho(
518 hICMP,
519 IP,
520 achReqData,
521 nDataLen,
522 lpstIPInfo,
523 achRepData,
524 sizeof(achRepData),
525 nTimeOut);
526 if (dwReplyCount != 0)
527 {
529 DWORD dwStatus;
530
531 pingTime = (*(UnsignedInt *) & (achRepData[8]));
532
533 // I've seen the ping time bigger than the timeout by a little
534 // bit. How lame.
535 if (pingTime > timeout)
536 pingTime = timeout;
537
538 dwStatus = *(DWORD *) & (achRepData[4]);
539 if (dwStatus != IP_SUCCESS)
540 {
541 DEBUG_LOG(("ICMPERR: %d\n", dwStatus));
542 }
543
544 }
545 else
546 {
547 DEBUG_LOG(("IcmpSendEcho() failed: %d\n", dwReplyCount));
548 // Ok we didn't get a packet, just say everything's OK
549 // and the time was -1
550 pingTime = -1;
551 goto cleanup;
552 }
553
554
555 /*
556 * IcmpCloseHandle - Close the ICMP handle
557 */
558 fRet = lpfnIcmpCloseHandle(hICMP);
559 if (fRet == FALSE)
560 {
561 DEBUG_LOG(("Error closing ICMP handle\n"));
562 }
563
564 // Say what you will about goto's but it's handy for stuff like this
565cleanup:
566
567 // Shut down...
568 if (hICMP_DLL)
569 FreeLibrary((HINSTANCE)hICMP_DLL);
570
571 return pingTime;
572}
