51void ParticlePanelParticle::InitPanel( void )
52{
53 CFileFind finder;
54
55 CComboBox *pWnd = (CComboBox*) GetDlgItem(IDC_PSEd_ParticleTypeParticle);
56 if (!pWnd) {
57 return;
58 }
59
60 // first, clear out any items.
61 pWnd->ResetContent();
62
63 std::string findString;
64 findString = PATH;
65 findString += PREFIX;
66 findString += POSTFIX;
67// DEBUG_LOG(("ParticlePanedParticle::InitPanel - looking for textures, search string is '%s'\n", findString.begin()));
68 BOOL bWorkin = finder.FindFile(findString.begin());
69 while (bWorkin) {
70 bWorkin = finder.FindNextFile();
71 pWnd->AddString(finder.GetFileName());
72 }
73 pWnd->AddString("SMUDGE RESERVED"); //smudges don't use textures so we're hardcoding one to tell them apart.
74}

76void ParticlePanelParticle::performUpdate( IN Bool toUI )
77{
78 static char buff[ARBITRARY_BUFF_SIZE];
79 DebugWindowDialog *pParent = (DebugWindowDialog*) GetParent();
80 if (!pParent) {
81 return;
82 }
83
84 { // update Particle parameters
85 CComboBox *pWnd;
86
87 // first Xmin
88 pWnd = (CComboBox*) GetDlgItem(IDC_PSEd_ParticleTypeParticle);
89 if (pWnd) {
90 if (toUI) {
91 pParent->getParticleNameFromSystem(buff, ARBITRARY_BUFF_SIZE - 1);
92 pWnd->SelectString(-1, buff);
93 } else {
94 int curSel = pWnd->GetCurSel();
95 if (curSel >= 0) {
96 pWnd->GetLBText(curSel, buff);
97 pParent->updateParticleNameToSystem(buff);
98 }
99 }
100 }
101 }
102}

129void ParticlePanelDrawable::performUpdate( IN Bool toUI )
130{
131 static char buff[ARBITRARY_BUFF_SIZE];
132 DebugWindowDialog *pParent = (DebugWindowDialog*) GetParent();
133 if (!pParent) {
134 return;
135 }
136
137 { // update Drawable parameters
138 CComboBox *pWnd = (CComboBox*) GetDlgItem(IDC_PSEd_ParticleTypeDrawable);
139 if (pWnd) {
140 if (pWnd->GetCount() == 0) {
141 // This is done here because InitPanel is called before ThingTemplates have been sent over.
142 std::list<std::string>::const_iterator cit;
143 pWnd->AddString(NONE_STRING);
144 const std::list<std::string> &r = pParent->getAllThingTemplates();
145 for (cit = r.begin(); cit != r.end(); ++cit) {
146 pWnd->AddString(cit->begin());
147 }
148 }
149
150
151 if (toUI) {
152 pParent->getDrawableNameFromSystem(buff, ARBITRARY_BUFF_SIZE - 1);
153 pWnd->SelectString(-1, buff);
154 } else {
155 int curSel = pWnd->GetCurSel();
156 if (curSel >= 0) {
157 pWnd->GetLBText(curSel, buff);
158 pParent->updateDrawableNameToSystem(buff);
