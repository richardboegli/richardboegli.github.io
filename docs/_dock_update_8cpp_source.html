66DockUpdate::DockUpdate( Thing *thing, const ModuleData* moduleData ) : UpdateModule( thing, moduleData )
67{
68
69 m_dockOpen = TRUE;
70
71 m_positionsLoaded = FALSE;
72 m_numberApproachPositionBones = -1;
73 m_activeDocker = INVALID_ID;
74 m_dockerInside = FALSE;
75 m_dockCrippled = FALSE;
76
77 const DockUpdateModuleData *md = (const DockUpdateModuleData *)moduleData;
78
79 m_exitPosition.zero();
80 m_dockPosition.zero();
81 m_enterPosition.zero();
82
83 m_numberApproachPositions = md->m_numberApproachPositionsData;
84 if( m_numberApproachPositions != DYNAMIC_APPROACH_VECTOR_FLAG )
85 {
86 // Not dynamic, so make this the size
87 m_approachPositions.resize(m_numberApproachPositions);
88 m_approachPositionOwners.resize(m_numberApproachPositions);
89 m_approachPositionReached.resize(m_numberApproachPositions);
90 }
91 else
92 {
93 //Otherwise, make a default size, and plan on growing it later
94 m_approachPositions.resize(DEFAULT_APPROACH_VECTOR_SIZE);
95 m_approachPositionOwners.resize(DEFAULT_APPROACH_VECTOR_SIZE);
96 m_approachPositionReached.resize(DEFAULT_APPROACH_VECTOR_SIZE);
97 }
98
99 for( Int vectorIndex = 0; vectorIndex < m_approachPositions.size(); ++vectorIndex )
100 {
101 // Whatever size we are, init everything.
102 m_approachPositions[vectorIndex].zero();
103 m_approachPositionOwners[vectorIndex] = INVALID_ID;
104 m_approachPositionReached[vectorIndex] = FALSE;
105 }
106}

112Bool DockUpdate::isClearToApproach( Object const* docker ) const
113{
114 // If we allow infinite approaches, we don't even need to look up. Just say yes.
115 // The reserve code will handle appending a free spot to the end.
116 if( m_numberApproachPositions == DYNAMIC_APPROACH_VECTOR_FLAG )
117 return TRUE;
118
119 ObjectID dockerID = docker->getID();
120
121 for( Int positionIndex = 0; positionIndex < m_approachPositionOwners.size(); ++positionIndex )
122 {
123 if( m_approachPositionOwners[positionIndex] == INVALID_ID )
124 {
125 return TRUE;
126 }
127 if( m_approachPositionOwners[positionIndex] == dockerID )
128 {
129 return TRUE;
130 }
131 }
132
133 return FALSE;
134}

136Bool DockUpdate::reserveApproachPosition( Object* docker, Coord3D *position, Int *index )
137{
138
139 // load dock positions if not loaded yet
140 if( m_positionsLoaded == FALSE )
141 loadDockPositions();
142
143 // sanity
144 if( position == NULL )
145 return FALSE;
146
147 ObjectID dockerID = docker->getID();
148
149 for( Int positionIndex = 0; positionIndex < m_approachPositionOwners.size(); ++positionIndex )
150 {
151 if( m_approachPositionOwners[positionIndex] == dockerID )
152 {
153 *position = computeApproachPosition( positionIndex, docker );
154 *index = positionIndex;
155 return TRUE;
156 }
157 if( m_approachPositionOwners[positionIndex] == INVALID_ID )
158 {
159 m_approachPositionOwners[positionIndex] = dockerID;
160 *position = computeApproachPosition( positionIndex, docker );
161 *index = positionIndex;
162 return TRUE;
163 }
164 }
165
166 // If I make it out of the loop, I am full, so dynamic approach buildings should make a new entry instead of saying no
167 if( m_numberApproachPositions == DYNAMIC_APPROACH_VECTOR_FLAG )
168 {
169 Coord3D zero;
170 zero.zero();
171 m_approachPositions.push_back( zero );
172 m_approachPositionOwners.push_back( INVALID_ID );
173 m_approachPositionReached.push_back( FALSE );
174
175 loadDockPositions();// refresh this new one
176
177 positionIndex = m_approachPositionOwners.size() - 1;// The new last spot
178 m_approachPositionOwners[positionIndex] = dockerID;
179 *position = computeApproachPosition( positionIndex, docker );
180 *index = positionIndex;
181 return TRUE;
182 }
183
184 return FALSE;
185}

187Bool DockUpdate::advanceApproachPosition( Object* docker, Coord3D *position, Int *index )
188{
189 // load dock positions if not loaded yet
190 if( m_positionsLoaded == FALSE )
191 loadDockPositions();
192
193 // sanity
194 if( position == NULL )
195 return FALSE;
196 if( *index <= 0 )
197 return FALSE;
198 if( m_approachPositionOwners[(*index) - 1] != INVALID_ID )
199 return FALSE;
200
201 Int hisIndex = *index;
202 Int previousIndex = hisIndex - 1;
203
204 m_approachPositionOwners[previousIndex] = docker->getID();
205 m_approachPositionReached[previousIndex] = FALSE;
206
207 m_approachPositionOwners[hisIndex] = INVALID_ID;
208 m_approachPositionReached[hisIndex] = FALSE;
209
210 *position = computeApproachPosition( previousIndex, docker );
211 *index = previousIndex;
212 return TRUE;
213}

221Bool DockUpdate::isClearToAdvance( Object const* docker, Int dockerIndex ) const
222{
223 if( dockerIndex < 0 )
224 return FALSE;
225
226 ObjectID dockerID = docker->getID();
227 Bool correctRequest = dockerID == m_approachPositionOwners[dockerIndex];
228 Bool approachReached = m_approachPositionReached[dockerIndex];
229 Bool nextSpotFree = (dockerIndex > 0) && (m_approachPositionOwners[dockerIndex - 1] == INVALID_ID);
230
231 return correctRequest && approachReached && nextSpotFree;

380void DockUpdate::cancelDock( Object* docker )
381{
382 ObjectID dockerID = docker->getID();
383 for( Int positionIndex = 0; positionIndex < m_approachPositionOwners.size(); ++positionIndex )
384 {
385 if( m_approachPositionOwners[positionIndex] == dockerID )
386 {
387 m_approachPositionOwners[positionIndex] = INVALID_ID;
388 m_approachPositionReached[positionIndex] = FALSE;
389 }
390 }
391 if( m_activeDocker == dockerID )
392 {
393 Object *dockingObject = TheGameLogic->findObjectByID(m_activeDocker);
394 m_activeDocker = INVALID_ID;
395 m_dockerInside = FALSE;
396 // clear any model conditions related to docking that may be set on us and them.
397 // (Normal clear is part of each stage, but we won't get there.)
398 ModelConditionFlags clear;
399 clear.set( MODELCONDITION_DOCKING_ENDING );
400 clear.set( MODELCONDITION_DOCKING_BEGINNING );
401 clear.set( MODELCONDITION_DOCKING_ACTIVE );
402 clear.set( MODELCONDITION_DOCKING );
403 getObject()->clearModelConditionFlags( clear );
404 if( dockingObject )
405 dockingObject->clearModelConditionFlags( clear );
406 }

415UpdateSleepTime DockUpdate::update()
416{
417 if( m_activeDocker == INVALID_ID && !m_dockCrippled )
418 {
419 // if setDockCrippled has been called, I will never give enterance permission.
420 for( Int positionIndex = 0; positionIndex < m_approachPositionReached.size(); ++positionIndex )
421 {
422 if( m_approachPositionReached[positionIndex] )
423 {
424 m_activeDocker = m_approachPositionOwners[positionIndex];
425 return UPDATE_SLEEP_NONE;
426 }
427 }
428 }
429 else if ( getObject()->isKindOf( KINDOF_SUPPLY_SOURCE ) )
430 {
431 Object *docker = TheGameLogic->findObjectByID( m_activeDocker );
432 if ( docker && docker->isKindOf( KINDOF_DOZER ) && docker->isKindOf( KINDOF_HARVESTER ))// a worker
433 {
434 ModelConditionFlags test;
435 test.set( MODELCONDITION_DOCKING_BEGINNING );
436 Drawable *dockerDraw = docker->getDrawable();
437 if ( dockerDraw && dockerDraw->getModelConditionFlags().anyIntersectionWith( test ) )
438 dockerDraw->clearModelConditionFlags( MAKE_MODELCONDITION_MASK(MODELCONDITION_MOVING) );
439 }
440 }
441
442
443 return UPDATE_SLEEP_NONE;
444}

446Coord3D DockUpdate::computeApproachPosition( Int positionIndex, Object *forWhom )
447{
448 // load dock positions if not loaded yet
449 if( m_positionsLoaded == FALSE )
450 loadDockPositions();
451
452 Coord3D bestPosition;// This answer is the best, as it includes findPositionAround
453 Coord3D workingPosition;// But if findPositionAround fails, we need to say something.
454
455 FindPositionOptions fpOptions;
456 // Start with the pristine bone, then convert it to the world, then find a clean spot around it.
457
458 Object *us = getObject();
459 us->convertBonePosToWorldPos( &m_approachPositions[positionIndex], NULL, &workingPosition, NULL );
460
461 if( m_numberApproachPositionBones == 0 )
462 {
463 Coord3D ourPosition = *us->getPosition();
464 Coord3D theirPosition = *forWhom->getPosition();
465 // A Boneless building wants to bias towards the caller for the arbitrary position
466 Vector3 offset( theirPosition.x - ourPosition.x, theirPosition.y - ourPosition.y, theirPosition.z - ourPosition.z );
467 offset.Normalize();
468 offset = offset * (us->getGeometryInfo().getMajorRadius() / 2);
469
470 workingPosition.x += offset.X;
471 workingPosition.y += offset.Y;
472 workingPosition.z += offset.Z;
473 }
474
475 fpOptions.minRadius = 0.0f;
476 fpOptions.maxRadius = 100.0f;
477 fpOptions.sourceToPathToDest = forWhom;// This makes it find a place forWhom can get to.
478 if( forWhom->isUsingAirborneLocomotor() )
479 fpOptions.ignoreObject = getObject();// Flyers can ignore us, so they can approach right over us if they want.
480
481 Bool spotFound = ThePartitionManager->findPositionAround( &workingPosition, &fpOptions, &bestPosition );
482
483 if( spotFound)
484 return bestPosition;
485
486 return workingPosition;
487}

491void DockUpdate::loadDockPositions()
492{
493 Object *obj = getObject();
494 Drawable *myDrawable = obj->getDrawable();
495
496 if (myDrawable)
497 {
498 //Patch 1.03 - Kris - Jan 19, 2005
499 //Some of the GLASupplyStash assets still have docking bones in them. When found, the docking positions must be
500 //observed. This occurs when upgrading to fortified structures which still have the bones. This has the negative
501 //side-effect of workers suddenly slowing down their gathering rate. The proper fix would be to remove the bones
502 //from the assets, but the artists could not find the original max files, hence the code solution.
503 if( !obj->isKindOf( KINDOF_IGNORE_DOCKING_BONES ) )
504 {
505
506 myDrawable->getPristineBonePositions( "DockStart", 0, &m_enterPosition, NULL, 1);
507 myDrawable->getPristineBonePositions( "DockAction", 0, &m_dockPosition, NULL, 1);
508 myDrawable->getPristineBonePositions( "DockEnd", 0, &m_exitPosition, NULL, 1);
509 if( m_numberApproachPositions != DYNAMIC_APPROACH_VECTOR_FLAG )
510 {
511 // Dynamic means no bones
512 Coord3D approachBones[DEFAULT_APPROACH_VECTOR_SIZE];
513 m_numberApproachPositionBones = myDrawable->getPristineBonePositions( "DockWaiting", 1, approachBones, NULL, m_numberApproachPositions);
514 if( m_numberApproachPositions == m_approachPositions.size() )//safeguard: will always be true
515 {
516 for( Int copyIndex = 0; copyIndex < m_numberApproachPositions; ++copyIndex )
517 {
518 m_approachPositions[copyIndex] = approachBones[copyIndex];
519 }
520 }
521 }
522 else
523 m_numberApproachPositionBones = 0;
524
525 m_positionsLoaded = TRUE;
526 }
527 else
528 {
529 m_numberApproachPositionBones = 0;
530 m_positionsLoaded = TRUE;
531 }
532 }
533}

556void DockUpdate::xfer( Xfer *xfer )
557{
558
559 // version
560 XferVersion currentVersion = 1;
561 XferVersion version = currentVersion;
562 xfer->xferVersion( &version, currentVersion );
563
564 // call base class
565 UpdateModule::xfer( xfer );
566
567 // enter position
568 xfer->xferCoord3D( &m_enterPosition );
569
570 // dock position
571 xfer->xferCoord3D( &m_dockPosition );
572
573 // exit position
574 xfer->xferCoord3D( &m_exitPosition );
575
576 // # approach positions
577 xfer->xferInt( &m_numberApproachPositions );
578
579 // positions loaded
580 xfer->xferBool( &m_positionsLoaded );
581
582 // approach positions
583 Int vectorSize = m_approachPositions.size();
584 xfer->xferInt( &vectorSize );
585 m_approachPositions.resize(vectorSize);
586 for( Int vectorIndex = 0; vectorIndex < vectorSize; ++vectorIndex )
587 {
588 // Okay, this is cool. On save, the size and a bunch of coords will be written.
589 // on load, vectorSize will be at 0 from the .size, but will then get set
590 // by the xfer, and properly control the number of Coords.
591 xfer->xferCoord3D( &m_approachPositions[vectorIndex] );
592 }
593
594 // approach position owners
595 vectorSize = m_approachPositionOwners.size();
596 xfer->xferInt( &vectorSize );
597 m_approachPositionOwners.resize(vectorSize);
598 for( vectorIndex = 0; vectorIndex < vectorSize; ++vectorIndex )
599 {
600 xfer->xferObjectID( &m_approachPositionOwners[vectorIndex] );
601 }
602
603 // approach positions reached
604 vectorSize = m_approachPositionReached.size();
605 xfer->xferInt( &vectorSize );
606 m_approachPositionReached.resize(vectorSize);
607 for( vectorIndex = 0; vectorIndex < vectorSize; ++vectorIndex )
608 {
609 // Vector of Bool gets packed as bitfield internally
610 Bool unpack = m_approachPositionReached[vectorIndex];
611 xfer->xferBool( &unpack );
612 }
613
614 // active docker
615 xfer->xferObjectID( &m_activeDocker );
616
617 // docker inside
618 xfer->xferBool( &m_dockerInside );
619
620 // docker crippled
621 xfer->xferBool( &m_dockCrippled );
622
623 // dock open
624 xfer->xferBool( &m_dockOpen );
625
626} // end xfer
