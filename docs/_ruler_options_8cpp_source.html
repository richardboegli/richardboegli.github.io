51void RulerOptions::setWidth(Real width)
52{
53 CString buf;
54 // Multiply by 2 because we are changing from radius to diameter.
55 buf.Format("%f", width * 2.0f);
56 if (m_staticThis && !m_staticThis->m_updating) {
57 CWnd *pEdit = m_staticThis->GetDlgItem(IDC_RULER_WIDTH);
58 if (pEdit && pEdit->IsWindowEnabled()) {
59 pEdit->SetWindowText(buf);
60 }
61 }

66BOOL RulerOptions::OnInitDialog()
67{
68 CDialog::OnInitDialog();
69
70 m_staticThis = this;
71 m_updating = false;
72 if (RulerTool::getType() != RULER_CIRCLE) {
73 CWnd *pEdit = GetDlgItem(IDC_RULER_WIDTH);
74 if (pEdit) {
75 // Disable the edit box since the ruler isn't a circle.
76 pEdit->EnableWindow(false);
77 }
78 } else {
79 // Only the circle rulers use the edit box to change the size.
80 setWidth(RulerTool::getLength());
81 }
82 return TRUE; // return TRUE unless you set the focus to a control
83 // EXCEPTION: OCX Property Pages should return FALSE
84}

86void RulerOptions::OnChangeWidthEdit()
87{
88 if (m_updating) return;
89 CWnd *pEdit = GetDlgItem(IDC_RULER_WIDTH);
90 char buffer[_MAX_PATH];
91 if (pEdit) {
92 pEdit->GetWindowText(buffer, sizeof(buffer));
93 float width;
94 m_updating = true;
95 // Pull out the length from the text.
96 if (1 == sscanf(buffer, "%f", &width)) {
97 // Change from diameter to radius.
98 RulerTool::setLength(width / 2.0f);
99 }
100 m_updating = false;
101 }
102}
103
104void RulerOptions::OnChangeCheckRuler()
105{
106 if (m_updating) return;
107 CWnd *pCheck = GetDlgItem(IDC_CHECK_RULER);
108 if (pCheck) {
109 if (RulerTool::switchType()) {
110 CWnd *pEdit = GetDlgItem(IDC_RULER_WIDTH);
111 if (pEdit) {
112 // The edit box is disabled when the ruler is a line.
113 pEdit->EnableWindow(!pEdit->IsWindowEnabled());
114 // Make sure that the length is updated.
115 if (pEdit->IsWindowEnabled()) {
116 setWidth(RulerTool::getLength());
117 }
118 }
119 }
120 }
121}
