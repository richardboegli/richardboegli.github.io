157extern "C" {
158 static int getQR2HostingStatus(void) { return 0; }
159}
160extern int isThreadHosting;

162Bool handleInGameSlashCommands(UnicodeString uText)
163{
164 AsciiString message;
165 message.translate(uText);
166
167 if (message.getCharAt(0) != '/')
168 {
169 return FALSE; // not a slash command
170 }
171
172 AsciiString remainder = message.str() + 1;
173 AsciiString token;
174 remainder.nextToken(&token);
175 token.toLower();
176
177 if (token == "host")
178 {
179 UnicodeString s;
180 s.format(L"Hosting qr2:%d thread:%d", getQR2HostingStatus(), isThreadHosting);
181 TheInGameUI->message(s);
182 return TRUE; // was a slash command
183 }
184
185 return FALSE; // not a slash command
186}

190void ToggleInGameChat( Bool immediate )
191{
192 static Bool justHid = false;
193 if (justHid)
194 {
195 justHid = false;
196 return;
197 }
198
199 if (TheGameLogic->isInReplayGame())
200 return;
201
202 if (!TheGameInfo->isMultiPlayer() && TheGlobalData->m_netMinPlayers)
203 return;
204
205 if (chatWindow)
206 {
207 Bool show = chatWindow->winIsHidden();
208 if (show)
209 ShowInGameChat( immediate );
210 else
211 {
212 if (chatTextEntry)
213 {
214 // Send what is there, clear it out, and hide the window
215 UnicodeString msg = GadgetTextEntryGetText( chatTextEntry );
216 msg.trim();
217 if (!msg.isEmpty() && !handleInGameSlashCommands(msg))
218 {
219 const Player *localPlayer = ThePlayerList->getLocalPlayer();
220 AsciiString playerName;
221 Int playerMask = 0;
222
223 for (Int i=0; i<MAX_SLOTS; ++i)
224 {
225 playerName.format("player%d", i);
226 const Player *player = ThePlayerList->findPlayerWithNameKey( TheNameKeyGenerator->nameToKey( playerName ) );
227 if (player && localPlayer)
228 {
229 switch (inGameChatType)
230 {
231 case INGAME_CHAT_EVERYONE:
232 if (!TheGameInfo->getConstSlot(i)->isMuted())
233 playerMask |= (1<<i);
234 break;
235 case INGAME_CHAT_ALLIES:
236 if ( (player->getRelationship(localPlayer->getDefaultTeam()) == ALLIES &&
237 localPlayer->getRelationship(player->getDefaultTeam()) == ALLIES) || player==localPlayer )
238 playerMask |= (1<<i);
239 break;
240 case INGAME_CHAT_PLAYERS:
241 if ( player == localPlayer )
242 playerMask |= (1<<i);
243 break;
244 }
245 }
246 }
247 TheLanguageFilter->filterLine(msg);
248 TheNetwork->sendChat(msg, playerMask);
249 }
250 GadgetTextEntrySetText( chatTextEntry, UnicodeString::TheEmptyString );
251 HideInGameChat( immediate );
252 justHid = true;
253 }
254 }
255 }
256 else
257 {
258 ShowInGameChat( immediate );
259 }
260}
