66// drawRadioButtonText ========================================================
68//=============================================================================
69static void drawRadioButtonText( GameWindow *window, WinInstanceData *instData )
70{
71 ICoord2D origin, size, textPos;
72 Int width, height;
73 Color textColor, dropColor;
74 DisplayString *text = instData->getTextDisplayString();
75
76 // sanity
77 if( text == NULL || text->getTextLength() == 0 )
78 return;
79
80 // get window position and size
81 window->winGetScreenPosition( &origin.x, &origin.y );
82 window->winGetSize( &size.x, &size.y );
83
84 // get the right text color
85 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
86 {
87 textColor = window->winGetDisabledTextColor();
88 dropColor = window->winGetDisabledTextBorderColor();
89 } // end if, disabled
90 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
91 {
92 textColor = window->winGetHiliteTextColor();
93 dropColor = window->winGetHiliteTextBorderColor();
94 } // end else if, hilited
95 else
96 {
97 textColor = window->winGetEnabledTextColor();
98 dropColor = window->winGetEnabledTextBorderColor();
99 } // end enabled only
100
101 // set our font to that of our parent if not the same
102 if( text->getFont() != window->winGetFont() )
103 text->setFont( window->winGetFont() );
104
105 // get text size
106 text->getSize( &width, &height );
107
108 // set the location for our text
109 textPos.x = origin.x + (size.x / 2) - (width / 2);
110 textPos.y = origin.y + (size.y / 2) - (height / 2);
111
112 // draw it
113 text->draw( textPos.x, textPos.y, textColor, dropColor );
114
115} // end drawRadioButtonText

126void W3DGadgetRadioButtonDraw( GameWindow *window, WinInstanceData *instData )
127{
128 Int checkOffsetFromLeft;
129 Color backColor,
130 backBorder,
131 boxColor,
132 boxBorder;
133 ICoord2D origin, size, start, end;
134
135 // get window position and size
136 window->winGetScreenPosition( &origin.x, &origin.y );
137 window->winGetSize( &size.x, &size.y );
138
139 // compute start of check offset
140 checkOffsetFromLeft = size.x / 16;
141
142 //
143 // get the colors we should be using to draw, see GadgetRadioButton.h
144 // draw appropriate state, see GadgetRadioButton.h for info
145 //
146 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
147 {
148
149 // disabled background
150 backColor = GadgetRadioGetDisabledColor( window );
151 backBorder = GadgetRadioGetDisabledBorderColor( window );
152
153 // check box
154 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
155 {
156 boxColor = GadgetRadioGetDisabledCheckedBoxColor( window );
157 boxBorder = GadgetRadioGetDisabledCheckedBoxBorderColor( window );
158 }
159 else
160 {
161 boxColor = GadgetRadioGetDisabledUncheckedBoxColor( window );
162 boxBorder = GadgetRadioGetDisabledUncheckedBoxBorderColor( window );
163 }
164
165 } // end if
166 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
167 {
168
169 // hilited background
170 backColor = GadgetRadioGetHiliteColor( window );
171 backBorder = GadgetRadioGetHiliteBorderColor( window );
172
173 // check box
174 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
175 {
176 boxColor = GadgetRadioGetHiliteCheckedBoxColor( window );
177 boxBorder = GadgetRadioGetHiliteCheckedBoxBorderColor( window );
178 }
179 else
180 {
181 boxColor = GadgetRadioGetHiliteUncheckedBoxColor( window );
182 boxBorder = GadgetRadioGetHiliteUncheckedBoxBorderColor( window );
183 }
184
185 } // end else if
186 else
187 {
188
189 // enabled background
190 backColor = GadgetRadioGetEnabledColor( window );
191 backBorder = GadgetRadioGetEnabledBorderColor( window );
192
193 // check box
194 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
195 {
196 boxColor = GadgetRadioGetEnabledCheckedBoxColor( window );
197 boxBorder = GadgetRadioGetEnabledCheckedBoxBorderColor( window );
198 }
199 else
200 {
201 boxColor = GadgetRadioGetEnabledUncheckedBoxColor( window );
202 boxBorder = GadgetRadioGetEnabledUncheckedBoxBorderColor( window );
203 }
204
205 } // end else
206
207 // draw background border
208 start.x = origin.x;
209 start.y = origin.y;
210 end.x = start.x + size.x;
211 end.y = start.y + size.y;
212 TheWindowManager->winOpenRect( backBorder, WIN_DRAW_LINE_WIDTH,
213 start.x, start.y, end.x, end.y );
214
215 // draw the background
216 start.x++;
217 start.y++;
218 end.x--;
219 end.y--;
220 TheWindowManager->winFillRect( backColor, WIN_DRAW_LINE_WIDTH,
221 start.x, start.y, end.x, end.y );
222
223
224
225 // draw box border
226 start.x = origin.x + size.y;
227 start.y = origin.y;
228 end.x = start.x;
229 end.y = start.y + size.y;
230 TheWindowManager->winDrawLine( backBorder, WIN_DRAW_LINE_WIDTH,
231 start.x, start.y, end.x, end.y );
232
233 // draw box for button
234 start.x = origin.x + 1;
235 start.y = origin.y + 1;
236 end.x = origin.x + size.y -1;
237 end.y = origin.y + size.y -1;
238 TheWindowManager->winFillRect( boxColor, WIN_DRAW_LINE_WIDTH,
239 start.x, start.y, end.x, end.y );
240
241 // draw box border
242 start.x = origin.x + size.x - size.y;
243 start.y = origin.y;
244 end.x = start.x;
245 end.y = start.y + size.y;
246 TheWindowManager->winDrawLine( backBorder, WIN_DRAW_LINE_WIDTH,
247 start.x, start.y, end.x, end.y );
248
249 // draw box for button
250 start.x = origin.x + size.x - size.y;
251 start.y = origin.y + 1;
252 end.x = origin.x + size.x -1;
253 end.y = origin.y + size.y -1;
254 TheWindowManager->winFillRect( boxColor, WIN_DRAW_LINE_WIDTH,
255 start.x, start.y, end.x, end.y );
256 // draw the button text
257 if( instData->getTextLength() )
258 drawRadioButtonText( window, instData );
259
260
261
262} // end W3DGadgetRadioButtonDraw

265void W3DGadgetRadioButtonImageDraw( GameWindow *window,
266 WinInstanceData *instData )
267{
268 const Image *leftImage, *rightImage, *centerImage;
269 ICoord2D origin, size, start, end;
270 Int xOffset, yOffset;
271 Int i;
272
273 // get screen position and size
274 window->winGetScreenPosition( &origin.x, &origin.y );
275 window->winGetSize( &size.x, &size.y );
276
277 IRegion2D clipLeft;
278
279 // get image offset
280 xOffset = instData->m_imageOffset.x;
281 yOffset = instData->m_imageOffset.y;
282
283 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
284 {
285 //backgroundImage = GadgetRadioGetEnabledCheckedBoxImage( window );
286 leftImage = GadgetRadioGetSelectedImage( window );
287 centerImage = GadgetRadioGetSelectedUncheckedBoxImage( window );
288 rightImage = GadgetRadioGetSelectedCheckedBoxImage( window );
289
290 }
291 else if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
292 {
293 // disabled background
294 leftImage = GadgetRadioGetDisabledImage( window );
295 centerImage = GadgetRadioGetDisabledUncheckedBoxImage( window );
296 rightImage = GadgetRadioGetDisabledCheckedBoxImage( window );
297
298 } // end if
299 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
300 {
301 // hilited background
302 leftImage = GadgetRadioGetHiliteImage( window );
303 centerImage = GadgetRadioGetHiliteUncheckedBoxImage( window );
304 rightImage = GadgetRadioGetHiliteCheckedBoxImage( window );
305
306 } // end else if
307 else
308 {
309 // enabled background
310 leftImage = GadgetRadioGetEnabledImage( window );
311 centerImage = GadgetRadioGetEnabledUncheckedBoxImage( window );
312 rightImage = GadgetRadioGetEnabledCheckedBoxImage( window );
313
314 } // end else
315
316 // sanity, we need to have these images to make it look right
317 if( leftImage == NULL || centerImage == NULL ||
318 rightImage == NULL )
319 return;
320
321 // get image sizes for the ends
322 ICoord2D leftSize, rightSize;
323 leftSize.x = leftImage->getImageWidth();
324 leftSize.y = leftImage->getImageHeight();
325 rightSize.x = rightImage->getImageWidth();
326 rightSize.y = rightImage->getImageHeight();
327
328 // get two key points used in the end drawing
329 ICoord2D leftEnd, rightStart;
330 leftEnd.x = origin.x + leftSize.x + xOffset;
331 leftEnd.y = origin.y + size.y + yOffset;
332 rightStart.x = origin.x + size.x - rightSize.x + xOffset;
333 rightStart.y = origin.y + size.y + yOffset;
334
335
336
337 // draw the center repeating bar
338 Int centerWidth, pieces;
339
340 // get width we have to draw our repeating center in
341 centerWidth = rightStart.x - leftEnd.x;
342
343 // how many whole repeating pieces will fit in that width
344 pieces = centerWidth / centerImage->getImageWidth();
345 pieces++;
346 // draw the pieces
347 start.x = leftEnd.x;
348 start.y = origin.y + yOffset;
349 end.y =origin.y + size.y + yOffset;
350
351 clipLeft.lo.x = leftEnd.x;
352 clipLeft.lo.y = origin.y;
353 clipLeft.hi.y = leftEnd.y;
354 clipLeft.hi.x = rightStart.x ;
355
356
357 TheDisplay->setClipRegion(&clipLeft);
358
359 for( i = 0; i < pieces; i++ )
360 {
361 end.x = start.x + centerImage->getImageWidth();
362 TheWindowManager->winDrawImage( centerImage,
363 start.x, start.y,
364 end.x, end.y );
365 start.x += centerImage->getImageWidth();
366 } // end for i
367
368 TheDisplay->enableClipping(FALSE);
369 // draw left end
370 start.x = origin.x + xOffset;
371 start.y = origin.y + yOffset;
372 end = leftEnd;
373 TheWindowManager->winDrawImage(leftImage, start.x, start.y, end.x, end.y);
374 // draw right end
375 start.x = rightStart.x;
376 start.y = origin.y + yOffset;
377 end.x = origin.x + size.x;
378 end.y = leftEnd.y;
379 TheWindowManager->winDrawImage(rightImage, start.x, start.y, end.x, end.y);
380
381 // draw the text
382 if( instData->getTextLength() )
383 drawRadioButtonText( window, instData );
384
385
386} // end W3DGadgetHorizontalSliderImageDraw
