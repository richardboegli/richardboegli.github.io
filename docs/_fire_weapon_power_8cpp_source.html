101void FireWeaponPower::doSpecialPower( UnsignedInt commandOptions )
102{
103 Object *self = getObject();
104 const FireWeaponPowerModuleData *data = getFireWeaponPowerModuleData();
105
106 if( self->isDisabled() )
107 return;
108
109 // call the base class action cause we are *EXTENDING* functionality
110 SpecialPowerModule::doSpecialPower( commandOptions );
111
112 self->reloadAllAmmo( TRUE );
113
114 AIUpdateInterface *ai = self->getAI();
115 if( ai )
116 {
117 ai->aiAttackPosition( NULL, data->m_maxShotsToFire, CMD_FROM_AI );
118
119 //Order any turrets to attack as well.
120 for( Int i = 0; i < MAX_TURRETS; i++ )
121 {
122 ai->setTurretTargetPosition( (WhichTurretType)i, self->getPosition() );
123 }
124 }
125}
