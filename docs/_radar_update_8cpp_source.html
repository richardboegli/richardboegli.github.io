74void RadarUpdate::extendRadar( void )
75{
76 const RadarUpdateModuleData *modData = getRadarUpdateModuleData();
77
78 // set the model condition for radar extension
79 Drawable *draw = getObject()->getDrawable();
80 if( draw )
81 draw->setModelConditionState( MODELCONDITION_RADAR_EXTENDING );
82
83 // mark the frame that the extension will be done on
84 m_extendDoneFrame = TheGameLogic->getFrame() + modData->m_radarExtendTime;
85
86 //Change this to make the radar active after extension...
87 m_radarActive = true;
88
89} // end extendRadar

93UpdateSleepTime RadarUpdate::update( void )
94{
96
97 // if no extend frame nothing to do
98 if( m_extendDoneFrame == 0 )
99 return UPDATE_SLEEP_NONE;
100
101 // check to see if our extension is already done
102 if( m_extendComplete == TRUE )
103 return UPDATE_SLEEP_NONE;
104
105 // see if it's time to stop the extension
106 if( TheGameLogic->getFrame() > m_extendDoneFrame )
107 {
108
109 // mark extend as done
110 m_extendComplete = TRUE;
111 m_extendDoneFrame = 0; // just to be clean
112
113 // remove the extending condition and set the extened condition
114 Drawable *draw = getObject()->getDrawable();
115 if( draw )
116 draw->clearAndSetModelConditionState( MODELCONDITION_RADAR_EXTENDING,
117 MODELCONDITION_RADAR_UPGRADED );
118
119 } // end if
120
121 return UPDATE_SLEEP_NONE;
122
123} // end update
