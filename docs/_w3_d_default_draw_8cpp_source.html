54W3DDefaultDraw::W3DDefaultDraw(Thing *thing, const ModuleData* moduleData) : DrawModule(thing, moduleData)
55{
56#ifdef LOAD_TEST_ASSETS
57 m_renderObject = NULL;
58 m_shadow = NULL;
59 if (!getDrawable()->getTemplate()->getLTAName().isEmpty())
60 {
61 m_renderObject = W3DDisplay::m_assetManager->Create_Render_Obj(getDrawable()->getTemplate()->getLTAName().str(), getDrawable()->getScale(), 0);
62
63 Shadow::ShadowTypeInfo shadowInfo;
64 shadowInfo.m_type=(ShadowType)SHADOW_VOLUME;
65 shadowInfo.m_sizeX=0; //use defaults
66 shadowInfo.m_sizeY=0;
67 shadowInfo.m_offsetX=0;
68 shadowInfo.m_offsetY=0;
69 m_shadow = TheW3DShadowManager->addShadow(m_renderObject, &shadowInfo);
70
71
72 DEBUG_ASSERTCRASH(m_renderObject, ("Test asset %s not found", getDrawable()->getTemplate()->getLTAName().str()));
73 if (m_renderObject)
74 {
75
76 W3DDisplay::m_3DScene->Add_Render_Object(m_renderObject);
77
78 m_renderObject->Set_User_Data(getDrawable()->getDrawableInfo());
79
80 Matrix3D transform;
83// transform.Set(Vector3(0,0,9999));
84 transform.Set(Vector3(0,0,0));
85 m_renderObject->Set_Transform(transform);
86 }
87 }
88#endif
89}

105W3DDefaultDraw::~W3DDefaultDraw(void)
106{
107#ifdef LOAD_TEST_ASSETS
108 if (TheW3DShadowManager && m_shadow)
109 {
110 TheW3DShadowManager->removeShadow(m_shadow);
111 m_shadow = NULL;
112 }
113 if (m_renderObject)
114 {
115 W3DDisplay::m_3DScene->Remove_Render_Object(m_renderObject);
116 REF_PTR_RELEASE(m_renderObject);
117 m_renderObject = NULL;
118 }
119#endif
120}

141void W3DDefaultDraw::doDrawModule(const Matrix3D* transformMtx)
142{
143#ifdef LOAD_TEST_ASSETS
144 if(m_renderObject)
145 {
146 Matrix3D scaledTransform;
147 if (getDrawable()->getInstanceScale() != 1.0f)
148 { //do custom scaling of the W3D model.
149 scaledTransform=*transformMtx;
150 scaledTransform.Scale(getDrawable()->getInstanceScale());
151 transformMtx = &scaledTransform;
152 m_renderObject->Set_ObjectScale(getDrawable()->getInstanceScale());
153 }
154 else
155 {
156 m_renderObject->Set_Transform(*transformMtx);
157 }
158 }
159#endif
160
161 return;
162}
