74// ImagePacker::createNewTexturePage ==========================================
76//=============================================================================
77TexturePage *ImagePacker::createNewTexturePage( void )
78{
79 TexturePage *page;
80
81 // allocate new page
82 page = new TexturePage( getTargetWidth(), getTargetHeight() );
83 if( page == NULL )
84 {
85
86 DEBUG_ASSERTCRASH( page, ("Unable to allocate new texture page.\n") );
87 return NULL;
88
89 } // end if
90
91 // link page to list
92 page->m_prev = NULL;
93 page->m_next = m_pageList;
94 if( m_pageList )
95 m_pageList->m_prev = page;
96 m_pageList = page;
97
98 // add the tail pointer if this is the first page
99 if( m_pageTail == NULL )
100 m_pageTail = page;
101
102 // we got a new page now
103 m_pageCount++;
104
105 // set page id as the current page count
106 page->setID( m_pageCount );
107
108 return page;
109
110} // end createNewTexturePage

112// ImagePacker::validateImages ================================================
121//=============================================================================
122Bool ImagePacker::validateImages( void )
123{
124 UnsignedInt i;
125 ImageInfo *image;
126 Bool errors = FALSE;
127 Bool proceed = TRUE;
128
129 // loop through all images
130 for( i = 0; i < m_imageCount; i++ )
131 {
132
133 // get this image
134 image = m_imageList[ i ];
135
136 // sanity
137 if( image == NULL )
138 {
139
140 DEBUG_ASSERTCRASH( image, ("Image in imagelist is NULL") );
141 continue; // should never happen
142
143 } // end if
144
145 //
146 // if this image is too big to fit in the target page size as a whole
147 // then there is nothing we can do about it
148 //
149 if( image->m_size.x > getTargetWidth() ||
150 image->m_size.y > getTargetHeight() )
151 {
152
153 errors = TRUE;
154 BitSet( image->m_status, ImageInfo::TOOBIG );
155 BitSet( image->m_status, ImageInfo::CANTPROCESS );
156
157 } // end if
158
159 //
160 // if this image is not the right format we can't process it, at
161 // present we only understand 32 and 24 bit images
162 //
163 if( image->m_colorDepth != 32 && image->m_colorDepth != 24 )
164 {
165
166 errors = TRUE;
167 BitSet( image->m_status, ImageInfo::INVALIDCOLORDEPTH );
168 BitSet( image->m_status, ImageInfo::CANTPROCESS );
169
170 } // end if
171
172 } // end for i
173
174 //
175 // if we have errors, build a list and show them to the user
176 //
177 if( errors == TRUE )
178 {
179
180 proceed = DialogBox( ApplicationHInstance,
181 (LPCTSTR)IMAGE_ERRORS,
182 TheImagePacker->getWindowHandle(),
183 (DLGPROC)ImageErrorProc );
184
185 } // end if
186
187 return proceed;
188
189} // end validateImages

191// ImagePacker::packImages ====================================================
194//=============================================================================
195Bool ImagePacker::packImages( void )
196{
197 UnsignedInt i;
198 TexturePage *page = NULL;
199 ImageInfo *image = NULL;
200
201 //
202 // first sanity check all images loaded, if there are images that cannot
203 // be processed the user will be given a list of these and asked wether
204 // or not to proceed
205 //
206 Bool proceed;
207 proceed = validateImages();
208 if( proceed == FALSE )
209 {
210
211 statusMessage( "Build Cancelled By User." );
212 return FALSE;
213
214 } // end if
215
216 // loop through all images
217 for( i = 0; i < m_imageCount; i++ )
218 {
219
220 // update status
221 sprintf( m_statusBuffer, "Fitting Image %d of %d.", i, m_imageCount );
222 statusMessage( m_statusBuffer );
223
224 // get this image out of the list
225 image = m_imageList[ i ];
226
227 // ignore images that we cannot process
228 if( BitTest( image->m_status, ImageInfo::CANTPROCESS) )
229 continue;
230
231 // try to put image on each page
232 for( page = m_pageTail; page; page = page->m_prev )
233 {
234
235 if( page->addImage( image ) == TRUE )
236 break; // page added, stop trying to add into pages
237
238 } // end for page
239
240 // if image was not able to go on any existing page create a new page for it
241 if( page == NULL )
242 {
243
244 page = createNewTexturePage();
245 if( page == NULL )
246 return FALSE;
247
248 // try to add the image to this page
249 if( page->addImage( image ) == FALSE )
250 {
251 char buffer[ _MAX_PATH ];
252
253 sprintf( buffer, "Unable to add image '%s' to a brand new page!\n", image->m_path );
254 DEBUG_ASSERTCRASH( 0, (buffer) );
255 MessageBox( NULL, buffer, "Internal Error", MB_OK | MB_ICONERROR );
256 return FALSE;
257
258 } // end if
259
260 } // end if
261
262 } // end for i
263
264 return TRUE; // success
265
266} // end packImages

268// ImagePacker::writeFinalTextures ============================================
272//=============================================================================
273void ImagePacker::writeFinalTextures( void )
274{
275 TexturePage *page;
276 Bool errors = FALSE;
277 char buffer[ 128 ];
278
279 //
280 // go through each page, let's start from the end of the list since
281 // that's where we packed first, but it doesn't matter
282 //
283 for( page = m_pageTail; page; page = page->m_prev )
284 {
285
286 // update status message
287 sprintf( buffer, "Generating texture #%d of %d.",
288 page->getID(), m_pageCount );
289 statusMessage( buffer );
290
291 // generate the final texture for this page
292 if( page->generateTexture() == FALSE )
293 {
294
295 errors = TRUE;
296 continue; // could not generate this page, but try to continue
297
298 } // end if
299
300 //
301 // write this page out to a file using the filename given by
302 // the user and the texture page ID to keep it unique
303 //
304 if( page->writeFile( m_outputFile ) == FALSE )
305 {
306
307 errors = TRUE;
308 continue; // could not write page, but try to go on
309
310 } // end if
311
312 } // end for page
313
314 // check for any errors and notify the user
315 if( errors == TRUE )
316 {
317
318 DialogBox( ApplicationHInstance,
319 (LPCTSTR)PAGE_ERRORS,
320 TheImagePacker->getWindowHandle(),
321 (DLGPROC)PageErrorProc );
322
323 } // end if
324
325} // end writeFinalTextures

327// sortImageCompare ===========================================================
333//=============================================================================
334static Int sortImageCompare( const void *aa, const void *bb )
335{
336 const ImageInfo **a = (const ImageInfo **)aa;
337 const ImageInfo **b = (const ImageInfo **)bb;
338
339 if( (*a)->m_area < (*b)->m_area )
340 return 1;
341 else if( (*a)->m_area > (*b)->m_area )
342 return -1;
343 else
344 return 0;
345
346} // end sortImageCompare

615// ImagePacker::addDirectory ==================================================

622void ImagePacker::addDirectory( char *path, Bool subDirs )
623{
624 char currDir[ _MAX_PATH ];
625 WIN32_FIND_DATA item; // search item
626 HANDLE hFile; // handle for search resources
627
628 // santiy
629 if( path == NULL )
630 return;
631
632 // check to see if path is already in list
633 ImageDirectory *dir;
634 for( dir = m_dirList; dir; dir = dir->m_next )
635 if( stricmp( dir->m_path, path ) == 0 )
636 return; // already in list
637
638 // save our current directory
639 GetCurrentDirectory( _MAX_PATH, currDir );
640
641 // set our directory to this one
642 if( SetCurrentDirectory( path ) == 0 )
643 return; // directory does not exist
644
645 // image is not in list, make a new entry and link to the list
646 dir = new ImageDirectory;
647 if( dir == NULL )
648 {
649
650 MessageBox( NULL, "Unable to allocate image directory", "Error",
651 MB_OK | MB_ICONERROR );
652 return;
653
654 } // end if
655
656 // allocate space for the path
657 Int len = strlen( path );
658 dir->m_path = new char[ len + 1 ];
659 strcpy( dir->m_path, path );
660 if( dir->m_path == NULL )
661 {
662
663 MessageBox( NULL, "Unable to allocate path for directory", "Error",
664 MB_OK | MB_ICONERROR );
665 delete dir;
666 return;
667
668 } // end if
669
670 // tie to list
671 dir->m_prev = NULL;
672 dir->m_next = m_dirList;
673 if( m_dirList )
674 m_dirList->m_prev = dir;
675 m_dirList = dir;
676
677 // increase our directory count
678 m_dirCount++;
679
680 // update status
681 sprintf( m_statusBuffer, "Folder Added: %d.", m_dirCount );
682 statusMessage( m_statusBuffer );
683
684 // count how many image files are in this directory
685 hFile = FindFirstFile( "*", &item);
686 if( hFile != INVALID_HANDLE_VALUE )
687 {
688
689 // if this is a file count it
690 if( !(item.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
691 strcmp( item.cFileName, "." ) &&
692 strcmp( item.cFileName, ".." ) )
693 {
694
695 len = strlen( item.cFileName );
696 if( len > 4 &&
697 item.cFileName[ len - 4 ] == '.' &&
698 (item.cFileName[ len - 3 ] == 't' || item.cFileName[ len - 3 ] == 'T') &&
699 (item.cFileName[ len - 2 ] == 'g' || item.cFileName[ len - 2 ] == 'G') &&
700 (item.cFileName[ len - 1 ] == 'a' || item.cFileName[ len - 1 ] == 'A') )
701 dir->m_imageCount++;
702
703 } // end if
704
705 // find the rest of the files
706 while( FindNextFile( hFile, &item ) != 0 )
707 {
708
709 // if this is a file count it
710 if( !(item.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
711 strcmp( item.cFileName, "." ) &&
712 strcmp( item.cFileName, ".." ) )
713 {
714
715 len = strlen( item.cFileName );
716 if( len > 4 &&
717 item.cFileName[ len - 4 ] == '.' &&
718 (item.cFileName[ len - 3 ] == 't' || item.cFileName[ len - 3 ] == 'T') &&
719 (item.cFileName[ len - 2 ] == 'g' || item.cFileName[ len - 2 ] == 'G') &&
720 (item.cFileName[ len - 1 ] == 'a' || item.cFileName[ len - 1 ] == 'A') )
721 dir->m_imageCount++;
722
723 } // end if
724
725 } // end while
726
727 // close search
728 FindClose( hFile );
729
730 } //end if, items found
731
732 // add the image count of this directory to the total image count
733 m_imagesInDirs += dir->m_imageCount;
734
735 // if we are adding subdirectories add them all
736 if( subDirs )
737 {
738 char subDir[ _MAX_PATH ];
739
740 // go through each item in the output directory
741 hFile = FindFirstFile( "*", &item);
742 if( hFile != INVALID_HANDLE_VALUE )

744
745 // if this is a file count it
746 if( (item.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
747 strcmp( item.cFileName, "." ) &&
748 strcmp( item.cFileName, ".." ) )
749 {
750
751 sprintf( subDir, "%s%s\\", path, item.cFileName );
752 addDirectory( subDir, subDirs );
753
754 } // end if
755
756 // find the rest of the files
757 while( FindNextFile( hFile, &item ) != 0 )
758 {
759
760 // if this is a file count it
761 if( (item.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
762 strcmp( item.cFileName, "." ) &&
763 strcmp( item.cFileName, ".." ) )
764 {
765
766 sprintf( subDir, "%s%s\\", path, item.cFileName );
767 addDirectory( subDir, subDirs );
768
769 } // end if
770
771 } // end while
772
773 // close search
774 FindClose( hFile );
775
776 } //end if, items found
777
778 } // end if

780 // restore our current directory
781 SetCurrentDirectory( currDir );

783} // end addDirectory

785// ImagePacker::addImage ======================================================
787//=============================================================================
788void ImagePacker::addImage( char *path )
789{
790
791 // sanity
792 if( path == NULL )
793 return;
794
795 // allocate a new entry
796 ImageInfo *info = new ImageInfo;
797 if( info == NULL )
798 {
799
800 MessageBox( NULL, "Unable to allocate image info", "Error",
801 MB_OK | MB_ICONERROR );
802 return;
803
804 } // end if
805
806 // allocate space for the path
807 Int len = strlen( path );
808 info->m_path = new char[ len + 1 ];
809 strcpy( info->m_path, path );
810 if( info->m_path == NULL )
811 {
812
813 MessageBox( NULL, "Unable to allcoate image path info", "Error",
814 MB_OK | MB_ICONERROR );
815 delete info;
816 return;
817
818 } // end if
819
820 // load just the header information from the targa
821 m_targa->Load( info->m_path, 0, TRUE );
822
823 // get the data we need out of the targa header
824 info->m_colorDepth = m_targa->Header.PixelDepth;
825 info->m_size.x = m_targa->Header.Width;
826 info->m_size.y = m_targa->Header.Height;
827 info->m_area = info->m_size.x * info->m_size.y;
828
829 // save the filename only without path
830 Int i;
831 char *c;
832 for( i = len - 1; i >= 0; i-- )
833 {
834
835 if( path[ i ] == '\\' )
836 {
837 c = &path[ i + 1 ];
838 break;
839 }
840
841 } // end for i
842
843 Int nameLen = strlen( c );
844 info->m_filenameOnly = new char[ nameLen + 1 ];
845 strcpy( info->m_filenameOnly, c );
846
847 info->m_filenameOnlyNoExt = new char[ nameLen - 4 + 1 ];
848 strncpy( info->m_filenameOnlyNoExt, c, nameLen - 4 );
849 info->m_filenameOnlyNoExt[ nameLen - 4 ] = '\0';
850
851 // assign to array
852 m_imageList[ m_imageCount++ ] = info;
853
854 // update status
855 sprintf( m_statusBuffer, "Loading Image %d of %d.",
856 m_imageCount, m_imagesInDirs );
857 statusMessage( m_statusBuffer );
858
859} // end addImage

861// ImagePacker::generateINIFile ===============================================
863//=============================================================================
864Bool ImagePacker::generateINIFile( void )
865{
866 FILE *fp;
867 char filename[ _MAX_PATH ];
868
869 // construct filename we'll use
870 sprintf( filename, "%s%s.INI", m_outputDirectory, m_outputFile );
871
872 // open the file
873 fp = fopen( filename, "w" );
874 if( fp == NULL )
875 {
876 char buffer[ _MAX_PATH + 64 ];
877
878 sprintf( buffer, "Cannot open INI file '%s' for writing.", filename );
879 MessageBox( NULL, buffer, "Error Opening File", MB_OK | MB_ICONERROR );
880 return FALSE;
881
882 } // end if
883
884 // print header for file
885 fprintf( fp, "; ------------------------------------------------------------\n" );
886 fprintf( fp, "; Do NOT edit by hand, ImagePacker.exe auto generated INI file\n" );
887 fprintf( fp, "; ------------------------------------------------------------\n\n" );
888
889 //
890 // loop through all the pages so that we write image definitions that
891 // are on the same page close together in the file, note we're
892 // going backwards through the page list because page 1 is at the
893 // tail and I want them to print out in number order, but it
894 // doesn't really matter
895 //
896 TexturePage *page;
897 ImageInfo *image;
898 for( page = m_pageTail; page; page = page->m_prev )
899 {
900
901 // ignore texture pages that generated errors
