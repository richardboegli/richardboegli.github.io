73void StickyBombUpdate::onObjectCreated()
74{
75 //This first step is an initialization step. Immediately after the stickybomb is created,
76 //the producerID is stored in the object to signify the shooter of this weapon. From there,
77 //we can get the target. The target information will be used to determine where the sticky
78 //bomb will go.
79 ObjectID shooterID = getObject()->getProducerID();
80 Object* shooter = TheGameLogic->findObjectByID( shooterID );
81 if( shooter )
82 {
83 //Find the shooters target!
84 AIUpdateInterface *ai = shooter->getAIUpdateInterface();
85 if( ai )
86 {
87 Object *target = ai->getGoalObject();
88 if( target )
89 {
90 initStickyBomb( target, NULL);
91 }
92 }
93 }
94}

97void StickyBombUpdate::initStickyBomb( Object *target, const Object *bomber, const Coord3D *specificPos )
98{
99 //Store the target.
100 m_targetID = target ? target->getID() : INVALID_ID;
101
102 //Set the target as the producer ID -- so we can access it from other areas.
103 getObject()->setProducer( target );
104 UnsignedInt now = TheGameLogic->getFrame();
105
106 //Also determine our lifetime.... for countdown purposes.
107 static NameKeyType key_LifetimeUpdate = NAMEKEY( "LifetimeUpdate" );
108 LifetimeUpdate *update = (LifetimeUpdate*)getObject()->findUpdateModule( key_LifetimeUpdate );
109 if( update )
110 {
111 //we are a timer bomb
112 m_dieFrame = update->getDieFrame();
113
114 //Calculate the number of seconds (rounded down)
115 UnsignedInt pings = (m_dieFrame - now) / LOGICFRAMES_PER_SECOND;
116 //Now determine the next frame we will make a "ping" sound.
117 m_nextPingFrame = m_dieFrame - (pings * LOGICFRAMES_PER_SECOND);
118 }
119 else
120 {
121 //we are a remotely triggered bomb.
122 m_dieFrame = 0;
123
124 //Because we don't die -- make our ping in a second.
125 m_nextPingFrame = now + LOGICFRAMES_PER_SECOND;
126 }
127 setWakeFrame( getObject(), UPDATE_SLEEP_NONE );
128
129 if( target )
130 {
131 const StickyBombUpdateModuleData* d = getStickyBombUpdateModuleData();
132 Coord3D pos = *target->getPosition();
133
134 if( specificPos )
135 {
136 pos = *specificPos;
137 pos.z = TheTerrainLogic->getGroundHeight(pos.x, pos.y);
138 }
139 else if(target->isKindOf( KINDOF_IMMOBILE ) && bomber )
140 {
141 // make this exception, if bomber has placed bomb on a structure
142 // let the bomb just stay where it was first put, so a mine clearing unit can get to it later
143 pos = *bomber->getPosition();
144 pos.z = TheTerrainLogic->getGroundHeight(pos.x, pos.y);
145 //keep it at ground height for mine clearing units to reach
146 }
147 else
148 pos.z += d->m_offsetZ; // ride on the roof of the truck/tank
149
150 getObject()->setPosition( &pos );
151
152 if( getObject()->isKindOf(KINDOF_BOOBY_TRAP) )
153 {
154 // This kind of sticky bomb needs to set a status, so the poor victim can trigger us from assorted places
155 target->setStatus( MAKE_OBJECT_STATUS_MASK(OBJECT_STATUS_BOOBY_TRAPPED) );
156 }
157
158 AudioEventRTS soundCreateBomb = *(getObject()->getTemplate()->getPerUnitSound("StickyBombCreated"));
159 soundCreateBomb.setPosition( getObject()->getPosition() );
160 TheAudio->addAudioEvent(&soundCreateBomb);
161
162 }
163}

166UpdateSleepTime StickyBombUpdate::update( void )
167{
168 // Continually reset position of stickybomb to match the position of the target.
169 const Object *target = getTargetObject();
170 Object *self = getObject();
171 if( target )
172 {
173 if( target->isEffectivelyDead() )
174 {
175 //If the target is dead, then
176 TheGameLogic->destroyObject( self );
177 return UPDATE_SLEEP_NONE;
178 }
179
180 if ( target->isKindOf( KINDOF_IMMOBILE) )
181 {
182 const Coord3D *pos = self->getPosition();
183 Coord3D newPos;
184 newPos.x = pos->x;
185 newPos.y = pos->y;
186 newPos.z = TheTerrainLogic->getGroundHeight(newPos.x, newPos.y);
187 //keep it at ground height for mine clearing units to reach
188
189 self->setPosition( &newPos );
190 }
191 else // make the bomb follow the target around
192 {
193 const StickyBombUpdateModuleData* d = getStickyBombUpdateModuleData();
194 const Coord3D *pos = target->getPosition();
195 Coord3D newPos;
196 newPos.x = pos->x;
197 newPos.y = pos->y;
198 newPos.z = pos->z + d->m_offsetZ;
199
200 self->setPosition( &newPos );
201 }
202 }
203
204 UnsignedInt now = TheGameLogic->getFrame();
205 if( now >= m_nextPingFrame )
206 {
207 m_nextPingFrame += LOGICFRAMES_PER_SECOND;
208
209 //Play the "ping" sound.
210 AudioEventRTS sound = *self->getTemplate()->getPerUnitSound( "UnitBombPing" );
211 sound.setObjectID( self->getID() );
212 TheAudio->addAudioEvent( &sound );
213 }
214
215 return UPDATE_SLEEP_NONE;
216}

231void StickyBombUpdate::detonate()
232{
233 const StickyBombUpdateModuleData *data = getStickyBombUpdateModuleData();
234
235 Object* boobyTrappedObject = getTargetObject();
236 if( data->m_geometryBasedDamageWeaponTemplate )
237 {
238 // We need to hurt people based on the size of the thing we are on. The radius in our weapon
239 // is the radius beyond our borders
240 if( boobyTrappedObject )
241 {
242 WeaponBonus nullBonus;
243 Real boundingCircle = boobyTrappedObject->getGeometryInfo().getBoundingCircleRadius();
244 Real primaryDamage = data->m_geometryBasedDamageWeaponTemplate->getPrimaryDamage(nullBonus);
245 Real secondaryDamage = data->m_geometryBasedDamageWeaponTemplate->getSecondaryDamage(nullBonus);
246 Real primaryDamageRange = data->m_geometryBasedDamageWeaponTemplate->getPrimaryDamageRadius(nullBonus);
247 Real secondaryDamageRange = data->m_geometryBasedDamageWeaponTemplate->getSecondaryDamageRadius(nullBonus);
248 primaryDamageRange += boundingCircle;
249 secondaryDamageRange += boundingCircle;
250 Real primaryDamageRangeSqr = sqr(primaryDamageRange);
251 Real radius = max(primaryDamageRange, secondaryDamageRange);
252
253 SimpleObjectIterator *iter;
254 iter = ThePartitionManager->iterateObjectsInRange(boobyTrappedObject->getPosition(), radius, FROM_BOUNDINGSPHERE_3D);
255 MemoryPoolObjectHolder hold(iter);
256 Real curVictimDistSqr;
257 Object *curVictim = iter->firstWithNumeric(&curVictimDistSqr);
258 DamageInfo damageInfo;
259 damageInfo.in.m_damageType = data->m_geometryBasedDamageWeaponTemplate->getDamageType();
260 damageInfo.in.m_deathType = data->m_geometryBasedDamageWeaponTemplate->getDeathType();
261 damageInfo.in.m_sourceID = getObject()->getID();
262 damageInfo.in.m_sourcePlayerMask = getObject()->getControllingPlayer()->getPlayerMask();
263 damageInfo.in.m_damageStatusType = data->m_geometryBasedDamageWeaponTemplate->getDamageStatusType();
264
265 for (; curVictim != NULL; curVictim = iter ? iter->nextWithNumeric(&curVictimDistSqr) : NULL)
266 {
267 damageInfo.in.m_amount = (curVictimDistSqr <= primaryDamageRangeSqr) ? primaryDamage : secondaryDamage;
268 curVictim->attemptDamage(&damageInfo);
269 }
270
271 if( data->m_geometryBasedDamageFX )
272 {
273 // And we make FX based on that size too.
274 FXList::doFXPos(data->m_geometryBasedDamageFX, boobyTrappedObject->getPosition(), NULL, 0, NULL, secondaryDamageRange);
275 }
276 }
277 }
278
279 if( getObject()->isKindOf(KINDOF_BOOBY_TRAP) && boobyTrappedObject )
280 {
281 // This kind of sticky bomb needs to set a status, so the poor victim can trigger us from assorted places
282 boobyTrappedObject->clearStatus( MAKE_OBJECT_STATUS_MASK(OBJECT_STATUS_BOOBY_TRAPPED) );
283 }
284
285 getObject()->kill();// Most things just fire weapons in their death modules
286}
