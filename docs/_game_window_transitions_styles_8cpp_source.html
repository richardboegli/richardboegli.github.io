116void FlashTransition::update( Int frame )
117{
118 m_drawState = -1;
119 if(frame < FLASHTRANSITION_START || frame > FLASHTRANSITION_END)
120 {
121 DEBUG_ASSERTCRASH(FALSE, ("FlashTransition::update - Frame is out of the range the this update can handle %d", frame));
122 return;
123 }
124 switch (frame) {
125 case FLASHTRANSITION_START:
126 {
127
128 if(m_isForward || !m_win)
129 break;
130 m_win->winHide(TRUE);
131 m_isFinished = TRUE;
132 }
133 break;
134 case FLASHTRANSITION_FADE_IN_1:
135 if(m_isForward)
136 {
137 AudioEventRTS buttonClick("GUIBoarderFadeIn");
138
139 if( TheAudio )
140 {
141 TheAudio->addAudioEvent( &buttonClick );
142 } // end if
143
144 }
145 case FLASHTRANSITION_FADE_IN_2:
146 case FLASHTRANSITION_FADE_IN_3:
147 {
148 if(!m_win)
149 break;
150 m_win->winHide(TRUE);
151 m_drawState = frame;
152 }
153 break;
154 case FLASHTRANSITION_FADE_TO_BACKGROUND_1:
155 case FLASHTRANSITION_FADE_TO_BACKGROUND_2:
156 case FLASHTRANSITION_FADE_TO_BACKGROUND_3:
157 case FLASHTRANSITION_FADE_TO_BACKGROUND_4:
158 {
159 if(!m_win)
160 break;
161 m_win->winHide(FALSE);
162 m_drawState = frame;
163 }
164 break;
165 case FLASHTRANSITION_END:
166 {
167 if(!m_isForward || !m_win)
168 break;
169 m_win->winHide(FALSE);
170 m_isFinished = TRUE;
171 }
172 break;
173 }
174
