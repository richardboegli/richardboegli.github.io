781void GameClient::iterateDrawablesInRegion( Region3D *region, GameClientFuncPtr userFunc, void *userData )
782{
783 Drawable *draw, *nextDrawable;
784
785 for( draw = m_drawableList; draw; draw=nextDrawable )
786 {
787 nextDrawable = draw->getNextDrawable();
788
789 Coord3D pos = *draw->getPosition();

1302

1363{

1365 // version
1366 XferVersion currentVersion = 3;
1367 XferVersion version = currentVersion;
1368 xfer->xferVersion( &version, currentVersion );
1369
1370 // client frame number
1371 xfer->xferUnsignedInt( &m_frame );
1372
1373 //
1374 // note that we do not do the id counter here, we did it in the game state block because
1375 // it's important to do that part very early in the load process
1376 //
1377 // !!!DON'T DO THIS!!! ----> xfer->xferDrawableID( &m_nextDrawableID ); <---- !!!DON'T DO THIS!!!
1378
1379 //
1380 // xfer a table of contents that contain thing template and indentifier pairs. this
1381 // table of contents is good for this save file only as unique numbers are only
1382 // generated and stored for the actual things that are on this map
1383 //
1384 xferDrawableTOC( xfer );
1385
1386 // drawable count
1387 Drawable *draw;
1388 UnsignedShort drawableCount = 0;
1389 for( draw = getDrawableList(); draw; draw = draw->getNextDrawable() )
1390 {
1391 if (xfer->getXferMode() == XFER_SAVE && !shouldSaveDrawable(draw))
1392 continue;
1393 drawableCount++;
1394 }
1395 xfer->xferUnsignedShort( &drawableCount );
1396
1397 // drawable data
1398 DrawableTOCEntry *tocEntry;
1399 ObjectID objectID;
1400 if( xfer->getXferMode() == XFER_SAVE )
1401 {
1402
1403 // iterate all drawables
1404 for( draw = getDrawableList(); draw; draw = draw->getNextDrawable() )
1405 {
1406 if (!shouldSaveDrawable(draw))
1407 continue;
1408
1409 // get TOC entry for this drawable
1410 tocEntry = findTOCEntryByName( draw->getTemplate()->getName() );
1411 if( tocEntry == NULL )
1412 {

1414 DEBUG_CRASH(( "GameClient::xfer - Drawable TOC entry not found for '%s'\n", draw->getTemplate()->getName().str() ));
1415 throw SC_INVALID_DATA;
1416
1417 } // end if
1418
1419 // xfer toc id entry
1420 xfer->xferUnsignedShort( &tocEntry->id );
1421
1422 // begin data block
1423 xfer->beginBlock();
1424
1425 // write the object ID this drawable is attached to
1426 objectID = draw->getObject() ? draw->getObject()->getID() : INVALID_ID;
1427 xfer->xferObjectID( &objectID );
1428
1429 // write snapshot data
1430 xfer->xferSnapshot( draw );
1431
1432 // end data block
1433 xfer->endBlock();
1434
1435 } // end for, draw
1436
1437 } // end if, save
1438 else
1439 {
1440 UnsignedShort tocID;
1441 const ThingTemplate *thingTemplate;
1442 Int dataSize;
1443
1444 // read all entries
1445 for( UnsignedShort i = 0; i < drawableCount; ++i )
1446 {
1447
1448 // read toc id entry
1449 xfer->xferUnsignedShort( &tocID );
1450

1453 if( tocEntry == NULL )
1454 {
1455

1457 throw SC_INVALID_DATA;
1458
1459 } // end if

1461 // read data block size
1462 dataSize = xfer->beginBlock();
1463

1466 if( thingTemplate == NULL )
1467 {
1468
1469 DEBUG_CRASH(( "GameClient::xfer - Unrecognized thing template '%s', skipping. ENGINEERS - Are you *sure* it's OK to be ignoring this object from the save file??? Think hard about it!\n",
1470 tocEntry->name.str() ));
1471 xfer->skip( dataSize );
1472 continue;
1473
1474 } // end if
1475
1476 // read the object ID this drawable is attached to (if any)
1477 xfer->xferObjectID( &objectID );
1478
1479 //
1480 // if we have an attached object ID, we won't create a new drawable, we'll use the
1481 // one that has been created and attached to the object already
1482 //
1483 if( objectID != INVALID_ID )
1484 {
1485 Object *object = TheGameLogic->findObjectByID( objectID );
1486
1487 // sanity
1488 if( object == NULL )
1489 {
1490
1491 DEBUG_CRASH(( "GameClient::xfer - Cannot find object '%d' that is supposed to be attached to this drawable '%s'\n",
1492 objectID, thingTemplate->getName().str() ));
1493 throw SC_INVALID_DATA;
1494
1495 } // end if
1496
1497 // get the drawable from the object
1498 draw = object->getDrawable();
1499 if( draw == NULL )
1500 {
1501
1502 DEBUG_CRASH(( "GameClient::xfer - There is no drawable attached to the object '%s' (%d) and there should be\n",
1503 object->getTemplate()->getName().str(), object->getID() ));
1504 throw SC_INVALID_DATA;
1505
1506 } // end if
1507

1509 //
1510 // note carefully: we do NOT want to use isEquivalentTo() here, because different object reskins
1511 // SHOULD count as different templates for our purposes here (which are purely visual). however, we
1512 // do need to compare getFinalOverride, because draw->getTemplate() is always gonna return the final
1513 // override, while TheThingFactory->findTemplate does not.
1514 //
1515 const ThingTemplate* drawTemplate = draw->getTemplate();
1516 if (drawTemplate->getFinalOverride() != thingTemplate->getFinalOverride())
1517 {
1518 TheGameClient->destroyDrawable( draw );

1521 }
1522
1523 } // end if

1528 // there was no object attached to this drawable when we saved, we need to create a
1529 // whole brand new drawable now
1530 //

1532
1533 // sanity
1534 if( draw == NULL )

1536

1538 thingTemplate->getName().str() ));

1540
1541 } // end if
1542
1543 } // end else
