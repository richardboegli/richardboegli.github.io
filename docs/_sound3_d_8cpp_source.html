162//
163// On_Frame_Update
164//
166bool
167Sound3DClass::On_Frame_Update (unsigned int milliseconds)
168{
169 Matrix3D prev_tm = m_PrevTransform;
170
171 if (m_bDirty && (m_PhysWrapper != NULL)) {
172 m_Scene->Update_Sound (m_PhysWrapper);
173 m_bDirty = false;
174 }
175
176 //
177 // Update the sound's position if its linked to a render object
178 //
179 Apply_Auto_Position ();
180
181 //
182 // Make sure the transform is initialized
183 //
184 if (m_IsTransformInitted == false) {
185 prev_tm = m_Transform;
186 }
187
188 //
189 // Update the current velocity if we are 'auto-calcing'.
190 //
191 if (m_bAutoCalcVel && Get_Class_ID () != CLASSID_LISTENER) {
192 Vector3 last_pos = prev_tm.Get_Translation ();
193 Vector3 curr_pos = m_Transform.Get_Translation ();
194
195 //
196 // Don't update the velocity if we haven't moved (optimization -- Miles calls
197 // can be really slow)
198 //
199 if (last_pos != curr_pos) {
200 Vector3 curr_vel;
201
202 //
203 // Extrapolate our current velocity given the last time slice and the distance
204 // we moved.
205 //
206 float secs_since_last_update = (::GetTickCount () - m_LastUpdate);
207 if (secs_since_last_update > 0) {
208 curr_vel = ((curr_pos - last_pos) / secs_since_last_update);
209 } else {
210 curr_vel.Set (0, 0, 0);
211 }
212
213 Set_Velocity (curr_vel);
214 }
215 }
216
217 // Remember when the last time we updated our 'auto-calc'
218 // variables.
219 m_LastUpdate = ::GetTickCount ();
220 m_PrevTransform = m_Transform;
221
222 // Allow the base class to process this call
223 return AudibleSoundClass::On_Frame_Update ();
224}

278//
279// Update_Miles_Transform
280//
282void
283Sound3DClass::Update_Miles_Transform (void)
284{
285 //
286 // Do we have a valid miles handle?
287 //
288 if (m_SoundHandle != NULL) {
289
290 //
291 // Build a matrix to transform coordinates from world-space to listener-space
292 //
293 Matrix3D world_to_listener_tm;
294 m_ListenerTransform.Get_Orthogonal_Inverse (world_to_listener_tm);
295
296 //
297 // Transform the object's TM into "listener-space"
298 //
299#ifdef ALLOW_TEMPORARIES
300 Matrix3D listener_space_tm = world_to_listener_tm * m_Transform;
301#else
302 Matrix3D listener_space_tm;
303 listener_space_tm.mul(world_to_listener_tm, m_Transform);
304#endif
305
306 //
307 // Pass the sound's position onto miles
308 //
309 Vector3 position = listener_space_tm.Get_Translation ();
310 ::AIL_set_3D_position (m_SoundHandle->Get_H3DSAMPLE (), -position.Y, position.Z, position.X);
311
312 //
313 // Pass the sound's orientation (facing) onto miles
314 //
315 Vector3 facing = listener_space_tm.Get_X_Vector ();
316 Vector3 up = listener_space_tm.Get_Z_Vector ();
317
318 ::AIL_set_3D_orientation (m_SoundHandle->Get_H3DSAMPLE (),
319 -facing.Y,
320 facing.Z,
321 facing.X,
322 -up.Y,
323 up.Z,
324 up.X);
325 }
326
327 return ;
328}

332//
333// Set_Position
334//
336void
337Sound3DClass::Set_Position (const Vector3 &position)
338{
339 //
340 // Pass the sound's position onto miles
341 //
342 if (m_Transform.Get_Translation () != position) {
343 // Update our internal transform
344 //
345 // SKB: 4/13/01 - Confirmed to be OK by Pat Smith.
346 // Took Set_Transform() outside of condition because even if sound is
347 // not playing I need to be able to change it's position.
348 // I had a problem that sounds would never be added to the scene because
349 // their positions stayed at 0,0,0 even after this Set_Postion() call.
350 m_PrevTransform = m_Transform;
351 m_Transform.Set_Translation (position);
352 Set_Dirty ();
353
354 if (m_IsTransformInitted == false) {
355 m_PrevTransform = m_Transform;
356 m_IsTransformInitted = true;
357 }
358
359 if (m_SoundHandle != NULL) {
360
361 //
362 // Transform the sound's position into 'listener-space'
363 //
364 Vector3 sound_pos = position;
365 Vector3 listener_space_pos;
366 Matrix3D::Inverse_Transform_Vector (m_ListenerTransform, sound_pos, &listener_space_pos);
367
368 //
369 // Update the object's position inside of Miles
370 //
371 ::AIL_set_3D_position (m_SoundHandle->Get_H3DSAMPLE (), -listener_space_pos.Y,
372 listener_space_pos.Z, listener_space_pos.X);
373 }
374 }
375
376 return ;
377}

456//
457// Initialize_Miles_Handle
458//
460void
461Sound3DClass::Initialize_Miles_Handle (void)
462{
463 MMSLockClass lock;
464
465 // If this sound is already playing, then update its
466 // playing position to make sure we really should
467 // be playing it... (it will free the miles handle if not)
468 if (m_State == STATE_PLAYING) {
469 Update_Play_Position ();
470 }
471
472 // Do we have a valid sample handle from miles?
473 if (m_SoundHandle != NULL) {
474
475 //
476 // Pass the actual sound data onto the sample
477 //
478 m_SoundHandle->Initialize (m_Buffer);
479
480 //
481 // Record the total length of the sample in milliseconds...
482 //
483 m_SoundHandle->Get_Sample_MS_Position ((S32 *)&m_Length, NULL);
484
485 //
486 // Pass our cached settings onto miles
487 //
488 float real_volume = Determine_Real_Volume ();
489 m_SoundHandle->Set_Sample_Volume (int(real_volume * 127.0F));
490 m_SoundHandle->Set_Sample_Pan (int(m_Pan * 127.0F));
491 m_SoundHandle->Set_Sample_Loop_Count (m_LoopCount);
492
493 //
494 // Pass attenuation settings onto miles
495 //
496 ::AIL_set_3D_sample_distances ( m_SoundHandle->Get_H3DSAMPLE (),
497 m_DropOffRadius,
498 (m_MaxVolRadius > 1.0F) ? m_MaxVolRadius : 1.0F);
499
500
501 //
502 // Assign the 3D effects level accordingly (for reverb, etc)
503 //
504 ::AIL_set_3D_sample_effects_level (m_SoundHandle->Get_H3DSAMPLE (),
505 WWAudioClass::Get_Instance ()->Get_Effects_Level ());
506
507 //
508 // Pass the sound's position and orientation onto Miles
509 //
510 Update_Miles_Transform ();
511
512 //
513 // Apply the pitch factor (if necessary)
514 //
515 if (m_PitchFactor != 1.0F) {
516 Set_Pitch_Factor (m_PitchFactor);
517 }
518
519 // If this sound is already playing (and just now got a handle)
520 // then make sure we start it.
521 if (m_State == STATE_PLAYING) {
522 m_SoundHandle->Start_Sample ();
523
524 // Update the loop count based on the number of loops left
525 m_SoundHandle->Set_Sample_Loop_Count (m_LoopsLeft);
526 }
527
528 // Seek to the position of the sound where we last left off.
529 // For example, this sound could have gotten bumped due to a low priority,
530 // but is now back and ready to resume at the position it would have been
531 // at if it was never bumped.
532 Seek (m_CurrentPosition);
533
534 // Associate this object instance with the handle
535 m_SoundHandle->Set_Sample_User_Data (INFO_OBJECT_PTR, (S32)this);
536 }
537
538 return ;
539}

664//
665// Load
666//
668bool
669Sound3DClass::Load (ChunkLoadClass &cload)
670{
671 while (cload.Open_Chunk ()) {
672 switch (cload.Cur_Chunk_ID ()) {
673
674 case CHUNKID_BASE_CLASS:
675 AudibleSoundClass::Load (cload);
676 break;
677
678 case CHUNKID_VARIABLES:
679 {
680 //
681 // Read all the variables from their micro-chunks
682 //
683 while (cload.Open_Micro_Chunk ()) {
684 switch (cload.Cur_Micro_Chunk_ID ()) {
685
686 READ_MICRO_CHUNK (cload, VARID_AUTO_CALC_VEL, m_bAutoCalcVel);
687 READ_MICRO_CHUNK (cload, VARID_CURR_VEL, m_CurrentVelocity);
688 READ_MICRO_CHUNK (cload, VARID_MAX_VOL_RADIUS, m_MaxVolRadius);
689 READ_MICRO_CHUNK (cload, VARID_IS_STATIC, m_IsStatic);
690 }
691
692 cload.Close_Micro_Chunk ();
693 }
694 }
695 break;
696 }
697
698 cload.Close_Chunk ();
699 }
700
701 return true;
702}
