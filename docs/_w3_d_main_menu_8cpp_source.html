89static void advancePosition(GameWindow *window, const Image *image, UnsignedInt posX, UnsignedInt posY, UnsignedInt sizeX, UnsignedInt sizeY)
90{
91 if(!image)
92 return;
93
94 static Bool goingForward = TRUE;
95
96 ICoord2D pos, size;
97 if(!window)
98 {
99 pos.x = posX;
100 pos.y = posY;
101 size.x = sizeX;
102 size.y = sizeY;
103 }
104 else
105 {
106 window->winGetScreenPosition(&pos.x, &pos.y);
107 window->winGetSize(&size.x,&size.y);
108 }
109 static Int Width = size.x + image->getImageWidth();
110
111 static Int x = -800;
112 static Int y = pos.y - (image->getImageHeight()/2);
113
114 static UnsignedInt m_startTime = timeGetTime();
115 Int time = timeGetTime() - m_startTime;
116 Real percentDone = INT_TO_REAL(time) / 10000;
117
118 if(goingForward)
119 {
120 if(percentDone >= 1)
121 {
122 y = pos.y + size.y - (image->getImageHeight()/2);
123 m_startTime = timeGetTime();
124 goingForward = FALSE;
125 }
126 else
127 {
128 y = pos.y - (image->getImageHeight()/2);
129 x = (percentDone * Width) - image->getImageWidth();
130 }
131 }
132 else
133 {
134 if(percentDone >= 1)
135 {
136 y = pos.y - (image->getImageHeight()/2);
137 m_startTime = timeGetTime();
138 goingForward = TRUE;
139 }
140 else
141 {
142 y = pos.y + size.y - (image->getImageHeight()/2);
143 x = size.x - (percentDone * Width);
144 }
145 }
146 TheDisplay->drawImage(image,x, y, x + image->getImageWidth(), y + image->getImageHeight());
147
148}

399void W3DMainMenuMapBorder( GameWindow *window, WinInstanceData *instData )
400{
401 enum
402 {
403 BORDER_CORNER_SIZE = 10,
404 BORDER_LINE_SIZE = 20,
405 };
406 //( Int x, Int y, Int width, Int height )
407
408// save original x, y
409 Int x, y, width, height;
410 window->winGetScreenPosition(&x, &y);
411 window->winGetSize(&width,&height);
412
413// TheDisplay->setClipRegion(&clipRegion);
414 Int originalX = x;
415 Int originalY = y;
416 Int maxX = x + width;
417 Int maxY = y + height;
418 Int x2, y2; // used for simultaneous drawing of line pairs
419 Int size = 20;
420 Int halfSize = size / 2;
421
422 const Image *image = NULL;
423
424 // Draw Horizontal Lines
425 // All border pieces are based on a 10 pixel offset from the centerline
426 y = originalY - 10;
427 y2 = maxY - 10;
428 x2 = maxX - (10 + BORDER_LINE_SIZE);
429
430 image = TheMappedImageCollection->findImageByName("FrameCornerHorizontal");
431 if(image)
432 {
433
434 for( x=(originalX + 10); x <= x2; x += BORDER_LINE_SIZE )
435 {
436
437 TheDisplay->drawImage( image,
438 x, y, x + size, y + size );
439 TheDisplay->drawImage( image,
440 x, y2, x + size, y2 + size );
441
442 }
443
444
445 x2 = maxX - BORDER_CORNER_SIZE;
446
447 // x == place to draw remainder if any
448 if( (x2 - x) >= (BORDER_LINE_SIZE / 2) )
449 {
450
451 //Blit Half piece
452 TheDisplay->drawImage( image,
453 x, y, x + halfSize, y + size );
454 TheDisplay->drawImage( image,
455 x, y2, x + halfSize, y2 + size );
456
457 x += (BORDER_LINE_SIZE / 2);
458
459 }
460
461
462 // x2 - x ... must now be less than a half piece
463 // check for equals and if not blit an adjusted half piece border pieces have
464 // a two pixel repeat so we will blit one pixel over if necessary to line up
465 // the art, but we'll cover-up the overlap with the corners
466 if( x < x2 )
467 {
468 x -= ((BORDER_LINE_SIZE / 2) - (((x2 - x) + 1) & ~1));
469
470 //Blit Half piece
471 TheDisplay->drawImage( image,
472 x, y, x + halfSize, y + size );
473 TheDisplay->drawImage( image,
474 x, y2, x + halfSize, y2 + size );
475
476 }
477 }
478 image = TheMappedImageCollection->findImageByName("FrameCornerVertical");
479
480 if( image )
481 {
482
483 // Draw Vertical Lines
484 // All border pieces are based on a 10 pixel offset from the centerline
485 x = originalX - 10;
486 x2 = maxX - 10;
487 y2 = maxY - (10 + BORDER_LINE_SIZE);
488
489 for( y=(originalY + 10); y <= y2; y += BORDER_LINE_SIZE )
490 {
491
492 TheDisplay->drawImage( image,
493 x, y, x + size, y + size );
494 TheDisplay->drawImage( image,
495 x2, y, x2 + size, y + size );
496
497 }
498
499 y2 = maxY - BORDER_CORNER_SIZE;
500
501 // y == place to draw remainder if any
502 if( (y2 - y) >= (BORDER_LINE_SIZE / 2) )
503 {
504
505 //Blit Half piece
506 TheDisplay->drawImage( image,
507 x, y, x + size, y + halfSize );
508 TheDisplay->drawImage( image,
509 x2, y, x2 + size, y + halfSize );
510
511 y += (BORDER_LINE_SIZE / 2);
512 }
513
514 // y2 - y ... must now be less than a half piece
515 // check for equals and if not blit an adjusted half piece border pieces have
516 // a two pixel repeat so we will blit one pixel over if necessary to line up
517 // the art, but we'll cover-up the overlap with the corners
518 if( y < y2 )
519 {
520 y -= ((BORDER_LINE_SIZE / 2) - (((y2 - y) + 1) & ~1));
521
522 //Blit Half piece
523 TheDisplay->drawImage( image,
524 x, y, x + size, y + halfSize );
525 TheDisplay->drawImage( image,
526 x2, y, x2 + size, y + halfSize );
527
528 }
529 }
530
531 // Draw Corners
532 x = originalX - BORDER_CORNER_SIZE;
533 y = originalY - BORDER_CORNER_SIZE;
534 TheDisplay->drawImage( TheMappedImageCollection->findImageByName("FrameCornerUL"),
535 x, y, x + size, y + size );
536 x = maxX - BORDER_CORNER_SIZE;
537 y = originalY - BORDER_CORNER_SIZE;
538 TheDisplay->drawImage( TheMappedImageCollection->findImageByName("FrameCornerUR"),
539 x, y, x + size, y + size );
540 x = originalX - BORDER_CORNER_SIZE;
541 y = maxY - BORDER_CORNER_SIZE;
542 TheDisplay->drawImage( TheMappedImageCollection->findImageByName("FrameCornerLL"),
543 x, y, x + size, y + size );
544 x = maxX - BORDER_CORNER_SIZE;
545 y = maxY - BORDER_CORNER_SIZE;
546 TheDisplay->drawImage(TheMappedImageCollection->findImageByName("FrameCornerLR"),
547 x, y, x + size, y + size );
548
549 TheDisplay->enableClipping(FALSE);
550
551}
