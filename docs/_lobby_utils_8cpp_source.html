147Bool HandleSortButton( NameKeyType sortButton )
148{
149 if (sortButton == buttonSortBuddiesID)
150 {
151 sortByBuddies( !sortBuddies );
152 return TRUE;
153 }
154 else if (sortButton == buttonSortAlphaID)
155 {
156 if (theGameSortType == GAMESORT_ALPHA_ASCENDING)
157 {
158 setSortMode(GAMESORT_ALPHA_DESCENDING);
159 }
160 else
161 {
162 setSortMode(GAMESORT_ALPHA_ASCENDING);
163 }
164 return TRUE;
165 }
166 else if (sortButton == buttonSortPingID)
167 {
168 if (theGameSortType == GAMESORT_PING_ASCENDING)
169 {
170 setSortMode(GAMESORT_PING_DESCENDING);
171 }
172 else
173 {
174 setSortMode(GAMESORT_PING_ASCENDING);
175 }
176 return TRUE;
177 }
178 return FALSE;
179}

199static void gameTooltip(GameWindow *window,
200 WinInstanceData *instData,
201 UnsignedInt mouse)
202{
203 Int x, y, row, col;
204 x = LOLONGTOSHORT(mouse);
205 y = HILONGTOSHORT(mouse);
206
207 GadgetListBoxGetEntryBasedOnXY(window, x, y, row, col);
208
209 if (row == -1 || col == -1)
210 {
211 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString);//TheGameText->fetch("TOOLTIP:GamesBeingFormed") );
212 return;
213 }
214
215 Int gameID = (Int)GadgetListBoxGetItemData(window, row, 0);
216 GameSpyStagingRoom *room = TheGameSpyInfo->findStagingRoomByID(gameID);
217 if (!room)
218 {
219 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:UnknownGame") );
220 return;
221 }
222
223 if (col == COLUMN_PING)
224 {
225#if 0 //def DEBUG_LOGGING
226 UnicodeString s;
227 s.format(L"Ping is %d ms (cutoffs are %d ms and %d ms\n%hs local pings\n%hs remote pings",
228 room->getPingAsInt(), TheGameSpyConfig->getPingCutoffGood(), TheGameSpyConfig->getPingCutoffBad(),
229 TheGameSpyInfo->getPingString().str(), room->getPingString().str()
230 );
231 TheMouse->setCursorTooltip( s, 10, NULL, 2.0f ); // the text and width are the only params used. the others are the default values.
232#else
233 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:PingInfo"), 10, NULL, 2.0f ); // the text and width are the only params used. the others are the default values.
234#endif
235 return;
236 }
237 if (col == COLUMN_NUMPLAYERS)
238 {
239 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:NumberOfPlayers"), 10, NULL, 2.0f ); // the text and width are the only params used. the others are the default values.
240 return;
241 }
242 if (col == COLUMN_PASSWORD)
243 {
244 if (room->getHasPassword())
245 {
246 UnicodeString checkTooltip =TheGameText->fetch("TOOTIP:Password");
247 if(!checkTooltip.compare(L"Password required to joing game"))
248 checkTooltip.set(L"Password required to join game");
249 TheMouse->setCursorTooltip( checkTooltip, 10, NULL, 2.0f ); // the text and width are the only params used. the others are the default values.
250 }
251 else
252 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString );
253 return;
254 }
255 if (col == COLUMN_USE_STATS)
256 {
257 if ( room->getUseStats() )
258 {
259 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:UseStatsOn") );
260 }
261 else
262 {
263 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:UseStatsOff") );
264 }
265 return;
266 }
267
268 UnicodeString tooltip;
269
270 UnicodeString mapName;
271 const MapMetaData *md = TheMapCache->findMap(room->getMap());
272 if (md)
273 {
274 mapName = md->m_displayName;
275 }
276 else
277 {
278 const char *start = room->getMap().reverseFind('\\');
279 if (start)
280 {
281 ++start;
282 }
283 else
284 {
285 start = room->getMap().str();
286 }
287 mapName.translate( start );
288 }
289 UnicodeString tmp;
290 tooltip.format(TheGameText->fetch("TOOLTIP:GameInfoGameName"), room->getGameName().str());
291 if (room->getLadderPort() != 0)
292 {
293 const LadderInfo *linfo = TheLadderList->findLadder(room->getLadderIP(), room->getLadderPort());
294 if (linfo)
295 {
296 tmp.format(TheGameText->fetch("TOOLTIP:GameInfoLadderName"), linfo->name.str());
297 tooltip.concat(tmp);
298 }
299 }
300 if (room->getExeCRC() != TheGlobalData->m_exeCRC || room->getIniCRC() != TheGlobalData->m_iniCRC)
301 {
302 tmp.format(TheGameText->fetch("TOOLTIP:InvalidGameVersion"), mapName.str());
303 tooltip.concat(tmp);
304 }
305 tmp.format(TheGameText->fetch("TOOLTIP:GameInfoMap"), mapName.str());
306 tooltip.concat(tmp);
307
308 AsciiString aPlayer;
309 UnicodeString player;
310 Int numPlayers = 0;
311 for (Int i=0; i<MAX_SLOTS; ++i)
312 {
313 GameSpyGameSlot *slot = room->getGameSpySlot(i);
314 if (i == 0 && (!slot || !slot->isHuman()))
315 {
316 DEBUG_CRASH(("About to tooltip a non-hosted game!\n"));
317 }
318 if (slot && slot->isHuman())
319 {
320 tmp.format(TheGameText->fetch("TOOLTIP:GameInfoPlayer"), slot->getName().str(), slot->getWins(), slot->getLosses());
321 tooltip.concat(tmp);
322 ++numPlayers;
323 }
324 else if (slot && slot->isAI())
325 {
326 ++numPlayers;
327 switch(slot->getState())
328 {
329 case SLOT_EASY_AI:
330 tooltip.concat(L'\n');
331 tooltip.concat(TheGameText->fetch("GUI:EasyAI"));
332 break;
333 case SLOT_MED_AI:
334 tooltip.concat(L'\n');
335 tooltip.concat(TheGameText->fetch("GUI:MediumAI"));
336 break;
337 case SLOT_BRUTAL_AI:
338 tooltip.concat(L'\n');
339 tooltip.concat(TheGameText->fetch("GUI:HardAI"));
340 break;
341 }
342 }
343 }
344 DEBUG_ASSERTCRASH(numPlayers, ("Tooltipping a 0-player game!\n"));
345
346 TheMouse->setCursorTooltip( tooltip, 10, NULL, 2.0f ); // the text and width are the only params used. the others are the default values.
347}

436typedef std::set<GameSpyStagingRoom *> BuddyGameSet;
437static BuddyGameSet *theBuddyGames = NULL;
438static void populateBuddyGames(void)
439{
440 BuddyInfoMap *m = TheGameSpyInfo->getBuddyMap();
441 theBuddyGames = NEW BuddyGameSet;
442 if (!m)
443 {
444 return;
445 }
446 for (BuddyInfoMap::const_iterator bit = m->begin(); bit != m->end(); ++bit)
447 {
448 BuddyInfo info = bit->second;
449 if (info.m_status == GP_STAGING)
450 {
451 StagingRoomMap *srm = TheGameSpyInfo->getStagingRoomList();
452 for (StagingRoomMap::iterator srmIt = srm->begin(); srmIt != srm->end(); ++srmIt)
453 {
454 GameSpyStagingRoom *game = srmIt->second;
455 game->cleanUpSlotPointers();
456 const GameSpyGameSlot *slot = game->getGameSpySlot(0);
457 if (slot && slot->getName() == info.m_locationString)
458 {
459 theBuddyGames->insert(game);
460 break;
461 }
462 }
463 }
464 }
465}

474struct GameSortStruct
475{
476 bool operator()(GameSpyStagingRoom *g1, GameSpyStagingRoom *g2)
477 {
478 // sort CRC mismatches to the bottom
479 Bool g1Good = (g1->getExeCRC() != TheGlobalData->m_exeCRC || g1->getIniCRC() != TheGlobalData->m_iniCRC);
480 Bool g2Good = (g1->getExeCRC() != TheGlobalData->m_exeCRC || g1->getIniCRC() != TheGlobalData->m_iniCRC);
481 if ( g1Good ^ g2Good )
482 {
483 return g1Good;
484 }
485
486 // sort games with private ladders to the bottom
487 Bool g1UnknownLadder = (g1->getLadderPort() && TheLadderList->findLadder(g1->getLadderIP(), g1->getLadderPort()) == NULL);
488 Bool g2UnknownLadder = (g2->getLadderPort() && TheLadderList->findLadder(g2->getLadderIP(), g2->getLadderPort()) == NULL);
489 if ( g1UnknownLadder ^ g2UnknownLadder )
490 {
491 return g2UnknownLadder;
492 }
493
494 // sort full games to the bottom
495 Bool g1Full = (g1->getNumNonObserverPlayers() == g1->getMaxPlayers() || g1->getNumPlayers() == MAX_SLOTS);
496 Bool g2Full = (g2->getNumNonObserverPlayers() == g2->getMaxPlayers() || g2->getNumPlayers() == MAX_SLOTS);
497 if ( g1Full ^ g2Full )
498 {
499 return g2Full;
500 }
501
502 if (sortBuddies)
503 {
504 Bool g1HasBuddies = (theBuddyGames->find(g1) != theBuddyGames->end());
505 Bool g2HasBuddies = (theBuddyGames->find(g2) != theBuddyGames->end());
506 if ( g1HasBuddies ^ g2HasBuddies )
507 {
508 return g1HasBuddies;
509 }
510 }
511
512 switch(theGameSortType)
513 {
514 case GAMESORT_ALPHA_ASCENDING:
515 return wcsicmp(g1->getGameName().str(), g2->getGameName().str()) < 0;
516 break;
517 case GAMESORT_ALPHA_DESCENDING:
518 return wcsicmp(g1->getGameName().str(),g2->getGameName().str()) > 0;
519 break;
520 case GAMESORT_PING_ASCENDING:
521 return g1->getPingAsInt() < g2->getPingAsInt();
522 break;
523 case GAMESORT_PING_DESCENDING:
524 return g1->getPingAsInt() > g2->getPingAsInt();
525 break;
526 }
527 return false;
528 }
529};

531static Int insertGame( GameWindow *win, GameSpyStagingRoom *game, Bool showMap )
532{
533 game->cleanUpSlotPointers();
534 Color gameColor = GameSpyColor[GSCOLOR_GAME];
535 if (game->getNumNonObserverPlayers() == game->getMaxPlayers() || game->getNumPlayers() == MAX_SLOTS)
536 {
537 gameColor = GameSpyColor[GSCOLOR_GAME_FULL];
538 }
539 if (game->getExeCRC() != TheGlobalData->m_exeCRC || game->getIniCRC() != TheGlobalData->m_iniCRC)
540 {
541 gameColor = GameSpyColor[GSCOLOR_GAME_CRCMISMATCH];
542 }
543 UnicodeString gameName = game->getGameName();
544
545 if(TheGameSpyInfo->getDisallowAsianText())
546 {
547 const WideChar *buff = gameName.str();
548 Int length = gameName.getLength();
549 for(Int i = 0; i < length; ++i)
550 {
551 if(buff[i] >= 256)
552 return -1;
553 }
554 }
555 else if(TheGameSpyInfo->getDisallowNonAsianText())
556 {
557 const WideChar *buff = gameName.str();
558 Int length = gameName.getLength();
559 Bool hasUnicode = FALSE;
560 for(Int i = 0; i < length; ++i)
561 {
562 if(buff[i] >= 256)
563 {
564 hasUnicode = TRUE;
565 break;
566 }
567 }
568 if(!hasUnicode)
569 return -1;
570 }
571
572
573
574 Int index = GadgetListBoxAddEntryText(win, game->getGameName(), gameColor, -1, COLUMN_NAME);
575 GadgetListBoxSetItemData(win, (void *)game->getID(), index);
576
577 UnicodeString s;
578
579 if (showMap)
580 {
581 UnicodeString mapName;
582 const MapMetaData *md = TheMapCache->findMap(game->getMap());
583 if (md)
584 {
585 mapName = md->m_displayName;
586 }
587 else
588 {
589 const char *start = game->getMap().reverseFind('\\');
590 if (start)
591 {
592 ++start;
593 }
594 else
595 {
596 start = game->getMap().str();
597 }
598 mapName.translate( start );
599 }
600 GadgetListBoxAddEntryText(win, mapName, gameColor, index, COLUMN_MAP);
601
602 const LadderInfo * li = TheLadderList->findLadder(game->getLadderIP(), game->getLadderPort());
603 if (li)
604 {
605 GadgetListBoxAddEntryText(win, li->name, gameColor, index, COLUMN_LADDER);
606 }
607 else if (game->getLadderPort())
608 {
609 GadgetListBoxAddEntryText(win, TheGameText->fetch("GUI:UnknownLadder"), gameColor, index, COLUMN_LADDER);
610 }
611 else
612 {
613 GadgetListBoxAddEntryText(win, TheGameText->fetch("GUI:NoLadder"), gameColor, index, COLUMN_LADDER);
614 }
615 }
616 else
617 {
618 GadgetListBoxAddEntryText(win, UnicodeString(L" "), gameColor, index, COLUMN_MAP);
619 GadgetListBoxAddEntryText(win, UnicodeString(L" "), gameColor, index, COLUMN_LADDER);
620 }
621
622 s.format(L"%d/%d", game->getReportedNumPlayers(), game->getReportedMaxPlayers());
623 GadgetListBoxAddEntryText(win, s, gameColor, index, COLUMN_NUMPLAYERS);
624
625 if (game->getHasPassword())
626 {
627 const Image *img = TheMappedImageCollection->findImageByName("Password");
628 Int width = 10, height = 10;
629 if (img)
630 {
631 width = img->getImageWidth();
632 height = img->getImageHeight();
633 }
634 GadgetListBoxAddEntryImage(win, img, index, COLUMN_PASSWORD, width, height);
635 }
636 else
637 {
638 GadgetListBoxAddEntryText(win, UnicodeString(L" "), gameColor, index, COLUMN_PASSWORD);
639 }
640
641 if (game->getAllowObservers())
642 {
643 const Image *img = TheMappedImageCollection->findImageByName("Observer");
644 GadgetListBoxAddEntryImage(win, img, index, COLUMN_OBSERVER);
645 }
646 else
647 {
648 GadgetListBoxAddEntryText(win, UnicodeString(L" "), gameColor, index, COLUMN_OBSERVER);
649 }
650
651 {
652 if (game->getUseStats())
653 {
654 const Image *img = TheMappedImageCollection->findImageByName("GoodStatsIcon");
655 GadgetListBoxAddEntryImage(win, img, index, COLUMN_USE_STATS, img->getImageHeight(), img->getImageWidth());
656 }
657
658 }
659
660 s.format(L"%d", game->getPingAsInt());
661 GadgetListBoxAddEntryText(win, s, gameColor, index, COLUMN_PING);
662 Int ping = game->getPingAsInt();
663 Int width = 10, height = 10;
664 if (pingImages[0])
665 {
666 width = pingImages[0]->getImageWidth();
667 height = pingImages[0]->getImageHeight();
668 }
669 // CLH picking an arbitrary number for our ping display
670 if (ping < TheGameSpyConfig->getPingCutoffGood())
671 {
672 GadgetListBoxAddEntryImage(win, pingImages[0], index, COLUMN_PING, width, height);
673 }
674 else if (ping < TheGameSpyConfig->getPingCutoffBad())
675 {
676 GadgetListBoxAddEntryImage(win, pingImages[1], index, COLUMN_PING, width, height);
677 }
678 else
679 {
680 GadgetListBoxAddEntryImage(win, pingImages[2], index, COLUMN_PING, width, height);
681 }

687{
688 if (!win)
689 return;
690
691 // save off selection
692 Int selectedIndex = -1;
693 Int indexToSelect = -1;
694 Int selectedID = 0;
695 GadgetListBoxGetSelected(win, &selectedIndex);
696 if (selectedIndex != -1 )
697 {
698 selectedID = (Int)GadgetListBoxGetItemData(win, selectedIndex);
699 }
700 int prevPos = GadgetListBoxGetTopVisibleEntry( win );
701
702 // empty listbox
703 GadgetListBoxReset(win);
704
705 // sort our games
706 typedef std::multiset<GameSpyStagingRoom *, GameSortStruct> SortedGameList;
707 SortedGameList sgl;
708 StagingRoomMap *srm = TheGameSpyInfo->getStagingRoomList();
709 populateBuddyGames();
710 for (StagingRoomMap::iterator srmIt = srm->begin(); srmIt != srm->end(); ++srmIt)
711 {
712 sgl.insert(srmIt->second);
713 }
714
715 // populate listbox
716 for (SortedGameList::iterator sglIt = sgl.begin(); sglIt != sgl.end(); ++sglIt)
717 {
718 GameSpyStagingRoom *game = *sglIt;
719 if (game)
720 {
721 Int index = insertGame(win, game, showMap);
722 if (game->getID() == selectedID)
723 {
724 indexToSelect = index;
725 }
726 }
727 }
728
729 clearBuddyGames();
730
731 // restore selection
732 GadgetListBoxSetSelected(win, indexToSelect); // even for -1, so we can disable the 'Join Game' button
733// if(prevPos > 10)
734 GadgetListBoxSetTopVisibleEntry( win, prevPos );//+ 1
735
736 if (indexToSelect < 0 && selectedID)
737 {
738 TheWindowManager->winSetLoneWindow(NULL);
739 }
