122void INI::parseMapCacheDefinition( INI* ini )
123{
124 const char *c;
125 AsciiString name;
126 MapMetaDataReader mdr;
127 MapMetaData md;
128
129 // read the name
130 c = ini->getNextToken(" \n\r\t");
131 name.set( c );
132 name = QuotedPrintableToAsciiString(name);
133 md.m_waypoints.clear();
134
135 ini->initFromINI( &mdr, mdr.getFieldParse() );
136
137 md.m_extent = mdr.m_extent;
138 md.m_isOfficial = mdr.m_isOfficial != 0;
139 md.m_isMultiplayer = mdr.m_isMultiplayer != 0;
140 md.m_numPlayers = mdr.m_numPlayers;
141 md.m_filesize = mdr.m_filesize;
142 md.m_CRC = mdr.m_CRC;
143 md.m_timestamp = mdr.m_timestamp;
144
145 md.m_waypoints[TheNameKeyGenerator->keyToName(TheKey_InitialCameraPosition)] = mdr.m_initialCameraPosition;
146
147// md.m_displayName = QuotedPrintableToUnicodeString(mdr.m_asciiDisplayName);
148// this string is never to be used, but we'll leave it in to allow people with an old mapcache.ini to parse it
149 md.m_nameLookupTag = QuotedPrintableToAsciiString(mdr.m_asciiNameLookupTag);
150
151 if (md.m_nameLookupTag.isEmpty())
152 {
153 // maps without localized name tags
154 AsciiString tempdisplayname;
155 tempdisplayname = name.reverseFind('\\') + 1;
156 md.m_displayName.translate(tempdisplayname);
157 if (md.m_numPlayers >= 2)
158 {
159 UnicodeString extension;
160 extension.format(L" (%d)", md.m_numPlayers);
161 md.m_displayName.concat(extension);
162 }
163 }
164 else
165 {
166 // official maps with name tags
167 md.m_displayName = TheGameText->fetch(md.m_nameLookupTag);
168 if (md.m_numPlayers >= 2)
169 {
170 UnicodeString extension;
171 extension.format(L" (%d)", md.m_numPlayers);
172 md.m_displayName.concat(extension);
173 }
174 }
175
176 AsciiString startingCamName;
177 for (Int i=0; i<md.m_numPlayers; ++i)
178 {
179 startingCamName.format("Player_%d_Start", i+1); // start pos waypoints are 1-based
180 md.m_waypoints[startingCamName] = mdr.m_waypoints[i];
181 }
182
183 Coord3DList::iterator it = mdr.m_supplyPositions.begin();
184 while( it != mdr.m_supplyPositions.end())
185 {
186 md.m_supplyPositions.push_front(*it);
187 it++;
188 }
189
190 it = mdr.m_techPositions.begin();
191 while( it != mdr.m_techPositions.end())
192 {
193 md.m_techPositions.push_front(*it);
194 it++;
195 }
196
197 if(TheMapCache && !md.m_displayName.isEmpty())
198 {
199 AsciiString lowerName = name;
200 lowerName.toLower();
201 md.m_fileName = lowerName;
202// DEBUG_LOG(("INI::parseMapCacheDefinition - adding %s to map cache\n", lowerName.str()));
203 (*TheMapCache)[lowerName] = md;
204 }
205}
