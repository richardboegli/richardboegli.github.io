193void GameSpyOpenOverlay( GSOverlayType overlay )
194{
195 if (overlay == GSOVERLAY_BUDDY)
196 {
197 if (!TheGameSpyBuddyMessageQueue->isConnected())
198 {
199 // not connected - is it because we were disconnected?
200 if (TheGameSpyBuddyMessageQueue->getLocalProfileID())
201 {
202 // used to be connected
203 GSMessageBoxYesNo(TheGameText->fetch("GUI:GPErrorTitle"), TheGameText->fetch("GUI:GPDisconnected"), buddyTryReconnect, NULL);
204 }
205 else
206 {
207 // no profile
208 GSMessageBoxOk(TheGameText->fetch("GUI:GPErrorTitle"), TheGameText->fetch("GUI:GPNoProfile"), NULL);
209 }
210 return;
211 }
212 AudioEventRTS buttonClick("GUICommunicatorOpen");
213
214 if( TheAudio )
215 {
216 TheAudio->addAudioEvent( &buttonClick );
217 } // end if
218 }
219 if (overlayLayouts[overlay])
220 {
221 overlayLayouts[overlay]->hide( FALSE );
222 overlayLayouts[overlay]->bringForward();
223 }
224 else
225 {
226 overlayLayouts[overlay] = TheWindowManager->winCreateLayout( AsciiString( gsOverlays[overlay] ) );
227 overlayLayouts[overlay]->runInit();
228 overlayLayouts[overlay]->hide( FALSE );
229 overlayLayouts[overlay]->bringForward();
230 }
231}

233void GameSpyCloseOverlay( GSOverlayType overlay )
234{
235 switch(overlay)
236 {
237 case GSOVERLAY_PLAYERINFO:
238 DEBUG_LOG(("Closing overlay GSOVERLAY_PLAYERINFO\n"));
239 break;
240 case GSOVERLAY_MAPSELECT:
241 DEBUG_LOG(("Closing overlay GSOVERLAY_MAPSELECT\n"));
242 break;
243 case GSOVERLAY_BUDDY:
244 DEBUG_LOG(("Closing overlay GSOVERLAY_BUDDY\n"));
245 break;
246 case GSOVERLAY_PAGE:
247 DEBUG_LOG(("Closing overlay GSOVERLAY_PAGE\n"));
248 break;
249 case GSOVERLAY_GAMEOPTIONS:
250 DEBUG_LOG(("Closing overlay GSOVERLAY_GAMEOPTIONS\n"));
251 break;
252 case GSOVERLAY_GAMEPASSWORD:
253 DEBUG_LOG(("Closing overlay GSOVERLAY_GAMEPASSWORD\n"));
254 break;
255 case GSOVERLAY_LADDERSELECT:
256 DEBUG_LOG(("Closing overlay GSOVERLAY_LADDERSELECT\n"));
257 break;
258 case GSOVERLAY_OPTIONS:
259 DEBUG_LOG(("Closing overlay GSOVERLAY_OPTIONS\n"));
260 if( overlayLayouts[overlay] )
261 {
262 SignalUIInteraction(SHELL_SCRIPT_HOOK_OPTIONS_CLOSED);
263 }
264 break;
265 }
266 if( overlayLayouts[overlay] )
267 {
268 overlayLayouts[overlay]->runShutdown();
269 overlayLayouts[overlay]->destroyWindows();
270 overlayLayouts[overlay]->deleteInstance();
271 overlayLayouts[overlay] = NULL;
272 }
273}
