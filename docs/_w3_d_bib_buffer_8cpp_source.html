87void W3DBibBuffer::loadBibsInVertexAndIndexBuffers(void)
88{
89 if (!m_indexBib || !m_vertexBib || !m_initialized) {
90 return;
91 }
92 if (!m_anythingChanged) {
93 return;
94 }
95
96 m_curNumBibVertices = 0;
97 m_curNumBibIndices = 0;
98 m_curNumNormalBibIndices = 0;
99 m_curNumNormalBibVertex = 0;
100
101 if (m_numBibs==0) {
102 return;
103 }
104
105 VertexFormatXYZDUV1 *vb;
106 UnsignedShort *ib;
107 // Lock the buffers.
108 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBib, D3DLOCK_DISCARD);
109 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexBib, D3DLOCK_DISCARD);
110 vb=(VertexFormatXYZDUV1*)lockVtxBuffer.Get_Vertex_Array();
111 ib = lockIdxBuffer.Get_Index_Array();
112 // Add to the index buffer & vertex buffer.
113 UnsignedShort *curIb = ib;
114
115 VertexFormatXYZDUV1 *curVb = vb;
116
117 Int curBib;
118
119 // Calculate a static lighting value to use for all the bibs.
120 Real shadeR, shadeG, shadeB;
121 shadeR = TheGlobalData->m_terrainAmbient[0].red;
122 shadeG = TheGlobalData->m_terrainAmbient[0].green;
123 shadeB = TheGlobalData->m_terrainAmbient[0].blue;
124 shadeR += TheGlobalData->m_terrainDiffuse[0].red;
125 shadeG += TheGlobalData->m_terrainDiffuse[0].green;
126 shadeB += TheGlobalData->m_terrainDiffuse[0].blue;
127 if (shadeR>1.0f) shadeR=1.0f;
128 if (shadeG>1.0f) shadeG=1.0f;
129 if (shadeB>1.0f) shadeB=1.0f;
130 shadeR*=255.0f;
131 shadeG*=255.0f;
132 shadeB*=255.0f;
133
134 Int diffuse = (REAL_TO_INT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16) | (255 << 24));
135 Int doHighlight;
136 try {
137 for (doHighlight=0; doHighlight<=1; doHighlight++)
138 {
139 if (doHighlight==1)
140 {
141 m_curNumNormalBibIndices = m_curNumBibIndices;
142 m_curNumNormalBibVertex = m_curNumBibVertices;
143 }
144 for (curBib=0; curBib<m_numBibs; curBib++) {
145 if (m_bibs[curBib].m_unused) continue;
146 if (m_bibs[curBib].m_highlight != (Bool)doHighlight) continue;
147 Int startVertex = m_curNumBibVertices;
148 Int i;
149 Int numVertex = 4;
150 if (m_curNumBibVertices+numVertex+2>= m_vertexBibSize) {
151 break;
152 }
153 Int numIndex = 6;
154 if (m_curNumBibIndices+numIndex+6 >= m_indexBibSize) {
155 break;
156 }
157
158 for (i=0; i<numVertex; i++) {
159
160 // Update the uv values. The W3D models each have their own texture, and
161 // we use one texture with all images in one, so we have to change the uvs to
162 // match.
163 Real U, V;
164 Vector3 vLoc=m_bibs[curBib].m_corners[i];
165 switch (i) {
166 case 0 :
167 U=0;V=1;
168 break;
169 case 1:
170 U=1;V=1;
171 break;
172 case 2:
173 U=1;V=0;
174 break;
175 case 3:
176 U=0;V=0;
177 break;
178 }
179
180 curVb->u1 = U;
181 curVb->v1 = V;
182 curVb->x = vLoc.X;
183 curVb->y = vLoc.Y;
184 curVb->z = vLoc.Z;
185 curVb->diffuse = diffuse;
186 curVb++;
187 m_curNumBibVertices++;
188 }
189
190 *curIb++ = startVertex + 0;
191 *curIb++ = startVertex + 1;
192 *curIb++ = startVertex + 2;
193 *curIb++ = startVertex + 0;
194 *curIb++ = startVertex + 2;
195 *curIb++ = startVertex + 3;
196 m_curNumBibIndices+=6;
197 }
198 }
199 IndexBufferExceptionFunc();
200 } catch(...) {
201 IndexBufferExceptionFunc();
202 }
203}

311void W3DBibBuffer::addBib(Vector3 corners[4], ObjectID id, Bool highlight)
312{
313 Int bibIndex;
314 for (bibIndex=0; bibIndex<m_numBibs; bibIndex++) {
315 if (!m_bibs[bibIndex].m_unused && m_bibs[bibIndex].m_objectID == id) {
316 break;
317 }
318 }
319 if (bibIndex==m_numBibs) {
320 for (bibIndex=0; bibIndex<m_numBibs; bibIndex++) {
321 if (m_bibs[bibIndex].m_unused) {
322 break;
323 }
324 }
325 }
326 if (bibIndex==m_numBibs) {
327 if (m_numBibs >= MAX_BIBS) {
328 return;
329 }
330 m_numBibs++;
331 }
332 m_anythingChanged = true;
333 m_bibs[bibIndex].m_corners[0] = corners[0];
334 m_bibs[bibIndex].m_corners[1] = corners[1];
335 m_bibs[bibIndex].m_corners[2] = corners[2];
336 m_bibs[bibIndex].m_corners[3] = corners[3];
337 m_bibs[bibIndex].m_highlight = highlight;
338 m_bibs[bibIndex].m_color = 0; // for now.
339 m_bibs[bibIndex].m_unused = false; // for now.
340 m_bibs[bibIndex].m_objectID = id;
341 m_bibs[bibIndex].m_drawableID = INVALID_DRAWABLE_ID;
342}

419//=============================================================================
420// W3DBibBuffer::drawBibs
421//=============================================================================
423//=============================================================================
424void W3DBibBuffer::renderBibs()
425{
426
427 loadBibsInVertexAndIndexBuffers();
428
429 if (m_curNumBibIndices == 0) {
430 return;
431 }
432 // Setup the vertex buffer, shader & texture.
433 DX8Wrapper::Set_Index_Buffer(m_indexBib,0);
434 DX8Wrapper::Set_Vertex_Buffer(m_vertexBib);
435 DX8Wrapper::Set_Shader(detailAlphaShader);
436 if (m_curNumNormalBibIndices) {
437 DX8Wrapper::Set_Texture(0,m_bibTexture);
438 DX8Wrapper::Draw_Triangles( 0, m_curNumNormalBibIndices/3, 0, m_curNumNormalBibVertex);
439 }
440 if (m_curNumBibIndices>m_curNumNormalBibIndices) {
441 DX8Wrapper::Set_Texture(0,m_highlightBibTexture);
442 DX8Wrapper::Draw_Triangles( m_curNumNormalBibIndices, (m_curNumBibIndices-m_curNumNormalBibIndices)/3,
443 m_curNumNormalBibVertex, m_curNumBibVertices-m_curNumNormalBibVertex);
444 }
