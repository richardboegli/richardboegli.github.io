74GameMessage::~GameMessage( )
75{
76 // free all arguments
77 GameMessageArgument *arg, *nextArg;
78
79 for( arg = m_argList; arg; arg=nextArg )
80 {
81 nextArg = arg->m_next;
82 arg->deleteInstance();
83 }
84
85 // detach message from list
86 if (m_list)
87 m_list->removeMessage( this );
88}

94const GameMessageArgumentType *GameMessage::getArgument( Int argIndex ) const
95{
96 static const GameMessageArgumentType junk = { 0 };
97
98 int i=0;
99 for( GameMessageArgument *a = m_argList; a; a=a->m_next, i++ )
100 if (i == argIndex)
101 return &a->m_data;
102
103 DEBUG_CRASH(("argument not found"));
104 return &junk;
105}

110GameMessageArgumentDataType GameMessage::getArgumentDataType( Int argIndex )
111{
112 if (argIndex >= m_argCount) {
113 return ARGUMENTDATATYPE_UNKNOWN;
114 }
115 int i=0;
116 for (GameMessageArgument *a = m_argList; a && (i < argIndex); a=a->m_next, ++i );
117
118 if (a != NULL)
119 {
120 return a->m_type;
121 }
122 return ARGUMENTDATATYPE_UNKNOWN;
123}

128GameMessageArgument *GameMessage::allocArg( void )
129{
130 // allocate a new argument
131 GameMessageArgument *arg = newInstance(GameMessageArgument);
132
133 // add to end of argument list
134 if (m_argTail)
135 m_argTail->m_next = arg;
136 else
137 {
138 m_argList = arg;
139 m_argTail = arg;
140 }
141
142 arg->m_next = NULL;
143 m_argTail = arg;
144
145 m_argCount++;
146
147 return arg;
148}

745void GameMessageList::insertMessage( GameMessage *msg, GameMessage *messageToInsertAfter )
746{
747 // First, set msg's next to be messageToInsertAfter's next.
748 msg->friend_setNext(messageToInsertAfter->next());
749
750 // Next, set msg's prev to be messageToInsertAfter
751 msg->friend_setPrev(messageToInsertAfter);
752
753 // Now update the next message's prev to be msg
754 if (msg->next())
755 msg->next()->friend_setPrev(msg);
756 else // if the friend wasn't there, then messageToInsertAfter is the last message. Update it to be msg.
757 m_lastMessage = msg;
758
759 // Finally, update the messageToInsertAfter's next to be msg
760 messageToInsertAfter->friend_setNext(msg);
761
762 // note containment within the message itself
763 msg->friend_setList(this);
764}

770{

773 else
774 m_lastMessage = msg->prev();
775
776 if (msg->prev())
777 msg->prev()->friend_setNext(msg->next());
778 else
779 m_firstMessage = msg->next();

894TranslatorID MessageStream::attachTranslator( GameMessageTranslator *translator,
895 UnsignedInt priority)
896{
897 MessageStream::TranslatorData *newSS = NEW MessageStream::TranslatorData;
898 MessageStream::TranslatorData *ss;
899
900 newSS->m_translator = translator;
901 newSS->m_priority = priority;
902 newSS->m_id = m_nextTranslatorID++;
903
904 if (m_firstTranslator == NULL)
905 {
906 // first Translator to be attached
907 newSS->m_prev = NULL;
908 newSS->m_next = NULL;
909 m_firstTranslator = newSS;
910 m_lastTranslator = newSS;
911 return newSS->m_id;
912 }
913
914 // seach the Translator list for our priority location
915 for( ss=m_firstTranslator; ss; ss=ss->m_next )
916 if (ss->m_priority > newSS->m_priority)
917 break;
918
919 if (ss)
920 {
921 // insert new Translator just BEFORE this one,
922 // therefore, m_lastTranslator cannot be affected
923 if (ss->m_prev)
924 {
925 ss->m_prev->m_next = newSS;
926 newSS->m_prev = ss->m_prev;
927 newSS->m_next = ss;
928 ss->m_prev = newSS;
929 }
930 else
931 {
932 // insert at head of list
933 newSS->m_prev = NULL;
934 newSS->m_next = m_firstTranslator;
935 m_firstTranslator->m_prev = newSS;
936 m_firstTranslator = newSS;
937 }
938 }
939 else
940 {
941 // append Translator to end of list
942 m_lastTranslator->m_next = newSS;
943 newSS->m_prev = m_lastTranslator;
944 newSS->m_next = NULL;
945 m_lastTranslator = newSS;
946 }
947
948 return newSS->m_id;
949}
