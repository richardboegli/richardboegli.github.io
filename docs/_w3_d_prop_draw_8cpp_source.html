90void W3DPropDraw::reactToTransformChange( const Matrix3D *oldMtx,
91 const Coord3D *oldPos,
92 Real oldAngle )
93{
94 Drawable *draw = getDrawable();
95 if (m_propAdded) {
96 return;
97 }
98 if (draw->getPosition()->x==0.0f && draw->getPosition()->y == 0.0f) {
99 return;
100 }
101 m_propAdded = true;
102 const W3DPropDrawModuleData *moduleData = getW3DPropDrawModuleData();
103 if (!moduleData) {
104 return;
105 }
106 Real scale = draw->getScale();
107 TheTerrainRenderObject->addProp((Int)draw->getID(), *draw->getPosition(),
108 draw->getOrientation(), scale, moduleData->m_modelName);
109
110}
