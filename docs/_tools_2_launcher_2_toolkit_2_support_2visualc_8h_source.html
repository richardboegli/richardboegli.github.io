43// "unreferenced inline function has been removed"
44// Inline functions are used in headers and this warning will appear everywhere
45// without explicitly being disabled.
46#pragma warning(disable:4514)
47
48// "conversion from 'double' to 'float', possible loss of data"
49// This occurs during non-constant floating point arithmetic. Since all floating
50// point math is silently upcasted to doubles, it should silently downcast
51// back to 'float' when complete -- hence this warning should not be displayed.
52#pragma warning(disable:4244)
53
54// "overflow in floating-point constant arithmetic"
55// This warning occurs even when there is no overflow. It occurs when a double
56// is downcasted to a float during constant arithmetic (this is not worthy of
57// a warning message).
58#pragma warning(disable:4056)
59
60// "argument trunation from const double to float"
61// This warning is of little use since the compiler uses doubles whenever
62// possible, therfore this warning will appear frequently. It is similar to
63// warning 4244 and is similarly irrelevant.
64#pragma warning(disable:4305)
65
66// "'this' used in base member initializer list"
67// Using "this" in a base member initializer is valid -- no need for this warning.
68#pragma warning(disable:4355)
69
70// "typedef-name used as a synonym for class-name"
71// This is by design and should not be a warning.
72#pragma warning(disable:4097)
73
74// "function not inlined"
75// This warning is typically useless. The inline keyword only serves as a
76// suggestion to the compiler and it may or may not inline a function on a
77// case by case basis. No need to be told of this.
78#pragma warning(disable:4710)
79
80// "function selected for automatic inline expansion"
81// There is no need to announce this with a warning message.
82#pragma warning(disable:4711)

84// "identifier was truncated to 'number' characters in the debug information"
85// The debugger cannot debug code with symbols longer than 255 characters.
86// In the debugger, you cannot view, evaluate, update, or watch the truncated symbols.
87#pragma warning(disable:4786)
