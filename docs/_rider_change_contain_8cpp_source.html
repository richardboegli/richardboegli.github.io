74void RiderChangeContainModuleData::parseRiderInfo( INI* ini, void *instance, void *store, const void* /*userData*/ )
75{
76 RiderInfo* rider = (RiderInfo*)store;
77 const char* name = ini->getNextToken();
78
79 //Template name
80 rider->m_templateName.format( name );
81
82 //Model condition state
83 INI::parseIndexList( ini, instance, &(rider->m_modelConditionFlagType), ModelConditionFlags::getBitNames() );
84
85 //Weaponset
86 INI::parseIndexList( ini, instance, &(rider->m_weaponSetFlag), WeaponSetFlags::getBitNames() );
87
88 //Object status
89 INI::parseIndexList( ini, instance, &(rider->m_objectStatusType), ObjectStatusMaskType::getBitNames() );
90
91 //Command set override
92 name = ini->getNextToken();
93 rider->m_commandSet.format( name );
94
95 //Locomotor set type
96 rider->m_locomotorSetType = (LocomotorSetType)INI::scanIndexList( ini->getNextToken(), TheLocomotorSetNames );
97}

161Bool RiderChangeContain::isValidContainerFor(const Object* rider, Bool checkCapacity) const
162{
163 //Don't check capacity because our rider will kick the other rider out!
164 if( TransportContain::isValidContainerFor( rider, FALSE ) )
165 {
166 if( m_scuttledOnFrame != 0 )
167 {
168 //Scuttled... too late!
169 return FALSE;
170 }
171
172 //We can enter this bike... but now we need to extend the base functionality by limiting
173 //which infantry can enter.
174 const RiderChangeContainModuleData *data = getRiderChangeContainModuleData();
175 for( int i = 0; i < MAX_RIDERS; i++ )
176 {
177 const ThingTemplate *thing = TheThingFactory->findTemplate( data->m_riders[ i ].m_templateName );
178 if( thing->isEquivalentTo( rider->getTemplate() ) )
179 {
180 //We found a valid rider, so return success.
181 return TRUE;
182 }
183 }
184 }
185 return FALSE;
186}

189//-------------------------------------------------------------------------------------------------
190void RiderChangeContain::onContaining( Object *rider, Bool wasSelected )
191{
192 Object *obj = getObject();
193 m_containing = TRUE;
194 //Remove our existing rider
195 if( m_payloadCreated )
196 {
197 obj->getAI()->aiEvacuateInstantly( TRUE, CMD_FROM_AI );
198 }
199
200 //If the rider is currently selected, transfer selection to the container and preserve other units
201 //that may be already selected. Note that containing the rider will automatically cause it to be
202 //deselected, so all we have to do is select the container (if not already selected)!
203 Drawable *containDraw = getObject()->getDrawable();
204 if( containDraw && wasSelected && !containDraw->isSelected() )
205 {
206 //Create the selection message
207 GameMessage *teamMsg = TheMessageStream->appendMessage( GameMessage::MSG_CREATE_SELECTED_GROUP );
208 teamMsg->appendBooleanArgument( FALSE );// not creating new team so pass false
209 teamMsg->appendObjectIDArgument( getObject()->getID() );
210 TheInGameUI->selectDrawable( containDraw );
211 TheInGameUI->setDisplayedMaxWarning( FALSE );
212 }
213
214 //Find the rider in the list and set the appropriate model condition
215 const RiderChangeContainModuleData *data = getRiderChangeContainModuleData();
216 for( int i = 0; i < MAX_RIDERS; i++ )
217 {
218 const ThingTemplate *thing = TheThingFactory->findTemplate( data->m_riders[ i ].m_templateName );
219 if( thing->isEquivalentTo( rider->getTemplate() ) )
220 {
221
222 //This is our rider, so set the correct model condition.
223 obj->setModelConditionState( data->m_riders[ i ].m_modelConditionFlagType );
224
225 //Also set the correct weaponset flag
226 obj->setWeaponSetFlag( data->m_riders[ i ].m_weaponSetFlag );
227
228 //Also set the object status
229 obj->setStatus( MAKE_OBJECT_STATUS_MASK( data->m_riders[ i ].m_objectStatusType ) );
230
231 //Set the new commandset override
232 obj->setCommandSetStringOverride( data->m_riders[ i ].m_commandSet );
233 TheControlBar->markUIDirty(); // Refresh the UI in case we are selected
234
235 //Change the locomotor.
236 AIUpdateInterface* ai = obj->getAI();
237 if( ai )
238 {
239 ai->chooseLocomotorSet( data->m_riders[ i ].m_locomotorSetType );
240 }
241
242 if( obj->getStatusBits().test( OBJECT_STATUS_STEALTHED ) )
243 {
244 StealthUpdate* stealth = obj->getStealth();
245 if( stealth )
246 {
247 stealth->markAsDetected();
248 }
249 }
250
251 //Transfer experience from the rider to the bike.
252 ExperienceTracker *riderTracker = rider->getExperienceTracker();
253 ExperienceTracker *bikeTracker = obj->getExperienceTracker();
254 bikeTracker->setVeterancyLevel( riderTracker->getVeterancyLevel(), FALSE );
255 riderTracker->setExperienceAndLevel( 0, FALSE );
256
257 break;
258 }
259 }
260
261 //Extend base class
262 TransportContain::onContaining( rider, wasSelected );
263
264 m_containing = FALSE;
265}

268//-------------------------------------------------------------------------------------------------
269void RiderChangeContain::onRemoving( Object *rider )
270{
271 Object *bike = getObject();
272 //Note if the bike dies, the rider dies too.
273 if( bike->isEffectivelyDead() )
274 {
275 TheGameLogic->destroyObject( rider );
276 return;
277 }
278
279 if( m_payloadCreated )
280 {
281 //Extend base class
282 TransportContain::onRemoving( rider );
283 }
284
285 //Find the rider in the list and clear various data.
286 const RiderChangeContainModuleData *data = getRiderChangeContainModuleData();
287 for( int i = 0; i < MAX_RIDERS; i++ )
288 {
289 const ThingTemplate *thing = TheThingFactory->findTemplate( data->m_riders[ i ].m_templateName );
290 if( thing->isEquivalentTo( rider->getTemplate() ) )
291 {
292 //This is our rider, so clear the current model condition.
293 bike->clearModelConditionFlags( MAKE_MODELCONDITION_MASK2( data->m_riders[ i ].m_modelConditionFlagType, MODELCONDITION_DOOR_1_CLOSING ) );
294
295 //Also clear the current weaponset flag
296 bike->clearWeaponSetFlag( data->m_riders[ i ].m_weaponSetFlag );
297
298 //Also clear the object status
299 bike->clearStatus( MAKE_OBJECT_STATUS_MASK( data->m_riders[ i ].m_objectStatusType ) );
300
301 if( rider->getControllingPlayer() != NULL )
302 {
303 //Wow, completely unforseeable game teardown order crash. SetVeterancyLevel results in a call to player
304 //about upgrade masks. So if we have a null player, it is game teardown, so don't worry about transfering exp.
305
306 //Transfer experience from the bike to the rider.
307 ExperienceTracker *riderTracker = rider->getExperienceTracker();
308 ExperienceTracker *bikeTracker = bike->getExperienceTracker();
309 riderTracker->setVeterancyLevel( bikeTracker->getVeterancyLevel(), FALSE );
310 bikeTracker->setExperienceAndLevel( 0, FALSE );
311 }
312
313 break;
314 }
315 }
316
317 //If we're not replacing the rider, then if the cycle is selected, transfer selection
318 //to the rider getting off (because the bike is gonna blow).
319 if( !m_containing )
320 {
321 Drawable *containDraw = bike->getDrawable();
322 Drawable *riderDraw = rider->getDrawable();
323 if( containDraw && riderDraw )
324 {
325 //Create the selection message for the rider if it's ours and SELECTED!
326 if( bike->getControllingPlayer() == ThePlayerList->getLocalPlayer() && containDraw->isSelected() )
327 {
328 GameMessage *teamMsg = TheMessageStream->appendMessage( GameMessage::MSG_CREATE_SELECTED_GROUP );
329 teamMsg->appendBooleanArgument( FALSE );// not creating new team so pass false
330 teamMsg->appendObjectIDArgument( rider->getID() );
331 TheInGameUI->selectDrawable( riderDraw );
332 TheInGameUI->setDisplayedMaxWarning( FALSE );
333
334 //Create the de-selection message for the container
335 teamMsg = TheMessageStream->appendMessage( GameMessage::MSG_REMOVE_FROM_SELECTED_GROUP );
336 teamMsg->appendObjectIDArgument( bike->getID() );
337 TheInGameUI->deselectDrawable( containDraw );
338 }
339
340 //Finally, scuttle the bike so nobody else can use it! <Design Spec>
341 m_scuttledOnFrame = TheGameLogic->getFrame();
342 bike->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNSELECTABLE ) );
343 bike->setModelConditionState( data->m_scuttleState );
344 if( !bike->getAI()->isMoving() )
345 {
346 bike->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_IMMOBILE ) );
347 }
348 }
349 }
350}

360// ------------------------------------------------------------------------------------------------
361// ------------------------------------------------------------------------------------------------
362UpdateSleepTime RiderChangeContain::update()
363{
364 if( m_scuttledOnFrame != 0 )
365 {
366 //Bike in the process of getting scuttled.
367 const RiderChangeContainModuleData *data = getRiderChangeContainModuleData();
368 UnsignedInt now = TheGameLogic->getFrame();
369 if( m_scuttledOnFrame + data->m_scuttleFrames <= now )
370 {
371 //We have scuttled the bike (at least as far as tipping it over via scuttle animation. Now
372 //kill the bike in a way that will cause it to sink into the ground without any real destruction.
373 getObject()->kill( DAMAGE_UNRESISTABLE, DEATH_TOPPLED ); //Sneaky, eh? Toppled heheh.
374 }
375 }
376 // extend base class
377 return TransportContain::update(); //extend
378}
