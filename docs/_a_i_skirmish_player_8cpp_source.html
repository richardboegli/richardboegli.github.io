99void AISkirmishPlayer::processBaseBuilding( void )
100{
101 //
102 // Refresh base buildings. Scan through list, if a building is missing,
103 // rebuild it, unless it's rebuild count is zero.
104 //
105 if (m_readyToBuildStructure)
106 {
107 const ThingTemplate *bldgPlan=NULL;
108 BuildListInfo *bldgInfo = NULL;
109 Bool isPriority = false;
110 Object *bldg = NULL;
111 const ThingTemplate *powerPlan=NULL;
112 BuildListInfo *powerInfo = NULL;
113 Bool isUnderPowered = !m_player->getEnergy()->hasSufficientPower();
114 Bool powerUnderConstruction = false;
115 for( BuildListInfo *info = m_player->getBuildList(); info; info = info->getNext() )
116 {
117 AsciiString name = info->getTemplateName();
118 if (name.isEmpty()) continue;
119 const ThingTemplate *curPlan = TheThingFactory->findTemplate( name );
120 if (!curPlan) {
121 DEBUG_LOG(("*** ERROR - Build list building '%s' doesn't exist.\n", name.str()));
122 continue;
123 }
124 bldg = TheGameLogic->findObjectByID( info->getObjectID() );
125 // check for hole.
126 if (info->getObjectID() != INVALID_ID) {
127 // used to have a building.
128 Object *bldg = TheGameLogic->findObjectByID( info->getObjectID() );
129 if (bldg==NULL) {
130 // got destroyed.
131 ObjectID priorID;
132 priorID = info->getObjectID();
133 info->setObjectID(INVALID_ID);
134 info->setObjectTimestamp(TheGameLogic->getFrame()+1);
135 // Scan for a GLA hole. KINDOF_REBUILD_HOLE
136 Object *obj;
137 for( obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject() ) {
138 if (!obj->isKindOf(KINDOF_REBUILD_HOLE)) continue;
139 RebuildHoleBehaviorInterface *rhbi = RebuildHoleBehavior::getRebuildHoleBehaviorInterfaceFromObject( obj );
140 if( rhbi ) {
141 ObjectID spawnerID = rhbi->getSpawnerID();
142 if (priorID == spawnerID) {
143 DEBUG_LOG(("AI Found hole to rebuild %s\n", curPlan->getName().str()));
144 info->setObjectID(obj->getID());
145 }
146 }
147 }
148 } else {
149 if (bldg->getControllingPlayer() == m_player) {
150 // Check for built or dozer missing.
151 if( bldg->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
152 {
153 if (bldg->isKindOf(KINDOF_FS_POWER) && !bldg->isKindOf(KINDOF_CASH_GENERATOR))
154 {
155 powerUnderConstruction = true;
156 }
157 // make sure dozer is working on him.
158 ObjectID builder = bldg->getBuilderID();
159 Object* myDozer = TheGameLogic->findObjectByID(builder);
160
161 if (myDozer && ( myDozer->getControllingPlayer() != m_player || myDozer->isDisabledByType( DISABLED_UNMANNED ) ) )
162 {//I don't expect this dozer to work well with me.
163 myDozer = NULL;
164 bldg->setBuilder( NULL );
165 }
166
167 if (myDozer==NULL) {
168 DEBUG_LOG(("AI's Dozer got killed (or captured). Find another dozer.\n"));
169 queueDozer();
170 myDozer = findDozer(bldg->getPosition());
171 if (myDozer==NULL || myDozer->getAI()==NULL) {
172 continue;
173 }
174 myDozer->getAI()->aiResumeConstruction(bldg, CMD_FROM_AI);
175 } else {
176 // make sure he is building.
177 myDozer->getAI()->aiResumeConstruction(bldg, CMD_FROM_AI);
178 }
179 }
180 } else {
181 // oops, got captured.
182 info->setObjectID(INVALID_ID);
183 info->setObjectTimestamp(TheGameLogic->getFrame()+1);
184 }
185 }
186 }
187 if (info->getObjectID()==INVALID_ID && info->getObjectTimestamp()>0) {
188 // this object was built at some time, and got destroyed at or near objectTimestamp.
189 // Wait a few seconds before initiating a rebuild.
190 if (info->getObjectTimestamp()+TheAI->getAiData()->m_rebuildDelaySeconds*LOGICFRAMES_PER_SECOND > TheGameLogic->getFrame()) {
191 continue;
192 } else {
193 DEBUG_LOG(("Enabling rebuild for %s\n", info->getTemplateName().str()));
194 info->setObjectTimestamp(0); // ready to build.
195 }
196 }
197 if (bldg) {
198 continue; // already built.
199 }
200 // Make sure it is safe to build here.
201 if (!isLocationSafe(info->getLocation(), curPlan)) {
202 continue;
203 }
204 if (info->isPriorityBuild()) {
205 // Always take priority build, unless we already have priority build.
206 if (!isPriority) {
207 bldgPlan = curPlan;
208 bldgInfo = info;
209 isPriority = true;
210 }
211 }
212 if (curPlan->isKindOf(KINDOF_FS_POWER)) {
213 if (powerPlan==NULL && !curPlan->isKindOf(KINDOF_CASH_GENERATOR)) {
214 if (isUnderPowered || info->isAutomaticBuild()) {
215 powerPlan = curPlan;
216 powerInfo = info;
217 }
218 }
219 }
220 if (!info->isAutomaticBuild()) {
221 continue; // marked to not build automatically.
222 }
223 Object *dozer = findDozer(info->getLocation());
224 if (dozer==NULL) {
225 if (isUnderPowered) {
226 queueDozer();
227 }
228 continue;
229 }
230 if (TheBuildAssistant->canMakeUnit(dozer, bldgPlan)!=CANMAKE_OK) {
231 if (info->isBuildable()) {
232 AsciiString bldgName = info->getTemplateName();
233 bldgName.concat(" - Dozer unable to build - money or technology missing.");
234 TheScriptEngine->AppendDebugMessage(bldgName, false);
235 }
236 continue;
237 }
238 // check if this building has any "rebuilds" left
239 if (info->isBuildable())
240 {
241 if (bldgPlan == NULL) {
242 bldgPlan = curPlan;
243 bldgInfo = info;
244 }
245 }
246 }
247 if (powerPlan && powerInfo && !powerPlan->isEquivalentTo(bldgPlan)) {
248 if (!powerUnderConstruction) {
249 bldgPlan = powerPlan;
250 bldgInfo = powerInfo;
251 DEBUG_LOG(("Forcing build of power plant.\n"));
252 }
253 }
254 if (bldgPlan && bldgInfo) {
255#ifdef USE_DOZER
256 // dozer-construct the building
257 bldg = buildStructureWithDozer(bldgPlan, bldgInfo);
258 // store the object with the build order
259 if (bldg)
260 {
261 bldgInfo->setObjectID( bldg->getID() );
262 bldgInfo->decrementNumRebuilds();
263
264 m_readyToBuildStructure = false;
265 m_structureTimer = TheAI->getAiData()->m_structureSeconds*LOGICFRAMES_PER_SECOND;
266 if (m_player->getMoney()->countMoney() < TheAI->getAiData()->m_resourcesPoor) {
267 m_structureTimer = m_structureTimer/TheAI->getAiData()->m_structuresPoorMod;
268 } else if (m_player->getMoney()->countMoney() > TheAI->getAiData()->m_resourcesWealthy) {
269 m_structureTimer = m_structureTimer/TheAI->getAiData()->m_structuresWealthyMod;
270 }
271 m_frameLastBuildingBuilt = TheGameLogic->getFrame();
272 // only build one building per delay loop
273 } // bldg built
274
275#else
276 // force delay between rebuilds
277 if (TheGameLogic->getFrame() - m_frameLastBuildingBuilt < framesToBuild)
278 {
279 m_buildDelay = framesToBuild - (TheGameLogic->getFrame() - m_frameLastBuildingBuilt);
280 return;
281 } else {
282 // building is missing, (re)build it
283 // deduct money to build, if we have it
284 Int cost = bldgPlan->calcCostToBuild( m_player );
285 if (m_player->getMoney()->countMoney() >= cost)
286 {
287 // we have the money, deduct it
288 m_player->getMoney()->withdraw( cost );
289
290 // inst-construct the building
291 bldg = buildStructureNow(bldgPlan, info, NULL);
292 // store the object with the build order
293 if (bldg)
294 {
295 info->setObjectID( bldg->getID() );
296 info->decrementNumRebuilds();
297
298 m_readyToBuildStructure = false;
299 m_structureTimer = TheAI->getAiData()->m_structureSeconds*LOGICFRAMES_PER_SECOND;
300 if (m_player->getMoney()->countMoney() < TheAI->getAiData()->m_resourcesPoor) {
301 m_structureTimer = m_structureTimer/TheAI->getAiData()->m_structuresPoorMod;
302 } else if (m_player->getMoney()->countMoney() > TheAI->getAiData()->m_resourcesWealthy) {
303 m_structureTimer = m_structureTimer/TheAI->getAiData()->m_structuresWealthyMod;
304 }
305 m_frameLastBuildingBuilt = TheGameLogic->getFrame();
306 // only build one building per delay loop
307 break;
308 } // bldg built
309 } // have money
310 } // rebuild delay ok
311#endif
312 }
313 }

357Bool AISkirmishPlayer::isAGoodIdeaToBuildTeam( TeamPrototype *proto )
358{
359 // Check condition.
360 if (!proto->evaluateProductionCondition()) {
361 return false;
362 }
363 // check build limit
364 if (proto->countTeamInstances() >= proto->getTemplateInfo()->m_maxInstances){
365 if (TheGlobalData->m_debugAI) {
366 AsciiString str;
367 str.format("Team %s not chosen - %d already exist.", proto->getName().str(), proto->countTeamInstances());
368 TheScriptEngine->AppendDebugMessage(str, false);
369 }
370 return false; // Max already built.
371 }
372
373 for ( DLINK_ITERATOR<TeamInQueue> iter = iterate_TeamBuildQueue(); !iter.done(); iter.advance())
374 {
375 TeamInQueue *team = iter.cur();
376 if (team->m_team->getPrototype() == proto) {
377 return false; // currently building one of these.
378 }
379 }
380 Bool needMoney;
381 if (!isPossibleToBuildTeam( proto, true, needMoney)) {
382 if (TheGlobalData->m_debugAI) {
383 AsciiString str;
384 if (needMoney) {
385 str.format("Team %s not chosen - Not enough money.", proto->getName().str());
386 } else {
387 str.format("Team %s not chosen - Factory/tech missing or busy.", proto->getName().str());
388 }
389 TheScriptEngine->AppendDebugMessage(str, false);
390 }
391 return false;
392 }
393 return true;

485void AISkirmishPlayer::acquireEnemy(void)
486{
487 Player *bestEnemy = NULL;
488 Real bestDistanceSqr = HUGE_DIST*HUGE_DIST;
489
490 if (m_currentEnemy) {
491 Bool inBadShape = !m_currentEnemy->hasAnyUnits() || !m_currentEnemy->hasAnyBuildFacility();
492 if (!inBadShape) return;
493 }
494
495 // look for the closest enemy.
496 Int i;
497 for (i=0; i<ThePlayerList->getPlayerCount(); i++) {
498 Player *curPlayer = ThePlayerList->getNthPlayer(i);
499 if (m_player->getRelationship(curPlayer->getDefaultTeam()) == ENEMIES) {
500 if (curPlayer->hasAnyObjects()==false) continue; // not much of an enemy.
501 // ok, we got an enemy;
502 // If a player is out of units, or out of build facilities, we can lower his priority.
503 Bool inBadShape = !curPlayer->hasAnyUnits() || !curPlayer->hasAnyBuildFacility();
504
505 Coord3D enemyPos = m_baseCenter;
506 Region2D bounds;
507 getPlayerStructureBounds(&bounds, i);
508 enemyPos.x = bounds.lo.x + bounds.width()/2;
509 enemyPos.y = bounds.lo.y + bounds.height()/2;
510 Real curDistSqr = sqr(enemyPos.x-m_baseCenter.x) + sqr(enemyPos.y-m_baseCenter.y);
511
512 //Fudge for in bad shape. If an enemy is crippled, concentrate on the other ones.
513 if (inBadShape) {
514 curDistSqr = HUGE_DIST*HUGE_DIST*0.5f;
515 }
516 // See if other ai's are attacking this target.
517 // We don't want the ai's to gang up on one enemy.
518 Int k;
519 for (k=0; k<ThePlayerList->getPlayerCount(); k++) {
520 if (k==i) continue; // don't count self.
521 Player *somePlayer = ThePlayerList->getNthPlayer(k);
522 if (somePlayer->isSkirmishAIPlayer() && (somePlayer->getCurrentEnemy()==curPlayer)) {
523 // Some ai is already targeting this guy. Add a distance penalty.
524 curDistSqr += (500*500);
525 }
526 if (somePlayer->isSkirmishAIPlayer() && (somePlayer->getCurrentEnemy()==m_player)) {
527 // he is attacking me. So I will (gently) prefer to attack him.
528 curDistSqr -= (25*25);
529 if (curDistSqr<0) curDistSqr = 0;
530 }
531 }
532
533 // Ai enemy - will take if we don't get a better offer.
534 if (curDistSqr<bestDistanceSqr) {
535 bestEnemy = curPlayer;
536 bestDistanceSqr = curDistSqr;
537 }
538 }
539 }
540 if (bestEnemy!=NULL && (bestEnemy!=m_currentEnemy)) {
541 m_currentEnemy = bestEnemy;
542 AsciiString msg = TheNameKeyGenerator->keyToName(m_player->getPlayerNameKey());
543 msg.concat(" acquiring target enemy player: ");
544 msg.concat(TheNameKeyGenerator->keyToName(m_currentEnemy->getPlayerNameKey()));
545 TheScriptEngine->AppendDebugMessage( msg, false);
546 }
547}

604void AISkirmishPlayer::buildAIBaseDefenseStructure(const AsciiString &thingName, Bool flank)
605{
606 const ThingTemplate *tTemplate = TheThingFactory->findTemplate(thingName);
607 if (tTemplate==NULL) {
608 DEBUG_CRASH(("Couldn't find base defense structure '%s' for side %s", thingName.str(), m_player->getSide().str()));
609 return;
610 }
611 do {
612 AsciiString pathLabel;
613 if (flank) {
614 if (m_curFlankBaseDefense&1) {
615 pathLabel.format("%s%d", SKIRMISH_FLANK, m_player->getMpStartIndex()+1);
616 } else {
617 pathLabel.format("%s%d", SKIRMISH_BACKDOOR, m_player->getMpStartIndex()+1);
618 }
619 } else {
620 pathLabel.format("%s%d", SKIRMISH_CENTER, m_player->getMpStartIndex()+1);
621 }
622
623 Coord3D goalPos = m_baseCenter;
624 Waypoint *way = TheTerrainLogic->getClosestWaypointOnPath( &goalPos, pathLabel );
625 if (way) {
626 goalPos = *way->getLocation();
627 } else {
628 if (flank) return;
629 Region2D bounds;
630 getPlayerStructureBounds(&bounds, getMyEnemyPlayerIndex());
631 goalPos.x = bounds.lo.x + bounds.width()/2;
632 goalPos.y = bounds.lo.y + bounds.height()/2;
633 }
634 Coord2D offset;
635 offset.x = goalPos.x-m_baseCenter.x;
636 offset.y = goalPos.y-m_baseCenter.y;
637 offset.normalize();
638 Real defenseDistance = m_baseRadius;
639 defenseDistance += TheAI->getAiData()->m_skirmishBaseDefenseExtraDistance;
640 offset.x *= defenseDistance;
641 offset.y *= defenseDistance;
642
643 Real structureRadius = tTemplate->getTemplateGeometryInfo().getBoundingCircleRadius();
644 Real baseCircumference = 2*PI*defenseDistance;
645 Real angleOffset = 2*PI*(structureRadius*4/baseCircumference);
646
647 Int selector;
648 Real angle;
649 if (flank) {
650 selector = m_curFlankBaseDefense>>1;
651 if (m_curFlankBaseDefense&1) {
652 if (selector&1) {
653 m_curLeftFlankRightDefenseAngle -= angleOffset;
654 angle = m_curLeftFlankRightDefenseAngle;
655 } else {
656 angle = m_curLeftFlankLeftDefenseAngle;
657 m_curLeftFlankLeftDefenseAngle += angleOffset;
658 }
659 } else {
660 if (selector&1) {
661 m_curRightFlankRightDefenseAngle -= angleOffset;
662 angle = m_curRightFlankRightDefenseAngle;
663 } else {
664 angle = m_curRightFlankLeftDefenseAngle;
665 m_curRightFlankLeftDefenseAngle += angleOffset;
666 }
667 }
668
669 } else {
670 selector = m_curFrontBaseDefense;
671 if (selector&1) {
672 m_curFrontRightDefenseAngle -= angleOffset;
673 angle = m_curFrontRightDefenseAngle;
674 } else {
675 angle = m_curFrontLeftDefenseAngle;
676 m_curFrontLeftDefenseAngle += angleOffset;
677 }
678 }
679
680 if (angle > PI/3) break;
681 Real s = sin(angle);
682 Real c = cos(angle);
683
684 DEBUG_LOG(("buildAIBaseDefenseStructure -- Angle is %f sin %f, cos %f \n", 180*angle/PI, s, c));
685 DEBUG_LOG(("buildAIBaseDefenseStructure -- Offset is %f %f, Final Position is %f, %f \n",
686 offset.x, offset.y,
687 offset.x*c - offset.y*s,
688 offset.y*c + offset.x*s
689 ));
690 Coord3D buildPos = m_baseCenter;
691 buildPos.x += offset.x*c - offset.y*s;
692 buildPos.y += offset.y*c + offset.x*s;
693
694 /* See if we can build there. */
695 Bool canBuild;
696 Real placeAngle = tTemplate->getPlacementViewAngle();
697 canBuild = LBC_OK == TheBuildAssistant->isLocationLegalToBuild(&buildPos, tTemplate, placeAngle,
698 BuildAssistant::TERRAIN_RESTRICTIONS|BuildAssistant::NO_OBJECT_OVERLAP, NULL, m_player);
699 TheTerrainVisual->removeAllBibs(); // isLocationLegalToBuild adds bib feedback, turn it off. jba.
700 if (flank) {
701 m_curFlankBaseDefense++;
702 } else {
703 m_curFrontBaseDefense++;
704 }
705 if (canBuild) {
706 m_player->addToPriorityBuildList(thingName, &buildPos, placeAngle);
707 break;
708 }
709 } while (true);
710
711}

718Bool AISkirmishPlayer::checkBridges(Object *unit, Waypoint *way)
719{
720 Coord3D unitPos = *unit->getPosition();
721 AIUpdateInterface *ai = unit->getAI();
722 if (!ai) return false; // no ai
723 const LocomotorSet& locoSet = ai->getLocomotorSet();

965void AISkirmishPlayer::adjustBuildList(BuildListInfo *list)
966{
967 Bool foundStart = false;
968 Coord3D startPos;
969
970 // Find our command center location.
971 Object *obj;
972 for( obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject() )
973 {
974
975 Player *owner = obj->getControllingPlayer();
976 if (owner==m_player) {
977 // See if it's a command center.
978 if (obj->isKindOf(KINDOF_COMMANDCENTER)) {
979 foundStart = true;
980 startPos = *obj->getPosition();
981 m_player->onStructureUndone(obj);
982 TheAI->pathfinder()->removeObjectFromPathfindMap(obj);
983 TheGameLogic->destroyObject(obj);
984 break;
985 }
986 }
987 }
988 if (!foundStart) {
989 DEBUG_LOG(("Couldn't find starting command center for ai player.\n"));
990 return;
991 }
992 // Find the location of the command center in the build list.
993 Bool foundInBuildList = false;
994 Coord3D buildPos;
995 BuildListInfo *cur = list;
996 while (cur) {
997 const ThingTemplate *tTemplate = TheThingFactory->findTemplate(cur->getTemplateName());
998 if (tTemplate && tTemplate->isKindOf(KINDOF_COMMANDCENTER)) {
999 foundInBuildList = true;
1000 buildPos = *cur->getLocation();
1001 cur->setInitiallyBuilt(true);
1002 }
1003 cur = cur->getNext();
1004 }
1005 Region3D bounds;
1006 TheTerrainLogic->getMaximumPathfindExtent(&bounds);
1007 /* calculate section of 3x3 grid:
1008 6 7 8
1009 3 4 5
1010 0 1 2 */
1011
1012 Int gridIndex = 0;
1013 if (startPos.x > bounds.lo.x + bounds.width()/3) {
1014 gridIndex++;
1015 }
1016 if (startPos.x > bounds.lo.x + 2*bounds.width()/3) {
1017 gridIndex++;
1018 }
1019
1020 if (startPos.y > bounds.lo.y + bounds.height()/3) {
1021 gridIndex+=3;
1022 }
1023 if (startPos.y > bounds.lo.y + 2*bounds.height()/3) {
1024 gridIndex+=3;
1025 }
1026
1027 Real angle = 0;
1028 if (TheAI->getAiData()->m_rotateSkirmishBases) {
1029 switch (gridIndex) {
1030 case 0 : angle = 0; break;
1031 case 1 : angle = PI/4; break;// 45 degrees.
1032 case 2 : angle = PI/2; break; // 90 degrees;
1033 case 3 : angle = -PI/4; break; // -45 degrees.
1034 case 4 : angle = 0; break;
1035 case 5 : angle = 3*PI/4; break; // 135 degrees.
1036 case 6 : angle = -PI/2; break; // -90 degrees;
1037 case 7 : angle = -3*PI/4; break; // -135 degrees.
1038 case 8 : angle = PI; break; // 180 degrees.
1039 }
1040 }
1041
1042 angle += 3*PI/4;
1043
1044 Real s = sin(angle);
1045 Real c = cos(angle);
1046
1047 cur = list;
1048 while (cur) {
1049 const ThingTemplate *tTemplate = TheThingFactory->findTemplate(list->getTemplateName());
1050 if (tTemplate && tTemplate->isKindOf(KINDOF_COMMANDCENTER)) {
1051 foundInBuildList = true;
1052 Coord3D curPos = *cur->getLocation();
1053 // Transform to new coords.
1054 curPos.x -= buildPos.x;
1055 curPos.y -= buildPos.y;
1056 Real newX = curPos.x*c - curPos.y*s;
1057 Real newY = curPos.y*c + curPos.x*s;
1058 curPos.x = newX + startPos.x;
1059 curPos.y = newY + startPos.y;
1060 cur->setLocation(curPos);
1061 cur->setAngle(cur->getAngle());
1062 }
1063 cur = cur->getNext();
1064 }
1065
1066}
