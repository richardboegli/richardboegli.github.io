64void ShadowOptions::setShadowColor(void)
65{
66 Int r, g, b, shift;
67
68 shift = (1.0-m_intensity)*255;
69 if (shift>255) shift = 255;
70 if (shift<0) shift = 0;
71
72 r = m_intensity*m_red*255 + shift;
73 if (r>255) r = 255;
74 if (r<0) r = 0;
75
76 g = m_intensity*m_green*255 + shift;
77 if (g>255) g = 255;
78 if (g<0) g = 0;
79
80 b = m_intensity*m_blue*255 + shift;
81 if (b>255) b = 255;
82 if (b<0) b = 0;
83
84 UnsignedInt clr = (255<<24) + (r<<16) + (g<<8) + b;
85 DEBUG_LOG(("Setting shadows to %x\n", clr));
86 TheW3DShadowManager->setShadowColor(clr);
87}

145BOOL ShadowOptions::OnInitDialog()
146{
147 CDialog::OnInitDialog();
148
149 UnsignedInt clr = TheW3DShadowManager->getShadowColor();
150 m_red = ((clr>>16)&0x00FF)/255.0f;
151 m_green = ((clr>>8)&0x00FF)/255.0f;
152 m_blue = ((clr)&0x00FF)/255.0f;
153
154 m_intensity = m_red;
155 if (m_green<m_red) m_intensity = m_green;
156 if (m_blue < m_intensity) m_intensity = m_blue;
157 m_intensity = 1.0f - m_intensity;
158 if (m_intensity < (1/256.0f)) {
159 m_intensity = 0;
160 m_red = m_green = m_blue = 0;
161 } else {
162 m_red -= (1.0-m_intensity);
163 m_red /= m_intensity;
164 m_green -= (1.0-m_intensity);
165 m_green /= m_intensity;
166 m_blue -= (1.0-m_intensity);
167 m_blue /= m_intensity;
168 }
169
170 CWnd *pEdit = GetDlgItem(IDC_RA_EDIT);
171 CString text;
172 if (pEdit) {
173 text.Format("%.2f", m_red);
174 pEdit->SetWindowText(text);
175 }
176 pEdit = GetDlgItem(IDC_BA_EDIT);
177 if (pEdit) {
178 text.Format("%.2f", m_blue);
179 pEdit->SetWindowText(text);
180 }
181 pEdit = GetDlgItem(IDC_GA_EDIT);
182 if (pEdit) {
183 text.Format("%.2f", m_green);
184 pEdit->SetWindowText(text);
185 }
186 pEdit = GetDlgItem(IDC_ALPHA_EDIT);
187 if (pEdit) {
188 text.Format("%.2f", m_intensity);
189 pEdit->SetWindowText(text);
190 }
191 return TRUE; // return TRUE unless you set the focus to a control
192 // EXCEPTION: OCX Property Pages should return FALSE
193}
