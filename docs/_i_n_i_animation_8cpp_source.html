39void INI::parseAnim2DDefinition( INI* ini )
40{
41 AsciiString name;
42 Anim2DTemplate *animTemplate;
43
44 // read the name
45 const char* c = ini->getNextToken();
46 name.set( c );
47
48 //
49 // find existing item if present, note that we do not support overrides
50 // in the animations like we do in systems that are more "design" oriented, images
51 // are assets as they are
52 //
53 if( !TheAnim2DCollection )
54 {
55
56 //We don't need it if we're in the builder... which doesn't have this.
57 return;
58
59 } // end if
60
61 // find existing animation template if present
62 animTemplate = TheAnim2DCollection->findTemplate( name );
63 if( animTemplate == NULL )
64 {
65
66 // item not found, create a new one
67 animTemplate = TheAnim2DCollection->newTemplate( name );
68 DEBUG_ASSERTCRASH( animTemplate, ("INI""parseAnim2DDefinition - unable to allocate animation template for '%s'\n",
69 name.str()) );
70
71 } // end if
72 else
73 {
74
75 // we're loading over an existing animation template ... something is probably wrong
76 DEBUG_CRASH(( "INI::parseAnim2DDefinition - Animation template '%s' already exists\n",
77 animTemplate->getName().str() ));
78 return;
79
80 } // end else
81
82 // parse the ini definition
83 ini->initFromINI( animTemplate, animTemplate->getFieldParse() );
84
85} // end parseAnim2DDefinition
