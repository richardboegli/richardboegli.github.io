107FrameDataReturnType FrameData::allCommandsReady(Bool debugSpewage) {
108 if (m_frameCommandCount == m_commandCount) {
109 m_lastFailedFrameCC = -2;
110 m_lastFailedCC = -2;
111 return FRAMEDATA_READY;
112 }
113
114 if (debugSpewage) {
115 if ((m_lastFailedFrameCC != m_frameCommandCount) || (m_lastFailedCC != m_commandCount)) {
116 DEBUG_LOG(("FrameData::allCommandsReady - failed, frame command count = %d, command count = %d\n", m_frameCommandCount, m_commandCount));
117 m_lastFailedFrameCC = m_frameCommandCount;
118 m_lastFailedCC = m_commandCount;
119 }
120 }
121
122 if (m_commandCount > m_frameCommandCount) {
123 DEBUG_LOG(("FrameData::allCommandsReady - There are more commands than there should be (%d, should be %d). Commands in command list are...\n", m_commandCount, m_frameCommandCount));
124 NetCommandRef *ref = m_commandList->getFirstMessage();
125 while (ref != NULL) {
126 DEBUG_LOG(("%s, frame = %d, id = %d\n", GetAsciiNetCommandType(ref->getCommand()->getNetCommandType()).str(), ref->getCommand()->getExecutionFrame(), ref->getCommand()->getID()));
127 ref = ref->getNext();
128 }
129 DEBUG_LOG(("FrameData::allCommandsReady - End of command list.\n"));
130 DEBUG_LOG(("FrameData::allCommandsReady - about to clear the command list\n"));
131 reset();
132 DEBUG_LOG(("FrameData::allCommandsReady - command list cleared. command list length = %d, command count = %d, frame command count = %d\n", m_commandList->length(), m_commandCount, m_frameCommandCount));
133 return FRAMEDATA_RESEND;
134 }
135 return FRAMEDATA_NOTREADY;
136}

163void FrameData::addCommand(NetCommandMsg *msg) {
164 // need to add the message in order of command ID
165 if (m_commandList == NULL) {
166 init();
167 }
168
169 // We don't need to worry about setting the relay since its not getting sent anywhere.
170 if (m_commandList->findMessage(msg) != NULL) {
171 // We don't want to add the same command twice.
172 return;
173 }
174 m_commandList->addMessage(msg);
175
176 ++m_commandCount;
177 //DEBUG_LOG(("added command %d, type = %d(%s), command count = %d, frame command count = %d\n", msg->getID(), msg->getNetCommandType(), GetAsciiNetCommandType(msg->getNetCommandType()).str(), m_commandCount, m_frameCommandCount));
178}
