77//=============================================================================
78static void drawTextEntryText( GameWindow *window, WinInstanceData *instData,
79 Color textColor, Color textDropColor,
80 Color compositeColor, Color compositeDropColor,
81 Int x, Int y, Int width, Int fontHeight )
82{
83 static Byte drawCnt = 0;
84 EntryData *e = (EntryData *)window->winGetUserData();
85// Int charPos = e->charPos;
86 Int cursorPos;
87
88// WideChar buffer[ ENTRY_TEXT_LEN + 1 ];
89// WideChar *bufptr = buffer;
90// Color constructColor = TheWindowManager->winMakeColor( 192, 0, 192, 255 );
91 DisplayString *text = e->text;
92 IRegion2D clipRegion;
93 ICoord2D origin, size;
94 Int compositeCursorPos = 0;
95
96 // Check to see if the IME manager is composing text
97 e->constructText->setText(UnicodeString::TheEmptyString);
98 if ( TheIMEManager && TheIMEManager->isAttachedTo( window) && TheIMEManager->isComposing())
99 {
100 // The user is composing a string.
101 // Show the composition in the text gadget.
102 UnicodeString composition;
103
104 TheIMEManager->getCompositionString( composition );
105
106 if ( e->secretText )
107 {
108 e->sText->setText( UnicodeString::TheEmptyString );
109 Int len = composition.getLength() + e->text->getTextLength();
110 for ( int i = 0; i < len; i++ )
111 {
112 e->sText->appendChar( '*' );
113 }
114 }
115 else
116 {
117 e->constructText->setText( composition );
118 compositeCursorPos = TheIMEManager->getCompositionCursorPosition();
119 }
120
121 }
122
123
124 // get out of here if no text color to show up
125 if( textColor == WIN_COLOR_UNDEFINED )
126 return;
127
128 // if our text is "secret" we will print only '*' characters
129 if( e->secretText )
130 text = e->sText;
131
132 // make sure our font is the same as our parents
133 if( text->getFont() != window->winGetFont() )
134 text->setFont( window->winGetFont() );
135 if( e->constructText->getFont() != window->winGetFont() )
136 e->constructText->setFont( window->winGetFont() );
137
138 // get the size of our text, and construct text
139 Int textWidth = text->getWidth();
140
141 if (!e->drawTextFromStart)
142 {
143 // clip the text to the edit window size
144 window->winGetScreenPosition( &origin.x, &origin.y );
145 window->winGetSize( &size.x, &size.y );
146 clipRegion.lo.x = x;
147 clipRegion.hi.x = x + width ;
148 clipRegion.lo.y = y;
149 clipRegion.hi.y = y + fontHeight;
150 text->setClipRegion( &clipRegion );
151 e->constructText->setClipRegion( &clipRegion );
152
153 // set construct window position if needed
154 //if( e->constructList && e->constructText->getTextLength() )
155 // e->constructList->winSetPosition( (x + textWidth1), (y + fontHeight) );
156
157 x+= 2;
158 // draw the text
159 if(textWidth < width)
160 {
161 text->draw( x, y, textColor, textDropColor );
162 cursorPos = textWidth + x;
163 }
164 else
165 {
166 Int div = textWidth / (width / 2) - 1;
167 text->draw(x - (div * (width/2)), y, textColor, textDropColor);
168 cursorPos = textWidth - (div * (width/2)) + x;
169 }
170
171 //cursorPos = x + textWidth;
172 }
173 else
174 {
175 window->winGetScreenPosition( &origin.x, &origin.y );
176 window->winGetSize( &size.x, &size.y );
177 clipRegion.lo.x = origin.x;
178 clipRegion.hi.x = origin.x + size.x;
179 clipRegion.lo.y = origin.y;
180 clipRegion.hi.y = origin.y + size.y;
181 text->setClipRegion( &clipRegion );
182 e->constructText->setClipRegion( &clipRegion );
183
184 // set construct window position if needed
185 //if( e->constructList && e->constructText->getTextLength() )
186 // e->constructList->winSetPosition( (x + textWidth1), (y + fontHeight) );
187
188 x+= 5;
189 // draw the text
190 text->draw( x, y, textColor, textDropColor );
191 cursorPos = textWidth + x;
192 }
193
194 if (e->constructText->getTextLength() > 0 )
195 {
196 e->constructText->draw( x + textWidth, y, compositeColor, compositeDropColor );
197 cursorPos += e->constructText->getWidth( compositeCursorPos );
198 }
199
200 // draw blinking cursor
201 GameWindow *parent;
202 parent = window->winGetParent();
203 if(parent && !BitTest(parent->winGetStyle(), GWS_COMBO_BOX))
204 parent = NULL;
205
206 if( (window == TheWindowManager->winGetFocus() || (parent && parent == TheWindowManager->winGetFocus())) && ((drawCnt++ >> 3) & 0x1) )
207 TheWindowManager->winFillRect( textColor, WIN_DRAW_LINE_WIDTH,
208 cursorPos, origin.y + 3,
209 cursorPos + 2, origin.y + size.y - 3 );
210 window->winSetCursorPosition( cursorPos + 2 - origin.x, 0 );
211
212} // end drawTextEntryText

221void W3DGadgetTextEntryDraw( GameWindow *window, WinInstanceData *instData )
222{
223 EntryData *e = (EntryData *)window->winGetUserData();
224 ICoord2D origin, size, start, end;
225 Color backBorder, backColor, textColor, textBorder,
226 compositeColor, compositeBorder;
227
228 // cancel unichar flag
229 e->receivedUnichar = FALSE;
230
231 // get size and position of window
232 window->winGetScreenPosition( &origin.x, &origin.y );
233 window->winGetSize( &size.x, &size.y );
234
235 // get the right colors
236 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
237 {
238
239 compositeColor = window->winGetDisabledTextColor();
240 compositeBorder = window->winGetDisabledTextBorderColor();
241 textColor = window->winGetDisabledTextColor();
242 textBorder = window->winGetDisabledTextBorderColor();
243 backColor = GadgetTextEntryGetDisabledColor( window );
244 backBorder = GadgetTextEntryGetDisabledBorderColor( window );
245
246 } // end if, disabled
247 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
248 {
249
250 compositeColor = window->winGetIMECompositeTextColor();
251 compositeBorder = window->winGetIMECompositeBorderColor();
252 textColor = window->winGetHiliteTextColor();
253 textBorder = window->winGetHiliteTextBorderColor();
254 backColor = GadgetTextEntryGetHiliteColor( window );
255 backBorder = GadgetTextEntryGetHiliteBorderColor( window );
256
257 } // end else if, hilited
258 else
259 {
260
261 compositeColor = window->winGetIMECompositeTextColor();
262 compositeBorder = window->winGetIMECompositeBorderColor();
263 textColor = window->winGetEnabledTextColor();
264 textBorder = window->winGetEnabledTextBorderColor();
265 backColor = GadgetTextEntryGetEnabledColor( window );
266 backBorder = GadgetTextEntryGetEnabledBorderColor( window );
267
268 } // end else, just enabled
269
270 // draw the back border
271 if( backBorder != WIN_COLOR_UNDEFINED )
272 {
273
274 start.x = origin.x;
275 start.y = origin.y;
276 end.x = start.x + size.x;
277 end.y = start.y + size.y;
278 TheWindowManager->winOpenRect( backBorder, WIN_DRAW_LINE_WIDTH,
279 start.x, start.y, end.x, end.y );
280
281 } // end if
282
283 // draw the filled back
284 if( backColor != WIN_COLOR_UNDEFINED )
285 {
286
287 start.x = origin.x + 1;
288 start.y = origin.y + 1;
289 end.x = start.x + size.x - 2;
290 end.y = start.y + size.y - 2;
291 TheWindowManager->winFillRect( backColor, WIN_DRAW_LINE_WIDTH,
292 start.x, start.y, end.x, end.y );
293
294 } // end if
295
296 // draw the text
297 Int fontHeight = TheWindowManager->winFontHeight( instData->getFont() );
298 Int startOffset = 5;
299 Int width;
300
301 width = size.x - (2 * startOffset);
302 start.x = origin.x + startOffset; // offset a little bit into the entry
303 if( BitTest( window->winGetStatus(), WIN_STATUS_ONE_LINE ) )
304 start.y = size.y / 2 - fontHeight / 2;
305 else
306 start.y = origin.y + startOffset; // offset a little bit into the entry
307
308 // draw the edit text
309 drawTextEntryText( window, instData, textColor, textBorder, compositeColor, compositeBorder,
310 start.x, start.y, width, fontHeight );
311
312
313
314} // end W3DGadgetTextEntryDraw

319void W3DGadgetTextEntryImageDraw( GameWindow *window, WinInstanceData *instData )
320{
321 EntryData *e = (EntryData *)window->winGetUserData();
322 ICoord2D origin, size, start, end;
323 Color textColor, textBorder;
324 Color compositeColor, compositeBorder;
325 const Image *leftImage, *rightImage, *centerImage, *smallCenterImage;
326 Int xOffset, yOffset;
327 Int i;
328
329 // cancel unichar flag
330 e->receivedUnichar = FALSE;
331
332 // get size and position of window
333 window->winGetScreenPosition( &origin.x, &origin.y );
334 window->winGetSize( &size.x, &size.y );
335
336 // get image offset
337 xOffset = instData->m_imageOffset.x;
338 yOffset = instData->m_imageOffset.y;
339
340 // get the right colors
341 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
342 {
343
344 textColor = window->winGetDisabledTextColor();
345 textBorder = window->winGetDisabledTextBorderColor();
346 compositeColor = window->winGetDisabledTextColor();
347 compositeBorder = window->winGetDisabledTextBorderColor();
348 leftImage = GadgetTextEntryGetDisabledImageLeft( window );
349 rightImage = GadgetTextEntryGetDisabledImageRight( window );
350 centerImage = GadgetTextEntryGetDisabledImageCenter( window );
351 smallCenterImage = GadgetTextEntryGetDisabledImageSmallCenter( window );
352
353 } // end if, disabled
354 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
355 {
356
357 textColor = window->winGetHiliteTextColor();
358 textBorder = window->winGetHiliteTextBorderColor();
359 compositeColor = window->winGetIMECompositeTextColor();
360 compositeBorder = window->winGetIMECompositeBorderColor();
361 leftImage = GadgetTextEntryGetHiliteImageLeft( window );
362 rightImage = GadgetTextEntryGetHiliteImageRight( window );
363 centerImage = GadgetTextEntryGetHiliteImageCenter( window );
364 smallCenterImage = GadgetTextEntryGetHiliteImageSmallCenter( window );
365
366 } // end else if, hilited
367 else
368 {
369
370 textColor = window->winGetEnabledTextColor();
371 textBorder = window->winGetEnabledTextBorderColor();
372 compositeColor = window->winGetIMECompositeTextColor();
373 compositeBorder = window->winGetIMECompositeBorderColor();
374 leftImage = GadgetTextEntryGetEnabledImageLeft( window );
375 rightImage = GadgetTextEntryGetEnabledImageRight( window );
376 centerImage = GadgetTextEntryGetEnabledImageCenter( window );
377 smallCenterImage = GadgetTextEntryGetEnabledImageSmallCenter( window );
378
379 } // end else, just enabled
380
381 // get image sizes for the ends
382 ICoord2D leftSize, rightSize;
383 leftSize.x = leftImage->getImageWidth();
384 leftSize.y = leftImage->getImageHeight();
385 rightSize.x = rightImage->getImageWidth();
386 rightSize.y = rightImage->getImageHeight();
387
388 // get two key points used in the end drawing
389 ICoord2D leftEnd, rightStart;
390 leftEnd.x = origin.x + leftSize.x + xOffset;
391 leftEnd.y = origin.y + size.y + yOffset;
392 rightStart.x = origin.x + size.x - rightSize.x + xOffset;
393 rightStart.y = origin.y + yOffset;
394
395 // draw the center repeating bar
396 Int centerWidth, pieces;
397
398 // get width we have to draw our repeating center in
399 centerWidth = rightStart.x - leftEnd.x;
400
401 // how many whole repeating pieces will fit in that width
402 pieces = centerWidth / centerImage->getImageWidth();
403
404 // draw the pieces
405 start.x = leftEnd.x;
406 start.y = origin.y + yOffset;
407 end.y = start.y + size.y;
408 for( i = 0; i < pieces; i++ )
409 {
410
411 end.x = start.x + centerImage->getImageWidth();
412 TheWindowManager->winDrawImage( centerImage,
413 start.x, start.y,
414 end.x, end.y );
415 start.x += centerImage->getImageWidth();
416
417 } // end for i
418
419 //
420 // how many small repeating pieces will fit in the gap from where the
421 // center repeating bar stopped and the right image, draw them
422 // and overlapping underneath where the right end will go
423 //
424 centerWidth = rightStart.x - start.x;
425 pieces = centerWidth / smallCenterImage->getImageWidth() + 1;
426 end.y = start.y + size.y;
427 for( i = 0; i < pieces; i++ )
428 {
429
430 end.x = start.x + smallCenterImage->getImageWidth();
431 TheWindowManager->winDrawImage( smallCenterImage,
432 start.x, start.y,
433 end.x, end.y );
434 start.x += smallCenterImage->getImageWidth();
435
436 } // end for i
437
438 // draw left end
439 start.x = origin.x + xOffset;
440 start.y = origin.y + yOffset;
441 end = leftEnd;
442 TheWindowManager->winDrawImage(leftImage, start.x, start.y, end.x, end.y);
443
444 // draw right end
445 start = rightStart;
446 end.x = start.x + rightSize.x;
447 end.y = start.y + size.y;
448 TheWindowManager->winDrawImage(rightImage, start.x, start.y, end.x, end.y);
449
450 // draw the text
451 Int fontHeight = TheWindowManager->winFontHeight( instData->getFont() );
452 Int startOffset = 5;
453 Int width;
454
455 width = size.x - (2 * startOffset);
456 start.x = origin.x + startOffset; // offset a little bit into the entry
457 if( BitTest( window->winGetStatus(), WIN_STATUS_ONE_LINE ) )
458 start.y = size.y / 2 - fontHeight / 2;
459 else
460 start.y = origin.y + startOffset; // offset a little bit into the entry
461
462 // draw the edit text
463 drawTextEntryText( window, instData, textColor, textBorder, compositeColor, compositeBorder,
464 start.x, start.y, width, fontHeight );
465
466
467
468} // end W3DGadgetTextEntryImageDraw
