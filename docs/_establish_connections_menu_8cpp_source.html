132 GadgetStaticTextSetText(control, TheGameText->fetch("GUI:WaitingForMangledPort"));

134 GadgetStaticTextSetText(control, TheGameText->fetch("GUI:WaitingForResponse"));

136 GadgetStaticTextSetText(control, TheGameText->fetch("GUI:ConnectionDone"));
