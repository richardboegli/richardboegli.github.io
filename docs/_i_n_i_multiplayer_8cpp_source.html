36void INI::parseMultiplayerSettingsDefinition( INI* ini )
37{
38 if( TheMultiplayerSettings )
39 {
40 //
41 // if the type of loading we're doing creates override data, we need to
42 // be loading into a new override item
43 //
44 if( ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES )
45 {
46 DEBUG_ASSERTCRASH(false, ("Creating an override of MultiplayerSettings!"));
47 }
48 } // end if
49 else
50 {
51 // we don't have any multiplayer settings instance at all yet, create one
52 TheMultiplayerSettings = NEW MultiplayerSettings;
53 } // end else
54
55 // parse the ini definition
56 ini->initFromINI( TheMultiplayerSettings, TheMultiplayerSettings->getFieldParse() );
57}

59void INI::parseMultiplayerColorDefinition( INI* ini )
60{
61 const char *c;
62 AsciiString name;
63 MultiplayerColorDefinition *multiplayerColorDefinition;
64
65 // read the name
66 c = ini->getNextToken();
67 name.set( c );
68
69 // find existing item if present, but this type does not allow overrides,
70 //so if it exists just overwrite it.
71 multiplayerColorDefinition = TheMultiplayerSettings->findMultiplayerColorDefinitionByName( name );
72 if( multiplayerColorDefinition == NULL )
73 multiplayerColorDefinition = TheMultiplayerSettings->newMultiplayerColorDefinition( name );
74
75 ini->initFromINI( multiplayerColorDefinition, multiplayerColorDefinition->getFieldParse() );
76
77 multiplayerColorDefinition->setColor(multiplayerColorDefinition->getRGBValue());
78 multiplayerColorDefinition->setNightColor(multiplayerColorDefinition->getRGBNightValue());
79}
