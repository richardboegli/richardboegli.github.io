133const GeneralPersona* ChallengeGenerals::getPlayerGeneralByCampaignName( AsciiString name ) const
134{
135 for (Int i = 0; i < NUM_GENERALS; i++)
136 {
137 AsciiString campaignName = m_position[i].getCampaign();
138 if (campaignName.compareNoCase( name.str() ) == 0)
139 return &m_position[i];
140 }
141 DEBUG_ASSERTCRASH(NULL, ("Can't find General by Campaign Name"));
142 return NULL;
143}
