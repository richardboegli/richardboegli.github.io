200void LANDisplayGameList( GameWindow *gameListbox, LANGameInfo *gameList )
201{
202 LANGameInfo *selectedPtr = NULL;
203 Int selectedIndex = -1;
204 Int indexToSelect = -1;
205 if (gameListbox)
206 {
207 GadgetListBoxGetSelected(gameListbox, &selectedIndex);
208
209 if (selectedIndex != -1 )
210 {
211 selectedPtr = (LANGameInfo *)GadgetListBoxGetItemData(gameListbox, selectedIndex, 0);
212 }
213
214 GadgetListBoxReset(gameListbox);
215
216 while (gameList)
217 {
218 UnicodeString txtGName;
219 txtGName = L"";
220 if( gameList->isGameInProgress() )
221 {
222 txtGName.concat(L"[");
223 }
224 txtGName.concat(gameList->getPlayerName(0));
225 if( gameList->isGameInProgress() )
226 {
227 txtGName.concat(L"]");
228 }
229 Int addedIndex = GadgetListBoxAddEntryText(gameListbox, txtGName, (gameList->isGameInProgress())?gameInProgressColor:gameColor, -1, -1);
230 GadgetListBoxSetItemData(gameListbox, (void *)gameList, addedIndex, 0 );
231
232 if (selectedPtr == gameList)
233 indexToSelect = addedIndex;
234
235 gameList = gameList->getNext();
236 }
237
238 if (indexToSelect >= 0)
239 GadgetListBoxSetSelected(gameListbox, indexToSelect);
240 else
241 HideGameInfoWindow(TRUE);
242 }
243}

253Bool ParseGameOptionsString(LANGameInfo *game, AsciiString options)
254{
255 if (!TheLAN || !game)
256 return false;
257
258 Int oldLocalSlotNum = (game->isInGame()) ? game->getLocalSlotNum() : -1;
259 Bool wasInGame = oldLocalSlotNum >= 0;
260// Int hadMap = wasInGame && game->getSlot(oldLocalSlotNum)->hasMap();
261 AsciiString oldMap = game->getMap();
262 UnsignedInt oldMapCRC, newMapCRC;
263 oldMapCRC = game->getMapCRC();
264
265 std::map<UnicodeString, UnicodeString> oldLogins, oldMachines;
266 std::map<UnicodeString, UnicodeString>::iterator mapIt;
267 Int i;
268 for (i=0; i<MAX_SLOTS; ++i)
269 {
270 LANGameSlot *slot = game->getLANSlot(i);
271 if (slot && slot->isHuman())
272 {
273 //DEBUG_LOG(("Saving off %ls@%ls for %ls\n", slot->getUser()->getLogin().str(), slot->getUser()->getHost().str(), slot->getName().str()));
274 oldLogins[slot->getName()] = slot->getUser()->getLogin();
275 oldMachines[slot->getName()] = slot->getUser()->getHost();
276 }
277 }
278
279 if (ParseAsciiStringToGameInfo(game, options))
280 {
281 Int newLocalSlotNum = (game->isInGame()) ? game->getLocalSlotNum() : -1;
282 Bool isInGame = newLocalSlotNum >= 0;
283 if (!TheLAN->AmIHost() && isInGame)
284 {
285// Int hasMap = game->getSlot(newLocalSlotNum)->hasMap();
286 newMapCRC = game->getMapCRC();
287 //DEBUG_LOG(("wasInGame:%d isInGame:%d hadMap:%d hasMap:%d oldMap:%s newMap:%s\n", wasInGame, isInGame, hadMap, hasMap, oldMap.str(), game->getMap().str()));
288 if ( (oldMapCRC ^ newMapCRC)/*(hasMap ^ hadMap)*/ || (!wasInGame && isInGame) )
289 {
290 // it changed. send it
291 TheLAN->RequestHasMap();
292 lanUpdateSlotList();
293 updateGameOptions();
294 }
295 }
296 // clean up LAN users, etc.
297 UnsignedInt now = timeGetTime();
298 for (i=0; i<MAX_SLOTS; ++i)
299 {
300 LANGameSlot *slot = game->getLANSlot(i);
301
302 if (slot->isHuman())
303 {
304 slot->setLastHeard(now);
305 mapIt = oldLogins.find(slot->getName());
306 if (mapIt != oldLogins.end())
307 slot->setLogin(mapIt->second);
308 mapIt = oldMachines.find(slot->getName());
309 if (mapIt != oldMachines.end())
310 slot->setHost(mapIt->second);
311 //DEBUG_LOG(("Restored %ls@%ls for %ls\n", slot->getUser()->getLogin().str(), slot->getUser()->getHost().str(), slot->getName().str()));
312 }
313 }
314
315 return true;
316 }
317
318 return false;
