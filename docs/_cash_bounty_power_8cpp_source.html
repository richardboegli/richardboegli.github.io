135void CashBountyPower::onObjectCreated()
136{
137 Player* controller = getObject()->getControllingPlayer();
138 if (controller && controller->hasScience(getRequiredScience()))
139 {
140 Real bounty = findBounty();
141 if (bounty > controller->getCashBounty())
142 controller->setCashBounty(bounty);
143 }
144}

148Real CashBountyPower::findBounty() const
149{
150 const CashBountyPowerModuleData* d = getCashBountyPowerModuleData();
151#ifdef NOT_IN_USE
152 const Player* controller = getObject()->getControllingPlayer();
153 if (controller != NULL)
154 {
155 for (std::vector<CashBountyPowerModuleData::Upgrades>::const_iterator it = d->m_upgrades.begin();
156 it != d->m_upgrades.end();
157 ++it)
158 {
159 if (controller->hasScience(it->m_science))
160 return it->m_bounty;
161 }
162 }
163#endif
164 return d->m_defaultBounty;
165}
