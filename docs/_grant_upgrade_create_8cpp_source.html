83void GrantUpgradeCreate::onCreate( void )
84{
85
86 ObjectStatusMaskType exemptStatus = getGrantUpgradeCreateModuleData()->m_exemptStatus;
87 ObjectStatusMaskType currentStatus = getObject()->getStatusBits();
88 if( exemptStatus.test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
89 {
90 if( !currentStatus.test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
91 {
92 const UpgradeTemplate *upgradeTemplate = TheUpgradeCenter->findUpgrade( getGrantUpgradeCreateModuleData()->m_upgradeName );
93 if( !upgradeTemplate )
94 {
95 DEBUG_ASSERTCRASH( 0, ("GrantUpdateCreate for %s can't find upgrade template %s.", getObject()->getName(), getGrantUpgradeCreateModuleData()->m_upgradeName ) );
96 return;
97 }
98
99 Player *player = getObject()->getControllingPlayer();
100 if( upgradeTemplate->getUpgradeType() == UPGRADE_TYPE_PLAYER )
101 {
102 // get the player
103 player->addUpgrade( upgradeTemplate, UPGRADE_STATUS_COMPLETE );
104 }
105 else
106 {
107 getObject()->giveUpgrade( upgradeTemplate );
108 }
109
110 player->getAcademyStats()->recordUpgrade( upgradeTemplate, TRUE );
111 }
112 }
113
114} // end onCreate
