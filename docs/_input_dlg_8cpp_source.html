116{
117 static InputDlg *dialog = NULL;
118
119 if (uMsg == WM_INITDIALOG)

121 dialog = (InputDlg*)lParam;
122 dialog->m_hWnd = hWnd;

124
125 if (dialog)

127 else

131BOOL CALLBACK InputDlg::DialogProc (HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
132{
133 int code = HIWORD(wParam);
134
135 switch (uMsg)
136 {
137 /*******************************************************************
138 * WM_INITDIALOG
139 *
140 * Initialize all of the custom controls for the dialog box
141 *
142 *******************************************************************/
143 case WM_INITDIALOG:
144
145 OnInitDialog(wParam, lParam);
146 return TRUE;
147
148
149 /*******************************************************************
150 * WM_COMMAND
151 *
152 *
153 *******************************************************************/
154 case WM_COMMAND:
155
156 switch (LOWORD(wParam))
157 {
158 case IDOK:
159 if (!OnOK())
160 return TRUE;
161
162 SetCursor(LoadCursor(NULL, IDC_WAIT));
163 EndDialog(m_hWnd, 1);
164 break;
165
166 case IDCANCEL:
167 EndDialog(m_hWnd, 0);
168 break;
169 }
170 return TRUE;
171 }
172
173 return FALSE;
174}

179LRESULT InputDlg::OnInitDialog (WPARAM wParam, LPARAM lParam)
180{
181 // Set the cursor to the normal arrow.
182 SetCursor(LoadCursor(NULL, IDC_ARROW));
183
184 // Set the dialog box caption.
185 SetWindowText(m_hWnd, m_Caption);
186
187 // Set the label text.
188 HWND hLabel = GetDlgItem(m_hWnd, IDC_LABEL);
189 assert(hLabel != NULL);
190 SetWindowText(hLabel, m_Label);
191
192 // Set the default value.
193 HWND hEdit = GetDlgItem(m_hWnd, IDC_VALUE);
194 assert(hEdit != NULL);
195 SetWindowText(hEdit, m_Value);
196
197 // Select all of the text in the edit box.
198 SendMessage(hEdit, EM_SETSEL, 0, -1);
199
200 return 0;
201}

209
210 // The dialog can be dismissed.
211 return TRUE;
