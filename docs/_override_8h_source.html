36/*
37 An OVERRIDE is a replacement for a pointer of its contained type, ie, rather than containing
38 a LocomotorTemplate*, you would contain an OVERRIDE<LocomotorTemplate>.
39
40 OVERRIDE pretends in all ways (dereference via *, -> and casting to type*) to be a type*, so
41 there should be very little code that needs to be rewritten to work with these.
42
43 In order to make something overridable, these are the steps:
44 1) Make the desired class derive from Overridable.
45 2) Make the container class contain an instance of OVERRIDE<Type>
46 3) Make the newOverride function (wherever an override is new'd) request the overridables lastOverride,
47 to ensure that no leaks are created.
48
49 See LocomotorTemplate for an example.
50*/
51
52template <class T> class OVERRIDE
53{
54 public:
55 // Provide useful constructores to go from a T* to an OVERRIDE<T>
56 OVERRIDE(const T *overridable = NULL);
57 // Copy constructor
58 OVERRIDE(OVERRIDE<T> &overridable);
59 // Operator= for copying from another OVERRIDE and T*
60 __inline OVERRIDE &operator=( const OVERRIDE<T>& override );
61 __inline OVERRIDE &operator=( const T* overridable );
62
63 // these are the methods which we can use to access data in a pointer. (Dereference*, ->, and cast
64 // to T*). They are all overloaded to recurse to the lowest override and use that.
65 __inline const T *operator->( void ) const; // overload const ->
66 __inline const T *operator*( void ) const; // overload const *(dereference operator)
67 __inline operator const T*( ) const; // overload casting to (const T*)
68
69 // this is useful in case you want to get the pointer that this object is actually looking at.
70 __inline const T *getNonOverloadedPointer( void ) const;
71
72 private:
73 // Because OVERRIDE is meant to live on the object and not in the store, it currently contains
74 // a constant pointer. We could change this if it seems weird.
75 const T *m_overridable;
76};
