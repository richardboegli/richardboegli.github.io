99// WinInstanceData::init ======================================================
101//=============================================================================
102void WinInstanceData::init( void )
103{
104 Int i;
105
106 // init our draw data images/colors for the states
107 for( i = 0; i < MAX_DRAW_DATA; i++ )
108 {
109
110 m_enabledDrawData[ i ].image = NULL;
111 m_enabledDrawData[ i ].color = WIN_COLOR_UNDEFINED;
112 m_enabledDrawData[ i ].borderColor = WIN_COLOR_UNDEFINED;
113
114 m_disabledDrawData[ i ].image = NULL;
115 m_disabledDrawData[ i ].color = WIN_COLOR_UNDEFINED;
116 m_disabledDrawData[ i ].borderColor = WIN_COLOR_UNDEFINED;
117
118 m_hiliteDrawData[ i ].image = NULL;
119 m_hiliteDrawData[ i ].color = WIN_COLOR_UNDEFINED;
120 m_hiliteDrawData[ i ].borderColor = WIN_COLOR_UNDEFINED;
121
122 } // end for i
123
124 // initialize text colors
125 m_enabledText.color = WIN_COLOR_UNDEFINED;
126 m_enabledText.borderColor = WIN_COLOR_UNDEFINED;
127 m_disabledText.color = WIN_COLOR_UNDEFINED;
128 m_disabledText.borderColor = WIN_COLOR_UNDEFINED;
129 m_hiliteText.color = WIN_COLOR_UNDEFINED;
130 m_hiliteText.borderColor = WIN_COLOR_UNDEFINED;
131
132 m_id = 0;
133 m_state = 0;
134 m_style = 0;
135 m_status = WIN_STATUS_NONE;
136 m_owner = NULL;
137 m_textLabelString.clear();
138 m_tooltipString.clear();
139 m_tooltipDelay = -1;
140 m_decoratedNameString.clear();
141
142 m_imageOffset.x = 0;
143 m_imageOffset.y = 0;
144
145 // reset all data for the text display strings and font for window
146 m_font = NULL;
147 if( m_text )
148 {
149
150 TheDisplayStringManager->freeDisplayString( m_text );
151 m_text = NULL;
152
153 } // end if
154 if( m_tooltip )
155 {
156
157 TheDisplayStringManager->freeDisplayString( m_tooltip );
158 m_tooltip = NULL;
159
160 } // end if
161
162 m_videoBuffer = NULL;
163
164
165} // end init
