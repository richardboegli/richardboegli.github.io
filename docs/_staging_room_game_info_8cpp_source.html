98/***********************************************************************************************
99 * Get_Local_Chat_Connection_Address -- Which address are we using to talk to the chat server? *
100 * *
101 * *
102 * *
103 * INPUT: Ptr to address to return local address * *
104 * *
105 * OUTPUT: True if success *
106 * *
107 * WARNINGS: None *
108 * *
109 * HISTORY: *
110 * 10/27/00 3:24PM ST : Created *
111 *=============================================================================================*/
112Bool GetLocalChatConnectionAddress(AsciiString serverName, UnsignedShort serverPort, UnsignedInt& localIP)
113{
114 //return false;
115 /*
116 ** Local defines.
117 */
118 enum {
119 CLOSED = 1,
120 LISTENING,
121 SYN_SENT,
122 SEN_RECEIVED,
123 ESTABLISHED,
124 FIN_WAIT,
125 FIN_WAIT2,
126 CLOSE_WAIT,
127 LAST_ACK,
128 CLOSING,
129 TIME_WAIT,
130 DELETE_TCB
131 };
132
133 enum {
134 tcpConnState = 1,
135 tcpConnLocalAddress,
136 tcpConnLocalPort,
137 tcpConnRemAddress,
138 tcpConnRemPort
139 };
140
141
142 /*
143 ** Locals.
144 */
145 unsigned char serverAddress[4];
146 unsigned char remoteAddress[4];
147 HANDLE trap_handle;
148 AsnObjectIdentifier first_supported_region;
149 std::vector<ConnInfoStruct> connectionVector;
150 int last_field;
151 int index;
152 AsnInteger error_status;
153 AsnInteger error_index;
154 int conn_entry_type_index;
155 int conn_entry_type;
156 Bool found;
157
158 /*
159 ** Statics.
160 */
161 static char _conn_state[][32] = {
162 "?",
163 "CLOSED",
164 "LISTENING",
165 "SYN_SENT",
166 "SEN_RECEIVED",
167 "ESTABLISHED",
168 "FIN_WAIT",
169 "FIN_WAIT2",
170 "CLOSE_WAIT",
171 "LAST_ACK",
172 "CLOSING",
173 "TIME_WAIT",
174 "DELETE_TCB"
175 };
176
177 DEBUG_LOG(("Finding local address used to talk to the chat server\n"));
178 DEBUG_LOG(("Current chat server name is %s\n", serverName.str()));
179 DEBUG_LOG(("Chat server port is %d\n", serverPort));
180
181 /*
182 ** Get the address of the chat server.
183 */
184 DEBUG_LOG( ("About to call gethostbyname\n"));
185 struct hostent *host_info = gethostbyname(serverName.str());
186
187 if (!host_info) {
188 DEBUG_LOG( ("gethostbyname failed! Error code %d\n", WSAGetLastError()));
189 return(false);
190 }
191
192 memcpy(serverAddress, &host_info->h_addr_list[0][0], 4);
193 unsigned long temp = *((unsigned long*)(&serverAddress[0]));
194 temp = ntohl(temp);
195 *((unsigned long*)(&serverAddress[0])) = temp;
196
197 DEBUG_LOG(("Host address is %d.%d.%d.%d\n", serverAddress[3], serverAddress[2], serverAddress[1], serverAddress[0]));
198
199 /*
200 ** Load the MIB-II SNMP DLL.
201 */
202 DEBUG_LOG(("About to load INETMIB1.DLL\n"));
203
204 HINSTANCE mib_ii_dll = LoadLibrary("inetmib1.dll");
205 if (mib_ii_dll == NULL) {
206 DEBUG_LOG(("Failed to load INETMIB1.DLL\n"));
207 return(false);
208 }
209
210 DEBUG_LOG(("About to load SNMPAPI.DLL\n"));
211
212 HINSTANCE snmpapi_dll = LoadLibrary("snmpapi.dll");
213 if (snmpapi_dll == NULL) {
214 DEBUG_LOG(("Failed to load SNMPAPI.DLL\n"));
215 FreeLibrary(mib_ii_dll);
216 return(false);
217 }
218
219 /*
220 ** Get the function pointers into the .dll
221 */
222 SnmpExtensionInitPtr = (int (__stdcall *)(unsigned long,void ** ,AsnObjectIdentifier *)) GetProcAddress(mib_ii_dll, "SnmpExtensionInit");
223 SnmpExtensionQueryPtr = (int (__stdcall *)(unsigned char,SnmpVarBindList *,long *,long *)) GetProcAddress(mib_ii_dll, "SnmpExtensionQuery");
224 SnmpUtilMemAllocPtr = (void *(__stdcall *)(unsigned long)) GetProcAddress(snmpapi_dll, "SnmpUtilMemAlloc");
225 SnmpUtilMemFreePtr = (void (__stdcall *)(void *)) GetProcAddress(snmpapi_dll, "SnmpUtilMemFree");
226 if (SnmpExtensionInitPtr == NULL || SnmpExtensionQueryPtr == NULL || SnmpUtilMemAllocPtr == NULL || SnmpUtilMemFreePtr == NULL) {
227 DEBUG_LOG(("Failed to get proc addresses for linked functions\n"));
228 FreeLibrary(snmpapi_dll);
229 FreeLibrary(mib_ii_dll);
230 return(false);
231 }
232
233
234 RFC1157VarBindList *bind_list_ptr = (RFC1157VarBindList *) SnmpUtilMemAllocPtr(sizeof(RFC1157VarBindList));
235 RFC1157VarBind *bind_ptr = (RFC1157VarBind *) SnmpUtilMemAllocPtr(sizeof(RFC1157VarBind));
236
237 /*
238 ** OK, here we go. Try to initialise the .dll
239 */
240 DEBUG_LOG(("About to init INETMIB1.DLL\n"));
241 int ok = SnmpExtensionInitPtr(GetCurrentTime(), &trap_handle, &first_supported_region);
242
243 if (!ok) {
244 /*
245 ** Aw crap.
246 */
247 DEBUG_LOG(("Failed to init the .dll\n"));
248 SnmpUtilMemFreePtr(bind_list_ptr);
249 SnmpUtilMemFreePtr(bind_ptr);
250 FreeLibrary(snmpapi_dll);
251 FreeLibrary(mib_ii_dll);
252 return(false);
253 }
254
255 /*
256 ** Name of mib_ii object we want to query. See RFC 1213.
257 **
258 ** iso.org.dod.internet.mgmt.mib-2.tcp.tcpConnTable.TcpConnEntry.tcpConnState
259 ** 1 3 6 1 2 1 6 13 1 1
260 */
261 unsigned int mib_ii_name[] = {1,3,6,1,2,1,6,13,1,1};
262 unsigned int *mib_ii_name_ptr = (unsigned int *) SnmpUtilMemAllocPtr(sizeof(mib_ii_name));
263 memcpy(mib_ii_name_ptr, mib_ii_name, sizeof(mib_ii_name));
264
265 /*
266 ** Get the index of the conn entry data.
267 */
268 conn_entry_type_index = ARRAY_SIZE(mib_ii_name) - 1;
269
270 /*
271 ** Set up the bind list.
272 */
273 bind_ptr->name.idLength = ARRAY_SIZE(mib_ii_name);
274 bind_ptr->name.ids = mib_ii_name;
275 bind_list_ptr->list = bind_ptr;
276 bind_list_ptr->len = 1;
277
278
279 /*
280 ** We start with the tcpConnLocalAddress field.
281 */
282 last_field = 1;
283
284 /*
285 ** First connection.
286 */
287 index = 0;
288
289 /*
290 ** Suck out that tcp connection info....
291 */
292 while (true) {
293
294 if (!SnmpExtensionQueryPtr(SNMP_PDU_GETNEXT, bind_list_ptr, &error_status, &error_index)) {
295 //if (!SnmpExtensionQueryPtr(ASN_RFC1157_GETNEXTREQUEST, bind_list_ptr, &error_status, &error_index)) {
296 DEBUG_LOG(("SnmpExtensionQuery returned false\n"));
297 SnmpUtilMemFreePtr(bind_list_ptr);
298 SnmpUtilMemFreePtr(bind_ptr);
299 FreeLibrary(snmpapi_dll);
300 FreeLibrary(mib_ii_dll);
301 return(false);
302 }
303
304 /*
305 ** If this is something new we aren't looking for then we are done.
306 */
307 if (bind_ptr->name.idLength < ARRAY_SIZE(mib_ii_name)) {
308 break;
309 }
310
311 /*
312 ** Get the type of info we are looking at. See RFC1213.
313 **
314 ** 1 = tcpConnState
315 ** 2 = tcpConnLocalAddress
316 ** 3 = tcpConnLocalPort
317 ** 4 = tcpConnRemAddress
318 ** 5 = tcpConnRemPort
319 **
320 ** tcpConnState is one of the following...
321 **
322 ** 1 closed
323 ** 2 listen
324 ** 3 synSent
325 ** 4 synReceived
326 ** 5 established
327 ** 6 finWait1
328 ** 7 finWait2
329 ** 8 closeWait
330 ** 9 lastAck
331 ** 10 closing
332 ** 11 timeWait
333 ** 12 deleteTCB
334 */
335 conn_entry_type = bind_ptr->name.ids[conn_entry_type_index];
336
337 if (last_field != conn_entry_type) {
338 index = 0;
339 last_field = conn_entry_type;
340 }
341
342 switch (conn_entry_type) {
343
344 /*
345 ** 1. First field in the entry. Need to create a new connection info struct
346 ** here to store this connection in.
347 */
348 case tcpConnState:
349 {
350 ConnInfoStruct new_conn;
351 new_conn.State = bind_ptr->value.asnValue.number;
352 connectionVector.push_back(new_conn);
353 break;
354 }
355
356 /*
357 ** 2. Local address field.
358 */
359 case tcpConnLocalAddress:
360 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
361 connectionVector[index].LocalIP = *((unsigned long*)bind_ptr->value.asnValue.address.stream);
362 index++;
363 break;
364
365 /*
366 ** 3. Local port field.
367 */
368 case tcpConnLocalPort:
369 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
370 connectionVector[index].LocalPort = bind_ptr->value.asnValue.number;
371 //connectionVector[index]->LocalPort = ntohs(connectionVector[index]->LocalPort);
372 index++;
373 break;
374
375 /*
376 ** 4. Remote address field.
377 */
378 case tcpConnRemAddress:
379 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
380 connectionVector[index].RemoteIP = *((unsigned long*)bind_ptr->value.asnValue.address.stream);
381 index++;
382 break;
383
384 /*
385 ** 5. Remote port field.
386 */
387 case tcpConnRemPort:
388 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
389 connectionVector[index].RemotePort = bind_ptr->value.asnValue.number;
390 //connectionVector[index]->RemotePort = ntohs(connectionVector[index]->RemotePort);
391 index++;
392 break;
393 }
394 }
395
396 SnmpUtilMemFreePtr(bind_list_ptr);
397 SnmpUtilMemFreePtr(bind_ptr);
398 SnmpUtilMemFreePtr(mib_ii_name_ptr);
399
400 DEBUG_LOG(("Got %d connections in list, parsing...\n", connectionVector.size()));
401
402 /*
403 ** Right, we got the lot. Lets see if any of them have the same address as the chat
404 ** server we think we are talking to.
405 */
406 found = false;
407 for (Int i=0; i<connectionVector.size(); ++i) {
408 ConnInfoStruct connection = connectionVector[i];
409
410 temp = ntohl(connection.RemoteIP);
411 memcpy(remoteAddress, (unsigned char*)&temp, 4);
412
413 /*
414 ** See if this connection has the same address as our server.
415 */
416 if (!found && memcmp(remoteAddress, serverAddress, 4) == 0) {
417 DEBUG_LOG(("Found connection with same remote address as server\n"));
418
419 if (serverPort == 0 || serverPort == (unsigned int)connection.RemotePort) {
420
421 DEBUG_LOG(("Connection has same port\n"));
422 /*
423 ** Make sure the connection is current.
424 */
425 if (connection.State == ESTABLISHED) {
426 DEBUG_LOG(("Connection is ESTABLISHED\n"));
427 localIP = connection.LocalIP;
428 found = true;
429 } else {
430 DEBUG_LOG(("Connection is not ESTABLISHED - skipping\n"));
431 }
432 } else {
433 DEBUG_LOG(("Connection has different port. Port is %d, looking for %d\n", connection.RemotePort, serverPort));
434 }
435 }
436 }
437
438 if (found) {
439 DEBUG_LOG(("Using address 0x%8.8X to talk to chat server\n", localIP));
440 }
441
442 FreeLibrary(snmpapi_dll);
443 FreeLibrary(mib_ii_dll);
444 return(found);
445}

537void GameSpyStagingRoom::startGame(Int gameID)
538{
539 DEBUG_ASSERTCRASH(m_inGame, ("Starting a game while not in game"));
540 DEBUG_LOG(("GameSpyStagingRoom::startGame - game id = %d\n", gameID));
541 DEBUG_ASSERTCRASH(m_transport == NULL, ("m_transport is not NULL when it should be"));
542 DEBUG_ASSERTCRASH(TheNAT == NULL, ("TheNAT is not NULL when it should be"));
543
544 UnsignedInt localIP = TheGameSpyInfo->getInternalIP();
545 setLocalIP(localIP);
546
547 if (TheNAT != NULL) {
548 delete TheNAT;
549 TheNAT = NULL;
550 }
551
552 // fill in GS-specific info
553 Int numHumans = 0;
554 for (Int i=0; i<MAX_SLOTS; ++i)
555 {
556 if (m_GameSpySlot[i].isHuman())
557 {
558 ++numHumans;
559 AsciiString gsName;
560 gsName.translate( m_GameSpySlot[i].getName() );
561 m_GameSpySlot[i].setLoginName( gsName );
562
563 if (m_isQM)
564 {
565 if (getLocalSlotNum() == i)
566 m_GameSpySlot[i].setProfileID(TheGameSpyInfo->getLocalProfileID()); // hehe - we know our own. the rest, they'll tell us.
567 }
568 else
569 {
570 PlayerInfoMap *pInfoMap = TheGameSpyInfo->getPlayerInfoMap();
571 PlayerInfoMap::iterator it = pInfoMap->find(gsName);
572 if (it != pInfoMap->end())
573 {
574 m_GameSpySlot[i].setProfileID(it->second.m_profileID);
575 m_GameSpySlot[i].setLocale(it->second.m_locale);
576 m_GameSpySlot[i].setSlotRankPoints(it->second.m_rankPoints);
577 m_GameSpySlot[i].setFavoriteSide(it->second.m_side);
578 }
579 else
580 {
581 DEBUG_CRASH(("No player info for %s", gsName.str()));
582 }
583 }
584 }
585 }
586
587//#if defined(_DEBUG) || defined(_INTERNAL)
588 if (numHumans < 2)
589 {
590 launchGame();
591 if (TheGameSpyInfo)
592 TheGameSpyInfo->leaveStagingRoom();
593 }
594 else
595//#endif defined(_DEBUG) || defined(_INTERNAL)
596 {
597 TheNAT = NEW NAT();
598 TheNAT->attachSlotList(m_slot, getLocalSlotNum(), m_localIP);
599 TheNAT->establishConnectionPaths();
600 }

603AsciiString GameSpyStagingRoom::generateGameSpyGameResultsPacket( void )
604{
605 Int i;
606 Int endFrame = TheVictoryConditions->getEndFrame();
607 Int localSlotNum = getLocalSlotNum();
608 Int winningTeam = -1;
609 Int numHumans = 0;
610 Int numPlayers = 0;
611 Int numAIs = 0;
612 Int numTeamsAtGameEnd = 0;
613 Int lastTeamAtGameEnd = -1;
614 for (i=0; i<MAX_SLOTS; ++i)
615 {
616 AsciiString playerName;
617 playerName.format("player%d", i);
618 Player *p = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
619 if (p)
620 {
621 ++numHumans;
622 if (TheVictoryConditions->hasAchievedVictory(p))
623 {
624 winningTeam = getSlot(i)->getTeamNumber();
625 }
626
627 // check if he lasted
628 GameSlot *slot = getSlot(i);
629 if (!slot->disconnected())
630 {
631 if (slot->getTeamNumber() != lastTeamAtGameEnd || numTeamsAtGameEnd == 0)
632 {
633 lastTeamAtGameEnd = slot->getTeamNumber();
634 ++numTeamsAtGameEnd;
635 }
636 }
637 }
638 else
639 {
640 if (m_GameSpySlot[i].isAI())
641 ++numAIs;
642 }
643 }
644 numPlayers = numHumans + numAIs;
645
646 AsciiString mapName;
647 for (i=0; i<getMap().getLength(); ++i)
648 {
649 char c = getMap().getCharAt(i);
650 if (c == '\\')
651 c = '/';
652 mapName.concat(c);
653 }
654
655 AsciiString results;
656 results.format("\\seed\\%d\\hostname\\%s\\mapname\\%s\\numplayers\\%d\\duration\\%d\\gamemode\\exiting\\localplayer\\%d",
657 getSeed(), m_GameSpySlot[0].getLoginName().str(), mapName.str(), numPlayers, endFrame, localSlotNum);
658
659 Int playerID = 0;
660 for (i=0; i<MAX_SLOTS; ++i)
661 {
662 AsciiString playerName;
663 playerName.format("player%d", i);
664 Player *p = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
665 if (p)
666 {
667 GameSpyGameSlot *slot = &(m_GameSpySlot[i]);
668 AsciiString playerName = (slot->isHuman())?slot->getLoginName():"AIPlayer";
669 Int gsPlayerID = slot->getProfileID();
670 Bool disconnected = slot->disconnected();
671
672 AsciiString result = "loss";
673 if (disconnected)
674 result = "discon";
675 else if (TheNetwork->sawCRCMismatch())
676 result = "desync";
677 else if (TheVictoryConditions->hasAchievedVictory(p))
678 result = "win";
679
680 AsciiString side = p->getPlayerTemplate()->getSide();
681 if (side == "America")
682 side = "USA"; //conform to GameSpy
683
684 AsciiString playerStr;
685 playerStr.format("\\player_%d\\%s\\pid_%d\\%d\\team_%d\\%d\\result_%d\\%s\\side_%d\\%s",
686 playerID, playerName.str(), playerID, gsPlayerID, playerID, slot->getTeamNumber(),
687 playerID, result.str(), playerID, side.str());
688 results.concat(playerStr);
689
690 ++playerID;
691 }
692 }
693
694 return results;
695}

697AsciiString GameSpyStagingRoom::generateLadderGameResultsPacket( void )
698{
699 Int i;
700 Int endFrame = TheVictoryConditions->getEndFrame();
701 Int localSlotNum = getLocalSlotNum();
702 Bool sawGameEnd = (endFrame > 0);
703 Int winningTeam = -1;
704 Int numPlayers = 0;
705 Int numTeamsAtGameEnd = 0;
706 Int lastTeamAtGameEnd = -1;
707 Player* p[MAX_SLOTS];
708 for (i=0; i<MAX_SLOTS; ++i)
709 {
710 AsciiString playerName;
711 playerName.format("player%d", i);
712 p[i] = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
713 if (p[i])
714 {
715 ++numPlayers;
716 if (TheVictoryConditions->hasAchievedVictory(p[i]))
717 {
718 winningTeam = getSlot(i)->getTeamNumber();
719 }
720
721 // check if he lasted
722 GameSlot *slot = getSlot(i);
723 if (!slot->disconnected())
724 {
725 if (slot->getTeamNumber() != lastTeamAtGameEnd || numTeamsAtGameEnd == 0)
726 {
727 lastTeamAtGameEnd = slot->getTeamNumber();
728 ++numTeamsAtGameEnd;
729 }
730 }
731 }
732 }
733
734 AsciiString results;
735 results.format("seed=%d,slotNum=%d,sawDesync=%d,sawGameEnd=%d,winningTeam=%d,disconEnd=%d,duration=%d,numPlayers=%d,isQM=%d,map=%s",
736 getSeed(), localSlotNum, TheNetwork->sawCRCMismatch(), sawGameEnd, winningTeam, (numTeamsAtGameEnd < 2),
737 endFrame, numPlayers, m_isQM, TheGameState->realMapPathToPortableMapPath(getMap()).str());
738
739 AsciiString tempStr;
740 tempStr.format("ladderIP=%s,ladderPort=%d", getLadderIP().str(), getLadderPort());
741 results.concat(tempStr);
742
743 Int playerID = 0;
744 for (i=0; i<MAX_SLOTS; ++i)
745 {
746 AsciiString playerName;
747 playerName.format("player%d", i);
748 if (p[i])
749 {
750 GameSpyGameSlot *slot = &(m_GameSpySlot[i]);
751 ScoreKeeper *keeper = p[i]->getScoreKeeper();
752 AsciiString playerName = slot->getLoginName();
753 Int gsPlayerID = slot->getProfileID();
754 PSPlayerStats stats = TheGameSpyPSMessageQueue->findPlayerStatsByID(gsPlayerID);
755 Int fps = TheNetwork->getAverageFPS();
756 Int unitsKilled = keeper->getTotalUnitsDestroyed();
757 Int unitsLost = keeper->getTotalUnitsLost();
758 Int unitsBuilt = keeper->getTotalUnitsBuilt();
759 Int buildingsKilled = keeper->getTotalBuildingsDestroyed();
760 Int buildingsLost = keeper->getTotalBuildingsLost();
761 Int buildingsBuilt = keeper->getTotalBuildingsBuilt();
762 Int earnings = keeper->getTotalMoneyEarned();
763 Int techCaptured = keeper->getTotalTechBuildingsCaptured();
764 Bool disconnected = slot->disconnected();
765
766 AsciiString playerStr;
767 playerStr.format(",player%d=%s,playerID%d=%d,locale%d=%d",
768 playerID, playerName.str(), playerID, gsPlayerID, playerID, stats.locale);
769 results.concat(playerStr);
770 playerStr.format(",unitsKilled%d=%d,unitsLost%d=%d,unitsBuilt%d=%d",
771 playerID, unitsKilled, playerID, unitsLost, playerID, unitsBuilt);
772 results.concat(playerStr);
773 playerStr.format(",buildingsKilled%d=%d,buildingsLost%d=%d,buildingsBuilt%d=%d",
774 playerID, buildingsKilled, playerID, buildingsLost, playerID, buildingsBuilt);
775 results.concat(playerStr);
776 playerStr.format(",fps%d=%d,cash%d=%d,capturedTech%d=%d,discon%d=%d,side%d=%s",
777 playerID, fps, playerID, earnings, playerID, techCaptured, playerID, disconnected, playerID, p[i]->getPlayerTemplate()->getSide().str());
778 results.concat(playerStr);
779
780 ++playerID;
781 }
782 }
783
784 // Add a trailing size value (so the server can ensure it got the entire packet)
785 results.concat(",size=");
786 int resultsLen = results.getLength()+10;
787 AsciiString tail;
788 tail.format("%10.10d", resultsLen);
789 results.concat(tail);
790
791 return results;

809 AsciiString user;
810 AsciiString userList;
811 DEBUG_ASSERTCRASH(TheNetwork == NULL, ("For some reason TheNetwork isn't NULL at the start of this game. Better look into that."));
812
813 if (TheNetwork != NULL) {

815 TheNetwork = NULL;
816 }
817
818 // Time to initialize TheNetwork for this game.
819 TheNetwork = NetworkInterface::createNetwork();

823 if (TheNAT)

855// TheShell->hideShell();
