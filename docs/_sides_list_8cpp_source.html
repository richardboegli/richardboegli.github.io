99SidesInfo& SidesInfo::operator=(const SidesInfo& that)
100{
101 if (this != &that)
102 {
103 this->clear();
104 this->m_dict = that.m_dict;
105
106 BuildListInfo* thisBLTail = NULL;
107 for (BuildListInfo* thatBL = that.m_pBuildList; thatBL; thatBL = thatBL->getNext())
108 {
109 BuildListInfo* thisBL = newInstance( BuildListInfo );
110 *thisBL = *thatBL;
111 thisBL->setNextBuildList(NULL);
112
113 if (thisBLTail)
114 thisBLTail->setNextBuildList(thisBL);
115 else
116 this->m_pBuildList = thisBL;
117
118 thisBLTail = thisBL;
119 }
120
121 if (that.m_scripts)
122 this->m_scripts = that.m_scripts->duplicate();
123 else
124 this->m_scripts = NULL;
125 }
126 return *this;
127}

133void SidesInfo::addToBuildList(BuildListInfo *pBuildList, Int position)
134{
135 DEBUG_ASSERTLOG(pBuildList->getNext()==NULL, ("WARNING***Adding already linked element."));
136 BuildListInfo *pCur = NULL;
137 while (position) {
138 position--;
139 if (pCur==NULL) {
140 pCur = m_pBuildList;
141 } else {
142 if (pCur->getNext()) {
143 pCur = pCur->getNext();
144 } else {
145 break; // at end of list.
146 }
147 }
148 }
149 if (pCur==NULL) {
150 // add to front of list.
151 pBuildList->setNextBuildList(m_pBuildList);
152 m_pBuildList = pBuildList;
153 } else {
154 pBuildList->setNextBuildList(pCur->getNext());
155 pCur->setNextBuildList(pBuildList);
156 }
157}

174Int SidesInfo::removeFromBuildList(BuildListInfo *pBuildList)
175{
176 DEBUG_ASSERTCRASH(pBuildList, ("Removing NULL list."));
177 if (pBuildList==NULL) return 0;
178
179 Int position = 0;
180
181 if (pBuildList == m_pBuildList) {
182 // First item in list, so update head.
183 m_pBuildList = pBuildList->getNext();
184 } else {
185 position = 1;
186 // Not the first item, so find the preceeding list element.
187 BuildListInfo *pPrev = m_pBuildList;
188 while (pPrev && (pPrev->getNext()!=pBuildList) ) {
189 pPrev = pPrev->getNext();
190 position++;
191 }
192 DEBUG_ASSERTCRASH(pPrev, ("Removing item not in list."));
193 if (pPrev) {
194 pPrev->setNextBuildList(pBuildList->getNext());
195 }
196 }
197 pBuildList->setNextBuildList(NULL);
198 return position;
199}

243Bool SidesList::ParseSidesDataChunk(DataChunkInput &file, DataChunkInfo *info, void *userData)
244{
245 DEBUG_ASSERTCRASH(TheSidesList, ("TheSidesList is null"));
246
247 if (TheSidesList==NULL)
248 return false;
249
250 TheSidesList->clear();
251 Int count = file.readInt();
252 Int i, j;
253 TheSidesList->emptySides();
254 for (i=0; i<count; i++)
255 {
256 if (i >= MAX_PLAYER_COUNT) break;
257 Dict d = file.readDict();
258 TheSidesList->addSide(&d);
259 BuildListInfo* pBuildList;
260 Int count = file.readInt();
261 for (j=0; j<count; j++)
262 {
263 pBuildList = newInstance( BuildListInfo );
264 pBuildList->setBuildingName(file.readAsciiString());
265 pBuildList->setTemplateName(file.readAsciiString());
266 Coord3D loc;
267 loc.x = file.readReal();
268 loc.y = file.readReal();
269 loc.z = file.readReal();
270 loc.z = 0; // force to ground level
271 pBuildList->setLocation(loc);
272 pBuildList->setAngle(file.readReal());
273 pBuildList->setInitiallyBuilt(file.readByte());
274 pBuildList->setNumRebuilds(file.readInt());
275 if (info->version >= K_SIDES_DATA_VERSION_3)
276 {
277 pBuildList->setScript(file.readAsciiString());
278 pBuildList->setHealth(file.readInt());
279 pBuildList->setWhiner(file.readByte());
280 pBuildList->setUnsellable(file.readByte());
281 pBuildList->setRepairable(file.readByte());
282 }
283 TheSidesList->getSideInfo(i)->addToBuildList(pBuildList, j);
284 }
285 }
286 if (info->version >= K_SIDES_DATA_VERSION_2)
287 {
288 count = file.readInt();
289 TheSidesList->emptyTeams();
290 for (i=0; i<count; i++)
291 {
292 Dict d = file.readDict();
293 TheSidesList->addTeam(&d);
294 }
295 }
296
297 file.registerParser( AsciiString("PlayerScriptsList"), info->label, ScriptList::ParseScriptsDataChunk );
298 if (!file.parse(NULL)) {
299 throw(ERROR_CORRUPT_FILE_FORMAT);
300 }
301 ScriptList *scripts[MAX_PLAYER_COUNT];
302 count = ScriptList::getReadScripts(scripts);
303 for (i=0; i<count; i++) {
304 if (i<TheSidesList->getNumSides()) {
305 ScriptList *pSL = TheSidesList->getSideInfo(i)->getScriptList();
306 pSL->deleteInstance();
307 TheSidesList->getSideInfo(i)->setScriptList(scripts[i]);
308 scripts[i] = NULL;
309 } else {
310 // Read in more players worth than we have.
311 scripts[i]->deleteInstance();
312 scripts[i] = NULL;
313 }
314 }
315 TheSidesList->validateSides();
316
317 DEBUG_ASSERTCRASH(file.atEndOfChunk(), ("Incorrect data file length."));
318 return true;
319}

329void SidesList::WriteSidesDataChunk(DataChunkOutput &chunkWriter)
330{
331 DEBUG_ASSERTCRASH(TheSidesList, ("TheSidesList is null"));
332 if (TheSidesList==NULL)
333 return;
334 /**********HEIGHT MAP DATA ***********************/
335 chunkWriter.openDataChunk("SidesList", K_SIDES_DATA_VERSION_3);
336
337 chunkWriter.writeInt(TheSidesList->getNumSides());
338 Int i;
339 for (i=0; i<TheSidesList->getNumSides(); i++) {
340 chunkWriter.writeDict(*TheSidesList->getSideInfo(i)->getDict());
341 BuildListInfo* pBuildList = TheSidesList->getSideInfo(i)->getBuildList();
342 Int count = 0;
343 while (pBuildList) {
344 count++;
345 pBuildList = pBuildList->getNext();
346 }
347 chunkWriter.writeInt(count);
348 pBuildList = TheSidesList->getSideInfo(i)->getBuildList();
349 while (pBuildList) {
350 chunkWriter.writeAsciiString(pBuildList->getBuildingName());
351 chunkWriter.writeAsciiString(pBuildList->getTemplateName());
352 chunkWriter.writeReal(pBuildList->getLocation()->x);
353 chunkWriter.writeReal(pBuildList->getLocation()->y);
354 chunkWriter.writeReal(pBuildList->getLocation()->z);
355 chunkWriter.writeReal(pBuildList->getAngle());
356 chunkWriter.writeByte(pBuildList->isInitiallyBuilt());
357 chunkWriter.writeInt(pBuildList->getNumRebuilds());
358 // BEGIN stuff new to K_SIDES_DATA_VERSION_3
359 chunkWriter.writeAsciiString(pBuildList->getScript());
360 chunkWriter.writeInt(pBuildList->getHealth());
361 chunkWriter.writeByte(pBuildList->getWhiner());
362 chunkWriter.writeByte(pBuildList->getUnsellable());
363 chunkWriter.writeByte(pBuildList->getRepairable());
364 // END stuff new to K_SIDES_DATA_VERSION_3
365
366 pBuildList = pBuildList->getNext();
367 }
368 }
369
370 // BEGIN stuff new to K_SIDES_DATA_VERSION_2
371 chunkWriter.writeInt(TheSidesList->getNumTeams());
372 for (i=0; i<TheSidesList->getNumTeams(); i++) {
373 chunkWriter.writeDict(*TheSidesList->getTeamInfo(i)->getDict());
374 }
375 // END stuff new to K_SIDES_DATA_VERSION_2
376
377 ScriptList *scripts[MAX_PLAYER_COUNT];
378 for (i=0; i<TheSidesList->getNumSides(); i++) {
379 scripts[i] = TheSidesList->getSideInfo(i)->getScriptList();
380 }
381 ScriptList::WriteScriptsDataChunk(chunkWriter, scripts, TheSidesList->getNumSides());
382 chunkWriter.closeDataChunk();
383
384 Bool modified = TheSidesList->validateSides();
385 DEBUG_ASSERTLOG(!modified, ("*** had to clean up sideslist on read"));
386 modified = false; // silence compiler warnings in release build
387
388}

478void SidesList::prepareForMP_or_Skirmish(void)
479{
480 m_skirmishTeamrec.clear();
481 Int i;
482 for (i = 0; i < getNumTeams(); i++)
483 {
484 Dict *tdict = getTeamInfo(i)->getDict();
485 m_skirmishTeamrec.addTeam(tdict);
486 }
487 m_teamrec.clear();
488
489 for (i = 0; i < MAX_PLAYER_COUNT; i++) {
490 m_skirmishSides[i].clear();
491 }
492 m_numSkirmishSides = 0;
493
494 for (i = 0; i < m_numSides; i++)
495 {
496 m_skirmishSides[m_numSkirmishSides] = m_sides[i];
497 m_numSkirmishSides++;
498 if (m_sides[i].getDict()->getAsciiString(TheKey_playerFaction) == "FactionCivilian") {
499 // Don't remove FactionCivilian.
500 continue;
501 }
502 if (m_numSides == 1) break; // can't remove the last side.
503 removeSide(i);
504 i--;
505 }
506 Bool gotScripts = false;
507 for (i=0; i<m_numSkirmishSides; i++) {
508 if (m_skirmishSides[i].getDict()->getAsciiString(TheKey_playerFaction) == "FactionCivilian") {
509 // Don't consider FactionCivilian.
510 continue;
511 }
512 if (m_skirmishSides[i].getScriptList()==NULL) continue;
513 if (m_skirmishSides[i].getScriptList()->getScript() != NULL ||
514 m_skirmishSides[i].getScriptList()->getScriptGroup()!=NULL) {
515 gotScripts = true;
516 }
517 }
518 if (!gotScripts) {
519 AsciiString path = "data\\Scripts\\SkirmishScripts.scb";
520 DEBUG_LOG(("Skirmish map using standard scripts\n"));
521 m_skirmishTeamrec.clear();
522 CachedFileInputStream theInputStream;
523 if (theInputStream.open(path)) {
524 ChunkInputStream *pStrm = &theInputStream;
525 DataChunkInput file( pStrm );
526 file.registerParser( AsciiString("PlayerScriptsList"), AsciiString::TheEmptyString, ScriptList::ParseScriptsDataChunk );
527 file.registerParser( AsciiString("ScriptsPlayers"), AsciiString::TheEmptyString, ParsePlayersDataChunk );
528 file.registerParser( AsciiString("ScriptTeams"), AsciiString::TheEmptyString, ParseTeamsDataChunk );
529 if (!file.parse(this)) {
530 DEBUG_LOG(("ERROR - Unable to read in skirmish scripts.\n"));
531 return;
532 }
533 ScriptList *scripts[MAX_PLAYER_COUNT];
534 Int count = ScriptList::getReadScripts(scripts);
535 Int i;
536 for (i=0; i<count; i++) {
537 Int curSide = -1;
538 Int j;
539 for (j=0; j<m_numSkirmishSides; j++) {
540 AsciiString name = getSkirmishSideInfo(j)->getDict()->getAsciiString(TheKey_playerName);
541 if (name == static_readPlayerNames[i]) {
542 curSide = j;
543 break;
544 }
545 }
546 if (curSide == -1) continue;
547
548 ScriptList *pSL = getSkirmishSideInfo(curSide)->getScriptList();
549 getSkirmishSideInfo(curSide)->setScriptList(scripts[i]);
550 scripts[i] = NULL;
551 if (pSL)
552 pSL->deleteInstance();
553 scripts[i] = NULL;
554 }
555 for (i=0; i<MAX_PLAYER_COUNT; i++) {
556 static_readPlayerNames[i].clear();
557 }
558 }
559
560
561 }

580 }
581 return false;

638Bool SidesList::validateAllyEnemyList(const AsciiString& tname, AsciiString& allies)
639{
640 // owners/allies/enemies must be players.
641
642 Bool modified = false;
643
644 AsciiString str, newstr, token;
645
646 str = allies;
647 newstr.clear();
648 while (str.nextToken(&token))
649 {
650 if (token == tname)
651 {
652 modified = true;
653 continue; // no allies/enemies with self
654 }
655
656 SidesInfo *si = findSideInfo(token);
657 if (!si)
658 {
659 modified = true;
660 continue; // player not found.
661 }
662
663 if (!newstr.isEmpty())
664 newstr.concat(" ");
665 newstr.concat(token);
666 }
667
668 allies = newstr;
669 return modified;
670}

672void SidesList::addPlayerByTemplate(AsciiString playerTemplateName)
673{
674 AsciiString playerName;
675 UnicodeString playerDisplayName;
676 Bool isHuman = false;
677
678 if (playerTemplateName.isEmpty())
679 {
680 playerName.set(""); // magic code for "neutral"
681 playerDisplayName = L"Neutral";
682 isHuman = false;
683 }
684 else
685 {
686 playerName.set("Plyr");
687 if (playerTemplateName.startsWith("Faction"))
688 {
689 playerName.concat(playerTemplateName.str() + 7);
690 }
691 else
692 {
693 playerName.concat(playerTemplateName);
694 }
695 playerDisplayName.translate(playerName);
696 isHuman = true;
697 // special-case "civilian"...
698 if (playerName == "PlyrCivilian")
699 isHuman = false;
700 }
701
702 Dict d;
703
704 d.clear();
705 d.setAsciiString(TheKey_playerName, playerName);
706 d.setBool(TheKey_playerIsHuman, isHuman);
707 d.setUnicodeString(TheKey_playerDisplayName, playerDisplayName);
708 d.setAsciiString(TheKey_playerFaction, playerTemplateName);
709 d.setAsciiString(TheKey_playerAllies, AsciiString::TheEmptyString);
710 d.setAsciiString(TheKey_playerEnemies, AsciiString::TheEmptyString);
711
712 addSide(&d);
713
714 AsciiString playerTeamName;
715 playerTeamName.set("team");
716 playerTeamName.concat(playerName);
717
718 d.clear();
719 d.setAsciiString(TheKey_teamName, playerTeamName);
720 d.setAsciiString(TheKey_teamOwner, playerName);
721 d.setBool(TheKey_teamIsSingleton, true);
722 addTeam(&d);
723}

725Bool SidesList::validateSides()
726{
727 Bool modified = false;
728
729 // ensure we have at least one player, and at least one neutral player.
730 Int i;
731 Int neutral = -1;
732 Int num = getNumSides();
733 for (i = 0; i < num; i++)
734 {
735 if (getSideInfo(i)->getDict()->getAsciiString(TheKey_playerName).isEmpty())
736 {
737 neutral = i;
738 break;
739 }
740 }
741 if (neutral == -1)
742 {
743 addPlayerByTemplate(AsciiString::TheEmptyString);
744 modified = true;
745 }
746
747 // now ensure that every player has a proper "default team"
748 for (i = 0; i < getNumSides(); i++)
749 {
750 Dict *pdict = getSideInfo(i)->getDict();
751 AsciiString pname = pdict->getAsciiString(TheKey_playerName);
752 AsciiString tname("team");
753 tname.concat(pname);
754 TeamsInfo *ti = findTeamInfo(tname);
755 if (ti)
756 {
757 // make sure the team owner points back to the player.
758 if (ti->getDict()->getAsciiString(TheKey_teamOwner) != pname)
759 {
760 DEBUG_CRASH(("hmm, team owner mismatch (%s) (%s), this should not normally be possible",ti->getDict()->getAsciiString(TheKey_teamOwner).str(), pname.str()));
761 ti->getDict()->setAsciiString(TheKey_teamOwner, pname);
762 modified = true;
763 }
764 // default teams are always singletons.
765 if (!ti->getDict()->getBool(TheKey_teamIsSingleton))
766 {
767 DEBUG_CRASH(("hmm, this should not normally be possible"));
768 ti->getDict()->setBool(TheKey_teamIsSingleton, true);
769 modified = true;
770 }
771 }
772 else
773 {
774 DEBUG_LOG(("*** default team for player %s missing (should not be possible), adding it...\n",tname.str()));
775 Dict d;
776 d.setAsciiString(TheKey_teamName, tname);
777 d.setAsciiString(TheKey_teamOwner, pname);
778 d.setBool(TheKey_teamIsSingleton, true);
779 addTeam(&d);
780 modified = true;
781 }
782
783 AsciiString allies = pdict->getAsciiString(TheKey_playerAllies);
784 AsciiString enemies = pdict->getAsciiString(TheKey_playerEnemies);
785
786 // ensure all teams have valid allies & enemies.
787 // (note that owners can be teams or players, but allies/enemies can only be teams.)
788 if (validateAllyEnemyList(pname, allies))
789 {
790 DEBUG_LOG(("bad allies...\n"));
