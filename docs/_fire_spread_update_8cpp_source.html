59//-------------------------------------------------------------------------------------------------
60Bool PartitionFilterFlammable::allow(Object *objOther)
61{
62 // It must be burnable in general, and burnable now
63 static NameKeyType key_FlammableUpdate = NAMEKEY("FlammableUpdate");
64 FlammableUpdate* fu = (FlammableUpdate*)objOther->findUpdateModule(key_FlammableUpdate);
65 if (fu == NULL)
66 return FALSE;
67
68 if( ! fu->wouldIgnite() )
69 return FALSE;
70
71 return TRUE;
72}

117UpdateSleepTime FireSpreadUpdate::update( void )
118{
119 const FireSpreadUpdateModuleData* d = getFireSpreadUpdateModuleData();
120 Object* me = getObject();
121
122 if( !me->getStatusBits().test( OBJECT_STATUS_AFLAME ) )
123 return UPDATE_SLEEP_FOREVER; // not on fire -- sleep forever
124 {
125 ObjectCreationList::create( d->m_oclEmbers, getObject(), NULL );
126
127 if( d->m_spreadTryRange != 0 )
128 {
129 // This will spread fire explicitly
130 PartitionFilterFlammable fFilter;
131 PartitionFilter *filters[] = { &fFilter, NULL };
132
133// SimpleObjectIterator *iter = NULL;
134// iter = ThePartitionManager->iterateObjectsInRange(getObject(),
135// d->m_spreadTryRange,
136// FROM_CENTER_3D,
137// filters,
138// ITER_SORTED_NEAR_TO_FAR
139// );
140// MemoryPoolObjectHolder hold(iter);
141// Object *objectToLight = iter->first();
142//
143// srj sez: the above code is stupid and slow. since we only want the closest object,
144// just ask for that; the above has to find ALL objects in range, but we ignore all
145// but the first (closest).
146//
147 Object* objectToLight = ThePartitionManager->getClosestObject(getObject(), d->m_spreadTryRange, FROM_CENTER_3D, filters);
148 if( objectToLight )
149 {
150 static NameKeyType key_FlammableUpdate = NAMEKEY("FlammableUpdate");
151 FlammableUpdate* fu = (FlammableUpdate*)objectToLight->findUpdateModule(key_FlammableUpdate);
152 if( fu )
153 fu->tryToIgnite();
154 }
155 }
156
157 return UPDATE_SLEEP(calcNextSpreadDelay());
158 }
159}

173UnsignedInt FireSpreadUpdate::calcNextSpreadDelay()
174{
175 const FireSpreadUpdateModuleData* d = getFireSpreadUpdateModuleData();
176 UnsignedInt delay = GameLogicRandomValue( d->m_minSpreadTryDelayData, d->m_maxSpreadTryDelayData );
177 if (delay < 1)
178 delay = 1;
179 return delay;
180}
