103static void parseTimeAndLocationInfo( INI *ini, void *instance,
104 TimeAndLocationInfo *timeAndLocationInfo )
105{
106
107 // delay label
108 const char *token = ini->getNextToken( ini->getSepsColon() );
109 if( stricmp( token, "Delay" ) != 0 )
110 {
111
112 DEBUG_CRASH(( "Expected 'Delay' token, found '%s'\n", token ));
113 throw INI_INVALID_DATA;
114
115 } // end if
116
117 // delay value
118 ini->parseDurationUnsignedInt( ini, instance, &timeAndLocationInfo->delay, NULL );
119
120 // get optional bone label
121 token = ini->getNextTokenOrNull( ini->getSepsColon() );
122 if( token )
123 {
124
125 // token must be a label for bone location
126 if( stricmp( token, "Bone" ) != 0 )
127 {
128
129 DEBUG_CRASH(( "Expected 'Bone' token, found '%s'\n", token ));
130 throw INI_INVALID_DATA;
131
132 } // end if
133
134 // read bone name and store
135 timeAndLocationInfo->boneName = ini->getNextAsciiString();
136
137 } // end if
138
139} // end parseTimeAndLocationInfo

143/*static*/ void BridgeBehaviorModuleData::parseFX( INI *ini,
144 void *instance,
145 void *store,
146 const void *userData )
147{
148 const char *token;
149
150 // create item we will read into and push on list
151 BridgeFXInfo item;
152 item.fx = NULL;
153
154 // get list to store at
155 BridgeFXList *bridgeFXList = (BridgeFXList *)store;
156
157 // fx list label
158 token = ini->getNextToken( ini->getSepsColon() );
159 if( stricmp( token, "FX" ) != 0 )
160 {
161
162 DEBUG_CRASH(( "Expected 'FX' token, found '%s'\n", token ));
163 throw INI_INVALID_DATA;
164
165 } // end if
166
167 // fx list name and store as pointer
168 FXList *fx;
169 ini->parseFXList( ini, instance, &fx, NULL );
170
171 // store fx list to item
172 item.fx = fx;
173
174 // parse the timing and location info
175 parseTimeAndLocationInfo( ini, instance, &item.timeAndLocationInfo );
176
177 // put on list
178 bridgeFXList->push_back( item );
179
180} // end parseFX

325void BridgeBehavior::resolveFX( void )
326{
327 Object *us = getObject();
328 Bridge *bridge = TheTerrainLogic->findBridgeAt( us->getPosition() );
329
330 // sanity
331 if( bridge == NULL )
332 return;
333
334 // get the bridge template name
335 AsciiString bridgeTemplateName = bridge->getBridgeTemplateName();
336
337 // find the bridge template
338 TerrainRoadType *bridgeTemplate = TheTerrainRoads->findBridge( bridgeTemplateName );
339
340 // sanity
341 if( bridgeTemplate == NULL )
342 return;
343
344 AsciiString name;
345 for( Int bodyState = BODY_PRISTINE; bodyState < BODYDAMAGETYPE_COUNT; ++bodyState )
346 {
347
348 // initialize the fx and ocl lists
349 for( Int i = 0; i < MAX_BRIDGE_BODY_FX; ++i )
350 {
351
352 name = bridgeTemplate->getDamageToOCLString( (BodyDamageType)bodyState, i );
353 m_damageToOCL[ bodyState ][ i ] = TheObjectCreationListStore->findObjectCreationList( name.str() );
354 if( name.isEmpty() == FALSE && m_damageToOCL[ bodyState ][ i ] == NULL )
355 DEBUG_CRASH(( "OCL list '%s' not found\n", name.str() ));
356
357 name = bridgeTemplate->getDamageToFXString( (BodyDamageType)bodyState, i );
358 m_damageToFX[ bodyState ][ i ] = TheFXListStore->findFXList( name.str() );
359 if( name.isEmpty() == FALSE && m_damageToFX[ bodyState ][ i ] == NULL )
360 DEBUG_CRASH(( "FX list '%s' not found\n", name.str() ));
361
362 name = bridgeTemplate->getRepairedToOCLString( (BodyDamageType)bodyState, i );
363 m_repairToOCL[ bodyState ][ i ] = TheObjectCreationListStore->findObjectCreationList( name.str() );
364 if( name.isEmpty() == FALSE && m_repairToOCL[ bodyState ][ i ] == NULL )
365 DEBUG_CRASH(( "OCL list '%s' not found\n", name.str() ));
366
367 name = bridgeTemplate->getRepairedToFXString( (BodyDamageType)bodyState, i );
368 m_repairToFX[ bodyState ][ i ] = TheFXListStore->findFXList( name.str() );;
369 if( name.isEmpty() == FALSE && m_repairToFX[ bodyState ][ i ] == NULL )
370 DEBUG_CRASH(( "FX list '%s' not found\n", name.str() ));
371
372 } // end for i
373
374 // audio sounds
375 name = bridgeTemplate->getDamageToSoundString( (BodyDamageType)bodyState );
376 m_damageToSound[ bodyState ].setEventName( name );
377 m_damageToSound[ bodyState ].setObjectID( us->getID() );
378
379 name = bridgeTemplate->getRepairedToSoundString( (BodyDamageType)bodyState );
380 m_repairToSound[ bodyState ].setEventName( name );
381 m_repairToSound[ bodyState ].setObjectID( us->getID() );
382
383 } // end for, bodyState
384
385 // fx are now "resolved"
386 m_fxResolved = TRUE;
387
388} // end resolveFX

433void BridgeBehavior::onDamage( DamageInfo *damageInfo )
434{
435
436 //
437 // get our body info so we now how much damage percent is being done to us ... we need this
438 // so that we can propagate the same damage percentage amont the towers and the bridge
439 //
440 BodyModuleInterface *body = getObject()->getBodyModule();
441 Real damagePercentage = damageInfo->in.m_amount / body->getMaxHealth();
442
443 //
444 // if the damage didn't come from a bridge tower, then we must propagate this damage
445 // to all our towers
446 //
447 Object *source = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
448 if( source == NULL || source->isKindOf( KINDOF_BRIDGE_TOWER ) == FALSE )
449 {
450 Object *tower;
451
452 for( Int i = 0; i < BRIDGE_MAX_TOWERS; i++ )
453 {
454
455 tower = TheGameLogic->findObjectByID( getTowerID( (BridgeTowerType)i ) );
456 if( tower )
457 {
458 BodyModuleInterface *towerBody = tower->getBodyModule();
459 DamageInfo towerDamage;
460
461 towerDamage.in.m_amount = damagePercentage * towerBody->getMaxHealth();
462 towerDamage.in.m_sourceID = getObject()->getID(); // we're now the source
463 towerDamage.in.m_damageType = damageInfo->in.m_damageType;
464 towerDamage.in.m_deathType = damageInfo->in.m_deathType;
465 tower->attemptDamage( &towerDamage );
466
467 } // end if
468
469 } // end for i
470
471 } // end if
472
473} // end onDamage

516void BridgeBehavior::getRandomSurfacePosition( TerrainRoadType *bridgeTemplate,
517 const BridgeInfo *bridgeInfo,
518 Coord3D *pos )
519{
520
521 // sanity
522 if( bridgeInfo == NULL || pos == NULL )
523 return;
524
525 //
526 // pick the spot by finding vectors along the edge of the bridge region, scaling
527 // them and then adding them together
528 //
529 Real scale;
530
531 Coord3D v1;
532 v1.x = bridgeInfo->toLeft.x - bridgeInfo->fromLeft.x;
533 v1.y = bridgeInfo->toLeft.y - bridgeInfo->fromLeft.y;
534 v1.z = bridgeInfo->toLeft.z - bridgeInfo->fromLeft.z;
535 scale = GameLogicRandomValueReal( 0.0f, 1.0f );
536 v1.x *= scale;
537 v1.y *= scale;
538 v1.z *= scale;
539
540 Coord3D v2;
541 v2.x = bridgeInfo->fromRight.x - bridgeInfo->fromLeft.x;
542 v2.y = bridgeInfo->fromRight.y - bridgeInfo->fromLeft.y;
543 v2.z = bridgeInfo->fromRight.z - bridgeInfo->fromLeft.z;
544 scale = GameLogicRandomValueReal( 0.0f, 1.0f );
545 v2.x *= scale;
546 v2.y *= scale;
547 v2.z *= scale;
548
549 // set the position
550 pos->x = v1.x + v2.x + bridgeInfo->fromLeft.x;
551 pos->y = v1.y + v2.y + bridgeInfo->fromLeft.y;
552 pos->z = v1.z + v2.z + bridgeInfo->fromLeft.z;
553
554 //
555 // we now have a position picked, the last thing to do is add in an additional
556 // Z component so that effects can be created in a "cube" area on and above the bridge
557 //
558 pos->z += GameLogicRandomValueReal( 0.0f, bridgeTemplate->getTransitionEffectsHeight() );
559
560} // end getRandomSurfacePosition

564void BridgeBehavior::doAreaEffects( TerrainRoadType *bridgeTemplate,
565 Bridge *bridge,
566 const ObjectCreationList *ocl,
567 const FXList *fx )
568{
569
570 // sanity
571 if( bridge == NULL )
572 return;
573
574 // if no effects, don't bother
575 if( ocl == NULL && fx == NULL )
576 return;
577
578 // get bridge info
579 const BridgeInfo *bridgeInfo = bridge->peekBridgeInfo();
580
581 // play effects in the bridge area
582 const Int maxEffects = bridgeTemplate->getNumFXPerType();
583 Coord3D pos;
584 for( Int i = 0; i < maxEffects; ++i )
585 {
586
587 // pick spot in the bridge area and do FX
588 if( fx )
589 {
590
591 getRandomSurfacePosition( bridgeTemplate, bridgeInfo, &pos );
592 FXList::doFXPos( fx, &pos );
593
594 } // end if
595
596 // pick spot in the bridge area and do OCL
597 if( ocl )
598 {
599
600 getRandomSurfacePosition( bridgeTemplate, bridgeInfo, &pos );
601 ObjectCreationList::create( ocl, getObject(), &pos, NULL, INVALID_ANGLE );
602
603 } // end if
604
605 } // end for i
606
607} // end doAreaEffects

611void BridgeBehavior::onBodyDamageStateChange( const DamageInfo* damageInfo,
612 BodyDamageType oldState,
613 BodyDamageType newState )
614{
615
616 //
617 // check for coming back from the dead, if our new state is not the rubble state, we can't
618 // possibly be dead
619 //
620 if( newState != BODY_RUBBLE )
621 m_deathFrame = 0;
622
623 // first resolve any fx stuff if we need to
624 if( m_fxResolved == FALSE )
625 resolveFX();
626
627 // sanity
628 if( m_fxResolved == FALSE )
629 return;
630
631 // sanity
632 DEBUG_ASSERTCRASH( oldState != newState, ("BridgeBehavior::onBodyDamageStateChange - oldState and newState should be different if this is getting called\n") );
633
634 Object *us = getObject();
635 Bridge *bridge = TheTerrainLogic->findBridgeAt( us->getPosition() );
636
637 // sanity
638 if( bridge == NULL )
639 {
640
641 DEBUG_CRASH(( "BridgeBehavior - Unable to find bridge\n" ));
642 return;
643
644 } // end if
645
646 // get the bridge template name
647 AsciiString bridgeTemplateName = bridge->getBridgeTemplateName();
648
649 // find the bridge template
650 TerrainRoadType *bridgeTemplate = TheTerrainRoads->findBridge( bridgeTemplateName );
651
652 // sanity
653 DEBUG_ASSERTCRASH( bridgeTemplate, ("BridgeBehavior: Unable to find bridge template '%s' in bridge object '%s'\n",
654 bridgeTemplateName,
655 us->getTemplate()->getName().str()) );
656
657 //
658 // given the old state and the new state, did we get worse (damaged) or did
659 // we get better (repaired)?
660 //
661 Bool gotRepaired = IS_CONDITION_WORSE( oldState, newState );
662
663 // get the effect data
664 AsciiString soundString;
665 AsciiString oclString[ MAX_BRIDGE_BODY_FX ];
666 AsciiString fxString[ MAX_BRIDGE_BODY_FX ];
667 if( gotRepaired )
668 {
669
670 // play the sound
671 TheAudio->addAudioEvent( &m_repairToSound[ newState ] );
672
673 for( Int i = 0; i < MAX_BRIDGE_BODY_FX; i++ )
674 doAreaEffects( bridgeTemplate, bridge, m_repairToOCL[ newState ][ i ], m_repairToFX[ newState ][ i ] );
675
676 } // end if
677 else
678 {
679
680 // play the sound
681 TheAudio->addAudioEvent( &m_damageToSound[ newState ] );
682
683 for( Int i = 0; i < MAX_BRIDGE_BODY_FX; i++ )
684 doAreaEffects( bridgeTemplate, bridge, m_damageToOCL[ newState ][ i ], m_damageToFX[ newState ][ i ] );
685
686 } // end else
687
688 // update bridge damage states
690 TheTerrainLogic->updateBridgeDamageStates();
691
692 //
693 // for the local player, if this bridge has switched from rubble, to usable, or from
694 // usable to rubble, we should reflect the change on the radar. note that we
695 // request that the radar queue a refresh sometime in the future because it keeps
696 // track of how often we makes requests to do a refresh and doesn't do them too
697 // often because it's expensive to refresh the terrain
698 //
699 if( oldState == BODY_RUBBLE || newState == BODY_RUBBLE )
700 TheRadar->queueTerrainRefresh();
701
702} // end onBodyDamageStateChange

704// ------------------------------------------------------------------------------------------------
705// ------------------------------------------------------------------------------------------------
706UpdateSleepTime BridgeBehavior::update( void )
707{
708
709 // if we're dead, we need to possibly throw off some effects
710 if( m_deathFrame != 0 )
711 {
712 AsciiString boneName;
713
714 // get object
715 Object *us = getObject();
716
717 // get module data
718 const BridgeBehaviorModuleData *modData = getBridgeBehaviorModuleData();
719
720 // get bridge information
721 Bridge *bridge = TheTerrainLogic->findBridgeAt( us->getPosition() );
722 const BridgeInfo *bridgeInfo = NULL;
723 TerrainRoadType *bridgeTemplate = NULL;
724 if ( bridge )
725 {
726 DEBUG_ASSERTCRASH( bridge, ("BridgeBehavior::update - Unable to find bridge\n") );
727
728 // get bridge info
729 bridgeInfo = bridge->peekBridgeInfo();
730
731 // get the bridge template info
732 AsciiString bridgeTemplateName = bridge->getBridgeTemplateName();
733 bridgeTemplate = TheTerrainRoads->findBridge( bridgeTemplateName );
734 DEBUG_ASSERTCRASH( bridgeTemplate, ("BridgeBeahvior::getRandomSurfacePosition - Encountered a bridge with no template!\n") );
735 }
736
737 // how much time has passed between now and our destruction frame
738 UnsignedInt deathTime = TheGameLogic->getFrame() - m_deathFrame;
739
740 // see if there are any fx visuals we need to execute
741 BridgeFXList::const_iterator fxIt;
742 for( fxIt = modData->m_fx.begin(); fxIt != modData->m_fx.end(); ++fxIt )
743 {
744
745 //
746 // we'll launch an fx list if our death time is equal to exactly the delay
747 // we're waiting for to launch the list
748 //
749 if( deathTime == (*fxIt).timeAndLocationInfo.delay )
750 {
751 Coord3D pos;
752
753 //
754 // if a bone name is present, we'll use the bone position, otherwise we'll pick a
755 // spot somewhere on the bridge surface
756 //
757 boneName = (*fxIt).timeAndLocationInfo.boneName;
758 if( boneName.isEmpty() == FALSE )
759 us->getSingleLogicalBonePosition( boneName.str(), &pos, NULL );
760 else if ( bridge && bridgeTemplate && bridgeInfo)//we have valid Terrain data for the bridge
761 getRandomSurfacePosition( bridgeTemplate, bridgeInfo, &pos );
762 else
763 pos.set( getObject()->getPosition() );
764
765
766 // launch the fx list
767 FXList::doFXPos( (*fxIt).fx, &pos );
768
769 } // end if
770
771 } // end for, fxIt
772
773 // see if there are any ocl visuals we need to execute
774 BridgeOCLList::const_iterator oclIt;
775 for( oclIt = modData->m_ocl.begin(); oclIt != modData->m_ocl.end(); ++oclIt )
776 {
777
778 //
779 // we'll launch an ocl list if our death time is equal to exactly the delay

1064 Real spacing = scaffoldTemplate->getTemplateGeometryInfo().getMajorRadius() * 2.0f;

1080 //
1081 Coord3D leftStart;
1082 leftStart.x = ((bridgeInfo.fromLeft.x - bridgeInfo.fromRight.x) / 2.0f) + bridgeInfo.fromRight.x;
1083 leftStart.y = ((bridgeInfo.fromLeft.y - bridgeInfo.fromRight.y) / 2.0f) + bridgeInfo.fromRight.y;

1087 rightStart.x = ((bridgeInfo.toLeft.x - bridgeInfo.toRight.x) / 2.0f) + bridgeInfo.toRight.x;
1088 rightStart.y = ((bridgeInfo.toLeft.y - bridgeInfo.toRight.y) / 2.0f) + bridgeInfo.toRight.y;
1089 rightStart.z = ((bridgeInfo.toLeft.z - bridgeInfo.toRight.z) / 2.0f) + bridgeInfo.toRight.z;
1090
1091 //
1092 // now that we have the left and right start points, we will compute two angles to
1093 // use for all the objects that we will create ... objects on the left side of the
1094 // bridge will be given a 'leftAngle' pointing from 'leftStart' to 'rightStart' and
1095 // the opposite will be given to objects on the right side of the bridge
1096 //
1097 Coord2D angleV;
1098 angleV.x = rightStart.x - leftStart.x;
1099 angleV.y = rightStart.y - leftStart.y;
1100 Real leftAngle = angleV.toAngle();
1101 Real rightAngle = leftAngle + TWO_PI;
1102
1103 // compute vector from left to right across bridge and the reverse
1104 Coord3D leftVector;
1105 leftVector.x = rightStart.x - leftStart.x;
1106 leftVector.y = rightStart.y - leftStart.y;
1107 leftVector.z = rightStart.z - leftStart.z;
1108 Coord3D rightVector;
1109 rightVector.x = leftStart.x - rightStart.x;
1110 rightVector.y = leftStart.y - rightStart.y;
1111 rightVector.z = leftStart.z - rightStart.z;

1114 // how many of these scaffold objects will take to tile from each of the endpoints
1115 // to the center area of the bridge
1116 //

1119
1120 //

1122 // go through the creation loop ceil( numObjects / 2.0f ) times, and each
1123 // time through the loop we'll create an object to move from each side of the
1124 // bridge, except the last object if the number of objects is odd is dead in the
1125 // center
1126 //

1128
1129 //
1130 // normalize left and right vectors as it is a vector that goes from our left start
1131 // position to the destination right start position ... we will multiply this vector by a
1132 // spacing amount and add to the left start position to find a destination position
1133 // for a particular scaffold object along the bridge surface
1134 //

1136 rightVector.normalize();

1161 // about any object reaching a destination before any other object

1164 destinationPos.y = leftVector.y * (spacing * i) + riseToPos->y;
1165 destinationPos.z = leftVector.z * (spacing * i) + riseToPos->z;
1166
1167 //
1168 // now that they key positions are calculated, set the rest of the position data
1169 // and movement speeds for the object
1170 //
1171 setScaffoldData( obj, angle, &scaffoldHeight, riseToPos, &destinationPos, center );
1172
1173 // keeping track of objects created
1174 scaffoldObjectsCreated++;
1175 m_scaffoldObjectIDList.push_back( obj->getID() );
1176
1177 //
1178 // create support object layers under the scaffold object, this object mirrors the scaffold
1179 // object except is on a lower layer
1180 //
1181 Real offset = riseToPos->z;
1182 Coord3D supportRiseToPos = *riseToPos;
1183 Coord3D supportDestinationPos = destinationPos;
1184 Coord3D supportBridgeCenter = *center;

1189 supportDestinationPos.z -= scaffoldSupportHeight;
1190 supportBridgeCenter.z -= scaffoldSupportHeight;

1193 angle,
1194 &scaffoldSupportHeight,
1195 &supportRiseToPos,
1196 &supportDestinationPos,
1197 &supportBridgeCenter );
1198 m_scaffoldObjectIDList.push_back( obj->getID() );
1199
1200
1201 // off to the next layer
1202 offset -= scaffoldSupportHeight;
1203
1204 } // end while
1205
1206 //
1207 // now create the object from the "right" side of the bridge ... but note that
1208 // we don't do this on the last iteration through this loop when we have an odd
1209 // number of objects we're tiling because all the space is already perfectly used up)
1210 //
1211 if( scaffoldObjectsCreated < numObjects )
1212 {
1213
1214 // sanity
1215 DEBUG_ASSERTCRASH( scaffoldObjectsCreated < numObjects,
1216 ("Creating too many scaffold objects\n") );
1217
1218 // create new object
1219 obj = TheThingFactory->newObject( scaffoldTemplate, us->getTeam() );
1220
1221 // this object is on the "right" side of the bridge
1222 riseToPos = &rightStart;
1223 angle = &rightAngle;
1224

1228 // can just assign the speeds so they line up perfectly and not have to worry

1232 destinationPos.y = rightVector.y * (spacing * i) + riseToPos->y;
1233 destinationPos.z = rightVector.z * (spacing * i) + riseToPos->z;
1234
1235 // set the rest scaffold data again
1236 setScaffoldData( obj, angle, &scaffoldHeight, riseToPos, &destinationPos, center );
1237
1238 // keeping track of objects created
1239 scaffoldObjectsCreated++;
1240 m_scaffoldObjectIDList.push_back( obj->getID() );
1241
1242 //
1243 // create support object layers under the scaffold object, this object mirrors the scaffold
1244 // object except is on a lower layer
1245 //
1246 Real offset = riseToPos->z;
1247 Coord3D supportRiseToPos = *riseToPos;
1248 Coord3D supportDestinationPos = destinationPos;
1249 Coord3D supportBridgeCenter = *center;
1250 while( offset >= 0.0f )
1251 {
1252
1253 supportRiseToPos.z -= scaffoldSupportHeight;
1254 supportDestinationPos.z -= scaffoldSupportHeight;
1255 supportBridgeCenter.z -= scaffoldSupportHeight;
