74static void drawHiliteBar( const Image *left, const Image *right,
75 const Image *center, const Image *smallCenter,
76 Int startX, Int startY,
77 Int endX, Int endY )
78{
79 ICoord2D barWindowSize; // end point of bar from window origin
80 Int xOffset = 0, yOffset = 0; // incase we want this functionality later
81 ICoord2D start, end;
82 Int i;
83 IRegion2D clipRegion;
84
85
86
87 barWindowSize.x = endX - startX;
88 barWindowSize.y = endY - startY;
89
90 //
91 // the bar window size will always be at least big enough to accomodate
92 // the left and right ends
93 //
94 if( barWindowSize.x < left->getImageWidth() + right->getImageWidth() )
95 barWindowSize.x = left->getImageWidth() + right->getImageWidth();
96
97 // get image sizes for the ends
98 ICoord2D leftSize, rightSize;
99 leftSize.x = left->getImageWidth();
100 leftSize.y = left->getImageHeight();
101 rightSize.x = right->getImageWidth();
102 rightSize.y = right->getImageHeight();
103
104 // get two key points used in the end drawing
105 ICoord2D leftEnd, rightStart;
106 leftEnd.x = startX + leftSize.x + xOffset;
107 leftEnd.y = startY + barWindowSize.y + yOffset;
108 rightStart.x = startX + barWindowSize.x - rightSize.x + xOffset;
109 rightStart.y = startY + yOffset;
110
111 // draw the center repeating bar
112 Int centerWidth, pieces;
113
114 // get width we have to draw our repeating center in
115 centerWidth = rightStart.x - leftEnd.x;
116
117 // how many whole repeating pieces will fit in that width
118 pieces = centerWidth / center->getImageWidth();
119
120
121
122 // draw the pieces
123 start.x = leftEnd.x;
124 start.y = startY + yOffset;
125 end.y = start.y + barWindowSize.y;
126 for( i = 0; i < pieces; i++ )
127 {
128
129 end.x = start.x + center->getImageWidth();
130 TheWindowManager->winDrawImage( center,
131 start.x, start.y,
132 end.x, end.y );
133 start.x += center->getImageWidth();
134
135 } // end for i
136
137 //
138 // how many small repeating pieces will fit in the gap from where the
139 // center repeating bar stopped and the right image, draw them
140 // and overlapping underneath where the right end will go
141 //
142 // set the text clip region to the outline of the listbox
143 clipRegion.lo.x = leftEnd.x;
144 clipRegion.lo.y = startY + yOffset;
145 clipRegion.hi.x = leftEnd.x + centerWidth;
146 clipRegion.hi.y = start.y + barWindowSize.y;
147 TheDisplay->setClipRegion(&clipRegion);
148 centerWidth = rightStart.x - start.x;
149 if( centerWidth )
150 {
151
152 pieces = centerWidth / smallCenter->getImageWidth() + 1;
153 end.y = start.y + barWindowSize.y;
154 for( i = 0; i < pieces; i++ )
155 {
156
157 end.x = start.x + smallCenter->getImageWidth();
158 TheWindowManager->winDrawImage( smallCenter,
159 start.x, start.y,
160 end.x, end.y );
161 start.x += smallCenter->getImageWidth();
162
163 } // end for i
164
165 } // end if
166 TheDisplay->enableClipping(FALSE);
167 // draw left end
168 start.x = startX + xOffset;
169 start.y = startY + yOffset;
170 end = leftEnd;
171 TheWindowManager->winDrawImage(left, start.x, start.y, end.x, end.y);
172
173 // draw right end
174 start = rightStart;
175 end.x = start.x + rightSize.x;
176 end.y = start.y + barWindowSize.y;
177 TheWindowManager->winDrawImage(right, start.x, start.y, end.x, end.y);
178
179} // end drawHiliteBar

184static void drawListBoxText( GameWindow *window, WinInstanceData *instData,
185 Int x, Int y, Int width, Int height,
186 Bool useImages )
187{
188 Int drawY;
189 ListboxData *list = (ListboxData *)window->winGetUserData();
190 Int i;
191 Bool selected;
192 Int listLineHeight;
193 Color textColor;
194// W3DGameWindow *w3dWindow = static_cast<W3DGameWindow *>(window);
195 IRegion2D clipRegion;
196 ICoord2D start, end;
197
198 //
199 // save the clipping information region cause we're going to use it here
200 // in drawing the text
201 //
202// TheWindowManager->winGetClipRegion( &clipRegion );
203
204 // set clip region to inside the outline box.
205// TheWindowManager->winSetClipRegion( x, y, width, height );
206
207 // set the text clip region to the outline of the listbox
208 clipRegion.lo.x = x + 1;
209 clipRegion.lo.y = y -3;
210 clipRegion.hi.x = x + width - 1;
211 clipRegion.hi.y = y + height - 1;
212
213 drawY = y - list->displayPos;
214
215 for( i = 0; ; i++ )
216 {
217
218 if( i > 0 )
219 if( list->listData[(i - 1)].listHeight >
220 (list->displayPos + list->displayHeight) )
221 break;
222
223 if( i == list->endPos )
224 break;
225
226 if( list->listData[i].listHeight < list->displayPos )
227 {
228 drawY += (list->listData[i].height + 1);
229 continue;
230 }
231
232 listLineHeight = list->listData[i].height + 1;
233 //textColor = list->listData[i].textColor;
234 selected = FALSE;
235
236 if( list->multiSelect )
237 {
238 Int j = 0;
239
240 while( list->selections[j] >= 0 )
241 {
242 if( i == list->selections[j] )
243 {
244 selected = TRUE;
245 break;
246 }
247
248 j++;
249 }
250 }
251 else
252 {
253 if( i == list->selectPos )
254 selected = TRUE;
255 }
256
257 // this item is selected, draw the selection color or image
258 if( selected )
259 {
260
261 if( useImages )
262 {
263 const Image *left, *right, *center, *smallCenter;
264
265 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
266 {
267
268 left = GadgetListBoxGetDisabledSelectedItemImageLeft( window );
269 right = GadgetListBoxGetDisabledSelectedItemImageRight( window );
270 center = GadgetListBoxGetDisabledSelectedItemImageCenter( window );
271 smallCenter = GadgetListBoxGetDisabledSelectedItemImageSmallCenter( window );
272
273 } // end if
274 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
275 {
276
277 left = GadgetListBoxGetHiliteSelectedItemImageLeft( window );
278 right = GadgetListBoxGetHiliteSelectedItemImageRight( window );
279 center = GadgetListBoxGetHiliteSelectedItemImageCenter( window );
280 smallCenter = GadgetListBoxGetHiliteSelectedItemImageSmallCenter( window );
281
282 } // end else if
283 else
284 {
285
286 left = GadgetListBoxGetEnabledSelectedItemImageLeft( window );
287 right = GadgetListBoxGetEnabledSelectedItemImageRight( window );
288 center = GadgetListBoxGetEnabledSelectedItemImageCenter( window );
289 smallCenter = GadgetListBoxGetEnabledSelectedItemImageSmallCenter( window );
290
291 } // end else
292
293 // draw select image across area
294
295 //
296 // where are we going to draw ... taking into account the clipping
297 // region of the edge of the listbox
298 //
299 start.x = x;
300 start.y = drawY;
301 end.x = start.x + width;
302 end.y = start.y + listLineHeight;
303
304 if( end.y > clipRegion.hi.y )
305 end.y = clipRegion.hi.y;
306 if( start.y < clipRegion.lo.y )
307 start.y = clipRegion.lo.y;
308
309 if( left && right && center && smallCenter )
310 drawHiliteBar( left, right, center, smallCenter, start.x + 1, start.y, end.x , end.y );
311
312 } // end if, use images
313 else
314 {
315 Color selectColor = WIN_COLOR_UNDEFINED,
316 selectBorder = WIN_COLOR_UNDEFINED;
317
318 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
319 {
320 selectColor = GadgetListBoxGetDisabledSelectedItemColor( window );
321 selectBorder = GadgetListBoxGetDisabledSelectedItemBorderColor( window );
322 } // end if, disabled
323 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
324 {
325 selectColor = GadgetListBoxGetHiliteSelectedItemColor( window );
326 selectBorder = GadgetListBoxGetHiliteSelectedItemBorderColor( window );
327 } // end else if, hilited
328 else
329 {
330 selectColor = GadgetListBoxGetEnabledSelectedItemColor( window );
331 selectBorder = GadgetListBoxGetEnabledSelectedItemBorderColor( window );
332 } // end else, enabled
333
334 // draw border
335
336 //
337 // where are we going to draw ... taking into account the clipping
338 // region of the edge of the listbox
339 //
340 start.x = x;
341 start.y = drawY;
342 end.x = start.x + width;
343 end.y = start.y + listLineHeight;
344
345 if( end.y > clipRegion.hi.y )
346 end.y = clipRegion.hi.y;
347 if( start.y < clipRegion.lo.y )
348 start.y = clipRegion.lo.y;
349
350 if( selectBorder != WIN_COLOR_UNDEFINED )
351 TheWindowManager->winOpenRect( selectBorder,
352 WIN_DRAW_LINE_WIDTH,
353 start.x, start.y,
354 end.x, end.y );
355
356 // draw filled inner rect
357
358 //
359 // where are we going to draw ... taking into account the clipping
360 // region of the edge of the listbox
361 //
362 start.x = x + 1;
363 start.y = drawY + 1;
364 end.x = start.x + width - 2;
365 end.y = start.y + listLineHeight - 2;
366
367 if( end.y > clipRegion.hi.y )
368 end.y = clipRegion.hi.y;
369 if( start.y < clipRegion.lo.y )
370 start.y = clipRegion.lo.y;
371
372 if( selectColor != WIN_COLOR_UNDEFINED )
373 TheWindowManager->winFillRect( selectColor,
374 WIN_DRAW_LINE_WIDTH,
375 start.x, start.y,
376 end.x, end.y );
377
378 } // end else, draw selection with colors
379
380 } // end if
381
382
383
384
385 Color dropColor = TheWindowManager->winMakeColor( 0, 0, 0, 255 );
386 DisplayString *string;
387
388 ListEntryCell *cells = list->listData[i].cell;
389 Int columnX = x;
390 IRegion2D columnRegion;
391 if( cells )
392 {
393 // loop through all the cells
394 for( Int j = 0; j < list->columns; j++ )
395 {
396 // setup the Clip Region size
397
398 columnRegion.lo.x = columnX;
399 columnRegion.lo.y = drawY;
400 if(list->columns == 1 && list->slider && list->slider->winIsHidden())
401 columnRegion.hi.x = columnX + width-3;
402 else
403 columnRegion.hi.x = columnX + list->columnWidth[j];
404 columnRegion.hi.y = drawY + list->listData[i].height;
405 if(columnRegion.lo.y < clipRegion.lo.y )
406 columnRegion.lo.y = clipRegion.lo.y;
407 if( columnRegion.hi.y > clipRegion.hi.y )
408 columnRegion.hi.y = clipRegion.hi.y;
409
410 // Display the Text Case;
411 if(cells[j].cellType == LISTBOX_TEXT)
412 {
413 textColor = cells[j].color;
414 string = (DisplayString *)cells[j].data;
415 if( BitTest( window->winGetStatus(), WIN_STATUS_ONE_LINE ) == TRUE )
416 {
417 string->setWordWrap(0);
418 // make sure the font of the text is the same as the windows
419 if( string->getFont() != window->winGetFont() )
420 string->setFont( window->winGetFont() );
421
422 // draw this text after setting the clip region for it
423 string->setClipRegion( &columnRegion );
424 string->draw( columnX + TEXT_X_OFFSET,
425 drawY,
426 textColor,
427 dropColor );
428
429 }
430 else
431 {
432
433 // make sure the font of the text is the same as the windows
434 if( string->getFont() != window->winGetFont() )
435 string->setFont( window->winGetFont() );
436
437 // set clip region and draw
438 string->setClipRegion( &columnRegion );
439 string->draw( columnX + TEXT_X_OFFSET,
440 drawY,
441 textColor,
442 dropColor );
443 }//else
444 }// if
445 else if(cells[j].cellType == LISTBOX_IMAGE && cells[j].data)
446 {
447 Int width, height;
448 if (cells[j].width > 0)
449 width = cells[j].width;
450 else
451 width = list->columnWidth[j];
452 if(cells[j].height > 0)
453 height = cells[j].height;
454 else
455 height = list->listData[i].height;
456 if(j == 0)
457 width--;
458 Int offsetX,offsetY;
459 if(width < list->columnWidth[j])
460 offsetX = columnX + ((list->columnWidth[j] - width) / 2);
461 else
462 offsetX = columnX;
463 if(height < list->listData[i].height)
464 offsetY = drawY + ((list->listData[i].height - height) / 2);
465 else
466 offsetY = drawY;
467
468 offsetY++;
469 if(offsetX <x+1)
470 offsetX = x+1;
471 TheDisplay->setClipRegion( &columnRegion );
472 TheWindowManager->winDrawImage( (const Image *)cells[j].data,
473 offsetX, offsetY,
474 offsetX + width, offsetY + height,cells[j].color );
475
476 }//else
477 columnX = columnX + list->columnWidth[j];
478 }// for
479 }//if
480
481
482 drawY += listLineHeight;
483 TheDisplay->enableClipping(FALSE);
484 }
485
486// TheWindowManager->winSetClipRegion( clipRegion.lo.x, clipRegion.lo.y,
487// clipRegion.hi.x, clipRegion.hi.y );
488
489} // end drawListBoxText

495// W3DGadgetListBoxDraw =======================================================
497//=============================================================================
498void W3DGadgetListBoxDraw( GameWindow *window, WinInstanceData *instData )
499{
500 Int width, height, fontHeight, x, y;
501 Color background, border, titleColor, titleBorder;
502 ListboxData *list = (ListboxData *)window->winGetUserData();
503 ICoord2D size;
504 DisplayString *title = instData->getTextDisplayString();
505
506 // get window position and size
507 window->winGetScreenPosition( &x, &y );
508 window->winGetSize( &size.x, &size.y );
509
510 // get font height
511 fontHeight = TheWindowManager->winFontHeight( instData->getFont() );
512
513 // alias width and height from size
514 width = size.x;
515 height = size.y;
516
517 // get the right colors
518 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
519 {
520 background = GadgetListBoxGetDisabledColor( window );
521 border = GadgetListBoxGetDisabledBorderColor( window );
522 titleColor = window->winGetDisabledTextColor();
523 titleBorder = window->winGetDisabledTextBorderColor();
524 } // end if, disabled
525 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
526 {
527 background = GadgetListBoxGetHiliteColor( window );
528 border = GadgetListBoxGetHiliteBorderColor( window );
529 titleColor = window->winGetHiliteTextColor();
530 titleBorder = window->winGetHiliteTextBorderColor();
531 } // end else if, hilited
532 else
533 {
534 background = GadgetListBoxGetEnabledColor( window );
535 border = GadgetListBoxGetEnabledBorderColor( window );
536 titleColor = window->winGetEnabledTextColor();
537 titleBorder = window->winGetEnabledTextBorderColor();
538 } // end else, enabled
539
540 // Draw the title
541 if( title && title->getTextLength() )
542 {
543
544 // set the font of this text to that of the window if not already
545 if( title->getFont() != window->winGetFont() )
546 title->setFont( window->winGetFont() );
547
548 // draw the text
549 title->draw( x + 1, y, titleColor, titleBorder );
550
551 y += fontHeight + 1;
552 height -= fontHeight + 1;
553
554 } // end if
555
556 // draw the back border
557 if( border != WIN_COLOR_UNDEFINED )
558 TheWindowManager->winOpenRect( border, WIN_DRAW_LINE_WIDTH,
559 x, y, x + width, y + height );
560
561 // draw background
562 if( background != WIN_COLOR_UNDEFINED )
563 TheWindowManager->winFillRect( background, WIN_DRAW_LINE_WIDTH,
564 x + 1, y + 1,
565 x + width - 1, y + height - 1 );
566
567 // If ScrollBar was requested ... adjust width.
568 if( list->slider && !list->slider->winIsHidden())
569 {
570 ICoord2D sliderSize;
571
572 list->slider->winGetSize( &sliderSize.x, &sliderSize.y );
573 width -= (sliderSize.x +3);
574
575 } // end if
576
577 // draw the text
578 drawListBoxText( window, instData, x, y + 4 , width, height-4, TRUE );
579
580
581
582} // end W3DGadgetListBoxDraw

587void W3DGadgetListBoxImageDraw( GameWindow *window, WinInstanceData *instData )
588{
589 Int width, height, x, y;

591 ListboxData *list = (ListboxData *)window->winGetUserData();

594 DisplayString *title = instData->getTextDisplayString();

596 // get window position and size

598 window->winGetSize( &size.x, &size.y );

600 // save off width and height so we can change them

624 titleColor = window->winGetHiliteTextColor();

629 image = GadgetListBoxGetEnabledImage( window );
630 titleColor = window->winGetEnabledTextColor();
631 titleBorder = window->winGetEnabledTextBorderColor();

633
634 // draw the back image

636 {

640 start.y = y + instData->m_imageOffset.y;

645 end.x, end.y );
646
647 } // end if
648
649 // Draw the title

652

654 if( title->getFont() != window->winGetFont() )
655 title->setFont( window->winGetFont() );
656
657 // draw the text
658 title->draw( x + 1, y, titleColor, titleBorder );
659
660 y += TheWindowManager->winFontHeight( instData->getFont() );
661 height -= TheWindowManager->winFontHeight( instData->getFont() ) + 1;
662
663 } // end if
664
665 // draw the listbox text
666 drawListBoxText( window, instData, x, y+4, width, height-4, TRUE );
667
668
669
