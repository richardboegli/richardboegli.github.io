88void AssaultTransportAIUpdate::aiDoCommand(const AICommandParms* parms)
89{
90 //Inspect the command and reset everything when necessary.
91 if( parms->m_cmdSource != CMD_FROM_AI )
92 {
93 //Now the only time we care about anything is if we were ordered to attack something or attack move.
94 switch( parms->m_cmd )
95 {
96 case AICMD_ATTACKMOVE_TO_POSITION:
97 //Reset because we have been ordered to do something.
98 reset();
99 m_attackMoveGoalPos = parms->m_pos;
100 m_isAttackMove = TRUE;
101 break;
102 case AICMD_ATTACK_OBJECT:
103 //Reset because we have been ordered to do something.
104 reset();
105 //m_designatedTarget = parms->m_obj ? parms->m_obj->getID() : INVALID_ID;
106 m_isAttackObject = TRUE;
107 break;
108 case AICMD_IDLE:
109 m_designatedTarget = INVALID_ID;
110 //Order all outside members back inside!
111 retrieveMembers();
112 reset();
113 break;
114 default:
115 //Reset because we have been ordered to do something we're not handling.
116 reset();
117 break;
118 }
119 }
120
121 //Note, in both cases, the transport will fire a dummy DEPLOY weapon that will trigger the
122 //evacuation of the troops.
123 AIUpdateInterface::aiDoCommand( parms );
124}

150UpdateSleepTime AssaultTransportAIUpdate::update( void )
151{
152 Object *transport = getObject();
153 //const AssaultTransportAIUpdateModuleData *data = getAssaultTransportAIUpdateModuleData();
154
155 if( transport->isEffectivelyDead() )
156 {
157 giveFinalOrders();
158 return UPDATE_SLEEP_FOREVER;
159 }
160
161 //First removing dead members or members that have been ordered to do something outside of this AI.
162 if( m_currentMembers )
163 {
164 for( int i = 0; i < m_currentMembers; i++ )
165 {
166 Object *member = TheGameLogic->findObjectByID( m_memberIDs[ i ] );
167 AIUpdateInterface *ai = member ? member->getAI() : NULL;
168 if( !member || member->isEffectivelyDead() || ai->getLastCommandSource() != CMD_FROM_AI )
169 {
170 //Member is toast -- so remove him from our list!
171 if( m_currentMembers - 1 > i )
172 {
173 //Move the last slot to this slot to keep array contiguous.
174 m_memberIDs[ i ] = m_memberIDs[ m_currentMembers - 1 ];
175 m_memberHealing[ i ] = m_memberHealing[ m_currentMembers - 1 ];
176 m_newMember[ i ] = m_newMember[ m_currentMembers - 1 ];
177 }
178 else
179 {
180 //Just clean out last slot.
181 m_memberIDs[ i ] = INVALID_ID;
182 m_memberHealing[ i ] = FALSE;
183 m_newMember[ i ] = FALSE;
184 }
185 if( ai )
186 {
187 //Important! Members of our assault transport must be allowed to chase down designated enemies.
188 //Generally only player commands allow this, so this flag allows AI commands to do the same.
189 //We need to turn this off though, because this ex-member is no longer under transport control.
190 ai->setAllowedToChase( FALSE );
191 }
192 m_currentMembers--;
193 }
194 }
195 }
196
197 //Now add any potentially new members to the group.
198 ContainModuleInterface *contain = transport->getContain();
199 if( contain )
200 {
201 const ContainedItemsList *passengerList = contain->getContainedItemsList();
202 ContainedItemsList::const_iterator passengerIterator;
203 passengerIterator = passengerList->begin();
204 while( passengerIterator != passengerList->end() )
205 {
206 Object *passenger = *passengerIterator;
207 //Advance to the next iterator
208 passengerIterator++;
209
210 //Make sure it isn't in our list already.
211 Bool found = FALSE;
212 for( int i = 0; i < m_currentMembers; i++ )
213 {
214 if( passenger->getID() == m_memberIDs[ i ] )
215 {
216 //He is in the list... so skip him.
217 found = TRUE;
218 break;
219 }
220 }
221 if( found )
222 {
223 //Get next passenger.
224 continue;
225 }
226
227 //It's possible to add members manually -- but if we already have 10 members, then wait!
228 if( m_currentMembers < MAX_TRANSPORT_SLOTS )
229 {
230 //Not in list, so add him!
231 m_memberIDs[ m_currentMembers ] = passenger->getID();
232 if( passenger->getAI() )
233 {
234 //Important! Members of our assault transport must be allowed to chase down designated enemies.
235 //Generally only player commands allow this, so this flag allows AI commands to do the same.
236 passenger->getAI()->setAllowedToChase( TRUE );
237 }
238
239 //Check if the passenger is wounded below threshhold (if so make sure we heal him before ordering him to fight!)
240 if( isMemberWounded( passenger ) )
241 {
242 m_memberHealing[ m_currentMembers ] = TRUE;
243 }
244 if( m_newOccupantsAreNewMembers )
245 {
246 //New members won't eject out until a new attack order is issued.
247 m_newMember[ m_currentMembers ] = TRUE;
248 }
249
250 m_currentMembers++;
251 }
252 }
253 m_newOccupantsAreNewMembers = TRUE;
254 }
255
256 if( isAttackPointless() )
257 {
258 aiIdle( CMD_FROM_AI );
259 return UPDATE_SLEEP_NONE;
260 }
261
262 //Keep track of the average position of all combat units assigned to me.
263 Coord3D fighterCentroidPos;
264 UnsignedInt fightingMembers = 0;
265 fighterCentroidPos.zero();
266
267 //If we're already in the process, reacquire the designated target again... see if
268 //it's still alive.
269 Object *designatedTarget = TheGameLogic->findObjectByID( m_designatedTarget );
270 if( designatedTarget && designatedTarget->isEffectivelyDead() )
271 {
272 designatedTarget = NULL;
273 }
274 if( designatedTarget )
275 {
276 //Look for members not currently attacking this target.
277 for( int i = 0; i < m_currentMembers; i++ )
278 {
279 Object *member = TheGameLogic->findObjectByID( m_memberIDs[ i ] );
280 AIUpdateInterface *ai = member ? member->getAI() : NULL;
281
282 if( member && ai )
283 {
284 Bool contained = member->isContained();
285 Bool wounded = isMemberWounded( member );
286 if( contained && isMemberHealthy( member ) && !m_newMember[ i ] )
287 {
288 //This contained member is healthy so order him to exit to start fighting!
289 //New members are exempt!
290 ai->aiExit( transport, CMD_FROM_AI );
291 }
292 if( !contained )
293 {
294 if( wounded )
295 {
296 if( ai->getAIStateType() != AI_ENTER )
297 {
298 //Order wounded members back to get healed.
299 ai->aiEnter( transport, CMD_FROM_AI );
300 }
301 }
302 else
303 {
304 //Increment the number of fighters and their position.
305 fighterCentroidPos.add( member->getPosition() );
306 fightingMembers++;
307
308 if( !ai->isMoving() )
309 {
310 if( ai->getGoalObject() != designatedTarget )
311 {
312 //Okay, this dude is outside and waiting... order him to attack the designated target
313 ai->aiAttackObject( designatedTarget, NO_MAX_SHOTS_LIMIT, CMD_FROM_AI );
314 }
315 }
316 }
317 }
318 }
319 }
320 }
321 else
322 {
323 if( m_isAttackMove && getAIStateType() != AI_ATTACK_MOVE_TO )
324 {
325 //Continue to move towards the attackmove area.
326 aiAttackMoveToPosition( &m_attackMoveGoalPos, NO_MAX_SHOTS_LIMIT, CMD_FROM_AI );
327 }
328 else if( m_isAttackObject )
329 {
330 retrieveMembers();
331 }
332 }
333
334 /*

336 if( !m_framesRemaining )
337 {
338 if( !isMoving() && fightingMembers && designatedTarget )
339 {
340 m_framesRemaining = 45;

342 //Get centriod pos now that we know the number of fighting members.
343 Real scale = 1.0f / (Real)fightingMembers;
344 fighterCentroidPos.scale( scale );
345
346 Coord3D designatedTargetPos = *designatedTarget->getPosition();

348 //Calculate a vector from the target passed the fighters to be at a safe place

385//-------------------------------------------------------------------------------------------------
386Bool AssaultTransportAIUpdate::isAttackPointless() const
387{
388 //If all members are new members (thus can't attack), and the transport itself
389 //is still attacking, stop!
390 const Object *transport = getObject();
391 if( transport->testStatus( OBJECT_STATUS_IS_ATTACKING ) )
392 {
393 for( int i = 0; i < m_currentMembers; i++ )
394 {
395 if( !m_newMember[ i ] )
396 {
397 //We have a non-new member, so attack is valid.
398 return FALSE;
399 }
400 }
401
402 //We are trying to attack, but can't because all our members are new.
403 return TRUE;
404 }
405
406 //We aren't trying to attack, so everything is good.
407 return FALSE;
408}

412{
413 const AssaultTransportAIUpdateModuleData *data = getAssaultTransportAIUpdateModuleData();
414 BodyModuleInterface *body = member->getBodyModule();

441void AssaultTransportAIUpdate::retrieveMembers()
442{
443 //Order all outside members back inside!
444 for( int i = 0; i < m_currentMembers; i++ )
445 {
446 Object *member = TheGameLogic->findObjectByID( m_memberIDs[ i ] );
447 AIUpdateInterface *ai = member ? member->getAI() : NULL;
448 if( member && ai )
449 {
450 Bool contained = member->isContained();
451 if( !contained )
452 {
453 //This contained member is healthy so order him to exit to start fighting!
454 if (ai->getAIStateType() != AI_ENTER) {
455 ai->aiEnter( getObject(), CMD_FROM_AI );
456 }
457 }
458 }
459 }

462//-------------------------------------------------------------------------------------------------
463void AssaultTransportAIUpdate::giveFinalOrders()
464{
465 //All members have been ejected outside already -- transfer the original order to the troops
466 for( int i = 0; i < m_currentMembers; i++ )
467 {
468 Object *member = TheGameLogic->findObjectByID( m_memberIDs[ i ] );
469 AIUpdateInterface *ai = member ? member->getAI() : NULL;
470 if( member && ai )
471 {
472 Object *designatedTarget = TheGameLogic->findObjectByID( m_designatedTarget );
473
474 if( m_isAttackObject && designatedTarget )
475 {
476 ai->aiAttackObject( designatedTarget, NO_MAX_SHOTS_LIMIT, CMD_FROM_PLAYER );
477 }
478 else if( m_isAttackMove )
479 {
480 ai->aiAttackMoveToPosition( &m_attackMoveGoalPos, NO_MAX_SHOTS_LIMIT, CMD_FROM_PLAYER );
481 }
482
483 ai->setAllowedToChase( FALSE );
484 }
485 }
