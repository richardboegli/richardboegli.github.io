124static enum CallbackType
125{
126 CALLBACK_CONNECT,
127 CALLBACK_ERROR,
128 CALLBACK_RECVMESSAGE,
129 CALLBACK_RECVREQUEST,
130 CALLBACK_RECVSTATUS,
131 CALLBACK_MAX
132};
133
134void callbackWrapper( GPConnection *con, void *arg, void *param )
135{
136 CallbackType info = (CallbackType)(Int)param;
137 BuddyThreadClass *thread = MESSAGE_QUEUE->getThread() ? MESSAGE_QUEUE->getThread() : NULL /*(TheGameSpyBuddyMessageQueue)?TheGameSpyBuddyMessageQueue->getThread():NULL*/;
138 if (!thread)
139 return;
140
141 switch (info)
142 {
143 case CALLBACK_CONNECT:
144 thread->connectCallback( con, (GPConnectResponseArg *)arg );
145 break;
146 case CALLBACK_ERROR:
147 thread->errorCallback( con, (GPErrorArg *)arg );
148 break;
149 case CALLBACK_RECVMESSAGE:
150 thread->messageCallback( con, (GPRecvBuddyMessageArg *)arg );
151 break;
152 case CALLBACK_RECVREQUEST:
153 thread->requestCallback( con, (GPRecvBuddyRequestArg *)arg );
154 break;
155 case CALLBACK_RECVSTATUS:
156 thread->statusCallback( con, (GPRecvBuddyStatusArg *)arg );
157 break;
158 }
159}

220Bool GameSpyBuddyMessageQueue::getRequest( BuddyRequest& req )
221{
222 MutexClass::LockClass m(m_requestMutex, 0);
223 if (m.Failed())
224 return false;
225
226 if (m_requests.empty())
227 return false;
228 req = m_requests.front();
229 m_requests.pop();
230 return true;
231}

267void BuddyThreadClass::Thread_Function()
268{
269 try {
270 _set_se_translator( DumpExceptionInfo ); // Hook that allows stack trace.
271 GPConnection gpCon;
272 GPConnection *con = &gpCon;
273 gpInitialize( con, 0, 0, GP_PARTNERID_GAMESPY );
274 m_isConnected = m_isConnecting = false;
275
276 gpSetCallback( con, GP_ERROR, callbackWrapper, (void *)CALLBACK_ERROR );
277 gpSetCallback( con, GP_RECV_BUDDY_MESSAGE, callbackWrapper, (void *)CALLBACK_RECVMESSAGE );
278 gpSetCallback( con, GP_RECV_BUDDY_REQUEST, callbackWrapper, (void *)CALLBACK_RECVREQUEST );
279 gpSetCallback( con, GP_RECV_BUDDY_STATUS, callbackWrapper, (void *)CALLBACK_RECVSTATUS );
280
281 GPEnum lastStatus = GP_OFFLINE;
282 std::string lastStatusString;
283
284 BuddyRequest incomingRequest;
285 while ( running )
286 {
287 // deal with requests
288 if (TheGameSpyBuddyMessageQueue->getRequest(incomingRequest))
289 {
290 switch (incomingRequest.buddyRequestType)
291 {
292 case BuddyRequest::BUDDYREQUEST_LOGIN:
293 m_isConnecting = true;
294 m_nick = incomingRequest.arg.login.nick;
295 m_email = incomingRequest.arg.login.email;
296 m_pass = incomingRequest.arg.login.password;
297 m_isConnected = (gpConnect( con, incomingRequest.arg.login.nick, incomingRequest.arg.login.email,
298 incomingRequest.arg.login.password, (incomingRequest.arg.login.hasFirewall)?GP_FIREWALL:GP_NO_FIREWALL,
299 GP_BLOCKING, callbackWrapper, (void *)CALLBACK_CONNECT ) == GP_NO_ERROR);
300 m_isConnecting = false;
301 break;
302
303 case BuddyRequest::BUDDYREQUEST_RELOGIN:
304 m_isConnecting = true;
305 m_isConnected = (gpConnect( con, m_nick.c_str(), m_email.c_str(), m_pass.c_str(), GP_FIREWALL,
306 GP_BLOCKING, callbackWrapper, (void *)CALLBACK_CONNECT ) == GP_NO_ERROR);
307 m_isConnecting = false;
308 break;
309 case BuddyRequest::BUDDYREQUEST_DELETEACCT:
310 m_isdeleting = true;
311 gpDeleteProfile( con, NULL, NULL );
312 break;
313 case BuddyRequest::BUDDYREQUEST_LOGOUT:
314 m_isConnecting = m_isConnected = false;
315 gpDisconnect( con );
316 break;
317 case BuddyRequest::BUDDYREQUEST_MESSAGE:
318 {
319 std::string s = WideCharStringToMultiByte( incomingRequest.arg.message.text );
320 DEBUG_LOG(("Sending a buddy message to %d [%s]\n", incomingRequest.arg.message.recipient, s.c_str()));
321 gpSendBuddyMessage( con, incomingRequest.arg.message.recipient, s.c_str() );
322 }
323 break;
324 case BuddyRequest::BUDDYREQUEST_LOGINNEW:
325 {
326 m_isConnecting = true;
327 m_nick = incomingRequest.arg.login.nick;
328 m_email = incomingRequest.arg.login.email;
329 m_pass = incomingRequest.arg.login.password;
330 m_isNewAccount = TRUE;
331 // Nwer API adds uniquenick and cdkey which are empty strings here.
332 m_isConnected = (gpConnectNewUser( con, incomingRequest.arg.login.nick, "", incomingRequest.arg.login.email,
333 incomingRequest.arg.login.password, "", (incomingRequest.arg.login.hasFirewall)?GP_FIREWALL:GP_NO_FIREWALL,
334 GP_BLOCKING, callbackWrapper, (void *)CALLBACK_CONNECT ) == GP_NO_ERROR);
335 if (m_isNewAccount) // if we didn't re-login
336 {
337 gpSetInfoMask( con, GP_MASK_NONE ); // don't share info
338 }
339 m_isConnecting = false;
340 }
341 break;
342 case BuddyRequest::BUDDYREQUEST_ADDBUDDY:
343 {
344 std::string s = WideCharStringToMultiByte( incomingRequest.arg.addbuddy.text );
345 gpSendBuddyRequest( con, incomingRequest.arg.addbuddy.id, s.c_str() );
346 }
347 break;
348 case BuddyRequest::BUDDYREQUEST_DELBUDDY:
349 {
350 gpDeleteBuddy( con, incomingRequest.arg.profile.id );
351 }
352 break;
353 case BuddyRequest::BUDDYREQUEST_OKADD:
354 {
355 gpAuthBuddyRequest( con, incomingRequest.arg.profile.id );
356 }
357 break;
358 case BuddyRequest::BUDDYREQUEST_DENYADD:
359 {
360 gpDenyBuddyRequest( con, incomingRequest.arg.profile.id );
361 }
362 case BuddyRequest::BUDDYREQUEST_SETSTATUS:
363 {
364 //don't blast our 'Loading' status with 'Online'.
365 if (lastStatus == GP_PLAYING && lastStatusString == "Loading" && incomingRequest.arg.status.status == GP_ONLINE)
366 break;
367
368 DEBUG_LOG(("BUDDYREQUEST_SETSTATUS: status is now %d:%s/%s\n",
369 incomingRequest.arg.status.status, incomingRequest.arg.status.statusString, incomingRequest.arg.status.locationString));
370 gpSetStatus( con, incomingRequest.arg.status.status, incomingRequest.arg.status.statusString,
371 incomingRequest.arg.status.locationString );
372 lastStatus = incomingRequest.arg.status.status;
373 lastStatusString = incomingRequest.arg.status.statusString;
374 }
375 break;
376 }
377 }
378
379 // update the network
380 GPEnum isConnected = GP_CONNECTED;
381 GPResult res = GP_NO_ERROR;
382 res = gpIsConnected( con, &isConnected );
383 if ( isConnected == GP_CONNECTED && res == GP_NO_ERROR )
384 gpProcess( con );
385
386 // end our timeslice
387 Switch_Thread();
388 }
389
390 gpDestroy( con );
391 } catch ( ... ) {
392 DEBUG_CRASH(("Exception in buddy thread!"));
393 }
394}

396void BuddyThreadClass::errorCallback( GPConnection *con, GPErrorArg *arg )
397{
398 // log the error
399 DEBUG_LOG(("GPErrorCallback\n"));
400 m_lastErrorCode = arg->errorCode;
401
402 char errorCodeString[256];
403 char resultString[256];
404
405 #define RESULT(x) case x: strcpy(resultString, #x); break;
406 switch(arg->result)
407 {
408 RESULT(GP_NO_ERROR)
409 RESULT(GP_MEMORY_ERROR)
410 RESULT(GP_PARAMETER_ERROR)
411 RESULT(GP_NETWORK_ERROR)
412 RESULT(GP_SERVER_ERROR)
413 default:
414 strcpy(resultString, "Unknown result!");
415 }
416 #undef RESULT
417
418 #define ERRORCODE(x) case x: strcpy(errorCodeString, #x); break;
419 switch(arg->errorCode)
420 {
421 ERRORCODE(GP_GENERAL)
422 ERRORCODE(GP_PARSE)
423 ERRORCODE(GP_NOT_LOGGED_IN)
424 ERRORCODE(GP_BAD_SESSKEY)
425 ERRORCODE(GP_DATABASE)
426 ERRORCODE(GP_NETWORK)
427 ERRORCODE(GP_FORCED_DISCONNECT)
428 ERRORCODE(GP_CONNECTION_CLOSED)
429 ERRORCODE(GP_LOGIN)
430 ERRORCODE(GP_LOGIN_TIMEOUT)
431 ERRORCODE(GP_LOGIN_BAD_NICK)
432 ERRORCODE(GP_LOGIN_BAD_EMAIL)
433 ERRORCODE(GP_LOGIN_BAD_PASSWORD)
434 ERRORCODE(GP_LOGIN_BAD_PROFILE)
435 ERRORCODE(GP_LOGIN_PROFILE_DELETED)
436 ERRORCODE(GP_LOGIN_CONNECTION_FAILED)
437 ERRORCODE(GP_LOGIN_SERVER_AUTH_FAILED)
438 ERRORCODE(GP_NEWUSER)
439 ERRORCODE(GP_NEWUSER_BAD_NICK)
440 ERRORCODE(GP_NEWUSER_BAD_PASSWORD)
441 ERRORCODE(GP_UPDATEUI)
442 ERRORCODE(GP_UPDATEUI_BAD_EMAIL)
443 ERRORCODE(GP_NEWPROFILE)
444 ERRORCODE(GP_NEWPROFILE_BAD_NICK)
445 ERRORCODE(GP_NEWPROFILE_BAD_OLD_NICK)
446 ERRORCODE(GP_UPDATEPRO)
447 ERRORCODE(GP_UPDATEPRO_BAD_NICK)
448 ERRORCODE(GP_ADDBUDDY)
449 ERRORCODE(GP_ADDBUDDY_BAD_FROM)
450 ERRORCODE(GP_ADDBUDDY_BAD_NEW)
451 ERRORCODE(GP_ADDBUDDY_ALREADY_BUDDY)
452 ERRORCODE(GP_AUTHADD)
453 ERRORCODE(GP_AUTHADD_BAD_FROM)
454 ERRORCODE(GP_AUTHADD_BAD_SIG)
455 ERRORCODE(GP_STATUS)
456 ERRORCODE(GP_BM)
457 ERRORCODE(GP_BM_NOT_BUDDY)
458 ERRORCODE(GP_GETPROFILE)
459 ERRORCODE(GP_GETPROFILE_BAD_PROFILE)
460 ERRORCODE(GP_DELBUDDY)
461 ERRORCODE(GP_DELBUDDY_NOT_BUDDY)
462 ERRORCODE(GP_DELPROFILE)
463 ERRORCODE(GP_DELPROFILE_LAST_PROFILE)
464 ERRORCODE(GP_SEARCH)
465 ERRORCODE(GP_SEARCH_CONNECTION_FAILED)
466 default:
467 strcpy(errorCodeString, "Unknown error code!");
468 }
469 #undef ERRORCODE
470
471 if(arg->fatal)
472 {
473 DEBUG_LOG(( "-----------\n"));
474 DEBUG_LOG(( "GP FATAL ERROR\n"));
475 DEBUG_LOG(( "-----------\n"));
476 }
477 else
478 {
479 DEBUG_LOG(( "-----\n"));
480 DEBUG_LOG(( "GP ERROR\n"));
481 DEBUG_LOG(( "-----\n"));
482 }
483 DEBUG_LOG(( "RESULT: %s (%d)\n", resultString, arg->result));
484 DEBUG_LOG(( "ERROR CODE: %s (0x%X)\n", errorCodeString, arg->errorCode));
485 DEBUG_LOG(( "ERROR STRING: %s\n", arg->errorString));
486
487 if (arg->fatal == GP_FATAL)
488 {
489 BuddyResponse errorResponse;
490 errorResponse.buddyResponseType = BuddyResponse::BUDDYRESPONSE_DISCONNECT;
491 errorResponse.result = arg->result;
492 errorResponse.arg.error.errorCode = arg->errorCode;
493 errorResponse.arg.error.fatal = arg->fatal;
494 strncpy(errorResponse.arg.error.errorString, arg->errorString, MAX_BUDDY_CHAT_LEN);
495 errorResponse.arg.error.errorString[MAX_BUDDY_CHAT_LEN-1] = 0;
496 m_isConnecting = m_isConnected = false;
497 TheGameSpyBuddyMessageQueue->addResponse( errorResponse );
498 if (m_isdeleting)
499 {
500 PeerRequest req;
501 req.peerRequestType = PeerRequest::PEERREQUEST_LOGOUT;
502 TheGameSpyPeerMessageQueue->addRequest( req );
503 m_isdeleting = false;
504 }
505 }

531void BuddyThreadClass::connectCallback( GPConnection *con, GPConnectResponseArg *arg )
532{
533 BuddyResponse loginResponse;
534 if (arg->result == GP_NO_ERROR)
535 {
536 loginResponse.buddyResponseType = BuddyResponse::BUDDYRESPONSE_LOGIN;
537 loginResponse.result = arg->result;
538 loginResponse.profile = arg->profile;
539 TheGameSpyBuddyMessageQueue->addResponse( loginResponse );
540 m_profileID = arg->profile;
541
542 if (!TheGameSpyPeerMessageQueue->isConnected() && !TheGameSpyPeerMessageQueue->isConnecting())
543 {
544 DEBUG_LOG(("Buddy connect: trying chat connect\n"));
545 PeerRequest req;
546 req.peerRequestType = PeerRequest::PEERREQUEST_LOGIN;
547 req.nick = m_nick.c_str();
548 req.password = m_pass;
549 req.email = m_email;
550 req.login.profileID = arg->profile;
551 TheGameSpyPeerMessageQueue->addRequest(req);
552 }
553 }
554 else
555 {
556 if (!TheGameSpyPeerMessageQueue->isConnected() && !TheGameSpyPeerMessageQueue->isConnecting())
557 {
558 if (m_lastErrorCode == GP_NEWUSER_BAD_NICK)
559 {
560 m_isNewAccount = FALSE;
561 // they just hit 'create account' instead of 'log in'. Fix them.
562 DEBUG_LOG(("User Error: Create Account instead of Login. Fixing them...\n"));
563 BuddyRequest req;
564 req.buddyRequestType = BuddyRequest::BUDDYREQUEST_LOGIN;
565 strcpy(req.arg.login.nick, m_nick.c_str());
566 strcpy(req.arg.login.email, m_email.c_str());
567 strcpy(req.arg.login.password, m_pass.c_str());
568 req.arg.login.hasFirewall = true;
569 TheGameSpyBuddyMessageQueue->addRequest( req );
570 return;
571 }
572 DEBUG_LOG(("Buddy connect failed (%d/%d): posting a failed chat connect\n", arg->result, m_lastErrorCode));
573 PeerResponse resp;
574 resp.peerResponseType = PeerResponse::PEERRESPONSE_DISCONNECT;
575 resp.discon.reason = DISCONNECT_COULDNOTCONNECT;
576 switch (m_lastErrorCode)
577 {
578 case GP_LOGIN_TIMEOUT:
579 resp.discon.reason = DISCONNECT_GP_LOGIN_TIMEOUT;
580 break;
581 case GP_LOGIN_BAD_NICK:
582 resp.discon.reason = DISCONNECT_GP_LOGIN_BAD_NICK;
583 break;
584 case GP_LOGIN_BAD_EMAIL:
585 resp.discon.reason = DISCONNECT_GP_LOGIN_BAD_EMAIL;
586 break;
587 case GP_LOGIN_BAD_PASSWORD:
588 resp.discon.reason = DISCONNECT_GP_LOGIN_BAD_PASSWORD;
589 break;
590 case GP_LOGIN_BAD_PROFILE:
591 resp.discon.reason = DISCONNECT_GP_LOGIN_BAD_PROFILE;
592 break;
593 case GP_LOGIN_PROFILE_DELETED:
594 resp.discon.reason = DISCONNECT_GP_LOGIN_PROFILE_DELETED;
595 break;
596 case GP_LOGIN_CONNECTION_FAILED:
597 resp.discon.reason = DISCONNECT_GP_LOGIN_CONNECTION_FAILED;
598 break;
599 case GP_LOGIN_SERVER_AUTH_FAILED:
600 resp.discon.reason = DISCONNECT_GP_LOGIN_SERVER_AUTH_FAILED;
601 break;
602 case GP_NEWUSER_BAD_NICK:
603 resp.discon.reason = DISCONNECT_GP_NEWUSER_BAD_NICK;
604 break;
605 case GP_NEWUSER_BAD_PASSWORD:
606 resp.discon.reason = DISCONNECT_GP_NEWUSER_BAD_PASSWORD;
607 break;
608 case GP_NEWPROFILE_BAD_NICK:
609 resp.discon.reason = DISCONNECT_GP_NEWPROFILE_BAD_NICK;
610 break;
611 case GP_NEWPROFILE_BAD_OLD_NICK:
612 resp.discon.reason = DISCONNECT_GP_NEWPROFILE_BAD_OLD_NICK;
613 break;
614 }
615 TheGameSpyPeerMessageQueue->addResponse(resp);
616 }
617 }

658void BuddyThreadClass::statusCallback( GPConnection *con, GPRecvBuddyStatusArg *arg )
659{
660 BuddyResponse response;
661
662 // get user's name
663 response.buddyResponseType = BuddyResponse::BUDDYRESPONSE_STATUS;
664 gpGetInfo( con, arg->profile, GP_CHECK_CACHE, GP_BLOCKING, (GPCallback)getInfoResponseForStatus, &response);
665
666 // get user's status
667 GPBuddyStatus status;
668 gpGetBuddyStatus( con, arg->index, &status );
669 strcpy(response.arg.status.location, status.locationString);
670 strcpy(response.arg.status.statusString, status.statusString);
671 response.arg.status.status = status.status;
672 DEBUG_LOG(("Got buddy status for %d(%s) - status %d\n", status.profile, response.arg.status.nick, status.status));
