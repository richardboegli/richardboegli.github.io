76void FireWeaponCollide::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
77{
78 if( other == NULL )
79 return; //Don't shoot the ground
80
81 Object *me = getObject();
82
83 // This will fire at you every frame, because multiple people could be colliding and we want
84 // to hurt them all. Another solution would be to keep a Map of other->objetIDs and
85 // delays for each individually. However, this solution here is so quick and simple that it
86 // warrants the "do it eventually if we need it" clause.
87 if( shouldFireWeapon() )
88 {
89 m_collideWeapon->loadAmmoNow( me );
90 m_collideWeapon->fireWeapon( me, other );
91 }
92}

94//-------------------------------------------------------------------------------------------------
95Bool FireWeaponCollide::shouldFireWeapon()
96{
97 const FireWeaponCollideModuleData *d = getFireWeaponCollideModuleData();
98
99 ObjectStatusMaskType status = getObject()->getStatusBits();
100
101 //We need all required status or else we fail
102 if( !status.testForAll( d->m_requiredStatus ) )
103 return FALSE;
104
105 //If we have any forbidden statii, then fail
106 if( status.testForAny( d->m_forbiddenStatus ) )
107 return FALSE;
108
109 if( m_everFired && d->m_fireOnce )
110 return FALSE;// can only fire once ever
111
112 return TRUE;
113}

131void FireWeaponCollide::xfer( Xfer *xfer )
132{
133
134 // version
135 XferVersion currentVersion = 1;

138
139 // extend base class
140 CollideModule::xfer( xfer );

144 xfer->xferBool( &collideWeaponPresent );
145 if( collideWeaponPresent )

147
148 DEBUG_ASSERTCRASH( m_collideWeapon != NULL,
149 ("FireWeaponCollide::xfer - m_collideWeapon present mismatch\n") );
150 xfer->xferSnapshot( m_collideWeapon );
151
152 } // end else
153 else
154 {
155
156 DEBUG_ASSERTCRASH( m_collideWeapon == NULL,
157 ("FireWeaponCollide::Xfer - m_collideWeapon missing mismatch\n" ));
158
159 } // end else
160
161 // ever fired
162 xfer->xferBool( &m_everFired );
163
164} // end xfer
