67/*
68** This class is the data that represents the marriage between a source art
69** palette and a destination palette/pixel-format. Facilities provied by this
70** class allow conversion from the source 8 bit pixel format to the correct
71** screen pixel format.
72**
73** Although this class can convert one pixel at a time through the conversion
74** function, the preferred way to convert pixels is through the translation
75** table provided. This table consists of 256 entries. Each entry is either
76** an 8 bit or 16 bit pixel in the correct screen-space format. Use the
77** Bytes_Per_Pixel() function to determine how to index into this translation
78** table.
79**
80** Expected use of this class would be to create separate objects of this class for
81** every source art palette. For an 8 bit display, an additional object will be
82** required for every additional palette set to the video DAC registers. It is
83** presumed that one general best-case palette will be used.
84*/
85class ConvertClass
86{
87 public:
88 ConvertClass(PaletteClass const & artpalette, PaletteClass const & screenpalette, Surface const & typicalsurface);
89 ~ConvertClass(void);
90
91 /*
92 ** Convert from source pixel to dest screen pixel.
93 */
94 int Convert_Pixel(int pixel) const {
95 if (BBP == 1) return(((unsigned char const *)Translator)[pixel]);
96 return(((unsigned short const *)Translator)[pixel]);
97 }
98
99 /*
100 ** Fetch a blitter object to use according to the draw flags
101 ** specified.
102 */
103 Blitter const * Blitter_From_Flags(ShapeFlags_Type flags) const;
104 RLEBlitter const * RLEBlitter_From_Flags(ShapeFlags_Type flags) const;
105
106 /*
107 ** This returns the bytes per pixel. Use this to determine how to index
108 ** through the translation table.
109 */
110 int Bytes_Per_Pixel(void) const {return(BBP);}
111
112 /*
113 ** Fetches the translation table. Sometimes the provided blitter objects
114 ** won't suffice and manual access to the translation process is necessary.
115 */
116 void const * Get_Translate_Table(void) const {return(Translator);}
117
118 /*
119 ** Sets the dynamic remap table so that the remapping blitters will use
120 ** it without having to recreate the blitter objects.
121 */
122 void Set_Remap(unsigned char const * remap) {RemapTable = remap;}
123
124 protected:
125 /*
126 ** Bytes per pixel in screen format.
127 */
128 int BBP;
129
130 /*
131 ** These are the blitter objects used to handle all the draw styles that this
132 ** drawing dispatcher implements.
133 */
134 Blitter * PlainBlitter; // No transparency (rarely used).
135 Blitter * TransBlitter; // Skips transparent pixels.
136 Blitter * ShadowBlitter; // Shadowizes the destination pixels.
137 Blitter * RemapBlitter; // Remaps source pixels then draws with transparency.
138 Blitter * Translucent1Blitter; // 25% translucent.
139 Blitter * Translucent2Blitter; // 50% translucent.
140 Blitter * Translucent3Blitter; // 75% translucent.
141
142 /*
143 ** These are the RLE aware blitters to handle all drawing styles that may
144 ** be used by RLE compressed images.
145 */
146 RLEBlitter * RLETransBlitter; // Skips transparent pixels.
147 RLEBlitter * RLEShadowBlitter; // Shadowizes the destination pixels.
148 RLEBlitter * RLERemapBlitter; // Remaps source pixels then draws with transparency.
149 RLEBlitter * RLETranslucent1Blitter; // 25% translucent.
150 RLEBlitter * RLETranslucent2Blitter; // 50% translucent.
151 RLEBlitter * RLETranslucent3Blitter; // 75% translucent.
152
153 /*
154 ** This is a translation table pointer. All source artwork is in 8 bit format.
155 ** This table will translate this source pixel into a screen dependant pixel
156 ** format datum.
157 */
158 void * Translator;
159
160 /*
161 ** This will shade an 8 bit pixel to about 1/2 intensity.
162 */
163 unsigned char * ShadowTable;
164
165 /*
166 ** Remap table pointer used for blits that require remapping. This value
167 ** will change according to the draw parameter. The blitting routines keep track
168 ** of this member object and use it to determine the remap table to use.
169 */
170 mutable unsigned char const * RemapTable;
171};
