115void CrateCollide::onCollide( Object *other, const Coord3D *, const Coord3D * )
116{
117 const CrateCollideModuleData *modData = getCrateCollideModuleData();
118 // If the crate can be picked up, perform the game logic and destroy the crate.
119 if( isValidToExecute( other ) )
120 {
121 if( executeCrateBehavior( other ) )
122 {
123 if( modData->m_executeFX != NULL )
124 {
125 // Note: We pass in other here, because the crate is owned by the neutral player, and
126 // we want to do things that only the other person can see.
127 FXList::doFXObj( modData->m_executeFX, other );
128 }
129
130 TheGameLogic->destroyObject( getObject() );
131 }
132
133 // play animation in the world at this spot if there is one
134 if( TheAnim2DCollection && modData->m_executionAnimationTemplate.isEmpty() == FALSE && TheGameLogic->getDrawIconUI() )
135 {
136 Anim2DTemplate *animTemplate = TheAnim2DCollection->findTemplate( modData->m_executionAnimationTemplate );
137
138 TheInGameUI->addWorldAnimation( animTemplate,
139 getObject()->getPosition(),
140 WORLD_ANIM_FADE_ON_EXPIRE,
141 modData->m_executeAnimationDisplayTimeInSeconds,
142 modData->m_executeAnimationZRisePerSecond );
143
144 }
145
146 }
147
148}

151Bool CrateCollide::isValidToExecute( const Object *other ) const
152{
153 //The ground never picks up a crate
154 if( other == NULL )
155 return FALSE;
156
157 //Nothing Neutral can pick up any type of crate
158 if( other->isNeutralControlled() )
159 return FALSE;
160
161 const CrateCollideModuleData* md = getCrateCollideModuleData();
162 Bool validBuildingAttempt = md->m_isBuildingPickup && other->isKindOf( KINDOF_STRUCTURE );
163
164 // Must be a "Unit" type thing. Real Game Object, not just Object
165 if( other->getAIUpdateInterface() == NULL && !validBuildingAttempt )// Building exception flag for Drop Zone
166 return FALSE;
167
168 // must match our kindof flags (if any)
169 if (md && !other->isKindOfMulti(md->m_kindof, md->m_kindofnot))
170 return FALSE;
171
172 if( other->isEffectivelyDead() )
173 return FALSE;
174
175 // crates cannot be claimed while in the air, except by buildings
176 if( getObject()->isAboveTerrain() && !validBuildingAttempt )
177 return FALSE;
178
179 if( md->m_isForbidOwnerPlayer && (getObject()->getControllingPlayer() == other->getControllingPlayer()) )
180 return FALSE; // Design has decreed this to not be picked up by the dead guy's team.
181
182 if( md->m_isHumanOnlyPickup && other->getControllingPlayer() && (other->getControllingPlayer()->getPlayerType() != PLAYER_HUMAN) )
183 return FALSE; // Human only mission crate
184
185 if( (md->m_pickupScience != SCIENCE_INVALID) && other->getControllingPlayer() && !other->getControllingPlayer()->hasScience(md->m_pickupScience) )
186 return FALSE; // Science required to pick this up
187
188 if( other->isKindOf( KINDOF_PARACHUTE ) )
189 return FALSE;
190
191 return TRUE;
192}

198void CrateCollide::doSabotageFeedbackFX( const Object *other, SabotageVictimType type )
199{
200
201 if ( ! getObject() )
202 return;
203 if ( ! other )
204 return;
205
206 AudioEventRTS soundToPlay;
207 switch ( type )
208 {
209 case CrateCollide::SAB_VICTIM_FAKE_BUILDING:
210 {
211 return; // THIS NEEDS NO ADD'L FEEDBACK
212 }
213 case CrateCollide::SAB_VICTIM_COMMAND_CENTER:
214 case CrateCollide::SAB_VICTIM_SUPERWEAPON:
215 {
216 soundToPlay = TheAudio->getMiscAudio()->m_sabotageResetTimerBuilding;
217 break;
218 }
219 case CrateCollide::SAB_VICTIM_DROP_ZONE:
220 case CrateCollide::SAB_VICTIM_SUPPLY_CENTER:
221 {
222 soundToPlay = TheAudio->getMiscAudio()->m_moneyWithdrawSound;
223 break;
224 }
225 case CrateCollide::SAB_VICTIM_INTERNET_CENTER:
226 case CrateCollide::SAB_VICTIM_MILITARY_FACTORY:
227 case CrateCollide::SAB_VICTIM_POWER_PLANT:
228 default:
229 {
230 soundToPlay = TheAudio->getMiscAudio()->m_sabotageShutDownBuilding;
231 break;
232 }
233 }
234
235 soundToPlay.setPosition( other->getPosition() );
236 TheAudio->addAudioEvent( &soundToPlay );
237
238 Drawable *draw = other->getDrawable();
239 if ( draw )
240 draw->flashAsSelected();
241
242}
