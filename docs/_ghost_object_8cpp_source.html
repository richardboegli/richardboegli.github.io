75void GhostObject::xfer( Xfer *xfer )
76{
77
78 // version
79 XferVersion currentVersion = 1;
80 XferVersion version = currentVersion;
81 xfer->xferVersion( &version, currentVersion );
82
83 // parent object
84 ObjectID parentObjectID = INVALID_ID;
85 if( m_parentObject )
86 parentObjectID = m_parentObject->getID();
87 xfer->xferObjectID( &parentObjectID );
88 if( xfer->getXferMode() == XFER_LOAD )
89 {
90
91 // tie up parent object pointer
92 m_parentObject = TheGameLogic->findObjectByID( parentObjectID );
93
94 // sanity
95 if( parentObjectID != INVALID_ID && m_parentObject == NULL )
96 {
97
98 DEBUG_CRASH(( "GhostObject::xfer - Unable to connect m_parentObject\n" ));
99 throw INI_INVALID_DATA;
100
101 } // end if
102
103 } // end if
104
105 // parent geometry type
106 xfer->xferUser( &m_parentGeometryType, sizeof( GeometryType ) );
107
108 // parent geometry is small
109 xfer->xferBool( &m_parentGeometryIsSmall );
110
111 // parent geometry major radius
112 xfer->xferReal( &m_parentGeometryMajorRadius );
113
114 // parent geometry minor radius
115 xfer->xferReal( &m_parentGeometryminorRadius );
116
117 // parent angle
118 xfer->xferReal( &m_parentAngle );
119
120 // parent position
121 xfer->xferCoord3D( &m_parentPosition );
122
123 // partition data
125 // PartitionData *m_partitionData; ///< our PartitionData
126
127} // end xfer
