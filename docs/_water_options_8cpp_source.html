141void WaterOptions::OnChangeWaterEdit()
142{
143 PolygonTrigger *theTrigger = WaypointOptions::getSingleSelectedPolygon();
144
145 // get the combo box
146 CComboBox *pCombo = (CComboBox*)GetDlgItem(IDC_WATERNAME_EDIT);
147 if (pCombo) {
148 // get the text out of the combo. If it is user-typed, sel will be -1, otherwise it will be >=0
149 CString theText;
150 Int sel = pCombo->GetCurSel();
151 if (sel >= 0) {
152 pCombo->GetLBText(sel, theText);
153 } else {
154 pCombo->GetWindowText(theText);
155 }
156 AsciiString name((LPCTSTR)theText);
157
158 // check to see if the user-entered name is already in use.
159 Bool didMatch = false;
160
161 // check trigger area objects
162 PolygonTrigger *pTrig;
163 for (pTrig=PolygonTrigger::getFirstPolygonTrigger(); !didMatch && pTrig; pTrig = pTrig->getNext()) {
164 if (pTrig==theTrigger) continue; // don't check against yourself.
165 AsciiString trigName = pTrig->getTriggerName();
166 if (name == trigName) {
167 if (pTrig->isValid()) {
168 didMatch = true;
169 } else {
170 PolygonTrigger::removePolygonTrigger(pTrig);
171 }
172 break;
173 }
174 }
175
176 // if there's a match, throw up a messagebox, otherwise set the name
177 if (didMatch) {
178 ::AfxMessageBox("Name already in use");
179 } else {
180 if (theTrigger) {
181 theTrigger->setTriggerName(name);
182 }
183 }
184 }
185}

194void WaterOptions::OnMakeRiver()
195{
196 CButton *pButton = (CButton*)GetDlgItem(IDC_MAKE_RIVER);
197 Bool river = (pButton->GetCheck()==1);
198 PolygonTrigger *theTrigger = WaypointOptions::getSingleSelectedPolygon();
199 if (theTrigger) {
200 theTrigger->setRiver(river);
201 if (river) {
202 Int curPoint = PolygonTool::getSelectedPointNdx();
203 if (curPoint >= 0) {
204 Real endLen=0;
205 Int newPoint = curPoint;
206 if (curPoint>0) curPoint--;
207 if (curPoint>0) curPoint--;
208 Int i;
209 for (i=curPoint; i<theTrigger->getNumPoints()-1 && i<curPoint+4; i++) {
210 ICoord3D innerPt = *theTrigger->getPoint(i);
211 ICoord3D outerPt = *theTrigger->getPoint(i+1);
212 Real dx = innerPt.x-outerPt.x;
213 Real dy = innerPt.y-outerPt.y;
214 Real curLen = sqrt(dx*dx+dy*dy);
215 if ( curLen>endLen) {
216 newPoint = i;
217 endLen = curLen;
218 }
219 }
220 theTrigger->setRiverStart(newPoint);
221
222 // Now find the other end.
223// Real sourceWidth = endLen;
224
225 endLen=0;
226 Int endPoint = 0;
227 for (i=0; i<theTrigger->getNumPoints()-1; i++) {
228 if (i>=newPoint-1 && i<=newPoint+1) continue;
229 ICoord3D innerPt = *theTrigger->getPoint(i);
230 ICoord3D outerPt = *theTrigger->getPoint(i+1);
231 Real dx = innerPt.x-outerPt.x;
232 Real dy = innerPt.y-outerPt.y;
233 Real curLen = sqrt(dx*dx+dy*dy);
234 if ( curLen>endLen) {
235 endPoint = i;
236 endLen = curLen;
237 }
238 }
239 Int pointsOut = endPoint - newPoint;
240 Int pointsIn = newPoint - endPoint;
241 if (pointsOut<0) pointsOut += theTrigger->getNumPoints();
242 if (pointsIn<0) pointsIn += theTrigger->getNumPoints();
243 Int delta = pointsIn-pointsOut;
244 if (delta<0) delta = -delta;
245 if (delta>1) {
246 PolygonTrigger *pNew;
247 if (pointsOut<pointsIn) {
248 pNew = adjustCount(theTrigger, newPoint+1, endPoint, pointsIn-1);
249 theTrigger->setRiverStart(pointsIn);
250 } else {
251 pNew = adjustCount(theTrigger, endPoint+1, newPoint, pointsOut-1);
252 theTrigger->setRiverStart(0);
253 }
254 while(theTrigger->getNumPoints()) theTrigger->deletePoint(theTrigger->getNumPoints()-1);
255 for (i=0; i<pNew->getNumPoints(); i++) {
256 theTrigger->addPoint(*pNew->getPoint(i));
257 }
258 pNew->deleteInstance();
259 }
260 }
261 }
262 }

266PolygonTrigger * WaterOptions::adjustCount(PolygonTrigger *trigger, Int firstPt, Int lastPt, Int desiredPointCount)
267{
268 PolygonTrigger *pNew = newInstance(PolygonTrigger)(trigger->getNumPoints());
269// Real endLen=0;
270 Real totalLen=0;
271 Real curSpacingLen = 10;
272 Int curPoint = lastPt;
273 ICoord3D pt;
274 while (curPoint != firstPt) {
275 pt = *trigger->getPoint(curPoint);
276 pNew->addPoint(pt);
277 curPoint++;
278 if (curPoint>=trigger->getNumPoints()) {
279 curPoint = 0;
280 }
281 }
282
283 curPoint = firstPt;
284 while (curPoint != lastPt) {
285 Int nextPoint = curPoint;
286 nextPoint++;
287 if (nextPoint>=trigger->getNumPoints()) {
288 nextPoint = 0;
289 }
290 ICoord3D curPt = *trigger->getPoint(curPoint);
291 ICoord3D nextPt = *trigger->getPoint(nextPoint);
292 Real dx = nextPt.x-curPt.x;
293 Real dy = nextPt.y-curPt.y;
294 Real curLen = sqrt(dx*dx+dy*dy);
295 totalLen += curLen;
296 curPoint = nextPoint;
297 }
298 Real spacing = totalLen/(desiredPointCount-1);
299
300 Bool didCurPoint = true;
301
302 curPoint = firstPt;
303 pt = *trigger->getPoint(curPoint);
304 pNew->addPoint(pt);
305 while (curPoint != lastPt) {
306 Int nextPoint = curPoint;
307 nextPoint++;
308 if (nextPoint>=trigger->getNumPoints()) {
309 nextPoint = 0;
310 }
311 ICoord3D curPt = *trigger->getPoint(curPoint);
312 ICoord3D nextPt = *trigger->getPoint(nextPoint);
313 Real dx = nextPt.x-curPt.x;
314 Real dy = nextPt.y-curPt.y;
315 Real curLen = sqrt(dx*dx+dy*dy);
316 if (curLen > 4*MAP_XY_FACTOR && curLen>2*spacing) {
317 if (!didCurPoint) pNew->addPoint(curPt);
318 else pNew->setPoint(curPt, pNew->getNumPoints()-1);
319 pNew->addPoint(nextPt);
320 didCurPoint = true;
321 curSpacingLen = spacing;
322 } else if (curSpacingLen>curLen) {
323 curSpacingLen -= curLen;
324 } else {
325 while (curLen >= curSpacingLen) {
326 // cur len > curSpacingLen.
327 Real factor = curSpacingLen/curLen;
328 curPt.x += dx*factor;
329 curPt.y += dy*factor;
330 pNew->addPoint(curPt);
331 didCurPoint = false;
332 dx = nextPt.x-curPt.x;
333 dy = nextPt.y-curPt.y;
334 curLen -= curSpacingLen;
335 curSpacingLen = spacing;
336 }
337 curSpacingLen -= curLen;
338 if ((curLen)<MAP_XY_FACTOR/2) {
339 didCurPoint = true;
340 }
341 }
342 curPoint = nextPoint;
343 }
344 return pNew;
345}

408void WaterOptions::startUpdateHeight(void)
409{
410 PolygonTrigger *theTrigger = WaypointOptions::getSingleSelectedPolygon();
411 if (!theTrigger) {
412 REF_PTR_RELEASE(m_moveUndoable);
413 return;
414 }
415 if (!theTrigger->isWaterArea()) {
416 REF_PTR_RELEASE(m_moveUndoable);
417 return;
418 }
419 if (m_moveUndoable && theTrigger == m_moveUndoable->getTrigger()) {
420 return;
421 }
422 m_originalHeight = theTrigger->getPoint(0)->z;
423 Int i;
424 for (i=0; i<theTrigger->getNumPoints(); i++) {
425 ICoord3D loc = *theTrigger->getPoint(i);
426 loc.z = m_originalHeight;
427 theTrigger->setPoint(loc, i);
428 }
429 m_moveUndoable = new MovePolygonUndoable(theTrigger);
430 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
431 pDoc->AddAndDoUndoable(m_moveUndoable);
432}

435void WaterOptions::updateHeight(void)
436{
437 PolygonTrigger *theTrigger = WaypointOptions::getSingleSelectedPolygon();
438 if (!theTrigger || !m_moveUndoable) {
439 REF_PTR_RELEASE(m_moveUndoable); // belongs to pDoc now.
440 return;
441 }
442 if (!theTrigger->isWaterArea()) {
443 REF_PTR_RELEASE(m_moveUndoable); // belongs to pDoc now.
444 return;
445 }
446 if (theTrigger != m_moveUndoable->getTrigger()) {
447 REF_PTR_RELEASE(m_moveUndoable); // belongs to pDoc now.
448 return;
449 }
450 ICoord3D iLoc;
451 Int dz = m_waterHeight - m_originalHeight;
452 iLoc.x = 0;
453 iLoc.y = 0;
454 iLoc.z = dz;
455 m_moveUndoable->SetOffset(iLoc);
456 WbView3d *pView = CWorldBuilderDoc::GetActive3DView();
457 pView->Invalidate();
458}
