71static LRESULT CALLBACK pushButtonPropertiesCallback( HWND hWndDialog,
72 UINT message,
73 WPARAM wParam,
74 LPARAM lParam )
75{
76 Int returnCode;
77
78 //
79 // handle any common messages between all property dialogs cause they
80 // are designed to have controls doing the same functionality
81 // and names
82 //
83 if( HandleCommonDialogMessages( hWndDialog, message,
84 wParam, lParam, &returnCode ) == TRUE )
85 return returnCode;
86
87 switch( message )
88 {
89
90 // ------------------------------------------------------------------------
91 case WM_COMMAND:
92 {
93// Int notifyCode = HIWORD( wParam ); // notification code
94 Int controlID = LOWORD( wParam ); // control ID
95// HWND hWndControl = (HWND)lParam; // control window handle
96
97 switch( controlID )
98 {
99
100 // --------------------------------------------------------------------
101 case IDOK:
102 {
103 GameWindow *window = TheEditor->getPropertyTarget();
104
105 // sanity
106 if( window )
107 {
108 ImageAndColorInfo *info;
109
110 // save the common properties
111 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
112 break;
113
114 // save the image and color data
115 // ----------------------------------------------------------------
116 info = GetStateInfo( BUTTON_ENABLED );
117 GadgetButtonSetEnabledImage( window, info->image );
118 GadgetButtonSetEnabledColor( window, info->color );
119 GadgetButtonSetEnabledBorderColor( window, info->borderColor );
120
121 info = GetStateInfo( BUTTON_ENABLED_PUSHED );
122 GadgetButtonSetEnabledSelectedImage( window, info->image );
123 GadgetButtonSetEnabledSelectedColor( window, info->color );
124 GadgetButtonSetEnabledSelectedBorderColor( window, info->borderColor );
125
126 // ----------------------------------------------------------------
127 info = GetStateInfo( BUTTON_DISABLED );
128 GadgetButtonSetDisabledImage( window, info->image );
129 GadgetButtonSetDisabledColor( window, info->color );
130 GadgetButtonSetDisabledBorderColor( window, info->borderColor );
131
132 info = GetStateInfo( BUTTON_DISABLED_PUSHED );
133 GadgetButtonSetDisabledSelectedImage( window, info->image );
134 GadgetButtonSetDisabledSelectedColor( window, info->color );
135 GadgetButtonSetDisabledSelectedBorderColor( window, info->borderColor );
136
137 // ----------------------------------------------------------------
138 info = GetStateInfo( BUTTON_HILITE );
139 GadgetButtonSetHiliteImage( window, info->image );
140 GadgetButtonSetHiliteColor( window, info->color );
141 GadgetButtonSetHiliteBorderColor( window, info->borderColor );
142
143 info = GetStateInfo( BUTTON_HILITE_PUSHED );
144 GadgetButtonSetHiliteSelectedImage( window, info->image );
145 GadgetButtonSetHiliteSelectedColor( window, info->color );
146 GadgetButtonSetHiliteSelectedBorderColor( window, info->borderColor );
147
148 UnsignedInt bit;
149 bit = WIN_STATUS_RIGHT_CLICK;
150 window->winClearStatus( bit );
151 if( IsDlgButtonChecked( hWndDialog, CHECK_RIGHT_CLICK ) )
152 window->winSetStatus( bit );
153
154 } // end if
155
156 DestroyWindow( hWndDialog );
157 break;
158
159 } // end OK
160
161 // --------------------------------------------------------------------
162 case IDCANCEL:
163 {
164
165 DestroyWindow( hWndDialog );
166 break;
167
168 } // end cancel
169
170 } // end switch( LOWORD( wParam ) )
171
172 return 0;
173
174 } // end of WM_COMMAND
175
176 // ------------------------------------------------------------------------
177 case WM_CLOSE:
178 {
179
180 DestroyWindow( hWndDialog );
181 return 0;
182
183 } // end close
184
185 // ------------------------------------------------------------------------
186 default:
187 return 0;
188
189 } // end of switch
190
191} // end pushButtonPropertiesCallback

197// InitPushButtonPropertiesDialog =============================================
199//=============================================================================
200HWND InitPushButtonPropertiesDialog( GameWindow *window )
201{
202 HWND dialog;
203
204 // create the dialog box
205 dialog = CreateDialog( TheEditor->getInstance(),
206 (LPCTSTR)PUSH_BUTTON_PROPERTIES_DIALOG,
207 TheEditor->getWindowHandle(),
208 (DLGPROC)pushButtonPropertiesCallback );
209 if( dialog == NULL )
210 return NULL;
211
212 // do the common initialization
213 CommonDialogInitialize( window, dialog );
214
215 //
216 // store in the image and color table the values for this putton
217 //
218 const Image *image;
219 Color color, borderColor;
220
221 image = GadgetButtonGetEnabledImage( window );
222 color = GadgetButtonGetEnabledColor( window );
223 borderColor = GadgetButtonGetEnabledBorderColor( window );
224 StoreImageAndColor( BUTTON_ENABLED, image, color, borderColor );
225
226 image = GadgetButtonGetEnabledSelectedImage( window );
227 color = GadgetButtonGetEnabledSelectedColor( window );
228 borderColor = GadgetButtonGetEnabledSelectedBorderColor( window );
229 StoreImageAndColor( BUTTON_ENABLED_PUSHED, image, color, borderColor );
230
231 image = GadgetButtonGetDisabledImage( window );
232 color = GadgetButtonGetDisabledColor( window );
233 borderColor = GadgetButtonGetDisabledBorderColor( window );
234 StoreImageAndColor( BUTTON_DISABLED, image, color, borderColor );
235
236 image = GadgetButtonGetDisabledSelectedImage( window );
237 color = GadgetButtonGetDisabledSelectedColor( window );
238 borderColor = GadgetButtonGetDisabledSelectedBorderColor( window );
239 StoreImageAndColor( BUTTON_DISABLED_PUSHED, image, color, borderColor );
240
241 image = GadgetButtonGetHiliteImage( window );
242 color = GadgetButtonGetHiliteColor( window );
243 borderColor = GadgetButtonGetHiliteBorderColor( window );
244 StoreImageAndColor( BUTTON_HILITE, image, color, borderColor );
245
246 image = GadgetButtonGetHiliteSelectedImage( window );
247 color = GadgetButtonGetHiliteSelectedColor( window );
248 borderColor = GadgetButtonGetHiliteSelectedBorderColor( window );
249 StoreImageAndColor( BUTTON_HILITE_PUSHED, image, color, borderColor );
250
251 // select the button enabled state for display
252 SwitchToState( BUTTON_ENABLED, dialog );
253
254 //
255 // initialize the dialog with values from the window
256 //
257
258 if( BitTest( window->winGetStatus(), WIN_STATUS_RIGHT_CLICK ) )
259 CheckDlgButton( dialog, CHECK_RIGHT_CLICK, BST_CHECKED );
260
261 return dialog;
262
263} // end InitPushButtonPropertiesDialog
