99void Anim2DTemplate::parseNumImages( INI *ini, void *instance, void *store, const void *userData )
100{
101
102 // parse the integer data from the file
103 UnsignedInt numFrames;
104 ini->parseUnsignedInt( ini, instance, &numFrames, userData );
105
106 // get the template we are to store into
107 Anim2DTemplate *animTemplate = (Anim2DTemplate *)instance;
108
109 // animations must have a minimum # of frames
110 Int minimumFrames = 1;
111 if( numFrames < minimumFrames )
112 {
113
114 DEBUG_CRASH(( "Anim2DTemplate::parseNumImages - Invalid animation '%s', animations must have '%d' or more frames defined\n",
115 animTemplate->getName().str(), minimumFrames ));
116 throw INI_INVALID_DATA;
117
118 } // end if
119
120 // allocate the image array
121 animTemplate->allocateImages( (UnsignedShort)numFrames );
122
123} // end parseNumImages

146void Anim2DTemplate::parseImage( INI *ini, void *instance, void *store, const void *userData )
147{
148
149 // parse the image name from the file and store as an image pointer
150 const Image *image;
151 ini->parseMappedImage( ini, instance, &image, userData );
152
153 // sanity
154 if( image == NULL )
155 {
156
157 //We don't care if we're in the builder
158 //DEBUG_CRASH(( "Anim2DTemplate::parseImage - Image not found\n" ));
159 //throw INI_INVALID_DATA;
160
161 } // end if
162
163 //
164 // assign the image to the animation template list of images ... note since we've pre-allocated
165 // the array of images and the index an image is loaded into depends on its order specified
166 // in INI, we need to get the number of images currently loaded into this animation template
167 // so that we can put it at the next free image spot ... and then tell the animation
168 // template we've loaded one more
169 //
170 Anim2DTemplate *animTemplate = (Anim2DTemplate *)instance;
171 animTemplate->storeImage( image );
172
173} // end parseImage

175// ------------------------------------------------------------------------------------------------
182// ------------------------------------------------------------------------------------------------
183/*static*/ void Anim2DTemplate::parseImageSequence( INI *ini, void *instance,
184 void *store, const void *userData )
185{
186
187 // get the animation template
188 Anim2DTemplate *animTemplate = (Anim2DTemplate *)instance;
189
190 //
191 // before we can read, allocate, and find all the images for the sequence ... we must
192 // know how many total images are in this animation. Verify that now
193 //
194 if( animTemplate->getNumFrames() == NUM_FRAMES_INVALID )
195 {
196
197 DEBUG_CRASH(( "Anim2DTemplate::parseImageSequence - You must specify the number of animation frames for animation '%s' *BEFORE* specifying the image sequence name\n",
198 animTemplate->getName().str() ));
199 throw INI_INVALID_DATA;
200
201 } // end if
202
203 //
204 // the image storage has already been allocated, all we have to do now is count from
205 // 0 to numImages - 1 and add those images to the template
206 //
207 AsciiString imageBaseName = ini->getNextAsciiString();
208 AsciiString imageName;
209 const Image *image;
210 for( Int i = 0; i < animTemplate->getNumFrames(); ++i )
211 {
212
213 // construct this name
214 imageName.format( "%s%03d", imageBaseName.str(), i );
215
216 // search for this image
217 image = TheMappedImageCollection->findImageByName( imageName );
218
219 // sanity
220 if( image == NULL )
221 {
222
223 DEBUG_CRASH(( "Anim2DTemplate::parseImageSequence - Image '%s' not found for animation '%s'. Check the number of images specified in INI and also make sure all the actual images exist.\n",
224 imageName.str(), animTemplate->getName().str() ));
225 throw INI_INVALID_DATA;
226
227 } // end if
228
229 // store the image in the next free frame
230 animTemplate->storeImage( image );
231
232 } // end if
233
234} // end parseImageSequence

236// ------------------------------------------------------------------------------------------------
238// ------------------------------------------------------------------------------------------------
239void Anim2DTemplate::storeImage( const Image *image )
240{
241
242 // sanity
243 if( image == NULL )
244 return;
245
246 // search through the image list and store at the next free spot
247 for( Int i = 0; i < m_numFrames; ++i )
248 {
249
250 if( m_images[ i ] == NULL )
251 {
252
253 m_images[ i ] = image;
254 return;
255
256 } // end if
257
258 } // end for i
259
260 // if we got here we tried to store an image in an array that was too small
261 DEBUG_CRASH(( "Anim2DTemplate::storeImage - Unable to store image '%s' into animation '%s' because the animation is setup to only support '%d' image frames\n",
262 image->getName().str(), getName().str(), m_numFrames ));
263 throw INI_INVALID_DATA;
264
265} // end storeImage

277

281

285
286 } // end if
287 else

289
290 // return the image frame
291 return m_images[ frameNumber ];
292
293 } // end else

304{
305
306 // sanity
307 DEBUG_ASSERTCRASH( animTemplate != NULL, ("Anim2D::Anim2D - NULL template\n") );
308
309 //Added By Sadullah Nader
310 //Initialization
311
312 m_currentFrame = 0;
313
314 //
315
316 // set the template
317 m_template = animTemplate;
318
319 // initialize members
320 m_status = ANIM_2D_STATUS_NONE;
321 m_alpha = 1.0f;
322
323 // set the initial frame for the animation based on the type of animation mode or randomize

325 randomizeCurrentFrame();

338
339 m_lastUpdateFrame = 0;
340
341 // if a system is present, register ourselves with that system
342 m_collectionSystem = collectionSystem;
343 if( m_collectionSystem )
344 m_collectionSystem->registerAnimation( this );

401// ------------------------------------------------------------------------------------------------
403// ------------------------------------------------------------------------------------------------
404void Anim2D::reset( void )
405{
406
407 // sanity
408 DEBUG_ASSERTCRASH( m_template != NULL, ("Anim2D::reset - No template for animation\n") );
409
410 switch( m_template->getAnimMode() )
411 {
412
413 // --------------------------------------------------------------------------------------------
414 case ANIM_2D_ONCE:
415 case ANIM_2D_LOOP:
416 case ANIM_2D_PING_PONG:
417 setCurrentFrame( m_minFrame );
418 break;
419
420 // --------------------------------------------------------------------------------------------
421 case ANIM_2D_ONCE_BACKWARDS:
422 case ANIM_2D_LOOP_BACKWARDS:
423 case ANIM_2D_PING_PONG_BACKWARDS:
424 setCurrentFrame( m_maxFrame );
425 break;
426
427 // --------------------------------------------------------------------------------------------
428 default:
429 DEBUG_CRASH(( "Anim2D::reset - Unknown animation mode '%d' for '%s'\n",
430 m_template->getAnimMode(), m_template->getName().str() ));
431 break;
432
433 } // end switch, animation mode
434
435} // end reset

437// ------------------------------------------------------------------------------------------------
440// ------------------------------------------------------------------------------------------------
441void Anim2D::tryNextFrame( void )
442{
443
444 // sanity
445 DEBUG_ASSERTCRASH( TheGameLogic != NULL,
446 ("Anim2D::tryNextFrame - TheGameLogic must exist to use animation instances (%s)\n",
447 m_template->getName().str()) );
448
449 // how many frames have passed since our last update
450 if( TheGameLogic->getFrame() - m_lastUpdateFrame >= m_framesBetweenUpdates )
451 {
452
453 switch( m_template->getAnimMode() )
454 {
455
456 // ------------------------------------------------------------------------------------------
457 case ANIM_2D_ONCE:
458 {
459
460 if( m_currentFrame < m_maxFrame )
461 setCurrentFrame( m_currentFrame + 1 );
462 else
463 setStatus( ANIM_2D_STATUS_COMPLETE );
464 break;
465
466 } // end once
467
468 // -------------------------------------------------------------------------------------------
469 case ANIM_2D_ONCE_BACKWARDS:
470 {
471
472 if( m_currentFrame > m_minFrame )
473 setCurrentFrame( m_currentFrame - 1 );
474 else
475 setStatus( ANIM_2D_STATUS_COMPLETE );
476 break;
477
478 } // end once backwards
479
480 // -------------------------------------------------------------------------------------------
481 case ANIM_2D_LOOP:
482 {
483
484 if( m_currentFrame == m_maxFrame )
485 setCurrentFrame( m_minFrame );
486 else
487 setCurrentFrame( m_currentFrame + 1 );
488 break;
489
490 } // end loop
491
492 // -------------------------------------------------------------------------------------------
493 case ANIM_2D_LOOP_BACKWARDS:
494 {
495
496 if( m_currentFrame > m_minFrame )
497 setCurrentFrame( m_currentFrame - 1 );
498 else
499 setCurrentFrame( m_maxFrame );
500 break;
501
502 } // end loop backwards
503
504 // -------------------------------------------------------------------------------------------
505 case ANIM_2D_PING_PONG:
506 case ANIM_2D_PING_PONG_BACKWARDS:
507 {
508
509 if( BitTest( m_status, ANIM_2D_STATUS_REVERSED ) )
510 {
511 //
512 // decrement frame, unless we're at frame 0 in which case we
513 // increment and reverse directions
514 //
515 if( m_currentFrame == m_minFrame )
516 {
517
518 setCurrentFrame( m_currentFrame + 1 );
519 clearStatus( ANIM_2D_STATUS_REVERSED );
520
521 } // end if
522 else
523 {
524
525 setCurrentFrame( m_currentFrame - 1 );
526
527 } // end else
528
529 } // end if
530 else
531 {
532
533 //
534 // increment frame, unless we're at the end in which case we decrement
535 // and reverse directions
536 //
537 if( m_currentFrame == m_maxFrame )
538 {
539
540 setCurrentFrame( m_currentFrame - 1 );
541 setStatus( ANIM_2D_STATUS_REVERSED );
542
543 } // end if
544 else
545 {
546
547 setCurrentFrame( m_currentFrame + 1 );
548
549 } // end else
550
551 } // end else
552
553 break;
554
555 } // end ping pong / ping pong backwards
556
557 // -------------------------------------------------------------------------------------------
558 default:
559 {
560
561 DEBUG_CRASH(( "Anim2D::tryNextFrame - Unknown animation mode '%d' for '%s'\n",
562 m_template->getAnimMode(), m_template->getName().str() ));
563 break;
564
565 } // end default
566
567 } // end switch
568
569 } // end if
570
571} // end tryNextFrame

623// ------------------------------------------------------------------------------------------------
625// ------------------------------------------------------------------------------------------------
626void Anim2D::draw( Int x, Int y )
627{
628
629 // get the current image
630 const Image *image = m_template->getFrame( m_currentFrame );
631
632 // sanity
633 DEBUG_ASSERTCRASH( image != NULL, ("Anim2D::draw - Image not found for frame '%d' on animation '%s'\n",
634 m_currentFrame, m_template->getName().str()) );
635
636 // get the natural width and height of this image
637 const ICoord2D *imageSize = image->getImageSize();
638
639 // draw the image
640 Color color = GameMakeColor( 255, 255, 255, 255 * m_alpha );
641 TheDisplay->drawImage( image, x, y, x + imageSize->x, y + imageSize->y, color );
642
643 //
644 // see if it's time for us to go to the next frame in the sequence, we do not update
645 // frame numbers for animation instances that are registered with a system as the
646 // system will update them during its update phase
647 //
648 if( m_collectionSystem == NULL && BitTest( m_status, ANIM_2D_STATUS_FROZEN ) == FALSE )
649 tryNextFrame();
650
651} // end draw

835// ------------------------------------------------------------------------------------------------
838// ------------------------------------------------------------------------------------------------
839void Anim2DCollection::registerAnimation( Anim2D *anim )
840{
841
842 // sanity
843 if( anim == NULL )
844 return;
845
846 // sanity
847 DEBUG_ASSERTCRASH( anim->m_collectionSystemNext == NULL &&
848 anim->m_collectionSystemPrev == NULL,
849 ("Registering animation instance, instance '%s' is already in a system\n",
850 anim->getAnimTemplate()->getName().str()) );
851
852 // tie to our list
853 anim->m_collectionSystemPrev = NULL;
854 anim->m_collectionSystemNext = m_instanceList;
855 if( m_instanceList )
856 m_instanceList->m_collectionSystemPrev = anim;
857 m_instanceList = anim;
858
859} // end registerAnimation

861// ------------------------------------------------------------------------------------------------
862// ------------------------------------------------------------------------------------------------
863void Anim2DCollection::unRegisterAnimation( Anim2D *anim )
864{
865
866 // sanity
867 if( anim == NULL )
868 return;
869
870 // if animation is not registered with us do nothing
871 if( anim->m_collectionSystem != this )
872 return;
873
874 // unlink from our instnace list
875 if( anim->m_collectionSystemNext )
876 anim->m_collectionSystemNext->m_collectionSystemPrev = anim->m_collectionSystemPrev;
877 if( anim->m_collectionSystemPrev )
878 anim->m_collectionSystemPrev->m_collectionSystemNext = anim->m_collectionSystemNext;
879 else
880 m_instanceList = anim->m_collectionSystemNext;
881
882} // end unRegisterAnimation
