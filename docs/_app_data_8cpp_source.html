60/*
61**
62** MAXScript Function:
63** wwCopyAppData - Usage: wwCopyAppData to_node from_node
64**
65** Copies all AppData associated with from_node to to_node.
66** This is needed for W3D flags such as Export Geometry
67** and Export Transform to get passed on to their
68** instances/copies/references.
69*/
70Value * copy_app_data_cf (Value **arg_list, int count)
71{
72 // Verify the number and type of the arguments.
73 check_arg_count("wwCopyAppData", 2, count);
74 type_check(arg_list[0], MAXNode, "Target INode");
75 type_check(arg_list[1], MAXNode, "Source INode");
76
77 // Get the INode pointers that were passed in.
78 INode *dest_node = arg_list[0]->to_node();
79 INode *src_node = arg_list[1]->to_node();
80
81 /*
82 ** Copy W3DAppData0Struct
83 */
84 W3DAppData0Struct *app_data_0 = GetW3DAppData0(src_node);
85 if (app_data_0 != NULL) {
86
87 // App Data 0 is now obsolete, not fatal if we don't find one
88 W3DAppData0Struct *copy_data_0 = new W3DAppData0Struct;
89 if (copy_data_0 == NULL)
90 throw RuntimeError("Out of memory.");
91
92 // Copy the app data and give it to the target node.
93 *copy_data_0 = *app_data_0;
94 dest_node->AddAppDataChunk(W3DUtilityClassID, UTILITY_CLASS_ID, W3D_APPDATA_0,
95 sizeof(W3DAppData0Struct), copy_data_0);
96 }
97
98 /*
99 ** Copy W3DAppData1Struct
100 */
101 W3DAppData1Struct *app_data_1 = GetW3DAppData1(src_node);
102 if (app_data_1 == NULL)
103 throw RuntimeError("Unable to retrieve W3DAppData1Struct from object: ", arg_list[1]);
104
105 W3DAppData1Struct *copy_data_1 = new W3DAppData1Struct;
106 if (copy_data_1 == NULL)
107 throw RuntimeError("Out of memory.");
108
109 // Copy the app data and give it to the target node.
110 *copy_data_1 = *app_data_1;
111 dest_node->AddAppDataChunk(W3DUtilityClassID, UTILITY_CLASS_ID, W3D_APPDATA_1,
112 sizeof(W3DAppData1Struct), copy_data_1);
113
114
115 /*
116 ** Copy W3DAppData2Struct
117 */
118 W3DAppData2Struct *app_data_2 = GetW3DAppData2(src_node);
119 if (app_data_2 == NULL)
120 throw RuntimeError("Unable to retrieve W3DAppData1Struct from object: ", arg_list[1]);
121
122 W3DAppData2Struct *copy_data_2 = new W3DAppData2Struct;
123 if (copy_data_2 == NULL)
124 throw RuntimeError("Out of memory.");
125
126 // Copy the app data and give it to the target node.
127 *copy_data_2 = *app_data_2;
128 dest_node->AddAppDataChunk(W3DUtilityClassID, UTILITY_CLASS_ID, W3D_APPDATA_2,
129 sizeof(W3DAppData2Struct), copy_data_2);
130
131 /*
132 ** Copy W3DDazzleAppDataStruct if one is present.
133 */
134 W3DDazzleAppDataStruct *dazzle_app_data = GetW3DDazzleAppData(src_node);
135 if (dazzle_app_data != NULL) {
136
137 W3DDazzleAppDataStruct *copy_dazzle_data = new W3DDazzleAppDataStruct;
138 if (copy_dazzle_data == NULL)
139 throw RuntimeError("Out of memory.");
140
141
142 // Copy the app data and give it to the target node.
143 *copy_dazzle_data = *dazzle_app_data;
144 dest_node->AddAppDataChunk(W3DUtilityClassID, UTILITY_CLASS_ID, W3D_DAZZLE_APPDATA,
145 sizeof(W3DDazzleAppDataStruct), copy_dazzle_data);
146 }
147
148 return &ok;
149}

153/*
154**
155** MAXScript Function:
156** wwSetOriginAppData - Usage: wwSetOriginAppData origin_node
157**
158** Sets the AppData associated with the given node to values
159** appropriate to an origin. (ie. turn off Export Geometry and
160** Export Transform)
161*/
162Value * set_origin_app_data_cf (Value **arg_list, int count)
163{
164 // Check the arguments that were passed to this function.
165 check_arg_count("wwSetOriginAppData", 1, count);
166 type_check(arg_list[0], MAXNode, "Origin INode");
167
168 // Get the INode that we were given.
169 INode *origin = arg_list[0]->to_node();
170
171 // Get the node's W3DAppData2Struct, and modify it accordingly.
172 W3DAppData2Struct *data = GetW3DAppData2(origin);
173 if (data == NULL)
174 throw RuntimeError("Unable to retrieve W3DAppData0Struct from object: ", arg_list[0]);
175
176 // Turn off Export Geometry and Export Hierarchy.
177 data->Enable_Export_Geometry(false);
178 data->Enable_Export_Transform(false);
179
180 return &ok;
181}

183/*
184**
185** MAXScript Function:
186** wwGetHierarchyFile - Usage: wwGetHierarchyFile()
187**
188** Returns the relative pathname of the file that will be loaded
189** during W3D export to get the hierarchy tree. If no such file
190** will be loaded, the return value is the MAXScript undefined.
191**
192*/
193Value * get_hierarchy_file_cf (Value **arg_list, int count)
194{
195 // Check that we weren't passed any arguments.
196 check_arg_count("wwGetHierarchyFile", 0, count);
197
198 // Retrieve the export options from the scene.
199 W3dExportOptionsStruct *options = NULL;
200 AppDataChunk * appdata = MAXScript_interface->GetScenePointer()->GetAppDataChunk(W3D_EXPORTER_CLASS_ID,SCENE_EXPORT_CLASS_ID,0);
201 if (appdata)
202 options = (W3dExportOptionsStruct*)(appdata->data);
203
204 // If we didn't get any options, return undefined.
205 if (!options)
206 return &undefined;
207
208 // Return the relative path to the htree file if it's available.
209 // Otherwise, return the absolute path.
210 one_typed_value_local(String* htree_file);
211 if (options->RelativeHierarchyFilename[0] != 0)
212 vl.htree_file = new String(options->RelativeHierarchyFilename);
213 else if (options->HierarchyFilename[0] != 0)
214 vl.htree_file = new String(options->HierarchyFilename);
215 else
216 {
217 // Neither filename is available, return undefined.
218 pop_value_locals();
219 return &undefined;
220 }
221 return_value(vl.htree_file);
222}
