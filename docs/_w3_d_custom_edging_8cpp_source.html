115void W3DCustomEdging::loadEdgingsInVertexAndIndexBuffers(WorldHeightMap *pMap, Int minX, Int maxX, Int minY, Int maxY)
116{
117 if (!m_indexEdging || !m_vertexEdging || !m_initialized) {
118 return;
119 }
120 if (!m_anythingChanged) {
121 return;
122 }
123 m_anythingChanged = false;
124 m_curNumEdgingVertices = 0;
125 m_curNumEdgingIndices = 0;
126 VertexFormatXYZDUV2 *vb;
127 UnsignedShort *ib;
128 // Lock the buffers.
129 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexEdging);
130 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexEdging);
131 vb=(VertexFormatXYZDUV2*)lockVtxBuffer.Get_Vertex_Array();
132 ib = lockIdxBuffer.Get_Index_Array();
133
134 UnsignedShort *curIb = ib;
135
136 VertexFormatXYZDUV2 *curVb = vb;
137
138 if (minX<0) minX = 0;
139 if (minY<0) minY = 0;
140 if (maxX >= pMap->getXExtent()) maxX = pMap->getXExtent()-1;
141 if (maxY >= pMap->getYExtent()) maxY = pMap->getYExtent()-1;
142 Int row;
143 Int column;
144 try {
145 for (row=minY; row<maxY-1; row++) {
146 for (column = minX; column < maxX-1; column++) {
147 Int cellNdx = column+row*pMap->getXExtent();
148 Int blend = pMap->m_blendTileNdxes[cellNdx];
149 if (blend == 0) continue; // no blend.
150
151 if (pMap->m_blendedTiles[blend].customBlendEdgeClass<0) continue; // alpha blend.
152 Int i, j;
153 Real uOffset;
154 Real vOffset;
155
156 if (pMap->m_blendedTiles[blend].horiz) {
157 uOffset = 0;
158 vOffset = 0.25f * (1 + (row&1));
159 if (pMap->m_blendedTiles[blend].inverted) {
160 uOffset = 0.75f;
161 }
162 } else if (pMap->m_blendedTiles[blend].vert) {
163 vOffset = 0.75;
164 uOffset = 0.25f * (1 + (column&1));
165 if (pMap->m_blendedTiles[blend].inverted) {
166 vOffset = 0.0f;
167 }
168 } else if (pMap->m_blendedTiles[blend].rightDiagonal) {
169 if (pMap->m_blendedTiles[blend].longDiagonal) {
170 vOffset = 0.25;
171 uOffset = 0.5;
172 if (pMap->m_blendedTiles[blend].inverted) {
173 uOffset = 0.5f;
174 vOffset = 0.5f;
175 }
176 } else {
177 vOffset = .75;
178 uOffset = 0;
179 if (pMap->m_blendedTiles[blend].inverted) {
180 uOffset = 0.0f;
181 vOffset = 0.0f;
182 }
183 }
184 } else if (pMap->m_blendedTiles[blend].leftDiagonal) {
185 if (pMap->m_blendedTiles[blend].longDiagonal) {
186 uOffset = 0.25f;
187 vOffset = 0.25f;
188 if (pMap->m_blendedTiles[blend].inverted) {
189 uOffset = 0.25f;
190 vOffset = 0.5f;
191 }
192 } else {
193 vOffset = 0.75;
194 uOffset = 0.75f;
195 if (pMap->m_blendedTiles[blend].inverted) {
196 uOffset = 0.75f;
197 vOffset = 0.0f;
198 }
199 }
200 } else {
201 continue;
202 }
203 Region2D range;
204 pMap->getUVForBlend(pMap->m_blendedTiles[blend].customBlendEdgeClass, &range);
205
206 uOffset = range.lo.x + range.width()*uOffset;
207 vOffset = range.lo.y + range.height()*vOffset;
208
209 UnsignedByte alpha[4];
210 float UA[4], VA[4];
211 Bool flipForBlend;
212 pMap->getAlphaUVData(column-pMap->getDrawOrgX(), row-pMap->getDrawOrgY(), UA, VA, alpha, &flipForBlend, false);
213
214
215 Int startVertex = m_curNumEdgingVertices;
216 for (j=0; j<2; j++) {
217 for (i=0; i<2; i++) {
218 if (m_curNumEdgingVertices >= MAX_EDGE_VERTEX) return;
219 cellNdx = column+i+(row+j)*pMap->getXExtent();
220
221 Int diffuse = TheTerrainRenderObject->getStaticDiffuse(column+i, row+j);
222 curVb->diffuse = 0x80000000 + (diffuse&0x00FFFFFF); // set alpha to 5b.
223 Real theZ;
224 theZ = ((float)pMap->getDataPtr()[cellNdx])*MAP_HEIGHT_SCALE;
225 Real X = (column+i)*MAP_XY_FACTOR;
226 Real Y = (row+j)*MAP_XY_FACTOR;
227 curVb->u2 = uOffset + i*0.25f*range.width();
228 curVb->v2 = vOffset + (1-j)*0.25f*range.height();
229 Int ndx;
230 if (j==0) ndx=i;
231 if (j==1) ndx = 3-i;
232 curVb->u1 = UA[ndx];
233 curVb->v1 = VA[ndx];
234 curVb->x = X;
235 curVb->y = Y;
236 curVb->z = theZ;
237 curVb++;
238 m_curNumEdgingVertices++;
239 }
240 }
241 Int yOffset = 2;
242 if (m_curNumEdgingIndices+6 > MAX_EDGE_INDEX) return;
243#ifdef FLIP_TRIANGLES // jba - reduces "diamonding" in some cases, not others. Better cliffs, though.
244 if (flipForBlend) {
245 *curIb++ = startVertex + 1;
246 *curIb++ = startVertex + yOffset;
247 *curIb++ = startVertex ;
248 *curIb++ = startVertex + 1;
249 *curIb++ = startVertex + 1+yOffset;
250 *curIb++ = startVertex + yOffset;
251 }
252 else
253#endif
254 {
255 *curIb++ = startVertex;
256 *curIb++ = startVertex + 1+yOffset;
257 *curIb++ = startVertex + yOffset;
258 *curIb++ = startVertex ;
259 *curIb++ = startVertex + 1;
260 *curIb++ = startVertex + 1+yOffset;
261 }
262 m_curNumEdgingIndices+=6;
263 }
264 }
265 IndexBufferExceptionFunc();
266 } catch(...) {
267 IndexBufferExceptionFunc();
268 }
269 m_anythingChanged = false;
270}

349 TextureClass * terrainTexture, TextureClass * cloudTexture, TextureClass * noiseTexture)
