117BOOL PS2GameMtlShaderDlg::Dialog_Proc (HWND dlg_wnd, UINT message, WPARAM wparam, LPARAM lparam)
118{
119 int cursel;
120 int i;
121 int id = LOWORD(wparam);
122 int code = HIWORD(wparam);
123
124 switch (message)
125 {
126
127 case WM_INITDIALOG:
128 for(i = 0; i <= NUM_PS2_SHADER_BLEND_PRESETS; i++) {
129 SendDlgItemMessage(dlg_wnd,IDC_PS2_PRESET_COMBO,CB_ADDSTRING,0,(LONG)_PS2ShaderBlendSettingPresetNames[i]);
130 }
131 SendDlgItemMessage(dlg_wnd,IDC_PS2_PRESET_COMBO,CB_SETCURSEL,0,0);
132 break;
133
134 case WM_LBUTTONDOWN:
135 case WM_LBUTTONUP:
136 case WM_MOUSEMOVE:
137 {
138 IParams->RollupMouseMessage(dlg_wnd,message,wparam,lparam);
139 }
140 return FALSE;
141
142 case WM_COMMAND:
143 {
144 if (code == CBN_SELCHANGE) {
145
146 switch (id)
147 {
148 // Shared by both shaders.
149 case IDC_DEPTHCOMPARE_COMBO:
150 cursel = SendDlgItemMessage(dlg_wnd,IDC_DEPTHCOMPARE_COMBO,CB_GETCURSEL,0,0);
151 TheMtl->Set_Depth_Compare(PassIndex,cursel);
152 break;
153 case IDC_PRIGRADIENT_COMBO:
154 cursel = SendDlgItemMessage(dlg_wnd,IDC_PRIGRADIENT_COMBO,CB_GETCURSEL,0,0);
155 TheMtl->Set_Pri_Gradient(PassIndex,cursel);
156 TheMtl->Notify_Changed();
157 break;
158 case IDC_SECGRADIENT_COMBO:
159 cursel = SendDlgItemMessage(dlg_wnd,IDC_SECGRADIENT_COMBO,CB_GETCURSEL,0,0);
160 TheMtl->Set_Sec_Gradient(PassIndex,cursel);
161 TheMtl->Notify_Changed();
162 break;
163 case IDC_DETAILCOLOR_COMBO:
164 cursel = SendDlgItemMessage(dlg_wnd,IDC_DETAILCOLOR_COMBO,CB_GETCURSEL,0,0);
165 TheMtl->Set_Detail_Color_Func(PassIndex,cursel);
166 TheMtl->Notify_Changed();
167 break;
168 case IDC_DETAILALPHA_COMBO:
169 cursel = SendDlgItemMessage(dlg_wnd,IDC_DETAILALPHA_COMBO,CB_GETCURSEL,0,0);
170 TheMtl->Set_Detail_Alpha_Func(PassIndex,cursel);
171 TheMtl->Notify_Changed();
172 break;
173
174 // Playstation 2 specific.
175 case IDC_PS2_PRESET_COMBO:
176 cursel = SendDlgItemMessage(dlg_wnd,IDC_PS2_PRESET_COMBO,CB_GETCURSEL,0,0);
177 Apply_Preset(cursel);
178 break;
179 case IDC_A_COMBO:
180 cursel = SendDlgItemMessage(dlg_wnd,IDC_A_COMBO,CB_GETCURSEL,0,0);
181 TheMtl->Set_PS2_Shader_Param_A(PassIndex,cursel);
182 TheMtl->Notify_Changed();
183 Set_Preset();
184 break;
185 case IDC_B_COMBO:
186 cursel = SendDlgItemMessage(dlg_wnd,IDC_B_COMBO,CB_GETCURSEL,0,0);
187 TheMtl->Set_PS2_Shader_Param_B(PassIndex,cursel);
188 TheMtl->Notify_Changed();
189 Set_Preset();
190 break;
191 case IDC_C_COMBO:
192 cursel = SendDlgItemMessage(dlg_wnd,IDC_C_COMBO,CB_GETCURSEL,0,0);
193 TheMtl->Set_PS2_Shader_Param_C(PassIndex,cursel);
194 TheMtl->Notify_Changed();
195 Set_Preset();
196 break;
197 case IDC_D_COMBO:
198 cursel = SendDlgItemMessage(dlg_wnd,IDC_D_COMBO,CB_GETCURSEL,0,0);
199 TheMtl->Set_PS2_Shader_Param_D(PassIndex,cursel);
200 TheMtl->Notify_Changed();
201 Set_Preset();
202 break;
203 }
204
205 } else {
206
207 switch(id) {
208
209 case IDC_DEPTHMASK_CHECK:
210 if (SendDlgItemMessage(dlg_wnd,IDC_DEPTHMASK_CHECK,BM_GETCHECK,0,0)) {
211 TheMtl->Set_Depth_Mask(PassIndex,W3DSHADER_DEPTHMASK_WRITE_ENABLE);
212 } else {
213 TheMtl->Set_Depth_Mask(PassIndex,W3DSHADER_DEPTHMASK_WRITE_DISABLE);
214 }
215 Set_Preset();
216 break;
217
218 case IDC_ALPHATEST_CHECK:
219 if (SendDlgItemMessage(dlg_wnd,IDC_ALPHATEST_CHECK,BM_GETCHECK,0,0)) {
220 TheMtl->Set_Alpha_Test(PassIndex,W3DSHADER_ALPHATEST_ENABLE);
221 } else {
222 TheMtl->Set_Alpha_Test(PassIndex,W3DSHADER_ALPHATEST_DISABLE);
223 }
224 Set_Preset();
225 break;
226
227 case IDC_SHADER_DEFAULTS_BUTTON:
228 Set_Advanced_Defaults();
229 break;
230 }
231 }
232 }
233 }
234
235 return FALSE;
236}

267/***********************************************************************************************
268 * PGMSD::Apply_Preset -- Notify the material of the values for the selected setting. *
269 * *
270 * *
271 * *
272 * *
273 * HISTORY: *
274 * 10/12/1999MLL: Created. *
275 *=============================================================================================*/
276void PS2GameMtlShaderDlg::Apply_Preset(int preset_index)
277{
278 if (preset_index < 0 || preset_index >= NUM_PS2_SHADER_BLEND_PRESETS) return;
279
280 const PS2ShaderBlendSettingPreset &preset = PS2ShaderBlendSettingPresets[preset_index];
281
282 int depth_mask = preset.DepthMask ? W3DSHADER_DEPTHMASK_WRITE_ENABLE : W3DSHADER_DEPTHMASK_WRITE_DISABLE;
283 TheMtl->Set_Depth_Mask(PassIndex, depth_mask);
284
285 int alpha_test = preset.AlphaTest ? W3DSHADER_ALPHATEST_ENABLE : W3DSHADER_ALPHATEST_DISABLE;
286 TheMtl->Set_Alpha_Test(PassIndex, alpha_test);
287
288 TheMtl->Set_PS2_Shader_Param_A(PassIndex, preset.A);
289 TheMtl->Set_PS2_Shader_Param_B(PassIndex, preset.B);
290 TheMtl->Set_PS2_Shader_Param_C(PassIndex, preset.C);
291 TheMtl->Set_PS2_Shader_Param_D(PassIndex, preset.D);
292
293 TheMtl->Notify_Changed();
294 ReloadDialog();
295
296 if (TheMtl->Compute_PC_Shader_From_PS2_Shader(PassIndex))
297 SetWindowText(GetDlgItem(m_hWnd,IDC_COMPATIBLE), " Compatible");
298 else
299 SetWindowText(GetDlgItem(m_hWnd,IDC_COMPATIBLE), " NOT Compatible");
300
301}

325/***********************************************************************************************
326 * PGMSD::CompareShaderToBlendPreset -- Determine if the settings conform to one of the presets*
327 * *
328 * *
329 * *
330 * *
331 * HISTORY: *
332 * 10/12/1999MLL: Created. *
333 *=============================================================================================*/
334bool PS2GameMtlShaderDlg::CompareShaderToBlendPreset(const PS2ShaderBlendSettingPreset &blend_preset)
335{
336 bool depthmask = TheMtl->Get_Depth_Mask(PassIndex) != W3DSHADER_DEPTHMASK_WRITE_DISABLE;
337 if (depthmask != blend_preset.DepthMask) return false;
338 bool alphatest = TheMtl->Get_Alpha_Test(PassIndex) != W3DSHADER_ALPHATEST_DISABLE;
339 if (alphatest != blend_preset.AlphaTest) return false;
340
341 if (TheMtl->Get_PS2_Shader_Param_A(PassIndex) != blend_preset.A) return false;
342 if (TheMtl->Get_PS2_Shader_Param_B(PassIndex) != blend_preset.B) return false;
343 if (TheMtl->Get_PS2_Shader_Param_C(PassIndex) != blend_preset.C) return false;
344 if (TheMtl->Get_PS2_Shader_Param_D(PassIndex) != blend_preset.D) return false;
345
346 return true;
347}
