53UpdateSleepTime ObjectDefectionHelper::update()
54{
55 Object* obj = getObject();
56 Drawable* draw = obj->getDrawable();
57
58 // if we are here, we should be an undetected defector, but
59 // just in case someone has changed us behind our backs...
60 if (!obj->getIsUndetectedDefector())
61 return UPDATE_SLEEP_FOREVER;
62
63 UnsignedInt now = TheGameLogic->getFrame();
64 if (now >= m_defectionDetectionEnd)
65 {
66 obj->friend_setUndetectedDefector(FALSE);
67
68 // timer has reached zero, so we flash white once!!!! -- lorenzen
69 if (draw && m_doDefectorFX)
70 {
71 RGBColor white = {1,1,1};
72 if (draw)
73 draw->flashAsSelected( &white ); //Whew! that's easier, now, isn't it!
74
75
76 AudioEventRTS defectorVulnerableSound = TheAudio->getMiscAudio()->m_defectorTimerDingSound;
77 defectorVulnerableSound.setObjectID( obj->getID() );
78 TheAudio->addAudioEvent(&defectorVulnerableSound);
79 }
80 return UPDATE_SLEEP_FOREVER; // hey, we're done.
81 }
82
83 // dead or attacking... our cover is blown.
84 if( obj->isEffectivelyDead() || obj->getStatusBits().test( OBJECT_STATUS_IS_FIRING_WEAPON ) )
85 {
86 // PLEASE NOTE:
87 // checking the is_attacking statusbit above, only handles weapon related attacks...
88 // I also set m_undetectedDefector = FALSE in the groupDoSpecialPower[...]( ) functions;
89 obj->friend_setUndetectedDefector(FALSE);
90 return UPDATE_SLEEP_FOREVER; // hey, we're done.
91 }
92
93 if (draw && m_doDefectorFX) // skip fx if merely 'invulnerable'
94 {
95 Bool lastPhase = ( ((Int)m_defectionDetectionFlashPhase) & 1 );// were we in a flashy phase last frame?
96 UnsignedInt timeLeft = m_defectionDetectionEnd - now;
97 m_defectionDetectionFlashPhase += 0.5f * ( 1.0f - ((Real)timeLeft / DEFECTION_DETECTION_TIME_MAX ) );
98 Bool thisPhase = ( ((Int)m_defectionDetectionFlashPhase) & 1 );// are we in a flashy phase this frame?
99
100 if ( lastPhase && ( ! thisPhase ) )
101 {
102 draw->flashAsSelected(); //Whew! that's easier, now, isn't it!
103
104 AudioEventRTS defectorTimerSound = TheAudio->getMiscAudio()->m_defectorTimerTickSound;
105 defectorTimerSound.setObjectID( obj->getID() );
106 TheAudio->addAudioEvent(&defectorTimerSound);
107
108 }
109 }
110
111 return UPDATE_SLEEP_NONE;
112}

116void ObjectDefectionHelper::startDefectionTimer(UnsignedInt numFrames, Bool withDefectorFX)
117{
118 Object* obj = getObject();
119
120 if (!obj->getIsUndetectedDefector())
121 {
122 setWakeFrame(obj, UPDATE_SLEEP_FOREVER);
123 return;
124 }
125
126 UnsignedInt now = TheGameLogic->getFrame();
127 m_defectionDetectionStart = now;
128 m_defectionDetectionEnd = now + numFrames;
129 m_defectionDetectionFlashPhase = 0.0f;
130 m_doDefectorFX = withDefectorFX;
131
132 setWakeFrame(obj, UPDATE_SLEEP_NONE);
133}

151void ObjectDefectionHelper::xfer( Xfer *xfer )
152{
153
154 // version
155 XferVersion currentVersion = 1;
156 XferVersion version = currentVersion;
157 xfer->xferVersion( &version, currentVersion );
158
159 // object helper base class

161
162 // detection start

164
165 // detection end
166 xfer->xferUnsignedInt( &m_defectionDetectionEnd );
167
168 // flash phase
169 xfer->xferReal( &m_defectionDetectionFlashPhase );
170
171 // do defector fx
172 xfer->xferBool( &m_doDefectorFX );

174} // end xfer
