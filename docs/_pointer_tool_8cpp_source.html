41// This function spiders out and un/picks all Waypoints that have some form of indirect contact with this point
42// Has a recursive helper function as well.
43//
44static void helper_pickAllWaypointsInPath( Int sourceID, CWorldBuilderDoc *pDoc, const Int numWaypointLinks, std::vector<Int>& alreadyTouched );
45
46static void pickAllWaypointsInPath( Int sourceID, Bool select )
47{
48 std::vector<Int> alreadyTouched;
49 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
50
51 helper_pickAllWaypointsInPath(sourceID, pDoc, pDoc->getNumWaypointLinks(), alreadyTouched);
52
53 // already touched should now be filled with waypointIDs that want to be un/selected
54 MapObject *pMapObj = MapObject::getFirstMapObject();
55 while (pMapObj) {
56 if (pMapObj->isWaypoint()) {
57 if (std::find(alreadyTouched.begin(), alreadyTouched.end(), pMapObj->getWaypointID()) != alreadyTouched.end()) {
58 pMapObj->setSelected(select);
59 }
60 }
61 pMapObj = pMapObj->getNext();
62 }
63}
64
65static void helper_pickAllWaypointsInPath( Int sourceID, CWorldBuilderDoc *pDoc, const Int numWaypointLinks, std::vector<Int>& alreadyTouched )
66{
67 if (std::find(alreadyTouched.begin(), alreadyTouched.end(), sourceID) != alreadyTouched.end() ) {
68 return;
69 }
70
71 alreadyTouched.push_back(sourceID);
72 for (int i = 0; i < numWaypointLinks; ++i) {
73 Int way1, way2;
74 pDoc->getWaypointLink(i, &way1, &way2);
75 if (way1 == sourceID) {
76 helper_pickAllWaypointsInPath(way2, pDoc, numWaypointLinks, alreadyTouched);
77 }
78
79 if (way2 == sourceID) {
80 helper_pickAllWaypointsInPath(way1, pDoc, numWaypointLinks, alreadyTouched);
81 }
82 }
83}

208Bool PointerTool::allowPick(MapObject* pMapObj, WbView* pView)
209{
210 EditorSortingType sort = ES_NONE;
211 if (!pMapObj) {
212 return false;
213 }
214 const ThingTemplate *tt = pMapObj->getThingTemplate();
215 if (tt && tt->getEditorSorting() == ES_AUDIO) {
216 if (pView->GetPickConstraint() == ES_NONE || pView->GetPickConstraint() == ES_AUDIO) {
217 return true;
218 }
219 }
220 if ((tt && !pView->getShowModels()) || (pMapObj->getFlags() & FLAG_DONT_RENDER)) {
221 return false;
222 }
223 if (pView->GetPickConstraint() != ES_NONE) {
224 if (tt) {
225 if (!pView->getShowModels()) {
226 return false;
227 }
228 sort = tt->getEditorSorting();
229 } else {
230 if (pMapObj->isWaypoint()) {
231 sort = ES_WAYPOINT;
232 }
233 if (pMapObj->getFlag(FLAG_ROAD_FLAGS)) {
234 sort = ES_ROAD;
235 }
236 }
237 if (sort != ES_NONE && sort != pView->GetPickConstraint()) {
238 return false;
239 }
240 }
241 return true;

259 Bool shiftKey = (0x8000 & ::GetAsyncKeyState(VK_SHIFT))!=0;
260 Bool ctrlKey = (0x8000 & ::GetAsyncKeyState(VK_CONTROL))!=0;
261
262 m_doPolyTool = false;

271 pView->snapPoint(&cpt);
272 }
273 m_poly_mouseDownPt = cpt;
274 m_poly_justPicked = true; // Makes poly tool move instead of inserting.
275 m_doPolyTool = true;
276 PolygonTool::startMouseDown(m, viewPt, pView, pDoc);
277 return;
278 }

280 m_poly_dragPointNdx = -1;

287 m_curObject = NULL;

289 MapObject *p3DObj = pView->picked3dObjectInView(viewPt);
290 MapObject *pClosestPicked = NULL;

292 pClosestPicked = p3DObj;
293 }
294 Real pickDistSqr = 10000*MAP_XY_FACTOR;
295 pickDistSqr *= pickDistSqr;
296
297 // Find the closest pick.
298 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
299 if (!allowPick(pObj, pView)) {
300 continue;
301 }
302 Bool picked = (pView->picked(pObj, cpt) != PICK_NONE);
303 if (picked) {
304 loc = *pObj->getLocation();
305 Real dx = m_downPt3d.x-loc.x;
306 Real dy = m_downPt3d.y-loc.y;
307 Real distSqr = dx*dx+dy*dy;
308 if (distSqr < pickDistSqr) {
309 pClosestPicked = pObj;
310 pickDistSqr = distSqr;
311 }
312 }
313 }
314
315 Bool anySelected = (pClosestPicked!=NULL);
316 if (shiftKey) {
317 if (pClosestPicked && pClosestPicked->isSelected()) {
318 pClosestPicked->setSelected(false);
319 if (ctrlKey && pClosestPicked->isWaypoint()) {
320 pickAllWaypointsInPath(pClosestPicked->getWaypointID(), false);
321 }
322 } else if (pClosestPicked) {
323 pClosestPicked->setSelected(true);
324 if (ctrlKey && pClosestPicked->isWaypoint()) {
325 pickAllWaypointsInPath(pClosestPicked->getWaypointID(), true);
326 }
327 }
328 } else if (pClosestPicked && pClosestPicked->isSelected()) {
329 // We picked a selected object
330 m_curObject = pClosestPicked;
331 } else {
332 clearSelection();
333 if (pClosestPicked) {
334 pClosestPicked->setSelected(true);
335 if (ctrlKey && pClosestPicked->isWaypoint()) {
336 pickAllWaypointsInPath(pClosestPicked->getWaypointID(), true);
337 }
338
339 }
340 }
341
342 // Grab both ends of a road.
343 if (pView->GetPickConstraint() == ES_NONE || pView->GetPickConstraint() == ES_ROAD) {
344 if (!shiftKey && pClosestPicked && (pClosestPicked->getFlags()&FLAG_ROAD_FLAGS) ) {
345 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
346 if (pObj->getFlags()&FLAG_ROAD_FLAGS) {
347 loc = *pObj->getLocation();
348 Real dx = pClosestPicked->getLocation()->x - loc.x;
349 Real dy = pClosestPicked->getLocation()->y - loc.y;
350 Real dist = sqrt(dx*dx+dy*dy);
351 if (dist < MAP_XY_FACTOR/100) {
352 pObj->setSelected(true);
353 }
354 }
355 }
356 }
357 }
358
359 if (anySelected) {
360 if (m_curObject) {
361 // See if we are picking on the arrow.
362 if (pView->picked(m_curObject, cpt) == PICK_ARROW) {
363 m_rotating = true;
364 }
365 } else {
366 pObj = MapObject::getFirstMapObject();
367 while (pObj) {
368 if (pObj->isSelected()) {
369 m_curObject = pObj;
370 break;
371 }
372 pObj = pObj->getNext();
373 }
374 }
375 if (m_curObject) {
376 // adjust the starting point so if we are snapping, the object snaps as well.
377 loc = *m_curObject->getLocation();
378 Coord3D snapLoc = loc;
379 pView->snapPoint(&snapLoc);
380 m_downPt3d.x += (loc.x-snapLoc.x);
381 m_downPt3d.y += (loc.y-snapLoc.y);
382 }
383 } else {
384 m_dragSelect = true;
385 }

389void PointerTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
390{
391 Coord3D cpt;
392 pView->viewToDocCoords(viewPt, &cpt, false);
393 if (m == TRACK_NONE) {
394 // See if the cursor is over an object.
395 MapObject *pObj = MapObject::getFirstMapObject();
396 m_mouseUpRotate = false;
397 m_mouseUpMove = false;
398 while (pObj) {
399 if (allowPick(pObj, pView)) {
400 TPickedStatus stat = pView->picked(pObj, cpt);
401 if (stat==PICK_ARROW) {
402 m_mouseUpRotate = true;
403 break;
404 }
405 if (stat==PICK_CENTER) {
406 m_mouseUpMove = true;
407 break;
408 }
409 }
410 pObj = pObj->getNext();
411 }
412 if (!m_mouseUpRotate) {
413 pObj = pView->picked3dObjectInView(viewPt);
414 if (allowPick(pObj, pView)) {
415 m_mouseUpMove = true;
416 }
417 }
418 if (pView->isPolygonTriggerVisible() && pickPolygon(cpt, viewPt, pView)) {
419 if (pView->GetPickConstraint() == ES_NONE || pView->GetPickConstraint() == ES_WAYPOINT) {
420 m_mouseUpMove = true;
421 m_mouseUpRotate = false;
422 }
423 }
424 return; // setCursor will use the value of m_mouseUpRotate. jba.
425 }
426
427 if (m != TRACK_L) return;
428 if (m_doPolyTool) {
429 PolygonTool::mouseMoved(m, viewPt, pView, pDoc);
430 return;
431 }
432
433 if (m_dragSelect) {
434 CRect box;
435 box.left = viewPt.x;
436 box.bottom = viewPt.y;
437 box.top = m_downPt2d.y;
438 box.right = m_downPt2d.x;
439 box.NormalizeRect();
440 pView->doRectFeedback(true, box);
441 pView->Invalidate();
442 return;
443 }
444
445 if (m_curObject == NULL) {
446 return;
447 }
448 pView->viewToDocCoords(viewPt, &cpt, !m_rotating);
449 if (!m_moving) {
450 // always use view coords (not doc coords) for hysteresis
451 Int dx = viewPt.x-m_downPt2d.x;
452 Int dy = viewPt.y-m_downPt2d.y;
453 if (abs(dx)>HYSTERESIS || abs(dy)>HYSTERESIS) {
454 m_moving = true;
455 m_modifyUndoable = new ModifyObjectUndoable(pDoc);
456 }
457 }
458 if (!m_moving || !m_modifyUndoable) return;
459
460 MapObject *curMapObj = MapObject::getFirstMapObject();
461 while (curMapObj) {
462 if (curMapObj->isSelected()) {
463 //pDoc->invalObject(curMapObj); // invaling in all views can be too slow.
464 pView->invalObjectInView(curMapObj);
465 }
466 curMapObj = curMapObj->getNext();
467 }
468
469 if (m_rotating) {
470 Coord3D center = *m_curObject->getLocation();
471 m_modifyUndoable->RotateTo(ObjectTool::calcAngle(center, cpt, pView));
472 } else {
473 pView->snapPoint(&cpt);
474 Real xOffset = (cpt.x-m_downPt3d.x);
475 Real yOffset = (cpt.y-m_downPt3d.y);
476 m_modifyUndoable->SetOffset(xOffset, yOffset);
477 }
478
479 curMapObj = MapObject::getFirstMapObject();
480 while (curMapObj) {
481 if (curMapObj->isSelected()) {
482 //pDoc->invalObject(curMapObj); // invaling in all views can be too slow.
483 pView->invalObjectInView(curMapObj);
484 }
485 curMapObj = curMapObj->getNext();
486 }
487
488 pDoc->updateAllViews();
489
490}

495void PointerTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
496{
497 if (m != TRACK_L) return;
498
499 if (m_doPolyTool) {
500 m_doPolyTool = false;
501 PolygonTool::mouseUp(m, viewPt, pView, pDoc);
502 checkForPropertiesPanel();
503 return;
504 }
505
506 Coord3D cpt;
507 pView->viewToDocCoords(viewPt, &cpt);
508
509 if (m_curObject && m_moving) {
510 pDoc->AddAndDoUndoable(m_modifyUndoable);
511 REF_PTR_RELEASE(m_modifyUndoable); // belongs to pDoc now.
512 } else if (m_dragSelect) {
513 CRect box;
514 box.left = viewPt.x;
515 box.top = viewPt.y;
516 box.right = m_downPt2d.x;
517 box.bottom = m_downPt2d.y;
518 box.NormalizeRect();
519 pView->doRectFeedback(false, box);
520 pView->Invalidate();
521
522 MapObject *pObj;
523 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
524 // Don't pick on invisible waypoints
525 if (pObj->isWaypoint() && !pView->isWaypointVisible()) {
526 continue;
527 }
528 if (!allowPick(pObj, pView)) {
529 continue;
530 }
531 Bool picked;
532 Coord3D loc = *pObj->getLocation();
533 CPoint viewPt;
534 if (pView->docToViewCoords(loc, &viewPt)){
535 picked = (viewPt.x>=box.left && viewPt.x<=box.right && viewPt.y>=box.top && viewPt.y<=box.bottom) ;
536 if (picked) {
537 if ((0x8000 && ::GetAsyncKeyState(VK_SHIFT))) {
538 pObj->setSelected(!pObj->isSelected());
539 } else {
540 pObj->setSelected(true);
541 }
542 pDoc->invalObject(pObj);
543 }
544 }
545 }
546
547 }
548 checkForPropertiesPanel();
