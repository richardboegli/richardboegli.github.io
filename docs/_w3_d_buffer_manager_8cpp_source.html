82void W3DBufferManager::freeAllSlots(void)
83{
84 Int i,j;
85
86 for (i=0; i<MAX_FVF; i++)
87 {
88 for (j=0; j<MAX_VB_SIZES; j++)
89 {
90 //Release all slots allocated for each size
91 W3DVertexBufferSlot *vbSlot = m_W3DVertexBufferSlots[i][j];
92 while (vbSlot)
93 {
94 if (vbSlot->m_prevSameVB)
95 vbSlot->m_prevSameVB->m_nextSameVB=vbSlot->m_nextSameVB;
96 else
97 vbSlot->m_VB->m_usedSlots=NULL;
98
99 if (vbSlot->m_nextSameVB)
100 vbSlot->m_nextSameVB->m_prevSameVB=vbSlot->m_prevSameVB;
101 vbSlot=vbSlot->m_nextSameSize;
102 m_numEmptySlotsAllocated--;
103 }
104 m_W3DVertexBufferSlots[i][j]=NULL;
105 }
106 }
107
108 for (j=0; j<MAX_IB_SIZES; j++)
109 {
110 //Release all slots allocated for each size
111 W3DIndexBufferSlot *ibSlot = m_W3DIndexBufferSlots[j];
112 while (ibSlot)
113 {
114 if (ibSlot->m_prevSameIB)
115 ibSlot->m_prevSameIB->m_nextSameIB=ibSlot->m_nextSameIB;
116 else
117 ibSlot->m_IB->m_usedSlots=NULL;
118
119 if (ibSlot->m_nextSameIB)
120 ibSlot->m_nextSameIB->m_prevSameIB=ibSlot->m_prevSameIB;
121 ibSlot=ibSlot->m_nextSameSize;
122 m_numEmptyIndexSlotsAllocated--;
123 }
124 m_W3DIndexBufferSlots[j]=NULL;
125 }
126
127 DEBUG_ASSERTCRASH(m_numEmptySlotsAllocated==0, ("Failed to free all empty vertex buffer slots"));
128 DEBUG_ASSERTCRASH(m_numEmptyIndexSlotsAllocated==0, ("Failed to free all empty index buffer slots"));
129}

131void W3DBufferManager::freeAllBuffers(void)
132{
133 Int i;
134
135 //Make sure all slots are free
136 freeAllSlots();
137
138 for (i=0; i<MAX_FVF; i++)
139 {
140 W3DVertexBuffer *vb = m_W3DVertexBuffers[i];
141 while (vb)
142 { DEBUG_ASSERTCRASH(vb->m_usedSlots == NULL, ("Freeing Non-Empty Vertex Buffer"));
143 REF_PTR_RELEASE(vb->m_DX8VertexBuffer);
144 m_numEmptyVertexBuffersAllocated--;
145 vb=vb->m_nextVB; //get next vertex buffer of this type
146 }
147 m_W3DVertexBuffers[i]=NULL;
148 }
149
150 W3DIndexBuffer *ib = m_W3DIndexBuffers;
151 while (ib)
152 { DEBUG_ASSERTCRASH(ib->m_usedSlots == NULL, ("Freeing Non-Empty Index Buffer"));
153 REF_PTR_RELEASE(ib->m_DX8IndexBuffer);
154 m_numEmptyIndexBuffersAllocated--;
155 ib=ib->m_nextIB; //get next vertex buffer of this type
156 }
157 m_W3DIndexBuffers=NULL;
158
159 DEBUG_ASSERTCRASH(m_numEmptyVertexBuffersAllocated==0, ("Failed to free all empty vertex buffers"));
160 DEBUG_ASSERTCRASH(m_numEmptyIndexBuffersAllocated==0, ("Failed to free all empty index buffers"));
161}

183Bool W3DBufferManager::ReAcquireResources(void)
184{
185 for (Int i=0; i<MAX_FVF; i++)
186 {
187 W3DVertexBuffer *vb = m_W3DVertexBuffers[i];
188 while (vb)
189 { DEBUG_ASSERTCRASH( vb->m_DX8VertexBuffer == NULL, ("ReAcquire of existing vertex buffer"));
190 vb->m_DX8VertexBuffer=NEW_REF(DX8VertexBufferClass,(FVFTypeIndexList[vb->m_format],vb->m_size,DX8VertexBufferClass::USAGE_DEFAULT));
191 DEBUG_ASSERTCRASH( vb->m_DX8VertexBuffer, ("Failed ReAcquire of vertex buffer"));
192 if (!vb->m_DX8VertexBuffer)
193 return FALSE;
194 vb=vb->m_nextVB; //get next vertex buffer of this type
195 }
196 }
197
198 W3DIndexBuffer *ib = m_W3DIndexBuffers;
199 while (ib)
200 { DEBUG_ASSERTCRASH( ib->m_DX8IndexBuffer == NULL, ("ReAcquire of existing index buffer"));
201 ib->m_DX8IndexBuffer=NEW_REF(DX8IndexBufferClass,(ib->m_size,DX8IndexBufferClass::USAGE_DEFAULT));
202 DEBUG_ASSERTCRASH( ib->m_DX8IndexBuffer, ("Failed ReAcquire of index buffer"));
203 if (!ib->m_DX8IndexBuffer)
204 return FALSE;
205 ib=ib->m_nextIB; //get next vertex buffer of this type
206 }
207
208 return TRUE;
209}

215W3DBufferManager::W3DVertexBufferSlot *W3DBufferManager::getSlot(VBM_FVF_TYPES fvfType, Int size)
216{
217 W3DVertexBufferSlot *vbSlot=NULL;
218
219 //round size to next multiple of minimum slot size.
220 //should help avoid fragmentation.
221 size = (size + (MIN_SLOT_SIZE-1)) & (~(MIN_SLOT_SIZE-1));
222 Int sizeIndex = (size >> MIN_SLOT_SIZE_SHIFT)-1;
223
224 DEBUG_ASSERTCRASH(sizeIndex < MAX_VB_SIZES && size, ("Allocating too large vertex buffer slot"));
225
226 if ((vbSlot=m_W3DVertexBufferSlots[fvfType][sizeIndex]) != 0)
227 { //found a previously allocated slot matching required size
228 m_W3DVertexBufferSlots[fvfType][sizeIndex]=vbSlot->m_nextSameSize;
229 if (vbSlot->m_nextSameSize)
230 vbSlot->m_nextSameSize->m_prevSameSize=NULL;
231 return vbSlot;
232 }
233 else
234 { //need to allocate a new slot
235 return allocateSlotStorage(fvfType, size);
236 }
237
238 return NULL;
239}

255W3DBufferManager::W3DVertexBufferSlot * W3DBufferManager::allocateSlotStorage(VBM_FVF_TYPES fvfType, Int size)
256{
257
258 W3DVertexBuffer *pVB;
259 W3DVertexBufferSlot *vbSlot;
260// Int sizeIndex = (size >> MIN_SLOT_SIZE_SHIFT)-1;
261
262 DEBUG_ASSERTCRASH(m_numEmptySlotsAllocated < MAX_NUMBER_SLOTS, ("Nore more VB Slots"));
263
264 pVB=m_W3DVertexBuffers[fvfType];
265 while (pVB)
266 {
267 if ((pVB->m_size - pVB->m_startFreeIndex) >= size)
268 { //found enough free space in this vertex buffer
269
270 if (m_numEmptySlotsAllocated < MAX_NUMBER_SLOTS)
271 { //we're allowing more slots to be allocated.
272 vbSlot=&m_W3DVertexBufferEmptySlots[m_numEmptySlotsAllocated];
273 vbSlot->m_size=size;
274 vbSlot->m_start=pVB->m_startFreeIndex;
275 vbSlot->m_VB=pVB;
276 //Link to VB list of slots
277 vbSlot->m_nextSameVB=pVB->m_usedSlots;
278 vbSlot->m_prevSameVB=NULL; //this will be the new head
279 if (pVB->m_usedSlots)
280 pVB->m_usedSlots->m_prevSameVB=vbSlot;
281 vbSlot->m_prevSameSize=vbSlot->m_nextSameSize=NULL;
282 pVB->m_usedSlots=vbSlot;
283 pVB->m_startFreeIndex += size;
284 m_numEmptySlotsAllocated++;
285 return vbSlot;
286 }
287 }
288 pVB = pVB->m_nextVB;
289 }
290
291 pVB=m_W3DVertexBuffers[fvfType]; //save old list head
292
293 //Didn't find any vertex buffers with room, create a new one
294 DEBUG_ASSERTCRASH(m_numEmptyVertexBuffersAllocated < MAX_VERTEX_BUFFERS_CREATED, ("Reached Max Static VB Shadow Geometry"));
295
296 if (m_numEmptyVertexBuffersAllocated < MAX_VERTEX_BUFFERS_CREATED)
297 {
298 m_W3DVertexBuffers[fvfType] = &m_W3DEmptyVertexBuffers[m_numEmptyVertexBuffersAllocated];
299 m_W3DVertexBuffers[fvfType]->m_nextVB=pVB; //link to list
300 m_numEmptyVertexBuffersAllocated++;
301
302 pVB=m_W3DVertexBuffers[fvfType]; //get new list head
303
304 Int vbSize=__max(DEFAULT_VERTEX_BUFFER_SIZE,size);
305
306 pVB->m_DX8VertexBuffer=NEW_REF(DX8VertexBufferClass,(FVFTypeIndexList[fvfType],vbSize,DX8VertexBufferClass::USAGE_DEFAULT));
307 pVB->m_format=fvfType;
308 pVB->m_startFreeIndex=size;
309 pVB->m_size=vbSize;
310 vbSlot=&m_W3DVertexBufferEmptySlots[m_numEmptySlotsAllocated];
311 m_numEmptySlotsAllocated++;
312 pVB->m_usedSlots=vbSlot;
313 vbSlot->m_size=size;
314 vbSlot->m_start=0;
315 vbSlot->m_VB=pVB;
316 vbSlot->m_prevSameVB=vbSlot->m_nextSameVB=NULL;
317 vbSlot->m_prevSameSize=vbSlot->m_nextSameSize=NULL;
318 return vbSlot;
319 }
320
321 return NULL;
322}
