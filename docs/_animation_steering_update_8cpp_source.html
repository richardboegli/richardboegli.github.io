70UpdateSleepTime AnimationSteeringUpdate::update( void )
71{
72
73 const AnimationSteeringUpdateModuleData *data = getAnimationSteeringUpdateModuleData();
74 PhysicsBehavior *physics = getObject()->getPhysics();
75 Drawable *draw = getObject()->getDrawable();
76 UnsignedInt now = TheGameLogic->getFrame();
77
78 if( physics && draw && now >= m_nextTransitionFrame )
79 {
80 PhysicsTurningType currentTurn = physics->getTurning();
81
82 switch( m_currentTurnAnim )
83 {
84 case MODELCONDITION_INVALID:
85 //We're currently going straight. Check if we want to turn.
86 if( currentTurn == TURN_NEGATIVE )
87 {
88 //Initiate a right turn
89 draw->setModelConditionState( MODELCONDITION_CENTER_TO_RIGHT );
90 m_nextTransitionFrame = now + data->m_transitionFrames;
91 m_currentTurnAnim = MODELCONDITION_CENTER_TO_RIGHT;
92 }
93 else if( currentTurn == TURN_POSITIVE )
94 {
95 //Initiate a left turn
96 draw->setModelConditionState( MODELCONDITION_CENTER_TO_LEFT );
97 m_nextTransitionFrame = now + data->m_transitionFrames;
98 m_currentTurnAnim = MODELCONDITION_CENTER_TO_LEFT;
99 }
100 break;
101 case MODELCONDITION_CENTER_TO_RIGHT:
102 //We're currently initiating a turn to the right. The only thing
103 //we can do go back to center or maintain the turn.
104 if( currentTurn != TURN_NEGATIVE )
105 {
106 //Recenter!
107 draw->clearAndSetModelConditionState( MODELCONDITION_CENTER_TO_RIGHT, MODELCONDITION_RIGHT_TO_CENTER );
108 m_nextTransitionFrame = now + data->m_transitionFrames;
109 m_currentTurnAnim = MODELCONDITION_RIGHT_TO_CENTER;
110 }
111 break;
112 case MODELCONDITION_CENTER_TO_LEFT:
113 //We're currently initiating a turn to the left. The only thing
114 //we can do go back to center or maintain the turn.
115 if( currentTurn != TURN_POSITIVE )
116 {
117 //Recenter!
118 draw->clearAndSetModelConditionState( MODELCONDITION_CENTER_TO_LEFT, MODELCONDITION_LEFT_TO_CENTER );
119 m_nextTransitionFrame = now + data->m_transitionFrames;
120 m_currentTurnAnim = MODELCONDITION_LEFT_TO_CENTER;
121 }
122 break;
123 case MODELCONDITION_LEFT_TO_CENTER:
124 case MODELCONDITION_RIGHT_TO_CENTER:
125 if( currentTurn == TURN_NONE )
126 {
127 //Finish the turn
128 draw->clearModelConditionFlags( MAKE_MODELCONDITION_MASK2( MODELCONDITION_LEFT_TO_CENTER, MODELCONDITION_RIGHT_TO_CENTER ) );
129 m_nextTransitionFrame = now;
130 m_currentTurnAnim = MODELCONDITION_INVALID;
131 }
132 break;
133 }
134 }
135
136 return UPDATE_SLEEP_NONE;
137}
