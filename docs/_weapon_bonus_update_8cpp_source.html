116void containIteratingDoTempWeaponBonus( Object *passenger, void *voidData)
117{
118 tempWeaponBonusData *data = (tempWeaponBonusData *)voidData;
119

126UpdateSleepTime WeaponBonusUpdate::update( void )
127{
128 const WeaponBonusUpdateModuleData * data = getWeaponBonusUpdateModuleData();
129 Object *me = getObject();
130
131 PartitionFilterRelationship relationship( me, PartitionFilterRelationship::ALLOW_ALLIES );
132 PartitionFilterSameMapStatus filterMapStatus(me);
133 PartitionFilterAlive filterAlive;
134
135 // Leaving this here commented out to show that I need to reach valid contents of invalid transports.
136 // So these checks are on an individual basis, not in the Partition query
137// PartitionFilterAcceptByKindOf filterKindof(data->m_requiredAffectKindOf,data->m_forbiddenAffectKindOf);
138 PartitionFilter *filters[] = { &relationship, &filterAlive, &filterMapStatus, NULL };
139
140 // scan objects in our region
141 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( me->getPosition(),
142 data->m_bonusRange,
143 FROM_CENTER_2D,
144 filters );
145 MemoryPoolObjectHolder hold( iter );
146 tempWeaponBonusData weaponBonusData;
147 weaponBonusData.m_type = data->m_bonusConditionType;
148 weaponBonusData.m_duration = data->m_bonusDuration;
149 weaponBonusData.m_requiredMask = data->m_requiredAffectKindOf;
150 weaponBonusData.m_forbiddenMask = data->m_forbiddenAffectKindOf;
151
152 for( Object *currentObj = iter->first(); currentObj != NULL; currentObj = iter->next() )
153 {
154 if( currentObj->isKindOfMulti(data->m_requiredAffectKindOf, data->m_forbiddenAffectKindOf) )
155 {
156 currentObj->doTempWeaponBonus(data->m_bonusConditionType, data->m_bonusDuration);
157 }
158
159 if( currentObj->getContain() )
160 {
161 currentObj->getContain()->iterateContained(containIteratingDoTempWeaponBonus, &weaponBonusData, FALSE);
162 }
163 }
164
165 return UPDATE_SLEEP(data->m_bonusDelay); // Only need an internal timer if there are external hooks for wakning us up, or a second thing we can do
166}
