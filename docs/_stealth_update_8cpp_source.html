202void StealthUpdate::receiveGrant( Bool active, UnsignedInt frames )
203{
204 Object *obj = getObject();
205 if ( obj == NULL )
206 return;
207
208 if (this->canDisguise())
209 return; //so bombtrucks and stuff do not get foiled by this.
210
211 //Kris: Turn it off if we pass in FALSE for active.
212 if( !active && m_enabled )
213 {
214 //markAsDetected();
215 }
216
217 m_enabled = active;
218
219 if( m_enabled )
220 {
221 //On
222 obj->setStatus( MAKE_OBJECT_STATUS_MASK2( OBJECT_STATUS_CAN_STEALTH, OBJECT_STATUS_STEALTHED ) );
223 m_stealthAllowedFrame = TheGameLogic->getFrame();
224 setWakeFrame( obj, UPDATE_SLEEP_NONE );
225 m_framesGranted = frames;
226 }
227 else
228 {
229 //Off
230 obj->clearStatus( MAKE_OBJECT_STATUS_MASK2( OBJECT_STATUS_CAN_STEALTH, OBJECT_STATUS_STEALTHED ) );
231 m_stealthAllowedFrame = FOREVER;
232 m_framesGranted = 0;
233 Drawable *draw = obj->getDrawable();
234 if( draw )
235 {
236 draw->setEffectiveOpacity( 1.0f );
237 }
238 }
239
240 const ContainModuleInterface *contain = obj->getContain();
241 if ( contain && contain->isRiderChangeContain() )
242 {
243 const Object *rider = contain->friend_getRider();
244 if ( rider )
245 {
246 StealthUpdate *riderStealth = rider->getStealth();
247 if ( riderStealth )
248 riderStealth->receiveGrant( active, frames );
249 }
250 }

258Bool StealthUpdate::allowedToStealth( Object *stealthOwner ) const
259{
260 const Object *self = getObject();
261 const StealthUpdateModuleData *data = getStealthUpdateModuleData();
262 UnsignedInt now = TheGameLogic->getFrame();
263
264 UnsignedInt flags = data->m_stealthLevel;
265 if( self != stealthOwner )
266 {
267 //Extract the rules from the rider's stealthupdate module data instead
268 //of our own, because the rider determines if the container can stealth or not.
269
270 const StealthUpdate *stealthUpdate = stealthOwner->getStealth();
271 if( stealthUpdate )
272 {
273 flags = stealthUpdate->getStealthLevel();
274 }
275 }
276
277 //With regards to slaves that stealth with us, we need to all be stealthed or not at all. If
278 //any of the slaves can't stealth, then reveal everyone!
279 if( self->isKindOf( KINDOF_SPAWNS_ARE_THE_WEAPONS ) )
280 {
281 SpawnBehaviorInterface *sbInterface = self->getSpawnBehaviorInterface();
282 if( sbInterface )
283 {
284 if( !sbInterface->areAllSlavesStealthed() )
285 {
286 sbInterface->revealSlaves();
287 return FALSE;
288 }
289 }
290 }
291
292 if( flags & STEALTH_NOT_WHILE_ATTACKING && self->getStatusBits().test( OBJECT_STATUS_IS_FIRING_WEAPON ) )
293 {
294 //Doesn't stealth while aggressive (includes approaching).
295 return FALSE;
296 }
297
298 if( flags & STEALTH_NOT_WHILE_USING_ABILITY && self->getStatusBits().test( OBJECT_STATUS_IS_USING_ABILITY ) )
299 {
300 //Doesn't stealth while using a special ability (starting with preparation, which takes place after unpacking).
301 return FALSE;
302 }
303
304 if( flags & STEALTH_ONLY_WITH_BLACK_MARKET && m_nextBlackMarketCheckFrame < now )
305 {
306 //randomize timer a little incase we have a whole bunch on the same frame.
307 m_nextBlackMarketCheckFrame += data->m_blackMarketCheckFrames + GameLogicRandomValue( 0, 10 );
308
309 //If we can't find an active black market, then we can't stealth.
310 Bool blackMarket = FALSE;
311 self->getControllingPlayer()->iterateObjects( isBlackMarket, &blackMarket );
312 if( !blackMarket )
313 {
314 return FALSE;
315 }
316 }
317
318 if( !stealthOwner->getStatusBits().test( OBJECT_STATUS_CAN_STEALTH ) )
319 {
320 return FALSE;
321 }
322
323 if( flags & STEALTH_NOT_WHILE_TAKING_DAMAGE && self->getBodyModule()->getLastDamageTimestamp() >= now - 1 )
324 {
325 //Only if it's not healing damage.
326 if( self->getBodyModule()->getLastDamageInfo()->in.m_damageType != DAMAGE_HEALING )
327 {
328 //Can't stealth if we just took damage in the last frame or two.
329 if( self->getBodyModule()->getLastDamageTimestamp() != 0xffffffff )
330 {
331 //But it's initialized to 0xffffffff so we don't think we took damage on the first frame.
332 return FALSE;
333 }
334 }
335 }
336
337 //We need all required status or else we fail
338 // If we have any requirements
339 if( data->m_requiredStatus.any() && !self->getStatusBits().testForAll( data->m_requiredStatus ) )
340 return FALSE;
341
342 //If we have any forbidden statii, then fail
343 if( self->getStatusBits().testForAny( data->m_forbiddenStatus ) )
344 return FALSE;
345
346
347 //Do a quick preliminary test to see if we are restricted by firing particular weapons and we fired a shot last frame or this frame.
348 if( flags & STEALTH_NOT_WHILE_FIRING_WEAPON && self->getStatusBits().test( OBJECT_STATUS_IS_FIRING_WEAPON ) )
349 {
350 if( (flags & STEALTH_NOT_WHILE_FIRING_WEAPON) == STEALTH_NOT_WHILE_FIRING_WEAPON )
351 {
352 //Not allowed to stealth while firing ANY weapon!
353 return FALSE;
354 }
355
356 //Now do weapon specific checks.
357 Weapon *weapon;
358 UnsignedInt lastFrame = TheGameLogic->getFrame() - 1;
359
360 if( flags & STEALTH_NOT_WHILE_FIRING_PRIMARY )
361 {
362 //Check primary weapon status
363 weapon = self->getWeaponInWeaponSlot( PRIMARY_WEAPON );
364 if( weapon && weapon->getLastShotFrame() >= lastFrame )
365 {
366 return FALSE;
367 }
368 }
369 if( flags & STEALTH_NOT_WHILE_FIRING_SECONDARY )
370 {
371 //Check secondary weapon status
372 weapon = self->getWeaponInWeaponSlot( SECONDARY_WEAPON );
373 if( weapon && weapon->getLastShotFrame() >= lastFrame )
374 {
375 return FALSE;
376 }
377 }
378 if( flags & STEALTH_NOT_WHILE_FIRING_TERTIARY )
379 {
380 //Check tertiary weapon status
381 weapon = self->getWeaponInWeaponSlot( TERTIARY_WEAPON );
382 if( weapon && weapon->getLastShotFrame() >= lastFrame )
383 {
384 return FALSE;
385 }
386 }
387 }
388
389 const Object *containedBy = self->getContainedBy();
390 if( containedBy )
391 {
392 ContainModuleInterface *contain = containedBy->getContain();
393 if( contain && !contain->isGarrisonable() )
394 {
395 return FALSE;
396 }
397 }
398
399 //new past-alpha feature, grr...
400 if( flags & STEALTH_NOT_WHILE_RIDERS_ATTACKING )
401 {
402 ContainModuleInterface *myContain = self->getContain();
403 if ( myContain && myContain->isPassengerAllowedToFire() )
404 {
405 if ( myContain->isAnyRiderAttacking() )
406 return FALSE;
407
408 }
409 }
410
411
412
413 const PhysicsBehavior *physics = self->getPhysics();
414 if ((flags & STEALTH_NOT_WHILE_MOVING) && physics != NULL &&
415 physics->getVelocityMagnitude() > getStealthUpdateModuleData()->m_stealthSpeed)
416 return FALSE;
417
418 if( self->testScriptStatusBit(OBJECT_STATUS_SCRIPT_UNSTEALTHED))
419 {
420 //We can't stealth because a script disabled this ability for this object!
421 return FALSE;
422 }
423
424 return TRUE;
425}

459// indicate how the given unit is "stealthed" with respect to a given player.
460StealthLookType StealthUpdate::calcStealthedStatusForPlayer(const Object* obj, const Player* player)
461{
462 /*
463 for stealthy things, there are these distinct "logical" states:
464
465 -- not stealthed at all (ie, totally visible)
466 -- stealthed
467 -- stealthed-but-detected
468
469 and the following visual states:
470
471 -- normal (n)
472 -- invisible (i)
473 -- stealthed-but-visible-to-friendly-folks (sv)
474 -- stealthed-but-visible-to-everyone-due-to-being-detected (sd)
475
476 Let's be ubergeeks and make a matrix of the possibilities:
477
478 Ally Nonally
479 normal: (n) (n)
480 stealthed: (sv) (i)
481 detected: (sd) (sd)
482
483
484 Or, to put it another way:
485
486 If normal, you always appear normal.
487 If stealthed (and not detected), you appear as (sv) to allies and (i) to others.
488 If detected, you always appears as (sd).
489
490 Sorry, there is one more condition, stealthed, but visible to friendly folks, YET detected
491 In this state we render outselves visible and we ovlerlay the detection effect as a warning
492 we'll call this STEALTHLOOK_VISIBLE_FRIENDLY_DETECTED
493
494
495 */
496
497
498 if (obj->isEffectivelyDead())
499 return STEALTHLOOK_NONE; // making sure he turns visible when he dies
500
501 if( obj->getStatusBits().test( OBJECT_STATUS_STEALTHED ) )
502 {
503 const Team* team = obj->getTeam();
504 Relationship r = team ? team->getRelationship(player->getDefaultTeam()) : NEUTRAL;
505 if( !player->isPlayerActive() )
506 {
507 //Observer players are friends to everyone!
508 r = ALLIES;
509 }
510
511// srj sez: disguised stuff doesn't work well when combined with the normal "detected" stuff.
512// so special case it here.
513 if (canDisguise())
514 {
515 if (r != ALLIES && isDisguised())
516 return STEALTHLOOK_DISGUISED_ENEMY;
517 else
518 return STEALTHLOOK_NONE;
519 }
520
521 if( obj->getStatusBits().test( OBJECT_STATUS_DETECTED ) ) // we're detected.
522 {
523 if (r == ALLIES)// if we're friendly to the given player, detection DOES matter though.
524 return STEALTHLOOK_VISIBLE_FRIENDLY_DETECTED;
525 else
526 return STEALTHLOOK_VISIBLE_DETECTED;
527 }
528 else
529 {
530 if (r == ALLIES)
531 {
532 // if we're friendly to the given player, detection doesn't matter.
533 return STEALTHLOOK_VISIBLE_FRIENDLY;
534 }
535 else
536 {
537// srj sez: disguised stuff doesn't work well when combined with the normal "detected" stuff.
538// so special case it above.
539// if( getStealthUpdateModuleData()->m_teamDisguised )
540// {
541// return STEALTHLOOK_DISGUISED_ENEMY;
542// }
543 // we're effectively hidden.
544 return STEALTHLOOK_INVISIBLE;
545 }
546 }
547 }
548 else
549 {
550 return STEALTHLOOK_NONE;
551 }
552}

591//-------------------------------------------------------------------------------------------------
592UpdateSleepTime StealthUpdate::update( void )
593{
594
595 // restore disguise if we need to from a game load
596 if( m_xferRestoreDisguise == TRUE )
597 {
598 Drawable *draw = getObject()->getDrawable();
599 Bool wasHidden = FALSE;
600
601 // hack! if drawable was hidden (such as if we're inside a container) we must keep that state
602 if( draw && draw->isDrawableEffectivelyHidden() )
603 wasHidden = TRUE;
604
605 // do the change (we get a new drawable from this)
606 changeVisualDisguise();
607
608 // restore hidden state in the new drawable
609 draw = getObject()->getDrawable();
610 if( wasHidden && draw )
611 draw->setDrawableHidden( TRUE );
612
613 } // end if
614
615 Object *self = getObject();
616 Object *stealthOwner = calcStealthOwner();
617
618 UnsignedInt stealthDelay;
619
620 if( self == stealthOwner )
621 {
622 const StealthUpdateModuleData *data = getStealthUpdateModuleData();
623 stealthDelay = data->m_stealthDelay;
624 }
625 else
626 {
627 //Extract the rules from the rider's stealthupdate module data instead
628 //of our own, because the rider determines if the container can stealth or not.
629 const StealthUpdate *stealthUpdate = stealthOwner->getStealth();
630 if( stealthUpdate )
631 {
632 stealthDelay = stealthUpdate->getStealthDelay();
633 }
634 }
635
636 UnsignedInt now = TheGameLogic->getFrame();
637
639// and demand every-frame attention when enabled. this could probably be smartened.
640
641 if( !m_enabled )
642 {
643 return calcSleepTime();
644 }
645
646 Drawable* draw = self->getDrawable();
647
648 if( draw )
649 {
650 const StealthUpdateModuleData *data = getStealthUpdateModuleData();
651 //Are we disguise transitioning (either gaining or losing disguise look?)
654 if (self->isKindOf(KINDOF_MINE))
655 {
656 // special case for mines
657 draw->setEffectiveOpacity( 0.0f, 0.0f );
658 }
659 else if( m_disguiseTransitionFrames )
660 {
661 m_disguiseTransitionFrames--;
662 Real factor;
663 if( m_transitioningToDisguise )
664 {
665 factor = 1.0f - ( (Real)m_disguiseTransitionFrames / (Real)data->m_disguiseTransitionFrames );
666 }
667 else
668 {
669 factor = 1.0f - ( (Real)m_disguiseTransitionFrames / (Real)data->m_disguiseRevealTransitionFrames );
670 }
671 if( factor >= 0.5f && !m_disguiseHalfpointReached )
672 {
673 //Switch models at the halfway point
674 changeVisualDisguise();
675 m_disguiseHalfpointReached = true;
676 }
677 //Opacity ranges from full to none at midpoint and full again at the end
678 Real opacity = fabs( 1.0f - (factor * 2.0f) );
679 Real overrideOpacity = opacity < 1.0f ? 0.0f : 1.0f;
680 draw->setEffectiveOpacity( opacity, overrideOpacity );
681 if( !m_disguiseTransitionFrames && !m_transitioningToDisguise )
682 {
683 //We're finished removing disguise so turn off stealth update.
684 m_enabled = false;
685 self->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_STEALTHED ) );
686 self->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DETECTED ) );
687 return calcSleepTime();
688 }
689 }
690 else
691 {
692 draw->setEffectiveOpacity( 0.5f + ( Sin( m_pulsePhase ) * 0.5f ) );
693 // between one half and full opacity
694 m_pulsePhase += m_pulsePhaseRate;
695 }
696 }
697
699 Real revealDistance = getRevealDistanceFromTarget();
700 if( revealDistance > 0.0f )
701 {
702 AIUpdateInterface *ai = self->getAI();
703 if( ai )
704 {
705 Object *target = ai->getCurrentVictim();
706 if( target )
707 {
708 Real distSqrd = ThePartitionManager->getDistanceSquared( self, target, FROM_CENTER_2D );
709 if( distSqrd <= revealDistance * revealDistance )
710 {
711 //We're close enough to reveal ourselves
712 markAsDetected();
713 return calcSleepTime();
714 }
715 }
716 }
717 }
718
719 //Deal with temporary stealth.
720 if( m_framesGranted > 0 )
721 {
722 m_framesGranted--;
723
724 //If the last AI command given was by the player... then LOSE the stealth now!
