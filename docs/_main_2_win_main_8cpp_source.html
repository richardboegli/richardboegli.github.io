306LRESULT CALLBACK WndProc( HWND hWnd, UINT message,
307 WPARAM wParam, LPARAM lParam )
308{
309
310 try
311 {
312 // First let the IME manager do it's stuff.
313 if ( TheIMEManager )
314 {
315 if ( TheIMEManager->serviceIMEMessage( hWnd, message, wParam, lParam ) )
316 {
317 // The manager intercepted an IME message so return the result
318 return TheIMEManager->result();
319 }
320 }
321
322#ifdef DO_COPY_PROTECTION
323 // Check for messages from the launcher
324 CopyProtect::checkForMessage(message, lParam);
325#endif
326
327#ifdef DEBUG_WINDOWS_MESSAGES
328 static msgCount=0;
329 char testString[256];
330 sprintf(testString,"\n%d: %s (%X,%X)", msgCount++,messageToString(message), wParam, lParam);
331 OutputDebugString(testString);
332#endif
333
334 // handle all window messages
335 switch( message )
336 {
337 //-------------------------------------------------------------------------
338 case WM_NCHITTEST:
339 // Prevent the user from selecting the menu in fullscreen mode
340 if( !ApplicationIsWindowed )
341 return HTCLIENT;
342 break;
343
344 //-------------------------------------------------------------------------
345 case WM_POWERBROADCAST:
346 switch( wParam )
347 {
348 #ifndef PBT_APMQUERYSUSPEND
349 #define PBT_APMQUERYSUSPEND 0x0000
350 #endif
351 case PBT_APMQUERYSUSPEND:
352 // At this point, the app should save any data for open
353 // network connections, files, etc., and prepare to go into
354 // a suspended mode.
355 return TRUE;
356
357 #ifndef PBT_APMRESUMESUSPEND
358 #define PBT_APMRESUMESUSPEND 0x0007
359 #endif
360 case PBT_APMRESUMESUSPEND:
361 // At this point, the app should recover any data, network
362 // connections, files, etc., and resume running from when
363 // the app was suspended.
364 return TRUE;
365 }
366 break;
367 //-------------------------------------------------------------------------
368 case WM_SYSCOMMAND:
369 // Prevent moving/sizing and power loss in fullscreen mode
370 switch( wParam )
371 {
372 case SC_MOVE:
373 case SC_SIZE:
374 case SC_MAXIMIZE:
375 case SC_KEYMENU:
376 case SC_MONITORPOWER:
377 if( FALSE == ApplicationIsWindowed )
378 return 1;
379 break;
380 }
381 break;
382
383 case WM_QUERYENDSESSION:
384 {
385 TheMessageStream->appendMessage(GameMessage::MSG_META_DEMO_INSTANT_QUIT);
386 return 0; //don't allow Windows to shutdown while game is running.
387 }
388
389 // ------------------------------------------------------------------------
390 case WM_CLOSE:
391 if (!TheGameEngine->getQuitting())
392 {
393 //user is exiting without using the menus
394
395 //This method didn't work in cinematics because we don't process messages.
396 //But it's the cleanest way to exit that's similar to using menus.
397 TheMessageStream->appendMessage(GameMessage::MSG_META_DEMO_INSTANT_QUIT);
398
399 //This method used to disable quitting. We just put up the options screen instead.
400 //TheMessageStream->appendMessage(GameMessage::MSG_META_OPTIONS);
401
402 //This method works everywhere but isn't as clean at shutting down.
403 //TheGameEngine->checkAbnormalQuitting(); //old way to log disconnections for ALT-F4
404 //TheGameEngine->reset();
405 //TheGameEngine->setQuitting(TRUE);
406 //_exit(EXIT_SUCCESS);
407 return 0;
408 }
409
410 // ------------------------------------------------------------------------
411 case WM_SETFOCUS:
412 {
413
414 //
415 // reset the state of our keyboard cause we haven't been paying
416 // attention to the keys while focus was away
417 //
418 if( TheKeyboard )
419 TheKeyboard->resetKeys();
420
421 if (TheWin32Mouse)
422 TheWin32Mouse->lostFocus(FALSE);
423
424 break;
425
426 } // end set focus
427
428 //-------------------------------------------------------------------------
429 case WM_SIZE:
430 // When W3D initializes, it resizes the window. So stop repainting.
431 if (!gInitializing)
432 gDoPaint = false;
433 break;
434
435 //-------------------------------------------------------------------------
436 case WM_KILLFOCUS:
437 {
438 if (TheKeyboard )
439 TheKeyboard->resetKeys();
440 if (TheWin32Mouse)
441 TheWin32Mouse->lostFocus(TRUE);
442
443 break;
444 }
445
446 //-------------------------------------------------------------------------
447 case WM_ACTIVATEAPP:
448 {
449// DWORD threadId=GetCurrentThreadId();
450 if ((bool) wParam != isWinMainActive)
451 { isWinMainActive = (BOOL) wParam;
452
453 if (TheGameEngine)
454 TheGameEngine->setIsActive(isWinMainActive);
455
456 Reset_D3D_Device(isWinMainActive);
457 if (isWinMainActive)
458 { //restore mouse cursor to our custom version.
459 if (TheWin32Mouse)
460 TheWin32Mouse->setCursor(TheWin32Mouse->getMouseCursor());
461 }
462 }
463 return 0;
464 }
465 //-------------------------------------------------------------------------
466 case WM_ACTIVATE:
467 {
468 Int active = LOWORD( wParam );
469
470 //
471 // when window is becoming deactivated we must release mouse cursor
472 // locks on our region, otherwise set the mouse limit region again
473 // which will clip the cursor to our window
474 //
475 if( active == WA_INACTIVE )
476 {
477
478 ClipCursor( NULL );
479 if (TheAudio)
480 TheAudio->loseFocus();
481 } // end if
482 else
483 {
484 if( TheMouse )
485 TheMouse->setMouseLimits();
486
487 if (TheAudio)
488 TheAudio->regainFocus();
489
490 } // end else
491 break;
492
493 } // end case activate
494
495 //-------------------------------------------------------------------------
496 case WM_KEYDOWN:
497 {
498 Int key = (Int)wParam;
499
500 switch( key )
501 {
502
503 //---------------------------------------------------------------------
504 case VK_ESCAPE:
505 {
506
507 PostQuitMessage( 0 );
508 break;
509
510 } // end VK_ESCAPE
511
512
513 } // end switch
514
515 return 0;
516
517 } // end WM_KEYDOWN
518
519 //-------------------------------------------------------------------------
520 case WM_LBUTTONDOWN:
521 case WM_LBUTTONUP:
522 case WM_LBUTTONDBLCLK:
523
524 case WM_MBUTTONDOWN:
525 case WM_MBUTTONUP:
526 case WM_MBUTTONDBLCLK:
527
528 case WM_RBUTTONDOWN:
529 case WM_RBUTTONUP:
530 case WM_RBUTTONDBLCLK:
531 {
532
533 if( TheWin32Mouse )
534 TheWin32Mouse->addWin32Event( message, wParam, lParam, TheMessageTime );
535
536 return 0;
537
538 } // end WM_LBUTTONDOWN
539
540 //-------------------------------------------------------------------------
541 case 0x020A: // WM_MOUSEWHEEL
542 {
543 long x = (long) LOWORD(lParam);
544 long y = (long) HIWORD(lParam);
545 RECT rect;
546
547 // ignore when outside of client area
548 GetWindowRect( ApplicationHWnd, &rect );
549 if( x < rect.left || x > rect.right || y < rect.top || y > rect.bottom )
550 return 0;
551
552 if( TheWin32Mouse )
553 TheWin32Mouse->addWin32Event( message, wParam, lParam, TheMessageTime );
554
555 return 0;
556
557 } // end WM_MOUSEWHEEL
558
559
560 //-------------------------------------------------------------------------
561 case WM_MOUSEMOVE:
562 {
563 Int x = (Int)LOWORD( lParam );
564 Int y = (Int)HIWORD( lParam );
565 RECT rect;
566// Int keys = wParam;
567
568 // ignore when outside of client area
569 GetClientRect( ApplicationHWnd, &rect );
570 if( x < rect.left || x > rect.right || y < rect.top || y > rect.bottom )
571 return 0;
572
573 if( TheWin32Mouse )
574 TheWin32Mouse->addWin32Event( message, wParam, lParam, TheMessageTime );
575
576 return 0;
577
578 } // end WM_MOUSEMOVE
579
580 //-------------------------------------------------------------------------
581 case WM_SETCURSOR:
582 {
583 if (TheWin32Mouse && (HWND)wParam == ApplicationHWnd)
584 TheWin32Mouse->setCursor(TheWin32Mouse->getMouseCursor());
585 return TRUE; //tell Windows not to reset mouse cursor image to default.
586 }
587
588 case WM_PAINT:
589 {
590 if (gDoPaint) {
591 PAINTSTRUCT paint;
592 HDC dc = ::BeginPaint(hWnd, &paint);
593#if 0
594 ::SetTextColor(dc, RGB(255,255,255));
595 ::SetBkColor(dc, RGB(0,0,0));
596 ::TextOut(dc, 30, 30, "Loading Command & Conquer Generals...", 37);
597#endif
598 if (gLoadScreenBitmap!=NULL) {
599 Int savContext = ::SaveDC(dc);
600 HDC tmpDC = ::CreateCompatibleDC(dc);
601 HBITMAP savBitmap = (HBITMAP)::SelectObject(tmpDC, gLoadScreenBitmap);
602 ::BitBlt(dc, 0, 0, DEFAULT_XRESOLUTION, DEFAULT_YRESOLUTION, tmpDC, 0, 0, SRCCOPY);
603 ::SelectObject(tmpDC, savBitmap);
604 ::DeleteDC(tmpDC);
605 ::RestoreDC(dc, savContext);
606 }
607 ::EndPaint(hWnd, &paint);
608 return TRUE;
609 }
610 break;
611 }
612
613 case WM_ERASEBKGND:
614 {
615 if (!gDoPaint)
616 return TRUE; //we don't need to erase the background because we always draw entire window.
617 break;
618 }

783static char* strtrim(char* buffer)
784{
785 if (buffer != NULL) {
786 // Strip leading white space from the string.
787 char * source = buffer;
788 while ((*source != 0) && ((unsigned char)*source <= 32))
789 {
790 source++;
791 }
792
793 if (source != buffer)
794 {
795 strcpy(buffer, source);
796 }
797
798 // Clip trailing white space from the string.
799 for (int index = strlen(buffer)-1; index >= 0; index--)
800 {
801 if ((*source != 0) && ((unsigned char)buffer[index] <= 32))
802 {
803 buffer[index] = '\0';
804 }
805 else
806 {
807 break;
808 }
809 }
810 }
811
812 return buffer;
813}

815char *nextParam(char *newSource, char *seps)
816{
817 static char *source = NULL;
818 if (newSource)
819 {
820 source = newSource;
821 }
822 if (!source)
823 {
824 return NULL;
825 }
826
827 // find first separator
828 char *first = source;//strpbrk(source, seps);
829 if (first)
830 {
831 // go past separator
832 char *firstSep = strpbrk(first, seps);
833 char firstChar[2] = {0,0};
834 if (firstSep == first)
835 {
836 firstChar[0] = *first;
837 while (*first == firstChar[0]) first++;
838 }
839
840 // find end
841 char *end;
842 if (firstChar[0])
843 end = strpbrk(first, firstChar);
844 else
845 end = strpbrk(first, seps);
846
847 // trim string & save next start pos
848 if (end)
849 {
850 source = end+1;
851 *end = 0;
852
853 if (!*source)
854 source = NULL;
855 }
856 else
857 {
858 source = NULL;
859 }
860
861 if (first && !*first)
862 first = NULL;
863 }
864
865 return first;
866}
