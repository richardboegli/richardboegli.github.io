135// This macro is to assist in the creation of new modules, contains the common
136// things that all module definitions must have in order to work with
137// the module creation factory
138//-------------------------------------------------------------------------------------------------
139#define MAKE_STANDARD_MODULE_MACRO( cls ) \
140public: \
141 static Module* friend_newModuleInstance( Thing *thing, const ModuleData* moduleData ) { return newInstance( cls )( thing, moduleData ); } \
142 virtual NameKeyType getModuleNameKey() const { static NameKeyType nk = NAMEKEY(#cls); return nk; } \
143protected: \
144 virtual void crc( Xfer *xfer ); \
145 virtual void xfer( Xfer *xfer ); \
146 virtual void loadPostProcess( void );
147
148// ------------------------------------------------------------------------------------------------
149// For the creation of abstract module classes
150// ------------------------------------------------------------------------------------------------
151#define MAKE_STANDARD_MODULE_MACRO_ABC( cls ) \
152protected: \
153 virtual void crc( Xfer *xfer ); \
154 virtual void xfer( Xfer *xfer ); \
155 virtual void loadPostProcess( void );
156
157//-------------------------------------------------------------------------------------------------
158// only use this macro for an ABC. for a real class, use MAKE_STANDARD_MODULE_MACRO_WITH_MODULE_DATA.
159#define MAKE_STANDARD_MODULE_DATA_MACRO_ABC( cls, clsmd ) \
160private: \
161 const clsmd* get##clsmd() const { return (clsmd*)getModuleData(); } \
162public: \
163 static ModuleData* friend_newModuleData(INI* ini) \
164 { \
165 clsmd* data = MSGNEW( "AllModuleData" ) clsmd; \
166 if (ini) ini->initFromINIMultiProc(data, clsmd::buildFieldParse); \
167 return data; \
168 }
169
170//-------------------------------------------------------------------------------------------------
171#define MAKE_STANDARD_MODULE_MACRO_WITH_MODULE_DATA( cls, clsmd ) \
172 MAKE_STANDARD_MODULE_MACRO(cls) \
173 MAKE_STANDARD_MODULE_DATA_MACRO_ABC(cls, clsmd)
