74WindowMsgHandledType GadgetTabControlInput( GameWindow *tabControl, UnsignedInt msg,
75 WindowMsgData mData1, WindowMsgData mData2 )
76{
77// WinInstanceData *instData = tabControl->winGetInstanceData();
78 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
79
80 Int tabX, tabY;
81 tabControl->winGetScreenPosition( &tabX, &tabY );
82 Int mouseX = LOLONGTOSHORT(mData1) - tabX;//mData1 is packedMouseCoords in screen space
83 Int mouseY = HILONGTOSHORT(mData1) - tabY;
84 Int tabsLeft = tabData->tabsLeftLimit;
85 Int tabsRight = tabData->tabsRightLimit;
86 Int tabsBottom = tabData->tabsBottomLimit;
87 Int tabsTop = tabData->tabsTopLimit;
88
89 switch( msg )
90 {
91 case GWM_LEFT_DOWN:
92 {
93 if( (mouseX < tabsLeft)
94 || (mouseX > tabsRight)
95 || (mouseY < tabsTop)
96 || (mouseY > tabsBottom)
97 )
98 {//I eat input on myself that isn't a tab (a button click would mean I don't see the input ever.)
99 return MSG_HANDLED;
100 }
101
102 Int distanceIn;
103 Int tabSize;
104 if( (tabData->tabEdge == TP_RIGHT_SIDE) || (tabData->tabEdge == TP_LEFT_SIDE) )
105 {//scan down to find which button
106 distanceIn = mouseY - tabsTop;
107 tabSize = tabData->tabHeight;
108 }
109 else
110 {//scan right to find which button
111 distanceIn = mouseX - tabsLeft;
112 tabSize = tabData->tabWidth;
113 }
114 Int tabPressed = distanceIn / tabSize;
115 if( ! tabData->subPaneDisabled[tabPressed] && (tabPressed != tabData->activeTab) )
116 GadgetTabControlShowSubPane( tabControl, tabPressed );
117 }
118
119 default:
120 {
121 return MSG_IGNORED;
122 }
123 }
124
125 return MSG_HANDLED;
126
127} // end GadgetTabControlInput

132WindowMsgHandledType GadgetTabControlSystem( GameWindow *tabControl, UnsignedInt msg,
133 WindowMsgData mData1, WindowMsgData mData2 )
134{
135 switch( msg )
136 {
137 // ------------------------------------------------------------------------
138 case GWM_CREATE:
139 break;
140
141 // ------------------------------------------------------------------------
142 case GWM_DESTROY:
143 {
144 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
145
146 // free tab control user data
147 delete tabData;
148
149 break;
150
151 } // end destroy
152
153 case GGM_RESIZED:
154 {//On resize, we need to upkeep the pane sizes and tabs since they are bound to us
155 GadgetTabControlResizeSubPanes( tabControl );
156 GadgetTabControlComputeTabRegion( tabControl );
157
158 break;
159 }
160
161 case GBM_SELECTED:
162 {//Pass buttons messages up
163 GameWindow *parent = tabControl->winGetParent();
164
165 if( parent )
166 return TheWindowManager->winSendSystemMsg( parent, msg, mData1, mData2 );
167 }
168
169 default:
170 return MSG_IGNORED;
171
172 } // end switch( msg )
173
174 return MSG_HANDLED;
175
176} // end GadgetTabControlSystem

178void GadgetTabControlComputeTabRegion( GameWindow *tabControl )
179{
180 Int winWidth, winHeight;
181 tabControl->winGetSize( &winWidth, &winHeight );
182
183 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
184
185 Int horzOffset = 0, vertOffset = 0;
186 if( (tabData->tabEdge == TP_TOP_SIDE) || (tabData->tabEdge == TP_BOTTOM_SIDE) )
187 {
188 if( tabData->tabOrientation == TP_CENTER )
189 {
190 horzOffset = winWidth - ( 2 * tabData->paneBorder ) - ( tabData->tabCount * tabData->tabWidth );
191 horzOffset /= 2;
192 }
193 else if( tabData->tabOrientation == TP_BOTTOMRIGHT )
194 {
195 horzOffset = winWidth - ( 2 * tabData->paneBorder ) - ( tabData->tabCount * tabData->tabWidth );
196 }
197 else if( tabData->tabOrientation == TP_TOPLEFT )
198 {
199 horzOffset = 0;
200 }
201 }
202 else
203 {
204 if( tabData->tabOrientation == TP_CENTER )
205 {
206 vertOffset = winHeight - ( 2 * tabData->paneBorder ) - ( tabData->tabCount * tabData->tabHeight );
207 vertOffset /= 2;
208 }
209 else if( tabData->tabOrientation == TP_BOTTOMRIGHT )
210 {
211 vertOffset = winHeight - ( 2 * tabData->paneBorder ) - ( tabData->tabCount * tabData->tabHeight );
212 }
213 else if( tabData->tabOrientation == TP_TOPLEFT )
214 {
215 vertOffset = 0;
216 }
217 }
218
219 if( tabData->tabEdge == TP_TOP_SIDE )
220 {
221 tabData->tabsTopLimit = tabData->paneBorder;
222 tabData->tabsBottomLimit = tabData->paneBorder + tabData->tabHeight;
223 tabData->tabsLeftLimit = tabData->paneBorder + horzOffset;
224 tabData->tabsRightLimit = tabData->paneBorder + horzOffset + ( tabData->tabWidth * tabData->tabCount );
225 }
226 else if( tabData->tabEdge == TP_BOTTOM_SIDE )
227 {
228 tabData->tabsTopLimit = winHeight - tabData->paneBorder - tabData->tabHeight;
229 tabData->tabsBottomLimit = winHeight - tabData->paneBorder;
230 tabData->tabsLeftLimit = tabData->paneBorder + horzOffset;
231 tabData->tabsRightLimit = tabData->paneBorder + horzOffset + ( tabData->tabWidth * tabData->tabCount );
232 }
233 else if( tabData->tabEdge == TP_RIGHT_SIDE )
234 {
235 tabData->tabsLeftLimit = winWidth - tabData->paneBorder - tabData->tabWidth;
236 tabData->tabsRightLimit = winWidth - tabData->paneBorder;
237 tabData->tabsTopLimit = tabData->paneBorder + vertOffset;
238 tabData->tabsBottomLimit = tabData->paneBorder + vertOffset + ( tabData->tabHeight * tabData->tabCount );
239 }
240 else if( tabData->tabEdge == TP_LEFT_SIDE )
241 {
242 tabData->tabsLeftLimit = tabData->paneBorder;
243 tabData->tabsRightLimit = tabData->paneBorder + tabData->tabWidth;
244 tabData->tabsTopLimit = tabData->paneBorder + vertOffset;
245 tabData->tabsBottomLimit = tabData->paneBorder + vertOffset + ( tabData->tabHeight * tabData->tabCount );
246 }
247
248}

250void GadgetTabControlComputeSubPaneSize( GameWindow *tabControl, Int *width, Int *height, Int *x, Int *y )
251{
252 Int winWidth, winHeight;
253 tabControl->winGetSize( &winWidth, &winHeight );
254
255 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
256
257 if( (tabData->tabEdge == TP_TOP_SIDE) || (tabData->tabEdge == TP_BOTTOM_SIDE) )
258 *height = winHeight - (2 * tabData->paneBorder) - tabData->tabHeight;
259 else
260 *height = winHeight - (2 * tabData->paneBorder);
261
262 if( (tabData->tabEdge == TP_LEFT_SIDE) || (tabData->tabEdge == TP_RIGHT_SIDE) )
263 *width = winWidth - (2 * tabData->paneBorder) - tabData->tabWidth;
264 else
265 *width = winWidth - (2 * tabData->paneBorder);
266
267 if( tabData->tabEdge == TP_LEFT_SIDE )
268 *x = tabData->paneBorder + tabData->tabWidth;
269 else
270 *x = tabData->paneBorder;
271
272 if( tabData->tabEdge == TP_TOP_SIDE )
273 *y = tabData->paneBorder + tabData->tabHeight;
274 else
275 *y = tabData->paneBorder;
276}

278void GadgetTabControlShowSubPane( GameWindow *tabControl, Int whichPane)
279{
280 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
281
282 for( Int paneIndex = 0; paneIndex < NUM_TAB_PANES; paneIndex++ )
283 {
284 if( tabData->subPanes[paneIndex] != NULL )
285 tabData->subPanes[paneIndex]->winHide( true );
286 }
287 if( tabData->subPanes[whichPane] )
288 tabData->activeTab = whichPane;
289 else
290 tabData->activeTab = 0;
291
292 tabData->activeTab = min( tabData->activeTab, tabData->tabCount - 1 );
293
294 tabData->subPanes[tabData->activeTab]->winHide( false );
295}

297void GadgetTabControlCreateSubPanes( GameWindow *tabControl )
298{//These two funcs are called after all the Editor set data is updated
299 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
300 Int width, height, x, y;
301 GadgetTabControlComputeSubPaneSize(tabControl, &width, &height, &x, &y);
302
303 for( Int paneIndex = 0; paneIndex < NUM_TAB_PANES; paneIndex++ )
304 {
305 if( (tabData->subPanes[paneIndex] == NULL) )//This one is blank
306 {
307 tabData->subPanes[paneIndex] = TheWindowManager->winCreate( tabControl,
308 WIN_STATUS_NONE, x, y,
309 width, height,
310 PassSelectedButtonsToParentSystem,
311 NULL);
312 WinInstanceData *instData = tabData->subPanes[paneIndex]->winGetInstanceData();
313 BitSet( instData->m_style, GWS_TAB_PANE );
314 char buffer[20];
315 sprintf( buffer, "Pane %d", paneIndex );
316 instData->m_decoratedNameString = buffer;
317 //set enabled status to that of Parent
318 tabData->subPanes[paneIndex]->winEnable( BitTest(tabControl->winGetStatus(), WIN_STATUS_ENABLED) );
319 }
320 else//this one exists, tabCount will control keeping extra panes perma-hidden
321 {
322 tabData->subPanes[paneIndex]->winSetSize( width, height );
323 tabData->subPanes[paneIndex]->winSetPosition( x, y );
324 }
325 }
326
327 GadgetTabControlShowSubPane( tabControl, tabData->activeTab );
328}

346void GadgetTabControlFixupSubPaneList( GameWindow *tabControl )
347{
348 Int childIndex =0;
349 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
350 GameWindow *child = tabControl->winGetChild();
351 if( child )
352 {//need to write down children, and they are reversed from our array
353 while( child->winGetNext() != NULL )
354 {
355 child = child->winGetNext();
356 }
357
358 while( child )
359 {
360 tabData->subPanes[childIndex] = child;
361 childIndex++;
362 child = child->winGetPrev();
363 }
364 }
