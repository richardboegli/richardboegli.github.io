159void BuildListTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
160{
161 if (m != TRACK_L) return;
162 m_moving = false;
163 m_rotating = false;
164 Coord3D cpt;
165 pView->viewToDocCoords(viewPt, &cpt);
166 m_downPt2d = viewPt;
167 m_downPt3d = cpt;
168 if (isDoingAdd()) return;
169 BuildListInfo *pInfo = pView->pickedBuildObjectInView(viewPt);
170 if (pInfo) {
171 m_curObject = pInfo;
172 Coord3D center = *pInfo->getLocation();
173 center.x -= m_downPt3d.x;
174 center.y -= m_downPt3d.y;
175 center.z = 0;
176 Real len = center.length();
177 // Check and see if we are within 1 cell size of the center.
178 if (pInfo->isSelected() && len>0.5f*MAP_XY_FACTOR && len < 1.5f*MAP_XY_FACTOR) {
179 m_rotating = true;
180 }
181 m_moving = true;
182 m_prevPt3d = m_downPt3d;
183 pView->snapPoint(&m_prevPt3d);
184 BuildList::setSelectedBuildList(pInfo);
185 } else {
186 PointerTool::clearSelection();
187 }
188}

190void BuildListTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
191{
192 Coord3D cpt;
193 pView->viewToDocCoords(viewPt, &cpt, false);
194
195 WbView3d *p3View = pDoc->GetActive3DView();
196 if (isDoingAdd()) {
197 Coord3D loc = cpt;
198 MapObject *pCur = ObjectOptions::getObjectNamed(m_pickBuildingDlg.getPickedUnit());
199 loc.z = 0;
200 if (pCur) {
201 // Display the transparent version of this object.
202 p3View->setObjTracking(pCur, loc, 0, true);
203 } else {
204 // Don't display anything.
205 p3View->setObjTracking(NULL, loc, 0, false);
206 }
207 return;
208 }
209 p3View->setObjTracking(NULL, cpt, 0, false);
210
211 if (m == TRACK_NONE) {
212 // See if the cursor is over an object.
213 BuildListInfo *pInfo = pView->pickedBuildObjectInView(viewPt);
214 m_mouseUpMove = false;
215 m_mouseUpRotate = false;
216 if (pInfo) {
217 Coord3D center = *pInfo->getLocation();
218 center.x -= cpt.x;
219 center.y -= cpt.y;
220 center.z = 0;
221 Real len = center.length();
222 // Check and see if we are within 1 cell size of the center.
223 if (pInfo->isSelected() && len>0.5f*MAP_XY_FACTOR && len < 1.5f*MAP_XY_FACTOR) {
224 m_mouseUpRotate = true;
225 } else {
226 m_mouseUpMove = true;
227 }
228 }
229 return; // setCursor will use the value of m_mouseUpRotate. jba.
230 }
231
232 if (m != TRACK_L) return;
233 if (!m_moving || !m_curObject) return;
234
235 Coord3D loc = *m_curObject->getLocation();
236 if (m_rotating) {
237 Real angle = ObjectTool::calcAngle(m_downPt3d, cpt, pView);
238 m_curObject->setAngle(angle);
239 } else {
240 pView->snapPoint(&cpt);
241 Real xOffset = (cpt.x-m_prevPt3d.x);
242 Real yOffset = (cpt.y-m_prevPt3d.y);
243 loc.x += xOffset;
244 loc.y += yOffset;
245 m_curObject->setLocation(loc);
246 }
247 p3View->invalBuildListItemInView(m_curObject);
248 m_prevPt3d = cpt;
249
250}

255void BuildListTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
256{
257 if (m != TRACK_L) return;
258 if (!isDoingAdd()) {
259 // Update the cursor.
260 mouseMoved(TRACK_NONE, viewPt, pView, pDoc);
261 setCursor();
262 return;
263 }
264 // always check hysteresis in view coords.
265 enum {HYSTERESIS = 3};
266 Bool justAClick = (abs(viewPt.x - m_downPt2d.x)<HYSTERESIS || abs(viewPt.x - m_downPt2d.x)<HYSTERESIS);
267
268 Coord3D cpt;
269 pView->viewToDocCoords(viewPt, &cpt, false); // Don't constrain.
270
271 Coord3D loc = m_downPt3d;
272 pView->snapPoint(&loc);
273 loc.z = ObjectOptions::getCurObjectHeight();
274 Real angle = justAClick ? 0 : ObjectTool::calcAngle(loc, cpt, pView);
275 if (!m_pickBuildingDlg.getPickedUnit().isEmpty()) {
276 BuildList::addBuilding(loc, angle, m_pickBuildingDlg.getPickedUnit());
277 }
278 //CMainFrame::GetMainFrame()->showOptionsDialog(IDD_BUILD_LIST_PANEL);
279}
