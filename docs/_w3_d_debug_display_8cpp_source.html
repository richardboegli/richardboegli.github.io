134void W3DDebugDisplay::drawText( Int x, Int y, Char *text )
135{
136 if ( m_font == NULL || m_displayString == NULL )
137 {
138 return ;
139 }
140
141 ::Color textColor = GameMakeColor( 255, 255, 255, 255 );
142 ::Color dropColor = GameMakeColor( 0, 0, 0, 255 );
143
144 UnicodeString unicode;
145
146 unicode.translate( AsciiString( text ) );
147 m_displayString->setText( unicode );
148 m_displayString->draw( x*m_fontWidth, 13 + y*m_fontHeight, textColor, dropColor );
149}
