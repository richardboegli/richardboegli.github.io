134UpdateSleepTime MobMemberSlavedUpdate::update( void )
135{
137
138 const MobMemberSlavedUpdateModuleData* data = getMobMemberSlavedUpdateModuleData();
139 Object *me = getObject();
140 if( !me )
141 {
142 return UPDATE_SLEEP_NONE;
143 }
144
145 Object *master = TheGameLogic->findObjectByID( m_slaver );
146 if( master == NULL )
147 {
148 stopSlavedEffects();
149
150 //TheGameLogic->destroyObject( me );
151 me->kill();
152 return UPDATE_SLEEP_NONE; // you cannot return SLEEP_FOREVER unless you make yourself sleepy...
153 }
154
155 AIUpdateInterface *myAI = me->getAIUpdateInterface();
156 AIUpdateInterface *masterAI = master->getAIUpdateInterface();
157 if( ! myAI || ! masterAI)
158 {
159 return UPDATE_SLEEP_NONE;
160 }
161
162 Drawable *myDraw = me->getDrawable();
163 Drawable *masterDraw = master->getDrawable();
164 if ( ! myDraw || ! masterDraw)
165 {
166 return UPDATE_SLEEP_NONE;
167 }
168
169// myDraw->colorTint( &m_personalColor );
170
171
172 const ModelConditionFlags flags = myDraw->getModelConditionFlags();
173 if (flags.anyIntersectionWith(MAKE_MODELCONDITION_MASK(MODELCONDITION_WEAPONSET_PLAYER_UPGRADE)))
174 {
175 ModelConditionFlags clearFlags;
176 clearFlags.clear();
177 clearFlags.set( MODELCONDITION_RELOADING_A );
178 clearFlags.set( MODELCONDITION_BETWEEN_FIRING_SHOTS_A );
179 clearFlags.set( MODELCONDITION_PREATTACK_A );
180 clearFlags.set( MODELCONDITION_FIRING_A );
181 clearFlags.set( MODELCONDITION_USING_WEAPON_A );
182 myDraw->clearModelConditionFlags( clearFlags );
183 }
184
185
186
187 if ( ++m_framesToWait < 16)
188 return UPDATE_SLEEP_NONE;
189
190 m_framesToWait = 0;
191
192
193 Locomotor *locomotor = myAI->getCurLocomotor();
194 if( !locomotor )
195 {
196 return UPDATE_SLEEP_NONE;
197 }
198
199 Object *victim = getObject()->getAIUpdateInterface()->getCurrentVictim();
200 Object *masterVictim = master->getAIUpdateInterface()->getCurrentVictim();
201
202 if (masterVictim)
203 {
204 m_primaryVictimID = masterVictim->getID();
205 }
206
207 Object *primaryVictim = TheGameLogic->findObjectByID(m_primaryVictimID);
208
209
210 //now, we don't know if master is standing still or going somewhere, so
211 Real masterPathDistToGoal = masterAI->getLocomotorDistanceToGoal();
212 Real myPathDistToGoal = myAI->getLocomotorDistanceToGoal();
213
214
215 Real catchUpRadiusSquared = ThePartitionManager->getDistanceSquared( me, master, FROM_CENTER_3D );
216 // I'm too far from the nexus... I need to catch up now!
217 if( catchUpRadiusSquared > sqr( data->m_mustCatchUpRadius ) )
218 {
219
220 if ( masterAI->isMoving() )// master is on the move
221 {
222 if ( masterPathDistToGoal > myPathDistToGoal ) // I'm getting ahead of master, need to slow down
223 myAI->chooseLocomotorSet(LOCOMOTORSET_WANDER);
224 else
225 myAI->chooseLocomotorSet(LOCOMOTORSET_PANIC); // I'm lagging, so I need to snap to it!
226
227 Coord3D nuPos = *masterAI->getGoalPosition();
228
229 if ( nuPos.length() < 1.0f ) //if a nasty error has sent me to map origin
230 {
231 myAI->aiMoveToPosition( master->getPosition(), CMD_FROM_AI ); // NASTY BEEHIVE EFFECT
232 }
233 else
234 {
235 Coord3D goalDelta = *myAI->getGoalPosition();
236 goalDelta.sub( &nuPos );
237
238 if ( goalDelta.length() > 5.0f * PATHFIND_CELL_SIZE_F )// only if I am not headed there already
239 {
240 myAI->aiMoveToPosition( &nuPos, CMD_FROM_AI ); // Whither thou goest... THis causes the mob to reconverge
241 }
242 }
243
244 // on the fly, instead of doubling back to reconverge
245 }
246 else // master is still, so let's re group in a hurry
247 {
248 myAI->chooseLocomotorSet(LOCOMOTORSET_PANIC);
249 myAI->aiMoveToPosition( master->getPosition(), CMD_FROM_AI );
250
251 }
252
253
254
255 if (catchUpRadiusSquared > sqr( data->m_mustCatchUpRadius * 3))// I am critically far, now!
256 {
257 ++ m_catchUpCrisisTimer; // I'm way too far from the nexus this frame
258
259 if ( m_catchUpCrisisTimer > data->m_catchUpCrisisBailTime)
260 {
261 me->kill();
262 return UPDATE_SLEEP_NONE;
263
264 // Here is the rethink:
265 // If the nexus has outrun me to the target by so much, //
266 // lets make the nexus come to me, and try to return to where I find it
267// Coord3D masterPosition = *master->getPosition();
268// master->setPosition( me->getPosition() );
269// if ( ! masterAI->isMoving() )
270// {
271// masterAI->aiMoveToPosition( &masterPosition, CMD_FROM_AI );
272// }
273// m_catchUpCrisisTimer = 0;
274 }
275 else if ( m_catchUpCrisisTimer > data->m_catchUpCrisisBailTime/3 )
276 {
277 myAI->aiMoveToPosition( master->getPosition(), CMD_FROM_AI ); // NASTY BEEHIVE EFFECT
278 }
279 }
280
281
282 }
283 else if ( myAI->isMoving() ) // we're all on a trip, together
284 {
285
286 m_catchUpCrisisTimer = 0; // I'm not too far from the nexus this frame
287
288 Int seed = GameLogicRandomValue( 0, 10 );
289 if ( seed == 1 )
290 myAI->chooseLocomotorSet(LOCOMOTORSET_WANDER);
291 else if ( seed == 2 )
292 myAI->chooseLocomotorSet(LOCOMOTORSET_PANIC);
293 else if ( seed == 3 )
294 myAI->chooseLocomotorSet(LOCOMOTORSET_NORMAL);
295 // else if ( seed >= 5 ) // go towards mommy's goal
296 // {
297 // Coord3D destination = *me->getPosition();
298 // TheAI->pathfinder()->adjustToPossibleDestination(me, myAI->getLocomotorSet(), &destination);
299 // myAI->aiMoveToPosition( &destination, CMD_FROM_AI ); // reconverge
300 // }
301
302 }
303 else // give me something to do while I'm standing here...
304 {
305 m_catchUpCrisisTimer = 0; // I'm not too far from the nexus this frame
306
307 SpawnBehaviorInterface *spawnerBehavior = master->getSpawnBehaviorInterface();
308
309 if ( spawnerBehavior ) // if I have a mommy
310 {
311
312 if ( masterAI->isIdle() ) // if controlling player has pressed stop, we stop! That's it!
313 {
314 myAI->aiIdle(CMD_FROM_AI);
315 primaryVictim = NULL;
316 m_primaryVictimID = INVALID_ID;
317 return UPDATE_SLEEP_NONE;
318 }
319
320 if ( spawnerBehavior->maySpawnSelfTaskAI( m_squirrellinessRatio ) ) // if mommy says it is okay
321 {
322 if ( myAI->getLastCommandSource() != CMD_FROM_AI ) // I may have been told to attack directly more recently
323 {
324 Object *newTarget = myAI->getNextMoodTarget( FALSE, FALSE );
325 if ( newTarget && ( newTarget != victim) ) // if there is someone else around to attack
326 {
327 victim = newTarget;
328 myAI->aiAttackObject( newTarget, 999, CMD_FROM_AI ); // go ahead and do it
329 m_isSelfTasking = TRUE;
330 }
331 }
332
333 }
334
335 if ( ! victim ) // If I still don't have anyone to shoot at
336 {
337 if ( primaryVictim ) // I remember the last target
338 {
339 myAI->aiAttackObject( primaryVictim, 999, CMD_FROM_AI );
340 }
341 else if( ! masterAI->isAttacking())// there Is no previous target and master isn't attacking
342 {
344 }
345 m_isSelfTasking = FALSE;
346 }
347 }
348 else
349 {
350 DEBUG_ASSERTCRASH(( spawnerBehavior != NULL ),("Hey!, why for this mob member got no spawner? MLorenzen"));
351 }
352 }
353
354 return UPDATE_SLEEP_NONE;
355}

360//-------------------------------------------------------------------------------------------------
361void MobMemberSlavedUpdate::doCatchUpLogic( Coord3D *pos )
362{
363 Coord3D nuPos;
364 const MobMemberSlavedUpdateModuleData* data = getMobMemberSlavedUpdateModuleData();
365 // recalc where we want to be if we wander around
366 Real randomDirection = GameLogicRandomValue( 0, 2*PI );
367 Real randomRadius = GameLogicRandomValue( 0, data->m_noNeedToCatchUpRadius );
368 nuPos.set(pos);
369 nuPos.x += randomRadius * Cos( randomDirection );
370 nuPos.y += randomRadius * Sin( randomDirection );
371 nuPos.z = TheTerrainLogic->getGroundHeight( nuPos.x, nuPos.y );
372
373 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
374 if( ai )
375 {
376 ai->aiMoveToPosition( &nuPos, CMD_FROM_AI );
377 }
378
379 setMobState(MOB_STATE_CATCHING_UP);
380
381}
