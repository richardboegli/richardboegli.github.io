81static WW3DFormat findFormat(const WW3DFormat formats[])
82{
83 for( Int i = 0; formats[ i ] != WW3D_FORMAT_UNKNOWN; i++ )
84 {
85
86 if( DX8Wrapper::Get_Current_Caps()->Support_Texture_Format( formats[ i ] ) )
87 {
88
89 return formats[ i ];
90
91 } // end if
92
93 } // end for i
94 DEBUG_CRASH(("WW3DRadar: No appropriate texture format\n") );
95 return WW3D_FORMAT_UNKNOWN;

178void W3DRadar::reconstructViewBox( void )
179{
180 Coord3D world[ 4 ];
181 ICoord2D radar[ 4 ];
182 Int i;
183
184 // get the 4 points of the view corners in the 3D world at the average Z height in the map
185 TheTacticalView->getScreenCornerWorldPointsAtZ( &world[ 0 ],
186 &world[ 1 ],
187 &world[ 2 ],
188 &world[ 3 ],
189 getTerrainAverageZ() );
190
191 // convert each of the 4 points in the world to radar cell positions
192 for( i = 0; i < 4; i++ )
193 {
194
195 // first convert to radar cells
196 radar[ i ].x = world[ i ].x / (m_mapExtent.width() / RADAR_CELL_WIDTH);
197 radar[ i ].y = world[ i ].y / (m_mapExtent.height() / RADAR_CELL_HEIGHT);
198
199 //
200 // store these points in the view box array which contains a first position
201 // of (0,0) and then offsets for each additional entry point
202 //
203 if( i == 0 )
204 {
205
206 m_viewBox[ i ].x = 0;
207 m_viewBox[ i ].y = 0;
208
209 } // end if
210 else
211 {
212
213 m_viewBox[ i ].x = radar[ i ].x - radar[ i - 1 ].x;
214 m_viewBox[ i ].y = radar[ i ].y - radar[ i - 1 ].y;
215
216 } // end else
217
218 } // end for i
219
220 //
221 // save the camera settings for this view box, we will need to make it again only
222 // if some of these change
223 //
224 m_viewAngle = TheTacticalView->getAngle();
225 Coord3D pos;
226 TheTacticalView->getPosition( &pos );
227 m_viewZoom = TheTacticalView->getZoom();
228 m_reconstructViewBox = FALSE;
229
230} // end reconstructViewBox

235void W3DRadar::radarToPixel( const ICoord2D *radar, ICoord2D *pixel,
236 Int radarUpperLeftX, Int radarUpperLeftY,
237 Int radarWidth, Int radarHeight )
238{
239
240 // sanity
241 if( radar == NULL || pixel == NULL )
242 return;
243
244 pixel->x = (radar->x * radarWidth / RADAR_CELL_WIDTH) + radarUpperLeftX;
245 // note the "inverted" y here to orient the way our world looks with +x=right and -y=down
246 pixel->y = ((RADAR_CELL_HEIGHT - 1 - radar->y) * radarHeight / RADAR_CELL_HEIGHT) + radarUpperLeftY;
247
248} // end radarToPixel

284void W3DRadar::drawViewBox( Int pixelX, Int pixelY, Int width, Int height )
285{
286 ICoord2D ulScreen;
287 ICoord2D ulRadar;
288 Coord3D ulWorld;
289 ICoord2D ulStart = { 0, 0 };
290 ICoord2D start, end;
291 ICoord2D clipStart, clipEnd;
292 Real lineWidth = 1.0f;
293 Color topColor = GameMakeColor( 225, 225, 0, 255 );
294 Color bottomColor = GameMakeColor( 158, 158, 0, 255 );
295
296 //
297 // setup the clipping region ... note that this clipping region is not over just the
298 // radar image area ... it's in the WHOLE window available for the radar
299 //
300 IRegion2D clipRegion;
301 ICoord2D radarWindowSize, radarWindowScreenPos;
302 m_radarWindow->winGetSize( &radarWindowSize.x, &radarWindowSize.y );
303 m_radarWindow->winGetScreenPosition( &radarWindowScreenPos.x, &radarWindowScreenPos.y );
304 clipRegion.lo.x = radarWindowScreenPos.x;
305 clipRegion.lo.y = radarWindowScreenPos.y;
306 clipRegion.hi.x = radarWindowScreenPos.x + radarWindowSize.x;
307 clipRegion.hi.y = radarWindowScreenPos.y + radarWindowSize.y;
308
309 // convert top left of screen into world position
310 TheTacticalView->getOrigin( &ulScreen.x, &ulScreen.y );
311 TheTacticalView->screenToWorldAtZ( &ulScreen, &ulWorld, getTerrainAverageZ() );
312
313 // convert world to radar coords
314 ulRadar.x = ulWorld.x / (m_mapExtent.width() / RADAR_CELL_WIDTH);
315 ulRadar.y = ulWorld.y / (m_mapExtent.height() / RADAR_CELL_HEIGHT);
316
317 //
318 // convert radar point to actual pixel coords on the screen, shifted
319 // into position on the radar for where the radar is drawn and the size of the
320 // area that the radar is drawn in
321 //
322 radarToPixel( &ulRadar, &ulStart, pixelX, pixelY, width, height );
323
324 //
325 // using our view box offset array, convert each of those radar cell offset points
326 // into screen pixels and draw the box. The view box array is setup with the
327 // first index containing (0,0) (the point we just converted in theory), with cell
328 // offsets to each of the other corners in the following order
329 // (upper left, upper right, lower right, lower left)
330 //
331 ICoord2D radar;
332
333 // top line
334 start = ulStart;
335 radar.x = ulRadar.x + m_viewBox[ 1 ].x;
336 radar.y = ulRadar.y + m_viewBox[ 1 ].y;
337 radarToPixel( &radar, &end, pixelX, pixelY, width, height );
338 if( ClipLine2D( &start, &end, &clipStart, &clipEnd, &clipRegion ) )
339 TheDisplay->drawLine( clipStart.x, clipStart.y, clipEnd.x, clipEnd.y,
340 lineWidth, topColor );
341
342 // right line
343 start = end;
344 radar.x += m_viewBox[ 2 ].x;
345 radar.y += m_viewBox[ 2 ].y;
346 radarToPixel( &radar, &end, pixelX, pixelY, width, height );
347 if( ClipLine2D( &start, &end, &clipStart, &clipEnd, &clipRegion ) )
348 TheDisplay->drawLine( clipStart.x, clipStart.y, clipEnd.x, clipEnd.y,
349 lineWidth, topColor, bottomColor );
350
351 // bottom line
352 start = end;
353 radar.x += m_viewBox[ 3 ].x;
354 radar.y += m_viewBox[ 3 ].y;
355 radarToPixel( &radar, &end, pixelX, pixelY, width, height );
356 if( ClipLine2D( &start, &end, &clipStart, &clipEnd, &clipRegion ) )
357 TheDisplay->drawLine( clipStart.x, clipStart.y, clipEnd.x, clipEnd.y,
358 lineWidth, bottomColor );
359
360 // left line
361 start = end;
362 end = ulStart;
363 if( ClipLine2D( &start, &end, &clipStart, &clipEnd, &clipRegion ) )
364 TheDisplay->drawLine( clipStart.x, clipStart.y, clipEnd.x, clipEnd.y,
365 lineWidth, bottomColor, topColor );
366
367} // end drawViewBox

371void W3DRadar::drawSingleBeaconEvent( Int pixelX, Int pixelY, Int width, Int height, Int index )
372{
373 RadarEvent *event = &(m_event[index]);
374 ICoord2D tri[ 3 ];
375 ICoord2D start, end;
376 Real angle, addAngle;
377 Color startColor, endColor;
378 Real lineWidth = 1.0f;
379 UnsignedInt currentFrame = TheGameLogic->getFrame();
380 UnsignedInt frameDiff; // frames the event has been alive for
381 Real maxEventSize = width / 10.0f; // max size of the event marker
382 Int minEventSize = 6; // min size of the event marker
383 Int eventSize; // current size of a marker to draw
384 const Real TIME_FROM_FULL_SIZE_TO_SMALL_SIZE = LOGICFRAMES_PER_SECOND * 1.5;
385 Real totalAnglesToSpin = 2.0f * PI;
386 UnsignedByte r, g, b, a;
387
388 // setup screen clipping region
389 IRegion2D clipRegion;
390 clipRegion.lo.x = pixelX;
391 clipRegion.lo.y = pixelY;
392 clipRegion.hi.x = pixelX + width;
393 clipRegion.hi.y = pixelY + height;
394
395 // get the difference in frame from the current frame to the frame we were created on
396 frameDiff = currentFrame - event->createFrame;
397
398 // compute the size of the event marker, it is largest when it starts and smallest at the end
399 eventSize = REAL_TO_INT( maxEventSize * ( 1.0f - frameDiff / TIME_FROM_FULL_SIZE_TO_SMALL_SIZE) );;
400
401 // we never let the event size get too small
402 if( eventSize < minEventSize )
403 eventSize = minEventSize;
404
405 // compute how much "angle" we will add to each point to make it rotate as it's getting small
406 addAngle = -totalAnglesToSpin * (frameDiff / TIME_FROM_FULL_SIZE_TO_SMALL_SIZE);
407
408 // create a triangle around the event
409 angle = 0.0f - addAngle;
410 tri[ 0 ].x = REAL_TO_INT( (DOUBLE_TO_REAL( Cos( angle ) ) * eventSize) + event->radarLoc.x );
411 tri[ 0 ].y = REAL_TO_INT( (DOUBLE_TO_REAL( Sin( angle ) ) * eventSize) + event->radarLoc.y );
412
413 angle = 2.0f * PI / 3.0f - addAngle;
414 tri[ 1 ].x = REAL_TO_INT( (DOUBLE_TO_REAL( Cos( angle ) ) * eventSize) + event->radarLoc.x );
415 tri[ 1 ].y = REAL_TO_INT( (DOUBLE_TO_REAL( Sin( angle ) ) * eventSize) + event->radarLoc.y );
416
417 angle = -2.0f * PI / 3.0f - addAngle;
418 tri[ 2 ].x = REAL_TO_INT( (DOUBLE_TO_REAL( Cos( angle ) ) * eventSize) + event->radarLoc.x );
419 tri[ 2 ].y = REAL_TO_INT( (DOUBLE_TO_REAL( Sin( angle ) ) * eventSize) + event->radarLoc.y );
420
421 // translate radar coords to screen coords
422 radarToPixel( &tri[ 0 ], &tri[ 0 ], pixelX, pixelY, width, height );
423 radarToPixel( &tri[ 1 ], &tri[ 1 ], pixelX, pixelY, width, height );
424 radarToPixel( &tri[ 2 ], &tri[ 2 ], pixelX, pixelY, width, height );
425
426 //
427 // make the colors we're going to use, when we're at our smallest size we will start to
428 // fade the alpha away to transparent so that at our lifetime frame we are completely gone
429 //
430
431 // color 1 ------------------
432 r = event->color1.red;
433 g = event->color1.green;
434 b = event->color1.blue;
435 a = event->color1.alpha;
436 if( currentFrame > event->fadeFrame )
437 {
438
439 a = REAL_TO_UNSIGNEDBYTE( (Real)a * (1.0f - (Real)(currentFrame - event->fadeFrame) /
440 (Real)(event->dieFrame - event->fadeFrame) ) );
441
442 } // end if
443 startColor = GameMakeColor( r, g, b, a );
444
445 // color 2 ------------------
446 r = event->color2.red;
447 g = event->color2.green;
448 b = event->color2.blue;
449 a = event->color2.alpha;
450 if( currentFrame > event->fadeFrame )
451 {
452
453 a = REAL_TO_UNSIGNEDBYTE( (Real)a * (1.0f - (Real)(currentFrame - event->fadeFrame) /
454 (Real)(event->dieFrame - event->fadeFrame) ) );
455
456 } // end if
457 endColor = GameMakeColor( r, g, b, a );
458
459 // draw the lines
460 if( ClipLine2D( &tri[ 0 ], &tri[ 1 ], &start, &end, &clipRegion ) )
461 TheDisplay->drawLine( start.x, start.y, end.x, end.y, lineWidth, startColor, endColor );
462 if( ClipLine2D( &tri[ 1 ], &tri[ 2 ], &start, &end, &clipRegion ) )
463 TheDisplay->drawLine( start.x, start.y, end.x, end.y, lineWidth, startColor, endColor );
464 if( ClipLine2D( &tri[ 2 ], &tri[ 0 ], &start, &end, &clipRegion ) )
465 TheDisplay->drawLine( start.x, start.y, end.x, end.y, lineWidth, startColor, endColor );
466}

590

592 drawSingleBeaconEvent( pixelX, pixelY, width, height, i );
593 else
594 drawSingleGenericEvent( pixelX, pixelY, width, height, i );
595
596 } // end if
597
598 } // end for i

620void W3DRadar::renderObjectList( const RadarObject *listHead, TextureClass *texture, Bool calcHero )
621{
622
623 // sanity
624 if( listHead == NULL || texture == NULL )
625 return;
626
627 // get surface for texture to render into
628 SurfaceClass *surface = texture->Get_Surface_Level();
629
630 // loop through all objects and draw
631 ICoord2D radarPoint;
632
633 Player *player = ThePlayerList->getLocalPlayer();
634 Int playerIndex=0;
635 if (player)
636 playerIndex=player->getPlayerIndex();
637
638 if( calcHero )
639 {
640 // clear all entries from the cached hero object list
641 m_cachedHeroPosList.clear();
642 }
643
644 for( const RadarObject *rObj = listHead; rObj; rObj = rObj->friend_getNext() )
645 {
646
647 if (rObj->isTemporarilyHidden())
648 continue;
649
650 // get object
651 const Object *obj = rObj->friend_getObject();
652
653 // cache hero object positions for drawing in icon layer
654 if( calcHero && obj->isHero() )
655 {
656 m_cachedHeroPosList.push_back(obj->getPosition());
657 }
658 Bool skip = FALSE;
659
660 // check for shrouded status
661 if (obj->getShroudedStatus(playerIndex) > OBJECTSHROUD_PARTIAL_CLEAR)
662 skip = TRUE; //object is fogged or shrouded, don't render it.
663
664 //
665 // objects with a local only unit priority will only appear on the radar if they
666 // are controlled by the local player, or if the local player is an observer (cause
667 // they are godlike and can see everything)
668 //
669
670
671 if( obj->getRadarPriority() == RADAR_PRIORITY_LOCAL_UNIT_ONLY &&
672 obj->getControllingPlayer() != ThePlayerList->getLocalPlayer() &&
673 ThePlayerList->getLocalPlayer()->isPlayerActive() )
674 skip = TRUE;
675
676 // get object position
677 const Coord3D *pos = obj->getPosition();
678
679 // compute object position as a radar blip
680 radarPoint.x = pos->x / (m_mapExtent.width() / RADAR_CELL_WIDTH);
681 radarPoint.y = pos->y / (m_mapExtent.height() / RADAR_CELL_HEIGHT);
682
683
684 if ( skip )
685 continue;
686
687 // get the color we're going to draw in
688 Color c = rObj->getColor();
689
690
691
692 // adjust the alpha for stealth units so they "fade/blink" on the radar for the controller
693 // if( obj->getRadarPriority() == RADAR_PRIORITY_LOCAL_UNIT_ONLY )
694 // ML-- What the heck is this? local-only and neutral-observier-viewed units are stealthy?? Since when?
695 // Now it twinkles for any stealthed object, whether locally controlled or neutral-observier-viewed
696 if( obj->testStatus( OBJECT_STATUS_STEALTHED ) )
697 {
698 if ( ThePlayerList->getLocalPlayer()->getRelationship(obj->getTeam()) == ENEMIES )
699 if( !obj->testStatus( OBJECT_STATUS_DETECTED ) && !obj->testStatus( OBJECT_STATUS_DISGUISED ) )
700 skip = TRUE;
701
702 UnsignedByte r, g, b, a;
703 GameGetColorComponents( c, &r, &g, &b, &a );
704
705 const UnsignedInt framesForTransition = LOGICFRAMES_PER_SECOND;
706 const UnsignedByte minAlpha = 32;
707
708 if (skip)
709 continue;
710
711 Real alphaScale = INT_TO_REAL(TheGameLogic->getFrame() % framesForTransition) / (framesForTransition / 2.0f);
712 if( alphaScale > 0.0f )
713 a = REAL_TO_UNSIGNEDBYTE( ((alphaScale - 1.0f) * (255.0f - minAlpha)) + minAlpha );
714 else
715 a = REAL_TO_UNSIGNEDBYTE( (alphaScale * (255.0f - minAlpha)) + minAlpha );
716 c = GameMakeColor( r, g, b, a );
717
718 } // end if
719
720
721
722
723 // draw the blip, but make sure the points are legal
724 if( legalRadarPoint( radarPoint.x, radarPoint.y ) )
725 surface->DrawPixel( radarPoint.x, radarPoint.y, c );
726
727 radarPoint.y++;
728 if( legalRadarPoint( radarPoint.x, radarPoint.y ) )
729 surface->DrawPixel( radarPoint.x, radarPoint.y, c );
730
731 radarPoint.x++;
732 if( legalRadarPoint( radarPoint.x, radarPoint.y ) )
733 surface->DrawPixel( radarPoint.x, radarPoint.y, c );
734
735 radarPoint.y--;
736 if( legalRadarPoint( radarPoint.x, radarPoint.y ) )
737 surface->DrawPixel( radarPoint.x, radarPoint.y, c );
738
739 } // end for
740 REF_PTR_RELEASE(surface);
741
742} // end renderObjectList

749//-------------------------------------------------------------------------------------------------
750void W3DRadar::interpolateColorForHeight( RGBColor *color,
751 Real height,
752 Real hiZ,
753 Real midZ,
754 Real loZ )
755{
756 const Real howBright = 0.95f; // bigger is brighter (0.0 to 1.0)
757 const Real howDark = 0.60f; // bigger is darker (0.0 to 1.0)
758
759 // sanity on map height (flat maps bomb)
760 if (hiZ == midZ)
761 hiZ = midZ+0.1f;
762 if (midZ == loZ)
763 loZ = midZ-0.1f;
764 if (hiZ == loZ)
765 hiZ = loZ+0.2f;
766
767 Real t;
768 RGBColor colorTarget;
769
770 // if "over" the middle height, interpolate lighter
771 if( height >= midZ )
772 {
773
774 // how far are we from the middleZ towards the hi Z
775 t = (height - midZ) / (hiZ - midZ);
776
777 // compute what our "lightest" color possible we want to use is
778 colorTarget.red = color->red + (1.0f - color->red) * howBright;
779 colorTarget.green = color->green + (1.0f - color->green) * howBright;
780 colorTarget.blue = color->blue + (1.0f - color->blue) * howBright;
781
782 } // end if
783 else // interpolate darker
784 {
785
786 // how far are we from the middleZ towards the low Z
787 t = (midZ - height) / (midZ - loZ);
788
789 // compute what the "darkest" color possible we want to use is
790 colorTarget.red = color->red + (0.0f - color->red) * howDark;
791 colorTarget.green = color->green + (0.0f - color->green) * howDark;
792 colorTarget.blue = color->blue + (0.0f - color->blue) * howDark;
793
794 } // end else
795
796 // interpolate toward the target color
797 color->red = color->red + (colorTarget.red - color->red) * t;
798 color->green = color->green + (colorTarget.green - color->green) * t;
799 color->blue = color->blue + (colorTarget.blue - color->blue) * t;
800
801 // keep the color real
802 if( color->red < 0.0f )
803 color->red = 0.0f;
804 if( color->red > 1.0f )
805 color->red = 1.0f;
806 if( color->green < 0.0f )
807 color->green = 0.0f;
808 if( color->green > 1.0f )
809 color->green = 1.0f;
810 if( color->blue < 0.0f )
811 color->blue = 0.0f;
812 if( color->blue > 1.0f )
813 color->blue = 1.0f;
814
815} // end interpolateColorForHeight

1021void W3DRadar::buildTerrainTexture( TerrainLogic *terrain )
1022{
1023 SurfaceClass *surface;
1024 RGBColor waterColor;
1025
1026 // we will want to reconstruct our new view box now
1027 m_reconstructViewBox = TRUE;
1028
1029 // setup our water color
1030 waterColor.red = TheWaterTransparency->m_radarColor.red;
1031 waterColor.green = TheWaterTransparency->m_radarColor.green;
1032 waterColor.blue = TheWaterTransparency->m_radarColor.blue;
1033
1034 // get the terrain surface to draw in
1035 surface = m_terrainTexture->Get_Surface_Level();
1036 DEBUG_ASSERTCRASH( surface, ("W3DRadar: Can't get surface for terrain texture\n") );
1037
1038 // build the terrain
1039 RGBColor sampleColor;
1040 RGBColor color;
1041 Int i, j, samples;
1042 Int x, y;
1043 ICoord2D radarPoint;
1044 Coord3D worldPoint;
1045 Bridge *bridge;
1046 for( y = 0; y < m_textureHeight; y++ )
1047 {
1048
1049 for( x = 0; x < m_textureWidth; x++ )
1050 {
1051
1052 // what point are we inspecting
1053 radarPoint.x = x;
1054 radarPoint.y = y;
1055 radarToWorld2D( &radarPoint, &worldPoint );
1056
1057 // check to see if this point is part of a working bridge
1058 Bool workingBridge = FALSE;
1059 bridge = TheTerrainLogic->findBridgeAt( &worldPoint );
1060 if( bridge != NULL )
1061 {
1062 Object *obj = TheGameLogic->findObjectByID( bridge->peekBridgeInfo()->bridgeObjectID );
1063
1064 if( obj )
1065 {
1066 BodyModuleInterface *body = obj->getBodyModule();
1067
1068 if( body->getDamageState() != BODY_RUBBLE )
1069 workingBridge = TRUE;
1070
1071 } // end if
1072
1073 } // end if
1074
1075 // create a color based on the Z height of the map
1076 Real waterZ;
1077 if( workingBridge == FALSE && terrain->isUnderwater( worldPoint.x, worldPoint.y, &waterZ ) )
1078 {
1079 const Int waterSamplesAway = 1; // how many "tiles" from the center tile we will sample away
1080 // to average a color for the tile color
1081
1082 sampleColor.red = sampleColor.green = sampleColor.blue = 0.0f;
1083 samples = 0;
1084
1085 for( j = y - waterSamplesAway; j <= y + waterSamplesAway; j++ )
1086 {
1087
1088 if( j >= 0 && j < m_textureHeight )
1089 {
1090
1091 for( i = x - waterSamplesAway; i <= x + waterSamplesAway; i++ )
1092 {
1093
1094 if( i >= 0 && i < m_textureWidth )
1095 {
1096
1097 // the the world point we are concerned with
1098 radarPoint.x = i;
1099 radarPoint.y = j;
1100 radarToWorld2D( &radarPoint, &worldPoint );
1101
1102 // get color for this Z and add to our sample color
1103 Real underwaterZ;
1104 if( terrain->isUnderwater( worldPoint.x, worldPoint.y, NULL, &underwaterZ ) )
1105 {
1106 // this is our "color" for water
1107 color = waterColor;
1108
1109 // interpolate the water color for height in the water table
1110 interpolateColorForHeight( &color, underwaterZ, waterZ,
1111 waterZ,
1112 m_mapExtent.lo.z );
1113
1114 // add color to our samples
1115 sampleColor.red += color.red;
1116 sampleColor.green += color.green;
1117 sampleColor.blue += color.blue;
1118 samples++;
1119
1120 } // end if
1121
1122 } // end if
1123
1124 } // end for i
1125
1126 } // end if
1127
1128 } // end for j
1129
1130 // prevent divide by zeros
1131 if( samples == 0 )
1132 samples = 1;
1133
1134 // set the color to an average of the colors read
1135 color.red = sampleColor.red / (Real)samples;
1136 color.green = sampleColor.green / (Real)samples;
1137 color.blue = sampleColor.blue / (Real)samples;
1138
1139 } // end if
1140 else // regular terrain ...
1141 {
1142 const Int samplesAway = 1; // how many "tiles" from the center tile we will sample away
1143 // to average a color for the tile color
1144
1145 sampleColor.red = sampleColor.green = sampleColor.blue = 0.0f;
1146 samples = 0;
1147
1148 for( j = y - samplesAway; j <= y + samplesAway; j++ )
1149 {
1150
1151 if( j >= 0 && j < m_textureHeight )
1152 {
1153
1154 for( i = x - samplesAway; i <= x + samplesAway; i++ )
1155 {
1156
1157 if( i >= 0 && i < m_textureWidth )
1158 {
1159
1160 // the the world point we are concerned with
1161 radarPoint.x = i;
1162 radarPoint.y = j;
1163 radarToWorld( &radarPoint, &worldPoint );
1164
1165 // get the color we're going to use here
1166 if( workingBridge )
1167 {
1168 AsciiString bridgeTName = bridge->getBridgeTemplateName();
1169 TerrainRoadType *bridgeTemplate = TheTerrainRoads->findBridge( bridgeTName );
1170
1171 // sanity
1172 DEBUG_ASSERTCRASH( bridgeTemplate, ("W3DRadar::buildTerrainTexture - Can't find bridge template for '%s'\n", bridgeTName.str()) );
1173
1174 // use bridge color
1175 if ( bridgeTemplate )
1176 color = bridgeTemplate->getRadarColor();
1177 else
1178 color.setFromInt(0xffffffff);
1179 //
1180 // we won't use the height of the terrain at this sample point, we will
1181 // instead use the height for the entire bridge
1182 //
1183 Real bridgeHeight = (bridge->peekBridgeInfo()->fromLeft.z +
1184 bridge->peekBridgeInfo()->fromRight.z +
1185 bridge->peekBridgeInfo()->toLeft.z +
1186 bridge->peekBridgeInfo()->toRight.z) / 4.0f;
1187
1188 // interpolate the color, but use the bridge height, not the terrain height
1189 interpolateColorForHeight( &color, bridgeHeight,
1190 getTerrainAverageZ(),
1191 m_mapExtent.hi.z, m_mapExtent.lo.z );
1192
1193 } // end if
1194 else
1195 {
1196
1197 // get the color at this point
1198 TheTerrainVisual->getTerrainColorAt( worldPoint.x, worldPoint.y, &color );
1199
1200 // interpolate the color for height
1201 interpolateColorForHeight( &color, worldPoint.z, getTerrainAverageZ(),
1202 m_mapExtent.hi.z, m_mapExtent.lo.z );
1203
1204 } // end else
1205
1206 // add color to our samples
1207 sampleColor.red += color.red;
1208 sampleColor.green += color.green;
1209 sampleColor.blue += color.blue;
1210 samples++;
1211
1212 } // end if
1213
1214 } // end for i
1215
1216 } // end if
1217
1218 } // end for j
1219
1220 // prevent divide by zeros
1221 if( samples == 0 )
1222 samples = 1;
1223
1224 // set the color to an average of the colors read
1225 color.red = sampleColor.red / (Real)samples;
1226 color.green = sampleColor.green / (Real)samples;
1227 color.blue = sampleColor.blue / (Real)samples;
1228
1229 } // end else
1230
1231 //
1232 // draw the pixel for the terrain at this point, note that because of the orientation
1233 // of our world we draw it with positive y in the "up" direction
1234 //
1235 // FYI: I tried making this faster by pulling out all the code inside DrawPixel
1236 // and locking only once ... but it made absolutely *no* performance difference,
1237 // the sampling and interpolation algorithm for generating pretty looking terrain
1238 // and water for the radar is just, well, expensive.
1239 //
1240 surface->DrawPixel( x, y, GameMakeColor( color.red * 255,
1241 color.green * 255,
1242 color.blue * 255,
1243 255 ) );
1244
1245 } // end for x
1246
1247 } // end for y
1248
1249 // all done with the surface
1250 REF_PTR_RELEASE(surface);
1251
1252} // end buildTerrainTexture
