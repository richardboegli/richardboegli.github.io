40Some info about partioning problems:
41
42 This problem is contained in a very interesting class of problems known as NP complete. The
43 basic problem is that there is no way to tell whether you have an optimal solution or not.
44 Worst case, you try out every possible solution and still don't find the optimal solution:
45 this takes 2^n time to find, where N is the number of elements you are attempting to place.
46 For this reason, a value near PREFER_FAST_SOLUTION should almost always be chosen. We will use
47 a flat multiply to determine how many solutions to attempt before giving up and returning our
48 best attempt. If you want more info, this site contains info on the problem:
49 http://odysseus.nat.uni-magdeburg.de/~mertens/npp/index.shtml
50**************************************************************************************************/

72void PartitionSolver::solve(void)
73{
74 m_bestSolution.clear();
75 m_currentSolution.clear();
76 m_currentSolutionLeftovers = 0x7fffffff;
77
78 Int minSizeForAllData = 0;
79 Int slotsAllotted = 0;
80 Int i, j;
81
82 // first, determine whether there is an actual solution, or we're going to have to fudge it.
83 for (i = 0; i < m_data.size(); ++i) {
84 minSizeForAllData += m_data[i].second;
85 }
86
87 for (i = 0; i < m_spacesForData.size(); ++i) {
88 slotsAllotted += m_spacesForData[i].second;
89 }
90
91 // we want to attempt to place the largest things first. This allows us to throw
92 // out whole classes of solutions
93
94 std::sort(m_data.begin(), m_data.end(), greater_than);
95
96 // Also make the largest partition first.
97 std::sort(m_spacesForData.begin(), m_spacesForData.end(), greater_than);
98
99 // work in our temporary vector.
100 SpacesVec spacesStillAvailable = m_spacesForData;
101
102 if (m_howToSolve == PREFER_FAST_SOLUTION)
103 {
104 // we prefer the fast, but not necessarily correct solution
105 // simply start placing the stuff. Skip things you can't place.
106 for (i = 0; i < m_data.size(); ++i)
107 {
108 for (j = 0; j < spacesStillAvailable.size(); ++j)
109 {
110 if (m_data[i].second <= spacesStillAvailable[j].second)
111 {
112 spacesStillAvailable[j].second -= m_data[i].second;
113 m_bestSolution.push_back(std::make_pair(m_data[i].first, spacesStillAvailable[j].first));
114 break;
115 }
116 }
117 }
118 } else {
119 DEBUG_CRASH(("PREFER_CORRECT_SOLUTION @todo impl"));
120 }
121}
