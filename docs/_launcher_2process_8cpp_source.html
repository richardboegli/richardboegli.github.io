30// Create a process
31bit8 Create_Process(Process &process)
32{
33 int retval;
34 STARTUPINFO si;
35 PROCESS_INFORMATION piProcess;
36 ZeroMemory(&si,sizeof(si));
37 si.cb=sizeof(si);
38
39 char cmdargs[513];
40 memset(cmdargs,0,513);
41 strcpy(cmdargs,process.command);
42 strcat(cmdargs,process.args);
43
44 DBGMSG("PROCESS CMD="<<cmdargs<<" DIR="<<process.directory);
45
46#ifndef COPY_PROTECT
47 retval=CreateProcess(NULL,cmdargs,NULL,NULL,FALSE, 0 ,NULL, NULL/*process.directory*/,&si,&piProcess);
48#else
49#ifdef USE_GAMEDIR_FROM_LCF
50 retval=CreateProcess(NULL,cmdargs,NULL,NULL,TRUE, 0 ,NULL, process.directory,&si,&piProcess);
51#else
52 retval=CreateProcess(NULL,cmdargs,NULL,NULL,TRUE, 0 ,NULL, NULL/*process.directory*/,&si,&piProcess);
53#endif
54#endif
55
56 DBGMSG("("<<retval<<") New process: HANDLE " << (void *)piProcess.hProcess << " ID "
57 << (DWORD)piProcess.dwProcessId);
58 DBGMSG("("<<retval<<") New thread: HANDLE " << (void *)piProcess.hThread << " ID "
59 << (DWORD)piProcess.dwThreadId);
60 if (retval==0)
61 {
62 char message_buffer[256];
63 FormatMessage( FORMAT_MESSAGE_FROM_SYSTEM, NULL, GetLastError(), MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), &message_buffer[0], 256, NULL );
64 DBGMSG("ERR: "<<message_buffer);
65 }
66 process.hProcess=piProcess.hProcess;
67 process.dwProcessID = piProcess.dwProcessId;
68 process.hThread=piProcess.hThread;
69 process.dwThreadID = piProcess.dwThreadId;
70 return(TRUE);
71}

73//
74// Wait for a process to complete, and fill in the exit code
75//
76bit8 Wait_Process(Process &process, DWORD *exit_code)
77{
78 DWORD retval;
79 retval=WaitForSingleObject(process.hProcess,INFINITE);
80 if (exit_code != NULL)
81 *exit_code=-1;
82 if (retval==WAIT_OBJECT_0) // process exited
83 {
84 // MDC 1/10/2003 Inserting sleep here to let game exit before applying patch
85 Sleep(3000);
86
87 if (exit_code != NULL)
88 GetExitCodeProcess(process.hProcess,exit_code);
89 return(TRUE);
90 }
91 else // can this happen?
92 return(FALSE);
93}

99bit8 Read_Process_Info(ConfigFile &config,OUT Process &info, IN char *key)
100{
101
102 Wstring keyStr = "RUN";
103 if (key)
104 keyStr = key;
105
106 Wstring procinfo;
107 if (config.getString(keyStr.get(), procinfo)==FALSE)
108 {
109 DBGMSG("Couldn't read the RUN line");
110 return(FALSE);
111 }
112 int offset=0;
113 Wstring dir;
114 Wstring executable;
115 Wstring args;
116 offset=procinfo.getToken(offset," ",dir);
117 offset=procinfo.getToken(offset," ",executable);
118 args=procinfo;
119 args.remove(0,offset);
120
123 DBGMSG("RUN: EXE = "<<executable.get()<<" DIR = "<<dir.get()<<
124 " ARGS = "<<args.get());
125 strcpy(info.command,executable.get());
126 strcpy(info.directory,dir.get());
127 strcpy(info.args,args.get());
128 return(TRUE);
129
130
131/*********************************************************
132 FILE *in;
133 char string[256];
134 bit8 found_space;
135 int i;
136
137 if ((in=fopen(config,"r"))==NULL)
138 return(FALSE);
139
140 while(fgets(string,256,in))
141 {
142 i=0;
143 while ((isspace(string[i]))&&(i<int(strlen(string))))
144 i++;
145
146 // do nothing with empty line or commented line
147 if ((string[i]=='#')||(i==int(strlen(string)-1)))
148 continue;
149
150 strcpy(info.directory,string);
151 found_space=FALSE;
152 for (; i<int(strlen(string)); i++)
153 {
154 if (isspace(info.directory[i]))
155 {
156 info.directory[i]=0;
157 found_space=TRUE;
158 }
159 else if (found_space)
160 break;
161 }
162
163 strcpy(info.command,string+i);
164 for (i=0; i<int(strlen(info.command)); i++)
165 if ((info.command[i]=='\n')||(info.command[i]=='\r'))
166 info.command[i]=0;
167
168 //printf("DIR = '%s' CMD = '%s'\n",info.directory,info.command);
169
170 // We only have 1 process for this
171 break;
172 }
173 fclose(in);
174 return(TRUE);
175**********************************************/
176}
