362UpgradeTemplate *UpgradeCenter::newUpgrade( const AsciiString& name )
363{
364 UpgradeTemplate *newUpgrade = newInstance(UpgradeTemplate);
365
366 // copy data from the default upgrade
367 const UpgradeTemplate *defaultUpgrade = findUpgrade( "DefaultUpgrade" );
368 if( defaultUpgrade )
369 *newUpgrade = *defaultUpgrade;
370
371 // assign name and starting data
372 newUpgrade->setUpgradeName( name );
373 newUpgrade->setUpgradeNameKey( TheNameKeyGenerator->nameToKey( name ) );
374
375 // Make a unique bitmask for this new template by keeping track of what bits have been assigned
376 // damn MSFT! proper ANSI syntax for a proper 64-bit constant is "1LL", but MSVC doesn't recognize it
377 UpgradeMaskType newMask;
378 newMask.set( m_nextTemplateMaskBit );
379 //Int64 newMask = 1i64 << m_nextTemplateMaskBit;
380 m_nextTemplateMaskBit++;
381 DEBUG_ASSERTCRASH( m_nextTemplateMaskBit < UPGRADE_MAX_COUNT, ("Can't have over %d types of Upgrades and have a Bitfield function.", UPGRADE_MAX_COUNT) );
382 newUpgrade->friend_setUpgradeMask( newMask );
383
384 // link upgrade
385 linkUpgrade( newUpgrade );
386
387 // return new upgrade
388 return newUpgrade;
389
390} // end newUnlinkedUpgrade

395void UpgradeCenter::linkUpgrade( UpgradeTemplate *upgrade )
396{
397
398 // sanity
399 if( upgrade == NULL )
400 return;
401
402 // link
403 upgrade->friend_setPrev( NULL );
404 upgrade->friend_setNext( m_upgradeList );
405 if( m_upgradeList )
406 m_upgradeList->friend_setPrev( upgrade );
407 m_upgradeList = upgrade;
408
409} // end linkUpgrade

414void UpgradeCenter::unlinkUpgrade( UpgradeTemplate *upgrade )
415{
416
417 // sanity
418 if( upgrade == NULL )
419 return;
420
421 if( upgrade->friend_getNext() )
422 upgrade->friend_getNext()->friend_setPrev( upgrade->friend_getPrev() );
423 if( upgrade->friend_getPrev() )
424 upgrade->friend_getPrev()->friend_setNext( upgrade->friend_getNext() );
425 else
426 m_upgradeList = upgrade->friend_getNext();
427
428} // end unlinkUpgrade

433Bool UpgradeCenter::canAffordUpgrade( Player *player, const UpgradeTemplate *upgradeTemplate, Bool displayReason ) const
434{
435
436 // sanity
437 if( player == NULL || upgradeTemplate == NULL )
438 return FALSE;
439
440 // money check
441 Money *money = player->getMoney();
442 if( money->countMoney() < upgradeTemplate->calcCostToBuild( player ) )
443 {
444 //Post reason why we can't make upgrade!
445 if( displayReason )
446 {
447 TheInGameUI->message( "GUI:NotEnoughMoneyToUpgrade" );
448 }
449 return FALSE;
450 }
451
453
454 return TRUE; // all is well
455
456} // end canAffordUpgrade

475void UpgradeCenter::parseUpgradeDefinition( INI *ini )
476{
477 // read the name
478 const char* c = ini->getNextToken();
479 AsciiString name = c;
480
481 // find existing item if present
482 UpgradeTemplate* upgrade = TheUpgradeCenter->findNonConstUpgradeByKey( NAMEKEY(name) );
483 if( upgrade == NULL )
484 {
485
486 // allocate a new item
487 upgrade = TheUpgradeCenter->newUpgrade( name );
488
489 } // end if
490
491 // sanity
492 DEBUG_ASSERTCRASH( upgrade, ("parseUpgradeDefinition: Unable to allocate upgrade '%s'\n", name.str()) );
493
494 // parse the ini definition
495 ini->initFromINI( upgrade, upgrade->getFieldParse() );
496
497}
