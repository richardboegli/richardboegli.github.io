169void FlightDeckBehavior::buildInfo(Bool createUnits)
170{
171 if (m_gotInfo)
172 return;
173
174 const FlightDeckBehaviorModuleData* data = getFlightDeckBehaviorModuleData();
175
176 m_thingTemplate = TheThingFactory->findTemplate( data->m_thingTemplateName );
177
178 if (getObject()->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) ||
179 getObject()->testStatus(OBJECT_STATUS_SOLD))
180 return;
181
182 //ProductionUpdateInterface* pu = getObject()->getProductionUpdateInterface();
183
184 m_spaces.reserve( data->m_numRows * data->m_numCols );
185
186 //Initialize the spaces that planes will eventually be assigned to for parking purposes
187 FlightDeckInfo flightDeckInfo;
188
189 //We want to sort the spaces so that we have runway 1 space 1, runway 2 space 1, R1S2, R2S2, R1S3...
190 for( Int row = 0; row < data->m_numRows; row++ )
191 {
192 for( Int col = 0; col < data->m_numCols; col++ )
193 {
194 std::vector<AsciiString> spaces = data->m_runwayInfo[ col ].m_spacesBoneNames;
195 std::vector<AsciiString>::const_iterator it;
196
197 Int counter = 0;
198 for( it = spaces.begin(); it != spaces.end(), counter < row; it++, counter++ )
199 {
200 //just iterate to the spaces.
201 }
202
203 AsciiString tmp;
204 Matrix3D mtx;
205
206 //Convert the module data bone names into coordinates that we can use
207 getObject()->getSingleLogicalBonePosition( it->str(), &flightDeckInfo.m_prep, &mtx );
208 flightDeckInfo.m_orientation = mtx.Get_Z_Rotation();
209
210 //Init basic runway stuff
211 flightDeckInfo.m_runway = col;
212 flightDeckInfo.m_objectInSpace = INVALID_ID;
213
214 //Create the payload
215 Object *jet = NULL;
216 if( m_thingTemplate && createUnits )
217 {
218 jet = TheThingFactory->newObject( m_thingTemplate, getObject()->getControllingPlayer()->getDefaultTeam() );
219 if( jet )
220 {
221 jet->setProducer( getObject() );
222 jet->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) );
223
224 //Init positioning.
225 jet->setPosition( &flightDeckInfo.m_prep );
226 jet->setOrientation( flightDeckInfo.m_orientation );
227
228 //Assign jet to space
229 flightDeckInfo.m_objectInSpace = jet->getID();
230 //validateAssignments();
231 }
232 }
233
234 m_spaces.push_back( flightDeckInfo );
235 }
236 }
237
238 //Now initialize the runway take-off and landing information.
239 RunwayInfo info;
240 m_runways.reserve(data->m_numCols);
241 for( Int col = 0; col < data->m_numCols; ++col )
242 {
243 AsciiString tmp;
244
245 getObject()->getSingleLogicalBonePosition( data->m_runwayInfo[ col ].m_takeoffBoneNames[ RUNWAY_START_BONE ].str(), &info.m_start, NULL);
246 getObject()->getSingleLogicalBonePosition( data->m_runwayInfo[ col ].m_takeoffBoneNames[ RUNWAY_END_BONE ].str(), &info.m_end, NULL);
247 getObject()->getSingleLogicalBonePosition( data->m_runwayInfo[ col ].m_landingBoneNames[ RUNWAY_START_BONE ].str(), &info.m_landingStart, NULL);
248 getObject()->getSingleLogicalBonePosition( data->m_runwayInfo[ col ].m_landingBoneNames[ RUNWAY_END_BONE ].str(), &info.m_landingEnd, NULL);
249
250 info.m_inUseByForTakeoff = INVALID_ID;
251 info.m_inUseByForLanding = INVALID_ID;
252
253 //Get the taxi bones and store them as well (possible to have none!)
254 std::vector<AsciiString> locations = data->m_runwayInfo[ col ].m_taxiBoneNames;
255 std::vector<AsciiString>::const_iterator it;
256 info.m_taxi.clear();
257 for( it = locations.begin(); it != locations.end(); it++ )
258 {
259 Coord3D taxiPos;
260 Matrix3D mtx;
261
262 //Get the position of the taxi bone.
263 getObject()->getSingleLogicalBonePosition( it->str(), &taxiPos, &mtx );
264
265 //Add it to the taxi vector
266 info.m_taxi.push_back( taxiPos );
267 }
268
269 //Get the creation bones and store them as well
270 locations = data->m_runwayInfo[ col ].m_creationBoneNames;
271 info.m_creation.clear();
272 Bool firstTime = TRUE;
273 for( it = locations.begin(); it != locations.end(); it++ )
274 {
275 Coord3D pos;
276 Matrix3D mtx;
277
278 //Get the position of the creation bone.
279 getObject()->getSingleLogicalBonePosition( it->str(), &pos, &mtx );
280
281 if( firstTime )
282 {
283 firstTime = FALSE;
284 info.m_startOrient = mtx.Get_Z_Rotation();
285 info.m_startTransform = mtx;
286 }
287
288 //Add it to the taxi vector
289 info.m_creation.push_back( pos );
290 }
291
292 m_runways.push_back(info);
293 }
294
295 m_gotInfo = true;
296}

299void FlightDeckBehavior::purgeDead()
300{
301 buildInfo();
302
303 for (std::vector<FlightDeckInfo>::iterator it = m_spaces.begin(); it != m_spaces.end(); ++it)
304 {
305 if (it->m_objectInSpace != INVALID_ID)
306 {
307 Object* obj = TheGameLogic->findObjectByID(it->m_objectInSpace);
308 if (obj == NULL || obj->isEffectivelyDead())
309 {
310 it->m_objectInSpace = INVALID_ID;
311 }
312 }
313 }
314
315 {
316 for (std::vector<RunwayInfo>::iterator it = m_runways.begin(); it != m_runways.end(); ++it)
317 {
318 if (it->m_inUseByForTakeoff != INVALID_ID)
319 {
320 Object* obj = TheGameLogic->findObjectByID(it->m_inUseByForTakeoff);
321 if (obj == NULL || obj->isEffectivelyDead())
322 {
323 it->m_inUseByForTakeoff = INVALID_ID;
324 }
325 }
326 if (it->m_inUseByForLanding != INVALID_ID)
327 {
328 Object* obj = TheGameLogic->findObjectByID(it->m_inUseByForLanding);
329 if (obj == NULL || obj->isEffectivelyDead())
330 {
331 it->m_inUseByForLanding = INVALID_ID;
332 }
333 }
334 }
335 }
336
337 {
338 Bool anythingPurged = false;
339 for (std::list<HealingInfo>::iterator it = m_healing.begin(); it != m_healing.end(); /*++it*/)
340 {
341 if (it->m_gettingHealedID != INVALID_ID)
342 {
343 Object* objToHeal = TheGameLogic->findObjectByID(it->m_gettingHealedID);
344 if (objToHeal == NULL || objToHeal->isEffectivelyDead())
345 {
346 it = m_healing.erase(it);
347 anythingPurged = true;
348 }
349 else
350 {
351 ++it;
352 }
353 }
354 }
355 if (anythingPurged)
356 resetWakeFrame();
357 }
358}

468{
469 buildInfo();
470 purgeDead();
471
472 const FlightDeckBehaviorModuleData* d = getFlightDeckBehaviorModuleData();
473

475 if (ppi == NULL)
476 {
477 ppi = findEmptyPPI();
478 if (ppi == NULL)
479 {
480 DEBUG_CRASH(("No parking places!"));
481 return false; // nothing available
482 }
483 }
484
485 ppi->m_objectInSpace = id;
486 //validateAssignments();
487
488 if( d->m_landingDeckHeightOffset )
489 {
490 Object *obj = TheGameLogic->findObjectByID( id );
491 if( obj )
492 {
493 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) );
494 }
495 }
496
497 if( info )
498 {
499 calcPPInfo( id, info );
500 if (parkingOffset != 0.0f)
501 {
502 info->parkingSpace.x += parkingOffset * Cos(ppi->m_orientation);
503 info->parkingSpace.y += parkingOffset * Sin(ppi->m_orientation);
504 }
505 }
506
507 return true;
508}

535//-------------------------------------------------------------------------------------------------
536void FlightDeckBehavior::calcPPInfo( ObjectID id, PPInfo *info )
537{
538 FlightDeckInfo *ppi = findPPI( id );
539 if( !ppi )
540 {
541 //Utter failure.
542 return;
543 }
544 const RunwayInfo& rr = m_runways[ ppi->m_runway ];
545
546 if( info )
547 {
548 const FlightDeckBehaviorModuleData* d = getFlightDeckBehaviorModuleData();
549 const Real APPROACH_DIST = 0.75f;
550 info->parkingSpace = ppi->m_prep;
551 info->runwayPrep = ppi->m_prep;
552 info->parkingOrientation = ppi->m_orientation;
553 info->runwayStart = rr.m_start;
554 info->runwayEnd = rr.m_end;
555 info->runwayExit = rr.m_end;
556 info->runwayExit.x += (rr.m_end.x - rr.m_start.x) * APPROACH_DIST;
557 info->runwayExit.y += (rr.m_end.y - rr.m_start.y) * APPROACH_DIST;
558 info->runwayExit.z = rr.m_end.z + d->m_approachHeight + d->m_landingDeckHeightOffset;
559
560 info->runwayLandingStart = rr.m_landingStart;
561 info->runwayLandingEnd = rr.m_landingEnd;
562 info->runwayApproach = rr.m_landingStart;
563 info->runwayApproach.x += (rr.m_landingStart.x - rr.m_landingEnd.x) * APPROACH_DIST;
564 info->runwayApproach.y += (rr.m_landingStart.y - rr.m_landingEnd.y) * APPROACH_DIST;
565 info->runwayApproach.z = rr.m_landingStart.z + d->m_approachHeight + d->m_landingDeckHeightOffset;
566
567 //Cache the runway's takeoff distance used by JetAIUpdate for calculating lift.
568 Coord3D vector = info->runwayStart;
569 vector.sub( &info->runwayEnd );
570 info->runwayTakeoffDist = vector.length();
571
572 for (std::vector<RunwayInfo>::iterator it = m_runways.begin(); it != m_runways.end(); ++it)
573 {
574 if (it->m_inUseByForTakeoff == id )
575 {
576 info->runwayStart = info->runwayPrep;
577 }
578 }
579 }

627//-------------------------------------------------------------------------------------------------
628Bool FlightDeckBehavior::reserveRunway(ObjectID id, Bool forLanding)
629{
630 buildInfo();
631 purgeDead();
632
633 Int runway = -1;
634
635 if( !forLanding )
636 {
637 //Only look at the front spaces for takeoff. You can't take off from the back!
638 const FlightDeckBehaviorModuleData *data = getFlightDeckBehaviorModuleData();
639 for( Int i = 0; i < data->m_numCols; i++ )
640 {
641 if( m_spaces[ i ].m_objectInSpace == id )
642 {
643 runway = m_spaces[ i ].m_runway;
644 break;
645 }
646 }
647 }
648 else
649 {
650 //Look at all spaces for landing.
651 for (std::vector<FlightDeckInfo>::iterator it = m_spaces.begin(); it != m_spaces.end(); ++it)
652 {
653 if (it->m_objectInSpace == id)
654 {
655 runway = it->m_runway;
656 break;
657 }
658 }
659 }
660 if (runway == -1)
661 {
662 if( forLanding )
663 {
664 DEBUG_CRASH(("only planes with reserved spaces can reserve runways"));
665 }
666 return false;
667 }
668
669 RunwayInfo& info = m_runways[runway];
670 if( info.m_inUseByForTakeoff == id && !forLanding || info.m_inUseByForLanding == id && forLanding )
671 {
672 return true;
673 }

879 //the back and keep looking at empty spaces until we find one with a plane blocking.

890 //Done, don't look at my spot, nor spots behind me.
891 if( bestIt )

893 myIt->m_objectInSpace = bestJet ? bestJet->getID() : INVALID_ID;

907
908 if( newIndex )
909 {
910 *newIndex = bestIndex;
911 }
912
913 //Promoted forward.
914 return TRUE;
915 }
916 //Did not get promoted forward.
917 return FALSE;
918 }
919 if( thatIt->m_runway == runway )

922 {
923 if( !checkForPlaneInWay )

943 bestIt = NULL;
944 }
945 }
946 }
947 }
948 //Did not get promoted forward
949 return FALSE;
950}

1079//-------------------------------------------------------------------------------------------------
1080UpdateSleepTime FlightDeckBehavior::update()
1081{
1082 // alas, we need to keep the buildInfo and dead-purged stuff pretty much up to date, for
1083 // the client to be able to peek at. at this late date, the most expedient way is to ensure
1084 // our update is run every frame, and do this manually. the extra cost should be trivial, since
1085 // there are generally at most only a few airfields at any given time.
1086 buildInfo();
1087 purgeDead();
1088
1089 const FlightDeckBehaviorModuleData* data = getFlightDeckBehaviorModuleData();
1090 UnsignedInt now = TheGameLogic->getFrame();
1091 if( now >= m_nextHealFrame )
1092 {
1093 m_nextHealFrame = now + HEAL_RATE_FRAMES;
1094 for (std::list<HealingInfo>::iterator it = m_healing.begin(); it != m_healing.end(); /*++it*/)
1095 {
1096 if (it->m_gettingHealedID != INVALID_ID)
1097 {
1098 Object* objToHeal = TheGameLogic->findObjectByID(it->m_gettingHealedID);
1099 if (objToHeal == NULL || objToHeal->isEffectivelyDead())
1100 {
1101 it = m_healing.erase(it);
1102 }
1103 else
1104 {
1105 DamageInfo healInfo;
1106 healInfo.in.m_damageType = DAMAGE_HEALING;
1107 healInfo.in.m_deathType = DEATH_NONE;
1108 healInfo.in.m_sourceID = getObject()->getID();
1109 healInfo.in.m_amount = HEAL_RATE_FRAMES * data->m_healAmount * SECONDS_PER_LOGICFRAME_REAL;
1110 BodyModuleInterface *body = objToHeal->getBodyModule();
1111 body->attemptHealing( &healInfo );
1112 ++it;
1113 }
1114 }
1115 }
1116 }
1117
1118 //Periodically, we want to look at all the aircraft assignments and keep them near the front
1119 //of the carrier at all times. We do this with the following steps:
1120 // 1) Detect non-idle plane assigned to frontmost spaces (implying spot is empty)
1121 // 2) Promote next idle plane behind it to move up and take it's spot. The original plane will
1122 // get assigned to it's old spot (thus bubblesorted eventually to the rear).
1123 if( now >= m_nextCleanupFrame )
1124 {
1125 m_nextCleanupFrame = now + data->m_cleanupFrames;
1126 std::vector<FlightDeckInfo>::iterator tempIt;
1127 Int spaceID = 0;
1128
1129 Bool complete[ MAX_RUNWAYS ] = { FALSE, FALSE };
1130 for( std::vector<FlightDeckInfo>::iterator it = m_spaces.begin(); it != m_spaces.end(); it++, spaceID++ )
1131 {
1132 Object *nonIdleJet = TheGameLogic->findObjectByID( it->m_objectInSpace );
1133 if( !nonIdleJet || isAbleToGiveUpParkingSpace( nonIdleJet ) )
1134 {
1135 //Either we don't have a jet, or the jet is busy (meaning it's not parked there). When a jet
1136 //isn't in his spot, we will look for jets behind him to move up to take up his spot.
1137 Int runwayCount = data->m_numCols;
1138 Int tempID = spaceID;
1139 for( tempIt = it; tempIt != m_spaces.end(); tempIt++, tempID++ )
1140 {
1141 if( runwayCount > 0 )
1142 {
1143 //Because the spaces are sorted Runway 1 Space 1, R2S1, R1S2, R2S2 etc, we simply iterate
1144 //every other space. If there were 3 or 4 runways, then we would jump the number of runways
1145 //before each check so we don't get aircraft from a different runway getting moved up!
1146 runwayCount--;
1147 continue;
1148 }
1149
1150 if( complete[ it->m_runway ] )
1151 {
1152 continue;
1153 }
1154
1155 //Now we have the correct runway. Check if that spot has an idle plane in it!
1156 Object *parkedJet = TheGameLogic->findObjectByID( tempIt->m_objectInSpace );
1157 if( parkedJet && isAbleToMoveForward( *parkedJet ) )
1158 {
1159 //We have found the best candidate to replace our empty space. Now handle assignment swap.
1160 it->m_objectInSpace = parkedJet->getID();
1161 tempIt->m_objectInSpace = nonIdleJet ? nonIdleJet->getID() : INVALID_ID;
1162 //validateAssignments();
1163
1164 //Give the parkedJet a move order to taxi over to the new spot.
1165 //However, we need to set a status bit to tell the AI the difference between taxiing from
1166 //a hangar and reassigning a parking space.
1167 parkedJet->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_REASSIGN_PARKING ) );
1168
1169 //Doesn't matter what we push in now, the JetOrHeliTaxiState::onEnter() will nuke it
1170 //and calculate the taxi point to move through.
1171 std::vector<Coord3D> exitPath;
1172 exitPath.push_back( it->m_prep );
1173 parkedJet->getAI()->aiFollowExitProductionPath( &exitPath, getObject(), CMD_FROM_AI );
1174
1175 //Determine if the first plane in each row has completed a move. We want to lag the
1176 //additional planes instead of moving them all simultaneously. So this is the exit
1177 //condition if one plane per runway has been bumped.
1178 complete[ it->m_runway ] = TRUE;
1179 m_nextCleanupFrame = now + data->m_humanFollowFrames;
1180 }
1181 //Break through and advance to the next spot!
1182 break;
1183 }
1184 }
1185 }
1186 }
1187
1188 //Handle automatic production of lost aircraft.
1189
1190 //Reset timer if we just finished building an aircraft.
1191 if( m_nextAllowedProductionFrame <= now )
1192 {
1193 m_startedProductionFrame = FOREVER;
1194 }
1195
1196 for( std::vector<FlightDeckInfo>::iterator it = m_spaces.begin(); it != m_spaces.end(); it++ )
1197 {
1198 //Unassigned space?... so we can build a replacement.
1199 if( it->m_objectInSpace == INVALID_ID )
1200 {
1201 //But are we already building one?
1202 ProductionUpdateInterface *pu = getObject()->getProductionUpdateInterface();
1203 if( pu == NULL )
1204 {
1205
1206 DEBUG_CRASH( ("MSG_QUEUE_UNIT_CREATE: Producer '%s' doesn't have a unit production interface\n", getObject()->getTemplate()->getName().str()) );
1207 break;
1208 } // end if
1209 DEBUG_ASSERTCRASH( m_thingTemplate != NULL, ("flightdeck has a null thingtemplate... no jets for you!\n") );
1210 if( !pu->getProductionCount() && now >= m_nextAllowedProductionFrame && m_thingTemplate != NULL )
1211 {
1212 //Queue the build
1213 pu->queueCreateUnit( m_thingTemplate, pu->requestUniqueUnitID() );
1214
1215 m_startedProductionFrame = now;
1216 m_nextAllowedProductionFrame = now + data->m_replacementFrames + data->m_dockAnimationFrames;
1217 }
1218
1219 break;
1220 }
1221 }
1222
1223 //If the carrier has at least one aircraft, then allow it to attack.
1224 Bool hasAircraft = FALSE;
1225 for( it = m_spaces.begin(); it != m_spaces.end(); it++ )
1226 {
1227 if( it->m_objectInSpace != INVALID_ID )
1228 {
1229 hasAircraft = TRUE;
1230 break;
1231 }
1232 }
1233 getObject()->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_ATTACK ), !hasAircraft );
1234 Drawable *draw = getObject()->getDrawable();
1235
1236 //Check for timer expiry -- are we allowed to launch the next wave yet?
1237 for( int i = 0; i < data->m_numCols; i++ )
1238 {
1239 Object *jet = TheGameLogic->findObjectByID( m_spaces[ i ].m_objectInSpace );
1240 if( jet && !isAbleToGiveUpParkingSpace( jet ) && isInPositionToTakeoff( *jet ) && hasTakeoffOrders() )
1241 {
1242 if( m_nextLaunchWaveFrame[ i ] <= now )
1243 {
1244 //Handle making the ramp going up and holding the launch until it's completely up.
1245 if( !m_rampUp[ i ] )
1246 {
1247 m_rampUp[ i ] = TRUE;
1248 m_rampUpFrame[ i ] = now + data->m_launchRampFrames;
1249 m_lowerRampFrame[ i ] = FOREVER;
1250 //****MAX_RUNWAYS***** if defined beyond 3, then this code will need to be rewritten or more modelcondition flags will
1251 // need to be added.
1252 ModelConditionFlagType opening = (ModelConditionFlagType)(MODELCONDITION_DOOR_2_OPENING + i * NUM_MODELCONDITION_DOOR_STATES);
1253 ModelConditionFlagType closing = (ModelConditionFlagType)(MODELCONDITION_DOOR_2_CLOSING + i * NUM_MODELCONDITION_DOOR_STATES);
1254 draw->clearAndSetModelConditionState( closing, opening );
1255 }
1256
1257 //Handle launching the wave of fighters.
1258 if( m_rampUp[ i ] && m_rampUpFrame[ i ] <= now )
1259 {
1260 AIUpdateInterface *jetAI = jet->getAI();
1261 if( jetAI )
1262 {
1263 propagateOrderToSpecificPlane( jet );
1264 m_nextLaunchWaveFrame[ i ] = now + data->m_launchWaveFrames;
1265 m_catapultSystemFrame[ i ] = now + data->m_catapultFireFrames;
1266 m_lowerRampFrame[ i ] = now + data->m_lowerRampFrames;
1267 }
1268 }
1269 }
1270 }
1271
1272 //Handle firing of the catapult steam effect upon launching planes.
1273 if( m_catapultSystemFrame[ i ] <= now && data->m_runwayInfo[ i ].m_catapultParticleSystem )
1274 {
1275 ParticleSystem *ps = TheParticleSystemManager->createParticleSystem( data->m_runwayInfo[ i ].m_catapultParticleSystem );
1276 m_catapultSystemFrame[ i ] = FOREVER;
1277 if( ps )
1278 {
1279 ps->setLocalTransform( &m_runways[ i ].m_startTransform );
1280 ps->setPosition( &m_runways[ i ].m_start );
1281 }
1282 }
1283
1284 //Handle lowering the ramp after the fighter has been launched.
1285 if( m_rampUp[ i ] && m_lowerRampFrame[ i ] <= now )
1286 {
1287 m_rampUp[ i ] = FALSE;
1288 //****MAX_RUNWAYS***** if defined beyond 3, then this code will need to be rewritten or more modelcondition flags will
1289 // need to be added.
1290 ModelConditionFlagType opening = (ModelConditionFlagType)(MODELCONDITION_DOOR_2_OPENING + i * NUM_MODELCONDITION_DOOR_STATES);
1291 ModelConditionFlagType closing = (ModelConditionFlagType)(MODELCONDITION_DOOR_2_CLOSING + i * NUM_MODELCONDITION_DOOR_STATES);
1292 draw->clearAndSetModelConditionState( opening, closing );
1293 }
1294 }
1295 return UPDATE_SLEEP_NONE;
1296}
