121//-------------------------------------------------------------------------------------------------
122// Normalizes angle to +- PI.
123//-------------------------------------------------------------------------------------------------
124static void normAngle(Real &angle)
125{
126 if (angle < -10*PI) {
127 angle = 0;
128 }
129 if (angle > 10*PI) {
130 angle = 0;
131 }
132 while (angle < -PI) {
133 angle += 2*PI;
134 }
135 while (angle > PI) {
136 angle -= 2*PI;
137 }
138}

140#define TERRAIN_SAMPLE_SIZE 40.0f
141static Real getHeightAroundPos(Real x, Real y)
142{
143 // terrain height + desired height offset == cameraOffset * actual zoom
144 Real terrainHeight = TheTerrainLogic->getGroundHeight(x, y);
145
146 // find best approximation of max terrain height we can see
147 Real terrainHeightMax = terrainHeight;
148 terrainHeightMax = max(terrainHeightMax, TheTerrainLogic->getGroundHeight(x+TERRAIN_SAMPLE_SIZE, y-TERRAIN_SAMPLE_SIZE));
149 terrainHeightMax = max(terrainHeightMax, TheTerrainLogic->getGroundHeight(x-TERRAIN_SAMPLE_SIZE, y-TERRAIN_SAMPLE_SIZE));
150 terrainHeightMax = max(terrainHeightMax, TheTerrainLogic->getGroundHeight(x+TERRAIN_SAMPLE_SIZE, y+TERRAIN_SAMPLE_SIZE));
151 terrainHeightMax = max(terrainHeightMax, TheTerrainLogic->getGroundHeight(x-TERRAIN_SAMPLE_SIZE, y+TERRAIN_SAMPLE_SIZE));
152
153 return terrainHeightMax;
154}

258#define MIN_CAPPED_ZOOM (0.5f) //WST 10.19.2002. JSC integrated 5/20/03.
259void W3DView::buildCameraTransform( Matrix3D *transform )
260{
261 Vector3 sourcePos, targetPos;
262
263 Real groundLevel = m_groundLevel; // 93.0f;
264
265 Real zoom = getZoom();
266 Real angle = getAngle();
267 Real pitch = getPitch();
268 Coord3D pos = *getPosition();
269
270 // add in the camera shake, if any
271 pos.x += m_shakeOffset.x;
272 pos.y += m_shakeOffset.y;
273
274 if (m_cameraConstraintValid)
275 {
276 pos.x = maxf(m_cameraConstraint.lo.x, pos.x);
277 pos.x = minf(m_cameraConstraint.hi.x, pos.x);
278 pos.y = maxf(m_cameraConstraint.lo.y, pos.y);
279 pos.y = minf(m_cameraConstraint.hi.y, pos.y);
280 }
281
282 // set position of camera itself
283 if (m_useRealZoomCam) //WST 10/10/2002 Real Zoom using FOV
284 {
285 sourcePos.X = m_cameraOffset.x;
286 sourcePos.Y = m_cameraOffset.y;
287 sourcePos.Z = m_cameraOffset.z;
288 Real capped_zoom = zoom;
289 if (capped_zoom > 1.0f)
290 {
291 capped_zoom= 1.0f;
292 }
293 if (capped_zoom < MIN_CAPPED_ZOOM)
294 {
295 capped_zoom = MIN_CAPPED_ZOOM;
296 }
297 m_FOV = 50.0f * PI/180.0f * capped_zoom * capped_zoom;
298 }
299 else
300 {
301 sourcePos.X = m_cameraOffset.x*zoom;
302 sourcePos.Y = m_cameraOffset.y*zoom;
303 sourcePos.Z = m_cameraOffset.z*zoom;
304 }
305
306#ifdef NOT_IN_USE
307 if (TheGlobalData->m_isOffsetCameraZ && TheTerrainLogic)
308 {
309 sourcePos.Z += TheTerrainLogic->getGroundHeight(pos.x, pos.y);
310 if (m_prevSourcePosZ != SOURCEPOS_INVALID)
311 {
312 const Real MAX_SPZ_VARIATION = 0.05f;
313 Real spzMin = m_prevSourcePosZ*(1.0-MAX_SPZ_VARIATION);
314 Real spzMax Coord3D center;
315 = m_prevSourcePosZ*(1.0+MAX_SPZ_VARIATION);
316 if (sourcePos.Z < spzMin) sourcePos.Z = spzMin;
317 if (sourcePos.Z > spzMax) sourcePos.Z = spzMax;
318 }
319 m_prevSourcePosZ = sourcePos.Z;
320 }
321#endif
322
323 // camera looking at origin
324 targetPos.X = 0;
325 targetPos.Y = 0;
326 targetPos.Z = 0;
327
328
329 Real factor = 1.0 - (groundLevel/sourcePos.Z );
330
331 // construct a matrix to rotate around the up vector by the given angle
332 Matrix3D angleTransform( Vector3( 0.0f, 0.0f, 1.0f ), angle );
333
334 // construct a matrix to rotate around the horizontal vector by the given angle
335 Matrix3D pitchTransform( Vector3( 1.0f, 0.0f, 0.0f ), pitch );
336
337 // rotate camera position (pitch, then angle)
338#ifdef ALLOW_TEMPORARIES
339 sourcePos = pitchTransform * sourcePos;
340 sourcePos = angleTransform * sourcePos;
341#else
342 pitchTransform.mulVector3(sourcePos);
343 angleTransform.mulVector3(sourcePos);
344#endif
345 sourcePos *= factor;
346
347 // translate to current XY position
348 sourcePos.X += pos.x;
349 sourcePos.Y += pos.y;
350 sourcePos.Z += groundLevel;
351
352 targetPos.X += pos.x;
353 targetPos.Y += pos.y;
354 targetPos.Z += groundLevel;
355
356 // do m_FXPitch adjustment.
357 //WST Real height = sourcePos.Z - targetPos.Z;
358 //WST height *= m_FXPitch;
359 //WST targetPos.Z = sourcePos.Z - height;
360
361
362 // The following code moves camera down and pitch up when player zooms in.
363 // Use scripts to switch to useRealZoomCam
364 if (m_useRealZoomCam)
365 {
366 Real pitch_adjust = 1.0f;
367
368 if (!TheDisplay->isLetterBoxed())
369 {
370 Real capped_zoom = zoom;
371 if (capped_zoom > 1.0f)
372 {
373 capped_zoom= 1.0f;
374 }
375 if (capped_zoom < MIN_CAPPED_ZOOM)
376 {
377 capped_zoom = MIN_CAPPED_ZOOM;
378 }
379 sourcePos.Z = sourcePos.Z * ( 0.5f + capped_zoom * 0.5f); // move camera down physically
380 pitch_adjust = capped_zoom; // adjust camera to pitch up
381 }
382 m_FXPitch = 1.0f * (0.25f + pitch_adjust*0.75f);
383 }
384
385
386 // do fxPitch adjustment
387 if (m_useRealZoomCam)
388 {
389 sourcePos.X = targetPos.X + ((sourcePos.X - targetPos.X) / m_FXPitch);
390 sourcePos.Y = targetPos.Y + ((sourcePos.Y - targetPos.Y) / m_FXPitch);
391 }
392 else
393 {
394 if (m_FXPitch <= 1.0f)
395 {
396 Real height = sourcePos.Z - targetPos.Z;
397 height *= m_FXPitch;
398 targetPos.Z = sourcePos.Z - height;
399 }
400 else
401 {
402 sourcePos.X = targetPos.X + ((sourcePos.X - targetPos.X) / m_FXPitch);
403 sourcePos.Y = targetPos.Y + ((sourcePos.Y - targetPos.Y) / m_FXPitch);
404 }
405 }
406
407 //m_3DCamera->Set_View_Plane(DEG_TO_RADF(50.0f));
408 //DEBUG_LOG(("zoom %f, SourceZ %f, posZ %f, groundLevel %f CamOffZ %f\n",
409 // zoom, sourcePos.Z, pos.z, groundLevel,m_cameraOffset.z));
410
411 // build new camera transform
412 transform->Make_Identity();
413 transform->Look_At( sourcePos, targetPos, 0 );
414
415 //WST 11/12/2002 New camera shaker system
416 CameraShakerSystem.Timestep(1.0f/30.0f);
417 CameraShakerSystem.Update_Camera_Shaker(sourcePos, &m_shakerAngles);
418 transform->Rotate_X(m_shakerAngles.X);
419 transform->Rotate_Y(m_shakerAngles.Y);
420 transform->Rotate_Z(m_shakerAngles.Z);
421
422 //if (m_shakerAngles.X >= 0.0f)
423 //{
424 // DEBUG_LOG(("m_shakerAngles %f, %f, %f\n", m_shakerAngles.X, m_shakerAngles.Y, m_shakerAngles.Z));
425 //}
426
427 // (gth) check if the camera is being controlled by an animation
428 if (m_isCameraSlaved) {
429 // find object named m_cameraSlaveObjectName
430 Object * obj = TheScriptEngine->getUnitNamed(m_cameraSlaveObjectName);
431
432 if (obj != NULL) {
433 // dig out the drawable
434 Drawable * draw = obj->getDrawable();
435 if (draw != NULL) {
436
437 // dig out the first draw module with an ObjectDrawInterface
438 for (DrawModule ** dm = draw->getDrawModules(); *dm; ++dm) {
439 const ObjectDrawInterface* di = (*dm)->getObjectDrawInterface();
440 if (di) {
441 Matrix3D tm;
442 di->clientOnly_getRenderObjBoneTransform(m_cameraSlaveObjectBoneName,&tm);
443
444 // Ok, slam it into the camera!
445 *transform = tm;
446
447 //--------------------------------------------------------------------
448 // WST 10.22.2002. Update the Listener positions used by audio system
449 //--------------------------------------------------------------------
450 Vector3 position = transform->Get_Translation();
451 m_pos.x = position.X;
452 m_pos.y = position.Y;
453 m_pos.z = position.Z;
454
455
456 //DEBUG_LOG(("mpos x%f, y%f, z%f\n", m_pos.x, m_pos.y, m_pos.z ));
457
458 break;
459 }
460 }
461
462 } else {
463 m_isCameraSlaved = false;
464 }
465 } else {
466 m_isCameraSlaved = false;
467 }
468 }
469}

473void W3DView::calcCameraConstraints()
474{
475// const Matrix3D& cameraTransform = m_3DCamera->Get_Transform();
476
477// DEBUG_LOG(("*** rebuilding cam constraints\n"));
478
479 // ok, now check to ensure that we can't see outside the map region,
480 // and twiddle the camera if needed
481 if (TheTerrainLogic)
482 {
483 Region3D mapRegion;
484 TheTerrainLogic->getExtent( &mapRegion );
485
486 /*
487 Note the following restrictions on camera constraints!
488
489 -- they assume that all maps are height 'm_groundLevel' at the edges.
490 (since you need to add some "buffer" around the edges of your map
491 anyway, this shouldn't be an issue.)
492
493 -- for angles/pitches other than zero, it may show boundaries.
494 since we currently plan the game to be restricted to this,
495 it shouldn't be an issue.
496
497 */
498 Real maxEdgeZ = m_groundLevel;
499// const Real BORDER_FUDGE = MAP_XY_FACTOR * 1.414f;
500 Coord3D center, bottom;
501 ICoord2D screen;
502
503 //Pick at the center
504 screen.x=0.5f*getWidth()+m_originX;
505 screen.y=0.5f*getHeight()+m_originY;
506
507 Vector3 rayStart,rayEnd;
508
509 getPickRay(&screen,&rayStart,&rayEnd);
510
511 center.x = Vector3::Find_X_At_Z(maxEdgeZ, rayStart, rayEnd);
512 center.y = Vector3::Find_Y_At_Z(maxEdgeZ, rayStart, rayEnd);
513 center.z = maxEdgeZ;
514
515 screen.y = m_originY+ 0.95f*getHeight();
516 getPickRay(&screen,&rayStart,&rayEnd);
517 bottom.x = Vector3::Find_X_At_Z(maxEdgeZ, rayStart, rayEnd);
518 bottom.y = Vector3::Find_Y_At_Z(maxEdgeZ, rayStart, rayEnd);
519 bottom.z = maxEdgeZ;
520 center.x -= bottom.x;
521 center.y -= bottom.y;
522
523 Real offset = center.length();
524
525 if (TheGlobalData->m_debugAI) {
526 offset = -1000; // push out the constraints so we can look at staging areas.
527 }
528
529 m_cameraConstraint.lo.x = mapRegion.lo.x + offset;
530 m_cameraConstraint.hi.x = mapRegion.hi.x - offset;
531 // this looks inverted, but is correct
532 m_cameraConstraint.lo.y = mapRegion.lo.y + offset;
533 m_cameraConstraint.hi.y = mapRegion.hi.y - offset;
534 m_cameraConstraintValid = true;
535 }
536}

543void W3DView::getPickRay(const ICoord2D *screen, Vector3 *rayStart, Vector3 *rayEnd)
544{
545 Real logX,logY;
546
547 //W3D Screen coordinates are -1 to 1, so we need to do some conversion:
548 PixelScreenToW3DLogicalScreen(screen->x - m_originX,screen->y - m_originY, &logX, &logY,getWidth(),getHeight());
549
550 *rayStart = m_3DCamera->Get_Position(); //get camera location
551 m_3DCamera->Un_Project(*rayEnd,Vector2(logX,logY)); //get world space point
552 *rayEnd -= *rayStart; //vector camera to world space point
553 rayEnd->Normalize(); //make unit vector
554 *rayEnd *= m_3DCamera->Get_Depth(); //adjust length to reach far clip plane
555 *rayEnd += *rayStart; //get point on far clip plane along ray from camera.
556}

561void W3DView::setCameraTransform( void )
562{
563 m_cameraHasMovedSinceRequest = true;
564 Matrix3D cameraTransform( 1 );
565
566 Real nearZ, farZ;
567 // m_3DCamera->Get_Clip_Planes(nearZ, farZ);
568 // Set the near to MAP_XY_FACTOR. Improves zbuffer resolution.
569 nearZ = MAP_XY_FACTOR;
570 farZ = 1200.0f;
571
572 if (m_useRealZoomCam) //WST 10.19.2002
573 {
574 if (m_FXPitch<0.95f)
575 {
576 farZ = farZ / m_FXPitch; //Extend far Z when we pitch up for RealZoomCam
577 }
578 }
579 else
580 {
581 if ((TheGlobalData && TheGlobalData->m_drawEntireTerrain) || (m_FXPitch<0.95f || m_zoom>1.05))
582 { //need to extend far clip plane so entire terrain can be visible
583 farZ *= MAP_XY_FACTOR;
584 }
585 }
586
587 m_3DCamera->Set_Clip_Planes(nearZ, farZ);
588#if defined(_DEBUG) || defined(_INTERNAL)
589 if (TheGlobalData->m_useCameraConstraints)
590#endif
591 {
592 if (!m_cameraConstraintValid)
593 {
594 buildCameraTransform(&cameraTransform);
595 m_3DCamera->Set_Transform( cameraTransform );
596 calcCameraConstraints();
597 }
598 DEBUG_ASSERTLOG(m_cameraConstraintValid,("*** cam constraints are not valid!!!\n"));
599
600 if (m_cameraConstraintValid)
601 {
602 Coord3D pos = *getPosition();
603 pos.x = maxf(m_cameraConstraint.lo.x, pos.x);
604 pos.x = minf(m_cameraConstraint.hi.x, pos.x);
605 pos.y = maxf(m_cameraConstraint.lo.y, pos.y);
606 pos.y = minf(m_cameraConstraint.hi.y, pos.y);
607 setPosition(&pos);
608 }
609 }
610
611#if defined(_DEBUG) || defined(_INTERNAL)
612 m_3DCamera->Set_View_Plane( m_FOV, -1 );
613#endif
614
615 // rebuild it (even if we just did it due to camera constraints)
616 buildCameraTransform( &cameraTransform );
617 m_3DCamera->Set_Transform( cameraTransform );
618
619 if (TheTerrainRenderObject)
620 {
621 RefRenderObjListIterator *it = W3DDisplay::m_3DScene->createLightsIterator();
622 TheTerrainRenderObject->updateCenter(m_3DCamera, it);
623 if (it)
624 {
625 W3DDisplay::m_3DScene->destroyLightsIterator(it);
626 it = NULL;
627 }
628 }
629}

1398void W3DView::getAxisAlignedViewRegion(Region3D &axisAlignedRegion)
1399{
1400 //
1401 // get the 4 points in 3D space of the 4 corners of the view, we will use a z = 0.0f
1402 // value so that we can get everything ... even stuff below the terrain
1403 //
1404 Coord3D box[ 4 ];
1405 getScreenCornerWorldPointsAtZ( &box[ 0 ], &box[ 1 ], &box[ 2 ], &box[ 3 ], 0.0f );
1406
1407 //
1408 // take those 4 corners projected into the world and create an axis aligned bounding
1409 // box, we will use this box to iterate the drawables in 3D space
1410 //
1411 axisAlignedRegion.lo = box[ 0 ];
1412 axisAlignedRegion.hi = box[ 0 ];
1413 for( Int i = 0; i < 4; i++ )
1414 {
1415
1416 if( box[ i ].x < axisAlignedRegion.lo.x )
1417 axisAlignedRegion.lo.x = box[ i ].x;
1418 if( box[ i ].y < axisAlignedRegion.lo.y )
1419 axisAlignedRegion.lo.y = box[ i ].y;
1420 if( box[ i ].x > axisAlignedRegion.hi.x )
1421 axisAlignedRegion.hi.x = box[ i ].x;
1422 if( box[ i ].y > axisAlignedRegion.hi.y )
1423 axisAlignedRegion.hi.y = box[ i ].y;
1424
1425 } // end for i
1426
1427 // low and high regions will be based of the extent of the map
1428 Region3D mapExtent;
1429 Real safeValue = 999999;
1430 TheTerrainLogic->getExtent( &mapExtent );
1431 axisAlignedRegion.lo.z = mapExtent.lo.z - safeValue;
1432 axisAlignedRegion.hi.z = mapExtent.hi.z + safeValue;
1433
1434 // we want to overscan a little bit so that we get objects that are partially offscreen
1435 axisAlignedRegion.lo.x -= (DRAWABLE_OVERSCAN + m_guardBandBias.x);
1436 axisAlignedRegion.lo.y -= (DRAWABLE_OVERSCAN + m_guardBandBias.y + 60.0f );
1437 axisAlignedRegion.hi.x += (DRAWABLE_OVERSCAN + m_guardBandBias.x);
1438 axisAlignedRegion.hi.y += (DRAWABLE_OVERSCAN + m_guardBandBias.y);
1439
1440}
