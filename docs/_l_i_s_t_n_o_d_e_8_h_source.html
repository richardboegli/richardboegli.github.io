55/*
56** This is a doubly linked list node. Typical use of this node is to derive
57** objects from this node. The interface class for this node can be used for
58** added convenience.
59*/
60class GenericList;
61class GenericNode {
62 public:
63 GenericNode(void) : NextNode(0), PrevNode(0) {}
64 virtual ~GenericNode(void) {Unlink();}
65 GenericNode(GenericNode & node) {node.Link(this);}
66 GenericNode & operator = (GenericNode & node) {
67 if (&node != this) {
68 node.Link(this);
69 }
70 return(*this);
71 }
72
73 void Unlink(void) {
74 // note that this means that the special generic node at the head
75 // and tail of the list can not be unlinked. This is done because
76 // the user should never unlink them -- it will destroy the list in
77 // an evil way.
78 if (Is_Valid()) {
79 PrevNode->NextNode = NextNode;
80 NextNode->PrevNode = PrevNode;
81 PrevNode = 0;
82 NextNode = 0;
83 }
84 }
85
86 GenericList * Main_List(void) const {
87 GenericNode const * node = this;
88 while (node->PrevNode) {
89 node = PrevNode;
90 }
91 return((GenericList *)this);
92 }
93 void Link(GenericNode * node) {
94 assert(node != (GenericNode *)0);
95 node->Unlink();
96 node->NextNode = NextNode;
97 node->PrevNode = this;
98 if (NextNode) NextNode->PrevNode = node;
99 NextNode = node;
100 }
101
102 GenericNode * Next(void) const {return(NextNode);}
103 GenericNode * Next_Valid(void) const {
104 return ((NextNode && NextNode->NextNode) ? NextNode : (GenericNode *)0);
105 }
106 GenericNode * Prev(void) const {return(PrevNode);}
107 GenericNode * Prev_Valid(void) const {
108 return ((PrevNode && PrevNode->PrevNode) ? PrevNode : (GenericNode *)0);
109 }
110 bool Is_Valid(void) const {return(this != (GenericNode *)0 && NextNode != (GenericNode *)0 && PrevNode != (GenericNode *)0);}
111
112 protected:
113 GenericNode * NextNode;
114 GenericNode * PrevNode;
115};

118/*
119** This is a generic list handler. It manages N generic nodes. Use the interface class
120** to the generic list for added convenience.
121*/
122class GenericList {
123 public:
124 GenericList(void) {
125 FirstNode.Link(&LastNode);
126 }
127
128
129 virtual ~GenericList(void) {
130 while (FirstNode.Next()->Is_Valid()) {
131 FirstNode.Next()->Unlink();
132 }
133 }
134
135 GenericNode * First(void) const {return(FirstNode.Next());}
136 GenericNode * First_Valid(void) const
137 {
138 GenericNode *node = FirstNode.Next();
139 return (node->Next() ? node : (GenericNode *)0);
140 }
141
142 GenericNode * Last(void) const {return(LastNode.Prev());}
143 GenericNode * Last_Valid(void) const
144 {
145 GenericNode *node = LastNode.Prev();
146 return (node->Prev() ? node : (GenericNode *)0);
147 }
148
149 bool Is_Empty(void) const {return(!FirstNode.Next()->Is_Valid());}
150 void Add_Head(GenericNode * node) {FirstNode.Link(node);}
151 void Add_Tail(GenericNode * node) {LastNode.Prev()->Link(node);}
152// void Delete(void) {while (FirstNode.Next()->Is_Valid()) delete FirstNode.Next();}
153
154 int Get_Valid_Count(void) const
155 {
156 GenericNode * node = First_Valid();
157 int counter = 0;
158 while(node) {
159 counter++;
160 node = node->Next_Valid();
161 }
162 return counter;
163 }
164
165 protected:
166 GenericNode FirstNode;
167 GenericNode LastNode;
168
169 private:
170 GenericList(GenericList & list);
171 GenericList & operator = (GenericList const &);
172};

176/*
177** This node class serves only as an "interface class" for the normal node
178** object. In order to use this interface class you absolutely must be sure
179** that the node is the root base object of the "class T". If it is true that the
180** address of the node is the same as the address of the "class T", then this
181** interface class will work. You can usually ensure this by deriving the
182** class T object from this node.
183*/
184template<class T> class List;
185template<class T>
186class Node : public GenericNode {
187 public:
188 List<T> * Main_List(void) const {return((List<T> *)GenericNode::Main_List());}
189 T Next(void) const {return((T)GenericNode::Next());}
190 T Next_Valid(void) const {return((T)GenericNode::Next_Valid());}
191 T Prev(void) const {return((T)GenericNode::Prev());}
192 T Prev_Valid(void) const {return((T)GenericNode::Prev_Valid());}
193 bool Is_Valid(void) const {return(GenericNode::Is_Valid());}
194};

198** This is an "interface class" for a list of nodes. The rules for the class T object
199** are the same as the requirements required of the node class.
200*/
201template<class T>
202class List : public GenericList {
203 public:
204 List(void) {};
205
206 T First(void) const {return((T)GenericList::First());}

281/*
282 The DoubleNode class has double everything a DataNode has!
283 Each DataNode's data member points to the DoubleNode object
284 that owns it. This type of object can therefore be in two distinct
285 lists, and when it is destroyed it is removed from both of them
286 automatically.
287
288 usage example:
289
290 typedef a symbol for the usage of the DoubleNode class as such:
291
292 typedef DoubleNode<int, bool> DOUBLENODE;
293
294 declare the head of the list as a DataNode with type DOUBLENODE
295
296 List<DOUBLENODE *> PrimaryList; // member of object1
297 List<DOUBLENODE *> SecondaryList; // member of object2
298
299 Iterate through the list starting at PrimaryList or SecondaryList.
300 These two lists are intended to be located in different
301 objects, such as an Active list and an Association list as used in G
302 Planet mode.
303*/
304template<class PRIMARY, class SECONDARY>
305class DoubleNode {
306 void Initialize() { Primary.Set(this); Secondary.Set(this); };
307 PRIMARY PrimaryValue;
308 SECONDARY SecondaryValue;
309
310public:
311 typedef DoubleNode<PRIMARY, SECONDARY> Type;
312
313 DataNode<Type *> Primary;
314 DataNode<Type *> Secondary;
315
316 DoubleNode() { Initialize(); };
317 DoubleNode(PRIMARY primary, SECONDARY secondary) { Initialize(); Set_Primary(primary); Set_Secondary(secondary); };
318 void Set_Primary(PRIMARY value) { PrimaryValue = value; };
319 void Set_Secondary(SECONDARY value) { SecondaryValue = value; };
320 PRIMARY Get_Primary() { return PrimaryValue; };
321 SECONDARY Get_Secondary() { return SecondaryValue; };
322 void Unlink() { Primary.Unlink(); Secondary.Unlink(); };
323};
