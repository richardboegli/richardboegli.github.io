84void FireWeaponWhenDeadBehavior::onDie( const DamageInfo *damageInfo )
85{
86 const FireWeaponWhenDeadBehaviorModuleData* d = getFireWeaponWhenDeadBehaviorModuleData();
87 Object *obj = getObject();
88
89 if (!isUpgradeActive())
90 return;
91
92 // right type?
93 if (!d->m_dieMuxData.isDieApplicable(getObject(), damageInfo))
94 return;
95
96 // This will never apply until built. Otherwise canceling construction sets it off, and killing
97 // a one hitpoint one percent building will too.
98 if( obj->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
99 return;
100
101
102 UpgradeMaskType activation, conflicting;
103 getUpgradeActivationMasks( activation, conflicting );
104
105 if( obj->getObjectCompletedUpgradeMask().testForAny( conflicting ) )
106 {
107 return;
108 }
109 if( obj->getControllingPlayer() && obj->getControllingPlayer()->getCompletedUpgradeMask().testForAny( conflicting ) )
110 {
111 return;
112 }
113
114 if (d->m_deathWeapon)
115 {
116 // fire the default weapon
117 TheWeaponStore->createAndFireTempWeapon(d->m_deathWeapon, obj, obj->getPosition());
118 }
119}
