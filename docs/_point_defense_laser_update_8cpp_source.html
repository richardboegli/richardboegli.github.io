97void PointDefenseLaserUpdate::onObjectCreated()
98{
99 const PointDefenseLaserUpdateModuleData *data = getPointDefenseLaserUpdateModuleData();
100
101 //Make sure we have a weapon template
102 if( !data->m_weaponTemplate )
103 {
104 DEBUG_CRASH( ("PointDefenseLaserUpdate for %s doesn't have a valid weapon template",
105 getObject()->getTemplate()->getName().str() ) );
106 return;
107 }
108
109 //Make sure our firing range is smaller than the scan range.
110 WeaponBonus bonus;
111 bonus.clear();
112 Real attackRange = data->m_weaponTemplate->getAttackRange( bonus );
113 if( data->m_scanRange <= attackRange )
114 {
115 DEBUG_CRASH( ("PointDefenseLaserUpdate for %s requires the scan range (%.1f) being larger than the firing range (%.1f)",
116 getObject()->getTemplate()->getName().str(), data->m_scanRange, attackRange ) );
117 }
118}

123UpdateSleepTime PointDefenseLaserUpdate::update()
124{
126 //*** HERE'S THE UPDATE PHILOSOPHY ***
127 //The point defense laser typically has short range, high rate of fire, and shoots at incoming projectiles
128 //that move fast. This amounts to a potentially very expensive system. Instead of frantically scanning for
129 //targets, we will scan less frequently (data->m_scanFrames) in a larger radius (data->m_scanRange). When
130 //this occurs, we'll store the "best" target, and track only that target until the next update or if it is
131 //killed.
132
133 Object *me = getObject();
134 if( me->isEffectivelyDead() )
135 return UPDATE_SLEEP_FOREVER;//No more laser fo you.
136
137 const PointDefenseLaserUpdateModuleData *data = getPointDefenseLaserUpdateModuleData();
138
139 //Optimized firing at acquired target
140 if( m_nextScanFrames > 0 )
141 {
142 m_nextScanFrames--;
143 fireWhenReady(); //Only happens if something is tracked.
144 return UPDATE_SLEEP_NONE;
145 }
146 m_nextScanFrames = data->m_scanFrames;
147
148 //Periodic scanning (expensive)
149 if( scanClosestTarget() )
150 {
151 //1 frame can make a big difference so fire ASAP!
152 fireWhenReady();
153 }
154 return UPDATE_SLEEP_NONE;
155}

158void PointDefenseLaserUpdate::fireWhenReady()
159{
160 const PointDefenseLaserUpdateModuleData *data = getPointDefenseLaserUpdateModuleData();
161
162 //Track our target
163 Object *target = TheGameLogic->findObjectByID( m_bestTargetID );
164 if( target )
165 {
166 WeaponBonus bonus;
167 bonus.clear();
168 Real fireRange = data->m_weaponTemplate->getAttackRange( bonus );
169 Object *me = getObject();
170 Real fDist = sqrt( ThePartitionManager->getDistanceSquared( me, target, FROM_CENTER_2D ) );
171 if( fDist < fireRange )
172 {
173 //We are currently in range!
174 m_inRange = true;
175 }
176 else
177 {
178 if( m_inRange )
179 {
180 //We were in range last frame, but the target has moved out of firing range, so
181 //re-evaluate by forcing a new scan.
182 m_nextScanFrames = GameLogicRandomValue( 0, 3 );
183 m_bestTargetID = INVALID_ID;
184 if( !m_nextScanFrames )
185 {
186 scanClosestTarget();
187 m_nextScanFrames = data->m_scanFrames;
188 target = NULL; //Set target to NULL so we don't shoot at it (might be out of range)
189 }
190 }
191 else
192 {
193 //Not in range
194 m_inRange = false;
195 }
196 }
197 }
198
199 if( m_nextShotAvailableInFrames > 0 )
200 {
201 //We can't fire this frame.
202 m_nextShotAvailableInFrames--;
203 return;
204 }
205
206 WeaponTemplate *wt = data->m_weaponTemplate;
207 if( wt )
208 {
209 WeaponBonus bonus;
210
211 //Fire control!
212 if( target && m_inRange )
213 {
214 if( !target->isEffectivelyDead() )
215 {
216 Weapon* w = TheWeaponStore->allocateNewWeapon( wt, TERTIARY_WEAPON );
217 w->loadAmmoNow( getObject() );
218 w->fireWeapon( getObject(), target );
219 w->deleteInstance();
220
221 // And now that we have shot, set our internal reload timer.
222 m_nextShotAvailableInFrames = wt->getDelayBetweenShots( bonus );
223 }
224
225 if( target->isEffectivelyDead() )
226 {
227 m_nextScanFrames = GameLogicRandomValue( 0, 3 );
228 m_bestTargetID = INVALID_ID;
229 if( !m_nextScanFrames )
230 {
231 scanClosestTarget();
232 m_nextScanFrames = data->m_scanFrames;
233 }
234 }
235 }
236 }

241Object* PointDefenseLaserUpdate::scanClosestTarget()
242{
243 const PointDefenseLaserUpdateModuleData *data = getPointDefenseLaserUpdateModuleData();
244 Object *me = getObject();
245 Object *bestTargetOutOfRange[2] = { NULL, NULL };
246 Object *bestTargetInRange[2] = { NULL, NULL };
247 Real closestDist[2];
248 Real closestOutsideRange[2];
249 Int index;
250 WeaponBonus bonus;
251 bonus.clear();
252 Real fireRange = data->m_weaponTemplate->getAttackRange( bonus );
253
254 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( me->getPosition(), data->m_scanRange, FROM_CENTER_2D );
255 MemoryPoolObjectHolder hold(iter);
256
257 for( Object *other = iter->first(); other; other = iter->next() )
258 {
259 if( other->isAnyKindOf( data->m_primaryTargetKindOf ) )
260 {
261 //Primary target type
262 index = 0;
263 }
264 else if( other->isAnyKindOf( data->m_secondaryTargetKindOf ) )
265 {
266 //Secondary target type (use only if we can't find any primary targets)
267 index = 1;
268 }
269 else
270 {
271 //Not a valid target.
272 continue;
273 }
274
275 // Since we don't have an actual weapon or a weapon set, we lose all of the automatic checks.
276 // "Borrow" the check for being an AA only laser to stop from shooting planes on airports.
277 if( !other->isAirborneTarget() && !(data->m_weaponTemplate->getAntiMask() & WEAPON_ANTI_GROUND) )
278 continue;
279
280 // order matters: we want to know if I consider it to be an enemy, not vice versa
281 if( getObject()->getRelationship( other ) != ENEMIES )
282 {
283 //Don't kill our friends!
284 continue;
285 }
286
287 if( other->testStatus( OBJECT_STATUS_STEALTHED ) && !other->testStatus( OBJECT_STATUS_DETECTED ) && !other->testStatus( OBJECT_STATUS_DISGUISED ) )
288 {
289 //We can't see it.
290 continue;
291 }
292
293 Real fDist = sqrt( ThePartitionManager->getDistanceSquared( me, other, FROM_CENTER_2D ) );
294
295 if( fDist <= fireRange )
296 {
297 //Inside fire range (which one is closest?)
298 if( !bestTargetInRange[index] || fDist < closestDist[index] )
299 {
300 closestDist[index] = fDist;
301 bestTargetInRange[index] = other;
302 }
303 }
304 else if( !bestTargetInRange[index] )
305 {
306 //Outside fire range.
307
308 //Determine where the target will be based on current velocity using (m_velocityFactor * frames)
309 if( data->m_velocityFactor != 0.0f && !other->isKindOf( KINDOF_IMMOBILE ) )
310 {
311 Coord3D pos;
312 PhysicsBehavior *physics = other->getPhysics();
313 if( physics )
314 {
315 pos.set( physics->getVelocity() );
316 pos.scale( data->m_velocityFactor );
317 pos.add( other->getPosition() );
318
319 //Recalculate the distance.
320 fDist = sqrt( ThePartitionManager->getDistanceSquared( me, other, FROM_CENTER_2D ) );
321 }
322 }
323
324 //Now calculate the best outside range target.
325 if( !bestTargetOutOfRange[index] || fDist < closestOutsideRange[index] )
326 {
327 closestOutsideRange[index] = fDist;
328 bestTargetOutOfRange[index] = other;
329 }
330 }
331 } // end for, other
332
333 if( bestTargetInRange[ 0 ] )
334 {
335 //This is the best primary target in range.
336 m_bestTargetID = bestTargetInRange[ 0 ]->getID();
337 m_inRange = true;
338 return bestTargetInRange[ 0 ];
339 }
340
341 if( bestTargetInRange[ 1 ] )
342 {
343 //This is the best secondary target in range.
344 m_bestTargetID = bestTargetInRange[ 1 ]->getID();
345 m_inRange = true;
346 return bestTargetInRange[ 1 ];
347 }
348
349 if( bestTargetOutOfRange[ 0 ] )
350 {
351 //This is the best primary target out of range.
352 m_bestTargetID = bestTargetOutOfRange[ 0 ]->getID();
353 m_inRange = false;
354 return bestTargetOutOfRange[ 0 ];
355 }
356
357 if( bestTargetOutOfRange[ 1 ] )
358 {
359 //This is the best secondary target out of range.
360 m_bestTargetID = bestTargetOutOfRange[ 1 ]->getID();
361 m_inRange = false;
362 return bestTargetOutOfRange[ 1 ];
363 }
364
365 //Utter failure -- nothing on the scope.
366 m_bestTargetID = INVALID_ID;
367 m_inRange = false;
368 return NULL;
369}
