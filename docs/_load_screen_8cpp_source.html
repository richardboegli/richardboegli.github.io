734// accepts the number of chars to advance, the window we're concerned with, the total text for final display, and the current position of the readout
735// returns the updated position of the readout
736Int updateTeletypeText( Int num_chars, GameWindow* window, UnicodeString full_text, Int current_text_pos )
737{
738 DEBUG_ASSERTCRASH(window, ("No window for teletype text update"));
739 UnicodeString currentText = GadgetStaticTextGetText(window);
740 WideChar wChar;
741 for (Int i = 0; i < num_chars; i++)
742 {
743 if (current_text_pos < full_text.getLength())
744 {
745 wChar = full_text.getCharAt(current_text_pos);
746 currentText.concat(wChar);
747 current_text_pos++;
748 }
749 }
750 GadgetStaticTextSetText(window, currentText);
751 return current_text_pos;
