57Int VeterancyCrateCollide::getLevelsToGain() const
58{
59 const VeterancyCrateCollideModuleData* d = getVeterancyCrateCollideModuleData();
60 if (!d || !d->m_addsOwnerVeterancy)
61 return 1;
62

69Bool VeterancyCrateCollide::isValidToExecute( const Object *other ) const
70{
71 const VeterancyCrateCollideModuleData* d = getVeterancyCrateCollideModuleData();
72 if( !d )
73 {
74 return false;
75 }
76
77 if(!CrateCollide::isValidToExecute(other))
78 return false;
79
80 if (other->isEffectivelyDead())
81 return false;
82
83 if( other->isSignificantlyAboveTerrain() )
84 {
85 return false;
86 }
87
88 Int levelsToGain = getLevelsToGain();
89 if (levelsToGain <= 0)
90 return false;
91
92 const ExperienceTracker *et = other->getExperienceTracker();
93 if( !et || !et->isTrainable() )
94 {
95 //If the other unit can't gain experience, then we can't help promote it!
96 return false;
97 }
98
99 if (!et || !et->canGainExpForLevel(levelsToGain))
100 return false;
101
102 if( d->m_isPilot )
103 {
104 if( other->getControllingPlayer() != getObject()->getControllingPlayer() )
105 {
106 //This is a pilot and we are checking to make sure the pilot is entering a vehicle on
107 //the same team. If it's not, then don't allow it.. this is particularly the case for
108 //pilots attempting to enter civilian vehicles.
109 return false;
110 }
111
112 if( other->isUsingAirborneLocomotor() )
113 {
114 // Can't upgrade a helicopter or plane, but we will think we can for a moment while it
115 // is on the ground from being built.
116 return false;
117 }
118 }
119
120 return true;
121}

125Bool VeterancyCrateCollide::executeCrateBehavior( Object *other )
126{
127 //Make sure the pilot is actually *TRYING* to enter the object
128 //unlike other crates
129 AIUpdateInterface *ai = (AIUpdateInterface*)getObject()->getAIUpdateInterface();
130 const VeterancyCrateCollideModuleData *md = getVeterancyCrateCollideModuleData();
131
132 if( !ai || ai->getGoalObject() != other )
133 {
134 return false;
135 }
136
137 Int levelsToGain = getLevelsToGain();
138 Real range = md->m_rangeOfEffect;
139 if (range == 0)
140 {
141 // do just the collider
142 if (other != NULL)
143 {
144 other->getExperienceTracker()->gainExpForLevel( levelsToGain, ( ! md->m_isPilot) );
145 }
146 }
147 else
148 {
149 PartitionFilterSamePlayer othersPlayerFilter( other->getControllingPlayer() );
150 PartitionFilterSameMapStatus filterMapStatus(other);
151 PartitionFilter *filters[] = { &othersPlayerFilter, &filterMapStatus, NULL };
152 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( other, range, FROM_CENTER_2D, filters, ITER_FASTEST );
153 MemoryPoolObjectHolder hold(iter);
154
155 for( Object *potentialObject = iter->first(); potentialObject; potentialObject = iter->next() )
156 {
157 // This function will give just enough exp for the Object to gain a level, if it can
158 potentialObject->getExperienceTracker()->gainExpForLevel( levelsToGain, ( ! md->m_isPilot) );
159 }
160 }
161
162 //In order to make things easier for the designers, we are going to transfer the terrorist name
163 //to the car... so the designer can control the car with their scripts.
164 if( md->m_isPilot )
165 {
166 TheScriptEngine->transferObjectName( getObject()->getName(), other );
167 }
168
169 return TRUE;
170}
