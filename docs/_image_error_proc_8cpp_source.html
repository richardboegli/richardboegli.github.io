69BOOL CALLBACK ImageErrorProc( HWND hWndDialog, UINT message,
70 WPARAM wParam, LPARAM lParam )
71{
72
73 switch( message )
74 {
75
76 // ------------------------------------------------------------------------
77 case WM_INITDIALOG:
78 {
79
80 //
81 // load the listbox with images that could not be processed
82 // and the reasons for it
83 //
84
85 // sanity
86 if( TheImagePacker == NULL )
87 return TRUE;
88
89 // go through all images
90 Int i, count;
91 HWND list = GetDlgItem( hWndDialog, LIST_IMAGES );
92 char buffer[ _MAX_PATH + 256 ];
93 char reason[ 32 ];
94 ImageInfo *image;
95
96 count = TheImagePacker->getImageCount();
97 for( i = 0; i < count; i++ )
98 {
99
100 // get image
101 image = TheImagePacker->getImage( i );
102
103 // sanity
104 if( image == NULL )
105 continue;
106
107 // if image can't be processed find out why
108 if( BitTest( image->m_status, ImageInfo::CANTPROCESS ) )
109 {
110
111 if( BitTest( image->m_status, ImageInfo::TOOBIG ) )
112 sprintf( reason, "Too Big" );
113 else if( BitTest( image->m_status, ImageInfo::INVALIDCOLORDEPTH ) )
114 sprintf( reason, "Unsupported Color Depth" );
115 else
116 sprintf( reason, "Unknown Reason" );
117
118 sprintf( buffer, "%s: (%dx%dx%d) %s",
119 reason, image->m_size.x, image->m_size.y, image->m_colorDepth,
120 image->m_path );
121
122 SendMessage( list, LB_INSERTSTRING, -1, (LPARAM)buffer );
123
124 } // end if
125
126 } // end for i
127
128 // set the extents for the horizontal scroll bar in the listbox
129 SendMessage( list, LB_SETHORIZONTALEXTENT, 1280, 0 );
130
131 return TRUE;
132
133 } // end init
134
135 // ------------------------------------------------------------------------
136 case WM_COMMAND:
137 {
138 Int controlID = LOWORD( wParam );
139// Int notifyCode = HIWORD( wParam );
140// HWND hWndControl = (HWND)lParam;
141
142 switch( controlID )
143 {
144
145 // --------------------------------------------------------------------
146 case BUTTON_PROCEED:
147 {
148
149 EndDialog( hWndDialog, TRUE );
150 break;
151
152 } // end proceed
153
154 // --------------------------------------------------------------------
155 case BUTTON_CANCEL:
156 {
157
158 EndDialog( hWndDialog, FALSE );
159 break;
160
161 } // end cancel
162
163 } // end switch
164
165 break;
166
167 } // end command
168
169 } // end switch message
170
171 return 0;
172
173} // end ImageErrorProc
