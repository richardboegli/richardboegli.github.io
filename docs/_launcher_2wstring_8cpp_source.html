123bit8 Wstring::cat(IN char *s)
124{
125 char *oldStr;
126 uint32 len;
127
128 if (s==NULL) // it's OK to cat nothing
129 return(TRUE);
130
131 // Save the contents of the string.
132 oldStr = str;
133
134 // Determine the length of the resultant string.
135 len = strlen(s) + 1;
136 if(oldStr)
137 len += strlen(oldStr);
138
139 // Allocate memory for the new string.
140
141 if(!(str = new char[(len * sizeof(char))]))
142 {
143 str = oldStr;
144 return(FALSE);
145 }
146
147 // Copy the contents of the old string and concatenate the passed
148 // string.
149 if(oldStr) strcpy(str, oldStr);
150 else str[0] = 0;
151
152 strcat(str, s);
153
154 // delete the old string.
155 if(oldStr)
156 delete[](oldStr);
157
158 return(TRUE);
159}

229//
230// This function deletes 'count' characters indexed by `pos' from the Wstring.
231// If `pos'+'count' is > the length of the array, the last 'count' characters
232// of the string are removed. If an error occurs, FALSE is returned.
233// Otherwise, TRUE is returned. Note: count has a default value of 1.
234//
235//
236char Wstring::remove(sint32 pos,sint32 count)
237{
238 char *s;
239 sint32 len;
240
241 if (!str)
242 return(FALSE);
243
244 len = (sint32)strlen(str);
245
246 if(pos+count > len)
247 pos = len - count;
248 if (pos < 0)
249 {
250 count+=pos; // If they remove before 0, ignore up till beginning
251 pos=0;
252 }
253 if (count<=0)
254 return(FALSE);
255
256 if(!(s = new char[len-count+1]))
257 {
258 //ErrorMessage(SET_EM, "Insufficient memory to modify Wstring.");
259 return(FALSE);
260 }
261
263
264 // put nulls on both ends of substring to be removed
265 str[pos] = 0;
266 str[pos+count-1] = 0;
267
268 strcpy(s, str);
269 strcat(s, &(str[pos + count]));
270 delete[](str);
271 str = s;
272
273 return(TRUE);
274}

276// Remove all instances of a char from the string
277bit8 Wstring::removeChar(char c)
278{
279 int len=0;
280 char *cptr=NULL;
281 bit8 removed=FALSE;
282
283 if (str==NULL)
284 return(FALSE);
285
286 len=strlen(str);
287 while ((cptr=strchr(str,c)) !=NULL)
288 {
289 memmove(cptr,cptr+1,len-1-((int)(cptr-str)));
290 len--;
291 str[len]=0;
292 removed=TRUE;
293 }
294 if (removed)
295 {
296 char *newStr=new char[strlen(str)+1];
297 strcpy(newStr,str);
298 delete[](str);
299 str=newStr;
300 }
301 return(removed);
302}

358// Insert at given position and shift old stuff to right
359bit8 Wstring::insert(char *instring, uint32 pos)
360{
361 if (str==NULL)
362 return(set(instring));
363 if (pos>strlen(str))
364 pos=strlen(str);
365 char *newstr=new char[strlen(str)+strlen(instring)+1];
366 memset(newstr,0,strlen(str)+strlen(instring)+1);
367 strcpy(newstr,str);
368 // move the old data out of the way
369 int bytes=strlen(str)+1-pos;
370 memmove(newstr+pos+strlen(instring),newstr+pos,bytes);
371 // move the new data into place
372 memmove(newstr+pos,instring,strlen(instring));
373 delete[](str);
374 str=newstr;
375 return(TRUE);
376}

378// This function inserts the character specified by `k' into the string at the
379// position indexed by `pos'. If `pos' is >= the length of the string, it is
380// appended to the string. If an error occurs, FALSE is returned. Otherwise,
381// TRUE is returned.
382bit8 Wstring::insert(char k, uint32 pos)
383{
384 char *s;
385 uint32 len;
386 char c[2];
387
388 if(!str)
389 {
390 c[0] = k;
391 c[1] = 0;
392 return(set(c));
393 }
394
395 len = (uint32)strlen(str);
396
397 if(pos > len)
398 pos = len;
399
400 if(!(s = (char *)new char[(len + 2)]))
401 {
402 //ErrorMessage(SET_EM, "Insufficient memory to modify Wstring.");
403 return(FALSE);
404 }
405
406 c[0] = str[pos];
407 str[pos] = 0;
408 c[1] = 0;
409 strcpy(s, str);
410 str[pos] = c[0];
411 c[0] = k;
412 strcat(s, c);
413 strcat(s, &(str[pos]));
414 delete[](str);
415 str = s;
416
417 return(TRUE);
418}

421// This function replaces any occurences of the string pointed to by
422// `replaceThis' with the string pointed to by `withThis'. If an error
423// occurs, FALSE is returned. Otherwise, TRUE is returned.
424bit8 Wstring::replace(char *replaceThis, char *withThis)
425{
426 Wstring dest;
427 char *foundStr, *src;
428 uint32 len;
429
430 src=get();
431 while(src && src[0])
432 {
433 foundStr = strstr(src, replaceThis);
434 if(foundStr)
435 {
436 len = (uint32)foundStr - (uint32)src;
437 if(len)
438 {
439 if(!dest.cat(len, src))
440 return(FALSE);
441 }
442 if(!dest.cat(withThis))
443 return(FALSE);
444 src = foundStr + strlen(replaceThis);
445 }
446 else
447 {
448 if(!dest.cat(src))
449 return(FALSE);
450
451 src=NULL;
452 }
453 }
454 return(set(dest.get()));
455}

564// Get a token from this string that's seperated by one or more
565// chars from the 'delim' string , start at offset & return offset
566sint32 Wstring::getToken(int offset,char *delim,Wstring &out)
567{
568 int i;
569 sint32 start;
570 sint32 stop;
571 for (i=offset; i<(int)length(); i++) {
572 if(strchr(delim,str[i])==NULL)
573 break;
574 }
575 if (i>=(int)length())
576 return(-1);
577 start=i;
578
579 for (; i<(int)length(); i++) {
580 if(strchr(delim,str[i])!=NULL)
581 break;
582 }
583 stop=i-1;
584 out.set(str+start);
585 out.truncate((uint32)stop-start+1);
586 return(stop+1);
587}

589// Get the first line of text after offset. Lines are terminated by '\r\n' or '\n'
590sint32 Wstring::getLine(int offset, Wstring &out)
591{
592 int i;
593 sint32 start;
594 sint32 stop;
595
596 start=i=offset;
597 if (start >= (int)length())
598 return(-1);
599
600 for (; i<(int)length(); i++) {
601 if(strchr("\r\n",str[i])!=NULL)
602 break;
603 }
604 stop=i;
605 if ((str[stop]=='\r')&&(str[stop+1]=='\n'))
606 stop++;
607
608 out.set(str+start);
609 out.truncate((uint32)stop-start+1);
610 return(stop+1);
611}
