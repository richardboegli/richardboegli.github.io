52void ObjectHelper::sleepUntil(UnsignedInt when)
53{
54 if( getObject()->getStatusBits().test( OBJECT_STATUS_DESTROYED ) )
55 return;
56
57 // note the setWakeFrame(NEVER) actually awakens immediately, since NEVER==0.
58 // when we get NEVER in this case, we really want to sleep forever.
59 // so just special case it.
60 UpdateSleepTime wakeDelay = (when == NEVER || when == FOREVER) ?
61 UPDATE_SLEEP_FOREVER :
62 UPDATE_SLEEP(when - TheGameLogic->getFrame());
63 setWakeFrame(getObject(), wakeDelay);
64}
