42GameMessageDisposition HintSpyTranslator::translateGameMessage(const GameMessage *msg)
43{
44 GameMessageDisposition disp = KEEP_MESSAGE;
45
47 switch( msg->getType() )
48 {
49 //-----------------------------------------------------------------------------
50 case GameMessage::MSG_MOUSEOVER_DRAWABLE_HINT:
51 {
52 TheInGameUI->createMouseoverHint( msg );
53
54 disp = DESTROY_MESSAGE; //hint no longer needed by anyone. Eat it.
55 }
56 break;
57 case GameMessage::MSG_MOUSEOVER_LOCATION_HINT:
58 {
59 TheInGameUI->createMouseoverHint( msg );
60
61 disp = DESTROY_MESSAGE; //hint no longer needed by anyone. Eat it.
62 }
63 break;
64
65 //-----------------------------------------------------------------------------
66 case GameMessage::MSG_DEFECTOR_HINT:
67
68 disp = DESTROY_MESSAGE; //hint no longer needed by anyone. Eat it.
69
70 case GameMessage::MSG_DO_MOVETO_HINT:
71 case GameMessage::MSG_DO_ATTACKMOVETO_HINT:
72 case GameMessage::MSG_DO_ATTACK_OBJECT_HINT:
73 case GameMessage::MSG_DO_ATTACK_OBJECT_AFTER_MOVING_HINT:
74 case GameMessage::MSG_DO_FORCE_ATTACK_OBJECT_HINT:
75 case GameMessage::MSG_DO_FORCE_ATTACK_GROUND_HINT:
76 case GameMessage::MSG_ADD_WAYPOINT_HINT:
77 case GameMessage::MSG_GET_REPAIRED_HINT:
78 case GameMessage::MSG_DOCK_HINT:
79 case GameMessage::MSG_GET_HEALED_HINT:
80 case GameMessage::MSG_DO_REPAIR_HINT:
81 case GameMessage::MSG_RESUME_CONSTRUCTION_HINT:
82 case GameMessage::MSG_ENTER_HINT:
83 case GameMessage::MSG_HIJACK_HINT:
84 case GameMessage::MSG_SABOTAGE_HINT:
85 case GameMessage::MSG_CONVERT_TO_CARBOMB_HINT:
86#ifdef ALLOW_SURRENDER
87 case GameMessage::MSG_PICK_UP_PRISONER_HINT:
88#endif
89 case GameMessage::MSG_VALID_GUICOMMAND_HINT:
90 case GameMessage::MSG_INVALID_GUICOMMAND_HINT:
91 case GameMessage::MSG_CAPTUREBUILDING_HINT:
92 case GameMessage::MSG_HACK_HINT:
93 case GameMessage::MSG_SET_RALLY_POINT_HINT:
94 case GameMessage::MSG_IMPOSSIBLE_ATTACK_HINT:
95 case GameMessage::MSG_DO_SPECIAL_POWER_OVERRIDE_DESTINATION_HINT:
96 case GameMessage::MSG_DO_SALVAGE_HINT:
97 case GameMessage::MSG_DO_INVALID_HINT:
98 TheInGameUI->createCommandHint( msg );
99 disp = DESTROY_MESSAGE; //hint no longer needed by anyone. Eat it.
100 break;
101
102 //-----------------------------------------------------------------------------
103 case GameMessage::MSG_AREA_SELECTION_HINT:
104 TheInGameUI->beginAreaSelectHint( msg );
105 break;
106
107 //-----------------------------------------------------------------------------
108 // An AREA_SELECTION_HINT is always followed by an AREA_SELECTION, so
109 // watch for it to stop hinting.
110 case GameMessage::MSG_AREA_SELECTION:
111 TheInGameUI->endAreaSelectHint( msg );
112 break;
113
114 //-----------------------------------------------------------------------------
115 case GameMessage::MSG_DO_MOVETO:
116 case GameMessage::MSG_DO_ATTACKMOVETO:
117 case GameMessage::MSG_DO_FORCEMOVETO:
118 case GameMessage::MSG_ADD_WAYPOINT:
119 TheInGameUI->createMoveHint( msg );
120 break;
121
122 //-----------------------------------------------------------------------------
123 case GameMessage::MSG_DO_ATTACK_OBJECT:
124 TheInGameUI->createAttackHint( msg );
125 break;
126
127 //-----------------------------------------------------------------------------
128 case GameMessage::MSG_DO_FORCE_ATTACK_GROUND:
129 case GameMessage::MSG_DO_FORCE_ATTACK_OBJECT:
130 TheInGameUI->createForceAttackHint( msg );
131 break;
132
133 //-----------------------------------------------------------------------------
134 case GameMessage::MSG_ENTER:
135 TheInGameUI->createGarrisonHint( msg );
136 break;
137
138 }
139 return disp;
140}
