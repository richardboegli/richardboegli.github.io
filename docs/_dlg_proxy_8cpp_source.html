38CBabylonDlgAutoProxy::CBabylonDlgAutoProxy()
39{
40 EnableAutomation();
41
42 // To keep the application running as long as an automation
43 // object is active, the constructor calls AfxOleLockApp.
44 AfxOleLockApp();
45
46 // Get access to the dialog through the application's
47 // main window pointer. Set the proxy's internal pointer
48 // to point to the dialog, and set the dialog's back pointer to
49 // this proxy.
50 ASSERT (AfxGetApp()->m_pMainWnd != NULL);
51 ASSERT_VALID (AfxGetApp()->m_pMainWnd);
52 ASSERT_KINDOF(CBabylonDlg, AfxGetApp()->m_pMainWnd);
53 m_pDialog = (CBabylonDlg*) AfxGetApp()->m_pMainWnd;
54 m_pDialog->m_pAutoProxy = this;
55}
56
57CBabylonDlgAutoProxy::~CBabylonDlgAutoProxy()
58{
59 // To terminate the application when all objects created with
60 // with automation, the destructor calls AfxOleUnlockApp.
61 // Among other things, this will destroy the main dialog
62 if (m_pDialog != NULL)
63 m_pDialog->m_pAutoProxy = NULL;
64 AfxOleUnlockApp();
65}
