41// Desc: Draw buffer to handle all the waypoints in the scene. Waypoints
42// are rendered after terrain, after roads & bridges, and after
43// global fog, but before structures, objects, units, trees, etc.
44// This way if we have two waypoints at the bottom of a hill but
45// going through the hill, the line won't get cut off. However,
46// structures and units on top of paths will render above it. Waypoints
47// are only shown for selected units while in waypoint plotting mode.
48//

155void W3DWaypointBuffer::drawWaypoints(RenderInfoClass &rinfo)
156{
157
158 if ( ! TheInGameUI )
159 return;
160
161
162 setDefaultLineStyle();
163
164
165
166 if( TheInGameUI->isInWaypointMode() )
167 {
168 //Create a default light environment with no lights and only full ambient.
169 //@todo: Fix later by copying default scene light environement from W3DScene.cpp.
170 LightEnvironmentClass lightEnv;
171 lightEnv.Reset(Vector3(0,0,0), Vector3(1.0f,1.0f,1.0f));
172 lightEnv.Pre_Render_Update(rinfo.Camera.Get_Transform());
173 RenderInfoClass localRinfo(rinfo.Camera);
174 localRinfo.light_environment=&lightEnv;
175 Vector3 points[ MAX_DISPLAY_NODES + 1 ]; //Lines have nodes + 1 points.
176
177 const DrawableList *selected = TheInGameUI->getAllSelectedDrawables();
178 Drawable *draw;
179 for( DrawableListCIt it = selected->begin(); it != selected->end(); ++it )
180 {
181 draw = *it;
182 Object *obj = draw->getObject();
183 Int numPoints = 1;
184 if( obj && ! obj->isKindOf( KINDOF_IGNORED_IN_GUI ))//so mobs and stuff sont make a gazillion lines
185 {
186 AIUpdateInterface *ai = obj->getAI();
187 Int goalSize = ai ? ai->friend_getWaypointGoalPathSize() : 0;
188 Int gpIdx = ai ? ai->friend_getCurrentGoalPathIndex() : 0;
189 if( ai && gpIdx >= 0 && gpIdx < goalSize )
190 {
191 const Coord3D *pos = obj->getPosition();
192 points[ 0 ].Set( Vector3( pos->x, pos->y, pos->z ) );
193
194 for( int i = gpIdx; i < goalSize; i++ )
195 {
196 const Coord3D *waypoint = ai->friend_getGoalPathPosition( i );
197 if( waypoint )
198 {
199 //Render line from previous point to current node.
200
201 if( numPoints < MAX_DISPLAY_NODES + 1 )
202 {
203 points[ numPoints ].Set( Vector3( waypoint->x, waypoint->y, waypoint->z ) );
204 numPoints++;
205 }
206
207 m_waypointNodeRobj->Set_Position(Vector3(waypoint->x,waypoint->y,waypoint->z));
208 WW3D::Render(*m_waypointNodeRobj,localRinfo);
209 }
210 }
211 //Now render the lines in one pass!
212 m_line->Set_Points( numPoints, points );
213 m_line->Render( localRinfo );
214 }
215 }
216 }
217 }
218 else // maybe we want to draw rally points, then?
219 {
220 //Create a default light environment with no lights and only full ambient.
221 //@todo: Fix later by copying default scene light environement from W3DScene.cpp.
222 LightEnvironmentClass lightEnv;
223 lightEnv.Reset(Vector3(0,0,0), Vector3(1.0f,1.0f,1.0f));
224 lightEnv.Pre_Render_Update(rinfo.Camera.Get_Transform());
225 RenderInfoClass localRinfo(rinfo.Camera);
226 localRinfo.light_environment=&lightEnv;
227 Vector3 points[ MAX_DISPLAY_NODES + 1 ]; //Lines have nodes + 1 points.
228
229 const DrawableList *selected = TheInGameUI->getAllSelectedDrawables();
230 Drawable *draw;
231 for( DrawableListCIt it = selected->begin(); it != selected->end(); ++it )
232 {
233 draw = *it;
234 Object *obj = draw->getObject();
235
236 Int numPoints = 0;
237 if( obj )
238 {
239 if ( ! obj->isLocallyControlled())
240 continue;
241
242
243
244 // WAIT! before we go browsing the drawable list for buildings that want to draw their rally points
245 // lets test for that very special case of having a listeningoutpost selected, and some enemy drawable moused-over
246 if ( obj->isKindOf( KINDOF_REVEALS_ENEMY_PATHS ) )
247 {
248
249 DrawableID enemyID = TheInGameUI->getMousedOverDrawableID();
250 Drawable *enemyDraw = TheGameClient->findDrawableByID( enemyID );
251 if ( enemyDraw )
252 {
253 Object *enemy = enemyDraw->getObject();
254 if ( enemy )
255 {
256 if ( enemy->getRelationship( obj ) == ENEMIES )
257 {
258
259 Coord3D delta = *obj->getPosition();
260 delta.sub( enemy->getPosition() );
261 if ( delta.length() <= obj->getVisionRange() ) // is listening outpost close enough to do this?
262 {
263
264
266 AIUpdateInterface *ai = enemy->getAI();
267 Int goalSize = ai ? ai->friend_getWaypointGoalPathSize() : 0;
268 Int gpIdx = ai ? ai->friend_getCurrentGoalPathIndex() : 0;
269 if( ai )
270 {
271 Bool lineExists = FALSE;
272
273 const Coord3D *pos = enemy->getPosition();
274 points[ numPoints++ ].Set( Vector3( pos->x, pos->y, pos->z ) );
275
276 if ( gpIdx >= 0 && gpIdx < goalSize )// Ooh, the enemy is in waypoint mode
277 {
278
279 for( int i = gpIdx; i < goalSize; i++ )
280 {
281 const Coord3D *waypoint = ai->friend_getGoalPathPosition( i );
282 if( waypoint )
283 {
284 //Render line from previous point to current node.
285
286 if( numPoints < MAX_DISPLAY_NODES + 1 )
287 {
288 points[ numPoints++ ].Set( Vector3( waypoint->x, waypoint->y, waypoint->z ) );
289 }
290
291 m_waypointNodeRobj->Set_Position(Vector3(waypoint->x,waypoint->y,waypoint->z));
292 WW3D::Render(*m_waypointNodeRobj,localRinfo);
293 lineExists = TRUE;
294 }
295 }
296 }
297 else // then enemy may be moving to a goal position
298 {
299 const Coord3D *destinationPoint = ai->getGoalPosition();
300 if ( destinationPoint->length() > 1.0f )
301 {
302 points[ numPoints++ ].Set( Vector3( destinationPoint->x, destinationPoint->y, destinationPoint->z ) );
303 m_waypointNodeRobj->Set_Position(Vector3(destinationPoint->x,destinationPoint->y,destinationPoint->z));
304 WW3D::Render(*m_waypointNodeRobj,localRinfo);
305 lineExists = TRUE;
306 }
307 }
308
309 if ( lineExists )
310 {
311 //Now render the lines in one pass!
312
313 m_line->Set_Color( Vector3( 0.95f, 0.5f, 0.0f ) );
314 m_line->Set_Width( 3.0f );
315
316 m_line->Set_Points( numPoints, points );
317 m_line->Render( localRinfo );
318 }
319 }
321
322
323
324
325 }
326 }
327 }
328 }
329
330 break;// dont even bother with the rest, since this one listening outpost satisfies the single path-line limit
331 }
332
333
334
335
336 ExitInterface *exitInterface = obj->getObjectExitInterface();
337 if( exitInterface )
338 {
339
340 Coord3D exitPoint;
341 if ( ! exitInterface->getExitPosition(exitPoint))
342 exitPoint = *obj->getPosition();
343
344 points[ numPoints ].Set( Vector3( exitPoint.x, exitPoint.y, exitPoint.z ) );
345 numPoints++;
346
347 Bool boxWrap = TRUE;
348 Coord3D naturalRallyPoint;
349 if (exitInterface->getNaturalRallyPoint(naturalRallyPoint, FALSE))//FALSE means "without the extra offset"
350 {
351 if( !naturalRallyPoint.equals( exitPoint ) )
352 {
353 points[ numPoints ].Set( Vector3( naturalRallyPoint.x, naturalRallyPoint.y, naturalRallyPoint.z ) );
354 numPoints++;
355 }
356 else
357 {
358 //Helipad rally point -- so don't use box wrapping.
359 boxWrap = FALSE;
360 }
361 }
362 else
363 continue; //next drawable
364
365 const Coord3D *rallyPoint = exitInterface->getRallyPoint();
366 if( rallyPoint )
367 {
368 if( boxWrap )
369 {
370
371 //test to se whether the rally point flanks the side of the natural rally point
372 //to do this we find the two corners of the geometry extents that are nearest the natural rally point
373 // these define the natural rally point edge
374 // an intermediate point should be inserted at the corner nearest the rally point
375 const GeometryInfo& geom = obj->getGeometryInfo();
376 const Coord3D *ctr = obj->getPosition();
377 Coord3D NRPDelta;
378 NRPDelta.x = naturalRallyPoint.x - exitPoint.x;
379 NRPDelta.y = naturalRallyPoint.y - exitPoint.y;
380 NRPDelta.z = 0.0f;
381
382
383 //This is a quick idiot test to se whether the rally line needs to wrap the box at all
384 Coord3D wayOutPoint = NRPDelta;
385 wayOutPoint.normalize();
386 wayOutPoint.scale( 99999.9f );
387 Real wayOutLength = wayOutPoint.length();
388 wayOutPoint.add(&naturalRallyPoint);
389
390
391 //if the rallypoint is closer to the wayoutpoint than it is to the natural rally point then we definitely do not wrap
392 Coord3D rallyToWayOutDelta = wayOutPoint;
393 rallyToWayOutDelta.sub(rallyPoint);
394 if ( (100.0f + rallyToWayOutDelta.length()) > wayOutLength)
395 {
396
397 //if we passed the above idiot test, now lets be sure by testing the dotproduct of the rp against the wayoutpoint
398 wayOutPoint.normalize();// a normal shooting straight out the door
399 //next comes the delta between the NRP and the RP
400 Coord3D NRPToRPDelta = naturalRallyPoint;
401 NRPToRPDelta.sub(rallyPoint);
402 NRPToRPDelta.normalize();
403 Real dot = NRPToRPDelta.x * wayOutPoint.x + NRPToRPDelta.y * wayOutPoint.y;
404 if (dot > 0)
405 {
406
407 Real angle = obj->getOrientation();
408 Real c = (Real)cos(angle);
409 Real s = (Real)sin(angle);
410
411 Coord3D NRPToCtrDelta;
412 NRPToCtrDelta.x = naturalRallyPoint.x - ctr->x;
413 NRPToCtrDelta.y = naturalRallyPoint.y - ctr->y;
414 NRPToCtrDelta.x = 0.0f;
415
416 Real exc = geom.getMajorRadius() * c;
417 Real eyc = geom.getMinorRadius() * c;
418 Real exs = geom.getMajorRadius() * s;
419 Real eys = geom.getMinorRadius() * s;
420
421 Coord2D corners[ 4 ];
422
423 corners[0].x = ctr->x - exc - eys;
424 corners[0].y = ctr->y + eyc - exs;
425 corners[1].x = ctr->x + exc - eys;
426 corners[1].y = ctr->y + eyc + exs;
427 corners[2].x = ctr->x + exc + eys;
428 corners[2].y = ctr->y - eyc + exs;
429 corners[3].x = ctr->x - exc + eys;
430 corners[3].y = ctr->y - eyc - exs;
431
432 Coord2D *pNearElbow = NULL;//find the closest corner to the rallyPoint same end as door
433 Coord2D *pFarElbow = NULL; //find the closest corner to the rallypoint away from door
434 Coord2D *nearCandidate = NULL;
435 Coord3D cornerToRPDelta, cornerToExitDelta;
436 cornerToRPDelta.z = 0.0f;
437 cornerToExitDelta.z = 0.0f;
438 Real elbowDistanceNear = 99999.9f;
439 Real elbowDistanceFar = 99999.9f;
440
441 for (UnsignedInt cornerIndex = 0; cornerIndex < 4; ++ cornerIndex)
442 {
443 nearCandidate = &corners[cornerIndex];//for quicker array access
444 cornerToExitDelta.x = exitPoint.x - nearCandidate->x;
445 cornerToExitDelta.y = exitPoint.y - nearCandidate->y;
446 cornerToExitDelta.normalize();
447
448 dot = cornerToExitDelta.x * wayOutPoint.x + cornerToExitDelta.y * wayOutPoint.y;
449 if ( dot < 0.0f )
450 {
451 cornerToRPDelta.x = rallyPoint->x - nearCandidate->x;
452 cornerToRPDelta.y = rallyPoint->y - nearCandidate->y;
453 if (cornerToRPDelta.length() < elbowDistanceNear)
454 {
455 elbowDistanceNear = cornerToRPDelta.length();
456 pNearElbow = nearCandidate;
457 }
458 }
459 else
460 {
461 cornerToRPDelta.x = rallyPoint->x - nearCandidate->x;
462 cornerToRPDelta.y = rallyPoint->y - nearCandidate->y;
463 if (cornerToRPDelta.length() < elbowDistanceFar)
464 {
465 elbowDistanceFar = cornerToRPDelta.length();
466 pFarElbow = nearCandidate;
467 }
468 }
469
470
471
472 }
473
474 if (pNearElbow)//did we find a nearest corner?
475 {
476 m_waypointNodeRobj->Set_Position(Vector3(pNearElbow->x,pNearElbow->y,ctr->z));
477 WW3D::Render(*m_waypointNodeRobj,localRinfo); //The little hockey puck
478 points[ numPoints ].Set( Vector3( pNearElbow->x, pNearElbow->y, ctr->z ) );
479 numPoints++;
480
481
482 //and for that matter did we find a far side coner?
483 if (pFarElbow)//did we find a nearest corner?
484 {
485 // but let's test the dot of the first elbow against the rally point to find out
486 //whethet the rally point wraps around this one, too
487 Coord3D firstElbowDelta;
488 firstElbowDelta.x = naturalRallyPoint.x - pNearElbow->x;
489 firstElbowDelta.y = naturalRallyPoint.y - pNearElbow->y;
490 firstElbowDelta.z = 0.0f;
491 firstElbowDelta.normalize();
492
493 Coord3D firstToRPDelta;
494 firstToRPDelta.x = pNearElbow->x - rallyPoint->x;
495 firstToRPDelta.y = pNearElbow->y - rallyPoint->y;
496 firstToRPDelta.z = 0.0f;
497 firstToRPDelta.normalize();
498
499 dot = firstToRPDelta.x * firstElbowDelta.x + firstToRPDelta.y * firstElbowDelta.y;
500 if (dot < 0)// we have a second elbow
501 {
502 m_waypointNodeRobj->Set_Position(Vector3(pFarElbow->x,pFarElbow->y,ctr->z));
503 WW3D::Render(*m_waypointNodeRobj,localRinfo); //The little hockey puck
504 points[ numPoints ].Set( Vector3( pFarElbow->x, pFarElbow->y, ctr->z ) );
505 numPoints++;
506 }
507 }
508
509 }
510
511 }
512
513 }
514 }
515
516 // Finally draw the line out to the RallyPoint
517 points[ numPoints ].Set( Vector3( rallyPoint->x, rallyPoint->y, rallyPoint->z ) );
518 numPoints++;
519
520 }
521 else
522 continue;
523
524 m_waypointNodeRobj->Set_Position(Vector3(naturalRallyPoint.x,naturalRallyPoint.y,naturalRallyPoint.z));
525 WW3D::Render(*m_waypointNodeRobj,localRinfo); //The little hockey puck
526
527
528 m_line->Set_Points( numPoints, points );
529 m_line->Render( localRinfo );
530
531 }// end if exit interface
532
533 }
534 }
