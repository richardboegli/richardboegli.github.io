126void ArchiveFileSystem::loadIntoDirectoryTree(const ArchiveFile *archiveFile, const AsciiString& archiveFilename, Bool overwrite)
127{
128
129 FilenameList filenameList;
130
131 archiveFile->getFileListInDirectory(AsciiString(""), AsciiString(""), AsciiString("*"), filenameList, TRUE);
132
133 FilenameListIter it = filenameList.begin();
134
135 while (it != filenameList.end()) {
136 // add this filename to the directory tree.
137 AsciiString path = *it;
138 path.toLower();
139 AsciiString token;
140 AsciiString debugpath;
141
142 ArchivedDirectoryInfo *dirInfo = &m_rootDirectory;
143
144 Bool infoInPath;
145 infoInPath = path.nextToken(&token, "\\/");
146
147 while (infoInPath && (!token.find('.') || path.find('.'))) {
148 ArchivedDirectoryInfoMap::iterator tempiter = dirInfo->m_directories.find(token);
149 if (tempiter == dirInfo->m_directories.end())
150 {
151 dirInfo->m_directories[token].clear();
152 dirInfo->m_directories[token].m_directoryName = token;
153 }
154
155 dirInfo = &(dirInfo->m_directories[token]);
156 debugpath.concat(token);
157 debugpath.concat('\\');
158 infoInPath = path.nextToken(&token, "\\/");
159 }
160
161 // token is the filename, and dirInfo is the directory that this file is in.
162 if (dirInfo->m_files.find(token) == dirInfo->m_files.end() || overwrite) {
163 AsciiString path2;
164 path2 = debugpath;
165 path2.concat(token);
166// DEBUG_LOG(("ArchiveFileSystem::loadIntoDirectoryTree - adding file %s, archived in %s\n", path2.str(), archiveFilename.str()));
167 dirInfo->m_files[token] = archiveFilename;
168 }
169
170 it++;
171 }
172}

174void ArchiveFileSystem::loadMods() {
175 if (TheGlobalData->m_modBIG.isNotEmpty())
176 {
177 ArchiveFile *archiveFile = openArchiveFile(TheGlobalData->m_modBIG.str());
178
179 if (archiveFile != NULL) {
180 DEBUG_LOG(("ArchiveFileSystem::loadMods - loading %s into the directory tree.\n", TheGlobalData->m_modBIG.str()));
181 loadIntoDirectoryTree(archiveFile, TheGlobalData->m_modBIG, TRUE);
182 m_archiveFileMap[TheGlobalData->m_modBIG] = archiveFile;
183 DEBUG_LOG(("ArchiveFileSystem::loadMods - %s inserted into the archive file map.\n", TheGlobalData->m_modBIG.str()));
184 }
185 else
186 {
187 DEBUG_LOG(("ArchiveFileSystem::loadMods - could not openArchiveFile(%s)\n", TheGlobalData->m_modBIG.str()));
188 }
189 }
190
191 if (TheGlobalData->m_modDir.isNotEmpty())
192 {
193#ifdef DEBUG_LOGGING
194 Bool ret =
195#endif
196 loadBigFilesFromDirectory(TheGlobalData->m_modDir, "*.big", TRUE);
197 DEBUG_ASSERTLOG(ret, ("loadBigFilesFromDirectory(%s) returned FALSE!\n", TheGlobalData->m_modDir.str()));
198 }
199}

201Bool ArchiveFileSystem::doesFileExist(const Char *filename) const
202{
203 AsciiString path = filename;
204 path.toLower();
205 AsciiString token;
206
207 const ArchivedDirectoryInfo *dirInfo = &m_rootDirectory;
208
209 path.nextToken(&token, "\\/");
210
211 while (!token.find('.') || path.find('.'))
212 {
213 ArchivedDirectoryInfoMap::const_iterator tempiter = dirInfo->m_directories.find(token);
214 if (tempiter != dirInfo->m_directories.end())
215 {
216 dirInfo = &tempiter->second;
217 path.nextToken(&token, "\\/");
218 }
219 else
220 {
221 // the directory doesn't exist, so return false
222 return FALSE;
223 }
224 }
225
226 // token is the filename, and dirInfo is the directory that this file is in.
227 if (dirInfo->m_files.find(token) == dirInfo->m_files.end()) {
228 return FALSE;
229 }
230 return TRUE;
231}

245Bool ArchiveFileSystem::getFileInfo(const AsciiString& filename, FileInfo *fileInfo) const
246{
247 if (fileInfo == NULL) {
248 return FALSE;
249 }
250
251 if (filename.getLength() <= 0) {
252 return FALSE;
253 }
254
255 AsciiString archiveFilename = getArchiveFilenameForFile(filename);
256 ArchiveFileMap::const_iterator it = m_archiveFileMap.find(archiveFilename);
257 if (it != m_archiveFileMap.end())
258 {
259 return it->second->getFileInfo(filename, fileInfo);
260 }
261 else
262 {
263 return FALSE;
264 }
