46void parseFactionObjectCreationList( INI *ini, void *instance, void *store, const void *userData )
47{
48 OCLUpdateModuleData::FactionOCLInfo info;
49 info.m_factionName = "";
50 info.m_ocl = 0;
51
52 const char *token = ini->getNextToken( ini->getSepsColon() );
53
54 if ( stricmp(token, "Faction") == 0 )
55 {
56 token = ini->getNextTokenOrNull( ini->getSepsColon() );
57 if (!token) throw INI_INVALID_DATA;
58
59 info.m_factionName = token;
60 }
61 else
62 throw INI_INVALID_DATA;
63
64
65 token = ini->getNextTokenOrNull( ini->getSepsColon() );
66 if ( stricmp(token, "OCL") == 0 )
67 ini->parseObjectCreationList( ini, instance, &info.m_ocl, NULL );
68 else
69 throw INI_INVALID_DATA;
70
71 // Insert the info into the ocl hashmap
72 OCLUpdateModuleData::FactionOCLList * theList = (OCLUpdateModuleData::FactionOCLList*)store;
73 theList->push_back(info);
74
75} // end parseFactionObjectCreationList

124UpdateSleepTime OCLUpdate::update( void )
125{
126 if( getObject()->isDisabled() )
127 {
128 m_nextCreationFrame++;
129 return UPDATE_SLEEP_NONE;
130 }
131
132 const OCLUpdateModuleData *data = getOCLUpdateModuleData();
133
134 // Test if the OCL update is faction dependant. If so, check for faction changes
135 if (data->m_isFactionTriggered)
136 {
137 Player *player = getObject()->getControllingPlayer();
138
139 // Test for when a player captures the building
140 if (m_isFactionNeutral)
141 {
142 if( player && player->isPlayableSide() )
143 {
144 m_currentPlayerColor = player->getPlayerColor();
145 m_isFactionNeutral = FALSE;
146 setNextCreationFrame();
147 }
148 }
149 // Test for when the building has been made neutral or when it changes faction
150 else
151 {
152 // If this is no longer under player control, then we set the faction to neutral
153 if( !player || !player->isPlayableSide() )
154 {
155 m_isFactionNeutral = TRUE;
156 }
157 // If another player has taken control, reset the timer
158 else if( player && player->getPlayerColor() != m_currentPlayerColor)
159 {
160 m_currentPlayerColor = player->getPlayerColor();
161 setNextCreationFrame();
162 }
163 }
164
165 // If the building is neutal, skip futher update
166 if (m_isFactionNeutral)
167 return UPDATE_SLEEP_NONE;
168 }
169
171 if( shouldCreate() )
172 {
173 if( m_nextCreationFrame == 0 )
174 {
175 // You don't get to actually spread the first try, you start on a timer, then go
176 setNextCreationFrame();
177 return UPDATE_SLEEP_NONE;
178 }
179
180 setNextCreationFrame();
181
182 Coord3D creationCoord;
183 if( getOCLUpdateModuleData()->m_isCreateAtEdge )
184 creationCoord = TheTerrainLogic->findClosestEdgePoint( getObject()->getPosition() );
185 else
186 creationCoord = *getObject()->getPosition();
187
188 // If this is faction triggered, search through the faction specific OCLs to find the match
189 if (data->m_isFactionTriggered)
190 {
191 std::string playerFactionName("");
192
193 Player *player = getObject()->getControllingPlayer();
194 if (!player) return UPDATE_SLEEP_NONE;
195
196 const PlayerTemplate *playerT = player->getPlayerTemplate();
197 if (!playerT) return UPDATE_SLEEP_NONE;
198
199 // Get and store the faction side to compare with the faction ocl list
200 if (playerT->getSide().str()) playerFactionName = playerT->getSide().str();
201
202 // Loop through the list of faction ocls to find the matching faction that triggeres the specific ocls
203 for (OCLUpdateModuleData::FactionOCLList::const_iterator it = data->m_factionOCL.begin(); it != data->m_factionOCL.end(); ++it)
204 {
205 OCLUpdateModuleData::FactionOCLInfo info = *it;
206 if (playerFactionName == info.m_factionName)
207 {
208 ObjectCreationList::create( info.m_ocl, getObject(), &creationCoord, getObject()->getPosition(), getObject()->getOrientation() );
209 break;
210 }
211 }
212 }
213 // Use the non faction OCL information
214 else
215 {
216 ObjectCreationList::create( data->m_ocl, getObject(), &creationCoord, getObject()->getPosition(), getObject()->getOrientation() );
217 }
218 }
219 return UPDATE_SLEEP_NONE;
220}

230Bool OCLUpdate::shouldCreate()
231{
232 if( TheGameLogic->getFrame() < m_nextCreationFrame )
233 return FALSE;//too soon
234
235 if( getObject()->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
236 return FALSE;// not built yet
237
238 return TRUE;
239}

243void OCLUpdate::setNextCreationFrame()
244{
245 UnsignedInt delay = GameLogicRandomValue( getOCLUpdateModuleData()->m_minDelay,
246 getOCLUpdateModuleData()->m_maxDelay );
247 m_timerStartedFrame = TheGameLogic->getFrame();
248 m_nextCreationFrame = m_timerStartedFrame + delay;
249
250}

254Real OCLUpdate::getCountdownPercent() const
255{
256 UnsignedInt now = TheGameLogic->getFrame();
257
258 return 1.0f - (( m_nextCreationFrame - now ) / (float)( m_nextCreationFrame - m_timerStartedFrame ));
259}
