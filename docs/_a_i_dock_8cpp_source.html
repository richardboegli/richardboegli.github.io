120/* static */ Bool AIDockMachine::ableToAdvance( State *thisState, void* userData )
121{
122 Object *goalObject = thisState->getMachineGoalObject();
123 AIDockMachine *myMachine = (AIDockMachine *)thisState->getMachine();
124
125 if( goalObject == NULL )
126 return FALSE;
127
128 DockUpdateInterface *dock = goalObject->getDockUpdateInterface();
129
130 // if we have nothing to dock with, fail
131 if( dock == NULL )
132 return FALSE;
133
134 // if the dock says we can advance, then sidetrack to the scoot forward state
135 if( dock->isClearToAdvance( thisState->getMachineOwner(), myMachine->m_approachPosition ) )
136 return TRUE;
137
138 // continue to wait
139 return FALSE;
140}

167StateReturnType AIDockApproachState::onEnter( void )
168{
169 Object *goalObject = getMachineGoalObject();
170
171 // sanity
172 if( goalObject == NULL )
173 return STATE_FAILURE;
174
175 // get dock update interface
176 DockUpdateInterface *dock = goalObject->getDockUpdateInterface();
177
178 // if we have nothing to dock with, fail
179 if (dock == NULL)
180 return STATE_FAILURE;
181
182 // fail if the dock is closed
183 if( dock->isDockOpen() == FALSE )
184 {
185 dock->cancelDock( getMachineOwner() );
186 return STATE_FAILURE;
187 }
188
189 // get a good place to wait from the dock
190 Bool reserved = dock->reserveApproachPosition( getMachineOwner(), &m_goalPosition, &(( (AIDockMachine*)getMachine() )->m_approachPosition) );
191 if( reserved == FALSE )
192 {
193 // dock is full
194 return STATE_FAILURE;
195 }
196
197 AIUpdateInterface *ai = getMachineOwner()->getAIUpdateInterface();
198 if (ai) {
199 ai->ignoreObstacle( NULL );
200 }
201 // this behavior is an extention of basic MoveTo
202 return AIInternalMoveToState::onEnter();
203}

235 }

258StateReturnType AIDockWaitForClearanceState::update( void )
259{
260 Object *goalObject = getMachineGoalObject();
261
262 if( goalObject == NULL )
263 return STATE_FAILURE;
264
265 DockUpdateInterface *dock = goalObject->getDockUpdateInterface();
266
267 // if we have nothing to dock with, fail
268 if (dock == NULL)
269 return STATE_FAILURE;
270
271 // fail if the dock is closed
272 if( dock->isDockOpen() == FALSE )
273 {
274 dock->cancelDock( getMachineOwner() );
275 return STATE_FAILURE;
276 }
277
278 // if the dock says we can enter, our wait is over
279 if (dock->isClearToEnter( getMachineOwner() ))
280 return STATE_SUCCESS;
281
282 if (m_enterFrame + 30*LOGICFRAMES_PER_SECOND < TheGameLogic->getFrame()) {
283 return STATE_FAILURE;
284 }
285 // continue to wait
286 return STATE_CONTINUE;

575void AIDockProcessDockState::setNextDockActionFrame()
576{
577 // If we have a SupplyTruck Interface, then we will ask for our specific delay time
578 SupplyTruckAIInterface *supplyTruck = getMachineOwner()->getAI()->getSupplyTruckAIInterface();
579 if( supplyTruck )
580 {
581 m_nextDockActionFrame = TheGameLogic->getFrame() + supplyTruck->getActionDelayForDock( getMachineGoalObject() );

583 }

585 // The default is that it is simply okay to Action right away
586 m_nextDockActionFrame = TheGameLogic->getFrame();

613StateReturnType AIDockProcessDockState::update( void )
614{
615 Object *goalObject = getMachineGoalObject();
616
617 DockUpdateInterface *dock = NULL;
618 if( goalObject )
619 dock = goalObject->getDockUpdateInterface();
620
621 // if we have nothing to dock with, fail
622 if (dock == NULL)
623 return STATE_FAILURE;
624
625 // Some dockers can have a delay built in
626 if( TheGameLogic->getFrame() < m_nextDockActionFrame )
627 return STATE_CONTINUE;
628 setNextDockActionFrame();
629
630 Object *drone = findMyDrone();
631
632 // invoke the dock's action until it tells us it is done or the dock becomes closed

634 return STATE_SUCCESS;
635
636 return STATE_CONTINUE;

640struct DroneInfo
641{
642 Object *owner;
643 Object *drone;
644 Bool found;
645};
646
647void findDrone( Object *obj, void *droneInfo )
648{
649 DroneInfo *dInfo = (DroneInfo*)droneInfo;
650
651 if( !dInfo->found && obj )
652 {
653 if( obj->isKindOf( KINDOF_DRONE ) && obj->getProducerID() == dInfo->owner->getID() )
654 {
655 dInfo->found = TRUE;
656 dInfo->drone = obj;
657 }
658 }
659}

661//----------------------------------------------------------------------------------------------
662Object* AIDockProcessDockState::findMyDrone()
663{
664 //First do the fast cached check.
665 Object *drone = TheGameLogic->findObjectByID( m_droneID );
666 if( drone )
667 {
668 return drone;
669 }
670
671 //Nope... look for a drone (perhaps we just finished building one after docking?)
672 Object *self = getMachineOwner();
673 Player *player = self->getControllingPlayer();
674 DroneInfo dInfo;
675 dInfo.found = FALSE;
676 dInfo.drone = NULL;
677 dInfo.owner = self;
678
679 //Iterate the objects in search for a drone with a producer ID of me.
680 if( player )
681 {
682 player->iterateObjects( findDrone, &dInfo );
683 }
684
685 //If we found a drone, store it's ID as cached.
686 if( dInfo.drone )
687 {
688 m_droneID = dInfo.drone->getID();
689 }
690 return dInfo.drone;
691}
