70{

72 if (Vector != NULL) {
73
74 // Pass the object back to the caller

76
77 //
78 // Move the head object to the end of the list
79 //
80 for (int index = 1; index < ActiveCount; index ++) {
81 Vector[index - 1] = Vector[index];
82 }
83 Vector[ActiveCount - 1] = object;
84
85 // Success!
86 retval = true;
87 }
88
