48Bool ImpassableOptions::ValidateSlope()
49{
50 if (m_slopeToShow < 0.0f) {
51 m_slopeToShow = 0.0f;
52 return FALSE;
53 }
54
55 if (m_slopeToShow >= 90.0f) {
56 m_slopeToShow = 89.9f;
57 return FALSE;
58 }
59
60 return TRUE;
61}

79void ImpassableOptions::OnAngleChange()
80{
81 CWnd* pWnd = GetDlgItem(IDC_ANGLE);
82 if (!pWnd) {
83 return;
84 }
85
86 CString cstr;
87 pWnd->GetWindowText(cstr);
88 char* buff = cstr.GetBuffer(0);
89
90 m_slopeToShow = (Real)atof(buff);
91
92 if (!ValidateSlope()) {
93 AsciiString astr;
94 astr.format("%.2f", m_slopeToShow);
95 pWnd->SetWindowText(astr.str());
96 }
97 TheTerrainRenderObject->setViewImpassableAreaSlope(m_slopeToShow);
98}
