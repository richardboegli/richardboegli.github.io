70void CreateCrateDie::onDie( const DamageInfo * damageInfo )
71{
72 if (!isDieApplicable(damageInfo))
73 return;
74
75 CrateTemplate const *currentCrateData = NULL;
76 Object *killer = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
77 Object *me = getObject();
78
79 if( killer && killer->getRelationship( me ) == ALLIES )
80 return; //Nope, no crate for killing ally at all.
81
82 for( AsciiStringListConstIterator iter = getCreateCrateDieModuleData()->m_crateNameList.begin();
83 iter != getCreateCrateDieModuleData()->m_crateNameList.end();
84 iter++
85 )
86 {
87 currentCrateData = TheCrateSystem->findCrateTemplate( *iter );
88 if( currentCrateData )
89 {
90 if( ! testCreationChance( currentCrateData ) )
91 continue;// always test this
92
93 if( (currentCrateData->m_veterancyLevel != LEVEL_INVALID) && ! testVeterancyLevel( currentCrateData ) )
94 continue; //If this is set up to test and it fails
95
96 if( KINDOFMASK_ANY_SET(currentCrateData->m_killedByTypeKindof) && !testKillerType( currentCrateData, killer ) )
97 continue; //If this is set up to test and it fails
98
99 if( (currentCrateData->m_killerScience != SCIENCE_INVALID) && !testKillerScience( currentCrateData, killer ) )
100 continue; //If this is set up to test and it fails
101
102 Object *crate = createCrate( currentCrateData );//Make the crate
103 if( crate )
104 {
105 // Design needs to set ownership of crates sometimes
106 if( currentCrateData->m_isOwnedByMaker )
107 {
108 crate->setTeam( me->getControllingPlayer()->getDefaultTeam() );
109 }
110
111 if (killer)
112 {
113 // If the killer is a computer controlled player, notify that the crate exists.
114 if (killer->getControllingPlayer() &&
115 killer->getControllingPlayer()->getPlayerType()==PLAYER_COMPUTER)
116 {
117 AIUpdateInterface *ai = killer->getAIUpdateInterface();
118 if (ai)
119 {
120 ai->notifyCrate( crate->getID() );
121 }
122 }
123 }
124 }
125 }
126 }
127}

145Bool CreateCrateDie::testKillerType( CrateTemplate const *currentCrateData, Object *killer )
146{
147 if( killer == NULL )
148 return FALSE;
149
150 // Must match the whole group of bits set in the KilledBy description (most likely One).
151 if( ! killer->getTemplate()->isKindOfMulti( currentCrateData->m_killedByTypeKindof, KINDOFMASK_NONE ) )
152 return FALSE;
153
154 return TRUE;
155}

157Bool CreateCrateDie::testKillerScience( CrateTemplate const *currentCrateData, Object *killer )
158{
159 if( killer == NULL )
160 return FALSE;
161
162 // killer's player must have the listed science
163 Player *killerPlayer = killer->getControllingPlayer();
164
165 if( killerPlayer == NULL )
166 return FALSE;
167
168 if( ! killerPlayer->hasScience( currentCrateData->m_killerScience ) )
169 return FALSE;
170
171 return TRUE;
172}

174Object *CreateCrateDie::createCrate( CrateTemplate const *currentCrateData )
175{
176 Coord3D centerPoint = *getObject()->getPosition();
177 PathfindLayerEnum layer = getObject()->getLayer();
178
179 // CreationChance is used for the success of this block, but this block can have any number of potential actual crates
180 Real multipleCratePick = GameLogicRandomValueReal( 0, 1 );
181 Real multipleCrateRunningTotal = 0;
182 AsciiString crateName = "";
183
184 for( crateCreationEntryConstIterator iter = currentCrateData->m_possibleCrates.begin();
185 iter != currentCrateData->m_possibleCrates.end();
186 iter++
187 )
188 {
189 multipleCrateRunningTotal += (*iter).crateChance;
190 if( multipleCrateRunningTotal > multipleCratePick )
191 {
192 // Run through the list of possibles, and if the sum of the chances is greater than my random pick,
193 // then this is the correct one. (This simulates contiguous %, or weighted distribution)
194 crateName = (*iter).crateName;
195 break;
196 }
197 }
198 // At this point, I could very well have a "" for the type, if the Designer didn't make the sum of chances 1
199
200 ThingTemplate const *crateType = TheThingFactory->findTemplate( crateName );
201 if( crateType == NULL )
202 return NULL;
203
204 Bool spotFound = FALSE;
205 Coord3D creationPoint;
206 FindPositionOptions fpOptions;
207
208 fpOptions.minRadius = 0.0f;
209 fpOptions.maxRadius = 5.0f;
210 fpOptions.relationshipObject = getObject();
211 fpOptions.flags = FPF_IGNORE_ALLY_OR_NEUTRAL_UNITS; // So the dead guy won't block, nor will his dead hulk.
212 if (layer != LAYER_GROUND) {
213 creationPoint = centerPoint;
214 spotFound = true;
215 } else if( ThePartitionManager->findPositionAround( &centerPoint, &fpOptions, &creationPoint ) )
216 {
217 spotFound = TRUE;
218 }
219 else
220 {
221 // If the tight ignore units scan fails, then try a great big scan so we appear on the edge
222 // of the large dead thing (building rubble)
223 fpOptions.minRadius = 0.0f;
224 fpOptions.maxRadius = 125.0f;
225 fpOptions.relationshipObject = NULL;
226 fpOptions.flags = FPF_NONE;
227 if( ThePartitionManager->findPositionAround( &centerPoint, &fpOptions, &creationPoint ) )
228 {
229 spotFound = TRUE;
230 }
231 }
232
233 if( spotFound )
234 {
235 Object *newCrate = TheThingFactory->newObject( crateType, NULL );
236 newCrate->setPosition( &creationPoint );
237 newCrate->setOrientation( GameLogicRandomValueReal( 0, 2*PI ) );
238 newCrate->setLayer(layer);
239
240 Drawable *crateDrawable = newCrate->getDrawable();
241
242 if( crateDrawable )
243 {
244 crateDrawable->setTerrainDecal(TERRAIN_DECAL_CRATE);
245 crateDrawable->setTerrainDecalSize(2.5f * newCrate->getGeometryInfo().getMajorRadius(),
246 2.5f * newCrate->getGeometryInfo().getMajorRadius() ) ;
247 crateDrawable->setTerrainDecalFadeTarget(1.0f, 0.03f);
248 }
249
250 return newCrate;
251 }
252 return NULL;
253}
