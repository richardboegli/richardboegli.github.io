92static Real calcDistSqr(const Coord3D& a, const Coord3D& b)
93{
94 return sqr(a.x-b.x) + sqr(a.y-b.y) + sqr(a.z-b.z);
95}

99{

102 {

224public:
225 ChinookTakeoffOrLandingState( StateMachine *machine, Bool landing ) : m_landing(landing), State( machine, "ChinookTakeoffOrLandingState" )
226 {
227 m_destLoc.zero();
228 }
229
230 virtual StateReturnType onEnter()
231 {
232 Object* obj = getMachineOwner();
233 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
234
235 ai->friend_setFlightStatus(m_landing ? CHINOOK_LANDING : CHINOOK_TAKING_OFF);
236
237 if( m_landing )
238 {
239 // A chinook given transport duty loses his supplies.
240 while( ai->loseOneBox() );
241 }
242
243 // kill any drift...
244 obj->getPhysics()->scrubVelocity2D(0);
245
246 ai->chooseLocomotorSet(LOCOMOTORSET_NORMAL);
247 Locomotor* loco = ai->getCurLocomotor();
248 loco->setUsePreciseZPos(true);
249 loco->setUltraAccurate(true);
250
251 m_destLoc = *obj->getPosition();
252 const Bool onlyHealthyBridges = true; // ignore dead bridges.
253 PathfindLayerEnum layerAtDest = TheTerrainLogic->getHighestLayerForDestination(&m_destLoc, onlyHealthyBridges);
254 m_destLoc.z = TheTerrainLogic->getLayerHeight(m_destLoc.x, m_destLoc.y, layerAtDest);
255 if (m_landing)
256 {
257 Coord3D tmp;
258 FindPositionOptions options;
259 options.maxRadius = obj->getGeometryInfo().getBoundingCircleRadius() * 100.0f;
260 if (ThePartitionManager->findPositionAround(&m_destLoc, &options, &tmp))
261 {
262 m_destLoc = tmp;
263 TheAI->pathfinder()->adjustToLandingDestination(obj, &m_destLoc);
264 }
265 // recalc, since it may have changed. note that findPositionAround() will ALWAYS
266 // return a position on the ground proper, so if our initial search start pos was
267 // above a bridge, this will put us below the bridge, which would be unfortunate.
268 // so recheck to be sure. (note that the partitionmgr is 2d-only, so if it sez that
269 // the position on the ground at that xy is clear, it will be clear for both the
270 // ground proper and the bridge itself.) also note: don't call objectInteractsWithBridgeLayer(),
271 // since it assumes that things that aren't close in z shouldn't interact.
272 tmp = m_destLoc;
273 tmp.z = obj->getPosition()->z;
274 layerAtDest = TheTerrainLogic->getHighestLayerForDestination(&tmp, onlyHealthyBridges);
275 m_destLoc.z = TheTerrainLogic->getLayerHeight(m_destLoc.x, m_destLoc.y, layerAtDest);
276 obj->setLayer(layerAtDest);
277 }
278 else
279 {
280 m_destLoc.z += loco->getPreferredHeight();
281 obj->setLayer(LAYER_GROUND);
282 }
283
284 return STATE_CONTINUE;
285 }
286
287 virtual StateReturnType update()
288 {
289 Object* obj = getMachineOwner();
290 if (obj->isEffectivelyDead())
291 return STATE_FAILURE;
292
293 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
294
295 ai->setLocomotorGoalPositionExplicit(m_destLoc);
296
297 const Real THRESH = 3.0f;
298 const Real THRESH_SQR = THRESH*THRESH;
299 if (calcDistSqr(*obj->getPosition(), m_destLoc) <= THRESH_SQR)
300 return STATE_SUCCESS;
301
302 return STATE_CONTINUE;
303 }
304
305 virtual void onExit( StateExitType status )
306 {
307 Object* obj = getMachineOwner();
308 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
309
310 ai->friend_setFlightStatus(m_landing ? CHINOOK_LANDED : CHINOOK_FLYING);
311
312 // Paranoia checks - sometimes onExit is called when we are
313 // shutting down, and not all pieces are valid. CurLocomotor
314 // is definitely null in some cases. jba.
315 Locomotor* loco = ai->getCurLocomotor();
316 if (loco)
317 {
318 loco->setUsePreciseZPos(false);
319 loco->setUltraAccurate(false);
320 // don't restore lift if dead -- this may fight with JetSlowDeathBehavior!
321 if (!obj->isEffectivelyDead())
322 loco->setMaxLift(BIGNUM);
323 }
324
325 if (m_landing)
326 {
327 ai->chooseLocomotorSet(LOCOMOTORSET_TAXIING);
328 }
329 else
330 {
331 // when takeoff is complete, always go back to layer-ground, rather than
332 // some bridge layer.
333 obj->setLayer(LAYER_GROUND);
334 }
335
336
337 }
338
339};

462 if (xfer->getXferMode() == XFER_LOAD)
463 {
464 info.ropeDrawable = NULL; // filled in via loadPostProcess
465 m_ropes[i] = info;
466 }
467 }

481public:
482 ChinookCombatDropState( StateMachine *machine ): State( machine, "ChinookCombatDropState" ) { }

484 // --------------
485 virtual StateReturnType onEnter()
486 {
487 Object* obj = getMachineOwner();
488 Drawable* draw = obj->getDrawable();
489 if (draw == NULL)
490 return STATE_FAILURE;
491
492 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
493 const ChinookAIUpdateModuleData* d = ai->friend_getData();
494
495 obj->setDisabled( DISABLED_HELD );
496 ai->friend_setFlightStatus(CHINOOK_DOING_COMBAT_DROP);
497
498 // A chinook given combat drop duty also loses his supplies.
499 while( ai->loseOneBox() );
500
501 UnsignedInt now = TheGameLogic->getFrame();
502
503 const ThingTemplate* ropeTmpl = TheThingFactory->findTemplate(d->m_ropeName);
504
505 const Int MAX_BONES = 32;
506 Coord3D ropePos[MAX_BONES];
507 Matrix3D dropMtx[MAX_BONES];
508
509 Int ropeCount = draw->getPristineBonePositions("RopeStart", 1, ropePos, NULL, MAX_BONES);
510 Int dropCount = draw->getPristineBonePositions("RopeEnd", 1, NULL, dropMtx, MAX_BONES);
511
512 Int numRopes = d->m_numRopes;
513 if (numRopes > ropeCount) numRopes = ropeCount;
514 if (numRopes > dropCount) numRopes = dropCount;
515 if (numRopes <= 0)
516 return STATE_FAILURE;
517
518 m_ropes.clear();
519 for (Int i = 0; i < numRopes; ++i)
520 {
521 RopeInfo info;
522
523 obj->convertBonePosToWorldPos( NULL, &dropMtx[i], NULL, &info.dropStartMtx );
524
525 info.ropeDrawable = ropeTmpl ? TheThingFactory->newDrawable(ropeTmpl) : NULL;
526 if (info.ropeDrawable)
527 {
528 obj->convertBonePosToWorldPos( &ropePos[i], NULL, &ropePos[i], NULL );
529 info.ropeDrawable->setPosition(&ropePos[i]);
530 info.ropeSpeed = 0.0f;
531 info.ropeLen = 1.0f;
532
533 const Bool onlyHealthyBridges = true; // ignore dead bridges.
534 PathfindLayerEnum layerAtDest = TheTerrainLogic->getHighestLayerForDestination(&ropePos[i], onlyHealthyBridges);
535
536 info.ropeLenMax = ropePos[i].z - TheTerrainLogic->getLayerHeight(ropePos[i].x, ropePos[i].y, layerAtDest) - d->m_ropeFinalHeight;
537
538 initRopeParms(info.ropeDrawable, info.ropeLenMax, d->m_ropeWidth, d->m_ropeColor, d->m_ropeWobbleLen, d->m_ropeWobbleAmp, d->m_ropeWobbleRate);
539 }
540
541 info.nextDropTime = now + GameLogicRandomValue(d->m_perRopeDelayMin, d->m_perRopeDelayMax) - d->m_perRopeDelayMin;
542 info.rappellerIDs.clear();
543
544 m_ropes.push_back(info);
545 }
546
547 return STATE_CONTINUE;
548 }
549
550 // --------------
551 virtual StateReturnType update()
552 {
553 Object* obj = getMachineOwner();
554 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
555 const ChinookAIUpdateModuleData* d = ai->friend_getData();
556
557 if (obj->isEffectivelyDead())
558 {
559 return STATE_FAILURE;
560 }
561
562 // first, eliminate "done" rappellers
563 removeDoneRappellers();
564
565 UnsignedInt now = TheGameLogic->getFrame();
566
567 // ok, now check each rope: if it's empty, or we're at the next drop time, spawn a new rappeller
568 Int numRopesInUse = 0;
569 for (std::vector<RopeInfo>::iterator it = m_ropes.begin(); it != m_ropes.end(); ++it)
570 {
571 if (it->ropeLen < it->ropeLenMax)
572 {
573 it->ropeSpeed += fabs(TheGlobalData->m_gravity);
574 if (it->ropeSpeed > d->m_ropeDropSpeed)
575 it->ropeSpeed = d->m_ropeDropSpeed;
576 it->ropeLen += it->ropeSpeed;
577 setRopeCurLen(it->ropeDrawable, it->ropeLen);
578 if (d->m_waitForRopesToDrop)
579 {
580 // can't use this rope till it's dropped all the way
581 ++it->nextDropTime;
582 continue;
583 }
584 }
585
586 if (now >= it->nextDropTime)
587 {
588 Object* rappeller = getPotentialRappeller(obj);
589 if (rappeller != NULL)
590 {
591 ExitInterface *exitInterface = obj->getObjectExitInterface();
592 ExitDoorType exitDoor = exitInterface ? exitInterface->reserveDoorForExit(rappeller->getTemplate(), rappeller) : DOOR_NONE_AVAILABLE;
593 if(exitDoor != DOOR_NONE_AVAILABLE)
594 {
595 exitInterface->exitObjectViaDoor(rappeller, exitDoor);
596 }
597 else
598 {
599 DEBUG_CRASH(("rappeller is not free to exit... what?"));
600 }
601
602 rappeller->setTransformMatrix(&it->dropStartMtx);
603
604 AIUpdateInterface* rappellerAI = rappeller ? rappeller->getAIUpdateInterface() : NULL;
605 if (rappellerAI)
606 {
607 rappellerAI->setDesiredSpeed(d->m_rappelSpeed);
608 rappellerAI->aiRappelInto(getMachineGoalObject(), *getMachineGoalPosition(), CMD_FROM_AI);
609 }
610
611 it->rappellerIDs.push_back(rappeller->getID());
612
613 it->nextDropTime = now + GameLogicRandomValue(d->m_perRopeDelayMin, d->m_perRopeDelayMax);
614 }
615 }
616
617 if (!it->rappellerIDs.empty())
618 {
619 ++numRopesInUse;
620 }
621 }
622
623 if (numRopesInUse == 0 && getPotentialRappeller(obj) == NULL)
624 {
625 // we're done!
626 return STATE_SUCCESS;
627 }
628
629 return STATE_CONTINUE;
630 }

632 // --------------
633 virtual void onExit( StateExitType status )
634 {
635 Object* obj = getMachineOwner();
636 ChinookAIUpdate* ai = (ChinookAIUpdate*)obj->getAIUpdateInterface();
637 const ChinookAIUpdateModuleData* d = ai->friend_getData();
638
639 obj->clearDisabled( DISABLED_HELD );
640 ai->friend_setFlightStatus(CHINOOK_FLYING);
641
642 if (obj->isEffectivelyDead())
643 {
644 // oops. drop the rangers.
645 for (std::vector<RopeInfo>::iterator it = m_ropes.begin(); it != m_ropes.end(); ++it)
646 {
647 for (std::list<ObjectID>::iterator oit = it->rappellerIDs.begin(); oit != it->rappellerIDs.end(); ++oit)
648 {
649 Object* rappeller = TheGameLogic->findObjectByID(*oit);
650 AIUpdateInterface* rappellerAI = rappeller ? rappeller->getAIUpdateInterface() : NULL;
651 if (rappellerAI != NULL)
652 {
653 rappellerAI->aiIdle(CMD_FROM_AI);
654 }
655 }
656 }
657 }
658
659 UnsignedInt now = TheGameLogic->getFrame();
660 for (Int i = 0; i < m_ropes.size(); ++i)
661 {
662 if (m_ropes[i].ropeDrawable)
663 {
664 const UnsignedInt ROPE_EXPIRATION_TIME = LOGICFRAMES_PER_SECOND * 5;
665 const Real initialSpeed = TheGlobalData->m_gravity * 30; // give it a little kick
666 setRopeSpeed(m_ropes[i].ropeDrawable, initialSpeed, d->m_ropeDropSpeed, TheGlobalData->m_gravity);
667 m_ropes[i].ropeDrawable->setExpirationDate(now + ROPE_EXPIRATION_TIME);
668 m_ropes[i].ropeDrawable = NULL; // we're done with it, so null it so we won't save it
669 }
670 }
671
672 m_ropes.clear();
673 }

723 m_oldPreferredHeight = loco->getPreferredHeight();

1256//-------------------------------------------------------------------------------------------------
1257void ChinookAIUpdate::aiDoCommand(const AICommandParms* parms)
1258{
1259 // this gets reset every time a command is issued.
1260 setAirfieldForHealing(INVALID_ID);
1261
1262 if (!isAllowedToRespondToAiCommands(parms))
1263 return;
1264
1265 if (m_flightStatus == CHINOOK_TAKING_OFF ||
1266 m_flightStatus == CHINOOK_LANDING ||
1267 m_flightStatus == CHINOOK_DOING_COMBAT_DROP)
1268 {
1269 // have to wait for takeoff or landing (or rappel) to complete, just store the sucker.
1270 m_pendingCommand.store(*parms);
1271 m_hasPendingCommand = true;
1272 return;
1273 }
1274
1275 Bool passItThru = true;
1276 switch (parms->m_cmd)
1277 {
1278 case AICMD_IDLE:
1279 case AICMD_BUSY:
1280 case AICMD_FOLLOW_EXITPRODUCTION_PATH:
1281 {
1282 // don't need (or want) to take off for these...
1283 // just pass it thru.
1284 }
1285 break;
1286
1287 case AICMD_MOVE_TO_POSITION_AND_EVACUATE:
1288 case AICMD_MOVE_TO_POSITION_AND_EVACUATE_AND_EXIT:
1289 {
1290 const Real THRESH = 3.0f;
1291 const Real THRESH_SQR = THRESH*THRESH;
1292 if (calcDistSqr(*getObject()->getPosition(), parms->m_pos) > THRESH_SQR &&
1293 m_flightStatus == CHINOOK_LANDED)
1294 {
1295 // gotta take off first!
1296 m_pendingCommand.store(*parms);
1297 m_hasPendingCommand = true;
1298
1299 setMyState(TAKING_OFF, NULL, NULL, CMD_FROM_AI);
1300 passItThru = false;
1301 }
1302 else
1303 {
1304 // do this INSTEAD of the standard stuff
1305 setMyState(
1306 (parms->m_cmd == AICMD_MOVE_TO_POSITION_AND_EVACUATE) ? MOVE_TO_AND_EVAC : MOVE_TO_AND_EVAC_AND_EXIT_INIT,
1307 NULL, &parms->m_pos, CMD_FROM_AI);
1308 passItThru = false;
1309 }
1310 }
1311 break;
1312
1313 case AICMD_EXIT:
1314 case AICMD_EVACUATE:
1315 {
1316 if (m_flightStatus != CHINOOK_LANDED)
1317 {
1318 // gotta land first!
1319 m_pendingCommand.store(*parms);
1320 m_hasPendingCommand = true;
1321
1322 setMyState(LANDING, NULL, NULL, CMD_FROM_AI);
1323 passItThru = false;
1324 }
1325 }
1326 break;
1327
1328 default:
1329 {
1330 if (m_flightStatus != CHINOOK_FLYING)
1331 {
1332 // gotta take off first!
1333 m_pendingCommand.store(*parms);
1334 m_hasPendingCommand = true;
1335
1336 setMyState(TAKING_OFF, NULL, NULL, CMD_FROM_AI);
1337 passItThru = false;
1338 }
1339 }
1340 break;
1341 }
1342
1343 if (passItThru)
1344 {
1345 m_hasPendingCommand = false;
1346 SupplyTruckAIUpdate::aiDoCommand(parms);
1347 }
1348}
