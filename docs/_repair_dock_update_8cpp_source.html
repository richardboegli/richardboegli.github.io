91Bool RepairDockUpdate::action( Object *docker, Object *drone )
92{
93
94 // sanity
95 if( docker == NULL )
96 return FALSE;
97
98 // get our module data
99 const RepairDockUpdateModuleData *modData = getRepairDockUpdateModuleData();
100
101 // get the body module for the docker
102 BodyModuleInterface *body = docker->getBodyModule();
103
104 //
105 // no matter if the object is damaged just a little bit, or on the brink of death, we will
106 // heal the object over a fixed amount of *TIME* that is specified in the INI file. Whenever
107 // we get a new docker, given its current health we figure out how much health we will add
108 // to this docked object each frame so that it is fully healed after the correct amount
109 // of time has passed
110 //
111 if( m_lastRepair == 0 )
112 {
113
114 // save ID of this docker as the last docker
115 m_lastRepair = docker->getID();
116
117 //
118 // figure out how much health we need to add each frame to this object so that it's
119 // fully healed at the right time
120 //
121 m_healthToAddPerFrame = (body->getMaxHealth() - body->getHealth()) / modData->m_framesForFullHeal;
122
123 } // end if
124
125 // if we're at max health we're done
126 if( body->getHealth() >= body->getMaxHealth() )
127 {
128
129 // repair is complete, clear our last docker
130 m_lastRepair = INVALID_ID;
131
132 // returning false will complete the docking process
133 return FALSE;
134
135 } // end if
136
137 // give us some health buddy
138 DamageInfo healingInfo;
139 healingInfo.in.m_amount = m_healthToAddPerFrame;
140 healingInfo.in.m_sourceID = getObject()->getID();
141 healingInfo.in.m_damageType = DAMAGE_HEALING;
142 healingInfo.in.m_deathType = DEATH_NONE;
143 body->attemptHealing( &healingInfo );
144 if( drone )
145 {
146 body = drone->getBodyModule();
147 healingInfo.in.m_amount = body->getMaxHealth();
148 body->attemptHealing( &healingInfo );
149 }
150
151 // stay docked
152 return TRUE;
153
154} // end action
