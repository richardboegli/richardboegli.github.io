59FireWeaponWhenDamagedBehavior::FireWeaponWhenDamagedBehavior( Thing *thing, const ModuleData* moduleData ) :
60 UpdateModule( thing, moduleData ),
61 m_reactionWeaponPristine( NULL ),
62 m_reactionWeaponDamaged( NULL ),
63 m_reactionWeaponReallyDamaged( NULL ),
64 m_reactionWeaponRubble( NULL ),
65 m_continuousWeaponPristine( NULL ),
66 m_continuousWeaponDamaged( NULL ),
67 m_continuousWeaponReallyDamaged( NULL ),
68 m_continuousWeaponRubble( NULL )
69{
70
71 const FireWeaponWhenDamagedBehaviorModuleData *d = getFireWeaponWhenDamagedBehaviorModuleData();
72 const Object* obj = getObject();
73
74 if ( d->m_reactionWeaponPristine )
75 {
76 m_reactionWeaponPristine = TheWeaponStore->allocateNewWeapon(
77 d->m_reactionWeaponPristine, PRIMARY_WEAPON);
78 m_reactionWeaponPristine->reloadAmmo( obj );
79 }
80 if ( d->m_reactionWeaponDamaged )
81 {
82 m_reactionWeaponDamaged = TheWeaponStore->allocateNewWeapon(
83 d->m_reactionWeaponDamaged, PRIMARY_WEAPON);
84 m_reactionWeaponDamaged->reloadAmmo( obj );
85 }
86 if ( d->m_reactionWeaponReallyDamaged )
87 {
88 m_reactionWeaponReallyDamaged = TheWeaponStore->allocateNewWeapon(
89 d->m_reactionWeaponReallyDamaged, PRIMARY_WEAPON);
90 m_reactionWeaponReallyDamaged->reloadAmmo( obj );
91 }
92 if ( d->m_reactionWeaponRubble )
93 {
94 m_reactionWeaponRubble = TheWeaponStore->allocateNewWeapon(
95 d->m_reactionWeaponRubble, PRIMARY_WEAPON);
96 m_reactionWeaponRubble->reloadAmmo( obj );
97 }
98
99
100 if ( d->m_continuousWeaponPristine )
101 {
102 m_continuousWeaponPristine = TheWeaponStore->allocateNewWeapon(
103 d->m_continuousWeaponPristine, PRIMARY_WEAPON);
104 m_continuousWeaponPristine->reloadAmmo( obj );
105 }

141 }

171void FireWeaponWhenDamagedBehavior::onDamage( DamageInfo *damageInfo )
172{
173 if (!isUpgradeActive())
174 return;
175
176 const FireWeaponWhenDamagedBehaviorModuleData* d = getFireWeaponWhenDamagedBehaviorModuleData();
177
178 // right type?
179 if (!getDamageTypeFlag(d->m_damageTypes, damageInfo->in.m_damageType))
180 return;
181
182 // right amount? (use actual [post-armor] damage dealt)
183 if (damageInfo->out.m_actualDamageDealt < d->m_damageAmount)
184 return;
185
186 const Object *obj = getObject();
187 BodyDamageType bdt = obj->getBodyModule()->getDamageState();
188
189 if ( bdt == BODY_RUBBLE )
190 {
191 if( m_reactionWeaponRubble && m_reactionWeaponRubble->getStatus() == READY_TO_FIRE )
192 {
193 m_reactionWeaponRubble->forceFireWeapon( obj, obj->getPosition() );
194 }
195
196 }
197 else if ( bdt == BODY_REALLYDAMAGED )
198 {
199 if( m_reactionWeaponReallyDamaged && m_reactionWeaponReallyDamaged->getStatus() == READY_TO_FIRE )
200 {
201 m_reactionWeaponReallyDamaged->forceFireWeapon( obj, obj->getPosition() );
202 }
203 }
204 else if ( bdt == BODY_DAMAGED )
205 {
206 if( m_reactionWeaponDamaged && m_reactionWeaponDamaged->getStatus() == READY_TO_FIRE )
207 {
208 m_reactionWeaponDamaged->forceFireWeapon( obj, obj->getPosition() );
209 }
210 }
211 else // not damaged yet
212 {
213 if( m_reactionWeaponPristine && m_reactionWeaponPristine->getStatus() == READY_TO_FIRE )
214 {
215 m_reactionWeaponPristine->forceFireWeapon( obj, obj->getPosition() );
216 }
217 }
218
219}

225{

238 {

240 }
241

243 else if ( bdt == BODY_REALLYDAMAGED )
244 {

246 {
247 m_continuousWeaponReallyDamaged->forceFireWeapon( obj, obj->getPosition() );

250 else if ( bdt == BODY_DAMAGED )
251 {
252 if( m_continuousWeaponDamaged && m_continuousWeaponDamaged->getStatus() == READY_TO_FIRE )
253 {
254 m_continuousWeaponDamaged->forceFireWeapon( obj, obj->getPosition() );
