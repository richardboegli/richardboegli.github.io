93Bool SupplyWarehouseDockUpdate::action( Object* docker, Object *drone )
94{
95 if( m_boxesStored == 0 )
96 return FALSE;
97
98 // Make sure that the docker is at least reasonably close to the dock.
99 // Basically, one bounding diameter of space or less between us.
100 Real closeEnoughSqr = sqr(docker->getGeometryInfo().getBoundingCircleRadius()*2);
101 Real curDistSqr = ThePartitionManager->getDistanceSquared(docker, getObject(), FROM_BOUNDINGSPHERE_2D);
102 if (curDistSqr > closeEnoughSqr) {
103 DEBUG_LOG(("Failing dock, dist %f, not close enough(%f).\n", sqrt(curDistSqr), sqrt(closeEnoughSqr)));
104 // Make it twitch a little.
105 Coord3D newPos = *docker->getPosition();
106 Real range = 0.4*PATHFIND_CELL_SIZE_F;
107 newPos.x += GameLogicRandomValue(-range, range);
108 newPos.y += GameLogicRandomValue(-range, range);
109 docker->setPosition(&newPos);
110 return FALSE; //not close enough.
111 }
112
113 --m_boxesStored;// so the docker sees that I am shy by one box (or empty) from within his gainOneBox()
114
115 SupplyTruckAIInterface *ai = docker->getAIUpdateInterface()->getSupplyTruckAIInterface();
116 if( ai && ai->gainOneBox( m_boxesStored ) )
117 {
118 if( m_boxesStored == 0 && getSupplyWarehouseDockUpdateModuleData()->m_deleteWhenEmpty )
119 {
120 TheGameLogic->destroyObject( getObject() );
121 return FALSE; //Yer done. And so am I.
122 }
123 else
124 {
125 Drawable *draw = getObject()->getDrawable();
126 if( draw )
127 {
128 draw->updateDrawableSupplyStatus( getSupplyWarehouseDockUpdateModuleData()->m_startingBoxesData, m_boxesStored );
129 }
130 }
131
132 return TRUE;
133 }
134 else
135 ++m_boxesStored; //take it back, since there was noone to gain the box
136 //this is important so that I have one less boxes as perceived by the docker when he gains one
137
138
139 return FALSE;
140}

142void SupplyWarehouseDockUpdate::setDockCrippled( Bool setting )
143{
144 // At this level, Crippling means I kill any activeDocker between enter and exit.
145 if( setting )
146 {
147 if( m_activeDocker != INVALID_ID )
148 {
149 Object *victim = TheGameLogic->findObjectByID( m_activeDocker );
150 if( victim )
151 {
152 if( m_dockerInside )
153 {
154 if( !victim->isUsingAirborneLocomotor() )
155 victim->kill();
156 }
157 else
158 {
159 // Else, he was between Approach and Enter. Lucky guy. Tell him to stop, but then
160 // remind him that he wants to try again later
161 SupplyTruckAIInterface* supplyTruckAI = victim->getAI()->getSupplyTruckAIInterface();
162 if( supplyTruckAI )
163 {
164 victim->getAI()->aiIdle( CMD_FROM_AI );
165 supplyTruckAI->setForceWantingState( TRUE );
166 }
167 }
168 }
169 }
170 }
171
172 DockUpdate::setDockCrippled( setting );
173}

176{
177 // A script can tell us our set value, and we need to figure out the boxes needed to provide that.
178 m_boxesStored = ceil(cashValue / (float)TheGlobalData->m_baseValuePerSupplyBox);
179 Drawable *draw = getObject()->getDrawable();

182 draw->updateDrawableSupplyStatus( getSupplyWarehouseDockUpdateModuleData()->m_startingBoxesData, m_boxesStored );
