83void INI::parseHeaderTemplateDefinition( INI *ini )
84{
85 AsciiString name;
86 HeaderTemplate *hTemplate;
87
88 // read the name
89 const char* c = ini->getNextToken();
90 name.set( c );
91
92 // find existing item if present
93 hTemplate = TheHeaderTemplateManager->findHeaderTemplate( name );
94 if( hTemplate == NULL )
95 {
96
97 // allocate a new item
98 hTemplate = TheHeaderTemplateManager->newHeaderTemplate( name );
99
100 } // end if
101 else
102 {
103 DEBUG_CRASH(( "[LINE: %d in '%s'] Duplicate header Template %s found!", ini->getLineNum(), ini->getFilename().str(), name.str() ));
104 }
105 // parse the ini definition
106 ini->initFromINI( hTemplate, TheHeaderTemplateManager->getFieldParse() );
107
108} // end parseCommandButtonDefinition

142void HeaderTemplateManager::init( void )
143{
144 INI ini;
145 AsciiString fname;
146 fname.format("Data\\%s\\HeaderTemplate.ini", GetRegistryLanguage().str());
147 OSVERSIONINFO osvi;
148 osvi.dwOSVersionInfoSize=sizeof(OSVERSIONINFO);
149 if (GetVersionEx(&osvi))
150 { //check if we're running Win9x variant since they may need different fonts
151 if (osvi.dwPlatformId == VER_PLATFORM_WIN32_WINDOWS)
152 { AsciiString tempName;
153
154 tempName.format("Data\\%s\\HeaderTemplate9x.ini", GetRegistryLanguage().str());
155 if (TheFileSystem->doesFileExist(tempName.str()))
156 fname = tempName;
157 }
158 }
159 ini.load( fname, INI_LOAD_OVERWRITE, NULL );
160 populateGameFonts();
161}

163HeaderTemplate *HeaderTemplateManager::findHeaderTemplate( AsciiString name )
164{
165 HeaderTemplateListIt it = m_headerTemplateList.begin();
166 while(it != m_headerTemplateList.end())
167 {
168 HeaderTemplate *hTemplate = *it;
169 if(hTemplate->m_name.compare(name) == 0)
170 return hTemplate;
171 ++it;
172 }
173 return NULL;
174}

176HeaderTemplate *HeaderTemplateManager::newHeaderTemplate( AsciiString name )
177{
178 HeaderTemplate *newHTemplate = NEW HeaderTemplate;
179 DEBUG_ASSERTCRASH(newHTemplate, ("Unable to create a new Header Template in HeaderTemplateManager::newHeaderTemplate"));
180 if(!newHTemplate)
181 return NULL;
182
183 newHTemplate->m_name = name;
184 m_headerTemplateList.push_front(newHTemplate);
185 return newHTemplate;
186
187}

236void HeaderTemplateManager::populateGameFonts( void )
237{
238 HeaderTemplateListIt it = m_headerTemplateList.begin();
239 while(it != m_headerTemplateList.end())
240 {
241 HeaderTemplate *hTemplate = *it;
242 Real pointSize = TheGlobalLanguageData->adjustFontSize(hTemplate->m_point);
243 GameFont *font = TheFontLibrary->getFont(hTemplate->m_fontName, pointSize,hTemplate->m_bold);
244 DEBUG_ASSERTCRASH(font,("HeaderTemplateManager::populateGameFonts - Could not find font %s %d",hTemplate->m_fontName, hTemplate->m_point));
245
246 hTemplate->m_font = font;
247
248 ++it;
249 }
