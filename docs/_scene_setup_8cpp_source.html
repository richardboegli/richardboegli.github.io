62/***********************************************************************************************
63 * scene_setup_cf - MAXScript function wwSceneSetup *
64 * *
65 * wwSceneSetup - Usage: wwSceneSetup arg_array *
66 * *
67 * INPUT: *
68 * The contents of the given array is assumed to be as follows: *
69 * lod_count (int) - the number of LOD models that will be created *
70 * lod_offset (float) - X offset to move the LODs by *
71 * lod_clone (int) - 1==copy 2==instance 3==reference *
72 * damage_count (int) - the number of damage models that will be created *
73 * damage_offset (float) - Y offset to move the LODs by *
74 * damage_clone (int) - 1==copy 2==instance 3==reference *
75 * *
76 * OUTPUT: *
77 * The function returns an array of the new values the user selected in the same format as *
78 * above.
79 * *
80 * WARNINGS: *
81 * *
82 * HISTORY: *
83 * 9/27/1999 AJA : Created. *
84 *=============================================================================================*/
85Value * scene_setup_cf (Value **arg_list, int count)
86{
87 // We don't want any arguments for this function.
88 check_arg_count("wwSceneSetup", 1, count);
89 type_check(arg_list[0], Array, "Parameter array");
90
91 SceneSetupDlg dlg(MAXScript_interface);
92 one_typed_value_local(Array* result);
93
94 // Read the initial values out of the array.
95 Array *args = (Array*)(arg_list[0]);
96 dlg.m_LodCount = (args->get(1))->to_int();
97 dlg.m_LodOffset = (args->get(2))->to_float();
98 dlg.m_LodProc = (args->get(3))->to_int();
99 dlg.m_DamageCount = (args->get(4))->to_int();
100 dlg.m_DamageOffset = (args->get(5))->to_float();
101 dlg.m_DamageProc = (args->get(6))->to_int();
102
103 // Display the dialog
104 if (dlg.DoModal() == IDCANCEL)
105 {
106 pop_value_locals();
107 return &undefined;
108 }
109
110 // Stuff the values the user chose into the array we'll return.
111 vl.result = new Array(6);
112 vl.result->append(Integer::intern(dlg.m_LodCount));
113 vl.result->append(Float::intern(dlg.m_LodOffset));
114 vl.result->append(Integer::intern(dlg.m_LodProc));
115 vl.result->append(Integer::intern(dlg.m_DamageCount));
116 vl.result->append(Float::intern(dlg.m_DamageOffset));
117 vl.result->append(Integer::intern(dlg.m_DamageProc));
118
119 // Return the array of new values.
120 return_value(vl.result);
121}
