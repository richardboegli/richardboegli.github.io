479void RecorderClass::updateRecord()
480{
481 Bool needFlush = FALSE;
482 static Int lastFrame = -1;
483 GameMessage *msg = TheCommandList->getFirstMessage();
484 while (msg != NULL) {
485 if (msg->getType() == GameMessage::MSG_NEW_GAME &&
486 msg->getArgument(0)->integer != GAME_SHELL &&
487 msg->getArgument(0)->integer != GAME_SINGLE_PLAYER && // Due to the massive amount of scripts that use <local player> in GC and single player, replays have been cut for them.
488 msg->getArgument(0)->integer != GAME_NONE)
489 {
490 m_originalGameMode = msg->getArgument(0)->integer;
491 DEBUG_LOG(("RecorderClass::updateRecord() - original game is mode %d\n", m_originalGameMode));
492 lastFrame = 0;
493 GameDifficulty diff = DIFFICULTY_NORMAL;
494 if (msg->getArgumentCount() >= 2)
495 diff = (GameDifficulty)msg->getArgument(1)->integer;
496 Int rankPoints = 0;
497 if (msg->getArgumentCount() >= 3)
498 rankPoints = msg->getArgument(2)->integer;
499 Int maxFPS = 0;
500 if (msg->getArgumentCount() >= 4)
501 maxFPS = msg->getArgument(3)->integer;
502
503 startRecording(diff, m_originalGameMode, rankPoints, maxFPS);
504 } else if (msg->getType() == GameMessage::MSG_CLEAR_GAME_DATA) {
505 if (m_file != NULL) {
506 lastFrame = -1;
507 writeToFile(msg);
508 stopRecording();
509 }
510 m_fileName.clear();
511 } else {
512 if (m_file != NULL) {
513 if ((msg->getType() > GameMessage::MSG_BEGIN_NETWORK_MESSAGES) &&
514 (msg->getType() < GameMessage::MSG_END_NETWORK_MESSAGES)) {
515 // Only write the important messages to the file.
516 writeToFile(msg);
517 needFlush = TRUE;
518 }
519 }
520 }
521 msg = msg->next();
522 }
523
524 if (needFlush) {
525 fflush(m_file);
526 }
527}

724void RecorderClass::writeToFile(GameMessage * msg) {
725 // Write the frame number for this command.
726 UnsignedInt frame = TheGameLogic->getFrame();
727 fwrite(&frame, sizeof(frame), 1, m_file);
728
729 // Write the command type
730 GameMessage::Type type = msg->getType();
731 fwrite(&type, sizeof(type), 1, m_file);
732
733 // Write the player index
734 Int playerIndex = msg->getPlayerIndex();
735 fwrite(&playerIndex, sizeof(playerIndex), 1, m_file);
736
737#ifdef DEBUG_LOGGING
738 AsciiString commandName = msg->getCommandAsAsciiString();
739 if (type < GameMessage::MSG_BEGIN_NETWORK_MESSAGES || type > GameMessage::MSG_END_NETWORK_MESSAGES)
740 {
741 commandName.concat(" (Non-Network message!)");
742 }
743 else if (type == GameMessage::MSG_BEGIN_NETWORK_MESSAGES)
744 {
745 AsciiString tmp;
746 tmp.format(" (CRC 0x%8.8X)", msg->getArgument(0)->integer);
747 commandName.concat(tmp);
748 }
749
750 //DEBUG_LOG(("RecorderClass::writeToFile - Adding %s command from player %d to TheCommandList on frame %d\n",
751 //commandName.str(), msg->getPlayerIndex(), TheGameLogic->getFrame()));
752#endif // DEBUG_LOGGING
753
754 GameMessageParser *parser = newInstance(GameMessageParser)(msg);
755 UnsignedByte numTypes = parser->getNumTypes();
756 fwrite(&numTypes, sizeof(numTypes), 1, m_file);
757
758 GameMessageParserArgumentType *argType = parser->getFirstArgumentType();
759 while (argType != NULL) {
760 UnsignedByte type = (UnsignedByte)(argType->getType());
761 fwrite(&type, sizeof(type), 1, m_file);
762
763 UnsignedByte argTypeCount = (UnsignedByte)(argType->getArgCount());

816Bool RecorderClass::readReplayHeader(ReplayHeader& header)
817{
818 AsciiString filepath = getReplayDir();
819 filepath.concat(header.filename.str());
820 m_file = fopen(filepath.str(), "rb");
821 if (m_file == NULL)
822 {
823 DEBUG_LOG(("Can't open %s (%s)\n", filepath.str(), header.filename.str()));
824 return FALSE;
825 }
826
827 // Read the GENREP header.
828 char genrep[7];
829 fread(&genrep, sizeof(char), 6, m_file);
830 genrep[6] = 0;
831 if (strncmp(genrep, "GENREP", 6)) {
832 DEBUG_LOG(("RecorderClass::readReplayHeader - replay file did not have GENREP at the start.\n"));
833 fclose(m_file);
834 m_file = NULL;
835 return FALSE;
836 }
837
838 // read in some stats
839 fread(&header.startTime, sizeof(time_t), 1, m_file);
840 fread(&header.endTime, sizeof(time_t), 1, m_file);
841
842 fread(&header.frameDuration, sizeof(UnsignedInt), 1, m_file);
843
844 fread(&header.desyncGame, sizeof(Bool), 1, m_file);
845 fread(&header.quitEarly, sizeof(Bool), 1, m_file);
846 for (Int i=0; i<MAX_SLOTS; ++i)
847 {
848 fread(&(header.playerDiscons[i]), sizeof(Bool), 1, m_file);
849 }
850
851 // Read the Replay Name. We don't actually do anything with it. Oh well.
852 header.replayName = readUnicodeString();
853
854 // Read the date and time. We don't really do anything with this either. Oh well.
855 fread(&header.timeVal, sizeof(SYSTEMTIME), 1, m_file);
856
857 // Read in the Version info
858 header.versionString = readUnicodeString();
859 header.versionTimeString = readUnicodeString();
860 fread(&header.versionNumber, sizeof(UnsignedInt), 1, m_file);
861 fread(&header.exeCRC, sizeof(UnsignedInt), 1, m_file);
862 fread(&header.iniCRC, sizeof(UnsignedInt), 1, m_file);
863
864 // Read in the GameInfo
865 header.gameOptions = readAsciiString();
866 m_gameInfo.reset();
867 m_gameInfo.enterGame();
868 DEBUG_LOG(("RecorderClass::readReplayHeader - GameInfo = %s\n", header.gameOptions.str()));
869 if (!ParseAsciiStringToGameInfo(&m_gameInfo, header.gameOptions))
870 {
871 DEBUG_LOG(("RecorderClass::readReplayHeader - replay file did not have a valid GameInfo string.\n"));
872 fclose(m_file);
873 m_file = NULL;
874 return FALSE;
875 }
876 m_gameInfo.startGame(0);
877
878 AsciiString playerIndex = readAsciiString();
879 header.localPlayerIndex = atoi(playerIndex.str());
880 if (header.localPlayerIndex < -1 || header.localPlayerIndex >= MAX_SLOTS)
881 {
882 DEBUG_LOG(("RecorderClass::readReplayHeader - invalid local slot number.\n"));
883 m_gameInfo.endGame();
884 m_gameInfo.reset();
885 fclose(m_file);
886 m_file = NULL;
887 return FALSE;
888 }
889 if (header.localPlayerIndex >= 0)
890 {
891 Int localIP = m_gameInfo.getSlot(header.localPlayerIndex)->getIP();
892 m_gameInfo.setLocalIP(localIP);
893 }
894
895 if (!header.forPlayback)
896 {
897 m_gameInfo.endGame();
898 m_gameInfo.reset();
899 fclose(m_file);
900 m_file = NULL;
901 }
902 return TRUE;
903}
