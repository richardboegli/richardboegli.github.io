153// Window status flags --------------------------------------------------------

155enum
156{
157
158 // when you edit this, remember to edit WindowStatusNames[]
159 WIN_STATUS_NONE = 0x00000000, // No status bits set at all
160 WIN_STATUS_ACTIVE = 0x00000001, // At the top of the window list
161 WIN_STATUS_TOGGLE = 0x00000002, // If set, click to toggle
162 WIN_STATUS_DRAGABLE = 0x00000004, // Window can be dragged
163 WIN_STATUS_ENABLED = 0x00000008, // Window can receive input
164 WIN_STATUS_HIDDEN = 0x00000010, // Window is hidden, no input
165 WIN_STATUS_ABOVE = 0x00000020, // Window is always above others
166 WIN_STATUS_BELOW = 0x00000040, // Window is always below others
167 WIN_STATUS_IMAGE = 0x00000080, // Window is drawn with images
168 WIN_STATUS_TAB_STOP = 0x00000100, // Window is a tab stop
169 WIN_STATUS_NO_INPUT = 0x00000200, // Window does not take input
170 WIN_STATUS_NO_FOCUS = 0x00000400, // Window does not take focus
171 WIN_STATUS_DESTROYED = 0x00000800, // Window has been destroyed
172 WIN_STATUS_BORDER = 0x00001000, // Window will be drawn with Borders & Corners
173 WIN_STATUS_SMOOTH_TEXT = 0x00002000, // Window text will be drawn with smoothing
174 WIN_STATUS_ONE_LINE = 0x00004000, // Window text will be drawn on only one line
175 WIN_STATUS_NO_FLUSH = 0x00008000, // Window images will not be unloaded when window is hidden
176 WIN_STATUS_SEE_THRU = 0x00010000, // Will not draw, but it NOT hidden ... does not apply to children
177 WIN_STATUS_RIGHT_CLICK = 0x00020000, // Window pays attention to right clicks
178 WIN_STATUS_WRAP_CENTERED = 0x00040000, // Text will be centered on each word wrap or \n
179 WIN_STATUS_CHECK_LIKE = 0x00080000, // Make push buttons "check-like" with dual state
180 WIN_STATUS_HOTKEY_TEXT = 0x00100000, // Make push buttons "check-like" with dual state
181 WIN_STATUS_USE_OVERLAY_STATES = 0x00200000, // Push buttons will use the global automatic rendering overlay for disabled, hilited, and pushed.
182 WIN_STATUS_NOT_READY = 0x00400000, // A disabled button that is available -- but not yet (power charge, fire delay).
183 WIN_STATUS_FLASHING = 0x00800000, // Used for buttons that do cameo flashes.
184 WIN_STATUS_ALWAYS_COLOR = 0x01000000, // Never render these buttons using greyscale renderer when button disabled.
185 WIN_STATUS_ON_MOUSE_DOWN = 0x02000000, // Pushbutton triggers on mouse down.
186 WIN_STATUS_SHORTCUT_BUTTON = 0x04000000, // Oh god... this is a total hack for shortcut buttons to handle rendering text top left corner...
187 // when you edit this, remember to edit WindowStatusNames[]
