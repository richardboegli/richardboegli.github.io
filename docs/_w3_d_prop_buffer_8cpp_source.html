86void W3DPropBuffer::cull(CameraClass * camera)
87{
88 Int curProp;
89
90 for (curProp=0; curProp<m_numProps; curProp++) {
91 Bool visible = !camera->Cull_Sphere(m_props[curProp].bounds);
92 m_props[curProp].visible=visible;
93 }
94}

153//=============================================================================
154// W3DPropBuffer::addPropTypes
155//=============================================================================
157//=============================================================================
158Int W3DPropBuffer::addPropType(const AsciiString &modelName)
159{
160 if (m_numPropTypes>=MAX_TYPES) {
161 DEBUG_CRASH(("Too many kinds of props in map. Reduce kinds of props, or raise prop limit. jba."));
162 return 0;
163 }
164
165 m_propTypes[m_numPropTypes].m_robj = WW3DAssetManager::Get_Instance()->Create_Render_Obj(modelName.str());
166 if (m_propTypes[m_numPropTypes].m_robj==NULL) {
167 DEBUG_CRASH(("Unable to find model for prop %s\n", modelName.str()));
168 return -1;
169 }
170 m_propTypes[m_numPropTypes].m_robjName = modelName;
171
172 SphereClass bounds = m_propTypes[m_numPropTypes].m_robj->Get_Bounding_Sphere();
173 m_propTypes[m_numPropTypes].m_bounds = bounds;
174 m_numPropTypes++;
175 return m_numPropTypes-1;
176}

178//=============================================================================
179// W3DPropBuffer::addProp
180//=============================================================================
183//=============================================================================
184void W3DPropBuffer::addProp(Int id, Coord3D location, Real angle,Real scale, const AsciiString &modelName)
185{
186 if (m_numProps >= MAX_PROPS) {
187 return;
188 }
189 if (!m_initialized) {
190 return;
191 }
192 Int propType = -1;
193 Int i;
194 for (i=0; i<m_numPropTypes; i++) {
195 if (m_propTypes[i].m_robjName.compareNoCase(modelName)==0) {
196 propType = i;
197 break;
198 }
199 }
200 if (propType<0) {
201 propType = addPropType(modelName);
202 if (propType<0) {
203 return;
204 }
205 }
206
207 Matrix3D mtx(true);
208 mtx.Rotate_Z(angle);
209 mtx.Scale(scale);
210 mtx.Set_Translation(Vector3(location.x, location.y, location.z));
211
212 m_props[m_numProps].location = location;
213 m_props[m_numProps].id = id;
214 m_props[m_numProps].ss = OBJECTSHROUD_INVALID;
215 m_props[m_numProps].m_robj = m_propTypes[propType].m_robj->Clone();
216 m_props[m_numProps].m_robj->Set_Transform(mtx);
217 m_props[m_numProps].m_robj->Set_ObjectScale(scale);
218 m_props[m_numProps].propType = propType;
219 // Translate the bounding sphere of the model.
220 m_props[m_numProps].bounds = m_propTypes[propType].m_bounds;
221 m_props[m_numProps].bounds.Center += Vector3(location.x, location.y, location.z);
222 // Initially set it invisible. cull will update it's visiblity flag.
223 m_props[m_numProps].visible = false;
224
225 m_numProps++;
226}

228//=============================================================================
229// W3DPropBuffer::updatePropPosition
230//=============================================================================
232//=============================================================================
233Bool W3DPropBuffer::updatePropPosition(Int id, const Coord3D &location, Real angle, Real scale)
234{
235 Int i;
236 for (i=0; i<m_numProps; i++) {
237 if (m_props[i].id == id) {
238 Matrix3D mtx(true);
239 mtx.Rotate_Z(angle);
240 mtx.Scale(scale);
241 mtx.Set_Translation(Vector3(location.x, location.y, location.z));
242 m_props[i].location = location;
243 m_props[i].m_robj->Set_Transform(mtx);
244 m_props[i].m_robj->Set_ObjectScale(scale);
245 // Translate the bounding sphere of the model.
246 m_props[i].bounds = m_propTypes[m_props[i].propType].m_bounds;
247 m_props[i].bounds.Center += Vector3(location.x, location.y, location.z);
248 m_anythingChanged = true;
249 return true;
250 }
251 }
252 return false;
253}

270 m_props[i].bounds.Radius = 1;
271 m_anythingChanged = true;

277// W3DPropBuffer::removePropsForConstruction
278//=============================================================================
280//=============================================================================
281void W3DPropBuffer::removePropsForConstruction(const Coord3D* pos, const GeometryInfo& geom, Real angle )
282{
283 // Just iterate all trees, as even non-collidable ones get removed. jba. [7/11/2003]
284 Int i;
285 for (i=0; i<m_numProps; i++) {
286 if (m_props[i].m_robj == NULL) {
287 continue; // already deleted.
288 }
289 Real radius = m_props[i].bounds.Radius;
290 GeometryInfo info(GEOMETRY_CYLINDER, false, 5*radius, 2*radius, 2*radius);
291 if (ThePartitionManager->geomCollidesWithGeom( pos, geom, angle, &m_props[i].location, info, 0.0f)) {
292 // remove it [7/11/2003]
293 m_props[i].location.set(0,0,0);
294 m_props[i].propType = -1;
295 REF_PTR_RELEASE(m_props[i].m_robj);
296 // Translate the bounding sphere of the model.
297 m_props[i].bounds.Center = Vector3(0,0,0);
298 m_props[i].bounds.Radius = 1;
299 m_anythingChanged = true;
300 }
301 }
302}

322//=============================================================================
323// W3DPropBuffer::drawProps
324//=============================================================================
326//=============================================================================
327void W3DPropBuffer::drawProps(RenderInfoClass &rinfo)
328{
329 USE_PERF_TIMER(Prop_Render)
330
331 Int i;
332 if (m_doCull) {
333 cull(&rinfo.Camera);
334 }
335 const GlobalData::TerrainLighting *objectLighting = TheGlobalData->m_terrainObjectsLighting[TheGlobalData->m_timeOfDay];
336
337 LightEnvironmentClass lightEnv;
338 Vector3 center(0,0,0); // arbitrary center point. [6/6/2003]
339 Vector3 ambient(objectLighting[0].ambient.red, objectLighting[0].ambient.green, objectLighting[0].ambient.blue);
340 lightEnv.Reset(center, ambient);
341
342 Matrix3D mtx;
343 const Vector3 zeroVector(0.0f, 0.0f, 0.0f);
344 const Vector3 xVector(1.0f, 0.0f, 0.0f);
345 const Vector3 yVector(0.0f, 1.0f, 0.0f);
346
347 for (i = 0; i < MAX_GLOBAL_LIGHTS; ++i)
348 {
349 m_light->Set_Ambient(zeroVector);
350 m_light->Set_Diffuse(Vector3(objectLighting[i].diffuse.red,
351 objectLighting[i].diffuse.green,
352 objectLighting[i].diffuse.blue));
353 m_light->Set_Specular(zeroVector);
354 mtx.Set(xVector, yVector, Vector3(objectLighting[i].lightPos.x, objectLighting[i].lightPos.y, objectLighting[i].lightPos.z), zeroVector);
355 m_light->Set_Transform(mtx);
356 lightEnv.Add_Light(*m_light);
357 }
358
359 rinfo.light_environment = &lightEnv;
360 for (i=0; i<m_numProps; i++) {
361 if (!m_props[i].visible) {
362 continue;
363 }
364 if (m_props[i].m_robj==NULL) {
365 continue;
366 }
367 if (!ThePlayerList || !ThePartitionManager) {
368 // in Worldbuilder. No shroud either. jba. [6/9/2003]
369 m_props[i].ss = OBJECTSHROUD_CLEAR;
370 }
371 if (m_props[i].ss == OBJECTSHROUD_INVALID) {
372 Int localPlayerIndex = ThePlayerList ? ThePlayerList->getLocalPlayer()->getPlayerIndex() : 0;
373 m_props[i].ss = ThePartitionManager->getPropShroudStatusForPlayer(localPlayerIndex, &m_props[i].location);
374 }
375 if (m_props[i].ss >= OBJECTSHROUD_SHROUDED) {
376 continue;
377 }
378 if (m_props[i].ss <= OBJECTSHROUD_INVALID) {
379 continue;
380 }
381 if (TheTerrainRenderObject->getShroud() && m_props[i].ss != CELLSHROUD_CLEAR) {
382 rinfo.Push_Material_Pass(m_propShroudMaterialPass);
383 m_props[i].m_robj->Render(rinfo);
384 rinfo.Pop_Material_Pass();
385 } else {
386 m_props[i].m_robj->Render(rinfo);
387 }
388 }
389 rinfo.light_environment = NULL;
390
391}
