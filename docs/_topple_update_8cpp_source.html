132static Real angleClosestTo(Real a1, Real a2, Real desired)
133{
134 a1 = normalizeAngle(a1);
135 a2 = normalizeAngle(a2);
136 return (fabs(stdAngleDiff(desired, a1)) < fabs(stdAngleDiff(desired, a2))) ? a1 : a2;
137}

142void ToppleUpdate::applyTopplingForce( const Coord3D* toppleDirection, Real toppleSpeed,
143 UnsignedInt options )
144{
145 if (getObject()->isEffectivelyDead())
146 return;
147
148 //DEBUG_LOG(("awaking ToppleUpdate %08lx\n",this));
149 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
150
151 const ToppleUpdateModuleData* d = getToppleUpdateModuleData();
152
153 if (d->m_killWhenStartToppled)
154 {
155 setWakeFrame(getObject(), UPDATE_SLEEP_FOREVER);
156 getObject()->kill();
157 return;
158 }
159
160 m_toppleDirection = *toppleDirection;
161 m_toppleDirection.normalize();
162 TheScriptEngine->adjustToppleDirection(getObject(), &m_toppleDirection);
163
164 m_angularVelocity = toppleSpeed * d->m_initialVelocityPercent;
165 m_angularAcceleration = toppleSpeed * d->m_initialAccelPercent;
166 m_toppleState = TOPPLE_FALLING;
167 m_options = options;
168
169 // tell the drawable to stop swaying
170 Drawable * draw = getObject()->getDrawable();
171 static NameKeyType nameKeySwayUpdate = NAMEKEY("SwayClientUpdate");
172
173 ClientUpdateModule ** clientModules = draw->getClientUpdateModules();
174 if (clientModules)
175 {
176 while (*clientModules)
177 {
178 if ((*clientModules)->getModuleNameKey() == nameKeySwayUpdate)
179 (*(SwayClientUpdate **)clientModules)->stopSway();
180
181 ++clientModules;
182 }
183 }
184
185 // rotate around the z-axis so that our x-axis is perpendicular to the topple direction.
186 // this is really a trick to ensure that relatively planar things (eg, streetlights)
187 // fall parallel to the ground, so that they don't up sticking thru the ground.
188 // yeah, it assumes the models are constructed appropriately, but is a cheap way
189 // of minimizing the problem. (srj)
190 Real curAngleX = normalizeAngle(getObject()->getOrientation());
191 Real toppleAngle = normalizeAngle(atan2(m_toppleDirection.y, m_toppleDirection.x));
192 if (d->m_toppleLeftOrRightOnly)
193 {
194 // it's a fence or such, and can only topple left or right, so pick the closest
195 toppleAngle = angleClosestTo(curAngleX + PI/2, curAngleX - PI/2, toppleAngle);
196 m_toppleDirection.x = Cos(toppleAngle);
197 m_toppleDirection.y = Sin(toppleAngle);
198
199 // go ahead and remove it from the pathfinder now, rather than waiting for the topple to
200 // finish.... since we might be in a slightly different position when toppled, which can
201 // confuse the pathfinder and not de-obstacle everything correctly
202 TheAI->pathfinder()->removeObjectFromPathfindMap(getObject());
203
204 }
205 // desired angle is toppleAngle +/- pi/2, whichever is closer to curangle
206 Real desiredAngleX = angleClosestTo(toppleAngle + PI/2, toppleAngle - PI/2, curAngleX);
207 m_numAngleDeltaX = REAL_TO_INT_FLOOR(ANGULAR_LIMIT / (m_angularVelocity * 2));
208 if (m_numAngleDeltaX < 1)
209 m_numAngleDeltaX = 1;
210 m_angleDeltaX = (desiredAngleX - curAngleX) / m_numAngleDeltaX;
211
212 getObject()->getDrawable()->setModelConditionState(MODELCONDITION_TOPPLED);
213 FXList::doFXObj(d->m_toppleFX, getObject());
214
215 // if this is a tree, create a stump
216 if (!d->m_stumpName.isEmpty())
217 {
218 const ThingTemplate* ttn = TheThingFactory->findTemplate(d->m_stumpName);
219 Object *stump = TheThingFactory->newObject( ttn, NULL );
220 if (stump)
221 {
222 stump->setPosition( getObject()->getPosition() );
223 stump->setOrientation( getObject()->getOrientation() );
224 m_stumpID = stump->getID();
225
226 // if we are "burned", then we will burn our stump
227 const Drawable* draw = getObject()->getDrawable();
228 if( draw )
229 {
230 if( draw->getModelConditionFlags().test( MODELCONDITION_BURNED ) == TRUE )
231 {
232 Drawable* stumpDraw = stump->getDrawable();
233
234 if( stumpDraw )
235 stumpDraw->setModelConditionState( MODELCONDITION_BURNED );
236
237 }
238 }
239 }
240 }
241}

252static void deathByToppling(Object* obj)
253{
254 // use a special "topppling" damage type here so that
255 // toppled stuff can have different damage/die modules
256 // for toppled-death vs other-death
257 DamageInfo damageInfo;
258 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
259 damageInfo.in.m_deathType = DEATH_TOPPLED;
260 damageInfo.in.m_sourceID = INVALID_ID;
261 damageInfo.in.m_amount = HUGE_DAMAGE_AMOUNT;

268UpdateSleepTime ToppleUpdate::update()
269{
270 //DEBUG_LOG(("updating ToppleUpdate %08lx\n",this));
271 DEBUG_ASSERTCRASH(m_toppleState != TOPPLE_UPRIGHT, ("hmm, we should be sleeping here"));
272 if ( (m_toppleState == TOPPLE_UPRIGHT) || (m_toppleState == TOPPLE_DOWN) )
273 return UPDATE_SLEEP_FOREVER;
274
275 const ToppleUpdateModuleData* d = getToppleUpdateModuleData();
276 const Real VELOCITY_BOUNCE_LIMIT = 0.01f; // if the velocity after a bounce will be this or lower, just stop at zero
277 const Real VELOCITY_BOUNCE_SOUND_LIMIT = 0.03f; // and if this low, then skip the bounce sound
278
279 Object* obj = getObject();
280 if (m_numAngleDeltaX)
281 {
282 Matrix3D xfrm = *obj->getTransformMatrix();
283 xfrm.In_Place_Pre_Rotate_Z(m_angleDeltaX);
284 obj->setTransformMatrix(&xfrm);
285 --m_numAngleDeltaX;
286 }
287
288 Real curVelToUse = m_angularVelocity;
289 if (m_angularAccumulation + curVelToUse > ANGULAR_LIMIT)
290 curVelToUse = ANGULAR_LIMIT - m_angularAccumulation;
291
292 Matrix3D xfrm = *obj->getTransformMatrix();
293 xfrm.In_Place_Pre_Rotate_X(-curVelToUse * m_toppleDirection.y);
294 xfrm.In_Place_Pre_Rotate_Y(curVelToUse * m_toppleDirection.x);
295 obj->setTransformMatrix(&xfrm);
296
297 m_angularAccumulation += curVelToUse;
298 if ((m_angularAccumulation >= ANGULAR_LIMIT) && (m_angularVelocity > 0))
299 {
300 // Hit so either bounce or stop if too little remaining velocity.
301 m_angularVelocity *= -d->m_bounceVelocityPercent;
302
303 if( BitTest( m_options, TOPPLE_OPTIONS_NO_BOUNCE ) == TRUE ||
304 fabs(m_angularVelocity) < VELOCITY_BOUNCE_LIMIT )
305 {
306 // too slow, just stop
307 m_angularVelocity = 0;
308 m_toppleState = TOPPLE_DOWN;
309
310 if (d->m_killWhenToppled)
311 {
312 deathByToppling(obj);
313 if (d->m_reorientToppledRubble)
314 {
315 // we have a separate rubble state that needs to be upright, and centered
316 // on the new "center" pos...
317 Vector3 pos;
318 pos.X = 0;
319 pos.Y = 0;
320 pos.Z = obj->getGeometryInfo().getMaxHeightAbovePosition();
321 Matrix3D::Transform_Vector(*obj->getTransformMatrix(), pos, &pos);
322
323 Coord3D tmp;
324 tmp.x = pos.X;
325 tmp.y = pos.Y;
326 tmp.z = pos.Z;
327 obj->setPosition(&tmp);
328
329 // this relies on the fact that setOrientation always forces us straight up in the Z axis!
330 obj->setOrientation(obj->getOrientation());
331
332 }
333 } // if kill when toppled
334
335 if (d->m_killStumpWhenToppled)
336 {
337 Object* stump = TheGameLogic->findObjectByID(m_stumpID);
338 if (stump)
339 {
340 deathByToppling(stump);
341 }
342 }
343 }
344 else if( fabs(m_angularVelocity) >= VELOCITY_BOUNCE_SOUND_LIMIT )
345 {
346 // fast enough bounce to warrant the bounce fx
347 if( BitTest( m_options, TOPPLE_OPTIONS_NO_FX ) == FALSE )
348 FXList::doFXObj(d->m_bounceFX, obj);
349 }
350 }
351 else
352 {
353 m_angularVelocity += m_angularAcceleration;
354 }
355
356 Drawable *draw=obj->getDrawable();
357 if (draw)
358 draw->setShadowsEnabled(false);
359
360 return UPDATE_SLEEP_NONE;
361}

366void ToppleUpdate::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
367{
368 // Note that other == null means "collide with ground"
369 //
370 if (other == NULL)
371 return;
372
373 //@todo JohnA -- Should you get around to adding trees to avoidance pathfinding, then you'll
374 //want to change this code:
375 //if( other->getCrusherLevel() > getObject()->getCrushableLevel() ) //<----proper tree method
376 if( other->getCrusherLevel() > 1 )
377 {
378
379 // Give a vector with direction to thing and my speed.
380 Coord3D toppleVector = *getObject()->getPosition();
381 toppleVector.x -= other->getPosition()->x;
382 toppleVector.y -= other->getPosition()->y;
383 toppleVector.z = 0;
384
385 Coord3D vel;
386 PhysicsBehavior* phys = other->getPhysics();
387 if (phys)
388 vel = *phys->getVelocity();
389 else
390 vel.zero();
391 getObject()->topple( &toppleVector, vel.length(), TOPPLE_OPTIONS_NONE );
392
393 }
394
395}
