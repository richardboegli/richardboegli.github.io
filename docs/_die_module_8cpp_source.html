71Bool DieMuxData::isDieApplicable(const Object* obj, const DamageInfo *damageInfo) const
72{
73 // wrong death type? punt
74 if (!getDeathTypeFlag(m_deathTypes, damageInfo->in.m_deathType))
75 return false;
76
77 // wrong vet level? punt
78 if (!getVeterancyLevelFlag(m_veterancyLevels, obj->getVeterancyLevel()))
79 return false;
80
81 // all 'exempt' bits must be clear for us to run.
82 if( m_exemptStatus.any() && obj->getStatusBits().testForAny( m_exemptStatus ) )
83 return false;
84
85 // all 'required' bits must be set for us to run.
86 // But only if we have a required status to check
87 if( m_requiredStatus.any() && !obj->getStatusBits().testForAll( m_requiredStatus ) )
88 return false;
89
90 return true;
91}
