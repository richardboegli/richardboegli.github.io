55/***********************************************************************************************
56 * RAMFileClass::RAMFileClass -- Construct a RAM buffer based "file" object. *
57 * *
58 * This routine will construct a "file" object that actually is just a front end processor *
59 * for a buffer. Access to the buffer will appear as if it was accessing a file. This *
60 * is different from the caching ability of the buffered file class in that this file *
61 * class has no real file counterpart. Typical use of this is for algorithms that were *
62 * originally designed for file processing, but are now desired to work with a buffer. *
63 * *
64 * INPUT: buffer -- Pointer to the buffer to use for this file. The buffer will already *
65 * contain data if the file is opened for READ. It will be considered *
66 * a scratch buffer if opened for WRITE. If the buffer pointer is NULL *
67 * but the length parameter is not, then a buffer will be allocated *
68 * of the specified length. This case is only useful for opening the *
69 * file for WRITE. *
70 * *
71 * length -- The length of the buffer submitted to this routine. *
72 * *
73 * OUTPUT: none *
74 * *
75 * WARNINGS: none *
76 * *
77 * HISTORY: *
78 * 07/03/1996 JLB : Created. *
79 *=============================================================================================*/
80RAMFileClass::RAMFileClass(void * buffer, int len) :
81 Buffer((char *)buffer),
82 MaxLength(len),
83 Length(len),
84 Offset(0),
85 Access(READ),
86 IsOpen(false),
87 IsAllocated(false)
88{
89 if (buffer == NULL && len > 0) {
90 Buffer = W3DNEWARRAY char[len];
91 IsAllocated = true;
92 }
93}

236/***********************************************************************************************
237 * RAMFileClass::Open -- Opens the RAM based file. *
238 * *
239 * This will open the ram based file for read or write. If the file is opened for write, *
240 * the the 'file' can be written up to the limit of the buffer's size. If the file is *
241 * opened for read, then the buffer is presumed to hold the data to be read. *
242 * *
243 * INPUT: *
244 * *
245 * OUTPUT: *
246 * *
247 * WARNINGS: *
248 * *
249 * HISTORY: *
250 * 07/03/1996 JLB : Created. *
251 *=============================================================================================*/
252int RAMFileClass::Open(int access)
253{
254 if (Buffer == NULL || Is_Open()) {
255 return(false);
256 }
257
258 Offset = 0;
259 Access = access;
260 IsOpen = true;
261
262 switch (access) {
263 default:
264 case READ:
265 break;
266
267 case WRITE:
268 Length = 0;
269 break;
270
271 case READ|WRITE:
272 break;
273 }
274
275 return(Is_Open());
276}

279/***********************************************************************************************
280 * RAMFileClass::Read -- Read data from the file. *
281 * *
282 * Use this routine just like a normal file read. It will copy the bytes from the ram *
283 * buffer to the destination specified. When the ram buffer is exhausted, less bytes than *
284 * requested will be read. *
285 * *
286 * INPUT: buffer -- Pointer to the buffer to store the data to. *
287 * *
288 * size -- The number of bytes to 'read' into the specified buffer. *
289 * *
290 * OUTPUT: Returns with the number of bytes copied to the destination buffer. If the number *
291 * of bytes returned is less than requested, then this indicates that the source *
292 * buffer is exhausted. *
293 * *
294 * WARNINGS: The read function only applies to ram 'files' opened for read access. *
295 * *
296 * HISTORY: *
297 * 07/03/1996 JLB : Created. *
298 *=============================================================================================*/
299int RAMFileClass::Read(void * buffer, int size)
300{
301 if (Buffer == NULL || buffer == NULL || size == 0) {
302 return(0);
303 }
304
305 bool hasopened = false;
306 if (!Is_Open()) {
307 Open(READ);
308 hasopened = true;
309 } else {
310 if ((Access & READ) == 0) {
311 return(0);
312 }
313 }
314
315 int tocopy = (size < (Length-Offset)) ? size : (Length-Offset);
316 memmove(buffer, &Buffer[Offset], tocopy);
317 Offset += tocopy;
318
319 if (hasopened) {
320 Close();
321 }
322
323 return(tocopy);
324}

327/***********************************************************************************************
328 * RAMFileClass::Seek -- Controls the ram file virtual read position. *
329 * *
330 * This routine will move the read/write position of the ram file to the location specified *
331 * by the offset and direction parameters. It functions similarly to the regular file *
332 * seek method. *
333 * *
334 * INPUT: pos -- The signed offset from the home position specified by the "dir" *
335 * parameter. *
336 * *
337 * dir -- The home position to base the position offset on. This will either be *
338 * the start of the file, the end of the file, or the current read/write *
339 * position. *
340 * *
341 * OUTPUT: Returns with the new file position. *
342 * *
343 * WARNINGS: none *
344 * *
345 * HISTORY: *
346 * 07/03/1996 JLB : Created. *
347 *=============================================================================================*/
348int RAMFileClass::Seek(int pos, int dir)
349{
350 if (Buffer == NULL || !Is_Open()) {
351 return(Offset);
352 }
353
354 int maxoffset = Length;
355 if ((Access & WRITE) != 0) {
356 maxoffset = MaxLength;
357 }
358
359 switch (dir) {
360 case SEEK_CUR:
361 Offset += pos;
362 break;
363
364 case SEEK_SET:
365 Offset = 0 + pos;
366 break;
367
368 case SEEK_END:
369 Offset = maxoffset + pos;
370 break;
371 }
372
373 if (Offset < 0) Offset = 0;
374 if (Offset > maxoffset) Offset = maxoffset;
375
376 if (Offset > Length) {
377 Length = Offset;
378 }
379
380 return(Offset);
381}

406/***********************************************************************************************
407 * RAMFileClass::Write -- Copies data to the ram file. *
408 * *
409 * This function similarly to the regular write operation supported for files. It copies *
410 * the data specified to the current write position in the ram file. *
411 * *
412 * INPUT: buffer -- Pointer to the data to be written. *
413 * *
414 * size -- The number of bytes to write to the file. *
415 * *
416 * OUTPUT: Returns with the actual number of bytes written. This will be less than requested *
417 * if the buffer is exhausted of space prematurely. *
418 * *
419 * WARNINGS: none *
420 * *
421 * HISTORY: *
422 * 07/03/1996 JLB : Created. *
423 *=============================================================================================*/
424int RAMFileClass::Write(void const * buffer, int size)
425{
426 if (Buffer == NULL || buffer == NULL || size == 0) {
427 return(0);
428 }
429
430 bool hasopened = false;
431 if (!Is_Open()) {
432 Open(WRITE);
433 hasopened = true;
434 } else {
435 if ((Access & WRITE) == 0) {
436 return(0);
437 }
438 }
439
440 int maxwrite = MaxLength - Offset;
441 int towrite = (size < maxwrite) ? size : maxwrite;
442 memmove(&Buffer[Offset], buffer, towrite);
443 Offset += towrite;
444
445 if (Offset > Length) {
446 Length = Offset;
447 }
448
449 if (hasopened) {
450 Close();
451 }
452
453 return(towrite);
454}
