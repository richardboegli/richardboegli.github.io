154__inline const AABoxClass &
155SoundCullObjClass::Get_Bounding_Box (void) const
156{
157 // Get the 'real' values from the
158 if (m_SoundObj != NULL) {
159 m_Transform = m_SoundObj->Get_Transform ();
160 m_AABox.Extent.X = m_SoundObj->Get_DropOff_Radius ();
161 m_AABox.Extent.Y = m_SoundObj->Get_DropOff_Radius ();
162 m_AABox.Extent.Z = m_SoundObj->Get_DropOff_Radius ();
163 } else {
164 m_AABox.Extent.X = 0;
165 m_AABox.Extent.Y = 0;
166 m_AABox.Extent.Z = 0;
167 }
168
169 m_AABox.Center = m_Transform.Get_Translation ();
170 return m_AABox;
171}
