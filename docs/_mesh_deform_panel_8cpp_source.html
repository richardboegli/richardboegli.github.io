52//
53// Message_Proc
54//
56BOOL WINAPI
57MeshDeformPanelClass::Message_Proc
58(
59 HWND hwnd,
60 UINT message,
61 WPARAM wparam,
62 LPARAM lparam
63)
64{
65 // Lookup the controlling object for this panel
66 MeshDeformPanelClass *panel_obj = MeshDeformPanelClass::Get_Object (hwnd);
67 BOOL result = FALSE;
68
69 switch (message)
70 {
71 // Create the controlling panel-object
72 case WM_INITDIALOG:
73 panel_obj = new MeshDeformPanelClass (hwnd);
74 SetProp (hwnd, PANEL_OBJ_PROP, (HANDLE)panel_obj);
75 break;
76
77 case WM_DESTROY:
78 result = panel_obj->On_Message (message, wparam, lparam);
79 RemoveProp (hwnd, PANEL_OBJ_PROP);
80 SAFE_DELETE (panel_obj);
81 break;
82 }
83
84 // Pass the message onto the controlling panel-object
85 if (panel_obj != NULL) {
86 result = panel_obj->On_Message (message, wparam, lparam);
87 }
88
89 // Return the TRUE/FALSE result code
90 return result;
91}

262void
263MeshDeformPanelClass::Set_Deformer (MeshDeformClass *obj)
264{
265 if (m_pMeshDeformer != obj) {
266 m_pMeshDeformer = obj;
267
268 // Set the slider position based on the current state of the deformer
269 float state = m_pMeshDeformer->Get_Deform_State ();
270 ::SendDlgItemMessage (m_hWnd, IDC_STATE_SLIDER, TBM_SETPOS, (WPARAM)TRUE, LPARAM(state * 10.0F));
271
272 // Now update the current vertex color
273 Update_Vertex_Color ();
274 }
275
276 return ;
