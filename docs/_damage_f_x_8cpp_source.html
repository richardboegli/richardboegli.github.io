100//-------------------------------------------------------------------------------------------------
101ConstFXListPtr DamageFX::getDamageFXList(DamageType t, Real damageAmount, const Object* source) const
102{
103 /*
104 if damage is zero, never do damage fx. this is by design, since "zero" damage can happen
105 with some special weapons, like the battleship, which is a "faux" weapon that never does damage.
106 if you really need to change this for some reason, consider carefully... (srj)
107 */
108 if (damageAmount == 0.0f)
109 return NULL;
110
111 const DFX& dfx = m_dfx[t][source ? source->getVeterancyLevel() : LEVEL_REGULAR];
112 ConstFXListPtr fx =
113 damageAmount >= dfx.m_amountForMajorFX ?
114 dfx.m_majorDamageFXList :
115 dfx.m_minorDamageFXList;
116
117 return fx;
118}

138//-------------------------------------------------------------------------------------------Static
139static void parseCommonStuff(
140 INI* ini,
141 ConstCharPtrArray names,
142 VeterancyLevel& vetFirst,
143 VeterancyLevel& vetLast,
144 DamageType& damageFirst,
145 DamageType& damageLast
146)
147{
148 if (names)
149 {
150 vetFirst = (VeterancyLevel)INI::scanIndexList(ini->getNextToken(), names);
151 vetLast = vetFirst;
152 }
153 else
154 {
155 vetFirst = LEVEL_FIRST;
156 vetLast = LEVEL_LAST;
157 }
158
159 const char* damageName = ini->getNextToken();
160 if (stricmp(damageName, "Default") == 0)
161 {
162 damageFirst = (DamageType)0;
163 damageLast = (DamageType)(DAMAGE_NUM_TYPES - 1);
164 }
165 else
166 {
167 damageFirst = (DamageType)DamageTypeFlags::getSingleBitFromName(damageName);
168 damageLast = damageFirst;
169 }
170}

173/*static*/ void DamageFX::parseAmount( INI* ini, void* instance, void* /*store*/, const void* userData )
174{
175 DamageFX* self = (DamageFX*)instance;
176 ConstCharPtrArray names = (ConstCharPtrArray)userData;
177
178 VeterancyLevel vetFirst, vetLast;
179 DamageType damageFirst, damageLast;
180 parseCommonStuff(ini, names, vetFirst, vetLast, damageFirst, damageLast);
181
182 Real amt = INI::scanReal(ini->getNextToken());
183
184 for (Int dt = damageFirst; dt <= damageLast; ++dt)
185 {
186 for (Int v = vetFirst; v <= vetLast; ++v)
187 {
188 self->m_dfx[dt][v].m_amountForMajorFX = amt;
189 }
190 }
191}
