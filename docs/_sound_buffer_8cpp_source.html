115//
116// Determine_Stats
117//
118void
119SoundBufferClass::Determine_Stats (unsigned char *buffer)
120{
121 MMSLockClass lock;
122
123 m_Duration = 0;
124 m_Rate = 0;
125 m_Channels = 0;
126 m_Bits = 0;
127 m_Type = WAVE_FORMAT_IMA_ADPCM;
128
129 // Attempt to get statistical information about this sound
130 AILSOUNDINFO info = { 0 };
131 if ((buffer != NULL) && (::AIL_WAV_info (buffer, &info) != 0)) {
132
133 // Cache this information
134 m_Rate = info.rate;
135 m_Channels = info.channels;
136 m_Bits = info.bits;
137 m_Type = info.format;
138
139 // Determine how long this sound will play for
140 float bytes_sec = float((m_Channels * m_Rate * m_Bits) >> 3);
141 m_Duration = (unsigned long)((((float)m_Length) / bytes_sec) * 1000.0F);
142 }
143
144 return ;
145}

193//
194// Load_From_File
195//
196bool
197SoundBufferClass::Load_From_File (FileClass &file)
198{
199 MMSLockClass lock;
200
201 // Assume failure
202 bool retval = false;
203
204 // Start from scratch
205 Free_Buffer ();
206 Set_Filename (file.File_Name ());
207
208 // Open the file if necessary
209 bool we_opened = false;
210 if (file.Is_Open () == false) {
211 we_opened = (file.Open () == TRUE);
212 }
213
214 // Determine the size of the buffer
215 m_Length = file.Size ();
216 WWASSERT (m_Length > 0L);
217 if (m_Length > 0L) {
218
219 // Allocate a new buffer of the correct length and read the contents
220 // of the file into the buffer
221 m_Buffer = W3DNEWARRAY unsigned char[m_Length];
222 retval = bool(file.Read (m_Buffer, m_Length) == (int)m_Length);
223
224 // If we failed, free the buffer
225 if (retval == false) {
226 Free_Buffer ();
227 }
228 Determine_Stats (m_Buffer);
229 }
230
231 // Close the file if necessary
232 if (we_opened) {
233 file.Close ();
234 }
235
236 // Return the true/false result code
237 return retval;
238}
