274Bool GameSpySendChat( UnicodeString message, Bool isAction, GameWindow *playerListbox )
275{
276 RoomType roomType = StagingRoom;
277 if (TheGameSpyChat->getCurrentGroupRoomID())
278 roomType = GroupRoom;
279
280 message.trim();
281 // Echo the user's input to the chat window
282 if (!message.isEmpty())
283 {
284 // Check for slash commands
285 if (handleSlashCommands(message, isAction, playerListbox))
286 {
287 return false; // already handled
288 }
289
290 if (!playerListbox)
291 {
292 // Public message
293 if (isAction)
294 {
295 peerMessageRoom(TheGameSpyChat->getPeer(), roomType, UnicodeStringToQuotedPrintable(message).str(), ActionMessage);
296 //if (roomType == StagingRoom)
297 //handleUnicodeMessage(TheGameSpyChat->getloginName().str(), message, true, true);
298 }
299 else
300 {
301 peerMessageRoom(TheGameSpyChat->getPeer(), roomType, UnicodeStringToQuotedPrintable(message).str(), NormalMessage);
302 //if (roomType == StagingRoom)
303 //handleUnicodeMessage(TheGameSpyChat->getloginName().str(), message, true, false);
304 }
305 return false;
306 }
307
308 // Get the selections (is this a private message?)
309 Int maxSel = GadgetListBoxGetListLength(playerListbox);
310 Int *selections;
311 GadgetListBoxGetSelected(playerListbox, (Int *)&selections);
312
313 if (selections[0] == -1)
314 {
315 // Public message
316 if (isAction)
317 {
318 peerMessageRoom(TheGameSpyChat->getPeer(), roomType, UnicodeStringToQuotedPrintable(message).str(), ActionMessage);
319 //if (roomType == StagingRoom)
320 //handleUnicodeMessage(TheGameSpyChat->getloginName().str(), message, true, true);
321 }
322 else
323 {
324 peerMessageRoom(TheGameSpyChat->getPeer(), roomType, UnicodeStringToQuotedPrintable(message).str(), NormalMessage);
325 //if (roomType == StagingRoom)
326 //handleUnicodeMessage(TheGameSpyChat->getloginName().str(), message, true, false);
327 }
328 return false;
329 }
330 else
331 {
332 // Private message
333
334 // Construct a list
335 AsciiString names = AsciiString::TheEmptyString;
336 AsciiString tmp = AsciiString::TheEmptyString;
337 AsciiString aStr; // AsciiString buf for translating Unicode entries
338 names.format("%s", TheGameSpyChat->getLoginName().str());
339 for (int i=0; i<maxSel; i++)
340 {
341 if (selections[i] != -1)
342 {
343 aStr.translate(GadgetListBoxGetText(playerListbox, selections[i], 0));
344 if (aStr.compareNoCase(TheGameSpyChat->getLoginName()))
345 {
346 tmp.format(",%s", aStr.str());
347 names.concat(tmp);
348 }
349 }
350 else
351 {
352 break;
353 }
354 }
355
356 if (!names.isEmpty())
357 {
358 if (isAction)
359 {
360 peerMessagePlayer(TheGameSpyChat->getPeer(), names.str(), UnicodeStringToQuotedPrintable(message).str(), ActionMessage);
361 }
362 else
363 {
364 peerMessagePlayer(TheGameSpyChat->getPeer(), names.str(), UnicodeStringToQuotedPrintable(message).str(), NormalMessage);
365 }
366 }
367
368 return true;
369 }
370 }
371 return false;
372}

390static handleUnicodeMessage( const char *nick, UnicodeString msg, Bool isPublic, Bool isAction )
391{
392 GameSpyColors style;
393
394 Bool isOwner = false;
395 Int flags = 0;
396 if (TheGameSpyChat->getCurrentGroupRoomID())
397 peerGetPlayerFlags(TheGameSpyChat->getPeer(), nick, GroupRoom, &flags);
398 else
399 peerGetPlayerFlags(TheGameSpyChat->getPeer(), nick, StagingRoom, &flags);
400 isOwner = flags & PEER_FLAG_OP;
401
402 if (isPublic && isAction)
403 {
404 style = (isOwner)?GSCOLOR_CHAT_OWNER_EMOTE:GSCOLOR_CHAT_EMOTE;
405 }
406 else if (isPublic)
407 {
408 style = (isOwner)?GSCOLOR_CHAT_OWNER:GSCOLOR_CHAT_NORMAL;
409 }
410 else if (isAction)
411 {
412 style = (isOwner)?GSCOLOR_CHAT_PRIVATE_OWNER_EMOTE:GSCOLOR_CHAT_PRIVATE_EMOTE;
413 }
414 else
415 {
416 style = (isOwner)?GSCOLOR_CHAT_PRIVATE_OWNER:GSCOLOR_CHAT_PRIVATE;
417 }
418
419 UnicodeString name;
420 name.translate(nick);
421
422 // filters language
423// if( TheGlobalData->m_languageFilterPref )
424// {
425 TheLanguageFilter->filterLine(msg);
426// }
427
428 UnicodeString fullMsg;
429 if (isAction)
430 {
431 fullMsg.format( L"%ls %ls", name.str(), msg.str() );
432 }
433 else
434 {
435 fullMsg.format( L"[%ls] %ls", name.str(), msg.str() );
436 }
437 GameSpyAddText(fullMsg, style);
438}
