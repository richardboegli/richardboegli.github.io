99MultiplayerColorDefinition * MultiplayerSettings::getColor(Int which)
100{
101 if (which == PLAYERTEMPLATE_RANDOM)
102 {
103 return &m_randomColor;
104 }
105 else if (which == PLAYERTEMPLATE_OBSERVER)
106 {
107 return &m_observerColor;
108 }
109 else if (which < 0 || which >= getNumColors())
110 {
111 return NULL;
112 }
113
114 return &m_colorList[which];
115}

117MultiplayerColorDefinition * MultiplayerSettings::findMultiplayerColorDefinitionByName(AsciiString name)
118{
119 MultiplayerColorIter iter = m_colorList.begin();
120
121 while (iter != m_colorList.end())
122 {
123 if (iter->second.getTooltipName() == name)
124 return &(iter->second);
125
126 ++iter;
127 }
128
129 return NULL;
130}
