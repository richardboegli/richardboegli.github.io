54/*-----------------------------------------------------------------------------
55** Windows Version Information class. This is a global object that is used to
56** store information about the specific OS that we are running under. This can
57** be used to make special allowances for differences between OS's, such as when
58** using the registry, or trying to work around a limitaion of a particular OS
59** (their APIs are slightly different...)
60**-----------------------------------------------------------------------------*/
61class WindowsVersionInfo
62{
63 public:
64 WindowsVersionInfo (void);
65 ~WindowsVersionInfo (void) {}
66
67 int Major ( void ) const { return( MajorVersionNumber ); }
68 int Minor ( void ) const { return( MinorVersionNumber ); }
69 int Build ( void ) const { return( BuildNumber ); }
70 bool Is_Win9x ( void ) const { return( IsWin9x ); } // Win 9x
71 bool Is_Win95 ( void ) const { return( IsWin95 ); } // Win 95
72 bool Is_Win98 ( void ) const { return( IsWin98 ); } // Win 98
73 bool Is_WinNT ( void ) const { return( IsWinNT ); } // Win NT
74 bool Is_WinNT4 ( void ) const { return( IsWinNT && MajorVersionNumber == 4 ); } // Win NT
75 bool Is_WinNT5 ( void ) const { return( IsWinNT && MajorVersionNumber == 5 ); } // Win NT
76 bool Is_Win_2000 ( void ) const { return( IsWin2000 ); } // Win 2000
77 bool Is_Win_XP ( void ) const { return( IsWinXP ); } // Win XP
78 int Version ( void ) const { return( WindowsVersion ); }
79 int IsOSR2Release ( void ) const { return( RunningOSR2 ); }
80 const char * Info ( void ) const { return( &AdditionalInfo[0] ); }
81 char * Version_String ( void );
82 char * Version_Name ( void );
83 bool Meets_Minimum_Version_Requirements ( void );
84
85 private:
86 /*-----------------------------------------------------------------------
87 ** Major version number; i.e. for 4.10.1721 this would be '4'
88 */
89 int MajorVersionNumber;
90
91 /*-----------------------------------------------------------------------
92 ** Minor version number; i.e. for 4.10.1721 this would be '10'
93 */
94 int MinorVersionNumber;
95
96 /*-----------------------------------------------------------------------
97 ** Version number expressed as a DWORD; i.e. for 4.10 this would be '410'
98 */
99 int WindowsVersion;
100
101 /*-----------------------------------------------------------------------
102 ** Build number; i.e. for 4.10.1721 this would be '1721'
103 */
104 int BuildNumber;
105
106 /*-----------------------------------------------------------------------
107 ** Is the system running OSR 2 or later release of Windows95.
108 */
109 int RunningOSR2;
110
111 /*-----------------------------------------------------------------------
112 ** Additional Info; i.e. for NT 4.0 with SP3, this would be
113 ** the string 'Service Pack 3'
114 */
115 char AdditionalInfo[128];
116
117 /*-----------------------------------------------------------------------
118 ** Windows 9x flag; true if running on non-NT system
119 */
120 bool IsWin9x;
121 bool IsWin95;
122 bool IsWin98;
123
124 /*-----------------------------------------------------------------------
125 ** Windows NT flag; true if running on Windows NT system
126 */
127 bool IsWinNT;
128
129 /*-----------------------------------------------------------------------
130 ** Windows 2000 (Formerly Windows NT 5.0)
131 ** As you've no doubt heard by now, Windows NT 5.0 has been officially
132 ** christened "Windows 2000."
133 */
134 bool IsWin2000;
135
136 /*-----------------------------------------------------------------------
137 ** Windows XP flag; true if running on Windows NT system
138 */
139 bool IsWinXP;
140
141 /*-----------------------------------------------------------------------
142 ** This array is used for formatting the version # as a string
143 */
144 char VersionName[30];
145};
