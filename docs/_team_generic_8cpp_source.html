69void TeamGeneric::_fillComboBoxesWithScripts()
70{
71 int i = 0;
72 while (s_allControls[i][1]) {
73 CComboBox *pCombo = (CComboBox*) GetDlgItem(s_allControls[i][1]);
74 if (!pCombo) {
75 continue;
76 }
77
78 // Load all the scripts, then add the NONE string.
79 EditParameter::loadScripts(pCombo, true);
80 pCombo->InsertString(0, NONE_STRING);
81 ++i;

85void TeamGeneric::_dictToScripts()
86{
87 CWnd *pText = NULL;
88 CComboBox *pCombo = NULL;
89
90 if (!m_teamDict) {
91 return;
92 }
93
94 int i = 0;
95 while (s_allControls[i][1]) {
96
97 pText = GetDlgItem(s_allControls[i][0]);
98 if (!pText) {
99 continue;
100 }
101
102 pCombo = (CComboBox*) GetDlgItem(s_allControls[i][1]);
103 if (!pCombo) {
104 continue;
105 }
106
107 Bool exists;
108 AsciiString scriptString;
109 AsciiString keyName;
110 keyName.format("%s%d", TheNameKeyGenerator->keyToName(TheKey_teamGenericScriptHook).str(), i);
111 scriptString = m_teamDict->getAsciiString(NAMEKEY(keyName), &exists);
112
113 pText->ShowWindow(SW_SHOW);
114 pCombo->ShowWindow(SW_SHOW);
115
116 if (exists) {
117 Int selNdx = pCombo->FindStringExact(-1, scriptString.str());
118 if (selNdx == LB_ERR) {
119 pCombo->SetCurSel(0);
120 } else {
121 pCombo->SetCurSel(selNdx);
122 }
123 } else {
124 break;
125 }
126
127 ++i;
128 }
129
130 if (!s_allControls[i][1]) {
131 // We filled everything.
132 return;
133 }
134
135 if (!pCombo) {
136 // We filled nothing, or there was an error.
137 return;
138 }
139
140 pCombo->SetCurSel(0);
141
142 ++i;
143 while (s_allControls[i][1]) {
144 pText = GetDlgItem(s_allControls[i][0]);
145 if (!pText) {
146 continue;
147 }
148
149 pCombo = (CComboBox*) GetDlgItem(s_allControls[i][1]);
150 if (!pCombo) {
151 continue;
152 }
153
154 pText->ShowWindow(SW_HIDE);
155 pCombo->ShowWindow(SW_HIDE);
156 ++i;
157 }
158}

160void TeamGeneric::_scriptsToDict()
161{
162 if (!m_teamDict) {
163 return;
164 }
165
166 CWnd *pText = NULL;
167 CComboBox *pCombo = NULL;
168
169 int scriptNum = 0;
170
171 int i = 0;
172 while (s_allControls[i][1]) {
173
174 pText = GetDlgItem(s_allControls[i][0]);
175 if (!pText) {
176 continue;
177 }
178
179 pCombo = (CComboBox*) GetDlgItem(s_allControls[i][1]);
180 if (!pCombo) {
181 continue;
182 }
183
184 // i should always be incremented, so just do it here.
185 ++i;
186
187 AsciiString keyName;
188 keyName.format("%s%d", TheNameKeyGenerator->keyToName(TheKey_teamGenericScriptHook).str(), scriptNum);
189
190 int curSel = pCombo->GetCurSel();
191 if (curSel == CB_ERR || curSel == 0) {
192 if (m_teamDict->known(NAMEKEY(keyName), Dict::DICT_ASCIISTRING)) {
193 // remove it if we know it.
194 m_teamDict->remove(NAMEKEY(keyName));
195 }
196
197 continue;
198 }
199
200 CString cstr;
201 pCombo->GetLBText(curSel, cstr);
202
203 AsciiString scriptString = cstr;
204 m_teamDict->setAsciiString(NAMEKEY(keyName), scriptString);
205 ++scriptNum;
206 }
207
208 for ( ; s_allControls[scriptNum][1]; ++scriptNum ) {
209 AsciiString keyName;
210 keyName.format("%s%d", TheNameKeyGenerator->keyToName(TheKey_teamGenericScriptHook).str(), scriptNum);
211
212 if (m_teamDict->known(NAMEKEY(keyName), Dict::DICT_ASCIISTRING)) {
213 m_teamDict->remove(NAMEKEY(keyName));
214 }
215 }
216}
