102void DisconnectManager::update(ConnectionManager *conMgr) {
103 if (m_lastFrameTime == -1) {
104 m_lastFrameTime = timeGetTime();
105 }
106
107 // The game logic stalls on the frame we are currently waiting for commands on,
108 // so we have to check for the current logic frame being one higher than
109 // the last one we had the commands ready for.
110 if (TheGameLogic->getFrame() == m_lastFrame) {
111 time_t curTime = timeGetTime();
112 if ((curTime - m_lastFrameTime) > TheGlobalData->m_networkDisconnectTime) {
113 if (m_disconnectState == DISCONNECTSTATETYPE_SCREENOFF) {
114 turnOnScreen(conMgr);
115 }
116 sendKeepAlive(conMgr);
117 }
118 } else {
119 nextFrame(TheGameLogic->getFrame(), conMgr);
120 }
121
122 if (m_disconnectState != DISCONNECTSTATETYPE_SCREENOFF) {
123 updateDisconnectStatus(conMgr);
124
125 // check to see if we need to send pings
126 if (m_pingFrame < TheGameLogic->getFrame())
127 {
128 time_t curTime = timeGetTime();
129 if ((curTime - m_lastFrameTime) > 10000)
130 {
131 m_pingFrame = TheGameLogic->getFrame();
132 m_pingsSent = 0;
133 m_pingsRecieved = 0;
134
135 // Send the pings
136 if (ThePinger)
137 {
138 //use next ping server
139 static int serverIndex = 0;
140 serverIndex++;
141 if( serverIndex >= TheGameSpyConfig->getPingServers().size() )
142 serverIndex = 0; //wrap back to first ping server
143
144 std::list<AsciiString>::iterator it = TheGameSpyConfig->getPingServers().begin();
145 for( int i = 0; i < serverIndex; i++ )
146 it++;
147
148 PingRequest req;
149 req.hostname = it->str();
150 req.repetitions = 5;
151 req.timeout = 2000;
152 m_pingsSent = req.repetitions;
153 ThePinger->addRequest(req);
154 DEBUG_LOG(("DisconnectManager::update() - requesting %d pings of %d from %s\n",
155 req.repetitions, req.timeout, req.hostname.c_str()));
156 }
157 }
158 }
159
160 // update the ping thread, tracking pings if we are on the same frame
161 if (ThePinger)
162 {
163 PingResponse resp;
164 while (ThePinger->getResponse(resp))
165 {
166 if (m_pingFrame != TheGameLogic->getFrame())
167 {
168 // wrong frame - we're not pinging yet
169 DEBUG_LOG(("DisconnectManager::update() - discarding ping of %d from %s (%d reps)\n",
170 resp.avgPing, resp.hostname.c_str(), resp.repetitions));
171 }
172 else
173 {
174 // right frame
175 DEBUG_LOG(("DisconnectManager::update() - keeping ping of %d from %s (%d reps)\n",
176 resp.avgPing, resp.hostname.c_str(), resp.repetitions));
177 if (resp.avgPing < 2000)
178 {
179 m_pingsRecieved += resp.repetitions;
180 }
181 }
182 }
183 }
184 }

203void DisconnectManager::updateDisconnectStatus(ConnectionManager *conMgr) {
204 for (Int i = 0; i < MAX_SLOTS; ++i) {
205 if (conMgr->isPlayerConnected(i)) {
206 Int slot = translatedSlotPosition(i, conMgr->getLocalPlayerID());
207 if (slot != -1) {
208 time_t curTime = timeGetTime();
209 time_t newTime = TheGlobalData->m_networkPlayerTimeoutTime - (curTime - m_playerTimeouts[slot]);
210
211// if someone is more than 2/3 timed out, lets get our frame numbers sync'd up. Also if someone is voted out
212// lets do the same thing.
213
214 if (m_haveNotifiedOtherPlayersOfCurrentFrame == FALSE) {
215 if ((newTime < TheGlobalData->m_networkPlayerTimeoutTime / 3) || (isPlayerVotedOut(slot, conMgr) == TRUE)) {
216 TheNetwork->notifyOthersOfCurrentFrame();
217 m_haveNotifiedOtherPlayersOfCurrentFrame = TRUE;
218 }
219
220 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - curTime = %d, m_timeOfDisconnectScreenOn = %d, curTime - m_timeOfDisconnectScreenOn = %d\n", curTime, m_timeOfDisconnectScreenOn, curTime - m_timeOfDisconnectScreenOn));
221
222 if (m_timeOfDisconnectScreenOn != 0) {
223 if ((curTime - m_timeOfDisconnectScreenOn) > TheGlobalData->m_networkDisconnectScreenNotifyTime) {
224 TheNetwork->notifyOthersOfCurrentFrame();
225 m_haveNotifiedOtherPlayersOfCurrentFrame = TRUE;
226 }
227 }
228 }
229
230 if ((newTime < 0) || (isPlayerVotedOut(slot, conMgr) == TRUE)) {
231 newTime = 0;
232 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - player %d(translated slot %d) has been voted out or timed out\n", i, slot));
233 if (allOnSameFrame(conMgr) == TRUE) {
234 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - all on same frame\n"));
235 if (isLocalPlayerNextPacketRouter(conMgr) == TRUE) {
236 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - local player is next packet router\n"));
237 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - about to do the disconnect procedure for player %d\n", i));
238 sendDisconnectCommand(i, conMgr);
239 disconnectPlayer(i, conMgr);
240 sendPlayerDestruct(i, conMgr);
241 } else {
242 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - local player is not the next packet router\n"));
243 }
244 } else {
245 DEBUG_LOG(("DisconnectManager::updateDisconnectStatus - not all on same frame\n"));
246 }
247 }
248 TheDisconnectMenu->setPlayerTimeoutTime(slot, newTime);
249 }
250 }
251 }

270void DisconnectManager::processDisconnectCommand(NetCommandRef *ref, ConnectionManager *conMgr) {
271 NetCommandMsg *msg = ref->getCommand();
272 if (msg->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTKEEPALIVE) {
273 processDisconnectKeepAlive(msg, conMgr);
274 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTPLAYER) {
275 processDisconnectPlayer(msg, conMgr);
276 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_PACKETROUTERQUERY) {
277 processPacketRouterQuery(msg, conMgr);
278 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_PACKETROUTERACK) {
279 processPacketRouterAck(msg, conMgr);
280 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTVOTE) {
281 processDisconnectVote(msg, conMgr);
282 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTFRAME) {
283 processDisconnectFrame(msg, conMgr);
284 } else if (msg->getNetCommandType() == NETCOMMANDTYPE_DISCONNECTSCREENOFF) {
285 processDisconnectScreenOff(msg, conMgr);
286 }

304void DisconnectManager::processPacketRouterQuery(NetCommandMsg *msg, ConnectionManager *conMgr) {
305 NetPacketRouterQueryCommandMsg *cmdMsg = (NetPacketRouterQueryCommandMsg *)msg;
306 DEBUG_LOG(("DisconnectManager::processPacketRouterQuery - got a packet router query command from player %d\n", msg->getPlayerID()));
307
308 if (conMgr->getPacketRouterSlot() == conMgr->getLocalPlayerID()) {
309 NetPacketRouterAckCommandMsg *ackmsg = newInstance(NetPacketRouterAckCommandMsg);
310 ackmsg->setPlayerID(conMgr->getLocalPlayerID());
311 if (DoesCommandRequireACommandID(ackmsg->getNetCommandType()) == TRUE) {
312 ackmsg->setID(GenerateNextCommandID());
313 }
314 DEBUG_LOG(("DisconnectManager::processPacketRouterQuery - We are the new packet router, responding with an packet router ack. Local player is %d\n", ackmsg->getPlayerID()));
315 conMgr->sendLocalCommandDirect(ackmsg, 1 << cmdMsg->getPlayerID());
316 ackmsg->detach();
317 } else {
318 DEBUG_LOG(("DisconnectManager::processPacketRouterQuery - We are NOT the new packet router, these are not the droids you're looking for.\n"));
319 }

322void DisconnectManager::processPacketRouterAck(NetCommandMsg *msg, ConnectionManager *conMgr) {
323 NetPacketRouterAckCommandMsg *cmdMsg = (NetPacketRouterAckCommandMsg *)msg;
324 DEBUG_LOG(("DisconnectManager::processPacketRouterAck - got packet router ack command from player %d\n", msg->getPlayerID()));
325
326 if (conMgr->getPacketRouterSlot() == cmdMsg->getPlayerID()) {
327 DEBUG_LOG(("DisconnectManager::processPacketRouterAck - packet router command is from who it should be.\n"));
328 resetPacketRouterTimeout();
329 Int currentPacketRouterSlot = conMgr->getPacketRouterSlot();
330 Int currentPacketRouterIndex = 0;
331 while ((currentPacketRouterSlot != conMgr->getPacketRouterFallbackSlot(currentPacketRouterIndex)) && (currentPacketRouterIndex < MAX_SLOTS)) {
332 ++currentPacketRouterIndex;
333 }
334 DEBUG_ASSERTCRASH((currentPacketRouterIndex < MAX_SLOTS), ("Invalid packet router index"));
335
336 DEBUG_LOG(("DisconnectManager::processPacketRouterAck - New packet router confirmed, resending pending commands\n"));
337 conMgr->resendPendingCommands();
338 m_currentPacketRouterIndex = currentPacketRouterIndex;
339 DEBUG_LOG(("DisconnectManager::processPacketRouterAck - Setting disconnect state to screen on.\n"));
340 m_disconnectState = DISCONNECTSTATETYPE_SCREENON;
341 }

357void DisconnectManager::processDisconnectFrame(NetCommandMsg *msg, ConnectionManager *conMgr) {
358 NetDisconnectFrameCommandMsg *cmdMsg = (NetDisconnectFrameCommandMsg *)msg;
359 UnsignedInt playerID = cmdMsg->getPlayerID();
360 if (m_disconnectFrames[playerID] >= cmdMsg->getDisconnectFrame()) {
361 // this message isn't valid, we have a disconnect frame that is later than this already.
362 return;
363 }
364
365 if (m_disconnectFramesReceived[playerID] == TRUE) {
366 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - Got two disconnect frames without an intervening disconnect screen off command from player %d. Frames are %d and %d\n", playerID, m_disconnectFrames[playerID], cmdMsg->getDisconnectFrame()));
367 }
368
369 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - about to call resetPlayersVotes for player %d\n", playerID));
370 resetPlayersVotes(playerID, cmdMsg->getDisconnectFrame()-1, conMgr);
371
372 m_disconnectFrames[playerID] = cmdMsg->getDisconnectFrame();
373 m_disconnectFramesReceived[playerID] = TRUE;
374 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - Got a disconnect frame for player %d, frame = %d, local player is %d, local disconnect frame = %d, command id = %d\n", cmdMsg->getPlayerID(), cmdMsg->getDisconnectFrame(), conMgr->getLocalPlayerID(), m_disconnectFrames[conMgr->getLocalPlayerID()], cmdMsg->getID()));
375
376 if (playerID == conMgr->getLocalPlayerID()) {
377 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - player %d is the local player\n", playerID));
378 // we just got the message from the local player, check to see if we need to send
379 // commands to anyone we already have heard from.
380 for (Int i = 0; i < MAX_SLOTS; ++i) {
381 if (i != playerID) {
382 Int transSlot = translatedSlotPosition(i, conMgr->getLocalPlayerID());
383 if (isPlayerInGame(transSlot, conMgr) == TRUE) {
384 if ((m_disconnectFrames[i] < m_disconnectFrames[playerID]) && (m_disconnectFramesReceived[i] == TRUE)) {
385 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - I have more frames than player %d, my frame = %d, their frame = %d\n", i, m_disconnectFrames[conMgr->getLocalPlayerID()], m_disconnectFrames[i]));
386 conMgr->sendFrameDataToPlayer(i, m_disconnectFrames[i]);
387 }
388 }
389 }
390 }
391 } else if ((m_disconnectFrames[playerID] < m_disconnectFrames[conMgr->getLocalPlayerID()]) && (m_disconnectFramesReceived[playerID] == TRUE)) {
392 DEBUG_LOG(("DisconnectManager::processDisconnectFrame - I have more frames than player %d, my frame = %d, their frame = %d\n", playerID, m_disconnectFrames[conMgr->getLocalPlayerID()], m_disconnectFrames[playerID]));
393 conMgr->sendFrameDataToPlayer(playerID, m_disconnectFrames[playerID]);
394 }
395}

397void DisconnectManager::processDisconnectScreenOff(NetCommandMsg *msg, ConnectionManager *conMgr) {
398 NetDisconnectScreenOffCommandMsg *cmdMsg = (NetDisconnectScreenOffCommandMsg *)msg;
399 UnsignedInt playerID = cmdMsg->getPlayerID();
400
401 DEBUG_LOG(("DisconnectManager::processDisconnectScreenOff - got a screen off command from player %d for frame %d\n", cmdMsg->getPlayerID(), cmdMsg->getNewFrame()));
402
403 if ((playerID < 0) || (playerID >= MAX_SLOTS)) {
404 return;
405 }
406
407 UnsignedInt newFrame = cmdMsg->getNewFrame();
408 if (newFrame >= m_disconnectFrames[playerID]) {
409 DEBUG_LOG(("DisconnectManager::processDisconnectScreenOff - resetting the disconnect screen status for player %d\n", playerID));
410 m_disconnectFramesReceived[playerID] = FALSE;
411 m_disconnectFrames[playerID] = newFrame; // just in case we get packets out of order and the disconnect screen off message gets here before the disconnect frame message.
412
413 DEBUG_LOG(("DisconnectManager::processDisconnectScreenOff - about to call resetPlayersVotes for player %d\n", playerID));
414 resetPlayersVotes(playerID, cmdMsg->getNewFrame(), conMgr);
415 }

546void DisconnectManager::disconnectPlayer(Int slot, ConnectionManager *conMgr) {
547 DEBUG_LOG(("DisconnectManager::disconnectPlayer - Disconnecting slot number %d on frame %d\n", slot, TheGameLogic->getFrame()));
548 DEBUG_ASSERTCRASH((slot >= 0) && (slot < MAX_SLOTS), ("Attempting to disconnect an invalid slot number"));
549 if ((slot < 0) || (slot >= (MAX_SLOTS))) {
550 return;
551 }
552
553 if (TheGameInfo)
554 {
555 GameSlot *gSlot = TheGameInfo->getSlot( slot );
556 if (gSlot)
557 {
558 gSlot->markAsDisconnected();
559 }
560 }
561
562 Int transSlot = translatedSlotPosition(slot, conMgr->getLocalPlayerID());
563
564 if (transSlot != -1) {
565 // Ignore any disconnect commands that tell us to disconnect ourselves.
566
567 // Get the disconnecting player off the disconnect window.
568 UnicodeString uname = conMgr->getPlayerName(slot);
569 TheRecorder->logPlayerDisconnect(uname, slot);
570 TheDisconnectMenu->removePlayer(transSlot, uname);
571
572 PlayerLeaveCode retcode = conMgr->disconnectPlayer(slot);
573 DEBUG_ASSERTCRASH((retcode != PLAYERLEAVECODE_UNKNOWN), ("Invalid player leave code"));
574
575 if (retcode == PLAYERLEAVECODE_PACKETROUTER) {
576 DEBUG_LOG(("DisconnectManager::disconnectPlayer - disconnecting player was packet router.\n"));
577
578 conMgr->resendPendingCommands();
579 }
580 }

583void DisconnectManager::sendDisconnectCommand(Int slot, ConnectionManager *conMgr) {
584 DEBUG_LOG(("DisconnectManager::sendDisconnectCommand - Sending disconnect command for slot number %d\n", slot));
585 DEBUG_ASSERTCRASH((slot >= 0) && (slot < MAX_SLOTS), ("Attempting to send a disconnect command for an invalid slot number"));
586 if ((slot < 0) || (slot >= (MAX_SLOTS))) {
587 return;
588 }
589

645Bool DisconnectManager::allOnSameFrame(ConnectionManager *conMgr) {
646 Bool retval = TRUE;
647 for (Int i = 0; (i < MAX_SLOTS) && (retval == TRUE); ++i) {
648 Int transSlot = translatedSlotPosition(i, conMgr->getLocalPlayerID());
649 if (transSlot == -1) {
650 continue;
651 }
652 if ((conMgr->isPlayerConnected(i) == TRUE) && (isPlayerInGame(transSlot, conMgr) == TRUE)) {
653 // ok, i is someone who is in the game and hasn't timed out yet or been voted out.
654 if (m_disconnectFramesReceived[i] == FALSE) {
655 // we don't know what frame they are on yet.
656 retval = FALSE;
657 }
658 if ((m_disconnectFramesReceived[i] == TRUE) && (m_disconnectFrames[conMgr->getLocalPlayerID()] != m_disconnectFrames[i])) {
659 // We know their frame, but they aren't on the same frame as us.
660 retval = FALSE;
661 }
662 }
663 }

667Bool DisconnectManager::isLocalPlayerNextPacketRouter(ConnectionManager *conMgr) {
668 UnsignedInt localSlot = conMgr->getLocalPlayerID();
669 UnsignedInt packetRouterSlot = conMgr->getPacketRouterSlot();
670 Int transSlot = translatedSlotPosition(packetRouterSlot, localSlot);
671
672 // stop when we have found a packet router that is connected
673 while ((transSlot != -1) && (isPlayerInGame(transSlot, conMgr) == FALSE)) {
674 packetRouterSlot = conMgr->getNextPacketRouterSlot(packetRouterSlot);
675 if ((packetRouterSlot >= MAX_SLOTS) || (packetRouterSlot < 0)) {
676 // don't know who the next packet router is going to be,
677 // so this game is not going to go anywhere anymore.
678 DEBUG_CRASH(("no more players left to be the packet router, this shouldn't happen."));
679 return FALSE;
680 }
681 transSlot = translatedSlotPosition(packetRouterSlot, localSlot);
682 }
683
684 if (packetRouterSlot == localSlot) {
685 return TRUE;
686 }
687
688 return FALSE;
