120void Mouse::moveMouse( Int x, Int y, Int relOrAbs )
121{
122
123 if( relOrAbs == MOUSE_MOVE_RELATIVE )
124 {
125 m_currMouse.pos.x += x;
126 m_currMouse.pos.y += y;
127 }
128 else
129 {
130 m_currMouse.pos.x = x;
131 m_currMouse.pos.y = y;
132 }
133
134 if( m_currMouse.pos.x > m_maxX )
135 m_currMouse.pos.x = m_maxX;
136 else if( m_currMouse.pos.x < m_minX )
137 m_currMouse.pos.x = m_minX;
138
139 if( m_currMouse.pos.y > m_maxY )
140 m_currMouse.pos.y = m_maxY;
141 else if( m_currMouse.pos.y < m_minY )
142 m_currMouse.pos.y = m_minY;
143
144} // end moveMouse

149void Mouse::updateMouseData( )
150{
151 static Bool busy = FALSE;
152 Int index = 0;
153 UnsignedByte result;
154
155 // prevent reentrancy in the event we make this mouse multi-threaded
156 if( busy == FALSE )
157 {
158
159 busy = TRUE;
160
161 // Get latest mouse events from DirectX
162 do
163 {
164 do
165 {
166 result = getMouseEvent( &m_mouseEvents[ index ], TRUE );
167 }
168 while( result == MOUSE_LOST );
169 index++;
170 }
171 while( (result != MOUSE_NONE) &&
172 (index < sizeof( m_mouseEvents ) / sizeof( MouseIO )) );
173
174 busy = FALSE;
175
176 } // end if
177
178 if( index > 0 )
179 m_eventsThisFrame = index - 1;
180 else
181 m_eventsThisFrame = 0;
182
183 if( index != 0 )
184 m_deadInputFrame = m_inputFrame;
185
186} // end updateMouseData

191void Mouse::processMouseEvent( Int index )
192{
193 Int movementType;
194
195 m_currMouse.leftEvent = MOUSE_EVENT_NONE;
196 m_currMouse.rightEvent = MOUSE_EVENT_NONE;
197 m_currMouse.middleEvent = MOUSE_EVENT_NONE;
198 m_currMouse.wheelPos = 0;
199
200 // what type of movement commands are we setup for
201 if( m_inputMovesAbsolute == TRUE )
202 movementType = MOUSE_MOVE_ABSOLUTE;
203 else
204 movementType = MOUSE_MOVE_RELATIVE;
205
206 // set the time of this event to the correct time
207 m_currMouse.time = m_mouseEvents[ index ].time;
208
209 if( index == 0 )
210 checkForDrag();
211
212 // add Mouse Position Changes to Master Position
213 moveMouse( m_mouseEvents[ index ].pos.x,
214 m_mouseEvents[ index ].pos.y,
215 movementType );
216
217 // Cumulate Wheel Adjustments
218 m_currMouse.wheelPos += m_mouseEvents[ index ].wheelPos;
219
220 // Check Left Mouse State
221 if( m_mouseEvents[ index ].leftFrame )
222 {
223 if( m_currMouse.leftState != m_mouseEvents[ index ].leftState )
224 {
225 // State Change
226 if( m_mouseEvents[ index ].leftState == MBS_Down )
227 {
228 // Mouse Down
229 m_currMouse.leftEvent = GWM_LEFT_DOWN;
230 m_currMouse.leftState = MBS_Down;
231 m_currMouse.leftFrame = m_inputFrame;
232 }
233 else if ( m_mouseEvents[ index ].leftState == MBS_DoubleClick )
234 {
235 // Mouse Double Click
236 m_currMouse.leftEvent = GWM_LEFT_DOUBLE_CLICK;
237 m_currMouse.leftState = MBS_DoubleClick;
238 m_currMouse.leftFrame = m_inputFrame;
239 }
240 else
241 {
242 // Mouse Up
243 m_currMouse.leftEvent = GWM_LEFT_UP;
244 m_currMouse.leftState = MBS_Up;
245 m_currMouse.leftFrame = m_inputFrame;
246 }
247 }
248 }
249 else if( m_currMouse.leftState != MBS_Up &&
250 ( (m_prevMouse.leftEvent == GWM_LEFT_DOWN) ||
251 (m_prevMouse.leftEvent == GWM_LEFT_DRAG) ) )
252 {
253 m_currMouse.leftEvent = GWM_LEFT_DRAG;
254 }
255
256 // Check Right Mouse State
257 if( m_mouseEvents[ index ].rightFrame )
258 {
259 if( m_currMouse.rightState != m_mouseEvents[ index ].rightState )
260 {
261 // State Change
262 if( m_mouseEvents[ index ].rightState == MBS_Down )
263 {
264 // Mouse Down
265 m_currMouse.rightEvent = GWM_RIGHT_DOWN;
266 m_currMouse.rightState = MBS_Down;
267 m_currMouse.rightFrame = m_inputFrame;
268 }
269 else if( m_mouseEvents[ index ].rightState == MBS_DoubleClick )
270 {
271 // Mouse Double Click
272 m_currMouse.rightEvent = GWM_RIGHT_DOUBLE_CLICK;
273 m_currMouse.rightState = MBS_DoubleClick;
274 m_currMouse.rightFrame = m_inputFrame;
275 }
276 else
277 {
278 // Mouse Up
279 m_currMouse.rightEvent = GWM_RIGHT_UP;
280 m_currMouse.rightState = MBS_Up;
281 m_currMouse.rightFrame = m_inputFrame;
282 }
283 }
284 }
285 else if( m_currMouse.rightState != MBS_Up &&
286 ( (m_prevMouse.rightEvent == GWM_RIGHT_DOWN) ||
287 (m_prevMouse.rightEvent == GWM_RIGHT_DRAG) ) )
288 {
289 m_currMouse.rightEvent = GWM_RIGHT_DRAG;
290 }
291
292 // Check Middle Mouse State
293 if( m_mouseEvents[ index ].middleFrame )
294 {
295 if( m_currMouse.middleState != m_mouseEvents[index].middleState )
296 {
297 // State Change
298 if( m_mouseEvents[index].middleState == MBS_Down )
299 {
300 m_currMouse.middleEvent = GWM_MIDDLE_DOWN;
301 m_currMouse.middleState = MBS_Down;
302 m_currMouse.middleFrame = m_inputFrame;
303 }
304 else if( m_mouseEvents[index].middleState == MBS_DoubleClick )
305 {
306 m_currMouse.middleEvent = GWM_MIDDLE_DOUBLE_CLICK;
307 m_currMouse.middleState = MBS_DoubleClick;
308 m_currMouse.middleFrame = m_inputFrame;
309 }
310 else
311 {
312 // Mouse Up
313 m_currMouse.middleEvent = GWM_MIDDLE_UP;
314 m_currMouse.middleState = MBS_Up;
315 m_currMouse.middleFrame = m_inputFrame;
316 }
317 }
318 }
319 else if( m_currMouse.middleState != MBS_Up &&
320 ( (m_prevMouse.middleEvent == GWM_MIDDLE_DOWN) ||
321 (m_prevMouse.middleEvent == GWM_MIDDLE_DRAG) ) )
322 {
323 m_currMouse.middleEvent = GWM_MIDDLE_DRAG;
324 }
325
326 m_currMouse.deltaPos.x = m_currMouse.pos.x - m_prevMouse.pos.x;
327 m_currMouse.deltaPos.y = m_currMouse.pos.y - m_prevMouse.pos.y;
328// DEBUG_LOG(("Mouse dx %d, dy %d, index %d, frame %d\n", m_currMouse.deltaPos.x, m_currMouse.deltaPos.y, index, m_inputFrame));
329// // check if mouse is still and flag tooltip
330// if( ((dx*dx) + (dy*dy)) < CURSOR_MOVE_TOL_SQ )

333// //m_stillTime++;
334// Int delay = m_tooltipDelayTime;

360
361} // end processMouseEvent

366void Mouse::checkForDrag( void )
367{
368
369 if( m_currMouse.leftState &&
370 ( (m_prevMouse.leftEvent == GWM_LEFT_DOWN) ||
371 (m_prevMouse.leftEvent == GWM_LEFT_DRAG) ) )
372 {
373 m_currMouse.leftEvent = GWM_LEFT_DRAG;
374 }
375
376 if( m_currMouse.rightState &&
377 ( (m_prevMouse.rightEvent == GWM_RIGHT_DOWN) ||
378 (m_prevMouse.rightEvent == GWM_RIGHT_DRAG) ) )
379 {
380 m_currMouse.rightEvent = GWM_RIGHT_DRAG;
381 }
382
383 if( m_currMouse.middleState &&
384 ( (m_prevMouse.middleEvent == GWM_MIDDLE_DOWN) ||
385 (m_prevMouse.middleEvent == GWM_MIDDLE_DRAG) ) )
386 {
387 m_currMouse.middleEvent = GWM_MIDDLE_DRAG;

390} // end checkForDrag

396Bool Mouse::isClick(const ICoord2D *anchor, const ICoord2D *dest, UnsignedInt previousMouseClick, UnsignedInt currentMouseClick)
397{
398 ICoord2D delta;
399 delta.x = anchor->x - dest->x;
400 delta.y = anchor->y - dest->y;
401
402
403 // if the mouse hasn't moved further than the tolerance distance
404 // or the click took less than the tolerance duration
405 if ( abs(delta.x) > m_dragTolerance
406 || abs(delta.y) > m_dragTolerance
407 || currentMouseClick - previousMouseClick > m_dragToleranceMS)
408 {
409 return FALSE;
410 }
411 return TRUE;
412}

660//-------------------------------------------------------------------------------------------------
665//-------------------------------------------------------------------------------------------------
666void Mouse::createStreamMessages( void )
667{
668
669 // santiy
670 if( TheMessageStream == NULL )
671 return; // no place to put messages
672
673 GameMessage *msg = NULL;
674 UnsignedInt now = timeGetTime();
675
676 // basic position messages are always created
677 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_MOUSE_POSITION );
678 msg->appendPixelArgument( m_currMouse.pos );
679 msg->appendIntegerArgument( TheKeyboard->getModifierFlags() );
680
681 Int delay = m_tooltipDelayTime;
682 if(m_tooltipDelay >= 0 )
683 delay = m_tooltipDelay;
684 if( TheGlobalData->m_scriptDebug )
685 {
686 //No delay while scriptdebugging!
687 delay = 0;
688 }
689
690 if( now - m_stillTime >= delay )
691 {
692 if (!m_displayTooltip)
693 {
694 m_highlightPos = 0;
695 m_highlightUpdateStart = timeGetTime();
696 }
697
698 // display tooltip for current window
699 m_displayTooltip = TRUE;
700 }
701 else
702 {
703 //DEBUG_LOG(("%d %d %d %d\n", TheGameClient->getFrame(), delay, now, m_stillTime));
704 m_displayTooltip = FALSE;
705 }
706
707 for (Int i = 0; i < m_eventsThisFrame; ++i)
708 {
709 processMouseEvent(i);
710 if (m_currMouse.deltaPos.x || m_currMouse.deltaPos.y)
711 m_stillTime = now;
712
713 // button messages
714 msg = NULL;
715 switch( m_currMouse.leftEvent )
716 {
717
718 case GWM_LEFT_DOWN:
719 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_DOWN );
720 msg->appendPixelArgument( m_currMouse.pos );
721 msg->appendIntegerArgument( TheKeyboard->getModifierFlags() );
722 msg->appendIntegerArgument( m_currMouse.time );
723 break;
724
725 case GWM_LEFT_DOUBLE_CLICK:
726 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_MOUSE_LEFT_DOUBLE_CLICK );
727 msg->appendPixelArgument( m_currMouse.pos );
728 msg->appendIntegerArgument( TheKeyboard->getModifierFlags() );
729 msg->appendIntegerArgument( m_currMouse.time );
730 break;
731
732 case GWM_LEFT_UP:
733 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_UP );
734 msg->appendPixelArgument( m_currMouse.pos );
735 msg->appendIntegerArgument( TheKeyboard->getModifierFlags() );
736 msg->appendIntegerArgument( m_currMouse.time );
737 break;
738
739 case GWM_LEFT_DRAG:
