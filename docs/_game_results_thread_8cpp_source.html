210void GameResultsThreadClass::Thread_Function()
211{
212 try {
213 _set_se_translator( DumpExceptionInfo ); // Hook that allows stack trace.
214 GameResultsRequest req;
215
216 WSADATA wsaData;
217
218 // Fire up winsock (prob already done, but doesn't matter)
219 WORD wVersionRequested = MAKEWORD(1, 1);
220 WSAStartup( wVersionRequested, &wsaData );
221
222 while ( running )
223 {
224 // deal with requests
225 if (TheGameResultsQueue && TheGameResultsQueue->getRequest(req))
226 {
227 // resolve the hostname
228 const char *hostnameBuffer = req.hostname.c_str();
229 UnsignedInt IP = 0xFFFFFFFF;
230 if (isdigit(hostnameBuffer[0]))
231 {
232 IP = inet_addr(hostnameBuffer);
233 in_addr hostNode;
234 hostNode.s_addr = IP;
235 DEBUG_LOG(("sending game results to %s - IP = %s\n", hostnameBuffer, inet_ntoa(hostNode) ));
236 }
237 else
238 {
239 HOSTENT *hostStruct;
240 in_addr *hostNode;
241 hostStruct = gethostbyname(hostnameBuffer);
242 if (hostStruct == NULL)
243 {
244 DEBUG_LOG(("sending game results to %s - host lookup failed\n", hostnameBuffer));
245
246 // Even though this failed to resolve IP, still need to send a
247 // callback.
248 IP = 0xFFFFFFFF; // flag for IP resolve failed
249 }
250 hostNode = (in_addr *) hostStruct->h_addr;
251 IP = hostNode->s_addr;
252 DEBUG_LOG(("sending game results to %s IP = %s\n", hostnameBuffer, inet_ntoa(*hostNode) ));
253 }
254
255 int result = sendGameResults( IP, req.port, req.results );
256 GameResultsResponse resp;
257 resp.hostname = req.hostname;
258 resp.port = req.port;
259 resp.sentOk = (result == req.results.length());
260
261 }
262
263 // end our timeslice
264 Switch_Thread();
265 }
266
267 WSACleanup();
268 } catch ( ... ) {
269 DEBUG_CRASH(("Exception in results thread!"));
270 }
271}

344Int GameResultsThreadClass::sendGameResults( UnsignedInt IP, UnsignedShort port, const std::string& results )
345{
346 int error = 0;
347
348 // create the socket
349 Int sock = socket( AF_INET, SOCK_STREAM, 0 );
350 if (sock < 0)
351 {
352 DEBUG_LOG(("GameResultsThreadClass::sendGameResults() - socket() returned %d(%s)\n", sock, getWSAErrorString(sock)));
353 return sock;
354 }
355
356 // fill in address info
357 struct sockaddr_in sockAddr;
358 memset( &sockAddr, 0, sizeof( sockAddr ) );
359 sockAddr.sin_family = AF_INET;
360 sockAddr.sin_addr.s_addr = IP;
361 sockAddr.sin_port = htons(port);
362
363 // Start the connection process....
364 if( connect( sock, (struct sockaddr *)&sockAddr, sizeof( sockAddr ) ) == -1 )
365 {
366 error = WSAGetLastError();
367 DEBUG_LOG(("GameResultsThreadClass::sendGameResults() - connect() returned %d(%s)\n", error, getWSAErrorString(error)));
368 if( ( error == WSAEWOULDBLOCK ) || ( error == WSAEINVAL ) || ( error == WSAEALREADY ) )
369 {
370 return( -1 );
371 }
372
373 if( error != WSAEISCONN )
374 {
375 closesocket( sock );
376 return( -1 );
377 }
378 }
379
380 if (send( sock, results.c_str(), results.length(), 0 ) == SOCKET_ERROR)
381 {
382 error = WSAGetLastError();
383 DEBUG_LOG(("GameResultsThreadClass::sendGameResults() - send() returned %d(%s)\n", error, getWSAErrorString(error)));
384 closesocket(sock);
385 return WSAGetLastError();
386 }
387
388 closesocket(sock);
389
390 return results.length();
391}
