72static inline float PcToFrac(int pc)
73{
74 return (float)pc/100.0f;
75}
76
77static inline int FracToPc(float f)
78{
79 if (f<0.0) return (int)(100.0f*f - .5f);
80 else return (int) (100.0f*f + .5f);
81}
