64// Pick a scorchmark.
65MapObject *ScorchTool::pickScorch(Coord3D loc){
66 // Tight check first.
67 MapObject *pObj;
68 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
69 if (!pObj->isScorch()) {
70 continue;
71 }
72 Coord3D cloc = *pObj->getLocation();
73 // Check and see if we are within 1/2 cell size of the center.
74 Coord3D cpt = loc;
75 cpt.x -= cloc.x;
76 cpt.y -= cloc.y;
77 cpt.z = 0;
78 if (cpt.length() < 0.5f*MAP_XY_FACTOR) {
79 return pObj;
80 }
81 }
82 // Loose check
83 for (pObj = MapObject::getFirstMapObject(); pObj; pObj = pObj->getNext()) {
84 if (!pObj->isScorch()) {
85 continue;
86 }
87 Coord3D cloc = *pObj->getLocation();
88 // Check and see if we are within 1 & 1/2 cell size of the center.
89 Coord3D cpt = loc;
90 cpt.x -= cloc.x;
91 cpt.y -= cloc.y;
92 cpt.z = 0;
93 if (cpt.length() < 1.5f*MAP_XY_FACTOR) {
94 return pObj;
95 }
96 }
97 return NULL;
98}

102void ScorchTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
103{
104 if (m != TRACK_L) return;
105 Coord3D docPt;
106 pView->viewToDocCoords(viewPt, &docPt);
107 MapObject *pObj = pickScorch(docPt);
108 if (pObj) {
109 pObj->setSelected(true);
110 docPt = *pObj->getLocation();
111 ScorchOptions::update();
112 } else {
113 pView->snapPoint(&docPt);
114 MapObject *pNew = newInstance(MapObject)(docPt, AsciiString("Scorch"), 0, 0, NULL, NULL );
115 pNew->getProperties()->setAsciiString(TheKey_originalOwner, NEUTRAL_TEAM_INTERNAL_STR);
116 pNew->setSelected(true);
117 pNew->setIsScorch();
118 pNew->getProperties()->setReal(TheKey_objectRadius, ScorchOptions::getScorchSize());
119 pNew->getProperties()->setInt(TheKey_scorchType, ScorchOptions::getScorchType());
120 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew);
121 pDoc->AddAndDoUndoable(pUndo);
122 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
123 pNew = NULL; // undoable owns it now.
124 ScorchOptions::update();
125 }
126 m_mouseDownPt = docPt;
127}
