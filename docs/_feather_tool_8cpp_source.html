140void FeatherTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
141{
142 Coord3D cpt;
143 pView->viewToDocCoords(viewPt, &cpt);
144 DrawObject::setFeedbackPos(cpt);
145 if (m != TRACK_L) return;
146
147 int brushWidth = m_feather;
148
149 CPoint ndx;
150 getCenterIndex(&cpt, m_feather, &ndx, pDoc);
151
152 //if (m_prevXIndex == ndx.x && m_prevYIndex == ndx.y) return;
153
154 m_prevXIndex = ndx.x;
155 m_prevYIndex = ndx.y;
156
157 int sub = brushWidth/2;
158 int add = brushWidth-sub;
159
160 // round brush
161 Int i, j;
162 Bool redoRate = false;
163
164 for (i= ndx.x-sub; i< ndx.x+add; i++) {
165 if (i<0 || i>=m_htMapEditCopy->getXExtent()) {
166 continue;
167 }
168 for (j=ndx.y-sub; j<ndx.y+add; j++) {
169 if (j<0 || j>=m_htMapEditCopy->getYExtent()) {
170 continue;
171 }
172 Real blendFactor;
173 blendFactor = calcRoundBlendFactor(ndx, i, j, m_feather, 0);
174 // m_htMapEditCopy is the output.
175 // m_htMapFeatherCopy is the original input.
176 // m_htMapRateCopy is how much we use of the feathered data.
177 if (blendFactor > 0.0f) {
178 Int rate = m_htMapRateCopy->getHeight(i, j);
179 rate += blendFactor * m_rate*5;
180 if (rate>255) {
181 rate = 255;
182 redoRate = true;
183 }
184 m_htMapRateCopy->setHeight(i,j,rate);
185 Int total=0;
186 Real numSamples=0;
187 Int ii, jj;
188 Int radius = m_radius;
189 if (radius<1) radius=1;
190 if (radius>FeatherOptions::MAX_RADIUS) radius = FeatherOptions::MAX_RADIUS;
191 for (ii = i-radius; ii < i+radius+1; ii++) {
192 for (jj = j-radius; jj<j+radius+1; jj++) {
193 Real factor;
194 if (i==ii && j==jj) {
195 factor = 1.0f;
196 } else {
197 Real dist = sqrt((ii-i)*(ii-i)+(jj-j)*(jj-j));
198 if (dist<1.0) dist = 1.0;
199 if (dist>radius) {
200 factor = 0;
201 } else {
202 factor = 1.0f - (dist-1)/radius;
203 }
204 }
205 int iNdx = ii;
206 if (iNdx<0) iNdx = 1;
207 if (iNdx >=m_htMapEditCopy->getXExtent()) {
208 iNdx = m_htMapEditCopy->getXExtent()-1;
209 }
210 int jNdx = jj;
211 if (jNdx<0) jNdx = 1;
212 if (jNdx >=m_htMapEditCopy->getYExtent()) {
213 jNdx = m_htMapEditCopy->getYExtent()-1;
214 }
215 total += m_htMapFeatherCopy->getHeight(iNdx, jNdx);
216 numSamples+=1;
217 }
218 }
219 total = floor((total/numSamples));
220 UnsignedByte origHeight = m_htMapFeatherCopy->getHeight(i, j);
221 float rateF = rate/255.0;
222 total = floor(origHeight*(1.0f-rateF) + total*rateF + 0.5f);
223 m_htMapEditCopy->setHeight(i, j, total);
224 pDoc->invalCell(i, j);
225 }
226 }
227 }
228
229 IRegion2D partialRange;
230 partialRange.lo.x = ndx.x - brushWidth;
231 partialRange.hi.x = ndx.x + brushWidth;
232 partialRange.lo.y = ndx.y - brushWidth;
233 partialRange.hi.y = ndx.y + brushWidth;
234 pDoc->updateHeightMap(m_htMapEditCopy, true, partialRange);
235 if (redoRate) {
236 Int size = m_htMapRateCopy->getXExtent() * m_htMapRateCopy->getYExtent();
237 UnsignedByte *pData = m_htMapRateCopy->getDataPtr();
238 UnsignedByte *pFeather = m_htMapFeatherCopy->getDataPtr();
239 UnsignedByte *pEdit = m_htMapEditCopy->getDataPtr();
240 Int i;
241 for (i=0; i<size; i++) {
242 *pData++ = 0;
243 *pFeather = *pEdit;
244 pFeather++; pEdit++;
245 }
246 }
