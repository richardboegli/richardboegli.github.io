299void Radar::update( void )
300{
301 Int i;
302 UnsignedInt thisFrame = TheGameLogic->getFrame();
303
304 //
305 // traverse the radar event list, if an event has a creationFrame it means that it
306 // exists ... check to see if it's time for that event to die and if so, just clear
307 // out the active status
308 //
309 for( i = 0; i < MAX_RADAR_EVENTS; i++ )
310 {
311
312 if( m_event[ i ].active == TRUE && m_event[ i ].createFrame &&
313 thisFrame > m_event[ i ].dieFrame )
314 m_event[ i ].active = FALSE;
315
316 } // end for i
317
318 // see if we should refresh the terrain
319 if( m_queueTerrainRefreshFrame != 0 &&
320 TheGameLogic->getFrame() - m_queueTerrainRefreshFrame > RADAR_QUEUE_TERRAIN_REFRESH_DELAY )
321 {
322
323 // refresh the terrain
324 refreshTerrain( TheTerrainLogic );
325
326 } // end if
327
328} // end update

333void Radar::newMap( TerrainLogic *terrain )
334{
335
336 // keep a pointer for our radar window
337 Int id = NAMEKEY( "ControlBar.wnd:LeftHUD" );
338 m_radarWindow = TheWindowManager->winGetWindowFromId( NULL, id );
339 DEBUG_ASSERTCRASH( m_radarWindow, ("Radar::newMap - Unable to find radar game window\n") );
340
341 // reset all the data in the radar
342 reset();
343
344 // get the extents of the new map
345 terrain->getExtent( &m_mapExtent );
346
347 // we will sample at these intervals across the map
348 m_xSample = m_mapExtent.width() / RADAR_CELL_WIDTH;
349 m_ySample = m_mapExtent.height() / RADAR_CELL_HEIGHT;
350
351 // find the "middle" height for the terrain (most used value) and water table
352 Int x, y;
353 Int terrainSamples = 0, waterSamples = 0;
354
355 m_terrainAverageZ = 0.0f;
356 m_waterAverageZ = 0.0f;
357 Coord3D worldPoint;
358
359 // since we're averaging let's skip every second sample...
360 worldPoint.y=0;
361 for( y = 0; y < RADAR_CELL_HEIGHT; y+=2, worldPoint.y+=2.0*m_ySample )
362 {
363 worldPoint.x=0;
364 for( x = 0; x < RADAR_CELL_WIDTH; x+=2, worldPoint.x+=2.0*m_xSample )
365 {
366 // don't use this, we don't really need the
367 // Z position by this function... radarToWorld( &radarPoint, &worldPoint );
368 // and this is done by isUnderwater anyway: z = terrain->getGroundHeight( worldPoint.x, worldPoint.y );
369 Real z,waterZ;
370 if( terrain->isUnderwater( worldPoint.x, worldPoint.y, &waterZ, &z ) )
371 {
372 m_waterAverageZ += z;
373 waterSamples++;
374 }
375 else
376 {
377 m_terrainAverageZ += z;
378 terrainSamples++;
379 }
380
381 } // end for x
382 }
383
384 // avoid divide by zeros
385 if( terrainSamples == 0 )
386 terrainSamples = 1;
387 if( waterSamples == 0 )
388 waterSamples = 1;
389
390 // compute averages
391 m_terrainAverageZ = m_terrainAverageZ / INT_TO_REAL( terrainSamples );
392 m_waterAverageZ = m_waterAverageZ / INT_TO_REAL( waterSamples );
393
394} // end newMap

400void Radar::addObject( Object *obj )
401{
402
403 // get the radar priority for this object
404 RadarPriorityType newPriority = obj->getRadarPriority();
405 if( isPriorityVisible( newPriority ) == FALSE )
406 return;
407
408 // if this object is on the radar, remove it in favor of the new add
409 RadarObject **list;
410 RadarObject *newObj;
411
412 // sanity
413 DEBUG_ASSERTCRASH( obj->friend_getRadarData() == NULL,
414 ("Radar: addObject - non NULL radar data for '%s'\n",
415 obj->getTemplate()->getName().str()) );
416
417 // allocate a new object
418 newObj = newInstance(RadarObject);
419
420 // set the object data
421 newObj->friend_setObject( obj );
422
423 // set color for this object on the radar
424 const Player *player = obj->getControllingPlayer();
425 Bool useIndicatorColor = true;
426
427 if( obj->isKindOf( KINDOF_DISGUISER ) )
428 {
429 //Because we have support for disguised units pretending to be units from another
430 //team, we need to intercept it here and make sure it's rendered appropriately
431 //based on which client is rendering it.
432 StealthUpdate *update = obj->getStealth();
433 if( update )
434 {
435 if( update->isDisguised() )
436 {
437 Player *clientPlayer = ThePlayerList->getLocalPlayer();
438 Player *disguisedPlayer = ThePlayerList->getNthPlayer( update->getDisguisedPlayerIndex() );
439 if( player->getRelationship( clientPlayer->getDefaultTeam() ) != ALLIES && clientPlayer->isPlayerActive() )
440 {
441 //Neutrals and enemies will see this disguised unit as the team it's disguised as.
442 player = disguisedPlayer;
443 if( player )
444 useIndicatorColor = false;
445 }
446 //Otherwise, the color will show up as the team it really belongs to (already set above).
447 }
448 }
449 }
450
451 if( obj->getContain() )
452 {
453 // To handle Stealth garrison, ask containers what color they are drawing with to the local player.
454 // Local is okay because radar display is not synced.
455 player = obj->getContain()->getApparentControllingPlayer( ThePlayerList->getLocalPlayer() );
456 if( player )
457 useIndicatorColor = false;
458 }
459
460 if( useIndicatorColor || (player == NULL) )
461 {
462 newObj->setColor( obj->getIndicatorColor() );
463 }
464 else
465 {
466 newObj->setColor( player->getPlayerColor() );
467 }
468
469 // set a chunk of radar data in the object
470 obj->friend_setRadarData( newObj );
471
472 //
473 // we will put this on either the local object list for objects that belong to the
474 // local player, or on the regular object list for all other objects
475 //
476 if( obj->isLocallyControlled() )
477 list = &m_localObjectList;
478 else
479 list = &m_objectList;
480
481 // link object to master list at the head of it's priority section
482 if( *list == NULL )
483 *list = newObj; // trivial case, an empty list
484 else
485 {
486 RadarPriorityType prevPriority, currPriority;
487 RadarObject *currObject, *prevObject, *nextObject;
488
489 prevObject = NULL;
490 prevPriority = RADAR_PRIORITY_INVALID;
491 for( currObject = *list; currObject; currObject = nextObject )
492 {
493
494 // get the next object
495 nextObject = currObject->friend_getNext();
496
497 // get the priority of this entry in the list (currPriority)
498 currPriority = currObject->friend_getObject()->getRadarPriority();
499
500 //
501 // if there is no previous object, or the previous priority is less than the
502 // our new priority, and the current object in the list has a priority
503 // higher than our equal to our own we need to be inserted here
504 //
505 if( (prevObject == NULL || prevPriority < newPriority ) &&
506 (currPriority >= newPriority) )
507 {
508
509 // insert into the list just ahead of currObject
510 if( prevObject )
511 {
512
513 // the new entry next points to what the previous one used to point to
514 newObj->friend_setNext( prevObject->friend_getNext() );
515
516 // the previous one next now points to the new entry
517 prevObject->friend_setNext( newObj );
518
519 } // end if
520 else
521 {
522
523 // the new object next points to the current object
524 newObj->friend_setNext( currObject );
525
526 // new list head is now newObj
527 *list = newObj;
528
529 } // end else
530
531 break; // exit for, stop the insert
532
533 } // end if
534 else if( nextObject == NULL )
535 {
536
537 // at the end of the list, put object here
538 currObject->friend_setNext( newObj );
539
540 } // end else if
541
542 // our current object is now the previous object
543 prevObject = currObject;
544 prevPriority = currPriority;
545
546 } // end if
547
548 } // end else
549
550} // end addObject

555Bool Radar::deleteFromList( Object *obj, RadarObject **list )
556{
557 RadarObject *radarObject, *prevObject = NULL;
558
559 // find the object in list
560 for( radarObject = *list; radarObject; radarObject = radarObject->friend_getNext() )
561 {
562
563 if( radarObject->friend_getObject() == obj )
564 {
565
566 // unlink the object from list
567 if( prevObject == NULL )
568 *list = radarObject->friend_getNext(); // removing head of list
569 else
570 prevObject->friend_setNext( radarObject->friend_getNext() );
571
572 // set the object radar data to NULL
573 obj->friend_setRadarData( NULL );
574
575 // delete the object instance
576 radarObject->deleteInstance();
577
578 // all done, object found and deleted
579 return TRUE;
580
581 } // end if
582
583 // save this object as previous one encountered in the list
584 prevObject = radarObject;
585
586 } // end for, radarObject
587
588 // object was not found in this list
589 return FALSE;
590
591} // end deleteFromList

627

635

642 y = 0;

644 y = RADAR_CELL_HEIGHT - 1;
645
646 // translate to world
647 world->x = x * m_xSample;
648 world->y = y * m_ySample;
649 return TRUE;

675Bool Radar::worldToRadar( const Coord3D *world, ICoord2D *radar )
676{
677
678 // sanity
679 if( world == NULL || radar == NULL )
680 return FALSE;
681
682 // sanity check the world position
683// if( world->x < m_mapExtent.lo.x || world->x > m_mapExtent.hi.x ||
684// world->y < m_mapExtent.lo.y || world->y > m_mapExtent.hi.y )
685// return FALSE;
686 // This is actually an insanity check. Nobody uses the return value, so this just leaves garbage in the
687 // return pointer. The reason the question gets asked is there are 60 partition cells to 128 radar cells
688 // (for example), and the radar wants to draw a horizontal line. This line ends up three pixels long
689 // at the right side, so the radar gives up and doesn't draw the middle one.
690 // We bind to on radar anyway, and we only ask if we are on radar, so don't intentionally add edge weirdness.
691
692 // convert
693 radar->x = world->x / m_xSample;
694 radar->y = world->y / m_ySample;
695
696 // keep it in bounds
697 if( radar->x < 0 )
698 radar->x = 0;
699 if( radar->x >= RADAR_CELL_WIDTH )
700 radar->x = RADAR_CELL_WIDTH - 1;
701 if( radar->y < 0 )
702 radar->y = 0;
703 if( radar->y >= RADAR_CELL_HEIGHT )
704 radar->y = RADAR_CELL_HEIGHT - 1;
705
706 return TRUE; // valid translation
707
708} // end worldToRadar

716Bool Radar::localPixelToRadar( const ICoord2D *pixel, ICoord2D *radar )
717{
718
719 // sanity
720 if( pixel == NULL || radar == NULL )
721 return FALSE;
722
723 // get window size of the radar
724 ICoord2D size;
725 m_radarWindow->winGetSize( &size.x, &size.y );
726
727 //
728 // act like we're going to draw and find the aspect ratio adjusted points of the
729 // terrain radar positions
730 //
731 ICoord2D start = { 0, 0 };
732 ICoord2D ul, lr;
733 findDrawPositions( start.x, start.y, size.x, size.y, &ul, &lr );
734
735 // get the scaled width and height
736 Int scaledWidth = lr.x - ul.x;
737 Int scaledHeight = lr.y - ul.y;
738
739 // if the pixel is outsize of the adjusted radar area there are no logical coords
740 if( pixel->x < ul.x || pixel->x > lr.x ||
741 pixel->y < ul.y || pixel->y > lr.y )
742 return FALSE;
743
744 if( scaledWidth >= scaledHeight )
745 {
746
747 // just normal conversion from full stretched to radar cells
748 radar->x = (pixel->x - ul.x)* RADAR_CELL_WIDTH / scaledWidth;
749
750 // conversion for scaled Y direction in map
751 radar->y = REAL_TO_INT( ((pixel->y - ul.y) / INT_TO_REAL( scaledHeight )) * size.y );
752
753 //
754 // radar->y now refers to a point that was "as if" the map was square, translate to radar
755 // note that y is inverted to have the radar align with the world (+x = right, -y = down)
756 //
757 radar->y = (size.y - radar->y) * RADAR_CELL_HEIGHT / size.y;
758
759
760 } // end if
761 else
762 {
763
764 // conversion for scaled Y direction in map
765 radar->x = REAL_TO_INT( ((pixel->x - ul.x) / INT_TO_REAL( scaledWidth )) * size.x );
766
767 // radar->x now refers to a point that was "as if" the map was square, translate to radar
768 radar->x = radar->x * RADAR_CELL_WIDTH / size.x;
769
770 //
771 // just normal conversion from full stretched to radar cells, note that y is inverted
772 // to have the radar align with the world (+x = right, -y = down)
773 //
774 radar->y = (size.y - pixel->y) * RADAR_CELL_HEIGHT / size.y;
775
776 } // end else
777
778 return TRUE;
779
780} // end localPixelToRadar

786Bool Radar::screenPixelToWorld( const ICoord2D *pixel, Coord3D *world )
787{
788
789 // sanity
790 if( pixel == NULL || world == NULL )
791 return FALSE;
792
793 // if we have no radar window can't do the conversion
794 if( m_radarWindow == NULL )
795 return FALSE;
796
797 // translate pixel coords to local pixel coords relative to the radar window
798 ICoord2D localPixel;
799 ICoord2D screenPos;
800 m_radarWindow->winGetScreenPosition( &screenPos.x, &screenPos.y );
801 localPixel.x = pixel->x - screenPos.x;
802 localPixel.y = pixel->y - screenPos.y;
803
804 // translate local pixel to radar
805 ICoord2D radar;
806 if( localPixelToRadar( &localPixel, &radar ) == FALSE )
807 return FALSE;
808
809 // translate radar to world
810 return radarToWorld( &radar, world );

1293Bool Radar::tryEvent( RadarEventType event, const Coord3D *pos )
1294{
1295
1296 // sanity
1297 if( event <= RADAR_EVENT_INVALID || event >= RADAR_EVENT_NUM_EVENTS || pos == NULL )
1298 return FALSE;
1299
1300 // see if there was an event of this type within the given range within the given time
1301 UnsignedInt currentFrame = TheGameLogic->getFrame();
1302 const Real closeEnoughDistanceSq = 250.0f * 250.0f;
1303 const UnsignedInt framesBetweenEvents = LOGICFRAMES_PER_SECOND * 10;
1304 //
1305
1306 // see if there was any matching radar event within range of this one
1307 // that wasn't too long ago, if there was we won't create another and get outta here
1308 //
1309 for( Int i = 0; i < MAX_RADAR_EVENTS; ++i )
1310 {
1311
1312 // only pay attention to under attack events
1313 if( m_event[ i ].type == event )
1314 {
1315
1316 // get distance from our new event location to this event location in 2D
1317 Real distSquared = m_event[ i ].worldLoc.x - pos->x * m_event[ i ].worldLoc.x - pos->x +
1318 m_event[ i ].worldLoc.y - pos->y * m_event[ i ].worldLoc.y - pos->y;
1319
1320 if( distSquared <= closeEnoughDistanceSq )
1321 {
1322
1323 // finally only reject making a new event of this existing one is "recent enough"
1324 if( currentFrame - m_event[ i ].createFrame < framesBetweenEvents )
1325 return FALSE; // reject it
1326
1327 } // end if
1328
1329 } // end if
1330
1331 } // end for i
1332
1333 // if we got here then we want to create a new event
1334 createEvent( pos, event );
1335
1336 // return TRUE for successfully created event
1337 return TRUE;
1338
1339} // end tryEvent
