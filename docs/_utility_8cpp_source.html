62/***********************************************************************************************
63 * get_absolute_path_cf -- Returns the absolute path based on a relative one. *
64 * *
65 * wwGetAbsolutePath - Usage: wwGetAbsolutePath <relative_path> <context> *
66 * *
67 * INPUT: A string containing a relative path, and its context. *
68 * *
69 * OUTPUT: The equivalent absolute path. *
70 * *
71 * WARNINGS: *
72 * *
73 * HISTORY: *
74 * 10/4/1999 AJA : Created. *
75 *=============================================================================================*/
76Value * get_absolute_path_cf (Value **arg_list, int count)
77{
78 // We want an array as an argument
79 check_arg_count("wwGetAbsolutePath", 2, count);
80 type_check(arg_list[0], String, "Relative path");
81 type_check(arg_list[1], String, "Context");
82
83 // Grab the arguments out of the array.
84 char absolute[MAX_PATH];
85 char *relative = arg_list[0]->to_string();
86 char *context = arg_list[1]->to_string();
87
88 // Turn the relative path into an absolute one.
89 Create_Full_Path(absolute, context, relative);
90
91 // Return the absolute path.
92 one_typed_value_local(String *abs);
93 vl.abs = new String(absolute);
94 return_value(vl.abs);
95}

98/***********************************************************************************************
99 * input_box_cf -- Retrive a string from the user in a nice friendly manner. *
100 * *
101 * wwInputBox - Usage: wwInputBox [caption:'caption'] [label:'label'] [value:'value'] *
102 * *
103 * INPUT: Three optional named arguments. If they are not given values, defaults are used. *
104 * *
105 * OUTPUT: Returns the string entered by the user. *
106 * *
107 * WARNINGS: *
108 * *
109 * HISTORY: *
110 *=============================================================================================*/
111Value * input_box_cf (Value **arg_list, int count)
112{
113 // Create the input box (but don't show it yet).
114 InputDlg input_box(MAXScript_interface->GetMAXHWnd());
115 Value *param = NULL;
116
117 // Check the 'caption' parameter.
118 param = key_arg(caption);
119 if (param && param != &unsupplied)
120 input_box.SetCaption(param->to_string());
121
122 // Check the 'label' parameter.
123 param = key_arg(label);
124 if (param && param != &unsupplied)
125 input_box.SetLabel(param->to_string());
126
127 // Check the 'value' parameter.
128 param = key_arg(value);
129 if (param && param != &unsupplied)
130 input_box.SetValue(param->to_string());
131
132 // Show the dialog and let the user enter a value.
133 if (input_box.DoModal() == IDCANCEL)
134 return &undefined;
135
136 // Return the value the user entered.
137 one_typed_value_local(String *val);
138 vl.val = new String(input_box.m_Value);
139 return_value(vl.val);
140}
