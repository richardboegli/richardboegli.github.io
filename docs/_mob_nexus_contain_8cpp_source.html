138Bool MobNexusContain::isValidContainerFor(const Object* rider, Bool checkCapacity) const
139{
140
141 // sanity
142 if (!rider)
143 return false;
144
145 //The point of this new code is to determine when something has a negative 1 contain max, to
146 //look at the object inside of it to use that as the valid check. There is a case, when a
147 //paratrooper (an infantry contained in a parachute). In this case, when we pass this object
148 //to contain in a --- plane, we want to check the infantry, not the parachute.
149// const MobNexusContainModuleData *modData = getMobNexusContainModuleData();
150
151 //Check if we are a fake container, and if so, get an object inside it to see what kind this object *is*.
152 if( rider->getContain() && rider->getContain()->isSpecialZeroSlotContainer() )
153 {
154 //Report the first thing inside it!
155 const ContainedItemsList *items = rider->getContain()->getContainedItemsList();
156 if( items )
157 {
158 ContainedItemsList::const_iterator it;
159 it = items->begin();
160 if( *it )
161 {
162 //Replace the object we are checking with the *first* object contained within it.
163 rider = *it;
164 }
165 }
166 }
167 else
168 {
169 //blech! This case may or may not occur... in which case, just use the supplied object.
170 }
171
172
173 // extend functionality
174 if( OpenContain::isValidContainerFor( rider, checkCapacity ) == false )
175 return false;
176
177 // only allied objects can be MobNexused.
178 // order matters: we want to know if IT considers ME to be an ally (a reversal of the usual situation)
179 if (rider->getRelationship(getObject()) != ALLIES)
180 return false;
181
182 Int mobNexusSlotCount = rider->getTransportSlotCount();
183
184 // if 0, this object isn't MobNexusable.
185 if (mobNexusSlotCount == 0)
186 return false;
187
188 if (checkCapacity)
189 {
190 return (m_extraSlotsInUse + getContainCount() + mobNexusSlotCount <= getContainMax());
191 }
192 else
193 {
194 return true;
195 }
196}

200void MobNexusContain::onContaining( Object *rider, Bool wasSelected )
201{
202 OpenContain::onContaining( rider, wasSelected );
203
204 // objects inside a MobNexus are held
205 rider->setDisabled( DISABLED_HELD );
206
207 Int mobNexusSlotCount = rider->getTransportSlotCount();
208
209 DEBUG_ASSERTCRASH(mobNexusSlotCount > 0, ("Hmm, this object isnt MobNexusable"));
210 m_extraSlotsInUse += mobNexusSlotCount - 1;
211 DEBUG_ASSERTCRASH(m_extraSlotsInUse >= 0 && m_extraSlotsInUse + getContainCount() <= getContainMax(), ("Hmm, bad slot count"));
212
213 //
214 // when we go from holding nothing to holding something we have a model condition
215 // to visually show the change
216 //
217 if( getContainCount() == 1 )
218 {
219 Drawable *draw = getObject()->getDrawable();
220
221 if( draw )
222 draw->setModelConditionState( MODELCONDITION_LOADED );
223
224 } // end if
225

230void MobNexusContain::onRemoving( Object *rider )
231{
232 OpenContain::onRemoving(rider);
233
234 // object is no longer held inside a MobNexus
235 rider->clearDisabled( DISABLED_HELD );
236
237 const MobNexusContainModuleData* d = getMobNexusContainModuleData();
238
239 if (!d->m_exitBone.isEmpty())
240 {
241 Drawable* draw = getObject()->getDrawable();
242 if (draw)
243 {
244 Coord3D bonePos, worldPos;
245 if (draw->getPristineBonePositions(d->m_exitBone.str(), 0, &bonePos, NULL, 1) == 1)
246 {
247 getObject()->convertBonePosToWorldPos(&bonePos, NULL, &worldPos, NULL);
248 rider->setPosition(&worldPos);
249 }
250 }
251 }
252
253 if (d->m_orientLikeContainerOnExit)
254 {
255 rider->setOrientation(getObject()->getOrientation());
256 }
257
258 if (d->m_keepContainerVelocityOnExit)
259 {
260 PhysicsBehavior* parent = getObject()->getPhysics();
261 PhysicsBehavior* child = rider->getPhysics();
262 if (parent && child)
263 {
264 Coord3D startingForce = *parent->getVelocity();
265 Real mass = child->getMass();
266 startingForce.x *= mass;
267 startingForce.y *= mass;
268 startingForce.z *= mass;
269 child->applyMotiveForce( &startingForce );
270
271 Real pitchRate = child->getCenterOfMassOffset() * d->m_exitPitchRate;
272 child->setPitchRate( pitchRate );
273 }
274 }
275
276 if (d->m_scatterNearbyOnExit)
277 scatterToNearbyPosition(rider);
278
279 Int mobNexusSlotCount = rider->getTransportSlotCount();
280 DEBUG_ASSERTCRASH(mobNexusSlotCount > 0, ("This object isnt MobNexusable"));
281 m_extraSlotsInUse -= mobNexusSlotCount - 1;
282 DEBUG_ASSERTCRASH(m_extraSlotsInUse >= 0 && m_extraSlotsInUse + getContainCount() <= getContainMax(), ("Bad slot count, MobNexus"));
283
284 // when we are empty again, clear the model condition for loaded
285 if( getContainCount() == 0 )
286 {
287 Drawable *draw = getObject()->getDrawable();
288
289 if( draw )
290 draw->clearModelConditionState( MODELCONDITION_LOADED );
291
292 } // end if
293
294 if (getObject()->isAboveTerrain())
295 {
296 // temporarily mark the guy as being allowed to fall
297 // (overriding his locomotor's stick-to-ground attribute).
298 // this will be reset (by PhysicsBehavior) when he touches the ground.
299 PhysicsBehavior* physics = rider->getPhysics();
300 if (physics)
301 physics->setAllowToFall(true);
302 }
303
304}

308void MobNexusContain::onObjectCreated()
309{
310 MobNexusContainModuleData* self = (MobNexusContainModuleData*)getMobNexusContainModuleData();
311
312 Int count = self->m_initialPayload.count;
313 const ThingTemplate* payloadTemplate = TheThingFactory->findTemplate( self->m_initialPayload.name );
314 Object* object = getObject();
315
316 for( int i = 0; i < count; i++ )
317 {
318 //We are creating a MobNexus that comes with a initial payload, so add it now!
319
320 Object* payload = TheThingFactory->newObject( payloadTemplate, object->getControllingPlayer()->getDefaultTeam() );
321 if( object->getContain() && object->getContain()->isValidContainerFor( payload, true ) )
322 {
323 object->getContain()->addToContain( payload );
324 }
325 else
326 {
327 DEBUG_CRASH( ( "DeliverPayload: PutInContainer %s is full, or not valid for the payload %s!", object->getName().str(), self->m_initialPayload.name.str() ) );
328 }
329 }
330}

335UpdateSleepTime MobNexusContain::update()
336{
337 MobNexusContainModuleData *moduleData = (MobNexusContainModuleData*)getModuleData();
338
339 if( moduleData && moduleData->m_healthRegen )
340 {
341 ContainModuleInterface *contain = getObject()->getContain();
342 if( contain )
343 {
344 //This MobNexus has a health regeneration value, so go through and heal all inside.
345 const ContainedItemsList* items = contain->getContainedItemsList();
346 if( items )
347 {
348 ContainedItemsList::const_iterator it;
349 it = items->begin();
350
351 while( *it )
352 {
353 Object *object = *it;
354
355 //Advance to the next iterator
356 it++;
357
358 //Determine if we need healing or not.
359 BodyModuleInterface *body = object->getBodyModule();
360 if( body->getHealth() < body->getMaxHealth() )
361 {
362 //Calculate the health to be regenerated on each unit.
363 Real regen = body->getMaxHealth() * moduleData->m_healthRegen / 100.0f * SECONDS_PER_LOGICFRAME_REAL;
364
365 //Perform the actual healing for this frame.
366// DamageInfo damageInfo;
367// damageInfo.in.m_damageType = DAMAGE_HEALING;
368// damageInfo.in.m_deathType = DEATH_NONE;
369// damageInfo.in.m_sourceID = getObject()->getID();
370// damageInfo.in.m_amount = regen;
371// object->attemptDamage( &damageInfo );
372 object->attemptHealing( regen, getObject() );
373 }
374 }
375 }
376 }
377 }

379 return OpenContain::update(); //extend

384ExitDoorType MobNexusContain::reserveDoorForExit( const ThingTemplate* objType, Object *specificObject )
385{
386 if( specificObject == NULL )
387 return DOOR_1;// I can, in general, exit people.
388
389 // This is an override, not an extend. I will check for game legality for
390 // okaying the call to exitObjectViaDoor.
391 Object *me = getObject();
392
393 // this is present solely for some MobNexuss to override, so that they can land before
394 // allowing people to exit...
395 AIUpdateInterface* ai = me->getAIUpdateInterface();
396 if (ai && ai->getAiFreeToExit(me) != FREE_TO_EXIT)
397 return DOOR_NONE_AVAILABLE;
398
399 // I can always kick people out if I am in the air, I know what I'm doing
400 if( me->isUsingAirborneLocomotor() )
401 return DOOR_1;
402
403 const Coord3D *myPosition = me->getPosition();
404 if( !specificObject->getAIUpdateInterface() )
405 {
406 return DOOR_NONE_AVAILABLE;
407 }
408 const Locomotor *hisLocomotor = specificObject->getAIUpdateInterface()->getCurLocomotor();
409 // He can't get to this spot naturally, so I can't force him there. (amphib MobNexus)
410 if( ! TheAI->pathfinder()->validMovementTerrain(me->getLayer(), hisLocomotor, myPosition ) )
411 return DOOR_NONE_AVAILABLE;
412
413 return DOOR_1;
414}

423// ------------------------------------------------------------------------------------------------
424// ------------------------------------------------------------------------------------------------
425Bool MobNexusContain::tryToEvacuate( Bool exposeStealthedUnits )
426{
427 Bool exitedAnyone = false;
428 ContainedItemsList::const_iterator it = getContainList().begin();
429 while( it != getContainList().end() )
430 {
431 // get the object
432 Object *obj = *it;
433 // increment the iterator, since removal will pull this guy from the list somewhere else
434 // and we might not actually kick anyone so we don't want to loop forever.
435 ++it;
436
437 ExitDoorType exitDoor = reserveDoorForExit(obj->getTemplate(), obj);
438 if(exitDoor != DOOR_NONE_AVAILABLE)
439 {
440 exitObjectViaDoor( obj, exitDoor );
441 exitedAnyone = true;
442
443 if( obj->isKindOf( KINDOF_STEALTH_GARRISON ) && exposeStealthedUnits )
444 {
445 StealthUpdate* stealth = obj->getStealth();
446 if( stealth )
447 {
448 stealth->markAsDetected();
449 }
450 }
451 }
452 }
453 return exitedAnyone;
454}
