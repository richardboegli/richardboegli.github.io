75WindowMsgHandledType GadgetStaticTextInput( GameWindow *window, UnsignedInt msg,
76 WindowMsgData mData1, WindowMsgData mData2 )
77{
78
79 switch( msg )
80 {
81 // ------------------------------------------------------------------------
82 case GWM_CHAR:
83 switch (mData1)
84 {
85
86 case KEY_DOWN:
87 case KEY_RIGHT:
88 case KEY_TAB:
89 // Just in case some fool sets static text as a tab stop
90 if( BitTest( mData2, KEY_STATE_DOWN ) )
91 window->winNextTab();
92 break;
93
94 case KEY_UP:
95 case KEY_LEFT:
96 if( BitTest( mData2, KEY_STATE_DOWN ) )
97 window->winPrevTab();
98 break;
99
100 default:
101 return MSG_IGNORED;
102 }
103
104 break;
105
106 default:
107 return MSG_IGNORED;
108
109 }
110 return MSG_HANDLED;
111
112
113} // end GadgetStaticTextInput

115// GadgetStaticTextSystem =====================================================
117//=============================================================================
118WindowMsgHandledType GadgetStaticTextSystem( GameWindow *window, UnsignedInt msg,
119 WindowMsgData mData1, WindowMsgData mData2 )
120{
121// WinInstanceData *instData = window->winGetInstanceData();
122
123 switch( msg )
124 {
125 // ------------------------------------------------------------------------
126 case GGM_GET_LABEL:
127 {
128 TextData *tData = (TextData *)window->winGetUserData();
129 if (tData && tData->text)
130 *(UnicodeString*)mData2 = tData->text->getText();
131 break;
132
133 } // end get label
134
135 // ------------------------------------------------------------------------
136 case GGM_SET_LABEL:
137 {
138 if( mData1 )
139 {
140 TextData *tData = (TextData *)window->winGetUserData();
141 if (tData && tData->text)
142 tData->text->setText( *(UnicodeString*)mData1 );
143 }
144
145 break;
146
147 } // end set label
148
149 // ------------------------------------------------------------------------
150 case GWM_CREATE:
151 break;
152
153 // ------------------------------------------------------------------------
154 case GWM_DESTROY:
155 {
156 TextData *data = (TextData *)window->winGetUserData();
157
158 // free the display string
159 TheDisplayStringManager->freeDisplayString( data->text );
160
161 // free text data
162 delete( data );
163
164 break;
165
166 } // end destroy
167
168 default:
169 return MSG_IGNORED;
170
171 } // end switch( msg )
172
173 return MSG_HANDLED;
174} // end GadgetStaticTextSystem

188{

217 {
218
219 dString = textData->text;

225} // end GadgetStaticTextSetFont
