73void RenderInfoClass::Push_Material_Pass(MaterialPassClass * matpass)
74{
75 // add to the end of the array
76 if (AdditionalMaterialPassCount<MAX_ADDITIONAL_MATERIAL_PASSES-1) {
77
78 if (matpass) {
79 matpass->Add_Ref();
80 }
81 AdditionalMaterialPassArray[AdditionalMaterialPassCount++]=matpass;
82 } else {
83 RejectedMaterialPasses++;
84 }
85}
86
87void RenderInfoClass::Pop_Material_Pass(void)
88{
89 if (RejectedMaterialPasses == 0) {
90 // remove from the end of the array
91 WWASSERT(AdditionalMaterialPassCount>0);
92 AdditionalMaterialPassCount--;
93 MaterialPassClass * mpass = AdditionalMaterialPassArray[AdditionalMaterialPassCount];
94 if (mpass != NULL) {
95 mpass->Release_Ref();
96 }
97 } else {
98 RejectedMaterialPasses--;
99 }
100}
