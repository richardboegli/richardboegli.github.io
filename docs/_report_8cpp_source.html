158void CReport::OnOK()
159{
160 int count;
161 int i;
162 char buffer[100];
163
164 count = list->GetSelItems ( num_langs, langindices );
165
166 if ( !count )
167 {
168 AfxMessageBox ( "No languages selected" );
169 return;
170 }
171
172// get the filename
173 CFileDialog fd ( FALSE , NULL, "*.txt", OFN_OVERWRITEPROMPT | OFN_NOCHANGEDIR );
174
175 if ( fd.DoModal () != IDOK )
176 {
177 return;
178 }
179
180 strcpy ( filename, fd.GetPathName ());
181
182 num_langs = 0;
183 for ( i = 0; i <count; i++ )
184 {
185 LANGINFO *info;
186
187 if ( info = GetLangInfo ( langindices[i] ))
188 {
189 langids[num_langs++] = info->langid;
190 }
191 }
192
193 langids[num_langs] = LANGID_UNKNOWN;
194
195 options.dialog = dialog_status->GetCheck ();
196 options.translations = trans_status->GetCheck ();
197 limit->GetWindowText( buffer, sizeof(buffer)-1);
198 options.limit = atoi ( buffer );
199
200 if ( !show_details->GetCheck () )
201 {
202 options.limit = 0;
203 }
204 else if ( !ifless->GetCheck () )
205 {
206 options.limit = INT_MAX;
207 }
208
209
210 CDialog::OnOK();
211}
