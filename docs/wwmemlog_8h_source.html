155class WWMemorySampleClass
156{
157 bool category_push;
158public:
159 WWMemorySampleClass(int category) : category_push(WWMemoryLogClass::Is_Memory_Log_Enabled())
160 {
161 if (category_push) {
162 WWMemoryLogClass::Push_Active_Category(category);
163 }
164 }
165
166 ~WWMemorySampleClass(void)
167 {
168 if (category_push) {
169 WWMemoryLogClass::Pop_Active_Category();
170 }
171 }
