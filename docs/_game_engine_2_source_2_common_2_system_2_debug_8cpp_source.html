235// ----------------------------------------------------------------------------
236// doCrashBox
237/*
238 present a messagebox with the given message. Depending on user selection,
239 we exit the app, break into debugger, or continue execution.
240*/
241// ----------------------------------------------------------------------------
242static int doCrashBox(const char *buffer, Bool logResult)
243{
244 int result;
245
246 if (!ignoringAsserts()) {
247 result = MessageBoxWrapper(buffer, "Assertion Failure", MB_ABORTRETRYIGNORE|MB_TASKMODAL|MB_ICONWARNING|MB_DEFBUTTON3);
248 //result = MessageBoxWrapper(buffer, "Assertion Failure", MB_ABORTRETRYIGNORE|MB_TASKMODAL|MB_ICONWARNING);
249 } else {
250 result = IDIGNORE;
251 }
252
253 switch(result)
254 {
255 case IDABORT:
256#ifdef DEBUG_LOGGING
257 if (logResult)
258 DebugLog("[Abort]\n");
259#endif
260 _exit(1);
261 break;
262 case IDRETRY:
263#ifdef DEBUG_LOGGING
264 if (logResult)
265 DebugLog("[Retry]\n");
266#endif
267 ::DebugBreak();
268 break;
269 case IDIGNORE:
270#ifdef DEBUG_LOGGING
271 // do nothing, just keep going
272 if (logResult)
273 DebugLog("[Ignore]\n");
274#endif
275 break;
276 }
277 return result;
278}

408// ----------------------------------------------------------------------------
409// DebugCrash
410// ----------------------------------------------------------------------------
411#ifdef DEBUG_CRASHING
417void DebugCrash(const char *format, ...)
418{
419 // Note: You might want to make this thread safe, but we cannot. The reason is that
420 // there is an implicit requirement on other threads that the message loop be running.
421
422 // make it not static so that it'll be thread-safe.
423 // make it big to avoid weird overflow bugs in debug mode
424 char theCrashBuffer[ LARGE_BUFFER ];
425 if (theDebugFlags == 0)
426 {
427 if (!DX8Wrapper_IsWindowed) {
428 if (ApplicationHWnd) {
429 ShowWindow(ApplicationHWnd, SW_HIDE);
430 }
431 }
432 MessageBoxWrapper("DebugCrash - Debug not inited properly", "", MB_OK|MB_TASKMODAL);
433 }
434
435 format = prepBuffer(format, theCrashBuffer);
436 strcat(theCrashBuffer, "ASSERTION FAILURE: ");
437
438 va_list arg;
439 va_start(arg, format);
440 vsprintf(theCrashBuffer + strlen(theCrashBuffer), format, arg);
441 va_end(arg);
442
443 if (strlen(theCrashBuffer) >= sizeof(theCrashBuffer))
444 {
445 if (!DX8Wrapper_IsWindowed) {
446 if (ApplicationHWnd) {
447 ShowWindow(ApplicationHWnd, SW_HIDE);
448 }
449 }
450 MessageBoxWrapper("String too long for debug buffers", "", MB_OK|MB_TASKMODAL);
451 }
452
453#ifdef DEBUG_LOGGING
454 if (ignoringAsserts())
455 {
456 doLogOutput("**** CRASH IN FULL SCREEN - Auto-ignored, CHECK THIS LOG!\n");
457 }
458 whackFunnyCharacters(theCrashBuffer);
459 doLogOutput(theCrashBuffer);
460#endif
461#ifdef DEBUG_STACKTRACE
462 if (!(TheGlobalData && TheGlobalData->m_debugIgnoreStackTrace))
463 {
464 doStackDump();
465 }
466#endif
467
468 strcat(theCrashBuffer, "\n\nAbort->exception; Retry->debugger; Ignore->continue\n");
469
470 int result = doCrashBox(theCrashBuffer, true);
471
472 if (result == IDIGNORE && TheCurrentIgnoreCrashPtr != NULL)
473 {
474 int yn;
475 if (!ignoringAsserts())
476 {
477 yn = MessageBoxWrapper("Ignore this crash from now on?", "", MB_YESNO|MB_TASKMODAL);
478 }
479 else
480 {
481 yn = IDYES;
482 }
483 if (yn == IDYES)
484 *TheCurrentIgnoreCrashPtr = 1;
485 if( TheKeyboard )
486 TheKeyboard->resetKeys();
487 if( TheMouse )
488 TheMouse->reset();
489 }
490
491}
492#endif

562// ----------------------------------------------------------------------------
563void SimpleProfiler::stop()
564{
565 if (m_startThisSession != 0)
566 {
567 __int64 stop;
568 QueryPerformanceCounter((LARGE_INTEGER*)&stop);
569 m_totalThisSession = stop - m_startThisSession;
570 m_totalAllSessions += stop - m_startThisSession;
571 m_startThisSession = 0;
572 ++m_numSessions;
573 }
574}

576// ----------------------------------------------------------------------------
577void SimpleProfiler::stopAndLog(const char *msg, int howOftenToLog, int howOftenToResetAvg)
578{
579 stop();
580 // howOftenToResetAvg==0 means "never reset"
581 if (howOftenToResetAvg > 0 && m_numSessions >= howOftenToResetAvg)
582 {
583 m_numSessions = 0;
584 m_totalAllSessions = 0;
585 DEBUG_LOG(("%s: reset averages\n",msg));
586 }
587 DEBUG_ASSERTLOG(m_numSessions % howOftenToLog != 0, ("%s: %f msec, total %f msec, avg %f msec\n",msg,getTime(),getTotalTime(),getAverageTime()));
588}
