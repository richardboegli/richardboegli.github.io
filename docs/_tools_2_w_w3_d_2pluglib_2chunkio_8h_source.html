61/************************************************************************************
62
63 ChunkIO
64
65 (gth) This module provides classes for reading and writing chunk-based files.
66 For example, all of the w3d files are stored in a hierarchical-chunk format.
67 Basically the format is similar to IFF. All data in the file has chunk headers
68 wrapped around it. A chunk header contains an ID, and a Size. The size
69 is the number of bytes in the chunk (not including the header). The
70 contents of a chunk may be either: more "sub-chunks" or raw data. These classes
71 will automatically keep track of your positions within all of the sub and parent
72 chunks (to some maximum recursion depth).
73
74 Sept 3, 1999
75 (gth) Adding the new concept of "micro-chunks". Instead of filling the contents of a
76 chunk with data, you can fill it with "micro-chunks" which contain a single byte
77 id and a single byte size. Micro-chunks are used for storing simple variables
78 in a form that can survive revisions to the file format without paying the price
79 for a full chunk header. You CANNOT recursively embed micro-chunks due to their
80 size limitations....
81
82 Sept 24, 1999
83 (gth) Using the MSB of the chunksize to indicate whether a chunk contains other
84 chunks or pure data. If the MSB is 0, the chunk contains data (so that the reader
85 I'm going to write doesn't break on older files) and if it is 1 then it is
86 assumed to contain other chunks. This does not apply to micro-chunks as they
87 are considered data.
88
89**************************************************************************************/
90
91struct ChunkHeader
92{
93 // Functions.
94 ChunkHeader() : ChunkType(0), ChunkSize(0) {}
95 ChunkHeader(uint32 type, uint32 size) {ChunkType = type; ChunkSize = size;}
96
97 // Use these accessors to ensure you correctly deal with the data in the chunk header
98 void Set_Type(uint32 type) { ChunkType = type; }
99 uint32 Get_Type(void) { return ChunkType; }
100 void Set_Size(uint32 size) { ChunkSize &= 0x80000000; ChunkSize |= (size & 0x7FFFFFFF); }
101 void Add_Size(uint32 add) { Set_Size(Get_Size() + add); }
102 uint32 Get_Size(void) { return (ChunkSize & 0x7FFFFFFF); }
103 void Set_Sub_Chunk_Flag(bool onoff) { if (onoff) { ChunkSize |= 0x80000000; } else { ChunkSize &= 0x7FFFFFFF; } }
104 int Get_Sub_Chunk_Flag(void) { return (ChunkSize & 0x80000000); }
105
106 // Chunk type and size.
107 // Note: MSB of ChunkSize is used to indicate whether this chunk
108 // contains other chunks or data.
109 uint32 ChunkType;
110 uint32 ChunkSize;
111};
112
113struct MicroChunkHeader
114{
115 MicroChunkHeader() {}
116 MicroChunkHeader(uint8 type, uint8 size) { ChunkType = type, ChunkSize = size; }
117
118 void Set_Type(uint8 type) { ChunkType = type; }
119 uint8 Get_Type(void) { return ChunkType; }
120 void Set_Size(uint8 size) { ChunkSize = size; }
121 void Add_Size(uint8 add) { Set_Size(Get_Size() + add); }
122 uint8 Get_Size(void) { return ChunkSize; }
123
124 uint8 ChunkType;
125 uint8 ChunkSize;
126};

230/*
231** WRITE_WWSTRING_CHUNK - use this one-line macro to easily create a chunk to save a potentially
232** long string. Note: This macro does NOT create a micro chunk...
233** Example:
234**
235** csave.Begin_Chunk(CHUNKID_PARENT);
236** ParentClass::Save (csave);
237** csave.End_Chunk();
238**
239** WRITE_WWSTRING_CHUNK(csave, CHUNKID_NAME, string);
240** WRITE_WIDESTRING_CHUNK(csave, CHUNKID_WIDE_NAME, wide_string);
241**
242** csave.Begin_Chunk(PHYSGRID_CHUNK_VARIABLES);
243** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_VERSION,version);
244** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_DUMMYVISID,DummyVisId);
245** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_BASEVISID,BaseVisId);
246** csave.End_Chunk();
247**
248*/
249#define WRITE_WWSTRING_CHUNK(csave,id,var) { \
250 csave.Begin_Chunk(id); \
251 csave.Write((const TCHAR *)var, var.Get_Length () + 1); \
252 csave.End_Chunk(); }
253
254#define WRITE_WIDESTRING_CHUNK(csave,id,var) { \
255 csave.Begin_Chunk(id); \
256 csave.Write((const WCHAR *)var, (var.Get_Length () + 1) * 2); \
257 csave.End_Chunk(); }

260/*
261** READ_WWSTRING_CHUNK - use this macro in a switch statement to read the contents
262** of a chunk into a string object.
263** Example:
264**
265** while (cload.Open_Chunk()) {
266**

282/*
283** WRITE_MICRO_CHUNK - use this one-line macro to easily make a micro chunk for an individual variable.
284** Note that you should always wrap your micro-chunks inside a normal chunk.
285** Example:
286**
287** csave.Begin_Chunk(PHYSGRID_CHUNK_VARIABLES);
288** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_VERSION,version);
289** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_DUMMYVISID,DummyVisId);
290** WRITE_MICRO_CHUNK(csave,PHYSGRID_VARIABLE_BASEVISID,BaseVisId);
291** csave.End_Chunk();
292*/
293#define WRITE_MICRO_CHUNK(csave,id,var) { \
294 csave.Begin_Micro_Chunk(id); \
295 csave.Write(&var,sizeof(var)); \
296 csave.End_Micro_Chunk(); }
297

302 csave.End_Micro_Chunk(); }

320/*
321** READ_MICRO_CHUNK - use this macro in a switch statement to read a micro chunk into a variable
322** Example:
323**
324** while (cload.Open_Micro_Chunk()) {
325**
326** switch(cload.Cur_Micro_Chunk_ID()) {
327** READ_MICRO_CHUNK(cload,PHYSGRID_VARIABLE_VERSION,version);
328** READ_MICRO_CHUNK(cload,PHYSGRID_VARIABLE_DUMMYVISID,DummyVisId);

331** cload.Close_Micro_Chunk();
332** }

334#define READ_MICRO_CHUNK(cload,id,var) \
335 case (id): cload.Read(&var,sizeof(var)); break; \

337/*

339*/
340#define READ_SAFE_MICRO_CHUNK(cload,id,var,type) \

342 void *temp_read_buffer_on_the_stack = _alloca(sizeof(var)); \
343 cload.Read(temp_read_buffer_on_the_stack, sizeof(var)); \
344 var = *((type*)temp_read_buffer_on_the_stack); \
345 break; \
346 }
