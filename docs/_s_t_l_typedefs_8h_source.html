119 // Generic hash functor. This should almost always be overridden for
120 // specific types.
121 template<typename T> struct hash
122 {
123 size_t operator()(const T& __t) const
124 {
125 std::hash<T> tmp;
126 return tmp(__t);
127 }
128 };
129
130 // Generic equal_to functor. This should be overridden if there is no
131 // operator==, or if that isn't the behavior desired. (For instance, in
132 // the case of pointers.)
133 template<typename T> struct equal_to
134 {
135 Bool operator()(const T& __t1, const T& __t2) const
136 {
137 return (__t1 == __t2);
138 }
139 };
140
141 // Generic less_than_nocase functor. This should be overridden if there is no
142 // operator<, or if that isn't the behavior desired. (For instance, in
143 // the case of pointers, or strings.)
144 template<typename T> struct less_than_nocase
145 {
146 bool operator()(const T& __t1, const T& __t2) const
147 {
148 return (__t1 < __t2);
149 }
150 };

179 // This is the equal_to overload for char* comparisons. We compare the
180 // strings to determine whether they are equal or not.
181 // Other overloads should go into specific header files, not here (unless
182 // they are ot be used in lots of places.)
183 template<> struct equal_to<const char*>
184 {
185 Bool operator()(const char* s1, const char* s2) const
186 {
187 return strcmp(s1, s2) == 0;
188 }
189 };
