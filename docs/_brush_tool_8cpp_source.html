139void BrushTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
140{
141 Coord3D cpt;
142 pView->viewToDocCoords(viewPt, &cpt);
143 DrawObject::setFeedbackPos(cpt);
144
145 if (m != TRACK_L) return;
146
147 pView->viewToDocCoords(viewPt, &cpt);
148
149 int brushWidth = m_brushWidth;
150 if (m_brushFeather>0) {
151 brushWidth += 2*m_brushFeather;
152 }
153 brushWidth += 2;
154
155 CPoint ndx;
156 getCenterIndex(&cpt, m_brushWidth, &ndx, pDoc);
157
158 if (m_prevXIndex == ndx.x && m_prevYIndex == ndx.y) return;
159
160 m_prevXIndex = ndx.x;
161 m_prevYIndex = ndx.y;
162
163 int sub = brushWidth/2;
164 int add = brushWidth-sub;
165
166 Int i, j;
167 for (i=ndx.x-sub; i<ndx.x+add; i++) {
168 if (i<0 || i>=m_htMapEditCopy->getXExtent()) {
169 continue;
170 }
171 for (j=ndx.y-sub; j<ndx.y+add; j++) {
172 if (j<0 || j>=m_htMapEditCopy->getYExtent()) {
173 continue;
174 }
175 Real blendFactor;
176 if (m_brushSquare) {
177 blendFactor = calcSquareBlendFactor(ndx, i, j, m_brushWidth, m_brushFeather);
178 } else {
179 blendFactor = calcRoundBlendFactor(ndx, i, j, m_brushWidth, m_brushFeather);
180 }
181 Int curHeight = m_htMapFeatherCopy->getHeight(i,j);
182 float fNewHeight = blendFactor*m_brushHeight+((1.0f-blendFactor)*curHeight) ;
183 Int newHeight = floor(fNewHeight+0.5);
184 if (m_brushHeight > curHeight) {
185 if (m_htMapEditCopy->getHeight(i,j)>newHeight) {
186 newHeight = m_htMapEditCopy->getHeight(i,j);
187 }
188 } else {
189 if (m_htMapEditCopy->getHeight(i,j)<newHeight) {
190 newHeight = m_htMapEditCopy->getHeight(i,j);
191 }
192 }
193 m_htMapEditCopy->setHeight(i, j, newHeight);
194 }
195 }
196 IRegion2D partialRange;
197 partialRange.lo.x = ndx.x - brushWidth;
198 partialRange.hi.x = ndx.x + brushWidth;
199 partialRange.lo.y = ndx.y - brushWidth;
200 partialRange.hi.y = ndx.y + brushWidth;
201 pDoc->updateHeightMap(m_htMapEditCopy, true, partialRange);
202}
