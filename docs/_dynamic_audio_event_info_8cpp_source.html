152void DynamicAudioEventInfo::xferNoName( Xfer * xfer )
153{
154 const XferVersion currentVersion = 1;
155 XferVersion version = currentVersion;
156 xfer->xferVersion( &version, currentVersion );
157
158 // The default BitFlags xfer function is really strange. It also requires us to name our
159 // bits, which I don't want to do, since it forces anyone who has the same number of bits
160 // as us to use our names. Instead, just transfer directly and count on our local version
161 // to keep things straight, If you change the list, be sure to change this code too...
162 if ( xfer->getXferMode() == XFER_LOAD )
163 {
164 UnsignedByte overriddenFlags;
165 DEBUG_ASSERTCRASH( OVERRIDE_COUNT <= sizeof( overriddenFlags ) * 8, ("Save/load code assumes override flags can fit into UnsignedByte, but it doesn't work anymore! Move up to larger integer type") );
166 xfer->xferUnsignedByte( &overriddenFlags );
167 Int field;
168 for ( field = 0; field < OVERRIDE_COUNT; field++ )
169 {
170 m_overriddenFields.set( field, BitTest( overriddenFlags, 1 << field ) );
171 }
172 }
173 else
174 {
175 UnsignedByte overriddenFlags = 0;
176 DEBUG_ASSERTCRASH( OVERRIDE_COUNT <= sizeof( overriddenFlags ) * 8, ("Save/load code assumes override flags can fit into UnsignedByte, but it doesn't work anymore! Move up to larger integer type") );
177 Int field;
178 for ( field = 0; field < OVERRIDE_COUNT; field++ )
179 {
180 if ( m_overriddenFields.test( field ) )
181 {
182 BitSet( overriddenFlags, 1 << field );
183 }
184 }
185 xfer->xferUnsignedByte( &overriddenFlags );
186 }
187
188 if ( wasLoopFlagOverriden() )
189 {
190 Bool loopFlag = BitTest( m_control, AC_LOOP );
191 xfer->xferBool( &loopFlag );
192 if ( loopFlag )
193 {
194 BitSet( m_control, AC_LOOP );
195 }
196 else
197 {
198 BitClear( m_control, AC_LOOP );
199 }
200 }
201
202 if ( wasLoopCountOverriden() )
203 {
204 xfer->xferInt( &m_loopCount );
205 }
206
207 if ( wasVolumeOverriden() )
208 {
209 xfer->xferReal( &m_volume );
210 }
211
212 if ( wasMinVolumeOverriden() )
213 {
214 xfer->xferReal( &m_minVolume );
215 }
216
217 if ( wasMinRangeOverriden() )
218 {
219 xfer->xferReal( &m_minDistance );
220 }
221
222 if ( wasMaxRangeOverriden() )
223 {
224 xfer->xferReal( &m_maxDistance );
225 }
226
227 if ( wasPriorityOverriden() )
228 {
229 UnsignedByte priority = (UnsignedByte)m_priority;
230 xfer->xferUnsignedByte( &priority );
231 m_priority = (AudioPriority)priority;
232 }
233}
