25void CButtonShowColor::OnPaint()
26{
27 try {
28 CPaintDC paintDC(this);
29
30 CPen pen(PS_SOLID, 1, 0xFFFFFF00);
31 CBrush brush(RGBtoBGR(m_color.getAsInt()));
32 // Select my stuff
33 CPen *oldPen = paintDC.SelectObject(&pen);
34
35 CRect rect;
36 GetWindowRect(&rect);
37 ScreenToClient(&rect);
38
39 paintDC.FillRect(&rect, &brush);
40 paintDC.MoveTo(rect.left, rect.top);
41 paintDC.LineTo(rect.right, rect.top);
42 paintDC.LineTo(rect.right, rect.bottom);
43 paintDC.LineTo(rect.left, rect.bottom);
44 paintDC.LineTo(rect.left, rect.top);
45
46 // Restore the states.
47 paintDC.SelectObject(oldPen);
48
49 } catch (...) {
50 // Unlikely, but possible.
51 return;
52 }
53}

60// Convert from 0x00RRGGBB to 0x00BBGGRR
61COLORREF CButtonShowColor::RGBtoBGR(Int color)
62{
63 return ((color & 0x00FF0000) >> 16 |
64 (color & 0x0000FF00) << 0 |
65 (color & 0x000000FF) << 16);
66}
