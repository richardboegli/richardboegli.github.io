153void HelixContain::createPayload()
154{
155 HelixContainModuleData* self = (HelixContainModuleData*)getHelixContainModuleData();
156
157
158 // Any number of different passengers can be loaded here at init time
159 Object* object = getObject();
160 ContainModuleInterface *contain = object->getContain();
161 if( contain )
162 {
163 contain->enableLoadSounds( FALSE );
164
165 TemplateNameList list = self->m_payloadTemplateNameData;
166 TemplateNameIterator iter = list.begin();
167 while ( iter != list.end() )
168 {
169 const ThingTemplate* temp = TheThingFactory->findTemplate( *iter );
170 if (temp)
171 {
172 Object* payload = TheThingFactory->newObject( temp, object->getTeam() );
173
174 if( contain->isValidContainerFor( payload, true ) )
175 {
176 contain->addToContain( payload );
177 }
178 else
179 {
180 DEBUG_CRASH( ( "HelixContain::createPayload: %s is full, or not valid for the payload %s!", object->getName().str(), self->m_initialPayload.name.str() ) );
181 }
182
183 }
184
185 ++iter;
186 }
187
188 contain->enableLoadSounds( TRUE );
189
190 } // endif contain
191
192 m_payloadCreated = TRUE;
193
194}

364Bool HelixContain::isPassengerAllowedToFire( ObjectID id ) const
365{
366 // WHETHER WE ARE ALLOWED TO FIRE DEPENDS ON WHO WE ARE
367 // PASSENGERS (PROPER) MAY ONLY IF THE FLAG IS TRUE
368 // RIDERS ARE ALWAYS ALLOWED TO FIRE (GATTLING CANNONS)
369
370 if ( getObject() && getObject()->getContainedBy() ) // nested containment voids firing, always
371 return FALSE;
372
373 if ( m_portableStructureID != INVALID_ID && m_portableStructureID == id )
374 return TRUE;
375 else
376 {
377 const Object *rider = TheGameLogic->findObjectByID( id );
378 if ( rider && rider->isKindOf( KINDOF_INFANTRY ))
379 return TransportContain::isPassengerAllowedToFire( id );//extend
380 }
381
382 return FALSE;
383
384}
