114//-----------------------------------------------------------------------------
115static Real calcDotProduct(const Coord3D& a, const Coord3D& b)
116{
117 return a.x*b.x + a.y*b.y + a.z*b.z;
118}
119
120//-----------------------------------------------------------------------------
121static Real calcDistSquared(const Coord3D& a, const Coord3D& b)
122{
123 return sqr(a.x - b.x) + sqr(a.y - b.y) + sqr(a.z - b.z);
124}
125
126//-----------------------------------------------------------------------------
127static Real calcPointToLineDistSquared(const Coord3D& pt, const Coord3D& lineStart, const Coord3D& lineEnd)
128{
129 Coord3D line, lineToPt, closest;
130
131 line.x = lineEnd.x - lineStart.x;
132 line.y = lineEnd.y - lineStart.y;
133 line.z = lineEnd.z - lineStart.z;
134
135 lineToPt.x = pt.x - lineStart.x;
136 lineToPt.y = pt.y - lineStart.y;
137 lineToPt.z = pt.z - lineStart.z;
138
139 Real dot = calcDotProduct(lineToPt, line);
140 if (dot <= 0.0f)
141 {
142 // angle is obtuse
143 return calcDistSquared(pt, lineStart);
144 }
145
146 Real lineLenSqr = calcDistSquared(lineStart, lineEnd);
147 DEBUG_ASSERTCRASH(lineLenSqr==calcDotProduct(line,line),("hmm"));
148 if (lineLenSqr <= dot)
149 {
150 return calcDistSquared(pt, lineEnd);
151 }
152
153 Real tmp = dot / lineLenSqr;
154
155 closest.x = lineStart.x + tmp * line.x;
156 closest.y = lineStart.y + tmp * line.y;
157 closest.z = lineStart.z + tmp * line.z;
158
159 return calcDistSquared(pt, closest);
160}

173// given an object with this geom, located at 'pos', and another obj with the given
174// pos & geom, calc the min and max pitches from this to that.
175void GeometryInfo::calcPitches(const Coord3D& thisPos, const GeometryInfo& that, const Coord3D& thatPos,
176 Real& minPitch, Real& maxPitch) const
177{
178 Coord3D thisCenter;
179 getCenterPosition(thisPos, thisCenter);
180
181 Real dxy = sqrt(sqr(thatPos.x - thisCenter.x) + sqr(thatPos.y - thisCenter.y));
182
183 Real dz;
184
187 dz = (thatPos.z + that.getMaxHeightAbovePosition()) - thisCenter.z;
188 maxPitch = atan2(dz, dxy);
189
190 dz = (thatPos.z - that.getMaxHeightBelowPosition()) - thisCenter.z;
191 minPitch = atan2(dz, dxy);
192}

271void GeometryInfo::get2DBounds(const Coord3D& geomCenter, Real angle, Region2D& bounds) const
272{
273 switch(m_type)
274 {
275 case GEOMETRY_SPHERE:
276 case GEOMETRY_CYLINDER:
277 {
278 bounds.lo.x = geomCenter.x - m_majorRadius;
279 bounds.lo.y = geomCenter.y - m_majorRadius;
280 bounds.hi.x = geomCenter.x + m_majorRadius;
281 bounds.hi.y = geomCenter.y + m_majorRadius;
282 break;
283 }
284
285 case GEOMETRY_BOX:
286 {
287 Real c = (Real)cos(angle);
288 Real s = (Real)sin(angle);
289 Real exc = m_majorRadius*c;
290 Real eyc = m_minorRadius*c;
291 Real exs = m_majorRadius*s;
292 Real eys = m_minorRadius*s;
293 Real x,y;
294 x = geomCenter.x - exc - eys;
295 y = geomCenter.y + eyc - exs;
296 bounds.lo.x = x;
297 bounds.lo.y = y;
298 bounds.hi.x = x;
299 bounds.hi.y = y;
300
301 x = geomCenter.x + exc - eys;
302 y = geomCenter.y + eyc + exs;
303 if (bounds.lo.x > x) bounds.lo.x = x;
304 if (bounds.lo.y > y) bounds.lo.y = y;
305 if (bounds.hi.x < x) bounds.hi.x = x;
306 if (bounds.hi.y < y) bounds.hi.y = y;
307
308 x = geomCenter.x + exc + eys;
309 y = geomCenter.y - eyc + exs;
310 if (bounds.lo.x > x) bounds.lo.x = x;
311 if (bounds.lo.y > y) bounds.lo.y = y;
312 if (bounds.hi.x < x) bounds.hi.x = x;
313 if (bounds.hi.y < y) bounds.hi.y = y;
314
315 x = geomCenter.x - exc + eys;
316 y = geomCenter.y - eyc - exs;
317 if (bounds.lo.x > x) bounds.lo.x = x;
318 if (bounds.lo.y > y) bounds.lo.y = y;
319 if (bounds.hi.x < x) bounds.hi.x = x;
320 if (bounds.hi.y < y) bounds.hi.y = y;
321
322 break;
323 }
324 }
325}

328void GeometryInfo::clipPointToFootprint(const Coord3D& geomCenter, Coord3D& ptToClip) const
329{
330 switch(m_type)
331 {
332 case GEOMETRY_SPHERE:
333 case GEOMETRY_CYLINDER:
334 {
335 Real dx = ptToClip.x - geomCenter.x;
336 Real dy = ptToClip.y - geomCenter.y;
337 Real radius = sqrt(sqr(dx) + sqr(dy));
338 if (radius > m_majorRadius)
339 {
340 Real ratio = m_majorRadius / radius;
341 ptToClip.x = geomCenter.x + dx * ratio;
342 ptToClip.y = geomCenter.y + dy * ratio;
343 }
344 break;
345 }
346
347 case GEOMETRY_BOX:
348 {
349 ptToClip.x = clamp(geomCenter.x - m_majorRadius, ptToClip.x, geomCenter.x + m_majorRadius);
350 ptToClip.y = clamp(geomCenter.y - m_minorRadius, ptToClip.y, geomCenter.y + m_minorRadius);
351 break;
352 }
353 };
354}

359//=============================================================================
360Bool GeometryInfo::isPointInFootprint(const Coord3D& geomCenter, const Coord3D& pt) const
361{
362 switch(m_type)
363 {
364 case GEOMETRY_SPHERE:
365 case GEOMETRY_CYLINDER:
366 {
367 Real dx = pt.x - geomCenter.x;
368 Real dy = pt.y - geomCenter.y;
369 Real radius = sqrt(sqr(dx) + sqr(dy));
370 return (radius <= m_majorRadius);
371 break;
372 }
373
374 case GEOMETRY_BOX:
375 {
376 return isWithin(geomCenter.x - m_majorRadius, pt.x, geomCenter.x + m_majorRadius) &&
377 isWithin(geomCenter.y - m_minorRadius, pt.y, geomCenter.y + m_minorRadius);
378 }
379 };
380 return false;
381}

384void GeometryInfo::makeRandomOffsetWithinFootprint(Coord3D& pt) const
385{
386 switch(m_type)
387 {
388 case GEOMETRY_SPHERE:
389 case GEOMETRY_CYLINDER:
390 {
391#if 1
392 // this is a better technique than the more obvious radius-and-angle
393 // one, below, because the latter tends to clump more towards the center.
394 Real maxDistSqr = sqr(m_majorRadius);
395 Real distSqr;
396 do
397 {
398 pt.x = GameLogicRandomValueReal(-m_majorRadius, m_majorRadius);
399 pt.y = GameLogicRandomValueReal(-m_majorRadius, m_majorRadius);
400 pt.z = 0.0f;
401 distSqr = sqr(pt.x) + sqr(pt.y);
402 } while (distSqr > maxDistSqr);
403#else
404 Real radius = GameLogicRandomValueReal(0.0f, m_boundingCircleRadius);
405 Real angle = GameLogicRandomValueReal(-PI, PI);
406 pt.x = radius * Cos(angle);
407 pt.y = radius * Sin(angle);
408 pt.z = 0.0f;
409#endif
410 break;
411 }
412
413 case GEOMETRY_BOX:
414 {
415 pt.x = GameLogicRandomValueReal(-m_majorRadius, m_majorRadius);
416 pt.y = GameLogicRandomValueReal(-m_minorRadius, m_minorRadius);
417 pt.z = 0.0f;
418 break;
419 }
420 };
421}

423//=============================================================================
424void GeometryInfo::makeRandomOffsetOnPerimeter(Coord3D& pt) const
425{
426 switch(m_type)
427 {
428 case GEOMETRY_SPHERE:
429 case GEOMETRY_CYLINDER:
430 {
431 DEBUG_CRASH( ("GeometryInfo::makeRandomOffsetOnPerimeter() not implemented for SPHERE or CYLINDER extents. Using position.") );
432
433 //Kris: Did not have time nor need to support non-box extents. I added this feature for script placement
434 // of boobytraps.
435 pt.x = 0.0f;
436 pt.y = 0.0f;
437 break;
438 }
439
440 case GEOMETRY_BOX:
441 {
442 if( GameLogicRandomValueReal( 0.0f, 1.0f ) < 0.5f )
443 {
444 //Pick random point on x axis.
445 pt.x = GameLogicRandomValueReal(-m_majorRadius, m_majorRadius);
446
447 //Min or max the y axis value
448 if( GameLogicRandomValueReal( 0.0f, 1.0f ) < 0.5f )
449 pt.y = -m_minorRadius;
450 else
451 pt.y = m_minorRadius;
452 }
453 else
454 {
455 //Pick random point on y axis.
456 pt.y = GameLogicRandomValueReal(-m_minorRadius, m_minorRadius);
457
458 //Min or max the x axis value
459 if( GameLogicRandomValueReal( 0.0f, 1.0f ) < 0.5f )
460 pt.x = -m_majorRadius;
461 else
462 pt.x = m_majorRadius;
463 }
464 pt.z = 0.0f;
465 break;
466 }
467 };
468}

491//=============================================================================
492void GeometryInfo::calcBoundingStuff()
493{
494 switch(m_type)
495 {
496 case GEOMETRY_SPHERE:
497 {
498 m_boundingSphereRadius = m_majorRadius;
499 m_boundingCircleRadius = m_majorRadius;
500 break;
501 }
502 case GEOMETRY_CYLINDER:
503 {
504 m_boundingCircleRadius = m_majorRadius;
505
506 m_boundingSphereRadius = m_height*0.5;
507 if (m_boundingSphereRadius < m_majorRadius)
508 m_boundingSphereRadius = m_majorRadius;
509 break;
510 }
511
512 case GEOMETRY_BOX:
513 {
514 m_boundingCircleRadius = sqrt(sqr(m_majorRadius) + sqr(m_minorRadius));
515 m_boundingSphereRadius = sqrt(sqr(m_majorRadius) + sqr(m_minorRadius) + sqr(m_height*0.5));
516 break;
517 }
518 };
519}
