91static void parseOCL( INI* ini, void *instance, void * /*store*/, const void* /*userData*/ )
92{
93 StructureToppleUpdateModuleData* self = (StructureToppleUpdateModuleData*)instance;
94 StructureTopplePhaseType stphase = (StructureTopplePhaseType)INI::scanIndexList(ini->getNextToken(), TheStructureTopplePhaseNames);
95 for (const char* token = ini->getNextToken(); token != NULL; token = ini->getNextTokenOrNull())
96 {
97 const ObjectCreationList *ocl = TheObjectCreationListStore->findObjectCreationList(token); // could be null! this is OK!
98 self->m_ocls[stphase].push_back(ocl);
99 }

103static void parseAngleFX(INI* ini, void *instance, void * /* store */, const void * /*userData*/)
104{
105 StructureToppleUpdateModuleData* self = (StructureToppleUpdateModuleData*)instance;
106 AngleFXInfo info;
107 INI::parseReal(ini, instance, &(info.angle), NULL);
108 info.angle = info.angle * PI / 180.0f; // convert from degrees to radians.
109 INI::parseFXList(ini, instance, &(info.fxList), NULL);
110 self->angleFX.push_back(info);

143void StructureToppleUpdate::beginStructureTopple(const DamageInfo *damageInfo)
144{
145 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();
146
147 if (d)
148 {
149 UnsignedInt now = TheGameLogic->getFrame();
150 m_toppleFrame = now + GameLogicRandomValue(d->m_minToppleDelay, d->m_maxToppleDelay);
151
152 Object *attacker = TheGameLogic->findObjectByID(damageInfo->in.m_sourceID);
153 Object *building = getObject();
154
155 Real toppleAngle = 0.0;
156 if (attacker == NULL) {
157 toppleAngle = GameLogicRandomValueReal(0.0, 2*PI);
158 } else {
159 const Coord3D *attackerPos = attacker->getPosition();
160 const Coord3D *buildingPos = building->getPosition();
161
162 // Calculate the topple direction to be the opposite of the direction fired from.
163 m_toppleDirection.x = buildingPos->x - attackerPos->x;
164 m_toppleDirection.y = buildingPos->y - attackerPos->y;
165
166 // Give it a little randomness...
167 toppleAngle = m_toppleDirection.toAngle();
168 toppleAngle += GameLogicRandomValueReal(-PI/8, PI/8);
169 }
170 m_toppleDirection.x = Cos(toppleAngle);
171 m_toppleDirection.y = Sin(toppleAngle);
172 TheScriptEngine->adjustToppleDirection(getObject(), &m_toppleDirection);
173
174 Real averageRadius = (building->getGeometryInfo().getMajorRadius() + building->getGeometryInfo().getMinorRadius()) / 2;
175 Real explosionRadius = averageRadius * 0.90;
176
177 m_delayBurstLocation.x = building->getPosition()->x + explosionRadius * Cos(toppleAngle);
178 m_delayBurstLocation.y = building->getPosition()->y + explosionRadius * Sin(toppleAngle);
179 m_delayBurstLocation.z = TheTerrainLogic->getGroundHeight(m_delayBurstLocation.x, m_delayBurstLocation.y);
180
181 doToppleStartFX(building, damageInfo);
182 m_nextBurstFrame = now + GameClientRandomValue(d->m_minToppleBurstDelay, d->m_maxToppleBurstDelay);
183
184 m_toppleState = TOPPLESTATE_WAITINGFORTOPPLESTART;
185
186 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
187 }
188}

192void StructureToppleUpdate::onDie( const DamageInfo *damageInfo )
193{
194 const StructureToppleUpdateModuleData* d = getStructureToppleUpdateModuleData();
195 if (!d->m_dieMuxData.isDieApplicable(getObject(), damageInfo))
196 return;
197
198 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
199 if (ai)
200 ai->markAsDead();
201
202 // Deselect the object for all players.
203 TheGameLogic->deselectObject(getObject(), PLAYERMASK_ALL, TRUE);
204
205 beginStructureTopple(damageInfo);
206}

210UpdateSleepTime StructureToppleUpdate::update( void )
211{
212 static const Real TOPPLE_ACCELERATION_FACTOR = 0.02f;
213
214 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();
215
216 if (m_toppleState == TOPPLESTATE_STANDING)
217 {
218 DEBUG_CRASH(("hmm, what?"));
219 return UPDATE_SLEEP_FOREVER;
220 }
221
222 // get last damage info
223 const DamageInfo *lastDamageInfo = getObject()->getBodyModule()->getLastDamageInfo();
224
225 // We are in the dramatic pause between when the building has lost all its hit points and
226 // when it starts toppling over.
227 if (m_toppleState == TOPPLESTATE_WAITINGFORTOPPLESTART) {
228 UnsignedInt now = TheGameLogic->getFrame();
229 if (now >= m_nextBurstFrame) {
230 doToppleDelayBurstFX();
231 // This uses a game client random value because the delay bursts are purely visual and aural effects.
232 m_nextBurstFrame = now + GameClientRandomValue(d->m_minToppleBurstDelay, d->m_maxToppleBurstDelay);
233 }
234 if (now >= m_toppleFrame) {
235 m_toppleState = TOPPLESTATE_TOPPLING;
236 m_structuralIntegrity = d->m_structuralIntegrity;
237 }
238 }
239
240 // The building is in the process of falling over.
241 if (m_toppleState == TOPPLESTATE_TOPPLING) {
242 UnsignedInt now = TheGameLogic->getFrame();
243 Real toppleAcceleration = TOPPLE_ACCELERATION_FACTOR * (Sin(m_accumulatedAngle) * (1.0 - m_structuralIntegrity));
244// DEBUG_LOG(("toppleAcceleration = %f\n", toppleAcceleration));
245 m_toppleVelocity += toppleAcceleration;
246// DEBUG_LOG(("m_toppleVelocity = %f\n", m_toppleVelocity));
247
248 // doesn't make sense to have a structural integrity less than zero.
249 if (m_structuralIntegrity > 0.0f) {
250 m_structuralIntegrity *= d->m_structuralDecay;
251 if (m_structuralIntegrity < 0.0f) {
252 m_structuralIntegrity = 0.0f;
253 }
254 }
255// DEBUG_LOG(("m_structuralIntegrity = %f\n\n", m_structuralIntegrity));
256
257 doAngleFX(m_accumulatedAngle, m_accumulatedAngle + m_toppleVelocity);
258
259 m_accumulatedAngle += m_toppleVelocity;
260
261 applyCrushingDamage(PI/2 - m_accumulatedAngle);
262
263 if (m_accumulatedAngle >= PI/2) {
264 m_toppleVelocity -= m_accumulatedAngle - PI/2;
265 m_accumulatedAngle = PI/2;
266 m_toppleState = TOPPLESTATE_WAITINGFORDONE;
267
268 applyCrushingDamage(0.0f);
269 doPhaseStuff(STPHASE_FINAL, getObject()->getPosition());
270
271 if( lastDamageInfo == NULL || getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
272 FXList::doFXObj(d->m_toppleDoneFXList, getObject());
273
274 m_toppleFrame = TheGameLogic->getFrame();
275 }
276
277 if (now >= m_nextBurstFrame) {
278 doToppleDelayBurstFX();
279 // This uses a game client random value because the delay bursts are purely visual and aural effects.
280 m_nextBurstFrame = now + GameClientRandomValue(d->m_minToppleBurstDelay, d->m_maxToppleBurstDelay);
281 }
282
283 Object *building = getObject();
284 Matrix3D xfrm = *building->getTransformMatrix();
285 xfrm.In_Place_Pre_Rotate_X(-m_toppleVelocity * m_toppleDirection.y);
286 xfrm.In_Place_Pre_Rotate_Y(m_toppleVelocity * m_toppleDirection.x);
287 building->setTransformMatrix(&xfrm);
288 }
289
290 // The building is now flat on the ground and done with all the crushing and all that.
291 if (m_toppleState == TOPPLESTATE_WAITINGFORDONE)
292 {
293 if (m_toppleFrame <= TheGameLogic->getFrame())
294 {
295 Object *building = getObject();
296 Drawable *drawable = building->getDrawable();
297 drawable->clearModelConditionState(MODELCONDITION_RUBBLE);
298 drawable->setModelConditionState(MODELCONDITION_POST_COLLAPSE);
299
300
301 // Need to update body particle systems, now
302 BodyModuleInterface *body = building->getBodyModule();
303 body->updateBodyParticleSystems();
304
305 doToppleDoneStuff();
306
307 m_toppleState = TOPPLESTATE_DONE;
308
309 return UPDATE_SLEEP_FOREVER;
310 }
311 }
312
313 return UPDATE_SLEEP_NONE;
314}

318void StructureToppleUpdate::doToppleDoneStuff()
319{
320 static NameKeyType key_BoneFXUpdate = NAMEKEY("BoneFXUpdate");
321 BoneFXUpdate *bfxu = (BoneFXUpdate *)getObject()->findUpdateModule(key_BoneFXUpdate);
322 if (bfxu != NULL) {
323 bfxu->stopAllBoneFX();
324 }
325
326 Object *building = getObject();
327
328 Real origAngle = building->getOrientation();
329 building->setOrientation(origAngle);
330
331 Real toppleAngle = m_toppleDirection.toAngle();
332
333 Matrix3D xfrm = *building->getTransformMatrix();
334 xfrm.In_Place_Pre_Rotate_Z(toppleAngle-origAngle);
335 building->setTransformMatrix(&xfrm);
336}

341{
342 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();

358// theta is the angle of the building with respect to the ground.
359void StructureToppleUpdate::applyCrushingDamage(Real theta)
360{
361// static const Real THETA_CEILING = PI/8; // This weapon won't do any damage until theta is less than this value.
362 static const Real THETA_CEILING = PI/6; // This weapon won't do any damage until theta is less than this value.
363 static const Real WEAPON_SPACING_PERPENDICULAR = 25; // The spacing between weapon firing locations,
364 // distance is perpendicular to the direction the
365 // building is falling in.
366
367 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();
368 if (theta > THETA_CEILING) {
369 return;
370 }
371
372 Object *building = getObject();
373 Real orientationAngle = building->getOrientation();
374 Real toppleAngle = m_toppleDirection.toAngle();
375
376 // Figure out the width of the projection of the boundary of the building along the topple direction.
377 // Do this because the amount of ground that is affected will be different if the building falls
378 // in different orientations.
379 Real angle = orientationAngle - toppleAngle;
380 Real minorComponent = building->getGeometryInfo().getMinorRadius() * Cos(angle);
381 Real majorComponent = building->getGeometryInfo().getMajorRadius() * Sin(angle);
382
383 Coord3D temp3D;
384 temp3D.x = majorComponent;
385 temp3D.y = minorComponent;
386 temp3D.z = 0.0f;
387
388 Real facingWidth = temp3D.length() / 2;
389
390 // Get the crushing weapon.
391 const WeaponTemplate* wt = TheWeaponStore->findWeaponTemplate(d->m_crushingWeaponName);
392 if (wt == NULL) {
393 return;
394 }
395
396 // The furthest away from the base of the building to explode on.
397 Real maxDistance = m_buildingHeight * (1.0 - Sin(theta));
398
399 /*
400 * Fire explosions at regular intervals across the area that the building is currently
401 * crushing. The explosions occur across the face and along the length of the building.
402 */
403 Real jcos;
404 Real jsin;
405// Coord3D target;
406 for (Real j = m_lastCrushedLocation; j < maxDistance; j += WEAPON_SPACING_PERPENDICULAR) {
407 jcos = j * Cos(toppleAngle);
408 jsin = j * Sin(toppleAngle);
409 doDamageLine(building, wt, jcos, jsin, facingWidth, toppleAngle);
410 }
411
412
413 jcos = maxDistance * Cos(toppleAngle);
414 jsin = maxDistance * Sin(toppleAngle);
415 doDamageLine(building, wt, jcos, jsin, facingWidth, toppleAngle);
416
417 m_lastCrushedLocation = j;
418}

421//-------------------------------------------------------------------------------------------------
422void StructureToppleUpdate::doDamageLine(Object *building, const WeaponTemplate* wt, Real jcos, Real jsin, Real facingWidth, Real toppleAngle)
423{
424 const DamageInfo *lastDamageInfo = getObject()->getBodyModule()->getLastDamageInfo();
425 static const Real WEAPON_SPACING_PARALLEL = 25; // The spacing between weapon firing locations,
426 // distance is parallel to the direction the building
427 // is falling in.
428
429 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();
430
431 Coord3D target;
432
433 for (Real i = -facingWidth; i < facingWidth; i += WEAPON_SPACING_PARALLEL)
434 {
435 target.x = building->getPosition()->x + jcos + (i * Sin(toppleAngle));
436 target.y = building->getPosition()->y + jsin + (i * Cos(toppleAngle));
437 target.z = TheTerrainLogic->getGroundHeight(target.x, target.y);
438
439 TheWeaponStore->createAndFireTempWeapon(wt, building, &target);
440
441 // do the crushing particle effects
442 if( lastDamageInfo == NULL || getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
443 FXList::doFXPos(d->m_crushingFXList, &target);
444 }
445
446 // Make sure there are weapons fired and FX done on the edge of the building.
447 target.x = building->getPosition()->x + jcos + (facingWidth * Sin(toppleAngle));
448 target.y = building->getPosition()->y + jsin + (facingWidth * Cos(toppleAngle));
449 target.z = TheTerrainLogic->getGroundHeight(target.x, target.y);
450
451 TheWeaponStore->createAndFireTempWeapon(wt, building, &target);
452
453 // do the crushing particle effects
454 if( lastDamageInfo == NULL || getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
455 FXList::doFXPos(d->m_crushingFXList, &target);
456
457 // Do the flying debris for this line.
458 target.x = building->getPosition()->x + jcos;
459 target.y = building->getPosition()->y + jsin;
460 target.z = TheTerrainLogic->getGroundHeight(target.x, target.y);
461
462 doPhaseStuff(STPHASE_FINAL, &target);
463}

480void StructureToppleUpdate::doToppleDelayBurstFX()
481{
482 const StructureToppleUpdateModuleData *d = getStructureToppleUpdateModuleData();
483 const DamageInfo *lastDamageInfo = getObject()->getBodyModule()->getLastDamageInfo();
484
485 DEBUG_LOG(("Doing topple delay burst on frame %d\n", TheGameLogic->getFrame()));
486 if( lastDamageInfo == NULL || getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
487 FXList::doFXPos(d->m_toppleDelayFXList, &m_delayBurstLocation);
488
489 Object *building = getObject();
490 Drawable *drawable = building->getDrawable();
491
492 if( lastDamageInfo == NULL || getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) )
493 {
494
495 for (std::vector<FXBoneInfo>::const_iterator it = d->fxbones.begin(); it != d->fxbones.end(); ++it)
496 {
497 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem(it->particleSystemTemplate);
498 if (sys != NULL)
499 {
500 Coord3D pos;
501 if (drawable->getPristineBonePositions(it->boneName.str(), 0, &pos, NULL, 1) == 1)
502 {
503 // got the bone position...
504 sys->setPosition(&pos);
505
506 // Attatch it to the object...
507 sys->attachToDrawable(drawable);
508 }
509 }
510 }
511 }
512
513 doPhaseStuff(STPHASE_DELAY, &m_delayBurstLocation);
514
515}

532{
533 for (Int i = 0; i < count; ++i)
534 {
535 Int idx;

538 idx = GameLogicRandomValue(0, range-1);
539 }
540 while (inList(idx, i, idxList));

547void StructureToppleUpdate::doPhaseStuff(StructureTopplePhaseType stphase, const Coord3D *target)
548{
549 const StructureToppleUpdateModuleData* d = getStructureToppleUpdateModuleData();
550 Int i, idx, count, listSize;
551 Int idxList[MAX_IDX];
552
553 listSize = d->m_ocls[stphase].size();
554 if (listSize > 0)
555 {
556 count = d->m_oclCount[stphase];
557 buildNonDupRandomIndexList(listSize, count, idxList);
558 for (i = 0; i < count; ++i)
559 {
560 idx = idxList[i];
561 const OCLVec& v = d->m_ocls[stphase];
562 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
563 const ObjectCreationList* ocl = v[idx];
564 ObjectCreationList::create(ocl, getObject(), target, NULL, INVALID_ANGLE );
565 }
566 }
