86static void parseFX( INI* ini, void *instance, void * /*store*/, const void* /*userData*/ )
87{
88 SlowDeathBehaviorModuleData* self = (SlowDeathBehaviorModuleData*)instance;
89 SlowDeathPhaseType sdphase = (SlowDeathPhaseType)INI::scanIndexList(ini->getNextToken(), TheSlowDeathPhaseNames);
90 for (const char* token = ini->getNextToken(); token != NULL; token = ini->getNextTokenOrNull())
91 {
92 const FXList *fxl = TheFXListStore->findFXList((token)); // could be null! this is OK!
93 self->m_fx[sdphase].push_back(fxl);
94 if (fxl)
95 self->m_maskOfLoadedEffects |= SlowDeathBehaviorModuleData::HAS_FX;
96 }
97}

182Int SlowDeathBehavior::getProbabilityModifier( const DamageInfo *damageInfo ) const
183{
184 // Calculating how far past dead we were allows us to pick more spectacular deaths when
185 // severly killed, and more sedate ones when only slightly killed.
186 // eg ( 200 hp max, had 10 left, took 50 damage, 40 overkill, (40/200) * 100 = 20 overkill %)
187 Int overkillDamage = damageInfo->out.m_actualDamageDealt - damageInfo->out.m_actualDamageClipped;
188 Real overkillPercent = (float)overkillDamage / (float)getObject()->getBodyModule()->getMaxHealth();
189 Int overkillModifier = overkillPercent * getSlowDeathBehaviorModuleData()->m_modifierBonusPerOverkillPercent;
190
191 return max( getSlowDeathBehaviorModuleData()->m_probabilityModifier + overkillModifier, 1 );
192}

195static void calcRandomForce(Real minMag, Real maxMag, Real minPitch, Real maxPitch, Coord3D& force)
196{
197 Real angle = GameLogicRandomValueReal(-PI, PI);
198 Real pitch = GameLogicRandomValueReal(minPitch, maxPitch);
199 Real mag = GameLogicRandomValueReal(minMag, maxMag);
200
201 Matrix3D mtx(1);
202 mtx.Scale(mag);
203 mtx.Rotate_Z(angle);
204 mtx.Rotate_Y(-pitch);
205
206 Vector3 v = mtx.Get_X_Vector();
207
208 force.x = v.X;
209 force.y = v.Y;
210 force.z = v.Z;
211}

215void SlowDeathBehavior::beginSlowDeath(const DamageInfo *damageInfo)
216{
217 if (!isSlowDeathActivated())
218 {
219 const SlowDeathBehaviorModuleData* d = getSlowDeathBehaviorModuleData();
220 Object* obj = getObject();
221
222 if (d->m_sinkRate && obj->isKindOf(KINDOF_INFANTRY))
223 {
224
225 Drawable *draw = getObject()->getDrawable();
226 if ( draw )
227 {
228 // this object sinks slowly after it dies so don't draw a
229 // floating shadow decal on the ground above it.
230 obj->getDrawable()->setShadowsEnabled(false);
231 draw->setTerrainDecalFadeTarget( 0.0f, -0.2f );
232 }
233
234
235
236 }
237
238
239 // Ask game detail manager if we need to speedup all deaths to improve performance
240 Real timeScale = TheGameLODManager->getSlowDeathScale();
241 m_acceleratedTimeScale = 1.0f; // assume normal death speed.
242
243 if (timeScale == 0.0f && !d->hasNonLodEffects())
244 {
245 // Deaths happen instantly so just delete the object and return
246 TheGameLogic->destroyObject(obj);
247 return;
248 }
249 else
250 {
251 // timescale is some non-zero value so we may need to speed up death
252 if( getObject()->isKindOf( KINDOF_HULK ) && TheGameLogic->getHulkMaxLifetimeOverride() != -1 )
253 {
254 //Scripts don't want hulks around, so start sinking immediately!
255 m_sinkFrame = 1;
256 m_midpointFrame = (LOGICFRAMES_PER_SECOND/2) + 1;
257 m_destructionFrame = LOGICFRAMES_PER_SECOND + 1;
258 m_acceleratedTimeScale = 1.0f;
259 }
260 else
261 {
262 m_sinkFrame = timeScale * (d->m_sinkDelay + GameLogicRandomValue(0, d->m_sinkDelayVariance));
263 m_destructionFrame = timeScale * (d->m_destructionDelay + GameLogicRandomValue(0, d->m_destructionDelayVariance));
264 m_midpointFrame = GameLogicRandomValue( BEGIN_MIDPOINT_RATIO * m_destructionFrame, END_MIDPOINT_RATIO * m_destructionFrame );
265 m_acceleratedTimeScale = timeScale;
266 }
267 }
268
269 UnsignedInt now = TheGameLogic->getFrame();
270
271 if (d->m_flingForce > 0)
272 {
273
274
275 //Just in case this is a stingersoldier or other HELD object, lets set them free so they will fly
276 // with their own physics during slow death
277 if( obj->isDisabledByType( DISABLED_HELD ) )
278 {
279 static NameKeyType key_SlavedUpdate = NAMEKEY( "SlavedUpdate" );
280 SlavedUpdate* slave = (SlavedUpdate*)obj->findUpdateModule( key_SlavedUpdate );
281 if( slave )
282 {
283 slave->onSlaverDie( NULL );
284 }
285 }
286
287 PhysicsBehavior* physics = obj->getPhysics();
288 if (physics)
289 {
290 // make sure we are at least a bit above the ground
291 const Real MIN_ALTITUDE = 1.0f;
292 Real altitude = obj->getHeightAboveTerrain();
293 if (altitude < MIN_ALTITUDE)
294 {
295 Coord3D pos = *obj->getPosition();
296 pos.z += MIN_ALTITUDE;
297 obj->setPosition(&pos);
298 }
299
300 Coord3D force;
301 calcRandomForce(d->m_flingForce, d->m_flingForce + d->m_flingForceVariance,
302 d->m_flingPitch, d->m_flingPitch + d->m_flingPitchVariance, force);
303 physics->setAllowToFall(true);
304 physics->applyForce(&force);
305 physics->setExtraBounciness(-1.0); // we don't want this guy to bounce at all
306 physics->setExtraFriction(-3 * SECONDS_PER_LOGICFRAME_REAL); // reduce his ground friction a bit
307 physics->setAllowBouncing(true);
308 Real orientation = atan2(force.y, force.x);
309 physics->setAngles(orientation, 0, 0);
310 obj->getDrawable()->setModelConditionState(MODELCONDITION_EXPLODED_FLAILING);
311 m_flags |= (1<<FLUNG_INTO_AIR);
312 }
313 setWakeFrame(obj, UPDATE_SLEEP_NONE);
314 }
315 else
316 {
317 // we don't need to wake up immediately, but only when the first of these
318 // counters wants to trigger....
319 Int whenToWakeTime = m_sinkFrame;
320 if (whenToWakeTime > m_destructionFrame)
321 whenToWakeTime = m_destructionFrame;
322 if (whenToWakeTime > m_midpointFrame)
323 whenToWakeTime = m_midpointFrame;
324 setWakeFrame(obj, UPDATE_SLEEP(whenToWakeTime));
325 }
326 m_sinkFrame += now;
327 m_destructionFrame += now;
328 m_midpointFrame += now;
329
330 m_flags |= (1<<SLOW_DEATH_ACTIVATED);
331
332 doPhaseStuff(SDPHASE_INITIAL);
333
334 }
335}

339void SlowDeathBehavior::doPhaseStuff(SlowDeathPhaseType sdphase)
340{
341 const SlowDeathBehaviorModuleData* d = getSlowDeathBehaviorModuleData();
342 Int idx, listSize;
343
344 if (!d->m_maskOfLoadedEffects)
345 return; //has no ocl, fx, or weapons.
346
347 listSize = d->m_fx[sdphase].size();
348 if (listSize > 0)
349 {
350 idx = GameLogicRandomValue(0, listSize-1);
351 const FXListVec& v = d->m_fx[sdphase];
352 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
353 const FXList* fxl = v[idx];
354 FXList::doFXObj(fxl, getObject(), NULL);
355 }
356
357 listSize = d->m_ocls[sdphase].size();
358 if (listSize > 0)
359 {
360 idx = GameLogicRandomValue(0, listSize-1);
361 const OCLVec& v = d->m_ocls[sdphase];
362 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
363 const ObjectCreationList* ocl = v[idx];
364 ObjectCreationList::create(ocl, getObject(), NULL);
365 }
366
367 listSize = d->m_weapons[sdphase].size();
368 if (listSize > 0)
369 {
370 idx = GameLogicRandomValue(0, listSize-1);
371 const WeaponTemplateVec& v = d->m_weapons[sdphase];
372 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
373 const WeaponTemplate* wt = v[idx];
374 if (wt)
375 {
376 TheWeaponStore->createAndFireTempWeapon(wt, getObject(), getObject()->getPosition());
377 }
378 }
379}

383UpdateSleepTime SlowDeathBehavior::update()
384{
385 //DEBUG_LOG(("updating SlowDeathBehavior %08lx\n",this));
386 DEBUG_ASSERTCRASH(isSlowDeathActivated(), ("hmm, this should not be possible"));
387
388 const SlowDeathBehaviorModuleData* d = getSlowDeathBehaviorModuleData();
389 Object* obj = getObject();
390
391 Real timeScale = TheGameLODManager->getSlowDeathScale();
392
393 // Check if we have normal time scale but LODManager is requeseting acceleration
394 if (timeScale != 1.0f && m_acceleratedTimeScale == 1.0f && !d->hasNonLodEffects())
395 {
396 // speed of deaths has been increased since beginning of death
397 // so adjust it to current levels.
398 if (timeScale == 0)
399 {
400 // instant death
401 TheGameLogic->destroyObject(obj);
402 return UPDATE_SLEEP_NONE;
403 }
404
405 m_sinkFrame = (Real)m_sinkFrame * timeScale;
406 m_midpointFrame = (Real)m_midpointFrame * timeScale;
407 m_destructionFrame = (Real)m_destructionFrame * timeScale;
408 m_acceleratedTimeScale = timeScale;
409 };
410
411 UnsignedInt now = TheGameLogic->getFrame();
412
413
414 if ((m_flags & (1<<FLUNG_INTO_AIR)) != 0)
415 {
416 if ((m_flags & (1<<BOUNCED)) == 0)
417 {
418 ++m_sinkFrame;
419 ++m_midpointFrame;
420 ++m_destructionFrame;
421 if (!obj->isAboveTerrain())
422 {
423 obj->clearAndSetModelConditionFlags(MAKE_MODELCONDITION_MASK(MODELCONDITION_EXPLODED_FLAILING),
424 MAKE_MODELCONDITION_MASK(MODELCONDITION_EXPLODED_BOUNCING));
425 m_flags |= (1<<BOUNCED);
426 }
427
428 // Here we want to make sure we die if we collide with a tree on the way down
429 PhysicsBehavior *phys = obj->getPhysics();
430 if ( phys )
431 {
432 ObjectID treeID = phys->getLastCollidee();
433 Object *tree = TheGameLogic->findObjectByID( treeID );
434 if ( tree )
435 {
436 if (tree->isKindOf( KINDOF_SHRUBBERY ) )
437 {
438 obj->setDisabled( DISABLED_HELD );
439 obj->clearModelConditionFlags( MAKE_MODELCONDITION_MASK(MODELCONDITION_EXPLODED_FLAILING) );
440 obj->clearModelConditionFlags( MAKE_MODELCONDITION_MASK(MODELCONDITION_EXPLODED_BOUNCING) );
441 obj->setModelConditionFlags( MAKE_MODELCONDITION_MASK(MODELCONDITION_PARACHUTING) ); //looks like he is snagged in a tree
442 obj->setPositionZ( obj->getPosition()->z - (d->m_sinkRate * 50.0f) );// make him sink faster
443 if ( !obj->isAboveTerrain() )
444 TheGameLogic->destroyObject(obj);
445
446 }
447 }
448 }
449
450
451
452 }
453 }
454
455 if ( (now >= m_sinkFrame && d->m_sinkRate > 0.0f) )
456 {
457 // disable Physics (if any) so that we can control the sink...
458 obj->setDisabled( DISABLED_HELD );
459 Coord3D pos = *obj->getPosition();
460 pos.z -= d->m_sinkRate / m_acceleratedTimeScale;
461 obj->setPosition( &pos );
462 }
463
464 if( now >= m_midpointFrame && (m_flags & (1<<MIDPOINT_EXECUTED)) == 0 )
465 {
466 doPhaseStuff(SDPHASE_MIDPOINT);
467 m_flags |= (1<<MIDPOINT_EXECUTED);
468 }
469
470 if (now >= m_destructionFrame)
471 {
472 doPhaseStuff(SDPHASE_FINAL);
473 TheGameLogic->destroyObject(obj);
474 }
475
476 return UPDATE_SLEEP_NONE;
477}

480void SlowDeathBehavior::onDie( const DamageInfo *damageInfo )
481{
482 Object *obj = getObject();
483
484 if (!isDieApplicable(damageInfo))
485 return;
486
487 AIUpdateInterface *ai = obj->getAIUpdateInterface();
488 if (ai)
489 {
490 // has another AI already handled us. (hopefully another SlowDeathBehavior)
491 if (ai->isAiInDeadState())
492 return;
493 ai->markAsDead();
494 }
495
496 // deselect this unit for all players.
497 TheGameLogic->deselectObject(obj, PLAYERMASK_ALL, TRUE);
498
499 Int total = 0;
500 for (BehaviorModule** update = obj->getBehaviorModules(); *update; ++update)
501 {
502 SlowDeathBehaviorInterface* sdu = (*update)->getSlowDeathBehaviorInterface();
503 if (sdu != NULL && sdu->isDieApplicable(damageInfo))
504 {
505 total += sdu->getProbabilityModifier( damageInfo );
506 }
507 }
508 DEBUG_ASSERTCRASH(total > 0, ("Hmm, this is wrong"));
509
510
511 // this returns a value from 1...total, inclusive
512 Int roll = GameLogicRandomValue(1, total);
513
514 for (/* UpdateModuleInterface** */ update = obj->getBehaviorModules(); *update; ++update)
515 {
516 SlowDeathBehaviorInterface* sdu = (*update)->getSlowDeathBehaviorInterface();
517 if (sdu != NULL && sdu->isDieApplicable(damageInfo))
518 {
519 roll -= sdu->getProbabilityModifier( damageInfo );
520 if (roll <= 0)
521 {
522 sdu->beginSlowDeath(damageInfo);
523 return;
524 }
525 }
526 }
527
528 DEBUG_CRASH(("We should never get here"));
529}
