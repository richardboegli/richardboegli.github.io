105static void doMoveTo( Object *obj, const Coord3D *pos )
106{
107 AIUpdateInterface *ai = obj->getAIUpdateInterface();
108 DEBUG_ASSERTCRASH(ai, ("Attemped doMoveTo() on an Object with no AI\n"));
109 if (ai)
110 {
111 if (theBuildPlan)
112 {
113 int i;
114
115 // if this object isn't in the buildPlan set, add it
116 for( i=0; i<thePlanSubjectCount; i++ )
117 if (thePlanSubject[i] == obj)
118 break;
119
120 if (i == thePlanSubjectCount)
121 thePlanSubject[ thePlanSubjectCount++ ] = obj;
122
123 ai->queueWaypoint( pos );
124 }
125 else
126 {
127 ai->clearWaypointQueue();
128 obj->leaveGroup();
129 obj->releaseWeaponLock(LOCKED_TEMPORARILY); // release any temporary locks.
130 ai->aiMoveToPosition( pos, CMD_FROM_PLAYER );
131 }
132 }
133}

137static void doSetRallyPoint( Object *obj, const Coord3D& pos )
138{
139 Bool isLocalPlayer = obj->isLocallyControlled();
140
141 //
142 // we must be able to find a path from the object to the point they have chosen, cause setting
143 // a rally point at a invalid location would suck. To be super nice, we have to make sure
144 // that every type of object that can be created from the thing setting the rally point
145 // can actually find a path from the thing to the point
146 //
147
148 // to see the never-finished code to check all locomotor sets, see past revs of GUICommandTranslator.cpp -MDC
149
150 //
151 // for now, just use the basic human locomotor ... and enable the above code when Steven
152 // tells me how to get the locomotor sets based on a thing template (CBD)
153 //
154 NameKeyType key = NAMEKEY( "BasicHumanLocomotor" );
155 LocomotorSet locomotorSet;
156 locomotorSet.addLocomotor( TheLocomotorStore->findLocomotorTemplate( key ) );
157 if( TheAI->pathfinder()->clientSafeQuickDoesPathExist( locomotorSet, obj->getPosition(), &pos ) == FALSE )
158 {
159
160 // user feedback
161 if( isLocalPlayer )
162 {
163
164 // display error message to user
165 TheInGameUI->message( TheGameText->fetch( "GUI:RallyPointNoPath" ) );
166
167 // play the no can do sound
168 static AudioEventRTS rallyNotSet("UnableToSetRallyPoint");
169 rallyNotSet.setPosition(&pos);
170 TheAudio->addAudioEvent(&rallyNotSet);
171
172 } // end if
173
174 return;
175
176 } // end if
177
178 // feedback to the player
179 if( isLocalPlayer )
180 {
181
182 // print a message to the user
183 UnicodeString info;
184 info.format( TheGameText->fetch( "GUI:RallyPointSet" ),
185 obj->getTemplate()->getDisplayName().str() );
186 TheInGameUI->message( info );
187
188 // play a sound for setting the rally point
189 static AudioEventRTS rallyPointSet("RallyPointSet");
190 rallyPointSet.setPosition(&pos);
191 rallyPointSet.setPlayerIndex(obj->getControllingPlayer()->getPlayerIndex());
192 TheAudio->addAudioEvent(&rallyPointSet);
193
194 // mark the UI as dirty so that we re-evaluate the selection and show the rally point
195 Drawable *draw = obj->getDrawable();
196 if( draw && draw->isSelected() )
197 TheControlBar->markUIDirty();
198
199 } // end if
200
201 // if this object has a ProductionExitUpdate interface, we are setting a rally point
202 ExitInterface *exitInterface = obj->getObjectExitInterface();
203 if( exitInterface )
204 {
205 // set the rally point
206 exitInterface->setRallyPoint( &pos );
207
208 }

352void GameLogic::logicMessageDispatcher( GameMessage *msg, void *userData )
353{
354#ifdef _DEBUG
355 DEBUG_ASSERTCRASH(msg != NULL && msg != (GameMessage*)0xdeadbeef, ("bad msg"));
356#endif
357
358 Player *thisPlayer = ThePlayerList->getNthPlayer( msg->getPlayerIndex() );
359 DEBUG_ASSERTCRASH( thisPlayer, ("logicMessageDispatcher: Processing message from unknown player (player index '%d')\n",
360 msg->getPlayerIndex()) );
361
362 AIGroup *currentlySelectedGroup = NULL;
363
364 if (isInGame())
365 {
366 if (msg->getType() >= GameMessage::MSG_BEGIN_NETWORK_MESSAGES && msg->getType() <= GameMessage::MSG_END_NETWORK_MESSAGES)
367 {
368 if (msg->getType() != GameMessage::MSG_LOGIC_CRC && msg->getType() != GameMessage::MSG_SET_REPLAY_CAMERA)
369 {
370 currentlySelectedGroup = TheAI->createGroup(); // can't do this outside a game - it'll cause sync errors galore.
371 CRCGEN_LOG(( "Creating AIGroup %d in GameLogic::logicMessageDispatcher()\n", (currentlySelectedGroup)?currentlySelectedGroup->getID():0 ));
372 thisPlayer->getCurrentSelectionAsAIGroup(currentlySelectedGroup);
373
374 // We can't issue commands to groups that contain units that don't belong the issuing player, so pretend like
375 // there's nothing selected. Also, if currentlySelectedGroup is empty, go ahead and delete it, so that we can skip
376 // any processing on it.
377 if (currentlySelectedGroup->isEmpty())
378 {
379 TheAI->destroyGroup(currentlySelectedGroup);
380 currentlySelectedGroup = NULL;
381 }
382
383 // If there are any units that the player doesn't own, then remove them from the "currentlySelectedGroup"
384 if (currentlySelectedGroup)
385 if (currentlySelectedGroup->removeAnyObjectsNotOwnedByPlayer(thisPlayer))
386 currentlySelectedGroup = NULL;
387
388 if(TheStatsCollector)
389 TheStatsCollector->collectMsgStats(msg);
390 }
391 }
392 }
393
394#ifdef DEBUG_LOGGING
395 AsciiString commandName;
396
397 commandName = msg->getCommandAsAsciiString();
398 if (msg->getType() < GameMessage::MSG_BEGIN_NETWORK_MESSAGES || msg->getType() > GameMessage::MSG_END_NETWORK_MESSAGES)
399 {
400 commandName.concat(" (NON-LOGIC-MESSAGE!!!)");
401 }
402 else if (msg->getType() == GameMessage::MSG_BEGIN_NETWORK_MESSAGES)
403 {
404 commandName = " (CRC message!)";
405 }
406#if 0
407 if (commandName.isNotEmpty() /*&& msg->getType() != GameMessage::MSG_FRAME_TICK*/)
408 {
409 DEBUG_LOG(("Frame %d: GameLogic::logicMessageDispatcher() saw a %s from player %d (%ls)\n", getFrame(), commandName.str(),
410 msg->getPlayerIndex(), thisPlayer->getPlayerDisplayName().str()));
411 }
412#endif
413#endif // DEBUG_LOGGING
414
415 // process the message
416 GameMessage::Type msgType = msg->getType();
417 switch( msgType )
418 {
419 //---------------------------------------------------------------------------------------------
420 case GameMessage::MSG_NEW_GAME:
421 {
422 //DEBUG_ASSERTCRASH(msg->getArgumentCount() == 1 || msg->getArgumentCount() == 2, ("%d arguments to MSG_NEW_GAME", msg->getArgumentCount()));
423 Int gameMode = msg->getArgument( 0 )->integer;
424 Int rankPoints = 0;
425 GameDifficulty diff = DIFFICULTY_NORMAL;
426 if ( msg->getArgumentCount() >= 2 )
427 diff = (GameDifficulty)msg->getArgument( 1 )->integer;
428 if ( msg->getArgumentCount() >= 3 )
429 rankPoints = msg->getArgument( 2 )->integer;
430
431 if ( msg->getArgumentCount() >= 4 )
432 {
433 Int maxFPS = msg->getArgument( 3 )->integer;
434 if (maxFPS < 1 || maxFPS > 1000)
435 maxFPS = TheGlobalData->m_framesPerSecondLimit;
436 DEBUG_LOG(("Setting max FPS limit to %d FPS\n", maxFPS));
437 TheGameEngine->setFramesPerSecondLimit(maxFPS);
438 TheWritableGlobalData->m_useFpsLimit = true;
439 }
440
441 // prepare for new game
442 prepareNewGame( gameMode, diff, rankPoints );
443
444 // start new game
445 startNewGame( FALSE );
446
447 break;
448
449 } // end new game
450
451 //---------------------------------------------------------------------------------------------
452 case GameMessage::MSG_CLEAR_GAME_DATA:
453 {
454
455#if defined(_DEBUG) || defined(_INTERNAL)
456 if (TheDisplay && TheGlobalData->m_dumpAssetUsage)
457 TheDisplay->dumpAssetUsage(TheGlobalData->m_mapName.str());
458#endif
459
460 if (currentlySelectedGroup)
461 TheAI->destroyGroup(currentlySelectedGroup);
462 currentlySelectedGroup = NULL;
463 TheGameLogic->clearGameData();
464 break;
465
466 } // end clear game data
467
468 //---------------------------------------------------------------------------------------------
469 case GameMessage::MSG_META_BEGIN_PATH_BUILD:
470 {
471 DEBUG_LOG(("META: begin path build\n"));
472 DEBUG_ASSERTCRASH(!theBuildPlan, ("mismatched theBuildPlan"));
473
474 if (theBuildPlan == false)
475 {
476 theBuildPlan = true;
477 thePlanSubjectCount = 0;
478 }
479
480 break;
481 }
482
483 //---------------------------------------------------------------------------------------------
484 case GameMessage::MSG_META_END_PATH_BUILD:
485 {
486 DEBUG_LOG(("META: end path build\n"));
487 DEBUG_ASSERTCRASH(theBuildPlan, ("mismatched theBuildPlan"));
488
489 // tell everyone who participated in the plan to move
490 for( int i=0; i<thePlanSubjectCount; i++ )
491 {
492 AIUpdateInterface *ai = thePlanSubject[i]->getAIUpdateInterface();
493 if (ai)
494 ai->executeWaypointQueue();
495 }
496
497 theBuildPlan = false;
498 thePlanSubjectCount = 0;
499
500 break;
501 }
502
503 //---------------------------------------------------------------------------------------------
504 case GameMessage::MSG_SET_RALLY_POINT:
505 {
506 Object *obj = TheGameLogic->findObjectByID( msg->getArgument( 0 )->objectID );
507 Coord3D dest = msg->getArgument( 1 )->location;
508 if (obj)
509 {
510 doSetRallyPoint( obj, dest );
511 }
512
513 break;
514
515 }
516
517 //---------------------------------------------------------------------------------------------
518 case GameMessage::MSG_DO_WEAPON:
519 {
520 WeaponSlotType weaponSlot = (WeaponSlotType)msg->getArgument( 0 )->integer;
521 Int maxShotsToFire = msg->getArgument( 1 )->integer;
522
523 // lock it just till the weapon is empty or the attack is "done"
524 if( currentlySelectedGroup && currentlySelectedGroup->setWeaponLockForGroup( weaponSlot, LOCKED_TEMPORARILY ))
525 {
526 currentlySelectedGroup->groupAttackPosition( NULL, maxShotsToFire, CMD_FROM_PLAYER );
527 }
528
529 break;
530 }
531
532 //---------------------------------------------------------------------------------------------
533 case GameMessage::MSG_COMBATDROP_AT_OBJECT:
534 {
535 Object *targetObject = TheGameLogic->findObjectByID( msg->getArgument( 0 )->objectID );
536
537 // issue command for either single object or for selected group
538 if( currentlySelectedGroup )
539 currentlySelectedGroup->groupCombatDrop( targetObject,
540 *targetObject->getPosition(),
541 CMD_FROM_PLAYER );
542

558 //---------------------------------------------------------------------------------------------
559 case GameMessage::MSG_COMBATDROP_AT_LOCATION:
560 {
561 Coord3D targetLoc = msg->getArgument( 0 )->location;
562
563 if( currentlySelectedGroup )
564 currentlySelectedGroup->groupCombatDrop( NULL, targetLoc, CMD_FROM_PLAYER );

566/*

576
577 break;
578
579 } // end GameMessage::MSG_COMBATDROP_AT_LOCATION
580
581 //---------------------------------------------------------------------------------------------
582 case GameMessage::MSG_DO_WEAPON_AT_OBJECT:
583 {
584 // Lock the weapon choice to the right weapon, then give an attack command
585
586 WeaponSlotType weaponSlot = (WeaponSlotType)msg->getArgument( 0 )->integer;
587 Object *targetObject = TheGameLogic->findObjectByID( msg->getArgument( 1 )->objectID );
588 Int maxShotsToFire = msg->getArgument( 2 )->integer;
589
590 // sanity
591 if( targetObject == NULL )
592 break;
593
594
595 // issue command for either single object or for selected group
596 if( currentlySelectedGroup )
597 {
598 // lock it just till the weapon is empty or the attack is "done"
599 if (currentlySelectedGroup->setWeaponLockForGroup( weaponSlot, LOCKED_TEMPORARILY ))
600 currentlySelectedGroup->groupAttackObject( targetObject, maxShotsToFire, CMD_FROM_PLAYER );
601 } // end if, command for group
602 break;
603
604 } // end do weapon at object
605
606 //---------------------------------------------------------------------------------------------
607 case GameMessage::MSG_SWITCH_WEAPONS:
608 {
609 // use the selected group
610 WeaponSlotType weaponSlot = (WeaponSlotType)msg->getArgument( 0 )->integer;
611 // lock until un-switched, or switched to something else.
612 if( currentlySelectedGroup )
613 currentlySelectedGroup->setWeaponLockForGroup( weaponSlot, LOCKED_PERMANENTLY );
614 break;
615 }

617 //---------------------------------------------------------------------------------------------
618 case GameMessage::MSG_SET_MINE_CLEARING_DETAIL:

642 case GameMessage::MSG_DO_WEAPON_AT_LOCATION:
643 {
644 WeaponSlotType weaponSlot = (WeaponSlotType)msg->getArgument( 0 )->integer;
645 Coord3D targetLoc = msg->getArgument( 1 )->location;
646 Int maxShotsToFire = msg->getArgument( 2 )->integer;
647
648 // issue command for either single object or for selected group
649 if( currentlySelectedGroup )
650 {
651 // lock it just till the weapon is empty or the attack is "done"
652 if (currentlySelectedGroup->setWeaponLockForGroup( weaponSlot, LOCKED_TEMPORARILY ))
653 currentlySelectedGroup->groupAttackPosition( &targetLoc, maxShotsToFire, CMD_FROM_PLAYER );

655
656 } // end if, command for group
657
658 break;
659
660 } //end do weapon at location
661
662 //---------------------------------------------------------------------------------------------
663 case GameMessage::MSG_DO_SPECIAL_POWER:
664 {
665
666 // first argument is the special power ID
667 UnsignedInt specialPowerID = msg->getArgument( 0 )->integer;
668
669 // Command button options -- special power may care about variance options
670 UnsignedInt options = msg->getArgument( 1 )->integer;
671
672 // check for possible specific source, ignoring selection.
673 ObjectID sourceID = msg->getArgument(2)->objectID;
674 Object* source = TheGameLogic->findObjectByID(sourceID);
675 if (source != NULL)
676 {
677 AIGroup* theGroup = TheAI->createGroup();
678 theGroup->add(source);
679 theGroup->groupDoSpecialPower( specialPowerID, options );
680 TheAI->destroyGroup(theGroup);
681 }
682 else
683 {
684 //Use the selected group!
685 if( currentlySelectedGroup )
686 {
687 currentlySelectedGroup->groupDoSpecialPower( specialPowerID, options );
688 }
689 }

691
692 } // end do special
693
694 //---------------------------------------------------------------------------------------------
695 case GameMessage::MSG_DO_SPECIAL_POWER_AT_LOCATION:
696 {
697 // first argument is the special power ID
698 UnsignedInt specialPowerID = msg->getArgument( 0 )->integer;
699
700 // Location argument 2 is destination
701 Coord3D targetCoord = msg->getArgument(1)->location;
702
703 // Angle argument 3 is the orientation of the special power (if applicable)
704 Real angle = msg->getArgument(2)->real;
705
706 // Object in way -- if applicable (some specials care, others don't)
707 ObjectID objectID = msg->getArgument( 3 )->objectID;
708 Object *objectInWay = TheGameLogic->findObjectByID( objectID );
709
710 // Command button options -- special power may care about variance options
711 UnsignedInt options = msg->getArgument( 4 )->integer;
712
713 // check for possible specific source, ignoring selection.
714 ObjectID sourceID = msg->getArgument(5)->objectID;
715 Object* source = TheGameLogic->findObjectByID(sourceID);
716 if (source != NULL)
717 {
718 AIGroup* theGroup = TheAI->createGroup();
719 theGroup->add(source);
720 theGroup->groupDoSpecialPowerAtLocation( specialPowerID, &targetCoord, angle, objectInWay, options );
721 TheAI->destroyGroup(theGroup);
722 }
723 else
724 {
725 //Use the selected group!
726 if( currentlySelectedGroup )

733 } // end do special at location
734
735 //---------------------------------------------------------------------------------------------
736 case GameMessage::MSG_DO_SPECIAL_POWER_AT_OBJECT:
737 {
738 // first argument is the special power ID
739 UnsignedInt specialPowerID = msg->getArgument( 0 )->integer;
740
741 // argument 2 is target object
742 ObjectID targetID = msg->getArgument(1)->objectID;
743 Object *target = TheGameLogic->findObjectByID( targetID );
744 if( !target )
745 {
746 break;
747 }
748
749 // Command button options -- special power may care about variance options

751
