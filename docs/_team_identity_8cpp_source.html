226void TeamIdentity::loadUnitsInfo(int idcMinUnit, NameKeyType keyMinUnit,
227 int idcMaxUnit, NameKeyType keyMaxUnit,
228 int idcUnitType, NameKeyType keyUnitType)
229{
230 CEdit *pEdit = (CEdit *)GetDlgItem(idcMinUnit);
231 CString text;
232 Bool exists;
233 text.Format("%d", m_teamDict->getInt(keyMinUnit, &exists));
234 pEdit->SetWindowText(text);
235
236 pEdit = (CEdit *)GetDlgItem(idcMaxUnit);
237 text.Format("%d", m_teamDict->getInt(keyMaxUnit, &exists));
238 pEdit->SetWindowText(text);
239
240 AsciiString type = m_teamDict->getAsciiString(keyUnitType, &exists);
241 if (type.isEmpty()) type = NONE_STRING;
242
243 CComboBox *pCombo = (CComboBox *)GetDlgItem(idcUnitType);
244 pCombo->ResetContent();
245
246 Bool found = false;
247
248 // add entries from the thing factory as the available UNITS to use
249 const ThingTemplate *tTemplate;
250 for( tTemplate = TheThingFactory->firstTemplate();
251 tTemplate;
252 tTemplate = tTemplate->friend_getNextTemplate() ) {
253
254 // next tier uses the editor sorting bits that design can specify in the INI
255 EditorSortingType sort = tTemplate->getEditorSorting();
256 if (( sort != ES_VEHICLE ) && (sort != ES_INFANTRY)) continue;
257
258 Int ndx = pCombo->AddString(tTemplate->getName().str());
259 if (type == tTemplate->getName()) {
260 found = true;
261 pCombo->SetCurSel(ndx);
262 }
263 }
264
265 Int ndx = pCombo->AddString(NONE_STRING);
266 if (!found) {
267 pCombo->SetCurSel(ndx);
268 }
269
270}

273BOOL TeamIdentity::OnCommand(WPARAM wParam, LPARAM lParam)
274{
275
276 Int wNotifyCode = HIWORD(wParam); // notification code
277 Int wID = LOWORD(wParam); // item, control, or accelerator identifier
278 NameKeyType key;
279 if (wNotifyCode == EN_CHANGE) {
280 Int editCtrl = wID;
281 switch (editCtrl) {
282 default: editCtrl = 0;
283 case IDC_MIN_UNIT1: key = TheKey_teamUnitMinCount1; break;
284 case IDC_MAX_UNIT1: key = TheKey_teamUnitMaxCount1; break;
285
286 case IDC_MIN_UNIT2: key = TheKey_teamUnitMinCount2; break;
287 case IDC_MAX_UNIT2: key = TheKey_teamUnitMaxCount2; break;
288
289 case IDC_MIN_UNIT3: key = TheKey_teamUnitMinCount3; break;
290 case IDC_MAX_UNIT3: key = TheKey_teamUnitMaxCount3; break;
291
292 case IDC_MIN_UNIT4: key = TheKey_teamUnitMinCount4; break;
293 case IDC_MAX_UNIT4: key = TheKey_teamUnitMaxCount4; break;
294
295 case IDC_MIN_UNIT5: key = TheKey_teamUnitMinCount5; break;
296 case IDC_MAX_UNIT5: key = TheKey_teamUnitMaxCount5; break;
297
298 case IDC_MIN_UNIT6: key = TheKey_teamUnitMinCount6; break;
299 case IDC_MAX_UNIT6: key = TheKey_teamUnitMaxCount6; break;
300
301 case IDC_MIN_UNIT7: key = TheKey_teamUnitMinCount7; break;
302 case IDC_MAX_UNIT7: key = TheKey_teamUnitMaxCount7; break;
303 }
304 if (editCtrl != 0) {
305 CEdit *pEdit = (CEdit *)GetDlgItem(editCtrl);
306 CString txt;
307 pEdit->GetWindowText(txt);
308 Int theInt;
309 if (1==sscanf(txt, "%d", &theInt)) {
310 m_teamDict->setInt(key, theInt);
311 }
312 return true;
313 }
314 } else if (wNotifyCode == CBN_SELCHANGE) {
315 Int cmboCtrl = wID;
316 switch (cmboCtrl) {
317 case IDC_UNIT_TYPE1: key = TheKey_teamUnitType1; break;
318 case IDC_UNIT_TYPE2: key = TheKey_teamUnitType2; break;
319 case IDC_UNIT_TYPE3: key = TheKey_teamUnitType3; break;
320 case IDC_UNIT_TYPE4: key = TheKey_teamUnitType4; break;
321 case IDC_UNIT_TYPE5: key = TheKey_teamUnitType5; break;
322 case IDC_UNIT_TYPE6: key = TheKey_teamUnitType6; break;
323 case IDC_UNIT_TYPE7: key = TheKey_teamUnitType7; break;
324 default: cmboCtrl = 0;
325 }
326 if (cmboCtrl != 0) {
327 CComboBox *pCombo = (CComboBox*)GetDlgItem(cmboCtrl);
328 CString txt;
329 pCombo->GetWindowText(txt);
330 AsciiString comboText = AsciiString(txt);
331 m_teamDict->setAsciiString(key, comboText);
332 return true;
333 }
334 }
335
336 return CPropertyPage::OnCommand(wParam, lParam);
337}

531void TeamIdentity::OnKillfocusTeamName()
532{
533 CWnd *pWnd = GetDlgItem(IDC_TEAM_NAME);
534 if (pWnd) {
535 CString val;
536 pWnd->GetWindowText(val);
537 AsciiString tnamenew = val;
538 AsciiString tnamecur = m_teamDict->getAsciiString(TheKey_teamName);
539 Bool set = true;
540 if (tnamecur != tnamenew) {
541 if (m_sides->findTeamInfo(tnamenew) || m_sides->findSideInfo(tnamenew))
542 {
543 ::AfxMessageBox(IDS_NAME_IN_USE);
544 set = false;
545 }
546 else
547 {
548 Int count = MapObject::countMapObjectsWithOwner(tnamecur);
549 if (count > 0)
550 {
551 set = false;
552 CString msg;
553 msg.Format(IDS_RENAMING_INUSE_TEAM, count);
554 if (::AfxMessageBox(msg, MB_YESNO) == IDYES)
555 set = true;
556 }
557 }
558 }
559 if (set) {
560 m_teamDict->setAsciiString(TheKey_teamName, tnamenew);
561 } else {
562 Bool exists;
563 AsciiString description = m_teamDict->getAsciiString(TheKey_teamName, &exists);
564 pWnd->SetWindowText(description.str());
565 }
566 }
567}
