146Bool Display::setDisplayMode( UnsignedInt xres, UnsignedInt yres, UnsignedInt bitdepth, Bool windowed )
147{
148 //Get old values
149 UnsignedInt oldDisplayHeight=getHeight();
150 UnsignedInt oldDisplayWidth=getWidth();
151 Int oldViewWidth=TheTacticalView->getWidth();
152 Int oldViewHeight=TheTacticalView->getHeight();
153 Int oldViewOriginX,oldViewOriginY;
154 TheTacticalView->getOrigin(&oldViewOriginX,&oldViewOriginY);
155
156 setWidth(xres);
157 setHeight(yres);
158
159 //Adjust view to match previous proportions
160 TheTacticalView->setWidth((Real)oldViewWidth/(Real)oldDisplayWidth*(Real)xres);
161 TheTacticalView->setHeight((Real)oldViewHeight/(Real)oldDisplayHeight*(Real)yres);
162 TheTacticalView->setOrigin((Real)oldViewOriginX/(Real)oldDisplayWidth*(Real)xres,
163 (Real)oldViewOriginY/(Real)oldDisplayHeight*(Real)yres);
164 return TRUE;
165}

268void Display::stopMovie( void )
269{
270 delete m_videoBuffer;
271 m_videoBuffer = NULL;
272
273 if ( m_videoStream )
274 {
275 m_videoStream->close();
276 m_videoStream = NULL;
277 }
278
279 if (!m_currentlyPlayingMovie.isEmpty()) {
280 //TheScriptEngine->notifyOfCompletedVideo(m_currentlyPlayingMovie); // Removing this sync-error cause MDC
281 m_currentlyPlayingMovie = AsciiString::TheEmptyString;
282 }
283 if(m_copyrightDisplayString)
284 {
285 TheDisplayStringManager->freeDisplayString(m_copyrightDisplayString);
286 m_copyrightDisplayString = NULL;
287 }
288 m_copyrightHoldTime = -1;
289 m_movieHoldTime = -1;
290}

295
296void Display::update( void )
297{
298 if ( m_videoStream && m_videoBuffer )
299 {
300 if ( m_videoStream->isFrameReady())
301 {
302 m_videoStream->frameDecompress();
303 m_videoStream->frameRender( m_videoBuffer );
304 if( m_videoStream->frameIndex() != m_videoStream->frameCount() - 1)
305 m_videoStream->frameNext();
306 else if( m_copyrightHoldTime >= 0 ||m_movieHoldTime >= 0 )
307 {
308 if( m_elapsedCopywriteTime == 0 && m_elapsedCopywriteTime >= 0)
309 {
310 //display the copyrighttext;
311 if(m_copyrightDisplayString)
312 m_copyrightDisplayString->deleteInstance();
313 m_copyrightDisplayString = TheDisplayStringManager->newDisplayString();
314 m_copyrightDisplayString->setText(TheGameText->fetch("GUI:EACopyright"));
315 if (TheGlobalLanguageData && TheGlobalLanguageData->m_copyrightFont.name.isNotEmpty())
316 { FontDesc *fontdesc=&TheGlobalLanguageData->m_copyrightFont;
317 m_copyrightDisplayString->setFont(TheFontLibrary->getFont(fontdesc->name,
318 TheGlobalLanguageData->adjustFontSize(fontdesc->size),
319 fontdesc->bold));
320 }
321 else
322 m_copyrightDisplayString->setFont(TheFontLibrary->getFont("Courier",
323 TheGlobalLanguageData->adjustFontSize(12), TRUE));
324 m_elapsedCopywriteTime = timeGetTime();
325 }
326 if(m_movieHoldTime + m_elapsedMovieTime < timeGetTime() &&
327 m_copyrightHoldTime + m_elapsedCopywriteTime < timeGetTime())
328 {
329 m_movieHoldTime = -1;
330 m_elapsedMovieTime = 0;
331 m_elapsedCopywriteTime = 0;
332 m_copyrightHoldTime = -1;
333 }
334 }
335 else
336 {
337 stopMovie();
338 }
339 }
340 }
