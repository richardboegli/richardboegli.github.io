122#define LINE_LEN 2048
123Bool UserPreferences::load(AsciiString fname)
124{
125// if (strstr(fname.str(), "\\"))
126// throw INI_INVALID_DATA; // must be a leaf name
127
128 m_filename = TheGlobalData->getPath_UserData();
129 m_filename.concat(fname);
130
131 FILE *fp = fopen(m_filename.str(), "r");
132 if (fp)
133 {
134 char buf[LINE_LEN];
135 while( fgets( buf, LINE_LEN, fp ) != NULL )
136 {
137 AsciiString line = buf;
138 line.trim();
139
140 AsciiString key, val;
141 line.nextToken(&key, "=");
142 val = line.str() + 1;
143
144 key.trim();
145 val.trim();
146
147 if (key.isEmpty() || val.isEmpty())
148 continue;
149
150 (*this)[key] = val;
151 } // end while
152 fclose(fp);
153 return true;
154 }
155 return false;
156}
157
158Bool UserPreferences::write( void )
159{
160 if (m_filename.isEmpty())
161 return false;
162
163 FILE *fp = fopen(m_filename.str(), "w");
164 if (fp)
165 {
166 PreferenceMap::const_iterator it = begin();
167 while (it != end())
168 {
169 fprintf(fp, "%s = %s\n", it->first.str(), it->second.str());
170 ++it;
171 }
172 fclose(fp);
173 return true;
174 }
175 return false;
176}

178Bool UserPreferences::getBool(AsciiString key, Bool defaultValue) const
179{
180 AsciiString val = getAsciiString(key, AsciiString::TheEmptyString);

518Int CustomMatchPreferences::getPreferredFaction(void)
519{
520 Int ret;
521 CustomMatchPreferences::const_iterator it = find("PlayerTemplate");
522 if (it == end())
523 {
524 return PLAYERTEMPLATE_RANDOM;
525 }
526
527 ret = atoi(it->second.str());
528 if (ret == PLAYERTEMPLATE_OBSERVER || ret < PLAYERTEMPLATE_MIN || ret >= ThePlayerTemplateStore->getPlayerTemplateCount())
529 ret = PLAYERTEMPLATE_RANDOM;
530
531 if (ret >= 0)
532 {
533 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(ret);
534 if (!fac)
535 ret = PLAYERTEMPLATE_RANDOM;
536 else if (fac->getStartingBuilding().isEmpty())
537 ret = PLAYERTEMPLATE_RANDOM;
538 else if (TheGameInfo && TheGameInfo->oldFactionsOnly() && !fac->isOldFaction())
539 ret = PLAYERTEMPLATE_RANDOM;
540 else {
541 // Prevent from loading the disabled Generals, in case you had previously selected one as your preferred faction.
542 // This is also enforced at GUI setup (GUIUtil.cpp and GameLogic.cpp).
543 // @todo: unlock these when something rad happens
544 Bool disallowLockedGenerals = TRUE;
545 const GeneralPersona *general = TheChallengeGenerals->getGeneralByTemplateName(fac->getName());
546 Bool startsLocked = general ? !general->isStartingEnabled() : FALSE;
547 if (disallowLockedGenerals && startsLocked)
548 ret = PLAYERTEMPLATE_RANDOM;
549 }
550 }
551
552 return ret;
553}

882Bool LadderPreferences::loadProfile( Int profileID )
883{
884 clear();
885 m_ladders.clear();
886 AsciiString userPrefFilename;
887 userPrefFilename.format("GeneralsOnline\\Ladders%d.ini", profileID);
888 Bool success = load(userPrefFilename);
889 if (!success)
890 return success;
891
892 // parse out our ladders
893 for (LadderPreferences::iterator it = begin(); it != end(); ++it)
894 {
895 LadderPref p;
896 AsciiString ladName = it->first;
897 AsciiString ladData = it->second;
898
899 DEBUG_LOG(("Looking at [%s] = [%s]\n", ladName.str(), ladData.str()));
900
901 const char *ptr = ladName.reverseFind(':');
902 DEBUG_ASSERTCRASH(ptr, ("Did not find ':' in ladder name - skipping"));
903 if (!ptr)
904 continue;
905
906 p.port = atoi( ptr + 1 );
907 for (Int i=0; i<strlen(ptr); ++i)
908 {
909 ladName.removeLastChar();
910 }
911 p.address = QuotedPrintableToAsciiString(ladName);
912
913 ptr = ladData.reverseFind(':');
914 DEBUG_ASSERTCRASH(ptr, ("Did not find ':' in ladder data - skipping"));
915 if (!ptr)
916 continue;
917
