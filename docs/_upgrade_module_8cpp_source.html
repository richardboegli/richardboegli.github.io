127// ------------------------------------------------------------------------------------------------
128// ------------------------------------------------------------------------------------------------
129Bool UpgradeMux::wouldUpgrade( UpgradeMaskType keyMask ) const
130{
131 UpgradeMaskType activation, conflicting;
132 getUpgradeActivationMasks(activation, conflicting);
133
134 //Make sure we have activation conditions and we haven't performed the upgrade already.
135 if( activation.any() && keyMask.any() && !m_upgradeExecuted )
136 {
137 //Okay, make sure we don't have any conflicting upgrades
138 if( !keyMask.testForAny( conflicting) )
139 {
140 //Finally check to see if our upgrade conditions match.
141 if( requiresAllActivationUpgrades() )
142 {
143 //Make sure ALL triggers requirements are upgraded
144 if( keyMask.testForAll( activation ) )
145 {
146 return TRUE;
147 }
148 }
149 else
150 {
151 //Check if ANY trigger requirements are met.
152 if( keyMask.testForAny( activation ) )
153 {
154 return TRUE;
155 }
156 }
157 }
158 }
159 //We can't upgrade!
160 return FALSE;
161}

174Bool UpgradeMux::testUpgradeConditions( UpgradeMaskType keyMask ) const
175{
176 UpgradeMaskType activation, conflicting;
177 getUpgradeActivationMasks(activation, conflicting);
178
179 //Okay, make sure we don't have any conflicting upgrades
180 if( !keyMask.any() || !keyMask.testForAny( conflicting ) )
181 {
182 //Make sure we have activation conditions
183 if( activation.any() )
184 {
185 //Finally check to see if our upgrade conditions match.
186 if( requiresAllActivationUpgrades() )
187 {
188 //Make sure ALL triggers requirements are upgraded
189 if( keyMask.testForAll( activation ) )
190 {
191 return TRUE;
192 }
193 }
194 else
195 {
196 //Check if ANY trigger requirements are met.
197 if( keyMask.testForAny( activation ) )
198 {
199 return TRUE;
200 }
201 }
202 }
203 else
204 {
205 //This *upgrade* is relying only on not having conflicts.
206 return true;
207 }
208 }
209 //We can't upgrade!
210 return false;
211}
