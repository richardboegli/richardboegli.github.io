106//
107// Initialize_Miles_Handle
108//
110void
111FilteredSoundClass::Initialize_Miles_Handle (void)
112{
113 SoundPseudo3DClass::Initialize_Miles_Handle ();
114 m_hFilter = WWAudioClass::Get_Instance ()->Get_Reverb_Filter ();
115 if ((m_SoundHandle != NULL) &&
116 (m_hFilter != INVALID_MILES_HANDLE)) {
117
118 //
119 // Pass the filter onto the sample
120 //
121 ::AIL_set_sample_processor (m_SoundHandle->Get_HSAMPLE (), DP_FILTER, m_hFilter);
122
123 //
124 // Change the reverb's settings to simulate a 'tinny' effect.
125 //
126 F32 reverb_level = 0.3F;
127 F32 reverb_reflect = 0.01F;
128 F32 reverb_decay = 0.535F;
129 ::AIL_set_filter_sample_preference (m_SoundHandle->Get_HSAMPLE (),
130 "Reverb level",
131 &reverb_level);
132
133 ::AIL_set_filter_sample_preference (m_SoundHandle->Get_HSAMPLE (),
134 "Reverb reflect time",
135 &reverb_reflect);
136
137 ::AIL_set_filter_sample_preference (m_SoundHandle->Get_HSAMPLE (),
138 "Reverb decay time",
139 &reverb_decay);
140 }
141
142 Update_Volume ();
143 return ;
144}

148//
149// Update_Volume
150//
152void
153FilteredSoundClass::Update_Volume (void)
154{
155 if (m_SoundHandle != NULL) {
156
157 // Determine the listener's position and the sound's position
158 SoundSceneClass *scene = WWAudioClass::Get_Instance ()->Get_Sound_Scene ();
159 Listener3DClass *listener = scene->Peek_2nd_Listener ();
160 if (listener != NULL) {
161 Vector3 listener_pos = listener->Get_Position ();
162 Vector3 sound_pos = m_Transform.Get_Translation ();
163
164 // Determine a normalized volume from the position
165 float distance = (sound_pos - listener_pos).Quick_Length ();
166 Update_Pseudo_Volume (distance);
167 }
168 }
169
170 return ;
171}
