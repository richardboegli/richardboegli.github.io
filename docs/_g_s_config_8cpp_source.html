136Bool SectionChecker::isInSection() {
137 Bool ret = FALSE;
138 for (SectionList::const_iterator it = m_bools.begin(); it != m_bools.end(); ++it)
139 {
140 ret = ret || **it;
141 }
142 return ret;

216 {

220 {
221 inQMMaps = TRUE;
222 }

224 {
225 inQMMaps = FALSE;
226 }
227 else if (!sections.isInSection() && line.compare("<Manglers>") == 0)
228 {
229 inManglers = TRUE;
230 }
231 else if (inManglers && line.compare("</Manglers>") == 0)
232 {
233 inManglers = FALSE;
234 }
235 else if (!sections.isInSection() && line.compare("<QMBot>") == 0)
236 {
237 inQMBot = TRUE;
238 }
239 else if (inQMBot && line.compare("</QMBot>") == 0)

244 {

248 {
249 inVIP = FALSE;
250 }
251 else if (!sections.isInSection() && line.compare("<NAT>") == 0)
252 {
253 inNAT = TRUE;
254 }
255 else if (inNAT && line.compare("</NAT>") == 0)
256 {
257 inNAT = FALSE;
258 }
259 else if (!sections.isInSection() && line.compare("<Custom>") == 0)
260 {
261 inCustom = TRUE;
262 }
263 else if (inCustom && line.compare("</Custom>") == 0)
264 {
265 inCustom = FALSE;
266 }
267 else if (inVIP)
268 {
269 line.toLower();
270 if (line.getLength())
271 {
272 Int val = atoi(line.str());
273 if (val > 0)
274 m_vip.insert(val);
275 }
276 }
277 else if (inPingServers)
278 {
279 line.toLower();
280 m_pingServers.push_back(line);

282 else if (inPingDuration)
283 {
284 line.toLower();
285 AsciiString key, val;
286 if (line.nextToken(&key, " ="))
287 {
288 if (key == "reps")

290 if (line.nextToken(&val, " ="))
291 {
292 m_pingReps = atoi(val.str());
293 }
294 }
295 else if (key == "timeout")
296 {
297 if (line.nextToken(&val, " ="))
298 {
299 m_pingTimeout = atoi(val.str());
300 }
301 }
302 else if (key == "low")
303 {
304 if (line.nextToken(&val, " ="))
305 {
306 m_pingCutoffGood = atoi(val.str());
307 }
308 }
309 else if (key == "med")
310 {
311 if (line.nextToken(&val, " ="))
312 {
313 m_pingCutoffBad = atoi(val.str());
314 }
315 }
316 }
317 }
318 else if (inManglers)
319 {
320 line.trim();
321 line.toLower();
322 AsciiString hostStr;
323 AsciiString portStr;
324 line.nextToken(&hostStr, ":");
325 line.nextToken(&portStr, ":\n\r");
326 if (hostStr.isNotEmpty() && portStr.isNotEmpty())
327 {
328 m_manglerHosts.push_back(hostStr);
329 m_manglerPorts.push_back(atoi(portStr.str()));
330 }
331 }
332 else if (inQMMaps)
333 {
334 line.toLower();
335 AsciiString mapName;
336 mapName.format("%s\\%s\\%s.map", TheMapCache->getMapDir().str(), line.str(), line.str());
337 mapName = TheGameState->portableMapPathToRealMapPath(TheGameState->realMapPathToPortableMapPath(mapName));
338 mapName.toLower();
339
340 const MapMetaData *md = TheMapCache->findMap(mapName);
341 if (md)
342 {
343 m_qmMaps.push_back(mapName);
344 }
345 }
346 else if (inQMBot)
347 {
348 line.toLower();
349 AsciiString key, val;
350 if (line.nextToken(&key, " ="))
351 {
352 if (key == "id")
353 {
354 if (line.nextToken(&val, " ="))
355 {
356 m_qmBotID = atoi(val.str());
357 }
358 }
359 }
360 }
361 else if (inNAT)
362 {

365 if (line.nextToken(&key, " ="))
366 {
367 if (key == "retryinterval")
368 {
369 if (line.nextToken(&val, " ="))
370 {
371 m_natRetryInterval = atoi(val.str());
372 }
373 }
374 else if (key == "manglerretries")
375 {
376 if (line.nextToken(&val, " ="))
377 {
378 m_natMaxManglerRetries = atoi(val.str());
379 }
380 }
381 else if (key == "manglerinterval")
382 {
383 if (line.nextToken(&val, " ="))
384 {
385 m_natManglerRetryInterval = atoi(val.str());
386 }
387 }
388 else if (key == "keepaliveinterval")
389 {
390 if (line.nextToken(&val, " ="))
391 {
392 m_natKeepaliveInterval = atoi(val.str());
393 }
394 }
395 else if (key == "porttimeout")
396 {
397 if (line.nextToken(&val, " ="))
398 {
399 m_natPortTimeout = atoi(val.str());
400 }
401 }
402 else if (key == "roundtimeout")
403 {
404 if (line.nextToken(&val, " ="))
405 {
406 m_natRoundTimeout = atoi(val.str());
407 }
408 }
409 else
410 {
411 DEBUG_LOG(("Unknown key '%s' = '%s' in NAT block of GameSpy Config\n", key.str(), val.str()));
412 }
413 }
414 else
415 {
416 DEBUG_LOG(("Key '%s' missing val in NAT block of GameSpy Config\n", key.str()));
417 }
418 }
419 else if (inCustom)
420 {
421 line.toLower();
422 AsciiString key, val;
423 if (line.nextToken(&key, " =") && line.nextToken(&val, " ="))
424 {
425 if (key == "restricted")
426 {
427 m_restrictGamesToLobby = atoi(val.str());
428 }
429 else
430 {
431 DEBUG_LOG(("Unknown key '%s' = '%s' in Custom block of GameSpy Config\n", key.str(), val.str()));
432 }
433 }
434 else
435 {
436 DEBUG_LOG(("Key '%s' missing val in Custom block of GameSpy Config\n", key.str()));
437 }
438 }
439 else
440 {
441 m_leftoverConfig.concat(line);
442 m_leftoverConfig.concat('\n');
443 }
444 }
