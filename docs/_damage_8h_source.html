99typedef BitFlags<DAMAGE_NUM_TYPES> DamageTypeFlags;

101inline Bool getDamageTypeFlag(DamageTypeFlags flags, DamageType dt)
102{
103 return flags.test(dt);
104}
105
106inline DamageTypeFlags setDamageTypeFlag(DamageTypeFlags flags, DamageType dt)
107{
108 flags.set(dt, TRUE);

110}
111
112inline DamageTypeFlags clearDamageTypeFlag(DamageTypeFlags flags, DamageType dt)
113{
114 flags.set(dt, FALSE);

116}

118// Instead of checking against a single damage type, gather the question here so we can have more than one
119inline Bool IsSubdualDamage( DamageType type )
120{
121 switch( type )
122 {
123 case DAMAGE_SUBDUAL_MISSILE:
124 case DAMAGE_SUBDUAL_VEHICLE:
125 case DAMAGE_SUBDUAL_BUILDING:
126 case DAMAGE_SUBDUAL_UNRESISTABLE:
127 return TRUE;
128 }
129
130 return FALSE;
131}
132
134inline Bool IsHealthDamagingDamage( DamageType type )
135{
136 // The need for this function brought to you by "Have the guy with no game experience write the weapon code" Foundation.
137 // Health Damage should be one type of WeaponEffect. Thinking "Weapons can only do damage" is why AoE is boring.
138 switch( type )
139 {
140 case DAMAGE_STATUS:
141 case DAMAGE_SUBDUAL_MISSILE:
142 case DAMAGE_SUBDUAL_VEHICLE:
143 case DAMAGE_SUBDUAL_BUILDING:
144 case DAMAGE_SUBDUAL_UNRESISTABLE:
145 case DAMAGE_KILLPILOT:
146 case DAMAGE_KILL_GARRISONED:
147 return FALSE;
148 }
149
150 return TRUE;
151}

234typedef UnsignedInt DeathTypeFlags;
235
236const DeathTypeFlags DEATH_TYPE_FLAGS_ALL = 0xffffffff;
237const DeathTypeFlags DEATH_TYPE_FLAGS_NONE = 0x00000000;
238
239inline Bool getDeathTypeFlag(DeathTypeFlags flags, DeathType dt)
240{
241 return (flags & (1UL << (dt - 1))) != 0;
242}
243
244inline DeathTypeFlags setDeathTypeFlag(DeathTypeFlags flags, DeathType dt)
245{
246 return (flags | (1UL << (dt - 1)));
247}
248
249inline DeathTypeFlags clearDeathTypeFlag(DeathTypeFlags flags, DeathType dt)
250{
251 return (flags & ~(1UL << (dt - 1)));
252}
