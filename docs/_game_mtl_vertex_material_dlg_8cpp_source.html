142BOOL GameMtlVertexMaterialDlg::Dialog_Proc(HWND dlg_wnd, UINT message, WPARAM wparam, LPARAM lparam)
143{
144 int val;
145 int id = LOWORD(wparam);
146 int code = HIWORD(wparam);
147
148 switch (message)
149 {
150 case WM_INITDIALOG:
151 {
152 AmbientSwatch = GetIColorSwatch(GetDlgItem(dlg_wnd, IDC_AMBIENT_COLOR),TheMtl->Get_Ambient(PassIndex,IParams->GetTime()),Get_String(IDS_AMBIENT_COLOR));
153 DiffuseSwatch = GetIColorSwatch(GetDlgItem(dlg_wnd, IDC_DIFFUSE_COLOR),TheMtl->Get_Diffuse(PassIndex,IParams->GetTime()),Get_String(IDS_DIFFUSE_COLOR));
154 SpecularSwatch = GetIColorSwatch(GetDlgItem(dlg_wnd, IDC_SPECULAR_COLOR),TheMtl->Get_Specular(PassIndex,IParams->GetTime()),Get_String(IDS_SPECULAR_COLOR));
155 EmissiveSwatch = GetIColorSwatch(GetDlgItem(dlg_wnd, IDC_EMISSIVE_COLOR),TheMtl->Get_Emissive(PassIndex,IParams->GetTime()),Get_String(IDS_EMISSIVE_COLOR));
156
157 OpacitySpin = SetupFloatSpinner(dlg_wnd,IDC_OPACITY_SPIN,IDC_OPACITY_EDIT,0.0f,1.0f,TheMtl->Get_Opacity(PassIndex,IParams->GetTime()),0.01f);
158 TranslucencySpin = SetupFloatSpinner(dlg_wnd,IDC_TRANSLUCENCY_SPIN,IDC_TRANSULCENCY_EDIT,0.0f,1.0f,TheMtl->Get_Translucency(PassIndex,IParams->GetTime()),0.01f);
159 ShininessSpin = SetupFloatSpinner(dlg_wnd,IDC_SHININESS_SPIN,IDC_SHININESS_EDIT,1.0f,1000.0f,TheMtl->Get_Shininess(PassIndex,IParams->GetTime()),1.0f);
160 UVChannelSpin[0] = SetupIntSpinner(dlg_wnd,IDC_STAGE0UVCHAN_SPIN,IDC_STAGE0UVCHAN_EDIT,1,99,1);
161 UVChannelSpin[1] = SetupIntSpinner(dlg_wnd,IDC_STAGE1UVCHAN_SPIN,IDC_STAGE1UVCHAN_EDIT,1,99,1);
162 break;
163 }
164
165 case WM_LBUTTONDOWN:
166 case WM_LBUTTONUP:
167 case WM_MOUSEMOVE:
168 {
169 IParams->RollupMouseMessage(dlg_wnd,message,wparam,lparam);
170 return FALSE;
171 }
172
173 case WM_COMMAND:
174 {
175 switch (id)
176 {
177 case IDC_COPY_SPECULAR_DIFFUSE:
178 TheMtl->Set_Copy_Specular_To_Diffuse(PassIndex,GetCheckBox(dlg_wnd, IDC_COPY_SPECULAR_DIFFUSE) == TRUE);
179 break;
180
181 case IDC_MAPPING0_COMBO:
182 if (code == CBN_SELCHANGE) {
183 val = SendDlgItemMessage(dlg_wnd,IDC_MAPPING0_COMBO,CB_GETCURSEL,0,0);
184 TheMtl->Set_Mapping_Type(PassIndex,0,val);
185 }
186 break;
187
188 case IDC_MAPPING1_COMBO:
189 if (code == CBN_SELCHANGE) {
190 val = SendDlgItemMessage(dlg_wnd,IDC_MAPPING1_COMBO,CB_GETCURSEL,0,0);
191 TheMtl->Set_Mapping_Type(PassIndex,1,val);
192 }
193 break;
194
195 case IDC_MAPPING0_ARGS_EDIT:
196 switch (code) {
197 case EN_SETFOCUS:
198 DisableAccelerators();
199 break;
200 case EN_KILLFOCUS:
201 EnableAccelerators();
202 break;
203 case EN_CHANGE:
204 int len = GetWindowTextLength(GetDlgItem(dlg_wnd, IDC_MAPPING0_ARGS_EDIT));
205 char *buffer = TheMtl->Get_Mapping_Arg_Buffer(PassIndex, 0, len);
206 GetWindowText(GetDlgItem(dlg_wnd, IDC_MAPPING0_ARGS_EDIT), buffer, len + 1);
207 break;
208 }
209 break;
210
211 case IDC_MAPPING1_ARGS_EDIT:
212 switch (code) {
213 case EN_SETFOCUS:
214 DisableAccelerators();
215 break;
216 case EN_KILLFOCUS:
217 EnableAccelerators();
218 break;
219 case EN_CHANGE:
220 int len = GetWindowTextLength(GetDlgItem(dlg_wnd, IDC_MAPPING1_ARGS_EDIT));
221 char *buffer = TheMtl->Get_Mapping_Arg_Buffer(PassIndex, 1, len);
222 GetWindowText(GetDlgItem(dlg_wnd, IDC_MAPPING1_ARGS_EDIT), buffer, len + 1);
223 break;
224 }
225 break;
226
227 case IDC_NO_TRANS:
228 TheMtl->Set_PSX_Translucency(PassIndex,GAMEMTL_PSX_TRANS_NONE);
229 break;
230
231 case IDC_100_TRANS:
232 TheMtl->Set_PSX_Translucency(PassIndex,GAMEMTL_PSX_TRANS_100);
233 break;
234
235 case IDC_50_TRANS:
236 TheMtl->Set_PSX_Translucency(PassIndex,GAMEMTL_PSX_TRANS_50);
237 break;
238
239 case IDC_25_TRANS:
240 TheMtl->Set_PSX_Translucency(PassIndex,GAMEMTL_PSX_TRANS_25);
241 break;
242
243 case IDC_MINUS_100_TRANS:
244 TheMtl->Set_PSX_Translucency(PassIndex,GAMEMTL_PSX_TRANS_MINUS_100);
245 break;
246
247 case IDC_NO_RT_LIGHTING:
248 TheMtl->Set_PSX_Lighting(PassIndex,!GetCheckBox(dlg_wnd, IDC_NO_RT_LIGHTING));
249 break;
250 }
251 break;
252 }
253
254 case CC_COLOR_CHANGE:
255 {
256 // just update all of the colors
257 TheMtl->Set_Ambient(PassIndex,IParams->GetTime(),AmbientSwatch->GetColor());
258 TheMtl->Set_Diffuse(PassIndex,IParams->GetTime(),DiffuseSwatch->GetColor());
259 TheMtl->Set_Specular(PassIndex,IParams->GetTime(),SpecularSwatch->GetColor());
260 TheMtl->Set_Emissive(PassIndex,IParams->GetTime(),EmissiveSwatch->GetColor());
261 TheMtl->Notify_Changed();
262 break;
263 }
264
265 case CC_SPINNER_CHANGE:
266 {
267 TheMtl->Set_Shininess(PassIndex,IParams->GetTime(),ShininessSpin->GetFVal());
268 TheMtl->Set_Opacity(PassIndex,IParams->GetTime(),OpacitySpin->GetFVal());
269 TheMtl->Set_Translucency(PassIndex,IParams->GetTime(),TranslucencySpin->GetFVal());
270 for (int i=0; i<MAX_STAGES; i++) {
271 TheMtl->Set_Map_Channel(PassIndex,i,UVChannelSpin[i]->GetIVal());
272 }
273 break;
274 }
275
276 case CC_SPINNER_BUTTONUP:
277 {
278 TheMtl->Notify_Changed();
279 break;
280 }
281
282 }
283 return FALSE;
284}

299void GameMtlVertexMaterialDlg::ReloadDialog(void)
300{
301 // Vertex Material Controls
302 DebugPrint("GameMtlVertexMaterialDlg::ReloadDialog\n");
303 assert(AmbientSwatch && DiffuseSwatch && SpecularSwatch && EmissiveSwatch);
304 assert(ShininessSpin && OpacitySpin && TranslucencySpin);
305
306 AmbientSwatch->InitColor(TheMtl->Get_Ambient(PassIndex,IParams->GetTime()));
307 DiffuseSwatch->InitColor(TheMtl->Get_Diffuse(PassIndex,IParams->GetTime()));
308 SpecularSwatch->InitColor(TheMtl->Get_Specular(PassIndex,IParams->GetTime()));
309 EmissiveSwatch->InitColor(TheMtl->Get_Emissive(PassIndex,IParams->GetTime()));
310
311 ShininessSpin->SetValue(TheMtl->Get_Shininess(PassIndex,IParams->GetTime()),FALSE);
312 OpacitySpin->SetValue(TheMtl->Get_Opacity(PassIndex,IParams->GetTime()),FALSE);
313 TranslucencySpin->SetValue(TheMtl->Get_Translucency(PassIndex,IParams->GetTime()),FALSE);
314 for (int i=0; i<MAX_STAGES; i++) {
315 UVChannelSpin[i]->SetValue(TheMtl->Get_Map_Channel(PassIndex,i),FALSE);
316 }
317
318 SetCheckBox(m_hWnd,IDC_COPY_SPECULAR_DIFFUSE, TheMtl->Get_Copy_Specular_To_Diffuse(PassIndex));
319
320 SendDlgItemMessage( m_hWnd,
321 IDC_MAPPING0_COMBO,
322 CB_SETCURSEL,
323 TheMtl->Get_Mapping_Type(PassIndex, 0),
324 0 );
325
326 SendDlgItemMessage( m_hWnd,
327 IDC_MAPPING1_COMBO,
328 CB_SETCURSEL,
329 TheMtl->Get_Mapping_Type(PassIndex, 1),
330 0 );
331
332 // PSX Controls
333 SetCheckBox(m_hWnd,IDC_NO_RT_LIGHTING, !TheMtl->Get_PSX_Lighting(PassIndex));
334 SetCheckBox(m_hWnd,IDC_NO_TRANS, false);
335 SetCheckBox(m_hWnd,IDC_100_TRANS, false);
336 SetCheckBox(m_hWnd,IDC_50_TRANS, false);
337 SetCheckBox(m_hWnd,IDC_25_TRANS, false);
338 SetCheckBox(m_hWnd,IDC_MINUS_100_TRANS, false);
339
340 switch (TheMtl->Get_PSX_Translucency(PassIndex)) {
341 case 0:
342 SetCheckBox(m_hWnd,IDC_NO_TRANS,true);
343 break;
344 case GAMEMTL_PSX_TRANS_100:
345 SetCheckBox(m_hWnd,IDC_100_TRANS,true);
346 break;
347 case GAMEMTL_PSX_TRANS_50:
348 SetCheckBox(m_hWnd,IDC_50_TRANS,true);
349 break;
350 case GAMEMTL_PSX_TRANS_25:
351 SetCheckBox(m_hWnd,IDC_25_TRANS,true);
352 break;
353 case GAMEMTL_PSX_TRANS_MINUS_100:
354 SetCheckBox(m_hWnd,IDC_MINUS_100_TRANS,true);
355 break;
356 }
357
358 // Reload contents of mapper args edit box:
359 char *buffer = TheMtl->Get_Mapping_Arg_Buffer(PassIndex, 0);
360 SetWindowText(GetDlgItem(m_hWnd, IDC_MAPPING0_ARGS_EDIT), buffer);
361 buffer = TheMtl->Get_Mapping_Arg_Buffer(PassIndex, 1);
362 SetWindowText(GetDlgItem(m_hWnd, IDC_MAPPING1_ARGS_EDIT), buffer);
363}
