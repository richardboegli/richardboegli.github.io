288static void populateQMSideComboBox(Int favSide, const LadderInfo *li = NULL)
289{
290 Int numPlayerTemplates = ThePlayerTemplateStore->getPlayerTemplateCount();
291 UnicodeString playerTemplateName;
292
293 GadgetComboBoxReset(comboBoxSide);
294
295 MultiplayerColorDefinition *def = TheMultiplayerSettings->getColor(PLAYERTEMPLATE_RANDOM);
296 Int newIndex = GadgetComboBoxAddEntry(comboBoxSide, TheGameText->fetch("GUI:Random"), def->getColor());
297 GadgetComboBoxSetItemData(comboBoxSide, newIndex, (void *)PLAYERTEMPLATE_RANDOM);
298
299 std::set<AsciiString> seenSides;
300
301 Int entryToSelect = 0; // select Random by default
302
303 for (Int c=0; c<numPlayerTemplates; ++c)
304 {
305 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(c);
306 if (!fac)
307 continue;
308
309 if (fac->getStartingBuilding().isEmpty())
310 continue;
311
312 AsciiString side;
313 side.format("SIDE:%s", fac->getSide().str());
314 if (seenSides.find(side) != seenSides.end())
315 continue;
316
317 if (li)
318 {
319 if (std::find(li->validFactions.begin(), li->validFactions.end(), fac->getSide()) == li->validFactions.end())
320 continue; // ladder doesn't allow it.
321 }
322
323 // Remove disallowed generals from the choice list.
324 // This is also enforced at GUI setup (GUIUtil.cpp and UserPreferences.cpp).
325 // @todo: unlock these when something rad happens
326 Bool disallowLockedGenerals = TRUE;
327 const GeneralPersona *general = TheChallengeGenerals->getGeneralByTemplateName(fac->getName());
328 Bool startsLocked = general ? !general->isStartingEnabled() : FALSE;
329 if (disallowLockedGenerals && startsLocked)
330 continue;
331
332 seenSides.insert(side);
333
334 newIndex = GadgetComboBoxAddEntry(comboBoxSide, TheGameText->fetch(side), def->getColor());
335 GadgetComboBoxSetItemData(comboBoxSide, newIndex, (void *)c);
336
337 if (c == favSide)
338 entryToSelect = newIndex;
339 }
340 seenSides.clear();
341
342 GadgetComboBoxSetSelectedPos(comboBoxSide, entryToSelect);
343 if (li && li->randomFactions)
344 comboBoxSide->winEnable(FALSE);
345 else
346 comboBoxSide->winEnable(TRUE);
347}

376static inline Bool isValidLadder( const LadderInfo *lad )
377{
378 if (lad && lad->index > 0 && lad->validQM)
379 {
380 PSPlayerStats stats = TheGameSpyPSMessageQueue->findPlayerStatsByID(TheGameSpyInfo->getLocalProfileID());
381 Int numWins = 0;
382 PerGeneralMap::iterator it;
383 for (it = stats.wins.begin(); it != stats.wins.end(); ++it)
384 {
385 numWins += it->second;
386 }
387 if (!lad->maxWins || lad->maxWins >=numWins)
388 {
389 if (!lad->minWins || lad->minWins<=numWins)
390 {
391 return TRUE;
392 }
393 }
394 }
395 return FALSE;
396}

398void PopulateQMLadderListBox( GameWindow *win )
399{
400 if (!parentWOLQuickMatch || !comboBoxLadder)
401 return;
402
403 isPopulatingLadderBox = true;
404
405 QuickMatchPreferences pref;
406 AsciiString userPrefFilename;
407 Int localProfile = TheGameSpyInfo->getLocalProfileID();
408
409 Color specialColor = GameSpyColor[GSCOLOR_MAP_SELECTED];
410 Color normalColor = GameSpyColor[GSCOLOR_MAP_UNSELECTED];
411 Color favoriteColor = GameSpyColor[GSCOLOR_MAP_UNSELECTED];
412 Int index;
413 GadgetListBoxReset( win );
414
415 std::set<const LadderInfo *> usedLadders;
416
417 // start with "No Ladder"
418 index = GadgetListBoxAddEntryText( win, TheGameText->fetch("GUI:NoLadder"), normalColor, -1 );
419 GadgetListBoxSetItemData( win, 0, index );
420
421 // add the last ladder
422 Int selectedPos = 0;
423 AsciiString lastLadderAddr = pref.getLastLadderAddr();
424 UnsignedShort lastLadderPort = pref.getLastLadderPort();
425 const LadderInfo *info = TheLadderList->findLadder( lastLadderAddr, lastLadderPort );
426 if (isValidLadder(info))
427 {
428 usedLadders.insert(info);
429 index = GadgetListBoxAddEntryText( win, info->name, favoriteColor, -1 );
430 GadgetListBoxSetItemData( win, (void *)(info->index), index );
431 selectedPos = index;
432 }
433
434 // our recent ladders
435 LadderPreferences ladPref;
436 ladPref.loadProfile( localProfile );
437 const LadderPrefMap recentLadders = ladPref.getRecentLadders();
438 for (LadderPrefMap::const_iterator cit = recentLadders.begin(); cit != recentLadders.end(); ++cit)
439 {
440 AsciiString addr = cit->second.address;
441 UnsignedShort port = cit->second.port;
442 if (addr == lastLadderAddr && port == lastLadderPort)
443 continue;
444 const LadderInfo *info = TheLadderList->findLadder( addr, port );
445 if (isValidLadder(info) && usedLadders.find(info) == usedLadders.end())
446 {
447 usedLadders.insert(info);
448 index = GadgetListBoxAddEntryText( win, info->name, favoriteColor, -1 );
449 GadgetListBoxSetItemData( win, (void *)(info->index), index );
450 }
451 }
452
453 // special ladders
454 const LadderInfoList *lil = TheLadderList->getSpecialLadders();
455 LadderInfoList::const_iterator lit;
456 for (lit = lil->begin(); lit != lil->end(); ++lit)
457 {
458 const LadderInfo *info = *lit;
459 if (isValidLadder(info) && usedLadders.find(info) == usedLadders.end())
460 {
461 usedLadders.insert(info);
462 index = GadgetListBoxAddEntryText( win, info->name, specialColor, -1 );
463 GadgetListBoxSetItemData( win, (void *)(info->index), index );
464 }
465 }
466
467 // standard ladders
468 lil = TheLadderList->getStandardLadders();
469 for (lit = lil->begin(); lit != lil->end(); ++lit)
470 {
471 const LadderInfo *info = *lit;
472 if (isValidLadder(info) && usedLadders.find(info) == usedLadders.end())
473 {
474 usedLadders.insert(info);
475 index = GadgetListBoxAddEntryText( win, info->name, normalColor, -1 );
476 GadgetListBoxSetItemData( win, (void *)(info->index), index );
477 }
478 }
479
480 GadgetListBoxSetSelected( win, selectedPos );
481 isPopulatingLadderBox = false;
482}
