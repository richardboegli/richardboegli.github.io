111void W3DParticleSystemManager::doParticles(RenderInfoClass &rinfo)
112{
113
114 if (m_readyToRender == false)
115 return;
116
117 // external mechanism must tell us when it's OK to render again...
118 m_readyToRender = false;
119
120 //reset each frame
122 m_onScreenParticleCount = 0;
123
124 Int visibleSmudgeCount = 0;
125 if (TheSmudgeManager)
126 TheSmudgeManager->setSmudgeCountLastFrame(0); //keep track of visible smudges
127
128 const FrustumClass & frustum = rinfo.Camera.Get_Frustum();
129 AABoxClass bbox;
130
131 //Get a bounding box around our visible universe. Bounded by terrain and the sky
132 //so much tighter fitting volume than what's actually visible. This will cull
133 //particles falling under the ground.
134
135 TheTerrainRenderObject->getMaximumVisibleBox(frustum, &bbox, TRUE);
136
137 //@todo lorenzen sez: put these in registers for sure
138 Real bcX = bbox.Center.X;
139 Real bcY = bbox.Center.Y;
140 Real bcZ = bbox.Center.Z;
141 Real beX = bbox.Extent.X;
142 Real beY = bbox.Extent.Y;
143 Real beZ = bbox.Extent.Z;
144
145 unsigned int personalities[MAX_POINTS_PER_GROUP];
146
147
148 m_fieldParticleCount = 0;
149
150 SmudgeSet *set=NULL;
151 if (TheSmudgeManager)
152 set=TheSmudgeManager->addSmudgeSet(); //global smudge set through which all smudges are rendered.
153
154 ParticleSystemManager::ParticleSystemList &particleSysList = TheParticleSystemManager->getAllParticleSystems();
155 for( ParticleSystemManager::ParticleSystemListIt it = particleSysList.begin(); it != particleSysList.end(); ++it)
156 {
157 ParticleSystem *sys = (*it);
158 if (!sys) {
159 continue;
160 }
161
162 // only look at particle/point style systems
163 if (sys->isUsingDrawables())
164 continue;
165
166 //temporary hack that checks if texture name starts with "SMUD" - if so, we can assume it's a smudge type
167 if (/*sys->isUsingSmudge()*/ *((DWORD *)sys->getParticleTypeName().str()) == 0x44554D53)
168 {
169 if (TheSmudgeManager && ((W3DSmudgeManager*)TheSmudgeManager)->getHardwareSupport() && TheGlobalData->m_useHeatEffects)
170 {
171 //set-up all the per-particle
172 for (Particle *p = sys->getFirstParticle(); p; p = p->m_systemNext)
173 {
174 const Coord3D *pos = p->getPosition();
175 Real psize = p->getSize();
176
177 //Cull particle to edges of screen and terrain.
178 if (WWMath::Fabs( pos->x - bcX ) > ( beX + psize ) )
179 continue;
180
181 if (WWMath::Fabs( pos->y - bcY ) > ( beY + psize ) )
182 continue;
183
184 if (WWMath::Fabs( pos->z - bcZ ) > ( beZ + psize ) )
185 continue;
186
187 Smudge *smudge = set->addSmudgeToSet();
188
189 smudge->m_pos.Set( pos->x, pos->y, pos->z );
190 smudge->m_offset.Set( GameClientRandomValueReal(-0.06f,0.06f), GameClientRandomValueReal(-0.03f,0.03f) );
191 smudge->m_size = psize;
192 smudge->m_opacity = p->getAlpha();
193 visibleSmudgeCount++;
194 }
195 }
196 continue;
197 }
198
200 // initialize them here still, of course
201 // build W3D particle buffer
202 Int count = 0;
203 Vector3 *posArray = m_posBuffer->Get_Array();
204 Real *sizeArray = m_sizeBuffer->Get_Array();
205 Vector4 *RGBAArray = m_RGBABuffer->Get_Array();
206 uint8 *angleArray = m_angleBuffer->Get_Array();
207 const Coord3D *pos;
208 const RGBColor *color;
209 Real psize;
210
211
212
213 //set-up all the per-particle
214 for (Particle *p = sys->getFirstParticle(); p; p = p->m_systemNext)
215 {
216 pos = p->getPosition();
217 psize = p->getSize();
218
219 //Cull particle to edges of screen and terrain.
220 if (WWMath::Fabs(pos->x - bcX) > (beX + psize))
221 continue;
222
223 if (WWMath::Fabs(pos->y - bcY) > (beY + psize))
224 continue;
225
226 if (WWMath::Fabs(pos->z - bcZ) > (beZ + psize))
227 continue;
228
229 m_fieldParticleCount += ( sys->getPriority() == AREA_EFFECT && sys->m_isGroundAligned != FALSE );
230
231 //@todo lorenzen sez: use pointer arithmetic for these arrays
232 personalities[count] = p->getPersonality();
233
234 posArray[count].X = pos->x;
235 posArray[count].Y = pos->y;
236 posArray[count].Z = pos->z;
237
238 sizeArray[count] = psize;
239
240 color = p->getColor();
241 RGBAArray[count].X = color->red;
242 RGBAArray[count].Y = color->green;
243 RGBAArray[count].Z = color->blue;
244 RGBAArray[count].W = p->getAlpha();
245
246 angleArray[count] = (uint8)(p->getAngle() * 255.0f / (2.0f * PI));
247
248 if (++count == MAX_POINTS_PER_GROUP)
249 break;
250 }
251
252 if ( count == 0 )
253 continue; //this system has no particles to render
254
255 TextureClass *texture = W3DDisplay::m_assetManager->Get_Texture( sys->getParticleTypeName().str() );
256
257 if ( m_streakLine && sys->isUsingStreak() && (count >= 2) )
258 {
259 m_streakLine->Reset_Line();
260
261 m_streakLine->Set_Texture( texture );
262 texture->Release_Ref();//release reference since it's held by streakline
263 switch( sys->getShaderType() )
264 {
265 case ParticleSystemInfo::ADDITIVE:
266 m_streakLine->Set_Shader( ShaderClass::_PresetAdditiveSpriteShader );
267 break;
268 case ParticleSystemInfo::ALPHA:
269 m_streakLine->Set_Shader( ShaderClass::_PresetAlphaSpriteShader );
270 break;
271 case ParticleSystemInfo::ALPHA_TEST:
272 m_streakLine->Set_Shader( ShaderClass::_PresetATestSpriteShader );
273 break;
274 case ParticleSystemInfo::MULTIPLY:
275 m_streakLine->Set_Shader( ShaderClass::_PresetMultiplicativeSpriteShader );
276 break;
277 }
278
279 //UPDATE THE STREAK'S ARRAYS
280 m_streakLine->Set_LocsWidthsColors(
281 count,
282 m_posBuffer->Get_Array(),
283 m_sizeBuffer->Get_Array(),
284 m_RGBABuffer->Get_Array(),
285 &personalities[0]
286 );
287
288 //WWASSERT( m_streakLine->Get_Num_Points() == count );
289
290 // This is the happy place for this!
291 RGBAArray[0].X = 0;//eliminates the scissor edge on the trailing edge of the streak
292 RGBAArray[0].Y = 0;
293 RGBAArray[0].Z = 0;
294 RGBAArray[0].W = 0;
295
296
297 //RENDER STREAK!
298 m_streakLine->Render( rinfo );
299
300 }
301 else
302 {
303
304 WWASSERT( m_pointGroup );
305
306 if ( m_pointGroup ) // this catches the particle and volumeparticle cases
307 {
308 // render all the systems' particles
309 m_pointGroup->Set_Texture( texture );
310 texture->Release_Ref();//release reference since it's held by pointGroup
311 m_pointGroup->Set_Flag( PointGroupClass::TRANSFORM, true ); // transform to screen space
312
313 switch( sys->getShaderType() )
314 {
315 case ParticleSystemInfo::ADDITIVE:
316 m_pointGroup->Set_Shader( ShaderClass::_PresetAdditiveSpriteShader );
317 break;
318 case ParticleSystemInfo::ALPHA:
319 m_pointGroup->Set_Shader( ShaderClass::_PresetAlphaSpriteShader );
320 break;
321 case ParticleSystemInfo::ALPHA_TEST:
322 m_pointGroup->Set_Shader( ShaderClass::_PresetATestSpriteShader );
323 break;
324 case ParticleSystemInfo::MULTIPLY:
325 m_pointGroup->Set_Shader( ShaderClass::_PresetMultiplicativeSpriteShader );
326 break;
327 }
328
330 m_pointGroup->Set_Point_Mode( PointGroupClass::QUADS );
331 m_pointGroup->Set_Arrays( m_posBuffer, m_RGBABuffer, NULL, m_sizeBuffer, m_angleBuffer, NULL, count );
332 m_pointGroup->Set_Billboard(sys->shouldBillboard());
333
336 m_pointGroup->Set_Point_Frame( 0 );
337
338 //RENDER IT!
339 if( sys->getVolumeParticleDepth() > 1 )
340 {
341 m_pointGroup->RenderVolumeParticle( rinfo, sys->getVolumeParticleDepth() );
342 }
343 else
344 m_pointGroup->Render( rinfo );
345
346 }
347 }
348
349
351 //add particle count to total
352 m_onScreenParticleCount += count;
353
354 /*
355 // draw the wind vector for this particle system on the screen
356 UnsignedInt width = TheDisplay->getWidth();
357 UnsignedInt height = TheDisplay->getHeight();
358 Coord3D worldStart, worldEnd;
359 ICoord2D pixelStart, pixelEnd;
360 sys->getPosition( &worldStart );
361 worldEnd.x = Cos( sys->getWindAngle() ) * 50.0f + worldStart.x;
362 worldEnd.y = Sin( sys->getWindAngle() ) * 50.0f + worldStart.y;
363 worldEnd.z = worldStart.z;
364 TheTacticalView->worldToScreen( &worldStart, &pixelStart );
365 TheTacticalView->worldToScreen( &worldEnd, &pixelEnd );
366 Color colorStart = GameMakeColor( 255, 255, 255, 255 );
367 Color colorEnd = GameMakeColor( 255, 128, 128, 255 );
368 TheDisplay->drawLine( pixelStart.x, pixelStart.y, pixelEnd.x, pixelEnd.y, 1.0f, colorStart, colorEnd );
369 */
370
371
372 }// next system
373
375 TheParticleSystemManager->setOnScreenParticleCount(m_onScreenParticleCount);
376
377 //Draw any particles belonging to weather effects
378 if (TheSnowManager)
379 ((W3DSnowManager *)TheSnowManager)->render(rinfo);
380
381 //Now process screen smudges which are particles that distort the background behind them.
382 if(TheSmudgeManager)
383 {
384 ((W3DSmudgeManager *)TheSmudgeManager)->render(rinfo);
385 TheSmudgeManager->reset(); //clear all the smudges after rendering since we fill again each frame.
386 TheSmudgeManager->setSmudgeCountLastFrame(visibleSmudgeCount);
387 }
388}
