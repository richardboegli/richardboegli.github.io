58void SpecialPowerCreate::onBuildComplete( void )
59{
60 if( ! shouldDoOnBuildComplete() )
61 return;
62
63 CreateModule::onBuildComplete(); // extend
64
65 for (BehaviorModule** m = getObject()->getBehaviorModules(); *m; ++m)
66 {
67 SpecialPowerModuleInterface* sp = (*m)->getSpecialPower();
68 if (!sp)
69 continue;
70
71 sp->onSpecialPowerCreation();
72 }
73}
