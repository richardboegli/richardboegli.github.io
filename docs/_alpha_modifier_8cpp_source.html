53void AlphaModifierClass::ModifyObject(TimeValue t, ModContext &mc, ObjectState *os, INode *node)
54{
55 if (!os->obj->IsSubClassOf(triObjectClassID))
56 {
57 return;
58 }
59
60
61 // Get a mesh from input object
62 TriObject *object = (TriObject*)os->obj;
63
64 Mesh *mesh = &object->mesh;
65
66 assert(mesh);
67
68 int numVert = mesh->getNumVerts();
69 int i = 0;
70 float *vdata = NULL;
71
72 // Get parameters from pblock
73 float sparam = 0.0f;
74 Interval valid = LocalValidity(t);
75 int pass = 1;
76
77 pblock->GetValue(DL_EDIT_VALUE, t, sparam, valid);
78
79 // If needed a control could be put into the dialog box to specify which
80 // pass to apply the alpha values to. At this time, it was decided to
81 // not implement this because of the complexity to the artist and the
82 // performance issues in game.
83 //pblock->GetValue(DL_EDIT_PASS, t, pass, valid);
84
85
86 // Start from 0.
87 pass -= 1;
88 assert(pass >= 0);
89
90 // Use a channel for each pass.
91 vdata = mesh->vertexFloat(ALPHA_VERTEX_CHANNEL + pass);
92
93 if (!vdata)
94 {
95 // Turn on the channel for vertex alpha support.
96 mesh->setVDataSupport(ALPHA_VERTEX_CHANNEL + pass);
97 vdata = mesh->vertexFloat(ALPHA_VERTEX_CHANNEL + pass);
98
99 assert(vdata);
100
101 for (i = 0; i < numVert; i++)
102 {
103 if (mesh->VertSel()[i])
104 {
105 vdata[i] = 0.0f;
106 }
107 }
108
109 }
110
111 // Tracks the state of the FIND check box.
112 int box_checked = 0;
113
114
115 if (Message == AM_UPDATE_DATA)
116 {
117 // The user has updated the dialog box, so update the data.
118 assert(vdata);
119
120 pblock->GetValue(DL_FIND_CHECK_BOX, t, box_checked, valid);
121 if (!box_checked)
122 {
123 for (i = 0; i < numVert; i++)
124 {
125 if (SelectedVertices[i])
126 {
127 vdata[i] = sparam;
128 }
129 }
130 }
131 }
132
133 if (Message == AM_BOX_CHECKED)
134 {
135 pblock->GetValue(DL_FIND_CHECK_BOX, t, box_checked, valid);
136 }
137
138
139 // The user is trying to find vertices with certain values.
140 if (box_checked)
141 {
142 assert(vdata);
143 // Find the vertices that have the user entered value.
144 for (i = 0; i < numVert; i++)
145 {
146 if (vdata[i] == sparam)
147 {
148 mesh->VertSel().Set(i);
149 SelectedVertices.Set(i);
150 }
151 else
152 {
153 mesh->VertSel().Clear(i);
154 SelectedVertices.Clear(i);
155 }
156
157 }
158
159 }
160
161 if (Message == AM_INITIALIZE)
162 {
163 assert(vdata);
164
165 SelectedVertices = mesh->VertSel();
166
167 for (i = 0; i < numVert; i++)
168 {
169 if (SelectedVertices[i])
170 {
171 // Set the value in the dialog box to the value of the
172 // first selected vertex.
173 pblock->SetValue(DL_EDIT_VALUE, t, vdata[i]);
174 break;
175 }
176 }
177
178 }
179
180
181 // Always select the vertices that have been saved by the modifier.
182 // This must be done because the mesh changes each time ModfiyObject is called.
183 for (i = 0; i < numVert; i++)
184 {
185 if (SelectedVertices[i])
186 {
187 mesh->VertSel().Set(i);
188 }
189 else
190 {
191 mesh->VertSel().Clear(i);
192 }
193
194 }
195
196 // Display the vertices.
197 mesh->SetDispFlag(DISP_SELVERTS | DISP_VERTTICKS);
198 mesh->selLevel = MESH_VERTEX;
199 object->UpdateValidity(SELECT_CHAN_NUM, object->ChannelValidity (t, SELECT_CHAN_NUM));
200
201 // Clear messages.
202 Message = AM_NOTHING;
203}

280 | Basic implementation of a dialog handler
281\*===========================================================================*/
282
283BOOL AlphaModDlgProc::DlgProc(TimeValue t, IParamMap2 *map, HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
284{
285 int id = LOWORD(wParam);
286 int code = HIWORD(wParam);
287
288 switch (msg)
289 {
290 case WM_INITDIALOG:
291 AlphaModifier->Message = AM_INITIALIZE;
292 break;
293 case WM_DESTROY:
294 break;
295 case WM_COMMAND:
296 switch (code)
297 {
298 case EN_UPDATE:
299 break;
300 case EN_SETFOCUS:
301 break;
302 case EN_KILLFOCUS:
303 break;
304 case EN_CHANGE:
305 break;
306 }
307 if (id == IDC_ALPHA_EDIT)
308 {
309 AlphaModifier->Message = AM_UPDATE_DATA;
310 }
311 if (id == IDC_ALPHA_CHECKBOX)
312 {
313 AlphaModifier->Message = AM_BOX_CHECKED;
314 }
315 break;
316
317 case WM_NOTIFY:
318 if (id == IDC_ALPHA_EDIT)
319 {
320 AlphaModifier->Message = AM_UPDATE_DATA;
321 }
322 break;
323
324 default:
325 break;
326
327
328 }
329 return FALSE;
330}
