142void BattleBusSlowDeathBehavior::beginSlowDeath( const DamageInfo *damageInfo )
143{
144 Object *me = getObject();
145 const BattleBusSlowDeathBehaviorModuleData * data = getBattleBusSlowDeathBehaviorModuleData();
146 if( !m_isRealDeath )
147 {
148 // We can intercept and do our extra slow death if this is not from a real onDie
149 m_isInFirstDeath = TRUE;
150 m_groundCheckFrame = TheGameLogic->getFrame() + GROUND_CHECK_DELAY;
151
152 // First do the special effects
153 FXList::doFXObj(data->m_fxStartUndeath, me );
154 ObjectCreationList::create(data->m_oclStartUndeath, me, NULL );
155
156 if( me->getAI() )
157 {
158 // Then stop what we were doing
159 me->getAI()->aiIdle(CMD_FROM_AI);
160 }
161
162 if( me->getPhysics() )
163 {
164 // Then stop physically
165 me->getPhysics()->clearAcceleration();
166 me->getPhysics()->scrubVelocity2D(0);
167
168 // Then get chucked in the air
169 Coord3D throwForce;
170 throwForce.x = 0;
171 throwForce.y = 0;
172 throwForce.z = data->m_throwForce;
173 me->getPhysics()->applyShock(&throwForce);
174 me->getPhysics()->applyRandomRotation();
175 }
176
177 // And finally hit those inside for some damage
178 if( me->getContain() )
179 me->getContain()->processDamageToContained(data->m_percentDamageToPassengers);
180
181 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
182 }
183 else
184 {
185 // If a real death, we aren't allowed to do anything
186 SlowDeathBehavior::beginSlowDeath( damageInfo );
187 }
188
189} // end beginSlowDeath

193UpdateSleepTime BattleBusSlowDeathBehavior::update( void )
194{
195 Object *me = getObject();
196 const BattleBusSlowDeathBehaviorModuleData * data = getBattleBusSlowDeathBehaviorModuleData();
197 if( m_isInFirstDeath )
198 {
199 // First death means we are doing our throw up in the air conversion
200
201 if( (m_groundCheckFrame< TheGameLogic->getFrame() ) && !me->isAboveTerrain() )
202 {
203 // We're done since we hit the ground
204 m_isInFirstDeath = FALSE;
205
206 // Do the special FX
207 FXList::doFXObj(data->m_fxHitGround, me );
208 ObjectCreationList::create(data->m_oclHitGround, me, NULL );
209 me->setModelConditionState(MODELCONDITION_SECOND_LIFE);
210
211 // And stop us forever
212 if( me->getAI() )
213 {
214 // Stop doing anything again (could try to move while in air)
215 me->getAI()->aiIdle(CMD_FROM_AI);
216 }
217 if( me->getPhysics() )
218 {
219 // And stop physically again
220 me->getPhysics()->clearAcceleration();
221 me->getPhysics()->scrubVelocity2D(0);
222 }
223 me->setDisabled(DISABLED_HELD);
224
225 // We can only sleep if we don't have to watch out for being empty.
226 if( data->m_emptyHulkDestructionDelay == 0 )
227 return UPDATE_SLEEP_FOREVER;
228 else
229 return UPDATE_SLEEP_NONE; // None, so we check for empty as soon as possible
230 }
231 return UPDATE_SLEEP_NONE;// None, since we are waiting to be able to check for ground collision
232 }
233 else if( m_isRealDeath )
234 {
235 // If a real death, we aren't allowed to do anything
236
237 return SlowDeathBehavior::update();
238 }
239 else
240 {
241 // If neither death is active, we must be awake to check for emptiness
242
243 const ContainModuleInterface *contain = me->getContain();
244 // Safety, no need to be awake if no special case to wait for
245 if( contain == NULL )
246 return UPDATE_SLEEP_FOREVER;
247
248 if( m_penaltyDeathFrame != 0 )
249 {
250 // We have been timed for death, see if we have re-filled first
251 if( contain->getContainCount() > 0 )
252 {
253 m_penaltyDeathFrame = 0;
254 return UPDATE_SLEEP(EMPTY_HULK_CHECK_DELAY);
255 }
256 else if( TheGameLogic->getFrame() > m_penaltyDeathFrame )
257 {
258 // No salvation, see if the timer is up to be killed. Penalty prevents effects, Extra 4 prevents this die module from recalling.
259 // Can't use Suicided death because a Terrorist actually inflicts Suicide deaths on others.
260 me->kill(DAMAGE_PENALTY, DEATH_EXTRA_4);
261 return UPDATE_SLEEP_FOREVER;// Forever, since we are dead and some other death module is in charge
262 }
263 else
264 {
265 return UPDATE_SLEEP(EMPTY_HULK_CHECK_DELAY);
266 }
267 }
268 else
269 {
270 // We are not marked for death, so see if we should be
271 if( contain->getContainCount() == 0 )
272 {
273 m_penaltyDeathFrame = TheGameLogic->getFrame() + data->m_emptyHulkDestructionDelay;
274 }
275 return UPDATE_SLEEP(EMPTY_HULK_CHECK_DELAY);// Stay awake regardless
276 }
277 }
278} // end update
