68enum UpdateSleepTime
69{
70 UPDATE_SLEEP_INVALID = 0,
71 UPDATE_SLEEP_NONE = 1,
72 // (we use 0x3fffffff so that we can add offsets and not overflow...
73 // and also 'cuz we shift the value up by two bits for the phase.
74 // note that at 30fps, this is ~414 days...
75 UPDATE_SLEEP_FOREVER = 0x3fffffff
76};
77#define UPDATE_SLEEP(numFrames) ((UpdateSleepTime)(numFrames))
78
79// this is used to declare in which "phase" sleepy updates are called.
80// all phase 0 are called before all phase 1 within a given frame, etc.
81// this is really only used to facilitate some internal operations
82// in an efficient way while still maintaining order-dependency; you should
83// really never specify anything other than PHASE_NORMAL without very
84// careful deliberation. If you need to, talk it over with folks first. (srj)
85enum SleepyUpdatePhase
86{
87 // reserve 2 bits for phase. this still leaves us 30 bits for frame counter,
88 // which, at 30fps, will still run for ~414 days without overflowing...
89 PHASE_INITIAL = 0,
90 PHASE_PHYSICS = 1,
91 PHASE_NORMAL = 2,
92 PHASE_FINAL = 3
93};

207 // anything greater than "forever" is still "forever"

209 if (frame > UPDATE_SLEEP_FOREVER)
210 frame = UPDATE_SLEEP_FOREVER;
