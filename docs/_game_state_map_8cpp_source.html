79static void embedPristineMap( AsciiString map, Xfer *xfer )
80{
81
82 // open the map file
83 File *file = TheFileSystem->openFile( map.str(), File::READ | File::BINARY );
84 if( file == NULL )
85 {
86
87 DEBUG_CRASH(( "embedPristineMap - Error opening source file '%s'\n", map.str() ));
88 throw SC_INVALID_DATA;
89
90 } // end if
91
92 // how big is the map file
93 Int fileSize = file->seek( 0, File::END );
94
95 // rewind to beginning of file
96 file->seek( 0, File::START );
97
98 // allocate buffer big enough to hold the entire map file
99 char *buffer = new char[ fileSize ];
100 if( buffer == NULL )
101 {
102
103 DEBUG_CRASH(( "embedPristineMap - Unable to allocate buffer for file '%s'\n", map.str() ));
104 throw SC_INVALID_DATA;
105
106 } // end if
107
108 // copy the file to the buffer
109 if( file->read( buffer, fileSize ) != fileSize )
110 {
111
112 DEBUG_CRASH(( "embeddPristineMap - Error reading from file '%s'\n", map.str() ));
113 throw SC_INVALID_DATA;
114
115 } // end if
116
117 // close the BIG file
118 file->close();
119
120 // write the contents to the save file
121 DEBUG_ASSERTCRASH( xfer->getXferMode() == XFER_SAVE, ("embedPristineMap - Unsupposed xfer mode\n") );
122 xfer->beginBlock();
123 xfer->xferUser( buffer, fileSize );
124 xfer->endBlock();
125
126 // delete the buffer
127 delete [] buffer;
128
129} // end embedPristineMap

135static void embedInUseMap( AsciiString map, Xfer *xfer )
136{
137 FILE *fp = fopen( map.str(), "rb" );
138
139 // sanity
140 if( fp == NULL )
141 {
142
143 DEBUG_CRASH(( "embedInUseMap - Unable to open file '%s'\n", map.str() ));
144 throw SC_INVALID_DATA;
145
146 } // end if
147
148 // how big is the file
149 fseek( fp, 0, SEEK_END );
150 Int fileSize = ftell( fp );
151
152 // rewind file back to start
153 fseek( fp, 0, SEEK_SET );
154
155 // allocate a buffer big enough for the entire file
156 char *buffer = new char[ fileSize ];
157 if( buffer == NULL )
158 {
159
160 DEBUG_CRASH(( "embedInUseMap - Unable to allocate buffer for file '%s'\n", map.str() ));
161 throw SC_INVALID_DATA;
162
163 } // end if
164
165 // read the entire file
166 if( fread( buffer, 1, fileSize, fp ) != fileSize )
167 {
168
169 DEBUG_CRASH(( "embedInUseMap - Error reading from file '%s'\n", map.str() ));
170 throw SC_INVALID_DATA;
171
172 } // end if
173
174 // embed file into xfer stream
175 xfer->beginBlock();
176 xfer->xferUser( buffer, fileSize );
177 xfer->endBlock();
178
179 // close the file
180 fclose( fp );
181
182 // delete buffer
183 delete [] buffer;
184
185} // embedInUseMap

190static void extractAndSaveMap( AsciiString mapToSave, Xfer *xfer )
191{
192 UnsignedInt dataSize;
193
194 // open handle to output file
195 FILE *fp = fopen( mapToSave.str(), "w+b" );
196 if( fp == NULL )
197 {
198
199 DEBUG_CRASH(( "extractAndSaveMap - Unable to open file '%s'\n", mapToSave.str() ));
200 throw SC_INVALID_DATA;
201
202 } // en
203
204 // read data size from file
205 dataSize = xfer->beginBlock();
206
207 // allocate buffer big enough for the entire map file
208 char *buffer = new char[ dataSize ];
209 if( buffer == NULL )
210 {
211
212 DEBUG_CRASH(( "extractAndSaveMap - Unable to allocate buffer for file '%s'\n", mapToSave.str() ));
213 throw SC_INVALID_DATA;
214
215 } // end if
216
217 // read map file
218 xfer->xferUser( buffer, dataSize );
219
220 // write contents of buffer to new file
221 if( fwrite( buffer, 1, dataSize, fp ) != dataSize )
222 {
223
224 DEBUG_CRASH(( "extractAndSaveMap - Error writing to file '%s'\n", mapToSave.str() ));
225 throw SC_INVALID_DATA;
226
227 } // end if
228
229 // close the new file
230 fclose( fp );
231
232 // end of data block
233 xfer->endBlock();
234
235 // delete the buffer
236 delete [] buffer;
237
238} // end extractAndSaveMap

248void GameStateMap::xfer( Xfer *xfer )
249{
250 if( xfer->getXferMode() == XFER_LOAD )
251 {
252 TheGameLogic->setLoadingSave( TRUE );
253 }
254
255 // version
256 const XferVersion currentVersion = 2;
257 XferVersion version = currentVersion;
258 xfer->xferVersion( &version, currentVersion );
259
260 // get save game info
261 SaveGameInfo *saveGameInfo = TheGameState->getSaveGameInfo();
262
263 //
264 // map filename, for purposes of saving we will always be saving a with a map filename
265 // that refers to map in the save directory so we must always save a filename into
266 // the file that is in the save directory
267 //
268 Bool firstSave = FALSE; // TRUE if we haven't yet saved a pristine load of a new map
269 if( xfer->getXferMode() == XFER_SAVE )
270 {
271
272 AsciiString mapLeafName = TheGameState->getMapLeafName(TheGlobalData->m_mapName);
273
274 // construct filename to map in the save directory
275 saveGameInfo->saveGameMapName = TheGameState->getFilePathInSaveDirectory(mapLeafName);
276
277 // write map name. For cross-machine compatibility, we always write
278 // it as just "Save\filename", not a full path.
279 {
280 AsciiString tmp = TheGameState->realMapPathToPortableMapPath(saveGameInfo->saveGameMapName);
281 xfer->xferAsciiString( &tmp );
282 }
283
284 //
285 // write pristine map name which is already in the member 'pristineMapName' from
286 // a previous load, or in the instance where we are saving for the first time
287 // and the global data map name refers to a pristine map we will copy it in there first
288 //
289 if (!TheGameState->isInSaveDirectory(TheGlobalData->m_mapName))
290 {
291
292 // copy the pristine name
293 saveGameInfo->pristineMapName = TheGlobalData->m_mapName;
294
295 //
296 // this is also an indication that we are saving for the first time a brand new
297 // map that has never been saved into this save file before (a save is also considered
298 // to be a first save as long as we are writing data to disk without having loaded
299 // this particluar map from the save file ... so if you load USA01 for the first
300 // time and save, that is a first save ... then, without quitting, if you save
301 // again that is *also* considered a first save). First save just determines
302 // whether the map file we embed in the save file is taken from the maps directory
303 // or from the temporary map extracted to the save directory from a load
304 //
305 firstSave = TRUE;
306
307 } // end if
308
309 // save the pristine name
310 // For cross-machine compatibility, we always write
311 // it as just "Save\filename", not a full path.
312 {
313 AsciiString tmp = TheGameState->realMapPathToPortableMapPath(saveGameInfo->pristineMapName);
314 xfer->xferAsciiString( &tmp );
315 }
316
317 if (currentVersion >= 2)
318 {
319 // save the game mode.
320 Int gameMode = TheGameLogic->getGameMode();
321 xfer->xferInt( &gameMode);
322 }
323
324 } // end if, save
325 else
326 {
327
328 // read the save game map name
329 AsciiString tmp;
330 xfer->xferAsciiString( &tmp );
331
332 saveGameInfo->saveGameMapName = TheGameState->portableMapPathToRealMapPath(tmp);
333
334 if (!TheGameState->isInSaveDirectory(saveGameInfo->saveGameMapName))
335 {
336 DEBUG_CRASH(("GameState::xfer - The map filename read from the file '%s' is not in the SAVE directory, but should be\n",
337 saveGameInfo->saveGameMapName.str()) );
338 throw SC_INVALID_DATA;
339 }
340
341 // set this map as the map to load in the global data
342 TheWritableGlobalData->m_mapName = saveGameInfo->saveGameMapName;
343
344 // read the pristine map filename
345 xfer->xferAsciiString( &saveGameInfo->pristineMapName );
346 saveGameInfo->pristineMapName = TheGameState->portableMapPathToRealMapPath(saveGameInfo->pristineMapName);
347
348 if (currentVersion >= 2)
349 {
350 // get the game mode.
351 Int gameMode;
352 xfer->xferInt(&gameMode);
353 TheGameLogic->setGameMode(gameMode);
354 }
355
356 } // end else, load
357
358 // map data
359 if( xfer->getXferMode() == XFER_SAVE )
360 {
361
362 //
363 // if this is a first save from a pristine map load, we need to copy the pristine
364 // map into the save game file
365 //
366 if( firstSave == TRUE )
367 {
368
369 embedPristineMap( saveGameInfo->pristineMapName, xfer );
370
371 } // end if, first save
372 else
373 {
374
375 //
376 // this is *NOT* a first save from a pristine map, just read the map file
377 // that was extracted from the save game file during the last load and embedd
378 // that into the save game file
379 //
380 embedInUseMap( saveGameInfo->saveGameMapName, xfer );
381
382 } // end else
383
384 } // end if, save
385 else
386 {
387
388 //
389 // take the embedded map file out of the save file, and save as its own .map file
390 // in the save directory temporarily
391 //
392 extractAndSaveMap( saveGameInfo->saveGameMapName, xfer );
393
394 } // end else
395
396 //
397 // it's important that early in the load process, we xfer the object ID counter
398 // in the game logic ... this is necessary because there are flows of code that
399 // create objects during the load of a save game that is *NOT FROM THE OBJECT BLOCK* of
400 // code, such as loading bridges (which creates the bridge and tower objects). We
401 // are fancy and "do the right thing" in those situations, but we don't want to run
402 // the risk of these objects being created and having overlapping IDs of anything
403 // that we will load from the save file
404 //
405 ObjectID highObjectID = TheGameLogic->getObjectIDCounter();
406 xfer->xferObjectID( &highObjectID );
407 TheGameLogic->setObjectIDCounter( highObjectID );
408
409 //
410 // it is also equally important to xfer the drawable id counter early in the load process
411 // because the act of creating objects also creates drawables, so when it comes time
412 // to load the block of drawables we want to make sure that newly created drawables
413 // at that time will never overlap IDs with any drawable created as part of an object
414 // which then had its ID transferred in from the save file
415 //
416 DrawableID highDrawableID = TheGameClient->getDrawableIDCounter();
417 xfer->xferDrawableID( &highDrawableID );
418 TheGameClient->setDrawableIDCounter( highDrawableID );
419
420 // Save the Game Info so the game can be started with the correct players on load
421 if( TheGameLogic->getGameMode()==GAME_SKIRMISH )
422 {
423 if( TheSkirmishGameInfo==NULL )
424 {
425 TheSkirmishGameInfo = NEW SkirmishGameInfo;
426 TheSkirmishGameInfo->init();
427 TheSkirmishGameInfo->clearSlotList();
428 TheSkirmishGameInfo->reset();
429 }
430 xfer->xferSnapshot(TheSkirmishGameInfo);
431 }
432 else
433 {
434 if( TheSkirmishGameInfo )
435 {
436 delete TheSkirmishGameInfo;
437 TheSkirmishGameInfo = NULL;
438 }
439 }
440
441 //
442 // for loading, start a new game (flagged from a save game) ... this will load the map and all the
443 // things in the map file that don't don't change (terrain, triggers, teams, script
444 // definitions) etc
445 //
446 if( xfer->getXferMode() == XFER_LOAD )
447 {
448 TheGameLogic->startNewGame( TRUE );
449 TheGameLogic->setLoadingSave( FALSE );
450 }
451
452} // end xfer

460void GameStateMap::clearScratchPadMaps( void )
461{
462
463 // remember the current directory
464 char currentDirectory[ _MAX_PATH ];
465 GetCurrentDirectory( _MAX_PATH, currentDirectory );
466
467 // switch into the save directory
468 SetCurrentDirectory( TheGameState->getSaveDirectory().str() );
469
470 // iterate all items in the directory
471 AsciiString fileToDelete;
472 WIN32_FIND_DATA item; // search item
473 HANDLE hFile = INVALID_HANDLE_VALUE; // handle for search resources
474 Bool done = FALSE;
475 Bool first = TRUE;
476 while( done == FALSE )
477 {
478
479 // first, clear flag for deleting file
480 fileToDelete.clear();
481
482 // if our first time through we need to start the search
483 if( first )
484 {
485
486 // start search
487 hFile = FindFirstFile( "*", &item );
488 if( hFile == INVALID_HANDLE_VALUE )
489 return;
490
491 // we are no longer on our first item
492 first = FALSE;
493
494 } // end if, first
495
496 // see if this is a file, and therefore a possible .map file
497 if( !(item.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) )
498 {
499
500 // see if there is a ".map" at end of this filename
501 Char *c = strrchr( item.cFileName, '.' );
502 if( c && stricmp( c, ".map" ) == 0 )
503 fileToDelete.set( item.cFileName ); // we want to delete this one
504
505 } // end if
506
507 //
508 // find the next file before we delete this one, this is probably not necessary
509 // to strcuture things this way so that the find next occurs before the file
510 // delete, but it seems more correct to do so
511 //
512 if( FindNextFile( hFile, &item ) == 0 )
513 done = TRUE;
514
515 // delete file if set
516 if( fileToDelete.isEmpty() == FALSE )
517 DeleteFile( fileToDelete.str() );
518
519 } // end while
520
521 // close search resources
522 FindClose( hFile );
523
524 // restore our directory to the current directory
525 SetCurrentDirectory( currentDirectory );
526
527} // end clearScratchPadMaps
