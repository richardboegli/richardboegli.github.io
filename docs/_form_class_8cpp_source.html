105BOOL WINAPI
106FormClass::fnFormProc
107(
108 HWND dlg_wnd,
109 UINT message,
110 WPARAM wparam,
111 LPARAM lparam
112)
113{
114 FormClass *pform = (FormClass *)::GetProp (dlg_wnd, "FORMCLASS");
115
116 if (message == WM_INITDIALOG) {
117 pform = (FormClass *)lparam;
118 ::SetProp (dlg_wnd, "FORMCLASS", (HANDLE)pform);
119 } else if (message == WM_DESTROY) {
120 ::RemoveProp (dlg_wnd, "FORMCLASS");
121 }
122
123 BOOL retval = FALSE;
124 if (pform) {
125 retval = pform->Dialog_Proc (dlg_wnd, message, wparam, lparam);
126 }
127
128 return retval;
129}

132/***********************************************************************************************
133 * FormClass::ExecuteDlgInit -- Initializes controls in the dialog template *
134 * *
135 * This code was lifted straight out of MFC. It does some "interesting" things like putting *
136 * strings into combo boxes which are typed in the resource editor. *
137 * *
138 * INPUT: *
139 * *
140 * OUTPUT: *
141 * *
142 * WARNINGS: *
143 * *
144 * HISTORY: *
145 * 11/6/98 GTH : Created. *
146 *=============================================================================================*/
147BOOL FormClass::ExecuteDlgInit(LPCTSTR lpszResourceName)
148{
149 // find resource handle
150 LPVOID lpResource = NULL;
151 HGLOBAL hResource = NULL;
152 if (lpszResourceName != NULL)
153 {
154 HINSTANCE hInst = AppInstance;
155 HRSRC hDlgInit = ::FindResource(hInst, lpszResourceName, RT_DLGINIT);
156 if (hDlgInit != NULL)
157 {
158 // load it
159 hResource = LoadResource(hInst, hDlgInit);
160 if (hResource == NULL)
161 return FALSE;
162 // lock it
163 lpResource = LockResource(hResource);
164 assert(lpResource != NULL);
165 }
166 }
167
168 // execute it
169 BOOL bResult = ExecuteDlgInit(lpResource);
170
171 // cleanup
172 if (lpResource != NULL && hResource != NULL)
173 {
174 UnlockResource(hResource);
175 FreeResource(hResource);
176 }
177 return bResult;
178}

181/***********************************************************************************************
182 * FormClass::ExecuteDlgInit -- Initializes the controls in the dialog template *
183 * *
184 * As the above ExecuteDlgInit function, this was lifted from MFC... *
185 * *
186 * INPUT: *
187 * *
188 * OUTPUT: *
189 * *
190 * WARNINGS: *
191 * *
192 * HISTORY: *
193 * 11/6/98 GTH : Created. *
194 *=============================================================================================*/
195BOOL FormClass::ExecuteDlgInit(LPVOID lpResource)
196{
197 BOOL bSuccess = TRUE;
198 if (lpResource != NULL)
199 {
200 UNALIGNED WORD* lpnRes = (WORD*)lpResource;
201 while (bSuccess && *lpnRes != 0)
202 {
203 WORD nIDC = *lpnRes++;
204 WORD nMsg = *lpnRes++;
205 DWORD dwLen = *((UNALIGNED DWORD*&)lpnRes)++;
206
207 // In Win32 the WM_ messages have changed. They have
208 // to be translated from the 32-bit values to 16-bit
209 // values here.
210 #define WIN16_LB_ADDSTRING 0x0401
211 #define WIN16_CB_ADDSTRING 0x0403
212
213 if (nMsg == WIN16_LB_ADDSTRING)
214 nMsg = LB_ADDSTRING;
215 else if (nMsg == WIN16_CB_ADDSTRING)
216 nMsg = CB_ADDSTRING;
217
218 // check for invalid/unknown message types
219 assert(nMsg == LB_ADDSTRING || nMsg == CB_ADDSTRING);
220
221 if (nMsg == LB_ADDSTRING || nMsg == CB_ADDSTRING)
222 {
223 // List/Combobox returns -1 for error
224 if (::SendDlgItemMessageA(m_hWnd, nIDC, nMsg, 0, (LONG)lpnRes) == -1)
225 bSuccess = FALSE;
226 }
227
228 // skip past data
229 lpnRes = (WORD*)((LPBYTE)lpnRes + (UINT)dwLen);
230 }
231 }
232
233 return bSuccess;
234}
