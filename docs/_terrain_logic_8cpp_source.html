129Object *Bridge::createTower( Coord3D *worldPos,
130 BridgeTowerType towerType,
131 const ThingTemplate *towerTemplate,
132 Object *bridge )
133{
134
135 // sanity
136 if( towerTemplate == NULL || bridge == NULL )
137 {
138
139 DEBUG_CRASH(( "Bridge::createTower(): Invalid params\n" ));
140 return NULL;
141
142 } // end if
143
144 // create the tower object
145 Object *tower = TheThingFactory->newObject( towerTemplate, bridge->getTeam() );
146
147 // location information
148 Real angle = 0;
149 switch( towerType )
150 {
151
152 // --------------------------------------------------------------------------------------------
153 case BRIDGE_TOWER_FROM_LEFT:
154 angle = bridge->getOrientation() + PI;
155 break;
156
157 // --------------------------------------------------------------------------------------------
158 case BRIDGE_TOWER_FROM_RIGHT:
159 angle = bridge->getOrientation() + PI;
160 break;
161
162 // --------------------------------------------------------------------------------------------
163 case BRIDGE_TOWER_TO_LEFT:
164 angle = bridge->getOrientation();
165 break;
166
167 // --------------------------------------------------------------------------------------------
168 case BRIDGE_TOWER_TO_RIGHT:
169 angle = bridge->getOrientation();
170 break;
171
172 // --------------------------------------------------------------------------------------------
173 default:
174 DEBUG_CRASH(( "Bridge::createTower - Unknown bridge tower type '%d'\n", towerType ));
175 return NULL;
176
177 } // end switch
178
179 // set the position and angle
180 tower->setPosition( worldPos );
181 tower->setOrientation( angle );
182
183 // tie it to the bridge
184 BridgeBehaviorInterface *bridgeInterface = BridgeBehavior::getBridgeBehaviorInterfaceFromObject( bridge );
185 DEBUG_ASSERTCRASH( bridgeInterface != NULL, ("Bridge::createTower - no 'BridgeBehaviorInterface' found\n") );
186 if( bridgeInterface )
187 bridgeInterface->setTower( towerType, tower );
188
189 // tie the bridge to us
190 BridgeTowerBehaviorInterface *bridgeTowerInterface = BridgeTowerBehavior::getBridgeTowerBehaviorInterfaceFromObject( tower );
191 DEBUG_ASSERTCRASH( bridgeTowerInterface != NULL, ("Bridge::createTower - no 'BridgeTowerBehaviorInterface' found\n") );
192 if( bridgeTowerInterface )
193 {
194
195 // set bridge object
196 bridgeTowerInterface->setBridge( bridge );
197
198 // save our position type
199 bridgeTowerInterface->setTowerType( towerType );
200
201 } // end if
202
203 // if the bridge is indestructible, so is this tower
204 BodyModuleInterface *bridgeBody = bridge->getBodyModule();
205 if( bridgeBody->isIndestructible() )
206 {
207 BodyModuleInterface *towerBody = tower->getBodyModule();
208
209 towerBody->setIndestructible( TRUE );
210
211 } // end if
212
213 // return the newly created tower
214 return tower;
215
216} // end createTower

220Bridge::Bridge(BridgeInfo &theInfo, Dict *props, AsciiString bridgeTemplateName) :
221m_bridgeInfo(theInfo)
222{
223
224 // save the template name
225 m_templateName = bridgeTemplateName;
226
227 //Coord3D fromLeft, fromRight, toLeft, toRight; /// The 4 corners of the rectangle that the bridge covers.
228 m_bounds.lo.x = m_bridgeInfo.fromLeft.x;
229 m_bounds.lo.y = m_bridgeInfo.fromLeft.y;
230 m_bounds.hi = m_bounds.lo;
231 if (m_bounds.lo.x > m_bridgeInfo.fromRight.x) m_bounds.lo.x = m_bridgeInfo.fromRight.x;
232 if (m_bounds.lo.y > m_bridgeInfo.fromRight.y) m_bounds.lo.y = m_bridgeInfo.fromRight.y;
233 if (m_bounds.hi.x < m_bridgeInfo.fromRight.x) m_bounds.hi.x = m_bridgeInfo.fromRight.x;
234 if (m_bounds.hi.y < m_bridgeInfo.fromRight.y) m_bounds.hi.y = m_bridgeInfo.fromRight.y;
235 if (m_bounds.lo.x > m_bridgeInfo.toLeft.x) m_bounds.lo.x = m_bridgeInfo.toLeft.x;
236 if (m_bounds.lo.y > m_bridgeInfo.toLeft.y) m_bounds.lo.y = m_bridgeInfo.toLeft.y;
237 if (m_bounds.hi.x < m_bridgeInfo.toLeft.x) m_bounds.hi.x = m_bridgeInfo.toLeft.x;
238 if (m_bounds.hi.y < m_bridgeInfo.toLeft.y) m_bounds.hi.y = m_bridgeInfo.toLeft.y;
239 if (m_bounds.lo.x > m_bridgeInfo.toRight.x) m_bounds.lo.x = m_bridgeInfo.toRight.x;
240 if (m_bounds.lo.y > m_bridgeInfo.toRight.y) m_bounds.lo.y = m_bridgeInfo.toRight.y;
241 if (m_bounds.hi.x < m_bridgeInfo.toRight.x) m_bounds.hi.x = m_bridgeInfo.toRight.x;
242 if (m_bounds.hi.y < m_bridgeInfo.toRight.y) m_bounds.hi.y = m_bridgeInfo.toRight.y;
243
244 m_bridgeInfo.curDamageState = BODY_PRISTINE;
245
246
247 static const ThingTemplate* genericBridgeTemplate = TheThingFactory->findTemplate("GenericBridge");
248 if (!genericBridgeTemplate) {
249 DEBUG_LOG(("*** GenericBridge template not found."));
250 return;
251 }
252 Object *bridge = TheThingFactory->newObject(genericBridgeTemplate, NULL);
253 Coord3D center;
254 center.x = (m_bridgeInfo.fromLeft.x + m_bridgeInfo.toRight.x)/2.0f;
255 center.y = (m_bridgeInfo.fromLeft.y + m_bridgeInfo.toRight.y)/2.0f;
256 center.z = (m_bridgeInfo.fromLeft.z + m_bridgeInfo.toRight.z)/2.0f;
257 bridge->setPosition(&center);
258 m_bridgeInfo.bridgeObjectID = bridge->getID();
259 bridge->updateObjValuesFromMapProperties(props);
260
261 //
262 // we'll say the angle of this object representing the bridge is from the 'from' side
263 // to the 'to' side.
264 //
265 Coord2D v;
266 v.x = m_bridgeInfo.toLeft.x - m_bridgeInfo.fromLeft.x;
267 v.y = m_bridgeInfo.toLeft.y - m_bridgeInfo.fromLeft.y;
268 bridge->setOrientation( v.toAngle() );
269
270 v.x = m_bridgeInfo.toLeft.x - m_bridgeInfo.toRight.x;
271 v.y = m_bridgeInfo.toLeft.y - m_bridgeInfo.toRight.y;
272 v.normalize();
273
274 // get the template of the bridge
275 TerrainRoadType *bridgeTemplate = TheTerrainRoads->findBridge( bridgeTemplateName );
276 if( bridgeTemplate == NULL ) {
277 DEBUG_LOG(( "*** Bridge Template Not Found '%s'.", bridgeTemplateName ));
278 return;
279 }
280
281#define no_BRIDGE_TOWERS // since they aren't destructable, don't need towers.
282#if BRIDGE_TOWERS
283 // initialize each of the tower positions to that of the bridge info bounding rect
284 Coord3D towerPos[ BRIDGE_MAX_TOWERS ];
285 towerPos[ BRIDGE_TOWER_FROM_LEFT ] = m_bridgeInfo.fromLeft;
286 towerPos[ BRIDGE_TOWER_FROM_RIGHT ] = m_bridgeInfo.fromRight;
287 towerPos[ BRIDGE_TOWER_TO_LEFT ] = m_bridgeInfo.toLeft;
288 towerPos[ BRIDGE_TOWER_TO_RIGHT ] = m_bridgeInfo.toRight;
289
290 // create objects targetable objects for the 4 tower pieces
291 const ThingTemplate *towerTemplate;
292 BridgeTowerType type;
293 Object *tower;
294 Real offset = PATHFIND_CELL_SIZE_F/2.0f;
295 for( Int i = 0; i < BRIDGE_MAX_TOWERS; ++i )
296 {
297
298 // create the tower
299 type = (BridgeTowerType)i;
300 towerTemplate = TheThingFactory->findTemplate( bridgeTemplate->getTowerObjectName( type ) );
301 if (towerTemplate) {
302 offset = towerTemplate->getTemplateGeometryInfo().getMajorRadius();
303 }
304 Coord3D pos = towerPos[type];
305 switch( type )
306 {
307 case BRIDGE_TOWER_FROM_LEFT:
308 case BRIDGE_TOWER_TO_LEFT:
309 pos.x += v.x*offset;
310 pos.y += v.y*offset;
311 break;
312 case BRIDGE_TOWER_FROM_RIGHT:
313 case BRIDGE_TOWER_TO_RIGHT:
314 pos.x -= v.x*offset;
315 pos.y -= v.y*offset;
316 break;
317
318 } // end switch
319 tower = createTower( &pos, type, towerTemplate, bridge );
320
321 // store the tower object ID
322 m_bridgeInfo.towerObjectID[ i ] = tower->getID();
323
324 } // end for, i
325#endif
326
327 m_next = NULL;
328} // end Bridge

474 return true;
475 }

477}

488Bool LineInRegion( const Coord2D *p1, const Coord2D *p2, const Region2D *clipRegion )
489{
490 enum { CLIP_LEFT = 0x01,
491 CLIP_RIGHT = 0x02,
492 CLIP_BOTTOM = 0x04,
493 CLIP_TOP = 0x08 };
494 Real x1, y1, x2, y2;
495 Real clipLeft;
496 Real clipRight;
497 Real clipTop;
498 Real clipBottom;
499 Int clipCode1;
500 Int clipCode2;
501 Real diff;
502
503 // Use clip window that includes bottom right pixel
504 clipLeft = clipRegion->lo.x;
505 clipRight = clipRegion->hi.x;
506 clipTop = clipRegion->lo.y;
507 clipBottom = clipRegion->hi.y;
508
509 x1 = p1->x;
510 y1 = p1->y;
511 x2 = p2->x;
512 y2 = p2->y;
513
514 // Test first point
515 clipCode1 = 0;
516
517 if (x1 < clipLeft)
518 clipCode1 = CLIP_LEFT;
519 else
520 if (x1 > clipRight)
521 clipCode1 = CLIP_RIGHT;
522
523 if (y1 < clipTop)
524 clipCode1 |= CLIP_TOP;
525 else
526 if (y1 > clipBottom)
527 clipCode1 |= CLIP_BOTTOM;
528
529
530 // Test second point
531 clipCode2 = 0;
532
533 if (x2 < clipLeft)
534 clipCode2 = CLIP_LEFT;
535 else
536 if (x2 > clipRight)
537 clipCode2 = CLIP_RIGHT;
538
539 if (y2 < clipTop)
540 clipCode2 |= CLIP_TOP;
541 else
542 if (y2 > clipBottom)
543 clipCode2 |= CLIP_BOTTOM;
544
545
546 // Both points inside window?
547 if ((clipCode1 | clipCode2) == 0)
548 {
549 return TRUE;
550 } // end if
551
552 // Both points outside window?
553 if (clipCode1 & clipCode2)
554 return FALSE;
555
556 // First point outside window?
557 if (clipCode1)
558 {
559 if (clipCode1 & CLIP_TOP)
560 {
561 if ((diff = (y2 - y1)) == 0)
562 return FALSE;
563 x1 += (x2 - x1) * (clipTop - y1) / diff;
564 y1 = clipTop;
565 }
566 else
567 if (clipCode1 & CLIP_BOTTOM)
568 {
569 if ((diff = (y2 - y1)) == 0)
570 return FALSE;
571 x1 += (x2 - x1) * (clipBottom - y1) / diff;
572 y1 = clipBottom;
573 }
574
575 if (x1 > clipRight)
576 {
577 if ((diff = (x2 - x1)) == 0)
578 return FALSE;
579 y1 += (y2 - y1) * (clipRight - x1) / diff;
580 x1 = clipRight;
581 }
582 else
583 if (x1 < clipLeft)
584 {
585 if ((diff = (x2 - x1)) == 0)
586 return FALSE;
587 y1 += (y2 - y1) * (clipLeft - x1) / diff;
588 x1 = clipLeft;
589 }
590 }
591
592 // Second point outside window?
593 if (clipCode2)
594 {
595 if (clipCode2 & CLIP_TOP)
596 {
597 if ((diff = (y2 - y1)) == 0)
598 return FALSE;
599 x2 += (x2 - x1) * (clipTop - y2) / diff;
600 y2 = clipTop;
601 }
602 else
603 if (clipCode2 & CLIP_BOTTOM)
604 {
605 if ((diff = (y2 - y1)) == 0)
606 return FALSE;
607 x2 += (x2 - x1) * (clipBottom - y2) / diff;
608 y2 = clipBottom;
609 }
610
611 if (x2 > clipRight)
612 {
613 if ((diff = (x2 - x1)) == 0)
614 return FALSE;
615 y2 += (y2 - y1) * (clipRight - x2) / diff;
616 x2 = clipRight;
617 }
618 else
619 if (x2 < clipLeft)
620 {
621 if ((diff = (x2 - x1)) == 0)
622 return FALSE;
623 y2 += (y2 - y1) * (clipLeft - x2) / diff;
624 x2 = clipLeft;
625 }
626 }
627
628 // Line is visible
629 return (x1 >= clipLeft && x1 <= clipRight &&
630 y1 >= clipTop && y1 <= clipBottom &&
631 x2 >= clipLeft && x2 <= clipRight &&
632 y2 >= clipTop && y2 <= clipBottom);
633
634} // end LineInRegion

1026//-------------------------------------------------------------------------------------------------
1027void TerrainLogic::update( void )
1028{
1029
1030 // bridge damage states have not changed this frame now
1031 m_bridgeDamageStatesChanged = false;
1032
1033 // update any water tables that we need to
1034 if( m_numWaterToUpdate )
1035 {
1036 const WaterHandle *water;
1037 Real changePerFrame,
1038 damageAmount,
1039 targetHeight,
1040 currentHeight;
1041 Bool finalTransition,
1042 doDamageThisFrame = (TheGameLogic->getFrame() % LOGICFRAMES_PER_SECOND) == 0;
1043
1044 for( Int i = m_numWaterToUpdate - 1; i >= 0; --i )
1045 {
1046
1047 // get the water info
1048 water = m_waterToUpdate[ i ].waterTable;
1049 changePerFrame = m_waterToUpdate[ i ].changePerFrame;
1050 targetHeight = m_waterToUpdate[ i ].targetHeight;
1051 damageAmount = m_waterToUpdate[ i ].damageAmount;
1052 currentHeight = m_waterToUpdate[ i ].currentHeight;
1053
1054 //
1055 // check to see if this change per frame will get us to our target height, if so
1056 // we adjust the changePerFrame to make us be exactly at our target height, and after
1057 // the change we will remove our entry from this update phase
1058 //
1059 finalTransition = FALSE;
1060 if( changePerFrame > 0 )
1061 {
1062
1063 if( currentHeight + changePerFrame >= targetHeight )
1064 finalTransition = TRUE;
1065
1066 } // end if
1067 else
1068 {
1069
1070 if( currentHeight + changePerFrame <= targetHeight )
1071 finalTransition = TRUE;
1072
1073 } // end else
1074
1075 if( finalTransition == TRUE )
1076 {
1077
1078 //
1079 // make the final water height change, note we do damage on the final transition
1080 // in all situations by passing a valid damage amount
1081 //
1082 setWaterHeight( water, targetHeight, damageAmount, TRUE );
1083
1084 //
1085 // remove our water entry from the per frame water list, we're processing this array
1086 // backwards which makes cleanup easy, we just move everything after our index
1087 // position up one
1088 //
1089 for( Int j = i; j < m_numWaterToUpdate; j++ )
1090 m_waterToUpdate[ i ] = m_waterToUpdate[ j ];
1091 m_numWaterToUpdate -= 1;
1092
1093 } // end if
1094 else
1095 {
1096
1097 //
1098 // we're not doing damage every frame (0 damage) from the water
1099 // because it's an expensive process
1100 //
1101 if( doDamageThisFrame == FALSE )
1102 damageAmount = 0.0f;
1103
1104 //
1105 // because some water implementation store the height as integers, some changes
1106 // are too small to keep track of in the actual water data structures so we have to
1107 // keep track of it outselves
1108 //
1109 currentHeight += changePerFrame;
1110 m_waterToUpdate[ i ].currentHeight = currentHeight;
1111
1112 // update actual water
1113 setWaterHeight( water, currentHeight, damageAmount, FALSE );
1114
1115 } // end else
1116
1117 } // end for i
1118
1119 } // end if
1120
1121} // end update

1469//-------------------------------------------------------------------------------------------------
1470void makeAlignToNormalMatrix( Real angle, const Coord3D& pos, const Coord3D& normal, Matrix3D& mtx)
1471{
1472 Coord3D x, y, z;
1473
1474 z = normal;
1475
1476 /*
1477 It is extremely important that the resulting matrix is such that
1478 the xvector points in the angle we specified; specifically,
1479 that atan2(xvec.y, xvec.x) == angle. So we must construct
1480 the matrix carefully to ensure this!
1481 */
1482 x.x = Cos( angle );
1483 x.y = Sin( angle );
1484 x.z = 0.0f;
1485//x.normalize(); -- redundant; is normalized by definition
1486
1487 // dot of two unit vectors is cos of angle between them;
1488 // we want there to be a 90-deg angle between the x and z
1489 // vectors, so calc x.z to satisfy this (ie, cos==0)
1490 /*
1491 xx*zx + xy*zy + xz*zz = 0
1492 xz = (-xx*zz - xy*zy)/zz
1493 */
1494 if (z.z != 0.0f)
1495 {
1496 x.z = -(x.x*z.x + x.y*z.y) / z.z;
1497 x.normalize();
1498 }
1499
1500 DEBUG_ASSERTCRASH(fabs(x.x*z.x + x.y*z.y + x.z*z.z)<0.0001,("dot is not zero (%f)\n",fabs(x.x*z.x + x.y*z.y + x.z*z.z)));
1501
1502 // now computing the y vector is trivial.
1503 y.crossProduct( &z, &x, &y );
1504 y.normalize();
1505
1506 mtx.Set( x.x, y.x, z.x, pos.x,
1507 x.y, y.y, z.y, pos.y,
1508 x.z, y.z, z.z, pos.z );
1509}
