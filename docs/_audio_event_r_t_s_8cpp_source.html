317//-------------------------------------------------------------------------------------------------
318void AudioEventRTS::generateFilename( void )
319{
320 // A Logic Random Value is used because we may ask "How long will it take to play this sound?"
321 // In that case, we need the same answer across all pcs.
322
323 if (!m_eventInfo) {
324 return;
325 }
326
327 m_filenameToLoad = generateFilenamePrefix(m_eventInfo->m_soundType, false);
328
329 Int which = 0;
330
331 if (m_eventInfo->m_soundType == AT_Music || m_eventInfo->m_soundType == AT_Streaming) {
332 m_filenameToLoad.concat(m_eventInfo->m_filename);
333 adjustForLocalization(m_filenameToLoad);
334 return;
335 } else {
336 if (m_eventInfo->m_sounds.size() == 0) {
337 m_filenameToLoad = AsciiString::TheEmptyString;
338 return;
339 }
340
341
342
343 if (BitTest(m_eventInfo->m_control, AC_RANDOM))
344 {
345 if (m_isLogicalAudio)
346 {
347 which = GameLogicRandomValue(0, m_eventInfo->m_sounds.size() - 1);
348 }
349 else
350 {
351 which = GameAudioRandomValue(0, m_eventInfo->m_sounds.size() - 1);
352 }
353
354 if (which == m_playingAudioIndex && m_eventInfo->m_sounds.size() > 2)
355 which = ( which + 1 ) % m_eventInfo->m_sounds.size();
356
357 m_playingAudioIndex = which;//caching random choice to compare next call
358
359 }
360 else
361 which = (++m_playingAudioIndex) % m_eventInfo->m_sounds.size();
362
363
364 }
365
366 m_filenameToLoad.concat(m_eventInfo->m_sounds[which]);
367 m_filenameToLoad.concat(generateFilenameExtension(m_eventInfo->m_soundType));
368 adjustForLocalization(m_filenameToLoad);
369
370 // Note: Also generate Delay when generating a filename, cause
371 // we want delay to apply between every loop of a sound.
372 m_delay = GameAudioRandomValueReal(m_eventInfo->m_delayMin, m_eventInfo->m_delayMax);
373}

381//-------------------------------------------------------------------------------------------------
382void AudioEventRTS::generatePlayInfo( void )
383{
384 m_pitchShift = GameAudioRandomValueReal(m_eventInfo->m_pitchShiftMin, m_eventInfo->m_pitchShiftMax);
385 m_volumeShift = GameAudioRandomValueReal(1.0f + m_eventInfo->m_volumeShift, 1.0f); // volume shifts are between 0 and 1
386 m_loopCount = m_eventInfo->m_loopCount;
387
388 m_portionToPlayNext = PP_Attack;
389 Int attackSize = m_eventInfo->m_attackSounds.size();
390 if (attackSize > 0) {
391 m_attackName = generateFilenamePrefix(m_eventInfo->m_soundType, false);
392 // needs to be logic because it needs to be the same on all systems.
393 Int attackToPlay;
394 if (m_isLogicalAudio) {
395 attackToPlay = GameLogicRandomValue(0, attackSize - 1);
396 } else {
397 attackToPlay = GameAudioRandomValue(0, attackSize - 1);
398 }
399
400 m_attackName.concat(m_eventInfo->m_attackSounds[attackToPlay]);
401 m_attackName.concat(generateFilenameExtension(m_eventInfo->m_soundType));
402 adjustForLocalization(m_attackName);
403 } else {
404 m_portionToPlayNext = PP_Sound;
405 }
406
407 Int decaySize = m_eventInfo->m_decaySounds.size();
408 if (decaySize > 0) {
409 m_decayName = generateFilenamePrefix(m_eventInfo->m_soundType, false);
410 // needs to be logic because it needs to be the same on all systems.
411 Int decayToPlay;
412 if (m_isLogicalAudio) {
413 decayToPlay = GameLogicRandomValue(0, decaySize - 1);
414 } else {
415 decayToPlay = GameAudioRandomValue(0, decaySize - 1);
416 }
417
418 m_decayName.concat(m_eventInfo->m_decaySounds[decayToPlay]);
419 m_decayName.concat(generateFilenameExtension(m_eventInfo->m_soundType));
420 adjustForLocalization(m_decayName);
421 }
422
423 m_isLogicalAudio = FALSE;
424}

468//-------------------------------------------------------------------------------------------------
469void AudioEventRTS::advanceNextPlayPortion( void )
470{
471 switch (m_portionToPlayNext)
472 {
473 case PP_Attack:
474 m_portionToPlayNext = PP_Sound;
475 break;
476 case PP_Sound:
477 if (m_eventInfo && BitTest(m_eventInfo->m_control, AC_ALL))
478 {
479 if (m_allCount == m_eventInfo->m_sounds.size()) {
480 m_portionToPlayNext = PP_Decay;
481 }
482
483 // Advance the all count so that we move to the next sound.
484 ++m_allCount;
485 }
486 if (!m_decayName.isEmpty()) {
487 m_portionToPlayNext = PP_Decay;
488 } else {
489 m_portionToPlayNext = PP_Done;
490 }
491 break;
492 case PP_Decay:
493 m_portionToPlayNext = PP_Done;
494 break;
495 }
496}

770AsciiString AudioEventRTS::generateFilenamePrefix( AudioType audioTypeToPlay, Bool localized )
771{
772 AsciiString retStr;
773 retStr = TheAudio->getAudioSettings()->m_audioRoot;
774 retStr.concat("\\");
775 if (audioTypeToPlay == AT_Music) {
776 retStr.concat(TheAudio->getAudioSettings()->m_musicFolder);
777 } else if (audioTypeToPlay == AT_Streaming) {
778 retStr.concat(TheAudio->getAudioSettings()->m_streamingFolder);
779 } else {
780 retStr.concat(TheAudio->getAudioSettings()->m_soundsFolder);
781 }
782 retStr.concat("\\");
783
784 if (localized) {
785 retStr.concat(GetRegistryLanguage());
786 retStr.concat("\\");
787 }
788
789 return retStr;
790}

804//-------------------------------------------------------------------------------------------------
805void AudioEventRTS::adjustForLocalization(AsciiString &strToAdjust)
806{
807 const char *str = strToAdjust.reverseFind('\\');
808 if (!str) {
809 return;
810 }
811
812 // try the localized version first so that we're guarenteed to get it
813 // even if the generic data directory holds a version of the file
814 AsciiString localizedFilePath = generateFilenamePrefix(m_eventInfo->m_soundType, TRUE);
815 AsciiString filename = str;
816 localizedFilePath.concat(filename);
817
