82//Dummy material override which we assign to all ghost objects to disable their
83//texture animation.
84static RenderObjClass::Material_Override animationDisableOverride;
85
86//Helper function used to disable all UV mapper animations on a given model.
87//Also use this pass to disable muzzle effects.
88void disableUVAnimations(RenderObjClass *robj)
89{
90 if (robj && robj->Class_ID() == RenderObjClass::CLASSID_HLOD)
91 {
92 //Also disable any animations that may be playing using mappers (texture scrolling)
93 for (Int i=0; i < robj->Get_Num_Sub_Objects(); i++)
94 {
95 RenderObjClass *subObj=robj->Get_Sub_Object(i);
96 if (subObj && subObj->Class_ID() == RenderObjClass::CLASSID_MESH)
97 { //check if sub-object has the correct material to do texture scrolling.
98 MaterialInfoClass *mat=subObj->Get_Material_Info();
99 if (mat)
100 { for (Int j=0; j<mat->Vertex_Material_Count(); j++)
101 {
102 VertexMaterialClass *vmaterial=mat->Peek_Vertex_Material(j);
103 LinearOffsetTextureMapperClass *mapper=(LinearOffsetTextureMapperClass *)vmaterial->Peek_Mapper();
104 if (mapper && mapper->Mapper_ID() == TextureMapperClass::MAPPER_ID_LINEAR_OFFSET)
105 {
106 subObj->Set_User_Data(&animationDisableOverride); //tell W3D about custom material settings
107 }
108 }
109 REF_PTR_RELEASE(mat);
110 }
111 //We don't want muzzle flashes visible inside fog, so turn them off.
112 if (subObj->Get_Name() && strstr(subObj->Get_Name(),"MUZZLEFX"))
113 subObj->Set_Hidden(true);
114 }
115 REF_PTR_RELEASE(subObj);
116 }
117 }
118}

122void W3DRenderObjectSnapshot::update(RenderObjClass *robj, DrawableInfo *drawInfo,
123 Bool cloneParentRobj)
124{
125 REF_PTR_RELEASE(m_robj);
126
127 if( cloneParentRobj == TRUE )
128 {
129
130 m_robj=robj->Clone();
131 m_robj->Set_ObjectColor(robj->Get_ObjectColor());
132#ifdef DEBUG_FOG_MEMORY
133 m_robjName = m_robj->Get_Name();
134#endif
135 //Set cloned object to same state as original object.
136 m_robj->Set_Transform(robj->Get_Transform());
137 if (robj->Class_ID() == RenderObjClass::CLASSID_HLOD)
138 {
139 float frame,mult;
140 int mode,numFrames;
141
142 HAnimClass *hanim=((HLodClass *)robj)->Peek_Animation_And_Info(frame,numFrames,mode,mult);
143 m_robj->Set_Animation(hanim,frame);
144 disableUVAnimations(m_robj);
145 } //HLOD
146 }
147 else
148 m_robj=robj;
149
150 m_robj->Set_User_Data(drawInfo);
151
152}

183// ------------------------------------------------------------------------------------------------
184void W3DRenderObjectSnapshot::xfer( Xfer *xfer )
185{
186
187 // version
188 XferVersion currentVersion = 1;
189 XferVersion version = currentVersion;
190 xfer->xferVersion( &version, currentVersion );
191
192 // sanity
193 DEBUG_ASSERTCRASH( m_robj, ("W3DRenderObjectSnapshot::xfer - invalid m_robj\n") );
194
195 // transform on the main render object
196 Matrix3D transform;
197 transform = m_robj->Get_Transform();
198 xfer->xferUser( &transform, sizeof( Matrix3D ) );
199 if( xfer->getXferMode() == XFER_LOAD )
200 m_robj->Set_Transform( transform );
201
202 // how many sub objects of data will follow
203 Int subObjectCount = m_robj->Get_Num_Sub_Objects();
204 xfer->xferInt( &subObjectCount );
205
206 Bool visible;
207 RenderObjClass *subObject;
208 AsciiString subObjectName;
209 for( Int i = 0; i < subObjectCount; ++i )
210 {
211
212 //
213 // when saving we get sub objects by index and xfer their name, when loading
214 // we read the name and find that sub object
215 //
216 if( xfer->getXferMode() == XFER_SAVE )
217 {
218
219 // get sub object
220 subObject = m_robj->Get_Sub_Object( i );
221
222 // xfer sub object name which is unique among those in this render object
223 subObjectName.set( subObject->Get_Name() );
224 xfer->xferAsciiString( &subObjectName );
225
226 } // end if, save
227 else
228 {
229
230 // read sub object name
231 xfer->xferAsciiString( &subObjectName );
232
233 // find this sub object on the object
234 subObject = m_robj->Get_Sub_Object_By_Name( subObjectName.str() );
235
236 } // end else load
237
238 //
239 // NOTE that the remainder of this xfer code works on a sub object only *if*
240 // it is present. It is possible that in future patches we change the artwork
241 // for some objects which could remove sub objects for which we have data saved
242 // for in the save game file. If we encounter data in the save file for
243 // sub objects that are no longer in the artwork we just read the data and
244 // throw it away
245 //
246
247 // visible/hidden status of this sub object
248 if( subObject )
249 visible = subObject->Is_Not_Hidden_At_All();
250 xfer->xferBool( &visible );
251 if( subObject && xfer->getXferMode() == XFER_LOAD )
252 subObject->Set_Hidden( !visible );
253
254 // transform of this sub object
255 if( subObject )
256 transform = subObject->Get_Transform();
257 xfer->xferUser( &transform, sizeof( Matrix3D ) );
258 if( subObject && xfer->getXferMode() == XFER_LOAD )
259 subObject->Set_Transform( transform );
260
261 // need to tell W3D that this sub object transforms are ok
262 if( subObject )
263 {
264
265 // need to cast to HLod if we can to validate the hierarchy
266 if( subObject->Class_ID() == RenderObjClass::CLASSID_HLOD )
267 ((HLodClass *)subObject)->Friend_Set_Hierarchy_Valid( TRUE );
268
269 } // end if
270
271 // release reference to sub object
272 if( subObject )
273 REF_PTR_RELEASE( subObject );
274
275 } // end for, i
276
277 // tell W3D that the transforms for our sub objects are all OK cause we've done them ourselves
278 m_robj->Set_Sub_Object_Transforms_Dirty( FALSE );
279
280} // end xfer

322// ------------------------------------------------------------------------------------------------
326// ------------------------------------------------------------------------------------------------
327void W3DGhostObject::snapShot(int playerIndex)
328{
329#ifndef DEBUG_FOG_MEMORY
330 if (playerIndex != TheGhostObjectManager->getLocalPlayerIndex())
331 return; //we only snapshot things for the initial local player because local player can't change in non-debug game.
332#endif
333
334 Drawable *draw=m_parentObject->getDrawable();
335 if (draw->isDrawableEffectivelyHidden())
336 return; //don't bother to snapshot things which nobody can see.
337
338 W3DRenderObjectSnapshot *snap=m_parentSnapshots[playerIndex],*prevSnap=NULL;
339
340 //walk through all W3D render objects used by this object
341 for (DrawModule ** dm = draw->getDrawModules(); *dm; ++dm)
342 {
343 const ObjectDrawInterface* di = (*dm)->getObjectDrawInterface();
344 if (di)
345 {
346 W3DModelDraw *w3dDraw= (W3DModelDraw *)di;
347 RenderObjClass *robj=NULL;
348
349 robj=w3dDraw->getRenderObject();
350 //robj may be null for modules which have no render objects such
351 //as for build-ups that are currently disabled.
352 if (robj)
353 {
354 if (snap == NULL)
355 {
356 snap = NEW W3DRenderObjectSnapshot(robj, &m_drawableInfo); // poolify
357 if (prevSnap)
358 prevSnap->m_next=snap;
359 else
360 m_parentSnapshots[playerIndex]=snap;
361 }
362 else
363 m_parentSnapshots[playerIndex]->update(robj, &m_drawableInfo);
364
365 //Adding and removing render objects to the scene is expensive
366 //so only do it for the real player watching the screen. There is
367 //also no point in displaying the other player's ghost objects to
368 //the current player.
369 if (playerIndex == TheGhostObjectManager->getLocalPlayerIndex())
370 {
371 robj->Remove(); //remove normal object from scene
372 snap->addToScene();
373 }
374
375 prevSnap=snap;
376 snap = snap->m_next;
377 }
378 }
379 }
380
381 //Check if we captured at least one snapshot
382 if (snap != m_parentSnapshots[playerIndex])
383 { //save off other info we may need in case the parent object is destroyed.
385 //looking at a different geometry info/orientation because ghostobjects
386 //are supposed to be used on immobile buildings.
387 m_parentGeometryType=m_parentObject->getGeometryInfo().getGeomType();
388 m_parentGeometryIsSmall=m_parentObject->getGeometryInfo().getIsSmall();
389 m_parentGeometryMajorRadius=m_parentObject->getGeometryInfo().getMajorRadius();
390 m_parentGeometryminorRadius=m_parentObject->getGeometryInfo().getMinorRadius();
391 m_parentPosition=*m_parentObject->getPosition();
392 m_parentAngle=m_parentObject->getOrientation();
393 }
394}

396// ------------------------------------------------------------------------------------------------
398// ------------------------------------------------------------------------------------------------
399void W3DGhostObject::removeParentObject(void)
400{
401
402 // sanity
403 if( m_parentObject == NULL )
404 return;
405
406 Drawable *draw=m_parentObject->getDrawable();
407
408 //After we remove the unfogged object, we also disable
409 //anything that should be hidden inside fog - shadow, particles, etc.
410 draw->setFullyObscuredByShroud(true);
411
412 //walk through all W3D render objects used by this object
413 for (DrawModule ** dm = draw->getDrawModules(); *dm; ++dm)
414 {
415 const ObjectDrawInterface* di = (*dm)->getObjectDrawInterface();
416 if (di)
417 {
418 W3DModelDraw *w3dDraw= (W3DModelDraw *)di;
419 RenderObjClass *robj=NULL;
420
421 robj=w3dDraw->getRenderObject();
422 if (robj)
423 {
424 DEBUG_ASSERTCRASH(robj->Peek_Scene() != NULL, ("Removing GhostObject parent not in scene "));
425 robj->Remove();
426 }
427 }
428 }
429}

431// ------------------------------------------------------------------------------------------------
433// ------------------------------------------------------------------------------------------------
434void W3DGhostObject::restoreParentObject(void)
435{
436 Drawable *draw=m_parentObject->getDrawable();
437 if (!draw)
438 return;
439
440 //Notify drawable that it's okay to render its render objects again.
441 draw->setFullyObscuredByShroud(false);
442
443 //walk through all W3D render objects used by this object
444 for (DrawModule ** dm = draw->getDrawModules(); *dm; ++dm)
445 {
446 const ObjectDrawInterface* di = (*dm)->getObjectDrawInterface();
447 if (di)
448 {
449 W3DModelDraw *w3dDraw= (W3DModelDraw *)di;
450 RenderObjClass *robj=NULL;
451
452 robj=w3dDraw->getRenderObject();
453 //robj may be null for modules which have no render objects such
454 //as for build-ups that are currently disabled.
455 if (robj)
456 { //if we have a render object that's not in the scene, it must have been
457 //removed by the ghost object manager, so restore it. If we have a render
458 //object that is in the scene, then it was probably added because the model
459 //changed while the object was ghosted (for damage states, garrison, etc.).
460 if (robj->Peek_Scene() == NULL)
461 ((SimpleSceneClass *)W3DDisplay::m_3DScene)->Add_Render_Object(robj);
462 }
463 }
464 }
465}

469void W3DGhostObject::freeAllSnapShots(void)
470{
471 Int playerIndex;
472
473#ifdef DEBUG_FOG_MEMORY
474 for (playerIndex=0; playerIndex<MAX_PLAYER_COUNT; playerIndex++)
475#else
476 playerIndex = TheGhostObjectManager->getLocalPlayerIndex();
477#endif
478 if (m_parentSnapshots[playerIndex])
479 { //if we have a snapshot for this object, remove it from
480 //scene.
481 removeFromScene(playerIndex);
482
483 //Restore actual objects assuming they are still alive.
484 if (m_parentObject)
485 restoreParentObject();
486
487 W3DRenderObjectSnapshot *snap=m_parentSnapshots[playerIndex];
488 W3DRenderObjectSnapshot *nextSnap;
489 while (snap)
490 { nextSnap = snap->m_next;
491 delete snap;
492 snap = nextSnap;
493 }
494 m_parentSnapshots[playerIndex]=NULL;
495 }
496}

607// ------------------------------------------------------------------------------------------------
611// ------------------------------------------------------------------------------------------------
612void W3DGhostObject::xfer( Xfer *xfer )
613{
614
615 // version
616 XferVersion currentVersion = 1;
617 XferVersion version = currentVersion;
618 xfer->xferVersion( &version, currentVersion );
619
620 // extend base class
621 GhostObject::xfer( xfer );
622
623 // xfer the drawable info object id
624 xfer->xferObjectID( &m_drawableInfo.m_shroudStatusObjectID );
625
626 // drawable info flags
627 xfer->xferInt( &m_drawableInfo.m_flags );
628
629 // drawable info drawable pointer
630 DrawableID drawableID = m_drawableInfo.m_drawable ? m_drawableInfo.m_drawable->getID() : INVALID_DRAWABLE_ID;
631 xfer->xferDrawableID( &drawableID );
632 if( xfer->getXferMode() == XFER_LOAD )
633 {
634
635 // reconnect the drawable pointer
636 m_drawableInfo.m_drawable = TheGameClient->findDrawableByID( drawableID );
637
638 // sanity
639 if( drawableID != INVALID_DRAWABLE_ID && m_drawableInfo.m_drawable == NULL )
640 DEBUG_CRASH(( "W3DGhostObject::xfer - Unable to find drawable for ghost object\n" ));
641
642 } // end if
643
644 //
645 // no need to mess with this "circular" back into itself pointer to the ghost object
646 // because it is already valid and assigned upon creation of this ghost object
647 //
648 // m_drawableInfo.m_ghostObject <--- no need to mess with this
649
650 // xfer snapshot array
651 UnsignedByte snapshotCount;
652 for( Int i = 0; i < MAX_PLAYER_COUNT; ++i )
653 {
654
655 // count the snapshots at this index
656 snapshotCount = 0;
657 W3DRenderObjectSnapshot *objectSnapshot = m_parentSnapshots[ i ];
658 while( objectSnapshot )
659 {
660
661 // increment count
662 snapshotCount++;
663
664 // on to the next snapshot
665 objectSnapshot = objectSnapshot->m_next;
666
667 } // end while
668
669 // xfer the snapshot count at this index
670 xfer->xferUnsignedByte( &snapshotCount );
671
672 //
673 // sanity, this catches when we read from the file a count of zero, but our data
674 // structure already has something allocated in this snapshot index
675 //
676 if( snapshotCount == 0 && m_parentSnapshots[ i ] != NULL )
677 {
678
679 DEBUG_CRASH(( "W3DGhostObject::xfer - m_parentShapshots[ %d ] has data present but the count from the xfer stream is empty\n" ));
680 throw INI_INVALID_DATA;
681
682 } // end if
683
684 // xfer each of the snapshots at this index
685 Real scale;
686 UnsignedInt color;
687 AsciiString name;
688 if( xfer->getXferMode() == XFER_SAVE )
689 {
690
691 // iterate through list
692 objectSnapshot = m_parentSnapshots[ i ];
693 while( objectSnapshot )
694 {
695
696 // write name from render object
697 name.set( objectSnapshot->m_robj->Get_Name() );
698 xfer->xferAsciiString( &name );
699
700 // write scale from render object
701 scale = objectSnapshot->m_robj->Get_ObjectScale();
702 xfer->xferReal( &scale );
703
704 // write color from render object
705 color = objectSnapshot->m_robj->Get_ObjectColor();
706 xfer->xferUnsignedInt( &color );
707
708 // xfer data
709 xfer->xferSnapshot( objectSnapshot );
710
711 // onto the next
712 objectSnapshot = objectSnapshot->m_next;
713
714 } // end while
715
716 } // end if, save
717 else
718 {
719 RenderObjClass *renderObject;
720 W3DRenderObjectSnapshot *prevObjectSnapshot = NULL;
721
722 for( UnsignedByte j = 0; j < snapshotCount; ++j )
723 {
724
725 // read render object name
726 xfer->xferAsciiString( &name );
727
728 // read scale
729 xfer->xferReal( &scale );
730
731 // read color
732 xfer->xferUnsignedInt( &color );
733
734 // create the render object
735 renderObject = W3DDisplay::m_assetManager->Create_Render_Obj( name.str(), scale, color );
736 disableUVAnimations(renderObject);
737
738 // we're loading, allocate new snapshot
739 objectSnapshot = NEW W3DRenderObjectSnapshot( renderObject, &m_drawableInfo, FALSE );
740
741 // attach to list
742 if( prevObjectSnapshot )
743 prevObjectSnapshot->m_next = objectSnapshot;
744 else
745 m_parentSnapshots[ i ] = objectSnapshot;
746 prevObjectSnapshot = objectSnapshot;
747
748 // xfer data
749 xfer->xferSnapshot( objectSnapshot );
750
751 // add snapshot to the scene
752 objectSnapshot->addToScene();
753
754 } // end for, j
755
756 } // end else, load
757
758 } // end for, i
759
760 //
761 // since there is a snapshot for this object, there cannot be a regular object/drawable
762 // in the world, we need to remove it
763 //
764 if( m_parentObject &&
765 m_parentSnapshots[ ThePlayerList->getLocalPlayer()->getPlayerIndex() ] != NULL &&
766 xfer->getXferMode() == XFER_LOAD )
767 removeParentObject();
768
769 // count of partition shroudedness info to follow
770 UnsignedByte shroudednessCount = 0;
771 UnsignedByte playerIndex;
772 for( playerIndex = 0; playerIndex < MAX_PLAYER_COUNT; ++playerIndex )
773 if( m_parentSnapshots[ playerIndex ] )
774 shroudednessCount++;
775 xfer->xferUnsignedByte( &shroudednessCount );
776
777 // shroudedness info
778 ObjectShroudStatus status;
779 if( xfer->getXferMode() == XFER_SAVE )
780 {
781
782 // iterate through all array points
783 for( playerIndex = 0; playerIndex < MAX_PLAYER_COUNT; ++playerIndex )
784 {
785
786 // is this a location with any info
787 if( m_parentSnapshots[ playerIndex ] )
788 {
789
790 // write this index
