143SeismicSimulationFilterBase::SeismicSimStatusCode DomeStyleSeismicFilter::filterCallback( WorldHeightMapInterfaceClass *heightMap, const SeismicSimulationNode *node )
144{
145
146 Int life = node->m_life;
147
148 if ( heightMap == NULL )
149 return SEISMIC_STATUS_INVALID;
150
151
152 if ( life == 0 )
153 return SEISMIC_STATUS_ACTIVE;
154 if ( life < 15 )
155 {
156 // ADD HEIGHT BECAUSE THE EXPLOSION IS PUSHING DIRT UP
157
158 Real magnitude = node->m_magnitude;
159
160 Real offsScalar = magnitude / (Real)life; // real-life, get it?
161 Int radius = node->m_radius;
162 Int border = heightMap->getBorderSize();
163 Int centerX = node->m_center.x + border ;
164 Int centerY = node->m_center.y + border ;
165
166 UnsignedInt workspaceWidth = radius*2;
167 Real *workspace = NEW( Real[ sqr(workspaceWidth) ] );
168 Real *workspaceEnd = workspace + sqr(workspaceWidth);
169
170
171 for ( Real *t = workspace; t < workspaceEnd; ++t ) *t = 0.0f;// clear the workspace
172
173 for (Int x = 0; x < radius; ++x)
174 {
175 for (Int y = 0; y < radius; ++y)
176 {
177
178 Real distance = sqrt( sqr(x) + sqr(y) );//Pythagoras
179
180 if ( distance < radius )
181 {
182 Real distScalar = cos( ( distance / radius * (PI/2) ) );
183 Real height = (offsScalar * distScalar);
184
185 workspace[ (radius + x) + workspaceWidth * (radius + y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX + x, centerY + y ) ;//kaleidoscope
186
187 if ( x != 0 ) // non-zero test prevents cross-shaped double stamp
188 {
189 workspace[ (radius - x) + workspaceWidth * (radius + y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX - x, centerY + y ) ;
190 if ( y != 0 )
191 workspace[ (radius - x) + workspaceWidth * (radius - y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX - x, centerY - y ) ;
192 }
193 if ( y != 0 )
194 workspace[ (radius + x) + workspaceWidth * (radius - y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX + x, centerY - y ) ;
195 }
196 }
197 }
198
199 // stuff the values from the workspace into the heightmap's velocities
200 for (x = 0; x < workspaceWidth; ++x)
201 for (Int y = 0; y < workspaceWidth; ++y)
202 heightMap->setSeismicZVelocity( centerX - radius + x, centerY - radius + y, MIN( 9.0f, workspace[ x + workspaceWidth * y ]) );
203
204 delete [] workspace;
205
206 return SEISMIC_STATUS_ACTIVE;
207 }
208 else
209 return SEISMIC_STATUS_ZERO_ENERGY;
210}
