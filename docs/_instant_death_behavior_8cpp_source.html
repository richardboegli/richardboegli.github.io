64{

123void InstantDeathBehavior::onDie( const DamageInfo *damageInfo )
124{
125 if (!isDieApplicable(damageInfo))
126 return;
127
128 AIUpdateInterface* ai = getObject()->getAIUpdateInterface();
129 if (ai)
130 {
131 // has another AI already handled us. (hopefully another InstantDeathBehavior)
132 if (ai->isAiInDeadState())
133 return;
134 ai->markAsDead();
135 }
136
137 const InstantDeathBehaviorModuleData* d = getInstantDeathBehaviorModuleData();
138
139 Int idx, listSize;
140
141 listSize = d->m_fx.size();
142 if (listSize > 0)
143 {
144 idx = GameLogicRandomValue(0, listSize-1);
145 const FXListVec& v = d->m_fx;
146 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
147 const FXList* fxl = v[idx];
148 FXList::doFXObj(fxl, getObject(), NULL);
149 }
150
151 listSize = d->m_ocls.size();
152 if (listSize > 0)
153 {
154 idx = GameLogicRandomValue(0, listSize-1);
155 const OCLVec& v = d->m_ocls;
156 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
157 const ObjectCreationList* ocl = v[idx];
158 ObjectCreationList::create(ocl, getObject(), NULL);
159 }
160
161 listSize = d->m_weapons.size();
162 if (listSize > 0)
163 {
164 idx = GameLogicRandomValue(0, listSize-1);
165 const WeaponTemplateVec& v = d->m_weapons;
166 DEBUG_ASSERTCRASH(idx>=0&&idx<v.size(),("bad idx"));
167 const WeaponTemplate* wt = v[idx];
168 if (wt)
169 {
170 TheWeaponStore->createAndFireTempWeapon(wt, getObject(), getObject()->getPosition());
171 }
172 }
173
174 TheGameLogic->destroyObject(getObject());
175}
