87// All this does is stop the call path if we haven't been cleared to draw yet
88void W3DDependencyModelDraw::doDrawModule(const Matrix3D* transformMtx)
89{
90 if( m_dependencyCleared )
91 {
92 // We've been cleared by the thing we were waiting to draw, so we can draw.
93 W3DModelDraw::doDrawModule( transformMtx );
94 m_dependencyCleared = FALSE;
95
96
97 // A handy place to synchronize my drawable with container's
98 Drawable *myDrawable = getDrawable();
99 if ( ! myDrawable )
100 return;
101
102 const Object *me = myDrawable->getObject();
103 if ( ! me )
104 return;
105
106 Drawable *theirDrawable = NULL;
107
108 if( me->getContainedBy() && !me->getContainedBy()->getContain()->isEnclosingContainerFor(me) )
109 theirDrawable = me->getContainedBy()->getDrawable();
110
111 if( ! theirDrawable )
112 return;
113
114 myDrawable->imitateStealthLook( *theirDrawable );
115
116 }
117}

125// ------------------------------------------------------------------------------------------------
126void W3DDependencyModelDraw::adjustTransformMtx(Matrix3D& mtx) const
127{
128 W3DModelDraw::adjustTransformMtx(mtx);
129
130 // We have an additional adjustment to make, we want to use a bone in our container if there is one
131 const Object *me = getDrawable()->getObject();
132 const W3DDependencyModelDrawModuleData *md = getW3DDependencyModelDrawModuleData();
133
134 if( md->m_attachToDrawableBoneInContainer.isNotEmpty()
135 && me
136 && me->getContainedBy()
137 && !me->getContainedBy()->getContain()->isEnclosingContainerFor(me)
138 )
139 {
140 // If we are currently "riding on", then our client position is determined by the client position of
141 // a particular bone in our container object. Our logic position is updated by OpenContain.
142 const Drawable *theirDrawable = me->getContainedBy()->getDrawable();
143 if( theirDrawable )
144 {
145 Matrix3D theirBoneMtx;
146 if( theirDrawable->getCurrentWorldspaceClientBonePositions( md->m_attachToDrawableBoneInContainer.str(), theirBoneMtx ) )
147 {
148 mtx = theirBoneMtx;
149 }
150 else
151 {
152 mtx = *theirDrawable->getTransformMatrix();//TransformMatrix();
153 DEBUG_LOG(("m_attachToDrawableBoneInContainer %s not found\n",getW3DDependencyModelDrawModuleData()->m_attachToDrawableBoneInContainer.str()));
154 }
155 }
156 }
157}
