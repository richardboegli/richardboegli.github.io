131/* Info structure describing bitmaps */
132
133#define GIMEX_FRAMENAME_SIZE 512
134#define GIMEX_COMMENT_SIZE 1024
135#define GIMEX_COLOURTBL_SIZE 256
136#define GIMEX_HOTSPOTTBL_SIZE 1024

139typedef struct
140{
141 int signature; /* signature of gimex ie 'tga ' (optional) */
142 int size; /* size of GINFO structure */
143 int version; /* version number of GINFO structure (300) */
144 int framenum; /* current frame */
145 int width; /* width of bitmap in pixels */
146 int height; /* height of bitmap in pixels */
147 int bpp; /* bits per pixel (8, 16 or 32) */
148 int originalbpp; /* bits per pixel in original image (1 to 32) */
149 int startcolour; /* first colour in palette */
150 int numcolours; /* number of colours in original indexed palette */
151 ARGB colourtbl[GIMEX_COLOURTBL_SIZE]; /* 8 bit palette */
152 int subtype; /* internal format sub-type 0-default */
153 int packed; /* type of packing on original image. 0 none, 1 run, n other */
154 int quality; /* quality of lossy packing 0..100 */
155 int framesize; /* size of frame in bytes */
156 int alphabits; /* number of bits in alpha channel */
157 int redbits; /* number of bits in red channel */
158 int greenbits; /* number of bits in green channel */
159 int bluebits; /* number of bits in blue channel */
160 int centerx; /* center point relative to upper left corner */
161 int centery;
162 int defaultx; /* default coordinate point */
163 int defaulty;
164 int numhotspots; /* number of hot spots defined */
165 char framename[GIMEX_FRAMENAME_SIZE]; /* null terminated name of frame/image */
166 char comment[GIMEX_COMMENT_SIZE]; /* null terminated multiline user comment */
167 int hotspottbl[GIMEX_HOTSPOTTBL_SIZE][GIMEX_HOTSPOTTBL_VALUES];/* up to 256 hot spots, XY pairs relative to upperleft */
168 float dpi; /* dots per inch ie 72.0 */
169 float fps; /* frame per second (one over duration). 15.0 is typical */
170 int reserved[3]; /* reserved for future use - set to zero */
171} GINFO;

374/* get motorola memory */
375static __inline unsigned int ggetm(const void *src, int bytes)
376{
377 if (bytes==1)
378 return (unsigned int) *(const unsigned char *) src;
379#if defined(__APPLE__) || (defined(__MWERKS__) && defined(__PPCGEKKO__))
380 else if (bytes==2)
381 return (unsigned int) *(const unsigned short *) src;
382 else if (bytes==4)
383 return (unsigned int) *(const unsigned int *) src;
384#else
385 else if (bytes==2)
386 return (((unsigned int) *(const unsigned char *) src)<<8) | (((unsigned int) *((const unsigned char *) src+1)));
387 else if (bytes==4)
388 return (((unsigned int) *(const unsigned char *) src)<<24) | (((unsigned int) *((const unsigned char *) src+1))<<16) | (((unsigned int) *((const unsigned char *) src+2))<<8) | (((unsigned int) *((const unsigned char *) src+3)));
389#endif
390 else if (bytes==3)
391 return (((unsigned int) *(const unsigned char *) src)<<16) | (((unsigned int) *((const unsigned char *) src+1))<<8) | (((unsigned int) *((const unsigned char *) src+2)));
392 return 0;
393}

419static __inline void gputm(void *dst, unsigned int data, int bytes)
420{
421 if (bytes==1)
422 {
423 ((unsigned char *) dst)[0] = (unsigned char) data;
424 }
425#if defined(__APPLE__) || (defined(__MWERKS__) && defined(__PPCGEKKO__))
426 else if (bytes==2)
427 {
428 ((unsigned short *) dst)[0] = (unsigned short) data;
429 }
430 else if (bytes==4)
431 {
432 ((unsigned int *) dst)[0] = (unsigned int) data;
433 }
434#else
435 else if (bytes==2)
436 {
437 ((unsigned char *) dst)[0] = (unsigned char) (data>>8);
438 ((unsigned char *) dst)[1] = (unsigned char) data;
439 }
440 else if (bytes==4)
441 {
442 ((unsigned char *) dst)[0] = (unsigned char) (data>>24);
443 ((unsigned char *) dst)[1] = (unsigned char) (data>>16);
444 ((unsigned char *) dst)[2] = (unsigned char) (data>>8);
445 ((unsigned char *) dst)[3] = (unsigned char) data;
446 }
447#endif
448 else if (bytes==3)
449 {
450 ((unsigned char *) dst)[0] = (unsigned char) (data>>16);
451 ((unsigned char *) dst)[1] = (unsigned char) (data>>8);
452 ((unsigned char *) dst)[2] = (unsigned char) data;
453 }
454}
