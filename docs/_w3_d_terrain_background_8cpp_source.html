108//=============================================================================
109// W3DTerrainBackground::doPartialUpdate
110//=============================================================================
115void W3DTerrainBackground::doPartialUpdate(const IRegion2D &partialRange, WorldHeightMap *htMap, Bool doTextures )
116{
117 if (m_map==NULL) return;
118 if (htMap) {
119 REF_PTR_SET(m_map, htMap);
120 }
121
122 if (!m_initialized) {
123 return;
124 }
125 doTesselatedUpdate(partialRange, htMap, doTextures);
126
127 return;
128
129 Int requiredVertexSize = (m_width+1) * (m_width+1) + 6;
130 if (m_vertexTerrainSize<requiredVertexSize || m_vertexTerrain==NULL) {
131 m_vertexTerrainSize = requiredVertexSize;
132 REF_PTR_RELEASE(m_vertexTerrain);
133 REF_PTR_RELEASE(m_indexTerrain);
134 m_vertexTerrain=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,m_vertexTerrainSize+4,DX8VertexBufferClass::USAGE_DEFAULT));
135 }
136
137 Int requiredIndexSize = (m_width+1) * (m_width+1) + 6;
138 if (m_indexTerrainSize<requiredIndexSize || m_indexTerrain==NULL) {
139 m_indexTerrainSize = requiredIndexSize;
140 REF_PTR_RELEASE(m_indexTerrain);
141 m_indexTerrain=NEW_REF(DX8IndexBufferClass,(m_indexTerrainSize+4,DX8IndexBufferClass::USAGE_DEFAULT));
142 }
143 Int minX = m_xOrigin;
144 Int minY = m_yOrigin;
145 Int maxX = m_xOrigin + m_width;
146 Int maxY = m_yOrigin + m_width;
147 Int limitX = m_map->getXExtent()-1;
148 Int limitY = m_map->getYExtent()-1;
149 if (maxX>limitX) maxX = limitX;
150 if (maxY>limitY) maxY = limitY;
151
152 if (partialRange.lo.x > maxX) return;
153 if (partialRange.lo.y > maxY) return;
154 if (partialRange.hi.x < minX) return;
155 if (partialRange.hi.y < minY) return;
156
157 m_curNumTerrainVertices = 0;
158 //m_curNumTerrainIndices = 0;
159 VertexFormatXYZDUV2 *vb;
160 UnsignedShort *ib;
161 // Lock the buffer.
162 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexTerrain);
163 vb=(VertexFormatXYZDUV2*)lockVtxBuffer.Get_Vertex_Array();
164 // Add to the vertex buffer.
165
166 VertexFormatXYZDUV2 *curVb = vb;
167 MinMaxAABoxClass bounds;
168 bounds.Init_Empty();
169
170 Int i, j;
171 for (j=minY; j<=maxY; j+=STEP) {
172 for (i=minX; i<=maxX; i+=STEP) {
173 if (m_curNumTerrainVertices >= m_vertexTerrainSize) return;
174 curVb->diffuse = (0<<24)|TheTerrainRenderObject->getStaticDiffuse(i,j);
175 Vector3 pos;
176 pos.Z = ((float)m_map->getHeight(i,j)*MAP_HEIGHT_SCALE);
177 pos.X = (i)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
178 pos.Y = (j)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
179 curVb->u1 = (float)(i-minX)/(float)(m_width);
180 curVb->v1 = 1.0f - (float)(j-minY)/(float)(m_width);
181 curVb->x = pos.X;
182 curVb->y = pos.Y;
183 curVb->z = pos.Z;
184 curVb++;
185 m_curNumTerrainVertices++;
186 bounds.Add_Point(pos);
187 }
188 }
189 m_bounds.Init(bounds);
190
191 if (m_terrainTexture == NULL || doTextures) {
192 REF_PTR_RELEASE(m_terrainTexture);
193 REF_PTR_RELEASE(m_terrainTexture2X);
194 REF_PTR_RELEASE(m_terrainTexture4X);
195 m_terrainTexture = m_map->getFlatTexture(m_xOrigin, m_yOrigin, m_width, PIXELS_PER_GRID);
196 // DEBUG ONLY. jba. m_terrainTexture = (TerrainTextureClass *)NEW_REF(TextureClass, ("TBBib.tga"));
197 m_terrainTexture->Get_Filter().Set_U_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
198 m_terrainTexture->Get_Filter().Set_V_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
199 }
200
201 if (m_curNumTerrainIndices == 0) {
202 // Only do the index buffer if it has never been done. Index values don't change. jba.
203 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexTerrain);
204 ib = lockIdxBuffer.Get_Index_Array();
205 UnsignedShort *curIb = ib;
206 Int yOffset = ((maxX - minX)/STEP+1);
207 Int width = yOffset;
208 Int height = (maxY - minY)/STEP;
209 *curIb++ = width-1;
210 m_curNumTerrainIndices++;
211 for (j=0; j<height; j++) {
212 *curIb++ = j*yOffset + yOffset + width-1;
213 m_curNumTerrainIndices++;
214 for (i=width-2; i>=0; i--) {
215 if (m_curNumTerrainIndices+2 > m_indexTerrainSize) return;
216 *curIb++ = j*yOffset + i;
217 *curIb++ = j*yOffset + i+yOffset;
218 m_curNumTerrainIndices+=2;
219 }
220 j++;
221 if (j<height) {
222 *curIb++ = j*yOffset + yOffset;
223 m_curNumTerrainIndices++;
224 for (i=1; i<width; i++) {
225 if (m_curNumTerrainIndices+2 > m_indexTerrainSize) return;
226 *curIb++ = j*yOffset + i;
227 *curIb++ = j*yOffset + i+yOffset;
228 m_curNumTerrainIndices+=2;
229 }
230 }
231 }
232 }
233}

240Bool W3DTerrainBackground::advanceLeft(ICoord2D &left, Int xOffset, Int yOffset, Int width)
241{
242 while (left.y < yOffset+width) {
243 left.y++;
244 if (m_map->getFlipState(left.x+m_xOrigin, left.y+m_yOrigin)) {
245 return true;
246 }
247 }
248 while (left.x < xOffset+width-1) {
249 left.x++;
250 if (m_map->getFlipState(left.x+m_xOrigin, left.y+m_yOrigin)) {
251 return true;
252 }
253 }
254 return false;
255}

257//=============================================================================
258// W3DTerrainBackground::fillVBRecursive
259//=============================================================================
262Bool W3DTerrainBackground::advanceRight(ICoord2D &right, Int xOffset, Int yOffset, Int width)
263{
264 while (right.x < xOffset+width) {
265 right.x++;
266 if (m_map->getFlipState(right.x+m_xOrigin, right.y+m_yOrigin)) {
267 return true;
268 }
269 }
270 while (right.y < yOffset+width-1) {
271 right.y++;
272 if (m_map->getFlipState(right.x+m_xOrigin, right.y+m_yOrigin)) {
273 return true;
274 }
275 }
276 return false;
277}

280// W3DTerrainBackground::fillVBRecursive
281//=============================================================================
284void W3DTerrainBackground::fillVBRecursive(UnsignedShort *ib, Int xOffset, Int yOffset,
285 Int width, UnsignedShort *ndx, Int &curIndex)
286{
287
288 Int bottomLeftNdx = ndx[xOffset+yOffset*(m_width+1)];
289 Int topRightNdx = ndx[xOffset+width + (yOffset+width)*(m_width+1)];
290
291 Int limitX = m_map->getXExtent()-1;
292 Int limitY = m_map->getYExtent()-1;
293 Int i, j;
294 Bool match = true;
295 Int minX = m_xOrigin+xOffset;
296 Int minY = m_yOrigin+yOffset;
297 Int cornerHeight = m_map->getHeight(minX, minY);
298
299 for (i=0; i<=width; i++) {
300 for (j=0; j<=width; j++) {
301 Int k = minX+i;
302 k = k<limitX?k:limitX;
303 Int l = minY+j;
304 l = l<limitY?l:limitY;
305 if (cornerHeight!=m_map->getHeight(k, l)) {
306 match = false;
307 break;
308 }
309 }
310 }
311 if (width==1) {
312 match = true;
313 }
314
315 if (match) {
316
317
318 UnsignedShort prevNdxLeft;
319 UnsignedShort prevNdxRight;
320 ICoord2D left;
321 left.x = xOffset;
322 left.y = yOffset;
323 ICoord2D right;
324 right.x = xOffset;
325 right.y = yOffset;
326 advanceLeft(left, xOffset, yOffset, width);
327 advanceRight(right, xOffset, yOffset, width);
328
329 if (ib) {
330 ib[curIndex] = bottomLeftNdx;
331 }
332 curIndex++;
333
334 prevNdxRight = ndx[right.x+right.y*(m_width+1)];
335 if (ib) {
336 ib[curIndex] = prevNdxRight;
337 }
338 curIndex++;
339
340 prevNdxLeft = ndx[left.x+left.y*(m_width+1)];
341 if (ib) {
342 ib[curIndex] = prevNdxLeft;
343 }
344 curIndex++;
345 Bool didLeft = true;
346 Bool didRight = true;
347 while (didLeft || didRight) {
348 didLeft = advanceLeft(left, xOffset, yOffset, width);
349 if (didLeft) {
350
351 if (ib) {
352 ib[curIndex] = prevNdxLeft;
353 }
354 curIndex++;
355
356 if (ib) {
357 ib[curIndex] = prevNdxRight;
358 }
359 curIndex++;
360
361 prevNdxLeft = ndx[left.x+left.y*(m_width+1)];
362 if (ib) {
363 ib[curIndex] = prevNdxLeft;
364 }
365 curIndex++;
366 }
367 didRight = advanceRight(right, xOffset, yOffset, width);
368 if (didRight) {
369
370 if (ib) {
371 ib[curIndex] = prevNdxLeft;
372 }
373 curIndex++;
374
375 if (ib) {
376 ib[curIndex] = prevNdxRight;
377 }
378 curIndex++;
379
380 prevNdxRight = ndx[right.x+right.y*(m_width+1)];
381 if (ib) {
382 ib[curIndex] = prevNdxRight;
383 }
384 curIndex++;
385 }
386
387 }
388
389 if (ib) {
390 ib[curIndex] = prevNdxLeft;
391 }
392
393 curIndex++;
394
395 if (ib) {
396 ib[curIndex] = prevNdxRight;
397 }
398 curIndex++;
399
400 if (ib) {
401 ib[curIndex] = topRightNdx;
402 }
403 curIndex++;
404
405 return;
406 }
407 Int halfWidth = width/2;
408
409 fillVBRecursive(ib, xOffset, yOffset, halfWidth, ndx, curIndex);
410 fillVBRecursive(ib, xOffset, yOffset+halfWidth, halfWidth, ndx, curIndex);
411 fillVBRecursive(ib, xOffset+halfWidth, yOffset, halfWidth, ndx, curIndex);
412 fillVBRecursive(ib, xOffset+halfWidth, yOffset+halfWidth, halfWidth, ndx, curIndex);
413
414}

421void W3DTerrainBackground::setFlipRecursive(Int xOffset, Int yOffset, Int width)
422{
423
424 Int limitX = m_map->getXExtent()-1;
425 Int limitY = m_map->getYExtent()-1;
426 Int i, j;
427 Bool match = true;
428 Int minX = m_xOrigin+xOffset;
429 Int minY = m_yOrigin+yOffset;
430 Int cornerHeight = m_map->getHeight(minX, minY);
431
432 for (i=0; i<=width; i++) {
433 for (j=0; j<=width; j++) {
434 Int k = minX+i;
435 k = k<limitX?k:limitX;
436 Int l = minY+j;
437 l = l<limitY?l:limitY;
438 if (cornerHeight!=m_map->getHeight(k, l)) {
439 match = false;
440 break;
441 }
442 }
443 }
444 if (width==1) {
445 match = true;
446 }
447
448 if (match) {
449 m_map->setFlipState(minX, minY, true);
450 m_map->setFlipState(minX+width, minY, true);
451 m_map->setFlipState(minX+width, minY+width, true);
452 m_map->setFlipState(minX, minY+width, true);
453 return;
454 }
455 Int halfWidth = width/2;
456
457
458 setFlipRecursive(xOffset, yOffset, halfWidth);
459 setFlipRecursive(xOffset, yOffset+halfWidth, halfWidth);
460 setFlipRecursive(xOffset+halfWidth, yOffset, halfWidth);
461 setFlipRecursive(xOffset+halfWidth, yOffset+halfWidth, halfWidth);
462
463}

466// W3DTerrainBackground::doTesselatedUpdate
467//=============================================================================
472void W3DTerrainBackground::doTesselatedUpdate(const IRegion2D &partialRange, WorldHeightMap *htMap, Bool doTextures )
473{
474 if (m_map==NULL) return;
475 if (htMap) {
476 REF_PTR_SET(m_map, htMap);
477 }
478 if (!m_initialized) {
479 return;
480 }
481 Int minX = m_xOrigin;
482 Int minY = m_yOrigin;
483 Int maxX = m_xOrigin + m_width;
484 Int maxY = m_yOrigin + m_width;
485 Int limitX = m_map->getXExtent()-1;
486 Int limitY = m_map->getYExtent()-1;
487
488 if (partialRange.lo.x > maxX) return;
489 if (partialRange.lo.y > maxY) return;
490 if (partialRange.hi.x < minX) return;
491 if (partialRange.hi.y < minY) return;
492
493 setFlip(htMap);
494
495 Int count = (m_width+1)*(m_width+1);
496
497 UnsignedShort *ndx = new UnsignedShort[count];
498
499 Int requiredVertex = 0;
500 Int i, j;
501 for (j=minY; j<=maxY; j++) {
502 for (i=minX; i<=maxX; i++) {
503 Int ndxNdx = i-minX + (m_width+1)*(j-minY);
504 DEBUG_ASSERTCRASH(ndxNdx<count, ("Bad ndxNdx"));
505 ndx[ndxNdx] = 0;
506 if (m_map->getFlipState(i, j)) {
507 requiredVertex++;
508 }
509 }
510 }
511
512 if (m_vertexTerrainSize<requiredVertex || m_vertexTerrain==NULL) {
513 m_vertexTerrainSize = requiredVertex;
514 REF_PTR_RELEASE(m_vertexTerrain);
515 m_vertexTerrain=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV2,m_vertexTerrainSize+4,DX8VertexBufferClass::USAGE_DEFAULT));
516 }
517
518 m_curNumTerrainVertices = 0;
519 VertexFormatXYZDUV2 *vb;
520 // Lock the buffer.
521 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexTerrain);
522 vb=(VertexFormatXYZDUV2*)lockVtxBuffer.Get_Vertex_Array();
523 VertexFormatXYZDUV2 *curVb = vb;
524 // Add to the vertex buffer.
525 for (j=minY; j<=maxY; j++) {
526 for (i=minX; i<=maxX; i++) {
527 if (m_map->getFlipState(i, j)) {
528 curVb->diffuse = (0<<24)|TheTerrainRenderObject->getStaticDiffuse(i,j);
529 Vector3 pos;
530 Int k = i<limitX?i:limitX;
531 Int l = j<limitY?j:limitY;
532 pos.Z = ((float)m_map->getHeight(k,l)*MAP_HEIGHT_SCALE);
533 pos.X = (i)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
534 pos.Y = (j)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
535 curVb->u1 = (float)(i-minX)/(float)(m_width);
536 curVb->v1 = 1.0f - (float)(j-minY)/(float)(m_width);
537 curVb->x = pos.X;
538 curVb->y = pos.Y;
539 curVb->z = pos.Z;
540 curVb++;
541 Int ndxNdx = i-minX + (m_width+1)*(j-minY);
542 DEBUG_ASSERTCRASH(ndxNdx<count, ("Bad ndxNdx"));
543 ndx[ndxNdx] = m_curNumTerrainVertices;
544 m_curNumTerrainVertices++;
545 }
546 }
547 }
548
549 Int requiredIndex = 0;
550
551 fillVBRecursive(NULL, 0, 0, m_width, ndx, requiredIndex);
552
553 if (m_indexTerrainSize<requiredIndex || m_indexTerrain==NULL) {
554 m_indexTerrainSize = requiredIndex;
555 REF_PTR_RELEASE(m_indexTerrain);
556 m_indexTerrain=NEW_REF(DX8IndexBufferClass,(m_indexTerrainSize+4,DX8IndexBufferClass::USAGE_DEFAULT));
557 }
558
559 m_curNumTerrainIndices = 0;
560
561 UnsignedShort *ib;
562 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexTerrain);
563 ib = lockIdxBuffer.Get_Index_Array();
564 fillVBRecursive(ib, 0, 0, m_width, ndx, m_curNumTerrainIndices);
565 delete ndx;
566 ndx = NULL;
567
568 MinMaxAABoxClass bounds;
569 bounds.Init_Empty();
570
571 for (j=minY; j<=maxY; j+=1) {
572 for (i=minX; i<=maxX; i+=1) {
573 Vector3 pos;
574 Int k = i<limitX?i:limitX;
575 Int l = j<limitY?j:limitY;
576 pos.Z = ((float)m_map->getHeight(k,l)*MAP_HEIGHT_SCALE);
577 pos.X = (i)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
578 pos.Y = (j)*MAP_XY_FACTOR - m_map->getBorderSizeInline()*MAP_XY_FACTOR;
579 bounds.Add_Point(pos);
580 }
581 }
582 m_bounds.Init(bounds);
583
584 if (m_terrainTexture == NULL || doTextures) {
585 REF_PTR_RELEASE(m_terrainTexture);
586 REF_PTR_RELEASE(m_terrainTexture2X);
587 REF_PTR_RELEASE(m_terrainTexture4X);
588 m_terrainTexture = m_map->getFlatTexture(m_xOrigin, m_yOrigin, m_width, PIXELS_PER_GRID);
589 // DEBUG ONLY. jba. m_terrainTexture = (TerrainTextureClass *)NEW_REF(TextureClass, ("TBBib.tga"));
590 m_terrainTexture->Get_Filter().Set_U_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
591 m_terrainTexture->Get_Filter().Set_V_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
592 }
593

668//=============================================================================
669// W3DTerrainBackground::updateCenter
670//=============================================================================
672//=============================================================================
673void W3DTerrainBackground::updateCenter(CameraClass *camera)
674{
675 if (camera->Cull_Box(m_bounds)) {
676 m_cullStatus = CULL_STATUS_INVISIBLE;
677 } else {
678 m_cullStatus = CULL_STATUS_VISIBLE;
679 }
680
681 if (m_cullStatus==CULL_STATUS_INVISIBLE) {
682 REF_PTR_RELEASE(m_terrainTexture2X);
683 REF_PTR_RELEASE(m_terrainTexture4X);
684 m_texMultiplier = TEX1X;
685 return;
686 }
687 Vector3 cameraPos = camera->Get_Position();
688 const Real mipDistance = 310;
689 const Real mipSlop = 40;
690 const Real mip4xDistanceSqr = sqr(mipDistance+mipSlop);
691 const Real mip2xDistanceSqr = sqr(2*mipDistance+mipSlop);
692 const Real mipLODDistanceSqr = sqr(4*mipDistance+mipSlop);
693 Real minDistSqr = 2*mip2xDistanceSqr;
694 Int i, j, k;
695 for (i=-1; i<2; i++) {
696 for (j=-1; j<2; j++) {
697 for (k=-1; k<2; k++) {
698 Vector3 corner = m_bounds.Center;
699 corner.X += m_bounds.Extent.X * i;
700 corner.Y += m_bounds.Extent.Y * j;
701 corner.Z += m_bounds.Extent.Z * k;
702 Real distSqr = (cameraPos-corner).Length2();
703 if (distSqr<minDistSqr) minDistSqr = distSqr;
704 }
705 }
706 }
707 m_texMultiplier = TEX1X;
708 if (minDistSqr<mip4xDistanceSqr) {
709 m_texMultiplier = TEX4X;
710 } else if (minDistSqr<mip2xDistanceSqr) {
711 m_texMultiplier = TEX2X;
712 } else {
713 REF_PTR_RELEASE(m_terrainTexture4X);
714 REF_PTR_RELEASE(m_terrainTexture2X);
715 Int LOD = 0;
716 if (minDistSqr>mipLODDistanceSqr) {
717 LOD = 1;
718 }
719 m_terrainTexture->setLOD(LOD);
720 }
721
722}
