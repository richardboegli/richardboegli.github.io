183//============================================================================
184// BinkVideoPlayer::createStream
185//============================================================================
186
187VideoStreamInterface* BinkVideoPlayer::createStream( HBINK handle )
188{
189
190 if ( handle == NULL )
191 {
192 return NULL;
193 }
194
195 BinkVideoStream *stream = NEW BinkVideoStream;
196
197 if ( stream )
198 {
199
200 stream->m_handle = handle;
201 stream->m_next = m_firstStream;
202 stream->m_player = this;
203 m_firstStream = stream;
204
205 // never let volume go to 0, as Bink will interpret that as "play at full volume".
206 Int mod = (Int) ((TheAudio->getVolume(AudioAffect_Speech) * 0.8f) * 100) + 1;
207 Int volume = (32768*mod)/100;
208 DEBUG_LOG(("BinkVideoPlayer::createStream() - About to set volume (%g -> %d -> %d\n",
209 TheAudio->getVolume(AudioAffect_Speech), mod, volume));
210 BinkSetVolume( stream->m_handle,0, volume);
211 DEBUG_LOG(("BinkVideoPlayer::createStream() - set volume\n"));
212 }
213
214 return stream;
215}

217//============================================================================
218// BinkVideoPlayer::open
219//============================================================================
220
221VideoStreamInterface* BinkVideoPlayer::open( AsciiString movieTitle )
222{
223 VideoStreamInterface* stream = NULL;
224
225 const Video* pVideo = getVideo(movieTitle);
226 if (pVideo) {
227 DEBUG_LOG(("BinkVideoPlayer::createStream() - About to open bink file\n"));
228
229 if (TheGlobalData->m_modDir.isNotEmpty())
230 {
231 char filePath[ _MAX_PATH ];
232 sprintf( filePath, "%s%s\\%s.%s", TheGlobalData->m_modDir.str(), VIDEO_PATH, pVideo->m_filename.str(), VIDEO_EXT );
233 HBINK handle = BinkOpen(filePath , BINKPRELOADALL );
234 DEBUG_ASSERTLOG(!handle, ("opened bink file %s\n", filePath));
235 if (handle)
236 {
237 return createStream( handle );
238 }
239 }
240
241 char localizedFilePath[ _MAX_PATH ];
242 sprintf( localizedFilePath, VIDEO_LANG_PATH_FORMAT, GetRegistryLanguage().str(), pVideo->m_filename.str(), VIDEO_EXT );
243 HBINK handle = BinkOpen(localizedFilePath , BINKPRELOADALL );
244 DEBUG_ASSERTLOG(!handle, ("opened localized bink file %s\n", localizedFilePath));
245 if (!handle)
246 {
247 char filePath[ _MAX_PATH ];
248 sprintf( filePath, "%s\\%s.%s", VIDEO_PATH, pVideo->m_filename.str(), VIDEO_EXT );
249 handle = BinkOpen(filePath , BINKPRELOADALL );
250 DEBUG_ASSERTLOG(!handle, ("opened bink file %s\n", localizedFilePath));
251 }
252
253 DEBUG_LOG(("BinkVideoPlayer::createStream() - About to create stream\n"));
254 stream = createStream( handle );
255 }
256
257 return stream;
258}

348//============================================================================
349// BinkVideoStream::frameRender
350//============================================================================
351
352void BinkVideoStream::frameRender( VideoBuffer *buffer )
353{
354 if ( buffer )
355 {
356 void *mem = buffer->lock();
357
358 unsigned int flags;
359
360 switch ( buffer->format())
361 {
362 case VideoBuffer::TYPE_X8R8G8B8:
363 flags = BINKSURFACE32;
364 break;
365
366 case VideoBuffer::TYPE_R8G8B8:
367 flags = BINKSURFACE24;
368 break;
369
370 case VideoBuffer::TYPE_R5G6B5:
371 flags = BINKSURFACE565;
372 break;
373
374 case VideoBuffer::TYPE_X1R5G5B5:
375 flags = BINKSURFACE555;
376 break;
377
378 default:
379 return;
380 }
381
382 if ( mem != NULL )
383 {
384
385 BinkCopyToBuffer ( m_handle, mem, buffer->pitch(), buffer->height(),
386 buffer->xPos(), buffer->yPos(), flags );
387 buffer->unlock();
388 }
389 }
390
391}
