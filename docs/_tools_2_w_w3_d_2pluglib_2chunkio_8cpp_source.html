95/***********************************************************************************************
96 * ChunkSaveClass::Begin_Chunk -- Begin a new chunk in the file *
97 * *
98 * INPUT: *
99 * id - id of the chunk *
100 * *
101 * OUTPUT: *
102 * *
103 * WARNINGS: *
104 * *
105 * HISTORY: *
106 * 07/17/1997 GH : Created. *
107 *=============================================================================================*/
108bool ChunkSaveClass::Begin_Chunk(uint32 id)
109{
110 ChunkHeader chunkh;
111 int filepos;
112
113 // If we have a parent chunk, set its 'Contains_Chunks' flag
114 if (StackIndex > 0) {
115 HeaderStack[StackIndex-1].Set_Sub_Chunk_Flag(true);
116 }
117
118 // Save the current file position and chunk header
119 // for the call to End_Chunk.
120 chunkh.Set_Type(id);
121 chunkh.Set_Size(0);
122 filepos = File->Seek(0);
123
124 PositionStack[StackIndex] = filepos;
125 HeaderStack[StackIndex] = chunkh;
126 StackIndex++;
127
128 // write a temporary chunk header (size = 0)
129 if (File->Write(&chunkh,sizeof(chunkh)) != sizeof(chunkh)) {
130 return false;
131 }
132 return true;
133}

136/***********************************************************************************************
137 * ChunkSaveClass::End_Chunk -- Close a chunk, computes the size and adds to the header *
138 * *
139 * INPUT: *
140 * *
141 * OUTPUT: *
142 * *
143 * WARNINGS: *
144 * *
145 * HISTORY: *
146 * 07/17/1997 GH : Created. *
147 *=============================================================================================*/
148bool ChunkSaveClass::End_Chunk(void)
149{
150 // If the user didn't close his micro chunks bad things are gonna happen
151 assert(!InMicroChunk);
152
153 // Save the current position
154 int curpos = File->Seek(0);
155
156 // Pop the position and chunk header off the stacks
157 StackIndex--;
158 int chunkpos = PositionStack[StackIndex];
159 ChunkHeader chunkh = HeaderStack[StackIndex];
160
161 // write the completed header
162 File->Seek(chunkpos,SEEK_SET);
163 if (File->Write(&chunkh,sizeof(chunkh)) != sizeof(chunkh)) {
164 return false;
165 }
166
167 // Add the total bytes written to any encompasing chunk
168 if (StackIndex != 0) {
169 HeaderStack[StackIndex-1].Add_Size(chunkh.Get_Size() + sizeof(chunkh));
170 }
171
172 // Go back to the end of the file
173 File->Seek(curpos,SEEK_SET);
174
175 return true;
176}

179/***********************************************************************************************
180 * ChunkSaveClass::Begin_Micro_Chunk -- begins a new "micro-chunk" *
181 * *
182 * Micro chunks are used to wrap individual variables. They aren't hierarchical so if you *
183 * attempt to open a micro chunk while already in one, an assert will occur. *
184 * *
185 * INPUT: *
186 * id - 8bit id *
187 * *
188 * OUTPUT: *
189 * *
190 * WARNINGS: *
191 * id is asserted to be between 0 and 255 *
192 * cannot nest micro chunks so it asserts that you are currently not in another micro-chunk *
193 * *
194 * HISTORY: *
195 * 9/3/99 GTH : Created. *
196 *=============================================================================================*/
197bool ChunkSaveClass::Begin_Micro_Chunk(uint32 id)
198{
199 assert(id < 256);
200 assert(!InMicroChunk);
201
202 // Save the current file position and chunk header
203 // for the call to End_Micro_Chunk.
204 MCHeader.Set_Type(id);
205 MCHeader.Set_Size(0);
206 MicroChunkPosition = File->Seek(0);
207
208 // Write a temporary chunk header
209 // NOTE: I'm calling the ChunkSaveClass::Write method so that the bytes for
210 // this header are tracked in the wrapping chunk. This is because micro-chunks
211 // are simply data inside the normal chunks...
212 if (Write(&MCHeader,sizeof(MCHeader)) != sizeof(MCHeader)) {
213 return false;
214 }
215
216 InMicroChunk = true;
217 return true;
218}

222 * ChunkSaveClass::End_Micro_Chunk -- close a micro-chunk *
223 * *
224 * INPUT: *
225 * *
226 * OUTPUT: *
227 * *
228 * WARNINGS: *
229 * *
230 * HISTORY: *
231 * 9/3/99 GTH : Created. *
232 *=============================================================================================*/
233bool ChunkSaveClass::End_Micro_Chunk(void)
234{
235 assert(InMicroChunk);
236
237 // Save the current position
238 int curpos = File->Seek(0);
239
240 // Seek back and write the micro chunk header
241 File->Seek(MicroChunkPosition,SEEK_SET);
242 if (File->Write(&MCHeader,sizeof(MCHeader)) != sizeof(MCHeader)) {
243 return false;
244 }
245
246 // Go back to the end of the file
247 File->Seek(curpos,SEEK_SET);
248 InMicroChunk = false;
249 return true;
250}

252/***********************************************************************************************
253 * ChunkSaveClass::Write -- Write data into the current chunk *
254 * *
255 * INPUT: *
256 * *
257 * OUTPUT: *
258 * *
259 * WARNINGS: *
260 * *
261 * HISTORY: *
262 * 07/17/1997 GH : Created. *
263 *=============================================================================================*/
264uint32 ChunkSaveClass::Write(const void * buf, uint32 nbytes)
265{
266 // If this assert hits, you mixed data and chunks within the same chunk NO NO!
267 assert(HeaderStack[StackIndex-1].Get_Sub_Chunk_Flag() == 0);
268
269 // If this assert hits, you didnt open any chunks yet
270 assert(StackIndex > 0);
271
272 // write the bytes into the file
273 if (File->Write(buf,nbytes) != (int)nbytes) return 0;
274
275 // track them in the wrapping chunk
276 HeaderStack[StackIndex-1].Add_Size(nbytes);
277
278 // track them if you are using a micro-chunk too.
279 if (InMicroChunk) {
280 assert(MCHeader.Get_Size() < 255 - nbytes); // micro chunks can only be 255 bytes
281 MCHeader.Add_Size(nbytes);
282 }
283
284 return nbytes;
285}

400/***********************************************************************************************
401 * ChunkLoadClass::Open_Chunk -- Open a chunk in the file, reads in the chunk header *
402 * *
403 * INPUT: *
404 * *
405 * OUTPUT: *
406 * *
407 * WARNINGS: *
408 * *
409 * HISTORY: *
410 * 07/17/1997 GH : Created. *
411 *=============================================================================================*/
412bool ChunkLoadClass::Open_Chunk()
413{
414 // if user didn't close any micro chunks that he opened, bad things could happen
415 assert(InMicroChunk == false);
416
417 // check for stack overflow
418 assert(StackIndex < MAX_STACK_DEPTH-1);
419
420 // if the parent chunk has been completely eaten, return false
421 if ((StackIndex > 0) && (PositionStack[StackIndex-1] == HeaderStack[StackIndex-1].Get_Size())) {
422 return false;
423 }
424
425 // read the chunk header
426 if (File->Read(&HeaderStack[StackIndex],sizeof(ChunkHeader)) != sizeof(ChunkHeader)) {
427 return false;
428 }
429
430 PositionStack[StackIndex] = 0;
431 StackIndex++;
432 return true;
433}

436/***********************************************************************************************
437 * ChunkLoadClass::Close_Chunk -- Close a chunk, seeks to the end if needed *
438 * *
439 * INPUT: *
440 * *
441 * OUTPUT: *
442 * *
443 * WARNINGS: *
444 * *
445 * HISTORY: *
446 * 07/17/1997 GH : Created. *
447 *=============================================================================================*/
448bool ChunkLoadClass::Close_Chunk()
449{
450 // if user didn't close any micro chunks that he opened, bad things could happen
451 assert(InMicroChunk == false);
452
453 // check for stack overflow
454 assert(StackIndex > 0);
455
456 int csize = HeaderStack[StackIndex-1].Get_Size();
457 int pos = PositionStack[StackIndex-1];
458
459 if (pos < csize) {
460 File->Seek(csize - pos,SEEK_CUR);
461 }
462
463 StackIndex--;
464 if (StackIndex > 0) {
465 PositionStack[StackIndex - 1] += csize + sizeof(ChunkHeader);
466 }
467
468 return true;
469}

574 * ChunkLoadClass::Close_Micro_Chunk -- closes a micro-chunk (seeks to end) *
575 * *
576 * INPUT: *
577 * *
578 * OUTPUT: *
579 * *
580 * WARNINGS: *
581 * *
582 * HISTORY: *
583 * 9/3/99 GTH : Created. *
584 *=============================================================================================*/
585bool ChunkLoadClass::Close_Micro_Chunk()
586{
587 assert(InMicroChunk);
588 InMicroChunk = false;
589
590 int csize = MCHeader.Get_Size();
591 int pos = MicroChunkPosition;
592
593 // seek the file past this micro chunk
594 if (pos < csize) {
595
596 File->Seek(csize - pos,SEEK_CUR);
597
598 // update the tracking variables for where we are in the normal chunk.
599 if (StackIndex > 0) {
600 PositionStack[StackIndex-1] += csize - pos;
601 }
602 }
603
604 return true;
605}

649// Seek over nbytes in the stream
650uint32 ChunkLoadClass::Seek(uint32 nbytes)
651{
652 assert(StackIndex >= 1);
653
654 // Don't seek if we would go past the end of the current chunk
655 if (PositionStack[StackIndex-1] + nbytes > (int)HeaderStack[StackIndex-1].Get_Size()) {
656 return 0;
657 }
658
659 // Don't read if we are in a micro chunk and would go past the end of it
660 if (InMicroChunk && MicroChunkPosition + nbytes > MCHeader.Get_Size()) {
661 return 0;
662 }
663
664 uint32 curpos=File->Tell();
665 if (File->Seek(nbytes,SEEK_CUR)-curpos != (int)nbytes) {
666 return 0;
667 }
668
669 // Update our position in the chunk
670 PositionStack[StackIndex-1] += nbytes;
671
672 // Update our position in the micro chunk if we are in one
673 if (InMicroChunk) {
674 MicroChunkPosition += nbytes;
675 }
676
677 return nbytes;
678}
