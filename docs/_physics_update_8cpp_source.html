81static Real angleBetweenVectors(const Coord3D& inCurDir, const Coord3D& inGoalDir)
82{
83 Vector3 curDir;
84 curDir.X = inCurDir.x;
85 curDir.Y = inCurDir.y;
86 curDir.Z = inCurDir.z;
87 curDir.Normalize();
88
89 Vector3 goalDir;
90 goalDir.X = inGoalDir.x;
91 goalDir.Y = inGoalDir.y;
92 goalDir.Z = inGoalDir.z;
93 goalDir.Normalize();
94
95 // dot of two unit vectors is cos of angle between them.
96 Real cosine = Vector3::Dot_Product(curDir, goalDir);
97
98 // bound it in case of numerical error
99 Real angleBetween = (Real)ACos(clamp(-1.0f, cosine, 1.0f));
100
101 return angleBetween;
102}

105static Real heightToSpeed(Real height)
106{
107 // don't bother trying to remember how far we've fallen; instead,
108 // back-calc it from our speed & gravity... v = sqrt(2*g*h)
109 return sqrt(fabs(2.0f * TheGlobalData->m_gravity * height));
110}

148{
149 // don't bother trying to remember how far we've fallen; instead,
150 // back-calc it from our speed & gravity... v = sqrt(2*g*h)
151 Real height = INI::scanReal(ini->getNextToken());
152 *(Real *)store = heightToSpeed(height);

322void PhysicsBehavior::applyForce( const Coord3D *force )
323{
324 DEBUG_ASSERTCRASH(!(_isnan(force->x) || _isnan(force->y) || _isnan(force->z)), ("PhysicsBehavior::applyForce force NAN!\n"));
325 if (_isnan(force->x) || _isnan(force->y) || _isnan(force->z)) {
326 return;
327 }
328 // F = ma --> a = F/m (divide force by mass)
329 Real mass = getMass();
330 Coord3D modForce = *force;
331 if (isMotive())
332 {
333 const Coord3D *dir = getObject()->getUnitDirectionVector2D();
334 // Only accept the lateral acceleration.
335 Real lateralDot = force->x * (-dir->y) + force->y * dir->x;
336 modForce.x = lateralDot * -dir->y;
337 modForce.y = lateralDot * dir->x;
338 }
339
340 Real massInv = 1.0f / mass;
341 m_accel.x += modForce.x * massInv;
342 m_accel.y += modForce.y * massInv;
343 m_accel.z += modForce.z * massInv;
344
345 //DEBUG_ASSERTCRASH(!(_isnan(m_accel.x) || _isnan(m_accel.y) || _isnan(m_accel.z)), ("PhysicsBehavior::applyForce accel NAN!\n"));
346 //DEBUG_ASSERTCRASH(!(_isnan(m_vel.x) || _isnan(m_vel.y) || _isnan(m_vel.z)), ("PhysicsBehavior::applyForce vel NAN!\n"));
347 //DEBUG_ASSERTCRASH(fabs(force->z) < 3, ("unlikely z-force"));
348#ifdef SLEEPY_PHYSICS
349 if (getFlag(IS_IN_UPDATE))
350 {
351 // we're applying a force from inside our own update (probably for a bounce or friction).
352 // just do nothing, since update will calculate the correct sleep behavior at the end.
353 }
354 else
355 {
356 // when a force is applied by an external module, we must wake up, even if the force is zero.
357 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
358 }
359#endif
360}

379void PhysicsBehavior::applyRandomRotation()
380{
381 // Ignore any pitch, roll & yaw rotation if behavior is stick to ground
382 if (getFlag(STICK_TO_GROUND)) return;
383
384 // Set bounce to true for a while until the unit is complete bouncing

386
387 Real randomModifier;
388

394
395 randomModifier = GameLogicRandomValue(-1.0f, 1.0f);
396 m_rollRate += getPhysicsBehaviorModuleData()->m_shockMaxRoll * randomModifier;
397
398#ifdef SLEEPY_PHYSICS
399 if (getFlag(IS_IN_UPDATE))
400 {
401 // we're applying a force from inside our own update (probably for a bounce or friction).
402 // just do nothing, since update will calculate the correct sleep behavior at the end.
403 }

405 {
406 // when a force is applied by an external module, we must wake up, even if the force is zero.
407 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
408 }
409#endif

450//-------------------------------------------------------------------------------------------------
451void PhysicsBehavior::applyFrictionalForces()
452{
453 //Are we a plane that is taxiing on a deck with a height offset?
454 Bool deckTaxiing = getObject()->testStatus( OBJECT_STATUS_DECK_HEIGHT_OFFSET )
455 && getObject()->getAI()
456 && getObject()->getAI()->getCurLocomotorSetType() == LOCOMOTORSET_TAXIING;
457
458 if (getFlag(APPLY_FRICTION2D_WHEN_AIRBORNE) || !getObject()->isSignificantlyAboveTerrain() || deckTaxiing )
459 {
460 applyYPRDamping(1.0f - DEFAULT_LATERAL_FRICTION);
461
462 if (m_vel.x || m_vel.y)
463 {
464 const Coord3D *dir = getObject()->getUnitDirectionVector2D();
465 Real mass = getMass();
466
467 Real lateralDot = m_vel.x * (-dir->y) + m_vel.y * dir->x;
468 Real lateralVel_x = lateralDot * -dir->y;
469 Real lateralVel_y = lateralDot * dir->x;
470
471 Real lf = mass * getLateralFriction();
472
473 Coord3D accel;
474 accel.x = -(lf * lateralVel_x);
475 accel.y = -(lf * lateralVel_y);
476 accel.z = 0.0f;
477
478 if (!isMotive())
479 {
480 Real forwardDot = m_vel.x * dir->x + m_vel.y * dir->y;
481 Real forwardVel_x = forwardDot * dir->x;
482 Real forwardVel_y = forwardDot * dir->y;
483 Real ff = mass * getForwardFriction();
484 accel.x += -(ff * forwardVel_x);
485 accel.y += -(ff * forwardVel_y);
486 }
487 applyForce(&accel);
488 }
489 }
490 else
491 {
492 Real aerodynamics = -getAerodynamicFriction(); // negated!
493
494 // Air resistance is proportional to velocity in the opposite direction
495 m_accel.x += m_vel.x * aerodynamics;
496 m_accel.y += m_vel.y * aerodynamics;
497 m_accel.z += m_vel.z * aerodynamics;
498
499 applyYPRDamping(1.0f + aerodynamics); // since aero is negated, this results in 1.0-getAerodynamicFriction()
500 }
501}

504//-------------------------------------------------------------------------------------------------
505Bool PhysicsBehavior::handleBounce(Real oldZ, Real newZ, Real groundZ, Coord3D* bounceForce)
506{
507 if (getFlag(ALLOW_BOUNCE) && newZ <= groundZ)
508 {
509 const Real MIN_STIFF = 0.01f;
510 const Real MAX_STIFF = 0.99f;
511 Real stiffness = TheGlobalData->m_groundStiffness;
512 if (stiffness < MIN_STIFF) stiffness = MIN_STIFF;
513 if (stiffness > MAX_STIFF) stiffness = MAX_STIFF;
514
515 Real desiredAccelZ = 0.0f;
516 Real vz = getVelocity()->z;
517 if (oldZ > groundZ && vz < 0.0f)
518 {
519 desiredAccelZ = fabs(vz) * stiffness;
520 }
521
522 bounceForce->x = 0.0f;
523 bounceForce->y = 0.0f;
524 bounceForce->z = getMass() * desiredAccelZ;
525
526 const Real DAMPING = 0.7f;
527 applyYPRDamping(DAMPING);
528
529 if (vz < 0.0f)
530 {
531 Vector3 zvec = getObject()->getTransformMatrix()->Get_Z_Vector();
532 const Real rollAngle = (zvec.Z > 0) ? 0 : PI;
533 // don't flip both pitch and roll... we'll "flip" twice.
534 const Real pitchAngle = 0;
535 Real yawAngle = getObject()->getTransformMatrix()->Get_Z_Rotation();
536 setAngles(yawAngle, pitchAngle, rollAngle);
537 }
538
539 if(bounceForce->z > 0.0f)
540 {
541 testStunnedUnitForDestruction();
542 return true;
543 }
544 else
545 {
546 setAllowBouncing(m_originalAllowBounce);
547 return false;
548 }
549 }
550 else
551 {
552 bounceForce->zero();
553 return false;
554 }
555}

626DECLARE_PERF_TIMER(PhysicsBehavior)
627UpdateSleepTime PhysicsBehavior::update()
628{
629 USE_PERF_TIMER(PhysicsBehavior)
630
631 Object* obj = getObject();
632 const PhysicsBehaviorModuleData* d = getPhysicsBehaviorModuleData();
633 Bool airborneAtStart = obj->isAboveTerrain();
634 Real activeVelZ = 0;
635 Coord3D bounceForce;
636 Bool gotBounceForce = false;
637
638 DEBUG_ASSERTCRASH(!getFlag(IS_IN_UPDATE), ("impossible"));
639 setFlag(IS_IN_UPDATE, true);
640
641 if (!getFlag(UPDATE_EVER_RUN))
642 {
643 // set the flag so that we don't get bogus "collisions" on the first frame.
644 setFlag(WAS_AIRBORNE_LAST_FRAME, airborneAtStart);
645 }
646
647 Coord3D prevPos = *obj->getPosition();
648 m_prevAccel = m_accel;
649
650 if (!obj->isDisabledByType(DISABLED_HELD))
651 {
652 Matrix3D mtx = *obj->getTransformMatrix();
653
654 applyGravitationalForces();
655 applyFrictionalForces();
656
657 // integrate acceleration into velocity
658 m_vel.x += m_accel.x;
659 m_vel.y += m_accel.y;
660 m_vel.z += m_accel.z;
661
662 // when vel gets tiny, just clamp to zero
663 const Real THRESH = 0.001f;
664 if (fabsf(m_vel.x) < THRESH) m_vel.x = 0.0f;
665 if (fabsf(m_vel.y) < THRESH) m_vel.y = 0.0f;
666 if (fabsf(m_vel.z) < THRESH) m_vel.z = 0.0f;
667
668 m_velMag = INVALID_VEL_MAG;
669
670 Real oldPosZ = mtx.Get_Z_Translation();
671
672 // integrate velocity into position
673 if (obj->testStatus(OBJECT_STATUS_BRAKING))
674 {
675 // Don't update position if the locomotor is braking.
676 if (!obj->isKindOf(KINDOF_PROJECTILE))
677 {
678 // Things other than projectiles don't cheat in z. jba.
679 mtx.Adjust_Z_Translation(m_vel.z);
680 }
681 }
682 else
683 {
684 mtx.Adjust_X_Translation(m_vel.x);
685 mtx.Adjust_Y_Translation(m_vel.y);
686 mtx.Adjust_Z_Translation(m_vel.z);
687 }
688
689 if (_isnan(mtx.Get_X_Translation()) || _isnan(mtx.Get_Y_Translation()) ||
690 _isnan(mtx.Get_Z_Translation())) {
691 DEBUG_CRASH(("Object position is NAN, deleting."));
692 TheGameLogic->destroyObject(obj);
693 }
694
695 // Check when to clear the stunned status
696 if (getIsStunned())
697 {
698 if ( (fabs(m_vel.x) < STUN_RELIEF_EPSILON &&
699 fabs(m_vel.y) < STUN_RELIEF_EPSILON &&
700 fabs(m_vel.z) < STUN_RELIEF_EPSILON)
701 ||
702 obj->isSignificantlyAboveTerrain() == FALSE )
703 {
704 setStunned(false);
705 getObject()->clearModelConditionState(MODELCONDITION_STUNNED);
706 }
707
708 }
709
710 if (getFlag(HAS_PITCHROLLYAW))
711 {
712
713 /*
714 You may be tempted to do something like this:
715
716 Real rollAngle = -mtx.Get_X_Rotation();
717 Real pitchAngle = mtx.Get_Y_Rotation();
718 Real yawAngle = mtx.Get_Z_Rotation();
719 // do stuff to angles, then rebuild the mtx with 'em
720
721 You must resist this temptation, because your code will be wrong!
722
723 The problem is that you can't use these calls to later reconstruct
724 the matrix... because doing such a thing is highly order-dependent,
725 and furthermore, you'd have to use Euler angles (Not the Get_?_Rotation
726 calls) to be able to reconstruct 'em, and that's too slow to do for
727 every object every frame.
728
729 The one exception is that it is OK to use Get_Z_Rotation() to get
730 the yaw angle.
731 */
732
733 // only update the position if we are not HELD
734 // (otherwise, slowdeath sinking into ground won't work)
735 Real yawRateToUse = m_yawRate * d->m_pitchRollYawFactor;
736 Real pitchRateToUse = m_pitchRate * d->m_pitchRollYawFactor;
737 Real rollRateToUse = m_rollRate * d->m_pitchRollYawFactor;
738
739 // With a center of mass listing, pitchRate needs to dampen towards straight down/straight up
740 Real offset = getCenterOfMassOffset();
741
742 // Magnitude sets initial rate, here we care about sign
743 if (offset != 0.0f)
744 {
745 Vector3 xvec = mtx.Get_X_Vector();
746 Real xy = sqrtf(sqr(xvec.X) + sqr(xvec.Y));
747 Real pitchAngle = atan2(xvec.Z, xy);
748 Real remainingAngle = (offset > 0) ? ((PI/2) - pitchAngle) : (-(PI/2) + pitchAngle);
749 Real s = Sin(remainingAngle);
750 pitchRateToUse *= s;
751 }
752
753 // update rotation
755 // note, we DON'T want to Pre-rotate (either inplace or not),
756 // since we want to add our mods to the existing matrix.
757 mtx.Rotate_X(rollRateToUse);
758 mtx.Rotate_Y(pitchRateToUse);
759 mtx.Rotate_Z(yawRateToUse);
760 }
761
762 // do not allow object to pass through the ground
763 Real groundZ = TheTerrainLogic->getLayerHeight(mtx.Get_X_Translation(), mtx.Get_Y_Translation(), obj->getLayer());
764 if( obj->getStatusBits().test( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
765 {
766 groundZ += obj->getCarrierDeckHeight();
767 }
768 gotBounceForce = handleBounce(oldPosZ, mtx.Get_Z_Translation(), groundZ, &bounceForce);
769
770 // remember our z-vel prior to doing ground-slam adjustment
771 activeVelZ = m_vel.z;
772 if (mtx.Get_Z_Translation() <= groundZ)
773 {
774 // Note - when vehicles are going down a slope, they will maintain a small negative
775 // z velocity as they go down. So don't slam it to 0 if they aren't slamming into the
776 // ground.
777 Real dz = groundZ - mtx.Get_Z_Translation(); // Our excess z velocity.
778 m_vel.z += dz; // Remove the excess z velocity.
779 if (m_vel.z > 0.0f)
780 m_vel.z = 0.0f;
781
782 m_velMag = INVALID_VEL_MAG;
783
784 mtx.Set_Z_Translation(groundZ);
785
786 // this flag is ALWAYS cleared once we hit the ground.
787 setFlag(ALLOW_TO_FALL, false);
788
789 // When a stunned object hits the ground the first time, chage it's model state from stunned flailing to just stunned.
790 if (getFlag(IS_STUNNED))
791 {
792 obj->clearModelConditionState(MODELCONDITION_STUNNED_FLAILING);
793 obj->setModelConditionState(MODELCONDITION_STUNNED);
794 }
795 }
796 else if (mtx.Get_Z_Translation() > groundZ)
797 {
798 if (getFlag(IS_IN_FREEFALL))
799 {
800 obj->setDisabled(DISABLED_FREEFALL);
801 obj->setModelConditionState(MODELCONDITION_FREEFALL);
802 }
803 else if (getFlag(STICK_TO_GROUND) && !getFlag(ALLOW_TO_FALL))
804 {
805 mtx.Set_Z_Translation(groundZ);
806 }
807 }
808
809 if (gotBounceForce)
810 {
811 // Right the object after the bounce since the pitch and roll may have been affected
812 Real yawAngle = getObject()->getTransformMatrix()->Get_Z_Rotation();
813 setAngles(yawAngle, 0.0f, 0.0f);
814
815 // Set the translation of the after bounce matrix to the one calculated above
816 Matrix3D afterBounceMatrix = *getObject()->getTransformMatrix();
817 afterBounceMatrix.Set_Translation(mtx.Get_Translation());
818
819 // Set the result of the after bounce matrix as the object's final matrix
820 obj->setTransformMatrix(&afterBounceMatrix);
821 }
822 else
823 {
824 obj->setTransformMatrix(&mtx);
825 }
826 } // if not held
827
828 // reset the acceleration for accumulation next frame
829 m_accel.zero();
830
831 // clear overlap object, which will be set by PhysicsCollide later
832 m_previousOverlap = m_currentOverlap;
833 m_currentOverlap = INVALID_ID;
834
835 if (gotBounceForce && getFlag(ALLOW_BOUNCE))
836 {
837 applyForce(&bounceForce);
838 }
839
840 Bool airborneAtEnd = obj->isAboveTerrain();
841
842 // it's not good enough to check for airborne being different between
843 // the start and end of this func... we have to compare since last frame,
844 // since (if we're held by a parachute, for instance) we might have been
845 // moved by other bits of code!
846 if (getFlag(WAS_AIRBORNE_LAST_FRAME) && !airborneAtEnd && !getFlag(IMMUNE_TO_FALLING_DAMAGE))
847 {
848 doBounceSound(prevPos);
849
850 // the normal always points straight down, though we could
851 // get the alignWithTerrain() normal if it proves interesting
852 Coord3D normal;
853 normal.x = normal.y = 0.0f;
854 normal.z = -1.0f;
855 obj->onCollide(NULL, obj->getPosition(), &normal);
856
857 //
858 // don't bother trying to remember how far we've fallen; instead,
859 // we back-calc it from our speed & gravity... v = sqrt(2*g*h).
860 // (note that m_minFallSpeedForDamage is always POSITIVE.)
861 //
862 // also note: since projectiles are immune to falling damage, don't
863 // even bother doing this check here.
864 //
865 Real netSpeed = -activeVelZ - d->m_minFallSpeedForDamage;
866
867 if (netSpeed > 0.0f && m_pui == NULL)
868 {
869 // only apply force if it's a pretty steep fall, so that things
870 // going down hills don't injure themselves (unless the hill is really steep)
871 const Real MIN_ANGLE_TAN = 3.0f; // roughly 71 degrees
872 const Real TINY_DELTA = 0.01f;
873 if ((fabs(m_vel.x) <= TINY_DELTA || fabs(activeVelZ / m_vel.x) >= MIN_ANGLE_TAN) &&
874 (fabs(m_vel.y) <= TINY_DELTA || fabs(activeVelZ / m_vel.y) >= MIN_ANGLE_TAN))
875 {
876 Real damageAmt = netSpeed * getMass() * d->m_fallHeightDamageFactor;
877
878 DamageInfo damageInfo;
879 damageInfo.in.m_damageType = DAMAGE_FALLING;
880 damageInfo.in.m_deathType = DEATH_SPLATTED;
881 damageInfo.in.m_sourceID = obj->getID();
882 damageInfo.in.m_amount = damageAmt;
883 damageInfo.in.m_shockWaveAmount = 0.0f;
884 obj->attemptDamage( &damageInfo );
885 //DEBUG_LOG(("Dealing %f (%f %f) points of falling damage to %s!\n",damageAmt,damageInfo.out.m_actualDamageDealt, damageInfo.out.m_actualDamageClipped,obj->getTemplate()->getName().str()));
886
887 // if this killed us, add SPLATTED to get a cool death.
888 if (obj->isEffectivelyDead())
889 {
890 obj->setModelConditionState(MODELCONDITION_SPLATTED);
891 }
892 }
893 }
894 }
895
896 if (!airborneAtEnd)
897 {
898 // just in case.
899 setFlag(IS_IN_FREEFALL, false);
900 if (obj->isDisabledByType(DISABLED_FREEFALL))
901 obj->clearDisabled(DISABLED_FREEFALL);
902 obj->clearModelConditionState(MODELCONDITION_FREEFALL);
903 }
904
905
906 // If we are effectively dead, we shouldn't recall kill.
907 if (d->m_killWhenRestingOnGround && !airborneAtEnd && isVerySmall3D(m_vel))
908 {
909 if( !obj->isKindOf( KINDOF_DRONE ) || obj->isEffectivelyDead() || obj->isDisabledByType( DISABLED_UNMANNED ) )
910 {
911 //Must be one of the following cases in order to splat:
912 //1) Not a drone
913 //2) Dead drone
914 //3) Unmanned drone
915 obj->kill();
916 }
917 }
918
919 setFlag(UPDATE_EVER_RUN, true);
920 setFlag(WAS_AIRBORNE_LAST_FRAME, airborneAtEnd);
921
922 setFlag(IS_IN_UPDATE, false);
923 return calcSleepTime();
924}

963Real PhysicsBehavior::getForwardSpeed2D() const
964{
965 const Coord3D *dir = getObject()->getUnitDirectionVector2D();
966
967 Real vx = m_vel.x * dir->x;
968 Real vy = m_vel.y * dir->y;
969
970 Real dot = vx + vy;
971
972 Real speedSquared = vx*vx + vy*vy;
973// DEBUG_ASSERTCRASH( speedSquared != 0, ("zero speedSquared will overflow sqrtf()!") );// lorenzen... sanity check
974
975 Real speed = (Real)sqrtf( speedSquared );
976
977 if (dot >= 0.0f)
978 return speed;
979
980 return -speed;
981}

989{

993 Real vy = m_vel.y * dir.Y;
994 Real vz = m_vel.z * dir.Z;
995
996 Real dot = vx + vy + vz;
997
998 Real speed = (Real)sqrtf( vx*vx + vy*vy + vz*vz );
999
1000 if (dot >= 0.0f)
1001 return speed;
1002
1003 return -speed;
1004}

1154//-------------------------------------------------------------------------------------------------
1164//DECLARE_PERF_TIMER(PhysicsBehavioronCollide)
1165void PhysicsBehavior::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
1166{
1167 //USE_PERF_TIMER(PhysicsBehavioronCollide)
1168 if (m_pui != NULL)
1169 {
1170 // projectiles always get a chance to handle their own collisions, and not go thru here
1171 if (m_pui->projectileHandleCollision(other))
1172 return;
1173 }
1174
1175 Object *obj = getObject();
1176
1177 Object* objContainedBy = obj->getContainedBy();
1178
1179 // Note that other == null means "collide with ground"
1180 if (other == NULL)
1181 {
1182 // if we are in a container, tell the container we collided with the ground.
1183 // (handy for parachutes.)
1184 if (objContainedBy)
1185 {
1186 objContainedBy->onCollide(other, loc, normal);
1187 }
1188 return;
1189 }
1190
1191 // if we are containing this object, ignore collisions with it.
1192 Object* otherContainedBy = other->getContainedBy();
1193 if (otherContainedBy == obj || objContainedBy == other)
1194 {
1195 return;
1196 }
1197
1198 // if we're both on parachutes, we never collide with each other.
1200 if (obj->testStatus(OBJECT_STATUS_PARACHUTING) && other->testStatus(OBJECT_STATUS_PARACHUTING))
1201 {
1202 return;
1203 }
1204
1205 // ignore collisions with our "ignore" thingie, if any (and vice versa)
1206 AIUpdateInterface* ai = obj->getAIUpdateInterface();
1207 if (ai != NULL && ai->getIgnoredObstacleID() == other->getID())
1208 {
1210 //Before we return, check for a very special case of an infantry colliding with an unmanned vehicle.
1211 //If this is the case, it'll become its new pilot!
1212 if( obj->isKindOf( KINDOF_INFANTRY ) && other->isDisabledByType( DISABLED_UNMANNED ) )
1213 {
1214 //This is in fact the case, and we are doing it here because it applies to all infantry in any unmanned vehicle.
1215 //This could be done via a special/new module, but doing it here doesn't require a new module update to every infantry.
1216 other->clearDisabled( DISABLED_UNMANNED );
1217
1218 //We need to be able to test whether an object on a team has been captured, so set here that this object
1219 //was captured.
1220 other->setCaptured(true);
1221
1222 other->defect( obj->getTeam(), 0 );
1223 //other->setTeam( obj->getTeam() );
1224
1225 //In order to make things easier for the designers, we are going to transfer the name
1226 //of the infantry to the vehicle... so the designer can control the vehicle with their scripts.
1227 TheScriptEngine->transferObjectName( obj->getName(), other );
1228
1229 TheGameLogic->destroyObject( obj );
1230 }
1231 return;
1232 }
1233
1234 AIUpdateInterface* aiOther = other->getAIUpdateInterface();
1235 if (aiOther != NULL && aiOther->getIgnoredObstacleID() == obj->getID())
1236 {
1237 return;
1238 }
1239
1240 if (isIgnoringCollisionsWith(other->getID()))
1241 {
1242 return;
1243 }
1244
1245 Bool immobile = obj->isKindOf( KINDOF_IMMOBILE );
1246 Bool otherImmobile = other->isKindOf( KINDOF_IMMOBILE );
1247
1248 PhysicsBehavior* otherPhysics = other->getPhysics();
1249 if (otherPhysics)
1250 {
1251 if (otherPhysics->isIgnoringCollisionsWith(obj->getID()))
1252 {
1253 return;
1254 }
1255 }
1256 else
1257 {
1258 // if the object we have collided with has no physics, it is insubstantial
1259 // (exception: if it's immobile.)
1260 if (!otherImmobile)
1261 {
1262 return;
1263 }
1264 }
1265
1266
1267 if (checkForOverlapCollision(other))
1268 {
1269 // we should overlap them, rather than bounce them, so punt here.
1270 // (note that we do this prior to checking for Physics update,
1271 // because Overlap doesn't require 'other' to have Physics...)
1272 return;
1273 }
1274
1275 Real mass = getMass();
1276 if (immobile)
1277 mass = 999999.0f;
1278
1279 if (ai)
1280 {
1281 // AI objects move under their own initiative, not by getting bounced. jba.
1282 // unless they are dead & colliding with something immobile. (srj)
1283 // or parachuting and colliding with something immobile. (srj)
1284 if (!((obj->isEffectivelyDead() || obj->testStatus(OBJECT_STATUS_PARACHUTING)) && otherImmobile))
1285 {
1286 Bool doForce = ai->processCollision(this, other);
1287 if (!doForce)
1288 return;
1289 }
1290
1291 }
1292
1293 Coord3D usCenter, themCenter;
1294 obj->getGeometryInfo().getCenterPosition(*obj->getPosition(), usCenter);
1295 other->getGeometryInfo().getCenterPosition(*other->getPosition(), themCenter);
1296
1297 Coord3D delta;
1298 delta.x = themCenter.x - usCenter.x;
1299 delta.y = themCenter.y - usCenter.y;
1300 delta.z = themCenter.z - usCenter.z;
1301
1302 Real distSqr, usRadius, themRadius;
1303 if (obj->isAboveTerrain())
1304 {
1305 // do 3d testing.
1306 usRadius = obj->getGeometryInfo().getBoundingSphereRadius();
1307 themRadius = other->getGeometryInfo().getBoundingSphereRadius();
1308 distSqr = sqr(delta.x) + sqr(delta.y) + sqr(delta.z);
1309 }
1310 else
1311 {
1312 // do 2d testing.
1313 usRadius = obj->getGeometryInfo().getBoundingCircleRadius();
1314 themRadius = other->getGeometryInfo().getBoundingCircleRadius();
1315 distSqr = sqr(delta.x) + sqr(delta.y);
1316 delta.z = 0;
1317 }
1318
1319 if (distSqr > sqr(usRadius + themRadius))
1320 {
1321 // We don't overlap at all. How did we get here?
1322 return;
1323 }
1324
1325
1326 m_lastCollidee = other->getID();
1327
1328 Real dist = sqrtf(distSqr);
1329 Real overlap = usRadius + themRadius - dist;
1330
1331 // if objects are coincident, dist is zero, so force would be infinite -- clearly
1332 // not what we want. so just cap it here for now. (srj)
1333 if (dist < 1.0f)
1334 dist = 1.0f;
1335
1336 if (getAllowCollideForce())
1337 {
1338 Real factor;
1339 Coord3D force;
1340 if (otherImmobile && !obj->isDestroyed())
1341 {
1342 if (obj->testStatus(OBJECT_STATUS_PARACHUTING))
1343 {
1344 // don't let us intersect buildings. cheat. applying a force won't work
1345 // cuz we are usually braking. jam it.
1346 Object* objToBounce = obj;
1347 while (objToBounce->getContainedBy() != NULL)
1348 objToBounce = objToBounce->getContainedBy();
1349
1350 Real bounceOutDist = usRadius * 0.1f;
1351 Coord3D tmp = *objToBounce->getPosition();
1352 tmp.x -= bounceOutDist * delta.x / dist;
1353 tmp.y -= bounceOutDist * delta.y / dist;
1354 objToBounce->setPosition(&tmp);
1355
1356 objToBounce->getPhysics()->scrubVelocity2D(0);
1357 return;
1358 }
1359
1360 // if other is immobile, we must apply enough force to at least stop our motion,
1361 // and preferably negate it, otherwise we will pass thru the object. so cheat,
1362 // and act like we're bouncing off the ground.
1363
1364 const Real MIN_STIFF = 0.01f;
1365 const Real MAX_STIFF = 0.99f;
