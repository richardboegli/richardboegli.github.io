43void INI::parseMusicTrackDefinition( INI* ini )
44{
45 AsciiString name;
46 AudioEventInfo *track;
47
48 // read the name
49 const char* c = ini->getNextToken();
50 name.set( c );
51
52 track = TheAudio->newAudioEventInfo( name );
53 if (!track) {
54 return;
55 }
56
57 AudioEventInfo *defaultInfo = TheAudio->findAudioEventInfo("DefaultMusicTrack");
58 if (defaultInfo) {
59 (*track) = (*defaultInfo);
60 TheAudio->addTrackName( name );
61 }
62
63 track->m_audioName = name;
64 track->m_soundType = AT_Music;
65
66 // parse the ini definition
67 ini->initFromINI( track, track->getFieldParse() );
68} // end parseMusicTrackDefinition

155static void parseDelay( INI* ini, void *instance, void *store, const void* /*userData*/ )
156{
157 AudioEventInfo *attribs = (AudioEventInfo*) store;
158
159 Int min = ini->scanInt(ini->getNextToken());
160 Int max = ini->scanInt(ini->getNextToken());
161
162 DEBUG_ASSERTCRASH( min >= 0 && max >= min, ("Bad delay values for audio event %s", attribs->m_audioName.str()));
163 attribs->m_delayMax = max;
164 attribs->m_delayMin = min;
165}
166
167//-------------------------------------------------------------------------------------------------
168static void parsePitchShift( INI* ini, void *instance, void *store, const void* /*userData*/ )
169{
170 AudioEventInfo *attribs = (AudioEventInfo*) store;
171
172 Real min = ini->scanReal(ini->getNextToken());
173 Real max = ini->scanReal(ini->getNextToken());
174
175 DEBUG_ASSERTCRASH( min > -100 && max >= min, ("Bad pitch shift values for audio event %s", attribs->m_audioName.str()));
176 attribs->m_pitchShiftMin = 1.0f + min/100;
177 attribs->m_pitchShiftMax = 1.0f + max/100;
178}
