49static void RTSFlagsToOSFlags(UnsignedInt buttonFlags, UnsignedInt otherFlags, UnsignedInt& outWindowsFlags)
50{
51 outWindowsFlags = 0;
52
53 if (BitTest(buttonFlags, OSDBT_OK)) {
54 outWindowsFlags |= MB_OK;
55 }
56
57 if (BitTest(buttonFlags, OSDBT_CANCEL)) {
58 outWindowsFlags |= MB_OKCANCEL;
59 }
60
61 //-----------------------------------------------------------------------------------------------
62 if (BitTest(otherFlags, OSDOF_SYSTEMMODAL)) {
63 outWindowsFlags |= MB_SYSTEMMODAL;
64 }
65
66 if (BitTest(otherFlags, OSDOF_APPLICATIONMODAL)) {
67 outWindowsFlags |= MB_APPLMODAL;
68 }
69
70 if (BitTest(otherFlags, OSDOF_TASKMODAL)) {
71 outWindowsFlags |= MB_TASKMODAL;
72 }
73
74 if (BitTest(otherFlags, OSDOF_EXCLAMATIONICON)) {
75 outWindowsFlags |= MB_ICONEXCLAMATION;
76 }
77
78 if (BitTest(otherFlags, OSDOF_INFORMATIONICON)) {
79 outWindowsFlags |= MB_ICONINFORMATION;
80 }
81
82 if (BitTest(otherFlags, OSDOF_ERRORICON)) {
83 outWindowsFlags |= MB_ICONERROR;
84 }
85
86 if (BitTest(otherFlags, OSDOF_STOPICON)) {
87 outWindowsFlags |= MB_ICONSTOP;
88 }
89
90}

93OSDisplayButtonType OSDisplayWarningBox(AsciiString p, AsciiString m, UnsignedInt buttonFlags, UnsignedInt otherFlags)
94{
95 if (!TheGameText) {
96 return OSDBT_ERROR;
97 }
98
99 UnicodeString promptStr = TheGameText->fetch(p);
100 UnicodeString mesgStr = TheGameText->fetch(m);
101
102 UnsignedInt windowsOptionsFlags = 0;
103 RTSFlagsToOSFlags(buttonFlags, otherFlags, windowsOptionsFlags);
104
105 // @todo Make this return more than just ok/cancel - jkmcd
106 // (we need a function to translate back the other way.)
107 Int returnResult = 0;
108 if (TheSystemIsUnicode)
109 {
110 returnResult = ::MessageBoxW(NULL, mesgStr.str(), promptStr.str(), windowsOptionsFlags);
111 }
112 else
113 {
114 // However, if we're using the default version of the message box, we need to
115 // translate the string into an AsciiString
116 AsciiString promptA, mesgA;
117 promptA.translate(promptStr);
118 mesgA.translate(mesgStr);
119 //Make sure main window is not TOP_MOST
120 ::SetWindowPos(ApplicationHWnd, HWND_NOTOPMOST, 0, 0, 0, 0, SWP_NOSIZE | SWP_NOMOVE);
121 returnResult = ::MessageBoxA(NULL, mesgA.str(), promptA.str(), windowsOptionsFlags);
122 }
123
124 if (returnResult == IDOK) {
125 return OSDBT_OK;
126 }
127
128 return OSDBT_CANCEL;
129}
