69void HiveStructureBody::attemptDamage( DamageInfo *damageInfo )
70{
71 const HiveStructureBodyModuleData *data = getHiveStructureBodyModuleData();
72 Object *hive = getObject();
73
74 if( getDamageTypeFlag( data->m_damageTypesToPropagateToSlaves, damageInfo->in.m_damageType ) )
75 {
76
77 //We have the right type of damage types incoming to propagate to slaves. Do we have slaves?
78 SpawnBehaviorInterface *spawnInterface = hive->getSpawnBehaviorInterface();
79 if( spawnInterface )
80 {
81 //We found the spawn interface, now get some slaves!
82 Object *shooter = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
83 if( shooter )
84 {
85 Object *slave = spawnInterface->getClosestSlave( shooter->getPosition() );
86 if( slave )
87 {
88 //Propagate damage and return!
89 slave->attemptDamage( damageInfo );
90 return;
91 }
92 else if( getDamageTypeFlag( data->m_damageTypesToSwallow, damageInfo->in.m_damageType ) )
93 {
94 //no slave to give to, so eat it
95 damageInfo->out.m_actualDamageDealt = 0.0f;
96 damageInfo->out.m_actualDamageClipped = 0.0f;
97 damageInfo->out.m_noEffect = true;
98 return;
99 }
100 }
101 }
102 else if ( hive->getContain() )
103 {
104 ContainModuleInterface *contain = hive->getContain();
105 //We found the spawn interface, now get some slaves!
106 Object *shooter = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
107 if( shooter )
108 {
109 Object *rider = contain->getClosestRider( shooter->getPosition() );
110 if( rider )
111 {
112 //Propagate damage and return!
113 rider->attemptDamage( damageInfo );
114 return;
115 }
116 else if( getDamageTypeFlag( data->m_damageTypesToSwallow, damageInfo->in.m_damageType ) )
117 {
118 //no slave to give to, so eat it
119 damageInfo->out.m_actualDamageDealt = 0.0f;
120 damageInfo->out.m_actualDamageClipped = 0.0f;
121 damageInfo->out.m_noEffect = true;
122 return;
123 }
124 }
125 }
126 else
127 {
128 DEBUG_CRASH( ("%s has a HiveStructureBody module, which requires a SpawnBehavior module. Thus it is unable to propagate damage to slaves.", hive->getTemplate()->getName().str() ) );
129 }
130
131
132 }
133
134 //Nothing to propagate (either different damage type or no slaves),
135 //so damage me instead!
136 StructureBody::attemptDamage( damageInfo );
137}
