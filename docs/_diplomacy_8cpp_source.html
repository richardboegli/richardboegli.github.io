170void UpdateDiplomacyBriefingText(AsciiString newText, Bool clear)
171{
172 GameWindow *listboxSolo = TheWindowManager->winGetWindowFromId(theWindow, NAMEKEY("Diplomacy.wnd:ListboxSolo"));
173
174 if (clear)
175 {
176 theBriefingList.clear();
177 if (listboxSolo)
178 GadgetListBoxReset(listboxSolo);
179 }
180
181 if (newText.isEmpty())
182 return;
183
184 if (std::find(theBriefingList.begin(), theBriefingList.end(), newText) != theBriefingList.end())
185 return;
186
187 theBriefingList.push_back(newText);
188 if (!listboxSolo)
189 return;
190
191 UnicodeString translated = TheGameText->fetch(newText);
192
193 Int numEntries = GadgetListBoxGetNumEntries(listboxSolo);
194 GadgetListBoxAddEntryText(listboxSolo, translated, TheInGameUI->getMessageColor(numEntries%2), -1);
195}

199void ShowDiplomacy( Bool immediate )
200{
201 if (!TheInGameUI->getInputEnabled() || TheGameLogic->isIntroMoviePlaying() ||
202 TheGameLogic->isLoadingMap())
203 return;
204
205
206 if (TheInGameUI->isQuitMenuVisible())
207 return;
208
209 if (TheDisconnectMenu && TheDisconnectMenu->isScreenVisible())
210 return;
211
212 if (theWindow)
213 {
214 theWindow->winHide(FALSE);
215 theWindow->winEnable(TRUE);
216 }
217 else
218 {
219 theLayout = TheWindowManager->winCreateLayout( "Diplomacy.wnd" );
220 theWindow = theLayout->getFirstWindow();
221 theLayout->setUpdate(updateFunc);
222 theAnimateWindowManager = NEW AnimateWindowManager;
223 radioButtonInGameID = TheNameKeyGenerator->nameToKey("Diplomacy.wnd:RadioButtonInGame");
224 radioButtonBuddiesID = TheNameKeyGenerator->nameToKey("Diplomacy.wnd:RadioButtonBuddies");
225 radioButtonInGame = TheWindowManager->winGetWindowFromId(NULL, radioButtonInGameID);
226 radioButtonBuddies = TheWindowManager->winGetWindowFromId(NULL, radioButtonBuddiesID);
227 winInGameID = TheNameKeyGenerator->nameToKey("Diplomacy.wnd:InGameParent");
228 winBuddiesID = TheNameKeyGenerator->nameToKey("Diplomacy.wnd:BuddiesParent");
229 winSoloID = TheNameKeyGenerator->nameToKey("Diplomacy.wnd:SoloParent");
230 winInGame = TheWindowManager->winGetWindowFromId(NULL, winInGameID);
231 winBuddies = TheWindowManager->winGetWindowFromId(NULL, winBuddiesID);
232 winSolo = TheWindowManager->winGetWindowFromId(NULL, winSoloID);
233
234 if (!TheRecorder->isMultiplayer())
235 {
236 GameWindow *listboxSolo = TheWindowManager->winGetWindowFromId(theWindow, NAMEKEY("Diplomacy.wnd:ListboxSolo"));
237 if (listboxSolo)
238 {
239 for (BriefingList::iterator it = theBriefingList.begin(); it != theBriefingList.end(); ++it)
240 {
241 UnicodeString translated = TheGameText->fetch(*it);
242 Int numEntries = GadgetListBoxGetNumEntries(listboxSolo);
243 GadgetListBoxAddEntryText(listboxSolo, translated, TheInGameUI->getMessageColor(numEntries%2), -1);
244 }
245 }
246 }
247 }
248 theLayout->hide(FALSE);
249
250 radioButtonInGame->winHide(TRUE);
251 radioButtonBuddies->winHide(TRUE);
252 GadgetRadioSetSelection(radioButtonInGame, FALSE);
253 if (TheRecorder->isMultiplayer())
254 {
255 winInGame->winHide(FALSE);
256 winBuddies->winHide(TRUE);
257 winSolo->winHide(TRUE);
258 }
259 else
260 {
261 winInGame->winHide(TRUE);
262 winBuddies->winHide(TRUE);
263 winSolo->winHide(FALSE);
264 }
265
266 theAnimateWindowManager->reset();
267 if (!immediate && TheGlobalData->m_animateWindows)
268 theAnimateWindowManager->registerGameWindow( theWindow, WIN_ANIMATION_SLIDE_TOP, TRUE, 200 );
269
270 TheInGameUI->registerWindowLayout(theLayout);
271 grabWindowPointers();
272 PopulateInGameDiplomacyPopup();
273
274 if(TheGameSpyInfo && TheGameSpyInfo->getLocalProfileID() != 0)
275 {
276 radioButtonInGame->winHide(FALSE);
277 radioButtonBuddies->winHide(FALSE);
278 InitBuddyControls(1);
279 PopulateOldBuddyMessages();
280 updateBuddyInfo();
281 }
282
283}

384WindowMsgHandledType DiplomacySystem( GameWindow *window, UnsignedInt msg,
385 WindowMsgData mData1, WindowMsgData mData2 )
386{
387 if(BuddyControlSystem(window, msg, mData1, mData2) == MSG_HANDLED)
388 {
389 return MSG_HANDLED;
390 }
391 switch( msg )
392 {
393 //---------------------------------------------------------------------------------------------
394 case GGM_FOCUS_CHANGE:
395 {
396// Bool focus = (Bool) mData1;
397 //if (focus)
398 //TheWindowManager->winSetGrabWindow( chatTextEntry );
399 break;
400 } // end focus change
401
402 //---------------------------------------------------------------------------------------------
403 case GWM_INPUT_FOCUS:
404 {
405 // if we're given the opportunity to take the keyboard focus we must say we don't want it
406 if( mData1 == TRUE )
407 *(Bool *)mData2 = FALSE;
408
409 return MSG_HANDLED;
410 }//case GWM_INPUT_FOCUS:
411
412 //---------------------------------------------------------------------------------------------
413 case GBM_SELECTED:
414 {
415 GameWindow *control = (GameWindow *)mData1;
416 NameKeyType controlID = (NameKeyType)control->winGetWindowId();
417 static NameKeyType buttonHideID = NAMEKEY( "Diplomacy.wnd:ButtonHide" );
418 if (controlID == buttonHideID)
419 {
420 HideDiplomacy( FALSE );
421 }
422 else if( controlID == radioButtonInGameID)
423 {
424 winInGame->winHide(FALSE);
425 winBuddies->winHide(TRUE);
426 }
427 else if( controlID == radioButtonBuddiesID)
428 {
429 winInGame->winHide(TRUE);
430 winBuddies->winHide(FALSE);
431 }
432
433 for (Int i=0; i<MAX_SLOTS; ++i)
434 {
435 if (controlID == buttonMuteID[i] && slotNumInRow[i] >= 0)
436 {
437 TheGameInfo->getSlot(slotNumInRow[i])->mute(TRUE);
438 PopulateInGameDiplomacyPopup();
439 break;
440 }
441 if (controlID == buttonUnMuteID[i] && slotNumInRow[i] >= 0)
442 {
443 TheGameInfo->getSlot(slotNumInRow[i])->mute(FALSE);
444 PopulateInGameDiplomacyPopup();
445 break;
446 }
447 }
448 break;
449
450 } // end button selected
451
452 //---------------------------------------------------------------------------------------------
453 default:
454 return MSG_IGNORED;
455
456 } // end switch( msg )
457
458 return MSG_HANDLED;
459
460} // end DiplomacySystem

462void PopulateInGameDiplomacyPopup( void )
463{
464 if (!TheGameInfo)
465 return;
466
467 Int rowNum = 0;
468 for (Int slotNum=0; slotNum<MAX_SLOTS; ++slotNum)
469 {
470 const GameSlot *slot = TheGameInfo->getConstSlot(slotNum);
471 if (slot && slot->isOccupied())
472 {
473 Bool isInGame = false;
474 // Note - for skirmish, TheNetwork == NULL. jba.
475 if (TheNetwork && TheNetwork->isPlayerConnected(slotNum)) {
476 isInGame = true;
477 } else if ((TheNetwork == NULL) && slot->isHuman()) {
478 // this is a skirmish game and it is the human player.
479 isInGame = true;
480 }
481 if (slot->isAI())
482 isInGame = true;
483 AsciiString playerName;
484 playerName.format("player%d", slotNum);
485 Player *player = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
486 Bool isAlive = !TheVictoryConditions->hasSinglePlayerBeenDefeated(player);
487 Bool isObserver = player->isPlayerObserver();
488
489 if (slot->isHuman() && TheGameInfo->getLocalSlotNum() != slotNum && isInGame)
490 {
491 // show mute button
492 if (buttonMute[rowNum])
493 {
494 buttonMute[rowNum]->winHide(slot->isMuted());
495 }
496 if (buttonUnMute[rowNum])
497 {
498 buttonUnMute[rowNum]->winHide(!slot->isMuted());
499 }
500 }
501 else
502 {
503 // can't mute self, AI players, or MIA humans
504 if (buttonMute[rowNum])
505 buttonMute[rowNum]->winHide(TRUE);
506 if (buttonUnMute[rowNum])
507 buttonUnMute[rowNum]->winHide(TRUE);
508 }
509
510 Color playerColor = TheMultiplayerSettings->getColor(slot->getApparentColor())->getColor();
511 Color backColor = GameMakeColor(0, 0, 0, 255);
512 Color aliveColor = GameMakeColor(0, 255, 0, 255);
513 Color deadColor = GameMakeColor(255, 0, 0, 255);
514 Color observerInGameColor = GameMakeColor(255, 255, 255, 255);
515 Color goneColor = GameMakeColor(196, 0, 0, 255);
516 Color observerGoneColor = GameMakeColor(196, 196, 196, 255);
517
518 if (staticTextPlayer[rowNum])
519 {
520 staticTextPlayer[rowNum]->winSetEnabledTextColors( playerColor, backColor );
521 GadgetStaticTextSetText(staticTextPlayer[rowNum], slot->getName());
522 }
523 if (staticTextSide[rowNum])
524 {
525 staticTextSide[rowNum]->winSetEnabledTextColors( playerColor, backColor );
526 GadgetStaticTextSetText(staticTextSide[rowNum], slot->getApparentPlayerTemplateDisplayName() );
527 }
528 if (staticTextTeam[rowNum])
529 {
530 staticTextTeam[rowNum]->winSetEnabledTextColors( playerColor, backColor );
531 AsciiString teamStr;
532 teamStr.format("Team:%d", slot->getTeamNumber() + 1);
533 if (slot->isAI() && slot->getTeamNumber() == -1)
534 teamStr = "Team:AI";
535 GadgetStaticTextSetText(staticTextTeam[rowNum], TheGameText->fetch(teamStr) );
536 }
537 if (staticTextStatus[rowNum])
538 {
539 staticTextStatus[rowNum]->winHide(FALSE);
540 if (isInGame)
541 {
542 if (isAlive)
543 {
544 staticTextStatus[rowNum]->winSetEnabledTextColors( aliveColor, backColor );
545 GadgetStaticTextSetText(staticTextStatus[rowNum], TheGameText->fetch("GUI:PlayerAlive"));
546 }
547 else
548 {
549 if (isObserver)
550 {
551 staticTextStatus[rowNum]->winSetEnabledTextColors( observerInGameColor, backColor );
552 GadgetStaticTextSetText(staticTextStatus[rowNum], TheGameText->fetch("GUI:PlayerObserver"));
553 }
554 else
555 {
556 staticTextStatus[rowNum]->winSetEnabledTextColors( deadColor, backColor );
557 GadgetStaticTextSetText(staticTextStatus[rowNum], TheGameText->fetch("GUI:PlayerDead"));
558 }
559 }
560 }
561 else
562 {
563 // not in game
564 if (isObserver)
565 {
566 staticTextStatus[rowNum]->winSetEnabledTextColors( observerGoneColor, backColor );
567 GadgetStaticTextSetText(staticTextStatus[rowNum], TheGameText->fetch("GUI:PlayerObserverGone"));
568 }
569 else
570 {
571 staticTextStatus[rowNum]->winSetEnabledTextColors( goneColor, backColor );
572 GadgetStaticTextSetText(staticTextStatus[rowNum], TheGameText->fetch("GUI:PlayerGone"));
573 }
574 }
575 }
576
577 slotNumInRow[rowNum++] = slotNum;
578 }
579 }
580
581 while (rowNum < MAX_SLOTS)
582 {
583 slotNumInRow[rowNum] = -1;
584 if (staticTextPlayer[rowNum])
585 staticTextPlayer[rowNum]->winHide(TRUE);
586 if (staticTextSide[rowNum])
587 staticTextSide[rowNum]->winHide(TRUE);
588 if (staticTextTeam[rowNum])
589 staticTextTeam[rowNum]->winHide(TRUE);
590 if (staticTextStatus[rowNum])
591 staticTextStatus[rowNum]->winHide(TRUE);
592 if (buttonMute[rowNum])
593 buttonMute[rowNum]->winHide(TRUE);
594 if (buttonUnMute[rowNum])
595 buttonUnMute[rowNum]->winHide(TRUE);
596
597 ++rowNum;
598 }
599}
