176void DebugHintObject::initData(void)
177{
178 freeMapResources(); //free old data and ib/vb
179
180 m_indexBuffer = NEW_REF(DX8IndexBufferClass,(3));
181
182 // Fill up the IB
183 {
184 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
185 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
186 ib[0]=0;
187 ib[1]=1;
188 ib[2]=2;
189 }
190
191 m_vertexBufferTile = NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,3,DX8VertexBufferClass::USAGE_DEFAULT));
192
193 //go with a preset material for now.
194 m_vertexMaterialClass = VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
195
196 //use a multi-texture shader: (text1*diffuse)*text2.
197 m_shaderClass = ShaderClass::ShaderClass(SC_ALPHA);
198}

200void DebugHintObject::setLocAndColorAndSize(const Coord3D *loc, Int argb, Int size)
201{
202 m_myLoc = *loc;
203 m_myColor = argb;
204 m_mySize = size;
205
206 if (m_myLoc.z < 0 && TheTerrainRenderObject)
207 {
208 m_myLoc.z = TheTerrainRenderObject->getHeightMapHeight(m_myLoc.x, m_myLoc.y, NULL);
209 }
210
211 if (m_vertexBufferTile)
212 {
213 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexBufferTile);
214 VertexFormatXYZDUV1 *vb = (VertexFormatXYZDUV1*)lockVtxBuffer.Get_Vertex_Array();
215
216 Real x1 = m_mySize * 0.866; // cos(30)
217 Real y1 = m_mySize * 0.5; // sin(30)
218
219 // note, pts must go in a counterclockwise order!
220 vb[0].x = 0;
221 vb[0].y = m_mySize;
222 vb[0].z = 0;
223 vb[0].diffuse = m_myColor;
224 vb[0].u1 = 0;
225 vb[0].v1 = 0;
226
227 vb[1].x = -x1;
228 vb[1].y = -y1;
229 vb[1].z = 0;
230 vb[1].diffuse = m_myColor;
231 vb[1].u1 = 0;
232 vb[1].v1 = 0;
233
234 vb[2].x = x1;
235 vb[2].y = -y1;
236 vb[2].z = 0;
237 vb[2].diffuse = m_myColor;
238 vb[2].u1 = 0;
239 vb[2].v1 = 0;
240 }
241}

243void DebugHintObject::Render(RenderInfoClass & rinfo)
244{
245 SphereClass bounds(Vector3(m_myLoc.x, m_myLoc.y, m_myLoc.z), m_mySize);
246 if (!rinfo.Camera.Cull_Sphere(bounds))
247 {
248 DX8Wrapper::Set_Material(m_vertexMaterialClass);
249 DX8Wrapper::Set_Shader(m_shaderClass);
250 DX8Wrapper::Set_Texture(0, NULL);
251 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
252 DX8Wrapper::Set_Vertex_Buffer(m_vertexBufferTile);
253
254 Matrix3D tm(Transform);
255 Vector3 vec(m_myLoc.x, m_myLoc.y, m_myLoc.z);
256 tm.Set_Translation(vec);
257 DX8Wrapper::Set_Transform(D3DTS_WORLD, tm);
258
259 DX8Wrapper::Draw_Triangles( 0, 1, 0, 3);
260 }

311static void loadText( char *filename, GameWindow *listboxText )
312{
313 if (!listboxText)
314 return;
315 GadgetListBoxReset(listboxText);
316
317 FILE *fp;
318
319 // open the file
320 fp = fopen( filename, "r" );
321 if( fp == NULL )
322 return;
323
324 char buffer[ 1024 ];
325 UnicodeString line;
326 Color color = GameMakeColor(255, 255, 255, 255);
327 while( fgets( buffer, 1024, fp ) != NULL )
328 {
329 line.translate(buffer);
330 line.trim();
331 if (line.isEmpty())
332 line = UnicodeString(L" ");
333 GadgetListBoxAddEntryText(listboxText, line, color, -1, -1);
334 } // end while
335
336 // close the file
337 fclose( fp );
338
339} // end loadText
