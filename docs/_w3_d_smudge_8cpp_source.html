83void W3DSmudgeManager::ReAcquireResources(void)
84{
85 ReleaseResources();
86
87 SurfaceClass *surface=DX8Wrapper::_Get_DX8_Back_Buffer();
88 SurfaceClass::SurfaceDescription surface_desc;
89
90 surface->Get_Description(surface_desc);
91 REF_PTR_RELEASE(surface);
92
93 #ifdef USE_COPY_RECTS
94 m_backgroundTexture = MSGNEW("TextureClass") TextureClass(TheTacticalView->getWidth(),TheTacticalView->getHeight(),surface_desc.Format,MIP_LEVELS_1,TextureClass::POOL_DEFAULT, true);
95 #endif
96
97 m_backBufferWidth = surface_desc.Width;
98 m_backBufferHeight = surface_desc.Height;
99
100 m_indexBuffer=NEW_REF(DX8IndexBufferClass,(SMUDGE_DRAW_SIZE*4*3)); //allocate 4 triangles per smudge, each with 3 indices.
101
102 // Fill up the IB with static vertex indices that will be used for all smudges.
103 {
104 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
105 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
106 //quad of 4 triangles:
107 // 0-----3
108 // |\ /|
109 // | 4 |
110 // |/ \|
111 // 1-----2
112 Int vbCount=0;
113 for (Int i=0; i<SMUDGE_DRAW_SIZE; i++)
114 {
115 //Top
116 ib[0]=vbCount;
117 ib[1]=vbCount+4;
118 ib[2]=vbCount+3;
119 //Right
120 ib[3]=vbCount+3;
121 ib[4]=vbCount+4;
122 ib[5]=vbCount+2;
123 //Bottom
124 ib[6]=vbCount+2;
125 ib[7]=vbCount+4;
126 ib[8]=vbCount+1;
127 //Left
128 ib[9]=vbCount+1;
129 ib[10]=vbCount+4;
130 ib[11]=vbCount+0;
131
132 vbCount += 5;
133 ib+=12;
134 }
135 }
136}

138/*Copies a portion of the current render target into a specified buffer*/
139Int copyRect(unsigned char *buf, Int bufSize, int oX, int oY, int width, int height)
140{
141 IDirect3DSurface8 *surface=NULL;
142 IDirect3DSurface8 *tempSurface=NULL;
143 Int result = 0;
144 HRESULT hr = S_OK;
145
146 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
147
148 if (!m_pDev)
149 goto error;
150
151 m_pDev->GetRenderTarget(&surface);
152
153 if (!surface)
154 goto error;
155
156 D3DSURFACE_DESC desc;
157
158 surface->GetDesc(&desc);
159
160 RECT srcRect;
161 srcRect.left=oX;
162 srcRect.top=oY;
163 srcRect.right=oX+width;
164 srcRect.bottom=oY+height;
165
166 POINT dstPoint;
167 dstPoint.x=0;
168 dstPoint.y=0;
169
170 hr=m_pDev->CreateImageSurface( width, height, desc.Format, &tempSurface);
171
172 if (hr != S_OK)
173 goto error;
174
175 hr=m_pDev->CopyRects(surface,&srcRect,1,tempSurface,&dstPoint);
176
177 if (hr != S_OK)
178 goto error;
179
180 D3DLOCKED_RECT lrect;
181
182 hr=tempSurface->LockRect(&lrect,NULL,D3DLOCK_READONLY);
183
184 if (hr != S_OK)
185 goto error;
186
187 tempSurface->GetDesc(&desc);
188
189 if (desc.Size < bufSize)
190 bufSize = desc.Size;
191
192 memcpy(buf,lrect.pBits,bufSize);
193 result = bufSize;
194
195error:
196 if (surface)
197 surface->Release();
198 if (tempSurface)
199 tempSurface->Release();
200
201 return result;
202}

207Bool W3DSmudgeManager::testHardwareSupport(void)
208{
209 if (m_hardwareSupportStatus == SMUDGE_SUPPORT_UNKNOWN)
210 { //we have not done the test yet.
211
212 IDirect3DTexture8 *backTexture=W3DShaderManager::getRenderTexture();
213 if (!backTexture)
214 { //do trivial test first to see if render target exists.
215 m_hardwareSupportStatus = SMUDGE_SUPPORT_NO;
216 return FALSE;
217 }
218
219 if (!W3DShaderManager::isRenderingToTexture())
220 return FALSE; //can't do the test unless we're rendering to texture.
221
222 VertexMaterialClass *vmat=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
223 DX8Wrapper::Set_Material(vmat);
224 REF_PTR_RELEASE(vmat); //no need to keep a reference since it's a preset.
225
226 ShaderClass shader=ShaderClass::_PresetOpaqueShader;
227 shader.Set_Depth_Compare(ShaderClass::PASS_ALWAYS);
228 shader.Set_Depth_Mask(ShaderClass::DEPTH_WRITE_DISABLE);
229 DX8Wrapper::Set_Shader(shader);
230 DX8Wrapper::Set_Texture(0,NULL);
231 DX8Wrapper::Apply_Render_State_Changes(); //force update of view and projection matrices
232
233 struct _TRANS_LIT_TEX_VERTEX {
234 Vector4 p;
235 DWORD color; // diffuse color
236 float u;
237 float v;
238 } v[4];
239
240 //bottom right
241 v[0].p = Vector4( BLOCK_SIZE-0.5f, BLOCK_SIZE-0.5f, 0.0f, 1.0f );
242 v[0].u = BLOCK_SIZE/(Real)TheDisplay->getWidth(); v[0].v = BLOCK_SIZE/(Real)TheDisplay->getHeight();;
243 //top right
244 v[1].p = Vector4( BLOCK_SIZE-0.5f, 0-0.5f, 0.0f, 1.0f );
245 v[1].u = BLOCK_SIZE/(Real)TheDisplay->getWidth(); v[1].v = 0;
246 //bottom left
247 v[2].p = Vector4( 0-0.5f, BLOCK_SIZE-0.5f, 0.0f, 1.0f );
248 v[2].u = 0; v[2].v = BLOCK_SIZE/(Real)TheDisplay->getHeight();
249 //top left
250 v[3].p = Vector4( 0-0.5f, 0-0.5f, 0.0f, 1.0f );
251 v[3].u = 0; v[3].v = 0;
252
253 v[0].color = UNIQUE_COLOR;
254 v[1].color = UNIQUE_COLOR;
255 v[2].color = UNIQUE_COLOR;
256 v[3].color = UNIQUE_COLOR;
257
258 LPDIRECT3DDEVICE8 pDev=DX8Wrapper::_Get_D3D_Device8();
259
260 //draw polygons like this is very inefficient but for only 2 triangles, it's
261 //not worth bothering with index/vertex buffers.
262 pDev->SetVertexShader(D3DFVF_XYZRHW | D3DFVF_DIFFUSE | D3DFVF_TEX1);
263
264 pDev->DrawPrimitiveUP(D3DPT_TRIANGLESTRIP, 2, v, sizeof(_TRANS_LIT_TEX_VERTEX));
265
266 DWORD refData[BLOCK_SIZE*BLOCK_SIZE];
267 memset(refData,0,sizeof(refData));
268 Int bufSize=copyRect((unsigned char *)refData,sizeof(refData),0,0,BLOCK_SIZE,BLOCK_SIZE); //copy area we just rendered using solid color
269 if (!bufSize)
270 {
271 m_hardwareSupportStatus = SMUDGE_SUPPORT_NO;
272 return FALSE;
273 }
274
275 DX8Wrapper::Set_DX8_Texture(0,backTexture);
276
277 DWORD testData[BLOCK_SIZE*BLOCK_SIZE];
278 memset(testData,0xff,sizeof(testData));
279
280 v[0].color = 0xffffffff;
281 v[1].color = 0xffffffff;
282 v[2].color = 0xffffffff;
283 v[3].color = 0xffffffff;
284
285 pDev->DrawPrimitiveUP(D3DPT_TRIANGLESTRIP, 2, v, sizeof(_TRANS_LIT_TEX_VERTEX));
286 bufSize=copyRect((unsigned char *)testData,sizeof(testData),0,0,BLOCK_SIZE,BLOCK_SIZE);
287
288 if (!bufSize)
289 {
290 m_hardwareSupportStatus = SMUDGE_SUPPORT_NO;
291 return FALSE;
292 }
293
294 //compare the 2 buffers to see if they match.
295 if (memcmp(testData,refData,bufSize) == 0)
296 {
297 m_hardwareSupportStatus = SMUDGE_SUPPORT_YES;
298 return TRUE;
299 }
300 m_hardwareSupportStatus = SMUDGE_SUPPORT_NO;
301 }
302
303 return (SMUDGE_SUPPORT_YES == m_hardwareSupportStatus);
304}

306void W3DSmudgeManager::render(RenderInfoClass &rinfo)
307{
308 //Verify that the card supports the effect.
309 if (!testHardwareSupport())
310 return;
311
312 CameraClass &camera=rinfo.Camera;
313 Vector3 vsVert;
314 Vector4 ssVert;
315 Real uvSpanX,uvSpanY;
316 Vector3 vertex_offsets[4] = {
317 Vector3(-0.5f, 0.5f, 0.0f),
318 Vector3(-0.5f, -0.5f, 0.0f),
319 Vector3(0.5f, -0.5f, 0.0f),
320 Vector3(0.5f, 0.5f, 0.0f)
321 };
322
323#define THE_COLOR (0x00ffeedd)
324
325 UnsignedInt vertexDiffuse[5]={THE_COLOR,THE_COLOR,THE_COLOR,THE_COLOR,THE_COLOR};
326
327 Matrix4x4 proj;
328 Matrix3D view;
329
330 camera.Get_View_Matrix(&view);
331 camera.Get_Projection_Matrix(&proj);
332
333 SurfaceClass::SurfaceDescription surface_desc;
334#ifdef USE_COPY_RECTS
335 SurfaceClass *background=m_backgroundTexture->Get_Surface_Level();
336 background->Get_Description(surface_desc);
337#else
338 D3DSURFACE_DESC D3DDesc;
339
340 IDirect3DTexture8 *backTexture=W3DShaderManager::getRenderTexture();
341 if (!backTexture || !W3DShaderManager::isRenderingToTexture())
342 return; //this card doesn't support render targets.
343
344 backTexture->GetLevelDesc(0,&D3DDesc);
345
346 surface_desc.Width = D3DDesc.Width;
347 surface_desc.Height = D3DDesc.Height;
348#endif
349
350 Real texClampX = (Real)TheTacticalView->getWidth()/(Real)surface_desc.Width;
351 Real texClampY = (Real)TheTacticalView->getHeight()/(Real)surface_desc.Height;
352
353 Real texScaleX = texClampX*0.5f;
354 Real texScaleY = texClampY*0.5f;
355
356 //Do a first pass over the smudges to determine how many are visible
357 //and to fill in their world-space positions and screen uv coordinates.
358 //TODO: Optimize out this extra pass!
359 //TODO: Find size of screen rectangle that actually needs copying.
360
361 SmudgeSet *set=m_usedSmudgeSetList.Head(); //first set that didn't fit into render batch.
362 Int count = 0;
363
364 if (set)
365 { //there are possibly some smudges to render, so make sure background particles have finished drawing.
366 SortingRendererClass::Flush(); //draw sorted translucent polys like particles.
367 }
368
369 while (set)
370 {
371 Smudge *smudge=set->getUsedSmudgeList().Head();
372
373 while (smudge)
374 {
375 //Get view-space center
376 Matrix3D::Transform_Vector(view,smudge->m_pos,&vsVert);
377
378 //Get 5 view-space vertices
379 Smudge::smudgeVertex *verts=smudge->m_verts;
380
381 //Do center vertex outside 'for' loop since it's different.
382 verts[4].pos = vsVert;
383
384 for (Int i=0; i<4; i++)
385 {
386 verts[i].pos = vsVert + vertex_offsets[i] * smudge->m_size;
387 //Ge uv coordinates for each vertex
388 ssVert = proj * verts[i].pos;
389 Real oow = 1.0f/ssVert.W;
390 ssVert *= oow; //returned in camera space which is -1,-1 (bottom-left) to 1,1 (top-right)
391 //convert camera space to uv space: 0,0 (top-left), 1,1 (bottom-right)
392 verts[i].uv.Set((ssVert.X+1.0f)*texScaleX,(1.0f-ssVert.Y)*texScaleY);
393
394 Vector2 &thisUV=verts[i].uv;
395
396 //Clamp coordinates so we're not referencing texels outside the view.
397 if (thisUV.X > texClampX)
398 smudge->m_offset.X = 0;
399 else
400 if (thisUV.X < 0)
401 smudge->m_offset.X = 0;
402
403 if (thisUV.Y > texClampY)
404 smudge->m_offset.Y = 0;
405 else
406 if (thisUV.Y < 0)
407 smudge->m_offset.Y = 0;
408
409 }
410
411 //Finish center vertex
412 //Ge uv coordinates by interpolating corner uv coordinates and applying desired offset.
413 uvSpanX=verts[3].uv.X - verts[0].uv.X;
414 uvSpanY=verts[1].uv.Y - verts[0].uv.Y;
415 verts[4].uv.X=verts[0].uv.X+uvSpanX*(0.5f+smudge->m_offset.X);
416 verts[4].uv.Y=verts[0].uv.Y+uvSpanY*(0.5f+smudge->m_offset.X);
417
418 count++; //increment visible smudge count.
419 smudge=smudge->Succ();
420 }
421
422 set=set->Succ(); //advance to next node.
423 }
424
425 if (!count)
426 {
427#ifdef USE_COPY_RECTS
428 REF_PTR_RELEASE(background);
429#endif
430 return; //nothing to render.
431 }
432
433#ifdef USE_COPY_RECTS
434 SurfaceClass *backBuffer=DX8Wrapper::_Get_DX8_Back_Buffer();
435
436 backBuffer->Get_Description(surface_desc);
437
438 //Copy the area of backbuffer occupied by smudges into an alternate buffer.
439 background->Copy(0,0,0,0,surface_desc.Width,surface_desc.Height,backBuffer);
440
441 REF_PTR_RELEASE(background);
442 REF_PTR_RELEASE(backBuffer);
443#endif
444
445 Matrix4x4 identity(true);
446 DX8Wrapper::Set_Transform(D3DTS_WORLD,identity);
447 DX8Wrapper::Set_Transform(D3DTS_VIEW,identity);
448
449 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
450 //DX8Wrapper::Set_Shader(ShaderClass::_PresetOpaqueSpriteShader);
451
452 DX8Wrapper::Set_Shader(ShaderClass::_PresetAlphaShader);
453#ifdef USE_COPY_RECTS
454 DX8Wrapper::Set_Texture(0,m_backgroundTexture);
455#else
456 DX8Wrapper::Set_DX8_Texture(0,backTexture);
457 //Need these states in case texture is non-power-of-2
458 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ADDRESSU, D3DTADDRESS_CLAMP);
459 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ADDRESSV, D3DTADDRESS_CLAMP);
460 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ADDRESSW, D3DTADDRESS_CLAMP);
461 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_MAGFILTER, D3DTEXF_LINEAR);
462 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_MINFILTER, D3DTEXF_LINEAR);
463 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_MIPFILTER, D3DTEXF_NONE);
464#endif
465 VertexMaterialClass *vmat=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
466 DX8Wrapper::Set_Material(vmat);
467 REF_PTR_RELEASE(vmat);
468 DX8Wrapper::Apply_Render_State_Changes();
469
470 //Disable reading texture alpha since it's undefined.
471 //DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_COLOROP,D3DTOP_SELECTARG1);
472 DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_ALPHAOP,D3DTOP_SELECTARG2);
473
474 Int smudgesRemaining=count;
475 set=m_usedSmudgeSetList.Head(); //first smudge set that needs rendering.
476 Smudge *remainingSmudgeStart=set->getUsedSmudgeList().Head(); //first smudge that needs rendering.
477
478 while (smudgesRemaining) //keep drawing smudges until we run out.
479 {
480 //Now that we know how many smudges need rendering, allocate vertex buffer space and copy verts.
481 count=smudgesRemaining;
482
483 if (count > SMUDGE_DRAW_SIZE)
484 count = SMUDGE_DRAW_SIZE;
485
486 Int smudgesInRenderBatch=0;
487
488 DynamicVBAccessClass vb_access(BUFFER_TYPE_DYNAMIC_DX8,dynamic_fvf_type,count*5); //allocate 5 verts per smudge.
489 {
490 DynamicVBAccessClass::WriteLockClass lock(&vb_access);
491 VertexFormatXYZNDUV2* verts=lock.Get_Formatted_Vertex_Array();
492
493 while (set)
494 {
495 Smudge *smudge=remainingSmudgeStart;
496
497 while (smudge)
498 {
499 Smudge::smudgeVertex *smVerts = smudge->m_verts;
500
501 //Check if we exceeded maximum number of smudges allowed per draw call.
502 if (smudgesInRenderBatch >= count)
503 { remainingSmudgeStart = smudge;
504 goto flushSmudges;
505 }
506
507 //Set center vertex opacity.
508 vertexDiffuse[4] = ((Int)(smudge->m_opacity * 255.0f) << 24) | THE_COLOR;
509
510 for (Int i=0; i<5; i++)
511 {
512 verts->x=smVerts->pos.X;
513 verts->y=smVerts->pos.Y;
514 verts->z=smVerts->pos.Z;
515 verts->nx=0; //keep AGP write-combining active
516 verts->ny=0;
517 verts->nz=0;
518 verts->diffuse=vertexDiffuse[i]; //set to transparent
519 verts->u1=smVerts->uv.X;
520 verts->v1=smVerts->uv.Y;
521 verts->u2=0; //keep AGP write-combining active
522 verts->v2=0;
523 verts++;
524 smVerts++;
525 }
526
527 smudgesInRenderBatch++;
528 smudge=smudge->Succ();
529 }
530
531 set=set->Succ(); //advance to next node.
532
533 if (set) //start next batch at beginning of set.
534 remainingSmudgeStart = set->getUsedSmudgeList().Head();
535 }
536flushSmudges:
537 DX8Wrapper::Set_Vertex_Buffer(vb_access);
538 }
539
540 DX8Wrapper::Draw_Triangles( 0,smudgesInRenderBatch*4, 0, smudgesInRenderBatch*5);
541
542//Debug Code which draws outline around smudge
543/* DX8Wrapper::_Get_D3D_Device8()->SetRenderState(D3DRS_FILLMODE,D3DFILL_WIREFRAME);
544 DX8Wrapper::_Get_D3D_Device8()->SetRenderState(D3DRS_ALPHABLENDENABLE,FALSE);
545 DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_COLOROP,D3DTOP_SELECTARG2);
546 DX8Wrapper::Draw_Triangles( 0,smudgesInRenderBatch*4, 0, smudgesInRenderBatch*5);
547 DX8Wrapper::_Get_D3D_Device8()->SetRenderState(D3DRS_FILLMODE,D3DFILL_SOLID);
548 DX8Wrapper::_Get_D3D_Device8()->SetRenderState(D3DRS_ALPHABLENDENABLE,TRUE);
549 DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_COLOROP,D3DTOP_SELECTARG1);
550*/
551 smudgesRemaining -= smudgesInRenderBatch;
552 }
553
554 DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_COLOROP,D3DTOP_MODULATE);
555 DX8Wrapper::Set_DX8_Texture_Stage_State(0,D3DTSS_ALPHAOP,D3DTOP_MODULATE);
