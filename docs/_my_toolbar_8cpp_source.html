42void CellSizeToolBar::CellSizeChanged(Int cellSize)
43{
44 Int newSize = 1;
45 Int i;
46 for (i=MIN_POS; i<MAX_POS; i++) {
47 newSize *= 2;
48 if (newSize >= cellSize) break;
49 }
50 i = MAX_POS - i + MIN_POS; // Invert the range
51 if (m_staticThis != NULL) {
52 m_staticThis->m_cellSlider.SetPos(i);
53 }
54}

61void CellSizeToolBar::SetupSlider(void)
62{
63 CWnd *pWnd = GetDlgItem(ID_SLIDER);
64 CRect rect;
65 pWnd->GetWindowRect(&rect);
66
67 pWnd->DestroyWindow();
68 ScreenToClient(&rect);
69
70 m_cellSlider.Create(TBS_VERT|TBS_AUTOTICKS|TBS_RIGHT, rect, this, ID_SLIDER);
71 m_cellSlider.SetRange(MIN_POS,MAX_POS);
72 m_cellSlider.SetPos(3);
73 m_cellSlider.ShowWindow(SW_SHOW);
74 m_staticThis = this;

77void CellSizeToolBar::OnVScroll(UINT nSBCode, UINT nPos, CScrollBar* pScrollBar)
78{
79 if (nSBCode != TB_THUMBTRACK) {
80 nPos = m_cellSlider.GetPos();
81 }
82 UnsignedInt i;
83 // invert
84 nPos = MAX_POS - nPos + MIN_POS;
85 int newSize = 1;
86 for (i=1; i<nPos; i++) {
87 newSize *= 2;
88 }
89 if (newSize>64) return;
90
91 CWorldBuilderView* pView = CWorldBuilderDoc::GetActive2DView();
92 if (pView == NULL || newSize == pView->getCellSize()) return;
93 pView->setCellSize(newSize);
94}
