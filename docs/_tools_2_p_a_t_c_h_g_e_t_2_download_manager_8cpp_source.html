30DownloadManager::DownloadManager()
31{
32 m_download = new CDownload(this);
33 m_wasError = m_sawEnd = false;
34 m_statusString = Fetch_String(FTP_StatusIdle);
35
36 // ----- Initialize Winsock -----
37 m_winsockInit = true;
38 WORD verReq = MAKEWORD(2, 2);
39 WSADATA wsadata;
40
41 int err = WSAStartup(verReq, &wsadata);
42 if (err != 0)
43 {
44 m_winsockInit = false;
45 }
46 else
47 {
48 if ((LOBYTE(wsadata.wVersion) != 2) || (HIBYTE(wsadata.wVersion) !=2))
49 {
50 WSACleanup();
51 m_winsockInit = false;
52 }
53 }
54
55}

57DownloadManager::~DownloadManager()

62 WSACleanup();
63 m_winsockInit = false;

99HRESULT DownloadManager::downloadNextQueuedFile( void )
100{
101 QueuedDownload q;
102 std::list<QueuedDownload>::iterator it = m_queuedDownloads.begin();
103 if (it != m_queuedDownloads.end())
104 {
105 q = *it;
106 m_queuedDownloads.pop_front();
107 m_wasError = m_sawEnd = false;
108 return downloadFile( q.server, q.userName, q.password, q.file, q.localFile, q.regKey, q.tryResume );
109 }
110 else
111 {
112 DEBUG_CRASH(("Starting non-existent download!"));
113 return S_OK;
114 }
115}

124HRESULT DownloadManager::OnError( int error )
125{
126 m_wasError = true;
127 std::string s = Fetch_String(FTP_UnknownError);
128 switch (error)
129 {
130 case DOWNLOADEVENT_NOSUCHSERVER:
131 s = Fetch_String(FTP_NoSuchServer);
132 break;
133 case DOWNLOADEVENT_COULDNOTCONNECT:
134 s = Fetch_String(FTP_CouldNotConnect);
135 break;
136 case DOWNLOADEVENT_LOGINFAILED:
137 s = Fetch_String(FTP_LoginFailed);
138 break;
139 case DOWNLOADEVENT_NOSUCHFILE:
140 s = Fetch_String(FTP_NoSuchFile);
141 break;
142 case DOWNLOADEVENT_LOCALFILEOPENFAILED:
143 s = Fetch_String(FTP_LocalFileOpenFailed);
144 break;
145 case DOWNLOADEVENT_TCPERROR:
146 s = Fetch_String(FTP_TCPError);
147 break;
148 case DOWNLOADEVENT_DISCONNECTERROR:
149 s = Fetch_String(FTP_DisconnectError);
150 break;
151 }
152 m_errorString = s;
153 DEBUG_LOG(("DownloadManager::OnError(): %s(%d)\n", s.c_str(), error));
154 return S_OK;
155}
