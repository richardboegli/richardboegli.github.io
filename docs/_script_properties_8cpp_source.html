77BOOL ScriptProperties::OnSetActive()
78{
79 if ( CPropertyPage::OnSetActive()) {
80 CWnd *pWnd = GetDlgItem(IDC_SCRIPT_COMMENT);
81 pWnd->SetWindowText(m_script->getComment().str());
82
83 pWnd = GetDlgItem(IDC_SCRIPT_NAME);
84 pWnd->SetWindowText(m_script->getName().str());
85 enableControls();
86 return true;
87 }

91void ScriptProperties::enableControls()
92{
93 Bool isSubroutine = m_script->isSubroutine();
94 CButton *pButton = (CButton*)GetDlgItem(IDC_SCRIPT_SUBROUTINE);
95 pButton->SetCheck(isSubroutine ? 1:0);
96 pButton = (CButton*)GetDlgItem(IDC_SCRIPT_ACTIVE);
97 pButton->SetCheck(m_script->isActive() ? 1:0);
98 //pButton->EnableWindow(!isSubroutine);
99 pButton = (CButton*)GetDlgItem(IDC_ONE_SHOT);
100 pButton->SetCheck(m_script->isOneShot() ? 1:0);
101 //pButton->EnableWindow(!isSubroutine);
102 pButton = (CButton*)GetDlgItem(IDC_EASY);
103 pButton->SetCheck(m_script->isEasy() ? 1:0);
104 //pButton->EnableWindow(!isSubroutine);
105 pButton = (CButton*)GetDlgItem(IDC_NORMAL);
106 pButton->SetCheck(m_script->isNormal() ? 1:0);
107 //pButton->EnableWindow(!isSubroutine);
108 pButton = (CButton*)GetDlgItem(IDC_HARD);
109 pButton->SetCheck(m_script->isHard() ? 1:0);
110 //pButton->EnableWindow(!isSubroutine);
111
112 m_updating = true;
113 Int delay = m_script->getDelayEvalSeconds();
114 pButton = (CButton*)GetDlgItem(IDC_EVERY_SECOND);
115 pButton->SetCheck(delay>0);
116 pButton = (CButton*)GetDlgItem(IDC_EVERY_FRAME);
117 pButton->SetCheck(delay==0);
118 CString text="";
119 if (delay>0) text.Format("%d", delay);
120 GetDlgItem(IDC_SECONDS_EDIT)->SetWindowText(text);
121 m_updating = false;
122}
