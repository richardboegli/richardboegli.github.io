152// Version 3.0 Material, A W3D_CHUNK_MATERIALS3 chunk will wrap a bunch of
153// W3D_CHUNK_MATERIAL3 chunks. Inside each chunk will be a name chunk, an 'info' chunk which
154// contains the following struct, and one or more map chunks. a mesh with 2 materials might
155// look like:
156//
157// W3D_CHUNK_MATERIALS3 <-- simply a wrapper around the array of mtls
158// W3D_CHUNK_MATERIAL3 <-- a wrapper around each material
159// W3D_CHUNK_STRING <-- name of the material
160// W3D_CHUNK_MATERIAL3_INFO <-- standard material properties, a W3dMaterial3Struct
161// W3D_CHUNK_MATERIAL3_DC_MAP <-- a map, W3dMap3Struct
162// W3D_CHUNK_STRING <-- filename of the map
163// W3D_CHUNK_MAP_INFO <-- map parameters
164// W3D_CHUNK_MATERIAL3_SC_MAP
165// W3D_CHUNK_STRING <-- filename of the map
166// W3D_CHUNK_MAP_INFO
167// W3D_CHUNK_MATERIAL3
168// W3D_CHUNK_MATERIAL3_NAME
169// W3D_CHUNK_MATERIAL3_INFO
170// W3D_CHUNK_MATERIAL3_SI_MAP
171//
173struct W3dMaterial3Struct
174{
175 uint32 Attributes; // flags,hints,etc.
176
177 W3dRGBStruct DiffuseColor; // diffuse color
178 W3dRGBStruct SpecularColor; // specular color
179
180 W3dRGBStruct EmissiveCoefficients; // emmissive coefficients, default to 0,0,0
181 W3dRGBStruct AmbientCoefficients; // ambient coefficients, default to 1,1,1
182 W3dRGBStruct DiffuseCoefficients; // diffuse coeficients, default to 1,1,1
183 W3dRGBStruct SpecularCoefficients; // specular coefficients, default to 0,0,0
184
185 float32 Shininess; // how tight the specular highlight will be, 1 - 1000 (default = 1)
186 float32 Opacity; // how opaque the material is, 0.0 = invisible, 1.0 = fully opaque (default = 1)
187 float32 Translucency; // how much light passes through the material. (default = 0)
188 float32 FogCoeff; // effect of fog (0.0=not fogged, 1.0=fogged) (default = 1)
189};
