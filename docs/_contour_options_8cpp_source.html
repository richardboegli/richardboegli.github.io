113void ContourOptions::OnHScroll(UINT nSBCode, UINT nPos, CScrollBar* pScrollBar)
114{
115 CString str;
116 m_updating = true;
117 Bool step = (pScrollBar && pScrollBar->m_hWnd == m_contourStepSlider.m_hWnd);
118 Bool offset = (pScrollBar && pScrollBar->m_hWnd == m_contourOffsetSlider.m_hWnd);
119 if (step) {
120 if (nSBCode != TB_THUMBTRACK) {
121 nPos = m_contourStepSlider.GetPos();
122 }
123 m_contourStep = MIN_CONTOUR_STEP + MAX_CONTOUR_STEP - nPos;
124 } else if (offset) { // width
125 if (nSBCode != TB_THUMBTRACK) {
126 nPos = m_contourOffsetSlider.GetPos();
127 }
128 m_contourOffset = nPos;
129 } else { // width
130 if (nSBCode != TB_THUMBTRACK) {
131 nPos = m_contourWidthSlider.GetPos();
132 }
133 m_contourWidth = nPos;
134 }
135 if (nSBCode!=TB_THUMBTRACK) {
136 CWorldBuilderView *pView = CWorldBuilderDoc::GetActive2DView();
137 if (pView)
138 pView->setCellSize(pView->getCellSize());
139 }
140 m_updating = false;
141}

156 CButton *pButton = (CButton*)GetDlgItem(IDC_SHOW_CONTOURS);

158 pView->setShowContours(showContours);
