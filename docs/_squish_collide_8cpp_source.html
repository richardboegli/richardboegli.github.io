67void SquishCollide::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
68{
69 // Note that other == null means "collide with ground"
70 if (other == NULL)
71 return;
72
73 Object *self = getObject();
74 AIUpdateInterface *ai = self->getAI();
75 if( ai && ai->getGoalObject() == other )
76 {
77 //We are actually targeting the other object to do something to! Don't allow them to crush us in the following
78 //special circumstances:
79
80 //Hijacking!
81 static NameKeyType key_HijackerUpdate = NAMEKEY( "HijackerUpdate" );
82 HijackerUpdate *hijackUpdate = (HijackerUpdate*)self->findUpdateModule( key_HijackerUpdate );
83 if( hijackUpdate )
84 {
85 return;
86 }
87
88 //TNT hunter placing a charge!
89 SpecialAbilityUpdate *saUpdate = self->findSpecialAbilityUpdate( SPECIAL_TANKHUNTER_TNT_ATTACK );
90 if( saUpdate && saUpdate->isActive() )
91 {
92 return;
93 }
94 }
95
96 // order matters: we want to know if IT considers ME to be an ally (a reversal of the usual situation)
97 if( other->getCrusherLevel() > 0 && other->getRelationship(getObject()) != ALLIES)
98 {
99 PhysicsBehavior *otherPhysics = other->getPhysics();
100 if (otherPhysics == NULL)
101 return;
102
103 // use a 1.0 crush radius so the tank has to actually hit the infantry.
104 GeometryInfo myGeom = self->getGeometryInfo();
105 myGeom.setMajorRadius(1.0f);
106 myGeom.setMinorRadius(1.0f);
107 if (!ThePartitionManager->geomCollidesWithGeom(other->getPosition(), other->getGeometryInfo(), other->getOrientation(),
108 self->getPosition(), myGeom, self->getOrientation())) {
109 return;
110 }
111
112 // only squish if tank is moving toward victim
113 const Coord3D *vel = otherPhysics->getVelocity();
114 const Coord3D *pos = other->getPosition();
115 const Coord3D *myPos = getObject()->getPosition();
116 Coord3D to;
117
118 to.x = myPos->x - pos->x;
119 to.y = myPos->y - pos->y;
120 to.z = myPos->z - pos->z;
121
122 if (to.x * vel->x + to.y * vel->y > 0.0f)
123 {
124 DamageInfo damageInfo;
125 damageInfo.in.m_damageType = DAMAGE_CRUSH;
126 damageInfo.in.m_deathType = DEATH_CRUSHED;
127 damageInfo.in.m_sourceID = other->getID();
128 damageInfo.in.m_amount = HUGE_DAMAGE_AMOUNT; // make sure they die
129 getObject()->attemptDamage( &damageInfo );
130 getObject()->friend_setUndetectedDefector( FALSE );// My secret is out
131 }
132
133
134 }
135}
