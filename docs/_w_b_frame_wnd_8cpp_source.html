43BOOL CWBFrameWnd::LoadFrame(UINT nIDResource,
44 DWORD dwDefaultStyle,
45 CWnd* pParentWnd,
46 CCreateContext* pContext) {
47 //dwDefaultStyle &= ~(WS_SIZEBOX|WS_MAXIMIZEBOX|WS_SYSMENU);
48
49 BOOL ret = CFrameWnd::LoadFrame(nIDResource, dwDefaultStyle, CMainFrame::GetMainFrame(), pContext);
50 if (ret) {
51 Int top = ::AfxGetApp()->GetProfileInt(TWO_D_WINDOW_SECTION, "Top", 10);
52 Int left =::AfxGetApp()->GetProfileInt(TWO_D_WINDOW_SECTION, "Left", 10);
53 this->SetWindowPos(NULL, left,
54 top, 0, 0,
55 SWP_NOZORDER|SWP_NOSIZE);
56 if (!m_cellSizeToolBar.Create(this, IDD_CELL_SLIDER, CBRS_LEFT, IDD_CELL_SLIDER))
57 {
58 DEBUG_CRASH(("Failed to create toolbar\n"));
59 }
60 EnableDocking(CBRS_ALIGN_ANY);
61 m_cellSizeToolBar.SetupSlider();
62 m_cellSizeToolBar.EnableDocking(CBRS_ALIGN_ANY);
63 DockControlBar(&m_cellSizeToolBar);
64 }
65 return(ret);
66}
