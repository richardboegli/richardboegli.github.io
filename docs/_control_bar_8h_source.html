76enum CommandOption
77{
78 COMMAND_OPTION_NONE = 0x00000000,
79 NEED_TARGET_ENEMY_OBJECT = 0x00000001, // command now needs user to select enemy target
80 NEED_TARGET_NEUTRAL_OBJECT = 0x00000002, // command now needs user to select neutral target
81 NEED_TARGET_ALLY_OBJECT = 0x00000004, // command now needs user to select ally target
82#ifdef ALLOW_SURRENDER
83 NEED_TARGET_PRISONER = 0x00000008, // needs user to now select prisoner object
84#endif
85 ALLOW_SHRUBBERY_TARGET = 0x00000010, // allow neutral shrubbery as a target
86 NEED_TARGET_POS = 0x00000020, // command now needs user to select target position
87 NEED_UPGRADE = 0x00000040, // command requires upgrade to be enabled
88 NEED_SPECIAL_POWER_SCIENCE = 0x00000080, // command requires a science in the special power specified
89 OK_FOR_MULTI_SELECT = 0x00000100, // command is ok to show when multiple objects selected
90 CONTEXTMODE_COMMAND = 0x00000200, // a context sensitive command mode that requires code to determine whether cursor is valid or not.
91 CHECK_LIKE = 0x00000400, // dynamically change the UI element push button to be "check like"
92 ALLOW_MINE_TARGET = 0x00000800, // allow (land)mines as a target
93 ATTACK_OBJECTS_POSITION = 0x00001000, // for weapons that need an object target but attack the position indirectly (like burning trees)
94 OPTION_ONE = 0x00002000, // User data -- option 1
95 OPTION_TWO = 0x00004000, // User data -- option 2
96 OPTION_THREE = 0x00008000, // User data -- option 3
97 NOT_QUEUEABLE = 0x00010000, // Option not build queueable meaning you can only build it when queue is empty!
98 SINGLE_USE_COMMAND = 0x00020000, // Once used, it can never be used again!
99 COMMAND_FIRED_BY_SCRIPT = 0x00040000, // Used only by code to tell special powers that they have been fired by a script.
100 SCRIPT_ONLY = 0x00080000, // Only a script can use this command (not by users)
101 IGNORES_UNDERPOWERED = 0x00100000, // this button isn't disabled if its object is merely underpowered
102 USES_MINE_CLEARING_WEAPONSET= 0x00200000, // uses the special mine-clearing weaponset, even if not current
103 CAN_USE_WAYPOINTS = 0x00400000, // button has option to use a waypoint path
104 MUST_BE_STOPPED = 0x00800000, // Unit must be stopped in order to be able to use button.
105
106 NUM_COMMAND_OPTIONS // keep this last
107};
