45/*
46** This class describes a point in 2 dimensional space using arbitrary
47** components. The interpretation of which is outside the scope
48** of this class. This class is the successor to the old style COORDINATE
49** and CELL types but also serves anywhere an X and Y value are treated
50** as a logical object (e.g., pixel location).
51*/
52template<class T>
53class TPoint2D {
54 public:
55 TPoint2D(void) {} // Default constructor does nothing by design.
56 TPoint2D(T x, T y) : X(x), Y(y) {}
57
58 // Equality comparison operators.
59 bool operator == (TPoint2D<T> const & rvalue) const {return(X==rvalue.X && Y==rvalue.Y);}
60 bool operator != (TPoint2D<T> const & rvalue) const {return(X!=rvalue.X || Y!=rvalue.Y);}
61
62 // Addition and subtraction operators.
63 TPoint2D<T> const & operator += (TPoint2D<T> const & rvalue) {X += rvalue.X;Y += rvalue.Y;return(*this);}
64 TPoint2D<T> const & operator -= (TPoint2D<T> const & rvalue) {X -= rvalue.X;Y -= rvalue.Y;return(*this);}
65 TPoint2D<T> const operator - (TPoint2D<T> const & rvalue) const {return(TPoint2D<T>(X - rvalue.X, Y - rvalue.Y));}
66 TPoint2D<T> const operator + (TPoint2D<T> const & rvalue) const {return(TPoint2D<T>(X + rvalue.X, Y + rvalue.Y));}
67
68 // Scalar multiplication and division.
69 TPoint2D<T> const operator * (T rvalue) const {return(TPoint2D<T>(X * rvalue, Y * rvalue));}
70 TPoint2D<T> const & operator *= (T rvalue) {X *= rvalue; Y *= rvalue;return(*this);}
71 TPoint2D<T> const operator / (T rvalue) const {if (rvalue == T(0)) return(TPoint2D<T>(0,0));return(TPoint2D<T>(X / rvalue, Y / rvalue));}
72 TPoint2D<T> const & operator /= (T rvalue) {if (rvalue != T(0)) {X /= rvalue;Y /= rvalue;}return(*this);}
73
74 // Dot and cross product.
75 TPoint2D<T> const operator * (TPoint2D<T> const & rvalue) const {return(TPoint2D<T>(X * rvalue.X, Y * rvalue.Y));}
76 TPoint2D<T> const Dot_Product(TPoint2D<T> const & rvalue) const {return(TPoint2D<T>(X * rvalue.X, Y * rvalue.Y));}
77 TPoint2D<T> const Cross_Product(TPoint2D<T> const & rvalue) const {return(TPoint2D<T>(Y - rvalue.Y, rvalue.X - X));}
78
79 // Negation operator -- simple and effective
80 TPoint2D<T> const operator - (void) const {return(TPoint2D<T>(-X, -Y));}
81
82 // Vector support functions.
83 T Length(void) const {return(T(sqrt(X*X + Y*Y)));}
84 TPoint2D<T> const Normalize(void) const {
85 double len = sqrt(X*X + Y*Y);
86 if (len != 0.0) {
87 return(TPoint2D<T>((T)((double)X / len), (T)((double)Y / len)));
88 } else {
89 return(*this);
90 }
91 }
92
93 // Bias a point into a clipping rectangle.
94 TPoint2D<T> const Bias_To(TRect<T> const & rect) const;
95
96 // Find distance between points.
97 T Distance_To(TPoint2D<T> const & point) const {return((*this - point).Length());}
98
99 public:
100 T X;
101 T Y;
102};

105TPoint2D<T> const operator * (T lvalue, TPoint2D<T> const & rvalue)
106{
107 return(rvalue * lvalue);
108}

118/*
119** This describes a point in 3 dimensional space using arbitrary
120** components. This is the successor to the COORDINATE type for those
121** times when height needs to be tracked.
122**
123** Notice that it is not implemented as a virtually derived class. This
124** is for efficiency reasons. This class chooses to be smaller and faster at the
125** expense of polymorphism. However, since it is publicly derived, inheritance is
126** the next best thing.
127*/
128template<class T>
129class TPoint3D : public TPoint2D<T> {
130 typedef TPoint2D<T> BASECLASS;
131
132 public:
133 TPoint3D(void) {} // Default constructor does nothing by design.
134 TPoint3D(T x, T y, T z) : TPoint2D<T>(x, y), Z(z) {}
135 TPoint3D(BASECLASS const & rvalue) : TPoint2D<T>(rvalue), Z(0) {}
136
137 // Equality comparison operators.
138 bool operator == (TPoint3D<T> const & rvalue) const {return(X==rvalue.X && Y==rvalue.Y && Z==rvalue.Z);}
139 bool operator != (TPoint3D<T> const & rvalue) const {return(X!=rvalue.X || Y!=rvalue.Y || Z!=rvalue.Z);}
140
141 // Addition and subtraction operators.
142 TPoint3D<T> const & operator += (TPoint3D<T> const & rvalue) {X += rvalue.X;Y += rvalue.Y;Z += rvalue.Z;return(*this);}
143 TPoint2D<T> const & operator += (TPoint2D<T> const & rvalue) {BASECLASS::operator += (rvalue);return(*this);}
144 TPoint3D<T> const & operator -= (TPoint3D<T> const & rvalue) {X -= rvalue.X;Y -= rvalue.Y;Z -= rvalue.Z;return(*this);}
145 TPoint2D<T> const & operator -= (TPoint2D<T> const & rvalue) {BASECLASS::operator -= (rvalue);return(*this);}
146 TPoint3D<T> const operator - (TPoint3D<T> const & rvalue) const {return(TPoint3D<T>(X - rvalue.X, Y - rvalue.Y, Z - rvalue.Z));}
147 TPoint3D<T> const operator - (TPoint2D<T> const & rvalue) const {return(TPoint3D<T>(X - rvalue.X, Y - rvalue.Y, Z));}
148 TPoint3D<T> const operator + (TPoint3D<T> const & rvalue) const {return(TPoint3D<T>(X + rvalue.X, Y + rvalue.Y, Z + rvalue.Z));}
149 TPoint3D<T> const operator + (TPoint2D<T> const & rvalue) const {return(TPoint3D<T>(X + rvalue.X, Y + rvalue.Y, Z));}
150
151 // Scalar multiplication and division.
152 TPoint3D<T> const operator * (T rvalue) const {return(TPoint3D<T>(X * rvalue, Y * rvalue, Z * rvalue));}
153 TPoint3D<T> const & operator *= (T rvalue) {X *= rvalue;Y *= rvalue;Z *= rvalue;return(*this);}
154 TPoint3D<T> const operator / (T rvalue) const {if (rvalue == T(0)) return(TPoint3D<T>(0,0,0));return(TPoint3D<T>(X / rvalue, Y / rvalue, Z / rvalue));}
155 TPoint3D<T> const & operator /= (T rvalue) {if (rvalue != T(0)) {X /= rvalue;Y /= rvalue;Z /= rvalue;}return(*this);}
156
157 // Dot and cross product.
158 TPoint3D<T> const operator * (TPoint3D<T> const & rvalue) const {return(TPoint3D<T>(X * rvalue.X, Y * rvalue.Y, Z * rvalue.Z));}
159 TPoint3D<T> const Dot_Product(TPoint3D<T> const & rvalue) const {return(TPoint3D<T>(X * rvalue.X, Y * rvalue.Y, Z * rvalue.Z));}
160 TPoint3D<T> const Cross_Product(TPoint3D<T> const & rvalue) const {return TPoint3D<T>(Y * rvalue.Z - Z * rvalue.Y, Z * rvalue.X - X * rvalue.Z, X * rvalue.Y - Y * rvalue.X);}
161
162 // Negation operator -- simple and effective
163 TPoint3D<T> const operator - (void) const {return(TPoint3D<T>(-X, -Y, -Z));}
164
165 // Vector support functions.
166 T Length(void) const {return(T(sqrt(X*X + Y*Y + Z*Z)));}
167 TPoint3D<T> const Normalize(void) const {
168 double len = sqrt(X*X + Y*Y + Z*Z);
169 if (len != 0.0) {
170 return(TPoint3D<T>(X / len, Y / len, Z / len));
171 } else {
172 return(*this);
173 }
174 }
175
176 // Find distance between points.
177 T Distance_To(TPoint3D<T> const & point) const {return((*this - point).Length());}
178 T Distance_To(TPoint2D<T> const & point) const {return(BASECLASS::Distance_To(point));}
179
180 public:
181
182 /*
183 ** The Z component of this point.
184 */
185 T Z;
186};
