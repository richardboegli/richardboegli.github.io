84Bool SabotageSupplyDropzoneCrateCollide::isValidToExecute( const Object *other ) const
85{
86 if( !CrateCollide::isValidToExecute(other) )
87 {
88 //Extend functionality.
89 return FALSE;
90 }
91
92 if( other->isEffectivelyDead() )
93 {
94 //Can't sabotage dead structures
95 return FALSE;
96 }
97
98 if( !other->isKindOf( KINDOF_FS_SUPPLY_DROPZONE ) )
99 {
100 //We can only sabotage supply dropzones.
101 return FALSE;
102 }
103
104 Relationship r = getObject()->getRelationship( other );
105 if( r != ENEMIES )
106 {
107 //Can only sabotage enemy buildings.
108 return FALSE;
109 }
110
111 return TRUE;
112}

116Bool SabotageSupplyDropzoneCrateCollide::executeCrateBehavior( Object *other )
117{
118 //Check to make sure that the other object is also the goal object in the AIUpdateInterface
119 //in order to prevent an unintentional conversion simply by having the terrorist walk too close
120 //to it.
121 //Assume ai is valid because CrateCollide::isValidToExecute(other) checks it.
122 Object *obj = getObject();
123 AIUpdateInterface* ai = obj->getAIUpdateInterface();
124 if (ai && ai->getGoalObject() != other)
125 {
126 return false;
127 }
128
129 TheRadar->tryInfiltrationEvent( other );
130
131 doSabotageFeedbackFX( other, CrateCollide::SAB_VICTIM_DROP_ZONE );
132
133 //Reset the timer on the dropzone. Um... only the dropzone has an OCLUpdate and one, so
134 //we can "assume" it's going to be safe. Otherwise, we'll have to write code to search for
135 //a specific OCLUpdate.
136 static NameKeyType key_ocl = NAMEKEY( "OCLUpdate" );
137 OCLUpdate *update = (OCLUpdate*)other->findUpdateModule( key_ocl );
138 if( update )
139 {
140 update->resetTimer();
141 }
142
143 //Steal cash!
144 Money *targetMoney = other->getControllingPlayer()->getMoney();
145 Money *objectMoney = obj->getControllingPlayer()->getMoney();
146 if( targetMoney && objectMoney )
147 {
148 UnsignedInt cash = targetMoney->countMoney();
149 UnsignedInt desiredAmount = getSabotageSupplyDropzoneCrateCollideModuleData()->m_stealCashAmount;
150 //Check to see if they have the cash, otherwise, take the remainder!
151 cash = min( desiredAmount, cash );
152 if( cash > 0 )
153 {
154 //Steal the cash
155 targetMoney->withdraw( cash );
156 objectMoney->deposit( cash );
157 Player* controller = obj->getControllingPlayer();
158 if (controller)
159 controller->getScoreKeeper()->addMoneyEarned( cash );
160
161 //Play the "cash stolen" EVA event if the local player is the victim!
162 if( other && other->isLocallyControlled() )
163 {
164 TheEva->setShouldPlay( EVA_CashStolen );
165 }
166
167 //Display cash income floating over the about to be deleted saboteur.
168 UnicodeString moneyString;
169 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), cash );
170 Coord3D pos;
171 pos.set( obj->getPosition() );
172 pos.z += 20.0f; //add a little z to make it show up above the unit.
173 TheInGameUI->addFloatingText( moneyString, &pos, GameMakeColor( 0, 255, 0, 255 ) );
174
175 //Display cash lost floating over the target
176 moneyString.format( TheGameText->fetch( "GUI:LoseCash" ), cash );
177 pos.set( other->getPosition() );
178 pos.z += 30.0f; //add a little z to make it show up above the unit.
179 TheInGameUI->addFloatingText( moneyString, &pos, GameMakeColor( 255, 0, 0, 255 ) );
180 }
181 else
182 {
183 if( other->isLocallyControlled() )
184 {
185 TheEva->setShouldPlay( EVA_BuildingSabotaged );
186 }
187 }
188 }
189
190
191 return TRUE;
192}
