178Player *ScriptConditions::playerFromParam(Parameter *pSideParm)
179{
180 DEBUG_ASSERTCRASH(Parameter::SIDE == pSideParm->getParameterType(), ("Wrong parameter type."));
181 Player *pPlayer=NULL;
182 UnsignedInt mask = (UnsignedInt)pSideParm->getInt();
183 if (mask) {
184 pPlayer = ThePlayerList->getPlayerFromMask(mask);
185 } else {
186 pPlayer = TheScriptEngine->getPlayerFromAsciiString(pSideParm->getString());
187 if (pPlayer) {
188 // Enemy player can change dynamically, so don't cache the player mask. jba.
189 if (pSideParm->getString()!=THIS_PLAYER_ENEMY) {
190 mask = pPlayer->getPlayerMask();
191 }
192 } else {
193 mask = 0xFFFF0000;
194 }
195 pSideParm->friend_setInt((Int)mask);
196 }
197 DEBUG_ASSERTCRASH(pPlayer, ("Couldn't find player %s", pSideParm->getString().str()));
198 return pPlayer;
199}

364Bool ScriptConditions::evaluateHasUnits(Parameter *pTeamParm)
365{
366 AsciiString desiredTeamName = pTeamParm->getString();
367 // If they are calling a <this team> condition, do it.
368 if (desiredTeamName == THIS_TEAM) {
369 Team *theTeam = TheScriptEngine->getTeamNamed( desiredTeamName );
370 // The team is the team based on the name, and the calling team (if any) and the team that
371 // is being considered for the condition. jba. :)
372 if (theTeam) {
373 return (theTeam->hasAnyUnits());
374 }
375 return false; // Non existent team has no units.
376 }
377 Team *thisTeam = TheScriptEngine->getTeamNamed(THIS_TEAM);
378 if (thisTeam && thisTeam->getName()==desiredTeamName) {
379 return thisTeam->hasAnyUnits();
380 }
381
382 // It isn't THIS_TEAM, and doesn't match the THIS_TEAM, so check if any team with this name
383 // has units.
384 TeamPrototype *pProto = NULL;
385 pProto = TheTeamFactory->findTeamPrototype(desiredTeamName);
386
387 if (pProto) {
388 // We have a team referred to in the conditions. Iterate over the instances of the team,
389 // applying the script conditions (and possibly actions) to each instance of the team.
390 for (DLINK_ITERATOR<Team> iter = pProto->iterate_TeamInstanceList(); !iter.done(); iter.advance()) {
391 if (iter.cur()->hasAnyUnits()) {
392 return true;
393 }
394 }
395 }
396 return false; // Non existent team has no units.
397}

444Bool ScriptConditions::evaluatePlayerHasUnitTypeInArea(Condition *pCondition, Parameter *pPlayerParm, Parameter *pComparisonParm, Parameter *pCountParm, Parameter *pTypeParm, Parameter *pTriggerParm )
445{
446 AsciiString triggerName = pTriggerParm->getString();
447 PolygonTrigger *pTrig = TheScriptEngine->getQualifiedTriggerAreaByName(pTriggerParm->getString());
448 if (pTrig == NULL) return false;
449
450 Player* pPlayer = playerFromParam(pPlayerParm);
451 if (!pPlayer) {
452 return false;
453 }
454 Player::PlayerTeamList::const_iterator it;
455 Bool anyChanges = false;
456
457
458 if (pCondition->getCustomData() == 0) anyChanges = true;
459
460 for (it = pPlayer->getPlayerTeams()->begin(); it != pPlayer->getPlayerTeams()->end(); ++it) {
461 if (anyChanges) break;
462 for (DLINK_ITERATOR<Team> iter = (*it)->iterate_TeamInstanceList(); !iter.done(); iter.advance()) {
463 if (anyChanges) break;
464 Team *team = iter.cur();
465 if (!team) {
466 continue;
467 }
468 if (team->didEnterOrExit()) {
469 anyChanges = true;
470 }
471 }
472 }
473
474 if (TheScriptEngine->getFrameObjectCountChanged() > pCondition->getCustomFrame()) {
475 anyChanges = true; // Objects were added/deleted last frame, so count could have changed. jba.
476 }
477
478 if (!anyChanges) {
479 if (pCondition->getCustomData()==-1) return false;
480 if (pCondition->getCustomData()==1) return true;
481 }
482
483 ObjectTypesTemp types;
484 objectTypesFromParam(pTypeParm, types.m_types);
485
486 Int count = 0;
487 for (it = pPlayer->getPlayerTeams()->begin(); it != pPlayer->getPlayerTeams()->end(); ++it) {
488 for (DLINK_ITERATOR<Team> iter = (*it)->iterate_TeamInstanceList(); !iter.done(); iter.advance()) {
489 Team *team = iter.cur();
490 if (!team) {
491 continue;
492 }
493 for (DLINK_ITERATOR<Object> iter = team->iterate_TeamMemberList(); !iter.done(); iter.advance()) {
494 Object *pObj = iter.cur();
495 if (!pObj) {
496 continue;
497 }
498
499 if (types.m_types->isInSet(pObj->getTemplate())) {
500 if (pObj->isInside(pTrig)) {
501
502 //
503 // dead objects will not be considered, except crates ... they are "dead" cause
504 // they have no body and health, but are a class of object we want to
505 // trigger this stuff
506 //
507 if (!(pObj->isEffectivelyDead() || pObj->isKindOf(KINDOF_INERT)) || pObj->isKindOf( KINDOF_CRATE ) ) {
508 count++;
509 }
510 }
511 }
512 }
513 }
514 }
515
516 Bool comparison = false;
517 switch (pComparisonParm->getInt())
518 {
519 case Parameter::LESS_THAN : comparison = (count < pCountParm->getInt()); break;
520 case Parameter::LESS_EQUAL : comparison = (count <= pCountParm->getInt()); break;
521 case Parameter::EQUAL : comparison = (count == pCountParm->getInt()); break;
522 case Parameter::GREATER_EQUAL : comparison = (count >= pCountParm->getInt()); break;
523 case Parameter::GREATER : comparison = (count > pCountParm->getInt()); break;
524 case Parameter::NOT_EQUAL : comparison = (count != pCountParm->getInt()); break;
525 }
526 pCondition->setCustomData(-1); // false.
527 if (comparison) {
528 pCondition->setCustomData(1); // true.
529 }
530 pCondition->setCustomFrame(TheScriptEngine->getFrameObjectCountChanged());
531 return comparison;
532}
