100WBDocUndoable::WBDocUndoable(CWorldBuilderDoc *pDoc, WorldHeightMapEdit *pNewHtMap, Coord3D *pObjOffset):
101 mPNewHeightMapData(NULL),
102 mPOldHeightMapData(NULL),
103 mPDoc(NULL)
104{
105 if (pObjOffset) {
106 m_offsetObjects = true;
107 m_objOffset = *pObjOffset;
108 } else {
109 m_offsetObjects = false;
110 m_objOffset.x = m_objOffset.y = m_objOffset.z = 0;
111 }
112 REF_PTR_SET(mPNewHeightMapData, pNewHtMap);
113 REF_PTR_SET(mPOldHeightMapData, pDoc->GetHeightMap());
114 mPDoc = pDoc; // not ref counted.

117//
119//
120void WBDocUndoable::Do(void)
121{
122 mPDoc->SetHeightMap(mPNewHeightMapData, false); // SetHeightMap but don't inval.
123 if (m_offsetObjects) {
124 MapObject *pCur = MapObject::getFirstMapObject();
125 while (pCur) {
126 Coord3D loc = *pCur->getLocation();
127 loc.x += m_objOffset.x;
128 loc.y += m_objOffset.y;
129 loc.z += m_objOffset.z;
130 pCur->setLocation(&loc);
131 pCur = pCur->getNext();
132 }
133 Int i;
134 // Update build lists.
135 for (i=0; i<TheSidesList->getNumSides(); i++) {
136 SidesInfo *pSide = TheSidesList->getSideInfo(i);
137 for (BuildListInfo *pBuild = pSide->getBuildList(); pBuild; pBuild = pBuild->getNext()) {
138 // Update.
139 Coord3D loc = *pBuild->getLocation();
140 loc.x += m_objOffset.x;
141 loc.y += m_objOffset.y;
142 loc.z += m_objOffset.z;
143 pBuild->setLocation(loc);
144 }
145 }
146 mPDoc->invalObject(NULL); // Inval all objects.
147 }
148 mPNewHeightMapData->dbgVerifyAfterUndo();
149}

230//
232//
233void AddObjectUndoable::Do(void)
234{
235// WorldHeightMapEdit *pMap = m_pDoc->GetHeightMap();
236 MapObject *pCur = m_objectToAdd;
237 MapObject *pLast = NULL;
238
239 // Clear selection.
240 PointerTool::clearSelection();
241 m_numObjects = 0;
242 while (pCur) {
243 if (!pCur->getFlag(FLAG_ROAD_FLAGS) && !pCur->getFlag(FLAG_BRIDGE_FLAGS) && !pCur->isWaypoint())
244 {
245 pCur->setSelected(true);
246 }
247 pLast = pCur;
248 m_numObjects++;
249 pCur = pCur->getNext();
250 }
251 if (pLast==NULL) {
252 return;
253 }
254 pLast->setNextMap(MapObject::getFirstMapObject());
255
256 MapObject::TheMapObjectListPtr = m_objectToAdd;
257 m_addedToList = true;
258
259 pCur = m_objectToAdd;
260 for (int i = 0; i < m_numObjects; i++) {
261 // validate the object, giving it a valid ID and team.
262 pCur->validate();
263 Dict *props = pCur->getProperties();
264 Bool dontCare;
265 TheLayersList->addMapObjectToLayersList(pCur, props->getAsciiString(TheKey_objectLayer, &dontCare));
266
267 m_pDoc->invalObject(pCur);
268 pCur = pCur->getNext();
269 }
270
271 pCur = MapObject::getFirstMapObject();
272
273 const Int flagsWeDontWant = FLAG_ROAD_FLAGS | FLAG_ROAD_CORNER_ANGLED | FLAG_BRIDGE_FLAGS | FLAG_ROAD_CORNER_TIGHT | FLAG_ROAD_JOIN;
274
275 Int numObjects = 0;
276 while (pCur) {
277 Int flags = pCur->getFlags();
278 if (!(pCur->isWaypoint() || (flags & flagsWeDontWant) != 0))
279 ++numObjects;
280
281 pCur = pCur->getNext();
282 }
283
284 if (numObjects >= MAX_OBJECTS_IN_MAP) {
285 CString str, loadStr;
286 loadStr.Format(IDS_MAX_OBJECTS, MAX_OBJECTS_IN_MAP);
287 AfxMessageBox(loadStr, MB_APPLMODAL | MB_ICONEXCLAMATION | MB_OK);
288 }
289}

291//
292// Restore the old mipping values, and inval.
293//
294void AddObjectUndoable::Undo(void)
295{
296// WorldHeightMapEdit *pMap = m_pDoc->GetHeightMap();
297 DEBUG_ASSERTCRASH(m_addedToList,("oops"));
298 DEBUG_ASSERTCRASH(m_objectToAdd == MapObject::TheMapObjectListPtr,("oops"));
299 if (!m_addedToList || m_objectToAdd!=MapObject::TheMapObjectListPtr) {
300 return;
301 }
302 MapObject *pCur = MapObject::TheMapObjectListPtr;
303 while (pCur && (m_numObjects > 1)) {
304 m_pDoc->invalObject(pCur);
305 pCur = pCur->getNext();
306 m_numObjects--;
307 }
308 if ((m_numObjects == 1) && pCur) {
309 MapObject::TheMapObjectListPtr = pCur->getNext();
310 pCur->setNextMap(NULL);
311 m_addedToList = false;
312 }
313 pCur = m_objectToAdd;
314 while (pCur) {
315 m_pDoc->invalObject(pCur);
316 // Note : This undoable can add multiple objects, and removes multiple objects.
317 TheLayersList->removeMapObjectFromLayersList(pCur);
318 pCur = pCur->getNext();
319 }
