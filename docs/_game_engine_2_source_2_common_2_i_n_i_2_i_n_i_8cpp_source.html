156Bool INI::isValidINIFilename( const char *filename )
157{
158 if( filename == NULL )
159 return FALSE;
160
161 Int len = strlen( filename );
162 if( len < 3 )
163 return FALSE;
164
165 if( filename[ len - 1 ] != 'I' && filename[ len - 1 ] != 'i' )
166 return FALSE;
167
168 if( filename[ len - 2 ] != 'N' && filename[ len - 2 ] != 'n' )
169 return FALSE;
170
171 if( filename[ len - 3 ] != 'I' && filename[ len - 3 ] != 'i' )
172 return FALSE;
173
174 return TRUE;
175
176}

217void INI::loadDirectory( AsciiString dirName, Bool subdirs, INILoadType loadType, Xfer *pXfer )
218{
219 // sanity
220 if( dirName.isEmpty() )
221 throw INI_INVALID_DIRECTORY;
222
223 try
224 {
225 FilenameList filenameList;
226 dirName.concat('\\');
227 TheFileSystem->getFileListInDirectory(dirName, "*.ini", filenameList, TRUE);
228 // Load the INI files in the dir now, in a sorted order. This keeps things the same between machines
229 // in a network game.
230 FilenameList::const_iterator it = filenameList.begin();
231 while (it != filenameList.end())
232 {
233 AsciiString tempname;
234 tempname = (*it).str() + dirName.getLength();
235
236 if ((tempname.find('\\') == NULL) && (tempname.find('/') == NULL)) {
237 // this file doesn't reside in a subdirectory, load it first.
238 load( *it, loadType, pXfer );
239 }
240 ++it;
241 }
242
243 it = filenameList.begin();
244 while (it != filenameList.end())
245 {
246 AsciiString tempname;
247 tempname = (*it).str() + dirName.getLength();
248
249 if ((tempname.find('\\') != NULL) || (tempname.find('/') != NULL)) {
250 load( *it, loadType, pXfer );
251 }
252 ++it;
253 }
254 }
255 catch (...)
256 {
257 // propagate the exception
258 throw;
259 }
260
261} // end loadDirectory

265void INI::prepFile( AsciiString filename, INILoadType loadType )
266{
267 // if we have a file open already -- we can't do another one
268 if( m_file != NULL )
269 {
270
271 DEBUG_CRASH(( "INI::load, cannot open file '%s', file already open\n", filename.str() ));
272 throw INI_FILE_ALREADY_OPEN;
273
274 } // end if
275
276 // open the file
277 m_file = TheFileSystem->openFile(filename.str(), File::READ);
278 if( m_file == NULL )
279 {
280
281 DEBUG_CRASH(( "INI::load, cannot open file '%s'\n", filename.str() ));
282 throw INI_CANT_OPEN_FILE;
283
284 } // end if
285
286 m_file = m_file->convertToRAMFile();
287
288 // save our filename
289 m_filename = filename;
290
291 // save our load time
292 m_loadType = loadType;
293}

311static INIBlockParse findBlockParse(const char* token)
312{
313 for (const BlockParse* parse = theTypeTable; parse->token; ++parse)
314 {
315 if (strcmp( parse->token, token ) == 0)
316 {
317 return parse->parse;
318 }
319 }
320 return NULL;
321}

324static INIFieldParseProc findFieldParse(const FieldParse* parseTable, const char* token, int& offset, const void*& userData)
325{
326 for (const FieldParse* parse = parseTable; parse->token; ++parse)
327 {
328 if (strcmp( parse->token, token ) == 0)
329 {
330 offset = parse->offset;
331 userData = parse->userData;
332 return parse->parse;
333 }
334 }
335
336 if (!parse->token && parse->parse)
337 {
338 offset = parse->offset;
339 userData = token;
340 return parse->parse;
341 }
342 else
343 {
344 return NULL;
345 }
346}

351void INI::load( AsciiString filename, INILoadType loadType, Xfer *pXfer )
352{
353 setFPMode(); // so we have consistent Real values for GameLogic -MDC
354
355 s_xfer = pXfer;
356 prepFile(filename, loadType);
357
358 try
359 {
360
361 // read all lines in the file
362 DEBUG_ASSERTCRASH( m_endOfFile == FALSE, ("INI::load, EOF at the beginning!\n") );
363 while( m_endOfFile == FALSE )
364 {
365 // read this line
366 readLine();
367
368 AsciiString currentLine = m_buffer;
369
370 // the first word is the type of data we're processing
371 const char *token = strtok( m_buffer, m_seps );
372 if( token )
373 {
374 INIBlockParse parse = findBlockParse(token);
375 if (parse)
376 {
377 #if defined(_DEBUG) || defined(_INTERNAL)
378 strcpy(m_curBlockStart, m_buffer);
379 #endif
380 try {
381 (*parse)( this );
382
383 } catch (...) {
384 DEBUG_CRASH(("Error parsing block '%s' in INI file '%s'\n", token, m_filename.str()) );
385 char buff[1024];
386 sprintf(buff, "Error parsing INI file '%s' (Line: '%s')\n", m_filename.str(), currentLine.str());
387
388 throw INIException(buff);
389 }
390 #if defined(_DEBUG) || defined(_INTERNAL)
391 strcpy(m_curBlockStart, "NO_BLOCK");
392 #endif
393 }
394 else
395 {
396 DEBUG_ASSERTCRASH( 0, ("[LINE: %d - FILE: '%s'] Unknown block '%s'\n",
397 getLineNum(), getFilename().str(), token ) );
398 throw INI_UNKNOWN_TOKEN;
399 }
400
401 } // end if
402
403 } // end while
404 }
405 catch (...)
406 {
407 unPrepFile();
408
409 // propagate the exception.
410 throw;
411 }
412
413 unPrepFile();
414
415} // end load

421void INI::readLine( void )
422{
423 // sanity
424 DEBUG_ASSERTCRASH( m_file, ("readLine(), file pointer is NULL\n") );
425
426 if (m_endOfFile)
427 *m_buffer=0;
428 else
429 {
430 char *p=m_buffer;
431 while (p!=m_buffer+INI_MAX_CHARS_PER_LINE)
432 {
433 // get next character
434 if (m_readBufferNext==m_readBufferUsed)
435 {
436 // refill buffer
437 m_readBufferNext=0;
438 m_readBufferUsed=m_file->read(m_readBuffer,INI_READ_BUFFER);
439
440 // EOF?
441 if (!m_readBufferUsed)
442 {
443 m_endOfFile=true;
444 *p=0;
445 break;
446 }
447 }
448 *p=m_readBuffer[m_readBufferNext++];
449
450 // CR?
451 if (*p=='\n')
452 {
453 *p=0;
454 break;
455 }
456
457 DEBUG_ASSERTCRASH(*p != '\t', ("tab characters are not allowed in INI files (%s). please check your editor settings. Line Number %d\n",m_filename.str(), getLineNum()));
458
459 // comment?
460 if (*p==';')
461 *p=0;
462 // whitespace?
463 else if (*p>0&&*p<32)
464 *p=' ';
465 p++;
466 }
467 *p=0;
468
469 // increase our line count
470 m_lineNum++;
471
472 // check for at the max
473 if ( p == m_buffer+INI_MAX_CHARS_PER_LINE )
474 {
475
476 DEBUG_ASSERTCRASH( 0, ("Buffer too small (%d) and was truncated, increase INI_MAX_CHARS_PER_LINE\n",
477 INI_MAX_CHARS_PER_LINE) );
478
479 } // end if
480 }
481
482 if (s_xfer)
483 {
484 s_xfer->xferUser( m_buffer, sizeof( char ) * strlen( m_buffer ) );
485 //DEBUG_LOG(("Xfer val is now 0x%8.8X in %s, line %s\n", ((XferCRC *)s_xfer)->getCRC(),
486 //m_filename.str(), m_buffer));
487 }
488}

715AsciiString INI::getNextQuotedAsciiString()
716{
717 AsciiString result;
718 char buff[INI_MAX_CHARS_PER_LINE];
719
720 const char *token = getNextTokenOrNull(); // if null, just leave an empty string
721 if (token != NULL)
722 {
723 if (token[0] != '\"')
724 {
725 // if token is simply "
726 result.set( token ); // Start following the "
727 }
728 else
729 { int strLen=0;
730 Bool done=FALSE;
731 if ((strLen=strlen(token)) > 1)
732 {
733 strcpy(buff, &token[1]); //skip the starting quote
734 //Check for end of quoted string. Checking here fixes cases where quoted string on same line with other data.
735 if (buff[strLen-2]=='"') //skip ending quote if present
736 { buff[strLen-2]='\0';
737 done=TRUE;
738 }
739 }
740
741 if (!done)
742 {
743 token = getNextToken(getSepsQuote());
744
745 if (strlen(token) > 1 && token[1] != '\t')
746 {
747 strcat(buff, " ");
748 strcat(buff, token);
749 }
750 else
751 { Int buflen=strlen(buff);
752 if (buff[buflen-1]=='\"')
753 buff[buflen-1]='\0';
754 }
755 }
756 result.set(buff);
757 }
758 }
759 return result;
760}

764AsciiString INI::getNextAsciiString()
765{
766 AsciiString result;
767
768 const char *token = getNextTokenOrNull(); // if null, just leave an empty string
769 if (token != NULL)
770 {
771 if (token[0] != '\"')
772 {
773 // if token is simply "
774 result.set( token ); // Start following the "
775 }
776 else
777 {
778 static char buff[INI_MAX_CHARS_PER_LINE];
779 buff[0] = 0;
780 if (strlen(token) > 1)
781 {
782 strcpy(buff, &token[1]);
783 }
784
785 token = getNextTokenOrNull(getSepsQuote());
786 if (token) {
787 if (strlen(token) > 1 && token[1] != '\t')
788 {
789 strcat(buff, " ");
790 }
791 strcat(buff, token);
792 result.set(buff);
793 } else {
794 Int len = strlen(buff);
795 if (len && buff[len-1] == '"') { // strip off trailing quote jba. [2/12/2003]
796 buff[len-1] = 0;
797 }
798 result.set(buff);
799 }
800 }
801 }
802 return result;
803}

902//-------------------------------------------------------------------------------------------------
903void INI::parseBitString32( INI* ini, void * /*instance*/, void *store, const void* userData )
904{
905 ConstCharPtrArray flagList = (ConstCharPtrArray)userData;
906 UnsignedInt *bits = (UnsignedInt *)store;
907
908 if( flagList == NULL || flagList[ 0 ] == NULL)
909 {
910 DEBUG_ASSERTCRASH( flagList, ("INTERNAL ERROR! parseBitString32: No flag list provided!\n") );
911 throw INI_INVALID_NAME_LIST;
912 }
913
914 Bool foundNormal = false;
915 Bool foundAddOrSub = false;
916
917 // loop through all tokens
918 for (const char *token = ini->getNextTokenOrNull(); token != NULL; token = ini->getNextTokenOrNull())
919 {
920 if (stricmp(token, "NONE") == 0)
921 {
922 if (foundNormal || foundAddOrSub)
923 {
924 DEBUG_CRASH(("you may not mix normal and +- ops in bitstring lists"));
925 throw INI_INVALID_NAME_LIST;
926 }
927 *bits = 0;
928 break;
929 }
930
931 if (token[0] == '+')
932 {
933 if (foundNormal)
934 {
935 DEBUG_CRASH(("you may not mix normal and +- ops in bitstring lists"));
936 throw INI_INVALID_NAME_LIST;
937 }
938 Int bitIndex = INI::scanIndexList(token+1, flagList); // this throws if the token is not found
939 *bits |= (1 << bitIndex);
940 foundAddOrSub = true;
941 }
942 else if (token[0] == '-')
943 {
944 if (foundNormal)
945 {
946 DEBUG_CRASH(("you may not mix normal and +- ops in bitstring lists"));
947 throw INI_INVALID_NAME_LIST;
948 }
949 Int bitIndex = INI::scanIndexList(token+1, flagList); // this throws if the token is not found
950 *bits &= ~(1 << bitIndex);
951 foundAddOrSub = true;
952 }
953 else
954 {
955 if (foundAddOrSub)
956 {
957 DEBUG_CRASH(("you may not mix normal and +- ops in bitstring lists"));
958 throw INI_INVALID_NAME_LIST;
959 }
960
961 if (!foundNormal)
962 *bits = 0;
963
964 Int bitIndex = INI::scanIndexList(token, flagList); // this throws if the token is not found
965 *bits |= (1 << bitIndex);
966 foundNormal = true;
967 }
968 }
969}

1480 {
1481 DEBUG_ASSERTCRASH( 0, ("INI::initFromINI - Invalid parameters supplied!\n") );
1482 throw INI_INVALID_PARAMS;
1483 }
1484
1485 // read each of the data fields
1486 while( !done )
1487 {
1488
1489 // read next line
1490 readLine();
1491
1492 // check for end token
1493 const char* field = strtok( m_buffer, INI::getSeps() );
1494 if( field )
1495 {
1496
1497 if( stricmp( field, m_blockEndToken ) == 0 )
1498 {
1499 done = TRUE;
1500 }
1501 else
1502 {
1503 Bool found = false;
1504 for (int ptIdx = 0; ptIdx < parseTableList.getCount(); ++ptIdx)
1505 {
1506 int offset = 0;
1507 void* userData = 0;
1508 INIFieldParseProc parse = findFieldParse(parseTableList.getNthFieldParse(ptIdx), field, offset, userData);
1509 if (parse)
1510 {
1511 // parse this block and check for parse errors
1512 try {
1513
1514 (*parse)( this, what, (char *)what + offset + parseTableList.getNthExtraOffset(ptIdx), userData );
1515
1516 } catch (...) {
1517 DEBUG_CRASH( ("[LINE: %d - FILE: '%s'] Error reading field '%s' of block '%s'\n",
1518 INI::getLineNum(), INI::getFilename().str(), field, m_curBlockStart) );
1519
1520
1521 char buff[1024];
1522 sprintf(buff, "[LINE: %d - FILE: '%s'] Error reading field '%s'\n", INI::getLineNum(), INI::getFilename().str(), field);
1523 throw INIException(buff);
1524 }
1525
1526 found = true;
1527 break;
1528
1529 }
1530 }
1531
1532 if (!found)
1533 {
1534 DEBUG_ASSERTCRASH( 0, ("[LINE: %d - FILE: '%s'] Unknown field '%s' in block '%s'\n",
1535 INI::getLineNum(), INI::getFilename().str(), field, m_curBlockStart) );
1536 throw INI_UNKNOWN_TOKEN;
1537 }
1538
1539 } // end else
1540
1541 } // end if
1542
1543 // sanity check for reaching end of file with no closing end token
1544 if( done == FALSE && INI::isEOF() == TRUE )
1545 {
1546
1547 done = TRUE;

1549 m_curBlockStart, getFilename().str(), m_blockEndToken) );

1552 } // end if

1554 } // end while
