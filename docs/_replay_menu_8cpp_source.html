89UnicodeString GetReplayFilenameFromListbox(GameWindow *listbox, Int index)
90{
91 UnicodeString fname = GadgetListBoxGetText(listbox, index);
92
93 if (fname == TheGameText->fetch("GUI:LastReplay"))
94 {
95 fname.translate(TheRecorder->getLastReplayFileName());
96 }
97
98 UnicodeString ext;
99 ext.translate(TheRecorder->getReplayExtention());
100 fname.concat(ext);
101
102 return fname;
103}

109void PopulateReplayFileListbox(GameWindow *listbox)
110{
111 if (!TheMapCache)
112 return;
113
114 GadgetListBoxReset(listbox);
115
116 enum {
117 COLOR_SP = 0,
118 COLOR_SP_CRC_MISMATCH,
119 COLOR_MP,
120 COLOR_MP_CRC_MISMATCH,
121 COLOR_MAX
122 };
123 Color colors[COLOR_MAX] = {
124 GameMakeColor( 255, 255, 255, 255 ),
125 GameMakeColor( 128, 128, 128, 255 ),
126 GameMakeColor( 255, 255, 255, 255 ),
127 GameMakeColor( 128, 128, 128, 255 )
128 };
129
130 AsciiString asciistr;
131 AsciiString asciisearch;
132 asciisearch = "*";
133 asciisearch.concat(TheRecorder->getReplayExtention());
134
135 FilenameList replayFilenames;
136 FilenameListIter it;
137
138 TheFileSystem->getFileListInDirectory(TheRecorder->getReplayDir(), asciisearch, replayFilenames, TRUE);
139
140 TheMapCache->updateCache();
141
142
143 for (it = replayFilenames.begin(); it != replayFilenames.end(); ++it)
144 {
145 // just want the filename
146 asciistr.set((*it).reverseFind('\\') + 1);
147
148 // lets get some info about the replay
149 RecorderClass::ReplayHeader header;
150 header.forPlayback = FALSE;
151 header.filename = asciistr;
152 Bool success = TheRecorder && TheMapCache && TheRecorder->readReplayHeader( header );
153 if (success)
154 {
155 ReplayGameInfo info;
156 if (ParseAsciiStringToGameInfo( &info, header.gameOptions ))
157 {
158
159 // columns are: name, date, version, map, extra
160
161 // name
162 header.replayName.translate(asciistr);
163 for (Int tmp=0; tmp < TheRecorder->getReplayExtention().getLength(); ++tmp)
164 header.replayName.removeLastChar();
165
166 UnicodeString replayNameToShow = header.replayName;
167
168 AsciiString lastReplayFName = TheRecorder->getLastReplayFileName();
169 lastReplayFName.concat(TheRecorder->getReplayExtention());
170 if (lastReplayFName.compareNoCase(asciistr) == 0)
171 replayNameToShow = TheGameText->fetch("GUI:LastReplay");
172
173 UnicodeString displayTimeBuffer = getUnicodeTimeBuffer(header.timeVal);
174
175 //displayTimeBuffer.format( L"%ls", timeBuffer);
176
177 // version (no-op)
178
179 // map
180 UnicodeString mapStr;
181 const MapMetaData *md = TheMapCache->findMap(info.getMap());
182 if (!md)
183 {
184 mapStr.translate(info.getMap());
185 }
186 else
187 {
188 mapStr = md->m_displayName;
189 }
190
191// // extra
192// UnicodeString extraStr;
193// if (header.localPlayerIndex >= 0)
194// {
195// // MP game
196// time_t totalSeconds = header.endTime - header.startTime;
197// Int mins = totalSeconds/60;
198// Int secs = totalSeconds%60;
199// Real fps = header.frameDuration/totalSeconds;
200// extraStr.format(L"%d:%d (%g fps) %hs", mins, secs, fps, header.desyncGame?"OOS ":"");
201//
202// for (Int i=0; i<MAX_SLOTS; ++i)
203// {
204// const GameSlot *slot = info.getConstSlot(i);
205// if (slot && slot->isHuman())
206// {
207// if (i)
208// extraStr.concat(L", ");
209// if (header.playerDiscons[i])
210// extraStr.concat(L'*');
211// extraStr.concat(info.getConstSlot(i)->getName());
212// }
213// }
214// }
215// else
216// {
217// // solo game
218// time_t totalSeconds = header.endTime - header.startTime;
219// Int mins = totalSeconds/60;
220// Int secs = totalSeconds%60;
221// Real fps = header.frameDuration/totalSeconds;
222// extraStr.format(L"%d:%d (%g fps)", mins, secs, fps);
223// }
224
225 // pick a color
226 Color color;
227 if (header.versionString == TheVersion->getUnicodeVersion() && header.versionNumber == TheVersion->getVersionNumber() &&
228 header.exeCRC == TheGlobalData->m_exeCRC && header.iniCRC == TheGlobalData->m_iniCRC)
229 {
230 // good version
231 if (header.localPlayerIndex >= 0)
232 {
233 // MP
234 color = colors[COLOR_MP];
235 }
236 else
237 {
238 // SP
239 color = colors[COLOR_SP];
240 }
241 }
242 else
243 {
244 // bad version
245 if (header.localPlayerIndex >= 0)
246 {
247 // MP
248 color = colors[COLOR_MP_CRC_MISMATCH];
249 }
250 else
251 {
252 // SP
253 color = colors[COLOR_SP_CRC_MISMATCH];
254 }
255 }
256
257 Int insertionIndex = GadgetListBoxAddEntryText(listbox, replayNameToShow, color, -1, 0);
258 GadgetListBoxAddEntryText(listbox, displayTimeBuffer, color, insertionIndex, 1);
259 GadgetListBoxAddEntryText(listbox, header.versionString, color, insertionIndex, 2);
260 GadgetListBoxAddEntryText(listbox, mapStr, color, insertionIndex, 3);
261 //GadgetListBoxAddEntryText(listbox, extraStr, color, insertionIndex, 4);
262 }
263 }
264 }
265 GadgetListBoxSetSelected(listbox, 0);
266}

375{
376
377 switch( msg )
378 {
379
380 // --------------------------------------------------------------------------------------------

382 {

384 UnsignedByte state = mData2;
385
386 switch( key )
387 {
388
389 // ----------------------------------------------------------------------------------------
390 case KEY_ESC:
391 {
392
393 //
394 // send a simulated selected event to the parent window of the
395 // back/exit button
396 //
397 if( BitTest( state, KEY_STATE_UP ) )
398 {
399
400 TheWindowManager->winSendSystemMsg( window, GBM_SELECTED,
401 (WindowMsgData)buttonBack, buttonBackID );
402
403 } // end if
404
405 // don't let key fall through anywhere else
406 return MSG_HANDLED;
407
408 } // end escape
409
410 } // end switch( key )
411
412 } // end char
413
414 } // end switch( msg )
415
416 return MSG_IGNORED;

418} // end ReplayMenuInput

618{
