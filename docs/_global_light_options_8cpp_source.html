52static void calcNewLight(Int lr, Int fb, Vector3 *newLight)
53{
54 newLight->Set(0,0,-1);
55 Real yAngle = PI*(lr-90)/180;
56 Real xAngle = PI*(fb-90)/180;
57 Real zAngle = xAngle * WWMath::Sin(yAngle);
58 xAngle *= WWMath::Cos(yAngle);
59 newLight->Rotate_Y(yAngle);
60 newLight->Rotate_X(xAngle);
61 newLight->Rotate_Z(zAngle);
62}

541void GlobalLightOptions::stuffValuesIntoFields(Int lightIndex)
542{
543 const GlobalData::TerrainLighting *tl;
544 if (m_lighting == K_OBJECTS || m_lighting == K_BOTH) {
545 tl = &TheGlobalData->m_terrainObjectsLighting[TheGlobalData->m_timeOfDay][lightIndex];
546 } else {
547 tl = &TheGlobalData->m_terrainLighting[TheGlobalData->m_timeOfDay][lightIndex];
548 }
549 Real azimuth = 90;
550 Real elevation = 90;
551
552 Vector3 light(tl->lightPos.x, tl->lightPos.y, tl->lightPos.z);
553 light.Normalize();
554
555
556 Real angleAzimuth = atan2(light.Y,light.X);//WWMath::Asin(light.Y);
557 azimuth = angleAzimuth*180.0f/PI;//90-(angleFB/PI)*180;
558 if (azimuth < 0) {
559 azimuth += 360;
560 }
561 Real angleElevation = acos(light.Z);//WWMath::Asin(light.X);
562 elevation = (angleElevation-PI/2.0f)*180.0f/PI;//90-(angleLR/PI)*180;
563
564 m_angleElevation[lightIndex] = elevation;
565 m_angleAzimuth[lightIndex] = azimuth;
566
567 updateEditFields();
568
569 m_updating = true;
570 CString str;
571 str.Format("XYZ: %.2f, %.2f, %.2f", light.X, light.Y, light.Z);
572 CWnd *pWnd = this->GetDlgItem(IDC_XYZ_STATIC);
573 if (pWnd && lightIndex==0) {
574 pWnd->SetWindowText(str);
575 }
576
577 switch (lightIndex)
578 {
579 case K_SUN:
580 default:
581 PutInt(IDC_RA_EDIT, PercentToComponent(tl->ambient.red));
582 PutInt(IDC_GA_EDIT, PercentToComponent(tl->ambient.green));
583 PutInt(IDC_BA_EDIT, PercentToComponent(tl->ambient.blue));
584
585 PutInt(IDC_RD_EDIT, PercentToComponent(tl->diffuse.red));
586 PutInt(IDC_GD_EDIT, PercentToComponent(tl->diffuse.green));
587 PutInt(IDC_BD_EDIT, PercentToComponent(tl->diffuse.blue));
588 m_colorButton.setColor(tl->ambient);
589 break;
590
591 case K_ACCENT1:
592 PutInt(IDC_RD_EDIT1, PercentToComponent(tl->diffuse.red));
593 PutInt(IDC_GD_EDIT1, PercentToComponent(tl->diffuse.green));
594 PutInt(IDC_BD_EDIT1, PercentToComponent(tl->diffuse.blue));
595 break;
596
597 case K_ACCENT2:
598 PutInt(IDC_RD_EDIT2, PercentToComponent(tl->diffuse.red));
599 PutInt(IDC_GD_EDIT2, PercentToComponent(tl->diffuse.green));
600 PutInt(IDC_BD_EDIT2, PercentToComponent(tl->diffuse.blue));
601 break;
602 }
603
604 m_updating = false;
605 pWnd = GetDlgItem(IDC_TIME_OF_DAY_CAPTION);
606 if (pWnd) {
607 switch (TheGlobalData->m_timeOfDay) {
608 default:
609 case TIME_OF_DAY_MORNING: pWnd->SetWindowText("Time of day: Morning."); break;
610 case TIME_OF_DAY_AFTERNOON: pWnd->SetWindowText("Time of day: Afternoon."); break;
611 case TIME_OF_DAY_EVENING: pWnd->SetWindowText("Time of day: Evening."); break;
612 case TIME_OF_DAY_NIGHT: pWnd->SetWindowText("Time of day: Night."); break;
613 }
614 }
615 showLightFeedback(lightIndex);
616}

701 tl.ambient.green = 0;
702 tl.ambient.blue = 0;

706 tl.diffuse.green = TheGlobalData->m_terrainDiffuse[lightIndex].green;
