59void EyedropperTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
60{
61 if (m != TRACK_L) return;
62
63 Coord3D cpt;
64 pView->viewToDocCoords(viewPt, &cpt);
65
66 CPoint ndx;
67 if (!pDoc->getCellIndexFromCoord(cpt, &ndx))
68 return;
69
70 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
71 Int texClass = pMap->getTextureClass(ndx.x, ndx.y, true);
72 TerrainMaterial::setFgTexClass(texClass);
73}
