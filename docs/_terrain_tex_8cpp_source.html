96int TerrainTextureClass::update(WorldHeightMap *htMap)
97{
98 // D3DTexture is our texture;
99
100 IDirect3DSurface8 *surface_level;
101 D3DSURFACE_DESC surface_desc;
102 D3DLOCKED_RECT locked_rect;
103 DX8_ErrorCode(Peek_D3D_Texture()->GetSurfaceLevel(0, &surface_level));
104 DX8_ErrorCode(surface_level->GetDesc(&surface_desc));
105 if (surface_desc.Width < TEXTURE_WIDTH) {
106 return 0;
107 }
108
109 DX8_ErrorCode(surface_level->LockRect(&locked_rect, NULL, 0));
110
111 Int tilePixelExtent = TILE_PIXEL_EXTENT;
112 Int tilesPerRow = surface_desc.Width/(2*TILE_PIXEL_EXTENT+TILE_OFFSET);
113 tilesPerRow *= 2;
114// Int numRows = surface_desc.Height/(tilePixelExtent+TILE_OFFSET);
115#ifdef _DEBUG
116 //DEBUG_ASSERTCRASH(tilesPerRow*numRows >= htMap->m_numBitmapTiles, ("Too many tiles."));
117 DEBUG_ASSERTCRASH((Int)surface_desc.Width >= tilePixelExtent*tilesPerRow, ("Bitmap too small."));
118#endif
119 if (surface_desc.Format == D3DFMT_A1R5G5B5) {
120#if 0
121 UnsignedInt cellX, cellY;
122 for (cellX = 0; cellX < surface_desc.Width; cellX++) {
123 for (cellY = 0; cellY < surface_desc.Height; cellY++) {
124 UnsignedByte *pBGR = ((UnsignedByte *)locked_rect.pBits)+(cellY*surface_desc.Width+cellX)*2;
125 *((Short*)pBGR) = (((255-2*cellY)>>3)<<10) + ((4*cellX)>>4);
126 }
127 }
128#endif
129 Int tileNdx;
130 Int pixelBytes = 2;
131 for (tileNdx=0; tileNdx < htMap->m_numBitmapTiles; tileNdx++) {
132 TileData *pTile = htMap->getSourceTile(tileNdx);
133 if (!pTile) continue;
134 ICoord2D position = pTile->m_tileLocationInTexture;
135 if (position.x<=0) continue; // all real tile offsets start at 2. jba.
136
137 Int i,j;
138 for (j=0; j<tilePixelExtent; j++) {
139 UnsignedByte *pBGR = pTile->getRGBDataForWidth(tilePixelExtent);
140 pBGR += (tilePixelExtent-1-j)*TILE_BYTES_PER_PIXEL*tilePixelExtent; // invert to match.
141 Int row = position.y+j;
142 UnsignedByte *pBGRX = ((UnsignedByte*)locked_rect.pBits) +
143 (row)*surface_desc.Width*pixelBytes;
144
145 Int column = position.x;
146 pBGRX += column*pixelBytes;
147 for (i=0; i<tilePixelExtent; i++) {
148 *((Short*)pBGRX) = 0x8000 + ((pBGR[2]>>3)<<10) + ((pBGR[1]>>3)<<5) + (pBGR[0]>>3);
149 pBGRX +=pixelBytes;
150 pBGR +=TILE_BYTES_PER_PIXEL;
151 }
152 }
153 }
154 // Now draw the 4 pixel border around each tile class.
155 Int texClass;
156 for (texClass=0; texClass<htMap->m_numTextureClasses; texClass++) {
157 Int width = htMap->m_textureClasses[texClass].width;
158 ICoord2D origin = htMap->m_textureClasses[texClass].positionInTexture;
159 if (origin.x<=0) continue;
160 width *= TILE_PIXEL_EXTENT;
161 // Duplicate 4 columns of pixels before and after.
162 Int j;
163 for (j=0; j<width; j++) {
164 Int row = origin.y+j;
165 UnsignedByte *pBGRX = ((UnsignedByte*)locked_rect.pBits) +
166 (row)*surface_desc.Width*pixelBytes;
167
168 Int column = origin.x;
169 pBGRX += column*pixelBytes;
170 // copy before
171 memcpy(pBGRX-(4)*pixelBytes, pBGRX+(width-4)*pixelBytes, 4*pixelBytes);
172 // copy after
173 memcpy(pBGRX+(width*pixelBytes), pBGRX, 4*pixelBytes);
174 }
175
176 // Duplicate 4 rows of pixels before and after.
177 for (j=0; j<4; j++) {
178 // copy before.
179 Int row = origin.y-j-1;
180 UnsignedByte *pBGRX = ((UnsignedByte*)locked_rect.pBits) +
181 (row)*surface_desc.Width*pixelBytes;
182 UnsignedByte *target = pBGRX+(origin.x-4)*pixelBytes;
183 memcpy(target, target+width*surface_desc.Width*pixelBytes, (width+8)*pixelBytes);
184 // copy after.
185 row = origin.y+j;
186 pBGRX = ((UnsignedByte*)locked_rect.pBits) +
187 (row)*surface_desc.Width*pixelBytes;
188 target = pBGRX+(origin.x-4)*pixelBytes;
189 memcpy(target+width*surface_desc.Width*pixelBytes, target, (width+8)*pixelBytes);
190 }
191
192 }
193
194 }
195 surface_level->UnlockRect();
196 surface_level->Release();
197 DX8_ErrorCode(D3DXFilterTexture(Peek_D3D_Texture(), NULL, 0, D3DX_FILTER_BOX));
198 if (TheWritableGlobalData->m_textureReductionFactor) {
199 Peek_D3D_Texture()->SetLOD(TheWritableGlobalData->m_textureReductionFactor);
200 }
201 return(surface_desc.Height);
202}

507void AlphaTerrainTextureClass::Apply(unsigned int stage)
508{
509 // Do the base apply.
510 TextureClass::Apply(stage);
511
512 // Set the bilinear or trilinear filtering.
513 if (TheGlobalData && TheGlobalData->m_bilinearTerrainTex || TheGlobalData->m_trilinearTerrainTex) {
514 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MINFILTER, D3DTEXF_LINEAR);
515 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MAGFILTER, D3DTEXF_LINEAR);
516 } else {
517 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MINFILTER, D3DTEXF_POINT);
518 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MAGFILTER, D3DTEXF_POINT);
519 }
520 if (TheGlobalData && TheGlobalData->m_trilinearTerrainTex) {
521 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MIPFILTER, D3DTEXF_LINEAR);
522 } else {
523 DX8Wrapper::Set_DX8_Texture_Stage_State(stage, D3DTSS_MIPFILTER, D3DTEXF_POINT);
524 }
525 // Since we are using multiple distinct tiles, the textures doesn't wrap, so clamp it.
526 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ADDRESSU, D3DTADDRESS_CLAMP);
527 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ADDRESSV, D3DTADDRESS_CLAMP);
528 // Now setup the texture pipeline.
529 if (stage==0) {
530 // Modulate the diffuse color with the texture as lighting comes from diffuse.
531 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLORARG1, D3DTA_TEXTURE );
532 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLORARG2, D3DTA_DIFFUSE );
533 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLOROP, D3DTOP_MODULATE );
534 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ALPHAOP, D3DTOP_MODULATE );
535 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_TEXCOORDINDEX, 1 );
536 // Blend the result using the alpha. (came from diffuse mod texture)
537 DX8Wrapper::Set_DX8_Render_State(D3DRS_ALPHABLENDENABLE,true);
538 DX8Wrapper::Set_DX8_Render_State(D3DRS_SRCBLEND,D3DBLEND_SRCALPHA);
539 DX8Wrapper::Set_DX8_Render_State(D3DRS_DESTBLEND,D3DBLEND_INVSRCALPHA);
540 // Disable stage 2.
541 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLOROP, D3DTOP_DISABLE );
542 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_ALPHAOP, D3DTOP_DISABLE );
543 } else if (stage==1) {
544
545 if (TheGlobalData && !TheGlobalData->m_multiPassTerrain)
546 {
548 //This method is a backdoor specific to Nvidia based cards. It will fail on
549 //other hardware. Allows single pass blend of 2 textures and post modulate diffuse.
550 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLOROP, D3DTOP_MODULATE);
551 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_TEXCOORDINDEX, 0);
552 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLORARG1, D3DTA_TEXTURE);
553 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_COLORARG2, D3DTA_DIFFUSE);
554 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ALPHAOP, D3DTOP_MODULATE);
555 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ALPHAARG1, D3DTA_TEXTURE);
556 DX8Wrapper::Set_DX8_Texture_Stage_State( 0, D3DTSS_ALPHAARG2, D3DTA_DIFFUSE);
557
558 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLOROP, D3DTOP_ADD);
559 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_TEXCOORDINDEX, 1);
560 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLORARG1, D3DTA_DIFFUSE | D3DTA_COMPLEMENT | D3DTA_ALPHAREPLICATE);
561 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLORARG2, D3DTA_DIFFUSE);
562 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_ALPHAOP, D3DTOP_ADD);
563 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_ALPHAARG1, D3DTA_TFACTOR | D3DTA_COMPLEMENT);
564 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_ALPHAARG2, D3DTA_TFACTOR);
