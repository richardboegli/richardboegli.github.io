66void TunnelTracker::iterateContained( ContainIterateFunc func, void *userData, Bool reverse )
67{
68 if (reverse)
69 {
70 // note that this has to be smart enough to handle items in the list being deleted
71 // via the callback function.
72 for(ContainedItemsList::reverse_iterator it = m_containList.rbegin(); it != m_containList.rend(); )
73 {
74 // save the obj...
75 Object* obj = *it;
76
77 // incr the iterator BEFORE calling the func (if the func removes the obj,
78 // the iterator becomes invalid)
79 ++it;
80
81 // call it
82 (*func)( obj, userData );
83 }
84 }
85 else
86 {
87 // note that this has to be smart enough to handle items in the list being deleted
88 // via the callback function.
89 for(ContainedItemsList::iterator it = m_containList.begin(); it != m_containList.end(); )

91 // save the obj...

93
94 // incr the iterator BEFORE calling the func (if the func removes the obj,
95 // the iterator becomes invalid)
96 ++it;
97

99 (*func)( obj, userData );

101 }

111void TunnelTracker::updateNemesis(const Object *target)
112{
113 if (getCurNemesis()==NULL) {
114 if (target) {
115 if (target->isKindOf(KINDOF_VEHICLE) || target->isKindOf(KINDOF_STRUCTURE) ||
116 target->isKindOf(KINDOF_INFANTRY) || target->isKindOf(KINDOF_AIRCRAFT)) {
117 m_curNemesisID = target->getID();
118 m_nemesisTimestamp = TheGameLogic->getFrame();
119 }
120 }
121 } else if (getCurNemesis()==target) {
122 m_nemesisTimestamp = TheGameLogic->getFrame();
123 }
124}

126// ------------------------------------------------------------------------
127Object *TunnelTracker::getCurNemesis(void)
128{
129 if (m_curNemesisID == INVALID_ID) {
130 return NULL;
131 }
132 if (m_nemesisTimestamp + 4*LOGICFRAMES_PER_SECOND < TheGameLogic->getFrame()) {
133 m_curNemesisID = INVALID_ID;
134 return NULL;
135 }
136 Object *target = TheGameLogic->findObjectByID(m_curNemesisID);
137 if (target) {
138 //If the enemy unit is stealthed and not detected, then we can't attack it!
139 if( target->testStatus( OBJECT_STATUS_STEALTHED ) &&
140 !target->testStatus( OBJECT_STATUS_DETECTED ) &&
141 !target->testStatus( OBJECT_STATUS_DISGUISED ) )
142 {
143 target = NULL;
144 }
145 }
146 if (target && target->isEffectivelyDead()) {
147 target = NULL;
148 }
149 if (target == NULL) {
150 m_curNemesisID = INVALID_ID;
151 }
152 return target;
153}

210// ------------------------------------------------------------------------
211void TunnelTracker::onTunnelDestroyed( const Object *deadTunnel )
212{
213 m_tunnelCount--;
214 m_tunnelIDs.remove( deadTunnel->getID() );
215
216 if( m_tunnelCount == 0 )
217 {
218 // Kill everyone in our contain list. Cave in!
219 iterateContained( destroyObject, NULL, FALSE );
220 m_containList.clear();
221 m_containListSize = 0;
222 }
223 else
224 {
225 Object *validTunnel = TheGameLogic->findObjectByID( m_tunnelIDs.front() );
226 // Otherwise, make sure nobody inside remembers the dead tunnel as the one they entered
227 // (scripts need to use so there must be something valid here)
228 for(ContainedItemsList::iterator it = m_containList.begin(); it != m_containList.end(); )
229 {
230 Object* obj = *it;
231 ++it;
232 if( obj->getContainedBy() == deadTunnel )
233 obj->onContainedBy( validTunnel );
234 }
235 }
236}

254// ------------------------------------------------------------------------
255 // heal one object within the tunnel network system
256void TunnelTracker::healObject( Object *obj, void *frames)
257{
258
259 //get the number of frames to heal
260 Real *framesForFullHeal = (Real*)frames;
261
262 // setup the healing damageInfo structure with all but the amount
263 DamageInfo healInfo;
264 healInfo.in.m_damageType = DAMAGE_HEALING;
265 healInfo.in.m_deathType = DEATH_NONE;
266 //healInfo.in.m_sourceID = getObject()->getID();
267
268 // get body module of the thing to heal
269 BodyModuleInterface *body = obj->getBodyModule();
270
271 // if we've been in here long enough ... set our health to max
272 if( TheGameLogic->getFrame() - obj->getContainedByFrame() >= *framesForFullHeal )
273 {
274
275 // set the amount to max just to be sure we're at the top
276 healInfo.in.m_amount = body->getMaxHealth();
277
278 // set max health
279 body->attemptHealing( &healInfo );
280
281 } // end if
282 else
283 {
284 //
285 // given the *whole* time it would take to heal this object, lets pretend that the
286 // object is at zero health ... and give it a sliver of health as if it were at 0 health
287 // and would be fully healed at 'framesForFullHeal'
288 //
289 healInfo.in.m_amount = body->getMaxHealth() / *framesForFullHeal;
290
291 // do the healing
292 body->attemptHealing( &healInfo );
293
294 } // end else
295}

310void TunnelTracker::xfer( Xfer *xfer )
311{
312
313 // version
314 XferVersion currentVersion = 1;
315 XferVersion version = currentVersion;
316 xfer->xferVersion( &version, currentVersion );
317
318 // tunnel object id list
319 xfer->xferSTLObjectIDList( &m_tunnelIDs );
320
321 // contain list count
322 xfer->xferInt( &m_containListSize );
323
324 // contain list data
325 ObjectID objectID;
326 if( xfer->getXferMode() == XFER_SAVE )
327 {
328 ContainedItemsList::const_iterator it;
329
330 for( it = m_containList.begin(); it != m_containList.end(); ++it )
331 {
332
333 objectID = (*it)->getID();
334 xfer->xferObjectID( &objectID );
335
336 } // end for, it
337
338 } // end if, save
339 else
340 {
341
342 for( UnsignedShort i = 0; i < m_containListSize; ++i )
343 {
344
345 xfer->xferObjectID( &objectID );
346 m_xferContainList.push_back( objectID );
347
348 } // end for, i
349
350 } // end else, load
351
352 // tunnel count
353 xfer->xferUnsignedInt( &m_tunnelCount );
354
355} // end xfer

360void TunnelTracker::loadPostProcess( void )
361{
362
363 // sanity, the contain list should be empty until we post process the id list
364 if( m_containList.size() != 0 )
365 {
366
367 DEBUG_CRASH(( "TunnelTracker::loadPostProcess - m_containList should be empty but is not\n" ));
368 throw SC_INVALID_DATA;
369
370 } // end if
371
372 // translate each object ids on the xferContainList into real object pointers in the contain list
373 Object *obj;
374 std::list< ObjectID >::const_iterator it;
375 for( it = m_xferContainList.begin(); it != m_xferContainList.end(); ++it )
376 {
377
378 obj = TheGameLogic->findObjectByID( *it );
379 if( obj == NULL )
380 {
381
382 DEBUG_CRASH(( "TunnelTracker::loadPostProcess - Unable to find object ID '%d'\n", *it ));
383 throw SC_INVALID_DATA;
384
385 } // end if
386
387 // push on the back of the contain list
388 m_containList.push_back( obj );
389
390 // Crap. This is in OpenContain as a fix, but not here.
391 {
392 // remove object from its group (if any)
393 obj->leaveGroup();
394
395 // remove rider from partition manager
396 ThePartitionManager->unRegisterObject( obj );
397
398 // hide the drawable associated with rider
399 if( obj->getDrawable() )
400 obj->getDrawable()->setDrawableHidden( true );
401
402 // remove object from pathfind map
403 if( TheAI )
404 TheAI->pathfinder()->removeObjectFromPathfindMap( obj );
405
406 }
407 } // end for, it
408
409 // we're done with the xfer contain list now
410 m_xferContainList.clear();
411
412} // end loadPostProcess
