184void HelicopterSlowDeathBehavior::beginSlowDeath( const DamageInfo *damageInfo )
185{
186
187 // extending functionality
188 SlowDeathBehavior::beginSlowDeath( damageInfo );
189
190 // stop the current movement sound
191 if (getObject()->getDrawable())
192 {
193 getObject()->getDrawable()->stopAmbientSound();
194 }
195
196 // get the module data
197 const HelicopterSlowDeathBehaviorModuleData *modData = getHelicopterSlowDeathBehaviorModuleData();
198
199 m_deathSound = modData->m_deathSound;
200
201 if (m_deathSound.getEventName().isEmpty() == false)
202 {
203 m_deathSound.setObjectID(getObject()->getID());
204 m_deathSound.setPlayingHandle(TheAudio->addAudioEvent(&m_deathSound));
205 }
206
207
208 // pick a frame we will fly the blade off at
209 m_bladeFlyOffFrame = GameLogicRandomValueReal( modData->m_minBladeFlyOffDelay,
210 modData->m_maxBladeFlyOffDelay );
211
212 // for now, make it always fall to the left
213 m_orbitDirection = ORBIT_DIRECTION_LEFT;
214
215 //
216 // record the current forward angle of the object to be the start angle of our downward
217 // spin circle angle
218 //
219 m_forwardAngle = getObject()->getOrientation();
220
221 // set or forward speed in the spiral orbit speed to that specified
222 m_forwardSpeed = modData->m_spiralOrbitForwardSpeed;
223
224 // start our self spinning at the min self spin rate
225 m_selfSpin = modData->m_minSelfSpin;
226
227 // we will start off changing the self spin towards the MaxSelfSpin
228 m_selfSpinTowardsMax = TRUE;
229
230 // the fall of the helicopter
231 if( !getObject()->getAIUpdateInterface() )
232 {
233 return;
234 }
235 Locomotor *locomotor = getObject()->getAIUpdateInterface()->getCurLocomotor();
236 locomotor->setMaxLift( -TheGlobalData->m_gravity * (1.0f - modData->m_fallHowFast) );
237 locomotor->setMaxBraking( modData->m_maxBraking );
238
239 // attach particle system to bone if present
240 if( modData->m_attachParticleSystem )
241 {
242 ParticleSystem *pSys = TheParticleSystemManager->createParticleSystem( modData->m_attachParticleSystem );
243 if( pSys )
244 {
245
246 // where do the offset attachment to
247 if( modData->m_attachParticleBone.isEmpty() == FALSE )
248 {
249 Drawable *draw = getObject()->getDrawable();
250
251 if( draw )
252 {
253 Coord3D pos;
254
255 if( draw->getPristineBonePositions( modData->m_attachParticleBone.str(), 0, &pos, NULL, 1 ) )
256 pSys->setPosition( &pos );
257
258 } // end if
259
260 } // end if
261 else
262 {
263
264 // use location coord specified ... it will be zero if not given which is center of obj anyway
265 pSys->setPosition( &modData->m_attachParticleLoc );
266
267 } // end else
268
269 // attach the particle system to the object
270 pSys->attachToObject( getObject() );
271
272 } // end if
273
274 } // end if
275
276
277
278} // end beginSlowDeath

285UpdateSleepTime HelicopterSlowDeathBehavior::update( void )
286{
288 // call the base class cause we're extending functionality
289 SlowDeathBehavior::update();
290
291 // get out of here if we're not activated yet
292 if( isSlowDeathActivated() == FALSE )
293 return UPDATE_SLEEP_NONE;
294
295 // get the module data
296 const HelicopterSlowDeathBehaviorModuleData *modData = getHelicopterSlowDeathBehaviorModuleData();
297
298 // get the helicopter for easy access in here
299 Object *copter = getObject();
300
301 // update self spin and orbit if we haven't hit the ground yet
302 if( m_hitGroundFrame == 0 )
303 {
304
305 //
306 // change the angle of the helicoptor based on the spiral rate for the turning on the
307 // large downward spin circle
308 //
309
310 //copter->setOrientation( copter->getOrientation() + m_selfSpin * m_orbitDirection );
311
312 Matrix3D xfrm = *copter->getTransformMatrix();
313 xfrm.In_Place_Pre_Rotate_Z(m_selfSpin * m_orbitDirection);
314 copter->setTransformMatrix( &xfrm );
315
316 //
317 // over time we change the rate at which we self spin around our center of gravity ... we
318 // will ping pong back and forth between the MinSelfSpin and MaxSelfSpin defined in INI
319 //
320 if( modData->m_selfSpinUpdateDelay &&
321 TheGameLogic->getFrame() - m_lastSelfSpinUpdateFrame > modData->m_selfSpinUpdateDelay )
322 {
323
324 // update the self spin
325 if( m_selfSpinTowardsMax == TRUE )
326 {
327
328 // we're going towards the max self spin, increase it
329 m_selfSpin += modData->m_selfSpinUpdateAmount / LOGICFRAMES_PER_SECOND;
330 if( m_selfSpin > modData->m_maxSelfSpin )
331 {
332
333 m_selfSpin = modData->m_maxSelfSpin; // cap at max
334 m_selfSpinTowardsMax = FALSE; // now start changing spin towards min again
335
336 } // end if
337
338 } // end if
339 else
340 {
341
342 // we're going towards the min self spin, decrease it
343 m_selfSpin -= modData->m_selfSpinUpdateAmount / LOGICFRAMES_PER_SECOND;
344 if( m_selfSpin < modData->m_minSelfSpin )
345 {
346
347 m_selfSpin = modData->m_minSelfSpin; // cap at min
348 m_selfSpinTowardsMax = TRUE; // now start chaning spin towards max again
349
350 } // end if
351
352 } // end else
353
354 // we have made a change to the self spinning
355 m_lastSelfSpinUpdateFrame = TheGameLogic->getFrame();
356
357 } // end if
358
359 // get the physics update module
360 PhysicsBehavior *physics = copter->getPhysics();
361 DEBUG_ASSERTCRASH( physics, ("HelicopterSlowDeathBehavior: object '%s' does not have a physics module\n",
362 copter->getTemplate()->getName().str()) );
363
364 //
365 // apply a force to the helicopter pushing it in a forward motion according to the
366 // forward angle & speed we are keeping track of to make our downward circle (that forward angle
367 // is *NOT* the angle the object is facing
368 //
369 Coord3D force;
370 force.x = DOUBLE_TO_REAL( Cos( m_forwardAngle ) ) * m_forwardSpeed;
371 force.y = DOUBLE_TO_REAL( Sin( m_forwardAngle ) ) * m_forwardSpeed;
372 force.z = 0.0f;
373 physics->applyMotiveForce( &force );
374
375 // update our forward angle for travelling along the large spiral downward circle
376 m_forwardAngle += (modData->m_spiralOrbitTurnRate * m_orbitDirection);
377
378 // adjust our forward spiral orbit by the damping factor specified
379 m_forwardSpeed *= modData->m_spiralOrbitForwardSpeedDamping;
380
381 // is it time to have the blade fly off
382 if( m_bladeFlyOffFrame > 0 )
383 {
384 // decrement count
385 m_bladeFlyOffFrame--;
386 if( m_bladeFlyOffFrame <= 0 )
387 {
388 Coord3D bladePos = *copter->getPosition();
389
390 // get the blade position from the bone in the model
391 Drawable *draw = copter->getDrawable();
392 if( draw )
393 {
394
395 draw->getPristineBonePositions( modData->m_bladeBone.str(), 0, &bladePos, NULL, 1 );
396 draw->convertBonePosToWorldPos( &bladePos, NULL, &bladePos, NULL );
397
398 } // end if
399
400 // create the blades flying through the air
401 // const ObjectCreationList *ocl = TheObjectCreationListStore->findObjectCreationList( "OCL_ComancheBlades" );
402 // ObjectCreationList::create( ocl, &bladePos );
403
404 // run the fx at the blade position
405 FXList::doFXPos( modData->m_fxBlade, &bladePos );
406 ObjectCreationList::create( modData->m_oclBlade, copter, &bladePos, NULL, INVALID_ANGLE );
407
408 //
409 // if we have (potentially) a pilot ejection, do it here.
410 // note that we call EjectPilotDie::ejectPilot() rather than ObjectCreationList::create(),
411 // because the former makes the right sounds, and also constrains to veteran-or-better status.
412 //
413 if( modData->m_oclEjectPilot && copter->getVeterancyLevel() > LEVEL_REGULAR )
414 EjectPilotDie::ejectPilot( modData->m_oclEjectPilot, copter, NULL );
415
416 } // end if
417
418 } // endif
419
420 } // end if, not on ground
421
422
423
424 Bool hitATree = FALSE;
425 // Here we want to make sure we crash if we collide with a tree on the way down
426 PhysicsBehavior *phys = copter->getPhysics();
427 if ( m_hitGroundFrame == 0 && phys )
428 {
429 ObjectID treeID = phys->getLastCollidee();
430 Object *tree = TheGameLogic->findObjectByID( treeID );
431 if ( tree )
432 {
433 if (tree->isKindOf( KINDOF_SHRUBBERY ) )
434 hitATree = TRUE;
435 }
436 }
437
438
439
440 // when we hit the ground
441 const Coord3D *pos = copter->getPosition();
442 if (m_hitGroundFrame == 0)
443 {
444 // srj sez: if we haven't yet hit the ground, adjust our layer properly so we crash on bridges
445 Coord3D tmpPt = *pos;
446 tmpPt.z = 99999.0f;
447 PathfindLayerEnum newLayer = TheTerrainLogic->getHighestLayerForDestination(&tmpPt);
448 copter->setLayer(newLayer);
449
450 Real ground = TheTerrainLogic->getLayerHeight( tmpPt.x, tmpPt.y, newLayer );
451
452 if (pos->z <= ground + 1.0f || hitATree )
453 {
454
455 // mark the frame we hit the ground on
456 m_hitGroundFrame = TheGameLogic->getFrame();
457
458 // make hit ground effect
459 FXList::doFXObj( modData->m_fxHitGround, copter );
460 ObjectCreationList::create( modData->m_oclHitGround, copter, NULL );
461
462 // hold the copter in place now
463 copter->setDisabled( DISABLED_HELD );
464
465 // special damage state ... has the blades no on anymore
466 Drawable *draw = copter->getDrawable();
467 if( draw )
468 draw->setModelConditionState( MODELCONDITION_SPECIAL_DAMAGED );
469
470 // Stop the sound from playing.
471 TheAudio->removeAudioEvent(m_deathSound.getPlayingHandle());
472
473 } // end if
474 }
475
476 // if we're on the ground, see if it's time for our final boom
477 if( m_hitGroundFrame &&
478 TheGameLogic->getFrame() - m_hitGroundFrame > modData->m_delayFromGroundToFinalDeath )
479 {
480
481 // make effect
482 FXList::doFXObj( modData->m_fxFinalBlowUp, copter );
483 ObjectCreationList::create( modData->m_oclFinalBlowUp, copter, NULL );
484
485 // we are now all done, destroy us and make a rubble shell copter
486 const ThingTemplate* ttn = TheThingFactory->findTemplate(modData->m_finalRubbleObject);
487 Object *rubble = TheThingFactory->newObject( ttn, copter->getTeam() );
488 if( rubble )
489 {
490
491 rubble->setTransformMatrix( copter->getTransformMatrix() );
492
493 } // end if
494
495 // destroy the copter finally
496 TheGameLogic->destroyObject( copter );
497
498 } // end if
499
500 return UPDATE_SLEEP_NONE;
501
502} // end update
