82static void TransformFile(granny_file_info *FileInfo)
83{
84 float Origin[] = {0, 0, 0};
85 float RightVector[] = {1, 0, 0}; //x
86 float UpVector[] = {0, 0, 1}; //y
87 float BackVector[] = {0, -1, 0};//z
88
89 return;
90
91 float Affine3[3];
92 float Linear3x3[3][3];
93 float InverseLinear3x3[3][3];
94 GrannyAutoComputeBasisConversion(FileInfo, 1.0f,
95 Origin,
96 RightVector,
97 UpVector,
98 BackVector,
99 Affine3, (float *)Linear3x3,
100 (float *)InverseLinear3x3);
101
102 GrannyAutoTransformFile(FileInfo, Affine3,
103 (float *)Linear3x3,
104 (float *)InverseLinear3x3);

253void GrannyRenderObjClass::Set_Animation( HAnimClass * motion, float frame, int anim_mode)
254{
255 GrannyAnimClass *gAnim=(GrannyAnimClass*)motion;
256
257 granny_animation *Animation=gAnim->Animation;
258
259 if(Animation)
260 {
261 Int trackIndex=-1;
262 //Check if this animation works with this model
263 trackIndex=FindTrackGroupFor(Animation,m_modelInstance);
264
265 if(trackIndex != -1)
266 {
267 //stop previous animations
268 if (m_animationControl)
269 { GrannyFreeControl(m_animationControl);
270 m_animationControl=NULL;
271 }
272
273 m_animationControl = GrannyPlayControlledAnimation(frame / gAnim->FrameRate, Animation, trackIndex, m_modelInstance, 0);
274 if (m_animationControl)
275 {
276 GrannySetControlSpeed(m_animationControl, 1.0f); //play at normal speed
277 if (anim_mode == ANIM_MODE_LOOP)
278 GrannySetControlLooping(m_animationControl, true);
279 else
280 GrannySetControlLooping(m_animationControl, false);
281 }
282 }
283 }

292void GrannyRenderObjClass::Render(RenderInfoClass & rinfo)
293{
294 TheGrannyRenderObjSystem->AddRenderObject(rinfo, this);
295 return;
296
297 //update the model
298 GrannySetModelClock(m_modelInstance, LastSyncTime);
299 LastSyncTime = WW3D::Get_Sync_Time()*0.001f; //convert to seconds
300
301 GrannySampleModelAnimations(m_modelInstance, 0, GrannyGetModelBoneCount(m_modelInstance),
302 GrannyGetModelLocalPose(m_modelInstance));
303 GrannyBuildModelWorldTransforms(m_modelInstance, 0,
304 GrannyGetModelBoneCount(m_modelInstance),
305 GrannyGetModelLocalPose(m_modelInstance),
306 GrannyGetModelRootTransform(m_modelInstance),
307 GrannyGetModelWorldPose(m_modelInstance));
308
309 Real *RootTransform = GrannyGetModelRootTransform(m_modelInstance);
310
311 RootTransform[12] = 0; //X
312 RootTransform[13] = 0; //Y
313 RootTransform[13] = 0; //Z
314 RootTransform[0] = ObjectScale;
315 RootTransform[5] = ObjectScale;
316 RootTransform[10] = ObjectScale;
317
318 Int MeshCount = GrannyGetModelMeshCount(m_modelInstance);
319 for(Int MeshIndex = 0; MeshIndex < MeshCount; ++MeshIndex)
320 {
321 granny_mesh_instance *Mesh = (granny_mesh_instance *)GrannyGetModelMeshCookie(m_modelInstance, MeshIndex);
322 granny_mesh_model_binding *Binding = GrannyGetMeshModelBinding(Mesh);
323
324 granny_pnt332_vertex *Vertices = 0;
325 if(GrannyMeshIsRigid(Mesh))
326 { assert(0); //have not coded support for rigid meshes yet - only soft skinned supported.
327// granny_matrix_4x4 TempBuffer;
328// glMultMatrixf(GrannyGetRigidMeshTransform(
329// Binding, GrannyGetModelWorldPose(m_modelInstance),
330// (granny_real32 *)TempBuffer));
331 Vertices = (granny_pnt332_vertex *)GrannyGetMeshVertices(Mesh);
332 }
333 else
334 {
335 GrannyDeformMesh(Binding, GrannyGetModelWorldPose(m_modelInstance),
336 0, g_blendingBuffer);
337 Vertices = g_blendingBuffer;
338 }
339
340 int GroupCount = GrannyGetMeshTriangleGroupCount(Mesh);
341 {for(Int GroupIndex = 0;
342 GroupIndex < GroupCount;
343 ++GroupIndex)
344 {
345 granny_material_instance *Material = (granny_material_instance *)
346 GrannyGetMeshMaterialCookie(
347 Mesh, GrannyGetMeshTriangleGroupMaterialIndex(
348 Mesh, GroupIndex));
349 if(Material)
350 {
351 Int TextureIndex;
352 if(GrannyGetMaterialTextureIndexByType(
353 Material, GrannyDiffuseColorTexture, &TextureIndex))
354 {
355 granny_texture_instance *TextureInstance =
356 (granny_texture_instance *)GrannyGetMaterialTextureCookie(
357 Material, TextureIndex);
358
359 if(TextureInstance)
360 {
361 DX8Wrapper::Set_Texture(0, (TextureClass *)GrannyGetTextureCookie(TextureInstance));
362 }
363 else
364 DX8Wrapper::Set_Texture(0, NULL);
365 }
366 }
367
368 DynamicVBAccessClass vb_access(BUFFER_TYPE_DYNAMIC_DX8,dynamic_fvf_type,m_vertexCount);
369 {
370 DynamicVBAccessClass::WriteLockClass lock(&vb_access);
371 VertexFormatXYZNDUV2* vb=lock.Get_Formatted_Vertex_Array();
372 if(vb)
373 {
374 for (Int i=0; i<m_vertexCount; i++)
375 {
376 vb->x=Vertices->Position[0];
377 vb->y=Vertices->Position[1];
378 vb->z=Vertices->Position[2];
379 vb->nx=Vertices->Normal[0];
380 vb->ny=Vertices->Normal[1];
381 vb->nz=Vertices->Normal[2];
382 vb->diffuse=0xffffffff;
383 vb->u1=Vertices->UV[0];
384 vb->v1=Vertices->UV[1];
385 vb++;
386 Vertices++;
387 }
388 }
389 }
390
391 Int indexCount=GrannyGetMeshTriangleGroupIndexCount(Mesh, GroupIndex);
392 UnsignedInt *indices=(UnsignedInt*)GrannyGetMeshTriangleGroupIndices(Mesh, GroupIndex);
393
394 DynamicIBAccessClass ib_access(BUFFER_TYPE_DYNAMIC_DX8,indexCount);
395 {
396 DynamicIBAccessClass::WriteLockClass lock(&ib_access);
397 unsigned short* ib=lock.Get_Index_Array();
398
399 if(ib)
400 {
401 for (Int i=0; i<indexCount; i++)
402 ib[i]=(UnsignedShort)indices[i];
403 }
404 }
405
406
407 Matrix3D tm(Transform);
408 DX8Wrapper::Set_Light_Environment(rinfo.light_environment);
409 DX8Wrapper::Set_Material(m_prototype.m_vertexMaterial);
410 DX8Wrapper::Set_Shader(ShaderClass::_PresetOpaqueShader);
411 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
412 DX8Wrapper::Set_Index_Buffer(ib_access,0);
413 DX8Wrapper::Set_Vertex_Buffer(vb_access);
414 DX8Wrapper::Draw_Triangles( 0,indexCount/3, 0, m_vertexCount); //draw a quad, 2 triangles, 4 verts
415 }}
416 }
417}

422PrototypeClass * GrannyLoaderClass::Load_W3D(const char *filename)
423{
424 char drive[_MAX_DRIVE],dir[_MAX_DIR],fname[_MAX_FNAME],ext[_MAX_EXT];
425
426 granny_file *File = GrannyReadEntireFile(filename);
427 granny_file_info *fileInfo;
428 AABoxClass box;
429 Int vertexCount=0;
430 Int indexCount=0;
431 Int meshCount=0;
432 GrannyPrototypeClass::grannyMeshDesc meshData[GRANNY_MAX_MESHES_PER_MODEL];
433
434 if(File)
435 {
436 fileInfo = GrannyGetFileInfo(File);
437 if (fileInfo)
438 {
439
440 TransformFile(fileInfo);
441
442 for(int TextureIndex = 0; TextureIndex < fileInfo->TextureCount; ++TextureIndex)
443 {
444 _splitpath(fileInfo->Textures[TextureIndex]->FromFileName, drive, dir, fname, ext );
445 TextureClass *TextureHandle=WW3DAssetManager::Get_Instance()->Get_Texture(strncat(fname,".tga",4));
446 granny_texture_instance *TextureInstance = GrannyInstantiateTexture(fileInfo->Textures[TextureIndex]);
447 GrannySetTextureCookie(TextureInstance, (granny_uint32)TextureHandle);
448 GrannyAddTextureToLibrary(TextureLibrary,TextureInstance);
449 }
450
451 //Calculate bounding box and find maximum number of vertices needed to render mesh.
452 for (Int modelIndex=0; modelIndex<fileInfo->ModelCount; modelIndex++)
453 {
454 granny_model *sourceModel = fileInfo->Models[modelIndex];
455 if (stricmp(sourceModel->Name,"AABOX") == 0)
456 { //found a collision box, copy out data
457 int MeshCount = sourceModel->MeshBindingCount;
458 if (MeshCount==1)
459 {
460 granny_mesh *sourceMesh = sourceModel->MeshBindings[0].Mesh;
461 granny_pn33_vertex *Vertices = (granny_pn33_vertex *)sourceMesh->PrimaryVertexData->Vertices;
462 Vector3 points[24];
463
464 assert (sourceMesh->PrimaryVertexData->VertexCount <= 24);
465
466 for (Int boxVertex=0; boxVertex<sourceMesh->PrimaryVertexData->VertexCount; boxVertex++)
467 { points[boxVertex].Set(Vertices[boxVertex].Position[0],
468 Vertices[boxVertex].Position[1],
469 Vertices[boxVertex].Position[2]);
470 }
471 box.Init(points,sourceMesh->PrimaryVertexData->VertexCount);
472 }
473 }
474 else
475 { //mesh is part of model
476 meshCount = sourceModel->MeshBindingCount;
477
478 for (Int meshIndex=0; meshIndex<meshCount; meshIndex++)
479 {
480 granny_mesh *sourceMesh = sourceModel->MeshBindings[meshIndex].Mesh;
481 meshData[meshIndex].vertex=NULL;
482 meshData[meshIndex].vertexCount=0;
483 if (sourceMesh->PrimaryVertexData)
484 { vertexCount+=sourceMesh->PrimaryVertexData->VertexCount;
485 meshData[meshIndex].vertex=(granny_pnt332_vertex *)sourceMesh->PrimaryVertexData->Vertices;
486 meshData[meshIndex].vertexCount=sourceMesh->PrimaryVertexData->VertexCount;
487 }
488 meshData[meshIndex].index=NULL;
489 meshData[meshIndex].indexCount=NULL;
490 if (sourceMesh->PrimaryTopology)
491 {
492 assert (sourceMesh->PrimaryTopology->GroupCount == 1); //should only have 1 material per mesh!
493 granny_tri_material_group &sourceGroup = sourceMesh->PrimaryTopology->Groups[0];
494
495 // This is the texture index (relative to the array we just
496 // built) for this group of triangles
497 // SourceGroup.MaterialIndex;
498
499 // This is the number of indices
500 if(sourceMesh->PrimaryTopology->IndexCount)
501 {
502 meshData[meshIndex].index=(const UnsignedInt*)&sourceMesh->PrimaryTopology->Indices[3*sourceGroup.TriFirst];
503 // These are the indices for this group
504 meshData[meshIndex].indexCount=3*sourceGroup.TriCount;
505 indexCount += meshData[meshIndex].indexCount; //keep track of total indices in this model
506 }
507 }
508 }
509 }
510 }
512// for(int MaterialIndex = 0; MaterialIndex < fileInfo->MaterialCount; ++MaterialIndex)
513// {
514// granny_material_instance *MaterialInstance = GrannyInstantiateMaterial(fileInfo->Materials[MaterialIndex]);
515// GrannyAutoBindAllMaterialTextures(MaterialInstance,TextureLibrary);
516// GrannyAddMaterialToLibrary(MaterialLibrary,MaterialInstance);
517// }
518
519
520 GrannyAutoBindAllMaterials(MaterialLibrary, fileInfo,TextureLibrary);
521
522 } //fileInfo
523
524 if (fileInfo == NULL) {
525 return NULL;
526 }
527
528 // ok, accept this model!
529 GrannyPrototypeClass * hproto = NEW GrannyPrototypeClass(File);
530 _splitpath(filename, drive, dir, fname, ext );
531 hproto->Set_Name(strcat(fname,ext));
532 hproto->setBoundingBox(box);
533 hproto->setBoundingSphere(SphereClass(box.Center,box.Extent.Length()));
534 hproto->setVertexCount(vertexCount);
535 hproto->setIndexCount(indexCount);
536 hproto->setMeshCount(meshCount);
537 for (Int i=0; i<meshCount; i++)
538 hproto->setMeshData(meshData[i],i);
539 return hproto;
540 }//FILE
541
542 return NULL;
543}

639{
640 GrannyAnimClass * newanim = NEW GrannyAnimClass;
641
642 if (newanim == NULL) {
643 goto Error;
644 }
645
646 SET_REF_OWNER( newanim );
647

649 { // load failed!
650 newanim->Release_Ref();
651 goto Error;
652 } else if (Peek_Anim(newanim->Get_Name()) != NULL)

654 newanim->Release_Ref(); // Release the one we just loaded
655 goto Error;
656 } else
657 { Add_Anim( newanim );
658 newanim->Release_Ref();
659 }
660
661 return 0;
662
663Error:
664 return 1;

723void GrannyRenderObjSystem::AddRenderObject(RenderInfoClass & rinfo, GrannyRenderObjClass *robj)
724{
725 if (m_renderObjectCount >= MAX_VISIBLE_GRANNY_MODELS)
726 { assert (m_renderObjectCount < MAX_VISIBLE_GRANNY_MODELS);
727 return; //can't add any more granny models this frame.
728 }
729
730 for (Int i=0; i<m_renderStateCount; i++)
731 { //search for other objects with same render state
732 //right now we can just sort by prototype since prototypes have fixed materials.
733 if (m_renderStateModelList[i].list->m_prototype.m_file == robj->m_prototype.m_file)
734 { //found model with same prototype. Add new model to same list.
735 robj->m_nextSystem=m_renderStateModelList[i].list;
736 m_renderStateModelList[i].list=robj;
737 if (rinfo.light_environment)
738 m_renderLocalLightEnv[m_renderObjectCount]=*rinfo.light_environment;
739 m_renderObjectCount++;
740 return;
741 }
742 }
743
744 //Found a new render state
745
746 assert (m_renderStateCount < MAX_GRANNY_RENDERSTATES);
747
748 if (m_renderStateCount > MAX_GRANNY_RENDERSTATES)
749 return; //reached maximum number of render states
750
751 //store this objects lighting information for use later during drawing.
752 if (rinfo.light_environment)
753 m_renderLocalLightEnv[m_renderObjectCount]=*rinfo.light_environment;
754
755 m_renderStateModelList[m_renderStateCount++].list=robj;
756 robj->m_nextSystem=NULL;
757
758 m_renderObjectCount++;
759}

764void GrannyRenderObjSystem::Flush(void)
765{
766 GrannyRenderObjClass *robj;
767 granny_model_instance *modelInstance;
768 Bool setMaterial;
769 Int modelCount=0;
770 Int indexCount; //per model index count
771
772 for (Int i=0; i<m_renderStateCount; i++)
773 {
774 robj=m_renderStateModelList[i].list; //set to start of objects using this material
775 setMaterial=true; //force rendering code to apply material
776
777 while (robj)
778 {
779 modelInstance=robj->m_modelInstance;
780
781 //update the model
782 GrannySetModelClock(modelInstance,robj->LastSyncTime);
783 robj->LastSyncTime = WW3D::Get_Sync_Time()*0.001f; //convert to seconds
784
785 GrannySampleModelAnimations(modelInstance, 0, GrannyGetModelBoneCount(modelInstance),
786 GrannyGetModelLocalPose(modelInstance));
787 GrannyBuildModelWorldTransforms(modelInstance, 0,
788 GrannyGetModelBoneCount(modelInstance),
789 GrannyGetModelLocalPose(modelInstance),
790 GrannyGetModelRootTransform(modelInstance),
791 GrannyGetModelWorldPose(modelInstance));
792

905 }//while
