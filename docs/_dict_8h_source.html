275 // danger... this is Plain Old Data and allocated in a skanky way;
276 // and thus the ctor/dtor for DictPair will never be called. so don't
277 // bother writing one.
278 struct DictPair
279 {
280 private:
281 DictPairKeyType m_key;
282 void* m_value;
283
284 inline static DictPairKeyType createKey(NameKeyType keyVal, DataType nt)
285 {
286 return (DictPairKeyType)((((UnsignedInt)(keyVal)) << 8) | ((UnsignedInt)nt));
287 }
288
289 inline static DataType getTypeFromKey(DictPairKeyType nk)
290 {
291 return (DataType)(((UnsignedInt)nk) & 0xff);
292 }
293
294 inline static NameKeyType getNameFromKey(DictPairKeyType nk)
295 {
296 return (NameKeyType)(((UnsignedInt)nk) >> 8);
297 }
298
299
300 public:
301 void clear();
302 void copyFrom(DictPair* that);
303 void setNameAndType(NameKeyType key, DataType type);
304 inline DataType getType() const { return getTypeFromKey(m_key); }
305 inline NameKeyType getName() const { return getNameFromKey(m_key); }
306 inline Bool* asBool() { return (Bool*)&m_value; }
307 inline Int* asInt() { return (Int*)&m_value; }
308 inline Real* asReal() { return (Real*)&m_value; }
309 inline AsciiString* asAsciiString() { return (AsciiString*)&m_value; }
310 inline UnicodeString* asUnicodeString() { return (UnicodeString*)&m_value; }
311 };
