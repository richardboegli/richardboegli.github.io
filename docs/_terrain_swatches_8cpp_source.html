51void TerrainSwatches::OnPaint()
52{
53 CPaintDC dc(this); // device context for painting
54
55 CRect clientRect;
56 GetClientRect(&clientRect);
57
58 CRect bgRect;
59 bgRect = clientRect;
60 bgRect.top = bgRect.bottom-TILE_PIXEL_EXTENT;
61 bgRect.left = bgRect.right-TILE_PIXEL_EXTENT;
62
63 CRect fgRect = clientRect;
64 fgRect.bottom = fgRect.top+TILE_PIXEL_EXTENT;
65 fgRect.right = fgRect.left+TILE_PIXEL_EXTENT;
66
67 CBrush brush;
68 brush.CreateSolidBrush(RGB(0,0,0));
69
70 Int fgTexClass = TerrainMaterial::getFgTexClass();
71 Int bgTexClass = TerrainMaterial::getBgTexClass();
72 UnsignedByte *pData;
73 pData = WorldHeightMapEdit::getPointerToClassTileData(bgTexClass);
74 if (pData) {
75 DrawMyTexture(&dc, bgRect.top, bgRect.left, TILE_PIXEL_EXTENT, pData);
76 } else {
77 dc.FillSolidRect(&bgRect, RGB(0,128,0));
78 }
79 dc.FrameRect(&bgRect, &brush);
80 pData = WorldHeightMapEdit::getPointerToClassTileData(fgTexClass);
81 if (pData) {
82 DrawMyTexture(&dc, fgRect.top, fgRect.left, TILE_PIXEL_EXTENT, pData);
83 } else {
84 dc.FillSolidRect(&fgRect, RGB(128,0,0));
85 }
86 dc.FrameRect(&fgRect, &brush);
87
88
89}
