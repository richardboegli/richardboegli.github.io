66void NeutronBlastBehavior::onDie( const DamageInfo *damageInfo )
67{
68 // On death, perform the Neutron Blast!!
69 Object *self = getObject();
70 if (!self)
71 return;
72
73 const NeutronBlastBehaviorModuleData *data = getNeutronBlastBehaviorModuleData();
74 Real blastRadius = data->m_blastRadius;
75 Bool hitAir = data->m_isAffectAirborne;
76
77 // setup scan filters
78 PartitionFilterSameMapStatus filterMapStatus( self );
79 PartitionFilterAlive filterAlive;
80 PartitionFilter *filters[] = { &filterAlive, &filterMapStatus, NULL };
81
82 // scan objects in our region
83 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( self->getPosition(), blastRadius, FROM_CENTER_2D, filters );
84 MemoryPoolObjectHolder hold( iter );
85
86 // Apply neutron blast to object
87 for( Object *obj = iter->first(); obj; obj = iter->next() )
88 {
89 if( hitAir || ( !obj->isKindOf(KINDOF_AIRCRAFT) && !obj->isAirborneTarget() ) )
90 {
91 neutronBlastToObject( obj );
92 }
93 }
94}

108void NeutronBlastBehavior::neutronBlastToObject( Object *obj )
109{
110 // early exit check
111 if ( !obj || obj == getObject() )
112 return;
113
114 // Check for allies and quick exit if we are not suppose to hurt our own.
115 const NeutronBlastBehaviorModuleData *data = getNeutronBlastBehaviorModuleData();
116 if (!data->m_affectAllies && getObject()->getRelationship( obj ) == ALLIES)
117 {
118 return;
119 }
120
121 // Kill if object is infantry
122 if (obj->isKindOf(KINDOF_INFANTRY))
123 {
124 obj->kill();
125 }
126
127 // Kill all contained if it is a container
128 ContainModuleInterface *contain = obj->getContain();
129 if( contain )
130 {
131 contain->killAllContained();
132 }
133
134 // Kill pilots of vehicles
135 if( obj->isKindOf( KINDOF_VEHICLE ) && !obj->isKindOf( KINDOF_DRONE ) )
136 {
137 // If the vehicle is a combat bike, kill the whole thing
138 if ( obj->isKindOf( KINDOF_CLIFF_JUMPER ) )
139 {
140 obj->kill();
141 }
142 // Just kill the pilot of the vehicle
143 else
144 {
145 // Make it unmanned, so units can easily check the ability to "take control of it"
146 obj->setDisabled( DISABLED_UNMANNED );
147
148 if ( obj->getAI() )
149 obj->getAI()->aiIdle( CMD_FROM_AI );
150
151 TheGameLogic->deselectObject(obj, PLAYERMASK_ALL, TRUE);
152
153 // Clear any terrain decals here
154 Drawable* draw = obj->getDrawable();
155 if (draw)
156 draw->setTerrainDecal(TERRAIN_DECAL_NONE);
157
158 // Convert it to the neutral team so it renders gray giving visual representation that it is unmanned.
159 obj->setTeam( ThePlayerList->getNeutralPlayer()->getDefaultTeam() );
160 }
161 }
162}
