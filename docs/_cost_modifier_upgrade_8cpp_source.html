120void CostModifierUpgrade::onDelete( void )
121{
122
123 // if we haven't been upgraded there is nothing to clean up
124 if( isAlreadyUpgraded() == FALSE )
125 return;
126
127 // remove the radar from the player
128 Player *player = getObject()->getControllingPlayer();
129 if( player )
130 player->removeKindOfProductionCostChange(getCostModifierUpgradeModuleData()->m_kindOf,getCostModifierUpgradeModuleData()->m_percentage );
131
132 // this upgrade module is now "not upgraded"
133 setUpgradeExecuted(FALSE);
134
135} // end onDelete

137//-------------------------------------------------------------------------------------------------
138//-------------------------------------------------------------------------------------------------
139void CostModifierUpgrade::onCapture( Player *oldOwner, Player *newOwner )
140{
141
142 // do nothing if we haven't upgraded yet
143 if( isAlreadyUpgraded() == FALSE )
144 return;
145
146 // remove radar from old player and add to new player
147 if( oldOwner )
148 {
149
150 oldOwner->removeKindOfProductionCostChange(getCostModifierUpgradeModuleData()->m_kindOf,getCostModifierUpgradeModuleData()->m_percentage );
151 setUpgradeExecuted(FALSE);
152
153 } // end if
154 if( newOwner )
155 {
156
157 newOwner->addKindOfProductionCostChange(getCostModifierUpgradeModuleData()->m_kindOf,getCostModifierUpgradeModuleData()->m_percentage );
158 setUpgradeExecuted(TRUE);
159
160 } // end if
161
162} // end onCapture
