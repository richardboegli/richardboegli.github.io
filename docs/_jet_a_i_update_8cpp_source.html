106Bool JetAIUpdate::isOutOfSpecialReloadAmmo() const
107{
108 const Object* jet = getObject();
109 // if we have at least one special reload weapon,
110 // AND all such weapons are out of ammo,
111 // return true.
112 Int specials = 0;
113 Int out = 0;
114 for( Int i = 0; i < WEAPONSLOT_COUNT; i++ )
115 {
116 const Weapon* weapon = jet->getWeaponInWeaponSlot((WeaponSlotType)i);
117 if (weapon == NULL || weapon->getReloadType() != RETURN_TO_BASE_TO_RELOAD)
118 continue;
119 ++specials;
120 if (weapon->getStatus() == OUT_OF_AMMO)
121 ++out;
122 }
123 return specials > 0 && out == specials;
124}

127static ParkingPlaceBehaviorInterface* getPP(ObjectID id, Object** airfieldPP = NULL)
128{
129 if (airfieldPP)
130 *airfieldPP = NULL;
131
132 Object* airfield = TheGameLogic->findObjectByID( id );
133 if (airfield == NULL || airfield->isEffectivelyDead() || !airfield->isKindOf(KINDOF_FS_AIRFIELD) || airfield->testStatus(OBJECT_STATUS_SOLD))
134 return NULL;
135
136 if (airfieldPP)
137 *airfieldPP = airfield;
138
139 ParkingPlaceBehaviorInterface* pp = NULL;
140 for (BehaviorModule** i = airfield->getBehaviorModules(); *i; ++i)
141 {
142 if ((pp = (*i)->getParkingPlaceBehaviorInterface()) != NULL)
143 break;
144 }
145
146 return pp;
147}

170static Object* findSuitableAirfield(Object* jet)
171{
172 PartitionFilterAcceptByKindOf filterKind(MAKE_KINDOF_MASK(KINDOF_FS_AIRFIELD), KINDOFMASK_NONE);
173 PartitionFilterRejectByObjectStatus filterStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNDER_CONSTRUCTION ), OBJECT_STATUS_MASK_NONE );
174 PartitionFilterRejectByObjectStatus filterStatusTwo( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_SOLD ), OBJECT_STATUS_MASK_NONE ); // Independent to make it an OR
175 PartitionFilterRelationship filterTeam(jet, PartitionFilterRelationship::ALLOW_ALLIES);
176 PartitionFilterAlive filterAlive;
177 PartitionFilterSameMapStatus filterMapStatus(jet);
178 PartitionFilterHasParkingPlace filterPP(jet->getID());
179
180 PartitionFilter *filters[16];
181 Int numFilters = 0;
182 filters[numFilters++] = &filterKind;
183 filters[numFilters++] = &filterStatus;
184 filters[numFilters++] = &filterStatusTwo;
185 filters[numFilters++] = &filterTeam;
186 filters[numFilters++] = &filterAlive;
187 filters[numFilters++] = &filterPP;
188 filters[numFilters++] = &filterMapStatus;
189 filters[numFilters] = NULL;
190
191 return ThePartitionManager->getClosestObject( jet, HUGE_DIST, FROM_CENTER_2D, filters );
192}

200 Success: we have runway clearance
201 Failure: no runway clearance
202*/
203class JetAwaitingRunwayState : public State
204{
205 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(JetAwaitingRunwayState, "JetAwaitingRunwayState")
206protected:
207 // snapshot interface STUBBED.
208 virtual void crc( Xfer *xfer ){};
209 virtual void xfer( Xfer *xfer ){XferVersion cv = 1; XferVersion v = cv; xfer->xferVersion( &v, cv );}
210 virtual void loadPostProcess(){};
211private:
212 const Bool m_landing;
213
214public:
215 JetAwaitingRunwayState( StateMachine *machine, Bool landing ) : m_landing(landing), State( machine, "JetAwaitingRunwayState") { }
216
217 virtual StateReturnType onEnter()
218 {
219 Object* jet = getMachineOwner();
220 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
221 if( !jetAI )
222 return STATE_FAILURE;
223
224 jetAI->friend_setTakeoffInProgress(!m_landing);
225 jetAI->friend_setLandingInProgress(m_landing);
226 jetAI->friend_setAllowCircling(true);
227 return STATE_CONTINUE;
228 }
229
230 virtual StateReturnType update()
231 {
232 Object* jet = getMachineOwner();
233 if (jet->isEffectivelyDead())
234 return STATE_FAILURE;
235
236 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
237 if( !jetAI )
238 return STATE_FAILURE;
239
240 ParkingPlaceBehaviorInterface* pp = getPP(jet->getProducerID());
241 if (pp == NULL)
242 {
243 // no producer? just skip this step.
244 return STATE_SUCCESS;
245 }
246
247 // gotta reserve a space in order to reserve a runway
248 if (!pp->reserveSpace(jet->getID(), jetAI->friend_getParkingOffset(), NULL))
249 {
250 DEBUG_ASSERTCRASH(m_landing, ("hmm, this should never happen for taking-off things"));
251 return STATE_FAILURE;
252 }
253
254 if (pp->reserveRunway(jet->getID(), m_landing))
255 {
256 return STATE_SUCCESS;
257 }
258 else if( jet->testStatus( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) && !m_landing )
259 {
260 //If we're trying to take off an aircraft carrier and fail to reserve a
261 //runway, it's because we need to be at the front of the carrier queue.
262 //Therefore, we need to move forward whenever possible until we are in
263 //the front.
264 Coord3D bestPos;
265 if( pp->calcBestParkingAssignment( jet->getID(), &bestPos ) )
266 {
267 jetAI->friend_setTaxiInProgress(true);
268 jetAI->friend_setAllowAirLoco(false);
269 jetAI->chooseLocomotorSet(LOCOMOTORSET_TAXIING);
270
271 jetAI->destroyPath();
272 Path *movePath;
273 movePath = newInstance(Path);
274 Coord3D pos = *jet->getPosition();
275 movePath->prependNode( &pos, LAYER_GROUND );
276 movePath->markOptimized();
277 movePath->appendNode( &bestPos, LAYER_GROUND );
278
279 TheAI->pathfinder()->setDebugPath(movePath);
280
281 jetAI->friend_setPath( movePath );
282 DEBUG_ASSERTCRASH(jetAI->getCurLocomotor(), ("no loco"));
283 jetAI->getCurLocomotor()->setUsePreciseZPos(true);
284 jetAI->getCurLocomotor()->setUltraAccurate(true);
285 jetAI->getCurLocomotor()->setAllowInvalidPosition(true);
286 jetAI->ignoreObstacleID(jet->getProducerID());
287 }
288 }
289
290 // can't get a runway? gotta wait.
291 jetAI->setLocomotorGoalNone();
292 return STATE_CONTINUE;
293 }
294
295 virtual void onExit(StateExitType status)
296 {
297 Object* jet = getMachineOwner();
298 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
299 if (jetAI)
300 {
301 jetAI->friend_setTakeoffInProgress(false);
302 jetAI->friend_setLandingInProgress(false);
303 jetAI->friend_setAllowCircling(false);
304 }
305 }
306
307};
308EMPTY_DTOR(JetAwaitingRunwayState)

315class JetOrHeliCirclingDeadAirfieldState : public State
316{
317 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(JetOrHeliCirclingDeadAirfieldState, "JetOrHeliCirclingDeadAirfieldState")
318protected:
319 // snapshot interface STUBBED.
320 // The state will check immediately after a load game, but I think that's ok. jba.
321 virtual void crc( Xfer *xfer ){};
322 virtual void xfer( Xfer *xfer ){XferVersion cv = 1; XferVersion v = cv; xfer->xferVersion( &v, cv );}
323 virtual void loadPostProcess(){};
324
325private:
326 Int m_checkAirfield;
327
328 enum
329 {
330 // only recheck for new airfields every second or so
331 HOW_OFTEN_TO_CHECK = LOGICFRAMES_PER_SECOND
332 };
333
334public:
335 JetOrHeliCirclingDeadAirfieldState( StateMachine *machine ) :
336 State( machine, "JetOrHeliCirclingDeadAirfieldState"),
337 m_checkAirfield(0) { }
338
339 virtual StateReturnType onEnter()
340 {
341 Object* jet = getMachineOwner();
342 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
343 if( !jetAI )
344 {
345 return STATE_FAILURE;
346 }
347
348 // obscure case: if the jet wasn't spawned, but just placed directly on the map,
349 // it might not have an owning airfield, and it might be trying to return
350 // simply due to being idle, not out of ammo. so check and don't die in that
351 // case, but just punt back out to idle.
352 if (!jetAI->isOutOfSpecialReloadAmmo() && jet->getProducerID() == INVALID_ID)
353 {
354 return STATE_FAILURE;
355 }
356
357 // just stay where we are.
358 jetAI->setLocomotorGoalNone();
359
360 m_checkAirfield = HOW_OFTEN_TO_CHECK;
361
362 //Play the "low fuel" voice whenever the craft is circling above the airfield.
363 AudioEventRTS soundToPlay = *jet->getTemplate()->getPerUnitSound( "VoiceLowFuel" );
364 soundToPlay.setObjectID( jet->getID() );
365 TheAudio->addAudioEvent( &soundToPlay );
366
367 return STATE_CONTINUE;
368 }
369
370 virtual StateReturnType update()
371 {
372 Object* jet = getMachineOwner();
373 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
374 if( !jetAI )
375 {
376 return STATE_FAILURE;
377 }
378
379 // just stay where we are.
380 jetAI->setLocomotorGoalNone();
381
382 Real damageRate = jetAI->friend_getOutOfAmmoDamagePerSecond();
383 if (damageRate > 0)
384 {
385 // convert to damage/sec to damage/frame
386 damageRate *= SECONDS_PER_LOGICFRAME_REAL;
387 // since it's a percentage, multiply times the max health
388 damageRate *= jet->getBodyModule()->getMaxHealth();
389
390 DamageInfo damageInfo;
391 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
392 damageInfo.in.m_deathType = DEATH_NORMAL;
393 damageInfo.in.m_sourceID = INVALID_ID;
394 damageInfo.in.m_amount = damageRate;
395 jet->attemptDamage( &damageInfo );
396 }
397
398 if (--m_checkAirfield <= 0)
399 {
400 m_checkAirfield = HOW_OFTEN_TO_CHECK;
401 Object* airfield = findSuitableAirfield( jet );
402 if (airfield)
403 {
404 jet->setProducer(airfield);
405 return STATE_SUCCESS;
406 }
407 }
408
409 return STATE_CONTINUE;
410 }

412};

445// This solution uses the
446// http://www.faqs.org/faqs/graphics/algorithms-faq/
447// Subject 1.03
448static Bool intersectInfiniteLine2D
449(
450 Real ax, Real ay, Real ao,
451 Real cx, Real cy, Real co,
452 Real& ix, Real& iy
453)
454{
455 Real bx = ax + Cos(ao);
456 Real by = ay + Sin(ao);
457 Real dx = cx + Cos(co);
458 Real dy = cy + Sin(co);
459
460 Real denom = ((bx - ax) * (dy - cy) - (by - ay) * (dx - cx));
461 if (denom == 0.0f)
462 {
463 // the lines are parallel.
464 return false;
465 }
466
467 // The lines intersect.
468 Real r = ((ay - cy) * (dx - cx) - (ax - cx) * (dy - cy) ) / denom;
469 ix = ax + r * (bx - ax);
470 iy = ay + r * (by - ay);
471 return true;
472}

479class JetOrHeliTaxiState : public AIMoveOutOfTheWayState
480{
481 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(JetOrHeliTaxiState, "JetOrHeliTaxiState")
482private:
483 TaxiType m_taxiMode;
484public:
485 JetOrHeliTaxiState( StateMachine *machine, TaxiType m ) : m_taxiMode(m), AIMoveOutOfTheWayState( machine ) { }
486
487 virtual StateReturnType onEnter()
488 {
489 Object* jet = getMachineOwner();
490 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
491 if( !jetAI )
492 return STATE_FAILURE;
493
494 jetAI->setCanPathThroughUnits(true);
495 jetAI->friend_setTakeoffInProgress(m_taxiMode != TO_PARKING);
496 jetAI->friend_setLandingInProgress(m_taxiMode == TO_PARKING);
497 jetAI->friend_setTaxiInProgress(true);
498
499 if( m_taxiMode == TO_PARKING )
500 {
501 //Instantly reload flares.
502 CountermeasuresBehaviorInterface *cbi = jet->getCountermeasuresBehaviorInterface();
503 if( cbi )
504 {
505 cbi->reloadCountermeasures();
506 }
507 }
508
509 jetAI->friend_setAllowAirLoco(false);
510 jetAI->chooseLocomotorSet(LOCOMOTORSET_TAXIING);
511 DEBUG_ASSERTCRASH(jetAI->getCurLocomotor(), ("no loco"));
512
513 Object* airfield;
514 ParkingPlaceBehaviorInterface* pp = getPP(jet->getProducerID(), &airfield);
515 if (pp == NULL)
516 return STATE_SUCCESS; // no airfield? just skip this step.
517
518 ParkingPlaceBehaviorInterface::PPInfo ppinfo;
519 if (!pp->reserveSpace(jet->getID(), jetAI->friend_getParkingOffset(), &ppinfo))
520 return STATE_FAILURE; // full?
521
522 Coord3D intermedPt;
523 Bool intermed = false;
524 Real orient = atan2(ppinfo.runwayPrep.y - ppinfo.parkingSpace.y, ppinfo.runwayPrep.x - ppinfo.parkingSpace.x);
525 if (fabs(stdAngleDiff(orient, ppinfo.parkingOrientation)) > PI/128)
526 {
527 intermedPt.z = (ppinfo.parkingSpace.z + ppinfo.runwayPrep.z) * 0.5f;
528 intermed = intersectInfiniteLine2D(
529 ppinfo.parkingSpace.x, ppinfo.parkingSpace.y, ppinfo.parkingOrientation,
530 ppinfo.runwayPrep.x, ppinfo.runwayPrep.y, ppinfo.parkingOrientation + PI/2,
531 intermedPt.x, intermedPt.y);
532 }
533
534 jetAI->destroyPath();
535 Path *movePath;
536 movePath = newInstance(Path);
537 Coord3D pos = *jet->getPosition();
538 movePath->prependNode( &pos, LAYER_GROUND );
539 movePath->markOptimized();
540
541 if (m_taxiMode == TO_PARKING)
542 {
543 if( jet->testStatus( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
544 {
545 //We're on an aircraft carrier.
546 const std::vector<Coord3D> *pTaxiLocations = pp->getTaxiLocations( jet->getID() );
547 if( pTaxiLocations )
548 {
549 std::vector<Coord3D>::const_iterator it;
550 for( it = pTaxiLocations->begin(); it != pTaxiLocations->end(); it++ )
551 {
552 movePath->appendNode( it, LAYER_GROUND );
553 }
554 }
555
556 //We just landed... see if we can get a better space forward so we don't stop and pause
557 //at our initially assigned spot.
558 Coord3D pos;
559 pp->calcBestParkingAssignment( jet->getID(), &pos );
560
561 movePath->appendNode( &pos, LAYER_GROUND );
562 }
563 else
564 {
565 //We're on a normal airfield
566 movePath->appendNode( &ppinfo.runwayPrep, LAYER_GROUND );
567 if (intermed)
568 movePath->appendNode( &intermedPt, LAYER_GROUND );
569 movePath->appendNode( &ppinfo.parkingSpace, LAYER_GROUND );
570 }
571 }
572 else if (m_taxiMode == FROM_PARKING)
573 {
574 if( jet->testStatus( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
575 {
576 if( !(ppinfo.runwayStart == ppinfo.runwayPrep) )
577 {
578 movePath->appendNode( &ppinfo.runwayStart, LAYER_GROUND );
579 }
580 }
581 else
582 {
583 if (intermed)
584 movePath->appendNode( &intermedPt, LAYER_GROUND );
585 movePath->appendNode( &ppinfo.runwayPrep, LAYER_GROUND );
586 movePath->appendNode( &ppinfo.runwayStart, LAYER_GROUND );
587 }
588 }
589 else if (m_taxiMode == FROM_HANGAR)
590 {
591 if( jet->testStatus( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
592 {
593 //Aircraft carrier
594 if( jet->testStatus( OBJECT_STATUS_REASSIGN_PARKING ) )
595 {
596 //This status means we are being reassigned a parking space. We're not actually moving from the
597 //hangar. So simply move to the new parking spot which was just switched from under us in
598 //FlightDeckBehavior::update()
599 jet->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_REASSIGN_PARKING ) );
600 movePath->appendNode( &ppinfo.runwayPrep, LAYER_GROUND );
601 }
602 else
603 {
604 const std::vector<Coord3D> *pCreationLocations = pp->getCreationLocations( jet->getID() );
605 if( !pCreationLocations )
606 {
607 DEBUG_CRASH( ("No creation locations specified for runway for JetAIBehavior -- taxiing from hanger (Kris).") );
608 return STATE_FAILURE;
609 }
610 std::vector<Coord3D>::const_iterator it;
611 Bool firstNode = TRUE;
612 for( it = pCreationLocations->begin(); it != pCreationLocations->end(); it++ )
613 {
614 if( firstNode )
615 {
616 //Skip the first node because it's the creation location.
617 firstNode = FALSE;
618 continue;
619 }
620 movePath->appendNode( it, LAYER_GROUND );
621 }
622 movePath->appendNode( &ppinfo.runwayPrep, LAYER_GROUND );
623 }
624 }
625 else
626 {
627 //Airfield
628 movePath->appendNode( &ppinfo.parkingSpace, LAYER_GROUND );
629 }
630 }
631
632 m_waitingForPath = FALSE;
633 TheAI->pathfinder()->setDebugPath(movePath);
634
635 setAdjustsDestination(false); // precision is necessary
636
637 jetAI->friend_setPath( movePath );
638 DEBUG_ASSERTCRASH(jetAI->getCurLocomotor(), ("no loco"));
639 jetAI->getCurLocomotor()->setUsePreciseZPos(true);
640 jetAI->getCurLocomotor()->setUltraAccurate(true);
641 jetAI->getCurLocomotor()->setAllowInvalidPosition(true);
642 jetAI->ignoreObstacleID(jet->getProducerID());
643
644 StateReturnType ret = AIMoveOutOfTheWayState::onEnter();
645 return ret;
646 }
647
648 virtual StateReturnType update()
649 {
650 Object* jet = getMachineOwner();
651 if (jet->isEffectivelyDead())
652 return STATE_FAILURE;
653
654 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
655 if( !jetAI )
656 return STATE_FAILURE;
657
658 if( m_taxiMode == TO_PARKING || m_taxiMode == FROM_HANGAR )
659 {
660 //Keep checking to see if there is a better spot as it moves forward. If we find a better spot, then
661 //append the position to our move.
662 ParkingPlaceBehaviorInterface* pp = getPP(jet->getProducerID());
663 Coord3D bestPos;
664 Int oldIndex, newIndex;
665 // Check pp for null, as it is possible for your airfield to get destroyed while taxiing.jba [8/27/2003]
666 if( pp!=NULL && pp->calcBestParkingAssignment( jet->getID(), &bestPos, &oldIndex, &newIndex ) )
667 {
668 Path *path = jetAI->friend_getPath();
669 if( path )
670 {
671 path->appendNode( &bestPos, LAYER_GROUND );
672 }
673 }
674 }
675
676 return AIMoveOutOfTheWayState::update();
677 }
678
679 virtual void onExit( StateExitType status )
680 {
681 Object* jet = getMachineOwner();
682 JetAIUpdate* jetAI = (JetAIUpdate*)jet->getAIUpdateInterface();
683 if (jetAI)
684 {
685 jetAI->getCurLocomotor()->setUsePreciseZPos(false);
686 jetAI->getCurLocomotor()->setUltraAccurate(false);
687 jetAI->getCurLocomotor()->setAllowInvalidPosition(false);
688 jetAI->friend_setTakeoffInProgress(false);
689 jetAI->friend_setLandingInProgress(false);
690 jetAI->friend_setTaxiInProgress(false);
691 jetAI->setCanPathThroughUnits(false);
692 }
693
694 AIMoveOutOfTheWayState::onExit(status);
695 }

697};

837#ifdef CIRCLE_FOR_LANDING

839#endif

843 {
844 ParkingPlaceBehaviorInterface* pp = getPP(jet->getProducerID());
845 Real zPos = jet->getPosition()->z;
846 Real zSlop = 0.25f;
847 PathfindLayerEnum layer = TheTerrainLogic->getHighestLayerForDestination( jet->getPosition() );
848 Real groundZ = TheTerrainLogic->getLayerHeight( jet->getPosition()->x, jet->getPosition()->y, layer );
849 if( pp )
850 {
851 groundZ += pp->getLandingDeckHeightOffset();
852 }
853
854 if( zPos - zSlop <= groundZ )
855 {
856 m_landingSoundPlayed = TRUE;
857 AudioEventRTS soundToPlay = TheAudio->getMiscAudio()->m_aircraftWheelScreech;
858 soundToPlay.setPosition( jet->getPosition() );
859 TheAudio->addAudioEvent( &soundToPlay );
860 }
861 }
862 }

864 {
865 ParkingPlaceBehaviorInterface* pp = getPP(jet->getProducerID());
866 if (pp)
867 pp->transferRunwayReservationToNextInLineForTakeoff(jet->getID());

869 //Calculate the distance of the jet from the end of the runway as a ratio from the start.
870 //As it approaches the end of the runway, the plane will gain more lift, even if it's already
871 //going quickly. Using speed for lift is bad in the case of the aircraft carrier, because
872 //we don't want it to take off quickly.
873 ParkingPlaceBehaviorInterface::PPInfo ppinfo;
874 pp->calcPPInfo( jet->getID(), &ppinfo );
875 Coord3D vector = ppinfo.runwayEnd;
876 vector.sub( jet->getPosition() );
877 Real dist = vector.length();
878
879 Real ratio = 1.0f - (dist / ppinfo.runwayTakeoffDist);
880 ratio *= ratio; //dampen it....
881 if (ratio < 0.0f) ratio = 0.0f;
882 if (ratio > 1.0f) ratio = 1.0f;
883 jetAI->getCurLocomotor()->setMaxLift(m_maxLift * ratio);
884 }

887 return ret;

1840UpdateSleepTime JetAIUpdate::update()
1841{
1842 const JetAIUpdateModuleData* d = getJetAIUpdateModuleData();
1843
1844 getProducerLocation();
1845
1846 Object* jet = getObject();
1847
1848 ParkingPlaceBehaviorInterface* pp = getPP(getObject()->getProducerID());
1849
1850 // If idle & out of ammo, return
1851 // have to call our parent's isIdle, because we override it to never return true
1852 // when we have a pending command...
1853 UnsignedInt now = TheGameLogic->getFrame();
1854
1855 // srj sez: not 100% sure on this. calling RELOAD_AMMO "idle" allows us to get healed while reloading,
1856 // but might have other side effects we didn't want. if this does prove to cause a bug, be sure
1857 // that jets (and ESPECIALLY comanches) are still getting healed at airfields.
1858 if (AIUpdateInterface::isIdle() || getStateMachine()->getCurrentStateID() == RELOAD_AMMO)
1859 {
1860 if (pp != NULL)
1861 {
1862 if (!getFlag(ALLOW_AIR_LOCO) &&
1863 !getFlag(HAS_PENDING_COMMAND) &&
1864 jet->isKindOf(KINDOF_PRODUCED_AT_HELIPAD) &&
1865 jet->getBodyModule()->getHealth() == jet->getBodyModule()->getMaxHealth())
1866 {
1867 // we're completely healed, so take off again
1868 pp->setHealee(jet, false);
1869 friend_setAllowAirLoco(true);
1870 getStateMachine()->clear();
1871 setLastCommandSource( CMD_FROM_AI );
1872 getStateMachine()->setState( TAKING_OFF_AWAIT_CLEARANCE );
1873 }
1874 else
1875 {
1876 pp->setHealee(jet, !getFlag(ALLOW_AIR_LOCO));
1877 }
1878 }
1879
1880 // note that we might still have weapons with ammo, but still be forced to return to reload.
1881 if (isOutOfSpecialReloadAmmo() && getFlag(ALLOW_AIR_LOCO))
1882 {
1883 m_returnToBaseFrame = 0;
1884
1885 // this is really a "just-in-case" to ensure the targeter list doesn't spin out of control (srj)
1886 pruneDeadTargeters();
1887
1888 setFlag(USE_SPECIAL_RETURN_LOCO, true);
1889 setLastCommandSource( CMD_FROM_AI );
1890 getStateMachine()->setState(RETURNING_FOR_LANDING);
1891 }
1892 else if (getFlag(HAS_PENDING_COMMAND)
1893 // srj sez: if we are reloading ammo, wait will we are done before processing the pending command.
1894 && getStateMachine()->getCurrentStateID() != RELOAD_AMMO)
1895 {
1896 m_returnToBaseFrame = 0;
1897
1898 AICommandParms parms(AICMD_MOVE_TO_POSITION, CMD_FROM_AI); // values don't matter, will be wiped by next line
1899 m_mostRecentCommand.reconstitute(parms);
1900 setFlag(HAS_PENDING_COMMAND, false);
1901
1902 aiDoCommand(&parms);
1903 }
1904 else if (m_returnToBaseFrame != 0 && now >= m_returnToBaseFrame && getFlag(ALLOW_AIR_LOCO))
1905 {
1906 m_returnToBaseFrame = 0;
1907 DEBUG_ASSERTCRASH(isOutOfSpecialReloadAmmo() == false, ("Hmm, this seems unlikely -- isOutOfSpecialReloadAmmo()==false"));
1908 setFlag(USE_SPECIAL_RETURN_LOCO, false);
1909 setLastCommandSource( CMD_FROM_AI );
1910 getStateMachine()->setState(RETURNING_FOR_LANDING);
1911 }
1912 else if (m_returnToBaseFrame == 0 && d->m_returnToBaseIdleTime > 0 && getFlag(ALLOW_AIR_LOCO))
1913 {
1914 m_returnToBaseFrame = now + d->m_returnToBaseIdleTime;
1915 }
1916 }
1917 else
1918 {
1919 if (pp != NULL)
1920 {
1921 pp->setHealee(getObject(), false);
1922 }
1923 m_returnToBaseFrame = 0;
1924 if (getFlag(ALLOW_INTERRUPT_AND_RESUME_OF_CUR_STATE_FOR_RELOAD) &&
1925 isOutOfSpecialReloadAmmo() && getFlag(ALLOW_AIR_LOCO))
1926 {
1927 setFlag(USE_SPECIAL_RETURN_LOCO, true);
1928 setFlag(HAS_PENDING_COMMAND, true);
1929 setFlag(ALLOW_INTERRUPT_AND_RESUME_OF_CUR_STATE_FOR_RELOAD, false);
1930 setLastCommandSource( CMD_FROM_AI );
1931 getStateMachine()->setState(RETURNING_FOR_LANDING);
1932 }
1933 }
1934
1935 Real minHeight = friend_getMinHeight();
1936 if( pp )
1937 {
1938 minHeight += pp->getLandingDeckHeightOffset();
1939 }
1940
1941 Drawable* draw = jet->getDrawable();
1942 if (draw != NULL)
1943 {
1944 StateID id = getStateMachine()->getCurrentStateID();
1945 Bool needToCheckMinHeight = (id >= JETAISTATETYPE_FIRST && id <= JETAISTATETYPE_LAST) ||
1946 !jet->isAboveTerrain() ||
1947 !getFlag(ALLOW_AIR_LOCO);
1948 if( needToCheckMinHeight || jet->getStatusBits().test( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
1949 {
1950 Real ht = jet->isAboveTerrain() ? jet->getHeightAboveTerrain() : 0;
1951 if (ht < minHeight)
1952 {
1953 Matrix3D tmp(1);
1954 tmp.Set_Z_Translation(minHeight - ht);
1955 draw->setInstanceMatrix(&tmp);
1956 }
1957 else
1958 {
1959 draw->setInstanceMatrix(NULL);
1960 }
1961 }
1962 else
1963 {
1964 draw->setInstanceMatrix(NULL);
1965 }
1966 }
1967
1968 PhysicsBehavior* physics = jet->getPhysics();
1969 if (physics->getVelocityMagnitude() > 0 && getFlag(ALLOW_AIR_LOCO))
1970 jet->setModelConditionState(MODELCONDITION_JETEXHAUST);
1971 else
1972 jet->clearModelConditionState(MODELCONDITION_JETEXHAUST);
1973
1974 if (jet->testStatus(OBJECT_STATUS_IS_ATTACKING))
1975 {
1976 m_attackLocoExpireFrame = now + d->m_attackLocoPersistTime;
1977 m_attackersMissExpireFrame = now + d->m_attackersMissPersistTime;
1978 }
1979 else
1980 {
1981 if (m_attackLocoExpireFrame != 0 && now >= m_attackLocoExpireFrame)
1982 {
1983 m_attackLocoExpireFrame = 0;
1984 }
1985 if (m_attackersMissExpireFrame != 0 && now >= m_attackersMissExpireFrame)
1986 {
1987 m_attackersMissExpireFrame = 0;
1988 }
