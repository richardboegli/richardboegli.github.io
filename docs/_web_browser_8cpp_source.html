240STDMETHODIMP WebBrowser::QueryInterface(REFIID iid, void** ppv)
241{
242 *ppv = NULL;
243
244 if ((iid == IID_IUnknown) || (iid == IID_IBrowserDispatch))
245 {
246 *ppv = static_cast<IBrowserDispatch*>(this);
247 }
248 else
249 {
250 return E_NOINTERFACE;
251 }
252
253 static_cast<IUnknown*>(*ppv)->AddRef();
254
255 return S_OK;
256}
