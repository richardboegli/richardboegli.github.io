83/***********************************************************************************************
84 * Get_Local_Chat_Connection_Address -- Which address are we using to talk to the chat server? *
85 * *
86 * *
87 * *
88 * INPUT: Ptr to address to return local address * *
89 * *
90 * OUTPUT: True if success *
91 * *
92 * WARNINGS: None *
93 * *
94 * HISTORY: *
95 * 10/27/00 3:24PM ST : Created *
96 *=============================================================================================*/
97Bool GetLocalChatConnectionAddress(AsciiString serverName, UnsignedShort serverPort, UnsignedInt& localIP)
98{
99 //return false;
100 /*
101 ** Local defines.
102 */
103 enum {
104 CLOSED = 1,
105 LISTENING,
106 SYN_SENT,
107 SEN_RECEIVED,
108 ESTABLISHED,
109 FIN_WAIT,
110 FIN_WAIT2,
111 CLOSE_WAIT,
112 LAST_ACK,
113 CLOSING,
114 TIME_WAIT,
115 DELETE_TCB
116 };
117
118 enum {
119 tcpConnState = 1,
120 tcpConnLocalAddress,
121 tcpConnLocalPort,
122 tcpConnRemAddress,
123 tcpConnRemPort
124 };
125
126
127 /*
128 ** Locals.
129 */
130 unsigned char serverAddress[4];
131 unsigned char remoteAddress[4];
132 HANDLE trap_handle;
133 AsnObjectIdentifier first_supported_region;
134 std::vector<ConnInfoStruct> connectionVector;
135 int last_field;
136 int index;
137 AsnInteger error_status;
138 AsnInteger error_index;
139 int conn_entry_type_index;
140 int conn_entry_type;
141 Bool found;
142
143 /*
144 ** Statics.
145 */
146 static char _conn_state[][32] = {
147 "?",
148 "CLOSED",
149 "LISTENING",
150 "SYN_SENT",
151 "SEN_RECEIVED",
152 "ESTABLISHED",
153 "FIN_WAIT",
154 "FIN_WAIT2",
155 "CLOSE_WAIT",
156 "LAST_ACK",
157 "CLOSING",
158 "TIME_WAIT",
159 "DELETE_TCB"
160 };
161
162 DEBUG_LOG(("Finding local address used to talk to the chat server\n"));
163 DEBUG_LOG(("Current chat server name is %s\n", serverName.str()));
164 DEBUG_LOG(("Chat server port is %d\n", serverPort));
165
166 /*
167 ** Get the address of the chat server.
168 */
169 DEBUG_LOG( ("About to call gethostbyname\n"));
170 struct hostent *host_info = gethostbyname(serverName.str());
171
172 if (!host_info) {
173 DEBUG_LOG( ("gethostbyname failed! Error code %d\n", WSAGetLastError()));
174 return(false);
175 }
176
177 memcpy(serverAddress, &host_info->h_addr_list[0][0], 4);
178 unsigned long temp = *((unsigned long*)(&serverAddress[0]));
179 temp = ntohl(temp);
180 *((unsigned long*)(&serverAddress[0])) = temp;
181
182 DEBUG_LOG(("Host address is %d.%d.%d.%d\n", serverAddress[3], serverAddress[2], serverAddress[1], serverAddress[0]));
183
184 /*
185 ** Load the MIB-II SNMP DLL.
186 */
187 DEBUG_LOG(("About to load INETMIB1.DLL\n"));
188
189 HINSTANCE mib_ii_dll = LoadLibrary("inetmib1.dll");
190 if (mib_ii_dll == NULL) {
191 DEBUG_LOG(("Failed to load INETMIB1.DLL\n"));
192 return(false);
193 }
194
195 DEBUG_LOG(("About to load SNMPAPI.DLL\n"));
196
197 HINSTANCE snmpapi_dll = LoadLibrary("snmpapi.dll");
198 if (snmpapi_dll == NULL) {
199 DEBUG_LOG(("Failed to load SNMPAPI.DLL\n"));
200 FreeLibrary(mib_ii_dll);
201 return(false);
202 }
203
204 /*
205 ** Get the function pointers into the .dll
206 */
207 SnmpExtensionInitPtr = (int (__stdcall *)(unsigned long,void ** ,AsnObjectIdentifier *)) GetProcAddress(mib_ii_dll, "SnmpExtensionInit");
208 SnmpExtensionQueryPtr = (int (__stdcall *)(unsigned char,SnmpVarBindList *,long *,long *)) GetProcAddress(mib_ii_dll, "SnmpExtensionQuery");
209 SnmpUtilMemAllocPtr = (void *(__stdcall *)(unsigned long)) GetProcAddress(snmpapi_dll, "SnmpUtilMemAlloc");
210 SnmpUtilMemFreePtr = (void (__stdcall *)(void *)) GetProcAddress(snmpapi_dll, "SnmpUtilMemFree");
211 if (SnmpExtensionInitPtr == NULL || SnmpExtensionQueryPtr == NULL || SnmpUtilMemAllocPtr == NULL || SnmpUtilMemFreePtr == NULL) {
212 DEBUG_LOG(("Failed to get proc addresses for linked functions\n"));
213 FreeLibrary(snmpapi_dll);
214 FreeLibrary(mib_ii_dll);
215 return(false);
216 }
217
218
219 RFC1157VarBindList *bind_list_ptr = (RFC1157VarBindList *) SnmpUtilMemAllocPtr(sizeof(RFC1157VarBindList));
220 RFC1157VarBind *bind_ptr = (RFC1157VarBind *) SnmpUtilMemAllocPtr(sizeof(RFC1157VarBind));
221
222 /*
223 ** OK, here we go. Try to initialise the .dll
224 */
225 DEBUG_LOG(("About to init INETMIB1.DLL\n"));
226 int ok = SnmpExtensionInitPtr(GetCurrentTime(), &trap_handle, &first_supported_region);
227
228 if (!ok) {
229 /*
230 ** Aw crap.
231 */
232 DEBUG_LOG(("Failed to init the .dll\n"));
233 SnmpUtilMemFreePtr(bind_list_ptr);
234 SnmpUtilMemFreePtr(bind_ptr);
235 FreeLibrary(snmpapi_dll);
236 FreeLibrary(mib_ii_dll);
237 return(false);
238 }
239
240 /*
241 ** Name of mib_ii object we want to query. See RFC 1213.
242 **
243 ** iso.org.dod.internet.mgmt.mib-2.tcp.tcpConnTable.TcpConnEntry.tcpConnState
244 ** 1 3 6 1 2 1 6 13 1 1
245 */
246 unsigned int mib_ii_name[] = {1,3,6,1,2,1,6,13,1,1};
247 unsigned int *mib_ii_name_ptr = (unsigned int *) SnmpUtilMemAllocPtr(sizeof(mib_ii_name));
248 memcpy(mib_ii_name_ptr, mib_ii_name, sizeof(mib_ii_name));
249
250 /*
251 ** Get the index of the conn entry data.
252 */
253 conn_entry_type_index = ARRAY_SIZE(mib_ii_name) - 1;
254
255 /*
256 ** Set up the bind list.
257 */
258 bind_ptr->name.idLength = ARRAY_SIZE(mib_ii_name);
259 bind_ptr->name.ids = mib_ii_name;
260 bind_list_ptr->list = bind_ptr;
261 bind_list_ptr->len = 1;
262
263
264 /*
265 ** We start with the tcpConnLocalAddress field.
266 */
267 last_field = 1;
268
269 /*
270 ** First connection.
271 */
272 index = 0;
273
274 /*
275 ** Suck out that tcp connection info....
276 */
277 while (true) {
278
279 if (!SnmpExtensionQueryPtr(SNMP_PDU_GETNEXT, bind_list_ptr, &error_status, &error_index)) {
280 //if (!SnmpExtensionQueryPtr(ASN_RFC1157_GETNEXTREQUEST, bind_list_ptr, &error_status, &error_index)) {
281 DEBUG_LOG(("SnmpExtensionQuery returned false\n"));
282 SnmpUtilMemFreePtr(bind_list_ptr);
283 SnmpUtilMemFreePtr(bind_ptr);
284 FreeLibrary(snmpapi_dll);
285 FreeLibrary(mib_ii_dll);
286 return(false);
287 }
288
289 /*
290 ** If this is something new we aren't looking for then we are done.
291 */
292 if (bind_ptr->name.idLength < ARRAY_SIZE(mib_ii_name)) {
293 break;
294 }
295
296 /*
297 ** Get the type of info we are looking at. See RFC1213.
298 **
299 ** 1 = tcpConnState
300 ** 2 = tcpConnLocalAddress
301 ** 3 = tcpConnLocalPort
302 ** 4 = tcpConnRemAddress
303 ** 5 = tcpConnRemPort
304 **
305 ** tcpConnState is one of the following...
306 **
307 ** 1 closed
308 ** 2 listen
309 ** 3 synSent
310 ** 4 synReceived
311 ** 5 established
312 ** 6 finWait1
313 ** 7 finWait2
314 ** 8 closeWait
315 ** 9 lastAck
316 ** 10 closing
317 ** 11 timeWait
318 ** 12 deleteTCB
319 */
320 conn_entry_type = bind_ptr->name.ids[conn_entry_type_index];
321
322 if (last_field != conn_entry_type) {
323 index = 0;
324 last_field = conn_entry_type;
325 }
326
327 switch (conn_entry_type) {
328
329 /*
330 ** 1. First field in the entry. Need to create a new connection info struct
331 ** here to store this connection in.
332 */
333 case tcpConnState:
334 {
335 ConnInfoStruct new_conn;
336 new_conn.State = bind_ptr->value.asnValue.number;
337 connectionVector.push_back(new_conn);
338 break;
339 }
340
341 /*
342 ** 2. Local address field.
343 */
344 case tcpConnLocalAddress:
345 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
346 connectionVector[index].LocalIP = *((unsigned long*)bind_ptr->value.asnValue.address.stream);
347 index++;
348 break;
349
350 /*
351 ** 3. Local port field.
352 */
353 case tcpConnLocalPort:
354 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
355 connectionVector[index].LocalPort = bind_ptr->value.asnValue.number;
356 //connectionVector[index]->LocalPort = ntohs(connectionVector[index]->LocalPort);
357 index++;
358 break;
359
360 /*
361 ** 4. Remote address field.
362 */
363 case tcpConnRemAddress:
364 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
365 connectionVector[index].RemoteIP = *((unsigned long*)bind_ptr->value.asnValue.address.stream);
366 index++;
367 break;
368
369 /*
370 ** 5. Remote port field.
371 */
372 case tcpConnRemPort:
373 DEBUG_ASSERTCRASH(index < connectionVector.size(), ("Bad connection index"));
374 connectionVector[index].RemotePort = bind_ptr->value.asnValue.number;
375 //connectionVector[index]->RemotePort = ntohs(connectionVector[index]->RemotePort);
376 index++;
377 break;
378 }
379 }
380
381 SnmpUtilMemFreePtr(bind_list_ptr);
382 SnmpUtilMemFreePtr(bind_ptr);
383 SnmpUtilMemFreePtr(mib_ii_name_ptr);
384
385 DEBUG_LOG(("Got %d connections in list, parsing...\n", connectionVector.size()));
386
387 /*
388 ** Right, we got the lot. Lets see if any of them have the same address as the chat
389 ** server we think we are talking to.
390 */
391 found = false;
392 for (Int i=0; i<connectionVector.size(); ++i) {
393 ConnInfoStruct connection = connectionVector[i];
394
395 temp = ntohl(connection.RemoteIP);
396 memcpy(remoteAddress, (unsigned char*)&temp, 4);
397
398 /*
399 ** See if this connection has the same address as our server.
400 */
401 if (!found && memcmp(remoteAddress, serverAddress, 4) == 0) {
402 DEBUG_LOG(("Found connection with same remote address as server\n"));
403
404 if (serverPort == 0 || serverPort == (unsigned int)connection.RemotePort) {
405
406 DEBUG_LOG(("Connection has same port\n"));
407 /*
408 ** Make sure the connection is current.
409 */
410 if (connection.State == ESTABLISHED) {
411 DEBUG_LOG(("Connection is ESTABLISHED\n"));
412 localIP = connection.LocalIP;
413 found = true;
414 } else {
415 DEBUG_LOG(("Connection is not ESTABLISHED - skipping\n"));
416 }
417 } else {
418 DEBUG_LOG(("Connection has different port. Port is %d, looking for %d\n", connection.RemotePort, serverPort));
419 }
420 }
421 }
422
423 if (found) {
424 DEBUG_LOG(("Using address 0x%8.8X to talk to chat server\n", localIP));
425 }
426
427 FreeLibrary(snmpapi_dll);
428 FreeLibrary(mib_ii_dll);
429 return(found);
430}

459{
460 if (TheGameSpyGame)
461 {
462 int i;
463
464 int numUsers = 0;
465 for (i=0; i<MAX_SLOTS; ++i)

480 return;
481 }
482
483 TheGameSpyGame->startGame(0);
484 }

567

570 DEBUG_LOG(("InitGameLogicRandom( %d )\n", TheGameSpyGame->getSeed()));
571
572 if (TheNAT != NULL) {
573 delete TheNAT;
574 TheNAT = NULL;
575 }
576 }

624void GameSpyGameInfo::startGame(Int gameID)
625{
626 DEBUG_LOG(("GameSpyGameInfo::startGame - game id = %d\n", gameID));
627 DEBUG_ASSERTCRASH(m_transport == NULL, ("m_transport is not NULL when it should be"));
628 DEBUG_ASSERTCRASH(TheNAT == NULL, ("TheNAT is not NULL when it should be"));
629
630 // fill in GS-specific info
631 for (Int i=0; i<MAX_SLOTS; ++i)
632 {
633 if (m_GameSpySlot[i].isHuman())
634 {
635 AsciiString gsName;
636 gsName.translate( m_GameSpySlot[i].getName() );
637 m_GameSpySlot[i].setLoginName( gsName );
638
639 PlayerInfoMap *pInfoMap = TheGameSpyInfo->getPlayerInfoMap();
640 PlayerInfoMap::iterator it = pInfoMap->find(gsName);
641 if (it != pInfoMap->end())
642 {
643 m_GameSpySlot[i].setProfileID(it->second.m_profileID);
644 m_GameSpySlot[i].setLocale(it->second.m_locale);
645 }
646 else
647 {
648 DEBUG_CRASH(("No player info for %s", gsName.str()));
649 }
650 }
651 }
652
653 if (TheNAT != NULL) {
654 delete TheNAT;
655 TheNAT = NULL;
656 }
657 TheNAT = NEW NAT();
658 TheNAT->attachSlotList(m_slot, getLocalSlotNum(), m_localIP);
659 TheNAT->establishConnectionPaths();
660}

662AsciiString GameSpyGameInfo::generateGameResultsPacket( void )
663{
664 Int i;
665 Int endFrame = TheVictoryConditions->getEndFrame();
666 Int localSlotNum = getLocalSlotNum();
667 //GameSlot *localSlot = getSlot(localSlotNum);
668 Bool sawGameEnd = (endFrame > 0);// && localSlot->lastFrameInGame() <= endFrame);
669 Int winningTeam = -1;
670 Int numPlayers = 0;
671 Int numTeamsAtGameEnd = 0;
672 Int lastTeamAtGameEnd = -1;
673 for (i=0; i<MAX_SLOTS; ++i)
674 {
675 AsciiString playerName;
676 playerName.format("player%d", i);
677 Player *p = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
678 if (p)
679 {
680 ++numPlayers;
681 if (TheVictoryConditions->hasAchievedVictory(p))
682 {
683 winningTeam = getSlot(i)->getTeamNumber();
684 }
685
686 // check if he lasted
687 GameSlot *slot = getSlot(i);
688 if (!slot->disconnected())
689 {
690 if (slot->getTeamNumber() != lastTeamAtGameEnd || numTeamsAtGameEnd == 0)
691 {
692 lastTeamAtGameEnd = slot->getTeamNumber();
693 ++numTeamsAtGameEnd;
694 }
695 }
696 }
697 }
698
699 AsciiString results;
700 results.format("seed=%d,slotNum=%d,sawDesync=%d,sawGameEnd=%d,winningTeam=%d,disconEnd=%d,duration=%d,numPlayers=%d,isQM=%d",
701 getSeed(), localSlotNum, TheNetwork->sawCRCMismatch(), sawGameEnd, winningTeam, (numTeamsAtGameEnd != 0),
702 endFrame, numPlayers, m_isQM);
703
704 Int playerID = 0;
705 for (i=0; i<MAX_SLOTS; ++i)
706 {
707 AsciiString playerName;
708 playerName.format("player%d", i);
709 Player *p = ThePlayerList->findPlayerWithNameKey(NAMEKEY(playerName));
710 if (p)
711 {
712 GameSpyGameSlot *slot = &(m_GameSpySlot[i]);
713 ScoreKeeper *keeper = p->getScoreKeeper();
714 AsciiString playerName = slot->getLoginName();
715 Int gsPlayerID = slot->getProfileID();
716 AsciiString locale = slot->getLocale();
717 Int fps = TheNetwork->getAverageFPS();
718 Int unitsKilled = keeper->getTotalUnitsDestroyed();
719 Int unitsLost = keeper->getTotalUnitsLost();
720 Int unitsBuilt = keeper->getTotalUnitsBuilt();
721 Int buildingsKilled = keeper->getTotalBuildingsDestroyed();
722 Int buildingsLost = keeper->getTotalBuildingsLost();
723 Int buildingsBuilt = keeper->getTotalBuildingsBuilt();
724 Int earnings = keeper->getTotalMoneyEarned();
725 Int techCaptured = keeper->getTotalTechBuildingsCaptured();
726 Bool disconnected = slot->disconnected();
727
728 AsciiString playerStr;
729 playerStr.format(",player%d=%s,playerID%d=%d,locale%d=%s",
730 playerID, playerName.str(), playerID, gsPlayerID, playerID, locale.str());
731 results.concat(playerStr);
732 playerStr.format(",unitsKilled%d=%d,unitsLost%d=%d,unitsBuilt%d=%d",
733 playerID, unitsKilled, playerID, unitsLost, playerID, unitsBuilt);
734 results.concat(playerStr);
735 playerStr.format(",buildingsKilled%d=%d,buildingsLost%d=%d,buildingsBuilt%d=%d",
736 playerID, buildingsKilled, playerID, buildingsLost, playerID, buildingsBuilt);
737 results.concat(playerStr);
738 playerStr.format(",fps%d=%d,cash%d=%d,capturedTech%d=%d,discon%d=%d",
739 playerID, fps, playerID, earnings, playerID, techCaptured, playerID, disconnected);
740 results.concat(playerStr);
741
742 ++playerID;
743 }
744 }
