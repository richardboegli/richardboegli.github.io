87// Function: Fix_Single_Ampersands()
88//
89// Purpose: To replace each "&" with "&&" for display in a dialog.
90// Some dialogs mistake a single "&" for an accelerator key.
91//
92// Input: pszString - any NULL terminated string.
93//
94// Returns: VOID (returns nothing)
95//
96// Comments: Modifies the characters in pszString.
97//
98//---------------------------------------------------------------------------
99
100void Fix_Single_Ampersands ( LPSTR pszString, bool upper_case )
101{
102 char pszTemp[ MAX_PATH ]; // variable to hold the string passed
103 char pszOld[ MAX_PATH ]; // variable to hold the string passed
104 char * letter;
105 int i = 0;
106
107 lstrcpy((LPSTR)pszOld, (LPSTR)pszString );
108 letter = pszOld;
109 memset ( pszTemp, '\0', MAX_PATH );
110
111 //----------------------------------------------------------------------
112 // While our ptr has not passed the end of the string...
113 //----------------------------------------------------------------------
114 while (*letter != '\0') {
115
116 if (*letter == '&') {
117
118 pszTemp[i++] = '&';
119 pszTemp[i++] = '&';
120 letter++;
121
122 } else {
123
124 if ( upper_case ) {
125 pszTemp[i++] = (char) toupper( *( letter++ ));
126 } else {
127 pszTemp[i++] = *(letter++);
128 }
129 }
130 }
131 strcpy((LPSTR)pszString, (LPSTR)pszTemp );
132}

237/******************************************************************************
238 * Load_Alloc_Data -- Allocates a buffer and loads the file into it. *
239 * *
240 * This is the C++ replacement for the Load_Alloc_Data function. It will *
241 * allocate the memory big enough to hold the file & read the file into it. *
242 * *
243 * INPUT: file -- The file to read. *
244 * mem -- The memory system to use for allocation. *
245 * *
246 * OUTPUT: Returns with a pointer to the allocated and filled memory block. *
247 * *
248 * WARNINGS: none *
249 * *
250 * HISTORY: *
251 * 10/17/1994 JLB : Created. *
252 *============================================================================*/
253
254void * Load_Alloc_Data( char *filename, long *filesize )
255{
256 int size, bytes_read;
257 void *ptr = NULL;
258 StandardFileClass file;
259
260 //-------------------------------------------------------------------------
261 // Open file in READ ONLY mode. If fails, exit.
262 //-------------------------------------------------------------------------
263 file.Open( filename, MODE_READ_ONLY );
264 if ( !file.Query_Open()) {
265 return( NULL );
266 }
267
268 //-------------------------------------------------------------------------
269 // Get filesize and create a buffer.
270 //-------------------------------------------------------------------------
271 size = file.Query_Size();
272 ptr = (void*)malloc(size + 1);
273 if ( !ptr ) {
274 return( NULL );
275 }
276
277 //-------------------------------------------------------------------------
278 // Read data into the buffer, close the file.
279 //-------------------------------------------------------------------------
280 memset( ptr, '\0', size + 1 );
281 bytes_read = file.Read( ptr, size );
282 file.Close();
283
284 //-------------------------------------------------------------------------
285 // Check return bytes. It should match the file size.
286 //-------------------------------------------------------------------------
287 assert( bytes_read == size );
288 if ( bytes_read != size ) {
289 free(ptr);
290 return( NULL );
291 }
292
293 if ( filesize != NULL ) {
294 *filesize = (long)size;
295 }
296 return( ptr );
297}
