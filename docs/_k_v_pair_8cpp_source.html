31std::string intToString(int val)
32{
33 std::string s = "";
34 bool neg = (val < 0);
35 if (val < 0)
36 {
37 val = -val;
38 }
39 if (val == 0)
40 return "0";
41
42 char buf[2];
43 buf[1] = 0;
44 while (val)
45 {
46 buf[0] = '0' + val%10;
47 val /= 10;
48 s.insert(0, buf);
49 }
50 if (neg)
51 s.insert(0, "-");
52 return s;
53}

55static std::string trim(std::string s, const std::string& delim)
56{
57 unsigned int i;
58 i = s.find_first_not_of(delim);
59 if (i != s.npos)
60 {
61 s = s.substr(i);
62 }
63
64 i = s.find_last_not_of(delim);
65 if (i>=0 && i<s.npos)
66 {
67 s = s.substr(0, i+1);
68 }
69
70 return s;
71}

73static KeyValueMap parseIntoKVPairs( std::string s, const std::string& delim )
74{
75 KeyValueMap m;
76 bool done = false;
77 std::string kv, k, v;
78 while (!done)
79 {
80 int comma;
81 comma = s.find_first_of(delim, 0);
82 if (comma < 1)
83 {
84 done = true;
85 kv = s;
86 }
87 else
88 {
89 kv = s.substr(0, comma);
90 s = s.substr(comma+1);
91 }
92 s = trim(s, delim);
93 kv = trim(kv, delim);
94
95 if (!kv.empty())
96 {
97 int equals = kv.find_first_of('=', 0);
98 if (equals < 1)
99 {
100 done = true;
101 }
102 else
103 {
104 k = trim(trim(kv.substr(0, equals), delim), " \t");
105 v = trim(trim(kv.substr(equals+1), delim), " \t");
106
107 if (!k.empty() && !v.empty())
108 {
109 m[k] = v;
110 }
111 }
112 }
113 }
114 return m;
115}

130void KVPairClass::readFromFile( const std::string& in, const std::string& delim )
131{
132 m_map.clear();
133 FILE *fp = fopen(in.c_str(), "rb");
134 if (fp)
135 {
136 std::string s = "";
137 fseek(fp, 0, SEEK_END);
138 int len = ftell(fp);
139 fseek(fp, 0, SEEK_SET);
140
141 char *buf = new char[len + 1];
142 memset(buf, 0, len+1);
143 fread(buf, 1, len, fp);
144 fclose(fp);
145
146 s.append(buf);
147 delete[] buf;
148 set(s, delim);
149 }
150}
