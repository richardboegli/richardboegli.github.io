137void TerrainMaterial::updateLabel(void)
138{
139 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
140 if (!pDoc) return;
141
142 AsciiString name = pDoc->GetHeightMap()->getTexClassUiName(m_currentFgTexture);
143 const char *tName = name.str();
144 if (tName == NULL || tName[0] == 0) {
145 tName = pDoc->GetHeightMap()->getTexClassUiName(m_currentFgTexture).str();
146 }
147 if (tName == NULL) {
148 return;
149 }
150 const char *leaf = tName;
151 while (*tName) {
152 if ((tName[0] == '\\' || tName[0] == '/')&& tName[1]) {
153 leaf = tName+1;
154 }
155 tName++;
156 }
157 CWnd *pLabel = GetDlgItem(IDC_TERRAIN_NAME);

244HTREEITEM TerrainMaterial::findOrAdd(HTREEITEM parent, char *pLabel)
245{
246 TVINSERTSTRUCT ins;
247 char buffer[_MAX_PATH];
248 ::memset(&ins, 0, sizeof(ins));
249 HTREEITEM child = m_terrainTreeView.GetChildItem(parent);
250 while (child != NULL) {
251 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
252 ins.item.hItem = child;
253 ins.item.pszText = buffer;
254 ins.item.cchTextMax = sizeof(buffer)-2;
255 m_terrainTreeView.GetItem(&ins.item);
256 if (strcmp(buffer, pLabel) == 0) {
257 return(child);
258 }
259 child = m_terrainTreeView.GetNextSiblingItem(child);
260 }
261
262 // not found, so add it.
263 ::memset(&ins, 0, sizeof(ins));
264 ins.hParent = parent;
265 ins.hInsertAfter = TVI_LAST;
266 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
267 ins.item.lParam = -1;
268 ins.item.pszText = pLabel;
269 ins.item.cchTextMax = strlen(pLabel);
270 child = m_terrainTreeView.InsertItem(&ins);
271 return(child);
272}

275void TerrainMaterial::addTerrain(char *pPath, Int terrainNdx, HTREEITEM parent)
276{
277 TerrainType *terrain = TheTerrainTypes->findTerrain( WorldHeightMapEdit::getTexClassName( terrainNdx ) );
278 Bool doAdd = FALSE;
279 char buffer[_MAX_PATH];
280 //
281 // if we have a 'terrain' entry, it means that our terrain index was properly defined
282 // in an INI file, otherwise it was from eval textures. We will sort all of
283 // the eval texture entries in a tree leaf all their own while the others are
284 // sorted according to a field specified in INI
285 //
286 if( terrain )
287 {
288 if (terrain->isBlendEdge()) {
289 return; // Don't add blend edges to the materials list.
290 }
291 for( TerrainClass i = TERRAIN_NONE; i < TERRAIN_NUM_CLASSES; i = (TerrainClass)(i + 1) )
292 {
293
294 if( terrain->getClass() == i )
295 {
296
297 parent = findOrAdd( parent, terrainTypeNames[ i ] );
298 break; // exit for
299
300 } // end if
301
302 } // end for i
303
304 // set the name in the tree view to that of the entry
305 strcpy( buffer, terrain->getName().str() );
306
307 doAdd = TRUE;
308 } // end if
309 else if (!WorldHeightMapEdit::getTexClassIsBlendEdge(terrainNdx))
310 {
311
312 // all these old entries we will put in a tree for eval textures
313 parent = findOrAdd( parent, "**Eval" );
314 Int i=0;
315 while (pPath[i] && i<sizeof(buffer)) {
316 if (pPath[i] == 0) {
317 return;
318 }
319 if (pPath[i] == '\\' || pPath[i] == '/') {
320 if (i>0 && (i>1 || buffer[0]!='.') ) { // skip the "." directory.
321 buffer[i] = 0;
322 parent = findOrAdd(parent, buffer);
323 }
324 pPath+= i+1;
325 i = 0;
326 }
327 buffer[i] = pPath[i];
328 buffer[i+1] = 0; // terminate at next character
329 doAdd = TRUE;
330 i++;
331 }
332 } // end else
333
334 Int tilesPerRow = TEXTURE_WIDTH/(2*TILE_PIXEL_EXTENT+TILE_OFFSET);
335 Int availableTiles = 4 * tilesPerRow * tilesPerRow;
336 Int percent = (WorldHeightMapEdit::getTexClassNumTiles(terrainNdx)*100 + availableTiles/2) / availableTiles;
337
338 char label[_MAX_PATH];
339 sprintf(label, "%d%% %s", percent, buffer);
340
341
342 if( doAdd )
343 {
344 if (percent<3 && defaultMaterialIndex==0) {
345 defaultMaterialIndex = terrainNdx;
346 }
347 TVINSERTSTRUCT ins;
348
349 ::memset(&ins, 0, sizeof(ins));
350 ins.hParent = parent;
351 ins.hInsertAfter = TVI_LAST;
352 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
353 ins.item.lParam = terrainNdx;
354 ins.item.pszText = label;
355 ins.item.cchTextMax = strlen(label)+2;
356 m_terrainTreeView.InsertItem(&ins);
357 }
358

464BOOL TerrainMaterial::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
465{
466 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
467 if (pHdr->hdr.hwndFrom == m_terrainTreeView.m_hWnd) {
468 if (pHdr->hdr.code == TVN_SELCHANGED) {
469 HTREEITEM hItem = m_terrainTreeView.GetSelectedItem();
470 TVITEM item;
471 ::memset(&item, 0, sizeof(item));
472 item.mask = TVIF_HANDLE|TVIF_PARAM;
473 item.hItem = hItem;
474 m_terrainTreeView.GetItem(&item);
475 if (item.lParam >= 0) {
476 Int texClass = item.lParam;
477 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
478 if (!pDoc) return 0;
479
480 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
481 if (!pMap) return 0;
482 if (m_updating) return 0;
483 if (pMap->canFitTexture(texClass)) {
484 m_currentFgTexture = texClass;
485 updateLabel();
486 m_terrainSwatches.Invalidate();
487 } else {
488 if (m_currentFgTexture != texClass) {
489 // Tried to switch to a too large texture.
490 ::AfxMessageBox(IDS_TEXTURE_TOO_LARGE);
491 ::AfxGetMainWnd()->SetFocus();
492 }
493 m_currentFgTexture = texClass;
494 updateLabel();
495 m_terrainSwatches.Invalidate();
496 }
497 } else if (!(item.state & TVIS_EXPANDEDONCE) ) {
498 HTREEITEM child = m_terrainTreeView.GetChildItem(hItem);
499 while (child != NULL) {
500 hItem = child;
501 child = m_terrainTreeView.GetChildItem(hItem);
502 }
503 if (hItem != m_terrainTreeView.GetSelectedItem()) {
504 m_terrainTreeView.SelectItem(hItem);
505 }
506 }
507 }
508 }
509
510 return CDialog::OnNotify(wParam, lParam, pResult);
511}
