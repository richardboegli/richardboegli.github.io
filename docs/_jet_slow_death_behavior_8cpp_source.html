152void JetSlowDeathBehavior::onDie( const DamageInfo *damageInfo )
153{
154 Object *us = getObject();
155
156 // if the jet is on the ground we do just our ground fx death
157 if( us->isSignificantlyAboveTerrain() == FALSE || us->getStatusBits().test( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
158 {
159 const JetSlowDeathBehaviorModuleData *modData = getJetSlowDeathBehaviorModuleData();
160
161 // execute fx
162 FXList::doFXObj( modData->m_fxOnGroundDeath, us );
163
164 // execute ocl
165 ObjectCreationList::create( modData->m_oclOnGroundDeath, us, NULL );
166
167 // destroy object
168 TheGameLogic->destroyObject( us );
169
170 } // end if
171 else
172 {
173
174 // extend base class for slow death and begin the slow death behavior
175 SlowDeathBehavior::onDie( damageInfo );
176
177 } // end else
178
179 getObject()->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) );
180
181} // end onDie

185void JetSlowDeathBehavior::beginSlowDeath( const DamageInfo *damageInfo )
186{
187
188 // extend functionality
189 SlowDeathBehavior::beginSlowDeath( damageInfo );
190
191 // get our info
192 Object *us = getObject();
193 const JetSlowDeathBehaviorModuleData *modData = getJetSlowDeathBehaviorModuleData();
194
195 // record the frame we died on
196 m_timerDeathFrame = TheGameLogic->getFrame();
197
198 // do some effects
199 FXList::doFXObj( modData->m_fxInitialDeath, us );
200 ObjectCreationList::create( modData->m_oclInitialDeath, us, NULL );
201
202 // start audio loop playing
203 m_deathLoopSound = modData->m_deathLoopSound;
204 if( m_deathLoopSound.getEventName().isEmpty() == FALSE )
205 {
206
207 m_deathLoopSound.setObjectID( us->getID() );
208 m_deathLoopSound.setPlayingHandle( TheAudio->addAudioEvent( &m_deathLoopSound ) );
209
210 } // end if
211
212 // initialize our roll rate to that defined as the initial value in the module data
213 m_rollRate = modData->m_rollRate;
214
215 // set the locomotor so that the plane starts falling
216 Locomotor *locomotor = us->getAIUpdateInterface()->getCurLocomotor();
217 locomotor->setMaxLift( -TheGlobalData->m_gravity * (1.0f - modData->m_fallHowFast) );
218
219 // do not allow the jet to turn anymore
220 locomotor->setMaxTurnRate( 0.0f );
221
222} // end beginSlowDeath

226UpdateSleepTime JetSlowDeathBehavior::update( void )
227{
228
229 // extend functionality of base class
230 SlowDeathBehavior::update();
231
232 // if the death is not activated, do nothing else
233 if( isSlowDeathActivated() == FALSE )
234 return UPDATE_SLEEP_NONE;
235
236 // get object info
237 Object *us = getObject();
238 const JetSlowDeathBehaviorModuleData *modData = getJetSlowDeathBehaviorModuleData();
239
240 // roll us around in the air
241 PhysicsBehavior *physics = us->getPhysics();
242 DEBUG_ASSERTCRASH( physics, ("JetSlowDeathBehavior::beginSlowDeath - '%s' has no physics\n",
243 us->getTemplate()->getName().str()) );
244 if( physics )
245 physics->setRollRate( m_rollRate );
246
247 // adjust the roll rate over time
248 m_rollRate *= modData->m_rollRateDelta;
249
250 // do effects for death while in the air
251 if( m_timerOnGroundFrame == 0 )
252 {
253
254 PathfindLayerEnum layer = TheTerrainLogic->getLayerForDestination(us->getPosition());
255 us->setLayer(layer);
256 Real height;
257 if (layer == LAYER_GROUND)
258 {
259 // (this is more efficient than getGroundHeight because the info is cached)
260 height = us->getHeightAboveTerrain();
261 }
262 else
263 {
264 Real layerHeight = TheTerrainLogic->getLayerHeight( us->getPosition()->x, us->getPosition()->y, layer );
265 height = us->getPosition()->z - layerHeight;
266 // slop a little bit for bridges, since we tend to end up fractionally
267 // above 'em, and it's easier to just slop it here
268 if (height >= 0.0f && height <= 1.0f)
269 height = 0.0f;
270 }
271
272
273
274 Bool hitATree = FALSE;
275 // Here we want to make sure we crash if we collide with a tree on the way down
276 PhysicsBehavior *phys = us->getPhysics();
277 if ( m_timerOnGroundFrame == 0 && phys )
278 {
279 ObjectID treeID = phys->getLastCollidee();
280 Object *tree = TheGameLogic->findObjectByID( treeID );
281 if ( tree )
282 {
283 if (tree->isKindOf( KINDOF_SHRUBBERY ) )
284 hitATree = TRUE;
285 }
286 }
287
288
289
290 // when we've hit the ground, we're totally done
291 if( height <= 0.0f || hitATree )
292 {
293
294 // stop the death looping sound at the right time
295 TheAudio->removeAudioEvent( m_deathLoopSound.getPlayingHandle() );
296
297 // do some effects
298 FXList::doFXObj( modData->m_fxHitGround, us );
299 ObjectCreationList::create( modData->m_oclHitGround, us, NULL );
300
301 // we are now on the ground
302 m_timerOnGroundFrame = TheGameLogic->getFrame();
303
304 // start us rolling on another axis too
305 if( physics )
306 physics->setPitchRate( modData->m_pitchRate );
307
308 } // end if
309
310 // timers for the secondary effect
311 if( m_timerDeathFrame != 0 &&
312 TheGameLogic->getFrame() - m_timerDeathFrame >= modData->m_delaySecondaryFromInitialDeath )
313 {
314
315 // do some effects
316 FXList::doFXObj( modData->m_fxSecondary, us );
317 ObjectCreationList::create( modData->m_oclSecondary, us, NULL );
318
319 // clear the death frame timer since we've already executed the event now
320 m_timerDeathFrame = 0;
321
322 } //end if
323
324 } // end if
325 else
326 {
327 // we are on the ground, pay attention to the final explosion timers
328 if( TheGameLogic->getFrame() - m_timerOnGroundFrame >= modData->m_delayFinalBlowUpFromHitGround )
329 {
330
331 // do some effects
332 FXList::doFXObj( modData->m_fxFinalBlowUp, us );
333 ObjectCreationList::create( modData->m_oclFinalBlowUp, us, NULL );
334
335 // we're all done now
336 TheGameLogic->destroyObject( us );
337
338 } // end if
339
340 } // end else
341
342 return UPDATE_SLEEP_NONE;
343
344} // end update
