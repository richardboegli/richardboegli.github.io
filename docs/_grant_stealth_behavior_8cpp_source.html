57struct GrantStealthPlayerScanHelper
58{
59 KindOfMaskType m_kindOfToTest;
60 Object *m_theGrantor;
61 ObjectPointerList *m_objectList;
62};
63
64static void checkForGrantStealth( Object *testObj, void *userData )
65{
66 GrantStealthPlayerScanHelper *helper = (GrantStealthPlayerScanHelper*)userData;
67 ObjectPointerList *listToAddTo = helper->m_objectList;
68
69 if( testObj->isEffectivelyDead() )
70 return;
71
72 if( testObj->getControllingPlayer() != helper->m_theGrantor->getControllingPlayer() )
73 return;
74
75 if( testObj->isOffMap() )
76 return;
77
78 if( !testObj->isAnyKindOf(helper->m_kindOfToTest) )
79 return;
80
81 listToAddTo->push_back(testObj);
82
83 if( testObj->getContain() )
84 {
85 // have to tag visible riders too, or they will float around and look silly.
86 Object *rider = (Object*)testObj->getContain()->friend_getRider();
87 if( rider )
88 {
89 listToAddTo->push_back(rider);
90 }
91 }
92}

139UpdateSleepTime GrantStealthBehavior::update( void )
140{
141
142 Object *self = getObject();
143
144 if ( self->isEffectivelyDead())
145 return UPDATE_SLEEP_FOREVER;
146
147 const GrantStealthBehaviorModuleData *d = getGrantStealthBehaviorModuleData();
148 // setup scan filters
149 PartitionFilterRelationship relationship( self, PartitionFilterRelationship::ALLOW_ALLIES );
150 PartitionFilterSameMapStatus filterMapStatus( self );
151 PartitionFilterAlive filterAlive;
152 PartitionFilter *filters[] = { &relationship, &filterAlive, &filterMapStatus, NULL };
153
154
155 m_currentScanRadius += d->m_radiusGrowRate;
156
157 Bool thisIsFinalScan = FALSE;
158 if ( m_currentScanRadius >= d->m_finalRadius )
159 {
160 m_currentScanRadius = d->m_finalRadius;
161 thisIsFinalScan = TRUE;
162 }
163
164 // scan objects in our region
165 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( self->getPosition(), m_currentScanRadius, FROM_CENTER_2D, filters );
166 MemoryPoolObjectHolder hold( iter );
167 // GRANT STEALTH TO FRIENDLIES IN RADIUS
168 for( Object *obj = iter->first(); obj; obj = iter->next() )
169 grantStealthToObject( obj );
170
171 if ( thisIsFinalScan )
172 {
173
174 TheGameLogic->destroyObject( self );
175 return UPDATE_SLEEP_FOREVER;
176 }
177
178 return UPDATE_SLEEP_NONE;
179}

183void GrantStealthBehavior::grantStealthToObject( Object *obj )
184{
185
186 if ( obj == getObject() )
187 return;
188
189
190 const GrantStealthBehaviorModuleData *d = getGrantStealthBehaviorModuleData();
191 if ( ! obj->isAnyKindOf( d->m_kindOf ) )
192 return;
193
194 StealthUpdate* stealth = obj->getStealth();
195 if( stealth )
196 {
197 stealth->receiveGrant();
198 Drawable *draw = obj->getDrawable();
199 if ( draw )
200 {
201 draw->flashAsSelected();
202 }
203 }
204
205
206}
