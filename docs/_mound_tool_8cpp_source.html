101void MoundTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
102{
103 if (m != TRACK_L) return;
104
105// WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
106 // just in case, release it.
107 REF_PTR_RELEASE(m_htMapEditCopy);
108 m_htMapEditCopy = pDoc->GetHeightMap()->duplicate();
109 m_prevXIndex = -1;
110 m_prevYIndex = -1;
111 REF_PTR_RELEASE(m_htMapSaveCopy);
112 m_htMapSaveCopy = m_htMapEditCopy->duplicate();
113 m_lastMoveTime = ::GetTickCount();
114 m_lastMoveTime -= MIN_DELAY_TIME + 1; // Make the tool fire the first time.
115 mouseMoved(m, viewPt, pView, pDoc);
116}

129void MoundTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
130{
131 Coord3D cpt;
132 pView->viewToDocCoords(viewPt, &cpt);
133 DrawObject::setFeedbackPos(cpt);
134 if (m != TRACK_L) return;
135
136 Int curTime = ::GetTickCount();
137 Int deltaTime = curTime - m_lastMoveTime;
138 if (deltaTime < MIN_DELAY_TIME) return;
139 m_lastMoveTime = curTime;
140
141
142 int brushWidth = m_brushWidth;
143 int setFeather = m_brushFeather;
144 if (setFeather>0) {
145 brushWidth += 2*setFeather;
146 brushWidth += 2; // for round brush.
147 }
148
149 CPoint ndx;
150 getCenterIndex(&cpt, m_brushWidth, &ndx, pDoc);
151
152 // do nothing if in the same cell....
153 // if (m_prevXIndex == ndx.x && m_prevYIndex == ndx.y) return;
154
155 m_prevXIndex = ndx.x;
156 m_prevYIndex = ndx.y;
157
158 int sub = brushWidth/2;
159 int add = brushWidth-sub;
160
161 Int htDelta = m_moundHeight;
162 if (!m_raising) htDelta = -m_moundHeight;
163 // round brush
164 Int i, j;
165 for (i=ndx.x-sub; i<ndx.x+add; i++) {
166 if (i<0 || i>=m_htMapEditCopy->getXExtent()) {
167 continue;
168 }
169 for (j=ndx.y-sub; j<ndx.y+add; j++) {
170 if (j<0 || j>=m_htMapEditCopy->getYExtent()) {
171 continue;
172 }
173#if 1
174 // New floating point based blending calculation. jba.
175 Real blendFactor;
176 blendFactor = calcRoundBlendFactor(ndx, i, j, m_brushWidth, m_brushFeather);
177 Int curHeight = m_htMapEditCopy->getHeight(i,j);
178 float fNewHeight = (blendFactor*(htDelta+curHeight))+((1.0f-blendFactor)*curHeight);
179 Int newHeight = floor(fNewHeight+0.5f);
180
181 // check boundary values
182 if (newHeight < m_htMapSaveCopy->getMinHeightValue()) newHeight = m_htMapSaveCopy->getMinHeightValue();
183 if (newHeight > m_htMapSaveCopy->getMaxHeightValue()) newHeight = m_htMapSaveCopy->getMaxHeightValue();
184
185 m_htMapEditCopy->setHeight(i, j, newHeight);
186 pDoc->invalCell(i, j);
187#else
188 // Previous integer calculated mounding and blending.
189 // If not re-enabled by Dec 2001, delete as obsolete.
190 Int xd = abs( (2*(i-(ndx.x-sub)))+1 - brushWidth);
191 Int yd = abs( (2*(j-(ndx.y-sub)))+1 - brushWidth);
192
193 float delta = (float)sqrt(xd*xd+yd*yd);
194 //Int curHeight = m_htMapSaveCopy->getHeight(i,j);
195 Int curHeight = m_htMapEditCopy->getHeight(i,j);
196 Int newHeight = curHeight + htDelta;
197
198 // check boundary values
199 if (newHeight < m_htMapSaveCopy->getMinHeightValue()) newHeight = m_htMapSaveCopy->getMinHeightValue();
200 if (newHeight > m_htMapSaveCopy->getMaxHeightValue()) newHeight = m_htMapSaveCopy->getMaxHeightValue();
201
202 if (delta<m_brushWidth) {
203 m_htMapEditCopy->setHeight(i, j, newHeight);
204 pDoc->invalCell(i, j);
205 } else if (delta<brushWidth+0.7 && setFeather) {
206 Int factor = setFeather+1;
207 factor *= 2;
208 float feather = delta-m_brushWidth;
209 float fNewHeight = ((factor-feather)*(htDelta+curHeight)+(feather*curHeight) )/factor;
210 newHeight = (Int)(fNewHeight);
211 if (newHeight < m_htMapSaveCopy->getMinHeightValue()) newHeight = m_htMapSaveCopy->getMinHeightValue();
212 if (newHeight > m_htMapSaveCopy->getMaxHeightValue()) newHeight = m_htMapSaveCopy->getMaxHeightValue();
213 if (newHeight > m_htMapEditCopy->getHeight(i,j)) {
214 if (htDelta<0) {
215 newHeight = m_htMapEditCopy->getHeight(i,j);
216 }
217 } else {
218 if (htDelta>0) {
219 newHeight = m_htMapEditCopy->getHeight(i,j);
220 }
221 }
222
223 m_htMapEditCopy->setHeight(i, j, newHeight);
224 pDoc->invalCell(i, j);
225 }
226#endif
227 }
228 }
229
230 IRegion2D partialRange;
231 partialRange.lo.x = ndx.x - brushWidth;
232 partialRange.hi.x = ndx.x + brushWidth;
233 partialRange.lo.y = ndx.y - brushWidth;
234 partialRange.hi.y = ndx.y + brushWidth;
235 pDoc->updateHeightMap(m_htMapEditCopy, true, partialRange);
236}
