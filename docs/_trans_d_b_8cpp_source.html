378{

382 ListNode *node;
383 int count = 0;
384
385 text_bin->Clear ();
386 text_id_bin->Clear ();
387 label_bin->Clear ();
388 obsolete_bin->Clear ();
389
390 while ( node = sh.FirstNode ( &labels ) )
391 {
392 node->Remove ();
393 label = (BabylonLabel *) node->Item ();
394 count++;
395 delete label;
396 delete node;
397 }
398
399 while ( node = sh.FirstNode ( &obsolete ) )
400 {
401 node->Remove ();
402 text = (BabylonText *) node->Item ();
403 count++;
404 delete text;
405 delete node;
406 }
407
408 num_obsolete = 0;
409
410 if ( next_string_id != -1 )
411 {
412 next_string_id = START_STRING_ID;
413 }
414
415 if ( count )
416 {
417 Changed ();
418 }
419
420 valid = TRUE;
421
422 return count;

475void TransDB::AddToTree ( CTreeCtrl *tc, HTREEITEM parent, int changes, void (*cb) ( void ) )
476{
477 HTREEITEM item;
478 HTREEITEM ilabels, iobsolete;
479 ListSearch sh;
480 BabylonLabel *label;
481 BabylonText *txt;
482
483 sprintf ( buffer, "%s%c (%d/%d)",name, ChangedSymbol(), NumLabelsChanged(), NumLabels() );
484 item = tc->InsertItem ( buffer, parent );
485 ilabels = tc->InsertItem ( "Labels", item );
486
487 label = FirstLabel ( sh );
488
489 while ( label )
490 {
491 if ( !changes || label->IsChanged ())
492 {
493 label->AddToTree ( tc, ilabels, changes );
494 }
495
496 if ( cb )
497 {
498 cb ( );
499 }
500
501 label = NextLabel ( sh );
502 }
503
504 if ( num_obsolete )
505 {
506 iobsolete = tc->InsertItem ( "Obsolete Strings", item );
507
508 txt = FirstObsolete ( sh );
509
510 while ( txt )
511 {
512 if ( !changes || txt->IsChanged ())
513 {
514 txt->AddToTree ( tc, iobsolete );
515 }
516

985 winfo->SetValid ( TRUE );

987 winfo->SetMissing ( FALSE );

990 FindClose ( handle );
