79Bool ConvertToHijackedVehicleCrateCollide::isValidToExecute( const Object *other ) const
80{
81 if( !CrateCollide::isValidToExecute(other) )
82 {
83 return FALSE;
84 }
85
86 if( other->isEffectivelyDead() )
87 {
88 return FALSE;// can't hijack a dead vehicle
89 }
90
91 if( other->isKindOf( KINDOF_IMMUNE_TO_CAPTURE ) )
92 {
93 return FALSE; //Kris: Patch 1.03 -- Prevent hijackers from being able to hijack battle buses.
94 }
95
96 if( other->isKindOf( KINDOF_AIRCRAFT ) || other->isKindOf( KINDOF_BOAT ) )
97 {
98 //Can't hijack planes and boats!
99 return FALSE;
100 }
101
102 if( other->isKindOf( KINDOF_DRONE ) )
103 {
104 //Can't hijack drones!
105 return FALSE;
106 }
107
108 if( other->getStatusBits().test( OBJECT_STATUS_HIJACKED ) )
109 {
110 return FALSE;// oops, sorry, I'll jack the next one.
111 }
112
113 Relationship r = getObject()->getRelationship( other );
114 //Only hijack enemy objects
115 if( r != ENEMIES )
116 {
117 return FALSE;
118 }
119
120 if( other->isKindOf( KINDOF_TRANSPORT ) )
121 {
122 //Kris: Allow empty transports to be hijacked.
123 if( other->getContain() && other->getContain()->getContainCount() > 0 )
124 {
125 return FALSE;// dustin sez: do not jack vehicles that may carry hostile passengers
126 }
127 }
128
129 //Kris: Make sure you can't hijack any aircraft (or hijack-enter).
130 if( other->isKindOf( KINDOF_AIRCRAFT ) )
131 {
132 return FALSE;
133 }
134
135 //VeterancyLevel veterancyLevel = other->getVeterancyLevel();
136 //if( veterancyLevel >= LEVEL_ELITE )
137 //{
138 // return FALSE;
139 //}
140
141 return TRUE;
142}

146Bool ConvertToHijackedVehicleCrateCollide::executeCrateBehavior( Object *other )
147{
148 //Check to make sure that the other object is also the goal object in the AIUpdateInterface
149 //in order to prevent an unintentional conversion simply by having the terrorist walk too close
150 //to it.
151 //Assume ai is valid because CrateCollide::isValidToExecute(other) checks it.
152 Object *obj = getObject();
153 AIUpdateInterface* ai = obj->getAIUpdateInterface();
154 if (ai && ai->getGoalObject() != other)
155 {
156 return false;
157 }
158
159 TheRadar->tryInfiltrationEvent( other );
160
161 //Before the actual defection takes place, play the "vehicle stolen" EVA
162 //event if the local player is the victim!
163 if( other->isLocallyControlled() )
164 {
165 TheEva->setShouldPlay( EVA_VehicleStolen );
166 }
167
168 other->setTeam( obj->getControllingPlayer()->getDefaultTeam() );
169 other->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_HIJACKED ) );// I claim this car in the name of the GLA
170
171 AIUpdateInterface* targetAI = other->getAIUpdateInterface();
172 targetAI->aiMoveToPosition( other->getPosition(), CMD_FROM_AI );
173 targetAI->aiIdle( CMD_FROM_AI );
174
175
176 //Just in case this target is a dozer, lets make him stop al his dozer tasks, like building and repairing,
177 //So the previous owner does not benefit from these tasks
178 DozerAIInterface * dozerAI = targetAI->getDozerAIInterface();
179 if ( dozerAI )
180 {
181 for (UnsignedInt task = DOZER_TASK_FIRST; task < DOZER_NUM_TASKS; ++task)
182 {
183 dozerAI->cancelTask( (DozerTask)task );
184 }
185 }
186
187 AudioEventRTS hijackEvent( "HijackDriver", obj->getID() );
188 TheAudio->addAudioEvent( &hijackEvent );
189
190 //In order to make things easier for the designers, we are going to transfer the hijacker's name
191 //to the car... so the designer can control the car with their scripts.
192 TheScriptEngine->transferObjectName( obj->getName(), other );
193
194 ExperienceTracker *targetExp = other->getExperienceTracker();
195 ExperienceTracker *jackerExp = obj->getExperienceTracker();
196 if ( targetExp && jackerExp )
197 {
198 VeterancyLevel highestLevel = MAX(targetExp->getVeterancyLevel(),jackerExp->getVeterancyLevel());
199 jackerExp->setVeterancyLevel( highestLevel, FALSE );
200 targetExp->setVeterancyLevel( highestLevel, FALSE );
201 }
202
203
204 Bool targetCanEject = FALSE;
205 BehaviorModule **dmi = NULL;
206 for( dmi = other->getBehaviorModules(); *dmi; ++dmi )
207 {
208 if( (*dmi)->getEjectPilotDieInterface() )
209 {
210 targetCanEject = TRUE;
211 break;
212 }
213 } // end for dmi
214
215 if ( ! targetCanEject )
216 {
217 TheGameLogic->destroyObject( obj );
218 return TRUE;
219 }
220
221 // I we have made it this far, we are going to ride in this vehicle for a while
222 // get the name of the hijackerupdate
223 static NameKeyType key_HijackerUpdate = NAMEKEY( "HijackerUpdate" );
224 HijackerUpdate *hijackerUpdate = (HijackerUpdate*)obj->findUpdateModule( key_HijackerUpdate );
225 if( hijackerUpdate )
226 {
227 hijackerUpdate->setTargetObject( other );
228 hijackerUpdate->setIsInVehicle( TRUE );
229 hijackerUpdate->setUpdate( TRUE );
230
231 // flag bits so hijacker won't be selectible or collideable
232 //while within the vehicle
233 obj->setStatus( MAKE_OBJECT_STATUS_MASK3( OBJECT_STATUS_NO_COLLISIONS, OBJECT_STATUS_MASKED, OBJECT_STATUS_UNSELECTABLE ) );
234 }
235
236 // THIS BLOCK HIDES THE HIJACKER AND REMOVES HIM FROM PARTITION MANAGER
237 // remove object from its group (if any)
238 obj->leaveGroup();
239 if( ai )
240 {
241 //By setting him to idle, we will prevent him from entering the target after this gets called.
242 ai->aiIdle( CMD_FROM_AI );
243 }
244
245 //This is kinda special... we will endow our new ride with our vision and shroud range, since we are driving
246 other->setVisionRange(getObject()->getVisionRange());
247 other->setShroudClearingRange(getObject()->getShroudClearingRange());
248
249 // remove rider from partition manager
250 ThePartitionManager->unRegisterObject( obj );
251
252 // hide the drawable associated with rider
253 if( obj->getDrawable() )
254 obj->getDrawable()->setDrawableHidden( true );
255
256 // By returning FALSE, we will not remove the object (Hijacker)
257 return FALSE;
258// return TRUE;
259}
