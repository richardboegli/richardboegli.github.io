72UnicodeString LANAPIInterface::getErrorStringFromReturnType( ReturnType ret )
73{
74 switch (ret)
75 {
76 case RET_OK:
77 return TheGameText->fetch("LAN:OK");
78 case RET_TIMEOUT:
79 return TheGameText->fetch("LAN:ErrorTimeout");
80 case RET_GAME_FULL:
81 return TheGameText->fetch("LAN:ErrorGameFull");
82 case RET_DUPLICATE_NAME:
83 return TheGameText->fetch("LAN:ErrorDuplicateName");
84 case RET_CRC_MISMATCH:
85 return TheGameText->fetch("LAN:ErrorCRCMismatch");
86 case RET_GAME_STARTED:
87 return TheGameText->fetch("LAN:ErrorGameStarted");
88 case RET_GAME_EXISTS:
89 return TheGameText->fetch("LAN:ErrorGameExists");
90 case RET_GAME_GONE:
91 return TheGameText->fetch("LAN:ErrorGameGone");
92 case RET_BUSY:
93 return TheGameText->fetch("LAN:ErrorBusy");
94 case RET_SERIAL_DUPE:
95 return TheGameText->fetch("WOL:ChatErrorSerialDup");
96 default:
97 return TheGameText->fetch("LAN:ErrorUnknown");
98 }
99}

103void LANAPI::OnAccept( UnsignedInt playerIP, Bool status )
104{
105 if( AmIHost() )
106 {
107
108 for (Int i = 0; i < MAX_SLOTS; i++)
109 {

121 RequestGameOptions( GenerateGameOptionsString(), false );
122 lanUpdateSlotList();

126 {
127 //i'm not the host but if the accept came from the host...

129 {
130 UnicodeString text;

132 OnChat(UnicodeString(L"SYSTEM"), m_localIP, text, LANCHAT_SYSTEM);
133 }

137void LANAPI::OnHasMap( UnsignedInt playerIP, Bool status )
138{
139 if( AmIHost() )
140 {
141
142 for (Int i = 0; i < MAX_SLOTS; i++)
143 {
144 if (m_currentGame->getIP(i) == playerIP)
145 {
146 m_currentGame->getLANSlot(i)->setMapAvailability( status );
147 break;
148 }// if
149 }// for
150 if (i != MAX_SLOTS )
151 {
152 UnicodeString mapDisplayName;
153 const MapMetaData *mapData = TheMapCache->findMap( m_currentGame->getMap() );
154 Bool willTransfer = TRUE;
155 if (mapData)
156 {
157 mapDisplayName.format(L"%ls", mapData->m_displayName.str());
158 if (mapData->m_isOfficial)
159 willTransfer = FALSE;
160 }
161 else
162 {
163 mapDisplayName.format(L"%hs", m_currentGame->getMap().str());
164 willTransfer = WouldMapTransfer(m_currentGame->getMap());
165 }
166 if (!status)
167 {
168 UnicodeString text;
169 if (willTransfer)
170 text.format(TheGameText->fetch("GUI:PlayerNoMapWillTransfer"), m_currentGame->getLANSlot(i)->getName().str(), mapDisplayName.str());
171 else
172 text.format(TheGameText->fetch("GUI:PlayerNoMap"), m_currentGame->getLANSlot(i)->getName().str(), mapDisplayName.str());
173 OnChat(UnicodeString(L"SYSTEM"), m_localIP, text, LANCHAT_SYSTEM);
174 }
175 lanUpdateSlotList();
176 }
177 }//if

190void LANAPI::OnGameStart( void )
191{
192 //DEBUG_LOG(("Map is '%s', preview is '%s'\n", m_currentGame->getMap().str(), GetPreviewFromMap(m_currentGame->getMap()).str()));
193 //DEBUG_LOG(("Map is '%s', INI is '%s'\n", m_currentGame->getMap().str(), GetINIFromMap(m_currentGame->getMap()).str()));
194
195 if (m_currentGame)
196 {
197 LANPreferences pref;
198 AsciiString option;
199 option.format("%d", m_currentGame->getLANSlot( m_currentGame->getLocalSlotNum() )->getPlayerTemplate());
200 pref["PlayerTemplate"] = option;
201 option.format("%d", m_currentGame->getLANSlot( m_currentGame->getLocalSlotNum() )->getColor());
202 pref["Color"] = option;
203 if (m_currentGame->amIHost())
204 {
205 pref["Map"] = AsciiStringToQuotedPrintable(m_currentGame->getMap());
206 pref.setSuperweaponRestricted( m_currentGame->getSuperweaponRestriction() > 0 );
207 pref.setStartingCash( m_currentGame->getStartingCash() );
208 }
209 pref.write();
210
211 m_isInLANMenu = FALSE;
212
213 //m_currentGame->startGame(0);
214
215 // Set up the game network
216 DEBUG_ASSERTCRASH(TheNetwork == NULL, ("For some reason TheNetwork isn't NULL at the start of this game. Better look into that."));
217
218 if (TheNetwork != NULL) {
219 delete TheNetwork;
220 TheNetwork = NULL;
221 }
222
223 // Time to initialize TheNetwork for this game.
224 TheNetwork = NetworkInterface::createNetwork();
225 TheNetwork->init();
226 TheNetwork->setLocalAddress(m_localIP, 8088);
227 TheNetwork->initTransport();
228
229 TheNetwork->parseUserList(m_currentGame);
230
231 if (TheGameLogic->isInGame())
232 TheGameLogic->clearGameData();
233
234 Bool filesOk = DoAnyMapTransfers(m_currentGame);
235
236 // see if we really have the map. if not, back out.
237 TheMapCache->updateCache();
238 if (!filesOk || TheMapCache->findMap(m_currentGame->getMap()) == NULL)
239 {
240 DEBUG_LOG(("After transfer, we didn't really have the map. Bailing...\n"));
241 OnPlayerLeave(m_name);
242 removeGame(m_currentGame);
243 m_currentGame = NULL;
244 m_inLobby = TRUE;
245 if (TheNetwork != NULL) {
246 delete TheNetwork;
247 TheNetwork = NULL;
248 }
249 OnChat(UnicodeString::TheEmptyString, 0, TheGameText->fetch("GUI:CouldNotTransferMap"), LANCHAT_SYSTEM);
250 return;
251 }
252
253 m_currentGame->startGame(0);
254
255 // shutdown the top, but do not pop it off the stack
256 //TheShell->hideShell();
257 // setup the Global Data with the Map and Seed
258 TheWritableGlobalData->m_pendingFile = m_currentGame->getMap();
259
260 // send a message to the logic for a new game
261 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_NEW_GAME );
262 msg->appendIntegerArgument(GAME_LAN);
263
264 TheWritableGlobalData->m_useFpsLimit = false;
265
266 // Set the random seed
267 InitGameLogicRandom( m_currentGame->getSeed() );
268 DEBUG_LOG(("InitGameLogicRandom( %d )\n", m_currentGame->getSeed()));
269 }
270}

272void LANAPI::OnGameOptions( UnsignedInt playerIP, Int playerSlot, AsciiString options )
273{
274 if (!m_currentGame)
275 return;
276
277 if (m_currentGame->getIP(playerSlot) != playerIP)
278 return; // He's not in our game?!?
279
280
281 if (m_currentGame->isGameInProgress())
282 return; // we don't want to process any game options while in game.
283
284 if (playerSlot == 0 && !m_currentGame->amIHost())
285 {
286 m_currentGame->setLastHeard(timeGetTime());
287 AsciiString oldOptions = GameInfoToAsciiString(m_currentGame); // save these off for if we get booted
288 if(ParseGameOptionsString(m_currentGame,options))
289 {
290 lanUpdateSlotList();
291 updateGameOptions();
292 }
293 Bool booted = true;
294 for(Int player = 1; player< MAX_SLOTS; player++)
295 {
296 if(m_currentGame->getIP(player) == m_localIP)
297 {
298 booted = false;
299 break;
300 }
301 }
302 if(booted)
303 {
304 // restore the options with us in so we can save prefs
305 ParseGameOptionsString(m_currentGame, oldOptions);
306 OnPlayerLeave(m_name);
307 }
308
309 }
310 else
311 {
312 // Check for user/host updates
313 {
314 AsciiString key;
315 AsciiString munkee = options;
316 munkee.nextToken(&key, "=");
317 //DEBUG_LOG(("GameOpt request: key=%s, val=%s from player %d\n", key.str(), munkee.str(), playerSlot));
318
319 LANGameSlot *slot = m_currentGame->getLANSlot(playerSlot);
320 if (!slot)
321 return;
322
323 if (key == "User")
324 {
325 slot->setLogin(munkee.str()+1);
326 return;
327 }
328 else if (key == "Host")
329 {
330 slot->setHost(munkee.str()+1);
331 return;
332 }
333 }
334
335 // Parse player requests (side, color, etc)
336 if( AmIHost() && m_localIP != playerIP)
337 {
338 if (options.compare("HELLO") == 0)
339 {
340 m_currentGame->setPlayerLastHeard(playerSlot, timeGetTime());
341 }
342 else
343 {
344 m_currentGame->setPlayerLastHeard(playerSlot, timeGetTime());
345 Bool change = false;
346 Bool shouldUnaccept = false;
347 AsciiString key;
348 options.nextToken(&key, "=");
349 Int val = atoi(options.str()+1);
350 DEBUG_LOG(("GameOpt request: key=%s, val=%s from player %d\n", key.str(), options.str(), playerSlot));
351
352 LANGameSlot *slot = m_currentGame->getLANSlot(playerSlot);
353 if (!slot)
354 return;
355
356 if (key == "Color")
357 {
358 if (val >= -1 && val < TheMultiplayerSettings->getNumColors() && val != slot->getColor() && slot->getPlayerTemplate() != PLAYERTEMPLATE_OBSERVER)
359 {
360 Bool colorAvailable = TRUE;
361 if(val != -1 )
362 {
363 for(Int i=0; i <MAX_SLOTS; i++)
364 {
365 LANGameSlot *checkSlot = m_currentGame->getLANSlot(i);
366 if(val == checkSlot->getColor() && slot != checkSlot)
367 {
368 colorAvailable = FALSE;
369 break;
370 }
371 }
372 }
373 if(colorAvailable)
374 slot->setColor(val);
375 change = true;
376 }
377 else
378 {
379 DEBUG_LOG(("Rejecting invalid color %d\n", val));
380 }
381 }
382 else if (key == "PlayerTemplate")
383 {
384 if (val >= PLAYERTEMPLATE_MIN && val < ThePlayerTemplateStore->getPlayerTemplateCount() && val != slot->getPlayerTemplate())
385 {
386 slot->setPlayerTemplate(val);
387 if (val == PLAYERTEMPLATE_OBSERVER)
388 {
389 slot->setColor(-1);
390 slot->setStartPos(-1);
391 slot->setTeamNumber(-1);
392 }
393 change = true;
394 shouldUnaccept = true;
395 }
396 else
397 {
398 DEBUG_LOG(("Rejecting invalid PlayerTemplate %d\n", val));
399 }
400 }
401 else if (key == "StartPos" && slot->getPlayerTemplate() != PLAYERTEMPLATE_OBSERVER)
402 {
403
404 if (val >= -1 && val < MAX_SLOTS && val != slot->getStartPos())
405 {
406 Bool startPosAvailable = TRUE;
407 if(val != -1)
408 for(Int i=0; i <MAX_SLOTS; i++)
409 {
410 LANGameSlot *checkSlot = m_currentGame->getLANSlot(i);
411 if(val == checkSlot->getStartPos() && slot != checkSlot)
412 {
413 startPosAvailable = FALSE;
414 break;
415 }
416 }
417 if(startPosAvailable)
418 slot->setStartPos(val);
419 change = true;
420 shouldUnaccept = true;
421 }
422 else
423 {
424 DEBUG_LOG(("Rejecting invalid startPos %d\n", val));
425 }
426 }
427 else if (key == "Team")
428 {
429 if (val >= -1 && val < MAX_SLOTS/2 && val != slot->getTeamNumber() && slot->getPlayerTemplate() != PLAYERTEMPLATE_OBSERVER)
430 {
431 slot->setTeamNumber(val);
432 change = true;
433 shouldUnaccept = true;
434 }
435 else
436 {
437 DEBUG_LOG(("Rejecting invalid team %d\n", val));
438 }
439 }
440 else if (key == "NAT")
441 {
442 if ((val >= FirewallHelperClass::FIREWALL_TYPE_SIMPLE) &&
443 (val <= FirewallHelperClass::FIREWALL_TYPE_DESTINATION_PORT_DELTA))
444 {
445 slot->setNATBehavior((FirewallHelperClass::FirewallBehaviorType)val);
446 DEBUG_LOG(("NAT behavior set to %d for player %d\n", val, playerSlot));
447 change = true;
448 }
449 else
450 {
451 DEBUG_LOG(("Rejecting invalid NAT behavior %d\n", (Int)val));
452 }
453 }
454
455 if (change)
456 {
457 if (shouldUnaccept)
458 m_currentGame->resetAccepted();
459 RequestGameOptions(GenerateGameOptionsString(), true);
460 lanUpdateSlotList();
461 DEBUG_LOG(("Slot value is color=%d, PlayerTemplate=%d, startPos=%d, team=%d\n",
462 slot->getColor(), slot->getPlayerTemplate(), slot->getStartPos(), slot->getTeamNumber()));
463 DEBUG_LOG(("Slot list updated to %s\n", GenerateGameOptionsString().str()));
464 }
465 }
466 }
467 }
468}

552void LANAPI::OnPlayerLeave( UnicodeString player )
553{
554 DEBUG_ASSERTCRASH(!m_inLobby && m_currentGame, ("Game info is gone!"));
555 if (m_inLobby || !m_currentGame || m_currentGame->isGameInProgress())
556 return;
557
558 if (m_name.compare(player) == 0)
559 {
560 // We're leaving. Save options and Pop the shell up a screen.
561 //DEBUG_ASSERTCRASH(false, ("Slot is %d\n", m_currentGame->getLocalSlotNum()));
562 if (m_currentGame && m_currentGame->isInGame() && m_currentGame->getLocalSlotNum() >= 0)
563 {
564 LANPreferences pref;
565 AsciiString option;
566 option.format("%d", m_currentGame->getLANSlot( m_currentGame->getLocalSlotNum() )->getPlayerTemplate());
567 pref["PlayerTemplate"] = option;
568 option.format("%d", m_currentGame->getLANSlot( m_currentGame->getLocalSlotNum() )->getColor());
569 pref["Color"] = option;
570 if (m_currentGame->amIHost())
571 pref["Map"] = AsciiStringToQuotedPrintable(m_currentGame->getMap());
572 pref.write();
573 }
574 LANbuttonPushed = true;
575 DEBUG_LOG(("OnPlayerLeave says we're leaving! pop away!\n"));
576 TheShell->pop();
577 }
578 else
579 {
580 if (m_currentGame && m_currentGame->getIP(0) == m_localIP)
581 {
582 // Force a new slotlist send
583 m_lastResendTime = 0;
584
585 lanUpdateSlotList();
586 RequestGameOptions( GenerateGameOptionsString(), true );
587
588 }
589 }

673void LANAPI::OnChat( UnicodeString player, UnsignedInt ip, UnicodeString message, ChatType format )
674{
675 GameWindow *chatWindow = NULL;
676
677 if (m_inLobby)
678 {
679 chatWindow = listboxChatWindow;
680 }
681 else if( m_currentGame && m_currentGame->isGameInProgress() && TheShell->isShellActive())
682 {
683 chatWindow = listboxChatWindowScoreScreen;
684 }
685 else if( m_currentGame && !m_currentGame->isGameInProgress())
686 {
687 chatWindow = listboxChatWindowLanGame;
688 }
689 if (chatWindow == NULL)
690 return;
691 Int index = -1;
692 UnicodeString unicodeChat;
693 switch (format)
694 {
695 case LANAPIInterface::LANCHAT_SYSTEM:
696 unicodeChat = L"";
697 unicodeChat.concat(message);
698 unicodeChat.concat(L"");
699 index =GadgetListBoxAddEntryText(chatWindow, unicodeChat, chatSystemColor, -1, -1);
700 break;
701 case LANAPIInterface::LANCHAT_EMOTE:
702 unicodeChat = player;
703 unicodeChat.concat(L' ');
704 unicodeChat.concat(message);
705 if (ip == m_localIP)
706 index =GadgetListBoxAddEntryText(chatWindow, unicodeChat, chatLocalActionColor, -1, -1);
707 else
708 index =GadgetListBoxAddEntryText(chatWindow, unicodeChat, chatActionColor, -1, -1);
709 break;
710 case LANAPIInterface::LANCHAT_NORMAL:
711 default:
712 {
713 // Do the language filtering.
714 TheLanguageFilter->filterLine(message);
715
716 Color chatColor = GameMakeColor(255, 255, 255, 255);
717 if (m_currentGame)
718 {
719 Int slotNum = m_currentGame->getSlotNum(player);
720 // it'll be -1 if its invalid.
721 if (slotNum >= 0) {
722 GameSlot *gs = m_currentGame->getSlot(slotNum);
723 if (gs) {
724 Int colorIndex = gs->getColor();
725 MultiplayerColorDefinition *def = TheMultiplayerSettings->getColor(colorIndex);
726 if (def)
727 chatColor = def->getColor();
728 }
729 }
730 }
731
732 unicodeChat = L"[";
733 unicodeChat.concat(player);
734 unicodeChat.concat(L"] ");
735 unicodeChat.concat(message);
736 if (ip == m_localIP)
737 index =GadgetListBoxAddEntryText(chatWindow, unicodeChat, chatColor, -1, -1);
738 else
739 index =GadgetListBoxAddEntryText(chatWindow, unicodeChat, chatColor, -1, -1);
