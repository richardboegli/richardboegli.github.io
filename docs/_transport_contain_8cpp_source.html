160Bool TransportContain::isValidContainerFor(const Object* rider, Bool checkCapacity) const
161{
162
163 // sanity
164 if (!rider)
165 return false;
166
167 // The point of this new code is to determine when something is a "fake" container, to
168 // look at the object inside of it to use that as the valid check. There is a case, when a
169 // paratrooper (an infantry contained in a parachute). In this case, when we pass this object
170 // to contain in a transport plane, we want to check the infantry, not the parachute.
171 if (rider->getContain() && rider->getContain()->isSpecialZeroSlotContainer())
172 {
173 // Report the first thing inside it!
174 const ContainedItemsList *items = rider->getContain()->getContainedItemsList();
175 if (items && !items->empty())
176 {
177 if (items->front())
178 {
179 // Replace the object we are checking with the *first* object contained within it.
180 rider = items->front();
181 }
182 }
183 }
184
185
186 // extend functionality
187 if( OpenContain::isValidContainerFor( rider, checkCapacity ) == false )
188 return false;
189
190// // only allied objects can be transported.
191// // order matters: we want to know if I consider it to be an ally, not vice versa
192// if (getObject()->getRelationship(rider) != ALLIES)
193// return false;
194
195// no... actually, only OUR OWN units can be transported.
196 if (rider->getControllingPlayer() != getObject()->getControllingPlayer())
197 return false;
198
199 Int transportSlotCount = rider->getTransportSlotCount();
200
201 // if 0, this object isn't transportable.
202 if (transportSlotCount == 0)
203 return false;
204
205 if (checkCapacity)
206 {
207 Int containMax = getContainMax();
208 Int containCount = getContainCount();
209
210 return (m_extraSlotsInUse + containCount + transportSlotCount <= containMax);
211
212 }
213 else
214 {
215 return true;
216 }
217}

223void TransportContain::letRidersUpgradeWeaponSet( void )
224{
225
226 const TransportContainModuleData * d = getTransportContainModuleData();
227
228 if ( ! d->m_armedRidersUpgradeWeaponSet )
229 return;
230
231 Object *self = getObject();
232 if ( self == NULL )
233 return;
234
235 Bool anyRiderHasViableWeapon = FALSE;
236
237 const ContainedItemsList* riderList = getContainedItemsList();
238 if( riderList )
239 {
240 ContainedItemsList::const_iterator it;
241 it = riderList->begin();
242
243 while( *it )
244 {
245 Object *rider = *it;
246
247 //Advance to the next iterator
248 it++;
249
250 if ( rider )
251 {
252 if(rider->isKindOf(KINDOF_INFANTRY) == false)
253 continue;
254 Weapon *weapon = NULL;
255 for ( Int w = PRIMARY_WEAPON; w < WEAPONSLOT_COUNT; ++ w )
256 {
257 weapon = rider->getWeaponInWeaponSlot( (WeaponSlotType)w );
258 if ( weapon )
259 {
260 if ( weapon->getTemplate()->isContactWeapon() == FALSE && weapon->isDamageWeapon() == TRUE ) // THIS MAY NEED TO CHECK MORE WEAPON ATTRIBUTES TO WORK BEST
261 {
262 anyRiderHasViableWeapon = TRUE;
263 break;
264 }
265 }
266 }
267 }//end if rider
268
269 }
270 }
271
272 if ( anyRiderHasViableWeapon )
273 self->setWeaponSetFlag( WEAPONSET_PLAYER_UPGRADE );
274 else
275 self->clearWeaponSetFlag( WEAPONSET_PLAYER_UPGRADE );
276
277}

283void TransportContain::onContaining( Object *rider, Bool wasSelected )
284{
285 OpenContain::onContaining( rider, wasSelected );
286
287 // objects inside a transport are held
288 rider->setDisabled( DISABLED_HELD );
289
290 Int transportSlotCount = rider->getTransportSlotCount();
291
292 DEBUG_ASSERTCRASH(transportSlotCount > 0, ("Hmm, this object isnt transportable"));
293 m_extraSlotsInUse += transportSlotCount - 1;
294 DEBUG_ASSERTCRASH(m_extraSlotsInUse >= 0 && m_extraSlotsInUse + getContainCount() <= getContainMax(), ("Hmm, bad slot count"));
295
296 //
297 // when we go from holding nothing to holding something we have a model condition
298 // to visually show the change
299 //
300 if( getContainCount() == 1 )
301 {
302 Drawable *draw = getObject()->getDrawable();
303
304 if( draw )
305 draw->setModelConditionState( MODELCONDITION_LOADED );
306
307 } // end if
308
309 if ( getTransportContainModuleData()->m_armedRidersUpgradeWeaponSet )
310 letRidersUpgradeWeaponSet();
311
312 //Kris: October 20, 2003 - Patch 1.01
313 //Force Jarmen Kell to transfer weapon timer for snipe to and from the combat bike.
314 if( getObject()->isKindOf( KINDOF_CLIFF_JUMPER ) && rider->isKindOf( KINDOF_HERO ) && rider->isKindOf( KINDOF_SALVAGER ) )
315 {
316 //Admittedly brutal hack, but considering the state of the game (post-ship), this is a particularly surgical fix.
317 Weapon *bikeWeapon = getObject()->getWeaponInWeaponSlot( SECONDARY_WEAPON );
318 Weapon *riderWeapon = rider->getWeaponInWeaponSlot( SECONDARY_WEAPON );
319 if( bikeWeapon && riderWeapon )
320 {
321 //Transfer the reload time from the rider to the bike
322 bikeWeapon->transferNextShotStatsFrom( *riderWeapon );
323 }
324 }
325
326}

330void TransportContain::onRemoving( Object *rider )
331{
332 OpenContain::onRemoving(rider);
333
334 // object is no longer held inside a transport
335 rider->clearDisabled( DISABLED_HELD );
336
337 const TransportContainModuleData* d = getTransportContainModuleData();
338
339 if (!d->m_exitBone.isEmpty())
340 {
341 Drawable* draw = getObject()->getDrawable();
342 if (draw)
343 {
344 Coord3D bonePos, worldPos;
345 if (draw->getPristineBonePositions(d->m_exitBone.str(), 0, &bonePos, NULL, 1) == 1)
346 {
347 getObject()->convertBonePosToWorldPos(&bonePos, NULL, &worldPos, NULL);
348 rider->setPosition(&worldPos);
349 }
350 }
351 }
352
353 if (d->m_orientLikeContainerOnExit)
354 {
355 rider->setOrientation(getObject()->getOrientation());
356 }
357
358 if (d->m_keepContainerVelocityOnExit)
359 {
360 PhysicsBehavior* parent = getObject()->getPhysics();
361 PhysicsBehavior* child = rider->getPhysics();
362 if (parent && child)
363 {
364 Coord3D startingForce = *parent->getVelocity();
365 Real mass = child->getMass();
366 startingForce.x *= mass;
367 startingForce.y *= mass;
368 startingForce.z *= mass;
369 child->applyMotiveForce( &startingForce );
370
371 Real pitchRate = child->getCenterOfMassOffset() * d->m_exitPitchRate;
372 child->setPitchRate( pitchRate );
373 }
374 }
375
376 Int transportSlotCount = rider->getTransportSlotCount();
377 DEBUG_ASSERTCRASH(transportSlotCount > 0, ("Hmm, this object isnt transportable"));
378 m_extraSlotsInUse -= transportSlotCount - 1;
379
380#if (defined(_DEBUG) || defined(_INTERNAL))
381 UnsignedInt containCount = getContainCount();
382 UnsignedInt containMax = getContainMax();
383 DEBUG_ASSERTCRASH(m_extraSlotsInUse >= 0 && m_extraSlotsInUse + containCount <= containMax, ("Hmm, bad slot count"));
384#endif
385
386 // when we are empty again, clear the model condition for loaded
387 if( getContainCount() == 0 )
388 {
389 Drawable *draw = getObject()->getDrawable();
390
391 if( draw )
392 draw->clearModelConditionState( MODELCONDITION_LOADED );
393
394 } // end if
395
396 if (getObject()->isAboveTerrain())
397 {
398 // temporarily mark the guy as being allowed to fall
399 // (overriding his locomotor's stick-to-ground attribute).
400 // this will be reset (by PhysicsBehavior) when he touches the ground.
401 PhysicsBehavior* physics = rider->getPhysics();
402 if (physics)
403 physics->setAllowToFall(true);
404 }
405
406 // AI might need help using this transport in a good way. Make the passengers aggressive.
407 //There is no computer player check since Aggressive only means something for computer players anyway
408 if( d->m_goAggressiveOnExit && rider->getAI() )
409 {
410 rider->getAI()->setAttitude( AI_AGGRESSIVE );
411 }
412 if (getObject()->isEffectivelyDead()) {
413 scatterToNearbyPosition(rider);
414 }
415 if( d->m_resetMoodCheckTimeOnExit && rider->getAI() )
416 {
417 rider->getAI()->wakeUpAndAttemptToTarget();
418 }
419
420 m_frameExitNotBusy = TheGameLogic->getFrame() + d->m_exitDelay;
421
422
423 if ( d->m_armedRidersUpgradeWeaponSet )
424 letRidersUpgradeWeaponSet();
425
426 //Kris: October 20, 2003 - Patch 1.01
427 //Force Jarmen Kell to transfer weapon timer for snipe to and from the combat bike.
428 if( getObject()->isKindOf( KINDOF_CLIFF_JUMPER ) && rider->isKindOf( KINDOF_HERO ) && rider->isKindOf( KINDOF_SALVAGER ) )
429 {
430 //Admittedly brutal hack, but considering the state of the game (post-ship), this is a particularly surgical fix.
431 Weapon *bikeWeapon = getObject()->getWeaponInWeaponSlot( SECONDARY_WEAPON );
432 Weapon *riderWeapon = rider->getWeaponInWeaponSlot( SECONDARY_WEAPON );
433 if( bikeWeapon && riderWeapon )
434 {
435 //Transfer the reload time from the bike to the rider
436 riderWeapon->transferNextShotStatsFrom( *bikeWeapon );
437 }
438 }
439
440}

444void TransportContain::createPayload()
445{
446 TransportContainModuleData* self = (TransportContainModuleData*)getTransportContainModuleData();
447
448 Int count = self->m_initialPayload.count;
449 const ThingTemplate* payloadTemplate = TheThingFactory->findTemplate( self->m_initialPayload.name );
450 Object* object = getObject();
451 ContainModuleInterface *contain = object->getContain();
452
453 if( contain )
454 {
455 contain->enableLoadSounds( FALSE );
456 for( int i = 0; i < count; i++ )
457 {
458 //We are creating a transport that comes with a initial payload, so add it now!
459 Object* payload = TheThingFactory->newObject( payloadTemplate, object->getControllingPlayer()->getDefaultTeam() );
460 if( contain->isValidContainerFor( payload, true ) )
461 {
462 contain->addToContain( payload );
463 }
464 else
465 {
466 DEBUG_CRASH( ( "DeliverPayload: PutInContainer %s is full, or not valid for the payload %s!", object->getName().str(), self->m_initialPayload.name.str() ) );
467 }
468 }
469 contain->enableLoadSounds( TRUE );
470 }
471
472 m_payloadCreated = TRUE;
473
474}

478UpdateSleepTime TransportContain::update()
479{
480 const TransportContainModuleData *moduleData = getTransportContainModuleData();
481
482 if( m_payloadCreated == FALSE )
483 createPayload();
484
485 if( moduleData && moduleData->m_healthRegen )
486 {
487 ContainModuleInterface *contain = getObject()->getContain();
488 if( contain )
489 {
490 //This transport has a health regeneration value, so go through and heal all inside.
491 const ContainedItemsList* items = contain->getContainedItemsList();
492 if( items )
493 {
494 ContainedItemsList::const_iterator it;
495 it = items->begin();
496
497 while( *it )
498 {
499 Object *object = *it;
500
501 //Advance to the next iterator
502 it++;
503
504 //Determine if we need healing or not.
505 BodyModuleInterface *body = object->getBodyModule();
506 if( body->getHealth() < body->getMaxHealth() )
507 {
508 //Calculate the health to be regenerated on each unit.
509 Real regen = body->getMaxHealth() * moduleData->m_healthRegen / 100.0f * SECONDS_PER_LOGICFRAME_REAL;
510
511 //Perform the actual healing for this frame.
512// DamageInfo damageInfo;
513// damageInfo.in.m_damageType = DAMAGE_HEALING;
514// damageInfo.in.m_deathType = DEATH_NONE;
515// damageInfo.in.m_sourceID = getObject()->getID();
516// damageInfo.in.m_amount = regen;
517// object->attemptDamage( &damageInfo );
518 object->attemptHealing( regen, getObject() );
519 }
520 }
521 }
522 }
523 }
524 return OpenContain::update(); //extend

536void TransportContain::killRidersWhoAreNotFreeToExit()
537{
538 const TransportContainModuleData* d = getTransportContainModuleData();
539 ContainedItemsList::const_iterator it = getContainList().begin();
540 while( it != getContainList().end() )
541 {
542 // get the object
543 Object *obj = *it;
544 // increment the iterator, since death will pull this guy from the list somewhere else
545 ++it;
546
547 if (!isSpecificRiderFreeToExit(obj)) // only TransportContain has a meaningful failure for isFreeToExit
548 {
549 // If we cannot exit this guy legally, kill the bastard before removeAllContained forces him out.
550 if (d->m_destroyRidersWhoAreNotFreeToExit)
551 TheGameLogic->destroyObject(obj);
552 else
553 obj->kill();
554 }
555 }

558// ------------------------------------------------------------------------------------------------
559// ------------------------------------------------------------------------------------------------
560Bool TransportContain::isSpecificRiderFreeToExit(Object* specificObject)
561{
562 if( specificObject == NULL )
563 return TRUE; // I can, in general, exit people.
564
565 // This is a override, not an extend. I will check for game legality for
566 // okaying the call to exitObjectViaDoor.
567 const Object* me = getObject();
568
569 // this is present solely for some transports to override, so that they can land before
570 // allowing people to exit...
571 const AIUpdateInterface* ai = me->getAIUpdateInterface();
572 if (ai && ai->getAiFreeToExit(specificObject) != FREE_TO_EXIT)
573 return FALSE;
574
575 // I can always kick people out if I am in the air, I know what I'm doing
576 if (me->isUsingAirborneLocomotor())
577 return TRUE;
578
579 const Coord3D *myPosition = me->getPosition();
580 if (!specificObject->getAIUpdateInterface())
581 return FALSE;
582
583 const Locomotor *hisLocomotor = specificObject->getAIUpdateInterface()->getCurLocomotor();
584 if( hisLocomotor == FALSE )
585 return FALSE;
586
587 // He can't get to this spot naturally, so I can't force him there. (amphib transport)
588 if (!TheAI->pathfinder()->validMovementTerrain(me->getLayer(), hisLocomotor, myPosition))
589 return FALSE;
590
591 return TRUE;
592}

594Bool TransportContain::isPassengerAllowedToFire( ObjectID id ) const
595{
596 Object *passenger = TheGameLogic->findObjectByID(id);
597
598 if( passenger != NULL )
599 {
600 //only allow infantry, and turrets and such. no vehicles.
601 if( passenger->isKindOf(KINDOF_INFANTRY) == FALSE )
602 return FALSE;
603 }
604
605 if ( ! getObject() )
606 return FALSE;
607
608 // but wait! I may be riding on an Overlord
609 // This code detects the case of whether the contained passenger is in a bunker riding on an overlord, inside a helix!
610 // Oh my God.
611 const Object *heWhoContainsMe = getObject()->getContainedBy();
612 if ( heWhoContainsMe)
613 {
614 ContainModuleInterface *hisContain = heWhoContainsMe->getContain();
615 DEBUG_ASSERTCRASH( hisContain,("TransportContain::isPassengerAllowedToFire()... CONTAINER WITHOUT A CONTAIN! AARRGH!") );
616 if ( hisContain && hisContain->isSpecialOverlordStyleContainer() )
617 return hisContain->isPassengerAllowedToFire( id );
618 }
619
620
621
622 return OpenContain::isPassengerAllowedToFire();
623}
