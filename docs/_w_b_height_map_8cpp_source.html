82//=============================================================================
83// WBHeightMap::flattenHeights
84//=============================================================================
86//=============================================================================
87void WBHeightMap::flattenHeights(void) {
88#ifndef USE_FLAT_HEIGHT_MAP
89 Real theZ = THE_Z;
90 Int i, j;
91 for (j=0; j<m_numVBTilesY; j++)
92 for (i=0; i<m_numVBTilesX; i++)
93 {
94 static int count = 0;
95 count++;
96 Int numVertex = (VERTEX_BUFFER_TILE_LENGTH*2)*(VERTEX_BUFFER_TILE_LENGTH*2);
97 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexBufferTiles[j*m_numVBTilesX+i]);
98 VERTEX_FORMAT *vbHardware = (VERTEX_FORMAT*)lockVtxBuffer.Get_Vertex_Array();
99 Int vtx;
100 for (vtx=0; vtx<numVertex; vtx++) {
101 vbHardware->z = theZ;
102 vbHardware++;
103 }
104 }
105#endif
106}

144//=============================================================================
145// WBHeightMap::Cast_Ray
146//=============================================================================
150//=============================================================================
151Bool WBHeightMap::Cast_Ray(RayCollisionTestClass & raytest)
152{
153 if (!m_flattenHeights) {
154 return BaseHeightMapRenderObjClass::Cast_Ray(raytest);
155 }
156 Real theZ = THE_Z;
157 TriClass tri;
158 Bool hit = false;
159 Int X,Y;
160 Vector3 normal,P0,P1,P2,P3;
161
162 if (!m_map)
163 return false; //need valid pointer to heightmap samples
164// HeightSampleType *pData = m_map->getDataPtr();
165 //Clip ray to extents of heightfield
166 AABoxClass hbox;
167 LineSegClass lineseg,lineseg2;
168 CastResultStruct result;
169 Int StartCellX;
170 Int EndCellX;
171 Int StartCellY;
172 Int EndCellY;
173 const Int overhang = 2*32; // Allow picking past the edge for scrolling & objects.
174 Vector3 minPt(MAP_XY_FACTOR*(-overhang), MAP_XY_FACTOR*(-overhang), -MAP_XY_FACTOR);
175 Vector3 maxPt(MAP_XY_FACTOR*(m_map->getXExtent()+overhang),
176 MAP_XY_FACTOR*(m_map->getYExtent()+overhang), MAP_HEIGHT_SCALE*m_map->getMaxHeightValue()+MAP_XY_FACTOR);
177 MinMaxAABoxClass mmbox(minPt, maxPt);
178 hbox.Init(mmbox);
179
180 lineseg=raytest.Ray;
181
182 //Set initial ray endpoints
183 P0 = raytest.Ray.Get_P0();
184 P1 = raytest.Ray.Get_P1();
185 result.ComputeContactPoint=true;
186
187 Int p;
188 for (p=0; p<3; p++) {
189 //find intersection point of ray and terrain bounding box
190 if (CollisionMath::Collide(lineseg,hbox,&result))
191 { //ray intersects terrain or starts inside the terrain.
192 if (!result.StartBad) //check if start point inside terrain
193 P0 = result.ContactPoint; //make intersection point the new start of the ray.
194
195 //reverse direction of original ray and clip again to extent of
196 //heightmap
197 result.Fraction=1.0f; //reset the result
198 result.StartBad=false;
199 lineseg2.Set(lineseg.Get_P1(),lineseg.Get_P0()); //reverse line segment
200 if (CollisionMath::Collide(lineseg2,hbox,&result))
201 { if (!result.StartBad) //check if end point inside terrain
202 P1 = result.ContactPoint; //make intersection point the new end pont of ray
203 }
204 } else {
205 return(false);
206 }
207
208 // Take the 2D bounding box of ray and check heights
209 // inside this box for intersection.
210 if (P0.X > P1.X) { //flip start/end points
211 StartCellX = floor(P1.X/MAP_XY_FACTOR);
212 EndCellX = ceil(P0.X/MAP_XY_FACTOR);
213 } else {
214 StartCellX = floor(P0.X/MAP_XY_FACTOR);
215 EndCellX = ceil(P1.X/MAP_XY_FACTOR);
216 }
217 if (P0.Y > P1.Y) { //flip start/end points
218 StartCellY=floor(P1.Y/MAP_XY_FACTOR);
219 EndCellY=ceil(P0.Y/MAP_XY_FACTOR);
220 } else {
221 StartCellY = floor(P0.Y/MAP_XY_FACTOR);
222 EndCellY = ceil(P1.Y/MAP_XY_FACTOR);
223 }
224
225 Vector3 minPt(MAP_XY_FACTOR*(StartCellX-1), MAP_XY_FACTOR*(StartCellY-1), theZ-1);
226 Vector3 maxPt(MAP_XY_FACTOR*(EndCellX+1), MAP_XY_FACTOR*(EndCellY+1), theZ+1);
227 MinMaxAABoxClass mmbox(minPt, maxPt);
228 hbox.Init(mmbox);
229 }
230
231 raytest.Result->ComputeContactPoint=true; //tell CollisionMath that we need point.
232
233 Int offset;
234 for (offset = 1; offset < 5; offset *= 3) {
235 for (Y=StartCellY-offset; Y<=EndCellY+offset; Y++) {
236 //if (Y<0) continue;
237 //if (Y>=m_map->getYExtent()-1) continue;
238
239 for (X=StartCellX-offset; X<=EndCellX+offset; X++) {
240 //test the 2 triangles in this cell
241 // 3-----2
242 // | /|
243 // | / |
244 // |/ |
245 // 0-----1
246
247 //bottom triangle first
248 P0.X=X*MAP_XY_FACTOR;
249 P0.Y=Y*MAP_XY_FACTOR;
250 P0.Z=THE_Z;
251
252 P1.X=(X+1)*MAP_XY_FACTOR;
253 P1.Y=Y*MAP_XY_FACTOR;
254 P1.Z=THE_Z;
255
256 P2.X=(X+1)*MAP_XY_FACTOR;
257 P2.Y=(Y+1)*MAP_XY_FACTOR;
258 P2.Z=THE_Z;
259
260 P3.X=X*MAP_XY_FACTOR;
261 P3.Y=(Y+1)*MAP_XY_FACTOR;
262 P3.Z=THE_Z;
263
264
265 tri.V[0] = &P0;
266 tri.V[1] = &P1;
267 tri.V[2] = &P2;
268
269 tri.N = &normal;
270
271 tri.Compute_Normal();
272
273 hit = hit | CollisionMath::Collide(raytest.Ray, tri, raytest.Result);
274
275 if (raytest.Result->StartBad)
276 return true;
277
278 //top triangle
279 tri.V[0] = &P2;
280 tri.V[1] = &P3;
281 tri.V[2] = &P0;
282
283 tri.N = &normal;
284
285 tri.Compute_Normal();
286
287 hit = hit | CollisionMath::Collide(raytest.Ray, tri, raytest.Result);
288
289 if (hit)
290 raytest.Result->SurfaceType = SURFACE_TYPE_DEFAULT;
291 }
292 if (hit) break;
293 }
294 if (hit) break;
295 }
296 return hit;
297}
