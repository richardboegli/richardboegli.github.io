126const ObjectCreationList* OCLSpecialPower::findOCL() const
127{
128 const OCLSpecialPowerModuleData* d = getOCLSpecialPowerModuleData();
129 const Player* controller = getObject()->getControllingPlayer();
130 if (controller != NULL)
131 {
132 for (std::vector<OCLSpecialPowerModuleData::Upgrades>::const_iterator it = d->m_upgradeOCL.begin();
133 it != d->m_upgradeOCL.end();
134 ++it)
135 {
136 if (controller->hasScience(it->m_science))
137 return it->m_ocl;
138 }
139 }
140 return d->m_defaultOCL;
141}

153void OCLSpecialPower::doSpecialPowerAtLocation( const Coord3D *loc, Real angle, UnsignedInt commandOptions )
154{
155 if (getObject()->isDisabled())
156 return;
157
158 // sanity
159 if( loc == NULL )
160 return;
161
162 // get the module data
163 const OCLSpecialPowerModuleData *modData = getOCLSpecialPowerModuleData();
164
165 Coord3D targetCoord = *loc;
166
167 if( modData->m_isOCLAdjustPositionToPassable )
168 {
169 FindPositionOptions fpOptions;
170 fpOptions.flags = FPF_CLEAR_CELLS_ONLY;
171 fpOptions.maxRadius = MAX_ADJUST_RADIUS;
172 if ( ! ThePartitionManager->findPositionAround(&targetCoord, &fpOptions, &targetCoord) )
173 { // if findPosition() fails, then don't monkey with target Coord!
174 targetCoord = *loc;
175 }
176
177
178 }
179
180 // call the base class action cause we are *EXTENDING* functionality
181 SpecialPowerModule::doSpecialPowerAtLocation( &targetCoord, angle, commandOptions );
182
183 const ObjectCreationList* ocl = findOCL();
184
185 // at what point will the "deliverer" come in
186 Coord3D creationCoord;
187 switch (modData->m_createLoc)
188 {
189 case CREATE_AT_EDGE_NEAR_SOURCE:
190 creationCoord = TheTerrainLogic->findClosestEdgePoint( getObject()->getPosition() );
191 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle );
192 break;
193 case CREATE_AT_EDGE_NEAR_TARGET:
194 creationCoord = TheTerrainLogic->findClosestEdgePoint(&targetCoord);
195 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle );
196 break;
197 case CREATE_AT_EDGE_FARTHEST_FROM_TARGET:
198 creationCoord = TheTerrainLogic->findFarthestEdgePoint(&targetCoord);
199 creationCoord.z += CREATE_ABOVE_LOCATION_HEIGHT;
200 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle );
201 break;
202 case CREATE_AT_LOCATION:
203 // this is the case where the special power stuff originates at the location of the mouse click
204 creationCoord = targetCoord;
205 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle );
206 break;
207 case USE_OWNER_OBJECT:
208 creationCoord.set( &targetCoord );
209 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle, false );
210 break;
211 case CREATE_ABOVE_LOCATION:
212 // this is the case where the special power stuff originates above the location of the mouse click
213 creationCoord = targetCoord;
214 creationCoord.z += CREATE_ABOVE_LOCATION_HEIGHT;
215 ObjectCreationList::create( ocl, getObject(), &creationCoord, &targetCoord, angle );
216 break;
217 }
218}

222void OCLSpecialPower::doSpecialPowerAtObject( Object *obj, UnsignedInt commandOptions )
223{
224 if (getObject()->isDisabled())
225 return;
226
227 // convert to a location
228 if( !obj )
229 return;
230 doSpecialPowerAtLocation( obj->getPosition(), INVALID_ANGLE, commandOptions );
231}

233// ------------------------------------------------------------------------------------------------
234void OCLSpecialPower::doSpecialPower( UnsignedInt commandOptions )
235{
236 if (getObject()->isDisabled())
237 return;
238
239 Coord3D creationCoord;
240 creationCoord.set( getObject()->getPosition() );
241
242 // call the base class action cause we are *EXTENDING* functionality
243 SpecialPowerModule::doSpecialPowerAtLocation( &creationCoord, INVALID_ANGLE, commandOptions );
244
245 const ObjectCreationList* ocl = findOCL();
246 ObjectCreationList::create( ocl, getObject(), &creationCoord, &creationCoord, false );
247}
