85void Win32GameEngine::update( void )
86{
87
88
89 // call the engine normal update
90 GameEngine::update();
91
92 extern HWND ApplicationHWnd;
93 if (ApplicationHWnd && ::IsIconic(ApplicationHWnd)) {
94 while (ApplicationHWnd && ::IsIconic(ApplicationHWnd)) {
95 // We are alt-tabbed out here. Sleep a bit, & process windows
96 // so that we can become un-alt-tabbed out.
97 Sleep(5);
98 serviceWindowsOS();
99
100 if (TheLAN != NULL) {
101 // BGC - need to update TheLAN so we can process and respond to other
102 // people's messages who may not be alt-tabbed out like we are.
103 TheLAN->setIsActive(isActive());
104 TheLAN->update();
105 }
106
107 // If we are running a multiplayer game, keep running the logic.
108 // There is code in the client to skip client redraw if we are
109 // iconic. jba.
110 if (TheGameEngine->getQuitting() || TheGameLogic->isInInternetGame() || TheGameLogic->isInLanGame()) {
111 break; // keep running.
112 }
113 }
114
115 // When we are alt-tabbed out... the MilesAudioManager seems to go into a coma sometimes
116 // and not regain focus properly when we come back. This seems to wake it up nicely.
117 AudioAffect aa = (AudioAffect)0x10;
118 TheAudio->setVolume(TheAudio->getVolume( aa ), aa );
119
120 }
121
122 // allow windows to perform regular windows maintenance stuff like msgs
123 serviceWindowsOS();
124
125} // end update

133void Win32GameEngine::serviceWindowsOS( void )
134{
135 MSG msg;
136 Int returnValue;
137
138 //
139 // see if we have any messages to process, a NULL window handle tells the
140 // OS to look at the main window associated with the calling thread, us!
141 //
142 while( PeekMessage( &msg, NULL, 0, 0, PM_NOREMOVE ) )
143 {
144
145 // get the message
146 returnValue = GetMessage( &msg, NULL, 0, 0 );
147
148 // this is one possible way to check for quitting conditions as a message
149 // of WM_QUIT will cause GetMessage() to return 0
150/*
151 if( returnValue == 0 )
152 {
153
154 setQuitting( true );
155 break;
156
157 }
158*/
159
160 TheMessageTime = msg.time;
161 // translate and dispatch the message
162 TranslateMessage( &msg );
163 DispatchMessage( &msg );
164 TheMessageTime = 0;
165
166 } // end while
167
168} // end ServiceWindowsOS
