101void DemoralizeSpecialPower::doSpecialPowerAtLocation( const Coord3D *loc, Real angle, UnsignedInt commandOptions )
102{
103 if (getObject()->isDisabled())
104 return;
105
106 // sanity
107 if( loc == NULL )
108 return;
109
110 // call the base class action cause we are *EXTENDING* functionality
111 SpecialPowerModule::doSpecialPowerAtLocation( loc, angle, commandOptions );
112
113 // the source of my fiendish power
114 Object *source = getObject();
115
116 // get module data
117 const DemoralizeSpecialPowerModuleData *modData = getDemoralizeSpecialPowerModuleData();
118
119 //
120 // the demoralize special power increases in range and duration of effect the more
121 // units we have captured within us
122 //
123 UnsignedInt duration = modData->m_baseDurationInFrames;
124 Real range = modData->m_baseRange;
125 ContainModuleInterface *contain = source->getContain();
126 if( contain )
127 {
128
129 // for every captured unit we get a bonus
130 duration += contain->getContainCount() * modData->m_bonusDurationPerCapturedInFrames;
131
132 // cap at the max
133 if( duration > modData->m_maxDurationInFrames )
134 duration = modData->m_maxDurationInFrames;
135
136 // increase the range for every prisoner we got
137 range += contain->getContainCount() * modData->m_bonusRangePerCaptured;
138
139 // cap at the max
140 if( range > modData->m_maxRange )
141 range = modData->m_maxRange;
142
143 } // end if
144
145 // scan the objects in the area range and demoralize them
146 PartitionFilterRelationship filter1( source, PartitionFilterRelationship::ALLOW_ENEMIES |
147 PartitionFilterRelationship::ALLOW_NEUTRAL );
148 PartitionFilterAcceptByKindOf filter2( MAKE_KINDOF_MASK( KINDOF_INFANTRY ),
149 KINDOFMASK_NONE );
150 PartitionFilterSameMapStatus filterMapStatus(source);
151 PartitionFilter *filters[] = { &filter1, &filter2, &filterMapStatus, NULL };
152 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( loc,
153 range,
154 FROM_CENTER_2D,
155 filters );
156
157 MemoryPoolObjectHolder hold( iter );
158 AIUpdateInterface *ai;
159 for( Object *obj = iter->first(); obj; obj = iter->next() )
160 {
161
162 // demoralize only affects with ai of course
163 ai = obj->getAIUpdateInterface();
164 if( ai )
165 ai->setDemoralized( duration );
166
167 } // end for obj
168
169 // play an effect at the destination location
170 if( modData->m_fxList )
171 {
172
173 // execute FX
174 FXList::doFXPos( modData->m_fxList, loc );
175
176 } // end if
177
178} // end doSpecialPowerAtLocation
