75struct SeismicSimulationNode
76{
77 SeismicSimulationNode()
78 {
79 m_center.x = 0;
80 m_center.y = 0;
81 m_radius = 0;
82 m_region.lo.x = 0;
83 m_region.lo.y = 0;
84 m_region.hi.x = 0;
85 m_region.hi.y = 0;
86 m_clean = FALSE;
87 callbackFilter = NULL;
88 m_life = 0;
89 m_magnitude = DEFAULT_SEISMIC_SIMULATION_MAGNITUDE;
90
91 }
92 SeismicSimulationNode( const SeismicSimulationNode &ssn )
93 {
94 m_center.x = ssn.m_center.x;
95 m_center.y = ssn.m_center.y;
96 m_radius = ssn.m_radius;
97 m_region.lo.x = ssn.m_region.lo.x;
98 m_region.lo.y = ssn.m_region.lo.y;
99 m_region.hi.x = ssn.m_region.hi.x;
100 m_region.hi.y = ssn.m_region.hi.y;
101 m_clean = ssn.m_clean;
102 callbackFilter= ssn.callbackFilter;
103 m_life = ssn.m_life;
104 m_magnitude = ssn.m_magnitude;
105
106 }
107 SeismicSimulationNode( const Coord3D* ctr, Real rad, Real mag, SeismicSimulationFilterBase *cbf = NULL )
108 {
109 m_center.x = REAL_TO_INT_FLOOR(ctr->x/MAP_XY_FACTOR);
110 m_center.y = REAL_TO_INT_FLOOR(ctr->y/MAP_XY_FACTOR);
111 m_radius = (rad-1)/MAP_XY_FACTOR;
112 UnsignedInt regionSize = rad/MAP_XY_FACTOR;
113 m_region.lo.x = m_center.x - regionSize;
114 m_region.lo.y = m_center.y - regionSize;
115 m_region.hi.x = m_center.x + regionSize;
116 m_region.hi.y = m_center.y + regionSize;
117 m_clean = false;
118 callbackFilter= cbf;
119 m_life = 0;
120 m_magnitude = mag;
121
122 }
123
124 SeismicSimulationFilterBase::SeismicSimStatusCode handleFilterCallback( WorldHeightMapInterfaceClass *heightMap )
125 {
126 if ( callbackFilter == NULL )
127 return SeismicSimulationFilterBase::SEISMIC_STATUS_INVALID;
128
129 ++m_life;
130
131 return callbackFilter->filterCallback( heightMap, this );
132 }
133
134 Real applyGravity( Real velocityIn )
135 {
136 DEBUG_ASSERTCRASH( callbackFilter, ("SeismicSimulationNode::applyGravity() has no callback filter!") );
137
138 if ( callbackFilter == NULL )
139 return velocityIn;//oops, we have no callback!
140
141 return callbackFilter->applyGravityCallback( velocityIn );
142
143 }
144
145 IRegion2D m_region;
146 ICoord2D m_center;
147 Bool m_clean;
148 Real m_magnitude;
149 UnsignedInt m_radius;
150 UnsignedInt m_life;
151
152 SeismicSimulationFilterBase *callbackFilter;
153
154};
