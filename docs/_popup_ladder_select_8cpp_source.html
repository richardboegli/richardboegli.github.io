154 PASS_NONE,
155 PASS_ENTRY,

159static PasswordMode s_currentMode = PASS_NONE;
160static void setPasswordMode(PasswordMode mode)
161{
162 s_currentMode = mode;
163 switch(mode)
164 {
165 case PASS_NONE:
166 if (passwordParent)
167 passwordParent->winHide(TRUE);
168 if (badPasswordParent)
169 badPasswordParent->winHide(TRUE);
170 if (buttonOk)
171 buttonOk->winEnable(TRUE);
172 if (buttonCancel)
173 buttonCancel->winEnable(TRUE);
174 if (textEntryPassword)
175 textEntryPassword->winEnable(FALSE);
176 if (listboxLadderSelect)
177 listboxLadderSelect->winEnable(TRUE);
178 TheWindowManager->winSetFocus(listboxLadderSelect);
179 break;
180 case PASS_ENTRY:
181 if (passwordParent)
182 passwordParent->winHide(FALSE);
183 if (badPasswordParent)
184 badPasswordParent->winHide(TRUE);
185 if (buttonOk)
186 buttonOk->winEnable(FALSE);
187 if (buttonCancel)
188 buttonCancel->winEnable(FALSE);
189 if (textEntryPassword)
190 {
191 textEntryPassword->winEnable(TRUE);
192 GadgetTextEntrySetText(textEntryPassword, UnicodeString::TheEmptyString);
193 }
194 if (listboxLadderSelect)
195 listboxLadderSelect->winEnable(FALSE);
196 TheWindowManager->winSetFocus(textEntryPassword);
197 break;
198 case PASS_ERROR:
199 if (passwordParent)
200 passwordParent->winHide(TRUE);
201 if (badPasswordParent)
202 badPasswordParent->winHide(FALSE);
203 if (buttonOk)
204 buttonOk->winEnable(FALSE);
205 if (buttonCancel)
206 buttonCancel->winEnable(FALSE);
207 if (textEntryPassword)
208 textEntryPassword->winEnable(FALSE);
209 if (listboxLadderSelect)
210 listboxLadderSelect->winEnable(FALSE);
211 TheWindowManager->winSetFocus(parent);
212 break;
213 }
214}

270WindowMsgHandledType PopupLadderSelectInput( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
271{
272 switch( msg )
273 {
274
275 // --------------------------------------------------------------------------------------------
276 case GWM_CHAR:
277 {
278 UnsignedByte key = mData1;
279 UnsignedByte state = mData2;
280// if (buttonPushed)
281// break;
282
283 switch( key )
284 {
285
286 // ----------------------------------------------------------------------------------------
287 case KEY_ESC:
288 {
289
290 //
291 // send a simulated selected event to the parent window of the
292 // back/exit button
293 //
294 if( BitTest( state, KEY_STATE_UP ) )
295 {
296 switch (s_currentMode)
297 {
298 case PASS_NONE:
299 // re-select whatever was chosen before
300 populateLadderComboBox();
301 GameSpyCloseOverlay(GSOVERLAY_LADDERSELECT);
302 break;
303 case PASS_ENTRY:
304 case PASS_ERROR:
305 setPasswordMode(PASS_NONE);
306 break;
307 }
308
309 } // end if
310
311 // don't let key fall through anywhere else
312 return MSG_HANDLED;
313
314 } // end escape
315
316 } // end switch( key )
317
318 } // end char
319
320 } // end switch( msg )
321
322 return MSG_IGNORED;
323

341WindowMsgHandledType PopupLadderSelectSystem( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
342{
343 switch( msg )
344 {
345 // --------------------------------------------------------------------------------------------
346 case GWM_CREATE:
347 {
348 break;
349 } // end create
350 //---------------------------------------------------------------------------------------------
351 case GWM_DESTROY:
352 {
353 parent = NULL;
354 listboxLadderSelect = NULL;
355 listboxLadderDetails = NULL;
356 CustomMatchHideHostPopup(FALSE);
357 break;
358 } // end case
359
360 //----------------------------------------------------------------------------------------------
361 case GWM_INPUT_FOCUS:
362 {
363 // if we're given the opportunity to take the keyboard focus we must say we want it
364 if( mData1 == TRUE )
365 *(Bool *)mData2 = TRUE;
366 break;
367 } // end input
368 //----------------------------------------------------------------------------------------------
369 case GBM_SELECTED:
370 {
371 GameWindow *control = (GameWindow *)mData1;
372 Int controlID = control->winGetWindowId();
373 if (controlID == buttonOkID)
374 {
375 // save selection
376 Int selectPos = -1;
377 GadgetListBoxGetSelected( listboxLadderSelect, &selectPos );
378 if (selectPos < 0)
379 break;
380
381 ladderIndex = (Int)GadgetListBoxGetItemData( listboxLadderSelect, selectPos, 0 );
382 const LadderInfo *li = TheLadderList->findLadderByIndex( ladderIndex );
383 if (li && li->cryptedPassword.isNotEmpty())
384 {
385 // need password asking
386 setPasswordMode(PASS_ENTRY);
387 }
388 else
389 {
390 ladderSelectedCallback();
391 }
392 }
393 else if (controlID == buttonCancelID)
394 {
395 // reset what had been
396 populateLadderComboBox();
397
398 // tear down overlay
399 GameSpyCloseOverlay( GSOVERLAY_LADDERSELECT );
400 }
401 else if (controlID == buttonPasswordOkID)
402 {
403 const LadderInfo *li = TheLadderList->findLadderByIndex( ladderIndex );
404 if (!li || li->cryptedPassword.isEmpty())
405 {
406 // eh? something's not right. just pretend they typed something wrong...
407 setPasswordMode(PASS_ERROR);
408 break;
409 }
410
411 AsciiString pass;
412 pass.translate(GadgetTextEntryGetText(textEntryPassword));
413 if ( pass.isNotEmpty() ) // password ok
414 {
415 AsciiString cryptPass = EncryptString(pass.str());
416 DEBUG_LOG(("pass is %s, crypted pass is %s, comparing to %s\n",
417 pass.str(), cryptPass.str(), li->cryptedPassword.str()));
418 if (cryptPass == li->cryptedPassword)
419 ladderSelectedCallback();
420 else
421 setPasswordMode(PASS_ERROR);
422 }
423 else
424 {
425 setPasswordMode(PASS_ERROR);
426 }
427 }
428 else if (controlID == buttonPasswordCancelID)
429 {
430 setPasswordMode(PASS_NONE);
431 }
432 else if (controlID == buttonBadPasswordOkID)
433 {
434 setPasswordMode(PASS_NONE);
435 }
436 break;
437 } // end input
438
439 //---------------------------------------------------------------------------------------------
440 case GLM_SELECTED:
441 {
442 Int selIndex, selID;
443 GadgetListBoxGetSelected(listboxLadderSelect, &selIndex);
444 if (selIndex < 0)
445 break;
446
447 selID = (Int)GadgetListBoxGetItemData(listboxLadderSelect, selIndex);
448 if (!selID)
449 break;
450
451 updateLadderDetails(selID, staticTextLadderName, listboxLadderDetails);
452 break;
453 } // end GLM_DOUBLE_CLICKED
454
455 //---------------------------------------------------------------------------------------------
456 case GLM_DOUBLE_CLICKED:
457 {
458 GameWindow *control = (GameWindow *)mData1;
459 Int controlID = control->winGetWindowId();
460 Int selectPos = (Int)mData2;
461 GadgetListBoxSetSelected(control, &selectPos);
462
463 if( controlID == listboxLadderSelectID )
464 {
465 TheWindowManager->winSendSystemMsg( parent, GBM_SELECTED,
466 (WindowMsgData)buttonOk, buttonOk->winGetWindowId() );
467 }
468 break;
469 }
470
471 //---------------------------------------------------------------------------------------------
472 case GEM_EDIT_DONE:
473 {
474 GameWindow *control = (GameWindow *)mData1;
475 Int controlID = control->winGetWindowId();
476 if (controlID == textEntryPasswordID)
477 {
478 TheWindowManager->winSendSystemMsg( parent, GBM_SELECTED,
479 (WindowMsgData)(TheWindowManager->winGetWindowFromId(passwordParent, buttonPasswordOkID)), buttonPasswordOkID );
480 }
481 break;
482 }
483
484 default:
485 return MSG_IGNORED;
486
487 } // end switch
488
489 return MSG_HANDLED;
490
491}

498static void updateLadderDetails( Int selID, GameWindow *staticTextLadderName, GameWindow *listboxLadderDetails )
499{
500 if (!staticTextLadderName || !listboxLadderDetails)
501 return;
502
503 GadgetStaticTextSetText(staticTextLadderName, UnicodeString::TheEmptyString);
504 GadgetListBoxReset(listboxLadderDetails);
505
506 const LadderInfo *info = TheLadderList->findLadderByIndex(selID);
507 if (!info)
508 return;
509
510 UnicodeString line;
511 Color color = GameMakeColor( 255, 255, 255, 255 );
512 Color captionColor = GameMakeColor( 0, 255, 255, 255 );
513
514 // name
515 line.format(TheGameText->fetch("GUI:LadderNameAndSize"), info->name.str(), info->playersPerTeam, info->playersPerTeam);
516 GadgetStaticTextSetText(staticTextLadderName, line);
517
518 // location
519 if (!info->location.isEmpty())
520 GadgetListBoxAddEntryText(listboxLadderDetails, info->location, captionColor, -1);
521
522 // homepage
523 line.format(TheGameText->fetch("GUI:LadderURL"), info->homepageURL.str());
524 GadgetListBoxAddEntryText(listboxLadderDetails, line, captionColor, -1);
525
526 // description
527 if (!info->description.isEmpty())
528 GadgetListBoxAddEntryText(listboxLadderDetails, info->description, color, -1);
529
530 // requires password?
531 if (info->cryptedPassword.isNotEmpty())
532 {
533 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch("GUI:LadderHasPassword"), captionColor, -1);
534 }
535
536 // wins limits
537 if (info->minWins)
538 {
539 line.format(TheGameText->fetch("GUI:LadderMinWins"), info->minWins);
540 GadgetListBoxAddEntryText(listboxLadderDetails, line, captionColor, -1);
541 }
542 if (info->maxWins)
543 {
544 line.format(TheGameText->fetch("GUI:LadderMaxWins"), info->maxWins);
545 GadgetListBoxAddEntryText(listboxLadderDetails, line, captionColor, -1);
546 }
547
548 // random factions?
549 if (info->randomFactions)
550 {
551 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch("GUI:LadderRandomFactions"), captionColor, -1);
552 }
553 else
554 {
555 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch("GUI:LadderFactions"), captionColor, -1);
556 }
557
558 // factions
559 AsciiStringList validFactions = info->validFactions;
560 for (AsciiStringListIterator it = validFactions.begin(); it != validFactions.end(); ++it)
561 {
562 AsciiString marker;
563 marker.format("INI:Faction%s", it->str());
564 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch(marker), color, -1);
565 }
566
567 // random maps?
568 if (info->randomMaps)
569 {
570 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch("GUI:LadderRandomMaps"), captionColor, -1);
571 }
572 else
573 {
574 GadgetListBoxAddEntryText(listboxLadderDetails, TheGameText->fetch("GUI:LadderMaps"), captionColor, -1);
575 }
576
577 // maps
578 AsciiStringList validMaps = info->validMaps;
579 for (it = validMaps.begin(); it != validMaps.end(); ++it)
580 {
581 const MapMetaData *md = TheMapCache->findMap(*it);
582 if (md)
583 {
584 GadgetListBoxAddEntryText(listboxLadderDetails, md->m_displayName, color, -1);
585 }
586 }
587}
