117 /*
118 ** Enumeration of firewall behaviors we can detect.
119 **
120 ** It is assumed that all port mangling firewalls change the mangled source port in response to
121 ** an application source port change.
122 */
123 typedef enum tFirewallBehaviorType {
124
125 FIREWALL_MIN = 0,
126
127 /*
128 ** Just used as an initialiser.
129 */
130 FIREWALL_TYPE_UNKNOWN = 0,
131
132 /*
133 ** This is a simple, non-port translating firewall, or there is no firewall at all.
134 */
135 FIREWALL_TYPE_SIMPLE = 1,
136
137 /*
138 ** This is a firewall/NAT with port mangling but it's pretty dumb - it uses the same mangled
139 ** source port regardless of the destination address.
140 */
141 FIREWALL_TYPE_DUMB_MANGLING = 2,
142
143 /*
144 ** This is a smarter firewall/NAT with port mangling that uses different mangled source ports
145 ** for different destination IPs.
146 */
147 FIREWALL_TYPE_SMART_MANGLING = 4,
148
149 /*
150 ** This is a firewall that exhibits the bug as seen in the Netgear firewalls. A previously good
151 ** source port mapping will change in response to unsolicited traffic from a known IP.
152 */
153 FIREWALL_TYPE_NETGEAR_BUG = 8,
154
155 /*
156 ** This firewall has a simple absolute offset port allocation scheme.
157 */
158 FIREWALL_TYPE_SIMPLE_PORT_ALLOCATION = 16,
159
160 /*
161 ** This firewall has a relative offset port allocation scheme. For these firewalls, we have to
162 ** subtract the actual source port from the mangled source port to discover the allocation scheme.
163 ** The mangled port number is based in part on the original source port number.
164 */
165 FIREWALL_TYPE_RELATIVE_PORT_ALLOCATION = 32,
166
167 /*
168 ** This firewall mangles source ports differently depending on the destination port.
169 */
170 FIREWALL_TYPE_DESTINATION_PORT_DELTA = 64,
171
172 FIREWALL_MAX = 128
173
174 } FirewallBehaviorType;

214 /*
215 ** Behavior query functions.
216 */
217 inline Bool isNAT(void) {
218 if (m_behavior == FIREWALL_TYPE_UNKNOWN || (m_behavior & FIREWALL_TYPE_SIMPLE) != 0) {
219 return(FALSE);
220 }
221 return(TRUE);
222 };
223
224 inline Bool isNAT(FirewallBehaviorType behavior) {
225 if (behavior == FIREWALL_TYPE_UNKNOWN || (behavior & FIREWALL_TYPE_SIMPLE) != 0) {
226 return(FALSE);
227 }
228 return(TRUE);
229 };

231 inline Bool isNetgear(FirewallBehaviorType behavior) {
232 if ((behavior & FIREWALL_TYPE_NETGEAR_BUG) != 0) {
233 return(TRUE);
234 }
235 return(FALSE);
236 };

238 inline Bool isNetgear(void) {
