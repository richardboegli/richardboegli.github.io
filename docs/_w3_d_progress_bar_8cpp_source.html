75void W3DGadgetProgressBarDraw( GameWindow *window, WinInstanceData *instData )
76{
77 ICoord2D origin, size, start, end;
78 Color backColor, backBorder, barColor, barBorder;
79 Int progress = (Int)window->winGetUserData();
80
81 // get window size and position
82 window->winGetScreenPosition( &origin.x, &origin.y );
83 window->winGetSize( &size.x, &size.y );
84
85 // get the right colors to use
86 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
87 {
88 backColor = GadgetProgressBarGetDisabledColor( window );
89 backBorder = GadgetProgressBarGetDisabledBorderColor( window );
90 barColor = GadgetProgressBarGetDisabledBarColor( window );
91 barBorder = GadgetProgressBarGetDisabledBarBorderColor( window );
92 } // end if, disabled
93 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
94 {
95 backColor = GadgetProgressBarGetHiliteColor( window );
96 backBorder = GadgetProgressBarGetHiliteBorderColor( window );
97 barColor = GadgetProgressBarGetHiliteBarColor( window );
98 barBorder = GadgetProgressBarGetHiliteBarBorderColor( window );
99 } // end else if, hilited
100 else
101 {
102 backColor = GadgetProgressBarGetEnabledColor( window );
103 backBorder = GadgetProgressBarGetEnabledBorderColor( window );
104 barColor = GadgetProgressBarGetEnabledBarColor( window );
105 barBorder = GadgetProgressBarGetEnabledBarBorderColor( window );
106 } // end else, enabled
107
108 // draw background border
109 if( backBorder != WIN_COLOR_UNDEFINED )
110 {
111
112 start.x = origin.x;
113 start.y = origin.y;
114 end.x = start.x + size.x;
115 end.y = start.y + size.y;
116 TheWindowManager->winOpenRect( backBorder, WIN_DRAW_LINE_WIDTH,
117 start.x, start.y, end.x, end.y );
118
119 } // end if
120
121 // draw background fill
122 if( backColor != WIN_COLOR_UNDEFINED )
123 {
124
125 start.x = origin.x + 1;
126 start.y = origin.y + 1;
127 end.x = start.x + size.x - 2;
128 end.y = start.y + size.y - 2;
129 TheWindowManager->winFillRect( backColor, WIN_DRAW_LINE_WIDTH,
130 start.x, start.y, end.x, end.y );
131
132 } // end if
133
134 // draw the progress so far
135 if( progress )
136 {
137
138 // draw bar border
139 if( barBorder != WIN_COLOR_UNDEFINED )
140 {
141
142 start.x = origin.x;
143 start.y = origin.y;
144 end.x = start.x + (size.x * progress) / 100;
145 end.y = start.y + size.y;
146 if(end.x- start.x > 1 )
147 {
148 TheWindowManager->winOpenRect( barBorder, WIN_DRAW_LINE_WIDTH,
149 start.x, start.y, end.x, end.y );
150 }
151
152 } // end if
153
154 // draw bar fill
155 if( barColor != WIN_COLOR_UNDEFINED )
156 {
157
158 start.x = origin.x + 1;
159 start.y = origin.y + 1;
160 end.x = start.x + (size.x * progress) / 100 - 2;
161 end.y = start.y + size.y - 2;
162// TheWindowManager->winOpenRect( barColor, WIN_DRAW_LINE_WIDTH,
163// start.x, start.y, end.x, end.y );
164
165 if(end.x- start.x > 1 )
166 {
167 TheWindowManager->winFillRect( barColor,WIN_DRAW_LINE_WIDTH,
168 start.x, start.y, end.x, end.y );
169
170 TheWindowManager->winDrawLine(GameMakeColor(255,255,255,255),WIN_DRAW_LINE_WIDTH, start.x, start.y, end.x, start.y);
171 TheWindowManager->winDrawLine(GameMakeColor(200,200,200,255),WIN_DRAW_LINE_WIDTH, start.x, start.y, start.x, end.y);
172 }
173
174 } // end if
175
176 } // end if
177
178
179
180} // end W3DGadgetProgressBarDraw

185void W3DGadgetProgressBarImageDrawA( GameWindow *window, WinInstanceData *instData )
186{
187 ICoord2D origin, size;
188 const Image *barCenter, *barRight, *left, *right, *center;
189 Int progress = (Int)window->winGetUserData();
190 Int xOffset, yOffset;
191 Int i;
192 // get window size and position
193 window->winGetScreenPosition( &origin.x, &origin.y );
194 window->winGetSize( &size.x, &size.y );
195
196 // get offset
197 xOffset = instData->m_imageOffset.x;
198 yOffset = instData->m_imageOffset.y;
199
200 barCenter = GadgetProgressBarGetEnabledBarImageCenter( window );
201 barRight = GadgetProgressBarGetEnabledBarImageRight( window );
202 left = GadgetProgressBarGetEnabledImageLeft( window );
203 right = GadgetProgressBarGetEnabledImageRight( window );
204 center = GadgetProgressBarGetEnabledImageCenter( window );
205
206 if(!barCenter || !barRight || !left || !right || !center)
207 return;
208
209 Int width = barCenter->getImageWidth();
210// Int height = barCenter->getImageHeight();
211
212 Int drawWidth = (size.x * progress) / 100;
213 Int pieces = drawWidth / width;
214 Int x = origin.x;
215 for( i = 0; i < pieces; i ++)
216 {
217
218 TheWindowManager->winDrawImage( barCenter,
219 x, origin.y,
220 x + width, origin.y + size.y );
221 x += width;
222 }
223
224
