76void UndeadBody::attemptDamage( DamageInfo *damageInfo )
77{
78 // If we are on our first life, see if this damage will kill us. If it will, bind it to one hitpoint
79 // remaining, then go ahead and take it.
80 Bool shouldStartSecondLife = FALSE;
81
82 if( damageInfo->in.m_damageType != DAMAGE_UNRESISTABLE
83 && !m_isSecondLife
84 && damageInfo->in.m_amount >= getHealth()
85 && IsHealthDamagingDamage(damageInfo->in.m_damageType)
86 )
87 {
88 damageInfo->in.m_amount = min( damageInfo->in.m_amount, getHealth() - 1 );
89 shouldStartSecondLife = TRUE;
90 }
91
92 ActiveBody::attemptDamage(damageInfo);
93
94 // After we take it (which allows for damaging special effects), we will do our modifications to the body module
95 if( shouldStartSecondLife )
96 startSecondLife(damageInfo);
97}

101void UndeadBody::startSecondLife(DamageInfo *damageInfo)
102{
103 const UndeadBodyModuleData *data = getUndeadBodyModuleData();
104
105 // Flag module as no longer intercepting damage
106 m_isSecondLife = TRUE;
107
108 // Modify ActiveBody's max health and initial health
109 setMaxHealth(data->m_secondLifeMaxHealth, FULLY_HEAL);
110
111 // Set Armor set flag to use second life armor
112 setArmorSetFlag(ARMORSET_SECOND_LIFE);
113
114 // Fire the Slow Death module. The fact that this is not the result of an onDie will cause the special behavior
115 Int total = 0;
116 for( BehaviorModule** update = getObject()->getBehaviorModules(); *update; ++update )
117 {
118 SlowDeathBehaviorInterface* sdu = (*update)->getSlowDeathBehaviorInterface();
119 if (sdu != NULL && sdu->isDieApplicable(damageInfo) )
120 {
121 total += sdu->getProbabilityModifier( damageInfo );
122 }
123 }
124 DEBUG_ASSERTCRASH(total > 0, ("Hmm, this is wrong"));
125
126
127 // this returns a value from 1...total, inclusive
128 Int roll = GameLogicRandomValue(1, total);
129
130 for( update = getObject()->getBehaviorModules(); *update; ++update)
131 {
132 SlowDeathBehaviorInterface* sdu = (*update)->getSlowDeathBehaviorInterface();
133 if (sdu != NULL && sdu->isDieApplicable(damageInfo))
134 {
135 roll -= sdu->getProbabilityModifier( damageInfo );
136 if (roll <= 0)
137 {
138 sdu->beginSlowDeath(damageInfo);
139 return;
140 }
141 }
142 }
143
144}
