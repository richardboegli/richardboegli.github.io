492void W3DCommandBarGenExpDraw( GameWindow *window, WinInstanceData *instData )
493{
494 Player *player = ThePlayerList->getLocalPlayer();
495 if(!player->isPlayerActive())
496 return;
497 static const Image *endBar = TheMappedImageCollection->findImageByName("GenExpBarTop1");
498 static const Image *beginBar = TheMappedImageCollection->findImageByName("GenExpBarBottom1");
499 static const Image *centerBar = TheMappedImageCollection->findImageByName("GenExpBar1");
500 Int progress;
501 progress = ((player->getSkillPoints() - player->getSkillPointsLevelDown()) * 100) /(player->getSkillPointsLevelUp() - player->getSkillPointsLevelDown());
502
503 if(progress <= 0)
504 return;
505
506 // GS This should never be necessary, but scripts can change the points required or even disable a level.
507 // A disabled level will be -1 for points required. Just be totally safe and bind to 100, and we will
508 // fix the scripts to bind the points gained later.
509 if( progress > 100 )
510 progress = 100;
511
512 ICoord2D pos, size;
513 window->winGetScreenPosition( &pos.x, &pos.y );
514 window->winGetSize( &size.x, &size.y );
515
516
517
518 if( !endBar || !beginBar || !centerBar)

520

522 range = size.y * progress / 100;
523
524
525 // get image sizes for the ends
526 ICoord2D topSize, bottomSize, start, end;
527 bottomSize.x = beginBar->getImageWidth();

534 bottomEnd.x = pos.x + size.x;
535 bottomEnd.y = pos.y + size.y - bottomSize.y;

537 topStart.y = pos.y +size.y - range - topSize.y;
538
539 // draw the center repeating bar
540 Int centerWidth, pieces;
541
542 // get width we have to draw our repeating center in
543 centerWidth = bottomEnd.y - topStart.y;
544
545 if( centerWidth <= 0)
546 {
547 // draw left end

549 start.y = pos.y + size.y - bottomSize.y;
550 end.y = pos.y + size.y;
551 end.x = pos.x + size.x;
552 TheWindowManager->winDrawImage(beginBar, start.x, start.y, end.x, end.y);
553
554 // draw right end
555 start.y = pos.y + size.y - bottomSize.y - topSize.y;
556 start.x = pos.x;
557 end.x = pos.x + size.x;
558 end.y = start.y + topSize.y;
559 TheWindowManager->winDrawImage(endBar, start.x, start.y, end.x, end.y);
560 }
561 else
562 {
563
564 // how many whole repeating pieces will fit in that width
565 pieces = centerWidth / centerBar->getImageHeight();
566
567 // draw the pieces
568 start.x = pos.x;
569 start.y = topStart.y;
570 end.x = start.x + size.x; //centerImage->getImageHeight() + yOffset;
571 for( Int i = 0; i < pieces; i++ )
572 {
573
574 end.y = start.y + centerBar->getImageHeight();
575 TheWindowManager->winDrawImage( centerBar,
576 start.x, start.y,
577 end.x, end.y );
578 start.y += centerBar->getImageHeight();
579
580 } // end for i
581
582 // we will draw the image but clip the parts we don't want to show
583 IRegion2D reg;
584 reg.lo.x = start.x;
585 reg.lo.y = start.y;
586 reg.hi.x = bottomEnd.x;
587 reg.hi.y = bottomEnd.y;
588 centerWidth = bottomEnd.y - start.y;
589 if( centerWidth > 0)
590 {
591 TheDisplay->setClipRegion(&reg);
592 end.y = start.y + centerBar->getImageHeight();
593 TheWindowManager->winDrawImage( centerBar,
594 start.x, start.y,
595 end.x, end.y );
596 TheDisplay->enableClipping(FALSE);
597 }
598
599 // draw left end
600 end.x = pos.x + size.x;
601 end.y = pos.y + size.y;
602 start.x = pos.x;
603 start.y = bottomEnd.y;
604 TheWindowManager->winDrawImage(beginBar, start.x, start.y, end.x, end.y);
605
606 // draw right end
607 start.x = pos.x;
608 start.y = pos.y +size.y - range;
609 end.x = pos.x + size.x;
610 end.y = pos.y +size.y - range - topSize.y;
611 TheWindowManager->winDrawImage(endBar, start.x, start.y, end.x, end.y);
612 }
