135void ProfCacheNewFrame ( ProfileData *prof )
136{
137 int total;
138 prof->frames++;
139
140 if ( prof->totalTime > prof->longestFrame )
141 {
142 prof->nextLongestFrame = prof->longestFrame;
143 prof->nextLongestFramePages = prof->longestFramePages;
144 prof->nextLongestFrameBytes = prof->longestFrameBytes;
145
146 prof->longestFrame = prof->totalTime ;
147 prof->longestFrameBytes = prof->totalDataBytes;
148 prof->longestFramePages = prof->pageCount;
149
150 }
151 else if ( prof->totalTime > prof->nextLongestFrame )
152 {
153 prof->nextLongestFrame = prof->totalTime ;
154 prof->nextLongestFrameBytes = prof->totalDataBytes;
155 prof->nextLongestFramePages = prof->pageCount;
156 }
157
158 prof->pageCount = 0;
159
160 if ( (total = prof->hits + prof->misses ))
161 {
162 prof->HitPercent = (prof->hits *100) / total;
163 }
164
165 prof->BytesPerFrame = prof->totalFrameBytes / prof->frames;
166
167 if ( prof->frames > 3*30 )
168 {
169 prof->totalFrameBytes = 0;
170 prof->frames = 0;
171 }
172
173 if ( ! (prof->frames % 30) )
174 {
175 prof->hits = 0;
176 prof->misses = 0;
177 }
178
179 if ( prof->totalTime > prof->update )
180 {
181 int ms;
182
183 ms = (int) ( (prof->totalTime *1000 )/prof->freq);
184 prof->TotalBytesPerSecond = (int) (((unsigned __int64) prof->totalDataBytes *1000)/ ms);
185 prof->totalDataBytes = 0;
186 prof->totalTime = 0;
187
188 ms = (int) ((prof->totalDecompTime *1000 )/prof->freq) ;
189
190 if ( ms )
191 {
192 prof->DecompBytesPerSecond = (int) (((unsigned __int64) prof->totalDecompBytes *1000)/ ms);
193 }
194 else
195 {
196 prof->DecompBytesPerSecond = 0;
197 }
198
199 prof->totalDecompBytes = 0;
200 prof->totalDecompTime = 0;
201
202 ms = (int) ((prof->totalLoadTime *1000 )/prof->freq );
203
204 if ( ms )
205 {
206 prof->LoadBytesPerSecond = (int) (((unsigned __int64)prof->totalLoadBytes *1000)/ ms);
207 }
208 else
209 {
210 prof->LoadBytesPerSecond = 0;
211 }
212 prof->totalLoadBytes = 0;
213 prof->totalLoadTime = 0;
214 }
215}

447void ProfileCPUInit ( ProfileCPU &prof )
448{
449
450 memset ( &prof, 0, sizeof ( ProfileCPU ));
451
452 if ( !QueryPerformanceFrequency( (LARGE_INTEGER*) &prof.freq ))
453 {
454 prof.freq = 0;
455 }
456 prof.updateInterval = SECONDS(1);
457
458 if ( prof.freq )
459 {
460 prof.overflowInterval = SECONDS ( (((ProfStamp) 0xffffffffffffffff) / prof.freq) );
461
462 if ( prof.overflowInterval < prof.updateInterval )
463 {
464 prof.updateInterval = prof.overflowInterval;
465 }
466 }

485{
486

539
540void ProfileCPUEnd ( ProfileCPU &prof )
541{
542 ProfStamp end;
543
544 QueryPerformanceCounter( (LARGE_INTEGER*) &end );
545
546 if ( prof.state != PROF_STATE_IDLE )
547 {
548 if ( prof.start && prof.totalCPUTicks )
549 {
550 prof.totalTicks += calc_ticks ( prof.start, end );
551 }
552 prof.state = PROF_STATE_IDLE;
553 }
554
555 TimeStamp now = AudioGetTime();
556 TimeStamp lastUpdate = now - prof.lastUpdate;
557
558 if ( lastUpdate > prof.updateInterval )
559 {
560 prof.lastUpdate = now;
561
562 if ( lastUpdate < prof.overflowInterval )
563 {
564 // we can use the data
565 prof.lastTicks = prof.totalTicks;
566 prof.lastCPU = prof.totalCPUTicks;
567 calc_stats ( prof );
568 }
569
570 prof.totalTicks = 0;
571 prof.totalCPUTicks = 0;
572
573 }
574}
