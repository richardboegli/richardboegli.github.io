225//
226// On_Event
227//
229__inline void
230SoundSceneObjClass::On_Event
231(
232 AudioCallbackClass::EVENTS event,
233 uint32 param1,
234 uint32 param2
235)
236{
237 if ((m_pCallback != NULL) && (m_RegisteredEvents & event)) {
238
239 switch (event)
240 {
241 case AudioCallbackClass::EVENT_SOUND_STARTED:
242 m_pCallback->On_Sound_Started (this);
243 break;
244
245 case AudioCallbackClass::EVENT_SOUND_ENDED:
246 m_pCallback->On_Sound_Ended (this);
247 break;
248
249 case AudioCallbackClass::EVENT_LOGICAL_HEARD:
250 m_pCallback->On_Logical_Heard ((LogicalListenerClass *)param1, (LogicalSoundClass *)param2);
251 break;
252 }
253 }
254
255 return ;
256}
