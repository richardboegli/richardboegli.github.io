85void CLLTreeCtrl::OnRButtonDown(UINT nFlags, CPoint point)
86{
87 // first, if there's something under the mouse, select it.
88
89 HTREEITEM item = HitTest(point);
90 Bool contextIsLayer = false;
91 SelectItem(item);
92
93 if (item) {
94 if (GetParentItem(item) == NULL) {
95 mLastClickedLayer = GetItemText(item);
96 mLastClickedObject = AsciiString::TheEmptyString;
97 contextIsLayer = true;
98 } else {
99 mLastClickedLayer = AsciiString::TheEmptyString;
100 mLastClickedObject = GetItemText(item);
101 }
102 } else {
103 mLastClickedLayer = AsciiString::TheEmptyString;
104 mLastClickedObject = AsciiString::TheEmptyString;
105 }
106
107 CMenu menu;
108 VERIFY(menu.LoadMenu(IDR_LAYERSLISTPOPUP));
109 CMenu* pPopup = menu.GetSubMenu(0);
110 if (!pPopup) {
111 return;
112 }
113 ClientToScreen(&point);
114
115 CMenu *moveViewSelectionMenu = pPopup->GetSubMenu(LAYERS_LIST_MERGE_VIEW_SELECTION_INTO);
116 moveViewSelectionMenu->RemoveMenu(0, MF_BYPOSITION);
117 buildMoveMenu(moveViewSelectionMenu, ID_LAYERSLIST_MERGEVIEWSELECTIONINTO_BEGIN);
118
119 if (item) {
120 if (contextIsLayer) {
121 CString itemText = GetItemText(item);
122 if ((itemText.CompareNoCase(LayersList::TheActiveLayerName.c_str()) == 0) ||
123 ((LayersList::TheActiveLayerName.c_str() == NULL) &&
124 (LayersList::TheUnmutableDefaultLayerName.compare(LayersList::TheDefaultLayerName) == 0))) {
125 // Don't allow the current layer to be hidden.
126 // Because the objects will immediately disappear when placed.
127 pPopup->EnableMenuItem(ID_HIDECURRENTLAYER, MF_GRAYED);
128 } else {
129 LayersList *ll = (LayersList*) GetParent();
130 Bool hidden = ll->isLayerHidden(AsciiString(itemText));
131 pPopup->CheckMenuItem(ID_HIDECURRENTLAYER, MF_BYCOMMAND | (hidden ? MF_CHECKED : MF_UNCHECKED));
132 }
133
134 CMenu *moveMenu = pPopup->GetSubMenu(LAYERS_LIST_MERGE_LAYER_INTO);
135 moveMenu->RemoveMenu(0, MF_BYPOSITION);
136 buildMoveMenu(moveMenu, ID_LAYERSLIST_MERGELAYERINTO_BEGIN);
137
138 pPopup->RemoveMenu(LAYERS_LIST_MERGE_VIEW_SELECTION_INTO, MF_BYPOSITION);
139 pPopup->RemoveMenu(LAYERS_LIST_MERGE_OBJECT_INTO, MF_BYPOSITION);
140 pPopup->RemoveMenu(LAYERS_LIST_SELECT_OBJECT, MF_BYPOSITION);
141
142 } else {
143 pPopup->EnableMenuItem(ID_DELETECURRENTLAYER, MF_GRAYED);
144 pPopup->EnableMenuItem(ID_HIDECURRENTLAYER, MF_GRAYED);
145
146 CMenu* moveMenu = pPopup->GetSubMenu(LAYERS_LIST_MERGE_OBJECT_INTO);
147 moveMenu->RemoveMenu(0, MF_BYPOSITION);
148 buildMoveMenu(moveMenu, ID_LAYERSLIST_MERGEOBJECTINTO_BEGIN);
149
150 pPopup->RemoveMenu(LAYERS_LIST_MERGE_LAYER_INTO, MF_BYPOSITION);
151 }
152 } else {
153 pPopup->RemoveMenu(LAYERS_LIST_HIDE_CURRENT_LAYER, MF_BYPOSITION);
154 pPopup->RemoveMenu(LAYERS_LIST_MERGE_OBJECT_INTO, MF_BYPOSITION);
155 pPopup->RemoveMenu(LAYERS_LIST_MERGE_LAYER_INTO, MF_BYPOSITION);
156 pPopup->RemoveMenu(LAYERS_LIST_DELETE_CURRENT_LAYER, MF_BYPOSITION);
157 pPopup->RemoveMenu(LAYERS_LIST_SELECT_ACTIVE_LAYER, MF_BYPOSITION);
158 pPopup->RemoveMenu(LAYERS_LIST_SELECT_OBJECT, MF_BYPOSITION);
159 }
160
161 pPopup->TrackPopupMenu(TPM_LEFTALIGN | TPM_LEFTBUTTON, point.x, point.y, GetParent());
162}

203void LayersList::addMapObjectToLayersList(MapObject *objToAdd, AsciiString layerToAddTo)
204{
205 if (!objToAdd || findMapObjectAndList(objToAdd)) {
206 DEBUG_CRASH(("MapObject added was NULL or object already in Layers List. jkmcd"));
207 return;
208 }
209 ListLayerIt layerIt;
210
211 if (!findLayerNamed(layerToAddTo, &layerIt)) {
212 // try to add a layer of that name.
213 addLayerNamed(layerToAddTo);
214 if (!findLayerNamed(layerToAddTo, &layerIt)) {
215 // Try to place it in the active layer.
216 if (!findLayerNamed(AsciiString(TheActiveLayerName.c_str()), &layerIt)) {
217 // if we still can't find that layer name, then try adding it to the default layer.
218 if (!findLayerNamed(AsciiString(TheDefaultLayerName.c_str()), &layerIt)) {
219 return;
220 }
221 }
222 }
223 }
224
225 addMapObjectToLayer(objToAdd, &layerIt);

320void LayersList::removeLayerNamed(IN AsciiString layerToRemove)
321{
322 ListLayerIt layerIt;
323 // Not allowed to remove the Default layer
324 if (layerToRemove.compareNoCase(TheDefaultLayerName.c_str()) == 0) {
325 return;
326 }
327
328 // Not allowed to remove the Default trigger layer
329 if (layerToRemove.compareNoCase(ThePolygonTriggerLayerName.c_str()) == 0) {
330 return;
331 }
332
333
334 // If we can't find the layer, how can we remove it?
335 if (!findLayerNamed(layerToRemove, &layerIt)) {
336 DEBUG_CRASH(("Couldn't find layer named %s", layerToRemove.str()));
337 return;
338 }
339
340 // Transfer all the objects from the old layer to the default layer.
341 for (ListMapObjectPtrIt obj = layerIt->objectsInLayer.begin(); obj != layerIt->objectsInLayer.end(); obj = layerIt->objectsInLayer.begin()) {
342 changeMapObjectLayer((*obj), AsciiString(TheDefaultLayerName.c_str()));
343 }
344
345 // Finally, remove the layer itself.
346 mLayers.erase(layerIt);
347}

364void LayersList::mergeLayerInto(IN ListLayerIt src, IN ListLayerIt dst)
365{
366 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
367 if (!pDoc) {
368 return;
369 }
370
371 ListMapObjectPtrIt mopIt;
372 for (mopIt = src->objectsInLayer.begin(); mopIt != src->objectsInLayer.end(); mopIt = src->objectsInLayer.begin()) {
373 changeMapObjectLayer(*mopIt, dst->layerName);
374 }
375
376 ListPolygonTriggerPtrIt ptpIt;
377 for (ptpIt = src->polygonTriggersInLayer.begin(); ptpIt != src->polygonTriggersInLayer.end(); ptpIt = src->polygonTriggersInLayer.begin()) {
378 changePolygonTriggerLayer(*ptpIt, dst->layerName);
379 }
380
381
382 // Don't remove these layers.
383 if (src->layerName.compareNoCase(TheDefaultLayerName.c_str()) == 0) {
384 return;
385 }
386 if (src->layerName.compareNoCase(ThePolygonTriggerLayerName.c_str()) == 0) {
387 return;
388 }
389
390 if (src->layerName.compareNoCase(TheActiveLayerName.c_str()) == 0) {
391 // The active layer is no longer available.
392 TheActiveLayerName = AsciiString::TheEmptyString.str();
393 }
394
395
396 // remove the layer from the tree view
397 HTREEITEM itemToRemove = findTreeLayerNamed(src->layerName);
398 if (!itemToRemove) {
399 return;
400 }
401
402 mTree->DeleteItem(itemToRemove);
403
404 mLayers.erase(src);
405}

419void LayersList::updateUIFromList(void)
420{
421 if (!m_performUpdates) {
422 return;
423 }
424 CTreeCtrl *pTree = (CTreeCtrl*) GetDlgItem(IDC_LL_TREE);

430 pTree->DeleteAllItems();
431
432// int index = 0;

434 for (ListLayerIt layersIt = mLayers.begin(); layersIt != mLayers.end(); ++layersIt) {
435 // Add a branch for each Layer.

437 HTREEITEM thisBranch = pTree->InsertItem(layersIt->layerName.str(), iconToShow, iconToShow);

439 for (ListMapObjectPtrIt objIt = layersIt->objectsInLayer.begin(); objIt != layersIt->objectsInLayer.end(); ++objIt) {

444 } else {
445 pTree->InsertItem((*objIt)->getName().str(), iconToShow, iconToShow, thisBranch);
446 }
447 }
448
449 for (ListPolygonTriggerPtrIt triggerIt = layersIt->polygonTriggersInLayer.begin(); triggerIt != layersIt->polygonTriggersInLayer.end(); ++triggerIt) {
450 AsciiString uniqueID = (*triggerIt)->getTriggerName();
451 pTree->InsertItem(uniqueID.str(), iconToShow, iconToShow, thisBranch);
452 }
453 }

458 if (!objectToFind) {

460 }

463 for (ListMapObjectPtrIt objPtrIt = layersIt->objectsInLayer.begin(); objPtrIt != layersIt->objectsInLayer.end(); ++objPtrIt) {
464 if ((*objPtrIt) == objectToFind) {
465 if (layerIt) {
466 (*layerIt) = layersIt;
467 }
468
469 if (objectIt) {
470 (*objectIt) = objPtrIt;
471 }
472
473 return true;
474 }
475 }
476 }
477
478 // if we get here, it means we didn't find the object anywhere in any of the layers
479 return false;

529void LayersList::addMapObjectToLayer(IN MapObject *objectToAdd, IN ListLayerIt *layerIt)
530{
531 if (!objectToAdd) {
532 DEBUG_CRASH(("No object to add in addMapObjectToLayer. This should not happen. jkmcd"));
533 return;
534 }
535
536 ListLayerIt layerToAddTo;
537 if (layerIt) {
538 layerToAddTo = *layerIt;
539 } else {
540 // By default, add to the default layer.
541 if (!findLayerNamed(AsciiString(TheDefaultLayerName.c_str()), &layerToAddTo)) {
542 // If we couldn't find that layer, then we're fuxored.
543 return;
544 }
545 }
546
547 Dict *objProps = objectToAdd->getProperties();
548
549 if (layerToAddTo->layerName.compareNoCase(AsciiString(TheDefaultLayerName.c_str())) == 0) {
550 objProps->setAsciiString(TheKey_objectLayer, AsciiString::TheEmptyString);
551 } else {
552 objProps->setAsciiString(TheKey_objectLayer, layerToAddTo->layerName);
553 }
554
555 // layerToAddTo now has a valid layer to add to.
556 layerToAddTo->objectsInLayer.push_back(objectToAdd);
557
558 // only update this object.
559 HTREEITEM hItem = findTreeLayerNamed(layerToAddTo->layerName);
560 if (hItem) {
561 Bool dontcare;
562 AsciiString objName = objProps->getAsciiString(TheKey_uniqueID, &dontcare);

564 mTree->InsertItem(objName.str(), iconToShow, iconToShow, hItem);
565 }
566
567 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
568 if (!pDoc) {
569 return;
570 }
571
572 if (!layerToAddTo->show) {
573 objectToAdd->setFlag(FLAG_DONT_RENDER);
574 } else {
575 objectToAdd->clearFlag(FLAG_DONT_RENDER);
576 }
577 pDoc->invalObject(objectToAdd);

631void LayersList::removeMapObjectFromLayer(IN MapObject *objectToRemove, IN ListLayerIt *layerIt, IN ListMapObjectPtrIt *objectIt)
632{
633 if (!objectToRemove) {
634 DEBUG_CRASH(("Attempted to remove NULL object from layers list. jkmcd"));
635 return;
636 }
637
638 ListLayerIt layerToRemoveFrom;
639 ListMapObjectPtrIt objectBeingRemove;
640
641 if (layerIt && objectIt) {
642 layerToRemoveFrom = (*layerIt);
643 objectBeingRemove = (*objectIt);
644 } else {
645 if (!findMapObjectAndList(objectToRemove, &layerToRemoveFrom, &objectBeingRemove)) {
646 DEBUG_CRASH(("Couldn't find the object anywhere. Why did we try to remove it? jkmcd"));
647 return;
648 }
649 }
650
651 // only remove this object
652 HTREEITEM layer = findTreeLayerNamed(layerToRemoveFrom->layerName);
653 if (layer) {
654 Bool dontcare;
655 AsciiString objectUID = (*objectBeingRemove)->getProperties()->getAsciiString(TheKey_uniqueID, &dontcare);
656 HTREEITEM itemToDelete = findTreeObjectNamed(objectUID.str(), layer);
657 if (itemToDelete) {
658 mTree->DeleteItem(itemToDelete);
659 }
660 }
661
662 // don't need to update the layer in the object because only 1 of 2 things can be happening:
663 // 1. We are about to delete the object
664 // 2. The object is about to be moved to the default layer (and will be updated anyways.)
665
666 // layerToRemoveFrom and objectToRemove are now valid. Remove em.
667 layerToRemoveFrom->objectsInLayer.erase(objectBeingRemove);
668
669}
