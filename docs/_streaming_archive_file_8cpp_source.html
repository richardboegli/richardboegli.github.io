163//============================================================================
164// StreamingArchiveFile::openFromArchive
165//============================================================================
166Bool StreamingArchiveFile::openFromArchive(File *archiveFile, const AsciiString& filename, Int offset, Int size)
167{
168 //USE_PERF_TIMER(StreamingArchiveFile)
169 if (archiveFile == NULL) {
170 return FALSE;
171 }
172
173 if (File::open(filename.str(), File::READ | File::BINARY | File::STREAMING) == FALSE) {
174 return FALSE;
175 }
176
177 m_file = archiveFile;
178 m_startingPos = offset;
179 m_size = size;
180 m_curPos = 0;
181
182 if (m_file->seek(offset, File::START) != offset) {
183 return FALSE;
184 }
185
186 if (m_file->seek(size) != m_startingPos + size) {
187 return FALSE;
188 }
189
190 // We know this will succeed.
191 m_file->seek(offset, File::START);
192
193 m_nameStr = filename;
194
195 return TRUE;
196}

212//=================================================================
213// StreamingArchiveFile::read
214//=================================================================
215// if buffer is null, just advance the current position by 'bytes'
216Int StreamingArchiveFile::read( void *buffer, Int bytes )
217{
218 if (!m_file) {
219 return 0;
220 }
221
222 // There shouldn't be a way that this can fail, because we've already verified that the file
223 // contains at least this many bits.
224 m_file->seek(m_startingPos + m_curPos, File::START);
225
226 if (bytes + m_curPos > m_size)
227 bytes = m_size - m_curPos;
228
229 Int bytesRead = m_file->read(buffer, bytes);
230
231 m_curPos += bytesRead;
232
233 return bytesRead;
234}

250Int StreamingArchiveFile::seek( Int pos, seekMode mode)
251{
252 Int newPos;
253
254 switch( mode )
255 {
256 case START:
257 newPos = pos;
258 break;
259 case CURRENT:
260 newPos = m_curPos + pos;
261 break;
262 case END:
263 DEBUG_ASSERTCRASH(pos <= 0, ("StreamingArchiveFile::seek - position should be <= 0 for a seek starting from the end."));
264 newPos = m_size + pos;
265 break;
266 default:
267 // bad seek mode
268 return -1;
269 }
270
271 if ( newPos < 0 )
272 {
273 newPos = 0;
274 }
275 else if ( newPos > m_size )
276 {
277 newPos = m_size;
278 }
279
280 m_curPos = newPos;
281
282 return m_curPos;
283
