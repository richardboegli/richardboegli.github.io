34static void doIt(void)
35{
36 // Generate passkey
37 char passKey[128];
38 passKey[0] = '\0';
39 unsigned char installPath[MAX_PATH] = "";
40
41 // Get game information
42 HKEY hKey;
43 bool usesHKeycurrentUser = false;
44 LONG result = RegOpenKeyEx(HKEY_LOCAL_MACHINE, "Software\\Electronic Arts\\EA Games\\Command and Conquer Generals Zero Hour", 0, KEY_READ, &hKey);
45 if (result != ERROR_SUCCESS)
46 {
47 result = RegOpenKeyEx(HKEY_CURRENT_USER, "Software\\Electronic Arts\\EA Games\\Command and Conquer Generals Zero Hour", 0, KEY_READ, &hKey);
48 usesHKeycurrentUser = true;
49 }
50 assert((result == ERROR_SUCCESS) && "Failed to open game registry key");
51
52 if (result != ERROR_SUCCESS)
53 {
54 return;
55 }
56
57 // Retrieve install path
58 DWORD type;
59 DWORD sizeOfBuffer = sizeof(installPath);
60 result = RegQueryValueEx(hKey, "InstallPath", NULL, &type, installPath, &sizeOfBuffer);
61
62 assert((result == ERROR_SUCCESS) && "Failed to obtain game install path!");
63 assert((strlen((const char*)installPath) > 0) && "Game install path invalid!");
64 DebugPrint("Game install path: %s\n", installPath);
65
66 // Retrieve Hard drive S/N
67 char drive[8];
68 _splitpath((const char*)installPath, drive, NULL, NULL, NULL);
69 strcat(drive, "\\");
70
71 DWORD volumeSerialNumber = 0;
72 DWORD maxComponentLength;
73 DWORD fileSystemFlags;
74 BOOL volInfoSuccess = GetVolumeInformation((const char*)drive, NULL, 0,
75 &volumeSerialNumber, &maxComponentLength, &fileSystemFlags, NULL, 0);
76
77 if (volInfoSuccess == FALSE)
78 {
79 PrintWin32Error("***** GetVolumeInformation() Failed!");
80 }
81
82 DebugPrint("Drive Serial Number: %lx\n", volumeSerialNumber);
83
84 // Add hard drive serial number portion
85 char volumeSN[16];
86 sprintf(volumeSN, "%lx-", volumeSerialNumber);
87 strcat(passKey, volumeSN);
88
89 // Retrieve game serial #
90 unsigned char gameSerialNumber[64];
91 gameSerialNumber[0] = '\0';
92 sizeOfBuffer = sizeof(gameSerialNumber);
93 if (usesHKeycurrentUser)
94 {
95 result = RegOpenKeyEx(HKEY_CURRENT_USER, "Software\\Electronic Arts\\EA Games\\Command and Conquer Generals Zero Hour\\ergc", 0, KEY_READ, &hKey);
96 }
97 else
98 {
99 result = RegOpenKeyEx(HKEY_LOCAL_MACHINE, "Software\\Electronic Arts\\EA Games\\Command and Conquer Generals Zero Hour\\ergc", 0, KEY_READ, &hKey);
100 }
101 assert((result == ERROR_SUCCESS) && "Failed to open game serial registry key");
102
103 if (result == ERROR_SUCCESS)
104 {
105 result = RegQueryValueEx(hKey, "", NULL, &type, gameSerialNumber, &sizeOfBuffer);
106 assert((result == ERROR_SUCCESS) && "Failed to obtain game serial number!");
107 assert((strlen((const char*)gameSerialNumber) > 0) && "Game serial number invalid!");
108 }
109
110 DebugPrint("Game serial number: %s\n", gameSerialNumber);
111
112 RegCloseKey(hKey);
113
114 // Add game serial number portion
115 strcat(passKey, (char*)gameSerialNumber);
116
117 // Obtain windows product ID
118 result = RegOpenKeyEx(HKEY_LOCAL_MACHINE, "Software\\Microsoft\\Windows\\CurrentVersion", 0, KEY_READ, &hKey);
119 assert((result == ERROR_SUCCESS) && "Failed to open windows registry key!");
120
121 if (result == ERROR_SUCCESS)
122 {
123 // Retrieve Windows Product ID
124 unsigned char winProductID[64];
125 winProductID[0] = '\0';
126
127 DWORD type;
128 DWORD sizeOfBuffer = sizeof(winProductID);
129 result = RegQueryValueEx(hKey, "ProductID", NULL, &type, winProductID, &sizeOfBuffer);
130
131 assert((result == ERROR_SUCCESS) && "Failed to obtain windows product ID!");
132 assert((strlen((const char*)winProductID) > 0) && "Invalid windows product ID");
133
134 DebugPrint("Windows Product ID: %s\n", winProductID);
135
136 RegCloseKey(hKey);
137
138 // Add windows product ID portion
139 strcat(passKey, "-");
140 strcat(passKey, (char*)winProductID);
141 }
142
143 DebugPrint("Retrieved PassKey: %s\n", passKey);
144
145 const char *plainText = "Play the \"Command & Conquer: Generals\" Multiplayer Test.";
146 int textLen = strlen(plainText);
147 char cypherText[128];
148
149 DebugPrint("Retrieved PassKey: %s\n", passKey);
150
151 // Decrypt protected data into the memory mapped file
152 BlowfishEngine blowfish;
153 int len = strlen(passKey);
154 if (len > BlowfishEngine::MAX_KEY_LENGTH)
155 len = BlowfishEngine::MAX_KEY_LENGTH;
156 blowfish.Submit_Key(passKey, len);
157
158 blowfish.Encrypt(plainText, textLen, cypherText);
159 cypherText[textLen] = 0;
160 DebugPrint("Encrypted data: %s\n", cypherText);
161
162 DebugPrint("Install dir = '%s'\n", installPath);
163
164 char *lastBackslash = strrchr((char *)installPath, '\\');
165 if (lastBackslash)
166 *lastBackslash = 0; // strip of \\game.exe from install path
167
168 strcat((char *)installPath, "\\Generals.dat");
169
170 DebugPrint("DAT file = '%s'\n", installPath);
171
172 FILE *fp = fopen((char *)installPath, "wb");
173 if (fp)
174 {
175 fwrite(cypherText, textLen, 1, fp);
176 fclose(fp);
177 }
178
179 CDAPFN_ENDMARK(doIt);
180}
