198void DebugDisplay::printf( Char *format, ...)
199{
200 va_list args;
201 int result;
202 static char text[5*1024];
203
204 va_start( args, format );
205 result = vsprintf( text, format, args );
206 va_end( args );
207
208 if ( result < 0 )
209 {
210 // error while printing string
211 return;
212 }
213
214 DEBUG_ASSERTCRASH( result < sizeof(text), ("text overflow in DebugDisplay::printf() - string too long"));
215
216 // find every line and print it
217 Char *ptr = text;;
218 Char *lineStart = ptr;
219 Int lineLen = 0;
220 Char ch;
221
222 while ( (ch = *ptr++) != 0 )
223 {
224 switch ( ch )
225 {
226 case '\n':
227 {
228 if ( lineLen > 0 )
229 {
230 *(ptr -1) = 0; // replace '/n' with null
231 drawText( m_rightMargin + m_xPos, m_yPos, lineStart );
232 lineLen = 0;
233 }
234 lineStart = ptr;
235 m_yPos++;
236 m_xPos = 0;
237 break;
238 }
239
240 default:
241 {
242 lineLen++;
243 break;
244 }
245 }
246 }
247
248 if ( lineLen > 0 )
249 {
250 drawText( m_rightMargin + m_xPos, m_yPos, lineStart );
251 m_xPos += lineLen;
252 }
253}
