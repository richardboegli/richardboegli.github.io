60void BlendEdgeTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
61{
62 CPoint from, to;
63
64 Coord3D cpt;
65 // Don't constrain this tool.
66 pView->viewToDocCoords(viewPt, &cpt, false);
67
68 if (!pDoc->getCellIndexFromCoord(m_downPt, &from) || !pDoc->getCellIndexFromCoord(cpt, &to)) {
69 return;
70 }
71
72 if (from.x == to.x && from.y == to.y) {
73 return;
74 }
75
76// WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
77 WorldHeightMapEdit *htMapEditCopy = pDoc->GetHeightMap()->duplicate();
78 if (m == TRACK_L)
79 htMapEditCopy->blendTile(to.x, to.y, from.x, from.y, -1, -1); // does all the work.
80 else
81 htMapEditCopy->blendTile(to.x, to.y, from.x, from.y, TerrainMaterial::getFgTexClass(), -1); // does all the work.
82 IRegion2D partialRange = {to.x, to.y, to.x+1, to.y+1};
83 pDoc->updateHeightMap(htMapEditCopy, true, partialRange); //update the render object with new tile and/or height data.
84 WBDocUndoable *pUndo = new WBDocUndoable(pDoc, htMapEditCopy);
85 pDoc->AddAndDoUndoable(pUndo);
86 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
87
88 REF_PTR_RELEASE(htMapEditCopy);
89
90}
