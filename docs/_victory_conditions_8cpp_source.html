68inline static Bool areAllies(const Player *p1, const Player *p2)
69{
70 if (p1 != p2 &&
71 p1->getRelationship(p2->getDefaultTeam()) == ALLIES &&
72 p2->getRelationship(p1->getDefaultTeam()) == ALLIES)
73 return true;
74
75 return false;
76}

146//-------------------------------------------------------------------------------------------------
147void VictoryConditions::update( void )
148{
149 if (!TheRecorder->isMultiplayer() || (m_localSlotNum == -1 && !m_isObserver))
150 return;
151
152 // Check for a single winning alliance
153 if (!m_singleAllianceRemaining)
154 {
155 Bool multipleAlliances = false;
156 Player *alive = NULL;
157 Player *player;
158 for (Int i=0; i<MAX_PLAYER_COUNT; ++i)
159 {
160 player = m_players[i];
161 if (player && !hasSinglePlayerBeenDefeated(player))
162 {
163 if (alive)
164 {
165 // check to verify they are on the same team
166 if (!areAllies(alive, player))
167 {
168 multipleAlliances = true;
169 break;
170 }
171 }
172 else
173 {
174 alive = player; // save this pointer to check against
175 }
176 }
177 }
178
179 if (!multipleAlliances)
180 {
181 m_singleAllianceRemaining = true; // don't check again
182 m_endFrame = TheGameLogic->getFrame();
183 }
184 }
185
186 // check for player eliminations
187 for (Int i=0; i<MAX_PLAYER_COUNT; ++i)
188 {
189 Player *p = m_players[i];
190 if (p && !m_isDefeated[i] && hasSinglePlayerBeenDefeated(p))
191 {
192 m_isDefeated[i] = true;
193 if (TheGameLogic->getFrame() > 1)
194 {
195 ThePartitionManager->revealMapForPlayerPermanently( p->getPlayerIndex() );
196 TheGameClient->updateFakeDrawables();
197
198 TheInGameUI->message("GUI:PlayerHasBeenDefeated", p->getPlayerDisplayName().str() );
199 // People are boneheads. Also play a sound
200 static AudioEventRTS leftGameSound("GUIMessageReceived");
201 TheAudio->addAudioEvent(&leftGameSound);
202 }
203
204 for (Int idx = 0; idx < MAX_SLOTS; ++idx)
205 {
206 AsciiString pName;
207 pName.format("player%d", idx);
208 if (p->getPlayerNameKey() == NAMEKEY(pName))
209 {
210 GameSlot *slot = (TheGameInfo)?TheGameInfo->getSlot(idx):NULL;
211 if (slot && slot->isAI())
212 {
213 DEBUG_LOG(("Marking AI player %s as defeated\n", pName.str()));
214 slot->setLastFrameInGame(TheGameLogic->getFrame());
215 }
216 }
217 }
218
219 // destroy any remaining units (infantry if its a short game, for example)
220 p->killPlayer();
221 PopulateInGameDiplomacyPopup();
222 }
223 }
224
225 // Check if the local player has been eliminated
226 if (!m_localPlayerDefeated && !m_isObserver)
227 {
228 Player *localPlayer = m_players[m_localSlotNum];
229 if (hasSinglePlayerBeenDefeated(localPlayer))
230 {
231 if (!m_singleAllianceRemaining)
232 {
233 //MessageBoxOk(TheGameText->fetch("GUI:Defeat"), TheGameText->fetch("GUI:LocalDefeat"), NULL);
234 }
235 m_localPlayerDefeated = true; // don't check again
236 TheRadar->forceOn(TRUE);
237 SetInGameChatType( INGAME_CHAT_EVERYONE ); // can't chat to allies after death. Only to other observers.
238 }
239 }
240}

242//-------------------------------------------------------------------------------------------------
243Bool VictoryConditions::hasAchievedVictory(Player *player)
244{
245 if (!player)
246 return false;
247
248 if (m_singleAllianceRemaining)
249 {
250 for (Int i=0; i<MAX_PLAYER_COUNT; ++i)
251 {
252 if ( m_players[i] && !hasSinglePlayerBeenDefeated(m_players[i]) &&
253 (player == m_players[i] || areAllies(m_players[i], player)) )
254 return true;
255 }
256 }
257
258 return false;
259}

261//-------------------------------------------------------------------------------------------------
262Bool VictoryConditions::hasBeenDefeated(Player *player)
263{
264 if (!player)
265 return false;
266
267 if (m_singleAllianceRemaining && !hasAchievedVictory(player))
268 return true;
269
270 return false;
271}

273//-------------------------------------------------------------------------------------------------
274Bool VictoryConditions::hasSinglePlayerBeenDefeated(Player *player)
275{
276 if (!player)
277 return false;
278
279 KindOfMaskType mask;
280 mask.set(KINDOF_MP_COUNT_FOR_VICTORY);
281
282 if ( ISSET(NOUNITS) && ISSET(NOBUILDINGS) )
283 {
284 if ( !player->hasAnyObjects() )
285 {
286 return true;
287 }
288 }
289 else if ( ISSET(NOUNITS) )
290 {
291 if ( !player->hasAnyUnits() )
292 {
293 return true;
294 }
295 }
296 else if ( ISSET(NOBUILDINGS) )
297 {
298 if ( !player->hasAnyBuildings(mask) )
299 {
300 return true;
301 }
302 }
303
304 return false;
305}

308void VictoryConditions::cachePlayerPtrs( void )
309{
310 if (!TheRecorder->isMultiplayer())
311 return;
312
313 Int playerCount = 0;
314 const PlayerTemplate *civTemplate = ThePlayerTemplateStore->findPlayerTemplate( NAMEKEY("FactionCivilian") );
315 for (Int i=0; i<MAX_PLAYER_COUNT; ++i)
316 {
317 Player *player = ThePlayerList->getNthPlayer(i);
318 DEBUG_LOG(("Checking whether to cache player %d - [%ls], house [%ls]\n", i, player?player->getPlayerDisplayName().str():L"<NOBODY>", (player&&player->getPlayerTemplate())?player->getPlayerTemplate()->getDisplayName().str():L"<NONE>"));
319 if (player && player != ThePlayerList->getNeutralPlayer() && player->getPlayerTemplate() && player->getPlayerTemplate() != civTemplate && !player->isPlayerObserver())
320 {
321 DEBUG_LOG(("Caching player\n"));
322 m_players[playerCount] = player;
323 if (m_players[playerCount]->isLocalPlayer())
324 m_localSlotNum = playerCount;
325 ++playerCount;
326 }
327 }
328 while (playerCount < MAX_PLAYER_COUNT)
329 {
330 m_players[playerCount++] = NULL;
331 }
332
333 if (m_localSlotNum < 0)
334 {
335 m_localPlayerDefeated = true; // if we have no local player, don't check for defeat
336 DEBUG_ASSERTCRASH(TheRadar, ("No Radar!"));
337 TheRadar->forceOn(TRUE);
338 m_isObserver = true;
339 }
340}
