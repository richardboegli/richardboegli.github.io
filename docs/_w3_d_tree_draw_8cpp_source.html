131void W3DTreeDraw::reactToTransformChange( const Matrix3D *oldMtx,
132 const Coord3D *oldPos,
133 Real oldAngle )
134{
135 Drawable *draw = getDrawable();
136 if (m_treeAdded) {
137 return;
138 }
139 if (draw->getPosition()->x==0.0f && draw->getPosition()->y == 0.0f) {
140 return;
141 }
142 m_treeAdded = true;
143 const W3DTreeDrawModuleData *moduleData = getW3DTreeDrawModuleData();
144 if (!moduleData) {
145 return;
146 }
147 Real scale = draw->getScale();
148 Real scaleRandomness = draw->getTemplate()->getInstanceScaleFuzziness();
149 scaleRandomness = 0.0f; // We use the scale fuzziness inside WB to generate random scales, so they don't change at load time. jba. [4/22/2003]
150 TheTerrainRenderObject->addTree(draw->getID(), *draw->getPosition(),
151 scale, draw->getOrientation(), scaleRandomness, moduleData);
152
153}
