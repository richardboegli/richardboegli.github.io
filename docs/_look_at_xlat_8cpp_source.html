73//-----------------------------------------------------------------------------
74void LookAtTranslator::setScrolling(Int x)
75{
76 if (!TheInGameUI->getInputEnabled())
77 return;
78
79 prevCursor = TheMouse->getMouseCursor();
80 m_isScrolling = true;
81 TheInGameUI->setScrolling( TRUE );
82 TheTacticalView->setMouseLock( TRUE );
83 m_scrollType = x;
84 if(TheStatsCollector)
85 TheStatsCollector->startScrollTime();
86}

88//-----------------------------------------------------------------------------

90{
91 m_isScrolling = false;
92 TheInGameUI->setScrolling( FALSE );
93 TheTacticalView->setMouseLock( FALSE );
94 TheMouse->setCursor(prevCursor);
95 m_scrollType = SCROLL_NONE;
96
97 // if we have a stats collectore increment the stats
98 if(TheStatsCollector)
99 TheStatsCollector->endScrollTime();
100
101}

157//-----------------------------------------------------------------------------
163GameMessageDisposition LookAtTranslator::translateGameMessage(const GameMessage *msg)
164{
165 GameMessageDisposition disp = KEEP_MESSAGE;
166
167 GameMessage::Type t = msg->getType();
168 switch (t)
169 {
170 //-----------------------------------------------------------------------------
171 case GameMessage::MSG_RAW_KEY_DOWN:
172 case GameMessage::MSG_RAW_KEY_UP:
173 {
174 // get key and state from args
175 UnsignedByte key = msg->getArgument( 0 )->integer;
176 UnsignedByte state = msg->getArgument( 1 )->integer;
177 Bool isPressed = !(BitTest( state, KEY_STATE_UP ));
178
179 if (TheShell && TheShell->isShellActive())
180 break;
181
182 switch (key)
183 {
184 case KEY_UP:
185 scrollDir[DIR_UP] = isPressed;
186 break;
187 case KEY_DOWN:
188 scrollDir[DIR_DOWN] = isPressed;
189 break;
190 case KEY_LEFT:
191 scrollDir[DIR_LEFT] = isPressed;
192 break;
193 case KEY_RIGHT:
194 scrollDir[DIR_RIGHT] = isPressed;
195 break;
196 }
197
198 if (TheInGameUI->isSelecting() || (m_isScrolling && m_scrollType != SCROLL_KEY))
199 break;
200
201 // see if we need to start/stop scrolling
202 Int numDirs = 0;
203 for (Int i=0; i<4; ++i)
204 {
205 if (scrollDir[i])
206 numDirs++;
207 }
208
209 if (numDirs && !m_isScrolling)
210 {
211 setScrolling( SCROLL_KEY );
212 }
213 else if (!numDirs && m_isScrolling)
214 {
215 stopScrolling();
216 }
217 break;
218 }
219
220 //-----------------------------------------------------------------------------
221 case GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_DOWN:
222 {
223 m_lastMouseMoveFrame = TheGameLogic->getFrame();
224
225 m_anchor = msg->getArgument( 0 )->pixel;
226 m_currentPos = msg->getArgument( 0 )->pixel;
227
228 if (!TheInGameUI->isSelecting() && !m_isScrolling)
229 {
230 setScrolling(SCROLL_RMB);
231 }
232 break;
233 }
234
235 //-----------------------------------------------------------------------------
236 case GameMessage::MSG_RAW_MOUSE_RIGHT_BUTTON_UP:
237 {
238 m_lastMouseMoveFrame = TheGameLogic->getFrame();
239
240 if (m_scrollType == SCROLL_RMB)
241 {
242 stopScrolling();
243 }
244 break;
245 }
246
247 //-----------------------------------------------------------------------------
248 case GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_DOWN:
249 {
250 m_lastMouseMoveFrame = TheGameLogic->getFrame();
251
252 m_isRotating = true;
253 m_anchor = msg->getArgument( 0 )->pixel;
254 m_originalAnchor = msg->getArgument( 0 )->pixel;
255 m_currentPos = msg->getArgument( 0 )->pixel;
256 m_timestamp = TheGameClient->getFrame();
257 break;
258 }
259
260 //-----------------------------------------------------------------------------
261 case GameMessage::MSG_RAW_MOUSE_MIDDLE_BUTTON_UP:
262 {
263 m_lastMouseMoveFrame = TheGameLogic->getFrame();
264
265 const UnsignedInt CLICK_DURATION = 5;
266 const UnsignedInt PIXEL_OFFSET = 5;
267
268 m_isRotating = false;
269 Int dx = m_currentPos.x-m_originalAnchor.x;
270 if (dx<0) dx = -dx;
271 Int dy = m_currentPos.y-m_originalAnchor.y;
272 Bool didMove = dx>PIXEL_OFFSET || dy>PIXEL_OFFSET;
273 // if middle button is "clicked", reset to "home" orientation
274 if (!didMove && TheGameClient->getFrame() - m_timestamp < CLICK_DURATION)
275 {
276 TheTacticalView->setAngleAndPitchToDefault();
277 TheTacticalView->setZoomToDefault();
278 }
279
280 break;
281 }
282
283 //-----------------------------------------------------------------------------
284 case GameMessage::MSG_RAW_MOUSE_POSITION:
285 {
286 if (m_currentPos.x != msg->getArgument( 0 )->pixel.x || m_currentPos.y != msg->getArgument( 0 )->pixel.y)
287 m_lastMouseMoveFrame = TheGameLogic->getFrame();
288
289 m_currentPos = msg->getArgument( 0 )->pixel;
290
291 UnsignedInt height = TheDisplay->getHeight();
292 UnsignedInt width = TheDisplay->getWidth();
293
294 if (TheInGameUI->getInputEnabled() == FALSE) {
295 // We don't care how we're scrolling, just stop.
296 if (m_isScrolling)
297 stopScrolling();
298 break;
299 }
300
301 if (!TheGlobalData->m_windowed)
302 {
303 if (m_isScrolling)
304 {
305 if ( m_scrollType == SCROLL_SCREENEDGE && (m_currentPos.x >= edgeScrollSize && m_currentPos.y >= edgeScrollSize && m_currentPos.y < height-edgeScrollSize && m_currentPos.x < width-edgeScrollSize) )
306 {
307 stopScrolling();
308 }
309 }
310 else
311 {
312 if ( m_currentPos.x < edgeScrollSize || m_currentPos.y < edgeScrollSize || m_currentPos.y >= height-edgeScrollSize || m_currentPos.x >= width-edgeScrollSize )
313 {
314 setScrolling(SCROLL_SCREENEDGE);
315 }
316 }
317 }
318
319 // rotate the view
320 if (m_isRotating)
321 {
322 const Real FACTOR = 0.01f;
323
324 Real angle = FACTOR * (m_currentPos.x - m_anchor.x);
325
326 TheTacticalView->setAngle( TheTacticalView->getAngle() + angle );
327 m_anchor = msg->getArgument( 0 )->pixel;
328 }
329
330 // rotate the view up/down
331 if (m_isPitching)
332 {
333 const Real FACTOR = 0.01f;
334
335 Real angle = FACTOR * (m_currentPos.y - m_anchor.y);
336
337 TheTacticalView->setPitch( TheTacticalView->getPitch() + angle );
338 m_anchor = msg->getArgument( 0 )->pixel;
339 }
340
341#if defined(_DEBUG) || defined(_INTERNAL)
342 // adjust the field of view
343 if (m_isChangingFOV)
344 {
345 const Real FACTOR = 0.01f;
346
347 Real angle = FACTOR * (m_currentPos.y - m_anchor.y);
348
349 TheTacticalView->setFieldOfView( TheTacticalView->getFieldOfView() + angle );
350 m_anchor = msg->getArgument( 0 )->pixel;
351 }
352#endif
353 break;
354 }
355
356 //-----------------------------------------------------------------------------
357 case GameMessage::MSG_RAW_MOUSE_WHEEL:
358 {
359 m_lastMouseMoveFrame = TheGameLogic->getFrame();
360
361 Int spin = msg->getArgument( 1 )->integer;
362
363 if (spin > 0)
364 {
365 for ( ; spin > 0; spin--)
366 TheTacticalView->zoomIn();
367 }
368 else
369 {
370 for ( ;spin < 0; spin++ )
371 TheTacticalView->zoomOut();
372 }
373 }
374
375 //-----------------------------------------------------------------------------
376 case GameMessage::MSG_META_OPTIONS:
377 {
378 // stop the scrolling
379 stopScrolling();
380 // let the message drop through, cause we need to process this message for
381 // selection as well.
382 break;
383 }
384
385 //-----------------------------------------------------------------------------
386 case GameMessage::MSG_FRAME_TICK:
387 {
388 Coord2D offset = {0, 0};
389
390 // If we've been forced to stop scrolling (script action?) then stop
391 if (m_isScrolling && !TheInGameUI->isScrolling())
392 {
393 TheInGameUI->setScrollAmount(offset);
394 stopScrolling();
395 }
396 else
397 // scroll the view
398 if (m_isScrolling)
399 {
400 switch (m_scrollType)
401 {
402 case SCROLL_RMB:
403 {
404 if (TheInGameUI->shouldMoveRMBScrollAnchor())
405 {
406 Int maxX = TheDisplay->getWidth()/2;
407 Int maxY = TheDisplay->getHeight()/2;
408
409 if (m_currentPos.x + maxX < m_anchor.x)
410 m_anchor.x = m_currentPos.x + maxX;
411 else if (m_currentPos.x - maxX > m_anchor.x)
412 m_anchor.x = m_currentPos.x - maxX;
413
414 if (m_currentPos.y + maxY < m_anchor.y)
415 m_anchor.y = m_currentPos.y + maxY;
416 else if (m_currentPos.y - maxY > m_anchor.y)
417 m_anchor.y = m_currentPos.y - maxY;
418 }
419
420 offset.x = TheGlobalData->m_horizontalScrollSpeedFactor * (m_currentPos.x - m_anchor.x);
421 offset.y = TheGlobalData->m_verticalScrollSpeedFactor * (m_currentPos.y - m_anchor.y);
422 Coord2D vec;
423 vec.x = offset.x;
424 vec.y = offset.y;
425 vec.normalize();
426 // Add in the window scroll amount as the minimum.
427 offset.x += TheGlobalData->m_horizontalScrollSpeedFactor * vec.x * sqr(TheGlobalData->m_keyboardScrollFactor);
428 offset.y += TheGlobalData->m_verticalScrollSpeedFactor * vec.y * sqr(TheGlobalData->m_keyboardScrollFactor);
429 }
430 break;
431 case SCROLL_KEY:
432 {
433 if (scrollDir[DIR_UP])
434 {
435 offset.y -= TheGlobalData->m_verticalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
436 }
437 if (scrollDir[DIR_DOWN])
438 {
439 offset.y += TheGlobalData->m_verticalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
440 }
441 if (scrollDir[DIR_LEFT])
442 {
443 offset.x -= TheGlobalData->m_horizontalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
444 }
445 if (scrollDir[DIR_RIGHT])
446 {
447 offset.x += TheGlobalData->m_horizontalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
448 }
449 }
450 break;
451 case SCROLL_SCREENEDGE:
452 {
453 UnsignedInt height = TheDisplay->getHeight();
454 UnsignedInt width = TheDisplay->getWidth();
455 if (m_currentPos.y < edgeScrollSize)
456 {
457 offset.y -= TheGlobalData->m_verticalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
458 }
459 if (m_currentPos.y >= height-edgeScrollSize)
460 {
461 offset.y += TheGlobalData->m_verticalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
462 }
463 if (m_currentPos.x < edgeScrollSize)
464 {
465 offset.x -= TheGlobalData->m_horizontalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
466 }
467 if (m_currentPos.x >= width-edgeScrollSize)
468 {
469 offset.x += TheGlobalData->m_horizontalScrollSpeedFactor * SCROLL_AMT * TheGlobalData->m_keyboardScrollFactor;
470 }
471 }
472 break;
473 }
474
475 TheInGameUI->setScrollAmount(offset);
476 TheTacticalView->scrollBy( &offset );
477 }
478 else //not scrolling so reset amount
479 TheInGameUI->setScrollAmount(offset);
480
481 //if (TheGlobalData->m_saveCameraInReplay /*&& TheRecorder->getMode() != RECORDERMODETYPE_PLAYBACK *//**/&& (TheGameLogic->isInSinglePlayerGame() || TheGameLogic->isInSkirmishGame())/**/)
482 //if (TheGlobalData->m_saveCameraInReplay && (TheGameLogic->isInMultiplayerGame() || TheGameLogic->isInSinglePlayerGame() || TheGameLogic->isInSkirmishGame()))
483 if (TheGlobalData->m_saveCameraInReplay && (TheGameLogic->isInSinglePlayerGame() || TheGameLogic->isInSkirmishGame()))
484 {
485 ViewLocation currentView;
486 TheTacticalView->getLocation(&currentView);
487 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_SET_REPLAY_CAMERA );
488 msg->appendLocationArgument( currentView.m_pos );
489 msg->appendRealArgument( currentView.m_angle );
490 msg->appendRealArgument( currentView.m_pitch );
491 msg->appendRealArgument( currentView.m_zoom );
492 msg->appendIntegerArgument( (Int)TheMouse->getMouseCursor() );
493 msg->appendPixelArgument( m_currentPos );
494 }
495 break;
496 }
497
498 // ------------------------------------------------------------------------
499#if defined(_DEBUG) || defined(_INTERNAL)
500 case GameMessage::MSG_META_DEMO_BEGIN_ADJUST_PITCH:
501 {
502 DEBUG_ASSERTCRASH(!m_isPitching, ("hmm, mismatched m_isPitching"));
503 m_isPitching = true;
504 disp = DESTROY_MESSAGE;
505 break;
506 }
507#endif // #if defined(_DEBUG) || defined(_INTERNAL)
508
509 // ------------------------------------------------------------------------
510#if defined(_DEBUG) || defined(_INTERNAL)
511 case GameMessage::MSG_META_DEMO_END_ADJUST_PITCH:
512 {
513 DEBUG_ASSERTCRASH(m_isPitching, ("hmm, mismatched m_isPitching"));
514 m_isPitching = false;
515 disp = DESTROY_MESSAGE;
516 break;
517 }
518#endif // #if defined(_DEBUG) || defined(_INTERNAL)
519
520 // ------------------------------------------------------------------------
521#if defined(_DEBUG) || defined(_INTERNAL)
522 case GameMessage::MSG_META_DEMO_DESHROUD:
523 {
524 ThePartitionManager->revealMapForPlayerPermanently( ThePlayerList->getLocalPlayer()->getPlayerIndex() );
525 break;
526 }
527#endif // #if defined(_DEBUG) || defined(_INTERNAL)
528
529 // ------------------------------------------------------------------------
530#if defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
531 case GameMessage::MSG_CHEAT_DESHROUD:
532 {
533 if (!TheGameLogic->isInMultiplayerGame())
534 {
535 ThePartitionManager->revealMapForPlayerPermanently( ThePlayerList->getLocalPlayer()->getPlayerIndex() );
536 }
537 break;
538 }
539#endif // #if defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
540
541 // ------------------------------------------------------------------------
542#if defined(_DEBUG) || defined(_INTERNAL)
543 case GameMessage::MSG_META_DEMO_ENSHROUD:
544 {
545 // Need to first undo the permanent Look laid down by DEMO_DESHROUD, then blast a shroud dollop.
546 ThePartitionManager->undoRevealMapForPlayerPermanently( ThePlayerList->getLocalPlayer()->getPlayerIndex() );
547 ThePartitionManager->shroudMapForPlayer( ThePlayerList->getLocalPlayer()->getPlayerIndex() );
548 break;
549 }
550#endif // #if defined(_DEBUG) || defined(_INTERNAL)
551
552 // ------------------------------------------------------------------------
553#if defined(_DEBUG) || defined(_INTERNAL)
554 case GameMessage::MSG_META_DEMO_BEGIN_ADJUST_FOV:
555 {
556 //DEBUG_ASSERTCRASH(!m_isChangingFOV, ("hmm, mismatched m_isChangingFOV"));
557 m_isChangingFOV = true;
558 m_anchor = m_currentPos;
559 break;
560 }
561#endif // #if defined(_DEBUG) || defined(_INTERNAL)
562
563 // ------------------------------------------------------------------------
564#if defined(_DEBUG) || defined(_INTERNAL)
565 case GameMessage::MSG_META_DEMO_END_ADJUST_FOV:
566 {
567 // DEBUG_ASSERTCRASH(m_isChangingFOV, ("hmm, mismatched m_isChangingFOV"));
568 m_isChangingFOV = false;
569 break;
570 }
571#endif // #if defined(_DEBUG) || defined(_INTERNAL)
572
573 //-----------------------------------------------------------------------------------------
574 case GameMessage::MSG_META_SAVE_VIEW1:
575 case GameMessage::MSG_META_SAVE_VIEW2:
576 case GameMessage::MSG_META_SAVE_VIEW3:
577 case GameMessage::MSG_META_SAVE_VIEW4:
578 case GameMessage::MSG_META_SAVE_VIEW5:
579 case GameMessage::MSG_META_SAVE_VIEW6:
580 case GameMessage::MSG_META_SAVE_VIEW7:
581 case GameMessage::MSG_META_SAVE_VIEW8:
582 {
583 Int slot = t - GameMessage::MSG_META_SAVE_VIEW1 + 1;
584 if ( slot > 0 && slot <= MAX_VIEW_LOCS )
585 {
586 TheTacticalView->getLocation( &m_viewLocation[slot-1] );
587 UnicodeString msg;
588 msg.format( TheGameText->fetch( "GUI:BookmarkXSet" ), slot );
589 TheInGameUI->message( msg );
590 }
591 disp = DESTROY_MESSAGE;
592 break;
593 }
594
595 //-----------------------------------------------------------------------------------------
596 case GameMessage::MSG_META_VIEW_VIEW1:
597 case GameMessage::MSG_META_VIEW_VIEW2:
598 case GameMessage::MSG_META_VIEW_VIEW3:
599 case GameMessage::MSG_META_VIEW_VIEW4:
600 case GameMessage::MSG_META_VIEW_VIEW5:
601 case GameMessage::MSG_META_VIEW_VIEW6:
602 case GameMessage::MSG_META_VIEW_VIEW7:
603 case GameMessage::MSG_META_VIEW_VIEW8:
604 {
605 Int slot = t - GameMessage::MSG_META_VIEW_VIEW1 + 1;
606 if ( slot > 0 && slot <= MAX_VIEW_LOCS )
607 {
608 TheTacticalView->setLocation( &m_viewLocation[slot-1] );
609 }
610 disp = DESTROY_MESSAGE;
611 break;
612 }
613
614 //-----------------------------------------------------------------------------
615#if defined(_DEBUG) || defined(_INTERNAL)
616 case GameMessage::MSG_META_DEMO_LOCK_CAMERA_TO_PLANES:
617 {
618 Drawable *first = NULL;
619
620 if (m_lastPlaneID)
621 first = TheGameClient->findDrawableByID( m_lastPlaneID );
622
623 if (first == NULL)
624 first = TheGameClient->firstDrawable();
625
626 if (first)
627 {
628 Drawable *d = first;
629 Bool done = false;
630
631 while(!done)
632 {
633 // get next Drawable, wrapping around to head of list if necessary
634 d = d->getNextDrawable();
635 if (d == NULL)
636 d = TheGameClient->firstDrawable();
637
638 // if we've found an airborne object, lock onto it
639// "isAboveTerrain" only indicates that we are currently in the air, but that
640// could be the case if we are a buggy jumping a hill, or a unit being paradropped.
641// the right thing would be to look at the locomotors.
642// so this isn't really right, but will suffice for demo purposes.
643 if (d->getObject() && d->getObject()->isAboveTerrain() )
644 {
645 Bool doLock = true;
646
647 // but don't lock onto projectiles
648 ProjectileUpdateInterface* pui = NULL;
649 for (BehaviorModule** u = d->getObject()->getBehaviorModules(); *u; ++u)
650 {
651 if ((pui = (*u)->getProjectileUpdateInterface()) != NULL)
652 {
653 doLock = false;
654 break;
655 }
656 }
657
658 if (doLock)
659 {
660 TheTacticalView->setCameraLock( d->getObject()->getID() );
661 m_lastPlaneID = d->getID();
662 done = true;
663 break;
664 }
665 } // if airborne found
666
667 // if we're back to the first, quit
668 if (d == first)
669 break;
670 } // while
671 } // end plane lock
672
673 disp = DESTROY_MESSAGE;
674 break;
675 }
676#endif // #if defined(_DEBUG) || defined(_INTERNAL)
677
678 } // end switch
679
680 return disp;
