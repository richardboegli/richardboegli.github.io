103void FenceOptions::updateObjectOptions()
104{
105 if (m_currentObjectIndex >= 0) {
106 MapObject *pObj = m_objectsList;
107 int count = 0;
108 while (pObj) {
109 if (count == m_currentObjectIndex) {
110 ObjectOptions::selectObject(pObj);
111 const ThingTemplate *t = pObj->getThingTemplate();
112 if (t && !m_customSpacing) {
113 if ( !m_customSpacing) {
114 m_fenceSpacing = t->getFenceWidth();
115 }
116 m_fenceOffset = t->getFenceXOffset();
117 }
118 break;
119 }
120 count++;
121 pObj = pObj->getNext();
122 }
123 }
124 CWnd *pWnd = GetDlgItem(IDC_FENCE_SPACING_EDIT);
125 if (pWnd) {
126 CString s;
127 s.Format("%f",m_fenceSpacing);
128 pWnd->SetWindowText(s);
129 }
130}

205HTREEITEM FenceOptions::findOrAdd(HTREEITEM parent, const char *pLabel)
206{
207 TVINSERTSTRUCT ins;
208 char buffer[_MAX_PATH];
209 ::memset(&ins, 0, sizeof(ins));
210 HTREEITEM child = m_objectTreeView.GetChildItem(parent);
211 while (child != NULL) {
212 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
213 ins.item.hItem = child;
214 ins.item.pszText = buffer;
215 ins.item.cchTextMax = sizeof(buffer)-2;
216 m_objectTreeView.GetItem(&ins.item);
217 if (strcmp(buffer, pLabel) == 0) {
218 return(child);
219 }
220 child = m_objectTreeView.GetNextSiblingItem(child);
221 }
222
223 // not found, so add it.
224 ::memset(&ins, 0, sizeof(ins));
225 ins.hParent = parent;
226 ins.hInsertAfter = TVI_SORT;
227 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
228 ins.item.lParam = -1;
229 ins.item.pszText = (char*)pLabel;
230 ins.item.cchTextMax = strlen(pLabel);
231 child = m_objectTreeView.InsertItem(&ins);
232 return(child);
233}

238void FenceOptions::addObject( MapObject *mapObject, const char *pPath, const char *name,
239 Int terrainNdx, HTREEITEM parent )
240{
241 char buffer[ _MAX_PATH ];
242 const char *leafName = NULL;
243
244 // sanity
245 if( mapObject == NULL )
246 return;
247
248 //
249 // if we have an thing template in mapObject, we've read it from the new INI database,
250 // we will sort those items into the tree based on properties of the template that
251 // make it easier for us to browse when building levels
252 //
253 // Feel free to reorganize how this tree is constructed from the template
254 // data at will, whatever makes it easier for design
255 //
256 const ThingTemplate *thingTemplate = mapObject->getThingTemplate();
257 if( thingTemplate )
258 {
259
260 // first check for test sorted objects
261 if( thingTemplate->getEditorSorting() == ES_TEST )
262 parent = findOrAdd( parent, "TEST" );
263
264 // first sort by side, either create or find the tree item with matching side name
265 AsciiString side = thingTemplate->getDefaultOwningSide();
266 DEBUG_ASSERTCRASH( !side.isEmpty(), ("NULL default side in template\n") );
267 strcpy( buffer, side.str() );
268 parent = findOrAdd( parent, buffer );
269
270 // next tier uses the editor sorting that design can specify in the INI
271 for( EditorSortingType i = ES_FIRST;
272 i < ES_NUM_SORTING_TYPES;
273 i = (EditorSortingType)(i + 1) )
274 {
275
276 if( thingTemplate->getEditorSorting() == i )
277 {
278
279 parent = findOrAdd( parent, EditorSortingNames[ i ] );
280 break; // exit for
281
282 } // end if
283
284 } // end for i
285
286 if( i == ES_NUM_SORTING_TYPES )
287 parent = findOrAdd( parent, "UNSORTED" );
288
289 // the leaf name is the name of the template
290 leafName = thingTemplate->getName().str();
291
292 } // end if
293
294 // add to the tree view
295 if( leafName )
296 {
297 TVINSERTSTRUCT ins;
298
299 ::memset(&ins, 0, sizeof(ins));
300 ins.hParent = parent;
301 ins.hInsertAfter = TVI_SORT;
302 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
303 ins.item.lParam = terrainNdx;
304 ins.item.pszText = (char*)leafName;
305 ins.item.cchTextMax = strlen(leafName)+2;
306 m_objectTreeView.InsertItem(&ins);
307
308 }

312Bool FenceOptions::hasSelectedObject(void)
313{
314 // If we have no selected object, return false.
315 if (m_currentObjectIndex==-1) return false;
316 // If the objects panel has no selected object, return false.
317 if (ObjectOptions::getCurGdfName() == AsciiString::TheEmptyString) return false;
318 // If our selection is valid, return true.
319 if (m_staticThis && m_currentObjectIndex >= 0) {
320 MapObject *pObj = m_staticThis->m_objectsList;
321 int count = 0;
322 while (pObj) {
323 if (count == m_currentObjectIndex) {
324 return(true);
325 }
326 count++;
327 pObj = pObj->getNext();
328 }
329 }
330 return false;
331}

334Bool FenceOptions::setObjectTreeViewSelection(HTREEITEM parent, Int selection)
335{
336 TVITEM item;
337 char buffer[NAME_MAX_LEN];
338 ::memset(&item, 0, sizeof(item));
339 HTREEITEM child = m_objectTreeView.GetChildItem(parent);
340 while (child != NULL) {
341 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT;
342 item.hItem = child;
343 item.pszText = buffer;
344 item.cchTextMax = sizeof(buffer)-2;
345 m_objectTreeView.GetItem(&item);
346 if (item.lParam == selection) {
347 m_objectTreeView.SelectItem(child);
348 return(true);
349 }
350 if (setObjectTreeViewSelection(child, selection))
351 {
352 updateObjectOptions();
353 return(true);
354 }
355 child = m_objectTreeView.GetNextSiblingItem(child);
356 }
357 return(false);
358}

361BOOL FenceOptions::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
362{
363 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
364 if (pHdr->hdr.hwndFrom == m_objectTreeView.m_hWnd) {
365
366 if (pHdr->hdr.code == TVN_ITEMEXPANDED) {
367 if (pHdr->action == TVE_COLLAPSE) {
368 TVITEM item;
369 ::memset(&item, 0, sizeof(item));
370 item.mask = TVIF_STATE;
371 item.hItem = pHdr->itemOld.hItem;
372 m_objectTreeView.GetItem(&item);
373 item.state &= (~TVIS_EXPANDEDONCE);
374 item.mask = TVIF_STATE;
375 m_objectTreeView.SetItem(&item);
376 }
377 }
378 if (pHdr->hdr.code == TVN_SELCHANGED) {
379 char buffer[NAME_MAX_LEN];
380 HTREEITEM hItem = m_objectTreeView.GetSelectedItem();
381 TVITEM item;
382 ::memset(&item, 0, sizeof(item));
383 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT|TVIF_STATE;
384 item.hItem = hItem;
385 item.pszText = buffer;
386 item.cchTextMax = sizeof(buffer)-2;
387 m_objectTreeView.GetItem(&item);
388 if (item.lParam >= 0) {
389 m_currentObjectIndex = item.lParam;
390 m_customSpacing = false;
391 } else if (m_objectTreeView.ItemHasChildren(item.hItem)) {
392 m_currentObjectIndex = -1;
393 }
394 updateObjectOptions();
395 }
396 }
397
398 return CDialog::OnNotify(wParam, lParam, pResult);
399}
