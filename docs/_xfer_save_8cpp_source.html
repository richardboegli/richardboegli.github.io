66XferSave::~XferSave( void )
67{
68
69 // warn the user if a file was left open

71 {
72
73 DEBUG_CRASH(( "Warning: Xfer file '%s' was left open\n", m_identifier.str() ));
74 close();
75
76 } // end if
77
78 //
79 // the block stack should be empty, if it's not that means we started blocks but never
80 // called enough matching end blocks
81 //
82 if( m_blockStack != NULL )
83 {
84
85 // tell the user there is an error
86 DEBUG_CRASH(( "Warning: XferSave::~XferSave - m_blockStack was not NULL!\n" ));
87
88 // delete the block stack
89 XferBlockData *next;
90 while( m_blockStack )
91 {
92
93 next = m_blockStack->next;
94 m_blockStack->deleteInstance();
95 m_blockStack = next;
96
97 } // end while
98
99 } // end if
100
101} // end ~XferSave

165Int XferSave::beginBlock( void )
166{
167
168 // sanity
169 DEBUG_ASSERTCRASH( m_fileFP != NULL, ("Xfer begin block - file pointer for '%s' is NULL\n",
170 m_identifier.str()) );
171
172 // get the current file position so we can back up here for the next end block call
173 XferFilePos filePos = ftell( m_fileFP );
174
175 // write a placeholder
176 XferBlockSize blockSize = 0;
177 if( fwrite( &blockSize, sizeof( XferBlockSize ), 1, m_fileFP ) != 1 )
178 {
179
180 DEBUG_CRASH(( "XferSave::beginBlock - Error writing block size in '%s'\n",
181 m_identifier.str() ));
182 return XFER_WRITE_ERROR;
183
184 } // end if
185
186 // save this block position on the top of the "stack"
187 XferBlockData *top = newInstance(XferBlockData);
188// impossible -- exception will be thrown (srj)
189// if( top == NULL )
190// {
191//
192// DEBUG_CRASH(( "XferSave - Begin block, out of memory - can't save block stack data\n" ));
193// return XFER_OUT_OF_MEMORY;
194//
195// } // end if
196
197 top->filePos = filePos;
198 top->next = m_blockStack;
199 m_blockStack = top;
200
201 return XFER_OK;
202
203} // end beginBlock

210void XferSave::endBlock( void )
211{
212
213 // sanity
214 DEBUG_ASSERTCRASH( m_fileFP != NULL, ("Xfer end block - file pointer for '%s' is NULL\n",
215 m_identifier.str()) );
216
217 // sanity, make sure we have a block started
218 if( m_blockStack == NULL )
219 {
220
221 DEBUG_CRASH(( "Xfer end block called, but no matching begin block was found\n" ));
222 throw XFER_BEGIN_END_MISMATCH;
223
224 } // end if
225
226 // save our current file position
227 XferFilePos currentFilePos = ftell( m_fileFP );
228
229 // pop the block descriptor off the top of the block stack
230 XferBlockData *top = m_blockStack;
231 m_blockStack = m_blockStack->next;
232
233 // rewind the file to the block position
234 fseek( m_fileFP, top->filePos, SEEK_SET );
235
236 // write the size in bytes between the block position and what is our current file position
237 XferBlockSize blockSize = currentFilePos - top->filePos - sizeof( XferBlockSize );
238 if( fwrite( &blockSize, sizeof( XferBlockSize ), 1, m_fileFP ) != 1 )
239 {
240
241 DEBUG_CRASH(( "Error writing block size to file '%s'\n", m_identifier.str() ));
242 throw XFER_WRITE_ERROR;
243
244 } // end if
245
246 // place the file pointer back to the current position
247 fseek( m_fileFP, currentFilePos, SEEK_SET );
248
249 // delete the block data as it's all used up now
250 top->deleteInstance();
251
252} // end endBlock
