64void PrintInfoRecursive( GameWindow *win, FILE *fp)
65{
66 if(!win)
67 return;
68 ICoord2D pos, size;
69 win->winGetSize(&size.x, &size.y);
70 win->winGetPosition(&pos.x, &pos.y);
71 fprintf(fp, "ControlBarResizer %s\n",win->winGetInstanceData()->m_decoratedNameString.str());
72 fprintf(fp, " AltPosition = X:%d Y:%d\n",pos.x, pos.y);
73 fprintf(fp, " AltSize = X:%d Y:%d\n",size.x, size.y);
74 fprintf(fp, "END\n\n");
75
76 PrintInfoRecursive(win->winGetChild(),fp);
77 PrintInfoRecursive(win->winGetNext(),fp);
78
79}
