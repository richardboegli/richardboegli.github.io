75{
76 if (scene_root == NULL) return;
77
78 PointFilterClass pointfilter;
79 INodeListClass pointlist(scene_root,time,&pointfilter);
80
81 if (pointlist.Num_Nodes() > 0) {
82
83 csave.Begin_Chunk(W3D_CHUNK_POINTS);
84
85 for (unsigned int ci=0; ci<pointlist.Num_Nodes(); ci++) {
86
87 W3dVectorStruct vect;

89 vect.X = pos.x;
90 vect.Y = pos.y;
91 vect.Z = pos.z;
92 csave.Write(&vect,sizeof(vect));
93
94 }
95
96 csave.End_Chunk();
97 }
