102UpdateSleepTime FireOCLAfterWeaponCooldownUpdate::update( void )
103{
104 const FireOCLAfterWeaponCooldownUpdateModuleData* data = getFireOCLAfterWeaponCooldownUpdateModuleData();
105 UpgradeMaskType activation, conflicting;
106 getUpgradeActivationMasks( activation, conflicting );
107 Bool validThisFrame = true;
108 Bool validToFireOCL = true;
109 Object *obj = getObject();
110
111 //Get current weapon
112 Weapon *weapon = obj->getCurrentWeapon();
113 if( weapon )
114 {
115 if( weapon->getWeaponSlot() != data->m_weaponSlot )
116 {
117 //Not the right weapon slot -- it's possible we switched weapons in which case it's
118 //possible to fire the OCL.
119 validThisFrame = false;
120 }
121 }
122 else
123 {
124 //No weapon selected -- possible to switch off weapons, possible to have just
125 //finished firing our weapon.
126 validThisFrame = false;
127 }
128
129 UpgradeMaskType objectMask = obj->getObjectCompletedUpgradeMask();
130 UpgradeMaskType playerMask = obj->getControllingPlayer()->getCompletedUpgradeMask();
131 UpgradeMaskType maskToCheck = playerMask;
132 maskToCheck.set( objectMask );
133 if( validThisFrame && !testUpgradeConditions( maskToCheck ) )
134 {
135 //Can't use this period if this object doesn't have any of the upgrades
136 validThisFrame = false;
137 validToFireOCL = false;
138 }
139
140 UnsignedInt now = TheGameLogic->getFrame();
141 if( validThisFrame )
142 {
143 if( weapon->getLastShotFrame() == now - 1 )
144 {
145 m_consecutiveShots++;
146 if( m_consecutiveShots == 1 )
147 {
148 //Our first shot -- so record the frame number so we can easily calculate the
149 //time we've been firing!
150 m_startFrame = now;
151 }
152 }
153 else if( weapon->getPossibleNextShotFrame() < now )
154 {
155 //Means we could have shot but didn't!
156
157 if( data->m_minShotsRequired <= m_consecutiveShots )
158 {
159 //We have fired enough shots to create the OCL.
160 fireOCL();
161 }
162 }
163 }
164 else if( validToFireOCL )
165 {
166 //We've stopped being valid -- which means we may have stopped firing our weapon.
167 Weapon *weapon = obj->getWeaponInWeaponSlot( data->m_weaponSlot );
168 if( weapon && data->m_minShotsRequired <= m_consecutiveShots )
169 {
170 //We switched weapons! Fire OCL!
171 fireOCL();
172 }
173 }
174
175 if( validThisFrame != m_valid )
176 {
177 //There has been a validity change, so reset all values!
178 m_valid = validThisFrame;
179 resetStats();
180 }
181
182 return UPDATE_SLEEP_NONE;
183}

193void FireOCLAfterWeaponCooldownUpdate::fireOCL()
194{
195 const FireOCLAfterWeaponCooldownUpdateModuleData* data = getFireOCLAfterWeaponCooldownUpdateModuleData();
196 Object *obj = getObject();
197
198 //Calculate the lifetime of the OCL.
199 UnsignedInt now = TheGameLogic->getFrame();
200 Real seconds = (now - m_startFrame) * SECONDS_PER_LOGICFRAME_REAL;
201 seconds *= data->m_oclLifetimePerSecond * 0.001f;
202 UnsignedInt oclFrames = (UnsignedInt)(seconds * LOGICFRAMES_PER_SECOND);
203 oclFrames = MIN( oclFrames, data->m_oclMaxFrames );
204
205 ObjectCreationList::create( data->m_ocl, obj, obj, oclFrames );
206
207 resetStats();
208}
