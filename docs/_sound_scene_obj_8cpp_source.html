219//
220// Apply_Auto_Position
221//
223void
224SoundSceneObjClass::Apply_Auto_Position (void)
225{
226 // If the sound is attached to an object, then update its transform
227 // based on this link.
228 if (m_AttachedObject != NULL) {
229
230 // Determine which transform to use
231 Matrix3D transform (1);
232 if (m_AttachedBone >= 0) {
233 transform = m_AttachedObject->Get_Bone_Transform (m_AttachedBone);
234 } else {
235 transform = m_AttachedObject->Get_Transform ();
236
237 //
238 // Convert the camera's transform to an object transform
239 //
240 if (m_AttachedObject->Class_ID () == RenderObjClass::CLASSID_CAMERA) {
241 Matrix3D cam_to_world (Vector3 (0, 0, -1), Vector3 (-1, 0, 0), Vector3 (0, 1, 0), Vector3 (0, 0, 0));
242#ifdef ALLOW_TEMPORARIES
243 transform = transform * cam_to_world;
244#else
245 transform.postMul(cam_to_world);
246#endif
247 }
248 }
249
250 // Update the sound's transform
251 Set_Transform (transform);
252 }
253
254 return ;
255}

282//
283// Load
284//
286bool
287SoundSceneObjClass::Load (ChunkLoadClass &cload)
288{
289 uint32 id = SOUND_OBJ_DEFAULT_ID;
290
291 while (cload.Open_Chunk ()) {
292 switch (cload.Cur_Chunk_ID ()) {
293
294 case CHUNKID_BASE_CLASS:
295 PersistClass::Load (cload);
296 break;
297
298 case CHUNKID_VARIABLES:
299 {
300 //
301 // Read all the variables from their micro-chunks
302 //
303 while (cload.Open_Micro_Chunk ()) {
304 switch (cload.Cur_Micro_Chunk_ID ()) {
305
306 READ_MICRO_CHUNK (cload, VARID_ATTACHED_OBJ, m_AttachedObject);
307 READ_MICRO_CHUNK (cload, VARID_ATTACHED_BONE, m_AttachedBone);
308 READ_MICRO_CHUNK (cload, VARID_USER_DATA, m_UserData);
309 READ_MICRO_CHUNK (cload, VARID_USER_OBJ, m_UserObj);
310 READ_MICRO_CHUNK (cload, VARID_ID, id);
311 }
312
313 cload.Close_Micro_Chunk ();
314 }
315 }
316 break;
317 }
318
319 cload.Close_Chunk ();
320 }
321
322 //
323 // Set the ID (this will cause the sound object to
324 // be re-inserted in the master sorted list)
325 //
326 if (id != SOUND_OBJ_DEFAULT_ID) {
327 Set_ID (id);
328 }
329
330 //
331 // Max sure the next available ID is the largest ID in existence
332 //
333 m_NextAvailableID = max (m_NextAvailableID, m_ID + 1);
334
335 //
336 // We need to 'swizzle' the attached object pointer. We saved the pointer's
337 // value, and need to map it (hopefully) to the new value.
338 //
339 if (m_AttachedObject != NULL) {
340 SaveLoadSystemClass::Request_Ref_Counted_Pointer_Remap ((RefCountClass **)&m_AttachedObject);
341 }
342
343 return true;
344}

388// Register_Sound_Object
389//
391void
392SoundSceneObjClass::Register_Sound_Object (SoundSceneObjClass *sound_obj)
393{
394 int sound_id = sound_obj->Get_ID ();
395 CriticalSectionClass::LockClass lock(m_IDListMutex);
396
397 //
398 // Special case a non-ID
399 //
400 if (sound_id == SOUND_OBJ_DEFAULT_ID) {
401 m_GlobalSoundList.Insert (0, sound_obj);
402 } else {
403
404 //
405 // Check to ensure the object isn't already in the list
406 //
407 int index = 0;
408 if (Find_Sound_Object (sound_id, &index) == false) {
409
410 //
411 // Insert the object into the list
412 //
413 m_GlobalSoundList.Insert (index, sound_obj);
414 }
415 }
416
417 return ;
418}

448//
449// Find_Sound_Object
450//
452bool
453SoundSceneObjClass::Find_Sound_Object (uint32 id_to_find, int *index)
454{
455 CriticalSectionClass::LockClass lock(m_IDListMutex);
456
457 bool found = false;
458 (*index) = 0;
459 int min_index = 0;
460 int max_index = m_GlobalSoundList.Count () - 1;
461
462 //
463 // Keep looping until we've closed the window of possiblity
464 //
465 bool keep_going = (max_index >= min_index);
466 while (keep_going) {
467
468 //
469 // Calculate what slot we are currently looking at
470 //
471 int curr_index = min_index + ((max_index - min_index) / 2);
472 uint32 curr_id = m_GlobalSoundList[curr_index]->Get_ID ();
473
474 //
475 // Did we find the right slot?
476 //
477 if (id_to_find == curr_id) {
478 (*index) = curr_index;
479 keep_going = false;
480 found = true;
481 } else {
482
483 //
484 // Stop if we've narrowed the window to one entry
485 //
486 keep_going = (max_index > min_index);
487
488 //
489 // Move the window to the appropriate side
490 // of the test index.
491 //
492 if (id_to_find < curr_id) {
493 max_index = curr_index - 1;
494 (*index) = curr_index;
495 } else {
496 min_index = curr_index + 1;
497 (*index) = curr_index + 1;
498 }
499 }
500 }
501
502 return found;
503}
