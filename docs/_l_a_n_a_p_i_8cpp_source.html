180void LANAPI::sendMessage(LANMessage *msg, UnsignedInt ip /* = 0 */)
181{
182 if (ip != 0)
183 {
184 m_transport->queueSend(ip, lobbyPort, (unsigned char *)msg, sizeof(LANMessage) /*, 0, 0 */);
185 }
186 else if ((m_currentGame != NULL) && (m_currentGame->getIsDirectConnect()))
187 {
188 Int localSlot = m_currentGame->getLocalSlotNum();
189 for (Int i = 0; i < MAX_SLOTS; ++i)
190 {
191 if (i != localSlot) {
192 GameSlot *slot = m_currentGame->getSlot(i);
193 if ((slot != NULL) && (slot->isHuman())) {
194 m_transport->queueSend(slot->getIP(), lobbyPort, (unsigned char *)msg, sizeof(LANMessage) /*, 0, 0 */);
195 }
196 }
197 }
198 }
199 else
200 {
201 m_transport->queueSend(m_broadcastAddr, lobbyPort, (unsigned char *)msg, sizeof(LANMessage) /*, 0, 0 */);
202 }
203}

317extern Bool LANSocketErrorDetected;
318void LANAPI::update( void )
319{
320 if(LANbuttonPushed)
321 return;
322 static const UnsignedInt LANAPIUpdateDelay = 200;
323 UnsignedInt now = timeGetTime();
324
325 if( now > m_lastUpdate + LANAPIUpdateDelay)
326 {
327 m_lastUpdate = now;
328 }
329 else
330 {
331 return;
332 }
333
334 // Let the UDP socket breathe
335 if ((m_transport->update() == FALSE) && (LANSocketErrorDetected == FALSE)) {
336 if (m_isInLANMenu == TRUE) {
337 LANSocketErrorDetected = TRUE;
338 }
339 }
340
341 // Handle any new messages
342 int i;
343 for (i=0; i<MAX_MESSAGES && !LANbuttonPushed; ++i)
344 {
345 if (m_transport->m_inBuffer[i].length > 0)
346 {
347 // Process the new message
348 UnsignedInt senderIP = m_transport->m_inBuffer[i].addr;
349 if (senderIP == m_localIP)
350 {
351 m_transport->m_inBuffer[i].length = 0;
352 continue;
353 }
354
355 LANMessage *msg = (LANMessage *)(m_transport->m_inBuffer[i].data);
356 //DEBUG_LOG(("LAN message type %s from %ls (%s@%s)\n", GetMessageTypeString(msg->LANMessageType).str(),
357 // msg->name, msg->userName, msg->hostName));
358 switch (msg->LANMessageType)
359 {
360 // Location specification
361 case LANMessage::MSG_REQUEST_LOCATIONS: // Hey, where is everybody?
362 DEBUG_LOG(("LANAPI::update - got a MSG_REQUEST_LOCATIONS from 0x%08x\n", senderIP));
363 handleRequestLocations( msg, senderIP );
364 break;
365 case LANMessage::MSG_GAME_ANNOUNCE: // Here someone is, and here's his game info!
366 DEBUG_LOG(("LANAPI::update - got a MSG_GAME_ANNOUNCE from 0x%08x\n", senderIP));
367 handleGameAnnounce( msg, senderIP );
368 break;
369 case LANMessage::MSG_LOBBY_ANNOUNCE: // Hey, I'm in the lobby!
370 DEBUG_LOG(("LANAPI::update - got a MSG_LOBBY_ANNOUNCE from 0x%08x\n", senderIP));
371 handleLobbyAnnounce( msg, senderIP );
372 break;
373 case LANMessage::MSG_REQUEST_GAME_INFO:
374 DEBUG_LOG(("LANAPI::update - got a MSG_REQUEST_GAME_INFO from 0x%08x\n", senderIP));
375 handleRequestGameInfo( msg, senderIP );
376 break;
377
378 // Joining games
379 case LANMessage::MSG_REQUEST_JOIN: // Let me in! Let me in!
380 DEBUG_LOG(("LANAPI::update - got a MSG_REQUEST_JOIN from 0x%08x\n", senderIP));
381 handleRequestJoin( msg, senderIP );
382 break;
383 case LANMessage::MSG_JOIN_ACCEPT: // Okay, you can join.
384 DEBUG_LOG(("LANAPI::update - got a MSG_JOIN_ACCEPT from 0x%08x\n", senderIP));
385 handleJoinAccept( msg, senderIP );
386 break;
387 case LANMessage::MSG_JOIN_DENY: // Go away! We don't want any!
388 DEBUG_LOG(("LANAPI::update - got a MSG_JOIN_DENY from 0x%08x\n", senderIP));
389 handleJoinDeny( msg, senderIP );
390 break;
391
392 // Leaving games, lobby
393 case LANMessage::MSG_REQUEST_GAME_LEAVE: // I'm outa here!
394 DEBUG_LOG(("LANAPI::update - got a MSG_REQUEST_GAME_LEAVE from 0x%08x\n", senderIP));
395 handleRequestGameLeave( msg, senderIP );
396 break;
397 case LANMessage::MSG_REQUEST_LOBBY_LEAVE: // I'm outa here!
398 DEBUG_LOG(("LANAPI::update - got a MSG_REQUEST_LOBBY_LEAVE from 0x%08x\n", senderIP));
399 handleRequestLobbyLeave( msg, senderIP );
400 break;
401
402 // Game options, chat, etc
403 case LANMessage::MSG_SET_ACCEPT: // I'm cool with everything as is.
404 handleSetAccept( msg, senderIP );
405 break;
406 case LANMessage::MSG_MAP_AVAILABILITY: // Map status
407 handleHasMap( msg, senderIP );
408 break;
409 case LANMessage::MSG_CHAT: // Just spouting my mouth off.
410 handleChat( msg, senderIP );
411 break;
412 case LANMessage::MSG_GAME_START: // Hold on; we're starting!
413 handleGameStart( msg, senderIP );
414 break;
415 case LANMessage::MSG_GAME_START_TIMER:
416 handleGameStartTimer( msg, senderIP );
417 break;
418 case LANMessage::MSG_GAME_OPTIONS: // Here's some info about the game.
419 DEBUG_LOG(("LANAPI::update - got a MSG_GAME_OPTIONS from 0x%08x\n", senderIP));
420 handleGameOptions( msg, senderIP );
421 break;
422 case LANMessage::MSG_INACTIVE: // someone is telling us that we're inactive.
423 handleInActive( msg, senderIP );
424 break;
425
426 default:
427 DEBUG_LOG(("Unknown LAN message type %d\n", msg->LANMessageType));
428 }
429
430 // Mark it as read
431 m_transport->m_inBuffer[i].length = 0;
432 }
433 } // End message handling
434 if(LANbuttonPushed)
435 return;
436 // Send out periodic I'm Here messages
437 if (now > s_resendDelta + m_lastResendTime)
438 {
439 m_lastResendTime = now;
440
441 if (m_inLobby)
442 {
443 RequestSetName(m_name);
444 }
445 else if (m_currentGame && !m_currentGame->isGameInProgress())
446 {
447 if (AmIHost())
448 {
449 RequestGameOptions( GenerateGameOptionsString(), true );
450 RequestGameAnnounce( );
451 }
452 else
453 {
454 AsciiString text;
455 text.format("User=%s", m_userName.str());
456 RequestGameOptions( text, true );
457 text.format("Host=%s", m_hostName.str());
458 RequestGameOptions( text, true );
459 RequestGameOptions( "HELLO", false );
460 }
461 }
462 else if (m_currentGame)
463 {
464 // game is in progress - RequestGameAnnounce will check if we should send it
465 RequestGameAnnounce();
466 }
467 }
468
469 Bool playerListChanged = false;
470 Bool gameListChanged = false;
471
472 // Weed out people we haven't heard from in a while
473 LANPlayer *player = m_lobbyPlayers;
474 while (player)
475 {
476 if (player->getLastHeard() + s_resendDelta*2 < now)
477 {
478 // He's gone!
479 removePlayer(player);
480 LANPlayer *nextPlayer = player->getNext();
481 delete player;
482 player = nextPlayer;
483 playerListChanged = true;
484 }
485 else
486 {
487 player = player->getNext();
488 }
489 }
490
491 // Weed out people we haven't heard from in a while
492 LANGameInfo *game = m_games;
493 while (game)
494 {
495 if (game != m_currentGame && game->getLastHeard() + s_resendDelta*2 < now)
496 {
497 // He's gone!
498 removeGame(game);
499 LANGameInfo *nextGame = game->getNext();
500 delete game;
501 game = nextGame;
502 gameListChanged = true;
503 }
504 else
505 {
506 game = game->getNext();
507 }
508 }
509 if ( m_currentGame && !m_currentGame->isGameInProgress() )
510 {
511 if ( !AmIHost() && (m_currentGame->getLastHeard() + s_resendDelta*16 < now) )
512 {
513 // We haven't heard from the host in a while. Bail.
514 // Actually, fake a host leaving message. :)
515 LANMessage msg;
516 fillInLANMessage( &msg );
517 msg.LANMessageType = LANMessage::MSG_REQUEST_GAME_LEAVE;
518 wcsncpy(msg.name, m_currentGame->getPlayerName(0).str(), g_lanPlayerNameLength);
519 msg.name[g_lanPlayerNameLength] = 0;
520 handleRequestGameLeave(&msg, m_currentGame->getIP(0));
521 UnicodeString text;
522 text = TheGameText->fetch("LAN:HostNotResponding");
523 OnChat(UnicodeString::TheEmptyString, m_localIP, text, LANCHAT_SYSTEM);
524 }
525 else if ( AmIHost() )
526 {
527 // Check each player for timeouts
528 for (int p=1; p<MAX_SLOTS; ++p)
529 {
530 if (m_currentGame->getIP(p) && m_currentGame->getPlayerLastHeard(p) + s_resendDelta*8 < now)
531 {
532 LANMessage msg;
533 fillInLANMessage( &msg );
534 UnicodeString theStr;
535 theStr.format(TheGameText->fetch("LAN:PlayerDropped"), m_currentGame->getPlayerName(p).str());
536 msg.LANMessageType = LANMessage::MSG_REQUEST_GAME_LEAVE;
537 wcsncpy(msg.name, m_currentGame->getPlayerName(p).str(), g_lanPlayerNameLength);
538 msg.name[g_lanPlayerNameLength] = 0;
539 handleRequestGameLeave(&msg, m_currentGame->getIP(p));
540 OnChat(UnicodeString::TheEmptyString, m_localIP, theStr, LANCHAT_SYSTEM);
541 }
542 }
543 }
544 }
545
546 if (playerListChanged)
547 {
548 OnPlayerList(m_lobbyPlayers);
549 }
550
551 if (gameListChanged)
552 {
553 OnGameList(m_games);
554 }
555
556 // Time out old actions
557 if (m_pendingAction != ACT_NONE && now > m_expiration)
558 {
559 switch (m_pendingAction)
560 {
561 case ACT_JOIN:
562 OnGameJoin(RET_TIMEOUT, NULL);
563 m_pendingAction = ACT_NONE;
564 m_currentGame = NULL;
565 m_inLobby = true;
566 break;
567 case ACT_LEAVE:
568 OnPlayerLeave(m_name);
569 m_pendingAction = ACT_NONE;
570 m_currentGame = NULL;
571 m_inLobby = true;
572 break;
573 case ACT_JOINDIRECTCONNECT:
574 OnGameJoin(RET_TIMEOUT, NULL);
575 m_pendingAction = ACT_NONE;
576 m_currentGame = NULL;
577 m_inLobby = true;
578 break;
579 default:
580 m_pendingAction = ACT_NONE;
581 }
582 }
583
584 // send out "game starting" messages
585 if ( m_gameStartTime && m_gameStartSeconds && m_gameStartTime <= now )
586 {
587 // m_gameStartTime is when the next message goes out
588 // m_gameStartSeconds is how many seconds remain in the message
589
590 RequestGameStartTimer( m_gameStartSeconds );
591 }
592 else if (m_gameStartTime && m_gameStartTime <= now)
593 {
594// DEBUG_LOG(("m_gameStartTime=%d, now=%d, m_gameStartSeconds=%d\n", m_gameStartTime, now, m_gameStartSeconds));
595 ResetGameStartTimer();
596 RequestGameStart();
597 }
598
599 // Check for an MOTD every few seconds
600 static UnsignedInt lastMOTDCheck = 0;
601 static const UnsignedInt motdInterval = 30000;
602 if (now > lastMOTDCheck + motdInterval)
603 {
604 checkMOTD();
605 lastMOTDCheck = now;
606 }
607}

618void LANAPI::RequestGameJoin( LANGameInfo *game, UnsignedInt ip /* = 0 */ )
619{
620 if ((m_pendingAction != ACT_NONE) && (m_pendingAction != ACT_JOINDIRECTCONNECT))
621 {
622 OnGameJoin( RET_BUSY, NULL );
623 return;
624 }
625
626 if (!game)
627 {
628 OnGameJoin( RET_GAME_GONE, NULL );
629 return;
630 }
631
632 LANMessage msg;
633 msg.LANMessageType = LANMessage::MSG_REQUEST_JOIN;
634 fillInLANMessage( &msg );
635 msg.GameToJoin.gameIP = game->getSlot(0)->getIP();
636 msg.GameToJoin.exeCRC = TheGlobalData->m_exeCRC;
637 msg.GameToJoin.iniCRC = TheGlobalData->m_iniCRC;
638
639 AsciiString s = "";
640 GetStringFromRegistry("\\ergc", "", s);
641 strncpy(msg.GameToJoin.serial, s.str(), g_maxSerialLength);
642 msg.GameToJoin.serial[g_maxSerialLength-1] = '\0';
643
644 sendMessage(&msg, ip);
645
646 m_pendingAction = ACT_JOIN;
647 m_expiration = timeGetTime() + m_actionTimeout;
648}
