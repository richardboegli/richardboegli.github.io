48BOOL BaseBuildProps::OnInitDialog()
49{
50 // add name
51 CWnd* pName = GetDlgItem(IDC_MAPOBJECT_Name);
52 if (pName) {
53 pName->SetWindowText(m_name.str());
54 }
55
56 // add script
57 CComboBox *pCombo = (CComboBox*)GetDlgItem(IDC_MAPOBJECT_Script);
58 EditParameter::loadScripts(pCombo, true);
59 pCombo->AddString("<none>");
60 if (m_script.isEmpty()) {
61 pCombo->SelectString(-1, "<none>");
62 } else {
63 pCombo->SelectString(-1, m_script.str());
64 }
65
66 // add health
67 CWnd* pHealth = GetDlgItem(IDC_MAPOBJECT_StartingHealthEdit);
68 static char buff[12];
69 sprintf(buff, "%d", m_health);
70 pHealth->SetWindowText(buff);
71
72 CButton* pItem;
73 pItem = (CButton*) GetDlgItem(IDC_MAPOBJECT_Unsellable);
74 if (pItem) {
75 pItem->SetCheck(m_unsellable);
76 }
77
78 return TRUE;
79}

97void BaseBuildProps::OnOK()
98{
99 CComboBox *combo;
100 CWnd* edit;
101 CString cstr;
102 static char buf[1024];
103
104 edit = (CComboBox*)GetDlgItem(IDC_MAPOBJECT_Name);
105 edit->GetWindowText(buf, sizeof(buf)-2);
106 m_name = AsciiString(buf);
107
108 combo = (CComboBox*)GetDlgItem(IDC_MAPOBJECT_Script);
109 combo->GetWindowText(buf, sizeof(buf)-2);
110 m_script = AsciiString(buf);
111
112 edit = GetDlgItem(IDC_MAPOBJECT_StartingHealthEdit);
113 edit->GetWindowText(cstr);
114 if (cstr.IsEmpty()) {
115 m_health = 100;
116 } else {
117 m_health = atoi(cstr.GetBuffer(0));
118 if (m_health < 0)
119 m_health = 0;
120 }
121
122 CButton *check;
123 check = (CButton*) GetDlgItem(IDC_MAPOBJECT_Unsellable);
124 m_unsellable = (check->GetCheck() != 0);
125
126 CDialog::OnOK();
127}
