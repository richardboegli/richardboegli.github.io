341void BaseHeightMapRenderObjClass::adjustTerrainLOD(Int adj)
342{
343 if (adj>0 && TheGlobalData->m_terrainLOD<TERRAIN_LOD_MAX) TheWritableGlobalData->m_terrainLOD=(TerrainLOD)(TheGlobalData->m_terrainLOD+1);
344 if (adj<0 && TheGlobalData->m_terrainLOD>TERRAIN_LOD_MIN) TheWritableGlobalData->m_terrainLOD=(TerrainLOD)(TheGlobalData->m_terrainLOD-1);
345
346 if (TheGlobalData->m_terrainLOD ==TERRAIN_LOD_AUTOMATIC) {
347 TheWritableGlobalData->m_terrainLOD=TERRAIN_LOD_MAX;
348 }
349
350 if (m_map==NULL) return;
351 if (m_shroud)
352 m_shroud->reset(); //need reset here since initHeightData will load new shroud.
353
354 BaseHeightMapRenderObjClass *newROBJ = NULL;
355 if (TheGlobalData->m_terrainLOD==7) {
356 newROBJ = TheHeightMap;
357 if (newROBJ==NULL) {
358 newROBJ = NEW_REF( HeightMapRenderObjClass, () );
359 }
360 } else {
361 newROBJ = TheFlatHeightMap;
362 if (newROBJ==NULL) {
363 newROBJ = NEW_REF( FlatHeightMapRenderObjClass, () );
364 }
365 }
366 if (TheGlobalData->m_terrainLOD == 5)
367 newROBJ = NULL;
368 RTS3DScene *pMyScene = (RTS3DScene *)Scene;
369 if (pMyScene) {
370 pMyScene->Remove_Render_Object(this);
371 pMyScene->Unregister(this, SceneClass::ON_FRAME_UPDATE);
372 // add our terrain render object to the scene
373 if (newROBJ) {
374 pMyScene->Add_Render_Object( newROBJ );
375 pMyScene->Register(newROBJ,SceneClass::ON_FRAME_UPDATE);
376 }
377 }
378
379 if (newROBJ) {
380 // apply the heightmap to the terrain render object
381 newROBJ->initHeightData( m_map->getDrawWidth(),
382 m_map->getDrawHeight(),
383 m_map,
384 NULL);
385 TheTerrainRenderObject = newROBJ;
386 newROBJ->staticLightingChanged();
387 newROBJ->m_roadBuffer->loadRoads();
388 }
389 if (TheTacticalView) {
390 TheTacticalView->setAngle(TheTacticalView->getAngle() + 1);
391 TheTacticalView->setAngle(TheTacticalView->getAngle() - 1);
392 }
393}

516void BaseHeightMapRenderObjClass::doTheLight(VERTEX_FORMAT *vb, Vector3*light, Vector3*normal, RefRenderObjListIterator *pLightsIterator, UnsignedByte alpha)
517{
518#ifdef USE_NORMALS
519 vb->nx = normal->X;
520 vb->ny = normal->Y;
521 vb->nz = normal->Z;
522#else
523 Real shadeR, shadeG, shadeB;
524 Real shade;
525 shadeR = TheGlobalData->m_terrainAmbient[0].red; //only the first terrain light contributes to ambient
526 shadeG = TheGlobalData->m_terrainAmbient[0].green;
527 shadeB = TheGlobalData->m_terrainAmbient[0].blue;
528
529 if (pLightsIterator) {
530 for (pLightsIterator->First(); !pLightsIterator->Is_Done(); pLightsIterator->Next())
531 {
532 LightClass *pLight = (LightClass*)pLightsIterator->Peek_Obj();
533 Vector3 lightDirection(vb->x, vb->y, vb->z);
534 Real factor = 1.0f;
535 switch(pLight->Get_Type()) {
536 case LightClass::POINT:
537 case LightClass::SPOT: {
538 Vector3 lightLoc = pLight->Get_Position();
539 lightDirection -= lightLoc;
540 double range, midRange;
541 pLight->Get_Far_Attenuation_Range(midRange, range);
542 if (vb->x < lightLoc.X-range) continue;
543 if (vb->x > lightLoc.X+range) continue;
544 if (vb->y < lightLoc.Y-range) continue;
545 if (vb->y > lightLoc.Y+range) continue;
546 Real dist = lightDirection.Length();
547 if (dist >= range) continue;
548 if (midRange < 0.1) continue;
549#if 1
550 factor = 1.0f - (dist - midRange) / (range - midRange);
551#else
552 // f = 1.0 / (atten0 + d*atten1 + d*d/atten2);
553 if (fabs(range-midRange)<1e-5) {
554 // if the attenuation range is too small assume uniform with cutoff
555 factor = 1.0;
556 } else {
557 factor = 1.0f/(0.1+dist/midRange + 5.0f*dist*dist/(range*range));
558 }
559#endif
560 factor = WWMath::Clamp(factor,0.0f,1.0f);
561 }
562 break;
563 case LightClass::DIRECTIONAL:
564 lightDirection = pLight->Get_Transform().Get_Z_Vector();
565 factor = 1.0;
566 break;
567 };
568 lightDirection.Normalize();
569 Vector3 lightRay(-lightDirection.X, -lightDirection.Y, -lightDirection.Z);
570 shade = Vector3::Dot_Product(lightRay, *normal);
571 shade *= factor;
572 Vector3 diffuse;
573 pLight->Get_Diffuse(&diffuse);
574 Vector3 ambient;
575 pLight->Get_Ambient(&ambient);
576 if (shade > 1.0) shade = 1.0;
577 if(shade < 0.0f) shade = 0.0f;
578 shadeR += shade*diffuse.X;
579 shadeG += shade*diffuse.Y;
580 shadeB += shade*diffuse.Z;
581 shadeR += factor*ambient.X;
582 shadeG += factor*ambient.Y;
583 shadeB += factor*ambient.Z;
584
585 }
586 }
587 // Add in global diffuse value.
588 const RGBColor *terrainDiffuse;
589 for (Int lightIndex=0; lightIndex < TheGlobalData->m_numGlobalLights; lightIndex++)
590 {
591 shade = Vector3::Dot_Product(light[lightIndex], *normal);
592 if (shade > 1.0) shade = 1.0;
593 if(shade < 0.0f) shade = 0.0f;
594 terrainDiffuse=&TheGlobalData->m_terrainDiffuse[lightIndex];
595 shadeR += shade*terrainDiffuse->red;
596 shadeG += shade*terrainDiffuse->green;
597 shadeB += shade*terrainDiffuse->blue;
598 }
599
600 if (shadeR > 1.0) shadeR = 1.0;
601 if(shadeR < 0.0f) shadeR = 0.0f;
602 if (shadeG > 1.0) shadeG = 1.0;
603 if(shadeG < 0.0f) shadeG = 0.0f;
604 if (shadeB > 1.0) shadeB = 1.0;
605 if(shadeB < 0.0f) shadeB = 0.0f;
606
607 if (m_useDepthFade && vb->z <= TheGlobalData->m_waterPositionZ)
608 { //height is below water level
609 //reduce lighting values based on light fall off as it travels through water.
610 float depthScale = (1.4f - vb->z)/TheGlobalData->m_waterPositionZ;
611 shadeR *= 1.0f - depthScale * (1.0f-m_depthFade.X);
612 shadeG *= 1.0f - depthScale * (1.0f-m_depthFade.Y);
613 shadeB *= 1.0f - depthScale * (1.0f-m_depthFade.Z);
614 }
615
616 shadeR*=255.0f;
617 shadeG*=255.0f;
618 shadeB*=255.0f;
619 vb->diffuse = REAL_TO_INT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16) | ((Int)alpha << 24);
620#endif
621}

690//=============================================================================
691bool BaseHeightMapRenderObjClass::Cast_Ray(RayCollisionTestClass & raytest)
692{
693 TriClass tri;
694 Bool hit = false;
695 Int X,Y;
696 Vector3 normal,P0,P1,P2,P3;
697
698 if (!m_map)
699 return false; //need valid pointer to heightmap samples
700//HeightSampleType *pData = m_map->getDataPtr();
701 //Clip ray to extents of heightfield
702 AABoxClass hbox;
703 LineSegClass lineseg,lineseg2;
704 CastResultStruct result;
705 Int StartCellX = 0;
706 Int EndCellX = 0;
707 Int StartCellY = 0;
708 Int EndCellY = 0;
709 const Int overhang = 2*32+m_map->getBorderSizeInline(); // Allow picking past the edge for scrolling & objects.
710 Vector3 minPt(MAP_XY_FACTOR*(-overhang), MAP_XY_FACTOR*(-overhang), -MAP_XY_FACTOR);
711 Vector3 maxPt(MAP_XY_FACTOR*(m_map->getXExtent()+overhang),
712 MAP_XY_FACTOR*(m_map->getYExtent()+overhang), MAP_HEIGHT_SCALE*m_map->getMaxHeightValue()+MAP_XY_FACTOR);
713 MinMaxAABoxClass mmbox(minPt, maxPt);
714 hbox.Init(mmbox);
715
716 lineseg=raytest.Ray;
717
718 //Set initial ray endpoints
719 P0 = raytest.Ray.Get_P0();
720 P1 = raytest.Ray.Get_P1();
721 result.ComputeContactPoint=true;
722
723 Int p;
724 for (p=0; p<3; p++) {
725 //find intersection point of ray and terrain bounding box
726 result.Reset();
727 result.ComputeContactPoint=true;

1003
1004 }

1501 return; //we must be in the builder so don't sort.

1506
1507 if (shoreLineSortInfosSize <= (m_map->getYExtent()-1))

1510 }
1511
1512 m_shoreLineSortInfosXMajor= shoreLineSortInfosXMajor;
1513

1518 delete [] m_shoreLineSortInfos; //old buffer was too small.

1521 m_shoreLineSortInfosSize = shoreLineSortInfosSize;
1522

1524 }

1530 {
1531 //scan all the tiles and record batches of tiles using same x value.
1532 m_shoreLineTileSortMaxCoordinate=m_shoreLineTileSortMinCoordinate=(m_shoreLineTilePositions[0].m_xy & 0xffff);
1533 for (Int i=0; i<m_numShoreLineTiles; i++)

1535 Int x=(m_shoreLineTilePositions[i].m_xy & 0xffff);
1536
1537 shoreLineTileSortInfo *sortInfo=&m_shoreLineSortInfos[x];
1538
1539 if (x > m_shoreLineTileSortMaxCoordinate)
1540 m_shoreLineTileSortMaxCoordinate=x;
1541

1543 Int j=i+1;

1545 minY=maxY=m_shoreLineTilePositions[i].m_xy >> 16;
1546

1548 { //keep track of highest y coordinate.
1549 Int y = m_shoreLineTilePositions[j].m_xy >> 16;
1550 if (y > maxY)
1551 maxY=y;
1552 j++;
1553 }
1554
1555 sortInfo->tileStartIndex=i;
1556 sortInfo->numTiles=j-i;
1557 sortInfo->minTileCoordinate=minY;
1558 sortInfo->maxTileCoordinate=maxY;
1559 i += sortInfo->numTiles-1; //skip tiles we just scanned.
1560 }
1561 }
1562 else //map is taller than wider so tiles are already sorted by y
1563 {
1564 //scan all the tiles and record batches of tiles using same y value.
1565 m_shoreLineTileSortMaxCoordinate=m_shoreLineTileSortMinCoordinate=(m_shoreLineTilePositions[0].m_xy >> 16);
1566 for (Int i=0; i<m_numShoreLineTiles; i++)
1567 {

1569
1570 shoreLineTileSortInfo *sortInfo=&m_shoreLineSortInfos[y];

1597void BaseHeightMapRenderObjClass::updateShorelineTile(Int i, Int j, Int border, WorldHeightMap *pMap)
1598{
1599 Int waterSide;
1600 Real waterZ0,waterZ1,waterZ2,waterZ3;
1601 Real terrainZ0, terrainZ1, terrainZ2, terrainZ3;
1602
1603 //Figure out maximum depth of water before we reach the m_minWaterOpacity value. Depths greater than this don't need
1604 //custom shoreline tiles because they will get their opacity from the default value stored in the frame buffer during
1605 //a screen clear operation.
1606 Real transparentDepth=TheWaterTransparency->m_transparentWaterDepth*TheWaterTransparency->m_minWaterOpacity;
1607 Real depthScaleFactor = 1.0f/transparentDepth;
1608
1609 Real X0=(i-border)*MAP_XY_FACTOR;
1610 Real Y0=(j-border)*MAP_XY_FACTOR;
1611 waterSide=(waterZ0=TheWaterRenderObj->getWaterHeight(X0,Y0)) > ((terrainZ0=MAP_HEIGHT_SCALE*pMap->getHeight(i,j)));
1612 Real X1=(i-border+1)*MAP_XY_FACTOR;
1613 Real Y1=(j-border+1)*MAP_XY_FACTOR;
1614 waterSide |=((waterZ1=TheWaterRenderObj->getWaterHeight(X1,Y0)) > ((terrainZ1=MAP_HEIGHT_SCALE*pMap->getHeight(i+1,j)))) << 1;
1615 waterSide |=((waterZ2=TheWaterRenderObj->getWaterHeight(X1,Y1)) > ((terrainZ2=MAP_HEIGHT_SCALE*pMap->getHeight(i+1,j+1)))) << 2;
1616 waterSide |=((waterZ3=TheWaterRenderObj->getWaterHeight(X0,Y1)) > ((terrainZ3=MAP_HEIGHT_SCALE*pMap->getHeight(i,j+1)))) << 3;
1617
1618 if (!waterSide || (waterZ0*waterZ1*waterZ2*waterZ3) <= 0)
1619 return; //all verts are on positive (surface) side of water so don't need blending. Or one of them is outside the water plane bounds (waterHeight <= 0!)
1620
1621 //Check if mix of under/over water vertices or some vertices within depth fade region.
1622 if (waterSide < 0xf || (waterZ0 - terrainZ0) < transparentDepth ||
1623 (waterZ1 - terrainZ1) < transparentDepth || (waterZ2 - terrainZ2) < transparentDepth
1624 || (waterZ3 - terrainZ3) < transparentDepth)
1625 { //add tile to set that needs shoreline blending.
1626 if (m_numShoreLineTiles >= m_shoreLineTilePositionsSize)
1627 { //no more room to store extra blend tiles so enlarge the buffer.
1628 shoreLineTileInfo *tempPositions=NEW shoreLineTileInfo[m_shoreLineTilePositionsSize+512];
1629 memcpy(tempPositions, m_shoreLineTilePositions, m_shoreLineTilePositionsSize*sizeof(shoreLineTileInfo));
1630 delete [] m_shoreLineTilePositions;
1631 //enlarge by more tiles to reduce memory trashing
1632 m_shoreLineTilePositions = tempPositions;
1633 m_shoreLineTilePositionsSize += 512;
1634 }
1635 //Pack x and y position into single integer since maps are limited in size
1636 shoreLineTileInfo *shoreInfo=&m_shoreLineTilePositions[m_numShoreLineTiles];
1637 shoreInfo->m_xy=i | (j <<16);
1638 shoreInfo->verts[0]=X0; shoreInfo->verts[1]=Y0; shoreInfo->verts[2]=terrainZ0;
1639 shoreInfo->t0=(waterZ0 - terrainZ0)*depthScaleFactor;
1640 shoreInfo->verts[3]=X1; shoreInfo->verts[4]=Y0; shoreInfo->verts[5]=terrainZ1;
1641 shoreInfo->t1=(waterZ1 - terrainZ1)*depthScaleFactor;
1642 shoreInfo->verts[6]=X1; shoreInfo->verts[7]=Y1; shoreInfo->verts[8]=terrainZ2;
1643 shoreInfo->t2=(waterZ2 - terrainZ2)*depthScaleFactor;
1644 shoreInfo->verts[9]=X0; shoreInfo->verts[10]=Y1; shoreInfo->verts[11]=terrainZ3;
1645 shoreInfo->t3=(waterZ3 - terrainZ3)*depthScaleFactor;
1646
1647 m_numShoreLineTiles++;
1648 }
1649}

1653void BaseHeightMapRenderObjClass::updateShorelineTiles(Int minX, Int minY, Int maxX, Int maxY, WorldHeightMap *pMap)
1654{
1655 Int border = pMap->getBorderSizeInline();
1656
1657 //Clamp region to valid terrain tiles
1658 if (minX<0)
1659 minX = 0;
1660 if (minY<0)
1661 minY = 0;
1662 if (maxX > (pMap->getXExtent() - 1))
1663 maxX = (pMap->getXExtent() - 1);
1664 if (maxY > (pMap->getYExtent() - 1))
1665 maxY = (pMap->getYExtent() - 1);
1666
1667 if (!m_shoreLineTilePositions)
1668 { //Need to allocate memory
1669 m_shoreLineTilePositions = NEW shoreLineTileInfo[DEFAULT_MAX_MAP_SHORELINE_TILES];
1670 m_shoreLineTilePositionsSize = DEFAULT_MAX_MAP_SHORELINE_TILES;
1671 }
1672
1673 //First remove any existing extra blend tiles within this partial region
1674 for (Int j=0; j<m_numShoreLineTiles; j++)
1675 { Int x = m_shoreLineTilePositions[j].m_xy & 0xffff;
1676 Int y = m_shoreLineTilePositions[j].m_xy >> 16;
1677 if (x >= minX && x < maxX &&
1678 y >= minY && y < maxY)
1679 { //this tile is inside region being updated so remove it by shifting tile array
1680 memcpy(m_shoreLineTilePositions+j,m_shoreLineTilePositions+j+1,(m_numShoreLineTiles-1-j)*sizeof(shoreLineTileInfo));
1681 m_numShoreLineTiles--;
1682 j--; //look at current tile again since it was removed.
1683 }
1684 }
1685
1686 if (TheWaterTransparency->m_transparentWaterDepth == 0 || !TheGlobalData->m_showSoftWaterEdge)
1687 return;
1688
1689 //we want to add the tiles in a certain order to make culling faster
1690 //by default they are inserted in x-order
1691 Bool shoreLineSortInfosXMajor=FALSE;
1692 if (!TheGlobalData->m_isWorldBuilder) //we're in the game, not the builder
1693 {
1694 if ((m_map->getXExtent()-1) > (m_map->getYExtent()-1))
1695 shoreLineSortInfosXMajor=TRUE;
1696 }
1697
1698 if (shoreLineSortInfosXMajor)
1699 for (Int i=minX; i<maxX; i++)
1700 for (j=minY; j<maxY; j++)
1701 {
1702 updateShorelineTile(i,j,border,pMap);
1703 }
1704 else
1705 for (j=minY; j<maxY; j++)
1706 for (Int i=minX; i<maxX; i++)
1707 {
1708 updateShorelineTile(i,j,border,pMap);
1709 }
1710
1711 recordShoreLineSortInfos();
1712}
