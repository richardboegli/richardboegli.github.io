64void SpawnPointProductionExitUpdate::exitObjectViaDoor( Object *newObj, ExitDoorType exitDoor )
65{
66 DEBUG_ASSERTCRASH(exitDoor == DOOR_1, ("multiple exit doors not supported here"));
67
68 if( !m_bonesInitialized )
69 initializeBonePositions();
70
71 Object *creationObject = getObject();
72 if (creationObject)
73 {
74 for( Int positionIndex = 0; positionIndex < m_spawnPointCount; positionIndex++ )
75 {
76 if( m_spawnPointOccupier[positionIndex] == INVALID_ID )
77 break;
78 }
79 if( positionIndex == m_spawnPointCount )
80 {
81 DEBUG_CRASH( ("A SpawnPoint exit thought it had room but then failed") );
82 return;
83 }
84
85 Coord3D createPoint;
86 Real createAngle;
87
88 // Get the location from our bone array
89 createPoint = m_worldCoordSpawnPoints[positionIndex];
90
91 // make sure the point is on the terrain
92 createPoint.z = TheTerrainLogic ? TheTerrainLogic->getLayerHeight( createPoint.x, createPoint.y, creationObject->getLayer()) : 0.0f;
93
94 // get the angle
95 createAngle = m_worldAngleSpawnPoints[positionIndex];
96
97 // record him as taking this spot
98 m_spawnPointOccupier[positionIndex] = newObj->getID();
99
100 // put him there
101 newObj->setPosition( &createPoint );
102 newObj->setOrientation( createAngle );
103 newObj->setLayer(creationObject->getLayer());
104
107 // tell the AI about it
108 TheAI->pathfinder()->addObjectToPathfindMap( newObj );
109
110 // You are stuck here, little man.
111 newObj->setDisabled( DISABLED_HELD );
112 }
113}

116ExitDoorType SpawnPointProductionExitUpdate::reserveDoorForExit( const ThingTemplate* objType, Object *specificObject )
117{
118 if( !m_bonesInitialized )
119 initializeBonePositions();
120
121 if( !m_bonesInitialized )
122 return DOOR_NONE_AVAILABLE; // Init failure
123
124 revalidateOccupiers();
125
126 for( Int positionIndex = 0; positionIndex < m_spawnPointCount; positionIndex++ )
127 {
128 if( m_spawnPointOccupier[positionIndex] == INVALID_ID )
129 return DOOR_1;
130 }
131
132 return DOOR_NONE_AVAILABLE;
133}

142void SpawnPointProductionExitUpdate::initializeBonePositions()
143{
144 Object *me = getObject();
145 Drawable *myDrawable = me->getDrawable();
146
147 // This fundamental failure will result in this never ever thinking it is free
148 if( myDrawable == NULL )
149 return;
150
151 Matrix3D boneTransforms[MAX_SPAWN_POINTS];
152 for( Int matrixIndex = 0; matrixIndex < MAX_SPAWN_POINTS; matrixIndex++ )
153 boneTransforms[matrixIndex].Make_Identity();
154
155 // Get all the bones of the right name
156 const SpawnPointProductionExitUpdateModuleData* md = getSpawnPointProductionExitUpdateModuleData();
157 m_spawnPointCount = myDrawable->getPristineBonePositions( md->m_spawnPointBoneNameData.str(), 1, NULL, boneTransforms, MAX_SPAWN_POINTS );
158
159 for( matrixIndex = 0; matrixIndex < m_spawnPointCount; matrixIndex++ )
160 {
161 Matrix3D *currentTransform = &(boneTransforms[matrixIndex]);
162 // Convert their matrix one by one
163 me->convertBonePosToWorldPos( NULL, currentTransform, NULL, currentTransform );
164
165 // Then save the world coord and angle
166 m_worldCoordSpawnPoints[matrixIndex].x = currentTransform->Get_X_Translation();
167 m_worldCoordSpawnPoints[matrixIndex].y = currentTransform->Get_Y_Translation();
168 m_worldCoordSpawnPoints[matrixIndex].z = 0; //set at creation time
169
170 m_worldAngleSpawnPoints[matrixIndex] = currentTransform->Get_Z_Rotation();
171 }
172
173 m_bonesInitialized = TRUE;
174}

177void SpawnPointProductionExitUpdate::revalidateOccupiers()
178{
179 for( Int positionIndex = 0; positionIndex < m_spawnPointCount; positionIndex++ )
180 {
181 if( m_spawnPointOccupier[positionIndex] == INVALID_ID )
182 continue;
183
184 if( TheGameLogic->findObjectByID( m_spawnPointOccupier[positionIndex] ) == NULL )
185 m_spawnPointOccupier[positionIndex] = INVALID_ID;
186 }
187}
