111void LaserUpdate::updateStartPos()
112{
113 Coord3D oldStartPos = m_startPos;
114
115 if( m_parentID == INVALID_DRAWABLE_ID )
116 return;// Can't update if not told to update
117
118 const Drawable *parentDrawable = TheGameClient->findDrawableByID(m_parentID);
119 if( parentDrawable == NULL )
120 return;// Can't update if no one to ask
121
122 if( m_parentBoneName.isNotEmpty() )
123 {
124 Matrix3D startPosMatrix;
125
126 if( !parentDrawable->getCurrentWorldspaceClientBonePositions( m_parentBoneName.str(), startPosMatrix ) )
127 {
128 // failed to find the required bone, so just die
129
130 //Kris: Doing this CRASHES THE GAME LATER!!!! Instead, let's set the position to the drawable, then
131 // create a nasty assert.
132 //TheGameClient->destroyDrawable( getDrawable() );
133
134 m_startPos.set( parentDrawable->getPosition() );
135 DEBUG_CRASH( ("LaserUpdate::updateStartPos() -- Drawable %s is expecting to find a bone %s but can't. Defaulting to position of drawable.",
136 parentDrawable->getTemplate()->getName().str(), m_parentBoneName.str() ) );
137
138 return;
139 }
140
141
142
143 m_startPos.x = startPosMatrix.Get_X_Translation();
144 m_startPos.y = startPosMatrix.Get_Y_Translation();
145 m_startPos.z = startPosMatrix.Get_Z_Translation();
146 }
147 else
148 {
149 // Just use parent position
150 m_startPos = *parentDrawable->getPosition();
151 }
152
153 if( !(m_startPos == oldStartPos) ) // No != operator. Heh.
154 m_dirty = TRUE;
155}

157//-------------------------------------------------------------------------------------------------
158void LaserUpdate::updateEndPos()
159{
160 const LaserUpdateModuleData *data = getLaserUpdateModuleData();
161 Coord3D oldEndPos = m_endPos;
162
163 if( m_targetID == INVALID_DRAWABLE_ID )
164 return;// Can't update if not told to update
165
166 const Drawable *targetDrawable = TheGameClient->findDrawableByID(m_targetID);
167 Bool targetDead = (targetDrawable && targetDrawable->getObject())
168 ? targetDrawable->getObject()->isEffectivelyDead()
169 : FALSE;
170 if( targetDrawable == NULL || targetDead )
171 {
172 // If here, we used to track something, but now it is gone. So make our end point pierce through
173 // the old spot, and then stop trying to find a target Drawable
174 if( data->m_punchThroughScalar > 0 )
175 {
176 Vector3 laserVector;
177 laserVector.Set(m_endPos.x, m_endPos.y, m_endPos.z);
178 laserVector = laserVector - Vector3(m_startPos.x, m_startPos.y, m_startPos.z);
179 laserVector *= data->m_punchThroughScalar;
180 laserVector = laserVector + Vector3(m_startPos.x, m_startPos.y, m_startPos.z);
181 m_endPos.x = laserVector.X;
182 m_endPos.y = laserVector.Y;
183 m_endPos.z = laserVector.Z;
184 }
185
186 m_targetID = INVALID_DRAWABLE_ID;
187 }
188 else
189 {
190 m_endPos = *targetDrawable->getPosition();
191 }
192
193 if( !(m_endPos == oldEndPos) ) // No != operator. Heh.
194 m_dirty = TRUE;
195}

200void LaserUpdate::clientUpdate( void )
201{
202 updateStartPos();
203 updateEndPos();
204
205 if( m_decaying )
206 {
207 UnsignedInt now = TheGameLogic->getFrame();
208 m_currentWidthScalar = 1.0f - (Real)(now - m_decayStartFrame) / (Real)(m_decayFinishFrame - m_decayStartFrame);
209 m_dirty = true;
210 if( m_currentWidthScalar <= 0.0f )
211 {
212 m_currentWidthScalar = 0.0f;
213
214 //When decay is finished... delete the laser.
215 //TheGameLogic->destroyObject( getObject() );
216 return;
217 }
218 }
219 else if( m_widening )
220 {
221 //We need to resize our laser width based on the growth ratio completed.
222 UnsignedInt now = TheGameLogic->getFrame();
223 m_currentWidthScalar = (Real)(now - m_widenStartFrame) / (Real)(m_widenFinishFrame - m_widenStartFrame);
224 m_dirty = true;
225 if( m_currentWidthScalar >= 1.0f )
226 {
227 m_currentWidthScalar = 1.0f;
228 m_widening = false;
229 }
230 }
231 return;
232}

247void LaserUpdate::initLaser( const Object *parent, const Object *target, const Coord3D *startPos, const Coord3D *endPos, AsciiString parentBoneName, Int sizeDeltaFrames )
248{
249 const LaserUpdateModuleData *data = getLaserUpdateModuleData();
250 ParticleSystem *system;
251 if( sizeDeltaFrames > 0 )
252 {
253 m_widening = true;
254 m_widenStartFrame = TheGameLogic->getFrame();
255 m_widenFinishFrame = m_widenStartFrame + sizeDeltaFrames;
256 m_currentWidthScalar = 0.0f;
257 }
258 else if( sizeDeltaFrames < 0 )
259 {
260 m_decaying = true;
261 m_decayStartFrame = TheGameLogic->getFrame();
262 m_decayFinishFrame = m_decayStartFrame - sizeDeltaFrames;
263 m_currentWidthScalar = 1.0f;
264 }
265
266 // Write down the bone name override
267 m_parentBoneName = parentBoneName;
268
269 //Record IDs if we have them, then figure out starting points
270 if( parent )
271 {
272 // If a source object, use it
273 if( parent->getDrawable() )
274 m_parentID = parent->getDrawable()->getID();
275
276 updateStartPos();
277 }
278 else if( startPos )
279 {
280 // or just use what they gave
281 m_startPos = *startPos;
282 }
283 else
284 {
285 // if they gave nothing, then we are screwed
286 TheGameClient->destroyDrawable( getDrawable() );
287 return;
288 }
289
290 if( target && !endPos )
291 {
292 // If a target object, use it (unless we override it!)
293 if( target->getDrawable() )
294 m_targetID = target->getDrawable()->getID();
295
296 m_endPos = *target->getPosition();
297 }
298 else if( endPos )
299 {
300 // just use what they gave, no override here
301 m_endPos = *endPos;
302 }
303 else
304 {
305 // if they gave nothing, then we are screwed
306 TheGameClient->destroyDrawable( getDrawable() );
307 return;
308 }
309
310 // Create special particle systems
311 //PLEASE NOTE You cannot check an ID for NULL. This should be a check against INVALID_PARTICLE_SYSTEM_ID. Can't change it on the last day without a bug though.
312 if( !m_particleSystemID )
313 {
314 const Player *localPlayer = ThePlayerList->getLocalPlayer();
315
316 //Make sure the laser flare is visible to the player. If no parent, assume laser owner will handle it.
317 if (!parent || parent->getShroudedStatus( localPlayer->getPlayerIndex() ) <= OBJECTSHROUD_PARTIAL_CLEAR )
318 {
319
320 //If we don't have a particle system for the lense flare (muzzle flare), create it.
321 if( data->m_particleSystemName.isNotEmpty() )
322 {
323 const ParticleSystemTemplate *tmp = TheParticleSystemManager->findTemplate( data->m_particleSystemName );
324 if( tmp )
325 {
326 system = TheParticleSystemManager->createParticleSystem( tmp );
327 if( system )
328 {
329 m_particleSystemID = system->getSystemID();
330 }
331 }
332 }
333
334 //If we don't have a particle system for the target effect, create it.
335 if( data->m_targetParticleSystemName.isNotEmpty() )
336 {
337 const ParticleSystemTemplate *tmp = TheParticleSystemManager->findTemplate( data->m_targetParticleSystemName );
338 if( tmp )
339 {
340 system = TheParticleSystemManager->createParticleSystem( tmp );
341 if( system )
342 {
343 m_targetParticleSystemID = system->getSystemID();
344 }
345 }
346 }
347 }
348 }
349
350 //Adjust the position of any existing particle system.
351 //PLEASE NOTE You cannot check an ID for NULL. This should be a check against INVALID_PARTICLE_SYSTEM_ID. Can't change it on the last day without a bug though.
352 if( m_particleSystemID )
353 {
354 system = TheParticleSystemManager->findParticleSystem( m_particleSystemID );
355 if( system )
356 {
357 system->setPosition( &m_startPos );
358 }
359 }
360
361 //PLEASE NOTE You cannot check an ID for NULL. This should be a check against INVALID_PARTICLE_SYSTEM_ID. Can't change it on the last day without a bug though.
362 if( m_targetParticleSystemID )
363 {
364 system = TheParticleSystemManager->findParticleSystem( m_targetParticleSystemID );
365 if( system )
366 {
367 system->setPosition( &m_endPos );
368 }
369 }
370
371 //Important! Set the laser position to the average of both points or else
372 //it probably won't get rendered!!!
373 // And as a client update, we cannot set the logic position.
374 Coord3D posToUse;
375 if( parent == NULL )
376 {
377 posToUse.set( startPos );
378 posToUse.add( endPos );
379 posToUse.scale( 0.5 );
380 }
381 else
382 {
383 posToUse = *parent->getPosition();
384 }
385
386 Drawable *draw = getDrawable();
387 if( draw )
388 {
389 draw->setPosition( &posToUse );
390 }
391
392 m_dirty = true;
393}

396Real LaserUpdate::getCurrentLaserRadius() const
397{
398 const Drawable *draw = getDrawable();
399 const LaserDrawInterface* ldi = NULL;
400 for( const DrawModule** d = draw->getDrawModules(); *d; ++d )
401 {
402 ldi = (*d)->getLaserDrawInterface();
403 if( ldi )
404 {
405 //***NOTE***
406 //While it appears the logic is accessing client data, it is actually accessing template module
407 //data from the client. This value is INI constant thus can't change. It's grouped with other
408 //laser defining attributes and having it there makes it easier for artists.
409 return ldi->getLaserTemplateWidth() * m_currentWidthScalar;
410 }
411 }
412 return 0.0f;
413}

431void LaserUpdate::xfer( Xfer *xfer )

459
460 // decaying
461 xfer->xferBool( &m_decaying );
462
463 // widen start frame
464 xfer->xferUnsignedInt( &m_widenStartFrame );
465
466 // widen finish frame
467 xfer->xferUnsignedInt( &m_widenFinishFrame );
468
469 // current width scalar
470 xfer->xferReal( &m_currentWidthScalar );
471
472 // decay start frame
473 xfer->xferUnsignedInt( &m_decayStartFrame );
474
475 // decay finish frame
476 xfer->xferUnsignedInt( &m_decayFinishFrame );
477
478 xfer->xferDrawableID(&m_parentID);
479 xfer->xferDrawableID(&m_targetID);
480
481 xfer->xferAsciiString(&m_parentBoneName);
482
483} // end xfer
