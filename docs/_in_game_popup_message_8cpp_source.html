89void InGamePopupMessageInit( WindowLayout *layout, void *userData )
90{
91
92 parentID = TheNameKeyGenerator->nameToKey(AsciiString("InGamePopupMessage.wnd:InGamePopupMessageParent"));
93 parent = TheWindowManager->winGetWindowFromId(NULL, parentID);
94
95 staticTextMessageID = TheNameKeyGenerator->nameToKey(AsciiString("InGamePopupMessage.wnd:StaticTextMessage"));
96 staticTextMessage = TheWindowManager->winGetWindowFromId(parent, staticTextMessageID);
97 buttonOkID = TheNameKeyGenerator->nameToKey(AsciiString("InGamePopupMessage.wnd:ButtonOk"));
98 buttonOk = TheWindowManager->winGetWindowFromId(parent, buttonOkID);
99
100 PopupMessageData *pMData = TheInGameUI->getPopupMessageData();
101
102 if(!pMData)
103 {
104 DEBUG_ASSERTCRASH(pMData, ("We're in InGamePopupMessage without a pointer to pMData\n") );
106 return;
107 }
108
109 DisplayString *tempString = TheDisplayStringManager->newDisplayString();
110 tempString->setText(pMData->message);
111 tempString->setFont(staticTextMessage->winGetFont());
112 tempString->setWordWrap(pMData->width - 14);
113 Int width, height;
114 tempString->getSize(&width, &height);
115 TheDisplayStringManager->freeDisplayString(tempString);
116
117 GadgetStaticTextSetText(staticTextMessage, pMData->message);
118 // set the positions/sizes
119 Int widthOk, heightOk;
120 buttonOk->winGetSize(&widthOk, &heightOk);
121 parent->winSetPosition( pMData->x, pMData->y);
122 parent->winSetSize( pMData->width, height + 7 + 2 + 2 + heightOk + 2 );
123 staticTextMessage->winSetPosition( 2, 2);
124 staticTextMessage->winSetSize( pMData->width - 4, height + 7);
125 buttonOk->winSetPosition(pMData->width - widthOk - 2, height + 7 + 2 + 2);
126 staticTextMessage->winSetEnabledTextColors(pMData->textColor, 0);
127 pause = pMData->pause;
128 if(pMData->pause)
129 TheWindowManager->winSetModal( parent );
130
131 TheWindowManager->winSetFocus( parent );
132
133 parent->winHide(FALSE);
134 parent->winBringToTop();
135}

140WindowMsgHandledType InGamePopupMessageInput( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
141{
142
143 switch( msg )
144 {
145
146 // --------------------------------------------------------------------------------------------
147 case GWM_CHAR:
148 {
149 UnsignedByte key = mData1;
150 UnsignedByte state = mData2;
151 // if (buttonPushed)
152 // break;
153
154 switch( key )
155 {
156
157 // ----------------------------------------------------------------------------------------
158 case KEY_ENTER:
159 case KEY_ESC:
160 {
161
162 //
163 // send a simulated selected event to the parent window of the
164 // back/exit button
165 //
166 if( BitTest( state, KEY_STATE_UP ) )
167 {
168 TheWindowManager->winSendSystemMsg( window, GBM_SELECTED,
169 (WindowMsgData)buttonOk, buttonOkID );
170
171 } // end if
172
173 // don't let key fall through anywhere else
174 return MSG_HANDLED;
175
176 } // end escape
177
178 } // end switch( key )
179
180 } // end char
181
182 } // end switch( msg )
183 return MSG_IGNORED;
184
185
186}

191WindowMsgHandledType InGamePopupMessageSystem( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
192{
193
194 switch( msg )
195 {
196
197 // --------------------------------------------------------------------------------------------
198 case GWM_CREATE:
199 {
200
201 break;
202
203 } // end create
204 //---------------------------------------------------------------------------------------------
205 case GWM_DESTROY:
206 {
207
208 break;
209
210 } // end case
211
212 //----------------------------------------------------------------------------------------------
213 case GWM_INPUT_FOCUS:
214 {
215
216 // if we're givin the opportunity to take the keyboard focus we must say we want it
217 if( mData1 == TRUE )
218 *(Bool *)mData2 = TRUE;
219
220 break;
221
222 } // end input
223 //---------------------------------------------------------------------------------------------
224 case GBM_SELECTED:
225 {
226 GameWindow *control = (GameWindow *)mData1;
227 Int controlID = control->winGetWindowId();
228
229 if( controlID == buttonOkID )
230 {
231 if(!pause)
232 TheMessageStream->appendMessage( GameMessage::MSG_CLEAR_INGAME_POPUP_MESSAGE );
233 else
234 TheInGameUI->clearPopupMessageData();
235 }
236 break;
237 }
238 default:
239 return MSG_IGNORED;
240
241 } // end switch
242
243
244 return MSG_HANDLED;
245
246}
