66// drawCheckBoxText ===========================================================
68//=============================================================================
69static void drawCheckBoxText( GameWindow *window, WinInstanceData *instData )
70{
71 ICoord2D origin, size, textPos;
72 Int width, height;
73 Color textColor, dropColor;
74 DisplayString *text = instData->getTextDisplayString();
75
76 // sanity
77 if( text == NULL || text->getTextLength() == 0 )
78 return;
79
80 // get window position and size
81 window->winGetScreenPosition( &origin.x, &origin.y );
82 window->winGetSize( &size.x, &size.y );
83
84 // get the right text color
85 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
86 {
87 textColor = window->winGetDisabledTextColor();
88 dropColor = window->winGetDisabledTextBorderColor();
89 } // end if, disabled
90 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
91 {
92 textColor = window->winGetHiliteTextColor();
93 dropColor = window->winGetHiliteTextBorderColor();
94 } // end else if, hilited
95 else
96 {
97 textColor = window->winGetEnabledTextColor();
98 dropColor = window->winGetEnabledTextBorderColor();
99 } // end enabled only
100
101 // set our font to that of our parent if not the same
102 if( text->getFont() != window->winGetFont() )
103 text->setFont( window->winGetFont() );
104
105 // get text size
106 text->getSize( &width, &height );
107
108 // where to draw
109 textPos.x = origin.x + size.y;//(size.x / 2) - (width / 2);
110 textPos.y = origin.y + (size.y / 2) - (height / 2);
111
112 // draw it
113 text->draw( textPos.x, textPos.y, textColor, dropColor );
114
115} // end drawCheckBoxText

123// W3DGadgetCheckBoxDraw ======================================================
125//=============================================================================
126void W3DGadgetCheckBoxDraw( GameWindow *window, WinInstanceData *instData )
127{
128 Int checkOffsetFromLeft;
129 Color backColor,
130 backBorder,
131 boxColor,
132 boxBorder;
133 ICoord2D origin, size, start, end;
134
135 // get window position and size
136 window->winGetScreenPosition( &origin.x, &origin.y );
137 window->winGetSize( &size.x, &size.y );
138
139 // compute start of check offset
140 checkOffsetFromLeft = size.x / 16;
141
142 //
143 // get the colors we should be using to draw, see GadgetCheckBox.h
144 // draw appropriate state, see GadgetCheckBox.h for info
145 //
146 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
147 {
148
149 // disabled background
150 backColor = GadgetCheckBoxGetDisabledColor( window );
151 backBorder = GadgetCheckBoxGetDisabledBorderColor( window );
152
153 // check box
154 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
155 {
156 boxColor = GadgetCheckBoxGetDisabledCheckedBoxColor( window );
157 boxBorder = GadgetCheckBoxGetDisabledCheckedBoxBorderColor( window );
158 }
159 else
160 {
161 boxColor = GadgetCheckBoxGetDisabledUncheckedBoxColor( window );
162 boxBorder = GadgetCheckBoxGetDisabledUncheckedBoxBorderColor( window );
163 }
164
165 } // end if
166 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
167 {
168
169 // hilited background
170 backColor = GadgetCheckBoxGetHiliteColor( window );
171 backBorder = GadgetCheckBoxGetHiliteBorderColor( window );
172
173 // check box
174 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
175 {
176 boxColor = GadgetCheckBoxGetHiliteCheckedBoxColor( window );
177 boxBorder = GadgetCheckBoxGetHiliteCheckedBoxBorderColor( window );
178 }
179 else
180 {
181 boxColor = GadgetCheckBoxGetHiliteUncheckedBoxColor( window );
182 boxBorder = GadgetCheckBoxGetHiliteUncheckedBoxBorderColor( window );
183 }
184
185 } // end else if
186 else
187 {
188
189 // enabled background
190 backColor = GadgetCheckBoxGetEnabledColor( window );
191 backBorder = GadgetCheckBoxGetEnabledBorderColor( window );
192
193 // check box
194 if( BitTest( instData->getState(), WIN_STATE_SELECTED ) )
195 {
196 boxColor = GadgetCheckBoxGetEnabledCheckedBoxColor( window );
197 boxBorder = GadgetCheckBoxGetEnabledCheckedBoxBorderColor( window );
198 }
199 else
200 {
201 boxColor = GadgetCheckBoxGetEnabledUncheckedBoxColor( window );
202 boxBorder = GadgetCheckBoxGetEnabledUncheckedBoxBorderColor( window );
203 }
204
205 } // end else
206
207 // draw background border
208 start.x = origin.x;
209 start.y = origin.y;
210 end.x = start.x + size.x;
211 end.y = start.y + size.y;
212 TheWindowManager->winOpenRect( backBorder, WIN_DRAW_LINE_WIDTH,
213 start.x, start.y, end.x, end.y );
214
215 // draw the background
216 start.x++;
217 start.y++;
218 end.x--;
219 end.y--;
220 TheWindowManager->winFillRect( backColor, WIN_DRAW_LINE_WIDTH,
221 start.x, start.y, end.x, end.y );
222
223 // draw box border
224 start.x = origin.x + checkOffsetFromLeft;
225 start.y = origin.y + (size.y / 3);
226 end.x = start.x + (size.y / 3);
227 end.y = start.y + (size.y / 3);
228 TheWindowManager->winOpenRect( boxBorder, WIN_DRAW_LINE_WIDTH,
229 start.x, start.y, end.x, end.y );
230
231 // draw "x" for button
232 if( boxColor != WIN_COLOR_UNDEFINED )
233 {
234
235 TheWindowManager->winDrawLine( boxColor, WIN_DRAW_LINE_WIDTH,
236 start.x, start.y, end.x, end.y );
237 TheWindowManager->winDrawLine( boxColor, WIN_DRAW_LINE_WIDTH,
238 start.x, end.y, end.x, start.y );
239
240 } // end if
241
242 // draw the button text
243 if( instData->getTextLength() )
244 drawCheckBoxText( window, instData );
245
246} // end W3DGadgetCheckBoxDraw
