99void MapPreview::interpolateColorForHeight( RGBColor *color,
100 Real height,
101 Real hiZ,
102 Real midZ,
103 Real loZ )
104{
105 const Real howBright = 0.30f; // bigger is brighter (0.0 to 1.0)
106 const Real howDark = 0.60f; // bigger is darker (0.0 to 1.0)
107
108 // sanity on map height (flat maps bomb)
109 if (hiZ == midZ)
110 hiZ = midZ+0.1f;
111 if (midZ == loZ)
112 loZ = midZ-0.1f;
113 if (hiZ == loZ)
114 hiZ = loZ+0.2f;
115
116// Real heightPercent = height / (hiZ - loZ);
117 Real t;
118 RGBColor colorTarget;
119
120 // if "over" the middle height, interpolate lighter
121 if( height >= midZ )
122 {
123
124 // how far are we from the middleZ towards the hi Z
125 t = (height - midZ) / (hiZ - midZ);
126
127 // compute what our "lightest" color possible we want to use is
128 colorTarget.red = color->red + (1.0f - color->red) * howBright;
129 colorTarget.green = color->green + (1.0f - color->green) * howBright;
130 colorTarget.blue = color->blue + (1.0f - color->blue) * howBright;
131
132 } // end if
133 else // interpolate darker
134 {
135
136 // how far are we from the middleZ towards the low Z
137 t = (midZ - height) / (midZ - loZ);
138
139 // compute what the "darkest" color possible we want to use is
140 colorTarget.red = color->red + (0.0f - color->red) * howDark;
141 colorTarget.green = color->green + (0.0f - color->green) * howDark;
142 colorTarget.blue = color->blue + (0.0f - color->blue) * howDark;
143
144 } // end else
145
146 // interpolate toward the target color
147 color->red = color->red + (colorTarget.red - color->red) * t;
148 color->green = color->green + (colorTarget.green - color->green) * t;
149 color->blue = color->blue + (colorTarget.blue - color->blue) * t;
150
151 // keep the color real
152 if( color->red < 0.0f )
153 color->red = 0.0f;
154 if( color->red > 1.0f )
155 color->red = 1.0f;
156 if( color->green < 0.0f )
157 color->green = 0.0f;
158 if( color->green > 1.0f )
159 color->green = 1.0f;
160 if( color->blue < 0.0f )
161 color->blue = 0.0f;
162 if( color->blue > 1.0f )
163 color->blue = 1.0f;
164
165} // end interpolateColorForHeight

167Bool MapPreview::mapPreviewToWorld(const ICoord2D *radar, Coord3D *world)
168{
169 Int x, y;
170
171 // sanity
172 if( radar == NULL || world == NULL )
173 return FALSE;
174
175 // get the coords
176 x = radar->x;
177 y = radar->y;
178
179 // more sanity
180 if( x < 0 )
181 x = 0;
182 if( x >= MAP_PREVIEW_WIDTH )
183 x = MAP_PREVIEW_WIDTH - 1;
184 if( y < 0 )
185 y = 0;
186 if( y >= MAP_PREVIEW_HEIGHT )
187 y = MAP_PREVIEW_HEIGHT - 1;
188 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
189 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
190 Real xSample, ySample;
191
192 xSample = INT_TO_REAL(pMap->getXExtent() - (2 * pMap->getBorderSize())) / MAP_PREVIEW_WIDTH;
193 ySample = INT_TO_REAL(pMap->getYExtent() - (2 * pMap->getBorderSize())) / MAP_PREVIEW_HEIGHT;
194 // translate to world
195 world->x = x * xSample + pMap->getBorderSize();
196 world->y = y * ySample + pMap->getBorderSize();
197
198 // find the terrain height here
199// world->z = pMap->gethigh
200// TheTerrainLogic->getGroundHeight( world->x, world->y );
201
202 return TRUE;
203
204}

206void MapPreview::buildMapPreviewTexture( CString tgaName )
207{
208// SurfaceClass *surface;
209 RGBColor waterColor;
210
211 // we will want to reconstruct our new view box now
212 //m_reconstructViewBox = TRUE;
213
214 // setup our water color
215 waterColor.red = 0.55f;
216 waterColor.green = 0.55f;
217 waterColor.blue = 1.0f;
218
219 // fill the terrain texture with a representation of the terrain
220// Real mapDepth = m_mapExtent.depth();
221
222 // build the terrain
223 RGBColor sampleColor;
224 RGBColor color;
225 Int i, j, samples;
226 Int x, y, z;
227 ICoord2D radarPoint, boundary;
228 Coord3D worldPoint;
229 Real getTerrainAverageZ = 0;
230 Real maxHeight = 0;
231 Real minHeight = 100.0f;
232 Real tempHeight = 0;
233 Int terrainCount = 0;
234 CWorldBuilderDoc* pDoc = CWorldBuilderDoc::GetActiveDoc();
235 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
236 pMap->getBoundary(0, &boundary );
237 for(i = 0; i < MAP_PREVIEW_HEIGHT; ++i)
238 {
239 for(j = 0; j < MAP_PREVIEW_WIDTH; ++j)
240 {
241 radarPoint.x = j;
242 radarPoint.y = i;
243 mapPreviewToWorld( &radarPoint, &worldPoint );
244 tempHeight = pMap->getHeight(worldPoint.x, worldPoint.y);
245 getTerrainAverageZ += tempHeight;
246 if(tempHeight > maxHeight)
247 maxHeight = tempHeight;
248 if(tempHeight < minHeight)
249 minHeight = tempHeight;
250 ++terrainCount;
251 }
252 }
253 getTerrainAverageZ = getTerrainAverageZ /terrainCount;
254
255
256
257// Color paddingColor = GameMakeColor( 100, 100, 100, 255 );
258 for( y = 0; y < MAP_PREVIEW_HEIGHT; y++ )
259 {
260
261 for( x = 0; x < MAP_PREVIEW_WIDTH; x++ )
262 {
263
264 // what point are we inspecting
265 radarPoint.x = x;
266 radarPoint.y = y;
267 mapPreviewToWorld( &radarPoint, &worldPoint );
268
269
270 // get height of the terrain at this sample point
271 z = pMap->getHeight(worldPoint.x, worldPoint.y);
272
273 // create a color based on the Z height of the map
274 //Real waterZ;
275
276 if( localIsUnderwater( MAP_XY_FACTOR *(worldPoint.x - pMap->getBorderSize()), MAP_XY_FACTOR * ( worldPoint.y - pMap->getBorderSize())) )
277 {
278 const Int waterSamplesAway = 1; // how many "tiles" from the center tile we will sample away
279 // to average a color for the tile color
280
281 sampleColor.red = sampleColor.green = sampleColor.blue = 0.0f;
282 samples = 0;
283
284 for( j = y - waterSamplesAway; j <= y + waterSamplesAway; j++ )
285 {
286
287 if( j >= 0 && j < MAP_PREVIEW_HEIGHT )
288 {
289
290 for( i = x - waterSamplesAway; i <= x + waterSamplesAway; i++ )
291 {
292
293 if( i >= 0 && i < MAP_PREVIEW_WIDTH )
294 {
295
296 // the the world point we are concerned with
297 radarPoint.x = i;
298 radarPoint.y = j;
299 mapPreviewToWorld( &radarPoint, &worldPoint );
300
301 // get Z at this sample height
302 Real underwaterZ = pMap->getHeight(worldPoint.x, worldPoint.y);
303
304 // get color for this Z and add to our sample color
305 if( localIsUnderwater( MAP_XY_FACTOR *(worldPoint.x - pMap->getBorderSize()), MAP_XY_FACTOR * ( worldPoint.y - pMap->getBorderSize())))
306 {
307
308 // this is our "color" for water
309 color = waterColor;
310
311 // interpolate the water color for height in the water table
312 interpolateColorForHeight( &color, underwaterZ, pMap->getMaxHeightValue(),getTerrainAverageZ, pMap->getMinHeightValue() );
313
314 // add color to our samples
315 sampleColor.red += color.red;
316 sampleColor.green += color.green;
317 sampleColor.blue += color.blue;
318 samples++;
319
320 } // end if
321
322 } // end if
323
324 } // end for i
325
326 } // end if
327
328 } // end for j
329
330 // prevent divide by zeros
331 if( samples == 0 )
332 samples = 1;
333
334 // set the color to an average of the colors read
335 color.red = sampleColor.red / (Real)samples;
336 color.green = sampleColor.green / (Real)samples;
337 color.blue = sampleColor.blue / (Real)samples;
338
339 } // end if
340 else // regular terrain ...
341 {
342 const Int samplesAway = 1; // how many "tiles" from the center tile we will sample away
343 // to average a color for the tile color
344
345 sampleColor.red = sampleColor.green = sampleColor.blue = 0.0f;
346 samples = 0;
347
348 for( j = y - samplesAway; j <= y + samplesAway; j++ )
349 {
350
351 if( j >= 0 && j < MAP_PREVIEW_HEIGHT )
352 {
353
354 for( i = x - samplesAway; i <= x + samplesAway; i++ )
355 {
356
357 if( i >= 0 && i < MAP_PREVIEW_WIDTH )
358 {
359
360 // the the world point we are concerned with
361 radarPoint.x = i;
362 radarPoint.y = j;
363// radarPoint.x = x;
364// radarPoint.y = y;
365 mapPreviewToWorld( &radarPoint, &worldPoint );
366
367 // get the color at this point
368 pMap->getTerrainColorAt( MAP_XY_FACTOR *(worldPoint.x - pMap->getBorderSize()) ,MAP_XY_FACTOR * ( worldPoint.y - pMap->getBorderSize()), &color );
369
370 // interpolate the color for height
371 interpolateColorForHeight( &color, z,
372 maxHeight, getTerrainAverageZ, minHeight);
373 //pMap->getMaxHeightValue(),getTerrainAverageZ, pMap->getMinHeightValue() );
374
375 // add color to our sample
376 sampleColor.red += color.red;
377 sampleColor.green += color.green;
378 sampleColor.blue += color.blue;
379 samples++;
380
381 } // end if
382
383 } // end for i
384
385 } // end if
386
387 } // end for j
388
389 // prevent divide by zeros
390 if( samples == 0 )
391 samples = 1;
392
393 // set the color to an average of the colors read
394 color.red = sampleColor.red / (Real)samples;
395 color.green = sampleColor.green / (Real)samples;
396 color.blue = sampleColor.blue / (Real)samples;
397
398 } // end else
399
400 //
401 // draw the pixel for the terrain at this point, note that because of the orientation
402 // of our world we draw it with positive y in the "up" direction
403 //
404 // FYI: I tried making this faster by pulling out all the code inside DrawPixel
405 // and locking only once ... but it made absolutely *no* performance difference,
406 // the sampling and interpolation algorithm for generating pretty looking terrain
407 // and water for the radar is just, well, expensive.
408 //
409 m_pixelBuffer[y][x]= 255 | (REAL_TO_INT(color.red *255) << 8) | (REAL_TO_INT(color.green *255) << 16) | REAL_TO_INT(color.blue * 255)<< 24;
410
411 } // end for x
412
413 } // end for y
414 {
415 Targa tga;
416 tga.Header.Width = MAP_PREVIEW_WIDTH;
417 tga.Header.Height = MAP_PREVIEW_HEIGHT;
418 tga.Header.PixelDepth = 32;
419 tga.Header.ImageType = TGA_TRUECOLOR;
420 tga.SetImage((char *)m_pixelBuffer);
421 tga.Save(tgaName,TGAF_IMAGE, FALSE);
422 }
423
424} // end buildTerrainTexture
