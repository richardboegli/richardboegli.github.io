129inline int MPEXPORT XMP_Digits_To_Bits(int digits)
130{
131 return(digits << LOG_UNITSIZE);
132}
133
134
135inline int MPEXPORT XMP_Bits_To_Digits(int bits)
136{
137 return ((bits + (UNITSIZE-1)) / UNITSIZE);
138}
139
140
141inline digit MPEXPORT XMP_Bits_To_Mask(int bits)
142{
143 if (!bits) return(0);
144 return(1 << ((bits-1) % UNITSIZE));
145}
146
147
148inline bool MPEXPORT XMP_Is_Negative(const digit * r, int precision)
149{
150 return((signeddigit) *(r + (precision-1)) < 0);
151}
152
153
154inline void MPEXPORT XMP_Set_Bit(digit * r, unsigned bit)
155{
156 r[bit >> LOG_UNITSIZE] |= ((digit)1 << (bit & (UNITSIZE-1)));
157}
158
159inline bool MPEXPORT XMP_Test_Bit(const digit * r, unsigned bit)
160{
161 return ((r[bit >> LOG_UNITSIZE] & ((digit)1 << (bit & (UNITSIZE-1)))) != 0);
162}
