81void TunnelContain::removeFromContain( Object *obj, Bool exposeStealthUnits )
82{
83
84 // sanity
85 if( obj == NULL )
86 return;
87
88 // trigger an onRemoving event for 'm_object' no longer containing 'itemToRemove->m_object'
89 if( getObject()->getContain() )
90 {
91 getObject()->getContain()->onRemoving( obj );
92 }
93
94 // trigger an onRemovedFrom event for 'remove'
95 obj->onRemovedFrom( getObject() );
96
97 //
98 // we can only remove this object from the contains list of this module if
99 // it is actually contained by this module
100 //
101 Player *owningPlayer = getObject()->getControllingPlayer();
102 if( owningPlayer == NULL )
103 return; //game tear down. We do the onRemove* stuff first because this is allowed to fail but that still needs to be done
104
105 if( ! owningPlayer->getTunnelSystem()->isInContainer( obj ) )
106 {
107 return;
108 }
109
110 owningPlayer->getTunnelSystem()->removeFromContain( obj, exposeStealthUnits );
111
112}

119void TunnelContain::harmAndForceExitAllContained( DamageInfo *info )
120{
121 Player *owningPlayer = getObject()->getControllingPlayer();
122 const ContainedItemsList *fullList = owningPlayer->getTunnelSystem()->getContainedItemsList();
123
124 Object *obj;
125 ContainedItemsList::const_iterator it;
126
127 //Kris: Patch 1.01 -- November 6, 2003
128 //No longer advances the iterator and saves it. The iterator is fetched from the beginning after
129 //each loop. This is to prevent a crash where dropping a bunker buster on a tunnel network containing
130 //multiple units (if they have the suicide bomb upgrade - demo general). In this case, multiple bunker
131 //busters would hit the tunnel network in close succession. Missile #1 would iterate the list, killing
132 //infantry #1. Infantry #1 would explode and destroy Missile #2. Missile #2 would start iterating the
133 //same list, killing the remaining units. When Missile #1 picked up and continued processing the list
134 //it would crash because it's iterator was deleted from under it.
135 it = (*fullList).begin();
136 while( it != (*fullList).end() )
137 {
138 obj = *it;
139 removeFromContain( obj, true );
140 obj->attemptDamage( info );
141 it = (*fullList).begin();
142 } // end while
143
144} // end removeAllContained

235void TunnelContain::onSelling()
236{
237 // A TunnelContain tells everyone to leave if this is the last tunnel
238 Player *owningPlayer = getObject()->getControllingPlayer();
239 if( owningPlayer == NULL )
240 return;
241 TunnelTracker *tunnelTracker = owningPlayer->getTunnelSystem();
242 if( tunnelTracker == NULL )
243 return;
244
245 // We are the last tunnel, so kick everyone out. This makes tunnels act like Palace and Bunker
246 // rather than killing the occupants as if the last tunnel died.
247 if( tunnelTracker->friend_getTunnelCount() == 1 )
248 removeAllContained(FALSE);// Can't be order to exit, as I have no time to organize their exits.
249 // If they don't go right now, I will delete them in a moment
250
251 // Unregister after the kick out, or else the unregistering will activate a cavein-kill.
252 // We need to do this in case someone sells their last two tunnels at the same time.
253 if( m_isCurrentlyRegistered )
254 {
255 tunnelTracker->onTunnelDestroyed( getObject() );
256 m_isCurrentlyRegistered = FALSE;
257 }
258}

297void TunnelContain::scatterToNearbyPosition(Object* obj)
298{
299 Object *theContainer = getObject();
300
301 //
302 // for now we will just set the position of the object that is being removed from us
303 // at a random angle away from our center out some distance
304 //
305
306 //
307 // pick an angle that is in the view of the current camera position so that
308 // the thing will come out "toward" the player and they can see it
309 // NOPE, can't do that ... all players screen angles will be different, unless
310 // we maintain the angle of each players screen in the player structure or something
311 //
312 Real angle = GameLogicRandomValueReal( 0.0f, 2.0f * PI );
313// angle = TheTacticalView->getAngle();
314// angle -= GameLogicRandomValueReal( PI / 3.0f, 2.0f * (PI / 3.0F) );
315
316 Real minRadius = theContainer->getGeometryInfo().getBoundingCircleRadius();
317 Real maxRadius = minRadius + minRadius / 2.0f;
318 const Coord3D *containerPos = theContainer->getPosition();
319 Real dist = GameLogicRandomValueReal( minRadius, maxRadius );
320
321 Coord3D pos;
322 pos.x = dist * Cos( angle ) + containerPos->x;
323 pos.y = dist * Sin( angle ) + containerPos->y;
324 pos.z = TheTerrainLogic->getGroundHeight( pos.x, pos.y );
325
326 // set orientation
327 obj->setOrientation( angle );
328
329 AIUpdateInterface *ai = obj->getAIUpdateInterface();
330 if( ai )
331 {
332 // set position of the object at center of building and move them toward pos
333 obj->setPosition( theContainer->getPosition() );
334 ai->ignoreObstacle(theContainer);
335 ai->aiMoveToPosition( &pos, CMD_FROM_AI );
336
337 } // end if
338 else
339 {
340
341 // no ai, just set position at the target pos
342 obj->setPosition( &pos );
343
344 } // end else
345}

350void TunnelContain::onDie( const DamageInfo * damageInfo )
351{
352 // override the onDie we inherit from OpenContain. no super call.
353 if (!getTunnelContainModuleData()->m_dieMuxData.isDieApplicable(getObject(), damageInfo))
354 return;
355
356 if( !m_isCurrentlyRegistered )
357 return;//it isn't registered as a tunnel
358
359 Player *owningPlayer = getObject()->getControllingPlayer();
360 if( owningPlayer == NULL )
361 return;
362 TunnelTracker *tunnelTracker = owningPlayer->getTunnelSystem();
363 if( tunnelTracker == NULL )
364 return;
365
366 tunnelTracker->onTunnelDestroyed( getObject() );
367 m_isCurrentlyRegistered = FALSE;
368}

440void TunnelContain::onCapture( Player *oldOwner, Player *newOwner )
441{
442 if( m_isCurrentlyRegistered )
443 {
444 TunnelTracker *oldTunnelTracker = oldOwner->getTunnelSystem();
445 if( oldTunnelTracker )
446 {
447 DEBUG_ASSERTCRASH( oldTunnelTracker->getContainCount() == 0, ("You shouldn't force a capture of a Tunnel with people in it. Future ExitFromContainer scripts will fail."));
448 oldTunnelTracker->onTunnelDestroyed(getObject());
449 }
450
451 TunnelTracker *newTunnelTracker = newOwner->getTunnelSystem();
452 if( newTunnelTracker )
453 {
454 newTunnelTracker->onTunnelCreated(getObject());
455 }
456 }
457
458 // extend base class
459 OpenContain::onCapture( oldOwner, newOwner );
460}

465UpdateSleepTime TunnelContain::update( void )
466{
467 // extending functionality to heal the units within the tunnel system
468 OpenContain::update();
469 const TunnelContainModuleData *modData = getTunnelContainModuleData();
470
471 Object *obj = getObject();
472 Player *controllingPlayer = NULL;
473 if (obj)
474 {
475 controllingPlayer = obj->getControllingPlayer();
476 }
477 if (controllingPlayer)
478 {
479 TunnelTracker *tunnelSystem = controllingPlayer->getTunnelSystem();
480 if (tunnelSystem)
481 {
482 tunnelSystem->healObjects(modData->m_framesForFullHeal);
483 }
484
485 // check for attacked.
486 BodyModuleInterface *body = obj->getBodyModule();
487 if (body) {
488 const DamageInfo *info = body->getLastDamageInfo();
489 if (info) {
490 if (body->getLastDamageTimestamp() + LOGICFRAMES_PER_SECOND > TheGameLogic->getFrame()) {
491 // winner.
492 ObjectID attackerID = info->in.m_sourceID;
493 Object *attacker = TheGameLogic->findObjectByID(attackerID);
494 if( attacker )
495 {
496 if (obj->getRelationship(attacker) == ENEMIES) {
497 tunnelSystem->updateNemesis(attacker);
498 }
499 }
500 }
501 }
502 }
503 }
504 return UPDATE_SLEEP_NONE;
505
506}
