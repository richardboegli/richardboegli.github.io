69// sliderPropertiesCallback ===================================================
71//=============================================================================
72static LRESULT CALLBACK sliderPropertiesCallback( HWND hWndDialog,
73 UINT message,
74 WPARAM wParam,
75 LPARAM lParam )
76{
77 Int returnCode;
78
79 //
80 // handle any common messages between all property dialogs cause they
81 // are designed to have controls doing the same functionality
82 // and names
83 //
84 if( HandleCommonDialogMessages( hWndDialog, message,
85 wParam, lParam, &returnCode ) == TRUE )
86 return returnCode;
87
88 switch( message )
89 {
90
91 // ------------------------------------------------------------------------
92 case WM_COMMAND:
93 {
94// Int notifyCode = HIWORD( wParam ); // notification code
95 Int controlID = LOWORD( wParam ); // control ID
96// HWND hWndControl = (HWND)lParam; // control window handle
97
98 switch( controlID )
99 {
100
101 // --------------------------------------------------------------------
102 case BUTTON_SUBCONTROL_COLOR:
103 {
104 ImageAndColorInfo *info;
105 GameWindow *window = TheEditor->getPropertyTarget();
106 Bool vert = FALSE;
107
108 if( window )
109 vert = BitTest( window->winGetStyle(), GWS_VERT_SLIDER );
110
111 //
112 // using the current colors in the base of the slider, assign a
113 // reasonable color scheme to all the sub control components
114 //
115 if( vert )
116 {
117
118 info = GetStateInfo( VSLIDER_ENABLED_TOP );
119 StoreColor( VSLIDER_THUMB_ENABLED, info->color, info->borderColor );
120 StoreColor( VSLIDER_THUMB_ENABLED_PUSHED, info->borderColor, info->color );
121
122 info = GetStateInfo( VSLIDER_DISABLED_TOP );
123 StoreColor( VSLIDER_THUMB_DISABLED, info->color, info->borderColor );
124 StoreColor( VSLIDER_THUMB_DISABLED_PUSHED, info->borderColor, info->color );
125
126 info = GetStateInfo( VSLIDER_HILITE_TOP );
127 StoreColor( VSLIDER_THUMB_HILITE, info->color, info->borderColor );
128 StoreColor( VSLIDER_THUMB_HILITE_PUSHED, info->borderColor, info->color );
129
130 } // end if, vertical slider
131 else
132 {
133
134 info = GetStateInfo( HSLIDER_ENABLED_LEFT );
135 StoreColor( HSLIDER_THUMB_ENABLED, info->color, info->borderColor );
136 StoreColor( HSLIDER_THUMB_ENABLED_PUSHED, info->borderColor, info->color );
137
138 info = GetStateInfo( HSLIDER_DISABLED_LEFT );
139 StoreColor( HSLIDER_THUMB_DISABLED, info->color, info->borderColor );
140 StoreColor( HSLIDER_THUMB_DISABLED_PUSHED, info->borderColor, info->color );
141
142 info = GetStateInfo( HSLIDER_HILITE_LEFT );
143 StoreColor( HSLIDER_THUMB_HILITE, info->color, info->borderColor );
144 StoreColor( HSLIDER_THUMB_HILITE_PUSHED, info->borderColor, info->color );
145
146 } // end else horizontal slider
147
148 break;
149
150 } // end case subcontrol color
151
152 // --------------------------------------------------------------------
153 case IDOK:
154 {
155 GameWindow *window = TheEditor->getPropertyTarget();
156
157 // sanity
158 if( window )
159 {
160 ImageAndColorInfo *info;
161 Bool vert = BitTest( window->winGetStyle(), GWS_VERT_SLIDER );
162
163 // save the common properties
164 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
165 break;
166
167 // save the image and color data
168 // ----------------------------------------------------------------
169 if( vert )
170 {
171 info = GetStateInfo( VSLIDER_ENABLED_TOP );
172 GadgetSliderSetEnabledImageTop( window, info->image );
173 GadgetSliderSetEnabledColor( window, info->color );
174 GadgetSliderSetEnabledBorderColor( window, info->borderColor );
175
176 info = GetStateInfo( VSLIDER_ENABLED_BOTTOM );
177 GadgetSliderSetEnabledImageBottom( window, info->image );
178 } // end if
179 else
180 {
181 info = GetStateInfo( HSLIDER_ENABLED_LEFT );
182 GadgetSliderSetEnabledImageLeft( window, info->image );
183 GadgetSliderSetEnabledColor( window, info->color );
184 GadgetSliderSetEnabledBorderColor( window, info->borderColor );
185
186 info = GetStateInfo( HSLIDER_ENABLED_RIGHT );
187 GadgetSliderSetEnabledImageRight( window, info->image );
188 } // end else
189
190 info = GetStateInfo( vert ? VSLIDER_ENABLED_CENTER : HSLIDER_ENABLED_CENTER );
191 GadgetSliderSetEnabledImageCenter( window, info->image );
192
193 info = GetStateInfo( vert ? VSLIDER_ENABLED_SMALL_CENTER : HSLIDER_ENABLED_SMALL_CENTER );
194 GadgetSliderSetEnabledImageSmallCenter( window, info->image );
195
196 // ----------------------------------------------------------------
197 if( vert )
198 {
199 info = GetStateInfo( VSLIDER_DISABLED_TOP );
200 GadgetSliderSetDisabledImageTop( window, info->image );
201 GadgetSliderSetDisabledColor( window, info->color );
202 GadgetSliderSetDisabledBorderColor( window, info->borderColor );
203
204 info = GetStateInfo( VSLIDER_DISABLED_BOTTOM );
205 GadgetSliderSetDisabledImageBottom( window, info->image );
206 } // end if
207 else
208 {
209 info = GetStateInfo( HSLIDER_DISABLED_LEFT );
210 GadgetSliderSetDisabledImageLeft( window, info->image );
211 GadgetSliderSetDisabledColor( window, info->color );
212 GadgetSliderSetDisabledBorderColor( window, info->borderColor );
213
214 info = GetStateInfo( HSLIDER_DISABLED_RIGHT );
215 GadgetSliderSetDisabledImageRight( window, info->image );
216 } // end else
217
218 info = GetStateInfo( vert ? VSLIDER_DISABLED_CENTER : HSLIDER_DISABLED_CENTER );
219 GadgetSliderSetDisabledImageCenter( window, info->image );
220
221 info = GetStateInfo( vert ? VSLIDER_DISABLED_SMALL_CENTER : HSLIDER_DISABLED_SMALL_CENTER );
222 GadgetSliderSetDisabledImageSmallCenter( window, info->image );
223
224 // ----------------------------------------------------------------
225 if( vert )
226 {
227 info = GetStateInfo( VSLIDER_HILITE_TOP );
228 GadgetSliderSetHiliteImageTop( window, info->image );
229 GadgetSliderSetHiliteColor( window, info->color );
230 GadgetSliderSetHiliteBorderColor( window, info->borderColor );
231
232 info = GetStateInfo( VSLIDER_HILITE_BOTTOM );
233 GadgetSliderSetHiliteImageBottom( window, info->image );
234 } // end if
235 else
236 {
237 info = GetStateInfo( HSLIDER_HILITE_LEFT );
238 GadgetSliderSetHiliteImageLeft( window, info->image );
239 GadgetSliderSetHiliteColor( window, info->color );
240 GadgetSliderSetHiliteBorderColor( window, info->borderColor );
241
242 info = GetStateInfo( HSLIDER_HILITE_RIGHT );
243 GadgetSliderSetHiliteImageRight( window, info->image );
244 } // end else
245
246 info = GetStateInfo( vert ? VSLIDER_HILITE_CENTER : HSLIDER_HILITE_CENTER );
247 GadgetSliderSetHiliteImageCenter( window, info->image );
248
249 info = GetStateInfo( vert ? VSLIDER_HILITE_SMALL_CENTER : HSLIDER_HILITE_SMALL_CENTER );
250 GadgetSliderSetHiliteImageSmallCenter( window, info->image );
251
252 // ----------------------------------------------------------------
253 info = GetStateInfo( vert ? VSLIDER_THUMB_ENABLED : HSLIDER_THUMB_ENABLED );
254 GadgetSliderSetEnabledThumbImage( window, info->image );
255 GadgetSliderSetEnabledThumbColor( window, info->color );
256 GadgetSliderSetEnabledThumbBorderColor( window, info->borderColor );
257
258 info = GetStateInfo( vert ? VSLIDER_THUMB_ENABLED_PUSHED : HSLIDER_THUMB_ENABLED_PUSHED );
259 GadgetSliderSetEnabledSelectedThumbImage( window, info->image );
260 GadgetSliderSetEnabledSelectedThumbColor( window, info->color );
261 GadgetSliderSetEnabledSelectedThumbBorderColor( window, info->borderColor );
262
263 // ----------------------------------------------------------------
264 info = GetStateInfo( vert ? VSLIDER_THUMB_DISABLED : HSLIDER_THUMB_DISABLED );
265 GadgetSliderSetDisabledThumbImage( window, info->image );
266 GadgetSliderSetDisabledThumbColor( window, info->color );
267 GadgetSliderSetDisabledThumbBorderColor( window, info->borderColor );
268
269 info = GetStateInfo( vert ? VSLIDER_THUMB_DISABLED_PUSHED : HSLIDER_THUMB_DISABLED_PUSHED );
270 GadgetSliderSetDisabledSelectedThumbImage( window, info->image );
271 GadgetSliderSetDisabledSelectedThumbColor( window, info->color );
272 GadgetSliderSetDisabledSelectedThumbBorderColor( window, info->borderColor );
273
274 // ----------------------------------------------------------------
275 info = GetStateInfo( vert ? VSLIDER_THUMB_HILITE : HSLIDER_THUMB_HILITE );
276 GadgetSliderSetHiliteThumbImage( window, info->image );
277 GadgetSliderSetHiliteThumbColor( window, info->color );
278 GadgetSliderSetHiliteThumbBorderColor( window, info->borderColor );
279
280 info = GetStateInfo( vert ? VSLIDER_THUMB_HILITE_PUSHED : HSLIDER_THUMB_HILITE_PUSHED );
281 GadgetSliderSetHiliteSelectedThumbImage( window, info->image );
282 GadgetSliderSetHiliteSelectedThumbColor( window, info->color );
283 GadgetSliderSetHiliteSelectedThumbBorderColor( window, info->borderColor );
284
285 // slider data
286 SliderData *sliderData = (SliderData *)window->winGetUserData();
287
288 sliderData->minVal = GetDlgItemInt( hWndDialog, EDIT_SLIDER_MIN, NULL, FALSE );
289 sliderData->maxVal = GetDlgItemInt( hWndDialog, EDIT_SLIDER_MAX, NULL, FALSE );
290
291 // sanity
292 if( sliderData->minVal > sliderData->maxVal )
293 {
294 Int temp = sliderData->minVal;
295 sliderData->minVal = sliderData->maxVal;
296 sliderData->maxVal = temp;
297
298 MessageBox( NULL, "Slider min greated than max, the values were swapped",
299 "Warning", MB_OK | MB_ICONINFORMATION );
300
301 } // end if
302
303 } // end if
304
305 DestroyWindow( hWndDialog );
306 break;
307
308 } // end OK
309
310 // --------------------------------------------------------------------
311 case IDCANCEL:
312 {
313
314 DestroyWindow( hWndDialog );
315 break;
316
317 } // end cancel
318
319 } // end switch( LOWORD( wParam ) )
320
321 return 0;
322
323 } // end of WM_COMMAND
324
325 // ------------------------------------------------------------------------
326 case WM_CLOSE:
327 {
328
329 DestroyWindow( hWndDialog );
330 return 0;
331
332 } // end close
333
334 // ------------------------------------------------------------------------
335 default:
336 return 0;
337
338 } // end of switch
339
340} // end sliderPropertiesCallback

350HWND InitSliderPropertiesDialog( GameWindow *window )
351{
352 HWND dialog;
353
354 // create the dialog box
355 dialog = CreateDialog( TheEditor->getInstance(),
356 (LPCTSTR)SLIDER_PROPERTIES_DIALOG,

358 (DLGPROC)sliderPropertiesCallback );
359 if( dialog == NULL )
360 return NULL;
361
362 // do the common initialization
363 CommonDialogInitialize( window, dialog );
364
365 //
366 // init values
367 //
368 const Image *image;
369 Color color, borderColor;
370 Bool vert = BitTest( window->winGetStyle(), GWS_VERT_SLIDER );
371
372 // --------------------------------------------------------------------------
