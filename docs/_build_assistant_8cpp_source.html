176static const Real FRAMES_TO_ALLOW_SCAFFOLD = LOGICFRAMES_PER_SECOND * 1.5f;
177static const Real TOTAL_FRAMES_TO_SELL_OBJECT = LOGICFRAMES_PER_SECOND * 3.0f;
178//-------------------------------------------------------------------------------------------------
180//-------------------------------------------------------------------------------------------------
181void BuildAssistant::update( void )
182{
183 ObjectSellInfo *sellInfo;
184 Object *obj;
185
186 // remove any objects from the sell list who's "sell time" is up
187 ObjectSellListIterator it, thisIterator;
188 for( it = m_sellList.begin(); it != m_sellList.end(); /*empty*/ )
189 {
190
191 // get this object info
192 sellInfo = (*it);
193
194 // increment the iterator to the next as we may remove it
195 thisIterator = it;
196 ++it;
197
198 // find the object
199 obj = TheGameLogic->findObjectByID( sellInfo->m_id );
200
201 //
202 // if object is not found, remove it from the list immediately ... this is valid as
203 // the object maybe was destroyed by other means during the sell process
204 //
205 if( obj == NULL )
206 {
207
208 sellInfo->deleteInstance();
209 m_sellList.erase( thisIterator );
210 continue;
211
212 } // end if
213
214 // decrement the construction percent
215 if( TheGameLogic->getFrame() - sellInfo->m_sellFrame >= FRAMES_TO_ALLOW_SCAFFOLD )
216 {
217 Real previousConstructionPercent = obj->getConstructionPercent();
218
219 // do the constructoin
220 obj->setConstructionPercent( previousConstructionPercent - (100.0f / TOTAL_FRAMES_TO_SELL_OBJECT) );
221
222 //
223 // did we cross the threshold from positive to negative ... if so, we will clear
224 // visual construction indicators
225 //
226 if( previousConstructionPercent > 0.0f && obj->getConstructionPercent() <= 0.0f )
227 {
228
229 obj->clearAndSetModelConditionFlags( MAKE_MODELCONDITION_MASK2( MODELCONDITION_PARTIALLY_CONSTRUCTED,
230 MODELCONDITION_ACTIVELY_BEING_CONSTRUCTED ),
231 MAKE_MODELCONDITION_MASK( MODELCONDITION_SOLD ) );
232
233 //
234 // set the animation durations so that the regular build up loop animations can be
235 // done a bit faster for selling
236 //
237 Drawable *draw = obj->getDrawable();
238 if( draw )
239 draw->setAnimationLoopDuration( TOTAL_FRAMES_TO_SELL_OBJECT / 2 );
240
241 } // end if
242
243 } // end if
244
245 //
246 // after we've reached zero ... the object has "sunk" back down into the ground ... but
247 // we need to keep the object around for a little while longer so that the scaffold
248 // animation can finish up and sink itself back into the ground
249 //
250 if( obj->getConstructionPercent() <= -50.0f )
251 {
252 // refund the money to the controlling player
253 Player *player = obj->getControllingPlayer();
254 if( player )
255 {
256 UnsignedInt sellValue;
257 // 0 is the init, and means you have no override. We would be marked unsellable if someone wanted us to not sell
258 if( obj->getTemplate()->getRefundValue() != 0 )
259 sellValue = obj->getTemplate()->getRefundValue();
260 else
261 sellValue = REAL_TO_UNSIGNEDINT( obj->getTemplate()->calcCostToBuild( player ) *
262 TheGlobalData->m_sellPercentage );
263
264 player->getMoney()->deposit( sellValue );
265 // this money shouldn't be scored since it wasn't really "earned."
266// player->getScoreKeeper()->addMoneyEarned( sellValue );
267
268 } // end if
269
270 // cancel any of the production items and refund to the controlling player
271 ProductionUpdateInterface *pui = obj->getProductionUpdateInterface();
272 if( pui )
273 pui->cancelAndRefundAllProduction();
274
275 // for debugging
276 // UnicodeString msg;
277 // msg.format( L"'%S' has been sold\n", obj->getTemplate()->getName().str() );
278 // TheInGameUI->message( msg );
279
280 // destroy the object
281 TheGameLogic->destroyObject( obj );
282
283 // remove this object from the sell list
284 sellInfo->deleteInstance();
285 m_sellList.erase( thisIterator );
286
287 } // end if
288
289 } // end for
290
291} // end update

295void BuildAssistant::xferTheSellList( Xfer *xfer )
296{
297 ObjectSellInfo *sellInfo;
298
299 Int count=0;
300 ObjectSellListIterator it, thisIterator;
301 for( it = m_sellList.begin(); it != m_sellList.end(); ++it ) {
302 count++;
303 }
304 xfer->xferInt(&count);
305
306 if (xfer->getXferMode() == XFER_LOAD) {
307 m_sellList.clear();
308 Int i;
309 for (i=0; i<count; i++) {
310 // add this object to the list of objects being sold
311 sellInfo = newInstance(ObjectSellInfo);
312 xfer->xferObjectID(&sellInfo->m_id);
313 xfer->xferUnsignedInt(&sellInfo->m_sellFrame);
314 m_sellList.push_back( sellInfo );
315
316 }
317 } else {
318 for( it = m_sellList.begin(); it != m_sellList.end(); ++it ) {
319 // get this object info
320 sellInfo = (*it);
321 xfer->xferObjectID(&sellInfo->m_id);
322 xfer->xferUnsignedInt(&sellInfo->m_sellFrame);
323 count--;
324 }
325 DEBUG_ASSERTCRASH(count==0, ("Inconsistent list size counts."));
326 }
327
328} // end xferTheSellList

333Object *BuildAssistant::buildObjectNow( Object *constructorObject, const ThingTemplate *what,
334 const Coord3D *pos, Real angle, Player *owningPlayer )
335{
336
337 // sanity
338 if( what == NULL || pos == NULL )
339 return NULL;
340
341 if( owningPlayer == NULL )
342 return NULL;// Invalid pointer. Won't happen.
343
344 // sanity
345 if( constructorObject )
346 {
347 DEBUG_ASSERTCRASH( constructorObject->getControllingPlayer() == owningPlayer,
348 ("buildObjectNow: Constructor object player is not the same as the controlling player passed in\n") );
349
350 } // end if
351
352 // Need to validate that we can make this in case someone fakes their CommandSet
353 // A NULL constructor Object means a script built building so let it slide.
354 if( (constructorObject != NULL) && !isPossibleToMakeUnit(constructorObject, what) )
355 return NULL;
356
357 // clear out any objects from the building area that are "auto-clearable" when building
358 clearRemovableForConstruction( what, pos, angle );
359
360 if ( moveObjectsForConstruction( what, pos, angle, owningPlayer ) == FALSE )
361 {
362 // totally bogus. We tried to move our units out of the way, but they wouldn't.
363 // Chode-boys.
364 if (owningPlayer->getPlayerType()==PLAYER_HUMAN) {
365 return NULL; // ai gets to cheat. jba.
366 }
367 }
368
369 // do the build
370 if( isDozer( constructorObject ) && isLineBuildTemplate( what ) == FALSE )
371 {
372 AIUpdateInterface *ai = constructorObject->getAIUpdateInterface();
373
374 if( ai )
375 {
376 ai->aiIdle(CMD_FROM_AI); // stop any current behavior.
377 return ai->construct( what, pos, angle, owningPlayer, FALSE );
378 }
379 return NULL;
380
381 } // end else if
382 else
383 {
384
385 // create the new object. We need to construct it with UnderConstruction set, since we are going to insta build it,
386 // but we don't want to send double construction type events like power creation. onStructureConstructionComplete
387 // is called below, so we need to simulate the proper object creation flow from the start. Be like Dozer.
388 ObjectStatusMaskType startingStatus;
389 if( what->isKindOf( KINDOF_STRUCTURE ) )
390 startingStatus.set( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNDER_CONSTRUCTION ) );
391
392 Object *obj = TheThingFactory->newObject( what, owningPlayer->getDefaultTeam(), startingStatus );
393 obj->setProducer(constructorObject);
394
395 // place on terrain surface
396 Coord3D groundPos;
397 groundPos.x = pos->x;
398 groundPos.y = pos->y;
399 groundPos.z = TheTerrainLogic->getGroundHeight( groundPos.x, groundPos.y );
400 obj->setPosition( &groundPos );
401
402 obj->setOrientation( angle );
403
404 TheAI->pathfinder()->addObjectToPathfindMap( obj );
405
406 // notify the player that this thing has come into existence
407 if( obj->isKindOf( KINDOF_STRUCTURE ) )
408 {
409
410 //
411 // this is a cheat since this method is currently used by the solo AI to make
412 // buildings out of thin air
413 //
414 owningPlayer->onStructureCreated( constructorObject, obj );
415 owningPlayer->onStructureConstructionComplete( constructorObject, obj, FALSE );
416
417 } // end if
418 else
419 {
420 owningPlayer->onUnitCreated( constructorObject, obj );
421
422 //Call the voice created for the 1st object -- because it's possible to create multiple objects like redguards!
423 AudioEventRTS sound = *obj->getTemplate()->getPerUnitSound( "VoiceCreated" );
424 sound.setObjectID( obj->getID() );
425 TheAudio->addAudioEvent( &sound );
426 }
427
428 // Now onCreates were called at the constructor. This magically created
429 // thing needs to be considered as Built for Game specific stuff.
430 for (BehaviorModule** m = obj->getBehaviorModules(); *m; ++m)
431 {
432 CreateModuleInterface* create = (*m)->getCreate();
433 if (!create)
434 continue;
435
436 create->onBuildComplete();
437 }
438
439 // Creation is another valid and essential time to call this. This building now Looks.
440 obj->handlePartitionCellMaintenance();
441
442 return obj;
443
444 } // end else
445
446 return NULL;
447
448} // end buildObjectNow

454void BuildAssistant::buildObjectLineNow( Object *constructorObject, const ThingTemplate *what,
455 const Coord3D *start, const Coord3D *end, Real angle,
456 Player *owningPlayer )
457{
458 TileBuildInfo *tileBuildInfo;
459
460 // sanity
461 if( what == NULL || start == NULL || end == NULL )
462 return;
463
464 // how big are each of our objects
465 Real objectSize = what->getTemplateGeometryInfo().getMajorRadius() * 2.0f;
466
467 // what is our max tiling length we can make
468 Int maxObjects = TheGlobalData->m_maxLineBuildObjects;
469
470 // build an array of locations that we want to build from start to end
471 tileBuildInfo = buildTiledLocations( what, angle, start, end,
472 objectSize, maxObjects, constructorObject );
473
474 // create an object at each position
475 for( Int i = 0; i < tileBuildInfo->tilesUsed; i++ )
476 buildObjectNow( constructorObject, what, &tileBuildInfo->positions[ i ], angle, owningPlayer );
477
478} // end buildObjectLineNow

486{
487 Region3D mapRegion;

496static void checkSampleBuildLocation( const Coord3D *samplePoint, void *userData )
497{
498 TerrainType *terrain;
499 SampleBuildData *sampleData = (SampleBuildData *)userData;
500
501 // get the terrain tile here
502 terrain = TheTerrainVisual->getTerrainTile( samplePoint->x, samplePoint->y );
503 if( terrain )
504 {
505
506 // check for the restricts building flag
507 if( terrain->getRestrictConstruction() )
508 sampleData->terrainRestricted = TRUE;
509
510 } // end if
511
512 Int cellX = REAL_TO_INT_FLOOR( samplePoint->x / PATHFIND_CELL_SIZE );
513 Int cellY = REAL_TO_INT_FLOOR( samplePoint->y / PATHFIND_CELL_SIZE );
514
515 PathfindCell* cell = TheAI->pathfinder()->getCell( LAYER_GROUND, cellX, cellY );
516 if (!cell) {
517 sampleData->terrainRestricted = TRUE;
518 } else {
519 enum PathfindCell::CellType type = cell->getType();
520 if ( (type == PathfindCell::CELL_WATER) || (type == PathfindCell::CELL_CLIFF) ||
521 (type == PathfindCell::CELL_IMPASSABLE)) {
522 sampleData->terrainRestricted = true;
523 }
524 }
525
526 //
527 // record the highest and lowest Z points from all the samples and do not allow
528 // building when the difference between them is too great
529 //
530 if( samplePoint->z < sampleData->loZ )
531 sampleData->loZ = samplePoint->z;
532 if( samplePoint->z > sampleData->hiZ )
533 sampleData->hiZ = samplePoint->z;
534
535 // too close to edge of map?
536 if (TheGlobalData->m_MinDistFromEdgeOfMapForBuild > 0.0f)
537 {
538 if (samplePoint->x < sampleData->mapRegion.lo.x + TheGlobalData->m_MinDistFromEdgeOfMapForBuild
539 || samplePoint->x > sampleData->mapRegion.hi.x - TheGlobalData->m_MinDistFromEdgeOfMapForBuild
540 || samplePoint->y < sampleData->mapRegion.lo.y + TheGlobalData->m_MinDistFromEdgeOfMapForBuild
541 || samplePoint->y > sampleData->mapRegion.hi.y - TheGlobalData->m_MinDistFromEdgeOfMapForBuild)
542 {
543 sampleData->terrainRestricted = TRUE;
544 }
545 }
546
547} // end checkSampleBuildLocation

554void BuildAssistant::iterateFootprint( const ThingTemplate *build,
555 Real buildOrientation,
556 const Coord3D *worldPos,
557 Real sampleResolution,
558 IterateFootprintFunc func,
559 void *funcUserData )
560{
561
562 // sanity
563 if( build == NULL || worldPos == NULL || func == NULL )
564 return;
565
566 //
567 // create a transform matrix for the object position at the specified angle, we will
568 // iterate the object footprints in "local object extent space" and transform those
569 // points using this matrix into the world coords for the object at the real
570 // location and specified angle
571 //
572 Matrix3D transform;
573 transform.Make_Identity();
574 transform.Adjust_Translation( Vector3( worldPos->x, worldPos->y, worldPos->z ) );
575 transform.Rotate_Z( buildOrientation );
576
577 // get the bounding footprint rectangle for the geometry we're looking at
578 Real halfFootprintHeight,
579 halfFootprintWidth;
580 if( build->getTemplateGeometryInfo().getGeomType() == GEOMETRY_BOX )
581 {
582
583 halfFootprintHeight = build->getTemplateGeometryInfo().getMinorRadius();
584 halfFootprintWidth = build->getTemplateGeometryInfo().getMajorRadius();
585
586 } // end if
587 else if( build->getTemplateGeometryInfo().getGeomType() == GEOMETRY_SPHERE ||
588 build->getTemplateGeometryInfo().getGeomType() == GEOMETRY_CYLINDER )
589 {
590
591 halfFootprintHeight = build->getTemplateGeometryInfo().getBoundingCircleRadius();
592 halfFootprintWidth = build->getTemplateGeometryInfo().getBoundingCircleRadius();
593
594 } // end else if
595 else
596 {
597
598 DEBUG_ASSERTCRASH( 0, ("iterateFootprint: Undefined geometry '%d' for '%s'\n",
599 build->getTemplateGeometryInfo().getGeomType(), build->getName().str()) );
600 return;
601
602 } // end else
603
604 //
605 // start at a corner of the extent ... box geometries have a major radius down
606 // the X axis with the minor radius on the Y axis. Note that we are allowing
607 // the sample process to go an "extra" sample resolution outside of the
608 // real extent here. This is so that we can be sure to sample all of the
609 // footprint area, if our sample point is outside of the actual extent footprint
610 // area because of this we snap it back to sample on the exact edge
611 //
612 Real x, y;
613 Vector3 v;
614 for( y = -halfFootprintHeight;
615 y < halfFootprintHeight + sampleResolution;
616 y += sampleResolution )
617 {
618
619 // snap it to the actual extent since we can go over by one sample resolution
620 if( y > halfFootprintHeight )
621 y = halfFootprintHeight;
622
623 for( x = -halfFootprintWidth;
624 x < halfFootprintWidth + sampleResolution;
625 x += sampleResolution )
626 {
627
628 // snap it to the actual extent since we can go over by one sample resolution
629 if( x > halfFootprintWidth )
630 x = halfFootprintWidth;
631
632 // transform to world
633 v.Set( x, y, TheTerrainLogic->getGroundHeight( x, y ) );
634 transform.Transform_Vector( transform, v, &v );
635
636 // for circular geometries we must actually be within the circle
637 if( build->getTemplateGeometryInfo().getGeomType() == GEOMETRY_SPHERE ||
638 build->getTemplateGeometryInfo().getGeomType() == GEOMETRY_CYLINDER )
639 {
640 Coord2D vector;
641
642 vector.x = v.X - worldPos->x;
643 vector.y = v.Y - worldPos->y;
644 if( vector.length() > halfFootprintWidth ) // could be height too, radius is all the same for circles
645 continue; // ignore this point
646
647 } // end if
648
649 // call the user callback
650 Coord3D pos;
651 pos.x = v.X;
652 pos.y = v.Y;
653 pos.z = TheTerrainLogic->getGroundHeight( pos.x, pos.y );
654 func( &pos, funcUserData );
655
656 } // end for x
657
658 } // end for y
659
660} // end iterateFootprint

666LegalBuildCode BuildAssistant::isLocationClearOfObjects( const Coord3D *worldPos,
667 const ThingTemplate *build,
668 Real angle,
669 Object *builderObject,
670 UnsignedInt options,
671 Player *thePlayer)
672{
673 ObjectIterator *iter =
674 ThePartitionManager->iteratePotentialCollisions( worldPos,
675 build->getTemplateGeometryInfo(),
676 angle );
677 Object *them;
678 Bool onlyCheckEnemies = (options == NO_ENEMY_OBJECT_OVERLAP);
679 MemoryPoolObjectHolder hold(iter);
680 for( them = iter->first(); them; them = iter->next() )
681 {
682 Bool feedbackWithFailure = TRUE;
683
684 Relationship rel = builderObject ? builderObject->getRelationship( them ) : NEUTRAL;
685
686 //Kris: If the object is stealthed and we can't see it, pretend we can build there.
687 if( BitTest( options, IGNORE_STEALTHED ) )
688 {
689 if( rel != ALLIES )
690 {
691 if( them->testStatus( OBJECT_STATUS_STEALTHED ) && !them->testStatus( OBJECT_STATUS_DETECTED ) && !them->testStatus( OBJECT_STATUS_DISGUISED ) )
692 {
693 if( BitTest( options, FAIL_STEALTHED_WITHOUT_FEEDBACK ) )
694 {
695 feedbackWithFailure = FALSE; //We want to fail now but without feedback
696 }
697 else
698 {
699 continue;
700 }
701 }
702 }
703 }
704
705 //Kris: Patch 1.01 - November 5, 2003
706 //Prevent busy units (black lotus hacking from being moved by trying to place a building -- exploit).
707 if( rel == ALLIES )
708 {
709 if( them->testStatus( OBJECT_STATUS_IS_USING_ABILITY ) || them->getAI() && them->getAI()->isBusy() )
710 {
711 return LBC_OBJECTS_IN_THE_WAY;
712 }
713 }
714
715 // ignore any kind of class of objects that we will "remove" for building
716 if( isRemovableForConstruction( them ) == TRUE )
717 continue;
718
719 // ignore land mines, cluster mines and demo traps, since you can build on them
720 // doing so will damage them during construction, by the way
721 if (them->isKindOf( KINDOF_MINE ))
722 continue;
723
724 // same story for inert things: ie, radiation fields, pings, and projectile streams (!)...
725 if (them->isKindOf(KINDOF_INERT))
726 continue;
727
728 // an immobile object may obstruct our building depending on flags.
729 if( them->isKindOf( KINDOF_IMMOBILE ) )
730 {
731 if (onlyCheckEnemies && builderObject && rel != ENEMIES )
732 {
733 continue;
734 }
735 if( feedbackWithFailure )
736 {
737 TheTerrainVisual->addFactionBib( them, TRUE );
738 return LBC_OBJECTS_IN_THE_WAY;
739 }
740 return LBC_GENERIC_FAILURE;
741 }
742
743 if( them->isDisabled() )
744 {
745 //Kris: If object is disabled, it can't move out of the way, thus you can't build there.
746 if( feedbackWithFailure )
747 {
748 TheTerrainVisual->addFactionBib( them, TRUE );
749 return LBC_OBJECTS_IN_THE_WAY;
750 }
751 return LBC_GENERIC_FAILURE;
752 }
753
754 //
755 // if this is an enemy object of the builder object (and therefore the thing
756 // that will be constructed) we can't build here
757 //
758 if( builderObject && rel == ENEMIES )
759 {
760 if( feedbackWithFailure )
761 {
762 TheTerrainVisual->addFactionBib( them, TRUE );
763 return LBC_OBJECTS_IN_THE_WAY;
764 }
765 return LBC_GENERIC_FAILURE;
766 }
767
768 } // end for, them
769
770 if (onlyCheckEnemies)
771 {
772 return LBC_OK;
773 }
774 // Check for overlapping exit areas.
775
776 Real range = 2*(build->getTemplateGeometryInfo().getMajorRadius()+build->getTemplateGeometryInfo().getMinorRadius());
777
778 PartitionFilterAcceptByKindOf f1(MAKE_KINDOF_MASK(KINDOF_STRUCTURE), KINDOFMASK_NONE);
779 PartitionFilter *filters[] = { &f1, NULL };
780
781 ObjectIterator *iter2 = ThePartitionManager->iterateObjectsInRange(worldPos, range, FROM_BOUNDINGSPHERE_2D, filters);
782 MemoryPoolObjectHolder hold2(iter2);
783
784 Real myFactoryExitWidth = build->getFactoryExitWidth();
785 Real myExtraWidth = build->getFactoryExtraBibWidth();
786
787 if (thePlayer && thePlayer->isSkirmishAIPlayer()) {
788 // Skirmish ai adds a little extra around the edges so it doesn't build itself into a corner.
789 if (myExtraWidth < 3*PATHFIND_CELL_SIZE_F) {
790 myExtraWidth = 3*PATHFIND_CELL_SIZE_F;
791 myFactoryExitWidth -= myExtraWidth;
792 if (myFactoryExitWidth<0) myFactoryExitWidth = 0;
793 }
794 }
795
796 Bool checkMyExit = false;
797 Coord3D myExitPos;
798 GeometryInfo myBounds = build->getTemplateGeometryInfo();
799 myBounds.setMajorRadius(myBounds.getMajorRadius()+myExtraWidth);
800 if (myBounds.getGeomType() != GEOMETRY_BOX) {
801 myBounds.set(GEOMETRY_BOX, false, 40, myBounds.getMajorRadius(), myBounds.getMajorRadius());
802 } else {
803 myBounds.setMinorRadius(myBounds.getMinorRadius()+myExtraWidth);
804 }
805 GeometryInfo myGeom = build->getTemplateGeometryInfo();
806 if (myGeom.getGeomType() != GEOMETRY_BOX) {
807 myGeom.setMinorRadius(myGeom.getMajorRadius());
808 }
809 myGeom.setMajorRadius(myFactoryExitWidth/2.0f);
810 if (myFactoryExitWidth>0) {
811 myExitPos = *worldPos;
812 checkMyExit = true;
813 Real c = (Real)cos(angle);
814 Real s = (Real)sin(angle);
815 Real offset = build->getTemplateGeometryInfo().getMajorRadius() + myFactoryExitWidth/2.0f;
816 myExitPos.x += c*offset;
817 myExitPos.y += s*offset;
818 }
819
820
821 for( them = iter2->first(); them; them = iter2->next() )
822 {
823
824 Relationship rel = builderObject ? builderObject->getRelationship( them ) : NEUTRAL;
825 //Kris: If the building is stealthed and we can't see it, pretend we can build there.
826 if( BitTest( options, IGNORE_STEALTHED ) )
827 {
828 if( rel != ALLIES )
829 {
830 if( them->testStatus( OBJECT_STATUS_STEALTHED ) && !them->testStatus( OBJECT_STATUS_DETECTED ) && !them->testStatus( OBJECT_STATUS_DISGUISED ) )
831 {
832 continue;
833 }
834 }
835 }
836
837 // ignore any kind of class of objects that we will "remove" for building
838 if( isRemovableForConstruction( them ) == TRUE )
839 continue;
840
841 Real themFactoryExitWidth = them->getTemplate()->getFactoryExitWidth();
842 Real hisExtraWidth = them->getTemplate()->getFactoryExtraBibWidth();
843
844 Bool checkHisExit = false;
845 Coord3D hisExitPos;
846 GeometryInfo hisBounds = them->getGeometryInfo();
847 hisBounds.setMajorRadius(hisBounds.getMajorRadius()+hisExtraWidth);
848 if (hisBounds.getGeomType() != GEOMETRY_BOX) {
849 hisBounds.set(GEOMETRY_BOX, false, 40, hisBounds.getMajorRadius(), hisBounds.getMajorRadius());
850 } else {
851 hisBounds.setMinorRadius(hisBounds.getMinorRadius()+myExtraWidth);
852 }
853 GeometryInfo hisGeom = them->getGeometryInfo();
854 hisGeom.setMajorRadius(themFactoryExitWidth/2.0f);
855 if (hisGeom.getGeomType() != GEOMETRY_BOX) {
856 hisGeom.setMinorRadius(them->getGeometryInfo().getMajorRadius());
857 }
858 if (themFactoryExitWidth>0) {
859 hisExitPos = *them->getPosition();
860 checkHisExit = true;
861 Real c = (Real)cos(them->getOrientation());
862 Real s = (Real)sin(them->getOrientation());
863 Real offset = them->getGeometryInfo().getMajorRadius() + themFactoryExitWidth/2.0f;
864 hisExitPos.x += c*offset;
865 hisExitPos.y += s*offset;
866 }
867 if (ThePartitionManager->geomCollidesWithGeom(them->getPosition(), hisBounds, them->getOrientation(),
868 worldPos, myBounds, angle)) {
869 TheTerrainVisual->addFactionBib(them, true);
870 return LBC_OBJECTS_IN_THE_WAY;
871 }
872 if (!checkMyExit && !checkHisExit && !hisExtraWidth && !myExtraWidth)
873 {
874 continue; // neither has extra exit space.
875 }
876
877 // an immobile object will obstruct our building no matter what team it's on
878 if ( them->isKindOf( KINDOF_IMMOBILE ) ) {
879 /* Check for overlap of my exit rectangle to his geom info. */
880 if (checkMyExit && ThePartitionManager->geomCollidesWithGeom(them->getPosition(), hisBounds, them->getOrientation(),
881 &myExitPos, myGeom, angle)) {
882 TheTerrainVisual->addFactionBib(them, true);
883 return LBC_OBJECTS_IN_THE_WAY;
884 }
885 // Check for overlap of his exit rectangle with my geom info
886 if (checkHisExit && ThePartitionManager->geomCollidesWithGeom(&hisExitPos, hisGeom, them->getOrientation(),
887 worldPos, myBounds, angle)) {
888 TheTerrainVisual->addFactionBib(them, true);
889 return LBC_OBJECTS_IN_THE_WAY;
890 }
891 // Check both exit rectangles together.
892 if (checkMyExit&&checkHisExit&&ThePartitionManager->geomCollidesWithGeom(&hisExitPos, hisGeom, them->getOrientation(),
893 &myExitPos, myGeom, angle)) {
894 TheTerrainVisual->addFactionBib(them, true);
895 return LBC_OBJECTS_IN_THE_WAY;
896 }
897 }
898
899 } // end for, them
900 return LBC_OK;

908LegalBuildCode BuildAssistant::isLocationLegalToBuild( const Coord3D *worldPos,
909 const ThingTemplate *build,
910 Real angle,
911 UnsignedInt options,
912 Object *builderObject,
913 Player *player)
914{
915
916 /* You just can't never build off the map, regardless of options. jba. */
917 Region3D mapExtent;
918 TheTerrainLogic->getMaximumPathfindExtent(&mapExtent);
919 if (!mapExtent.isInRegionNoZ(worldPos)) {
920 return LBC_RESTRICTED_TERRAIN;
921 }
922
923 // check shroud level
924 // This should be the first check, since returning other errors for shrouded areas could be used to game the system
925 if( BitTest( options, SHROUD_REVEALED ) )
926 {
927 {
928 Int x, y;
929 ThePartitionManager->worldToCell(worldPos->x, worldPos->y, &x, &y);
930 Int playerIndex = -1;
931 if (builderObject && builderObject->getControllingPlayer())
932 playerIndex = builderObject->getControllingPlayer()->getPlayerIndex();
933 DEBUG_ASSERTCRASH(playerIndex >= 0, ("isLocationLegalToBuild() needs a builderObject with a team to check for shroud"));
934 if( ThePartitionManager->getShroudStatusForPlayer(playerIndex, x, y) != CELLSHROUD_CLEAR )
935 {
936 return LBC_SHROUD;
937 }
938 }
939 }
940
941 //
942 // if NO_OBJECT_OVERLAP is set, we are not allowed to construct 'build' if it would overlap
943 // any immobile objects, or an enemy object. Friendly objects should politely
944 // "move out of the way" when you build something where they're standing
945 //
946 if( BitTest( options, NO_OBJECT_OVERLAP ) )
947 {
948 LegalBuildCode code = isLocationClearOfObjects(worldPos, build, angle, builderObject, options, player);
949 if( code != LBC_OK )
950 {
951 return code;
952 }
953
954 } // end if
955 //
956 // if NO_ENEMY_OBJECT_OVERLAP is set, we are not allowed to construct 'build' if it would overlap
957 // any enemy objects. Friendly objects are ignored.
958 //
959 if( BitTest( options, NO_ENEMY_OBJECT_OVERLAP ) )
960 {
961 LegalBuildCode code = isLocationClearOfObjects(worldPos, build, angle, builderObject, options, player);
962 if( code != LBC_OK )
963 {
964 return code;
965 }
966 } // end if
967
968 if (build->isKindOf(KINDOF_CANNOT_BUILD_NEAR_SUPPLIES) && TheGlobalData->m_SupplyBuildBorder > 0)
969 {
970 // special case for supply centers: can't build too close to supply sources
971 PartitionFilterAcceptByKindOf f1(MAKE_KINDOF_MASK(KINDOF_SUPPLY_SOURCE), KINDOFMASK_NONE);
972 PartitionFilter *filters[] = { &f1, NULL };
973
974 // see if there are any reasonably close by
975 Real range = build->getTemplateGeometryInfo().getBoundingCircleRadius() + TheGlobalData->m_SupplyBuildBorder*2;
976 Object* tooClose = ThePartitionManager->getClosestObject(worldPos, range, FROM_BOUNDINGSPHERE_2D, filters);
977 if (tooClose != NULL)
978 {
979 // yep, see if we would collide with an expanded version
980 GeometryInfo tooCloseGeom = tooClose->getGeometryInfo();
981 tooCloseGeom.expandFootprint(TheGlobalData->m_SupplyBuildBorder);
982 if (ThePartitionManager->geomCollidesWithGeom(
983 worldPos,
984 build->getTemplateGeometryInfo(),
985 angle,
986 tooClose->getPosition(),
987 tooCloseGeom,
988 tooClose->getOrientation()))
989 {
990 TheTerrainVisual->addFactionBib(tooClose, true, TheGlobalData->m_SupplyBuildBorder);
991 return LBC_TOO_CLOSE_TO_SUPPLIES;
992 }
993 }
994 }
995
996 // if clear path is requested check to see if the builder object can get there (unless it's a structure)
997 if( BitTest( options, CLEAR_PATH ) && builderObject && !builderObject->isKindOf( KINDOF_IMMOBILE ) )
998 {
999 AIUpdateInterface *ai = builderObject->getAIUpdateInterface();
1000
1001 //
1002 // if there is no AI interface for this object, it cannot possible pass a clear path
1003 // check since it will never be able to move there
1004 //
1007 //
1008 if( ai == NULL )
1009 return LBC_NO_CLEAR_PATH;
1010
1011 if( ai->isQuickPathAvailable( worldPos ) == FALSE )
1012 return LBC_NO_CLEAR_PATH;
1013
1014 } // end if
1015
1016 // check basic terrain restrctions
1017 if( BitTest( options, TERRAIN_RESTRICTIONS ) )
1018 {
1019
1020 //
1021 // we will take "samples" at this resolution across the footprint of where we are
1022 // going to build the structure
1023 //
1024 Real sampleResolution = MAP_XY_FACTOR;
1025
1026 // get the terrain extents
1027 Region3D terrainExtent;
1028 TheTerrainLogic->getExtent( &terrainExtent );
1029
1030 if (TheTerrainLogic->getLayerForDestination(worldPos) != LAYER_GROUND) {
1031 // we're on a bridge. This is somewhat restricted.
1032 return LBC_RESTRICTED_TERRAIN;
1033 }
1034
1035 //
1036 // check the footprint of where the structure would go to be clear of any non-buildable
1037 // tiles and to make sure there isn't a restricted tile and to make sure it's "flat" enough
1038 //
1039 SampleBuildData sampleData;
1040 TheTerrainLogic->getExtent( &sampleData.mapRegion );
1041 sampleData.terrainRestricted = FALSE;
1042 sampleData.hiZ = terrainExtent.lo.z; // note we set hi point to lowest point
1043 sampleData.loZ = terrainExtent.hi.z; // note we set lo point to highest point
1044
1045 // quick check at triple res.
1046 iterateFootprint( build, angle, worldPos, 3*sampleResolution,
1047 checkSampleBuildLocation, &sampleData );
1048 if( sampleData.terrainRestricted == TRUE )
1049 return LBC_RESTRICTED_TERRAIN;
1050 // check if the height across the whole footprint area is too varied (not flat enough)
1051 if( sampleData.hiZ - sampleData.loZ > TheGlobalData->m_allowedHeightVariationForBuilding )
1052 return LBC_NOT_FLAT_ENOUGH;
1053
1054 // careful check at full res.
1055 iterateFootprint( build, angle, worldPos, sampleResolution,
1056 checkSampleBuildLocation, &sampleData );
1057 if( sampleData.terrainRestricted == TRUE )
1058 return LBC_RESTRICTED_TERRAIN;
1059 // check if the height across the whole footprint area is too varied (not flat enough)
1060 if( sampleData.hiZ - sampleData.loZ > TheGlobalData->m_allowedHeightVariationForBuilding )
1061 return LBC_NOT_FLAT_ENOUGH;
1062
1063 } // end if
1064
1065 // we passed all the checks
1066 return LBC_OK;
1067
1068} // end isLocationLegalToBuild

1114BuildAssistant::TileBuildInfo *BuildAssistant::buildTiledLocations( const ThingTemplate *thingBeingTiled,
1115 Real angle,
1116 const Coord3D *start,
1117 const Coord3D *end,
1118 Real tilingSize,
1119 Int maxTiles,
1120 Object *builderObject )
1121{
1122
1123 // sanity
1124 if( start == NULL || end == NULL )
1125 return 0;
1126
1127 //
1128 // we will fill out our own internal array of positions, it better be big enough to
1129 // accomodate max tiles, if it's not lets make it bigger!
1130 //
1131 if( maxTiles > m_buildPositionSize )
1132 {
1133
1134 // delete the old array
1135 delete [] m_buildPositions;
1136
1137 // create a new one
1138 m_buildPositions = NEW Coord3D[ maxTiles ];
1139 m_buildPositionSize = maxTiles;
1140
1141 //
1142 // lets try to at least keep sanity here so that we don't have a completely unbounded
1143 // allocation spot in the code here
1144 //
1145 DEBUG_ASSERTCRASH( m_buildPositionSize < 200, ("Do you really need to tile this many objects!!!") );
1146
1147 } // end if
1148 Coord3D *positions = m_buildPositions;
1149
1150 // compute a vector from the start of the line in the world to the end
1151 Coord3D placementVector;
1152 placementVector.x = end->x - start->x;
1153 placementVector.y = end->y - start->y;
1154 placementVector.z = 0.0f; //end->z - start->z;
1155
1156 //
1157 // get the lengh of the placement vector in the world, we'll use this to see how
1158 // many objects we'll need to fill the entire line
1159 //
1160 Real placementLength = placementVector.length();
1161
1162 //
1163 // first see how many objects we're going to need to go from the placement
1164 // line start to the placement line end, we are guaranteed to have one thing (+1)
1165 //
1166 Int tilesNeeded = REAL_TO_INT(placementLength / tilingSize) + 1;
1167
1168 // we have a max that we are allowed to build to
1169 if( tilesNeeded > maxTiles )
1170 tilesNeeded = maxTiles;
1171
1172 // we're guaranteed to have at least one object so lets take care of that position now
1173 positions[ 0 ] = *start;
1174
1175 //
1176 // march down each object and set the position based on its position in the line.
1177 // note that we're skipping the first one because we've already set it explicitly
1178 // and are guaranteed to have at least one object
1179 //
1180 Int tilesUsed = 1; // actual objects "tiled", we are guaranteed to have at least one
1181 Coord3D pos;
1182 Coord3D v = placementVector;
1183 v.normalize();
1184 for( Int i = 1; i < tilesNeeded; i++ )
1185 {
1186
1187 // compute position of object
1188 pos.x = v.x * (tilingSize * i) + start->x;
1189 pos.y = v.y * (tilingSize * i) + start->y;
1190 pos.z = TheTerrainLogic->getGroundHeight( pos.x, pos.y );
1191
1192 // check for a legal position to be at and stop the tiling process if that becomes broken
1193 if( isLocationLegalToBuild( &pos, thingBeingTiled, angle,
1194 BuildAssistant::USE_QUICK_PATHFIND |
1195 BuildAssistant::TERRAIN_RESTRICTIONS |
1196 BuildAssistant::CLEAR_PATH |
1197 BuildAssistant::NO_OBJECT_OVERLAP |
1198 BuildAssistant::SHROUD_REVEALED,
1199 builderObject,
1200 NULL) != LBC_OK )
1201 break;
1202
1203 // save the position in the output array
1204 positions[ i ] = pos;
1205
1206 // we have now actually used one more "tile"
1207 tilesUsed++;
1208
1209 } // end for i
1210
1211 // return a struct filled out with the actual tiles used and the array of locations
1212 static TileBuildInfo tileInfo;
1213 tileInfo.tilesUsed = tilesUsed;
1214 tileInfo.positions = positions;
1215 return &tileInfo;
1216
1217} // end buildTiledLocations

1411// ------------------------------------------------------------------------------------------------
1415// ------------------------------------------------------------------------------------------------
1416Bool BuildAssistant::moveObjectsForConstruction( const ThingTemplate *whatToBuild,
1417 const Coord3D *pos,
1418 Real angle,
1419 Player *playerToBuild )
1420{
1421 GeometryInfo gi (GEOMETRY_BOX, false, 10, whatToBuild->getTemplateGeometryInfo().getMajorRadius(),
1422 whatToBuild->getTemplateGeometryInfo().getMajorRadius());
1423 if (whatToBuild->getTemplateGeometryInfo().getGeomType()==GEOMETRY_BOX) {
1424 gi = whatToBuild->getTemplateGeometryInfo();
1425 }
1426 ObjectIterator *iter =
1427 ThePartitionManager->iteratePotentialCollisions( pos,
1428 gi,
1429 angle );
1430 Bool anyUnmovables = false;
1431 MemoryPoolObjectHolder hold( iter );
1432
1433 Real radius = sqrt(pow(gi.getMajorRadius(), 2) + pow(gi.getMinorRadius(), 2));
1434 radius *= 1.4f; // Fudge the distance,
1435
1436 for( Object *them = iter->first(); them; them = iter->next() )
1437 {
1438 //ignore land mines, cluster mines and demo traps, since you can build on them
1439 // doing so will damage them during construction, by the way
1440 if (them->isKindOf( KINDOF_MINE ))
1441 continue;
1442
1443 // same story for inert things: ie, radiation fields, pings, and projectile streams (!)...
1444 if (them->isKindOf(KINDOF_INERT))
1445 continue;
1446
1447 // Skip KINDOF_ALWAYS_SELECTABLE and isRemovableForConstruction, because if it is
1448 // RemovableForConstruction, it just got destroyed, and won't actually be gone until
1449 // the end of the frame. jba.
1450 if ( !them->isKindOf( KINDOF_ALWAYS_SELECTABLE ) && !isRemovableForConstruction(them) )
1451 {
1452 Relationship rel = playerToBuild->getRelationship(them->getTeam());
1453
1454 if (rel == NEUTRAL || rel == ALLIES)
1455 {
1456 // Pick a point outside of the construction and tell those fools to move there.
1457 // Pick an arbitrary direction and tell the unit to move there, at a distance greater
1458 // than the object's radius.
1459 AIUpdateInterface *ai = them->getAIUpdateInterface();
1460 if (ai)
1461 {
1462 // Vary the distance to move between one half the diameter of the building (roughly)
1463 // and 1.5 times the diameter of the building
1464 Real variedRadius = GameLogicRandomValueReal(0.5, 1.5) * radius;
1465
1466 Coord3D destPos;
