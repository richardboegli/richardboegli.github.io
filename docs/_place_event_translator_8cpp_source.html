68GameMessageDisposition PlaceEventTranslator::translateGameMessage(const GameMessage *msg)
69{
70 GameMessageDisposition disp = KEEP_MESSAGE;
71
72 switch(msg->getType())
73 {
74
75 //---------------------------------------------------------------------------------------------
76 case GameMessage::MSG_RAW_MOUSE_LEFT_BUTTON_DOWN:
77 {
78 // if we're in a building placement mode, do the place and send to all players
79 const ThingTemplate *build = TheInGameUI->getPendingPlaceType();
80 if( build && TheInGameUI->isPlacementAnchored() == FALSE )
81 {
82 ICoord2D mouse = msg->getArgument(0)->pixel;
83 Coord3D world;
84
85 // translate mouse position to world position
86 TheTacticalView->screenToTerrain( &mouse, &world );
87
88 //
89 // placing things causes a dozer to go over and build it ... get the dozer in question
90 // from the in game UI
91 //
92 Object *builderObject = TheGameLogic->findObjectByID( TheInGameUI->getPendingPlaceSourceObjectID() );
93
94 // if our source object is gone cancel this whole placement process
95 if( builderObject == NULL )
96 {
97
98 TheInGameUI->placeBuildAvailable( NULL, NULL );
99 break;
100
101 } // end if
102
103 // set this location as the placement anchor
104 TheInGameUI->setPlacementStart( &mouse );
105
106/*
107//
108// This block of code checks for valid placement on a down mouse click, but since we can
109// rotate a building into a valid location, this check prevents us from placing things
110// down in some legal locations
111//
112 // get the type of thing we want to build
113 const ThingTemplate *whatToBuild = TheInGameUI->getPendingPlaceType();
114
115 //
116 // if the spot at which they choose to place this thing is illegal we won't start
117 // the placement anchor, instead we play a "can't do that" sound
118 //
119 LegalBuildCode lbc;
120 lbc = TheBuildAssistant->isLocationLegalToBuild( &world,
121 whatToBuild,
122 TheInGameUI->getPlacementAngle(),
123 BuildAssistant::USE_QUICK_PATHFIND |
124 BuildAssistant::TERRAIN_RESTRICTIONS |
125 BuildAssistant::CLEAR_PATH |
126 BuildAssistant::NO_OBJECT_OVERLAP,
127 builderObject );
128 if( lbc != LBC_OK )
129 {
130 static const Sound *noCanDoSound = TheAudio->Sounds->getSound( "NoCanDoSound" );
131
132 // play a can't do that sound
133 TheAudio->Sounds->playSound( noCanDoSound );
134
135 // display a message to the user as to why you can't build there
136 TheInGameUI->displayCantBuildMessage( lbc );
137
138 } // end if
139 else
140 {
141
142 // start placement anchor
143 TheInGameUI->setPlacementStart(&mouse);
144
145 } // end else
146*/
147
148 // used the input
149 disp = DESTROY_MESSAGE;
150
151 }
152 break;
153 }
154
155 //---------------------------------------------------------------------------------------------
156 case GameMessage::MSG_MOUSE_LEFT_DOUBLE_CLICK:
157 case GameMessage::MSG_MOUSE_LEFT_CLICK:
158 {
159 // if we're in a building placement mode, do the place and send to all players
160 const ThingTemplate *build = TheInGameUI->getPendingPlaceType();
161
162 // ... and also remove any radius cursor that is active.
163 // (srj sez: not sure if this is always necessary... more of a failsafe to make it go away.)
164 TheInGameUI->setRadiusCursorNone();
165
166 if (build && TheInGameUI->isPlacementAnchored())
167 {
168 GameMessage *placeMsg;
169// Player *player = ThePlayerList->getLocalPlayer();
170 Coord3D world;
171 Real angle;
172 ICoord2D anchorStart, anchorEnd;
173 Bool isLineBuild = TheBuildAssistant->isLineBuildTemplate( build );
174
175 // get the angle of the drawable at the cursor to use as the initial angle
176 angle = TheInGameUI->getPlacementAngle();
177
178 // get start point from the anchor arrow used to place and select angles
179 TheInGameUI->getPlacementPoints( &anchorStart, &anchorEnd );
180
181 // translate the screen position of start to world target location
182 TheTacticalView->screenToTerrain( &anchorStart, &world );
183
184 // get the source object ID of the thing that is "building" the object
185 ObjectID builderID = INVALID_ID;
186 Object *builderObj = TheGameLogic->findObjectByID( TheInGameUI->getPendingPlaceSourceObjectID() );
187 if( builderObj )
188 builderID = builderObj->getID();
189
190 //Kris: September 27, 2002
191 //Make sure we have enough CASH to build it! It's possible that between the
192 //time we initiated it and the time we confirm it, a hacker has stolen some of
193 //our cash!
194 CanMakeType cmt = TheBuildAssistant->canMakeUnit( builderObj, build );
195 if( cmt != CANMAKE_OK )
196 {
197 if (cmt == CANMAKE_NO_MONEY)
198 {
199 TheEva->setShouldPlay(EVA_InsufficientFunds);
200 TheInGameUI->message( "GUI:NotEnoughMoneyToBuild" );
201 break;
202 }
203 else if (cmt == CANMAKE_QUEUE_FULL)
204 {
205 TheInGameUI->message( "GUI:ProductionQueueFull" );
206 break;
207 }
208 else if (cmt == CANMAKE_PARKING_PLACES_FULL)
209 {
210 TheInGameUI->message( "GUI:ParkingPlacesFull" );
211 break;
212 }
213 else if( cmt == CANMAKE_MAXED_OUT_FOR_PLAYER )
214 {
215 TheInGameUI->message( "GUI:UnitMaxedOut" );
216 break;
217 }
218 // get out of pending placement mode, this will also clear the arrow anchor status
219 TheInGameUI->placeBuildAvailable( NULL, NULL );
220 break;
221 }
222
223 // check to see if this is a legal location to build something at
224 LegalBuildCode lbc;
225 lbc = TheBuildAssistant->isLocationLegalToBuild( &world,
226 build,
227 angle,
228 BuildAssistant::USE_QUICK_PATHFIND |
229 BuildAssistant::TERRAIN_RESTRICTIONS |
230 BuildAssistant::CLEAR_PATH |
231 BuildAssistant::NO_OBJECT_OVERLAP |
232 BuildAssistant::SHROUD_REVEALED |
233 BuildAssistant::IGNORE_STEALTHED |
234 BuildAssistant::FAIL_STEALTHED_WITHOUT_FEEDBACK,
235 builderObj, NULL );
236 if( lbc == LBC_OK )
237 {
238 //Are we building this structure via the special power system? (special case for sneak attack)
239 if( builderObj )
240 {
241 ProductionUpdateInterface *puInterface = builderObj->getProductionUpdateInterface();
242 if( puInterface )
243 {
244 const CommandButton *commandButton = puInterface->getSpecialPowerConstructionCommandButton();
245 if( commandButton )
246 {
247 //If we get this far, then we aren't going to really build the object using the production update
248 //interface. Instead, we're going to trigger the special power to create it magically without a
249 //dozer/worker.
250 placeMsg = TheMessageStream->appendMessage( GameMessage::MSG_DO_SPECIAL_POWER_AT_LOCATION );
251 placeMsg->appendIntegerArgument( commandButton->getSpecialPowerTemplate()->getID() ); //The ID of the special power template.
252 placeMsg->appendLocationArgument( world ); //Position of special to be fired.
253 placeMsg->appendRealArgument( angle ); //Angle of special to be fired.
254 placeMsg->appendObjectIDArgument( INVALID_ID ); //There is no object in the way.
255 placeMsg->appendIntegerArgument( commandButton->getOptions() ); //Command button options.
256 placeMsg->appendObjectIDArgument( builderObj->getID() ); //The source object responsible for firing the special.
257
258 // get out of pending placement mode, this will also clear the arrow anchor status
259 TheInGameUI->placeBuildAvailable( NULL, NULL );
260
261 // used the input
262 disp = DESTROY_MESSAGE;
263 m_frameOfUpButton = TheGameLogic->getFrame();
264 break;
265 }
266 }
267 }
268
269 // create the right kind of message
270 if( isLineBuild )
271 placeMsg = TheMessageStream->appendMessage( GameMessage::MSG_DOZER_CONSTRUCT_LINE );
272 else
273 placeMsg = TheMessageStream->appendMessage( GameMessage::MSG_DOZER_CONSTRUCT );
274
275 placeMsg->appendIntegerArgument(build->getTemplateID());
276 placeMsg->appendLocationArgument(world);
277 placeMsg->appendRealArgument(angle);
278 if( isLineBuild )
279 {
280 Coord3D worldEnd;
281
282 TheTacticalView->screenToTerrain( &anchorEnd, &worldEnd );
283 placeMsg->appendLocationArgument( worldEnd );
284
285 } // end if
286
287 pickAndPlayUnitVoiceResponse( TheInGameUI->getAllSelectedDrawables(), placeMsg->getType() );
288
289 // get out of pending placement mode, this will also clear the arrow anchor status
290 TheInGameUI->placeBuildAvailable( NULL, NULL );
291
292 } // end if, location legal to build at
293 else
294 {
295 // can't place, display why
296 TheInGameUI->displayCantBuildMessage( lbc );
297
298 //Cannot build here -- play the voice sound from the dozer
299 AudioEventRTS sound = *builderObj->getTemplate()->getPerUnitSound( "VoiceNoBuild" );
300 sound.setObjectID( builderObj->getID() );
301 TheAudio->addAudioEvent( &sound );
302
303 // play a can't do that sound (UI beep type sound)
304 static AudioEventRTS noCanDoSound( "NoCanDoSound" );
305 TheAudio->addAudioEvent( &noCanDoSound );
306
307 // unhook the anchor so they can try again
308 TheInGameUI->setPlacementStart( NULL );
309
310 } // end else
311
312 // used the input
313 disp = DESTROY_MESSAGE;
314 m_frameOfUpButton = TheGameLogic->getFrame();
315
316 }
317
318 if (disp == DESTROY_MESSAGE)
319 TheInGameUI->clearAttackMoveToMode();
320
321 break;
322
323 }
324
325 //---------------------------------------------------------------------------------------------
326 case GameMessage::MSG_RAW_MOUSE_POSITION:
327 {
328 // if a building placement is in progress update the destination position
329 if (TheInGameUI->isPlacementAnchored())
330 {
331 const Int PLACEMENT_DRAG_THRESHOLD_DIST = 5; // in pixels away from anchor point
332 ICoord2D mouse = msg->getArgument(0)->pixel;
333
334 //
335 // we will only process placement end point sets (clicking, and dragging to set angles)
336 // if we have moved far enough away from the start point
337 //
338 ICoord2D start;
339 TheInGameUI->getPlacementPoints( &start, NULL );
340
341 Int x, y;
342 x = mouse.x - start.x;
343 y = mouse.y - start.y;
344 if( sqrt( (x * x) + (y * y) ) >= PLACEMENT_DRAG_THRESHOLD_DIST )
345 {
346
347 TheInGameUI->setPlacementEnd(&mouse);
348 disp = DESTROY_MESSAGE;
349
350 } // end if
351
352 }
353 break;
354 }
355 }
356
357 return disp;
358}
