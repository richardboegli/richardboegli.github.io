32FrameGrabClass::FrameGrabClass(const char *filename, MODE mode, int width, int height, int bitcount, float framerate)
33{
34 HRESULT hr;
35
36 Mode = mode;
37 Filename = filename;
38 FrameRate = framerate;
39 Counter = 0;
40
41 Stream = 0;
42 AVIFile = 0;
43
44 if(Mode != AVI) return;
45
46 AVIFileInit(); // opens AVIFile library
47
48 // find the first free file with this prefix
49 int counter = 0;
50 int result;
51 char file[256];
52 do {
53 sprintf(file, "%s%d.AVI", filename, counter++);
54 result = _access(file, 0);
55 } while(result != -1);
56
57 // Create new AVI file using AVIFileOpen.
58 hr = AVIFileOpen(&AVIFile, file, OF_WRITE | OF_CREATE, NULL);
59 if (hr != 0) {
60 char buf[256];
61 sprintf(buf, "Unable to open %s\n", Filename);
62 OutputDebugString(buf);
63 CleanupAVI();
64 return;
65 }
66
67
68 // Create a stream using AVIFileCreateStream.
69 AVIStreamInfo.fccType = streamtypeVIDEO;
70 AVIStreamInfo.fccHandler = mmioFOURCC('M','S','V','C');
71 AVIStreamInfo.dwFlags = 0;
72 AVIStreamInfo.dwCaps = 0;
73 AVIStreamInfo.wPriority = 0;
74 AVIStreamInfo.wLanguage = 0;
75 AVIStreamInfo.dwScale = 1;
76 AVIStreamInfo.dwRate = (int)FrameRate;
77 AVIStreamInfo.dwStart = 0;
78 AVIStreamInfo.dwLength = 0;
79 AVIStreamInfo.dwInitialFrames = 0;
80 AVIStreamInfo.dwSuggestedBufferSize = 0;
81 AVIStreamInfo.dwQuality = 0;
82 AVIStreamInfo.dwSampleSize = 0;
83 SetRect(&AVIStreamInfo.rcFrame, 0, 0, width, height);
84 AVIStreamInfo.dwEditCount = 0;
85 AVIStreamInfo.dwFormatChangeCount = 0;
86 sprintf(AVIStreamInfo.szName,"G");
87
88 hr = AVIFileCreateStream(AVIFile, &Stream, &AVIStreamInfo);
89 if (hr != 0) {
90 CleanupAVI();
91 return;
92 }
93
94 // Set format of new stream
95 BitmapInfoHeader.biWidth = width;
96 BitmapInfoHeader.biHeight = height;
97 BitmapInfoHeader.biBitCount = (unsigned short)bitcount;
98 BitmapInfoHeader.biSizeImage = ((((UINT)BitmapInfoHeader.biBitCount * BitmapInfoHeader.biWidth + 31) & ~31) / 8) * BitmapInfoHeader.biHeight;
99 BitmapInfoHeader.biSize = sizeof(BITMAPINFOHEADER); // size of structure
100 BitmapInfoHeader.biPlanes = 1; // must be set to 1
101 BitmapInfoHeader.biCompression = BI_RGB; // uncompressed
102 BitmapInfoHeader.biXPelsPerMeter = 1; // not used
103 BitmapInfoHeader.biYPelsPerMeter = 1; // not used
104 BitmapInfoHeader.biClrUsed = 0; // all colors are used
105 BitmapInfoHeader.biClrImportant = 0; // all colors are important
106
107 hr = AVIStreamSetFormat(Stream, 0, &BitmapInfoHeader, sizeof(BitmapInfoHeader));
108 if (hr != 0) {
109 CleanupAVI();
110 return;
111 }
112
113 Bitmap = (long *) GlobalAllocPtr(GMEM_MOVEABLE, BitmapInfoHeader.biSizeImage);
114}

123void FrameGrabClass::CleanupAVI() {

167void FrameGrabClass::ConvertFrame(void *BitmapPointer)
168{
169
170 int width = BitmapInfoHeader.biWidth;
171 int height = BitmapInfoHeader.biHeight;
172 long *image = (long *) BitmapPointer;
173
174 // copy the data, doing a vertical flip & byte re-ordering of the pixel longwords
175 int y = height;
176 while(y--) {
177 int x = width;
178 int yoffset = y * width;
179 int yoffset2 = (height - y) * width;
180 while(x--) {
181 long *source = &image[yoffset + x];
182 long *dest = &Bitmap[yoffset2 + x];
183 *dest = *source;
184 unsigned char *c = (unsigned char *) dest;
185 c[3] = c[0];
186 c[0] = c[2];
187 c[2] = c[3];
188 c[3] = 0;
189 }
190 }
