49/*
50** Byte order controlled long integer. This integer is constructed
51** so that character 0 (C0) is the most significant byte of the
52** integer. This is biased toward big endian architecture, but that
53** just happens to be how the Blowfish algorithm was designed.
54*/
55typedef union {
56 unsigned long Long;
57 struct {
58 unsigned char C3;
59 unsigned char C2;
60 unsigned char C1;
61 unsigned char C0;
62 } Char;
63} Int;

89/***********************************************************************************************
90 * BlowfishEngine::Submit_Key -- Submit a key that will allow data processing. *
91 * *
92 * This routine must be called before any data can be encrypted or decrypted. This routine *
93 * need only be called when the key is to be changed or set for the first time. Once the *
94 * key has been set, the engine may be used to encrypt, decrypt, or both operations *
95 * indefinitely. The key must be 56 bytes or less in length. This is necessary because *
96 * any keys longer than that will not correctly affect the encryption process. *
97 * *
98 * If the key pointer is NULL, then the S-Box tables are reset to identity. This will *
99 * mask the previous key setting. Use this method to clear the engine after processing in *
100 * order to gain a measure of security. *
101 * *
102 * INPUT: key -- Pointer to the key data block. *
103 * *
104 * length -- The length of the submitted key. *
105 * *
106 * OUTPUT: none *
107 * *
108 * WARNINGS: This is a time consuming process. *
109 * *
110 * HISTORY: *
111 * 04/14/1996 JLB : Created. *
112 *=============================================================================================*/
113void BlowfishEngine::Submit_Key(void const * key, int length)
114{
115 assert(length <= MAX_KEY_LENGTH);
116
117 /*
118 ** Initialize the permutation and S-Box tables to a known
119 ** constant value.
120 */
121 memcpy(P_Encrypt, P_Init, sizeof(P_Init));
122 memcpy(P_Decrypt, P_Init, sizeof(P_Init));
123 memcpy(bf_S, S_Init, sizeof(S_Init));
124
125 /*
126 ** Validate parameters.
127 */
128 if (key == 0 || length == 0) {
129 IsKeyed = false;
130 return;
131 }
132
133 /*
134 ** Combine the key with the permutation table. Wrap the key
135 ** as many times as necessary to ensure that the entire
136 ** permutation table has been modified. The key is lifted
137 ** into a long by using endian independent means.
138 */
139 int j = 0;
140 unsigned char const * key_ptr = (unsigned char const *)key;
141 unsigned long * p_ptr = &P_Encrypt[0];
142 for (int index = 0; index < ROUNDS+2; index++) {
143 unsigned long data = 0;
144
145 data = (data << CHAR_BIT) | key_ptr[j++ % length];
146 data = (data << CHAR_BIT) | key_ptr[j++ % length];
147 data = (data << CHAR_BIT) | key_ptr[j++ % length];
148 data = (data << CHAR_BIT) | key_ptr[j++ % length];
149
150 *p_ptr++ ^= data;
151 }
152
153 /*
154 ** The permutation table must be scrambled by means of the key. This
155 ** is how the key is factored into the encryption -- by merely altering
156 ** the permutation (and S-Box) tables. Because this transformation alters
157 ** the table data WHILE it is using the table data, the tables are
158 ** thoroughly obfuscated by this process.
159 */
160 unsigned long left = 0x00000000L;
161 unsigned long right = 0x00000000L;
162 unsigned long * p_en = &P_Encrypt[0]; // Encryption table.
163 unsigned long * p_de = &P_Decrypt[ROUNDS+1]; // Decryption table.
164 for (int p_index = 0; p_index < ROUNDS+2; p_index += 2) {
165 Sub_Key_Encrypt(left, right);
166
167 *p_en++ = left;
168 *p_en++ = right;
169
170 *p_de-- = left;
171 *p_de-- = right;
172 }
173
174 /*
175 ** Perform a similar transmutation to the S-Box tables. Also notice that the
176 ** working 64 bit number is carried into this process from the previous
177 ** operation.
178 */
179 for (int sbox_index = 0; sbox_index < 4; sbox_index++) {
180 for (int ss_index = 0; ss_index < UCHAR_MAX+1; ss_index += 2) {
181 Sub_Key_Encrypt(left, right);
182 bf_S[sbox_index][ss_index] = left;
183 bf_S[sbox_index][ss_index + 1] = right;
184 }
185 }
186
187 IsKeyed = true;
188}

191/***********************************************************************************************
192 * BlowfishEngine::Encrypt -- Encrypt an arbitrary block of data. *
193 * *
194 * Use this routine to encrypt an arbitrary block of data. The block must be an even *
195 * multiple of 8 bytes. Any bytes left over will not be encrypted. The 8 byte requirement *
196 * is necessary because the underlying algorithm processes blocks in 8 byte chunks. *
197 * Partial blocks are unrecoverable and useless. *
198 * *
199 * INPUT: plaintext-- Pointer to the data block to be encrypted. *
200 * *
201 * length -- The length of the data block. *
202 * *
203 * cyphertext- Pointer to the output buffer that will hold the encrypted data. *
204 * *
205 * OUTPUT: Returns with the actual number of bytes encrypted. *
206 * *
207 * WARNINGS: You must submit the key before calling this routine. This will only encrypt *
208 * the plaintext in 8 byte increments. Modulo bytes left over are not processed. *
209 * *
210 * HISTORY: *
211 * 04/14/1996 JLB : Created. *
212 *=============================================================================================*/
213int BlowfishEngine::Encrypt(void const * plaintext, int length, void * cyphertext)
214{
215 if (plaintext == 0 || length == 0) {
216 return(0);
217 }
218 if (cyphertext == 0) cyphertext = (void *)plaintext;
219
220 if (IsKeyed) {
221
222 /*
223 ** Validate parameters.
224 */
225 int blocks = length / BYTES_PER_BLOCK;
226
227 /*
228 ** Process the buffer in 64 bit chunks.
229 */
230 for (int index = 0; index < blocks; index++) {
231 Process_Block(plaintext, cyphertext, P_Encrypt);
232 plaintext = ((char *)plaintext) + BYTES_PER_BLOCK;
233 cyphertext = ((char *)cyphertext) + BYTES_PER_BLOCK;
234 }
235 int encrypted = blocks * BYTES_PER_BLOCK;
236
237 /*
238 ** Copy over any trailing left over appendix bytes.
239 */
240 if (encrypted < length) {
241 memmove(cyphertext, plaintext, length - encrypted);
242 }
243
244 return(encrypted);
245 }
246
247 /*
248 ** Non-keyed processing merely copies the data.
249 */
250 if (plaintext != cyphertext) {
251 memmove(cyphertext, plaintext, length);
252 }
253 return(length);
254}

257/***********************************************************************************************
258 * BlowfishEngine::Decrypt -- Decrypt an arbitrary block of data. *
259 * *
260 * Use this routine to decrypt an arbitrary block of data. The block must be an even *
261 * multiple of 8 bytes. Any bytes left over will not be decrypted. The 8 byte requirement *
262 * is necessary because the underlying algorithm processes blocks in 8 byte chunks. *
263 * Partial blocks are unrecoverable and useless. *
264 * *
265 * INPUT: cyphertext- Pointer to the data block to be decrypted. *
266 * *
267 * length -- The length of the data block. *
268 * *
269 * plaintext-- Pointer to the output buffer that will hold the decrypted data. *
270 * *
271 * OUTPUT: Returns with the actual number of bytes decrypted. *
272 * *
273 * WARNINGS: You must submit the key before calling this routine. This will only decrypt *
274 * the cyphertext in 8 byte increments. Modulo bytes left over are not processed. *
275 * *
276 * HISTORY: *
277 * 04/14/1996 JLB : Created. *
278 *=============================================================================================*/
279int BlowfishEngine::Decrypt(void const * cyphertext, int length, void * plaintext)
280{
281 if (cyphertext == 0 || length == 0) {
282 return(0);
283 }
284 if (plaintext == 0) plaintext = (void *)cyphertext;
285
286 if (IsKeyed) {
287
288 /*
289 ** Validate parameters.
290 */
291 int blocks = length / BYTES_PER_BLOCK;
292
293 /*
294 ** Process the buffer in 64 bit chunks.
295 */
296 for (int index = 0; index < blocks; index++) {
297 Process_Block(cyphertext, plaintext, P_Decrypt);
298 cyphertext = ((char *)cyphertext) + BYTES_PER_BLOCK;
299 plaintext = ((char *)plaintext) + BYTES_PER_BLOCK;
300 }
301 int encrypted = blocks * BYTES_PER_BLOCK;
302
303 /*
304 ** Copy over any trailing left over appendix bytes.
305 */
306 if (encrypted < length) {
307 memmove(plaintext, cyphertext, length - encrypted);
308 }
309
310 return(encrypted);
311 }
312
313 /*
314 ** Non-keyed processing merely copies the data.
315 */
316 if (plaintext != cyphertext) {
317 memmove(plaintext, cyphertext, length);
318 }
319 return(length);
320}

323/***********************************************************************************************
324 * BlowfishEngine::Process_Block -- Process a block of data using Blowfish algorithm. *
325 * *
326 * This is the main processing routine for encryption and decryption. The algorithm *
327 * consists of a 16 round Feistal network and uses mathematics from different algebraic *
328 * groups (strengthens against differential cryptanalysis). The large S-Boxes and the *
329 * rounds strengthen it against linear cryptanalysis. *
330 * *
331 * INPUT: plaintext -- Pointer to the source text (it actually might be a pointer to *
332 * the cyphertext if this is called as a decryption process). *
333 * *
334 * cyphertext -- Pointer to the output buffer that will hold the processed block. *
335 * *
336 * ptable -- Pointer to the permutation table. This algorithm will encrypt *
337 * and decrypt using the same S-Box tables. The encryption control *
338 * is handled by the permutation table. *
339 * *
340 * OUTPUT: none *
341 * *
342 * WARNINGS: The source and destination buffers must be 8 bytes long. *
343 * *
344 * HISTORY: *
345 * 04/19/1996 JLB : Created. *
346 *=============================================================================================*/
347void BlowfishEngine::Process_Block(void const * plaintext, void * cyphertext, unsigned long const * ptable)
348{
349 /*
350 ** Input the left and right halves of the source block such that
351 ** the byte order is constant regardless of the endian
352 ** persuasion of the current processor. The blowfish algorithm is
353 ** biased toward "big endian" architecture and some optimizations
354 ** could be done for big endian processors in that case.
355 */
356 unsigned char const * source = (unsigned char const *)plaintext;
357 Int left;
358 left.Char.C0 = *source++;
359 left.Char.C1 = *source++;
360 left.Char.C2 = *source++;
361 left.Char.C3 = *source++;
362
363 Int right;
364 right.Char.C0 = *source++;
365 right.Char.C1 = *source++;
366 right.Char.C2 = *source++;
367 right.Char.C3 = *source;
368
369 /*
370 ** Perform all Feistal rounds on the block. This is the encryption/decryption
371 ** process. Since there is an exchange that occurs after each round, two
372 ** rounds are combined in this loop to avoid unnecessary exchanging.
373 */
374 for (int index = 0; index < ROUNDS/2; index++) {
375 left.Long ^= *ptable++;
376 right.Long ^= ((( bf_S[0][left.Char.C0] + bf_S[1][left.Char.C1]) ^ bf_S[2][left.Char.C2]) + bf_S[3][left.Char.C3]);
377 right.Long ^= *ptable++;
378 left.Long ^= ((( bf_S[0][right.Char.C0] + bf_S[1][right.Char.C1]) ^ bf_S[2][right.Char.C2]) + bf_S[3][right.Char.C3]);
379 }
380
381 /*
382 ** The final two longs in the permutation table are processed into the block.
383 ** The left and right halves are still reversed as a side effect of the last
384 ** round.
385 */
386 left.Long ^= *ptable++;
387 right.Long ^= *ptable;
388
389 /*
390 ** The final block data is output in endian architecture
391 ** independent format. Notice that the blocks are output as
392 ** right first and left second. This is to counteract the final
393 ** superfluous exchange that occurs as a side effect of the
394 ** encryption rounds.
395 */
396 unsigned char * out = (unsigned char *)cyphertext;
397 *out++ = right.Char.C0;
398 *out++ = right.Char.C1;
399 *out++ = right.Char.C2;
400 *out++ = right.Char.C3;
401
402 *out++ = left.Char.C0;
403 *out++ = left.Char.C1;
404 *out++ = left.Char.C2;
405 *out = left.Char.C3;
406}

409/***********************************************************************************************
410 * BlowfishEngine::Sub_Key_Encrypt -- Encrypts a block for use in S-Box processing. *
411 * *
412 * This is the same as the normal process block function but it doesn't have the endian *
413 * fixup logic. Since this routine is only called for S-Box table generation and it is *
414 * known that the S-Box initial data is already in local machine endian format, the *
415 * byte order fixups are not needed. This also has a tendency to speed up S-Box generation *
416 * as well. *
417 * *
418 * INPUT: left -- The left half of the data block. *
419 * *
420 * right -- The right half of the data block. *
421 * *
422 * OUTPUT: none, but the processed block is stored back into the left and right half *
423 * integers. *
424 * *
425 * WARNINGS: none *
426 * *
427 * HISTORY: *
428 * 04/19/1996 JLB : Created. *
429 *=============================================================================================*/
430void BlowfishEngine::Sub_Key_Encrypt(unsigned long & left, unsigned long & right)
431{
432 Int l;
433 l.Long = left;
434
435 Int r;
436 r.Long = right;
437
438 for (int index = 0; index < ROUNDS; index += 2) {
439 l.Long ^= P_Encrypt[index];
440 r.Long ^= ((( bf_S[0][l.Char.C0] + bf_S[1][l.Char.C1]) ^ bf_S[2][l.Char.C2]) + bf_S[3][l.Char.C3]);
441 r.Long ^= P_Encrypt[index+1];
442 l.Long ^= ((( bf_S[0][r.Char.C0] + bf_S[1][r.Char.C1]) ^ bf_S[2][r.Char.C2]) + bf_S[3][r.Char.C3]);
443 }
444 left = r.Long ^ P_Encrypt[ROUNDS+1];
445 right = l.Long ^ P_Encrypt[ROUNDS];
446}
