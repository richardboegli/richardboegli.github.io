43/***********************************************************************************************
44 * CRCEngine::operator() -- Submits one byte of data to the CRC engine. *
45 * *
46 * This routine will take the specified byte of data and submit it to the CRC engine *
47 * for processing. This routine is designed to be as fast as possible since the typical *
48 * use of this routine is to feed one of presumably many byte sized chunks of data to the *
49 * CRC engine. *
50 * *
51 * INPUT: datum -- One byte of data to submit to the CRC engine. *
52 * *
53 * OUTPUT: none *
54 * *
55 * WARNINGS: If possible, use the buffer/size operator to submit data rather than repeated *
56 * calls to this routine. *
57 * *
58 * HISTORY: *
59 * 03/02/1996 JLB : Created. *
60 *=============================================================================================*/
61void CRCEngine::operator() (char datum)
62{
63 StagingBuffer.Buffer[Index++] = datum;
64
65 if (Index == sizeof(long)) {
66 CRC = Value();
67 StagingBuffer.Composite = 0;
68 Index = 0;
69 }
70}

73/***********************************************************************************************
74 * CRCEngine::operator() -- Submits an arbitrary data block to the CRC engine. *
75 * *
76 * This routine will submit the specified block to the CRC engine. The block can be of *
77 * arbitrary length. *
78 * *
79 * INPUT: buffer -- Pointer to the buffer that contains the data. The buffer will not *
80 * be modified. *
81 * *
82 * length -- The length of the buffer (in bytes). *
83 * *
84 * OUTPUT: Returns with the current CRC value accumulated so far. *
85 * *
86 * WARNINGS: none *
87 * *
88 * HISTORY: *
89 * 03/02/1996 JLB : Created. *
90 *=============================================================================================*/
91long CRCEngine::operator() (void const * buffer, int length)
92{
93 if (buffer != NULL && length > 0) {
94 char const * dataptr = (char const *)buffer;
95 int bytes_left = length;
96
97 /*
98 ** If there are any leader bytes (needed to fill the staging buffer)
99 ** then process those by first using them to fill up the staging
100 ** buffer. The bulk of the data block will be processed by the high
101 ** speed longword processing loop.
102 */
103 while (bytes_left && Buffer_Needs_Data()) {
104 operator()(*dataptr);
105 dataptr++;
106 bytes_left--;
107 }
108
109 /*
110 ** Perform the fast 'bulk' processing by reading long word sized
111 ** data blocks.
112 */
113 long const * longptr = (long const *)dataptr;
114 int longcount = bytes_left / sizeof(long); // Whole 'long' elements remaining.
115 while (longcount--) {
116 CRC = _lrotl(CRC, 1) + *longptr++;
117 bytes_left -= sizeof(long);
118 }
119
120 /*
121 ** If there are remainder bytes, then process these by adding them
122 ** to the staging buffer.
123 */
124 dataptr = (char const *)longptr;
125 while (bytes_left) {
126 operator()(*dataptr);
127 dataptr++;
128 bytes_left--;
129 }
130 }
131
132 /*
133 ** Return the current CRC value.
134 */
135 return(Value());
136}

206unsigned long CRC::Memory( unsigned char *data, unsigned long length, unsigned long crc )
207{
208 crc ^= 0xFFFFFFFF; // invert previous CRC
209 while ( length-- ) {
210 crc = CRC32( *data++, crc ); // calc crc for each byte
211 }
212 return (crc ^ 0xFFFFFFFF); // invert new CRC and return it
213}
