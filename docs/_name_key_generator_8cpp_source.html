76void NameKeyGenerator::freeSockets()
77{
78 for (Int i = 0; i < SOCKET_COUNT; ++i)
79 {
80 Bucket *next;
81 for (Bucket *b = m_sockets[i]; b; b = next)
82 {
83 next = b->m_nextInSocket;
84 b->deleteInstance();
85 }
86 m_sockets[i] = NULL;
87 }
88
89} // end freeSockets

92inline UnsignedInt calcHashForString(const char* p)
93{
94 UnsignedInt result = 0;
95 Byte *pp = (Byte*)p;
96 while (*pp)
97 result = (result << 5) + result + *pp++;
98 return result;
99}

101/* ------------------------------------------------------------------------ */
102inline UnsignedInt calcHashForLowercaseString(const char* p)
103{
104 UnsignedInt result = 0;
105 Byte *pp = (Byte*)p;
106 while (*pp)
107 result = (result << 5) + result + tolower(*pp++);
108 return result;
109}

112AsciiString NameKeyGenerator::keyToName(NameKeyType key)
113{
114 for (Int i = 0; i < SOCKET_COUNT; ++i)
115 {
116 for (Bucket *b = m_sockets[i]; b; b = b->m_nextInSocket)
117 {
118 if (key == b->m_key)
119 return b->m_nameString;
120 }
121 }
122 return AsciiString::TheEmptyString;
123}

126NameKeyType NameKeyGenerator::nameToKey(const char* nameString)
127{
128 Bucket *b;
129
130 UnsignedInt hash = calcHashForString(nameString) % SOCKET_COUNT;
131
132 // hmm, do we have it already?
133 for (b = m_sockets[hash]; b; b = b->m_nextInSocket)
134 {
135 if (strcmp(nameString, b->m_nameString.str()) == 0)
136 return b->m_key;
137 }
138
139 // nope, guess not. let's allocate it.
140 b = newInstance(Bucket);
141 b->m_key = (NameKeyType)m_nextID++;
142 b->m_nameString = nameString;
143 b->m_nextInSocket = m_sockets[hash];
144 m_sockets[hash] = b;
145
146 NameKeyType result = b->m_key;
147
148#if defined(_DEBUG) || defined(_INTERNAL)
149 // reality-check to be sure our hasher isn't going bad.
150 const Int maxThresh = 3;
151 Int numOverThresh = 0;
152 for (Int i = 0; i < SOCKET_COUNT; ++i)
153 {
154 Int numInThisSocket = 0;
155 for (b = m_sockets[i]; b; b = b->m_nextInSocket)
156 ++numInThisSocket;
157
158 if (numInThisSocket > maxThresh)
159 ++numOverThresh;
160 }
161
162 // if more than a small percent of the sockets are getting deep, probably want to increase the socket count.
163 if (numOverThresh > SOCKET_COUNT/20)
164 {
165 DEBUG_CRASH(("hmm, might need to increase the number of bucket-sockets for NameKeyGenerator (numOverThresh %d = %f%%)\n",numOverThresh,(Real)numOverThresh/(Real)(SOCKET_COUNT/20)));
166 }
167#endif
168
169 return result;
170
171} // end nameToKey
