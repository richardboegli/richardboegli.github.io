50void Keyboard::createStreamMessages( void )
51{
52
53 // santiy
54 if( TheMessageStream == NULL )
55 return;
56
57 KeyboardIO *key = getFirstKey();
58 GameMessage *msg = NULL;
59 while( key->key != KEY_NONE )
60 {
61
62 // add message to stream
63 if( BitTest( key->state, KEY_STATE_DOWN ) )
64 {
65
66 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_KEY_DOWN );
67 DEBUG_ASSERTCRASH( msg, ("Unable to append key down message to stream\n") );
68
69 } // end if
70 else if( BitTest( key->state, KEY_STATE_UP ) )
71 {
72
73 msg = TheMessageStream->appendMessage( GameMessage::MSG_RAW_KEY_UP );
74 DEBUG_ASSERTCRASH( msg, ("Unable to append key up message to stream\n") );
75
76 } // end else if
77 else
78 {
79
80 DEBUG_CRASH(( "Unknown key state when creating msg stream\n" ));
81
82 } // end else
83
84 // fill out message arguments
85 if( msg )
86 {
87 msg->appendIntegerArgument( key->key );
88 msg->appendIntegerArgument( key->state );
89 }
90
91 // next key please
92 key++;
93
94 } // end while
95
96} // end createStreamMessages

101void Keyboard::updateKeys( void )
102{
103 Int index = 0;
104
105 // get latest keys
106 do
107 {
108 do
109 {
110
111 // get a single key
112 getKey( &m_keys[ index ] );
113
114 //
115 // if we get back that our device was lost, we haven't been paying
116 // attention to the keyboard, lets start over now, we only need to
117 // do this reset once in this loop though just to be efficient
118 //
119 if( m_keys[ index ].key == KEY_LOST)
120 {
121
122 resetKeys();
123 index = 0;
124
125 } // end if
126
127 }
128 while( m_keys[ index ].key == KEY_LOST );
129 }
130 while( m_keys[ index++ ].key != KEY_NONE );
131
132 // update keyboard status array
133 index = 0;
134 while( m_keys[ index ].key != KEY_NONE )
135 {
136
139
140 m_keyStatus[ m_keys[ index ].key ].state = m_keys[ index ].state;
141 m_keyStatus[ m_keys[ index ].key ].status = m_keys[ index ].status;
142 m_keyStatus[ m_keys[ index ].key ].sequence = m_inputFrame;
143
144 // prevent ALT-TAB from causing a TAB event
145 if( m_keys[ index ].key == KEY_TAB )
146 {
147 if( BitTest( m_keyStatus[ KEY_LALT ].state, KEY_STATE_DOWN ) ||
148 BitTest( m_keyStatus[ KEY_RALT ].state, KEY_STATE_DOWN ) )
149 {
150 m_keys[index].status = KeyboardIO::STATUS_USED;
151 }
152 }
153 else if( m_keys[ index ].key == KEY_CAPS ||
154 m_keys[ index ].key == KEY_LCTRL ||
155 m_keys[ index ].key == KEY_RCTRL ||
156 m_keys[ index ].key == KEY_LSHIFT ||
157 m_keys[ index ].key == KEY_RSHIFT ||
158 m_keys[ index ].key == KEY_LALT ||
159 m_keys[ index ].key == KEY_RALT )
160
161 {
162
163 //
164 // this keeps our internal key state accurate event though we don't
165 // use the returned translation ... kinda weird I think
166 //
167 translateKey( m_keys[ index ].key );
168
169 } // end else if
170
171 index++;
172
173 } // end while
174
175 // check for key repeats
176 checkKeyRepeat();
177
178 if( m_modifiers )
179 {
180 index = 0;
181 while( m_keys[ index ].key != KEY_NONE )
182 {
183
184 // set in the modifier data into the already existing up/down state
185 BitSet( m_keys[ index ].state, m_modifiers );
186
187 // next key
188 index++;
189
190 } // end while
191
192 } // end if
193
194} // end updateKeys

199Bool Keyboard::checkKeyRepeat( void )
200{
201 Bool retVal = FALSE;
202 Int index = 0;
203 Int key;
204
207// if( currentFocus == FOCUS_OUT )
208// return FALSE;
209
210 // Find end of real keys for this frame
211 while( m_keys[ index ].key != KEY_NONE )
212 index++;
213
214 // Scan Keyboard status array for first key down
215 // long enough to repeat
216 for( key = 0; key < 256; key++ )
217 {
218
219 if( BitTest( m_keyStatus[ key ].state, KEY_STATE_DOWN ) )
220 {
221
222 if( (m_inputFrame - m_keyStatus[ key ].sequence) > Keyboard::KEY_REPEAT_DELAY )
223 {
224 // Add key to this frame
225 m_keys[ index ].key = (UnsignedByte)key;
226 m_keys[ index ].state = KEY_STATE_DOWN | KEY_STATE_AUTOREPEAT; // note: not a bitset; this is an assignment
227 m_keys[ index ].status = KeyboardIO::STATUS_UNUSED;
228
229 // Set End Flag
230 m_keys[ ++index ].key = KEY_NONE;
231
232 // Set all keys as new to prevent multiple keys repeating
233 for( index = 0; index< NUM_KEYS; index++ )
234 m_keyStatus[ index ].sequence = m_inputFrame;
235
236 // Set repeated key so it will repeat again in two frames
237 m_keyStatus[ key ].sequence = m_inputFrame - (Keyboard::KEY_REPEAT_DELAY + 2);
238
239 retVal = TRUE;
240 break; // exit for key
241
242 } // end if
243
244 } // end if, key down
245
246 } // end for key
247
248 return retVal;
249
250} // end checkKeyRepeat

814WideChar Keyboard::translateKey( WideChar keyCode )
815{
816
817 if( keyCode > 0x00FF )
818 return keyCode;
819
820 UnsignedByte ubKeyCode = (UnsignedByte)keyCode;
821 switch( ubKeyCode )
822 {
823
824 case KEY_CAPS:
825 if( getKeyStatusData( KEY_CAPS ) == KeyboardIO::STATUS_UNUSED )
826 {
827 if( getKeyStateBit( KEY_CAPS, KEY_STATE_DOWN ) )
828 {
829 if( m_modifiers & KEY_STATE_CAPSLOCK )
830 {
831 // Toggle caplocks off
832 m_modifiers &= ~KEY_STATE_CAPSLOCK;
833 }
834 else
835 {
836 // Toggle caplocks on
837 m_modifiers |= KEY_STATE_CAPSLOCK;
838 }
839 }
840
841 setKeyStatusData( KEY_CAPS, KeyboardIO::STATUS_USED );
842 }
843 return 0;
844
845 case KEY_LSHIFT:
846 if( getKeyStateBit( KEY_LSHIFT, KEY_STATE_DOWN ) )
847 {
848 m_modifiers |= KEY_STATE_LSHIFT;
849 }
850 else
851 {
852 m_modifiers &= ~KEY_STATE_LSHIFT;
853 }
854 return 0;
855
856 case KEY_RSHIFT:
857 if( getKeyStateBit( KEY_RSHIFT, KEY_STATE_DOWN ) )
858 {
859 m_modifiers |= KEY_STATE_RSHIFT;
860 }
861 else
862 {
863 m_modifiers &= ~KEY_STATE_RSHIFT;
864 }
865 return 0;
866
867 case KEY_LCTRL:
868 if( getKeyStateBit( KEY_LCTRL, KEY_STATE_DOWN ) )
869 {
870 m_modifiers |= KEY_STATE_LCONTROL;
871 }
872 else
873 {
874 m_modifiers &= ~KEY_STATE_LCONTROL;
875 }
876 return 0;
877
878 case KEY_RCTRL:
879 if( getKeyStateBit( KEY_RCTRL, KEY_STATE_DOWN ) )
880 {
881 m_modifiers |= KEY_STATE_RCONTROL;
882 }
883 else
884 {
885 m_modifiers &= ~KEY_STATE_RCONTROL;
886 }
887 return 0;
888
889 case KEY_LALT:
890 if( getKeyStateBit( KEY_LALT, KEY_STATE_DOWN ) )
891 {
892 m_modifiers |= KEY_STATE_LALT;
893 }
894 else
895 {
896 m_modifiers &= ~KEY_STATE_LALT;
897 }
898 return 0;
899
900 case KEY_RALT:
901 if( getKeyStateBit( KEY_RALT, KEY_STATE_DOWN ) )
902 {
903 m_modifiers |= KEY_STATE_RALT;
904 }
905 else
906 {
907 m_modifiers &= ~KEY_STATE_RALT;
908 }
909 return 0;
910
911 default:
912 if( ubKeyCode == m_shift2Key )
913 {
914 if( getKeyStateBit( m_shift2Key, KEY_STATE_DOWN ) )
915 {
916 m_modifiers |= KEY_STATE_SHIFT2;
917 }
918 else
919 {
920 m_modifiers &= ~KEY_STATE_SHIFT2;
921 }
922 return 0;
923 }
924
925 if( m_modifiers & KEY_STATE_SHIFT2 )
926 {
927 return( m_keyNames[ ubKeyCode ].shifted2 );
928 }
929
930 if( isShift() || getCapsState() && GameIsAlpha( m_keyNames[ ubKeyCode ].stdKey ) )
931 {
932 return( m_keyNames[ ubKeyCode ].shifted );
933 }
934
935 return( m_keyNames[ubKeyCode ].stdKey );
936 } // end switch( ubKeyCode )
937} // end translateKey
