271//Public inline function declerations
272inline Bool WaterRenderObjClass::worldToGridSpace(Real worldX, Real worldY, Real &gridX, Real &gridY)
273{
274 Real dx,dy;
275 Real ooGridCellSize = 1.0f/m_gridCellSize;
276
277 dx=worldX - m_gridOrigin.X;
278 dy=worldY - m_gridOrigin.Y;
279 gridX = ooGridCellSize * (dx * m_gridDirectionX.X + dy * m_gridDirectionX.Y);
280 gridY = ooGridCellSize * (dx * m_gridDirectionY.X + dy * m_gridDirectionY.Y);
281
282 if (gridX < 0)
283 return FALSE;
284 if (gridX > m_gridCellsX-1 )
285 return FALSE;
286 if (gridY < 0)
287 return FALSE;
288 if (gridY > m_gridCellsY-1 )
289 return FALSE;
290
291 return TRUE;
292}
