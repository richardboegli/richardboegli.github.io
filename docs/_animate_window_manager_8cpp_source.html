180void AnimateWindowManager::update( void )
181{
182
183 ProcessAnimateWindow *processAnim = NULL;
184
185 // if we need to update the windows that need to finish, update that list
186 if(m_needsUpdate)
187 {
188 AnimateWindowList::iterator it = m_winMustFinishList.begin();
189 m_needsUpdate = FALSE;
190
191 while (it != m_winMustFinishList.end())
192 {
193 AnimateWindow *animWin = *it;
194 if (!animWin)
195 {
196 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
197 return;
198 }
199 processAnim = getProcessAnimate( animWin->getAnimType() );
200 if(processAnim)
201 {
202 if(m_reverse)
203 {
204 if(!processAnim->reverseAnimateWindow(animWin))
205 m_needsUpdate = TRUE;
206 }
207 else
208 {
209 if(!processAnim->updateAnimateWindow(animWin))
210 m_needsUpdate = TRUE;
211 }
212 }
213
214 it ++;
215 }
216 }
217
218 AnimateWindowList::iterator it = m_winList.begin();
219
220 while (it != m_winList.end())
221 {
222 AnimateWindow *animWin = *it;
223 if (!animWin)
224 {
225 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
226 return;
227 }
228 processAnim = getProcessAnimate( animWin->getAnimType() );
229 if(m_reverse)
230 {
231 if(processAnim)
232 processAnim->reverseAnimateWindow(animWin);
233 }
234 else
235 {
236 if(processAnim)
237 processAnim->updateAnimateWindow(animWin);
238 }
239 it ++;
240 }
241}

244void AnimateWindowManager::registerGameWindow(GameWindow *win, AnimTypes animType, Bool needsToFinish, UnsignedInt ms, UnsignedInt delayMs)
245{
246 if(!win)
247 {
248 DEBUG_CRASH(("Win was NULL as it was passed into registerGameWindow... not good indeed"));
249 return;
250 }
251 if(animType <= WIN_ANIMATION_NONE || animType >= WIN_ANIMATION_COUNT )
252 {
253 DEBUG_CRASH(("an Invalid WIN_ANIMATION type was passed into registerGameWindow... please fix me "));
254 return;
255 }
256
257 // Create a new AnimateWindow class and fill in it's data.
258 AnimateWindow *animWin = newInstance(AnimateWindow);
259 animWin->setGameWindow(win);
260 animWin->setAnimType(animType);
261 animWin->setNeedsToFinish(needsToFinish);
262 animWin->setDelay(delayMs);
263
264 // Run the window through the processAnim's init function.
265 ProcessAnimateWindow *processAnim = getProcessAnimate( animType );
266 if(processAnim)
267 {
268 processAnim->setMaxDuration(ms);
269 processAnim->initAnimateWindow( animWin );
270 }
271
272 // Add the Window to the proper list
273 if(needsToFinish)
274 {
275 m_winMustFinishList.push_back(animWin);
276 m_needsUpdate = TRUE;
277 }
278 else
279 m_winList.push_back(animWin);
280}

299 return m_slideFromTop;
300 }
301 case WIN_ANIMATION_SLIDE_BOTTOM:
302 {
303 return m_slideFromBottom;
304 }
305 case WIN_ANIMATION_SPIRAL:
306 {
307 return m_spiral;
308 }
309 case WIN_ANIMATION_SLIDE_BOTTOM_TIMED:
310 {
311 return m_slideFromBottomTimed;
312 }
313 case WIN_ANIMATION_SLIDE_TOP_FAST:
314 {
315 return m_slideFromTopFast;
316 }
317 default:
318 return NULL;
319 }

322void AnimateWindowManager::reverseAnimateWindow( void )
323{
324
325 m_reverse = TRUE;
326 m_needsUpdate = TRUE;
327 ProcessAnimateWindow *processAnim = NULL;
328
329 UnsignedInt maxDelay = 0;
330 AnimateWindowList::iterator it = m_winMustFinishList.begin();
331 while (it != m_winMustFinishList.end())
332 {
333 AnimateWindow *animWin = *it;
334 if (!animWin)
335 {
336 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
337 return;
338 }
339 if(animWin->getDelay() > maxDelay)
340 maxDelay = animWin->getDelay();
341 it ++;
342 }
343
344 it = m_winMustFinishList.begin();
345 while (it != m_winMustFinishList.end())
346 {
347 AnimateWindow *animWin = *it;
348 if (!animWin)
349 {
350 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
351 return;
352 }
353 // Run the window through the processAnim's init function.
354 processAnim = getProcessAnimate( animWin->getAnimType() );
355 if(processAnim)
356 {
357 processAnim->initReverseAnimateWindow( animWin, maxDelay );
358 }
359
360 animWin->setFinished(FALSE);
361 it ++;
362 }
363
364 it = m_winList.begin();
365
366 while (it != m_winList.end())
367 {
368 AnimateWindow *animWin = *it;
369 if (!animWin)
370 {
371 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
372 return;
373 }
374 processAnim = getProcessAnimate( animWin->getAnimType() );
375
376 if(processAnim)
377 processAnim->initReverseAnimateWindow(animWin);
378 animWin->setFinished(FALSE);
379 it ++;
380 }

384void AnimateWindowManager::resetToRestPosition( void )
385{
386
387 m_reverse = TRUE;
388 m_needsUpdate = TRUE;
389
390 AnimateWindowList::iterator it = m_winMustFinishList.begin();
391 while (it != m_winMustFinishList.end())
392 {
393 AnimateWindow *animWin = *it;
394 if (!animWin)
395 {
396 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));
397 return;
398 }
399 ICoord2D restPos = animWin->getRestPos();
400 GameWindow *win = animWin->getGameWindow();
401 if(win)
402 win->winSetPosition(restPos.x, restPos.y);
403 it ++;
404 }
405 it = m_winList.begin();
406 while (it != m_winList.end())
407 {
408 AnimateWindow *animWin = *it;
409 if (!animWin)
410 {
411 DEBUG_CRASH(("There's No AnimateWindow in the AnimateWindow List"));

414 ICoord2D restPos = animWin->getRestPos();
415 GameWindow *win = animWin->getGameWindow();
