50void NetCommandList::appendList(NetCommandList *list) {
51 if (list == NULL) {
52 return;
53 }
54
55 // Need to do it this way because of the reference counting that needs to happen in appendMessage.
56 NetCommandRef *msg = list->getFirstMessage();
57 NetCommandRef *next = NULL;
58 while (msg != NULL) {
59 next = msg->getNext();
60 NetCommandRef *temp = addMessage(msg->getCommand());
61 if (temp != NULL) {
62 temp->setRelay(msg->getRelay());
63 }
64
65 msg = next;
66 }

79void NetCommandList::removeMessage(NetCommandRef *msg) {
80 if (m_lastMessageInserted == msg) {
81 m_lastMessageInserted = msg->getNext();
82 }
83
84 if (msg->getPrev() != NULL) {
85 msg->getPrev()->setNext(msg->getNext());
86 }
87 if (msg->getNext() != NULL) {
88 msg->getNext()->setPrev(msg->getPrev());
89 }
90
91 if (msg == m_first) {
92 m_first = msg->getNext();
93 }
94 if (msg == m_last) {
95 m_last = msg->getPrev();
96 }
97
98 msg->setNext(NULL);
99 msg->setPrev(NULL);

129NetCommandRef * NetCommandList::addMessage(NetCommandMsg *cmdMsg) {
130 if (cmdMsg == NULL) {
131 DEBUG_ASSERTCRASH(cmdMsg != NULL, ("NetCommandList::addMessage - command message was NULL"));
132 return NULL;
133 }
134
135// UnsignedInt id = cmdMsg->getID();
136
137 NetCommandRef *msg = NEW_NETCOMMANDREF(cmdMsg);
138
139 if (m_first == NULL) {
140 // this is the first node, so we don't have to worry about ordering it.
141 m_first = msg;
142 m_last = msg;
143 m_lastMessageInserted = msg;
144 return msg;
145 }
146
147 if (m_lastMessageInserted != NULL) {
148 // Messages that are inserted in order should just be put in one right after the other.
149 // So saving the placement of the last message inserted can give us a huge boost in
150 // efficiency.
151 NetCommandRef *theNext = m_lastMessageInserted->getNext();
152 if ((m_lastMessageInserted->getCommand()->getNetCommandType() == msg->getCommand()->getNetCommandType()) &&
153 (m_lastMessageInserted->getCommand()->getPlayerID() == msg->getCommand()->getPlayerID()) &&
154 (m_lastMessageInserted->getCommand()->getID() < msg->getCommand()->getID()) &&
155 ((theNext == NULL) || ((theNext->getCommand()->getNetCommandType() > msg->getCommand()->getNetCommandType()) ||
156 (theNext->getCommand()->getPlayerID() > msg->getCommand()->getPlayerID()) ||
157 (theNext->getCommand()->getID() > msg->getCommand()->getID())))) {
158
159 // Make sure this command isn't already in the list.
160 if (isEqualCommandMsg(m_lastMessageInserted->getCommand(), msg->getCommand())) {
161
162 // This command is already in the list, don't duplicate it.
163 msg->deleteInstance();
164 msg = NULL;
165 return NULL;
166 }
167
168 if (theNext == NULL) {
169 // this means that m_lastMessageInserted == m_last, so m_last should point to the msg that is being inserted.
170 msg->setNext(m_lastMessageInserted->getNext());
171 msg->setPrev(m_lastMessageInserted);
172 m_lastMessageInserted->setNext(msg);
173 m_lastMessageInserted = msg;
174 m_last = msg;
175 } else {
176 msg->setNext(m_lastMessageInserted->getNext());
177 msg->setPrev(m_lastMessageInserted);
178 m_lastMessageInserted->setNext(msg);
179 msg->getNext()->setPrev(msg);
180 m_lastMessageInserted = msg;
181 }
182 return msg;
183 }
184 }
185
186 if (msg->getCommand()->getNetCommandType() > m_last->getCommand()->getNetCommandType()) {
187 // easy optimization for a command that goes at the end of the list
188 // since they are likely to be added in order.
189
190 // Make sure this command isn't already in the list.
191 if (isEqualCommandMsg(m_last->getCommand(), msg->getCommand())) {
192
193 // This command is already in the list, don't duplicate it.
194 msg->deleteInstance();
195 msg = NULL;
196 return NULL;
197 }
198
199 msg->setPrev(m_last);
200 msg->setNext(NULL);
201 m_last->setNext(msg);
202 m_last = msg;
203 m_lastMessageInserted = msg;
204 return msg;
205 }
206
207 if (msg->getCommand()->getNetCommandType() < m_first->getCommand()->getNetCommandType()) {
208 // Make sure this command isn't already in the list.
209 if (isEqualCommandMsg(m_first->getCommand(), msg->getCommand())) {
210
211 // This command is already in the list, don't duplicate it.
212 msg->deleteInstance();
213 msg = NULL;
214 return NULL;
215 }
216
217 // The command goes at the head of the list.
218 msg->setNext(m_first);
219 msg->setPrev(NULL);
220 m_first->setPrev(msg);
221 m_first = msg;
222 m_lastMessageInserted = msg;
223 return msg;
224 }
225
226
227 // Find the start of the command type we're looking for.
228 NetCommandRef *tempmsg = m_first;
229 while ((tempmsg != NULL) && (msg->getCommand()->getNetCommandType() > tempmsg->getCommand()->getNetCommandType())) {
230 tempmsg = tempmsg->getNext();
231 }
232
233 if (tempmsg == NULL) {
234 // Make sure this command isn't already in the list.
235 if (isEqualCommandMsg(m_last->getCommand(), msg->getCommand())) {
236
237 // This command is already in the list, don't duplicate it.
238 msg->deleteInstance();
239 msg = NULL;
240 return NULL;
241 }
242
243 // message goes at the end of the list.
244 msg->setPrev(m_last);
245 msg->setNext(NULL);
246 m_last->setNext(msg);
247 m_last = msg;
248 m_lastMessageInserted = msg;
249 return msg;
250 }
251
252 // Now find the player position. munkee.
253 while ((tempmsg != NULL) && (msg->getCommand()->getNetCommandType() == tempmsg->getCommand()->getNetCommandType()) && (msg->getCommand()->getPlayerID() > tempmsg->getCommand()->getPlayerID())) {
254 tempmsg = tempmsg->getNext();
255 }
256
257 if (tempmsg == NULL) {
258 // Make sure this command isn't already in the list.
259 if (isEqualCommandMsg(m_last->getCommand(), msg->getCommand())) {
260
261 // This command is already in the list, don't duplicate it.
262 msg->deleteInstance();
263 msg = NULL;
264 return NULL;
265 }
266
267 // message goes at the end of the list.
268 msg->setPrev(m_last);
269 msg->setNext(NULL);
270 m_last->setNext(msg);
271 m_last = msg;
272 m_lastMessageInserted = msg;
273 return msg;
274 }
275
276 // Find the position within the player's section based on the command ID.
277 // If the command type doesn't require a command ID, sort by whatever it should be sorted by.
278 while ((tempmsg != NULL) && (msg->getCommand()->getNetCommandType() == tempmsg->getCommand()->getNetCommandType()) && (msg->getCommand()->getPlayerID() == tempmsg->getCommand()->getPlayerID()) && (msg->getCommand()->getSortNumber() > tempmsg->getCommand()->getSortNumber())) {
279 tempmsg = tempmsg->getNext();
280 }
281
282 if (tempmsg == NULL) {
283 // Make sure this command isn't already in the list.
284 if (isEqualCommandMsg(m_last->getCommand(), msg->getCommand())) {
285
286 // This command is already in the list, don't duplicate it.
287 msg->deleteInstance();
288 msg = NULL;
289 return NULL;
290 }
291
292 // This message goes at the end of the list.
293 msg->setPrev(m_last);
294 msg->setNext(NULL);
295 m_last->setNext(msg);
296 m_last = msg;
297 m_lastMessageInserted = msg;
298 return msg;
299 }
300
301 if (tempmsg == m_first) {
302 // Make sure this command isn't already in the list.
303 if (isEqualCommandMsg(m_first->getCommand(), msg->getCommand())) {
304
305 // This command is already in the list, don't duplicate it.
306 msg->deleteInstance();
307 return NULL;
308 }
309
310 // This message goes at the head of the list.
311 msg->setNext(m_first);
312 msg->setPrev(NULL);
313 m_first->setPrev(msg);
314 m_first = msg;
315 m_lastMessageInserted = msg;
316 return msg;
317 }
318
319 // Make sure this command isn't already in the list.
320 if (isEqualCommandMsg(tempmsg->getCommand(), msg->getCommand())) {
321
322 // This command is already in the list, don't duplicate it.
323 msg->deleteInstance();
324 msg = NULL;
325 return NULL;
326 }
327
328 // Insert message before tempmsg.
329 msg->setNext(tempmsg);
330 msg->setPrev(tempmsg->getPrev());
331 msg->getPrev()->setNext(msg);
332 tempmsg->setPrev(msg);
333 m_lastMessageInserted = msg;
334
335 return msg;
336}

373Bool NetCommandList::isEqualCommandMsg(NetCommandMsg *msg1, NetCommandMsg *msg2) {
374 if (DoesCommandRequireACommandID(msg1->getNetCommandType()) != DoesCommandRequireACommandID(msg2->getNetCommandType())) {
375 return FALSE;
376 }
377
378 // At this point we know that the commands both do or do not require a command id.
379 // Do or do not, there is no try.
380 if (DoesCommandRequireACommandID(msg1->getNetCommandType())) {
381 // Are the commands from the same player?
382 if (msg1->getPlayerID() != msg2->getPlayerID()) {
383 return FALSE;
384 }
385
386 // Do they have the same command ID?
387 if (msg1->getID() != msg2->getID()) {
388 return FALSE;
389 }
390 return TRUE;
391 }
392
393 // If we've gotten this far, we know that the commands do not require a command id.
394 // So now our equality checking becomes type-specific.
395
396 // Are they the same type?
397 if (msg1->getNetCommandType() != msg2->getNetCommandType()) {
398 return FALSE;
399 }
400
401 // Are they from the same player?
402 if (msg1->getPlayerID() != msg2->getPlayerID()) {
403 return FALSE;
404 }
405
406 // They are the same type and from the same player.
407 // Time for the type specific stuff.
408 if (msg1->getNetCommandType() == NETCOMMANDTYPE_ACKSTAGE1) {
409 NetAckStage1CommandMsg *ack1 = (NetAckStage1CommandMsg *)msg1;
410 NetAckStage1CommandMsg *ack2 = (NetAckStage1CommandMsg *)msg2;
411
412 if (ack1->getOriginalPlayerID() != ack2->getOriginalPlayerID()) {
413 return FALSE;
414 }
415
416 if (ack1->getCommandID() != ack2->getCommandID()) {
417 return FALSE;
418 }
419 return TRUE;
420 }
421
422 // They are the same type and from the same player.
423 // Time for the type specific stuff.
424 if (msg1->getNetCommandType() == NETCOMMANDTYPE_ACKSTAGE2) {
425 NetAckStage2CommandMsg *ack1 = (NetAckStage2CommandMsg *)msg1;
426 NetAckStage2CommandMsg *ack2 = (NetAckStage2CommandMsg *)msg2;
427
428 if (ack1->getOriginalPlayerID() != ack2->getOriginalPlayerID()) {
429 return FALSE;
430 }
431
432 if (ack1->getCommandID() != ack2->getCommandID()) {
433 return FALSE;
434 }
435 return TRUE;
436 }
437
438 // They are the same type and from the same player.
439 // Time for the type specific stuff.
440 if (msg1->getNetCommandType() == NETCOMMANDTYPE_ACKBOTH) {
441 NetAckBothCommandMsg *ack1 = (NetAckBothCommandMsg *)msg1;
442 NetAckBothCommandMsg *ack2 = (NetAckBothCommandMsg *)msg2;
443
444 if (ack1->getOriginalPlayerID() != ack2->getOriginalPlayerID()) {
445 return FALSE;
446 }
447
448 if (ack1->getCommandID() != ack2->getCommandID()) {
449 return FALSE;
450 }
451 return TRUE;
452 }
453
454 return FALSE;
