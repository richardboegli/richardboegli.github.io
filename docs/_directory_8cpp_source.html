32static void TimetToFileTime( time_t t, FILETIME& ft )
33{
34 LONGLONG ll = Int32x32To64(t, 10000000) + 116444736000000000;
35 ft.dwLowDateTime = (DWORD) ll;
36 ft.dwHighDateTime = ll >>32;
37}

39static time_t FileTimeToTimet( const FILETIME& ft )
40{
41 LONGLONG ll = (ft.dwHighDateTime << 32) + ft.dwLowDateTime - 116444736000000000;
42 ll /= 10000000;
43 return (time_t)ll;
44}

61Directory::Directory( const AsciiString& dirPath ) : m_dirPath(dirPath)
62{
63 WIN32_FIND_DATA item; // search item
64 HANDLE hFile; // handle for search resources
65 char currDir[ MAX_PATH ];
66
67 // sanity
68 if( m_dirPath.isEmpty() )
69 {
70 DEBUG_LOG(( "Empty dirname\n"));
71 return;
72 }
73
74 // save the current directory
75 GetCurrentDirectory( MAX_PATH, currDir );
76
77 // switch into the directory provided
78 if( SetCurrentDirectory( m_dirPath.str() ) == 0 )
79 {
80 DEBUG_LOG(( "Can't set directory '%s'\n", m_dirPath.str() ));
81 return;
82 }
83
84 // go through each item in the output directory
85 bool done = false;
86 hFile = FindFirstFile( "*", &item);
87 if( hFile == INVALID_HANDLE_VALUE )
88 {
89 DEBUG_LOG(( "Can't search directory '%s'\n", m_dirPath.str() ));
90 done = true;
91 }
92
93 FileInfo info;
94
95 while (!done)
96 {
97 // if this is a subdirectory keep the name around till the end
98 if( BitTest( item.dwFileAttributes, FILE_ATTRIBUTE_DIRECTORY ) )
99 {
100 if ( strcmp( item.cFileName, "." ) && strcmp( item.cFileName, ".." ) )
101 {
102 info.set(item);
103 m_subdirs.insert( info );
104 }
105 }
106 else
107 {
108 info.set(item);
109 m_files.insert( info );
110 }
111
112 if ( FindNextFile( hFile, &item ) == 0 )
113 {
114 done = true;
115 }
116 }
117
118 // close search
119 FindClose( hFile );
120
121 // restore the working directory to what it was when we started here
122 SetCurrentDirectory( currDir );
123}
