161//-----------------------------------------------------------------------------
162//* TintEnvelope handles the fading of the tint color up, down stable etc...
163//* assumes that 0,0,0, is the color for the AT REST state, used as decay target
164//* works like an ADSR envelope,
165//* except that SUSTAIN and RELEASE are randomly (or never) triggered, externally
166class TintEnvelope : public MemoryPoolObject, public Snapshot
167{
168 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(TintEnvelope, "TintEnvelope" )
169public:
170
171 TintEnvelope(void);
172 void update(void);
173 void play(const RGBColor *peak,
174 UnsignedInt atackFrames = DEF_ATTACK_FRAMES,
175 UnsignedInt decayFrames = DEF_DECAY_FRAMES,
176 UnsignedInt sustainAtPeak = DEF_SUSTAIN_FRAMES ); // ask MLorenzen
177 void sustain(void) { m_envState = ENVELOPE_STATE_SUSTAIN; }
178 void release(void) { m_envState = ENVELOPE_STATE_DECAY; }
179 void rest(void) { m_envState = ENVELOPE_STATE_REST; } // goes away now!
180 Bool isEffective() const { return m_affect; }
181 const Vector3* getColor() const { return &m_currentColor; }
182
183protected:
184
185 // snapshot methods
186 virtual void crc( Xfer *xfer );
187 virtual void xfer( Xfer *xfer );
188 virtual void loadPostProcess( void );
189
190private:
191
192 void setAttackFrames(UnsignedInt frames);
193 void setDecayFrames( UnsignedInt frames);
194 void setPeakColor( const RGBColor *peak) {m_peakColor = Vector3( peak->red, peak->green, peak->blue );};
195 void setPeakColor( Real r, Real g, Real b ) {m_peakColor.Set( r, g, b );};
196

199 ENVELOPE_STATE_REST,
200 ENVELOPE_STATE_ATTACK,
201 ENVELOPE_STATE_DECAY,
202 ENVELOPE_STATE_SUSTAIN
203 };
204
205 Vector3 m_attackRate;
206 Vector3 m_decayRate;
207 Vector3 m_peakColor;
208 Vector3 m_currentColor;
209 UnsignedInt m_sustainCounter;
210 Byte m_envState;
211 Bool m_affect;
212};
