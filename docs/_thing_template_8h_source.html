143class AudioArray
144{
145public:
146 DynamicAudioEventRTS* m_audio[TTAUDIO_COUNT];
147
148 AudioArray()
149 {
150 for (Int i = 0; i < TTAUDIO_COUNT; ++i)
151 m_audio[i] = NULL;
152 }
153
154 ~AudioArray()
155 {
156 for (Int i = 0; i < TTAUDIO_COUNT; ++i)
157 if (m_audio[i])
158 m_audio[i]->deleteInstance();
159 }
160
161 AudioArray(const AudioArray& that)
162 {
163 for (Int i = 0; i < TTAUDIO_COUNT; ++i)
164 {
165 if (that.m_audio[i])
166 m_audio[i] = newInstance(DynamicAudioEventRTS)(*that.m_audio[i]);
167 else
168 m_audio[i] = NULL;
169 }
170 }
171
172 AudioArray& operator=(const AudioArray& that)
173 {
174 if (this != &that)
175 {
176 for (Int i = 0; i < TTAUDIO_COUNT; ++i)
177 {
178 if (that.m_audio[i])
179 {
180 if (m_audio[i])
181 *m_audio[i] = *that.m_audio[i];
182 else
183 m_audio[i] = newInstance(DynamicAudioEventRTS)(*that.m_audio[i]);
184 }
185 else
186 {
187 m_audio[i] = NULL;
188 }
189 }
190 }
191 return *this;
192 }
193};
