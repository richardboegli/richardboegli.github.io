19/******************************************************************************
20*
21* FILE
22* $Archive: $
23*
24* DESCRIPTION
25* RefPtr<> and RefPtrConst<> are const-friendly, polymorphic reference
26* counting smart pointers.
27*
28* The pointed-to class must be derived from RefCount.
29*
30* RefPtr<X> replaces X*
31* RefPtrConst<X> replaces const X*
32*
33* Dynamic_Cast<X> replaces dynamic_cast<X*> and dynamic_cast<X&>
34* Reinterpret_Cast<X> replaces reinterpret_cast<X*> and reinterpret_cast<X&>
35* Const_Cast<X> replaces const_cast<X*> and const_cast<X&>
36*
37* IsValid() replaces (x != NULL)
38*
39* Member function Attach() or assigning RefPtr<X>() will NULL a pointer.
40*
41* Generally, RefPtr<> and RefPtrConst<> behave like their raw pointer
42* counterparts, except of course they are reference counted and will delete
43* the pointed-to object when the last reference is lost. The major
44* syntatical differences are the use of RefPtrConst<> to represent a pointer
45* to a constant object (I found it impossible to represent this within rc_ptr)
46* and the use of the upper-case cast functions (it is not possible to overload
47* these built-in functions).
48*
49* An explicit goal of this class is to completely avoid the "new" and "delete"
50* operators in client code. The constructors for these pointers are private;
51* they are friends of the pointed-to class. This forces the use of Factory
52* Method functions (or similar) in the pointed-to class. Pointed-to classes
53* should make the constructor protected or private to disallow clients from
54* creating an instance with "new". If this is done, it becomes very difficult
55* for the client to accidentally leak objects and/or misuse the pointed-to
56* class or the reference counting pointers.
57*

59* Steven Clinard

79class RefPtrBase
80 {
81 public:
82 inline bool operator==(const RefPtrBase& rhs) const
83 {return (mRefObject == rhs.mRefObject);}
84
85 inline bool operator!=(const RefPtrBase& rhs) const
86 {return !operator==(rhs);}
87
88 inline bool IsValid(void) const
89 {return (mRefObject != NULL);}
90
91 inline void Detach(void)
92 {
93 if (IsValid())
94 {
95 mRefObject->Release();
96 mRefObject = NULL;
97 }
98 }
99
100 protected:
101 RefPtrBase()
102 : mRefObject(NULL)
103 {}
104
105 RefPtrBase(RefCounted* object)
106 : mRefObject(object)
107 {
108 assert((mRefObject == NULL) || (mRefObject->mRefCount == 0));
109
110 if (IsValid())
111 {
112 mRefObject->AddReference();
113 }
114 }
115
116 RefPtrBase(const RefPtrBase& object)
117 : mRefObject(object.mRefObject)
118 {
119 assert(false); // why is this being called?
120
121 if (IsValid())
122 {
123 mRefObject->AddReference();
124 }
125 }
126
127 virtual ~RefPtrBase()
128 {Detach();}
129
130 const RefPtrBase& operator=(const RefPtrBase&);
131
132 inline RefCounted* const GetRefObject(void)
133 {return mRefObject;}
134
135 inline const RefCounted* const GetRefObject(void) const
136 {return mRefObject;}
137
138 inline void Attach(RefCounted* object)
139 {
140 // If objects are different
141 if (object != mRefObject)
142 {
143 // Add reference to new object
144 if (object != NULL)
145 {
146 object->AddReference();
147 }
148
149 // Release reference to old object
150 Detach();
151
152 // Assign new object
153 mRefObject = object;
154 }
155 }
156
157 private:
158 RefCounted* mRefObject;
159
160 template<typename Derived>
161 friend RefPtr<Derived> Dynamic_Cast(RefPtrBase&);
162
163 template<typename Type>
164 friend RefPtr<Type> Reinterpret_Cast(RefPtrBase&);
165 };
