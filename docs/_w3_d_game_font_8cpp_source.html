71// W3DFontLibrary::loadFontData ===============================================
73//=============================================================================
74Bool W3DFontLibrary::loadFontData( GameFont *font )
75{
76 FontCharsClass *fontChar;
77
78 // sanity
79 if( font == NULL )
80 return FALSE;
81
82 if ((UnsignedInt)font->pointSize > 100) //sanity check the size - anything over 100 is probably wrong. -MW
83 fontChar = NULL;
84 else
85 { // get the font data from the asset manager
86 fontChar = WW3DAssetManager::
87 Get_Instance()->Get_FontChars( font->nameString.str(), font->pointSize,
88 font->bold ? true : false );
89 }
90
91 if( fontChar == NULL )
92 {
93
94 DEBUG_LOG(( "W3D load font: unable to find font '%s' from asset manager\n",
95 font->nameString.str() ));
96 DEBUG_ASSERTCRASH(fontChar, ("Missing or Corrupted Font. Pleas see log for details"));
97 return FALSE;
98
99 } // end if
100
101 // assign font data
102 font->fontData = fontChar;
103 font->height = fontChar->Get_Char_Height();
104
105 FontCharsClass *unicodeFontChar = NULL;
106
107 // load unicode of same point size
108 if(TheGlobalLanguageData)
109 unicodeFontChar = WW3DAssetManager::
110 Get_Instance()->Get_FontChars( TheGlobalLanguageData->m_unicodeFontName.str(), font->pointSize,
111 font->bold ? true : false );
112 else
113 unicodeFontChar = WW3DAssetManager::
114 Get_Instance()->Get_FontChars( "Arial Unicode MS", font->pointSize,
115 font->bold ? true : false );
116
117 if ( unicodeFontChar )
118 {
119 fontChar->AlternateUnicodeFont = unicodeFontChar;
120 }
121
122 // all done and loaded
123 return TRUE;
124
125} // end loadFont

127// W3DFontLibrary::releaseFontData ============================================
129//=============================================================================
130void W3DFontLibrary::releaseFontData( GameFont *font )
131{
132
133 // presently we don't need to do anything because fonts are handled in
134 // the W3D asset manager which is all taken for of us
135 if (font && font->fontData)
136 {
137 if(((FontCharsClass *)(font->fontData))->AlternateUnicodeFont)
138 ((FontCharsClass *)(font->fontData))->AlternateUnicodeFont->Release_Ref();
139 ((FontCharsClass *)(font->fontData))->Release_Ref();
140 }
141 font->fontData = NULL;
142
143} // end releaseFont
