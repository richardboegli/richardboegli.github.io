95void INI::parseCampaignDefinition( INI *ini )
96{
97 AsciiString name;
98 Campaign *campaign;
99
100 // read the name
101 const char* c = ini->getNextToken();
102 name.set( c );
103
104 // find existing item if present
105 DEBUG_ASSERTCRASH( TheCampaignManager, ("parseCampaignDefinition: Unable to Get TheCampaignManager\n") );
106 if( !TheCampaignManager )
107 return;
108
109 // If we have a previously allocated Campaign
110 campaign = TheCampaignManager->newCampaign( name );
111
112 // sanity
113 DEBUG_ASSERTCRASH( campaign, ("parseCampaignDefinition: Unable to allocate campaign '%s'\n", name.str()) );
114
115 // parse the ini definition
116 ini->initFromINI( campaign, TheCampaignManager->getFieldParse() );
117
118} // end parseCampaignDefinition

149Mission *Campaign::newMission( AsciiString name )
150{
151 MissionListIt it;
152 it = m_missions.begin();
153 name.toLower();
154 while(it != m_missions.end())
155 {
156 Mission *mission = *it;
157 if(mission->m_name.compare(name) == 0)
158 {
159 m_missions.erase( it );
160 mission->deleteInstance();
161 break;
162 }
163 else
164 ++it;
165 }
166 Mission *newMission = newInstance(Mission);
167 newMission->m_name.set(name);
168 m_missions.push_back(newMission);
169 return newMission;

172//-----------------------------------------------------------------------------
173Mission *Campaign::getMission( AsciiString missionName )
174{
175 if(missionName.isEmpty())
176 return NULL;
177 MissionListIt it;
178 it = m_missions.begin();
179 // we've reached the end of the campaign
180 while(it != m_missions.end())
181 {
182 Mission *mission = *it;
183 if(mission->m_name.compare(missionName) == 0)
184 return mission;
185 ++it;
186 }
187 DEBUG_ASSERTCRASH(FALSE, ("getMission couldn't find %s", missionName.str()));
188 return NULL;
189}

427void CampaignManager::xfer( Xfer *xfer )
428{
429
430 // version
431 const XferVersion currentVersion = 5;
432 XferVersion version = currentVersion;
433 xfer->xferVersion( &version, currentVersion );
434
435 // current campaign
436 AsciiString currentCampaign;
437 if( m_currentCampaign )
438 currentCampaign = m_currentCampaign->m_name;
439 xfer->xferAsciiString( &currentCampaign );
440
441 // current mission
442 AsciiString currentMission;
443 if( m_currentMission )
444 currentMission = m_currentMission->m_name;
445 xfer->xferAsciiString( &currentMission );
446
447 // version 2 and above has rank points!
448 if(version >= 2)
449 xfer->xferInt( &m_currentRankPoints );
450
451 if(version >= 3)
452 xfer->xferUser( &m_difficulty, sizeof(m_difficulty) );
453
454 // when loading, need to set the current campaign and mission
455 if( xfer->getXferMode() == XFER_LOAD )
456 setCampaignAndMission( currentCampaign, currentMission );
457
458 if( version >= 4 )
459 {
460 // The saving of SkirmishInfo in GameStateMap is in the "wrong" place, but I can't move it.
461 // We need to ensure this is saved in a mission save as well as a normal save.
462 // Singletons are bad. THis is here because a) it is one of two blocks in a Mission Save
463 // b) It is not the block that is loaded for every save to get desc's in populating the saveload window.
464 // So it is here. <sob> I've got nowhere else to go!
465 Bool isChallengeCampaign = m_currentCampaign ? m_currentCampaign->m_isChallengeCampaign : FALSE;
466 xfer->xferBool(&isChallengeCampaign);
467
468 if( isChallengeCampaign )
469 {
470 if( TheChallengeGameInfo==NULL )
471 {
472 TheChallengeGameInfo = NEW SkirmishGameInfo;
473 TheChallengeGameInfo->init();
474 TheChallengeGameInfo->clearSlotList();
475 TheChallengeGameInfo->reset();
476 }
477 xfer->xferSnapshot(TheChallengeGameInfo);
478 }
479 else
480 {
481 if( TheChallengeGameInfo )
482 {
483 delete TheChallengeGameInfo;
484 TheChallengeGameInfo = NULL;
485 }
486 }
487 }
488
489 if( version >= 5 )
490 {
491 // Even more singleton goodness. TheChallengeGenerals is not a subsystem, nor even a snapshot.
492 // I need to know what side I am for use by The Continue and Play Again buttons. I can only just
493 // reach in and save it here.
494 Int playerTemplateNum = TheChallengeGenerals->getCurrentPlayerTemplateNum();
495 xfer->xferInt(&playerTemplateNum);
496 m_xferChallengeGeneralsPlayerTemplateNum = playerTemplateNum;
497 }
498
499} // end xfer
