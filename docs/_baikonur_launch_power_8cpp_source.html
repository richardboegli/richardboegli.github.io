100// ------------------------------------------------------------------------------------------------
101void BaikonurLaunchPower::doSpecialPowerAtLocation( const Coord3D *loc, Real angle, UnsignedInt commandOptions )
102{
103 if (getObject()->isDisabled())
104 return;
105
106 const BaikonurLaunchPowerModuleData *data = getBaikonurLaunchPowerModuleData();
107
108 // call the base class action cause we are *EXTENDING* functionality
109 SpecialPowerModule::doSpecialPowerAtLocation( loc, angle, commandOptions );
110
111 //Create the detonation
112 const ThingTemplate *thing = TheThingFactory->findTemplate( data->m_detonationObject );
113 if( thing )
114 {
115 Object *detonation = TheThingFactory->newObject( thing, getObject()->getTeam() );
116 if( detonation )
117 {
118 detonation->setPosition( loc );
119 }
120 }
121}
