65void FXListDie::onDie( const DamageInfo *damageInfo )
66{
67 if (!isUpgradeActive())
68 return;
69 if (!isDieApplicable(damageInfo))
70 return;
71 const FXListDieModuleData* d = getFXListDieModuleData();
72
73 UpgradeMaskType activation, conflicting;
74 getUpgradeActivationMasks( activation, conflicting );
75 Object *obj = getObject();
76 if( obj->getObjectCompletedUpgradeMask().testForAny( conflicting ) )
77 {
78 return;
79 }
80 if( obj->getControllingPlayer() && obj->getControllingPlayer()->getCompletedUpgradeMask().testForAny( conflicting ) )
81 {
82 return;
83 }
84
85 if (d->m_defaultDeathFX)
86 {
87 if (d->m_orientToObject)
88 {
89 Object *damageDealer = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
90 FXList::doFXObj(getFXListDieModuleData()->m_defaultDeathFX, getObject(), damageDealer);
91 }
92 else
93 {
94 FXList::doFXPos(getFXListDieModuleData()->m_defaultDeathFX, getObject()->getPosition());
95 }
96 }
97}
