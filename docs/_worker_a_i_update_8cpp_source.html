250UpdateSleepTime WorkerAIUpdate::update( void )
251{
252
253 //
254 // NOTE: Any changes to DozerAIUpdate::* you probably want to reflect and copy into
255 // WorkerAIUPdate:* as well ... sigh
256 //
257
258 //
259 // now that we're really executing we have all the necessary object modules in place to
260 // correctly create a state machine and set the default state
261 //
262 // create all the machines if they don't yet exist
263 createMachines();
264
265 // DO NOT set us as being to able to move with super precision off grid locations
266 // Causes workers to get stuck. jba.
267 //if( getCurLocomotor() )
268 //getCurLocomotor()->setUltraAccurate( TRUE );
269
270 // extend the normal AI system
271 AIUpdateInterface::update();
272
273 // do nothing if we're dead
275 if( getObject()->isEffectivelyDead() )
276 return UPDATE_SLEEP_NONE;
277
278 // run our own state machine, and the appropriate sub machine
279 m_workerMachine->updateStateMachine();
280
281 if( m_workerMachine->getCurrentStateID() == AS_DOZER )
282 {
283
284 // get and validate our current task
285 DozerTask currentTask = getCurrentTask();
286 if( currentTask != DOZER_TASK_INVALID )
287 {
288 ObjectID taskTarget = getTaskTarget( currentTask );
289 Object *targetObject = TheGameLogic->findObjectByID( taskTarget );
290 Bool invalidTask = FALSE;
291
292 // validate the task and the target
293 if( currentTask == DOZER_TASK_REPAIR &&
294 TheActionManager->canRepairObject( getObject(), targetObject, getLastCommandSource() ) == FALSE )
295 invalidTask = TRUE;
296
297 // cancel the task if it's now invalid
298 if( invalidTask == TRUE )
299 cancelTask( currentTask );
300
301 } // end if
302
303 // update dozer behavior
304 m_dozerMachine->updateStateMachine();
305
306 } // end if
307 else
308 {
309 m_supplyTruckStateMachine->updateStateMachine();
310 // If we are harvesting, we can be diverted to clear mines. jba.
311 getObject()->setWeaponSetFlag(WEAPONSET_MINE_CLEARING_DETAIL);//maybe go clear some mines, if I feel like it
312 }
313 return UPDATE_SLEEP_NONE;
314}

327Object *WorkerAIUpdate::construct( const ThingTemplate *what,
328 const Coord3D *pos,
329 Real angle,
330 Player *owningPlayer,
331 Bool isRebuild )
332{
333
334 // !!! NOTE: If you modify this you must modify the dozer too !!!
335 // !!! Graham: Please please please have inspiration for how to *not* duplicate this code
336 // GS - Construct needs to be an AI primitive. Inheriting off of AIUpdate means you are writing a
337 // master brain that will call AI primitives on the object, not something that does stuff itself.
338 // SupplyTruckAI decides who to call AIDock on. Worker should decide to AIDock or AIConstruct
339 // or AIRepair. Dozer should just use the latter two. No construction logic should be in
340 // the inherited AIUpdates at all.
341
342 // create our machines if they don't yet exist
344
345 m_isRebuild = isRebuild;
346
347 createMachines();
348
349 // sanity
350 if( what == NULL || pos == NULL || owningPlayer == NULL )
351 return NULL;
352
353 // sanity
354 DEBUG_ASSERTCRASH( getObject()->getControllingPlayer() == owningPlayer,
355 ("Dozer::Construct - The controlling player of the Dozer is not the owning player passed in\n") );
356
357 // if we're not rebuilding, we have a few checks to pass first for sanity
358 if( isRebuild == FALSE )
359 {
360
361 // AI has weaker restriction on building
362 Bool dozerIsAI = owningPlayer->getPlayerType() == PLAYER_COMPUTER;
363 if( dozerIsAI )
364 {
365
366 // validate the the position to build at is valid
367 if( TheBuildAssistant->isLocationLegalToBuild( pos, what, angle,
368 BuildAssistant::CLEAR_PATH |
369 BuildAssistant::NO_OBJECT_OVERLAP,
370 getObject(), NULL ) != LBC_OK )
371 return NULL;
372
373 } // end if
374 else
375 {
376
377 // make sure the player is capable of building this
378 if( TheBuildAssistant->canMakeUnit( getObject(), what ) != CANMAKE_OK )
379 return NULL;
380
381 // validate the the position to build at is valid
382 if( TheBuildAssistant->isLocationLegalToBuild( pos, what, angle,
383 BuildAssistant::TERRAIN_RESTRICTIONS |
384 BuildAssistant::CLEAR_PATH |
385 BuildAssistant::NO_OBJECT_OVERLAP |
386 BuildAssistant::SHROUD_REVEALED,
387 getObject(), NULL ) != LBC_OK )
388 return NULL;
389
390 } // end else
391
392 } // end if
393
394 // what will our initial status bits
395 ObjectStatusMaskType statusBits = MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNDER_CONSTRUCTION );
396 if( isRebuild )
397 statusBits.set( OBJECT_STATUS_RECONSTRUCTING );
398
399 // create an object at the destination location
400 Object *obj = TheThingFactory->newObject( what, owningPlayer->getDefaultTeam(), statusBits );
401
402 // even though we haven't actually built anything yet, this keeps things tidy
403 obj->setProducer( getObject() );
404 obj->setBuilder( getObject() );
405
406 // leave the supply truck state and now behave like a dozer.
407 exitingSupplyTruckState();
408
409 // take the required money away from the player
410 if( isRebuild == FALSE )
411 {
412 Money *money = owningPlayer->getMoney();
413
414 money->withdraw( what->calcCostToBuild( owningPlayer ) );
415
416 } // end if
417
418 //
419 // set a bit that this object is under construction, it is important to do this early
420 // before the hooks add/subtract power from a player are executed
421 //
422 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNDER_CONSTRUCTION ) );
423
424 // initialize object
425 obj->setPosition( pos );
426 obj->setOrientation( angle );
427
428 // Flatten the terrain underneath the object, then adjust to the flattened height. jba.
429 TheTerrainLogic->flattenTerrain(obj);
430 Coord3D adjustedPos = *pos;
431 adjustedPos.z = TheTerrainLogic->getGroundHeight(pos->x, pos->y);
432 obj->setPosition(&adjustedPos);
433
434 // Note - very important that we add to map AFTER we flatten terrain. jba.
435 TheAI->pathfinder()->addObjectToPathfindMap( obj );
436
437 // "callback" event for structure created (note that it's not yet "complete")
438 owningPlayer->onStructureCreated( getObject(), obj );
439
440 // set a construction percent for the new object to zero and a status for under construction
441 obj->setConstructionPercent( 0.0 );
442
443 // newly constructed objects start at one hit point
444 BodyModuleInterface *body = obj->getBodyModule();
445 body->internalChangeHealth( -body->getHealth() + 1.0f );
446
447 // set the model action state to awaiting construction
448 obj->clearAndSetModelConditionFlags(
449 MAKE_MODELCONDITION_MASK2(MODELCONDITION_PARTIALLY_CONSTRUCTED, MODELCONDITION_ACTIVELY_BEING_CONSTRUCTED),
450 MAKE_MODELCONDITION_MASK(MODELCONDITION_AWAITING_CONSTRUCTION)
451 );
452
453 // we have a construction pending
454 newTask( DOZER_TASK_BUILD, obj );
455
456 return obj;
457
458}

465{
466 if( m_workerMachine->getCurrentStateID() == AS_SUPPLY_TRUCK )

486Bool WorkerAIUpdate::canAcceptNewRepair( Object *obj )
487{
488
489 // sanity
490 if( obj == NULL )
491 return FALSE;
492
493 // if we're not repairing right now, we don't have any accept restrictions
494 if( getCurrentTask() != DOZER_TASK_REPAIR )
495 return TRUE;
496
497 // get current repair target
498 Object *currentRepair = TheGameLogic->findObjectByID( m_task[ DOZER_TASK_REPAIR ].m_targetObjectID );
499
500 if( currentRepair )
501 {
502
503 // check for same object
504 if( currentRepair == obj )
505 return FALSE;
506
507 // check for repairing any tower on the same bridge
508 if( currentRepair->isKindOf( KINDOF_BRIDGE_TOWER ) &&
509 obj->isKindOf( KINDOF_BRIDGE_TOWER ) )
510 {
511 BridgeTowerBehaviorInterface *currentTowerInterface = NULL;
512 BridgeTowerBehaviorInterface *newTowerInterface = NULL;
513
514 currentTowerInterface = BridgeTowerBehavior::getBridgeTowerBehaviorInterfaceFromObject( currentRepair );
515 newTowerInterface = BridgeTowerBehavior::getBridgeTowerBehaviorInterfaceFromObject( obj );
516
517 // sanity
518 if( currentTowerInterface == NULL || newTowerInterface == NULL )
519 {
520
521 DEBUG_CRASH(( "Unable to find bridge tower interface on object\n" ));
522 return FALSE;
523
524 } // end if
525
526 // if they are part of the same bridge, ignore this repair command
527 if( currentTowerInterface->getBridgeID() == newTowerInterface->getBridgeID() )
528 return FALSE;
529
530 } // end if
531
532 } // end if, currentRepair object exists
533
534 // all is well
535 return TRUE;
536
537} // end canAcceptNewRepair

616void WorkerAIUpdate::newTask( DozerTask task, Object* target )
617{
618
619 // sanity
620 DEBUG_ASSERTCRASH( task >= 0 && task < DOZER_NUM_TASKS, ("Illegal dozer task '%d'\n", task) );
621
622 // sanity
623 if( target == NULL )
624 return;
625
626 m_preferredDock = INVALID_ID; // If we are dozing, we don't want any supply truck stuff going on. jba.
627
628 //
629 // special check for the build task, we should never be given more than one of them ...
630 // for the other tasks we just forget what we were doing and the new target takes
631 // precedence for the task
632 //
633 if( task == DOZER_TASK_BUILD || task == DOZER_TASK_REPAIR )
634 {
635
636 // handle getting two tasks
637 if( isTaskPending( task ) == TRUE )
638 cancelTask( task );
639
640 // get our object
641 Object *me = getObject();
642
643 Coord3D position;
644 target = DozerAIUpdate::findGoodBuildOrRepairPositionAndTarget(me, target, position);
645 if (target == NULL)
646 return; // could happen for some bridges
647
648 //
649 // for building, we say that even "thinking" about building or rebuilding an object
650 // sets us as the current builder of that object. this allows any dozers that are
651 // ordered later to resume construction on something to see that somebody is already taking
652 // care of it and then they won't be even try to resume a build since we don't allow
653 // multiple dozers/workers to double up on construction efforts
654 //
655 if( task == DOZER_TASK_BUILD )
656 target->setBuilder( me );
657
658 m_dockPoint[ task ][ DOZER_DOCK_POINT_START ].valid = TRUE;
659 m_dockPoint[ task ][ DOZER_DOCK_POINT_START ].location = position;
660 m_dockPoint[ task ][ DOZER_DOCK_POINT_ACTION ].valid = TRUE;
661 m_dockPoint[ task ][ DOZER_DOCK_POINT_ACTION ].location = position;
662 m_dockPoint[ task ][ DOZER_DOCK_POINT_END ].valid = TRUE;
663 m_dockPoint[ task ][ DOZER_DOCK_POINT_END ].location = position;
664
665 } // end if, build task
666
667 // set the new task target and the frame in which we got this order
668 m_task[ task ].m_targetObjectID = target->getID();
669 m_task[ task ].m_taskOrderFrame = TheGameLogic->getFrame();
670
671 // reset the dozer behavior so that it can re-evluate which task to continue working on
672 m_dozerMachine->resetToDefaultState();
673
674 // reset the workermachine, if we've been acting like a supply truck
675 if( m_workerMachine->getCurrentStateID() == AS_SUPPLY_TRUCK )
676 {
677 // We've been given a Dozer specific order that the Supply Truck machine doesn't recognize
678 // as BUSY (because this command also recognizes its own busy and is likewise waiting).
679 // Explicitly slap it upside the head.
680 if( getObject()->getAIUpdateInterface() )
681 {
682 getObject()->getAIUpdateInterface()->aiIdle(CMD_FROM_AI);
683 }
684 m_workerMachine->setState( AS_DOZER );
685 // To clarify, I leave supply truck mode when I notice I am doing something not supply
686 // truck related. When given a construct command, I wait to do anything until I notice
687 // I'm not busy. Both states are being polite, so I must force the switch.
688 }
689
690}
