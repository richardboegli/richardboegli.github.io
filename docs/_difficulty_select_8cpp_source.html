85static void SetDifficultyRadioButton( void )
86{
87 OptionPreferences pref;
88 if (!TheScriptEngine)
89 {
90 s_AIDiff = DIFFICULTY_NORMAL;
91 }
92 else
93 {
94 switch (pref.getCampaignDifficulty())
95 {
96 case DIFFICULTY_EASY:
97 {
98 GadgetRadioSetSelection(radioButtonEasyAI, FALSE);
99 s_AIDiff = DIFFICULTY_EASY;
100 break;
101 }
102 case DIFFICULTY_NORMAL:
103 {
104 GadgetRadioSetSelection(radioButtonMediumAI, FALSE);
105 s_AIDiff = DIFFICULTY_NORMAL;
106 break;
107 }
108 case DIFFICULTY_HARD:
109 {
110 GadgetRadioSetSelection(radioButtonHardAI, FALSE);
111 s_AIDiff = DIFFICULTY_HARD;
112 break;
113 }
114
115 default:
116 {
117 DEBUG_CRASH(("unrecognized difficulty level in the script engine"));
118 }
119
120 }
121 } // if (TheScriptEngine)
122

214WindowMsgHandledType DifficultySelectSystem( GameWindow *window, UnsignedInt msg,
215 WindowMsgData mData1, WindowMsgData mData2 )
216{
217
218 switch( msg )
219 {
220
221 // --------------------------------------------------------------------------------------------
222 case GWM_CREATE:
223 {
224 break;
225
226 } // end create
227
228 //---------------------------------------------------------------------------------------------
229 case GWM_DESTROY:
230 {
231 break;
232
233 } // end case
234
235 // --------------------------------------------------------------------------------------------
236 case GWM_INPUT_FOCUS:
237 {
238
239 // if we're givin the opportunity to take the keyboard focus we must say we want it
240 if( mData1 == TRUE )
241 *(Bool *)mData2 = TRUE;
242
243 return MSG_HANDLED;
244
245 } // end input
246
247 //---------------------------------------------------------------------------------------------
248 case GBM_SELECTED:
249 {
250 // this isn't fixed yet
251 GameWindow *control = (GameWindow *)mData1;
252 Int controlID = control->winGetWindowId();
253
254 if ( controlID == buttonOkID )
255 {
256 OptionPreferences pref;
257 pref.setCampaignDifficulty(s_AIDiff);
258 pref.write();
259 //TheScriptEngine->setGlobalDifficulty(s_AIDiff); // CANNOT DO THIS! REPLAYS WILL BREAK!
260 WindowLayout *layout = window->winGetLayout();
261 layout->destroyWindows();
262 layout->deleteInstance();
263 setupGameStart(TheCampaignManager->getCurrentMap(), s_AIDiff);
264 // start the game
265 }
266 else if ( controlID == buttonCancelID )
267 {
268 TheCampaignManager->setCampaign( AsciiString::TheEmptyString );
269 TheWindowManager->winUnsetModal(window);
270 WindowLayout *layout = window->winGetLayout();
271 layout->destroyWindows();
272 layout->deleteInstance();
273
274 }
275 else if ( controlID == radioButtonEasyAIID )
276 {
277 s_AIDiff = DIFFICULTY_EASY;
278 }
279 else if ( controlID == radioButtonMediumAIID )
280 {
281 s_AIDiff = DIFFICULTY_NORMAL;
282 }
283 else if ( controlID == radioButtonHardAIID )
284 {
285 s_AIDiff = DIFFICULTY_HARD;
286 }
287
288 break;
289
290 } // end selected
291
292 default:
293 return MSG_IGNORED;
294
295 } // end switch
296
297 return MSG_HANDLED;
