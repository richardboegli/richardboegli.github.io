71void TeamBehavior::updateScript(NameKeyType keyScript, int idcScript)
72{
73 CComboBox *pCombo = (CComboBox*)GetDlgItem(idcScript);
74 CString txt;
75 Int curSel = pCombo->GetCurSel();
76 pCombo->GetLBText(curSel, txt);
77 AsciiString comboText = AsciiString(txt);
78 if (comboText == NONE_STRING) {
79 comboText.clear();
80 }
81 m_teamDict->setAsciiString(keyScript, comboText);
82}

84void TeamBehavior::setupScript(NameKeyType keyScript, int idcScript)
85{
86 CComboBox *pCombo = (CComboBox*)GetDlgItem(idcScript);
87 // Load the subroutine scripts into the combo box.
88 EditParameter::loadScripts(pCombo, true);
89 Int stringNdx = pCombo->AddString(NONE_STRING);
90 pCombo->SetFocus();
91 Bool exists;
92 AsciiString script = m_teamDict->getAsciiString(keyScript, &exists);
93 if (exists && !script.isEmpty()) {
94 Int ndx = pCombo->FindStringExact(-1, script.str());
95 if (ndx != CB_ERR) {
96 stringNdx = ndx;
97 } else {
98 AsciiString badName = "*";
99 badName.concat(script);
100 stringNdx = pCombo->AddString(badName.str());
101 }
102 }
103 pCombo->SetCurSel(stringNdx);
104}

107BOOL TeamBehavior::OnInitDialog()
108{
109 CPropertyPage::OnInitDialog();
110
111 setupScript(TheKey_teamOnCreateScript, IDC_ON_CREATE_SCRIPT);
112 setupScript(TheKey_teamOnIdleScript, IDC_ON_IDLE_SCRIPT);
113 setupScript(TheKey_teamEnemySightedScript, IDC_ON_ENEMY_SIGHTED);
114 setupScript(TheKey_teamOnDestroyedScript, IDC_ON_DESTROYED);
115 setupScript(TheKey_teamAllClearScript, IDC_ON_ALL_CLEAR);
116 setupScript(TheKey_teamOnUnitDestroyedScript, IDC_ON_UNIT_DESTROYED_SCRIPT);
117
118 Bool exists;
119
120 CButton *pCheck = (CButton *) GetDlgItem(IDC_TRANSPORTS_RETURN);
121 Bool transportsReturn = m_teamDict->getBool(TheKey_teamTransportsReturn, &exists);
122 pCheck->SetCheck(transportsReturn?1:0);
123
124 pCheck = (CButton *) GetDlgItem(IDC_BASE_DEFENSE);
125 if( pCheck )
126 {
127 Bool baseDef = m_teamDict->getBool(TheKey_teamIsBaseDefense, &exists);
128 pCheck->SetCheck(baseDef?1:0);
129 }
130
131 pCheck = (CButton *) GetDlgItem(IDC_PERIMETER_DEFENSE);
132 if( pCheck )
133 {
134 Bool perimeter = m_teamDict->getBool(TheKey_teamIsPerimeterDefense, &exists);
135 pCheck->SetCheck(perimeter?1:0);
136 }
137
138 pCheck = (CButton *) GetDlgItem(IDC_AVOID_THREATS);
139 Bool avoid = m_teamDict->getBool(TheKey_teamAvoidThreats, &exists);
140 pCheck->SetCheck(avoid?1:0);
141
142 pCheck = (CButton *) GetDlgItem(IDC_ATTACK_COMMON_TARGET);
143 Bool attack = m_teamDict->getBool(TheKey_teamAttackCommonTarget, &exists);
144 pCheck->SetCheck(attack?1:0);
145
146 AsciiString description;
147
148 CWnd *pWnd = GetDlgItem(IDC_PERCENT_DESTROYED);
149 Real threshold = m_teamDict->getReal(TheKey_teamDestroyedThreshold, &exists);
150 if (!exists) threshold = 0.5f;
151 Int percent = floor((threshold*100)+0.5);
152 description.format("%d", percent);
153 pWnd->SetWindowText(description.str());
154
155 CComboBox *pCombo = (CComboBox*)GetDlgItem(IDC_ENEMY_INTERACTIONS);
156 pCombo->SetCurSel(m_teamDict->getInt(TheKey_teamAggressiveness, &exists) - AI_SLEEP);
157
158
159 return FALSE; // return TRUE unless you set the focus to a control
160 // EXCEPTION: OCX Property Pages should return FALSE
161}

197void TeamBehavior::OnPerimeterDefense()
198{
199 CButton *pCheck = (CButton *) GetDlgItem(IDC_PERIMETER_DEFENSE);
200 if( pCheck )
201 {
202 Bool checked = pCheck->GetCheck()==1;
203 m_teamDict->setBool(TheKey_teamIsPerimeterDefense, checked);
204 if (checked) { // Can't be both base & perimeter defense.
205 pCheck = (CButton *) GetDlgItem(IDC_BASE_DEFENSE);
206 if( pCheck )
207 {
208 pCheck->SetCheck(0);
209 m_teamDict->setBool(TheKey_teamIsBaseDefense, false);
210 }
211 }
212 }
