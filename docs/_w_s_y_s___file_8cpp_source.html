130Bool File::open( const Char *filename, Int access )
131{
132 if( m_open )
133 {
134 return FALSE;
135 }
136
137 setName( filename );
138
139 if( (access & ( TEXT | BINARY)) == ( TEXT | BINARY ))
140 {
141 // illegal access
142 return FALSE;
143 }
144
145 if ( !(access & (READ|APPEND)) )
146 {
147 access |= TRUNCATE;
148 }
149
150 if ( (access & (READ|WRITE)) == 0 )
151 {
152 access = READ;
153 }
154
155 if ( (access & (TEXT|BINARY)) == 0 )
156 {
157 access = BINARY;
158 }
159
160 m_access = access;
161 m_open = TRUE;
162 return TRUE;
163}
