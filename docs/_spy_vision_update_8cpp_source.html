66void SpyVisionUpdate::activateSpyVision( UnsignedInt duration )
67{
68 UnsignedInt now = TheGameLogic->getFrame();
69 if( duration == 0 )
70 m_deactivateFrame = UINT_MAX;
71 else
72 m_deactivateFrame = now + duration;
73
74 doActivationWork( getObject()->getControllingPlayer(), TRUE );
75
76 if( duration == 0 )
77 setWakeFrame(getObject(), UPDATE_SLEEP_FOREVER);
78 else
79 setWakeFrame( getObject(), UPDATE_SLEEP(duration) );
80}

84void SpyVisionUpdate::setDisabledUntilFrame( UnsignedInt frame )
85{
86 UnsignedInt now = TheGameLogic->getFrame();
87 if( frame > now )
88 {
89 //Turn off the spy vision now since we are disabled
90 if( m_currentlyActive )
91 {
92 doActivationWork( getObject()->getControllingPlayer(), FALSE );
93 }
94
95 //When should we wake up?
96 m_disabledUntilFrame = frame;
97
98 //Mark it so we can turn it on again on next update.
99 m_resetTimersNextUpdate = TRUE;
100
101 //Now sleep until the disabled has expired. If it's expired again when we come back due to another
102 //sabotage or something else... we'll do the same thing over again.
103 setWakeFrame( getObject(), (UpdateSleepTime)(m_disabledUntilFrame - now) );
104 }
105 else
106 {
107 // Else it is a wakeup message. Update does the turning on.
108 m_disabledUntilFrame = now;
109 m_resetTimersNextUpdate = TRUE;
110 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
111 }
112}

138UpdateSleepTime SpyVisionUpdate::update( void )
139{
140 const SpyVisionUpdateModuleData *data = getSpyVisionUpdateModuleData();
141 UnsignedInt now = TheGameLogic->getFrame();
142
143 //Once disable has complete and we are waking up from it, then reset the interval so
144 //it has to wait before going on again. If it doesn't have an interval, we turn it on
145 //immediately, but only for self powered objects.
146 if( m_resetTimersNextUpdate )
147 {
148 m_resetTimersNextUpdate = FALSE;
149
150 if( data->m_selfPowered )
151 {
152 if( data->m_selfPoweredInterval == 0 )
153 {
154 //It's an always on self-powered special. So turn it back on now.
155 doActivationWork( getObject()->getControllingPlayer(), TRUE );
156 return UPDATE_SLEEP( UPDATE_SLEEP_FOREVER );
157 }
158 else
159 {
160 //Reset the interval timer via sleeping (so we have to wait a longer time before going on)
161 return UPDATE_SLEEP( data->m_selfPoweredInterval );
162 }
163 }
164 }
165
166 if( m_currentlyActive && (m_deactivateFrame <= TheGameLogic->getFrame()) )
167 {
168 // Turn off SpyVision.
169 doActivationWork( getObject()->getControllingPlayer(), FALSE );
170
171 m_deactivateFrame = 0;
172 }
173 else if( !m_currentlyActive && data->m_selfPowered )
174 {
175 doActivationWork( getObject()->getControllingPlayer(), TRUE );
176 if( data->m_selfPoweredDuration == 0 )
177 m_deactivateFrame = UINT_MAX;
178 else
179 m_deactivateFrame = now + data->m_selfPoweredDuration;
180 }
181
182 if( data->m_selfPowered )
183 {
184 if( m_currentlyActive )
185 return UPDATE_SLEEP(data->m_selfPoweredDuration);
186 else
187 return UPDATE_SLEEP(data->m_selfPoweredInterval);
188 }
189
190 return UPDATE_SLEEP_FOREVER;
191}

194void SpyVisionUpdate::doActivationWork( Player *playerToSetFor, Bool setting )
195{
196 const SpyVisionUpdateModuleData *data = getSpyVisionUpdateModuleData();
197 if( playerToSetFor == NULL || ThePlayerList == NULL )
198 return;
199
200 for (Int i=0; i < ThePlayerList->getPlayerCount(); ++i)
201 {
202 Player *player = ThePlayerList->getNthPlayer(i);
203 if( playerToSetFor->getRelationship(player->getDefaultTeam()) == ENEMIES )
204 {
205 player->setUnitsVisionSpied( setting, data->m_spyOnKindof, playerToSetFor->getPlayerIndex() );
206 }
207 }
208
209 m_currentlyActive = setting;
210}
