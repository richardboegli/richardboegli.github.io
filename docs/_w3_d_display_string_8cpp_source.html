164void W3DDisplayString::draw( Int x, Int y, Color color, Color dropColor, Int xDrop, Int yDrop )
165{
166 Bool needNewPolys = FALSE;
167
168 // sanity
169 if( getTextLength() == 0 )
170 return; // nothing to draw
171
172 // if our font or text has changed we need to build a new sentence
173 if( m_fontChanged || m_textChanged )
174 {
175 if(m_useHotKey)
176 {
177 m_textRenderer.Set_Hot_Key_Parse(TRUE);
178 m_textRenderer.Build_Sentence( getText().str(), &m_hotKeyPos.x, &m_hotKeyPos.y );
179 m_hotkey.translate(TheHotKeyManager->searchHotKey(getText()));
180 if(!m_hotkey.isEmpty())
181 m_textRendererHotKey.Build_Sentence(m_hotkey.str(), NULL, NULL);
182 else
183 {
184 m_useHotKey = FALSE;
185 m_textRendererHotKey.Reset();
186 }
187 }
188 else
189 m_textRenderer.Build_Sentence( getText().str(), NULL, NULL );
190 m_fontChanged = FALSE;
191 m_textChanged = FALSE;
192 needNewPolys = TRUE;
193
194 } // end if
195
196 //
197 // if our position has changed, or our colors have chagned, or our
198 // text data has changed, we need to redo the texture quads
199 //
200 if( needNewPolys ||
201 x != m_textPos.x ||
202 y != m_textPos.y ||
203 color != m_currTextColor ||
204 dropColor != m_currDropColor )
205 {
206
207 // save the new attributes of the text position and color
208 m_textPos.x = x;
209 m_textPos.y = y;
210 m_currTextColor = color;
211 m_currDropColor = dropColor;
212
213 // reset the quads
214 m_textRenderer.Reset_Polys();
215
216 // draw the shadow
217 m_textRenderer.Set_Location( Vector2( m_textPos.x + xDrop, m_textPos.y + yDrop) );
218 m_textRenderer.Draw_Sentence( m_currDropColor );
219
220 // draw the text
221 m_textRenderer.Set_Location( Vector2( m_textPos.x, m_textPos.y ) );
222 m_textRenderer.Draw_Sentence( m_currTextColor );
223
224 if(m_useHotKey)
225 {
226 m_textRendererHotKey.Reset_Polys();
227 m_textRendererHotKey.Set_Location( Vector2( m_textPos.x + m_hotKeyPos.x , m_textPos.y +m_hotKeyPos.y) );
228 m_textRendererHotKey.Draw_Sentence( m_hotKeyColor );
229 m_textRendererHotKey.Render();
230 }
231
232 } // end if
233
234 // render the text
235 m_textRenderer.Render();
236
237 // we are for sure using display resources now
238 if( TheGameClient )
239 usingResources( TheGameClient->getFrame() );
240
241} // end draw

258// DisplayString::appendChar ==================================================
260//=============================================================================
261
262Int W3DDisplayString::getWidth( Int charPos )
263{
264 FontCharsClass * font;
265 Int width = 0;
266 Int count = 0;
267
268 font = m_textRenderer.Peek_Font();
269
270 if ( font )
271 {
272 const WideChar *text = m_textString.str();
273 WideChar ch;
274
275 while ( (ch = *text++ ) != 0 && ( charPos == -1 || count < charPos ) )
276 {
277 if ( ch != (WideChar)'\n' )
278 {
279 width += font->Get_Char_Spacing( ch );
280 }
281 count++;
282 }
283 }
284
285 return width;
286}

288// W3DDisplayString::setFont ==================================================
290//=============================================================================
291void W3DDisplayString::setFont( GameFont *font )
292{
293
294 // sanity
295 if( font == NULL )
296 return;
297
298 // if the new font is the same as our existing font do nothing
299 if( m_font == font )
300 return;
301
302 // extending functionality
303 DisplayString::setFont( font );
304
305 // set the font in our renderer
306 m_textRenderer.Set_Font( static_cast<FontCharsClass *>(m_font->fontData) );
307
308 m_textRendererHotKey.Set_Font( static_cast<FontCharsClass *>(TheFontLibrary->getFont(font->nameString,font->pointSize, TRUE)->fontData) );
309 // recompute extents for text with new font
310 computeExtents();
311
312 // set flag telling us the font has changed since last render
313 m_fontChanged = TRUE;
314
315} // end setFont
