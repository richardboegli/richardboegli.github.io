67//DECLARE_PERF_TIMER(shadowsRender)
68void DoShadows(RenderInfoClass & rinfo, Bool stencilPass)
69{
70 //USE_PERF_TIMER(shadowsRender)
71 shadowCameraFrustum=&rinfo.Camera.Get_Frustum();
72 Int projectionCount=0;
73
74 //Projected shadows render first because they may fill the stencil buffer
75 //which will be used by the shadow volumes
76 if (stencilPass == FALSE && TheW3DProjectedShadowManager)
77 {
78 if (TheW3DShadowManager->isShadowScene())
79 projectionCount=TheW3DProjectedShadowManager->renderShadows(rinfo);
80 }
81
82 if (stencilPass == TRUE && TheW3DVolumetricShadowManager)
83 {
84
85// TheW3DShadowManager->loadTerrainShadows();
86
87 //This function gets called many times by the W3D renderer
88 //so we use this flag to make sure shadows rendered only once per frame.
89 if (TheW3DShadowManager->isShadowScene())
90 TheW3DVolumetricShadowManager->renderShadows(projectionCount);
91 }
92 if (TheW3DShadowManager && stencilPass) //reset so no more shadow processing this frame.
93 TheW3DShadowManager->queueShadows(FALSE);
94
95}

97W3DShadowManager::W3DShadowManager( void )
98{
99 DEBUG_ASSERTCRASH(TheW3DVolumetricShadowManager == NULL && TheW3DProjectedShadowManager == NULL,
100 ("Creating new shadow managers without deleting old ones"));
101
102 m_shadowColor = 0x7fa0a0a0;
103 m_isShadowScene = FALSE;
104 m_stencilShadowMask = 0; //all bits can be used for storing shadows.
105
106 Vector3 lightRay(-TheGlobalData->m_terrainLightPos[0].x,
107 -TheGlobalData->m_terrainLightPos[0].y, -TheGlobalData->m_terrainLightPos[0].z);
108 lightRay.Normalize();
109
110 LightPosWorld[0]=lightRay*SUN_DISTANCE_FROM_GROUND;
111
112 TheW3DVolumetricShadowManager = NEW W3DVolumetricShadowManager;
113 TheProjectedShadowManager = TheW3DProjectedShadowManager = NEW W3DProjectedShadowManager;
114}

175Shadow *W3DShadowManager::addShadow( RenderObjClass *robj, Shadow::ShadowTypeInfo *shadowInfo, Drawable *draw)
176{
177 ShadowType type = SHADOW_VOLUME;
178
179 if (shadowInfo)
180 type = shadowInfo->m_type;
181
182 switch(type)
183 {
184 case SHADOW_VOLUME:
185 if (TheW3DVolumetricShadowManager)
186 return (Shadow *)TheW3DVolumetricShadowManager->addShadow(robj, shadowInfo, draw);
187 break;
188 case SHADOW_PROJECTION:
189 case SHADOW_DECAL:
190 if (TheW3DProjectedShadowManager)
191 return (Shadow *)TheW3DProjectedShadowManager->addShadow(robj, shadowInfo, draw);
192 break;
193 default:
194 return NULL;
195 }
196
197 return NULL;
198}
