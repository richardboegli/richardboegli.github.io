97UpdateSleepTime FireWeaponUpdate::update( void )
98{
99
100 if ( TheGameLogic->getFrame() < m_initialDelayFrame )
101 return UPDATE_SLEEP_NONE;
102
103
104 // If my weapon is ready, shoot it.
105 if( isOkayToFire() )
106 {
107 m_weapon->forceFireWeapon( getObject(), getObject()->getPosition() );
108 }
109 return UPDATE_SLEEP_NONE;
110}

114Bool FireWeaponUpdate::isOkayToFire()
115{
116 const Object *me = getObject();
117 const FireWeaponUpdateModuleData *data = getFireWeaponUpdateModuleData();
118
119 if( m_weapon == NULL )
120 return FALSE;
121
122 // Weapon is reloading
123 if( m_weapon->getStatus() != READY_TO_FIRE )
124 return FALSE;
125
126 if( me->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) )
127 return FALSE; // no hitting with a 0% building, cheater
128
129 // Firing a real weapon surpresses this module
130 if( data->m_exclusiveWeaponDelay > 0 && ( TheGameLogic->getFrame() < (me->getLastShotFiredFrame() + data->m_exclusiveWeaponDelay) ) )
131 return FALSE;
132
133 return TRUE;
134}
