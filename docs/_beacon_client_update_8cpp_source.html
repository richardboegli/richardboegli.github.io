86static ParticleSystem* createParticleSystem( Drawable *draw )
87{
88 ParticleSystem *system = NULL;
89 if (draw)
90 {
91 Object *obj = draw->getObject();
92 if (obj)
93 {
94 AsciiString templateName;
95 templateName.format("BeaconSmoke%6.6X", (0xffffff & obj->getIndicatorColor()));
96 const ParticleSystemTemplate *particleTemplate = TheParticleSystemManager->findTemplate( templateName );
97
98 DEBUG_ASSERTCRASH(particleTemplate, ("Could not find particle system %s\n", templateName.str()));
99
100 if (particleTemplate)
101 {
102 system = TheParticleSystemManager->createParticleSystem( particleTemplate );
103 if (system)
104 system->attachToDrawable( draw );
105 }
106 else// This is a failsafe... if someone has monkeyed with the particle system names, or the MP house colors
107 {// THis this will whip up a new particle system to match the house color provided
108 templateName.format("BeaconSmokeFFFFFF");
109 const ParticleSystemTemplate *failsafeTemplate = TheParticleSystemManager->findTemplate( templateName );
110 DEBUG_ASSERTCRASH(failsafeTemplate, ("Doh, this is bad \n I Could not even find the white particle system to make a failsafe system out of."));
111 if (failsafeTemplate)
112 {
113 system = TheParticleSystemManager->createParticleSystem( failsafeTemplate );
114 if (system)
115 {
116 system->attachToDrawable( draw );
117 system->tintAllColors( obj->getIndicatorColor() );
118 }
119 }
120 }
121 }
122 }
123 return system;
124}

128void BeaconClientUpdate::hideBeacon( void )
129{
130 Drawable *draw = getDrawable();
131 if (draw)
132 {
133 draw->setDrawableHidden( true );
134 draw->setShadowsEnabled( false );
135 }
136
137 ParticleSystem *system;
138 if (draw && m_particleSystemID == INVALID_PARTICLE_SYSTEM_ID)
139 {
140 system = createParticleSystem( draw );
141 if (system)
142 m_particleSystemID = system->getSystemID();
143 }
144
145 // clean up particle system
146 if (m_particleSystemID != INVALID_PARTICLE_SYSTEM_ID)
147 {
148 ParticleSystem *system = TheParticleSystemManager->findParticleSystem( m_particleSystemID );
149
150 if( system )
151 system->stop();
152
153 } // end if
154
155// DEBUG_LOG(("in hideBeacon(): draw=%d, m_particleSystemID=%d\n", draw, m_particleSystemID));
156
157}

162void BeaconClientUpdate::clientUpdate( void )
163{
164 Drawable *draw = getDrawable();
165 if (!draw)
166 return;
167
168 if (m_particleSystemID == INVALID_PARTICLE_SYSTEM_ID)
169 {
170 ParticleSystem *system = createParticleSystem( draw );
171 if( system )
172 m_particleSystemID = system->getSystemID();
173 }
174
175 if (!draw->isDrawableEffectivelyHidden())
176 {
177 BeaconClientUpdateModuleData *moduleData = (BeaconClientUpdateModuleData *)getModuleData();
178 if (TheGameLogic->getFrame() > m_lastRadarPulse + moduleData->m_framesBetweenRadarPulses)
179 {
180 TheRadar->createEvent( draw->getPosition(), RADAR_EVENT_BEACON_PULSE, moduleData->m_radarPulseDuration * SECONDS_PER_LOGICFRAME_REAL );
181 m_lastRadarPulse = TheGameLogic->getFrame();
182 }
183 }
184}
