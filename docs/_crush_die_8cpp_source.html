48// Figure out which crush point was hit so the correct crushed object can be swapped in
49static CrushEnum crushLocationCheck( Object* crusherObject, Object* victimObject )
50{
51 if( (crusherObject == NULL) || (victimObject == NULL) )
52 return NO_CRUSH;
53
54 Bool frontCrushed = victimObject->getBodyModule()->getFrontCrushed();
55 Bool backCrushed = victimObject->getBodyModule()->getBackCrushed();
56
57// const Coord3D *dir = crusherObject->getUnitDirectionVector2D();
58 const Coord3D *otherDir = victimObject->getUnitDirectionVector2D();
59 const Coord3D *pos = crusherObject->getPosition();
60 const Coord3D *otherPos = victimObject->getPosition();
61
62 Real crushPointOffsetDistance = victimObject->getGeometryInfo().getMajorRadius() * 0.5;
63
64 Coord3D crushPointOffset;
65 crushPointOffset.x = otherDir->x * crushPointOffsetDistance;
66 crushPointOffset.y = otherDir->y * crushPointOffsetDistance;
67 crushPointOffset.z = 0;
68
69 Coord3D comparisonCoord;
70 Real dx, dy;
71
72 CrushEnum retval = NO_CRUSH;
73 Real bestDist = 99999;
74
75 // PhysicsCollide has already done the logic of which point to smoosh and waited until we crossed that point
76 // so at this point we just need to know which crush point is closest.
77
78 if( !frontCrushed && !backCrushed )
79 {
80 // Check the middle crush point
81 comparisonCoord = *otherPos;//copy so can move to each crush point
82
83 dx = comparisonCoord.x - pos->x;
84 dy = comparisonCoord.y - pos->y;
85 Real dist = (Real)( dx*dx + dy*dy );
86
87 //otherwise we want to make sure we get the closest valid crush point
88 retval = TOTAL_CRUSH;
89 bestDist = dist;
90 }
91
92 if( !frontCrushed )
93 {
94 // Check the front point.
95 comparisonCoord = *otherPos;
96 comparisonCoord.x += crushPointOffset.x;
97 comparisonCoord.y += crushPointOffset.y;
98
99 dx = comparisonCoord.x - pos->x;
100 dy = comparisonCoord.y - pos->y;
101 Real dist = (Real)( dx*dx + dy*dy );
102
103 if( dist < bestDist )//closer
104 {
105 if( backCrushed )
106 {
107 retval = TOTAL_CRUSH;
108 bestDist = dist;
109 }
110 else
111 {
112 retval = FRONT_END_CRUSH;
113 bestDist = dist;
114 }
115 }
116 }
117
118 if( !backCrushed )
119 {
120 // Check back point
121 comparisonCoord = *otherPos;
122 comparisonCoord.x -= crushPointOffset.x;
123 comparisonCoord.y -= crushPointOffset.y;
124
125 dx = comparisonCoord.x - pos->x;
126 dy = comparisonCoord.y - pos->y;
127 Real dist = (Real)( dx*dx + dy*dy );
128
129 if( dist < bestDist )//closer
130 {
131 if( frontCrushed )
132 {
133 retval = TOTAL_CRUSH;
134 bestDist = dist;
135 }
136 else
137 {
138 retval = BACK_END_CRUSH;
139 bestDist = dist;
140 }
141 }
142 }
143
144 return retval;
145}

163void CrushDie::onDie( const DamageInfo * damageInfo )
164{
165 if (!isDieApplicable(damageInfo))
166 return;
167
168 DEBUG_ASSERTCRASH(damageInfo->in.m_damageType == DAMAGE_CRUSH, ("this should only be used for crush damage\n"));
169 if (damageInfo->in.m_damageType != DAMAGE_CRUSH)
170 return;
171
172 Object *damageDealer = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
173 DEBUG_ASSERTCRASH(damageDealer,("You must have a damageDealer source for this effect"));
174
175 CrushEnum crushType = damageDealer ? crushLocationCheck(damageDealer, getObject()) : TOTAL_CRUSH;
176
177 if (crushType != NO_CRUSH)
178 {
179 if (getCrushDieModuleData()->m_crushSounds[crushType].getEventName().isEmpty() == false)
180 {
181 // be sure that 0==never, 100==always
182 // MDC: moving to GameLogicRandomValue. This does not need to be synced, but having it so makes searches *so* much nicer.
183 if (GameLogicRandomValue(0, 99) < getCrushDieModuleData()->m_crushSoundPercent[crushType])
184 {
185 AudioEventRTS crushSound(getCrushDieModuleData()->m_crushSounds[crushType]);
186 crushSound.setObjectID(getObject()->getID());
187 TheAudio->addAudioEvent(&crushSound);
188 }
189 }
190 {
191 Object *me = getObject();
192
193 if (me)
194 {
195 me->getBodyModule()->setFrontCrushed(crushType == TOTAL_CRUSH || crushType == FRONT_END_CRUSH);
196 me->getBodyModule()->setBackCrushed(crushType == TOTAL_CRUSH || crushType == BACK_END_CRUSH);
197
198 ModelConditionFlags newCrushed;
199 newCrushed.set(MODELCONDITION_FRONTCRUSHED, (crushType == TOTAL_CRUSH || crushType == FRONT_END_CRUSH));
200 newCrushed.set(MODELCONDITION_BACKCRUSHED, crushType == TOTAL_CRUSH || crushType == BACK_END_CRUSH);
201
202 me->getDrawable()->clearAndSetModelConditionFlags(
203 MAKE_MODELCONDITION_MASK2(MODELCONDITION_BACKCRUSHED, MODELCONDITION_FRONTCRUSHED),
204 newCrushed);
205 }
206 }
207 }
208}
