145//-------------------------------------------------------------------------------------------------
146// enable the appropriate buttons, make sure they aren't hidden, and set the correct images
147//-------------------------------------------------------------------------------------------------
148void setEnabledButtons()
149{
150 for (Int i = 0; i < NUM_GENERALS; i++)
151 {
152 const GeneralPersona* generals = TheChallengeGenerals->getChallengeGenerals();
153 buttonGeneralPosition[i]->winEnable(generals[i].isStartingEnabled());
154 buttonGeneralPosition[i]->winHide(! generals[i].isStartingEnabled());
155
156 Int templateNum = ThePlayerTemplateStore->getTemplateNumByName(generals[i].getPlayerTemplateName());
157 const PlayerTemplate *playerTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(templateNum);
158 if (playerTemplate)
159 {
160 const Image *enabledImage = TheMappedImageCollection->findImageByName( playerTemplate->getMedallionNormal() );
161 GadgetCheckBoxSetEnabledImage( buttonGeneralPosition[i], enabledImage );
162 if (enabledImage)
163 // image size keeps changing, so it'll drive the window size directly
164 buttonGeneralPosition[i]->winSetSize( enabledImage->getImageWidth(), enabledImage->getImageWidth() );
165
166 const Image *selectedImage = TheMappedImageCollection->findImageByName( playerTemplate->getMedallionSelected() );
167 GadgetCheckBoxSetHiliteUncheckedBoxImage( buttonGeneralPosition[i], selectedImage);
168 GadgetCheckBoxSetDisabledUncheckedBoxImage( buttonGeneralPosition[i], selectedImage);
169
170 const Image *hilitedImage = TheMappedImageCollection->findImageByName( playerTemplate->getMedallionHilite() );
171 GadgetCheckBoxSetHiliteImage( buttonGeneralPosition[i], hilitedImage);
172 }
173 }

177//-------------------------------------------------------------------------------------------------
178// sets the appropriate campaign for the chosen general
179//-------------------------------------------------------------------------------------------------
180void setGeneralCampaign( Int buttonIndex )
181{
182 if (buttonIndex < 0 || buttonIndex >= NUM_GENERALS)
183 return;
184
185 // determine which general and player template is selected and store it
186 const GeneralPersona* generals = TheChallengeGenerals->getChallengeGenerals();
187 TheCampaignManager->setCampaign(generals[buttonIndex].getCampaign());
188 Int templateNum = ThePlayerTemplateStore->getTemplateNumByName(generals[buttonIndex].getPlayerTemplateName());
189 TheChallengeGenerals->setCurrentPlayerTemplateNum(templateNum);
190
191 // set up the skirmish games single player slot
192 GameSlot slot;
193 const PlayerTemplate *playerTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(templateNum);
194 slot.setState(SLOT_PLAYER, playerTemplate->getDisplayName());
195 slot.setPlayerTemplate(templateNum);
196 TheChallengeGameInfo->setSlot(0, slot);
197}

231//-------------------------------------------------------------------------------------------------
232// update the intro button sequence UNFINISHED
233//-------------------------------------------------------------------------------------------------
234void updateButtonSequence(Int stepsPerUpdate)
235{
236 const static Int cleanupStates = 2;
237 if (buttonSequenceStep > NUM_GENERALS + cleanupStates)
238 return;
239
240 const GeneralPersona* generals = TheChallengeGenerals->getChallengeGenerals();
241
242 for (Int i = 0; i < stepsPerUpdate; i++)
243 {
244 // selected look
245 Int pos = buttonSequenceStep;
246 if (pos < NUM_GENERALS && !buttonGeneralPosition[pos]->winIsHidden())
247 {
248 Int templateNum = ThePlayerTemplateStore->getTemplateNumByName(generals[pos].getPlayerTemplateName());
249 const PlayerTemplate *playerTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(templateNum);
250 if (playerTemplate)
251 GadgetCheckBoxSetEnabledImage( buttonGeneralPosition[pos], TheMappedImageCollection->findImageByName( playerTemplate->getMedallionSelected() ) );
252 }
253
254 // mouseover look
255 if (--pos > 0 && pos < NUM_GENERALS && !buttonGeneralPosition[pos]->winIsHidden())
256 {
257 Int templateNum = ThePlayerTemplateStore->getTemplateNumByName(generals[pos].getPlayerTemplateName());
258 const PlayerTemplate *playerTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(templateNum);
259 if (playerTemplate)
260 GadgetCheckBoxSetEnabledImage( buttonGeneralPosition[pos], TheMappedImageCollection->findImageByName( playerTemplate->getMedallionHilite() ) );
261 }
262
263 // regular look
264 if (--pos > 0 && pos < NUM_GENERALS && !buttonGeneralPosition[pos]->winIsHidden())
265 {
266 Int templateNum = ThePlayerTemplateStore->getTemplateNumByName(generals[pos].getPlayerTemplateName());
267 const PlayerTemplate *playerTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(templateNum);
268 if (playerTemplate)
269 GadgetCheckBoxSetEnabledImage( buttonGeneralPosition[pos], TheMappedImageCollection->findImageByName( playerTemplate->getMedallionNormal() ) );
270 }
271
272 buttonSequenceStep++;
273 }
274}

276//-------------------------------------------------------------------------------------------------
277// update the general's bio, teletype style, call this as often as you want frames advanced
278// NOTE: static text windows must be initialized to empty before calling
279// accepts the number of frames to advance
280// returns TRUE if an update occurred (the update was necessary)
281//-------------------------------------------------------------------------------------------------
282Bool updateBio(Int frames)
283{
284 Bool ret = FALSE;
285
286 for (Int i = 0; i < frames; i++)
287 {
288 if (bioTextPosition < bioTotalLength)
289 {
290 UnicodeString text;
291 WideChar wChar;
292 GameWindow *window;
293 if (bioTextPosition < bioLine1.getLength())
294 {
295 text = GadgetStaticTextGetText(bioLine1Entry);
296 wChar = bioLine1.getCharAt(bioTextPosition);
297 window = bioLine1Entry;
298 }
299 else if (bioTextPosition < bioLine1.getLength() + bioLine2.getLength())
300 {
301 text = GadgetStaticTextGetText(bioLine2Entry);
302 wChar = bioLine2.getCharAt(bioTextPosition - bioLine1.getLength());
303 window = bioLine2Entry;
304 }
305 else if (bioTextPosition < bioLine1.getLength() + bioLine2.getLength() + bioLine3.getLength())
306 {
307 text = GadgetStaticTextGetText(bioLine3Entry);
308 wChar = bioLine3.getCharAt(bioTextPosition - bioLine1.getLength() - bioLine2.getLength());
309 window = bioLine3Entry;
310 }
311 else
312 {
313 text = GadgetStaticTextGetText(bioLine4Entry);
314 wChar = bioLine4.getCharAt(bioTextPosition - bioLine1.getLength() - bioLine2.getLength() - bioLine3.getLength());
315 window = bioLine4Entry;
316 }
317
318 text.concat(wChar);
319 GadgetStaticTextSetText(window, text);
320 bioTextPosition++;
321 ret = TRUE;
322 }
323 }
324
325 return ret;
326}

513

528 } // end escape
529
530 } // end switch( key )
531
532 } // end char
533
534 } // end switch( msg )

544WindowMsgHandledType ChallengeMenuSystem( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
545{
546 switch( msg )
547 {
548 case GWM_CREATE: break;
549
550 case GWM_DESTROY: break;
551
552 case GWM_INPUT_FOCUS:
553 {
554 // if we're givin the opportunity to take the keyboard focus we must say we want it
555 if( mData1 == TRUE )
556 *(Bool *)mData2 = TRUE;
557
558 return MSG_HANDLED;
559 }
560
561 case GBM_MOUSE_ENTERING:
562 {
563 // preview the bio for this position
564 GameWindow *control = (GameWindow *)mData1;
565 Int controlID = control->winGetWindowId();
566 Int buttonIndex = findPositionButton(controlID);
567 if( buttonIndex != -1 && buttonIndex != lastButtonIndex )
568 {
569 setGeneralBio(buttonIndex);
570
571 // special sound for Harvard
572 AudioEventRTS event( "GUILogoMouseOver" );
573 TheAudio->addAudioEvent( &event );
574
575 lastHilitedIndex = buttonIndex;
576 }
577
578 break;
579 }
580
581 case GBM_MOUSE_LEAVING:
582 {
583 // set the bio back to the selected one
584 GameWindow *control = (GameWindow *)mData1;
585 Int controlID = control->winGetWindowId();
586 Int buttonIndex = findPositionButton(controlID);
587 if( buttonIndex != -1 && buttonIndex != lastButtonIndex )
588 {
589 if (lastButtonIndex == -1)
590 {
591 // they're just browsing and haven't made a selection yet
592 // @todo: make this a nice transition
593// bioParent->winHide(TRUE);
594// buttonPlay->winHide(TRUE);
595 }
596
597 setGeneralBio(lastButtonIndex);
598 }
599 break;
600 }
601
602 case GBM_SELECTED:
603 {
604 UnicodeString filename;
605 GameWindow *control = (GameWindow *)mData1;
606 Int controlID = control->winGetWindowId();
607
608 // we don't need to handle these
609 if ( isAutoSelecting )
610 {
611 isAutoSelecting = FALSE;
612 break;
613 }
614
615 Int buttonIndex = findPositionButton(controlID);
616 if( buttonIndex != -1)
617 {
618 // partial radio button behavior (I'm not using actual radio buttons
619 // because it limits options on how the interface can work, which is a
620 // bad thing when the design is constantly in flux.)
621 // ...basically this just makes you have exactly one button selected
622 // once the first choice has been made.
623 if (lastButtonIndex != -1)
624 {
625 isAutoSelecting = TRUE;
626 GameWindow *lastControl = TheWindowManager->winGetWindowFromId( NULL, buttonGeneralPositionID[lastButtonIndex]);
627 GadgetCheckBoxToggle(lastControl);
628 }
629
630 // play audio to indicate selection
631 TheAudio->removeAudioEvent(lastSelectionSound);
632 TheAudio->removeAudioEvent(lastPreviewSound);
633 const GeneralPersona* generals = TheChallengeGenerals->getChallengeGenerals();
634 AudioEventRTS event( generals[buttonIndex].getPreviewSound() );
635 lastPreviewSound = TheAudio->addAudioEvent( &event );
636
637/*
638 // play audio to indicate selection
639 TheAudio->removeAudioEvent(lastSelectionSound);
640 TheAudio->removeAudioEvent(lastPreviewSound);
641 const GeneralPersona* generals = TheChallengeGenerals->getChallengeGenerals();
642 AudioEventRTS event( generals[buttonIndex].getSelectionSound() );
643 lastSelectionSound = TheAudio->addAudioEvent( &event );
644*/
645
646 lastButtonIndex = buttonIndex;
647
648 buttonPlay->winHide(FALSE);
649
650 //@todo: outro transitions
651 }
652 else if( controlID == buttonPlayID )
653 {
654 if( TheChallengeGameInfo == NULL )
655 {
656 // If this is NULL, then we must be on the way back out of this menu.
657 // Don't crash, just eat the button click message.
658 return MSG_HANDLED;
659 }
660
661 setGeneralCampaign(lastButtonIndex);
662 TheWritableGlobalData->m_pendingFile = TheCampaignManager->getCurrentMap();
663 TheChallengeGameInfo->setMap(TheCampaignManager->getCurrentMap());
664
665 // turn off the last button so the screen will be pristine when the user returns
666 isAutoSelecting = TRUE;
667 GameWindow *lastControl = TheWindowManager->winGetWindowFromId( NULL, buttonGeneralPositionID[lastButtonIndex]);
668 GadgetCheckBoxSetChecked(lastControl, FALSE);
669 lastButtonIndex = -1;
670// introAudioHasPlayed = FALSE;
671
672 buttonSequenceStep = 0;
673
674 if (TheGameLogic->isInGame())
675 TheGameLogic->clearGameData();
676
677 // If the campaign has been reset, so has the campaign difficulty. Restore it, just in case.
678 DEBUG_ASSERTCRASH(TheChallengeGenerals, ("TheChallengeGenerals are not initialized."));
679 if (TheChallengeGenerals)
680 {
681 TheCampaignManager->setGameDifficulty(TheChallengeGenerals->getCurrentDifficulty());
682 TheScriptEngine->setGlobalDifficulty(TheChallengeGenerals->getCurrentDifficulty());
683 }
684
685 // put a request in the message stream for a new game
686 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_NEW_GAME );
687 msg->appendIntegerArgument(GAME_SINGLE_PLAYER);
688 msg->appendIntegerArgument(TheCampaignManager->getGameDifficulty());
689 msg->appendIntegerArgument(TheCampaignManager->getRankPoints());
690
691
692 // Added so that, even though a ChallengeGame is really a SkirmishGame in SinglePlayerGame's clothing,
693 // GameEngine will still apply the default "FRAME CAP" as it does during "Solo Missions."
694 msg->appendIntegerArgument(LOGICFRAMES_PER_SECOND); // FPS limit
695
696 InitRandom(0);
697 }
698 else if( controlID == buttonBackID )
699 {
700 TheShell->pop();
701 }
702 break;
703 }
704
705 default: return MSG_IGNORED;
706 }
707
708 return MSG_HANDLED;
