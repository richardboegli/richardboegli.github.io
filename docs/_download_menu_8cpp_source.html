132HRESULT DownloadManagerMunkee::downloadFile( AsciiString server, AsciiString username, AsciiString password, AsciiString file, AsciiString localfile, AsciiString regkey, Bool tryResume )
133{
134 // see if we'll need to restart
135 if (strstr(localfile.str(), "patches\\") != NULL)
136 {
137 m_shouldQuitOnSuccess = true;
138 }
139
140 if (staticTextFile)
141 {
142 AsciiString bob = file;
143
144 // just get the filename, not the pathname
145 const char *tmp = bob.reverseFind('/');
146 if (tmp)
147 bob = tmp+1;
148 tmp = bob.reverseFind('\\');
149 if (tmp)
150 bob = tmp+1;
151
152 UnicodeString fileString;
153 fileString.translate(bob);
154 GadgetStaticTextSetText(staticTextFile, fileString);
155 }
156
157 password.format("-%s", password.str());
158 return DownloadManager::downloadFile( server, username, password, file, localfile, regkey, tryResume );
159}

164 MessageBoxOk(TheGameText->fetch("GUI:DownloadErrorTitle"), getErrorString(), errorCallback);
165 return ret;
166}
167HRESULT DownloadManagerMunkee::OnEnd()
168{
169 HRESULT ret = DownloadManager::OnEnd();
170
171 if (isFileQueuedForDownload())
172 {
173 return downloadNextQueuedFile();
174 }
175 if (m_shouldQuitOnSuccess)
176 MessageBoxOk(TheGameText->fetch("GUI:DownloadSuccessTitle"), TheGameText->fetch("GUI:DownloadSuccessMustQuit"), successQuitCallback);
177 else
178 MessageBoxOk(TheGameText->fetch("GUI:DownloadSuccessTitle"), TheGameText->fetch("GUI:DownloadSuccess"), successNoQuitCallback);
179 return ret;
180}

184HRESULT DownloadManagerMunkee::OnProgressUpdate( Int bytesread, Int totalsize, Int timetaken, Int timeleft )
185{
186 HRESULT ret = DownloadManager::OnProgressUpdate( bytesread, totalsize, timetaken, timeleft );
187
188 if (progressBarMunkee)
189 {
190 Int percent = bytesread * 100 / totalsize;
191 GadgetProgressBarSetProgress( progressBarMunkee, percent );
192 }
193
194 if (staticTextSize)
195 {
196 UnicodeString sizeString;
197 sizeString.format(TheGameText->fetch("GUI:DownloadBytesRatio"), bytesread, totalsize);
198 GadgetStaticTextSetText(staticTextSize, sizeString);
199 }
200 timeLeft = timeleft;
201 if (staticTextTime && GadgetStaticTextGetText(staticTextTime).isEmpty()) // only update immediately the first time
202 {
203 lastUpdate = time(NULL);
204 UnicodeString timeString;
205 if (timeleft)
206 {
207 DEBUG_ASSERTCRASH(timeleft > 0, ("Time left is negative!"));
208 timeleft = max(1, timeleft);
209 Int takenHour, takenMin, takenSec;
210 takenHour = timeleft / 60 / 60;
211 takenMin = timeleft / 60;
212 takenSec = timeleft % 60;
213 timeString.format(TheGameText->fetch("GUI:DownloadTimeLeft"), takenHour, takenMin, takenSec);
214 }
215 else
216 {
217 timeString = TheGameText->fetch("GUI:DownloadUnknownTime");
218 }
219 GadgetStaticTextSetText(staticTextTime, timeString);
220 }
221 return ret;
222}

327WindowMsgHandledType DownloadMenuInput( GameWindow *window, UnsignedInt msg,
328 WindowMsgData mData1, WindowMsgData mData2 )
329{
330
331 switch( msg )
332 {
333
334 // --------------------------------------------------------------------------------------------

339
340 switch( key )

342
343 // ----------------------------------------------------------------------------------------

345 {
346
347 //
348 // send a simulated selected event to the parent window of the
349 // back/exit button

354 NameKeyType buttonID = TheNameKeyGenerator->nameToKey( buttonName );

356
357 TheWindowManager->winSendSystemMsg( window, GBM_SELECTED,
358 (WindowMsgData)button, buttonID );
359
360 } // end if
361
362 // don't let key fall through anywhere else
363 return MSG_HANDLED;

365 } // end escape

380WindowMsgHandledType DownloadMenuSystem( GameWindow *window, UnsignedInt msg,
381 WindowMsgData mData1, WindowMsgData mData2 )
382{
383
384 switch( msg )
385 {
386
387 // --------------------------------------------------------------------------------------------
388 case GWM_CREATE:
389 {
390
391 break;
392
393 } // end create
394 //---------------------------------------------------------------------------------------------
395 case GWM_DESTROY:
396 {
397
398 break;
399
400 } // end case
401
402 //----------------------------------------------------------------------------------------------
403 case GWM_INPUT_FOCUS:
404 {
405
406 // if we're givin the opportunity to take the keyboard focus we must say we want it
407 if( mData1 == TRUE )
408 *(Bool *)mData2 = TRUE;
409
410 break;
411
412 } // end input
413 //---------------------------------------------------------------------------------------------
414 case GBM_SELECTED:
415 {
416 GameWindow *control = (GameWindow *)mData1;
417 Int controlID = control->winGetWindowId();
418
419 if( controlID == buttonCancelID )
420 {
421 HandleCanceledDownload();
422 closeDownloadWindow();
423 } // end if
424
425 break;
426
427 } // end selected
428
429 default:
430 return MSG_IGNORED;
431
432 } // end switch
433
434 return MSG_HANDLED;
