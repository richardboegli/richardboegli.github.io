383//-------------------------------------------------------------------------------------------------
384void FunctionLexicon::loadTable( TableEntry *table,
385 TableIndex tableIndex )
386{
387
388 // sanity
389 if( table == NULL )
390 return;
391
392 // loop through all entries
393 TableEntry *entry = table;
394 while( entry->name )
395 {
396
397 // assign key from name key based on name provided in table
398 entry->key = TheNameKeyGenerator->nameToKey( AsciiString(entry->name) );
399
400 // next table entry please
401 entry++;
402
403 } // end while
404
405 // assign table to the index specified
406 m_tables[ tableIndex ] = table;
407
408} // end loadTable

410//-------------------------------------------------------------------------------------------------
412//-------------------------------------------------------------------------------------------------
413void *FunctionLexicon::keyToFunc( NameKeyType key, TableEntry *table )
414{
415
416 // sanity
417 if( key == NAMEKEY_INVALID )
418 return NULL;
419
420 // search table for key
421 TableEntry *entry = table;
422 while( entry && entry->key != NAMEKEY_INVALID )
423 {
424
425 if( entry->key == key )
426 return entry->func;
427 entry++;
428
429 } // end if
430
431 return NULL; // not found
432
433} // end keyToFunc

435//-------------------------------------------------------------------------------------------------
439//-------------------------------------------------------------------------------------------------
440void *FunctionLexicon::findFunction( NameKeyType key, TableIndex index )
441{
442 void *func = NULL;
443
444 // sanity
445 if( key == NAMEKEY_INVALID )
446 return NULL;
447
448 // search ALL tables for function if the index paramater allows if
449 if( index == TABLE_ANY )
450 {
451
452 Int i;
453 for( i = 0; i < MAX_FUNCTION_TABLES; i++ )
454 {
455
456 func = keyToFunc( key, m_tables[ i ] );
457 if( func )
458 break; // exit for i
459
460 } // end for i
461
462 } // end if
463 else
464 {
465
466 // do NOT search all tables, just the one specified by the parameter
467 func = keyToFunc( key, m_tables[ index ] );
468
469 } // end else
470
471 // return function, if found
472 return func;
473
474} // end findFunction
