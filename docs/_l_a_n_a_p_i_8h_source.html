293#pragma pack(push, 1)
294struct LANMessage
295{
296 enum
297 {
298 // Locating everybody
299 MSG_REQUEST_LOCATIONS,
300 MSG_GAME_ANNOUNCE,
301 MSG_LOBBY_ANNOUNCE,
302
303 // Joining games
304 MSG_REQUEST_JOIN,
305 MSG_JOIN_ACCEPT,
306 MSG_JOIN_DENY,
307
308 // Leaving games
309 MSG_REQUEST_GAME_LEAVE,
310 MSG_REQUEST_LOBBY_LEAVE,
311
312 // Game options, chat, etc
313 MSG_SET_ACCEPT,
314 MSG_MAP_AVAILABILITY,
315 MSG_CHAT,
316 MSG_GAME_START,
317 MSG_GAME_START_TIMER,
318 MSG_GAME_OPTIONS,
319 MSG_INACTIVE,
320
321 MSG_REQUEST_GAME_INFO,
322
323 MSG_MAX
324 } LANMessageType;
325
326 WideChar name[g_lanPlayerNameLength+1];
327 char userName[g_lanLoginNameLength+1];
328 char hostName[g_lanHostNameLength+1];
329
330 // No additional data is required for REQUEST_LOCATIONS, LOBBY_ANNOUNCE,
331 // REQUEST_LOBBY_LEAVE, GAME_START.
332 union
333 {
334 // StartTimer is sent with GAME_START_TIMER
335 struct
336 {
337 Int seconds;
338 } StartTimer;
339
340 // GameJoined is sent with REQUEST_GAME_LEAVE
341 struct
342 {
343 WideChar gameName[g_lanGameNameLength+1];
344 } GameToLeave;
345
346 // GameInfo if sent with GAME_ANNOUNCE
347 struct
348 {
349 WideChar gameName[g_lanGameNameLength+1];
350 Bool inProgress;
351 char options[m_lanMaxOptionsLength+1];
352 Bool isDirectConnect;
353 } GameInfo;
354
355 // PlayerInfo is sent with REQUEST_GAME_INFO for direct connect games.
356 struct
357 {
358 UnsignedInt ip;
359 WideChar playerName[g_lanPlayerNameLength+1];
360 } PlayerInfo;
361
362 // GameToJoin is sent with REQUEST_JOIN
363 struct
364 {
365 UnsignedInt gameIP;
366 UnsignedInt exeCRC;
367 UnsignedInt iniCRC;
368 char serial[g_maxSerialLength];
369 } GameToJoin;
370
371 // GameJoined is sent with JOIN_ACCEPT
372 struct
373 {
374 WideChar gameName[g_lanGameNameLength+1];
375 UnsignedInt gameIP;
376 UnsignedInt playerIP;
377 Int slotPosition;
378 } GameJoined;
379
380 // GameNotJoined is sent with JOIN_DENY
381 struct
382 {
383 WideChar gameName[g_lanGameNameLength+1];
384 UnsignedInt gameIP;
385 UnsignedInt playerIP;
386 LANAPIInterface::ReturnType reason;
387 } GameNotJoined;
388
389 // Accept is sent with SET_ACCEPT
390 struct
391 {
392 WideChar gameName[g_lanGameNameLength+1];
393 Bool isAccepted;
394 } Accept;
395
396 // Accept is sent with MAP_AVAILABILITY
397 struct
398 {
399 WideChar gameName[g_lanGameNameLength+1];
400 UnsignedInt mapCRC; // to make sure we're talking about the same map
401 Bool hasMap;
402 } MapStatus;
403
404 // Chat is sent with CHAT
405 struct
406 {
407 WideChar gameName[g_lanGameNameLength+1];
408 LANAPIInterface::ChatType chatType;
409 WideChar message[g_lanMaxChatLength+1];
410 } Chat;
411
412 // GameOptions is sent with GAME_OPTIONS
413 struct
414 {
415 char options[m_lanMaxOptionsLength+1];
416 } GameOptions;
417
418 };
419};
