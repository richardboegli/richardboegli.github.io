68struct SelectObjectsInfo
69{
70 const ThingTemplate *thingTemplate;
71 GameMessage *msg;
72};

74//-------------------------------------------------------------------------------------------------
75
76void selectObjectOfType( Object* obj, void* selectObjectsInfo )
77{
78 SelectObjectsInfo *soInfo = (SelectObjectsInfo*)selectObjectsInfo;
79 if( !obj || !soInfo )
80 {
81 return;
82 }
83
84 //Do the templates match?
85 if( obj->getTemplate()->isEquivalentTo( soInfo->thingTemplate ) )
86 {
87 //Okay, then add it to the selected group.
88 soInfo->msg->appendObjectIDArgument( obj->getID() );
89
90 Drawable *draw = obj->getDrawable();
91 if( draw )
92 {
93 TheInGameUI->selectDrawable( draw );
94 }
95 }
96}

105CBCommandStatus ControlBar::processCommandTransitionUI( GameWindow *control, GadgetGameMessage gadgetMessage )
106{
107 // sanity, we won't process messages if we have no source object
108 if( m_currContext != CB_CONTEXT_MULTI_SELECT &&
109 (m_currentSelectedDrawable == NULL ||
110 m_currentSelectedDrawable->getObject() == NULL) )
111 {
112
113 if( m_currContext != CB_CONTEXT_NONE &&
114 m_currContext != CB_CONTEXT_OBSERVER_INFO &&
115 m_currContext != CB_CONTEXT_OBSERVER_LIST)
116 switchToContext( CB_CONTEXT_NONE, NULL );
117 return CBC_COMMAND_NOT_USED;
118
119 } // end if
120
121 return CBC_COMMAND_USED;
122
123}

129CBCommandStatus ControlBar::processCommandUI( GameWindow *control,
130 GadgetGameMessage gadgetMessage )
131{
132 // get the command pointer from the control user data we put in the button
133 const CommandButton *commandButton = (const CommandButton *)GadgetButtonGetData(control);
134 if( !commandButton )
135 {
136 DEBUG_CRASH( ("ControlBar::processCommandUI() -- Button activated has no data. Ignoring...") );
137 return CBC_COMMAND_NOT_USED;
138 }
139
140 // sanity, we won't process messages if we have no source object,
141 // unless we're CB_CONTEXT_PURCHASE_SCIENCE or GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT
142 if( m_currContext != CB_CONTEXT_MULTI_SELECT &&
143 commandButton->getCommandType() != GUI_COMMAND_PURCHASE_SCIENCE &&
144 commandButton->getCommandType() != GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT &&
145 commandButton->getCommandType() != GUI_COMMAND_SPECIAL_POWER_CONSTRUCT_FROM_SHORTCUT &&
146 commandButton->getCommandType() != GUI_COMMAND_SELECT_ALL_UNITS_OF_TYPE &&
147 (m_currentSelectedDrawable == NULL || m_currentSelectedDrawable->getObject() == NULL) )
148 {
149
150 if( m_currContext != CB_CONTEXT_NONE )
151 switchToContext( CB_CONTEXT_NONE, NULL );
152 return CBC_COMMAND_NOT_USED;
153
154 } // end if
155
156 // sanity
157 if( control == NULL )
158 return CBC_COMMAND_NOT_USED;
159
160 // the context sensitive gui only is only made of buttons ... sanity
161 if( control->winGetInputFunc() != GadgetPushButtonInput )
162 return CBC_COMMAND_NOT_USED;
163
164
165 if( commandButton == NULL )
166 return CBC_COMMAND_NOT_USED;
167
168 // if the button is flashing, tell it to stop flashing
169 commandButton->setFlashCount(0);
170 TheControlBar->setFlash( FALSE );
171
172 if( commandButton->getCommandType() != GUI_COMMAND_EXIT_CONTAINER )
173 {
174 GadgetButtonSetEnabledImage( control, commandButton->getButtonImage() );
175 }
176
177 //
178 // get the object that is driving the context sensitive UI if we're not in a multi
179 // select context
180 //
181 Object *obj = NULL;
182 if( m_currContext != CB_CONTEXT_MULTI_SELECT &&
183 commandButton->getCommandType() != GUI_COMMAND_PURCHASE_SCIENCE &&
184 commandButton->getCommandType() != GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT &&
185 commandButton->getCommandType() != GUI_COMMAND_SPECIAL_POWER_CONSTRUCT_FROM_SHORTCUT &&
186 commandButton->getCommandType() != GUI_COMMAND_SELECT_ALL_UNITS_OF_TYPE )
187 obj = m_currentSelectedDrawable->getObject();
188
189 //@todo Kris -- Special case code so convoy trucks can detonate nuke trucks -- if other things need this,
190 //rethink it.
191 if( obj && BitTest( commandButton->getOptions(), SINGLE_USE_COMMAND ) )
192 {
197
198 //Make sure the command button is marked as used if it's a single use command. That way
199 //we can never press the button again. This was added specifically for nuke convoy trucks.
200 //When you click to detonate the nuke, it takes a few seconds to detonate in order to play
201 //a sound. But we want to disable the button after the first click.
202 obj->markSingleUseCommandUsed(); //Yeah, an object can only use one single use command...
203 }
204
205 TheInGameUI->placeBuildAvailable( NULL, NULL );
206
207 //Play any available unit specific sound for button
208 Player *player = ThePlayerList->getLocalPlayer();
209 if( player )
210 {
211 AudioEventRTS sound = *commandButton->getUnitSpecificSound();
212 sound.setPlayerIndex( player->getPlayerIndex() );
213 TheAudio->addAudioEvent( &sound );
214 }
215
216 if( BitTest( commandButton->getOptions(), COMMAND_OPTION_NEED_TARGET ) )
217 {
218 if (commandButton->getOptions() & USES_MINE_CLEARING_WEAPONSET)
219 {
220 TheMessageStream->appendMessage( GameMessage::MSG_SET_MINE_CLEARING_DETAIL );
221 }
222
223 //June 06, 2002 -- Major change
224 //I've added support for specific context sensitive commands which need targets just like
225 //other options may need. When we need a target, the user must move the cursor to a position
226 //where he wants the GUI command to take place. Older commands such as napalm strikes or daisy
227 //cutter drops simply needed the user to click anywhere he desired.
228 //
229 //Now, we have new commands that will only work when the user clicks on valid targets to interact
230 //with. For example, the terrorist can jack a car and convert it into a carbomb, but he has to
231 //click on a valid car. In this case the doCommandOrHint code will determine if the mode is valid
232 //or not and the cursor modes will be set appropriately.
233 TheInGameUI->setGUICommand( commandButton );
234 }
235 else switch( commandButton->getCommandType() )
236 {
237
238 //---------------------------------------------------------------------------------------------
239 case GUI_COMMAND_DOZER_CONSTRUCT:
240 {
241
242 // sanity
243 if( m_currentSelectedDrawable == NULL )
244 break;
245
246 //Kris: September 27, 2002
247 //Make sure we have enough CASH to build it WHEN we click the button to build it,
248 //before actually previewing the purchase, otherwise, cancel altogether.
249 const ThingTemplate *whatToBuild = commandButton->getThingTemplate();
250 CanMakeType cmt = TheBuildAssistant->canMakeUnit( obj, whatToBuild );
251 if (cmt == CANMAKE_NO_MONEY)
252 {
253 TheEva->setShouldPlay(EVA_InsufficientFunds);
254 TheInGameUI->message( "GUI:NotEnoughMoneyToBuild" );
255 break;
256 }
257 else if (cmt == CANMAKE_QUEUE_FULL)
258 {
259 TheInGameUI->message( "GUI:ProductionQueueFull" );
260 break;
261 }
262 else if (cmt == CANMAKE_PARKING_PLACES_FULL)
263 {
264 TheInGameUI->message( "GUI:ParkingPlacesFull" );
265 break;
266 }
267 else if( cmt == CANMAKE_MAXED_OUT_FOR_PLAYER )
268 {
269 TheInGameUI->message( "GUI:UnitMaxedOut" );
270 break;
271 }
272
273 // tell the UI that we want to build something so we get a building at the cursor
274 TheInGameUI->placeBuildAvailable( commandButton->getThingTemplate(), m_currentSelectedDrawable );
275
276 break;
277
278 } // end dozer construct
279
280
281 case GUI_COMMAND_SPECIAL_POWER_CONSTRUCT_FROM_SHORTCUT:
282 {
283 //Determine the object that would construct it.
284 const SpecialPowerTemplate *spTemplate = commandButton->getSpecialPowerTemplate();
285 SpecialPowerType spType = spTemplate->getSpecialPowerType();
286 Object* obj = ThePlayerList->getLocalPlayer()->findMostReadyShortcutSpecialPowerOfType( spType );
287 if( !obj )
288 break;
289 Drawable *draw = obj->getDrawable();
290
291 const ThingTemplate *whatToBuild = commandButton->getThingTemplate();
292
293 CanMakeType cmt = TheBuildAssistant->canMakeUnit( obj, whatToBuild );
294 if (cmt == CANMAKE_NO_MONEY)
295 {
296 TheEva->setShouldPlay(EVA_InsufficientFunds);
297 TheInGameUI->message( "GUI:NotEnoughMoneyToBuild" );
298 break;
299 }
300 else if (cmt == CANMAKE_QUEUE_FULL)
301 {
302 TheInGameUI->message( "GUI:ProductionQueueFull" );
303 break;
304 }
305 else if (cmt == CANMAKE_PARKING_PLACES_FULL)
306 {
307 TheInGameUI->message( "GUI:ParkingPlacesFull" );
308 break;
309 }
310 else if( cmt == CANMAKE_MAXED_OUT_FOR_PLAYER )
311 {
312 TheInGameUI->message( "GUI:UnitMaxedOut" );
313 break;
314 }
315
316 // tell the UI that we want to build something so we get a building at the cursor
317 TheInGameUI->placeBuildAvailable( commandButton->getThingTemplate(), draw );
318
319 ProductionUpdateInterface* pu = obj->getProductionUpdateInterface();
320 if( pu )
321 {
322 pu->setSpecialPowerConstructionCommandButton( commandButton );
323 }
324
325 break;
326 }
327 case GUI_COMMAND_SPECIAL_POWER_CONSTRUCT:
328 {
329 // sanity
330 if( m_currentSelectedDrawable == NULL )
331 break;
332
333 const ThingTemplate *whatToBuild = commandButton->getThingTemplate();
334
335 CanMakeType cmt = TheBuildAssistant->canMakeUnit( obj, whatToBuild );
336 if (cmt == CANMAKE_NO_MONEY)
337 {
338 TheEva->setShouldPlay(EVA_InsufficientFunds);
339 TheInGameUI->message( "GUI:NotEnoughMoneyToBuild" );
340 break;
341 }
342 else if (cmt == CANMAKE_QUEUE_FULL)
343 {
344 TheInGameUI->message( "GUI:ProductionQueueFull" );
345 break;
346 }
347 else if (cmt == CANMAKE_PARKING_PLACES_FULL)
348 {
349 TheInGameUI->message( "GUI:ParkingPlacesFull" );
350 break;
351 }
352 else if( cmt == CANMAKE_MAXED_OUT_FOR_PLAYER )
353 {
354 TheInGameUI->message( "GUI:UnitMaxedOut" );
355 break;
356 }
357
358 // tell the UI that we want to build something so we get a building at the cursor
359 TheInGameUI->placeBuildAvailable( commandButton->getThingTemplate(), m_currentSelectedDrawable );
360
361 ProductionUpdateInterface* pu = obj->getProductionUpdateInterface();
362 if( pu )
363 {
364 pu->setSpecialPowerConstructionCommandButton( commandButton );
365 }
366
367 break;
368
369 } // end dozer construct

373 //---------------------------------------------------------------------------------------------
374 case GUI_COMMAND_DOZER_CONSTRUCT_CANCEL:
375 {
376
377 // get the object we have selected
378 Object *building = obj;
379 if( building == NULL )
380 break;
381
382 // sanity check, the building must be under our control to cancel construction
383 if( building->getControllingPlayer() != ThePlayerList->getLocalPlayer() )
384 break;
385
386 // do the message
387 TheMessageStream->appendMessage( GameMessage::MSG_DOZER_CANCEL_CONSTRUCT );
388
389 break;

393 //---------------------------------------------------------------------------------------------
394 case GUI_COMMAND_UNIT_BUILD:
395 {
396 //
397 const ThingTemplate *whatToBuild = commandButton->getThingTemplate();
398
399 // get the "factory" object that is going to make the thing
400 Object *factory = obj;
401 if( factory == NULL )
402 break;
403
404 // sanity, we must have something to build
405 DEBUG_ASSERTCRASH( whatToBuild, ("Undefined BUILD command for object '%s'\n",
406 commandButton->getThingTemplate()->getName().str()) );
407
408 CanMakeType cmt = TheBuildAssistant->canMakeUnit(factory, whatToBuild);
409
410 if (cmt == CANMAKE_NO_MONEY)
411 {
412 TheEva->setShouldPlay(EVA_InsufficientFunds);
413 TheInGameUI->message( "GUI:NotEnoughMoneyToBuild" );
414 break;
415 }
416 else if (cmt == CANMAKE_QUEUE_FULL)
417 {
418 TheInGameUI->message( "GUI:ProductionQueueFull" );
419 break;
420 }
421 else if (cmt == CANMAKE_PARKING_PLACES_FULL)
422 {
423 TheInGameUI->message( "GUI:ParkingPlacesFull" );
424 break;
425 }
426 else if( cmt == CANMAKE_MAXED_OUT_FOR_PLAYER )
427 {
428 TheInGameUI->message( "GUI:UnitMaxedOut" );
429 break;
430 }
431 else if (cmt != CANMAKE_OK)
432 {
433 DEBUG_ASSERTCRASH( 0, ("Cannot create '%s' because the factory object '%s' returns false for canMakeUnit\n",
434 whatToBuild->getName().str(),
435 factory->getTemplate()->getName().str()) );
436 break;
437 }
438
439 // get the production interface from the factory object
440 ProductionUpdateInterface *pu = factory->getProductionUpdateInterface();
441
442 // sanity, we can't build things if we can't produce units
443 if( pu == NULL )
444 {
445
446 DEBUG_ASSERTCRASH( 0, ("Cannot create '%s' because the factory object '%s' is not capable of producting units\n",
447 whatToBuild->getName().str(),
448 factory->getTemplate()->getName().str()) );
449 break;
450
451 } // end if
452
453 // get a new production id to assign to this
454 ProductionID productionID = pu->requestUniqueUnitID();
455
456 // create a message to build this thing
457
458 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_QUEUE_UNIT_CREATE );
459 msg->appendIntegerArgument( whatToBuild->getTemplateID() );
460 msg->appendIntegerArgument( productionID );
461
462 break;
463
464 } // end build unit
465
466 //---------------------------------------------------------------------------------------------
467 case GUI_COMMAND_CANCEL_UNIT_BUILD:
468 {
469 Int i;
470
471 // find out which index (i) in the queue represents the button clicked
472 for( i = 0; i < MAX_BUILD_QUEUE_BUTTONS; i++ )
473 if( m_queueData[ i ].control == control )
474 break;
475
476 // sanity, control not found
477 if( i == MAX_BUILD_QUEUE_BUTTONS )
478 {
479
480 DEBUG_ASSERTCRASH( 0, ("Control not found in build queue data\n") );
481 break;
482
483 } // end if
484
485 // sanity
486 if( m_queueData[ i ].type != PRODUCTION_UNIT )
487 break;
488
489 // the the production ID to cancel
490 ProductionID productionIDToCancel = m_queueData[ i ].productionID;
491
492 // get the object that is the producer
493 Object *producer = obj;
494 if( producer == NULL )
495 break;
496
497 // sanity, we must control the producer ... if this isn't true they might be hacking the game
498 if( producer->getControllingPlayer() != ThePlayerList->getLocalPlayer() )
499 break;
500
501 // send a message to cancel that particular production entry
502 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_CANCEL_UNIT_CREATE );
503 msg->appendIntegerArgument( productionIDToCancel );
504
505 break;
506
507 } // end cancel unit build

511 {
512 const UpgradeTemplate *upgradeT = commandButton->getUpgradeTemplate();
513 DEBUG_ASSERTCRASH( upgradeT, ("Undefined upgrade '%s' in player upgrade command\n", "UNKNOWN") );
514
515 // sanity
516 if( obj == NULL || upgradeT == NULL )
517 break;
518
519 // make sure the player can really make this
520 if( TheUpgradeCenter->canAffordUpgrade( ThePlayerList->getLocalPlayer(), upgradeT, TRUE ) == FALSE )
521 {
522 break;
523 }
524
525 ProductionUpdateInterface* pu = obj ? obj->getProductionUpdateInterface() : NULL;
526 if (pu != NULL)
527 {
528 CanMakeType cmt = pu->canQueueUpgrade(upgradeT);

533 }
534 }
535
536 // send the message
537 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_QUEUE_UPGRADE );
538 msg->appendObjectIDArgument( obj->getID() );
539 msg->appendIntegerArgument( upgradeT->getUpgradeNameKey() );
540
541 break;
542
543 } // command player upgrade

545 //---------------------------------------------------------------------------------------------
546 case GUI_COMMAND_OBJECT_UPGRADE:
547 {
548 const UpgradeTemplate *upgradeT = commandButton->getUpgradeTemplate();
549 DEBUG_ASSERTCRASH( upgradeT, ("Undefined upgrade '%s' in object upgrade command\n", "UNKNOWN") );
550 // sanity
551 if( upgradeT == NULL )
552 break;
553
554 //Make sure the player can really make this
555 if( TheUpgradeCenter->canAffordUpgrade( ThePlayerList->getLocalPlayer(), upgradeT, TRUE ) == FALSE )
556 {
557 //Kris: Disabled because we can get a valid reason for not being able to afford the upgrade!
558 //TheInGameUI->message( "upgrade unsupported in commandprocessing." );
559 break;
560 }
561
562 ProductionUpdateInterface* pu = obj ? obj->getProductionUpdateInterface() : NULL;
563 if (pu != NULL)
564 {
565 CanMakeType cmt = pu->canQueueUpgrade(upgradeT);
566 if (cmt == CANMAKE_QUEUE_FULL)
567 {
568 TheInGameUI->message( "GUI:ProductionQueueFull" );
569 break;
570 }
571 }
572
573 ObjectID objID = INVALID_ID;
574 if (obj)
575 objID = obj->getID();
576
577 // make sure that the this object can actually build the upgrade
578 if( obj && (obj->hasUpgrade( upgradeT ) == TRUE || obj->affectedByUpgrade( upgradeT ) == FALSE) )
579 break;
580
581 // send the message
582 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_QUEUE_UPGRADE );
583 msg->appendObjectIDArgument( objID );
584 msg->appendIntegerArgument( upgradeT->getUpgradeNameKey() );
585
586 break;
587
588 } // end object upgrade

686 //
687 // find the object ID that wants to exit by scanning through the transport data and looking

690 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )

694 // get the actual object

705 //
706 m_containData[ i ].control = NULL;
707 m_containData[ i ].objectID = INVALID_ID;
708 break; // exit case
709
710 } // end if

712 //what if container is subdued... assert a logic failure, perhaps?
713
714 // send message to exit
715 GameMessage *exitMsg = TheMessageStream->appendMessage( GameMessage::MSG_EXIT );
716 exitMsg->appendObjectIDArgument( objWantingExit->getID() ); // 0 is the thing inside coming out
