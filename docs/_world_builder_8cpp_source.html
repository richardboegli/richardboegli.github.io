494// CWorldBuilderApp message handlers
495
496BOOL CWorldBuilderApp::OnCmdMsg(UINT nID, int nCode, void* pExtra,
497 AFX_CMDHANDLERINFO* pHandlerInfo)
498{
499 // If pHandlerInfo is NULL, then handle the message
500 if (pHandlerInfo == NULL)
501 {
502 for (Int i=0; i<NUM_VIEW_TOOLS; i++) {
503 Tool *pTool = m_tools[i];
504 if (pTool==NULL) continue;
505 if ((Int)nID == pTool->getToolID()) {
506 if (nCode == CN_COMMAND)
507 {
508 // Handle WM_COMMAND message
509 setActiveTool(pTool);
510 }
511 else if (nCode == CN_UPDATE_COMMAND_UI)
512 {
513 // Update UI element state
514 CCmdUI *pUI = (CCmdUI*)pExtra;
515 pUI->SetCheck(m_curTool == pTool?1:0);
516 pUI->Enable(true);
517 }
518 return TRUE;
519 }
520 }
521 }
522
523 // If we didn't process the command, call the base class
524 // version of OnCmdMsg so the message-map can handle the message
525 return CWinApp::OnCmdMsg(nID, nCode, pExtra, pHandlerInfo);
526}

545void CWorldBuilderApp::setActiveTool(Tool *pNewTool)
546{
547 if (m_curTool == pNewTool) {
548 // same tool
549 return;
550 }
551 if (m_selTool && m_selTool != pNewTool) {
552 m_selTool->deactivate();
553 }
554 if (pNewTool) {
555 pNewTool->activate();
556 }
557 m_curTool = pNewTool;
558 m_selTool = pNewTool;
559}

561//=============================================================================
562// CWorldBuilderApp::updateCurTool
563//=============================================================================
566//=============================================================================
567void CWorldBuilderApp::updateCurTool(Bool forceHand)
568{
569 Tool *curTool = m_curTool;
570 DEBUG_ASSERTCRASH((m_lockCurTool>=0),("oops"));
571 if (!m_lockCurTool) { // don't change tools that are doing something.
572 if (forceHand || (0x8000 & ::GetAsyncKeyState(VK_SPACE))) {
573 // Space bar gives scroll hand.
574 m_curTool = &m_handScrollTool;
575 } else if (0x8000 & ::GetAsyncKeyState(VK_MENU)) {
576 // Alt key gives eyedropper.
577 m_curTool = &m_eyedropperTool;
578 } else if (0x8000 & ::GetAsyncKeyState(VK_CONTROL)) {
579 // Control key gives pointer.
580 m_curTool = &m_pointerTool;
581 } else {
582 // Else the tool selected in the tool palette.
583 m_curTool = m_selTool;
584 }
585 }
586 if (curTool != m_curTool) {
587 m_curTool->activate();
588 }
589}

691void CWorldBuilderApp::OnFileOpen()
692{
693#ifdef DO_MAPS_IN_DIRECTORIES
694 TOpenMapInfo info;
695 OpenMap mapDlg(&info);
696 if (mapDlg.DoModal() == IDOK) {
697 if (!info.browse) {
698 OpenDocumentFile(info.filename);
699 return;
700 }
701 } else {
702 // cancelled so return.
703 return;
704 }
705#endif
706
707 CFileStatus status;
708 if (m_currentDirectory != AsciiString("")) try {
709 if (CFile::GetStatus(m_currentDirectory.str(), status)) {
710 if (status.m_attribute & CFile::directory) {
711 ::SetCurrentDirectory(m_currentDirectory.str());
712 }
713 }
