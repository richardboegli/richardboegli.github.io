50** uses gun values from 0 to 255.
51*/

54 public:
55 RGBClass(void) : Red(0), Green(0), Blue(0) {}
56 RGBClass(unsigned char red, unsigned char green, unsigned char blue) : Red(red), Green(green), Blue(blue) {}
57 operator HSVClass (void) const;

59 if (this == &rgb) return(*this);
60
61 Red = rgb.Red;
62 Green = rgb.Green;
63 Blue = rgb.Blue;
64 return(*this);
65 }
