183void CGenerateDlg::OnOK()
184{
185 char buffer[30];
186 int count;
187 int i;
188 // TODO: Add extra validation here
189
190 edit->GetWindowText ( buffer, sizeof ( filename) -1 );
191 _getcwd ( filename, sizeof (filename ) -1 );
192 strcat ( filename, "\\" );
193 strcat ( filename, buffer );
194
195 count = list->GetSelItems ( num_langs, langindices );
196
197 if ( !count )
198 {
199 AfxMessageBox ( "No languages selected" );
200 return;
201 }
202
203 num_langs = 0;
204 for ( i = 0; i <count; i++ )
205 {
206 LANGINFO *info;
207
208 if ( info = GetLangInfo ( langindices[i] ))
209 {
210 langids[num_langs++] = info->langid;
211 }
212 }
213
214 langids[num_langs] = LANGID_UNKNOWN;
215
216
217 CDialog::OnOK();
