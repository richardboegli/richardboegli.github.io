386int __declspec(dllexport) NextParticleEditorBehavior( void )
387{
388 try {
389 if (HasUpdatedSelectedParticleSystem()) {
390 return PEB_UpdateCurrentSystem;
391 }
392
393 if (ShouldWriteINI()) {
394 return PEB_SaveAllSystems;
395 }
396
397 if (HasRequestedReload()) {
398 return PEB_ReloadCurrentSystem;
399 }
400
401 if (ShouldBusyWait()) {
402 return PEB_BusyWait;
403 }
404
405 if (ShouldUpdateParticleCap()) {
406 return PEB_SetParticleCap;
407 }
408
409 if (ShouldReloadTextures()) {
410 return PEB_ReloadTextures;
411 }
412
413 if (HasRequestedKillAllSystems()) {
414 return PEB_KillAllSystems;
415 }
416
417 return PEB_Continue;
418 } catch (...) { }
419 return PEB_Error;
420}
