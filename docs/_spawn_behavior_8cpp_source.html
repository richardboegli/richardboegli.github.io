108void SpawnBehavior::onDelete()
109{
110 const SpawnBehaviorModuleData *modData = getSpawnBehaviorModuleData();
111
112 // destroy anything that we have spawned that is not already dead
113 if( modData->m_spawnedRequireSpawner )
114 {
115 Object *obj;
116
117 for( objectIDListIterator it = m_spawnIDs.begin(); it != m_spawnIDs.end(); /*empty*/ )
118 {
119
120 // get object
121 obj = TheGameLogic->findObjectByID( *it );
122
123 // increment iterator incase the id list is alterd because what we're about to do
124 ++it;
125
126 //
127 // destroy this if it's alive, in the usual case this object "dies" and is
128 // not "destroyed". on *death* we *KILL* our spawned things, but this is here
129 // and will *DESTROY* our spawned things if we ourselves are destroyed and they
130 // are still alive (such a case would be calling destroy object on us directly)
131 //
132 if( obj && obj->isEffectivelyDead() == FALSE )
133 TheGameLogic->destroyObject( obj );
134
135 } // end for, it
136
137 } // end if
138

142void SpawnBehavior::onDie( const DamageInfo *damageInfo )
143{
144 const SpawnBehaviorModuleData *modData = getSpawnBehaviorModuleData();
145
147 if( modData->m_dieMuxData.isDieApplicable( getObject(), damageInfo ) == FALSE )
148 return;
149
150 for( objectIDListIterator iter = m_spawnIDs.begin();
151 iter != m_spawnIDs.end();
152 iter++
153 )
154 {
155 Object *currentSpawn = TheGameLogic->findObjectByID( (*iter) );
156 if( currentSpawn )
157 {
158 // Go through all my spawns and see if they have a SlavedUpdate I can tell I was killed to
159 for (BehaviorModule** update = currentSpawn->getBehaviorModules(); *update; ++update)
160 {
161 SlavedUpdateInterface* sdu = (*update)->getSlavedUpdateInterface();
162 if (sdu != NULL)
163 {
164 sdu->onSlaverDie( damageInfo );
165 break;
166 }
167 }
168
169 // our spawner has died, we must invalidate the ID now in the spawned object
170 currentSpawn->setProducer( NULL );
171
172 }
173 }
174
175 // kill anything that we have spawned if our module data directs us to do so
176 if( modData->m_spawnedRequireSpawner )
177 {
178 Object *obj;
179
180 for( objectIDListIterator it = m_spawnIDs.begin(); it != m_spawnIDs.end(); /*emtpy*/ )
181 {
182
183 // get object
184 obj = TheGameLogic->findObjectByID( *it );
185
186 // increment iterator incase the id list is alterd because what we're about to do
187 ++it;
188
189 // kill it if not already dead
190 if( obj && obj->isEffectivelyDead() == FALSE )
191 obj->kill();
192
193 } // end for, it
194
195 } // end if

200UpdateSleepTime SpawnBehavior::update( void )
201{
203
204 //EVERY FRAME
205 if ( m_aggregateHealth )
206 {
207 computeAggregateStates();
208 }
209
210 const SpawnBehaviorModuleData* md = getSpawnBehaviorModuleData();
211 if ( ! m_initialBurstTimesInited )
212 {
213 m_initialBurstTimesInited = TRUE;
214
215 Bool runtimeProduced = getObject()->getProducerID()!=INVALID_ID; //this was produced by a production module, rather than a script or worldbuilder
216 Int now = TheGameLogic->getFrame();
217 Int burstInitCount = m_initialBurstCountdown;
218 for( int listIndex = 0; listIndex < md->m_spawnNumberData; listIndex++ )
219 {
220 if ( md->m_initialBurst > 0 )
221 {
222 UnsignedInt birthFrame = now;
223 if ( runtimeProduced && burstInitCount > 0)
224 {
225 --burstInitCount;
226 birthFrame += (listIndex*SPAWN_DELAY_MIN_FRAMES);
227 }
228 m_replacementTimes.push_back( runtimeProduced );// Set all spawns to be created in rapid succession
229 }
230 else
231 m_replacementTimes.push_back( listIndex );// Set all spawns to be created as soon as possible
232 }
233 }
234
235
236
237 //SPARSELY SOLVED
238 if (--m_framesToWait > 0)
239 {
240 return UPDATE_SLEEP_NONE;
241 }
242
243 m_framesToWait = SPAWN_UPDATE_RATE;
244
245 //Go through the list and make a spawn for each number that is less than now's frame
246 if( shouldTryToSpawn() )
247 {
248 // Due to STL's list vomiting when you erase in a for loop,
249 // Start at the beginning, and if you erase one, use the return
250 // value to manually step to the next. Otherwise, step by yourself.
251 // begin == end means empty list, so no special check is needed.
252 intListIterator iterator = m_replacementTimes.begin();
253 while( iterator != m_replacementTimes.end() )
254 {
255 Int replacementTime = *iterator;
256 UnsignedInt currentTime = TheGameLogic->getFrame();
257 if( currentTime > replacementTime )
258 {
259 //If you create one, you pop the number off the list
260 if( createSpawn() )
261 iterator = m_replacementTimes.erase( iterator );
262 else
263 iterator++;
264 }
265 else
266 iterator++;
267 }
268
269 if( md->m_isOneShotData && (m_oneShotCountdown <= 0) )
270 stopSpawning(); //I only trigger one batch. ie on Creation.
271 }
272 return UPDATE_SLEEP_NONE;
273}

276Bool SpawnBehavior::maySpawnSelfTaskAI( Real maxSelfTaskersRatio )
277{
278 if ( m_spawnCount == 0)
279 return FALSE;
280 if ( maxSelfTaskersRatio == 0)
281 return FALSE;
282
283
284 //if my last attack command was from player or script, I need to forbid my spawn from disobeying that command
285 //otherwise (since my attack state was autoacquired my ny own ai), let them deviate by the ratio specified.
286 Object* obj = getObject();
287 if ( ! obj )
288 return FALSE;
289 AIUpdateInterface *ai = obj->getAI();
290 if ( ! ai )
291 return FALSE;
292
293 CommandSourceType lastAttackCommandSource = ai->getLastCommandSource();
294
295
296 if ( lastAttackCommandSource != CMD_FROM_AI )
297 return FALSE;
298
299
300 Real curSelfTaskersRatio = (Real)m_selfTaskingSpawnCount / (Real)m_spawnCount;
301
302 return ( curSelfTaskersRatio < maxSelfTaskersRatio );
303}

546static void findClosestOrphan( Object *obj, void *userData )
547{
548 OrphanData *orphanData = (OrphanData *)userData;
549
550 // if template doesn't match do nothing
551 if( obj->getTemplate()->isEquivalentTo( orphanData->m_matchTemplate ) == FALSE )
552 return;
553
554 // this object must be orphaned
555 if( obj->getProducerID() != INVALID_ID )
556 return;
557
558 // is this the closest one so far
559 Real distSq = ThePartitionManager->getDistanceSquared( orphanData->m_source, obj, FROM_CENTER_2D );
560 if( distSq < orphanData->m_closestDistSq )
561 {
562
563 orphanData->m_closest = obj;
564 orphanData->m_closestDistSq = distSq;
565
566 } // end if
567
568} // findClosestOrphan

570// ------------------------------------------------------------------------------------------------
571Object *SpawnBehavior::reclaimOrphanSpawn( void )
572{
573 Player *player = getObject()->getControllingPlayer();
574 const SpawnBehaviorModuleData *md = getSpawnBehaviorModuleData();
575
576 //
577 // iterate all the objects my controlling player has and look for any orphaned things
578 // that we would normally spawn, if found we'll just make it our own
579 //
580 // EVEN MORE NEW AND DIFFERENT
581 // This block scans the list for matchTemplates
582 //
583
584 OrphanData orphanData;
585 AsciiString prevName = "";
586 for (std::vector<AsciiString>::const_iterator tempName = md->m_spawnTemplateNameData.begin();
587 tempName != md->m_spawnTemplateNameData.end();
588 ++tempName)
589 {
590 if (prevName.compare(*tempName)) // the list may have redundancy, this will skip some of it
591 continue;
592 orphanData.m_matchTemplate = TheThingFactory->findTemplate( *tempName );;
593 orphanData.m_source = getObject();
594 orphanData.m_closest = NULL;
595 orphanData.m_closestDistSq = BIG_DISTANCE;
596 player->iterateObjects( findClosestOrphan, &orphanData );
597 prevName = *tempName;
598 }
599
600 return orphanData.m_closest;
601}

604Bool SpawnBehavior::createSpawn()
605{
606 Object *parent = getObject();
607 const SpawnBehaviorModuleData *md = getSpawnBehaviorModuleData();
608
609 ExitInterface* exitInterface = parent->getObjectExitInterface();
610 if( exitInterface == NULL )
611 {
612 DEBUG_ASSERTCRASH( exitInterface != NULL, ("Something cannot have SpawnBehavior without an exit interface") );
613 return FALSE;
614 }
615
616 ExitDoorType exitDoor = exitInterface->reserveDoorForExit(NULL, NULL);
617 if (exitDoor == DOOR_NONE_AVAILABLE)
618 return FALSE;
619
620 Object *newSpawn = NULL;
621
622 // try to reclaim orphaned objects if possible
623 Bool reclaimedOrphan = FALSE;
624 if( md->m_canReclaimOrphans && md->m_isOneShotData == FALSE )
625 {
626 newSpawn = reclaimOrphanSpawn();
627 if( newSpawn )
628 reclaimedOrphan = TRUE;
629 }
630
631 // This assures that an orphan has not just been reclaimed,
632 if( newSpawn == NULL ) // and that we really want a new spawn, here.
633 {
634 m_spawnTemplate = TheThingFactory->findTemplate( *m_templateNameIterator );
635
636 //newSpawn = TheThingFactory->newObject( m_spawnTemplate, parent->getControllingPlayer()->getDefaultTeam() );
637 newSpawn = TheThingFactory->newObject( m_spawnTemplate, parent->getTeam() ); // just a little worried about this...
638
639 // Count this unit towards our score.
640 newSpawn->getControllingPlayer()->onUnitCreated(parent, newSpawn);
641
642 // Gee, this is new...
643 // Now, we can have a list of template names.
644 // Each time m_spawn template is used, it will increment m_templateNameIterator,
645 // thus scanning through the ordered list of template names, and
646 // looping back to the beginning
647
648 ++m_templateNameIterator;
649 if ( m_templateNameIterator == md->m_spawnTemplateNameData.end())
650 {
651 m_templateNameIterator = md->m_spawnTemplateNameData.begin();
652 }
653 }
654
655
656 newSpawn->setProducer(parent);
657
658 // If they have a SlavedUpdate, then I have to tell them who their daddy is from now on.
659 for (BehaviorModule** update = newSpawn->getBehaviorModules(); *update; ++update)
660 {
661 SlavedUpdateInterface* sdu = (*update)->getSlavedUpdateInterface();
662 if (sdu != NULL)
663 {
664 sdu->onEnslave( parent );
665 break;
666 }
667 }
668
669 m_spawnIDs.push_back( newSpawn->getID() );
670
671 if( reclaimedOrphan == FALSE )
672 {
673 if ( md->m_exitByBudding )
674 {
675
676 Bool barracksExitSuccess = FALSE;
677
678 if ( m_initialBurstCountdown > 0 )
679 {
680 Object *barracks = TheGameLogic->findObjectByID( parent->getProducerID() );
681 if ( barracks && barracks->isKindOf( KINDOF_STRUCTURE ) )
682 {
683 ExitInterface* barracksExitInterface = barracks->getObjectExitInterface();
684 if ( barracksExitInterface )
685 {
686 ExitDoorType barracksDoor = barracksExitInterface->reserveDoorForExit(NULL, NULL);
687 barracksExitInterface->exitObjectViaDoor( newSpawn, barracksDoor );
688 newSpawn->setProducer(parent);//let parents producer exit him, but he thinks it was me
689 --m_initialBurstCountdown;
690 barracksExitSuccess = TRUE;
691 }
692
693 }
694
695 }
696
697 if ( ! barracksExitSuccess )
698 {
699 // find the closest spawn to the nexus...
700 //there is probably a more elegant way to choose the budHost, but oh well
701 Object *budHost = NULL;
702 Object *curSpawn = NULL;
703 Real tapeMeasure = 99999;
704 Real closest = 999999.9f; // 1000 * 1000
705 objectIDListIterator iter;
706 for( iter = m_spawnIDs.begin(); iter != m_spawnIDs.end(); iter++)
707 {
708 curSpawn = TheGameLogic->findObjectByID( *iter );
709 if ( curSpawn )
710 {
711 if (curSpawn == newSpawn )
712 continue;
713
714 tapeMeasure = ThePartitionManager->getDistanceSquared( curSpawn, parent, FROM_CENTER_2D );
715 if ( tapeMeasure < closest )
716 {
717 closest = tapeMeasure;
718 budHost = curSpawn;
719 }
720 }
721 }
722 exitInterface->exitObjectByBudding( newSpawn, budHost );// also handles the NULL pointer okay
723
724 }
725
726 }
727 else
728 {
729 exitInterface->exitObjectViaDoor( newSpawn, exitDoor );
730 }
731
732 }
733 else
734 exitInterface->unreserveDoorForExit( exitDoor );
735
736 if( md->m_isOneShotData )
737 m_oneShotCountdown--;
738
739
740 if ( m_spawnCount == NONE_SPAWNED_YET )
741 {
742 m_spawnCount = 1;
743 }
744 else
745 {
746 ++ m_spawnCount;
747 }
748
749
750 return TRUE;
751}

754void SpawnBehavior::onSpawnDeath( ObjectID deadSpawn, DamageInfo *damageInfo )
755{
756 objectIDListIterator it = std::find(m_spawnIDs.begin(), m_spawnIDs.end(), deadSpawn);
757
758 // If the iterator is at the end, we didn't find deadSpawn, so bail out.
759 // Otherwise, bad crash stuff will happen.
760 if (it == m_spawnIDs.end())
761 return;
762
763 //When one dies, you push (now + delay) as the time a new one should be made
764 const SpawnBehaviorModuleData* md = getSpawnBehaviorModuleData();
765
766 Int replacementTime = md->m_spawnReplaceDelayData + TheGameLogic->getFrame();
767 m_replacementTimes.push_back( replacementTime );
768
769 m_spawnIDs.erase( it );
770
771 --m_spawnCount;
772
773 if ( (m_spawnCount == 0) && m_aggregateHealth) // I'm dead without my spawn
774 {
775 Object *killer = TheGameLogic->findObjectByID(damageInfo->in.m_sourceID);
776 if (killer != NULL) {
777 killer->scoreTheKill(getObject());
778 }
779 TheGameLogic->destroyObject(getObject());
780 //getObject()->kill();
781 return;
782 }
783}

849//********************************************************************
850//* This function allows various object states to be either inherited
851//* from the spawner to the spawn, or vice versa
852//* Selection is distributed across all spawn and spawner at once
853//* Health for the spawner is calc'd by aggregating the sum of all
854//* spawn health, and dividing by optimal health of full population
855//* (that is, the max spawn, SpawnBehaviorModuleData::m_spawnNumberData)
856//* at full health.
857//* Veterancy is sucked out of any unit that has any and put into the
858//* Spawner, scaled by 1/SpawnBehaviorModuleData::m_spawnNumberData;
859//* The HealthBoxPosition (maybe to include moodicon, vet icon) is calc'd
860//* as an average position of all the spawn.
861//********************************************************************
862
863void SpawnBehavior::computeAggregateStates(void)
864{
865 if ( ! m_aggregateHealth ) // sanity
866 return;
867
868 Object* obj = getObject();
869 const SpawnBehaviorModuleData* md = getSpawnBehaviorModuleData();
870
871 Int spawnCount = 0;
872 Int spawnCountMax = md->m_spawnNumberData;
873 Coord3D avgSpawnPos;
874
875 avgSpawnPos.set(0,0,0);
876 Real acrHealth = 0.0f;
877 Real avgHealthMax = 0.0f;
878
879 ExperienceTracker *expTracker = obj->getExperienceTracker();
880 VeterancyLevel vetLevel = expTracker->getVeterancyLevel();
881
882 Bool SomebodyIsSelected = FALSE;
883 Bool SomebodyIsNotSelected = FALSE;
884
885 Drawable *spawnDraw = NULL;
886 Object *currentSpawn = NULL;
887
888 WeaponBonusConditionFlags spawnWeaponBonus;
889
890
891 m_selfTaskingSpawnCount = 0;
892
893 for( objectIDListIterator iter = m_spawnIDs.begin(); iter != m_spawnIDs.end(); iter++)
894 {
895 currentSpawn = TheGameLogic->findObjectByID( (*iter) );
896
897 if( currentSpawn )
898 {
899 //m_selfTaskingSpawnCount += ( currentSpawn->isSelf);
900
901 for (BehaviorModule** update = currentSpawn->getBehaviorModules(); *update; ++update)
902 {
903 SlavedUpdateInterface* sdu = (*update)->getSlavedUpdateInterface();
904 if (sdu != NULL)
905 {
906 m_selfTaskingSpawnCount += ( sdu->isSelfTasking());;
907 break;
908 }
909 }
910
911
912 // VETERANCY LEVEL *************************************
913 VeterancyLevel spawnVetLevel = currentSpawn->getVeterancyLevel();
914 if ( spawnVetLevel > vetLevel) // whoever has more, set the other guy
915 {
916 expTracker->setVeterancyLevel(spawnVetLevel);
917 }
918 else if ( spawnVetLevel < vetLevel) // whoever has more, set the other guy
919 {
920 currentSpawn->getExperienceTracker()->setVeterancyLevel(vetLevel);
921 }
922
923 spawnWeaponBonus = currentSpawn->getWeaponBonusCondition();
924
925 avgSpawnPos.add(currentSpawn->getPosition());
926
927 BodyModuleInterface *body = currentSpawn->getBodyModule();
928 acrHealth += body->getHealth();
929 avgHealthMax += body->getMaxHealth();
930
931 spawnDraw = currentSpawn->getDrawable();
932
933 if ( spawnDraw->isSelected() )
934 {
935 SomebodyIsSelected = TRUE;
936 }
937 else
938 {
939 SomebodyIsNotSelected = TRUE;
940 }
941
942 spawnCount++;
943 }
944 } // next iter
945
946
947
948 // SELECTION STATE *****************************************
949 // THIS LOGIC IS SIMPLE
950 // IF ANY ONE OF MY SPAWN ARE SELECTED RIGHT NOW,
951 // I JUST SET THEM ALL SELECTED
952 // THIS WAY WE ARE ALWAYS ONE BIG HAPPY MOB
953 // BUT I ONLY PROCEED IF THERE IS ONE AMONG ALL OF US WHO IS NOT SELECTED
954
955 if ( SomebodyIsSelected && ( ! obj->getDrawable()->isSelected() || SomebodyIsNotSelected ) )
956 {
957 GameMessage *teamMsg = TheMessageStream->appendMessage( GameMessage::MSG_CREATE_SELECTED_GROUP );
958 teamMsg->appendBooleanArgument( FALSE );// not creating new team so pass false
959
960 if ( SomebodyIsNotSelected ) // lets select everybody
961 {
962 for( objectIDListIterator iter = m_spawnIDs.begin(); iter != m_spawnIDs.end(); iter++)
963 {
964 currentSpawn = TheGameLogic->findObjectByID( (*iter) );
965
966 if( currentSpawn )
967 {
968 spawnDraw = currentSpawn->getDrawable();
969
970
971 if ( ! spawnDraw->isSelected())
972 {
973 TheInGameUI->selectDrawable( spawnDraw );
974 TheInGameUI->setDisplayedMaxWarning( FALSE );
975 teamMsg->appendBooleanArgument( FALSE );// not creating new team so pass false
976 teamMsg->appendObjectIDArgument( currentSpawn->getID() );
977 }
978 }
979 } // next iter
980 }
981 // if somebody is selected then I sure need to be!
982 if ( ! obj->getDrawable()->isSelected())
983 {
984 TheInGameUI->selectDrawable( obj->getDrawable() );
985 TheInGameUI->setDisplayedMaxWarning( FALSE );
986 teamMsg->appendBooleanArgument( FALSE );// not creating new team so pass false
987 teamMsg->appendObjectIDArgument( obj->getID() );
988 }
989
990 }
991
992
993
994 // HEALTH BOX POSITION *****************************
995 // pick a centered, average spot to draw the health box
996 avgSpawnPos.scale(1.0f / spawnCount);
997 avgSpawnPos.sub(obj->getPosition());
998 obj->setHealthBoxOffset(avgSpawnPos);
999
1000
1001
1002 // HEALTH STATE *************************************
1003 // make my health an aggregate of all my spawns' healths
1004 if ( spawnCount )
1005 {
1006 avgHealthMax /= spawnCount;
1007 Real perfectTotalHealth = avgHealthMax * spawnCountMax;
1008 Real actualHealth = acrHealth / perfectTotalHealth;
1009 obj->getBodyModule()->setInitialHealth(100.0f * actualHealth);
1010 }
1011 else
1012 {
1013 obj->getBodyModule()->setInitialHealth(0);// I been sick <
1014 }
1015
1016
1017 // HOUSEKEEPING *************************************
1018 //make sure no enemies are shooting at the nexus, since it doesn't 'exist'
1019 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_MASKED ) );
1020
1021}
