104void RebuildHoleExposeDie::onDie( const DamageInfo *damageInfo )
105{
106 if (!isDieApplicable(damageInfo))
107 return;
108
109
110#if defined(_DEBUG) || defined(_INTERNAL) || defined(_ALLOW_DEBUG_CHEATS_IN_RELEASE)
111 if(TheSelectionTranslator->isHandOfGodSelectionMode())
112 {
113 if ( getObject()->isKindOf( KINDOF_STRUCTURE ) )
114 {
115 if ( damageInfo->in.m_damageType == DAMAGE_UNRESISTABLE )
116 return;
117 }
118 }
119#endif
120
121
122
123
124
125 const RebuildHoleExposeDieModuleData *modData = getRebuildHoleExposeDieModuleData();
126 Object *us = getObject();
127
128 //
129 // if we are being constructed from either the first time or from a hole reconstruction
130 // we do not "spawn" a hole object
131 //
132 if( us->getControllingPlayer() != ThePlayerList->getNeutralPlayer()
133 && us->getControllingPlayer()->isPlayerActive()
134 && !us->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
135 {
136 Object *hole;
137
138 // create the hole
139 hole = TheThingFactory->newObject( TheThingFactory->findTemplate( modData->m_holeName ),
140 getObject()->getTeam() );
141
142 // put the hole at our position and angle
143 hole->setPosition( us->getPosition() );
144 hole->setOrientation( us->getOrientation() );
145
146 //
147 // modify the hole extents to be the same as ours because we need to preserve the
148 // same amount of space for the rebuilding process
149 //
150 hole->setGeometryInfo( us->getGeometryInfo() );
151
152 //
153 // Transfer the building's name to the hole
154 //
155 TheScriptEngine->transferObjectName( us->getName(), hole );
156
157 //
158 // add to pathfind map, this really should be wrapped up somewhere in the creation
159 // pipeline of the object automagically!
160 //
161 TheAI->pathfinder()->addObjectToPathfindMap( hole );
162
163 // set the health of the hole to that defined by our data
164 BodyModuleInterface *body = hole->getBodyModule();
165 body->setMaxHealth( modData->m_holeMaxHealth );
166
167 // set the information in the hole about what to build
168 RebuildHoleBehaviorInterface *rhbi = RebuildHoleBehavior::getRebuildHoleBehaviorInterfaceFromObject( hole );
169
170 // sanity
171 DEBUG_ASSERTCRASH( rhbi, ("RebuildHoleExposeDie: No Rebuild Hole Behavior interface on hole\n") );
172
173 // start the rebuild process
174 if( rhbi )
175 rhbi->startRebuildProcess( us->getTemplate(), us->getID() );
176
177 if (modData->m_transferAttackers)
178 {
179 for ( Object *obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject() )
180 {
181 AIUpdateInterface* ai = obj->getAI();
182 if (!ai)
183 continue;
184
185 ai->transferAttack(us->getID(), hole->getID());
186 }
187 }
188
189 } // end if
190
191} // end onDie
