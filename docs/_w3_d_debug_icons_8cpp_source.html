169void W3DDebugIcons::compressIconsArray(void)
170{
171 if (m_debugIcons && m_numDebugIcons > 0) {
172 Int newNum = 0;
173 Int i;
174 for (i=0; i<m_numDebugIcons; i++) {
175 if (m_debugIcons[i].endFrame >= TheGameLogic->getFrame() && i>newNum) {
176 m_debugIcons[newNum] = m_debugIcons[i];
177 newNum++;
178 }
179 }
180 m_numDebugIcons = newNum;
181 }

186void W3DDebugIcons::addIcon(const Coord3D *pos, Real width, Int numFramesDuration, RGBColor color)
187{
188 if (pos==NULL) {
189 if (m_numDebugIcons > maxIcons) {
190 DEBUG_LOG(("Max icons %d\n", m_numDebugIcons));
191 maxIcons = m_numDebugIcons;
192 }
193 m_numDebugIcons = 0;
194 return;
195 }
196 if (m_numDebugIcons>= MAX_ICONS) return;
197 if (m_debugIcons==NULL) return;
198 m_debugIcons[m_numDebugIcons].position = *pos;
199 m_debugIcons[m_numDebugIcons].width = width;
200 m_debugIcons[m_numDebugIcons].color = color;
201 m_debugIcons[m_numDebugIcons].endFrame = TheGameLogic->getFrame()+numFramesDuration;
202 m_numDebugIcons++;
203}

206void W3DDebugIcons::Render(RenderInfoClass & rinfo)
207{
208 //
209 if (WW3D::Are_Static_Sort_Lists_Enabled()) {
210 WW3D::Add_To_Static_Sort_List(this, 1);
211 return;
212 }
213 //
214 Bool anyVanished = false;
215 if (m_numDebugIcons==0) return;
216 DX8Wrapper::Apply_Render_State_Changes();
217
218 DX8Wrapper::Set_Material(m_vertexMaterialClass);
219 DX8Wrapper::Set_Texture(0, NULL);
220 DX8Wrapper::Apply_Render_State_Changes();
221
222 Matrix3D tm(Transform);
223 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
224
225 Int numRect = m_numDebugIcons;
226 static Real offset = 30;
227 const Int MAX_RECT = 5000; // cap drawing n rects.
228 if (numRect > MAX_RECT) numRect = MAX_RECT;
229 offset+= 0.5f;
230 Int k;
231 for (k=0; k<m_numDebugIcons;) {
232 Int curIndex = 0;
233 Int numVertex = 0;
234 DynamicVBAccessClass vb_access(BUFFER_TYPE_DYNAMIC_DX8,DX8_FVF_XYZNDUV2,numRect*4);
235 DynamicIBAccessClass ib_access(BUFFER_TYPE_DYNAMIC_DX8,numRect*6);
236 {
237 DynamicVBAccessClass::WriteLockClass lock(&vb_access);
238 VertexFormatXYZNDUV2* vb= lock.Get_Formatted_Vertex_Array();
239 DynamicIBAccessClass::WriteLockClass lockib(&ib_access);
240 if (!vb) return;
241
242 UnsignedShort *ib=lockib.Get_Index_Array();
243 UnsignedShort *curIb = ib;
244
245// VertexFormatXYZNDUV2 *curVb = vb;
246 Real shadeR, shadeG, shadeB;
247 shadeR = 0;
248 shadeG = 0;
249 shadeB = 255;
250 try {
251 for(; numVertex<numRect*4 && k<m_numDebugIcons; k++) {
252 Int theAlpha = 64;
253 const Int FADE_FRAMES = 100;
254 Int framesLeft = m_debugIcons[k].endFrame - TheGameLogic->getFrame();
255 if (framesLeft < 1) {
256 anyVanished = true;
257 continue;
258 }
259 if (framesLeft<FADE_FRAMES) {
260 theAlpha *= (Real)framesLeft/FADE_FRAMES;
261 }
262 RGBColor clr = m_debugIcons[k].color;
263 Real halfWidth = m_debugIcons[k].width/2;
264 Int diffuse = clr.getAsInt() | ((int)theAlpha << 24);
265 Coord3D pt1 = m_debugIcons[k].position;
266 vb->x= pt1.x-halfWidth;
267 vb->y= pt1.y-halfWidth;
268 vb->z= pt1.z;
269 vb->diffuse=diffuse; // b g<<8 r<<16 a<<24.
270 vb->u1=0 ;
271 vb->v1=0 ;
272 vb++;
273 vb->x= pt1.x+halfWidth;
274 vb->y= pt1.y-halfWidth;
275 vb->z= pt1.z;
276 vb->diffuse=diffuse; // b g<<8 r<<16 a<<24.
277 vb->u1=0 ;
278 vb->v1=0 ;
279 vb++;
280 vb->x= pt1.x+halfWidth;
281 vb->y= pt1.y+halfWidth;
282 vb->z= pt1.z;
283 vb->diffuse=diffuse; // b g<<8 r<<16 a<<24.
284 vb->u1=0 ;
285 vb->v1=0 ;
286 vb++;
287 vb->x= pt1.x-halfWidth;
288 vb->y= pt1.y+halfWidth;
289 vb->z= pt1.z;
290 vb->diffuse=diffuse; // b g<<8 r<<16 a<<24.
291 vb->u1=0 ;
292 vb->v1=0 ;
293 vb++;
294 *curIb++ = numVertex;
295 *curIb++ = numVertex+1;
296 *curIb++ = numVertex+2;
297 *curIb++ = numVertex;
298 *curIb++ = numVertex+2;
299 *curIb++ = numVertex+3;
300 curIndex += 6;
301 numVertex += 4;
302 }
303 IndexBufferExceptionFunc();
304 } catch(...) {
305 IndexBufferExceptionFunc();
306 }
307
308 }
309 if (numVertex == 0) break;
310 DX8Wrapper::Set_Shader(ShaderClass(SC_ALPHA));
311 DX8Wrapper::Set_Index_Buffer(ib_access,0);
312 DX8Wrapper::Set_Vertex_Buffer(vb_access);
313 DX8Wrapper::Draw_Triangles( 0,curIndex/3, 0, numVertex); //draw a quad, 2 triangles, 4 verts
314 }
315
316 if (anyVanished) {
317 compressIconsArray();
318 }
319}
