73// Add with carry. SUM is replaced with A + B + C, C is replaced with 1 if there was a carry, 0 if there wasn't. A carry occurred if the sum is less than one of the inputs. This is addition, so carry can never be more than one.
74#define ADC(SUM, A, B, C) SUM = (A) + (B) + (C); C = ((SUM < (A)) || (SUM < (B)))
75
76static UnsignedInt randomValue(UnsignedInt *seed)
77{
78 UnsignedInt ax;
79 UnsignedInt c = 0;
80
81
82 ADC(ax, seed[5], seed[4], c); /* mov ax,seed+20 */
83 /* add ax,seed+16 */
84 seed[4] = ax; /* mov seed+8,ax */
85
86 ADC(ax, ax, seed[3], c); /* adc ax,seed+12 */
87 seed[3] = ax; /* mov seed+12,ax */
88
89 ADC(ax, ax, seed[2], c); /* adc ax,seed+8 */
90 seed[2] = ax; /* mov seed+8,ax */
91
92 ADC(ax, ax, seed[1], c); /* adc ax,seed+4 */
93 seed[1] = ax; /* mov seed+4,ax */
94
95 ADC(ax, ax, seed[0], c); /* adc ax,seed+0 */
96 seed[0] = ax; /* mov seed+0,ax */
97
98 /* Increment seed array, bubbling up the carries. */
99 if (!++seed[5])
100 {
101 if (!++seed[4])
102 {
103 if (!++seed[3])
104 {
105 if (!++seed[2])
106 {
107 if (!++seed[1])
108 {
109 ++seed[0];
110 ++ax;
111 }
112 }
113 }
114 }
115 }
116 return(ax);
117}

119static void seedRandom(UnsignedInt SEED, UnsignedInt *seed)
120{
121 UnsignedInt ax;
122
123 ax = SEED; /* mov eax,SEED */
124 ax += 0xf22d0e56; /* add eax,0f22d0e56h */
125 seed[0] = ax; /* mov seed,eax */
126 ax += 0x883126e9 - 0xf22d0e56; /* add eax,0883126e9h-0f22d0e56h */
127 seed[1] = ax; /* mov seed+4,eax */
128 ax += 0xc624dd2f - 0x883126e9; /* add eax,0c624dd2fh-0883126e9h */
129 seed[2] = ax; /* mov seed+8,eax */
130 ax += 0x0702c49c - 0xc624dd2f; /* add eax,00702c49ch-0c624dd2fh */
131 seed[3] = ax; /* mov seed+12,eax */
132 ax += 0x9e353f7d - 0x0702c49c; /* add eax,09e353f7dh-00702c49ch */
133 seed[4] = ax; /* mov seed+16,eax */
134 ax += 0x6fdf3b64 - 0x9e353f7d; /* add eax,06fdf3b64h-09e353f7dh */
135 seed[5] = ax; /* mov seed+20,eax */
136}

285Real GetGameLogicRandomValueReal( Real lo, Real hi, char *file, int line )
286{
287 Real delta = hi - lo;
288 Real rval;
289

293 rval = ((Real)(randomValue(theGameLogicSeed)) * theMultFactor ) * delta + lo;

377Real GameClientRandomVariable::getValue( void ) const
378{
379 switch( m_type )
380 {
381 case CONSTANT:
382 DEBUG_ASSERTLOG(m_low == m_high, ("m_low != m_high for a CONSTANT GameClientRandomVariable\n"));
383 if (m_low == m_high) {
384 return m_low;
385 } // else return as though a UNIFORM.
386
387 case UNIFORM:
388 return GameClientRandomValueReal( m_low, m_high );
389
390 default:
392 DEBUG_CRASH(("unsupported DistributionType in GameClientRandomVariable::getValue\n"));
393 return 0.0f;
394 }
395}
