184// CWorldBuilderView::getColorForHeight
185//=============================================================================
188//=============================================================================
189DWORD CWorldBuilderView::getColorForHeight(UnsignedByte ht)
190{
191 Int fromR, fromG, fromB;
192 Int toR, toG, toB;
193 Int min, max;
194 // black blue red green magenta cyan yellow white
195 // 0x00 0x24 0x48 0x6c 0x90 0xb4 0xd8 0xff
196 if (ht<0x24) {
197 // 0 - 0x24 goes from black to blue
198 fromR = fromG = fromB = toR = toG = 0;
199 toB = 0xff;
200 min = 0;
201 max = 0x24;
202 } else if (ht < 0x48) {
203 // 0x24-0x48 goes from blue to red
204 fromR = 0; fromG = 0; fromB = 0xFF;
205 toR = 0xff; toG = 0; toB = 0x00;
206 min = 0x24;
207 max = 0x48;
208 } else if (ht < 0x6c) {
209 // 0x48-0x6c goes from red to green
210 fromR = 0xFF; fromG = 0; fromB = 0;
211 toR = 0; toG = 0xff; toB = 0;
212 min = 0x48;
213 max = 0x6c;
214 } else if (ht < 0x90) {
215 // 0x6c-0x90 goes from green to magenta
216 fromR = 0; fromG =0xFF; fromB = 0;
217 toR = 0xff; toG = 0; toB = 0xff;
218 min = 0x6c;
219 max = 0x90;
220 } else if (ht < 0xb4) {
221 // 0x90-0xb4 goes from magenta to cyan
222 fromR = 0xff; fromG =0; fromB = 0xff;
223 toR = 0; toG = 0xFF; toB = 0xff;
224 min = 0x90;
225 max = 0xb4;
226 } else if (ht < 0xd8) {
227 // 0xb4-0xd8 goes from cyan to yellow
228 fromR = 0; fromG =0xf; fromB = 0xff;
229 toR = 0xff; toG = 0xFF; toB = 0;
230 min = 0xb4;
231 max = 0xd8;
232 } else {
233 // 0xd8-0xff goes from yellow to white
234 fromR = 0xff; fromG =0xff; fromB = 0;
235 toR = 0xff; toG = 0xFF; toB = 0xFF;
236 min = 0xd8;
237 max = 0xff;
238 }
239 Int delta = max-min;
240 Int to = ht-min;
241 Int from = max-ht;
242 // Interpolate red value.
243 Int r = (fromR*from + toR*to +delta/2)/delta;
244 if (r<0) r = 0;
245 if (r>255) r = 255;
246 // Interpolate green value.
247 Int g = (fromG*from + toG*to +delta/2)/delta;
248 if (g<0) g = 0;
249 if (g>255) g = 255;
250 // Interpolate blue value.
251 Int b = (fromB*from + toB*to +delta/2)/delta;
252 if (b<0) b = 0;
253 if (b>255) b = 255;
254 return(RGB(r,g,b));
255}

257//=============================================================================
258// CWorldBuilderView::OnPaint
259//=============================================================================
261//=============================================================================
262void CWorldBuilderView::OnPaint()
263{
264 CWorldBuilderDoc* pDoc = WbDoc();
265 ASSERT_VALID(pDoc);
266
267 // If we are editing with a tool, draw the map being edited, if one exists.
268 WorldHeightMapEdit *pMap = getTrackingHeightMap();
269
270 // If no map yet, there is nothing to draw.
271 if (pMap==NULL) return;
272
273 Int minJ = 0;
274 Int minI = 0;
275 Int maxJ = pMap->getYExtent();
276 Int maxI = pMap->getXExtent();
277
278 CRect updateRect;
279 CRgn updateRgn;
280 updateRgn.CreateRectRgn(0,0,1,1);
281 if (NULLREGION != GetUpdateRgn(&updateRgn)) {
282 // The region isn't null, so get the bounding box.
283 updateRgn.GetRgnBox(&updateRect);
284 // Get the visible pixel bounds.
285 Int left = updateRect.left + mXScrollOffset;
286 Int right = updateRect.right + mXScrollOffset;
287 Int top = updateRect.top + mYScrollOffset;
288 Int bottom = updateRect.bottom + mYScrollOffset;
289 // Convert the visible pixel bounds to the index bounds for the map.
290 minI = (left/m_cellSize);
291 maxI = (right/m_cellSize)+1;
292 // Clip the bounds to the size of the actual height map.
293 if (minI<0) minI=0;
294 if (maxI>=pMap->getXExtent()) maxI = pMap->getXExtent()-1;
295
296 // screen goes down, map goes up, so gotta dance a little on the verticals.
297 Int min = (top/m_cellSize);
298 Int max = (bottom/m_cellSize)+1;
299
300 if (min<0) min=0;
301 // Clip the bounds to the size of the actual height map.
302 if (max>=pMap->getYExtent()) max = pMap->getYExtent();
303
304
305 maxJ = pMap->getYExtent() - min;
306 minJ = pMap->getYExtent() - max;
307 if (maxJ>=pMap->getYExtent()) maxJ = pMap->getYExtent()-1;
308 }
309
310
311 CPaintDC dc(this); // device context for painting
312 // Offset the origin so that we can draw on 0 based coordinates.
313 dc.SetViewportOrg(-mXScrollOffset, -mYScrollOffset);
314 updateRgn.OffsetRgn(mXScrollOffset, mYScrollOffset);
315 Int i, j;
316 CRect rect;
317 CBrush brush;
318 // gray brush for drawing the grid.
319 brush.CreateSolidBrush(RGB(128,128,128));
320 CRect clientRect;
321 GetClientRect(&clientRect);
322 clientRect.OffsetRect(mXScrollOffset, mYScrollOffset);
323 if (maxJ == pMap->getYExtent()-1) {
324 rect = clientRect; // Do top.
325 rect.bottom = (1)*m_cellSize;
326 dc.FillSolidRect(&rect, RGB(255,255,255));
327 }
328 if (minJ == 0) {
329 rect = clientRect; // Do bottom.
330 rect.top = (pMap->getYExtent())*m_cellSize;
331 dc.FillSolidRect(&rect, RGB(255,255,255));
332 }
333 if (minI == 0) {
334 rect = clientRect; // Do left;
335 rect.right = 0;
336 dc.FillSolidRect(&rect, RGB(255,255,255));
337 }
338 if (maxI == pMap->getXExtent()-1) {
339 rect = clientRect; // Do left;
340 rect.left = (pMap->getYExtent()-1)*m_cellSize;
341 dc.FillSolidRect(&rect, RGB(255,255,255));
342 }
343
344 for (j=maxJ-1; j>=minJ; j--) { // map goes up, screen goes down.
345
346 rect.bottom = (pMap->getYExtent()-j)*m_cellSize;
347 rect.top = rect.bottom-m_cellSize;
348 for (i=minI; i<maxI; i++) {
349 UnsignedByte ht = pMap->getHeight(i, j);
350 rect.left = i*m_cellSize;
351 rect.right = rect.left+m_cellSize;
352 // If this cell is not visible, don't bother drawing it.
353 if (!updateRgn.RectInRegion(&rect)) {
354 continue;
355 }
356#ifdef INTENSE_DEBUG
357 dc.FillSolidRect(&rect, RGB(0,255,0));
358 ::Sleep(5);
359#endif
360 Int width = m_cellSize;
361 UnsignedByte *pData = NULL;
362 if (m_showTexture) {
363 }
364 // Draw the texture if we have one, else the height color.
365 if ((pData!=NULL)) {
366 drawMyTexture(&dc, &rect, width, pData);
367 } else {
368 dc.FillSolidRect(&rect, getColorForHeight(ht));
369 }
370 // Draw the grid. It doesn't make sense to draw it if the width
371 // in pixels is less than 3, as all you see is the grid.
372 if (mShowGrid && m_cellSize>=MIN_GRID_SIZE) {
373 CRect frameRect=rect;
374 frameRect.bottom++; frameRect.right++;
375 dc.FrameRect(&frameRect, &brush);
376 }
377 }
378 }
379
380 // Draw the contours if they are on.
381 if (m_cellSize>=MIN_GRID_SIZE && m_showContours) {
382 drawContours(&dc, &updateRgn, minI, maxI, minJ, maxJ);
383 }
384
385 // Draw the icons for the objects if they are turned on.
386 if (m_showObjects) {
387 MapObject *pMapObj = MapObject::getFirstMapObject();
388 while (pMapObj) {
389 // Assume objects draw approx grids across.
390 CPoint lpt;
391 docToViewCoords(*pMapObj->getLocation(), &lpt);
392 lpt.x += mXScrollOffset;
393 lpt.y += mYScrollOffset;
394 // Create a rect centered on the object and 2 cells wide & tall.
395 CRect rect;
396 rect.bottom = lpt.y - m_cellSize;
397 rect.top = lpt.y + m_cellSize;
398 rect.left = lpt.x - m_cellSize;
399 rect.right = lpt.x + m_cellSize;
400 // If visible in the update region, draw it.
401 if (updateRgn.RectInRegion(&rect))
402 {
403 drawObjectInView(&dc, pMapObj);
404 }
405 pMapObj = pMapObj->getNext();
406 }
407 }
408
409}

417void CWorldBuilderView::invalObjectInView(MapObject *pMapObj)
418{
419 if (pMapObj == NULL) {
420 Invalidate(false);
421 return;
422 }
423
424 CPoint lpt;
425 docToViewCoords(*pMapObj->getLocation(), &lpt);
426
427 // Calculate the bounds of the circle.
428 CRect bounds(lpt, lpt);
429 bounds.InflateRect( m_cellSize/2+2, m_cellSize/2+2);
430
431 // Calculate the bounds of the arrowhead.
432 Coord3D ac;
433 WbDoc()->getObjArrowPoint(pMapObj, &ac);
434 CPoint arrpt;
435 docToViewCoords(ac, &arrpt);
436
437 CRect arrow(arrpt.x, arrpt.y, arrpt.x, arrpt.y);
438 arrow.InflateRect(1,1);
439 // Union the two bounds.
440 CRect fullBounds;
441 fullBounds.UnionRect(&bounds, &arrow);
442// fullBounds.OffsetRect(-mXScrollOffset, -mYScrollOffset);
443 // Invalidate it.
444 InvalidateRect(&fullBounds, false);
445}

452void CWorldBuilderView::drawObjectInView(CDC *pDc, MapObject *pMapObj)
453{
454 CPoint lpt;
455 docToViewCoords(*pMapObj->getLocation(), &lpt);
456 // un-adjuse by scroll offset since we are called from OnPaint, which adjusts by this amt
457 lpt.x += mXScrollOffset;
458 lpt.y += mYScrollOffset;
459
460 CRect circle(lpt, lpt);
461 circle.InflateRect(m_cellSize/2, m_cellSize/2);
462 CPen ovalPen;
463 CBrush brush;
464 CGdiObject *savPen;
465 CGdiObject *savBrush;
466 // Create a pen to draw the icon of the object's color.
467 ovalPen.CreatePen(PS_SOLID, 1, pMapObj->getColor());
468 // Null brush, as the object icon is not filled.
469 brush.CreateStockObject(NULL_BRUSH);
470 // Select the pen & brush.
471 savPen = pDc->SelectObject(&ovalPen);
472 savBrush = pDc->SelectObject(&brush);
473 // Draw the object's circle.
474 pDc->Ellipse(&circle);
475
476 // Draw the arrow
477 Int delta = (m_cellSize+4)/4;
478 Int smallDelta = (delta+4)/4;
479
480 float angle = pMapObj->getAngle();
481
482 Vector3 head1(m_cellSize - delta, smallDelta, 0);
483 Vector3 head2(m_cellSize - delta, -smallDelta, 0);
484 head1.Rotate_Z(-angle);
485 head2.Rotate_Z(-angle);
486
487 Coord3D cpt2;
488 WbDoc()->getObjArrowPoint(pMapObj, &cpt2);
489 CPoint pt2;
490 docToViewCoords(cpt2, &pt2);
491 // un-adjuse by scroll offset since we are called from OnPaint, which adjusts by this amt
492 pt2.x += mXScrollOffset;
493 pt2.y += mYScrollOffset;
494 pDc->MoveTo(lpt.x,lpt.y);
495 pDc->LineTo(pt2.x, pt2.y);
496
497 CPoint pt1;
498 pt1.x = pt2.x - ROUND(head1.X);
499 pt1.y = pt2.y - ROUND(head1.Y);
500 pDc->LineTo(pt1);
501
502 pDc->MoveTo(pt2.x, pt2.y);
503 pt1.x = pt2.x - ROUND(head2.X);
504 pt1.y = pt2.y - ROUND(head2.Y);
505 pDc->LineTo(pt1);
506
507 if (pMapObj->isSelected()) {
508 // The object is selected, so draw selection feedback.
509 CPoint pts[3];
510 pts[0].x = lpt.x;
511 pts[0].y = lpt.y;
512 // Solid yellow pen and brush, as the selection feedback is filled.
513 CPen selectPen;
514 selectPen.CreatePen(PS_SOLID, 1, RGB(255,255,0));
515 CBrush selectBrush;
516 selectBrush.CreateSolidBrush(RGB(255,255,0));
517 pDc->SelectObject(&selectPen);
518 pDc->SelectObject(&selectBrush);
519 // Draw a triangle centered about the object center.
520 pts[0].x += delta;
521 pts[0].y += delta;
522 pts[1] = pts[0];
523 pts[1].x -= 2*delta;
524 pts[2].x = lpt.x;
525 pts[2].y = lpt.y;
526 pts[2].y -= 2*delta;
527 pDc->Polygon(pts, 3);
528 // Restore the previous brush & pen to prevent leaks.
529 pDc->SelectObject(&ovalPen);
530 pDc->SelectObject(&brush);
531 }
532
533 // Restore the previous brush & pen to prevent leaks.
534 pDc->SelectObject(savPen);
535 pDc->SelectObject(savBrush);
536}

538//=============================================================================
539// CWorldBuilderView::drawContours
540//=============================================================================
542//=============================================================================
543void CWorldBuilderView::drawContours(CDC *pDc, CRgn *pRgn, Int minX, Int maxX, Int minY, Int maxY)
544{
545 CWorldBuilderDoc* pDoc = WbDoc();
546 ASSERT_VALID(pDoc);
547
548 WorldHeightMapEdit *pMap = getTrackingHeightMap();
549 // We do cell and cell+1 in the loop, so trim back the max limits.
550 if (maxX == pMap->getXExtent()) maxX--;
551 if (maxY == pMap->getYExtent()) maxY--;
552
553 Int curHeight;
554 Bool didWater = false; // We do the water level first, then step through the contours.
555 for (curHeight = ContourOptions::getContourOffset(); curHeight < 255;
556 curHeight+= ContourOptions::getContourStep()) {
557 Bool doingWater = false;
558 CPen pen;
559 CGdiObject *savObj;
560 Int width = ContourOptions::getContourWidth();
561 if (!didWater) {
562 // Do the water first.
563 didWater = true;
564 doingWater = true;
565 curHeight = ROUND(TheGlobalData->m_waterPositionZ/MAP_HEIGHT_SCALE);
566 pen.CreatePen(PS_SOLID, width+1, RGB(0, 255, 255));
567 } else {
568 pen.CreatePen(PS_SOLID, width, getColorForHeight(curHeight));
569 }
570 savObj = pDc->SelectObject(&pen);
571 Int i, j;
572 CRect rect;
573 for (i=minX; i<maxX; i++) {
574 for (j=minY; j<maxY; j++) {
575 rect.bottom = (pMap->getYExtent()-j)*m_cellSize;
576 rect.top = rect.bottom-m_cellSize;
577 rect.left = i*m_cellSize;
578 rect.right = rect.left+m_cellSize;
579 if (!pRgn->RectInRegion(&rect)) {
580 continue;
581 }
582 CPoint lowerLeftPt, lowerRightPt, upperLeftPt, upperRightPt;
583 Int lowerLeft = pMap->getHeight(i, j);
584 Int upperLeft = pMap->getHeight(i, j+1);
585 Int upperRight = pMap->getHeight(i+1, j+1);
586 Int lowerRight = pMap->getHeight(i+1, j);
587 lowerLeftPt.x = rect.left;
588 lowerLeftPt.y = rect.bottom;
589 upperLeftPt.x = rect.left;
590 upperLeftPt.y = rect.top;
591 lowerRightPt.x = rect.right;
592 lowerRightPt.y = rect.bottom;
593 upperRightPt.x = rect.right;
594 upperRightPt.y = rect.top;
595
596 // if the square is flat, we have no contours.
597 if (lowerLeft==upperRight && lowerLeft == upperLeft && upperLeft == lowerRight) {
598 continue;
599 }
600 // If it is all below or all above, we have no contours.
601 if (lowerLeft<curHeight && upperLeft<curHeight && upperRight<curHeight && lowerRight<curHeight) {
602 continue;
603 }
604 if (lowerLeft>curHeight && upperLeft>curHeight && upperRight>curHeight && lowerRight>curHeight) {
605 continue;
606 }
607 Bool drawLine = false;
608 CPoint pt1, pt2;
609 // Consider the lower left, upper left, upper right triangle.
610 while (true) {
611 if (lowerLeft != upperLeft || lowerLeft != upperRight) {
612 if (lowerLeft == curHeight) {
613 if (isBetween(curHeight, upperRight, upperLeft)) {
614 pt1 = lowerLeftPt;
615 interpolate(&pt2, curHeight, upperLeftPt, upperLeft, upperRightPt, upperRight);
616 drawLine = true;
617 }
618 break;
619 }
620 if (upperLeft == curHeight) {
621 if (isBetween(curHeight, upperRight, lowerLeft)) {
622 pt1 = upperLeftPt;
623 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerLeftPt, lowerLeft);
624 drawLine = true;
625 }
626 break;
627 }
628 if (upperRight == curHeight) {
629 if (isBetween(curHeight, lowerLeft, upperLeft)) {
630 pt1 = upperRightPt;
631 interpolate(&pt2, curHeight, lowerLeftPt, lowerLeft, upperLeftPt, upperLeft);
632 drawLine = true;
633 }
634 break;
635 }
636 if (isBetween(curHeight, lowerLeft, upperLeft)) {
637 interpolate(&pt1, curHeight, lowerLeftPt, lowerLeft, upperLeftPt, upperLeft);
638 if (isBetween(curHeight, upperRight, upperLeft)) {
639 interpolate(&pt2, curHeight, upperLeftPt, upperLeft, upperRightPt, upperRight);
640 drawLine = true;
641 }
642 if (isBetween(curHeight, upperRight, lowerLeft)) {
643 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerLeftPt, lowerLeft);
644 drawLine = true;
645 }
646 break;
647 }
648 if (isBetween(curHeight, upperRight, upperLeft)) {
649 interpolate(&pt1, curHeight, upperLeftPt, upperLeft, upperRightPt, upperRight);
650 if (isBetween(curHeight, upperRight, lowerLeft)) {
651 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerLeftPt, lowerLeft);
652 drawLine = true;
653 }
654 break;
655 }
656 }
657 break;
658 }
659 if (drawLine) {
660 pDc->MoveTo(pt1);
661 pDc->LineTo(pt2);
662 }
663
664 drawLine = false;
665 // Consider the lower left, upper right, lowerRight triangle.
666 while (true) {
667 if (lowerLeft != lowerRight || lowerLeft != upperRight) {
668 if (lowerLeft == curHeight) {
669 if (isBetween(curHeight, upperRight, lowerRight)) {
670 pt1 = lowerLeftPt;
671 interpolate(&pt2, curHeight, lowerRightPt, lowerRight, upperRightPt, upperRight);
672 drawLine = true;
673 }
674 break;
675 }
676 if (lowerRight == curHeight) {
677 if (isBetween(curHeight, upperRight, lowerLeft)) {
678 pt1 = lowerRightPt;
679 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerLeftPt, lowerLeft);
680 drawLine = true;
681 }
682 break;
683 }
684 if (upperRight == curHeight) {
685 if (isBetween(curHeight, lowerLeft, lowerRight)) {
686 pt1 = upperRightPt;
687 interpolate(&pt2, curHeight, lowerLeftPt, lowerLeft, lowerRightPt, lowerRight);
688 drawLine = true;
689 }
690 break;
691 }
692
693 if (isBetween(curHeight, lowerLeft, lowerRight)) {
694 interpolate(&pt1, curHeight, lowerLeftPt, lowerLeft, lowerRightPt, lowerRight);
695 if (isBetween(curHeight, upperRight, lowerLeft)) {
696 interpolate(&pt2, curHeight, lowerLeftPt, lowerLeft, upperRightPt, upperRight);
697 drawLine = true;
698 }
699 if (isBetween(curHeight, upperRight, lowerRight)) {
700 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerRightPt, lowerRight);
701 drawLine = true;
702 }
703 break;
704 }
705 if (isBetween(curHeight, upperRight, lowerRight)) {
706 interpolate(&pt1, curHeight, lowerRightPt, lowerRight, upperRightPt, upperRight);
707 if (isBetween(curHeight, upperRight, lowerLeft)) {
708 interpolate(&pt2, curHeight, upperRightPt, upperRight, lowerLeftPt, lowerLeft);
709 drawLine = true;
710 }
711 break;
712 }
713 }
714 break;
715 }
716 if (drawLine) {
717 pDc->MoveTo(pt1);
718 pDc->LineTo(pt2);
719 }
720 }
721 }
722 pDc->SelectObject(savObj);
723 if (doingWater) {
724 curHeight = ContourOptions::getContourOffset() - ContourOptions::getContourStep();

732//=============================================================================
735//=============================================================================
736void CWorldBuilderView::interpolate(CPoint *pt, Int ht, CPoint pt1, Int ht1, CPoint pt2, Int ht2)
737{
738 DEBUG_ASSERTCRASH((ht1!=ht2),("oops"));
739 // Paranoid check to avoid divide by zero.
740 if (ht1==ht2) {
741 *pt = pt1;
742 return;
743 }
744 Int delta = ht2 - ht1;
745 Int d1 = ht2-ht;
746 Int d2 = ht-ht1;
747 DEBUG_ASSERTCRASH((d1+d2==delta),("oops"));
748 // Interpolate between pt1 and pt2.
749 pt->x = (pt1.x*d1 + pt2.x*d2)/delta;
750 pt->y = (pt1.y*d1 + pt2.y*d2)/delta;
751}

807void CWorldBuilderView::adjustDocSize()
808{
809 CWorldBuilderDoc* pDoc = WbDoc();
810 ASSERT_VALID(pDoc);
811 // Initialize to a sensible default as we may get redrawn before we load a map.
812 Int height = m_cellSize*256;
813 Int width = m_cellSize*256;
814
815 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
816 // If we have a map loaded, use the actual size.
817 if (pMap) {
818 height = m_cellSize * pMap->getYExtent();
819 width = m_cellSize * pMap->getXExtent();
820 }
821 CRect clientRect;
822 GetClientRect(&clientRect);
823 Int flippedY = m_centerPt.Y;
824 if (pMap && flippedY>0) {
825 flippedY = pMap->getYExtent()-flippedY;
826 }
827 // Calculate the scroll offset to place the cell m_centerX, m_centerY in the middle.
828 mXScrollOffset = (m_centerPt.X*m_cellSize)-clientRect.Width()/2;
829 mYScrollOffset = (flippedY*m_cellSize)-clientRect.Height()/2;
830 // Calculate the scroll bar limits so we can have up to 1/2 screen of white space.
831 m_scrollMin.x = -clientRect.Width()/2;
832 m_scrollMax.x = width-clientRect.Width()/2;
833 m_scrollMin.y = -clientRect.Height()/2;
834 m_scrollMax.y = width-clientRect.Height()/2;
835 // Clip the offset so it fits inside the range.
836 if (mXScrollOffset > m_scrollMax.x) {
837 mXScrollOffset = m_scrollMax.x;
838 }
839 if (mYScrollOffset > m_scrollMax.y) {
840 mYScrollOffset = m_scrollMax.y;
841 }
842 // Set the scroll range
843 SetScrollRange(SB_HORZ, m_scrollMin.x, m_scrollMax.x, false);
844 SetScrollRange(SB_VERT, m_scrollMin.y, m_scrollMax.y, false);
845 // Set the position of the scroll button in the scroll bars.
846 SetScrollPos(SB_HORZ, mXScrollOffset, true);
847 SetScrollPos(SB_VERT, mYScrollOffset, true);
848}

905//=============================================================================
906// CWorldBuilderView::scroll
907//=============================================================================
909//=============================================================================
910void CWorldBuilderView::scrollInView(Real xScroll, Real yScroll, Bool end)
911{
912 // Calculate new scroll offset.
913 Int newOffset = mXScrollOffset + xScroll*m_cellSize;
914 // Clip it to our maximum scroll range.
915 if (newOffset < m_scrollMin.x) newOffset = m_scrollMin.x;
916 if (newOffset > m_scrollMax.x) newOffset = m_scrollMax.x;
917 // Back calculate the number of pixels to scroll.
918 xScroll = newOffset - mXScrollOffset;
919 mXScrollOffset = newOffset;
920
921 // Calculate new scroll offset.
922 newOffset = mYScrollOffset - yScroll*m_cellSize;
923 // Clip it to our maximum scroll range.
924 if (newOffset < m_scrollMin.y) newOffset = m_scrollMin.y;
925 if (newOffset > m_scrollMax.y) newOffset = m_scrollMax.y;
926 // Back calculate the number of pixels to scroll.
927 yScroll = newOffset - mYScrollOffset;
928 mYScrollOffset = newOffset;
929
930 if (xScroll || yScroll) {
931 ScrollWindow(-xScroll, -yScroll);
932 SetScrollPos(SB_HORZ, mXScrollOffset, true);
933 SetScrollPos(SB_VERT, mYScrollOffset, true);
934 // Update the center indexes.
935 CRect client;
936 GetClientRect(&client);
937 // pt is the center point in pixels, 0 based.
938 CPoint pt((client.left+client.right)/2+mXScrollOffset, (client.bottom+client.top)/2+mYScrollOffset);
939 CWorldBuilderDoc* pDoc = WbDoc();
940 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
941 if (pMap==NULL) return;
942 m_centerPt.X = (Real)(pt.x)/m_cellSize;
943 m_centerPt.Y = pMap->getYExtent() - (Real)(pt.y)/m_cellSize;
944 constrainCenterPt();
945 }
946 if (end)
947 WbDoc()->syncViewCenters(m_centerPt.X, m_centerPt.Y);
948}

1030Bool CWorldBuilderView::viewToDocCoords(CPoint curPt, Coord3D *newPt, Bool constrain)
1031{
1032 // Flip the y.
1033 WorldHeightMapEdit *pMap = WbDoc()->GetHeightMap();
1034 newPt->x = (Real)(curPt.x + mXScrollOffset)/m_cellSize;
1035 newPt->y = pMap->getYExtent() - (Real)(curPt.y + mYScrollOffset)/m_cellSize;
1036 newPt->x *= MAP_XY_FACTOR;
1037 newPt->y *= MAP_XY_FACTOR;
1038 newPt->z = MAGIC_GROUND_Z;
1039 if (constrain) {
1040 if (m_doLockAngle) {
1041 Real dy = fabs(m_mouseDownDocPoint.y - newPt->y);
1042 Real dx = fabs(m_mouseDownDocPoint.x - newPt->x);
1043 if (dx>2*dy) {
1044 // lock to dx.
1045 newPt->y = m_mouseDownDocPoint.y;
1046 } else if (dy>2*dx) {
1047 //lock to dy.
1048 newPt->x = m_mouseDownDocPoint.x;
1049 } else {
1050 // Lock to 45 degree.
1051 dx = (dx+dy)/2;
1052 dy = dx;
1053 if (newPt->x < m_mouseDownDocPoint.x) dx = -dx;
1054 if (newPt->y < m_mouseDownDocPoint.y) dy = -dy;
1055 newPt->x = m_mouseDownDocPoint.x+dx;
1056 newPt->y = m_mouseDownDocPoint.y+dy;
1057 }
1058 }
1059 }
1060#ifdef X_DEBUG
1061CPoint curPt2;
1062docToViewCoords(*newPt, &curPt2);
1063DEBUG_ASSERTCRASH((curPt.x==curPt2.x),("oops"));
1064DEBUG_ASSERTCRASH((curPt.y==curPt2.y),("oops"));
1065#endif
1066 return true;
1067}
