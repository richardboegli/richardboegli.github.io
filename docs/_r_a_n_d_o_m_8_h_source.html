47/*
48** This class functions like a 'magic' int value that returns a random number
49** every time it is read. To set the "random seed" for this, just assign a number
50** to the object (just as you would if it were a true 'int' value). Take note that although
51** this class will return an 'int', the actual significance of the random number is
52** limited to 15 bits (0..32767).
53*/
54class RandomClass {
55 public:
56 RandomClass(unsigned seed=0);
57
58 operator int(void) {return(operator()());};
59 int operator() (void);
60 int operator() (int minval, int maxval);
61
62 enum {
63 SIGNIFICANT_BITS=15 // Random number bit significance.
64 };
65
66 protected:
67 unsigned long Seed;
68
69 /*
70 ** Internal working constants that are used to generate the next
71 ** random number.
72 */
73 enum {
74 MULT_CONSTANT=0x41C64E6D, // K multiplier value.
75 ADD_CONSTANT=0x00003039, // K additive value.
76 THROW_AWAY_BITS=10 // Low bits to throw away.
77 };
78};

194/***********************************************************************************************
195 * Pick_Random_Number -- Picks a random number between two values (inclusive). *
196 * *
197 * This is a utility template that works with one of the random number classes. Given a *
198 * random number generator, this routine will return with a value that lies between the *
199 * minimum and maximum values specified (subject to the bit precision limitations of the *
200 * random number generator itself). *
201 * *
202 * INPUT: generator -- Reference to the random number generator to use for the pick *
203 * process. *
204 * *
205 * minval -- The minimum legal return value (inclusive). *
206 * *
207 * maxval -- The maximum legal return value (inclusive). *
208 * *
209 * OUTPUT: Returns with a random number between the minval and maxval (inclusive). *
210 * *
211 * WARNINGS: none *
212 * *
213 * HISTORY: *
214 * 05/23/1997 JLB : Created. *
215 *=============================================================================================*/
216template<class T>
217int Pick_Random_Number(T & generator, int minval, int maxval)
218{
219 /*
220 ** Test for shortcut case where the range is null and thus
221 ** the number to return is actually implicit from the
222 ** parameters.
223 */
224 if (minval == maxval) return(minval);
225
226 /*
227 ** Ensure that the min and max range values are in proper order.
228 */
229 if (minval > maxval) {
230 int temp = minval;
231 minval = maxval;
232 maxval = temp;
233 }
234
235 /*
236 ** Find the highest bit that fits within the magnitude of the
237 ** range of random numbers desired. Notice that the scan is
238 ** limited to the range of significant bits returned by the
239 ** random number algorithm.
240 */
241 int magnitude = maxval - minval;
242 int highbit = T::SIGNIFICANT_BITS-1;
243 while ((magnitude & (1 << highbit)) == 0 && highbit > 0) {
244 highbit--;
245 }
246
247 /*
248 ** Create a full bit mask pattern that has all bits set that just
249 ** barely covers the magnitude of the number range desired.
250 */
251 int mask = ~( (~0L) << (highbit+1));
252
253 /*
254 ** Keep picking random numbers until it fits within the magnitude desired. With a
255 ** good random number generator, it will have to perform this loop an average
256 ** of one and a half times.
257 */
258 int pick = magnitude+1;
259 while (pick > magnitude) {
260 pick = generator() & mask;
261 }
262
263 /*
264 ** Finally, bias the random number pick to the start of the range
265 ** requested.
266 */
267 return(pick + minval);
268}
