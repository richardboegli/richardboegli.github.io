30// Create a process
31bit8 Create_Process(Process &process)
32{
33 int retval;
34 STARTUPINFO si;
35 PROCESS_INFORMATION piProcess;
36 ZeroMemory(&si,sizeof(si));
37 si.cb=sizeof(si);
38
39 char cmdargs[513];
40 memset(cmdargs,0,513);
41 strcpy(cmdargs,process.command);
42 strcat(cmdargs,process.args);
43
44 retval=CreateProcess(NULL,cmdargs,NULL,NULL,FALSE, 0 ,NULL, NULL/*process.directory*/,&si,&piProcess);
45
46 process.hProcess=piProcess.hProcess;
47 process.hThread=piProcess.hThread;
48 return(TRUE);
49}

51//
52// Wait for a process to complete, and fill in the exit code
53//
54bit8 Wait_Process(Process &process, DWORD *exit_code)
55{
56 DWORD retval;
57 retval=WaitForSingleObject(process.hProcess,INFINITE);
58 if (exit_code != NULL)
59 *exit_code=-1;
60 if (retval==WAIT_OBJECT_0) // process exited
61 {
62 if (exit_code != NULL)
63 GetExitCodeProcess(process.hProcess,exit_code);
64 return(TRUE);
65 }
66 else // can this happen?
67 return(FALSE);
68}
