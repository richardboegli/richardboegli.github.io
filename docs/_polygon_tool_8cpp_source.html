99// Pick a polygon.
100Bool PolygonTool::poly_pickPoly(PolygonTrigger *pTrig, Coord3D loc, Int tolerance) {
101 ICoord3D iDocPt;
102 iDocPt.x = floor(loc.x+0.5f);
103 iDocPt.y = floor(loc.y+0.5f);
104 iDocPt.z = floor(loc.z+0.5f);
105 Int i, j;
106 ICoord3D ptArray[9];
107 for (i=0; i<3; i++) {
108 for (j=0; j<3; j++) {
109 ptArray[i+3*j].x = iDocPt.x + (1-i)*tolerance;
110 ptArray[i+3*j].y = iDocPt.y + (1-j)*tolerance;
111 ptArray[i+3*j].z = iDocPt.z;
112 }
113 }
114 int count = 0;
115 for (i=0; i<9; i++) {
116 if (pTrig->pointInTrigger(ptArray[i])) {
117 count++;
118 }
119 }
120 if (count>0 && count<8) {
121 return true;
122 }
123 return false;
124}

128// Pick a polygon.
129PolygonTrigger *PolygonTool::pickPolygon(Coord3D loc, CPoint viewPt, WbView* pView) {
130 for (PolygonTrigger *pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
131 // For water, only do water
132 if (WaterTool::isActive() && !pTrig->isWaterArea()) {
133 continue;
134 }
135 if (poly_pickPoly(pTrig, loc, SNAP_DISTANCE/2)) {
136 return pTrig;
137 }
138 }
139 for (pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
140 if (WaterTool::isActive() && !pTrig->isWaterArea()) {
141 continue;
142 }
143 Coord3D docPt = loc;
144 Int pt = poly_pickPoint(pTrig, viewPt, pView);
145 if (pt>=0) {
146 return pTrig;
147 }
148 if (poly_pickPoly(pTrig, docPt, 3*SNAP_DISTANCE/2)) {
149 return pTrig;
150 }
151 }
152 return NULL;
153}

155// Snap to other polys.
156Bool PolygonTool::poly_snapToPoly(Coord3D *pLoc) {
157 for (PolygonTrigger *pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
158 Int i;
159 // Dont' snap to self.
160 if (pTrig == m_poly_curSelectedPolygon) continue;
161 for (i=0; i<pTrig->getNumPoints(); i++) {
162 ICoord3D iLoc = *pTrig->getPoint(i);
163 Coord3D cpt = *pLoc;
164 cpt.x -= iLoc.x;
165 cpt.y -= iLoc.y;
166 cpt.z = 0;
167 if (cpt.length() < 1.0f*SNAP_DISTANCE) {
168 pLoc->x = iLoc.x;
169 pLoc->y = iLoc.y;
170 return true;
171 }
172 }
173 }
174 return false;
175}

177// Pick a point.
178Int PolygonTool::poly_pickPoint(PolygonTrigger *pTrig, CPoint viewPt, WbView* pView) {
179 if (pTrig==NULL) return -1;
180 Int i;
181 const Int PICK_PIXELS = pView->getPickPixels();
182 for (i=0; i<pTrig->getNumPoints(); i++) {
183 ICoord3D iLoc = *pTrig->getPoint(i);
184 Coord3D cpt;
185 cpt.x = iLoc.x;
186 cpt.y = iLoc.y;
187 cpt.z = iLoc.z;
188 cpt.z = 0;
189 CPoint screenLoc;
190 pView->docToViewCoords(cpt, &screenLoc);
191 Int dx = screenLoc.x-viewPt.x;
192 Int dy = screenLoc.y - viewPt.y;
193 if (dy<0) dy = -dy;
194 if (dx<0) dx = -dx;
195 if (dx<=PICK_PIXELS && dy<= PICK_PIXELS) {
196 return i;
197 }
198 }
199 return -1;
200}

202// Get the point to insert closest to loc.
203Int PolygonTool::poly_getInsertIndex(PolygonTrigger *pTrig, Coord3D loc) {
204 if (pTrig==NULL) return 0;
205 static Int tolerance = SNAP_DISTANCE;
206
207 Int i;
208 for (i=0; i<pTrig->getNumPoints(); i++) {
209 ICoord3D pt1 = *pTrig->getPoint(i);
210 ICoord3D pt2;
211 Int theNdx = i+1;
212 if (i==pTrig->getNumPoints()-1) {
213 pt2 = *pTrig->getPoint(0);
214 theNdx = 0;
215 } else {
216 pt2 = *pTrig->getPoint(i+1);
217 }
218 Int minX = loc.x-tolerance;
219 if (pt1.x<minX && pt2.x<minX) continue;
220 Int maxX = loc.x+tolerance;
221 if (pt1.x>maxX && pt2.x>maxX) continue;
222 Int minY = loc.y-tolerance;
223 if (pt1.y<minY && pt2.y<minY) continue;
224 Int maxY = loc.y+tolerance;
225 if (pt1.y>maxY && pt2.y>maxY) continue;
226 Int dy = pt2.y-pt1.y;
227 Int dx = pt2.x-pt1.x;
228 if (abs(dy)>abs(dx)) {
229 Real intersectionX = pt1.x + (dx * (loc.y-pt1.y)) / ((Real)dy);
230 if (intersectionX >= minX && intersectionX <= maxX) {
231 return theNdx;
232 }
233 } else {
234 Real intersectionY = pt1.y + (dy * (loc.x-pt1.x)) / ((Real)dx);
235 if (intersectionY >= minY && intersectionY <= maxY) {
236 return theNdx;
237 }
238 }
239 }
240 return 0;
241}

244void PolygonTool::poly_pickOnMouseDown(CPoint viewPt, WbView* pView)
245{
246 m_poly_dragPointNdx = -1;
247 if (m_poly_curSelectedPolygon) {
248 Bool found = false;
249 for (PolygonTrigger *pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
250 if (m_poly_curSelectedPolygon == pTrig) {
251 found = true;
252 break;
253 }
254 }
255 if (!found) {
256 m_poly_curSelectedPolygon = NULL; // Undo probably made it go away.
257 }
258 }
259 m_poly_justPicked = false; // Always false here. Pointer tool sets to true.
260 if (m_poly_curSelectedPolygon == NULL || !m_poly_isAdding) {
261 PolygonTrigger *pSel;
262 Coord3D docPt = m_poly_unsnappedMouseDownPt;
263 if (m_poly_curSelectedPolygon && poly_pickPoly(m_poly_curSelectedPolygon, docPt, SNAP_DISTANCE)) {
264 pSel = m_poly_curSelectedPolygon;
265 } else {
266 pSel = pickPolygon(docPt, viewPt, pView);
267 }
268 m_poly_curSelectedPolygon = pSel;
269 m_poly_isAdding = false;
270 }
271}

277{
278 if (m != TRACK_L) return;

281 m_poly_unsnappedMouseDownPt = docPt;
282 poly_pickOnMouseDown(viewPt, pView);

287 startMouseDown(m, viewPt, pView, pDoc);

291void PolygonTool::startMouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
292{
293 // Clear any selected map objects.
294 if (!m_poly_curSelectedPolygon) PointerTool::clearSelection();
295
296 ICoord3D iDocPt;
297 iDocPt.x = floor(m_poly_mouseDownPt.x+0.5f);
298 iDocPt.y = floor(m_poly_mouseDownPt.y+0.5f);
299 iDocPt.z = floor(m_poly_mouseDownPt.z+0.5f);
300 if (m_poly_curSelectedPolygon && m_poly_curSelectedPolygon->isWaterArea()) {
301 iDocPt.z = m_poly_curSelectedPolygon->getPoint(0)->z;
302 }
303 m_poly_dragPointNdx = -1;
304 m_poly_moveUndoable = NULL;
305 if (m_poly_curSelectedPolygon==NULL) {
306 // adding a new polygon.
307 m_poly_curSelectedPolygon = newInstance(PolygonTrigger)(32);
308 AsciiString name;
309 name.format("Area %d", m_poly_curSelectedPolygon->getID());
310 m_poly_curSelectedPolygon->setTriggerName(name);
311 m_poly_curSelectedPolygon->addPoint(iDocPt);
312 m_poly_dragPointNdx = 0;
313 AddPolygonUndoable *pUndo = new AddPolygonUndoable(m_poly_curSelectedPolygon);
314 pDoc->AddAndDoUndoable(pUndo);
315 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
316 m_poly_isAdding = true;
317 } else {
318 m_poly_dragPointNdx = poly_pickPoint(m_poly_curSelectedPolygon, viewPt, pView);
319 if (m_poly_isAdding) {
320 if (m_poly_dragPointNdx == 0) {
321 // closing the poly.
322 m_poly_isAdding = false;
323 } else {
324 m_poly_dragPointNdx = -1;
325 }
326 }
327 if (m_poly_isAdding && m_poly_dragPointNdx == -1) {
328 // Continuing to add points.
329 m_poly_dragPointNdx = m_poly_curSelectedPolygon->getNumPoints();
330 AddPolygonPointUndoable *pUndo = new AddPolygonPointUndoable(m_poly_curSelectedPolygon, iDocPt);
331 pDoc->AddAndDoUndoable(pUndo);
332 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
333 } else if (m_poly_dragPointNdx != -1) {
334 // Moving a point on the polygon.
335 ModifyPolygonPointUndoable *pUndo = new ModifyPolygonPointUndoable(m_poly_curSelectedPolygon, m_poly_dragPointNdx);
336 pDoc->AddAndDoUndoable(pUndo);
337 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
338 } else if (m_poly_justPicked) {
339 // Move the entire polygon.
340 m_poly_moveUndoable = new MovePolygonUndoable(m_poly_curSelectedPolygon);
341 pDoc->AddAndDoUndoable(m_poly_moveUndoable);
342 } else {
343 // Insert a point.
344 Coord3D docPt = m_poly_unsnappedMouseDownPt;
345 m_poly_dragPointNdx = poly_getInsertIndex(m_poly_curSelectedPolygon, docPt);
346 InsertPolygonPointUndoable *pUndo = new InsertPolygonPointUndoable(m_poly_curSelectedPolygon, iDocPt, m_poly_dragPointNdx);
347 pDoc->AddAndDoUndoable(pUndo);
348 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
349 }
350 }
351 WaypointOptions::update();
352}

355Bool PolygonTool::deleteSelectedPolygon(void)
356{
357 if (m_poly_curSelectedPolygon) {
358 Bool found = false;
359 for (PolygonTrigger *pTrig=PolygonTrigger::getFirstPolygonTrigger(); pTrig; pTrig = pTrig->getNext()) {
360 if (m_poly_curSelectedPolygon == pTrig) {
361 found = true;
362 break;
363 }
364 }
365 if (!found) {
366 m_poly_curSelectedPolygon = NULL; // Undo probably made it go away.
367 }
368 }
369 if (m_poly_curSelectedPolygon == NULL) {
370 return false;
371 }
372 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
373 if (!pDoc) return false;
374 if (m_poly_dragPointNdx != -1) {
375 // Moving a point on the polygon.
376 DeletePolygonPointUndoable *pUndo = new DeletePolygonPointUndoable(m_poly_curSelectedPolygon, m_poly_dragPointNdx);
377 pDoc->AddAndDoUndoable(pUndo);
378 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
379 m_poly_dragPointNdx = -1;
380 } else {
381 // delete the polygon.
382 DeletePolygonUndoable *pUndo = new DeletePolygonUndoable(m_poly_curSelectedPolygon);
383 pDoc->AddAndDoUndoable(pUndo);
384 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
385 m_poly_curSelectedPolygon = NULL;
386 m_poly_dragPointNdx = -1;
387 }
388 WaypointOptions::update();
389 return true;
390}

413void PolygonTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
414{
415 Coord3D docPt;
416 pView->viewToDocCoords(viewPt, &docPt);
417 if (m == TRACK_NONE) {
418 PolygonTrigger *pCur = m_poly_curSelectedPolygon;
419 Int curPt = m_poly_dragPointNdx;
420 m_poly_unsnappedMouseDownPt = docPt;
421 poly_pickOnMouseDown(viewPt, pView);
422 m_poly_mouseUpPlus = false;
423 m_poly_mouseUpMove = false;
424 if (m_poly_curSelectedPolygon) {
425 Int pt = poly_pickPoint(m_poly_curSelectedPolygon, viewPt, pView);
426 if (pt >= 0) {
427 m_poly_mouseUpMove = true;
428 } else {
429 m_poly_mouseUpPlus = true;
430 }
431 }
432 m_poly_curSelectedPolygon = pCur;
433 m_poly_dragPointNdx = curPt;
434 return; // setCursor will use the value of m_mouseUpRotate. jba.
435 }
436
437 if (m != TRACK_L) return;
438 if (!poly_snapToPoly(&docPt)) {
439 pView->snapPoint(&docPt);
440 }
441 if (m_poly_moveUndoable) {
442 ICoord3D iDocPt;
443 iDocPt.x = floor(docPt.x+0.5f-m_poly_mouseDownPt.x);
444 iDocPt.y = floor(docPt.y+0.5f-m_poly_mouseDownPt.y);
445 iDocPt.z = 0;
446 m_poly_moveUndoable->SetOffset(iDocPt);
447 pView->Invalidate();
448 return;
449 }
450 if (m_poly_dragPointNdx >= 0 && m_poly_curSelectedPolygon) {
451 ICoord3D iDocPt;
452 iDocPt.x = floor(docPt.x+0.5f);
453 iDocPt.y = floor(docPt.y+0.5f);
454 iDocPt.z = floor(docPt.z+0.5f);
455 if (m_poly_curSelectedPolygon->isWaterArea()) {
456 iDocPt.z = m_poly_curSelectedPolygon->getPoint(m_poly_dragPointNdx)->z;
457 }
458 m_poly_curSelectedPolygon->setPoint(iDocPt, m_poly_dragPointNdx);
459 pView->Invalidate();
460 }
461}
