123Int LANPreferences::getPreferredFaction(void)
124{
125 Int ret;
126 LANPreferences::const_iterator it = find("PlayerTemplate");
127 if (it == end())
128 {
129 return PLAYERTEMPLATE_RANDOM;
130 }
131
132 ret = atoi(it->second.str());
133 if (ret == PLAYERTEMPLATE_OBSERVER || ret < PLAYERTEMPLATE_MIN || ret >= ThePlayerTemplateStore->getPlayerTemplateCount())
134 ret = PLAYERTEMPLATE_RANDOM;
135
136 if (ret >= 0)
137 {
138 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(ret);
139 if (!fac)
140 ret = PLAYERTEMPLATE_RANDOM;
141 else if (fac->getStartingBuilding().isEmpty())
142 ret = PLAYERTEMPLATE_RANDOM;
143 }
144
145 return ret;
146}

195UnicodeString LANPreferences::getRemoteIPEntry(Int i)
196{
197 UnicodeString ret;
198 AsciiString key;
199 key.format("RemoteIP%d", i);
200
201 AsciiString ipstr;
202 AsciiString asciientry;
203
204 LANPreferences::const_iterator it = find(key.str());
205 if (it == end())
206 {
207 asciientry = "";
208 return ret;
209 }
210
211 asciientry = it->second;
212
213 asciientry.nextToken(&ipstr, ":");
214 asciientry.set(asciientry.str() + 1); // skip the ':'
215
216 ret.translate(ipstr);
217 if (asciientry.getLength() > 0)
218 {
219 ret.concat(L"(");
220 ret.concat(QuotedPrintableToUnicodeString(asciientry));
221 ret.concat(L")");
222 }
223
224 return ret;
225}

312static void playerTooltip(GameWindow *window,
313 WinInstanceData *instData,
314 UnsignedInt mouse)
315{
316 Int x, y, row, col;
317 x = LOLONGTOSHORT(mouse);
318 y = HILONGTOSHORT(mouse);
319
320 GadgetListBoxGetEntryBasedOnXY(window, x, y, row, col);
321
322 if (row == -1 || col == -1)
323 {
324 //TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:LobbyPlayers") );
325 return;
326 }
327
328 UnsignedInt playerIP = (UnsignedInt)GadgetListBoxGetItemData( window, row, col );
329 LANPlayer *player = TheLAN->LookupPlayer(playerIP);
330 if (!player)
331 {
332 DEBUG_CRASH(("No player info in listbox!"));
333 //TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:LobbyPlayers") );
334 return;
335 }
336 UnicodeString tooltip;
337 tooltip.format(TheGameText->fetch("TOOLTIP:LANPlayer"), player->getName().str(), player->getLogin().str(), player->getHost().str());
338 TheMouse->setCursorTooltip( tooltip );
339}

579void LanLobbyMenuUpdate( WindowLayout * layout, void *userData)
580{
581 if (TheGameLogic->isInShellGame() && TheGameLogic->getFrame() == 1)
582 {

584 }
585

587 {
588 if(initialGadgetDelay == 1)
589 {
590 TheTransitionHandler->setGroup("LanLobbyFade");
591 initialGadgetDelay = 2;
592 justEntered = FALSE;
593 }
594 else
595 initialGadgetDelay--;
596 }
597
598 if(LANisShuttingDown && TheShell->isAnimFinished() && TheTransitionHandler->isFinished())
599 shutdownComplete(layout);
600
601 if (TheShell->isAnimFinished() && !LANbuttonPushed && TheLAN)
602 TheLAN->update();
603
604 if (LANSocketErrorDetected == TRUE) {
605 LANSocketErrorDetected = FALSE;
606 DEBUG_LOG(("SOCKET ERROR! BAILING!\n"));
607 MessageBoxOk(TheGameText->fetch("GUI:NetworkError"), TheGameText->fetch("GUI:SocketError"), NULL);
608
609 // we have a socket problem, back out to the main menu.
610 TheWindowManager->winSendSystemMsg(buttonBack->winGetParent(), GBM_SELECTED,
611 (WindowMsgData)buttonBack, buttonBackID);
612 }

635 {
636
637 // ----------------------------------------------------------------------------------------

661 } // end switch( msg )
