177Bool W3DMouse::loadD3DCursorTextures(MouseCursor cursor)
178{
179 if (cursor == NONE || cursorTextures[cursor][0])
180 return TRUE; //no texture for this cursor or texture already loaded
181
182 WW3DAssetManager *am=WW3DAssetManager::Get_Instance();
183 Int animFrames=m_cursorInfo[cursor].numFrames;
184
185 if (!animFrames)
186 return FALSE; //no animation frames defined.
187
188 const char *baseName=m_cursorInfo[cursor].textureName.str();
189 char FrameName[64];
190
191 //Clamp to reasonable number
192 if (animFrames > MAX_2D_CURSOR_ANIM_FRAMES)
193 animFrames = MAX_2D_CURSOR_ANIM_FRAMES;
194
195 m_currentFrames=0;
196
197 if (animFrames == 1)
198 { //single animation frame without trailing numbers
199 sprintf(FrameName,"%s.tga",baseName);
200 cursorTextures[cursor][0]= am->Get_Texture(FrameName);
201 m_currentD3DSurface[0]=cursorTextures[cursor][0]->Get_Surface_Level();
202 m_currentFrames = 1;
203 }
204 else
205 for (Int i=0; i<animFrames; i++)
206 {
207 sprintf(FrameName,"%s%04d.tga",baseName,i);
208 if ((cursorTextures[cursor][i]=am->Get_Texture(FrameName)) != NULL)
209 { m_currentD3DSurface[m_currentFrames]=cursorTextures[cursor][i]->Get_Surface_Level();
210 m_currentFrames++;
211 }
212 }
213 return TRUE;
214}

259void W3DMouse::initW3DAssets(void)
260{
261 CriticalSectionClass::LockClass m(mutex);
262
263 //don't allow the mouse thread to initialize
264 //wait for main app to do initialization.
265 if (isThread)
266 return;
267
268 //Check if model assets already loaded
269 if ((cursorModels[1] == NULL && W3DDisplay::m_assetManager))
270 {
271 for (Int i=1; i<NUM_MOUSE_CURSORS; i++)
272 {
273 if (!m_cursorInfo[i].W3DModelName.isEmpty())
274 {
275 if (m_orthoCamera)
276 cursorModels[i] = W3DDisplay::m_assetManager->Create_Render_Obj(m_cursorInfo[i].W3DModelName.str(), m_cursorInfo[i].W3DScale*m_orthoZoom, 0);
277 else
278 cursorModels[i] = W3DDisplay::m_assetManager->Create_Render_Obj(m_cursorInfo[i].W3DModelName.str(), m_cursorInfo[i].W3DScale, 0);
279 if (cursorModels[i])
280 {
281 cursorModels[i]->Set_Position(Vector3(0.0f, 0.0f, -1.0f));
282 //W3DDisplay::m_3DInterfaceScene->Add_Render_Object(cursorModels[i]);
283 }
284 }
285 }
286 }
287 if ((cursorAnims[1] == NULL && W3DDisplay::m_assetManager))
288 {
289 for (Int i=1; i<NUM_MOUSE_CURSORS; i++)
290 {
291 if (!m_cursorInfo[i].W3DAnimName.isEmpty())
292 {
293 DEBUG_ASSERTCRASH(cursorAnims[i] == NULL, ("hmm, leak festival"));
294 cursorAnims[i] = W3DDisplay::m_assetManager->Get_HAnim(m_cursorInfo[i].W3DAnimName.str());
295 if (cursorAnims[i] && cursorModels[i])
296 {
297 cursorModels[i]->Set_Animation(cursorAnims[i], 0, (m_cursorInfo[i].loop) ? RenderObjClass::ANIM_MODE_LOOP : RenderObjClass::ANIM_MODE_ONCE);
298 }
299 }
300 }
301 }
302
303 // create the camera
304 m_camera = NEW_REF( CameraClass, () );
305 m_camera->Set_Position( Vector3( 0, 1, 1 ) );
306 Vector2 min = Vector2( -1, -1 );
307 Vector2 max = Vector2( +1, +1 );
308 m_camera->Set_View_Plane( min, max );
309 m_camera->Set_Clip_Planes( 0.995f, 20.0f );
310 if (m_orthoCamera)
311 m_camera->Set_Projection_Type( CameraClass::ORTHO );
312}

364void W3DMouse::setCursor( MouseCursor cursor )
365{
366
367 CriticalSectionClass::LockClass m(mutex);
368
369 m_directionFrame=0;
370 if (m_currentRedrawMode == RM_WINDOWS)
371 { //Windows default cursor needs to refreshed whenever we get a WM_SETCURSOR
372 m_currentD3DCursor=NONE;
373 m_currentW3DCursor=NONE;
374 m_currentPolygonCursor=NONE;
375 setCursorDirection(cursor);
376 if (m_drawing) //only allow cursor to change when drawing the cursor (once per frame) to fix flickering.
377 Win32Mouse::setCursor(cursor);
378 m_currentCursor = cursor;
379 return;
380 }
381
382 // extend
383 Mouse::setCursor( cursor );
384
385 // if we're already on this cursor ignore the rest of code to stop cursor flickering.
386 if( m_currentCursor == cursor && m_currentD3DCursor == cursor)
387 return;
388
389 //make sure Windows didn't reset our cursor
390 if (m_currentRedrawMode == RM_DX8)
391 {
392 SetCursor(NULL); //Kill Windows Cursor
393
394 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
395 Bool doImageChange=FALSE;
396
397 if (m_pDev != NULL)
398 {
399 m_pDev->ShowCursor(FALSE); //disable DX8 cursor
400 if (cursor != m_currentD3DCursor)
401 { if (!isThread)
402 { releaseD3DCursorTextures(m_currentD3DCursor);
403 //Since this type of cursor is updated from a non-D3D thread, we need
404 //to preallocate all surfaces in main thread.
405 loadD3DCursorTextures(cursor);
406 }
407 }
408 if (m_currentD3DSurface[0])
409 doImageChange=TRUE;
410 }
411 //For DX8 Cursors, we continually set the image on every call even when
412 //it didn't change. This is needed to prevent the cursor from flickering.
413 if (doImageChange)
414 {
415 HRESULT res;
416 m_currentHotSpot = m_cursorInfo[cursor].hotSpotPosition;
417 m_currentFMS = m_cursorInfo[cursor].fps/1000.0f;
418 m_currentAnimFrame = 0; //reset animation when cursor changes
419 res = m_pDev->SetCursorProperties(m_currentHotSpot.x,m_currentHotSpot.y,m_currentD3DSurface[(Int)m_currentAnimFrame]->Peek_D3D_Surface());
420 m_pDev->ShowCursor(TRUE); //Enable DX8 cursor
421 m_currentD3DFrame=(Int)m_currentAnimFrame;
422 m_currentD3DCursor = cursor;
423 m_lastAnimTime=timeGetTime();
424 }
425 }
426 else if (m_currentRedrawMode == RM_POLYGON)
427 {
428 SetCursor(NULL); //Kill Windows Cursor
429 m_currentD3DCursor=NONE;
430 m_currentW3DCursor=NONE;
431 m_currentPolygonCursor = cursor;
432 m_currentHotSpot = m_cursorInfo[cursor].hotSpotPosition;
433 }
434 else if (m_currentRedrawMode == RM_W3D)
435 {
436 SetCursor(NULL); //Kill Windows Cursor
437 m_currentD3DCursor=NONE;
438 m_currentPolygonCursor=NONE;
439 if (cursor != m_currentW3DCursor)
440 {
441 // set the new model visible
442 if (!cursorModels[1])
443 initW3DAssets();
444
445 if (cursorModels[1])
446 {
447 if (cursorModels[m_currentW3DCursor])
448 {
449 W3DDisplay::m_3DInterfaceScene->Remove_Render_Object(cursorModels[m_currentW3DCursor]);
450 }
451
452 m_currentW3DCursor=cursor;
453
454 if (cursorModels[m_currentW3DCursor])
455 {
456 W3DDisplay::m_3DInterfaceScene->Add_Render_Object(cursorModels[m_currentW3DCursor]);
457 if (m_cursorInfo[m_currentW3DCursor].loop == FALSE && cursorAnims[m_currentW3DCursor])
458 {
459 cursorModels[m_currentW3DCursor]->Set_Animation(cursorAnims[m_currentW3DCursor], 0, RenderObjClass::ANIM_MODE_ONCE);
460 }
461 }
462 }
463 }
464 else
465 {
466 m_currentW3DCursor=cursor;
467 }
468 }
469
470 // save current cursor
471 m_currentCursor = cursor;
472
473} // end setCursor

477void W3DMouse::draw(void)
478{
479 CriticalSectionClass::LockClass m(mutex);
480
481 m_drawing = TRUE;
482
483 //make sure the correct cursor image is selected
484 setCursor(m_currentCursor);
485
486 if (m_currentRedrawMode == RM_DX8 && m_currentD3DCursor != NONE)
487 {
488 //called from upate thread or rendering loop. Tells D3D where
489 //to draw the mouse cursor.
490 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
491 if (m_pDev)
492 { m_pDev->ShowCursor(TRUE); //Enable DX8 cursor
493
494 if (TheDisplay && !TheDisplay->getWindowed())
495 { //if we're full-screen, need to manually move cursor image
496 POINT ptCursor;
497
498 GetCursorPos( &ptCursor );
499 ScreenToClient( ApplicationHWnd, &ptCursor );
500 m_pDev->SetCursorPosition( ptCursor.x, ptCursor.y, D3DCURSOR_IMMEDIATE_UPDATE);
501 }
502 //Check if animated cursor and new frame
503 if (m_currentFrames > 1)
504 {
505 Int msTime=timeGetTime();
506 m_currentAnimFrame += (msTime-m_lastAnimTime) * m_currentFMS;
507 m_currentAnimFrame=fmod(m_currentAnimFrame,m_currentFrames);
508 m_lastAnimTime=msTime;
509
510 if ((Int)m_currentAnimFrame != m_currentD3DFrame)
511 {
512 m_currentD3DFrame=(Int)m_currentAnimFrame;
513 m_pDev->SetCursorProperties(m_currentHotSpot.x,m_currentHotSpot.y,m_currentD3DSurface[m_currentD3DFrame]->Peek_D3D_Surface());
514 }
515 }
516 }
517 }
518 else if (m_currentRedrawMode == RM_POLYGON)
519 {
520 const Image *image=cursorImages[m_currentPolygonCursor];
521 if (image)
522 {
523 TheDisplay->drawImage(image,m_currMouse.pos.x-m_currentHotSpot.x,m_currMouse.pos.y-m_currentHotSpot.y,
524 m_currMouse.pos.x+image->getImageWidth()-m_currentHotSpot.x, m_currMouse.pos.y+image->getImageHeight()-m_currentHotSpot.y);
525 }
526 }
527 else if (m_currentRedrawMode == RM_WINDOWS)
528 {
529 }
530 else if (m_currentRedrawMode == RM_W3D)
531 {
532 if ( W3DDisplay::m_3DInterfaceScene && m_camera && m_visible)
533 {
534 if (cursorModels[m_currentW3DCursor])
535 {
536 Real xPercent = (1.0f - (TheDisplay->getWidth() - m_currMouse.pos.x) / (Real)TheDisplay->getWidth());
537 Real yPercent = ((TheDisplay->getHeight() - m_currMouse.pos.y) / (Real)TheDisplay->getHeight());
538
539 Real x, y, z = -1.0f;
540
541 if (m_orthoCamera)
542 {
543 x = xPercent*2 - 1;
544 y = yPercent*2;
545 }
546 else
547 {
548 //W3D Screen coordinates are -1 to 1, so we need to do some conversion:
549 Real logX, logY;
550 PixelScreenToW3DLogicalScreen(m_currMouse.pos.x - 0, m_currMouse.pos.y - 0, &logX, &logY, TheDisplay->getWidth(), TheDisplay->getHeight());
551
552 Vector3 rayStart;
553 Vector3 rayEnd;
554 rayStart = m_camera->Get_Position(); //get camera location
555 m_camera->Un_Project(rayEnd,Vector2(logX,logY)); //get world space point
556 rayEnd -= rayStart; //vector camera to world space point
557 rayEnd.Normalize(); //make unit vector
558 rayEnd *= m_camera->Get_Depth(); //adjust length to reach far clip plane
559 rayEnd += rayStart; //get point on far clip plane along ray from camera.
560
561 x = Vector3::Find_X_At_Z(z, rayStart, rayEnd);
562 y = Vector3::Find_Y_At_Z(z, rayStart, rayEnd);
563 }
564
565 Matrix3D tm(1);
566 tm.Set_Translation(Vector3(x, y, z));
567 Coord2D offset = {0, 0};
568 if (TheInGameUI && TheInGameUI->isScrolling())
569 {
570 offset = TheInGameUI->getScrollAmount();
571 offset.normalize();
572 Real theta = atan2(-offset.y, offset.x);
573 theta -= (Real)M_PI/2;
574 tm.Rotate_Z(theta);
575 }
576 cursorModels[m_currentW3DCursor]->Set_Transform(tm);
577
578 WW3D::Render( W3DDisplay::m_3DInterfaceScene, m_camera );
579 }
580 }
581 }
582
583 //@todo: In DX8 mode the mouse is drawn in another thread which isn't allowed
584 //access to D3D so we can't do any drawing here.
585 // draw the cursor text
586 if (!isThread)
587 drawCursorText();
588
589 // draw tooltip text
590 if (m_visible && !isThread)
591 drawTooltip();
592
593 m_drawing = FALSE;
594}

596void W3DMouse::setRedrawMode(RedrawMode mode)
597{
598 MouseCursor cursor = getMouseCursor();
599
600 //Turn off the previous cursor mode
601 setCursor(NONE);
602
603 m_currentRedrawMode=mode;
604
605 switch (mode)
606 {
607 case RM_WINDOWS:
608 { //Windows mouse doesn't need an update thread.
609 if (thread.Is_Running())
610 thread.Stop();
611 freeD3DAssets(); //using Windows resources
612 freeW3DAssets();
613 freePolygonAssets();
614 m_currentD3DCursor = NONE;
615 m_currentW3DCursor = NONE;
616 m_currentPolygonCursor = NONE;
617 }
618 break;
619
620 case RM_W3D:
621 { //Model mouse updated only at render time so doesn't
622 //require thread.
623 if (thread.Is_Running())
624 thread.Stop();
625 freeD3DAssets(); //using packed Image data, not textures.
626 freePolygonAssets();
627 m_currentD3DCursor = NONE;
628 m_currentPolygonCursor = NONE;
629 initW3DAssets();
630 }
631 break;
632
633 case RM_POLYGON:
634 { //Polygon mouse updated only at render time so doesn't
635 //require thread.
636 if (thread.Is_Running())
637 thread.Stop();
638 freeD3DAssets(); //using packed Image data, not textures.
639 freeW3DAssets();
640 m_currentD3DCursor = NONE;
641 m_currentW3DCursor = NONE;
642 m_currentPolygonCursor = NONE;
643 initPolygonAssets();
644 }
645 break;
646
647 case RM_DX8:
648 { //this cursor type is drawn by DX8 and can be refreshed
649 //independent of rendering rate. Uses another thread to do
650 //position updates.
651 initD3DAssets(); //make sure textures loaded.
652 freeW3DAssets();
653 freePolygonAssets();
654 if (!thread.Is_Running())
655 thread.Execute();
656 m_currentW3DCursor = NONE;
657 m_currentPolygonCursor = NONE;
658 break;
659 }
660 }
661
662 setCursor(NONE);
663 setCursor(cursor);
664}

666void W3DMouse::setCursorDirection(MouseCursor cursor)
667{
668 Coord2D offset = {0, 0};
669 //Check if we have a directional cursor that needs different images for each direction
670 if (m_cursorInfo[cursor].numDirections > 1 && TheInGameUI && TheInGameUI->isScrolling())
671 {
672 offset = TheInGameUI->getScrollAmount();
673 if (offset.x || offset.y)
674 {
675 offset.normalize();
676 Real theta = atan2(offset.y, offset.x);
677 theta = fmod(theta+M_PI*2,M_PI*2);
678 Int numDirections=m_cursorInfo[m_currentCursor].numDirections;
679 //Figure out which of our predrawn cursor orientations best matches the
680 //actual cursor direction. Frame 0 is assumed to point right and continue
681 //clockwise.
682 m_directionFrame=(Int)(theta/(2.0f*M_PI/(Real)numDirections)+0.5f);
683 if (m_directionFrame >= numDirections)
684 m_directionFrame = 0;
685 }
686 else
687 {
688 m_directionFrame=0;
689 }
690 }
691 else
692 m_directionFrame = 0;
693}
