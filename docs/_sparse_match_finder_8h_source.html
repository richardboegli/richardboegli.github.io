55 struct HashMapHelper
56 {
57 size_t operator()(const BITSET& p) const
58 {
60 size_t result = 0;
61
62 for (int i = 0; i < p.size(); ++i)
63 if (p.test(i))
64 result += (i + 1);
65
66 return result;
67 }
68
69 Bool operator()(const BITSET& a, const BITSET& b) const
70 {
71 return (a == b);
72 }
73 };
74
75 struct MapHelper
76 {
77 bool operator()(const BITSET& a, const BITSET& b) const
78 {
79 int i;
80 if (a.size() < b.size()) {
81 return true;
82 }
83 for (i = 0; i < a.size(); ++i) {
84 bool aVal = a.test(i);
85 bool bVal = b.test(i);
86 if (aVal && bVal) continue;
87 if (!aVal && !bVal) continue;
88 if (!aVal) return true;
89 return false;
90 }
91 return false; // all bits match.
92 }
93 };

123 const MATCHABLE* findBestInfoSlow(const std::vector<MATCHABLE>& v, const BITSET& bits) const
124 {
125 const MATCHABLE* result = NULL;
126 Int bestYesMatch = 0; // want to maximize this
127 Int bestYesExtraneousBits = 999; // want to minimize this
128
129 #ifdef SPARSEMATCH_DEBUG
130 Int numDupMatches = 0;
131 AsciiString curBestMatchStr, dupMatchStr;
132 #endif
133
134 for (std::vector<MATCHABLE>::const_iterator it = v.begin(); it != v.end(); ++it)
135 {
136 for (Int i = it->getConditionsYesCount()-1; i >= 0; --i)
137 {
138 const BITSET& yesFlags = it->getNthConditionsYes(i);
139
140 // the best match has the most "yes" matches and the smallest number of "no" matches.
141 // if there are ties, then prefer the model with the smaller number of irrelevant 'yes' bits.
142 // (example of why tiebreaker is necessary: if we want to match FRONTCRUSHED,
143 // it would tie with both FRONTCRUSHED and FRONTCRUSHED|BACKCRUSHED, since they
144 // both match a single YES bit.)
145 Int yesMatch = countConditionIntersection(bits, yesFlags);
146 Int yesExtraneousBits = countConditionInverseIntersection(bits, yesFlags);
147
148 #ifdef SPARSEMATCH_DEBUG
149 if (yesMatch == bestYesMatch &&
150 yesExtraneousBits == bestYesExtraneousBits)
151 {
152 ++numDupMatches;
153 dupMatchStr = it->getDescription();
154 }
155 #endif
156
157 if ((yesMatch > bestYesMatch) ||
158 (yesMatch >= bestYesMatch && yesExtraneousBits < bestYesExtraneousBits))
159 {
160 result = &(*it);
161 bestYesMatch = yesMatch;
162 bestYesExtraneousBits = yesExtraneousBits;
163 #ifdef SPARSEMATCH_DEBUG
164 numDupMatches = 0;
165 curBestMatchStr = it->getDescription();
166 #endif
167 }
168 } // end for i
169
170 } // end for it
171
172#ifdef SPARSEMATCH_DEBUG
173 if (numDupMatches > 0)
174 {
175 AsciiString curConditionStr;
176 bits.buildDescription(&curConditionStr);
177 DEBUG_CRASH(("ambiguous model match in findBestInfoSlow \n\nbetween \n(%s)\n<and>\n(%s)\n\n(%d extra matches found)\n\ncurrent bits are (\n%s)\n",
178 curBestMatchStr.str(),
179 dupMatchStr.str(),
180 numDupMatches,
181 curConditionStr.str()));
182 }
183#endif
184
185 return result;
186 }

198 const MATCHABLE* findBestInfo(const std::vector<MATCHABLE>& v, const BITSET& bits) const
199 {
200 MatchMap::const_iterator it = m_bestMatches.find(bits);
201
202 const MATCHABLE *first = NULL;
203 if (it != m_bestMatches.end())
204 {
205 first = (*it).second;
206 }
207 if (first != NULL) {
208 return first;
209 }
210
211 const MATCHABLE* info = findBestInfoSlow(v, bits);
212
213 DEBUG_ASSERTCRASH(info != NULL, ("no suitable match for criteria was found!\n"));
214 if (info != NULL) {
215 m_bestMatches[bits] = info;
216 }
217
218 return info;
219 }
