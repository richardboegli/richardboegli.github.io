468void Network::GetCommandsFromCommandList() {
469 GameMessage *msg = TheCommandList->getFirstMessage();
470 GameMessage *next = NULL;
471 while (msg != NULL) {
472 next = msg->next();
473 if (isTransferCommand(msg)) { // Is this something we should be sending to the other players?
474 if (m_localStatus == NETLOCALSTATUS_INGAME) {
475 m_conMgr->sendLocalGameMessage(msg, getExecutionFrame());
476 }
477 TheCommandList->removeMessage(msg); // This does not destroy msg's prev and next pointers, so they should still be valid.
478 msg->deleteInstance();
479 } else {
480 if (processCommand(msg)) {
481 TheCommandList->removeMessage(msg);
482 msg->deleteInstance();
483 }
484 }
485 msg = next;
486 }
487}

504Bool Network::processCommand(GameMessage *msg)
505{
506 if ((m_lastFrame != TheGameLogic->getFrame()) || (m_localStatus == NETLOCALSTATUS_PREGAME)) {
507 // If this is the start of a new game logic frame, then tell the connection manager that the last
508 // frame is over and that it should now produce a frame info packet for the other players.
509
510 if (m_localStatus == NETLOCALSTATUS_PREGAME) {
511 // a sort-of-hack that prevents extraneous frames from being executed before the game actually starts.
512 // Idealy this shouldn't be necessary, but I don't think its hurting anything by being here.
513 if (TheGameLogic->getFrame() == 1) {
514 m_localStatus = NETLOCALSTATUS_INGAME;
515 NetCommandList *netcmdlist = m_conMgr->getFrameCommandList(0); // clear out frame 0 since we skipped it
516 netcmdlist->deleteInstance();
517 } else {
518 return FALSE;
519 }
520 }
521
522 // Only send frame info packets for frames where we are actually going to be in the game.
523 // The reason is so we don't have to wait for frame info packets to be sent that aren't going to
524 // matter anyways when we actually try to leave the game ourselves.
525 if (m_localStatus == NETLOCALSTATUS_INGAME) {
526 Int executionFrame = getExecutionFrame();
527
528 // Send command counts for all the frames we can.
529 for (Int i = m_lastFrameCompleted + 1; i < executionFrame; ++i) {
530 m_conMgr->processFrameTick(i);
531 //DEBUG_LOG(("Network::processCommand - calling processFrameTick for frame %d\n", i));
532 m_lastFrameCompleted = i;
533 }
534 }
535
536 //DEBUG_LOG(("Next Execution Frame - %d, last frame completed - %d\n", getExecutionFrame(), m_lastFrameCompleted));
537 m_lastFrame = TheGameLogic->getFrame();
538 }
539
540 // Are we leaving the game?
541 // This has to happen after the check to see if this is the start of a new logic frame.
542 // The reason is that we have to send all the frame info packets necessary to get to the
543 // frame where everyone else is going to see that we left.
544 if ((msg->getType() == GameMessage::MSG_CLEAR_GAME_DATA) && (m_localStatus == NETLOCALSTATUS_INGAME)) {
545 Int executionFrame = getExecutionFrame();
546 DEBUG_LOG(("Network::processCommand - local player leaving, executionFrame = %d, player leaving on frame %d\n", executionFrame, executionFrame+1));
547
548 m_conMgr->handleLocalPlayerLeaving(executionFrame+1);
549 m_conMgr->processFrameTick(executionFrame); // This is the last command we will execute, so send the command count.
550 // Also, we are guaranteed not to send any more commands for this frame
551 // since the local status will change to "Leaving" so we don't have to
552 // worry about messing up the other players.
553 m_conMgr->processFrameTick(executionFrame+1); // since we send it for executionFrame+1, we need to process both ticks
554 m_lastFrameCompleted = executionFrame;
555 DEBUG_LOG(("Network::processCommand - player leaving on frame %d\n", executionFrame));
556 m_localStatus = NETLOCALSTATUS_LEAVING;
557 return TRUE;
558 }
559 return FALSE;
560}

619void Network::processFrameSynchronizedNetCommand(NetCommandRef *msg) {
620 NetCommandMsg *cmdMsg = msg->getCommand();
621 if (cmdMsg->getNetCommandType() == NETCOMMANDTYPE_PLAYERLEAVE) {
622 PlayerLeaveCode retval = m_conMgr->processPlayerLeave((NetPlayerLeaveCommandMsg *)cmdMsg);
623 if (retval == PLAYERLEAVECODE_LOCAL) {
624 DEBUG_LOG(("Network::processFrameSynchronizedNetCommand - Local player left the game on frame %d.\n", TheGameLogic->getFrame()));
625 m_localStatus = NETLOCALSTATUS_LEFT;
626 } else if (retval == PLAYERLEAVECODE_PACKETROUTER) {
627 DEBUG_LOG(("Network::processFrameSynchronizedNetCommand - Packet router left the game on frame %d\n", TheGameLogic->getFrame()));
628 } else {
629 DEBUG_LOG(("Network::processFrameSynchronizedNetCommand - Client left the game on frame %d\n", TheGameLogic->getFrame()));
630 }
631 }
632 else if (cmdMsg->getNetCommandType() == NETCOMMANDTYPE_RUNAHEAD) {
633 NetRunAheadCommandMsg *netmsg = (NetRunAheadCommandMsg *)cmdMsg;
634 processRunAheadCommand(netmsg);
635 DEBUG_LOG(("command to set run ahead to %d and frame rate to %d on frame %d actually executed on frame %d\n", netmsg->getRunAhead(), netmsg->getFrameRate(), netmsg->getExecutionFrame(), TheGameLogic->getFrame()));
636 }
637 else if (cmdMsg->getNetCommandType() == NETCOMMANDTYPE_DESTROYPLAYER) {
638 NetDestroyPlayerCommandMsg *netmsg = (NetDestroyPlayerCommandMsg *)cmdMsg;
639 processDestroyPlayerCommand(netmsg);
640 //DEBUG_LOG(("CRC command (%8.8X) on frame %d actually executed on frame %d\n", netmsg->getCRC(), netmsg->getExecutionFrame(), TheGameLogic->getFrame()));
641 }
642}

644void Network::processRunAheadCommand(NetRunAheadCommandMsg *msg) {
645 m_runAhead = msg->getRunAhead();
646 m_frameRate = msg->getFrameRate();
647 time_t frameGrouping = (1000 * m_runAhead) / m_frameRate; // number of miliseconds between packet sends
648 frameGrouping = frameGrouping / 2; // since we only want the latency for one way to be a factor.
649// DEBUG_LOG(("Network::processRunAheadCommand - trying to set frame grouping to %d. run ahead = %d, m_frameRate = %d\n", frameGrouping, m_runAhead, m_frameRate));
650 if (frameGrouping < 1) {
651 frameGrouping = 1; // Having a value less than 1 doesn't make sense.
652 }
653 if (frameGrouping > 500) {
654 frameGrouping = 500; // Max of a half a second.
655 }
656 m_conMgr->setFrameGrouping(frameGrouping);
657}

678 msg->getExecutionFrame(), TheGameLogic->getFrame()));
679}

684void Network::update( void )
685{
686//
687// 1. Take Commands off TheCommandList, hand them off to the ConnectionManager.
688// 2. Call ConnectionManager->update;
689// 3. Check to see if all the commands for the next frame are there.
690// 4. If all commands are there, put that frame's commands on TheCommandList.
691//
692 m_frameDataReady = FALSE;
693
694#if defined(_DEBUG) || defined(_INTERNAL)
695 if (m_networkOn == FALSE) {
696 return;
697 }
698#endif
699
700 GetCommandsFromCommandList(); // Remove commands from TheCommandList and send them to the connection manager.
701 if (m_conMgr != NULL) {
702 if (m_localStatus == NETLOCALSTATUS_INGAME) {
703 m_conMgr->updateRunAhead(m_runAhead, m_frameRate, m_didSelfSlug, getExecutionFrame());
704 m_didSelfSlug = FALSE;
705 }
706 //m_conMgr->update(); // Do the priority thing, packetize thing. This also calls the Transport::update function.
707 // depacketizes the incoming packets and puts them on the Network command list.
708 }
709
710 liteupdate();
711
712 if ((m_localStatus == NETLOCALSTATUS_LEFT)) {// || (m_localStatus == NETLOCALSTATUS_LEAVING)) {
713 endOfGameCheck();
714 }
715
716 if (AllCommandsReady(TheGameLogic->getFrame())) { // If all the commands are ready for the next frame...
717 m_conMgr->handleAllCommandsReady();
718// DEBUG_LOG(("Network::update - frame %d is ready\n", TheGameLogic->getFrame()));
719 if (timeForNewFrame()) { // This needs to come after any other pre-frame execution checks as this changes the timing variables.
720 RelayCommandsToCommandList(TheGameLogic->getFrame()); // Put the commands for the next frame on TheCommandList.
721 m_frameDataReady = TRUE; // Tell the GameEngine to run the commands for the new frame.
722 }
723 }
724}

760Bool Network::timeForNewFrame() {
761 __int64 curTime;
762 QueryPerformanceCounter((LARGE_INTEGER *)&curTime);
763 __int64 frameDelay = m_perfCountFreq / m_frameRate;
764
765 /*
766 * If we're pushing up against the edge of our run ahead, we should slow the framerate down a bit
767 * to avoid being frozen by spikes in network lag. This will happen if another user's computer is
768 * running too far behind us, so we need to slow down to let them catch up.
769 */
770 if (m_conMgr != NULL) {
771 Real cushion = m_conMgr->getMinimumCushion();
772 Real runAheadPercentage = m_runAhead * (TheGlobalData->m_networkRunAheadSlack / (Real)100.0); // If we are at least 50% into our slack, we need to slow down.
773 if (cushion < runAheadPercentage) {
774// DEBUG_LOG(("Average cushion = %f, run ahead percentage = %f. Adjusting frameDelay from %I64d to ", cushion, runAheadPercentage, frameDelay));
775 frameDelay += frameDelay / 10; // temporarily decrease the frame rate by 20%.
776// DEBUG_LOG(("%I64d\n", frameDelay));
777 m_didSelfSlug = TRUE;
778// } else {
779// DEBUG_LOG(("Average cushion = %f, run ahead percentage = %f\n", cushion, runAheadPercentage));
780 }
781 }
782
783 // Check to see if we can run another frame.
784 if (curTime >= m_nextFrameTime) {
785// DEBUG_LOG(("Allowing a new frame, frameDelay = %I64d, curTime - m_nextFrameTime = %I64d\n", frameDelay, curTime - m_nextFrameTime));
786
787// if (m_nextFrameTime + frameDelay < curTime) {
788 if ((m_nextFrameTime + (2 * frameDelay)) < curTime) {
789 // If we get too far behind on our framerate we need to reset the nextFrameTime thing.
790 m_nextFrameTime = curTime;
791// DEBUG_LOG(("Initializing m_nextFrameTime to %I64d\n", m_nextFrameTime));
792 } else {
793 // Set the soonest possible starting time for the next frame.
794 m_nextFrameTime += frameDelay;
795// DEBUG_LOG(("m_nextFrameTime = %I64d\n", m_nextFrameTime));
796 }
797
798 return TRUE;
799 }
800// DEBUG_LOG(("Slowing down frame rate. frame rate = %d, frame delay = %I64d, curTime - m_nextFrameTime = %I64d\n", m_frameRate, frameDelay, curTime - m_nextFrameTime));
801 return FALSE;
802}
