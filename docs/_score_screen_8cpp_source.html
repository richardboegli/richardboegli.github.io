509WindowMsgHandledType ScoreScreenSystem( GameWindow *window, UnsignedInt msg,
510 WindowMsgData mData1, WindowMsgData mData2 )
511{
512 UnicodeString txtInput;
513
514 switch( msg )
515 {
516 // --------------------------------------------------------------------------------------------
517 case GWM_DESTROY:
518 {
519 break;
520 }
521
522 // --------------------------------------------------------------------------------------------
523 case GWM_INPUT_FOCUS:
524 {
525
526 // if we're givin the opportunity to take the keyboard focus we must say we want it
527 if( mData1 == TRUE )
528 *(Bool *)mData2 = TRUE;
529
530 break;
531 } // end input
532
533 // --------------------------------------------------------------------------------------------
534 case GBM_SELECTED:
535 {
536 TheTransitionHandler->remove("ScoreScreenShow", TRUE);
537 ReplayWasPressed = FALSE;
538
539 GameWindow *control = (GameWindow *)mData1;
540 Int controlID = control->winGetWindowId();

622 }
623 break;
624 }
625 }
626 }
627
628 case GEM_EDIT_DONE:
629 {
630 GameWindow *control = (GameWindow *)mData1;

632
633 // Take the user's input and echo it into the chat window as well as

637

647 TheLAN->RequestChat(txtInput, LANAPIInterface::LANCHAT_NORMAL);
648 //add the gamespy chat request here
649
650 }// if ( controlID == textEntryChatID )
651 }
652 }
653 return MSG_HANDLED;
654}

685void PlayMovieAndBlock(AsciiString movieTitle)
686{
687 VideoStreamInterface *videoStream = TheVideoPlayer->open( movieTitle );
688 if ( videoStream == NULL )
689 {
690 return;
691 }
692
693 // Create the new buffer
694 VideoBuffer *videoBuffer = TheDisplay->createVideoBuffer();
695 if ( videoBuffer == NULL ||
696 !videoBuffer->allocate( videoStream->width(),
697 videoStream->height())
698 )
699 {
700 delete videoBuffer;
701 videoBuffer = NULL;
702
703 if ( videoStream )
704 videoStream->close();
705 videoStream = NULL;
706
707 return;
708 }
709
710 GameWindow *movieWindow = s_blankLayout->getFirstWindow();
711 TheWritableGlobalData->m_loadScreenRender = TRUE;
712 while (videoStream->frameIndex() < videoStream->frameCount() - 1)
713 {
714 TheGameEngine->serviceWindowsOS();
715
716 if(!videoStream->isFrameReady())
717 {
718 Sleep(1);
719 continue;
720 }
721
722 if (!TheGameEngine->isActive())
723 { //we are alt-tabbed out, so just increment the frame
724 videoStream->frameNext();
725 videoStream->frameDecompress();
726 continue;
727 }
728
729 videoStream->frameDecompress();
730 videoStream->frameRender(videoBuffer);
