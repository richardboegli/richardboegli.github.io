90void NetCommandWrapperListNode::copyChunkData(NetWrapperCommandMsg *msg) {
91 if (msg == NULL) {
92 DEBUG_CRASH(("Trying to copy data from a non-existent wrapper command message"));
93 return;
94 }
95
96 DEBUG_ASSERTCRASH(msg->getChunkNumber() < m_numChunks, ("MunkeeChunk %d of %d\n",
97 msg->getChunkNumber(), m_numChunks));
98 if (msg->getChunkNumber() >= m_numChunks)
99 return;
100
101 DEBUG_LOG(("NetCommandWrapperListNode::copyChunkData() - copying chunk %d\n",
102 msg->getChunkNumber()));
103
104 if (m_chunksPresent[msg->getChunkNumber()] == TRUE) {
105 // we already received this chunk, no need to recopy it.
106 return;
107 }
108
109 m_chunksPresent[msg->getChunkNumber()] = TRUE;
110 UnsignedInt offset = msg->getDataOffset();
111 memcpy(m_data + offset, msg->getData(), msg->getDataLength());
112 ++m_numChunksPresent;
113}

163void NetCommandWrapperList::processWrapper(NetCommandRef *ref) {
164 NetCommandWrapperListNode *temp = m_list;
165 NetWrapperCommandMsg *msg = (NetWrapperCommandMsg *)(ref->getCommand());
166
167 while ((temp != NULL) && (temp->getCommandID() != msg->getWrappedCommandID())) {
168 temp = temp->m_next;
169 }
170
171 if (temp == NULL) {
172 temp = newInstance(NetCommandWrapperListNode)(msg);
173 temp->m_next = m_list;
174 m_list = temp;
175 }
176
177 temp->copyChunkData(msg);
178}

180NetCommandList * NetCommandWrapperList::getReadyCommands()
181{
182 NetCommandList *retlist = newInstance(NetCommandList);
183 retlist->init();
184
185 NetCommandWrapperListNode *temp = m_list;
186 NetCommandWrapperListNode *next = NULL;
187
188 while (temp != NULL) {
189 next = temp->m_next;
190 if (temp->isComplete()) {
191 NetCommandRef *msg = NetPacket::ConstructNetCommandMsgFromRawData(temp->getRawData(), temp->getRawDataLength());
192 NetCommandRef *ret = retlist->addMessage(msg->getCommand());
193 ret->setRelay(msg->getRelay());
194
195 msg->deleteInstance();
196 msg = NULL;
197
198 removeFromList(temp);
199 temp = NULL;
200 }
201 temp = next;
202 }
203
204 return retlist;
205}

207void NetCommandWrapperList::removeFromList(NetCommandWrapperListNode *node) {
208 if (node == NULL) {
209 return;
210 }
211
212 NetCommandWrapperListNode *temp = m_list;
213 NetCommandWrapperListNode *prev = NULL;
214
215 while ((temp != NULL) && (temp->getCommandID() != node->getCommandID())) {
216 prev = temp;
217 temp = temp->m_next;
218 }
219
220 if (temp == NULL) {
221 return;
222 }
223
224 if (prev == NULL) {
225 m_list = temp->m_next;
226 temp->deleteInstance();
227 temp = NULL;
228 } else {
229 prev->m_next = temp->m_next;
230 temp->deleteInstance();
231 temp = NULL;
232 }
