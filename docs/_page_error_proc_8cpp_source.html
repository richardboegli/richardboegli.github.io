69BOOL CALLBACK PageErrorProc( HWND hWndDialog, UINT message,
70 WPARAM wParam, LPARAM lParam )
71{
72
73 switch( message )
74 {
75
76 // ------------------------------------------------------------------------
77 case WM_INITDIALOG:
78 {
79
80 //
81 // load the listbox with pages that could not be processed
82 // and the reasons for it
83 //
84
85 // sanity
86 if( TheImagePacker == NULL )
87 return TRUE;
88
89 // go through all pages
90 TexturePage *page;
91 HWND list = GetDlgItem( hWndDialog, LIST_PAGES );
92 char buffer[ _MAX_PATH + 256 ];
93 char reason[ 32 ];
94
95 for( page = TheImagePacker->getFirstTexturePage();
96 page;
97 page = page->m_next )
98 {
99
100 // if image can't be processed find out why
101 if( BitTest( page->m_status, TexturePage::PAGE_ERROR ) )
102 {
103
104 if( BitTest( page->m_status, TexturePage::CANT_ALLOCATE_PACKED_IMAGE ) )
105 sprintf( reason, "Can't allocate image memory" );
106 else if( BitTest( page->m_status, TexturePage::CANT_ADD_IMAGE_DATA ) )
107 sprintf( reason, "Can't add image(s) data" );
108 else if( BitTest( page->m_status, TexturePage::NO_TEXTURE_DATA ) )
109 sprintf( reason, "No texture data to write" );
110 else if( BitTest( page->m_status, TexturePage::ERROR_DURING_SAVE ) )
111 sprintf( reason, "Error writing texture file" );
112 else
113 sprintf( reason, "Unknown Reason" );
114
115 sprintf( buffer, "%s: (%dx%d) %s%d",
116 reason, page->getWidth(), page->getHeight(),
117 TheImagePacker->getOutputFile(), page->getID() );
118
119 SendMessage( list, LB_INSERTSTRING, -1, (LPARAM)buffer );
120
121 } // end if
122
123 } // end for i
124
125 // set the extents for the horizontal scroll bar in the listbox
126 SendMessage( list, LB_SETHORIZONTALEXTENT, 1280, 0 );
127
128 return TRUE;
129
130 } // end init
131
132 // ------------------------------------------------------------------------
133 case WM_COMMAND:
134 {
135 Int controlID = LOWORD( wParam );
136// Int notifyCode = HIWORD( wParam );
137// HWND hWndControl = (HWND)lParam;
138
139 switch( controlID )
140 {
141
142 // --------------------------------------------------------------------
143 case IDOK:
144 {
145
146 EndDialog( hWndDialog, TRUE );
147 break;
148
149 } // end proceed
150
151 } // end switch
152
153 break;
154
155 } // end command
156
157 } // end switch message
158
159 return 0;
160
161} // end PageErrorProc
