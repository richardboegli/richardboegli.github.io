182void MapObject::verifyValidTeam(void)
183{
184 // if this map object has a valid team, then do nothing.
185 // if it has an invalid team, the place it on the default neutral team, (by clearing the
186 // existing team name.)
187 Bool exists;
188 AsciiString teamName = getProperties()->getAsciiString(TheKey_originalOwner, &exists);
189 if (exists) {
190 Bool valid = false;
191
192 int numSides = TheSidesList->getNumTeams();
193
194 for (int i = 0; i < numSides; ++i) {
195 TeamsInfo *teamInfo = TheSidesList->getTeamInfo(i);
196 if (!teamInfo) {
197 continue;
198 }
199
200 Bool itBetter;
201 AsciiString testAgainstTeamName = teamInfo->getDict()->getAsciiString(TheKey_teamName, &itBetter);
202 if (itBetter) {
203 if (testAgainstTeamName.compare(teamName) == 0) {
204 valid = true;
205 }
206 }
207 }
208
209 if (!valid) {
210 getProperties()->remove(TheKey_originalOwner);
211 }
212 }

215void MapObject::verifyValidUniqueID(void)
216{
217 Bool exists;
218 AsciiString uniqueID = getProperties()->getAsciiString(TheKey_uniqueID, &exists);
219 MapObject *obj = MapObject::getFirstMapObject();
220
221 // -1 is the sentinel
222 int highestIndex = -1;
223
224 while (obj) {
225 if (obj == this) {
226 // the first object is THIS OBJECT, cause we've already been added.
227 obj = obj->getNext();
228 continue;
229 }
230
231 if (obj->isWaypoint()) {
232 // waypoints throw this off. Sad but true. :-(
233 obj = obj->getNext();
234 continue;
235 }
236
237 Bool iterateExists;
238 AsciiString tempStr = obj->getProperties()->getAsciiString(TheKey_uniqueID, &iterateExists);
239 const char* lastSpace = tempStr.reverseFind(' ');
240
241 int testIndex = -1;
242 if (lastSpace) {
243 testIndex = atoi(lastSpace);
244 }
245
246 if (testIndex > highestIndex) {
247 highestIndex = testIndex;
248 }
249 break;
250 }
251
252 int indexOfThisObject = highestIndex + 1;
253
254 const char* thingName;
255 if (getThingTemplate()) {
256 thingName = getThingTemplate()->getName().str();
257 } else if (isWaypoint()) {
258 thingName = getWaypointName().str();
259 } else {
260 thingName = getName().str();
261 }
262 const char* pName = thingName;
263
264 while (*thingName) {
265 if ((*thingName) == '/') {
266 pName = thingName + 1;
267 }
268 ++thingName;
269 }
270
271 AsciiString newID;
272 if (isWaypoint()) {
273 newID.format("%s", pName);
274 } else {
275 newID.format("%s %d", pName, indexOfThisObject);
276 }
277 getProperties()->setAsciiString(TheKey_uniqueID, newID);
278}

280void MapObject::fastAssignAllUniqueIDs(void)
281{
282 // here's what we do. Take all of them, push them onto a stack. Then, pop each one, setting its id.
283 // should be much faster than what we currently do.
284
285 MapObject *pMapObj = getFirstMapObject();
286
287 std::stack<MapObject*> objStack;
288 Int actualNumObjects = 0;
289
290 while (pMapObj) {
291 ++actualNumObjects;
292 objStack.push(pMapObj);
293 pMapObj = pMapObj->getNext();
294 }
295
296 Int indexOfThisObject = 0;
297 while (actualNumObjects) {
298 MapObject *obj = objStack.top();
299
300
301 const char* thingName;
302 if (obj->getThingTemplate()) {
303 thingName = obj->getThingTemplate()->getName().str();
304 } else if (obj->isWaypoint()) {
305 thingName = obj->getWaypointName().str();
306 } else {
307 thingName = obj->getName().str();
308 }
309 const char* pName = thingName;
310
311 while (*thingName) {
312 if ((*thingName) == '/') {
313 pName = thingName + 1;
314 }
315 ++thingName;
316 }
317
318 AsciiString newID;
319 if (obj->isWaypoint()) {
320 newID.format("%s", pName);
321 } else {
322 newID.format("%s %d", pName, indexOfThisObject);
323 }
324
325 obj->getProperties()->setAsciiString(TheKey_uniqueID, newID);
326 objStack.pop();
327
328 ++indexOfThisObject;
329 --actualNumObjects;
330 }
331}

486WorldHeightMap::WorldHeightMap(ChunkInputStream *pStrm, Bool logicalDataOnly):
487 m_width(0), m_height(0), m_dataSize(0), m_data(NULL), m_cellFlipState(NULL), m_seismicUpdateFlag(NULL), m_seismicZVelocities(NULL),
488 m_drawOriginX(0), m_cellCliffState(NULL), m_drawOriginY(0),

495#endif

497 m_terrainTex(NULL), m_alphaTerrainTex(NULL), m_numBitmapTiles(0), m_numBlendedTiles(1)
498{
499
500 int i;
501 for (i=0; i<NUM_SOURCE_TILES; i++) {
502 m_sourceTiles[i]=NULL;

504 }
505 if (TheGlobalData && TheGlobalData->m_stretchTerrain) {
506 m_drawWidthX=STRETCH_DRAW_WIDTH;
507 m_drawHeightY=STRETCH_DRAW_HEIGHT;
508 }
509
510 DataChunkInput file( pStrm );
511
512 if (logicalDataOnly) {
513 file.registerParser( AsciiString("HeightMapData"), AsciiString::TheEmptyString, ParseSizeOnlyInChunk );
514 file.registerParser( AsciiString("WorldInfo"), AsciiString::TheEmptyString, ParseWorldDictDataChunk );
515 file.registerParser( AsciiString("ObjectsList"), AsciiString::TheEmptyString, ParseObjectsDataChunk );
516 freeListOfMapObjects(); // just in case.
517 file.registerParser( AsciiString("PolygonTriggers"), AsciiString::TheEmptyString, PolygonTrigger::ParsePolygonTriggersDataChunk );
518 PolygonTrigger::deleteTriggers(); // just in case.
519 TheSidesList->emptySides();
520 file.registerParser(AsciiString("SidesList"), AsciiString::TheEmptyString, SidesList::ParseSidesDataChunk );
521 } else {
522 file.registerParser( AsciiString("HeightMapData"), AsciiString::TheEmptyString, ParseHeightMapDataChunk );
523 file.registerParser( AsciiString("BlendTileData"), AsciiString::TheEmptyString, ParseBlendTileDataChunk );
524#ifdef EVAL_TILING_MODES

527 file.registerParser( AsciiString("GlobalLighting"), AsciiString::TheEmptyString, ParseLightingDataChunk );
528 }
529 if (!file.parse(this)) {
530
531 throw(ERROR_CORRUPT_FILE_FORMAT);
532 }
533 // patch bad maps.
534 if (!logicalDataOnly) {
535 for(i=0; i<m_dataSize; i++) {
536 if (m_cliffInfoNdxes[i]<0 || m_cliffInfoNdxes[i]>= m_numCliffInfo) {

538 }

544 }
545 }
546 }
547 if (TheGlobalData && TheGlobalData->m_drawEntireTerrain) {
548 m_drawWidthX=m_width;

654Real WorldHeightMap::getBilinearSampleSeismicZVelocity( Int x, Int y)
655{
656 if ( x < 0 || y < 0 ) return 0;
657 if ( y >= m_height ) return 0;
658 if ( x >= m_width ) return 0;
659 if (!m_seismicZVelocities) return 0;
660
661 Real collector = 0.0f;
662 Real divisor = 0.0f;
663
664 collector += m_seismicZVelocities[ y * m_width + x ];
665 ++divisor;
666
667 if ( y > 0 )
668 {
669 collector += m_seismicZVelocities[ (y-1) * m_width + x ];//bottom
670 ++divisor;
671
672 if( x > 0 )
673 {
674 collector += m_seismicZVelocities[ (y-1) * m_width + (x-1) ];//lower left
675 ++divisor;
676 }
677 if ( x < m_width-1 )
678 {
679 collector += m_seismicZVelocities[ (y-1) * m_width + (x+1) ];//lower right
680 ++divisor;
681 }
682 }
683 if ( y < m_height-1 )
684 {
685 collector += m_seismicZVelocities[ (y+1) * m_width + x ];//top
686 ++divisor;
687
688 if( x > 0 )
689 {
690 collector += m_seismicZVelocities[ (y+1) * m_width + (x-1) ];//upper left
691 ++divisor;
692 }
693 if ( x < m_width-1 )
694 {
695 collector += m_seismicZVelocities[ (y+1) * m_width + (x+1) ];//upper right
696 ++divisor;
697 }
698 }
699 if( x > 0 )
700 {
701 collector += m_seismicZVelocities[ y * m_width + (x-1) ];//left
702 ++divisor;
703 }
704 if ( x < m_width-1 )
705 {
706 collector += m_seismicZVelocities[ y * m_width + (x+1) ];//right
707 ++divisor;
708 }
709
710 collector /= divisor;
711
712 return collector;
713

1379/*Break down a .tga file into a collection of tiles. numRows * numRows total tiles.*/
1380Bool WorldHeightMap::readTiles(InputStream *pStr, TileData **tiles, Int numRows)
1381{

1383 pStr->read(&hdr, sizeof(hdr));
1384 Int tileWidth = hdr.imageWidth/TILE_PIXEL_EXTENT;
1385 Int tileHeight = hdr.imageHeight/TILE_PIXEL_EXTENT;
1386
1387 if (hdr.imageHeight==TILE_PIXEL_EXTENT/2) {
1388 tileHeight = 1;
1389 }
1390 if (hdr.imageWidth==TILE_PIXEL_EXTENT/2) {
1391 tileWidth = 1;
1392 }
1393
1394 if (tileWidth<numRows && tileHeight<numRows) {
1395 return(false);
1396 }
1397 Bool compressed = false;
1398 if (hdr.imageType & 0x08) {
1399 compressed = true;
1400 }
1401 int row = 0;
1402 int column = 0;
1403 int bytesPerPixel = (hdr.pixelDepth+7)/8;
1404 if (bytesPerPixel < 3) return(false);
1405 if (bytesPerPixel > 4) return(false);
1406 int i;
1407 for (i=0; i<numRows*numRows; i++) {
1408 if (tiles[i] == NULL)
1409 tiles[i] = MSGNEW("WorldHeightMap_readTiles") TileData;

1413// Bool read = false;
1414 Bool running = false;

1416 for (column=0; column<hdr.imageWidth; column++) {

1419 if (compressed && repeatCount==0) {
1420 UnsignedByte flag;
1421 pStr->read(&flag, 1);
1422 repeatCount = flag&0x7f;
1423 repeatCount++;
1424 if (flag&0x80) {
1425 running = true;
1426 pStr->read(buf, bytesPerPixel);
1427 } else {
1428 running = false;
1429 }
1430 }
1431 if (compressed) repeatCount--;
1432 if (!running) {
1433 pStr->read(buf, bytesPerPixel);
1434 }
1435 r = buf[2]; g = buf[1]; b = buf[0];
1436 if (bytesPerPixel==4) {
1437 a = buf[3];
1438 } else {
1439 a = 255;// solid alpha.
1440 }
1441 } else {
1442 r = g = b = a = 0;
1443 }
1444 if (column >= (numRows*TILE_PIXEL_EXTENT)) continue;

1449
1450 pixel += pixelNdx*TILE_BYTES_PER_PIXEL;
1451 *pixel++ = b;
1452 *pixel++ = g;
1453 *pixel++ = r;
1454 *pixel = a;
1455
1456 }
1457 DEBUG_ASSERTCRASH(repeatCount==0, ("Invalid tga."));
1458 }
1459 for (i=0; i<numRows*numRows; i++) {
1460 tiles[i]->updateMips();
1461 }
1462 return(true);
