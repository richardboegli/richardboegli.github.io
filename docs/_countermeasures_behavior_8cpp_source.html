65static void checkForCountermeasures( Object *testObj, void *userData )
66{
67 CountermeasuresPlayerScanHelper *helper = (CountermeasuresPlayerScanHelper*)userData;
68 ObjectPointerList *listToAddTo = helper->m_objectList;
69
70 if( testObj->isEffectivelyDead() )
71 return;
72
73 if( testObj->getControllingPlayer() != helper->m_theHealer->getControllingPlayer() )
74 return;
75
76 if( testObj->isOffMap() )
77 return;
78
79 if( !testObj->isAnyKindOf(helper->m_kindOfToTest) )
80 return;
81
82 if( testObj->getBodyModule()->getHealth() >= testObj->getBodyModule()->getMaxHealth() )
83 return;
84
85 listToAddTo->push_back(testObj);
86}

110void CountermeasuresBehavior::reportMissileForCountermeasures( Object *missile )
111{
112 if( !missile )
113 {
114 return;
115 }
116
117 //Record the number of missiles that have been fired at us
118 m_incomingMissiles++;
119
120 if( m_availableCountermeasures + m_activeCountermeasures > 0 )
121 {
122 //We have countermeasures we can use. Determine now whether or not the incoming missile will
123 //be diverted.
124 const CountermeasuresBehaviorModuleData *data = getCountermeasuresBehaviorModuleData();
125
126 if( GameLogicRandomValueReal( 0.0f, 1.0f ) < data->m_evasionRate )
127 {
128 //This missile will be diverted!
129 ProjectileUpdateInterface* pui = NULL;
130 for( BehaviorModule** u = missile->getBehaviorModules(); *u; ++u )
131 {
132 if( (pui = (*u)->getProjectileUpdateInterface()) != NULL )
133 {
134 //Make sure the missile diverts after a delay. The delay needs to be larger than
135 //the countermeasure reaction time or else the missile won't have a countermeasure to divert to!
136 DEBUG_ASSERTCRASH( data->m_countermeasureReactionFrames < data->m_missileDecoyFrames,
137 ("MissileDecoyDelay needs to be less than CountermeasureReactionTime in order to function properly.") );
138 pui->setFramesTillCountermeasureDiversionOccurs( data->m_missileDecoyFrames );
139 m_divertedMissiles++;
140
141 if( m_activeCountermeasures == 0 && m_reactionFrame == 0 )
142 {
143 //We need to launch our first volley of countermeasures, but we can't do it now. If we
144 //do, it'll look too artificial. Instead, we need to set up a timer to fake a reaction
145 //delay.
146 m_reactionFrame = TheGameLogic->getFrame() + data->m_countermeasureReactionFrames;
147 }
148 break;
149 }
150 }
151 }
152 }
153}

156ObjectID CountermeasuresBehavior::calculateCountermeasureToDivertTo( const Object& victim )
157{
158 const CountermeasuresBehaviorModuleData *data = getCountermeasuresBehaviorModuleData();
159
160 //Flares are pushed to the front of the list, but we only want to acquire the "newest" of the flares, therefore
161 //stop iterating after we've reached size of a single volley.
162 Int iteratorMax = MAX( data->m_volleySize, 1 );
163
164 Real closestDist = 1e15f;
165 Object *closestFlare = NULL;
166
167 //Start at the end of the list and go towards the beginning.
168 CountermeasuresVec::iterator it = m_counterMeasures.end();
169 //end is actually the end so advance the iterator.
170 if( it )
171 {
172 --it;
173 while( iteratorMax-- )
174 {
175 Object *obj = TheGameLogic->findObjectByID( *it );
176 if( obj )
177 {
178 Real dist = ThePartitionManager->getDistanceSquared( obj, getObject(), FROM_CENTER_2D );
179 if( dist < closestDist )
180 {
181 closestDist = dist;
182 closestFlare = obj;
183 }
184 }
185 else
186 {
187 --it;
188 }
189 }
190 }
191
192 if( closestFlare )
193 {
194 return closestFlare->getID();
195 }
196 return INVALID_ID;
197}

208UpdateSleepTime CountermeasuresBehavior::update( void )
209{
210 UnsignedInt now = TheGameLogic->getFrame();
211 const CountermeasuresBehaviorModuleData *data = getCountermeasuresBehaviorModuleData();
212 Object *obj = getObject();
213
214 if( obj->isEffectivelyDead() )
215 {
216 return UPDATE_SLEEP_FOREVER;
217 }
218 if( !isUpgradeActive() )
219 {
220 return UPDATE_SLEEP_FOREVER;
221 }
222
223 //Validate all existing flares, and clean them up as needed.
224 for (CountermeasuresVec::iterator it = m_counterMeasures.begin(); it != m_counterMeasures.end(); /*nothing*/ )
225 {
226 Object *obj = TheGameLogic->findObjectByID( *it );
227 if( !obj )
228 {
229 it = m_counterMeasures.erase( it );
230 m_activeCountermeasures--;
231 }
232 else
233 {
234 ++it;
235 }
236 }
237
238 if( obj->isAirborneTarget() )
239 {
240
241 //Handle flare volley launching (initial reaction, and continuation firing).
242 if( m_availableCountermeasures )
243 {
244 //Deal with the initial volley, but wait until we are permitted to react.
245 if( m_reactionFrame )
246 {
247 if( m_reactionFrame == now )
248 {
249 //We have been shot at and now that the reaction timer has expired, fire a full volley of
250 //countermeasures.
251 launchVolley();
252 m_nextVolleyFrame = now + data->m_framesBetweenVolleys;
253 m_reactionFrame = 0;
254 }
255 }
256
257 //Handle subsequent volley launching.
258 if( m_nextVolleyFrame == now )
259 {
260 launchVolley();
261 m_nextVolleyFrame = now + data->m_framesBetweenVolleys;
262 }
263 }
264 }
265
266 //Handle auto-reloading (data->m_reloadFrames of zero means it's not possible to auto-reload).
267 //Aircraft that don't auto-reload require landing at an airfield for resupply.
268 if( !m_availableCountermeasures && data->m_reloadFrames )
269 {
270 if( m_reloadFrame != 0 )
271 {
272 if( m_reloadFrame <= now )
273 {
274 //We've successfully reloaded automatically.
275 reloadCountermeasures();
276 }
277 }
278 else
279 {
280 //We just started reloading, so set the frame it'll be ready.
281 m_reloadFrame = now + data->m_reloadFrames;
282 }
283 }
284
285 return UPDATE_SLEEP( UPDATE_SLEEP_NONE );
286}

297void CountermeasuresBehavior::launchVolley()
298{
299 const CountermeasuresBehaviorModuleData *data = getCountermeasuresBehaviorModuleData();
300 Object *obj = getObject();
301
302 Real volleySize = (Real)data->m_volleySize;
303 for( int i = 0; i < data->m_volleySize; i++ )
304 {
305 //Each flare in a volley will calculate a different vector to fly out. We have a +/- angle to
306 //spread out equally. With only one flare, it'll come straight out the back. Two flares will
307 //launch at the extreme positive and negative angle. Three flares will launch at extreme angles
308 //plus straight back. Four or more will divy it up equally.
309 Real currentVolley = (Real)i;
310 Real ratio = 0.0f;
311 if( volleySize != 1.0f )
312 {
313 //ratio between -1.0 and +1.0f
314 ratio = currentVolley / (volleySize - 1.0f) * 2.0f - 1.0f;
315 }
316 //Now calculate the angle. Simply multiply it by the ratio!
317 Real angle = ratio * data->m_volleyArcAngle;
318
319 Coord3D vel;
320 PhysicsBehavior *physics = obj->getPhysics();
321
322 //Calculate the angle to fire the flare by taking the facing angle and rotating it
323 //and then scaling it by it's velocity (if it's moving).
324 obj->getUnitDirectionVector3D( vel );
325 Vector2 flareVector;
326 flareVector.X = vel.x;
327 flareVector.Y = vel.y;
328 flareVector.Normalize();
329 flareVector.Rotate( angle );
330 //Give it back to the Coord3D
331 vel.x = flareVector.X;
332 vel.y = flareVector.Y;
333 vel.z = 0.0f;
334
335 Real velocity = physics->getVelocityMagnitude();
336 if( velocity < 1.0f )
337 {
338 velocity = -10.0f;
339 }
340 vel.scale( velocity * data->m_volleyVelocityFactor );
341
342 const ThingTemplate *thing = TheThingFactory->findTemplate( data->m_flareTemplateName );
343 if( thing )
344 {
345 Object *flare = TheThingFactory->newObject( thing, obj->getControllingPlayer()->getDefaultTeam() );
346 flare->setPosition( obj->getPosition() );
347 flare->setOrientation( obj->getOrientation() );
348 physics->transferVelocityTo( flare->getPhysics() );
349 flare->getPhysics()->applyMotiveForce( &vel );
350 m_activeCountermeasures++;
351 m_availableCountermeasures--;
352 m_counterMeasures.push_back( flare->getID() );
353 }
354 }
355}
