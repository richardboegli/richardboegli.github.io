178typedef BitFlags<KINDOF_COUNT> KindOfMaskType;
179
180#define MAKE_KINDOF_MASK(k) KindOfMaskType(KindOfMaskType::kInit, (k))
181
182inline Bool TEST_KINDOFMASK(const KindOfMaskType& m, KindOfType t)
183{
184 return m.test(t);
185}
186
187inline Bool TEST_KINDOFMASK_ANY(const KindOfMaskType& m, const KindOfMaskType& mask)
188{
189 return m.anyIntersectionWith(mask);
190}
191
192inline Bool TEST_KINDOFMASK_MULTI(const KindOfMaskType& m, const KindOfMaskType& mustBeSet, const KindOfMaskType& mustBeClear)
193{
194 return m.testSetAndClear(mustBeSet, mustBeClear);
195}
196
197inline Bool KINDOFMASK_ANY_SET(const KindOfMaskType& m)
198{
199 return m.any();
200}
201
202inline void CLEAR_KINDOFMASK(KindOfMaskType& m)
203{
204 m.clear();
205}
206
207inline void SET_ALL_KINDOFMASK_BITS(KindOfMaskType& m)
208{
209 m.clear();
210 m.flip();
211}
212
213inline void FLIP_KINDOFMASK(KindOfMaskType& m)
214{
215 m.flip();
216}
