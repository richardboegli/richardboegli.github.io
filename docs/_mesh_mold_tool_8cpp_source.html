90void MeshMoldTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
91{
92 if (m != TRACK_L) return;
93 Coord3D cpt;
94 pView->viewToDocCoords(viewPt, &cpt);
95 if (!m_tracking) {
96 m_tracking = true;
97 m_toolPos = cpt;
98 Real height = TheTerrainRenderObject->getHeightMapHeight(cpt.x, cpt.y, NULL);
99 MeshMoldOptions::setHeight(height);
100 }
101 DrawObject::setDoMeshFeedback(true);
102 m_prevDocPt = cpt;
103 Bool shiftKey = (0x8000 & ::GetAsyncKeyState(VK_SHIFT))!=0;
104 if (shiftKey) {
105 m_offsettingZ = true;
106 m_prevViewPt = viewPt;
107 } else {
108 m_offsettingZ = false;
109 }
110
111 mouseMoved(m, viewPt, pView, pDoc);
112}

124void MeshMoldTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
125{
126 if (m != TRACK_L) {
127 return;
128 }
129 Coord3D cpt;
130 pView->viewToDocCoords(viewPt, &cpt);
131 if (m_offsettingZ) {
132 Real deltaZ = (m_prevViewPt.y - viewPt.y)*MAP_HEIGHT_SCALE*0.5f;
133 MeshMoldOptions::setHeight(MeshMoldOptions::getHeight()+deltaZ);
134 } else {
135 m_toolPos.x += cpt.x-m_prevDocPt.x;
136 m_toolPos.y += cpt.y-m_prevDocPt.y;
137 }
138 updateMeshLocation(false);
139 m_prevViewPt = viewPt;
140 m_prevDocPt = cpt;
141}

145void MeshMoldTool::updateMeshLocation(Bool changePreview)
146{
147 if (m_tracking) {
148 m_toolPos.z = MeshMoldOptions::getHeight();
149 DrawObject::setFeedbackPos(m_toolPos);
150 if (MeshMoldOptions::isDoingPreview() || changePreview) {
151 // doing preview
152 CWorldBuilderDoc *pDoc = CWorldBuilderDoc::GetActiveDoc();
153 if (MeshMoldOptions::isDoingPreview()) {
154 applyMesh(pDoc);
155 } else if (m_htMapEditCopy) {
156 // Unapply the mesh :)
157 Int i, j;
158 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
159 for (j=0; j<m_htMapEditCopy->getYExtent(); j++) {
160 for (i=0; i<m_htMapEditCopy->getXExtent(); i++) {
161 m_htMapEditCopy->setHeight(i, j, pMap->getHeight(i, j));
162 }
163 }
164 IRegion2D partialRange = {0,0,0,0};
165 pDoc->updateHeightMap(m_htMapEditCopy, false, partialRange);
166 }
167 }
168 }

185void MeshMoldTool::applyMesh(CWorldBuilderDoc *pDoc)
186{
187 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
188
189 HCURSOR old = SetCursor(::LoadCursor(0, IDC_WAIT));
190 if (m_htMapEditCopy == NULL) {
191 m_htMapEditCopy = pDoc->GetHeightMap()->duplicate();
192 } else {
193 // Restore original heights to edit copy.
194 Int i, j;
195 for (j=0; j<m_htMapEditCopy->getYExtent(); j++) {
196 for (i=0; i<m_htMapEditCopy->getXExtent(); i++) {
197 m_htMapEditCopy->setHeight(i, j, pMap->getHeight(i, j));
198 }
199 }
200 }
201 Int border = m_htMapEditCopy->getBorderSize();
202 WbView3d *p3View = pDoc->GetActive3DView();
203 if (p3View) {
204 DrawObject *pDraw = p3View->getDrawObject();
205 if (pDraw) {
206 MeshClass *pMesh = pDraw->peekMesh();
207 if (pMesh) {
208 SphereClass bounds;
209 pDraw->getMeshBounds(&bounds);
210 bounds.Center *= MeshMoldOptions::getScale();
211 bounds.Radius *= MeshMoldOptions::getScale();
212 Int minX = ceil((m_toolPos.x+bounds.Center.X-bounds.Radius)/MAP_XY_FACTOR);
213 Int minY = ceil((m_toolPos.y+bounds.Center.Y-bounds.Radius)/MAP_XY_FACTOR);
214 Int maxX = floor((m_toolPos.x+bounds.Center.X+bounds.Radius)/MAP_XY_FACTOR);
215 Int maxY = floor((m_toolPos.y+bounds.Center.Y+bounds.Radius)/MAP_XY_FACTOR);
216 maxX++; maxY++;
217 if (minX<0) minX = 0;
218 if (minY<0) minY = 0;
219 if (maxX > m_htMapEditCopy->getXExtent()) {
220 maxX = m_htMapEditCopy->getXExtent();
221 }
222 if (maxY > m_htMapEditCopy->getYExtent()) {
223 maxY = m_htMapEditCopy->getYExtent();
224 }
225 Int i, j;
226 for (j=minY; j<maxY; j++) {
227 for (i=minX; i<maxX; i++) {
228 Real X, Y;
229 X = i*MAP_XY_FACTOR;
230 X -= m_toolPos.x;
231 Y = j*MAP_XY_FACTOR;
232 Y -= m_toolPos.y;
233 X /= MeshMoldOptions::getScale();
234 Y /= MeshMoldOptions::getScale();
235 Vector3 vLoc(X, Y, 10000);
236 vLoc.Rotate_Z(-MeshMoldOptions::getAngle()*PI/180.0f);
237
238 LineSegClass ray(vLoc, Vector3(vLoc.X, vLoc.Y, -10000));
239 CastResultStruct castResult;
240 castResult.ComputeContactPoint = true;
241 RayCollisionTestClass rayCollide(ray, &castResult) ;
242 rayCollide.CollisionType = 0xFFFFFFFF;
243 if (pMesh->Cast_Ray(rayCollide)) {
244 // get the point of intersection according to W3D
245 Vector3 intersection = castResult.ContactPoint;
246 intersection.Z *= MeshMoldOptions::getScale();
247 Int newHeight = floor( ((intersection.Z+MeshMoldOptions::getHeight())/MAP_HEIGHT_SCALE)+0.5);
248 // check boundary values
249 if (newHeight < m_htMapEditCopy->getMinHeightValue()) newHeight = m_htMapEditCopy->getMinHeightValue();
250 if (newHeight > m_htMapEditCopy->getMaxHeightValue()) newHeight = m_htMapEditCopy->getMaxHeightValue();
251 Bool apply = true;
252 if (MeshMoldOptions::isRaisingOnly()) {
253 apply = newHeight > m_htMapEditCopy->getHeight(i+border, j+border);
254 } else if (MeshMoldOptions::isLoweringOnly()) {
255 apply = newHeight < m_htMapEditCopy->getHeight(i+border, j+border);
256 }
257 if (apply) {
258 m_htMapEditCopy->setHeight(i+border, j+border, newHeight);
259 }
260 }
261 }
262 }
263 IRegion2D partialRange;
264 // Offset by 2, because the normals change out two past the actual height change.
265 partialRange.lo.x = minX-2+border;
266 partialRange.hi.x = maxX+2+border;
267 partialRange.lo.y = minY-2+border;
268 partialRange.hi.y = maxY+2+border;
269 pDoc->updateHeightMap(m_htMapEditCopy, true, partialRange);
270
271 }
272 }
273 }
274 if (old) SetCursor(old);
