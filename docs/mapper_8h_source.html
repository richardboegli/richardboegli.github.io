59** TextureMapperClass
60** Base class for all texture mappers.
61*/

63{
64 public:
65
66 enum {
67 MAPPER_ID_UNKNOWN,
68 MAPPER_ID_LINEAR_OFFSET,
69 MAPPER_ID_CLASSIC_ENVIRONMENT,
70 MAPPER_ID_ENVIRONMENT,
71 MAPPER_ID_SCREEN,
72 MAPPER_ID_ANIMATING_1D,
73 MAPPER_ID_AXIAL,
74 MAPPER_ID_SILHOUETTE,
75 MAPPER_ID_SCALE,
76 MAPPER_ID_GRID,
77 MAPPER_ID_ROTATE,
78 MAPPER_ID_SINE_LINEAR_OFFSET,
79 MAPPER_ID_STEP_LINEAR_OFFSET,
80 MAPPER_ID_ZIGZAG_LINEAR_OFFSET,
81 MAPPER_ID_WS_CLASSIC_ENVIRONMENT,
82 MAPPER_ID_WS_ENVIRONMENT,
83 MAPPER_ID_GRID_CLASSIC_ENVIRONMENT,
84 MAPPER_ID_GRID_ENVIRONMENT,
85 MAPPER_ID_RANDOM,
86 MAPPER_ID_EDGE,
87 MAPPER_ID_BUMPENV,
88 MAPPER_ID_GRID_WS_CLASSIC_ENVIRONMENT,
89 MAPPER_ID_GRID_WS_ENVIRONMENT,
90 };
91
92 TextureMapperClass(unsigned int stage=0);
93 TextureMapperClass(const TextureMapperClass & src) : Stage(src.Stage) { }
94
95 virtual ~TextureMapperClass(void) { }
96
97 virtual int Mapper_ID(void) const { return MAPPER_ID_UNKNOWN;}
98
99 virtual TextureMapperClass * Clone(void) const = 0;
100
101 virtual bool Is_Time_Variant(void) { return false; }
102 virtual void Apply(int uv_array_index) = 0;
103 virtual void Reset(void) { }
104 virtual bool Needs_Normals(void) { return false; }
105 void Set_Stage(int stage) { Stage = stage; }
106 int Get_Stage(void) const { return Stage; }
107
108 // This is called by Apply(). It should not be called externally except
109 // in unusual circumstances.
110 virtual void Calculate_Texture_Matrix(Matrix4x4 &tex_matrix) = 0;
111
112 protected:
113 unsigned int Stage;
114};
