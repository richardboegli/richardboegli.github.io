74typedef struct
75{
76 long signature; /* signature of gimex ie 'tga ' (optional) */
77 long size; /* size of GINFO structure */
78 int version; /* version number of GINFO structure (200) */
79 int framenum; /* current frame */
80 int width; /* width of bitmap in pixels */
81 int height; /* height of bitmap in pixels */
82 int bpp; /* bits per pixel (8 or 32) */
83 int originalbpp; /* bits per pixel in original image (1 to 32) */
84 int startcolour; /* first colour in palette */
85 int numcolours; /* number of colours in original indexed palette */
86 ARGB colourtbl[GIMEX_COLOURTBL_SIZE]; /* 8 bit palette */
87 int subtype; /* internal format sub-type 0-default */
88 int packed; /* type of packing on original image. 0 none, 1 run, n other */
89 int quality; /* quality of lossy packing 0..100 */
90 int framesize; /* size of frame in bytes */
91 int alphabits; /* number of bits in alpha channel */
92 int redbits; /* number of bits in red channel */
93 int greenbits; /* number of bits in green channel */
94 int bluebits; /* number of bits in blue channel */
95 int centerx; /* center point relative to upper left corner */
96 int centery;
97 int defaultx; /* default coordinate point */
98 int defaulty;
99 int numhotspots; /* number of hot spots defined */
100 char framename[GIMEX_FRAMENAME_SIZE]; /* null terminated name of frame/image */
101 char comment[GIMEX_COMMENT_SIZE]; /* null terminated multiline user comment */
102 int hotspottbl[GIMEX_HOTSPOTTBL_SIZE][GIMEX_HOTSPOTTBL_VALUES];/* up to 256 hot spots, XY pairs relative to upperleft */
103 float dpi; /* dots per inch ie 72.0 */
104 int reserved[3]; /* reserved for future use - set to zero */
105} GINFO;

124#define MAXMACTYPES 8
125#define MAXEXTENSIONS 8

127#define GIMEX_EXTENSION_SIZE 8
128#define GIMEX_AUTHORSTR_SIZE 32
129#define GIMEX_VERSIONSTR_SIZE 8
130#define GIMEX_SHORTTYPESTR_SIZE 8
131#define GIMEX_WORDTYPESTR_SIZE 16
132#define GIMEX_LONGTYPESTR_SIZE 32
133
134typedef struct
135{
136 long signature; /* signature of gimex ie 'tga ' (optional) */
137 long size; /* size of GABOUT structure */
138 int version; /* version number of GABOUT structure (200) */
139 unsigned int canimport :1; /* supports importing */
140 unsigned int canexport :1; /* supports exporting */
141 unsigned int importpacked :2; /* max import packed field 0..3 */
142 unsigned int exportpacked :2; /* max export packed field 0..3 */
143 unsigned int import8 :1; /* supports importing 8 bit indexed */
144 unsigned int export8 :1; /* supports exporting 8 bit indexed */
145 unsigned int import32 :1; /* supports importing 32 bit direct rgb */
146 unsigned int export32 :1; /* supports exporting 32 bit direct rgb */
147 unsigned int multiframe :1; /* supports multiple frames */
148 unsigned int multifile :1; /* format requires additional files or resource fork */
149 unsigned int multisize :1; /* supports different size per frame */
150 unsigned int framebuffer :1; /* module requires memory to buffer entire frame */
151 unsigned int external :1; /* module requires external tool or plugin */
152 unsigned int usesfile :1; /* module is file based vs ads/printer/generator */
153 unsigned int singlepalette :1; /* limited to a single palette per file */
154 unsigned int greyscale :1; /* use maxcolours for number of levels */
155 unsigned int startcolour :1; /* supports start colour */
156 unsigned int dotsubtype :1; /* subtype based on extension */
157 unsigned int resizable :1; /* read will respect ginfo width & height */
158 unsigned int pad :11; /* pad bitfield to 32 bit boundary for inter compiler compatibility */
159 int maxcolours; /* only use in 8 bit, 0 if module does not care */
160 int maxframename; /* maximum characters in ginfo framename */
161 int defaultquality; /* default pack quality */
162 long mactype[MAXMACTYPES]; /* mac file system types used */
163 char extensions[MAXEXTENSIONS][GIMEX_EXTENSION_SIZE]; /* null terminated extensions with '.' */
164 char authorstr[GIMEX_AUTHORSTR_SIZE]; /* name of gimex module author */
165 char versionstr[GIMEX_VERSIONSTR_SIZE]; /* version number of gimex module ie 1.00 */
166 char shorttypestr[GIMEX_SHORTTYPESTR_SIZE]; /* 3 or 4 character type string ie TGA */
167 char wordtypestr[GIMEX_WORDTYPESTR_SIZE]; /* single word type string ie Targa */
168 char longtypestr[GIMEX_LONGTYPESTR_SIZE]; /* full name of data format ie True Vision Targa */
169} GABOUT;
