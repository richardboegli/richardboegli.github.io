215void Particle::computeAlphaRate( void )
216{
217 if (m_alphaKey[ m_alphaTargetKey ].frame == 0)
218 {
219 m_alphaRate = 0.0f;
220 return;
221 }
222
223 Real delta = m_alphaKey[ m_alphaTargetKey ].value - m_alphaKey[ m_alphaTargetKey-1 ].value;
224 UnsignedInt time = m_alphaKey[ m_alphaTargetKey ].frame - m_alphaKey[ m_alphaTargetKey-1 ].frame;
225
226 m_alphaRate = delta/time;
227}

229// ------------------------------------------------------------------------------------------------
231// ------------------------------------------------------------------------------------------------
232void Particle::computeColorRate( void )
233{
234 if (m_colorKey[ m_colorTargetKey ].frame == 0)
235 {
236 m_colorRate.red = 0.0f;
237 m_colorRate.green = 0.0f;
238 m_colorRate.blue = 0.0f;
239 return;
240 }
241
242 UnsignedInt time = m_colorKey[ m_colorTargetKey ].frame - m_colorKey[ m_colorTargetKey-1 ].frame;
243 Real delta = m_colorKey[ m_colorTargetKey ].color.red - m_colorKey[ m_colorTargetKey-1 ].color.red;
244 m_colorRate.red = delta/time;
245
246 delta = m_colorKey[ m_colorTargetKey ].color.green - m_colorKey[ m_colorTargetKey-1 ].color.green;
247 m_colorRate.green = delta/time;
248
249 delta = m_colorKey[ m_colorTargetKey ].color.blue - m_colorKey[ m_colorTargetKey-1 ].color.blue;
250 m_colorRate.blue = delta/time;
251}

358Bool Particle::update( void )
359{
360 // integrate acceleration into velocity
361 m_vel.x += m_accel.x;
362 m_vel.y += m_accel.y;
363 m_vel.z += m_accel.z;
364
365 m_vel.x *= m_velDamping;
366 m_vel.y *= m_velDamping;
367 m_vel.z *= m_velDamping;
368
369 // integrate velocity into position
370 const Coord3D *driftVel = m_system->getDriftVelocity();
371 m_pos.x += m_vel.x + driftVel->x;
372 m_pos.y += m_vel.y + driftVel->y;
373 m_pos.z += m_vel.z + driftVel->z;
374
375 // integrate the wind (if specified) into position
376 ParticleSystemInfo::WindMotion windMotion = m_system->getWindMotion();
377
378 // see if we should even do anything
379 if( windMotion != ParticleSystemInfo::WIND_MOTION_NOT_USED )
380 doWindMotion();
381
382 // update orientation
383 m_angleZ += m_angularRateZ;
384 m_angularRateZ *= m_angularDamping;
385
386 if (m_particleUpTowardsEmitter) {
387 // adjust the up position back towards the particle
388 static const Coord2D upVec = { 0.0f, 1.0f };
389 Coord2D emitterDir;
390 emitterDir.x = m_pos.x - m_emitterPos.x;
391 emitterDir.y = m_pos.y - m_emitterPos.y;
392 m_angleZ = (angleBetween(&upVec, &emitterDir) + PI);
393
394
395 }
396
397 // update size
398 m_size += m_sizeRate;
399 m_sizeRate *= m_sizeRateDamping;
400
401 //
402 // Update alpha (if used)
403 //
404
405 if (m_system->getShaderType() != ParticleSystemInfo::ADDITIVE)
406 {
407 m_alpha += m_alphaRate;
408
409 if (m_alphaTargetKey < MAX_KEYFRAMES && m_alphaKey[ m_alphaTargetKey ].frame)
410 {
411 if (TheGameClient->getFrame() - m_createTimestamp >= m_alphaKey[ m_alphaTargetKey ].frame)
412 {
413 m_alpha = m_alphaKey[ m_alphaTargetKey ].value;
414 m_alphaTargetKey++;
415 computeAlphaRate();
416 }
417 }
418 else
419 m_alphaRate = 0.0f;
420
421 if (m_alpha < 0.0f)
422 m_alpha = 0.0f;
423 else if (m_alpha > 1.0f)
424 m_alpha = 1.0f;
425 }
426
427
428 //
429 // Update color
430 //
431 m_color.red += m_colorRate.red;
432 m_color.green += m_colorRate.green;
433 m_color.blue += m_colorRate.blue;
434
435 if (m_colorTargetKey < MAX_KEYFRAMES && m_colorKey[ m_colorTargetKey ].frame)
436 {
437 if (TheGameClient->getFrame() - m_createTimestamp >= m_colorKey[ m_colorTargetKey ].frame)
438 {
439 // can't set, because of colorscale
440 // m_color = m_colorKey[ m_colorTargetKey ].color;
441 m_colorTargetKey++;
442 computeColorRate();
443 }
444 }
445 else
446 {
447 m_colorRate.red = 0.0f;
448 m_colorRate.green = 0.0f;
449 m_colorRate.blue = 0.0f;
450 }
451
453 m_color.red += m_colorScale;
454 m_color.green += m_colorScale;
455 m_color.blue += m_colorScale;
456
457 if (m_color.red < 0.0f)
458 m_color.red = 0.0f;
459 else if (m_color.red > 1.0f)
460 m_color.red = 1.0f;
461
462 if (m_color.red < 0.0f)
463 m_color.green = 0.0f;
464 else if (m_color.green > 1.0f)
465 m_color.green = 1.0f;
466
467 if (m_color.blue < 0.0f)
468 m_color.blue = 0.0f;
469 else if (m_color.blue > 1.0f)
470 m_color.blue = 1.0f;
471
472
473 // reset the acceleration for accumulation next frame
474 m_accel.z=m_accel.y=m_accel.x= 0.0f;
475
476 // monitor lifetime
477 if (m_lifetimeLeft && --m_lifetimeLeft == 0)
478 return false;
479
480 DEBUG_ASSERTCRASH( m_lifetimeLeft, ( "A particle has an infinite lifetime..." ));
481
482 // if we've gone totally invisible, destroy ourselves
483 if (isInvisible())
484 return false;
485 return true;
486}

491void Particle::doWindMotion( void )
492{
493
494 // get the angle of the wind
495 Real windAngle = m_system->getWindAngle();
496
497 // get the system position
498 Coord3D systemPos;
499 m_system->getPosition( &systemPos );
500
501 // when we're attached objects and drawables we offset by that position as well
502 if( ObjectID attachedObj = m_system->getAttachedObject() )
503 {
504 Object *obj = TheGameLogic->findObjectByID( attachedObj );
505
506 if( obj )
507 {
508 const Coord3D *objPos = obj->getPosition();
509
510 systemPos.x += objPos->x;
511 systemPos.y += objPos->y;
512 systemPos.z += objPos->z;
513
514 } // end if
515
516 } // end if
517 else if( DrawableID attachedDraw = m_system->getAttachedDrawable() )
518 {
519 Drawable *draw = TheGameClient->findDrawableByID( attachedDraw );
520
521 if( draw )
522 {
523 const Coord3D *drawPos = draw->getPosition();
524
525 systemPos.x += drawPos->x;
526 systemPos.y += drawPos->y;
527 systemPos.z += drawPos->z;
528
529 } // end if
530
531 } // end else if
532
533 //
534 // compute a vector from the system position in the world to the particle ... we will use
535 // this to compute how much force we apply
536 //
537 Coord3D v;
538 v.x = m_pos.x - systemPos.x;
539 v.y = m_pos.y - systemPos.y;
540 v.z = m_pos.z - systemPos.z;
541
542 // distance amounts for full force from wind and no force at all
543 Real fullForceDistance = 75.0f;
544 Real noForceDistance = 200.0f;
545
546 //
547 // given the distance from the wind position to the particle ... figure out how much
548 // force we're going to apply to it. When it's further away (outside of the full force
549 // distance) we will apply only a fraction of the force
550 //
551
552 Real distFromWind = v.length();
553 if( distFromWind < noForceDistance )
554 {
555 Real windForceStrength = 2.0f * m_windRandomness;
556
557 // only apply force if still within the circle of influence
558 if( distFromWind > fullForceDistance )
559 windForceStrength *= (1.0f - ((distFromWind - fullForceDistance) /
560 (noForceDistance - fullForceDistance)));
561
562 // integate the wind motion into the position
563 m_pos.x += (Cos( windAngle ) * windForceStrength);
564 m_pos.y += (Sin( windAngle ) * windForceStrength);
565
566 } // end if
567
568} // doWindMotion

581Bool Particle::isInvisible( void )
582{
583 switch (m_system->getShaderType())
584 {
585 case ParticleSystemInfo::ADDITIVE:
586 // if color is black, this particle is invisible
587
588 // check that we're not in the process of going to another color
589 if (m_colorKey[ m_colorTargetKey ].frame == 0)
590 {
591 if ((m_color.red + m_color.green + m_color.blue) <= 0.06f)
592 return true;
593 }
594 return false;
595
596 case ParticleSystemInfo::ALPHA:
597 // if alpha is zero, this particle is invisible
598 if (m_alpha < 0.02f)
599 return true;
600 return false;
601
602 case ParticleSystemInfo::ALPHA_TEST:
603 // hmm... assume these particles are never invisible
604 return false;
605
606 case ParticleSystemInfo::MULTIPLY:
607 // if color is white, this particle is invisible
608
609 // check that we're not in the process of going to another color
610 if (m_colorKey[ m_colorTargetKey ].frame == 0)
611 {
612 if ((m_color.red * m_color.green * m_color.blue) > 0.95f)
613 return true;
614 }
615 return false;
616 }
617
618 // should never get here - if we do, data is incorrect
619 return true;
620}

1392// ------------------------------------------------------------------------------------------------
1394// ------------------------------------------------------------------------------------------------
1395const Coord3D *ParticleSystem::computeParticleVelocity( const Coord3D *pos )
1396{
1397 static Coord3D newVel;
1398
1399 switch( m_emissionVelocityType )
1400 {
1401 case ORTHO:
1402 newVel.x = m_emissionVelocity.ortho.x.getValue();
1403 newVel.y = m_emissionVelocity.ortho.y.getValue();
1404 newVel.z = m_emissionVelocity.ortho.z.getValue();
1405 break;
1406
1407 case CYLINDRICAL:
1408 {
1409 Real radialSpeed, angle;
1410 radialSpeed = m_emissionVelocity.cylindrical.radial.getValue();
1411
1412 angle = GameClientRandomValueReal( 0, 2.0f*PI );
1413 newVel.x = radialSpeed * cos( angle );
1414 newVel.y = radialSpeed * sin( angle );
1415
1416 newVel.z = m_emissionVelocity.cylindrical.normal.getValue();
1417 break;
1418 }
1419
1420 // "outward" velocity is directed along the surface normal of the emission volume
1421 case OUTWARD:
1422 {
1423 Real speed = m_emissionVelocity.outward.speed.getValue();
1424 Real otherSpeed = m_emissionVelocity.outward.otherSpeed.getValue();
1425 Coord3D sysPos;
1426
1427 /*
1428 sysPos.x = m_localTransform.Get_X_Translation();
1429 sysPos.y = m_localTransform.Get_Y_Translation();
1430 sysPos.z = m_localTransform.Get_Z_Translation();
1431 */
1432 sysPos.x = 0.0f;
1433 sysPos.y = 0.0f;
1434 sysPos.z = 0.0f;
1435
1436 switch( m_emissionVolumeType )
1437 {
1438 case CYLINDER:
1439 Coord2D disk;
1440
1441 disk.x = pos->x - sysPos.x;
1442 disk.y = pos->y - sysPos.y;
1443 disk.normalize();
1444
1445 newVel.x = speed * disk.x;
1446 newVel.y = speed * disk.y;
1447 newVel.z = otherSpeed;
1448 break;
1449
1450 case BOX:
1451 case SPHERE:
1452 {
1453 newVel.x = pos->x - sysPos.x;
1454 newVel.y = pos->y - sysPos.y;
1455 newVel.z = pos->z - sysPos.z;
1456 newVel.normalize();
1457
1458 newVel.x *= speed;
1459 newVel.y *= speed;
1460 newVel.z *= speed;
1461 break;
1462 }
1463
1464 case LINE:
1465 {
1466 Coord3D along; // unit vector along line direction
1467
1468 along.x = m_emissionVolume.line.end.x - m_emissionVolume.line.start.x;
1469 along.y = m_emissionVolume.line.end.y - m_emissionVolume.line.start.y;
1470 along.z = m_emissionVolume.line.end.z - m_emissionVolume.line.start.z;
1471 along.normalize();
1472
1473 Coord3D perp; // unit vector perpendicular to the along/up plane
1474 Coord3D up; // unit vector in the up direction (Z)
1475 up.x = 0.0;
1476 up.y = 0.0;
1477 up.z = 1.0;
1478 perp.crossProduct( &up, &along, &perp );
1479 up.crossProduct( &along, &perp, &up );
1480
1481 // "speed" is in 'horizontal' plane, and "otherSpeed" is 'vertical'
1482 newVel.x = speed * perp.x + otherSpeed * up.x;
1483 newVel.y = speed * perp.y + otherSpeed * up.y;
1484 newVel.z = speed * perp.z + otherSpeed * up.z;
1485 break;
1486 }
1487
1488 case POINT:
1489 {
1490 Coord3D vel = *computePointOnUnitSphere();
1491
1492 newVel.x = speed * vel.x;
1493 newVel.y = speed * vel.y;
1494 newVel.z = speed * vel.z;
1495 break;
1496 }
1497 }
1498
1499 break;
1500 }
1501
1502 case SPHERICAL:
1503 {
1504 Real speed = m_emissionVelocity.spherical.speed.getValue();
1505 Coord3D vel = *computePointOnUnitSphere();
1506
1507 newVel.x = speed * vel.x;
1508 newVel.y = speed * vel.y;
1509 newVel.z = speed * vel.z;
1510 break;
1511 }
1512
1513 case HEMISPHERICAL:
1514 {
1515 Coord3D vel;
1516 Real speed = m_emissionVelocity.spherical.speed.getValue();
1517
1518 do
1519 {
1520 vel.x = GameClientRandomValueReal( -1.0f, 1.0f );
1521 vel.y = GameClientRandomValueReal( -1.0f, 1.0f );
1522 vel.z = GameClientRandomValueReal( 0.0f, 1.0f );
1523 }
1524 while (vel.x == 0.0f && vel.y == 0.0f && vel.z == 0.0f);
1525
1526 vel.normalize();
1527
1528 newVel.x = speed * vel.x;
1529 newVel.y = speed * vel.y;
1530 newVel.z = speed * vel.z;
1531 break;
1532 }
1533
1534 default:
1535 newVel.x = 0.0f;
1536 newVel.y = 0.0f;
1537 newVel.z = 0.0f;
1538 break;
1539 }
1540
1541 // scale the velocity by the velocity multiplier
1542 newVel.x *= m_velCoeff.x*(0.5f+TheGlobalData->m_particleScale/2.0f);
1543 newVel.y *= m_velCoeff.y*(0.5f+TheGlobalData->m_particleScale/2.0f);
1544 newVel.z *= m_velCoeff.z*(0.5f+TheGlobalData->m_particleScale/2.0f);
1545
1546 return &newVel;
1547}

1571 Real halfLength = m_emissionVolume.cylinder.length/2.0f;
1572 newPos.z = GameClientRandomValueReal( -halfLength, halfLength );
1573

1575 }
1576
1577 case SPHERE:
1578 {

1580

1583 else
1584 radius = GameClientRandomValueReal( 0.0f, m_emissionVolume.sphere.radius );
1585
1586 newPos = *computePointOnUnitSphere();
1587
1588 newPos.x *= radius;
1589 newPos.y *= radius;
1590 newPos.z *= radius;
1591
1592 break;
1593 }
1594
1595 case BOX:
1596 {
1597 if (m_isEmissionVolumeHollow) {
1598 // determine which side to generate on.
1599 // 0 is bottom, 3 is top,
1600 // 1 is left , 4 is right
1601 // 2 is front, 5 is right back
1602
1603 int side = GameClientRandomValue(0, 6);
1604 if (side % 3 == 0) {
1605 // generate X, Y
1606 newPos.x = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.x, m_emissionVolume.box.halfSize.x );
1607 newPos.y = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.y, m_emissionVolume.box.halfSize.y );
1608 if (side == 0) {
1609 newPos.z = -m_emissionVolume.box.halfSize.z;
1610 } else {
1611 newPos.z = m_emissionVolume.box.halfSize.z;
1612 }
1613
1614 } else if (side % 3 == 1) {
1615 // generate Y, Z
1616 newPos.y = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.y, m_emissionVolume.box.halfSize.y );
1617 newPos.z = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.z, m_emissionVolume.box.halfSize.z );
1618 if (side == 1) {
1619 newPos.x = -m_emissionVolume.box.halfSize.x;
1620 } else {
1621 newPos.x = m_emissionVolume.box.halfSize.y;
1622 }
1623
1624 } else if (side % 3 == 2) {
1625 // generate X, Z
1626 newPos.x = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.x, m_emissionVolume.box.halfSize.x );
1627 newPos.z = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.z, m_emissionVolume.box.halfSize.z );
1628 if (side == 2) {
1629 newPos.y = -m_emissionVolume.box.halfSize.y;
1630 } else {
1631 newPos.y = m_emissionVolume.box.halfSize.y;
1632 }
1633 }
1634 } else {
1635 newPos.x = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.x, m_emissionVolume.box.halfSize.x );
1636 newPos.y = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.y, m_emissionVolume.box.halfSize.y );
1637 newPos.z = GameClientRandomValueReal( -m_emissionVolume.box.halfSize.z, m_emissionVolume.box.halfSize.z );
1638 break;
1639 }
1640 }
1641
1642 case LINE:
1643 {
1644 Coord3D delta, start, end;
1645
1646 start = m_emissionVolume.line.start;
1647 end = m_emissionVolume.line.end;
1648
1649 delta.x = end.x - start.x;
1650 delta.y = end.y - start.y;
1651 delta.z = end.z - start.z;
1652
1653 Real t = GameClientRandomValueReal( 0.0f, 1.0f );
1654
1655 newPos.x = start.x + t * delta.x;
1656 newPos.y = start.y + t * delta.y;
1657 newPos.z = start.z + t * delta.z;
1658 break;
1659 }
1660
1661 case POINT:
1662 default:
1663 newPos.x = 0.0f;
1664 newPos.y = 0.0f;
1665 newPos.z = 0.0f;
1666 break;
1667 }
1668 newPos.x *= (0.5f+TheGlobalData->m_particleScale/2.0f);
1669 newPos.y *= (0.5f+TheGlobalData->m_particleScale/2.0f);
1670 newPos.z *= (0.5f+TheGlobalData->m_particleScale/2.0f);
1671 return &newPos;
1672}

2131 //
2132 // given our distance from the center ... we need to compute how much we will change
2133 // the angle. When we are closer to the center we change it faster (more), and when

2156

2178 // subtract angle
