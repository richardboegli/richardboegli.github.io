72class PartitionFilterHordeMember : public PartitionFilter
73{
74private:
75 Object* m_obj;
76 const HordeUpdateModuleData* m_data;
77
78public:
79
80 PartitionFilterHordeMember(Object* obj, const HordeUpdateModuleData* data) : m_obj(obj), m_data(data) { }
81
82#if defined(_DEBUG) || defined(_INTERNAL)
83 virtual const char* debugGetName() { return "PartitionFilterHordeMember"; }
84#endif
85
86 virtual Bool allow(Object *objOther)
87 {
88 // must be exact same type as us (well, maybe)
89 if (m_data->m_exactMatch && m_obj->getTemplate() != objOther->getTemplate())
90 return false;
91
92 // can only horde with other horde-able things
93 HordeUpdateInterface *hui = getHUI(objOther);
94 if( !hui )
95 return false;
96
97 // must match the kindof flags (if any)
98 if (!objOther->isKindOfMulti(m_data->m_kindof, KINDOFMASK_NONE))
99 return false;
100
101 // must be our ally (well, maybe)
102 if (m_data->m_alliesOnly && m_obj->getRelationship(objOther) != ALLIES)
103 return false;
104
105 // doh
106 if (m_obj->isOffMap() != objOther->isOffMap())
107 return false;
108
109 // guess it's horde-worthy!
110 return true;
111 }
112};

196void HordeUpdate::joinOrLeaveHorde(SimpleObjectIterator *iter, Bool join)
197{
198 Bool prevInHorde = m_inHorde;
199
200 m_inHorde = join;
201
202
203
204
205 const HordeUpdateModuleData* d = getHordeUpdateModuleData();
206 switch (d->m_action)
207 {
208 case HORDEACTION_HORDE:
209 {
210
211 // give/remove bonus effects
212 if( prevInHorde != m_inHorde )
213 {
214 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
215
216 if( ai )
217 ai->evaluateMoraleBonus();
218 else
219 DEBUG_CRASH(( "HordeUpdate::joinOrLeaveHorde - We (%s) must have an AI to benefit from horde\n",
220 getObject()->getTemplate()->getName().str() ));
221 } // end if
222
223 }
224 break;
225 }

259UpdateSleepTime HordeUpdate::update( void )
260{
261
262
263 //This code handles decals and ONLY decals!
264 Object *obj = getObject();
265
266 if ( ! obj )
267 return UPDATE_SLEEP_FOREVER;
268
269 const HordeUpdateModuleData *md = getHordeUpdateModuleData();
270
271 Bool wasInHorde = m_inHorde;
272
273 // This is a sticky situation, where refreshing the model state (like from default to damaged, for example)
274 // will rebuild the terraindecal and set its size to the default size.... since Vehicles have a special size,
275 // we want to keep it fresh, here, but not do the horde-ing test every frame...
276 Bool isInfantry = ( obj->isKindOf(KINDOF_INFANTRY) );
277 if ( isInfantry || (TheGameLogic->getFrame() > m_lastHordeRefreshFrame + md->m_updateRate) )
278 {
279 m_lastHordeRefreshFrame = TheGameLogic->getFrame();
280
281 PartitionFilterHordeMember hmFilter(getObject(), md);
282 PartitionFilter *filters[] = { &hmFilter, NULL };
283 SimpleObjectIterator *iter = ThePartitionManager->iterateObjectsInRange(getObject(), md->m_minDist, FROM_BOUNDINGSPHERE_3D, filters);
284 MemoryPoolObjectHolder hold(iter);
285
286 if ((iter->getCount() >= md->m_minCount - 1) )//we really are in the thick part of the horde
287 {
288 m_inHorde = TRUE;
289 m_trueHordeMember = TRUE;
290 }
291 else
292 {
293 m_inHorde = FALSE;
294 m_trueHordeMember = FALSE;
295
296 Real rubOffRadiusSq = sqr(md->m_rubOffRadius);
297 for (Object* other = iter->first(); other; other = iter->next())
298 {
299 HordeUpdateInterface* hui = getHUI(other);
300 if ( hui != NULL && hui->isTrueHordeMember() )
301 {
302 Real dist = ThePartitionManager->getDistanceSquared(getObject(), other, FROM_CENTER_2D);
303
304 if (dist <= rubOffRadiusSq )
305 {
306 m_inHorde = TRUE;
307 break;
308 }
309
310 }
311 }
312 }
313
314 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
315 if( ai )
316 ai->evaluateMoraleBonus();
317
318 }
319
320
321
322
323
324
325 // This is a sticky situation, where refreshing the model state (like from default to damaged, for example)
326 // will rebuild the terraindecal and set its size to the default size.... since Vehicles have a special size,
327 // we want to keep it fresh, here, but not do the horde-ing test every frame...
328 // This is a weak solution, in that It causes this update to fight the defualt behavior of the modelstate methods,
329 // But in the interest of not breaking the modelstate changing logic for five hundred other units a week before golden,
330 // this is my solution... If anyone gets this note on the next project... please please please, fix the resetting of the
331 // shadows/terraindecals in W3DModelDraw. THanks, ML
332
333 Drawable* draw = getObject()->getDrawable();
334 if ( draw && ! obj->isEffectivelyDead() )
335 {
336 if( TheGameLogic->getDrawIconUI() )
337 {
338 if(m_inHorde && !obj->isKindOf( KINDOF_PORTABLE_STRUCTURE ) )// this not is a ride-on for overlord
339 {
340
341 TerrainDecalType nuType;// uninitialized
342
343 if ( isInfantry )
344 {
345 if( obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_NATIONALISM ) == TRUE )
346 {
347 if ( obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_FANATICISM ) == TRUE )
348 nuType = ( TERRAIN_DECAL_HORDE_WITH_FANATICISM_UPGRADE );
349 else
350 nuType = ( TERRAIN_DECAL_HORDE_WITH_NATIONALISM_UPGRADE );
351 }
352 else
353 nuType =(TERRAIN_DECAL_HORDE);
354
355 if (nuType != draw->getTerrainDecalType())
356 draw->setTerrainDecal(nuType);
357 }
358 else
359 {
360 Real size = 3.5f * obj->getGeometryInfo().getMajorRadius();
361 draw->setTerrainDecalSize( size, size );
362
363 if( obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_NATIONALISM ) == TRUE )
364 {
365 if ( obj->testWeaponBonusCondition( WEAPONBONUSCONDITION_FANATICISM ) == TRUE )
366 nuType = ( TERRAIN_DECAL_HORDE_WITH_FANATICISM_UPGRADE );
367 else
368 nuType = ( TERRAIN_DECAL_HORDE_WITH_NATIONALISM_UPGRADE );
369 }
370 else
371 nuType = (TERRAIN_DECAL_HORDE_VEHICLE);
372
373 if (nuType != draw->getTerrainDecalType())
374 draw->setTerrainDecal(nuType);
375 }
376 }
377
378
379 }
380 else
381 {
382 //Scripts have disabled icons so don't draw these!
383 draw->setTerrainDecal( TERRAIN_DECAL_NONE );
384 }
385
386 if ( ! wasInHorde && m_inHorde )
387 draw->setTerrainDecalFadeTarget(1.0f, 0.03f);
388 else if ( wasInHorde && ! m_inHorde )
389 draw->setTerrainDecalFadeTarget(0.0f, -0.03f);
390 }
391
392
393 return isInfantry ? UPDATE_SLEEP(md->m_updateRate) : UPDATE_SLEEP_NONE;
394
395}
