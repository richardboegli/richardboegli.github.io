215// ------------------------------------------------------------------------------------------------
216// ------------------------------------------------------------------------------------------------
217UpdateSleepTime NeutronMissileSlowDeathBehavior::update( void )
218{
220 // get the module data
221 const NeutronMissileSlowDeathBehaviorModuleData *modData = getNeutronMissileSlowDeathBehaviorModuleData();
222
223 // call the base class cause we're extending functionality
224 SlowDeathBehavior::update();
225
226 // get out of here if we're not activated yet
227 if( isSlowDeathActivated() == FALSE )
228 return UPDATE_SLEEP_NONE;
229
230 // get the current frame
231 UnsignedInt currFrame = TheGameLogic->getFrame();
232
233 // when we become activated we want to do a few things
234 if( m_activationFrame == 0 )
235 {
236 Coord3D pos;
237 const Coord3D *missilePos = getObject()->getPosition();
238
239 // get the position to play the effect at on the ground
240 pos.x = missilePos->x;
241 pos.y = missilePos->y;
242 pos.z = TheTerrainLogic->getGroundHeight( pos.x, pos.y );
243
244 // record the frame
245 m_activationFrame = currFrame;
246
247 FXList::doFXPos( modData->m_fxList, &pos );
248 } // end if
249
250 // see if it's time for any explosions
251 for( Int i = 0; i < MAX_NEUTRON_BLASTS; i++ )
252 {
253
254 // if this blast is not enabled we will not bother with it
255 if( modData->m_blastInfo[ i ].enabled == FALSE )
256 continue;
257
258 // has the time of this blast come
259 if( m_completedBlasts[ i ] == FALSE &&
260 (currFrame - m_activationFrame > modData->m_blastInfo[ i ].delay) )
261 {
262
263 // do the blast
264 doBlast( &modData->m_blastInfo[ i ] );
265
266 // mark this blast as complete now
267 m_completedBlasts[ i ] = TRUE;
268
269 } // end if
270
271 // has the time for a scorch blast come
272 if( m_completedScorchBlasts[ i ] == FALSE &&
273 (currFrame - m_activationFrame > modData->m_blastInfo[ i ].scorchDelay) )
274 {
275
276 // do the scorch blast
277 doScorchBlast( &modData->m_blastInfo[ i ] );
278
279 // mark this scorch blast as complete now
280 m_completedScorchBlasts[ i ] = TRUE;
281
282 } // end if
283
284 } // end for i
285
286 return UPDATE_SLEEP_NONE;
287
288} // end update

293void NeutronMissileSlowDeathBehavior::doBlast( const BlastInfo *blastInfo )
294{
295
296 // sanity
297 if( blastInfo == NULL )
298 return;
299
300 // get the module data
301 const NeutronMissileSlowDeathBehaviorModuleData *modData = getNeutronMissileSlowDeathBehaviorModuleData();
302
303 // get the object and position
304 Object *missile = getObject();
305 const Coord3D *missilePos = missile->getPosition();
306
307 // setup a damage info structure to do some damage
308 DamageInfo damageInfo;
309 damageInfo.in.m_damageType = DAMAGE_EXPLOSION;
310 damageInfo.in.m_deathType = DEATH_EXPLODED;
311 damageInfo.in.m_sourceID = missile->getID();
312 damageInfo.in.m_amount = blastInfo->minDamage;
313
314 // scan objects around us and do damage to objects we have "passed over" and are behind us
315 if( blastInfo->outerRadius )
316 {
317 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( missilePos,
318 blastInfo->outerRadius,
319 FROM_CENTER_2D,
320 NULL );
321 MemoryPoolObjectHolder hold( iter );
322 Object *other;
323 const Coord3D *otherPos;
324 Coord3D forceVector;
325 Real dist;
326 for( other = iter->first(); other; other = iter->next() )
327 {
328
329 // get other position
330 otherPos = other->getPosition();
331
332 // compute vector from the missile to other object
333 forceVector.x = otherPos->x - missilePos->x;
334 forceVector.y = otherPos->y - missilePos->y;
335 forceVector.z = otherPos->z - missilePos->z;
336
337 // try to topple other object
338 other->topple( &forceVector, blastInfo->toppleSpeed, TOPPLE_OPTIONS_NO_BOUNCE |
339 TOPPLE_OPTIONS_NO_FX );
340
341 //
342 // compute how much damage we're going to do to this object ... if the object
343 // is inside the inner radius we do the full damage. Outside of the inner radius
344 // we do a percentage based on how far away from the inner radius it is, but we
345 // will always do at least blastInfo->minDamage amount of damage
346 //
347 dist = forceVector.length();
348 if( dist <= blastInfo->innerRadius )
349 damageInfo.in.m_amount = blastInfo->maxDamage;
350 else
351 {
352 Real percent;
353
354 percent = 1.0f - ((dist - blastInfo->innerRadius) / (blastInfo->outerRadius - blastInfo->innerRadius + 0.01f));
355 damageInfo.in.m_amount = blastInfo->maxDamage * percent;
356 if( damageInfo.in.m_amount < blastInfo->minDamage )
357 damageInfo.in.m_amount = blastInfo->minDamage;
358
359 } // end else
360
361 // do actual damage
362 if( damageInfo.in.m_amount )
363 {
364
365 // do damage
366 other->attemptDamage( &damageInfo );
367
368 // place a scorch mark if we haven't already
369 if( m_scorchPlaced == FALSE )
370 {
371
372 TheGameClient->addScorch( missilePos, modData->m_scorchSize, SCORCH_1 );
373 m_scorchPlaced = TRUE;
374
375 } // end if
376
377 } // end if
378
379/*
380 // apply a small force to the object from the shockwave center
381 PhysicsBehavior *physics = other->getPhysics();
382 if( physics )
383 {
384 Coord3D physicsForce = forceVector;
385
386 // normalize the physics force
387 physicsForce.normalize();
388
389 //
390 // change the magnitude of the physics force to the force amount we want to apply from
391 // the shockwave
392 //
393 physicsForce.x *= blastInfo->pushForceMag;
394 physicsForce.y *= blastInfo->pushForceMag;
395 physicsForce.z *= blastInfo->pushForceMag;
396
397 // apply the force
398 physics->applyForce( &physicsForce );
399
400 } // end if, physics
401*/
402
403 } // end for, other
404
405 } // end if, an outer radius exists
406
407} // end doBlast
