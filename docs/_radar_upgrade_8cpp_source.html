72void RadarUpgrade::onDelete( void )
73{
74 const RadarUpgradeModuleData *md = getRadarUpgradeModuleData();
75
76 // if we haven't been upgraded there is nothing to clean up
77 if( isAlreadyUpgraded() == FALSE )
78 return;
79
80 // If we're currently disabled, we shouldn't do anything, because we've already done it.
81 if ( getObject()->isDisabled() )
82 return;
83
84 // remove the radar from the player
85 Player *player = getObject()->getControllingPlayer();
86 if( player )
87 player->removeRadar( md->m_isDisableProof );
88
89 // this upgrade module is now "not upgraded"
90 setUpgradeExecuted(FALSE);
91
92} // end onDelete

96void RadarUpgrade::onCapture( Player *oldOwner, Player *newOwner )
97{
98 const RadarUpgradeModuleData *md = getRadarUpgradeModuleData();
99
100 // do nothing if we haven't upgraded yet
101 if( isAlreadyUpgraded() == FALSE )
102 return;
103
104 // If we're currently disabled, we shouldn't do anything, because we've already done it.
105 if ( getObject()->isDisabled() )
106 return;
107
108 // remove radar from old player and add to new player
109 if( oldOwner )
110 {
111
112 oldOwner->removeRadar( md->m_isDisableProof );
113 setUpgradeExecuted(FALSE);
114
115 } // end if
116 if( newOwner )
117 {
118
119 newOwner->addRadar( md->m_isDisableProof );
120 setUpgradeExecuted(TRUE);
121
122 } // end if
123
124} // end onCapture

128void RadarUpgrade::upgradeImplementation( void )
129{
130 const RadarUpgradeModuleData *md = getRadarUpgradeModuleData();
131
132 Player *player = getObject()->getControllingPlayer();
133
134 // update the player with another radar facility
135 player->addRadar( md->m_isDisableProof );
136
137 // find the radar update module of this object
138 NameKeyType radarUpdateKey = NAMEKEY( "RadarUpdate" );
139 RadarUpdate *radarUpdate = (RadarUpdate *)getObject()->findUpdateModule( radarUpdateKey );
140 if( radarUpdate )
141 radarUpdate->extendRadar();
142
143} // end upgradeImplementation
