61BOOL SelectMacrotexture::OnInitDialog()
62{
63 CDialog::OnInitDialog();
64
65 CWnd *pWnd = GetDlgItem(IDC_TEXTURE_TREEVIEW);
66 CRect rect;
67 pWnd->GetWindowRect(&rect);
68
69 ScreenToClient(&rect);
70 rect.DeflateRect(2,2,2,2);
71 m_textureTreeView.Create(TVS_HASLINES|TVS_LINESATROOT|TVS_HASBUTTONS|
72 TVS_SHOWSELALWAYS|TVS_DISABLEDRAGDROP, rect, this, IDC_TERRAIN_TREEVIEW);
73 m_textureTreeView.ShowWindow(SW_SHOW);
74
75 {
76 char dirBuf[_MAX_PATH];
77 char findBuf[_MAX_PATH];
78 char fileBuf[_MAX_PATH];
79
80 strcpy(dirBuf, "..\\TestArt");
81 int len = strlen(dirBuf);
82
83 if (len > 0 && dirBuf[len - 1] != '\\') {
84 dirBuf[len++] = '\\';
85 dirBuf[len] = 0;
86 }
87 strcpy(findBuf, dirBuf);
88
89 FilenameList filenameList;
90 TheFileSystem->getFileListInDirectory(AsciiString(findBuf), AsciiString("*.tga"), filenameList, FALSE);
91
92 if (filenameList.size() > 0) {
93 TVINSERTSTRUCT ins;
94 HTREEITEM child = NULL;
95 FilenameList::iterator it = filenameList.begin();
96 do {
97 AsciiString filename = *it;
98 len = filename.getLength();
99 if (len<5) continue;
100 strcpy(fileBuf, filename.str());
101 ::memset(&ins, 0, sizeof(ins));
102 ins.hParent = TVI_ROOT;
103 ins.hInsertAfter = TVI_SORT;
104 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
105 ins.item.lParam = -1;
106 ins.item.pszText = fileBuf;
107 ins.item.cchTextMax = strlen(fileBuf);
108 child = m_textureTreeView.InsertItem(&ins);
109 ++it;
110 } while (it != filenameList.end());
111
112 ::memset(&ins, 0, sizeof(ins));
113 ins.hParent = TVI_ROOT;
114 ins.hInsertAfter = TVI_SORT;
115 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
116 ins.item.lParam = -1;
117 ins.item.pszText = DEFAULT;
118 ins.item.cchTextMax = strlen(DEFAULT);
119 child = m_textureTreeView.InsertItem(&ins);
120
121 }
122 }
123
124
125 return TRUE; // return TRUE unless you set the focus to a control
126 // EXCEPTION: OCX Property Pages should return FALSE
127}

129BOOL SelectMacrotexture::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
130{
131 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
132 if (pHdr->hdr.hwndFrom == m_textureTreeView.m_hWnd) {
133 if (pHdr->hdr.code == TVN_SELCHANGED) {
134 char buffer[_MAX_PATH];
135 HTREEITEM hItem = m_textureTreeView.GetSelectedItem();
136 TVITEM item;
137 ::memset(&item, 0, sizeof(item));
138 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT|TVIF_STATE;
139 item.hItem = hItem;
140 item.pszText = buffer;
141 item.cchTextMax = sizeof(buffer)-2;
142 m_textureTreeView.GetItem(&item);
143 if (0==strcmp(buffer, DEFAULT)) {
144 TheTerrainRenderObject->updateMacroTexture(AsciiString(""));
145 } else {
146 TheTerrainRenderObject->updateMacroTexture(AsciiString(buffer));
147 }
148 TheWritableGlobalData->m_useLightMap = true;
149 }
150 }
151 return CDialog::OnNotify(wParam, lParam, pResult);
152}
