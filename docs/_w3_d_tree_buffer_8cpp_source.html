132int W3DTreeBuffer::W3DTreeTextureClass::update(W3DTreeBuffer *buffer)
133{
134
135 //Set to clamp.
136 Get_Filter().Set_U_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
137 Get_Filter().Set_V_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
138
139 IDirect3DSurface8 *surface_level;
140 D3DSURFACE_DESC surface_desc;
141 D3DLOCKED_RECT locked_rect;
142 DX8_ErrorCode(Peek_D3D_Texture()->GetSurfaceLevel(0, &surface_level));
143 DX8_ErrorCode(surface_level->GetDesc(&surface_desc));
144
145 DX8_ErrorCode(surface_level->LockRect(&locked_rect, NULL, 0));
146
147 Int tilePixelExtent = TILE_PIXEL_EXTENT;
148// Int numRows = surface_desc.Height/(tilePixelExtent+TILE_OFFSET);
149#ifdef _DEBUG
150 //DASSERT_MSG(tilesPerRow*numRows >= htMap->m_numBitmapTiles,Debug::Format ("Too many tiles."));
151 //DEBUG_ASSERTCRASH((Int)surface_desc.Width >= tilePixelExtent*tilesPerRow, ("Bitmap too small."));
152#endif
153 if (surface_desc.Format == D3DFMT_A8R8G8B8) {
154 Int tileNdx;
155 Int pixelBytes = 4;
156#if 0 // Fill unused texture for debug display.
157 UnsignedInt cellX, cellY;
158 for (cellX = 0; cellX < surface_desc.Width; cellX++) {
159 for (cellY = 0; cellY < surface_desc.Height; cellY++) {
160 UnsignedByte *pBGR = ((UnsignedByte *)locked_rect.pBits)+(cellY*surface_desc.Width+cellX)*pixelBytes;
161 //*((Short*)pBGR) = 0x8000 + (((255-2*cellY)>>3)<<10) + ((4*cellX)>>4);
162 *((Int*)pBGR) = 0xFF000000 | ( (((255-cellY))<<16) + ((cellX)) );
163
164 }
165 }
166#endif
167 for (tileNdx=0; tileNdx < buffer->getNumTiles(); tileNdx++) {
168 TileData *pTile = buffer->getSourceTile(tileNdx);
169 if (!pTile) continue;
170 ICoord2D position = pTile->m_tileLocationInTexture;
171 if (position.x<0) {
172 continue;
173 }
174 Int i,j;
175 for (j=0; j<tilePixelExtent; j++) {
176 UnsignedByte *pBGR = pTile->getRGBDataForWidth(tilePixelExtent);
177 pBGR += (tilePixelExtent-(1+j))*TILE_BYTES_PER_PIXEL*tilePixelExtent; // invert to match.
178 Int row = position.y+j;
179 UnsignedByte *pBGRA = ((UnsignedByte*)locked_rect.pBits) +
180 (row)*surface_desc.Width*pixelBytes;
181
182 Int column = position.x;
183 pBGRA += column*pixelBytes;
184 for (i=0; i<tilePixelExtent; i++) {
185 // 15 bit color *((Short*)pBGRA) = 0x8000 + ((pBGR[2]>>3)<<10) + ((pBGR[1]>>3)<<5) + (pBGR[0]>>3);
186 *((Int *)pBGRA) = (pBGR[3]<<24) + (pBGR[2]<<16) + (pBGR[1]<<8) + (pBGR[0]);
187 pBGRA +=pixelBytes;
188 pBGR +=TILE_BYTES_PER_PIXEL;
189 }
190 }
191 }
192
193 }
194 DX8_ErrorCode(surface_level->UnlockRect());
195 surface_level->Release();
196 DX8_ErrorCode(D3DXFilterTexture(Peek_D3D_Texture(), NULL, (UINT)0, D3DX_FILTER_BOX));
197 if (TheWritableGlobalData->m_textureReductionFactor) {
198 DX8_ErrorCode(Peek_D3D_Texture()->SetLOD((DWORD)TheWritableGlobalData->m_textureReductionFactor));
199 }
200 return(surface_desc.Height);
201}

301void W3DTreeBuffer::cull(const CameraClass * camera)
302{
303 Int curTree;
304
305 // Calulate the vector direction that the camera is looking at.
306 Matrix3D camera_matrix = camera->Get_Transform();
307 float zmod = -1;
308 float x = zmod * camera_matrix[0][2] ;
309 float y = zmod * camera_matrix[1][2] ;
310 float z = zmod * camera_matrix[2][2] ;
311 m_cameraLookAtVector.Set(x,y,z);
312
313 for (curTree=0; curTree<m_numTrees; curTree++) {
314 Bool doKey = false; // We calculate the key when a tree becomes visible.
315 Bool visible = !camera->Cull_Sphere(m_trees[curTree].bounds);
316 if (visible != m_trees[curTree].visible) {
317 m_trees[curTree].visible=visible;
318 m_anythingChanged = true;
319 if (visible) {
320 doKey = true;
321 }
322 }
323 // Also calculate sort key if a tree is visible, and the view changed setting m_updateAllKeys to true.
324 if (doKey || (visible&&m_updateAllKeys)) {
325 // The sort key is essentially the distance of location in the direction of the
326 // camera look at.
327 m_trees[curTree].sortKey = Vector3::Dot_Product(m_trees[curTree].location, m_cameraLookAtVector);
328 }
329 }
330 m_updateAllKeys = false;
331}

336//=============================================================================
337Int W3DTreeBuffer::getPartitionBucket(const Coord3D &pos) const
338{
339 Real x = pos.x;
340 Real y = pos.y;
341 if (x<m_bounds.lo.x) x = m_bounds.lo.x;
342 if (y<m_bounds.lo.y) y = m_bounds.lo.y;
343 if (x>m_bounds.hi.x) x = m_bounds.hi.x;
344 if (y>m_bounds.hi.y) y = m_bounds.hi.y;
345 Int xIndex = REAL_TO_INT_FLOOR ( (x/(m_bounds.hi.x-m_bounds.lo.x)) * (PARTITION_WIDTH_HEIGHT-0.1f) );
346 Int yIndex = REAL_TO_INT_FLOOR ( (y/(m_bounds.hi.y-m_bounds.lo.y)) * (PARTITION_WIDTH_HEIGHT-0.1f) );
347 DEBUG_ASSERTCRASH(xIndex>=0 && yIndex>=0 && xIndex<PARTITION_WIDTH_HEIGHT && yIndex<PARTITION_WIDTH_HEIGHT, ("Invalid range."));
348 return yIndex*PARTITION_WIDTH_HEIGHT + xIndex;
349}

357void W3DTreeBuffer::updateSway(const BreezeInfo& info)
358{
359 Int i;
360 for (i=0; i<NUM_SWAY_ENTRIES; i++) {
361 Real factor = Cos(i*2.0f*PI/(NUM_SWAY_ENTRIES+1.0f));
362 Real angle = info.m_lean + (info.m_intensity * factor);
363 Real S = Sin(angle);
364 Real C = Cos(angle);
365 m_swayOffsets[i].X = info.m_directionVec.x * S;
366 m_swayOffsets[i].Y = info.m_directionVec.y * S;
367 m_swayOffsets[i].Z = C - 1.0f;
368 }
369
370 Real delta = info.m_randomness * 0.5f;
371 for (i=0; i<m_numTrees; i++) {
372 m_trees[i].swayType = 1+GameClientRandomValue(0, MAX_SWAY_TYPES-1);
373 }
374 for (i=0; i<MAX_SWAY_TYPES; i++) {
375 m_curSwayStep[i] = NUM_SWAY_ENTRIES / (Real)info.m_breezePeriod;
376 m_curSwayStep[i] *= GameClientRandomValueReal(1.0f-delta, 1.0f+delta);
377 if (m_curSwayStep[i]<0.0f) {
378 m_curSwayStep[i] = 0.0f;
379 }
380 m_curSwayOffset[i] = 0;
381 m_curSwayFactor[i] = GameClientRandomValueReal(1.0f-delta, 1.0f+delta);
382 }
383 m_curSwayVersion = info.m_breezeVersion;
384}

438//=============================================================================
439// W3DTreeBuffer::updateTexture
440//=============================================================================

443void W3DTreeBuffer::updateTexture(void)
444{
445
446 const Int MAX_TEX_WIDTH = 2048;
447
448 Int i, j;
449 Int maxHeight = 0;
450 const Int maxTilesPerRow = MAX_TEX_WIDTH/(TILE_PIXEL_EXTENT);
451
452 REF_PTR_RELEASE(m_treeTexture);
453
454 Bool availableGrid[maxTilesPerRow][maxTilesPerRow];
455 Int row, column;
456 for (row=0; row<maxTilesPerRow; row++) {
457 for (column=0; column<maxTilesPerRow; column++) {
458 availableGrid[row][column] = true;
459 }
460 }
461
462 for (i=0; i<m_numTiles; i++) {
463 REF_PTR_RELEASE (m_sourceTiles[i]);
464 }
465 m_numTiles = 0;
466 File *theFile = NULL;
467 for (i=0; i<m_numTreeTypes; i++) {
468 char texturePath[ _MAX_PATH ];
469 m_treeTypes[i].m_numTiles = 0;
470 sprintf( texturePath, "%s%s", TERRAIN_TGA_DIR_PATH, m_treeTypes[i].m_data->m_textureName.str() );
471 theFile = TheFileSystem->openFile( texturePath, File::READ|File::BINARY);
472 if (theFile==NULL) {
473 sprintf( texturePath, "%s%s", TGA_DIR_PATH, m_treeTypes[i].m_data->m_textureName.str() );
474 theFile = TheFileSystem->openFile( texturePath, File::READ|File::BINARY);
475 }
476 if (theFile != NULL) {
477 GDIFileStream2 theStream(theFile);
478 InputStream *pStr = &theStream;
479 Bool halfTile;
480 Int numTiles = WorldHeightMap::countTiles(pStr, &halfTile);
481 Int width;
482 for (width = 10; width >= 1; width--) {
483 if (numTiles >= width*width) {
484 numTiles = width*width;
485 break;
486 }
487 }
488 Bool texFound = false;
489 for (j=0; j<i; j++) {
490 if (m_treeTypes[j].m_data->m_textureName.compareNoCase(m_treeTypes[i].m_data->m_textureName)==0) {
491 m_treeTypes[i].m_firstTile = 0;
492 m_treeTypes[i].m_tileWidth = width;
493 m_treeTypes[i].m_numTiles = 0;
494 texFound = true;
495 break;
496 }
497 }
498 if (texFound) {
499 theFile->close();
500 continue;
501 }
502 if (m_numTiles+numTiles<=MAX_TILES) {
503 theFile->seek(0, File::START);
504 m_treeTypes[i].m_firstTile = m_numTiles;
505 m_treeTypes[i].m_tileWidth = width;
506 m_treeTypes[i].m_numTiles = numTiles;
507 m_treeTypes[i].m_halfTile = halfTile;
508 WorldHeightMap::readTiles(pStr, m_sourceTiles+m_treeTypes[i].m_firstTile, width);
509 m_numTiles += numTiles;
510 } else {
511 m_treeTypes[i].m_firstTile = 0;
512 m_treeTypes[i].m_tileWidth = 0;
513 m_treeTypes[i].m_numTiles = 0;
514 }
515 theFile->close();
516 } else {
517 DEBUG_CRASH(("Could not find texture %s\n", m_treeTypes[i].m_data->m_textureName.str()));
518 m_treeTypes[i].m_firstTile = 0;
519 m_treeTypes[i].m_tileWidth = 0;
520 m_treeTypes[i].m_numTiles = 0;
521 }
522 }
523
524 Int tmpWidth = 8;
525 while (tmpWidth*tmpWidth<m_numTiles) {
526 tmpWidth*=2;
527 }
528 Int tilesPerRow = tmpWidth;
529 m_textureWidth = tmpWidth*TILE_PIXEL_EXTENT;
530
531 if (m_textureWidth>MAX_TEX_WIDTH) {
532 m_textureWidth = 64;
533 m_textureHeight = 64;
534 if (m_treeTexture==NULL) {
535 m_treeTexture = new TextureClass("missing.tga");
536 }
537 DEBUG_CRASH(("Too many trees in a scene."));
538 return;
539 }
540
541 for (i=0; i<m_numTiles; i++) {
542 if (m_sourceTiles[i]) {
543 m_sourceTiles[i]->m_tileLocationInTexture.x = -1;
544 m_sourceTiles[i]->m_tileLocationInTexture.y = -1;
545 }
546 }
547
548 /* put the tree tiles into the texture */
549 Int texClass;
550 Int tileWidth;
551 for (tileWidth = tilesPerRow; tileWidth>0; tileWidth--) {
552 for (texClass=0; texClass<m_numTreeTypes; texClass++) {
553 Int width = m_treeTypes[texClass].m_tileWidth;
554 if (width != tileWidth) continue;
555 Bool texFound = false;
556 for (i=0; i<texClass; i++) {
557 if (m_treeTypes[i].m_data->m_textureName.compareNoCase(m_treeTypes[texClass].m_data->m_textureName)==0) {
558 m_treeTypes[texClass].m_textureOrigin.x = m_treeTypes[i].m_textureOrigin.x;
559 m_treeTypes[texClass].m_textureOrigin.y = m_treeTypes[i].m_textureOrigin.y;
560 texFound = true;
561 break;
562 }
563 }
564 if (texFound) {
565 continue;
566 }
567
568 // Find an available block of space.
569 Bool found = false;
570 for (row=0; row<(tilesPerRow-width)+1 && !found; row++) {
571 for (column=0; column<(tilesPerRow-width)+1 && !found; column++) {
572 if (availableGrid[row][column]) {
573 Bool open = true;
574 for (i=0; i<width && open; i++) {
575 for (j=0; j<width&&open; j++) {

635//=============================================================================
636// W3DTreeBuffer::doLighting
637//=============================================================================
639//=============================================================================
640UnsignedInt W3DTreeBuffer::doLighting(const Vector3 *normal,
641 const GlobalData::TerrainLighting *objectLighting,
642 const Vector3 *emissive, UnsignedInt vertDiffuse, Real scale) const
643{
644
645 Real shadeR, shadeG, shadeB;
646 Real shade;
647 shadeR = objectLighting[0].ambient.red+emissive->X; //only the first light contributes to ambient
648 shadeG = objectLighting[0].ambient.green+emissive->Y;
649 shadeB = objectLighting[0].ambient.blue+emissive->Z;
650
651 Int i;
652 for (i=0; i<MAX_GLOBAL_LIGHTS; i++) {
653 Vector3 lightDirection(objectLighting[i].lightPos.x, objectLighting[i].lightPos.y, objectLighting[i].lightPos.z);
654 lightDirection.Normalize();
655 Vector3 lightRay(-lightDirection.X, -lightDirection.Y, -lightDirection.Z);
656 shade = Vector3::Dot_Product(lightRay, *normal);
657
658 if (shade > 1.0) shade = 1.0;
659 if(shade < 0.0f) shade = 0.0f;
660 shadeR += shade*objectLighting[i].diffuse.red;
661 shadeG += shade*objectLighting[i].diffuse.green;
662 shadeB += shade*objectLighting[i].diffuse.blue;
663 }
664
665 shadeR *= scale;
666 shadeG *= scale;
667 shadeB *= scale;
668
669 if (shadeR > 1.0) shadeR = 1.0;
670 if(shadeR < 0.0f) shadeR = 0.0f;
671 if (shadeG > 1.0) shadeG = 1.0;
672 if(shadeG < 0.0f) shadeG = 0.0f;
673 if (shadeB > 1.0) shadeB = 1.0;
674 if(shadeB < 0.0f) shadeB = 0.0f;
675
676 if (vertDiffuse!=0xFFFFFFFF) {
677 shade = vertDiffuse&0xff; //blue;
678 shadeB *= shade/255.0f;
679 shade = (vertDiffuse>>8)&0xFF; // green;
680 shadeG *= shade/255.0f;
681 shade = (vertDiffuse>>16)&0xFF; // red;
682 shadeR *= shade/255.0f;
683 }
684
685 shadeR*=255.0f;
686 shadeG*=255.0f;
687 shadeB*=255.0f;
688 const Real alpha = 255.0;
689 return REAL_TO_UNSIGNEDINT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16) | ((Int)alpha << 24);
690
691}

1158{
1159 if (unit->isKindOf(KINDOF_IMMOBILE)) {
1160 // This is the initial positioning of the object, and we don't care. jba. [6/5/2003]
1161 return;
1162 }
1163 Real radius = unit->getGeometryInfo().getMajorRadius();
1164 if (unit->getGeometryInfo().getGeomType()==GEOMETRY_BOX) {
1165 if (radius>unit->getGeometryInfo().getMinorRadius()) {
1166 radius = unit->getGeometryInfo().getMinorRadius();
1167 }
1168 }
1169 // Value to assume for the tree radius.
1170#define TREE_RADIUS_APPROX 7.0f
1171 radius += TREE_RADIUS_APPROX;
1172
1173 Coord3D pos = *unit->getPosition();
1174 Real x = pos.x-radius;
1175 Real y = pos.y-radius;
1176 if (x<m_bounds.lo.x) x = m_bounds.lo.x;

1426 return;

1429 }
1430 if (data->m_framesToMoveOutward > 2 || data->m_doTopple) {

1447 m_trees[m_numTrees].scale = scale;
1448 }
1449 m_trees[m_numTrees].location = Vector3(location.x, location.y, location.z);
1450 m_trees[m_numTrees].treeType = treeType;

1461 m_trees[m_numTrees].bufferNdx = -1;
1462

1465 m_trees[m_numTrees].lastFrameUpdated = 0;

1864void W3DTreeBuffer::updateTopplingTree(TTree *tree)
1865{
1866 //DLOG(Debug::Format("updating W3DTreeBuffer %08lx\n",this));
1867 DEBUG_ASSERTCRASH(tree->m_toppleState != TOPPLE_UPRIGHT, ("hmm, we should be sleeping here"));
1868 if ( (tree->m_toppleState == TOPPLE_UPRIGHT) || (tree->m_toppleState == TOPPLE_DOWN) )
1869 return;
1870
1871 const W3DTreeDrawModuleData* d = m_treeTypes[tree->treeType].m_data;
1872 Int localPlayerIndex = ThePlayerList ? ThePlayerList->getLocalPlayer()->getPlayerIndex() : 0;
1873 Coord3D pos;
1874 pos.set(tree->location.X, tree->location.Y, tree->location.Z);
1875 ObjectShroudStatus ss = ThePartitionManager->getPropShroudStatusForPlayer(localPlayerIndex, &pos);
1876 if (ss==OBJECTSHROUD_FOGGED) {
1877 // Don't update fogged trees. [8/11/2003]
1878 tree->m_toppleState = TOPPLE_FOGGED;
1879 return;
1880 } else if (tree->m_toppleState == TOPPLE_FOGGED) {
1881 // was fogged, now isn't.
1882 tree->m_angularVelocity = 0;
1883 tree->m_toppleState = TOPPLE_DOWN;
1884 tree->m_mtx.In_Place_Pre_Rotate_X(-ANGULAR_LIMIT * tree->m_toppleDirection.y);
1885 tree->m_mtx.In_Place_Pre_Rotate_Y(ANGULAR_LIMIT * tree->m_toppleDirection.x);
1886 if (d->m_killWhenToppled) {
1887 // If got killed in the fog, just remove. jba [8/11/2003]
1888 tree->m_sinkFramesLeft = 0;
1889 }
1890 return;
1891 }
1892 const Real VELOCITY_BOUNCE_LIMIT = 0.01f; // if the velocity after a bounce will be this or lower, just stop at zero
1893 const Real VELOCITY_BOUNCE_SOUND_LIMIT = 0.03f; // and if this low, then skip the bounce sound
1894
1895 Real curVelToUse = tree->m_angularVelocity;
1896 if (tree->m_angularAccumulation + curVelToUse > ANGULAR_LIMIT)
1897 curVelToUse = ANGULAR_LIMIT - tree->m_angularAccumulation;
1898
1899 tree->m_mtx.In_Place_Pre_Rotate_X(-curVelToUse * tree->m_toppleDirection.y);
1900 tree->m_mtx.In_Place_Pre_Rotate_Y(curVelToUse * tree->m_toppleDirection.x);
1901
1902
1903 tree->m_angularAccumulation += curVelToUse;
1904 if ((tree->m_angularAccumulation >= ANGULAR_LIMIT) && (tree->m_angularVelocity > 0))
1905 {
1906 // Hit so either bounce or stop if too little remaining velocity.
1907 tree->m_angularVelocity *= -d->m_bounceVelocityPercent;
1908
1909 if( BitTest( tree->m_options, W3D_TOPPLE_OPTIONS_NO_BOUNCE ) == TRUE ||
1910 fabs(tree->m_angularVelocity) < VELOCITY_BOUNCE_LIMIT )
1911 {
1912 // too slow, just stop
1913 tree->m_angularVelocity = 0;
1914 tree->m_toppleState = TOPPLE_DOWN;
1915 if (d->m_killWhenToppled) {
1916 tree->m_sinkFramesLeft = d->m_sinkFrames;
1917 }
1918 }
1919 else if( fabs(tree->m_angularVelocity) >= VELOCITY_BOUNCE_SOUND_LIMIT )
1920 {
1921 // fast enough bounce to warrant the bounce fx
1922 if( BitTest( tree->m_options, W3D_TOPPLE_OPTIONS_NO_FX ) == FALSE ) {
1923 Vector3 loc(0, 0, 3*TREE_RADIUS_APPROX); // Kinda towards the top of the tree. jba. [7/11/2003]
1924 Vector3 xloc;
1925 tree->m_mtx.Transform_Vector(tree->m_mtx, loc, &xloc);
1926 Coord3D pos;
1927 pos.set(xloc.X, xloc.Y, xloc.Z);
1928 FXList::doFXPos(d->m_bounceFX, &pos);
1929 }
1930 }
1931 }
1932 else
1933 {
1934 tree->m_angularVelocity += tree->m_angularAcceleration;
1935 }
