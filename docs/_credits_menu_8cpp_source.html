85void CreditsMenuInit( WindowLayout *layout, void *userData )
86{
87 TheShell->showShellMap(FALSE);
88 if(TheCredits)
89 delete TheCredits;
90 TheCredits = new CreditsManager;
91 TheCredits->load();
92 TheCredits->init();
93
94 parentMainMenuID = TheNameKeyGenerator->nameToKey( AsciiString("CreditsMenu.wnd:ParentCreditsWindow") );
95 parentMainMenu = TheWindowManager->winGetWindowFromId( NULL, parentMainMenuID );
96
97
98 // show menu
99 layout->hide( FALSE );
100
101 // set keyboard focus to main parent
102 TheWindowManager->winSetFocus( parentMainMenu );
103
104
105
106 TheAudio->removeAudioEvent( AHSV_StopTheMusicFade );
107 AudioEventRTS event( AsciiString( "Credits" ) );
108 event.setShouldFade( TRUE );
109 TheAudio->addAudioEvent( &event );
110
111
112} // end CreditsMenuInit

117void CreditsMenuShutdown( WindowLayout *layout, void *userData )
118{
119 TheCredits->reset();
120 delete TheCredits;
121 TheCredits = NULL;
122 TheShell->showShellMap(TRUE);
123
124 // hide menu
125 layout->hide( TRUE );
126
127 // our shutdown is complete
128 TheShell->shutdownComplete( layout );
129
130 TheAudio->removeAudioEvent( AHSV_StopTheMusicFade );

132} // end CreditsMenuShutdown

155WindowMsgHandledType CreditsMenuInput( GameWindow *window, UnsignedInt msg,
156 WindowMsgData mData1, WindowMsgData mData2 )
157{
158
159 switch( msg )
160 {
161
162 // --------------------------------------------------------------------------------------------
163 case GWM_CHAR:
164 {
165 UnsignedByte key = mData1;
166 UnsignedByte state = mData2;
167
168 switch( key )
169 {
170
171 // ----------------------------------------------------------------------------------------
172 case KEY_ESC:
173 {
174
175 //
176 // send a simulated selected event to the parent window of the
177 // back/exit button
178 //
179 if( BitTest( state, KEY_STATE_UP ) )
180 {
181
182 TheShell->pop();
183
184 } // end if
185
186 // don't let key fall through anywhere else
187 return MSG_HANDLED;
188
189 } // end escape
190
191 } // end switch( key )
192
193 } // end char
194
195 } // end switch( msg )
196
197 return MSG_IGNORED;
198
199} // end CreditsMenuInput
