71void EnableAcceptControls(Bool Enabled, GameInfo *myGame, GameWindow *comboPlayer[],
72 GameWindow *comboColor[], GameWindow *comboPlayerTemplate[],
73 GameWindow *comboTeam[], GameWindow *buttonAccept[], GameWindow *buttonStart,
74 GameWindow *buttonMapStartPosition[], Int slotNum)
75{
76 if(slotNum == -1 || slotNum >= MAX_SLOTS )
77 slotNum = myGame->getLocalSlotNum();
78
79 Bool isObserver = myGame->getConstSlot(slotNum)->getPlayerTemplate() == PLAYERTEMPLATE_OBSERVER;
80
81 if( !myGame->amIHost() && (buttonStart != NULL) )
82 buttonStart->winEnable(Enabled);
83 if(comboColor[slotNum])
84 {
85 if (isObserver)
86 {
87 GadgetComboBoxHideList(comboColor[slotNum]);
88 }
89 comboColor[slotNum]->winEnable(Enabled && !isObserver);
90 }
91 if(comboPlayerTemplate[slotNum])
92 comboPlayerTemplate[slotNum]->winEnable(Enabled);
93 if(comboTeam[slotNum])
94 {
95 if (isObserver)
96 {
97 GadgetComboBoxHideList(comboTeam[slotNum]);
98 }
99 comboTeam[slotNum]->winEnable(Enabled && !isObserver);
100 }
101
102 Bool canChooseStartSpot = FALSE;
103 if (!isObserver)
104 canChooseStartSpot = TRUE;
105 for (Int i=0; i<MAX_SLOTS && !canChooseStartSpot && myGame->amIHost(); ++i)
106 {
107 if (myGame->getConstSlot(i) && myGame->getConstSlot(i)->isAI())
108 canChooseStartSpot = TRUE;
109 }
110
111 if (slotNum == myGame->getLocalSlotNum())
112 {
113 if (myGame->getConstSlot(myGame->getLocalSlotNum())->hasMap())
114 {
115 for (Int i=0; i<MAX_SLOTS; ++i)
116 {
117 if (buttonMapStartPosition[i])
118 {
119 buttonMapStartPosition[i]->winEnable(Enabled && canChooseStartSpot);
120 }
121 }
122 }
123 else
124 {
125 for (Int i=0; i<MAX_SLOTS; ++i)
126 {
127 if (buttonMapStartPosition[i])
128 buttonMapStartPosition[i]->winEnable(FALSE);
129 }
130 }
131 }
132}

136void ShowUnderlyingGUIElements( Bool show, const char *layoutFilename, const char *parentName,
137 const char **gadgetsToHide, const char **perPlayerGadgetsToHide )
138{
139 AsciiString parentNameStr;
140 parentNameStr.format("%s:%s", layoutFilename, parentName);
141 NameKeyType parentID = NAMEKEY(parentNameStr);
142 GameWindow *parent = TheWindowManager->winGetWindowFromId( NULL, parentID );
143 if (!parent)
144 {
145 DEBUG_CRASH(("Window %s not found\n", parentNameStr.str()));
146 return;
147 }
148
149 // hide some GUI elements of the screen underneath
150 GameWindow *win;
151
152 Int player;
153 const char **text;
154
155 text = gadgetsToHide;
156 while (*text)
157 {
158 AsciiString gadgetName;
159 gadgetName.format("%s:%s", layoutFilename, *text);
160 win = TheWindowManager->winGetWindowFromId( parent, NAMEKEY(gadgetName) );
161 //DEBUG_ASSERTCRASH(win, ("Cannot find %s to show/hide it", gadgetName.str()));
162 if (win)
163 {
164 win->winHide( !show );
165 }
166 ++text;
167 }
168
169 text = perPlayerGadgetsToHide;
170 while (*text)
171 {
172 for (player = 0; player < MAX_SLOTS; ++player)
173 {
174 AsciiString gadgetName;
175 gadgetName.format("%s:%s%d", layoutFilename, *text, player);
176 win = TheWindowManager->winGetWindowFromId( parent, NAMEKEY(gadgetName) );
177 //DEBUG_ASSERTCRASH(win, ("Cannot find %s to show/hide it", gadgetName.str()));
178 if (win)
179 {
180 win->winHide( !show );
181 }
182 }
183 ++text;
184 }
185}

189void PopulateColorComboBox(Int comboBox, GameWindow *comboArray[], GameInfo *myGame, Bool isObserver)
190{
191 Int numColors = TheMultiplayerSettings->getNumColors();
192 UnicodeString colorName;
193 std::vector<bool> availableColors;
194
195 for (Int i = 0; i < numColors; i++)
196 availableColors.push_back(true);
197
198 for (i = 0; i < MAX_SLOTS; i++)
199 {
200 GameSlot *slot = myGame->getSlot(i);
201 if( slot && (i != comboBox) && (slot->getColor() >= 0 )&& (slot->getColor() < numColors))
202 {
203 DEBUG_ASSERTCRASH(slot->getColor() >= 0,("We've tried to access array %d and that ain't good",slot->getColor()));
204 availableColors[slot->getColor()] = false;
205 }
206 }
207
208 Bool wasObserver = (GadgetComboBoxGetLength(comboArray[comboBox]) == 1);
209 GadgetComboBoxReset(comboArray[comboBox]);
210
211 MultiplayerColorDefinition *def = TheMultiplayerSettings->getColor(PLAYERTEMPLATE_RANDOM);
212 Int newIndex = GadgetComboBoxAddEntry(comboArray[comboBox],
213 (isObserver)?TheGameText->fetch("GUI:None"):TheGameText->fetch("GUI:???"), def->getColor());
214 GadgetComboBoxSetItemData(comboArray[comboBox], newIndex, (void *)-1);
215
216 if (isObserver)
217 {
218 GadgetComboBoxSetSelectedPos(comboArray[comboBox], 0);
219 return;
220 }
221
222 for (Int c=0; c<numColors; ++c)
223 {
224 def = TheMultiplayerSettings->getColor(c);
225 if (!def || availableColors[c] == false)
226 continue;
227
228 colorName = TheGameText->fetch(def->getTooltipName().str());
229 newIndex = GadgetComboBoxAddEntry(comboArray[comboBox], colorName, def->getColor());
230 GadgetComboBoxSetItemData(comboArray[comboBox], newIndex, (void *)c);
231 }
232 if (wasObserver)
233 GadgetComboBoxSetSelectedPos(comboArray[comboBox], 0);
234}

238void PopulatePlayerTemplateComboBox(Int comboBox, GameWindow *comboArray[], GameInfo *myGame, Bool allowObservers)
239{
240 Int numPlayerTemplates = ThePlayerTemplateStore->getPlayerTemplateCount();
241 UnicodeString playerTemplateName;
242
243 GadgetComboBoxReset(comboArray[comboBox]);
244
245 MultiplayerColorDefinition *def = TheMultiplayerSettings->getColor(PLAYERTEMPLATE_RANDOM);
246 Int newIndex = GadgetComboBoxAddEntry(comboArray[comboBox], TheGameText->fetch("GUI:Random"), def->getColor());
247 GadgetComboBoxSetItemData(comboArray[comboBox], newIndex, (void *)PLAYERTEMPLATE_RANDOM);
248
249 std::set<AsciiString> seenSides;
250
251 for (Int c=0; c<numPlayerTemplates; ++c)
252 {
253 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(c);
254 if (!fac)
255 continue;
256
257 if (fac->getStartingBuilding().isEmpty())
258 continue;
259
260 if ( myGame->oldFactionsOnly() && !fac->isOldFaction() )
261 continue;
262
263 // Prevent players from selecting the disabled Generals for use.
264 // This is also enforced at game loading (GameLogic.cpp and UserPreferences.cpp).
265 // @todo: unlock these when something rad happens
266 Bool disallowLockedGenerals = TRUE;
267 const GeneralPersona *general = TheChallengeGenerals->getGeneralByTemplateName(fac->getName());
268 Bool startsLocked = general ? !general->isStartingEnabled() : FALSE;
269 if (disallowLockedGenerals && startsLocked)
270 continue;
271
272
273 AsciiString side;
274 side.format("SIDE:%s", fac->getSide().str());
275 if (seenSides.find(side) != seenSides.end())
276 continue;
277
278 seenSides.insert(side);
279
280 newIndex = GadgetComboBoxAddEntry(comboArray[comboBox], TheGameText->fetch(side), def->getColor());
281 GadgetComboBoxSetItemData(comboArray[comboBox], newIndex, (void *)c);
282 }
283 seenSides.clear();
284
285 // disabling observers for Multiplayer test
286 if (allowObservers)
287 {
288 def = TheMultiplayerSettings->getColor(PLAYERTEMPLATE_OBSERVER);
289 newIndex = GadgetComboBoxAddEntry(comboArray[comboBox], TheGameText->fetch("GUI:Observer"), def->getColor());
290 GadgetComboBoxSetItemData(comboArray[comboBox], newIndex, (void *)PLAYERTEMPLATE_OBSERVER);
291 }
292 GadgetComboBoxSetSelectedPos(comboArray[comboBox], 0);
293
294}

334void PopulateStartingCashComboBox(GameWindow *comboBox, GameInfo *myGame)
335{
336 GadgetComboBoxReset(comboBox);
337
338 const MultiplayerStartingMoneyList & startingCashMap = TheMultiplayerSettings->getStartingMoneyList();
339 Int currentSelectionIndex = -1;
340
341 for ( MultiplayerStartingMoneyList::const_iterator it = startingCashMap.begin(); it != startingCashMap.end(); it++ )
342 {
343 Int newIndex = GadgetComboBoxAddEntry(comboBox, formatMoneyForStartingCashComboBox( *it ),
344 comboBox->winGetEnabled() ? comboBox->winGetEnabledTextColor() : comboBox->winGetDisabledTextColor());
345 GadgetComboBoxSetItemData(comboBox, newIndex, (void *)it->countMoney());
346
347 if ( myGame->getStartingCash().amountEqual( *it ) )
348 {
349 currentSelectionIndex = newIndex;
350 }
351 }
352
353 if ( currentSelectionIndex == -1 )
354 {
355 DEBUG_CRASH( ("Current selection for starting cash not found in list") );
356 currentSelectionIndex = GadgetComboBoxAddEntry(comboBox, formatMoneyForStartingCashComboBox( myGame->getStartingCash() ),
357 comboBox->winGetEnabled() ? comboBox->winGetEnabledTextColor() : comboBox->winGetDisabledTextColor());
358 GadgetComboBoxSetItemData(comboBox, currentSelectionIndex, (void *)it->countMoney() );
359 }
360
361 GadgetComboBoxSetSelectedPos(comboBox, currentSelectionIndex);
362}

369void UpdateSlotList( GameInfo *myGame, GameWindow *comboPlayer[],
370 GameWindow *comboColor[], GameWindow *comboPlayerTemplate[],
371 GameWindow *comboTeam[], GameWindow *buttonAccept[],
372 GameWindow *buttonStart, GameWindow *buttonMapStartPosition[] )
373{
374 if(!AreSlotListUpdatesEnabled())
375 return;
376 //LANGameInfo *myGame = TheLAN->GetMyGame();
377
378 const MapMetaData *mapData = TheMapCache->findMap( myGame->getMap() );
379 Bool willTransfer = TRUE;
380 if (mapData)
381 {
382 willTransfer = !mapData->m_isOfficial;
383 }
384 else
385 {
386 willTransfer = WouldMapTransfer(myGame->getMap());
387 }
388
389 if (myGame)
390 {
391 for( int i =0; i < MAX_SLOTS; i++ )
392 {
393 GameSlot * slot = myGame->getSlot(i);
394 // if i'm host, enable the controls for AI
395 if(myGame->amIHost() && slot && slot->isAI())
396 {
397 EnableAcceptControls(TRUE, myGame, comboPlayer, comboColor, comboPlayerTemplate,
398 comboTeam, buttonAccept, buttonStart, buttonMapStartPosition, i);
399 }
400 else if (slot && myGame->getLocalSlotNum() == i)
401 {
402 if(slot->isAccepted() && !myGame->amIHost())
403 {
404 EnableAcceptControls(FALSE, myGame, comboPlayer, comboColor, comboPlayerTemplate,
405 comboTeam, buttonAccept, buttonStart, buttonMapStartPosition);
406 }
407 else
408 {
409 if (slot->hasMap()) {
410 EnableAcceptControls(TRUE, myGame, comboPlayer, comboColor, comboPlayerTemplate,
411 comboTeam, buttonAccept, buttonStart, buttonMapStartPosition);
412 }
413 else
414 {
415 EnableAcceptControls(willTransfer, myGame, comboPlayer, comboColor, comboPlayerTemplate,
416 comboTeam, buttonAccept, buttonStart, buttonMapStartPosition);
417 }
418 }
419
420 }
421 else if(myGame->amIHost())
422 {
423 EnableAcceptControls(FALSE, myGame, comboPlayer, comboColor, comboPlayerTemplate,
424 comboTeam, buttonAccept, buttonStart, buttonMapStartPosition, i);
425 }
426 if(slot && slot->isHuman())
427 {
428 UnicodeString newName = slot->getName();
429 UnicodeString oldName = GadgetComboBoxGetText(comboPlayer[i]);
430 if (comboPlayer[i] && newName.compare(oldName))
431 {
432 GadgetComboBoxSetText(comboPlayer[i], newName);

434 if(i!= 0 && buttonAccept && buttonAccept[i])
435 {

455 {
456 GadgetComboBoxSetSelectedPos(comboPlayer[i], slot->getState(), TRUE);
457 if( buttonAccept && buttonAccept[i] )
458 buttonAccept[i]->winHide(TRUE);
459 }
460/*
461 if (myGame->getLocalSlotNum() == i && i!=0)

463 if (comboPlayer[i])
464 comboPlayer[i]->winEnable( TRUE );
465 }
466 else*/ if (!myGame->amIHost())
467 {
468 if (comboPlayer[i])
469 comboPlayer[i]->winEnable( FALSE );
470 }
471 //if( i == myGame->getLocalSlotNum())
472 if((comboColor[i] != NULL) && BitTest(comboColor[i]->winGetStatus(), WIN_STATUS_ENABLED))
