179//=============================================================================
180// TerrainTracksRenderObjClass::addCapEdgeToTrack
181//=============================================================================
187//=============================================================================
188void TerrainTracksRenderObjClass::addCapEdgeToTrack(Real x, Real y)
189{
191
192 if (m_haveCap)
193 { //we already have a cap or there are no segments to cap
194 return;
195 }
196
197 if (m_activeEdgeCount == 1)
198 { //if we only have one edge, then it must be the current anchor edge.
199 //since achnors are caps, there is not point in adding another.
200 m_haveCap=TRUE;
201 m_haveAnchor=false; //recreate a new anchor when track resumes.
202 return;
203 }
204
205 Vector3 vPos,vZ;
206 Coord3D vZTmp;
207 PathfindLayerEnum objectLayer;
208 Real eHeight;
209
210 if (m_ownerDrawable && (objectLayer=m_ownerDrawable->getObject()->getLayer()) != LAYER_GROUND)
211 eHeight=BRIDGE_OFFSET_FACTOR+TheTerrainLogic->getLayerHeight(x,y,objectLayer,&vZTmp);
212 else
213 eHeight=TheTerrainLogic->getGroundHeight(x,y,&vZTmp);
214
215 vZ.X = vZTmp.x;
216 vZ.Y = vZTmp.y;
217 vZ.Z = vZTmp.z;
218
219 vPos.X=x;
220 vPos.Y=y;
221 vPos.Z=eHeight;
222
223 Vector3 vDir=Vector3(x,y,eHeight)-m_lastAnchor;
224 Int maxEdgeCount=TheTerrainTracksRenderObjClassSystem->m_maxTankTrackEdges;
225
226 //avoid sqrt() by checking distance squared since last track mark
227 if (vDir.Length2() < sqr(m_length))
228 { //not far enough from anchor to add track
229 //since this is a cap, we'll force the previous segment to transparent
230 Int lastAddedEdge=m_topIndex-1;
231 if (lastAddedEdge < 0)
232 lastAddedEdge = maxEdgeCount-1;
233 m_edges[lastAddedEdge].alpha=0.0f; //force the last added edge to transparent.
234 m_haveCap=TRUE;
235 m_haveAnchor=false; //recreate a new anchor when track resumes.
236 return;
237 }
238
239 if (m_activeEdgeCount >= maxEdgeCount)
240 { //no more room in buffer so release oldest edge
241 m_bottomIndex++;
242 m_activeEdgeCount--;
243
244 if (m_bottomIndex >= maxEdgeCount)
245 m_bottomIndex=0; //roll buffer back to start
246 }
247
248 if (m_topIndex >= maxEdgeCount)
249 m_topIndex=0; //roll around buffer
250
251 //we traveled far enough from last point.
252 //accept new point
253 vDir.Z=0; //ignore height
254 vDir.Normalize();
255
256 Vector3 vX;
257
258 Vector3::Cross_Product(vDir,vZ,&vX);
259
260 //calculate left end point
261 edgeInfo& topEdge = m_edges[m_topIndex];
262
263 topEdge.endPointPos[0]=vPos-(m_width*0.5f*vX);
264 topEdge.endPointPos[0].Z += 0.2f * MAP_XY_FACTOR; //raise above terrain slightly
265
266 if (m_totalEdgesAdded&1) //every other edge has different set of UV's
267 {
268 topEdge.endPointUV[0].X=0.0f;
269 topEdge.endPointUV[0].Y=0.0f;
270 }
271 else
272 {
273 topEdge.endPointUV[0].X=0.0f;
274 topEdge.endPointUV[0].Y=1.0f;
275 }
276
277 //calculate right end point
278 topEdge.endPointPos[1]=vPos+(m_width*0.5f*vX);
279 topEdge.endPointPos[1].Z += 0.2f * MAP_XY_FACTOR; //raise above terrain slightly
280
281 if (m_totalEdgesAdded&1) //every other edge has different set of UV's
282 {
283 topEdge.endPointUV[1].X=1.0f;
284 topEdge.endPointUV[1].Y=0.0f;
285 }
286 else
287 {
288 topEdge.endPointUV[1].X=1.0f;
289 topEdge.endPointUV[1].Y=1.0f;
290 }
291
292 topEdge.timeAdded=WW3D::Get_Sync_Time();
293 topEdge.alpha=0.0f; //fully transparent at cap.
294 m_lastAnchor=vPos;
295 m_activeEdgeCount++;
296 m_totalEdgesAdded++;
297 m_topIndex++; //make space for new edge
298 m_haveCap=TRUE;
299 m_haveAnchor=false;
300}

310void TerrainTracksRenderObjClass::addEdgeToTrack(Real x, Real y)
311{
313
314 if (!m_haveAnchor)
315 { //no anchor yet, make this point an anchor.
316 PathfindLayerEnum objectLayer;
317 if (m_ownerDrawable && (objectLayer=m_ownerDrawable->getObject()->getLayer()) != LAYER_GROUND)
318 m_lastAnchor=Vector3(x,y,TheTerrainLogic->getLayerHeight(x,y,objectLayer)+BRIDGE_OFFSET_FACTOR);
319 else
320 m_lastAnchor=Vector3(x,y,TheTerrainLogic->getGroundHeight(x,y));
321
322 m_haveAnchor=true;
323 m_airborne = true;
324 m_haveCap = true; //single segment tracks are always capped because nothing is drawn.
325 return;
326 }
327
328 m_haveCap = false; //have more than 1 segment now so will need to cap if it's interrupted.
329
330 Vector3 vPos,vZ;
331 Coord3D vZTmp;
332 Real eHeight;
333 PathfindLayerEnum objectLayer;
334
335 if (m_ownerDrawable && (objectLayer=m_ownerDrawable->getObject()->getLayer()) != LAYER_GROUND)
336 eHeight=BRIDGE_OFFSET_FACTOR+TheTerrainLogic->getLayerHeight(x,y,objectLayer,&vZTmp);
337 else
338 eHeight=TheTerrainLogic->getGroundHeight(x,y,&vZTmp);
339
340 vZ.X = vZTmp.x;
341 vZ.Y = vZTmp.y;
342 vZ.Z = vZTmp.z;
343
344 vPos.X=x;
345 vPos.Y=y;
346 vPos.Z=eHeight;
347
348 Vector3 vDir=Vector3(x,y,eHeight)-m_lastAnchor;
349
350 //avoid sqrt() by checking distance squared since last track mark
351 if (vDir.Length2() < sqr(m_length))
352 return; //not far enough from anchor to add track
353
354 Int maxEdgeCount=TheTerrainTracksRenderObjClassSystem->m_maxTankTrackEdges;
355
356 if (m_activeEdgeCount >= maxEdgeCount)
357 { //no more room in buffer so release oldest edge
358 m_bottomIndex++;
359 m_activeEdgeCount--;
360
361 if (m_bottomIndex >= maxEdgeCount)
362 m_bottomIndex=0; //roll buffer back to start
363 }
364
365 if (m_topIndex >= maxEdgeCount)
366 m_topIndex=0; //roll around buffer
367
368 //we traveled far enough from last point.
369 //accept new point
370 vDir.Z=0; //ignore height
371 vDir.Normalize();
372
373 Vector3 vX;
374
375 Vector3::Cross_Product(vDir,vZ,&vX);
376
377 edgeInfo& topEdge = m_edges[m_topIndex];
378
379 //calculate left end point
380 topEdge.endPointPos[0]=vPos-(m_width*0.5f*vX);
381 topEdge.endPointPos[0].Z += 0.2f * MAP_XY_FACTOR; //raise above terrain slightly
382
383 if (m_totalEdgesAdded&1) //every other edge has different set of UV's
384 {
385 topEdge.endPointUV[0].X=0.0f;
386 topEdge.endPointUV[0].Y=0.0f;
387 }
388 else
389 {
390 topEdge.endPointUV[0].X=0.0f;
391 topEdge.endPointUV[0].Y=1.0f;
392 }
393
394 //calculate right end point
395 topEdge.endPointPos[1]=vPos+(m_width*0.5f*vX);
396 topEdge.endPointPos[1].Z += 0.2f * MAP_XY_FACTOR; //raise above terrain slightly
397
398 if (m_totalEdgesAdded&1) //every other edge has different set of UV's
399 {
400 topEdge.endPointUV[1].X=1.0f;
401 topEdge.endPointUV[1].Y=0.0f;
402 }
403 else
404 {
405 topEdge.endPointUV[1].X=1.0f;
406 topEdge.endPointUV[1].Y=1.0f;
407 }
408
409 topEdge.timeAdded=WW3D::Get_Sync_Time();
410 topEdge.alpha=1.0f; //fully opaque at start.
411 if (m_airborne || m_activeEdgeCount <= 1) {
412 topEdge.alpha=0.0f; //smooth out track restarts by setting transparent
413 }
414 m_airborne = false;
415
416 m_lastAnchor=vPos;
417 m_activeEdgeCount++;
418 m_totalEdgesAdded++;
419 m_topIndex++; //make space for new edge
420}

435#define DEFAULT_TRACK_SPACING (MAP_XY_FACTOR * 1.4f)
436#define DEFAULT_TRACK_WIDTH 4.0f;
437
441static Real computeTrackSpacing(RenderObjClass *renderObj)
442{
443 Real trackSpacing = DEFAULT_TRACK_SPACING;
444 Int leftTrack;
445 Int rightTrack;
446
447 if ((leftTrack=renderObj->Get_Bone_Index( "TREADFX01" )) != 0 && (rightTrack=renderObj->Get_Bone_Index( "TREADFX02" )) != 0)
448 { //both bones found, determine distance between them.
449 Vector3 leftPos,rightPos;
450 leftPos=renderObj->Get_Bone_Transform( leftTrack ).Get_Translation();
451 rightPos=renderObj->Get_Bone_Transform( rightTrack ).Get_Translation();
452 rightPos -= leftPos; //get distance between centers of tracks
453 trackSpacing = rightPos.Length() + DEFAULT_TRACK_WIDTH; //add width of each track
455 };
456
457 return trackSpacing;
458}

474TerrainTracksRenderObjClass *TerrainTracksRenderObjClassSystem::bindTrack( RenderObjClass *renderObject, Real length, const Char *texturename)
475{
476 TerrainTracksRenderObjClass *mod;
477
478 mod = m_freeModules;
479 if( mod )
480 {
481 // take module off the free list
482 if( mod->m_nextSystem )
483 mod->m_nextSystem->m_prevSystem = mod->m_prevSystem;
484 if( mod->m_prevSystem )
485 mod->m_prevSystem->m_nextSystem = mod->m_nextSystem;
486 else
487 m_freeModules = mod->m_nextSystem;
488
489 // put module on the used list
490 mod->m_prevSystem = NULL;
491 mod->m_nextSystem = m_usedModules;
492 if( m_usedModules )
493 m_usedModules->m_prevSystem = mod;
494 m_usedModules = mod;
495
496 mod->init(computeTrackSpacing(renderObject),length,texturename);
497 mod->m_bound=true;
498 m_TerrainTracksScene->Add_Render_Object( mod);
499 } // end if
500
501 return mod;
502
503} //end bindTrack

587//=============================================================================
588// TerrainTracksRenderObjClassSystem::ReAcquireResources
589//=============================================================================
591//=============================================================================
592void TerrainTracksRenderObjClassSystem::ReAcquireResources(void)
593{
594 Int i;
595 const Int numModules=TheGlobalData->m_maxTerrainTracks;
596
597 // just for paranoia's sake.
598 REF_PTR_RELEASE(m_indexBuffer);
599 REF_PTR_RELEASE(m_vertexBuffer);
600
601 //Create static index buffers. These will index the vertex buffers holding the track segments
602 m_indexBuffer=NEW_REF(DX8IndexBufferClass,((m_maxTankTrackEdges-1)*6));
603
604 // Fill up the IB
605 {
606 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
607 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
608
609 for (i=0; i<(m_maxTankTrackEdges-1); i++)
610 {
611 ib[3]=ib[0]=i*2;
612 ib[1]=i*2+1;
613 ib[4]=ib[2]=(i+1)*2+1;
614 ib[5]=(i+1)*2;
615 ib+=6; //skip the 6 indices we just filled
616 }
617 }
618
619 DEBUG_ASSERTCRASH(numModules*m_maxTankTrackEdges*2 < 65535, ("Too many terrain track edges"));
620
621 m_vertexBuffer=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,numModules*m_maxTankTrackEdges*2,DX8VertexBufferClass::USAGE_DYNAMIC));
622}

637//=============================================================================
638// TerrainTracksRenderObjClassSystem::init
639//=============================================================================
641//=============================================================================
642void TerrainTracksRenderObjClassSystem::init( SceneClass *TerrainTracksScene )
643{
644 const Int numModules=TheGlobalData->m_maxTerrainTracks;
645
646 Int i;
647 TerrainTracksRenderObjClass *mod;
648
649 m_TerrainTracksScene=TerrainTracksScene;
650
651 ReAcquireResources();
652 //go with a preset material for now.
653 m_vertexMaterialClass=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
654
655 //use a multi-texture shader: (text1*diffuse)*text2.
656 m_shaderClass = ShaderClass::_PresetAlphaShader;//_PresetATestSpriteShader;//_PresetOpaqueShader;
657
658 // we cannot initialize a system that is already initialized
659 if( m_freeModules || m_usedModules )
660 {
661
662 // system already online!
663 assert( 0 );
664 return;
665
666 } // end if
667
668 // allocate our modules for this system
669 for( i = 0; i < numModules; i++ )
670 {
671
672 mod = NEW_REF( TerrainTracksRenderObjClass, () );
673
674 if( mod == NULL )
675 {
676
677 // unable to allocate modules needed
678 assert( 0 );
679 return;
680
681 } // end if
682
683 mod->m_prevSystem = NULL;
684 mod->m_nextSystem = m_freeModules;
685 if( m_freeModules )
686 m_freeModules->m_prevSystem = mod;
687 m_freeModules = mod;
688
689 } // end for i
690
691} // end init

693//=============================================================================
694// TerrainTracksRenderObjClassSystem::shutdown
695//=============================================================================
697//=============================================================================
698void TerrainTracksRenderObjClassSystem::shutdown( void )
699{
700 TerrainTracksRenderObjClass *nextMod,*mod;
701
702 //release unbound tracks that may still be fading out
703 mod=m_usedModules;
704
705 while(mod)
706 {
707 nextMod=mod->m_nextSystem;
708
709 if (!mod->m_bound)
710 releaseTrack(mod);
711
712 mod = nextMod;
713 } // end while
714
715
716 // free all attached things and used modules
717 assert( m_usedModules == NULL );
718
719 // free all module storage
720 while( m_freeModules )
721 {
722
723 nextMod = m_freeModules->m_nextSystem;
724 REF_PTR_RELEASE (m_freeModules);
725 m_freeModules = nextMod;
726
727 } // end while
728
729 REF_PTR_RELEASE(m_indexBuffer);
730 REF_PTR_RELEASE(m_vertexMaterialClass);
731 REF_PTR_RELEASE(m_vertexBuffer);
732
733} // end shutdown

735//=============================================================================
736// TerrainTracksRenderObjClassSystem::update
737//=============================================================================
739//=============================================================================
740void TerrainTracksRenderObjClassSystem::update()
741{
742
743 Int iTime=WW3D::Get_Sync_Time();
744 Real iDiff;
745 TerrainTracksRenderObjClass *mod=m_usedModules,*nextMod;
746
747 //first update all the tracks
748 while( mod )
749 {
750 Int i,index;
751 Vector3 *endPoint;
752 Vector2 *endPointUV;
753
754 nextMod = mod->m_nextSystem;
755
756 if (!TheGlobalData->m_makeTrackMarks)
757 mod->m_haveAnchor=false; //force a track restart next time around.
758
759 for (i=0,index=mod->m_bottomIndex; i<mod->m_activeEdgeCount; i++,index++)
760 {
761 if (index >= m_maxTankTrackEdges)
762 index=0;
763
764 endPoint=&mod->m_edges[index].endPointPos[0]; //left endpoint
765 endPointUV=&mod->m_edges[index].endPointUV[0];
766 iDiff=(float)(iTime-mod->m_edges[index].timeAdded);
767 iDiff = 1.0f - iDiff/(Real)m_maxTankTrackFadeDelay;
768 if (iDiff < 0.0)
769 iDiff=0.0f;
770 if (mod->m_edges[index].alpha>0.0f) {
771 mod->m_edges[index].alpha=iDiff;
772 }
773
774 if (iDiff == 0.0f)
775 { //this edge was invisible, we can remove it
776 mod->m_bottomIndex++;
777 mod->m_activeEdgeCount--;
778
779 if (mod->m_bottomIndex >= m_maxTankTrackEdges)
780 mod->m_bottomIndex=0; //roll buffer back to start
781 }
782 if (mod->m_activeEdgeCount == 0 && !mod->m_bound)
783 releaseTrack(mod);
784 }

795void TerrainTracksRenderObjClassSystem::flush()
796{
801
802 Int diffuseLight;
803 TerrainTracksRenderObjClass *mod=m_usedModules;
804 if (!mod)
805 return; //nothing to render
806
807 Int trackStartIndex;
808 Real distanceFade;
809
810 if (ShaderClass::Is_Backface_Culling_Inverted())
811 return; //don't render track marks in reflections.
812
813 // adjust shading for time of day.
814 Real shadeR, shadeG, shadeB;
815 shadeR = TheGlobalData->m_terrainAmbient[0].red;
816 shadeG = TheGlobalData->m_terrainAmbient[0].green;
817 shadeB = TheGlobalData->m_terrainAmbient[0].blue;
818 shadeR += TheGlobalData->m_terrainDiffuse[0].red/2;
819 shadeG += TheGlobalData->m_terrainDiffuse[0].green/2;
820 shadeB += TheGlobalData->m_terrainDiffuse[0].blue/2;
821 shadeR*=255.0f;
822 shadeG*=255.0f;
823 shadeB*=255.0f;
824
825 diffuseLight = REAL_TO_INT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16);
826 Real numFadedEdges=m_maxTankTrackEdges-m_maxTankTrackOpaqueEdges;
827
828 //check if there is anything to draw and fill vertex buffer
829 if (m_edgesToFlush >= 2)
830 {
831 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(m_vertexBuffer);
832 VertexFormatXYZDUV1 *verts = (VertexFormatXYZDUV1*)lockVtxBuffer.Get_Vertex_Array();
833 trackStartIndex=0;
834
835 mod=m_usedModules;
836 //Fill our vertex buffer with all the tracks
837 while( mod )
838 {
839 Int i,index;
840 Vector3 *endPoint;
841 Vector2 *endPointUV;
842
843 if (mod->m_activeEdgeCount >= 2 && mod->Is_Really_Visible())
844 {
845 for (i=0,index=mod->m_bottomIndex; i<mod->m_activeEdgeCount; i++,index++)
846 {
847 if (index >= m_maxTankTrackEdges)
848 index=0;
849
850 endPoint=&mod->m_edges[index].endPointPos[0]; //left endpoint
851 endPointUV=&mod->m_edges[index].endPointUV[0];
852
853 distanceFade=1.0f;
854
855 if ((mod->m_activeEdgeCount -1 -i) >= m_maxTankTrackOpaqueEdges)// && i < (MAX_PER_TRACK_EDGE_COUNT-FORCE_FADE_AT_EDGE))
856 { //we're getting close to the limit on the number of track pieces allowed
857 //so force it to fade out.
858 distanceFade=1.0f-(float)((mod->m_activeEdgeCount -i)-m_maxTankTrackOpaqueEdges)/numFadedEdges;
859 }
860
861 distanceFade *= mod->m_edges[index].alpha; //adjust fade with distance from start of track
862
863 verts->x=endPoint->X;
864 verts->y=endPoint->Y;
865 verts->z=endPoint->Z;
866
867 verts->u1=endPointUV->X;
868 verts->v1=endPointUV->Y;
869
870 //fade the alpha channel with distance
871 verts->diffuse=diffuseLight | ( REAL_TO_INT(distanceFade*255.0f) <<24);
872 verts++;
873
874 endPoint=&mod->m_edges[index].endPointPos[1]; //right endpoint
875 endPointUV=&mod->m_edges[index].endPointUV[1];
876
877 verts->x=endPoint->X;
878 verts->y=endPoint->Y;
879 verts->z=endPoint->Z;
880
881 verts->u1=endPointUV->X;
882 verts->v1=endPointUV->Y;
883
884 verts->diffuse=diffuseLight | ( REAL_TO_INT(distanceFade*255.0f) <<24);
885 verts++;
886 }//for
887 }// mod has edges to render
888 mod = mod->m_nextSystem;
889 } //while (mod)
890 }//edges to flush
891
892 //draw the filled vertex buffers
893 if (m_edgesToFlush >= 2)
894 {
895 ShaderClass::Invalidate();
896 DX8Wrapper::Set_Material(m_vertexMaterialClass);
897 DX8Wrapper::Set_Shader(m_shaderClass);
898 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
899 DX8Wrapper::Set_Vertex_Buffer(m_vertexBuffer);
900
901 trackStartIndex=0;
902 mod=m_usedModules;
903 Matrix3D tm(mod->Transform);
904 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
905 while (mod)
906 {
907 if (mod->m_activeEdgeCount >= 2 && mod->Is_Really_Visible())
908 {
909 DX8Wrapper::Set_Texture(0,mod->m_stageZeroTexture);
910 DX8Wrapper::Set_Index_Buffer_Index_Offset(trackStartIndex);
911 DX8Wrapper::Draw_Triangles( 0,(mod->m_activeEdgeCount-1)*2, 0, mod->m_activeEdgeCount*2);
912
913 trackStartIndex += mod->m_activeEdgeCount*2;
914 }
915 mod=mod->m_nextSystem;
916 }
917 } //there are some edges to render in pool.
918
919 m_edgesToFlush=0; //reset count for next flush
920}
