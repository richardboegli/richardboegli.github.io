139void SoundManager::addAudioEvent(AudioEventRTS *eventToAdd)
140{
141 if (m_num2DSamples == 0 && m_num3DSamples == 0) {
142 m_num2DSamples = TheAudio->getNum2DSamples();
143 m_num3DSamples = TheAudio->getNum3DSamples();
144 }
145
146 if (canPlayNow(eventToAdd)) {
147#ifdef INTENSIVE_AUDIO_DEBUG
148 DEBUG_LOG((" - appended to request list with handle '%d'.\n", (UnsignedInt) eventToAdd->getPlayingHandle()));
149#endif
150 AudioRequest *audioRequest = TheAudio->allocateAudioRequest( true );
151 audioRequest->m_pendingEvent = eventToAdd;
152 audioRequest->m_request = AR_Play;
153 TheAudio->appendAudioRequest(audioRequest);
154 } else {
155 TheAudio->releaseAudioEventRTS(eventToAdd);
156 }
157}

206Bool SoundManager::canPlayNow( AudioEventRTS *event )
207{
208 Bool retVal = false;
209 // 1) Are we muted because we're beyond our maximum distance?
210 // 2) Are we shrouded and this is a shroud sound?
211 // 3) Are we violating our voice count or are we playing above the limit? (If so, stop now)
212 // 4) is there an avaiable channel open?
213 // 5) if not, then determine if there is anything of lower priority that we can kill
214 // 6) if not, are we an interrupt-sound type?
215 // if so, are there any sounds of our type playing right now that we can interrupt?
216 // potentially here: Are there any sounds that are playing that are now beyond their distance?
217 // if so, kill them and start our sound
218 // if not, we're done. Can't play dude.
219
220 if( event->isPositionalAudio() && !BitTest( event->getAudioEventInfo()->m_type, ST_GLOBAL) && event->getAudioEventInfo()->m_priority != AP_CRITICAL )
221 {
222 Coord3D distance = *TheAudio->getListenerPosition();
223 const Coord3D *pos = event->getCurrentPosition();
224 if (pos)
225 {
226 distance.sub(pos);
227 if (distance.length() >= event->getAudioEventInfo()->m_maxDistance)
228 {
229#ifdef INTENSIVE_AUDIO_DEBUG
230 DEBUG_LOG(("- culled due to distance (%.2f).\n", distance.length()));
231#endif
232 return false;
233 }
234
235 Int localPlayerNdx = ThePlayerList->getLocalPlayer()->getPlayerIndex();
236 if( (event->getAudioEventInfo()->m_type & ST_SHROUDED) &&
237 ThePartitionManager->getShroudStatusForPlayer(localPlayerNdx, pos) != CELLSHROUD_CLEAR )
238 {
239#ifdef INTENSIVE_AUDIO_DEBUG
240 DEBUG_LOG(("- culled due to shroud.\n"));
241#endif
242 return false;
243 }
244 }
245 }
246
247 if (violatesVoice(event))
248 {
249 retVal = isInterrupting(event);
250 if (retVal)
251 {
252 return true;
253 }
254 else
255 {
256#ifdef INTENSIVE_AUDIO_DEBUG
257 DEBUG_LOG(("- culled due to voice.\n"));
258#endif
259 return false;
260 }
261 }
262
263 if( TheAudio->doesViolateLimit( event ) )
264 {
265#ifdef INTENSIVE_AUDIO_DEBUG
266 DEBUG_LOG(("- culled due to limit.\n" ));
267#endif
268 return false;
269 }
270 else if( isInterrupting( event ) )
271 {
272 return true;
273 }
274
275 if (event->isPositionalAudio())
276 {
277 if (m_numPlaying3DSamples < m_num3DSamples)
278 {
279 return true;
280 }
281#ifdef INTENSIVE_AUDIO_DEBUG
282 DEBUG_LOG(("- %d samples playing, %d samples available", m_numPlaying3DSamples, m_num3DSamples));
283#endif
284 }
285 else
286 {
287 // its a UI sound (and thus, 2-D)
288 if (m_numPlaying2DSamples < m_num2DSamples)
289 {
290 return true;
291 }
292 }
293
294 if (TheAudio->isPlayingLowerPriority(event))
295 {
296 return true;
297 }
298
299 if (isInterrupting(event))
300 {
301 retVal = TheAudio->isPlayingAlready(event);
302 if (retVal)
303 {
304 return true;
305 }
306 else
307 {
308#ifdef INTENSIVE_AUDIO_DEBUG
309 DEBUG_LOG(("- culled due to no channels available and non-interrupting.\n" ));
310#endif
311 return false;
312 }
313 }
314#ifdef INTENSIVE_AUDIO_DEBUG
315 DEBUG_LOG(("culled due to unavailable channels"));
316#endif
317 return false;
318}

321Bool SoundManager::violatesVoice( AudioEventRTS *event )
322{
323 if (event->getAudioEventInfo()->m_type & ST_VOICE) {
324 return (event->getObjectID() && TheAudio->isObjectPlayingVoice(event->getObjectID()));
325 }
326 return false;
327}
