100{

111

132 }

134 }
135 }
136 else
137 {
138 LANGameInfo *game = LookupGame(UnicodeString(msg->GameInfo.gameName));
139 if (!game)
140 {

208void LANAPI::handleRequestJoin( LANMessage *msg, UnsignedInt senderIP )
209{
210 UnsignedInt responseIP = senderIP; // need this cause the player may or may not be
211 // in the player list at the sendMessage.
212
213 if (msg->GameToJoin.gameIP != m_localIP)
214 {
215 return; // Not us. Ignore it.
216 }
217 LANMessage reply;
218 fillInLANMessage( &reply );
219 if (!m_inLobby && m_currentGame && m_currentGame->getIP(0) == m_localIP)
220 {
221 if (m_currentGame->isGameInProgress())
222 {
223 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
224 reply.GameNotJoined.reason = LANAPIInterface::RET_GAME_STARTED;
225 reply.GameNotJoined.gameIP = m_localIP;
226 reply.GameNotJoined.playerIP = senderIP;
227 DEBUG_LOG(("LANAPI::handleRequestJoin - join denied because game already started.\n"));
228 }
229 else
230 {
231 int player;
232 Bool canJoin = true;
233
234 // see if the CRCs match
235#if defined(_DEBUG) || defined(_INTERNAL)
236 if (TheGlobalData->m_netMinPlayers > 0) {
237#endif
238/* if (msg->GameToJoin.iniCRC != TheGlobalData->m_iniCRC ||
239 msg->GameToJoin.exeCRC != TheGlobalData->m_exeCRC)
240 {
241 DEBUG_LOG(("LANAPI::handleRequestJoin - join denied because of CRC mismatch. CRCs are them/us INI:%X/%X exe:%X/%X\n",
242 msg->GameToJoin.iniCRC, TheGlobalData->m_iniCRC,
243 msg->GameToJoin.exeCRC, TheGlobalData->m_exeCRC));
244 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
245 reply.GameNotJoined.reason = LANAPIInterface::RET_CRC_MISMATCH;
246 reply.GameNotJoined.gameIP = m_localIP;
247 reply.GameNotJoined.playerIP = senderIP;
248 canJoin = false;
249 }
250*/
251#if defined(_DEBUG) || defined(_INTERNAL)
252 }
253#endif
254
255 // check for a duplicate serial
256 AsciiString s;
257 for (player = 0; canJoin && player<MAX_SLOTS; ++player)
258 {
259 LANGameSlot *slot = m_currentGame->getLANSlot(player);
260 s.clear();
261 if (player == 0)
262 {
263 GetStringFromRegistry("\\ergc", "", s);
264 }
265 else if (slot->isHuman())
266 {
267 s = slot->getSerial();
268 if (s.isEmpty())
269 s = "<Munkee>";
270 }
271
272 if (s.isNotEmpty())
273 {
274 DEBUG_LOG(("Checking serial '%s' in slot %d\n", s.str(), player));
275
276 if (!strncmp(s.str(), msg->GameToJoin.serial, g_maxSerialLength))
277 {
278 // serials match! kick the punk!
279 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
280 reply.GameNotJoined.reason = LANAPIInterface::RET_SERIAL_DUPE;
281 reply.GameNotJoined.gameIP = m_localIP;
282 reply.GameNotJoined.playerIP = senderIP;
283 canJoin = false;
284
285 DEBUG_LOG(("LANAPI::handleRequestJoin - join denied because of duplicate serial # (%s).\n", s.str()));
286 break;
287 }
288 }
289 }
290
291 // We're the host, so see if he has a duplicate name
292 for (player = 0; canJoin && player<MAX_SLOTS; ++player)
293 {
294 LANGameSlot *slot = m_currentGame->getLANSlot(player);
295 if (slot->isHuman() && slot->getName().compare(msg->name) == 0)
296 {
297 // just deny duplicates
298 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
299 reply.GameNotJoined.reason = LANAPIInterface::RET_DUPLICATE_NAME;
300 reply.GameNotJoined.gameIP = m_localIP;
301 reply.GameNotJoined.playerIP = senderIP;
302 canJoin = false;
303
304 DEBUG_LOG(("LANAPI::handleRequestJoin - join denied because of duplicate names.\n"));
305 break;
306 }
307 }
308
309 // See if there's room
310 // First get the number of players currently in the room.
311 Int numPlayers = 0;
312 for (player = 0; player < MAX_SLOTS; ++player)
313 {
314 if (m_currentGame->getLANSlot(player)->isOccupied()
315 && !(m_currentGame->getLANSlot(player)->getPlayerTemplate() == PLAYERTEMPLATE_OBSERVER))
316 {
317 ++numPlayers;
318 }
319 }
320
321 // now get the number of starting spots on the map.
322 Int numStartingSpots = MAX_SLOTS;
323 const MapMetaData *md = TheMapCache->findMap(m_currentGame->getMap());
324 if (md != NULL)
325 {
326 numStartingSpots = md->m_numPlayers;
327 }
328
329 if (numPlayers < numStartingSpots) {
330 for (player = 0; canJoin && player<MAX_SLOTS; ++player)
331 {
332 if (m_currentGame->getLANSlot(player)->isOpen())
333 {
334 // OK, add him in.
335 reply.LANMessageType = LANMessage::MSG_JOIN_ACCEPT;
336 wcsncpy(reply.GameJoined.gameName, m_currentGame->getName().str(), g_lanGameNameLength);
337 reply.GameJoined.gameName[g_lanGameNameLength] = 0;
338 reply.GameJoined.slotPosition = player;
339 reply.GameJoined.gameIP = m_localIP;
340 reply.GameJoined.playerIP = senderIP;
341
342 LANGameSlot newSlot;
343 newSlot.setState(SLOT_PLAYER, UnicodeString(msg->name));
344 newSlot.setIP(senderIP);
345 newSlot.setPort(NETWORK_BASE_PORT_NUMBER);
346 newSlot.setLastHeard(timeGetTime());
347 newSlot.setSerial(msg->GameToJoin.serial);
348 m_currentGame->setSlot(player,newSlot);
349 DEBUG_LOG(("LANAPI::handleRequestJoin - added player %ls at ip 0x%08x to the game\n", msg->name, senderIP));
350
351 OnPlayerJoin(player, UnicodeString(msg->name));
352 responseIP = 0;
353
354 break;
355 }
356 }
357 }
358 if (canJoin && player == MAX_SLOTS)
359 {
360 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
361 wcsncpy(reply.GameNotJoined.gameName, m_currentGame->getName().str(), g_lanGameNameLength);
362 reply.GameNotJoined.gameName[g_lanGameNameLength] = 0;
363 reply.GameNotJoined.reason = LANAPIInterface::RET_GAME_FULL;
364 reply.GameNotJoined.gameIP = m_localIP;
365 reply.GameNotJoined.playerIP = senderIP;
366 DEBUG_LOG(("LANAPI::handleRequestJoin - join denied because game is full.\n"));
367 }
368 }
369 }
370 else
371 {
372 reply.LANMessageType = LANMessage::MSG_JOIN_DENY;
373 reply.GameNotJoined.reason = LANAPIInterface::RET_GAME_GONE;
374 reply.GameNotJoined.gameIP = m_localIP;
375 reply.GameNotJoined.playerIP = senderIP;
376 }
377 sendMessage(&reply, responseIP);
378 RequestGameOptions(GenerateGameOptionsString(), true);
379}

381void LANAPI::handleJoinAccept( LANMessage *msg, UnsignedInt senderIP )
382{
383 if (msg->GameJoined.playerIP == m_localIP) // Is it for us?
384 {
385 if (m_pendingAction == ACT_JOIN) // Are we trying to join?
386 {
387 m_currentGame = LookupGame(UnicodeString(msg->GameJoined.gameName));
388
389 if (!m_currentGame)
390 {
391 DEBUG_ASSERTCRASH(false, ("Could not find game to join!"));
392 OnGameJoin(RET_UNKNOWN, NULL);
393 }
394 else
395 {
396 m_inLobby = false;
397 AsciiString options = GameInfoToAsciiString(m_currentGame);
398 m_currentGame->enterGame();
399 ParseAsciiStringToGameInfo(m_currentGame, options);
400
401 Int pos = msg->GameJoined.slotPosition;
402
403 LANGameSlot slot;
404 slot.setState(SLOT_PLAYER, m_name);
405 slot.setIP(m_localIP);
406 slot.setPort(NETWORK_BASE_PORT_NUMBER);
407 slot.setLastHeard(0);
408 slot.setLogin(m_userName);
409 slot.setHost(m_hostName);
410 m_currentGame->setSlot(pos, slot);
411
412 m_currentGame->getLANSlot(0)->setHost(msg->hostName);
413 m_currentGame->getLANSlot(0)->setLogin(msg->userName);
414
415 LANPreferences prefs;
416 AsciiString entry;
417 entry.format("%d.%d.%d.%d:%s", senderIP >> 24, (senderIP & 0xff0000) >> 16, (senderIP & 0xff00) >> 8, senderIP & 0xff, UnicodeStringToQuotedPrintable(m_currentGame->getSlot(0)->getName()).str());
418 prefs["RemoteIP0"] = entry;
419 prefs.write();
420
421 OnGameJoin(RET_OK, m_currentGame);
422 //DEBUG_ASSERTCRASH(false, ("setting host to %ls@%ls\n", m_currentGame->getLANSlot(0)->getUser()->getLogin().str(),
423 // m_currentGame->getLANSlot(0)->getUser()->getHost().str()));
424 }
425 m_pendingAction = ACT_NONE;
426 m_expiration = 0;
427 }
428 }

444void LANAPI::handleRequestGameLeave( LANMessage *msg, UnsignedInt senderIP )
445{
446 if (!m_inLobby && m_currentGame && !m_currentGame->isGameInProgress())
447 {
448 int player;
449 for (player = 0; player < MAX_SLOTS; ++player)
450 {
451 if (m_currentGame->getIP(player) == senderIP)
452 {
453 if (player == 0)
454 {
455 OnHostLeave();
456 removeGame(m_currentGame);
457 delete m_currentGame;
458 m_currentGame = NULL;
459
461 LANPlayer *lanPlayer = LookupPlayer(m_localIP);
462 if (!lanPlayer)
463 {
464 lanPlayer = NEW LANPlayer;
465 lanPlayer->setIP(m_localIP);
466 }
467 else
468 {
469 removePlayer(lanPlayer);
470 }
471 lanPlayer->setName(UnicodeString(m_name));
472 lanPlayer->setHost(m_hostName);
473 lanPlayer->setLogin(m_userName);
474 lanPlayer->setLastHeard(timeGetTime());
475 addPlayer(lanPlayer);
476
477 }
478 else
479 {
480 if (AmIHost())
481 {
482 // remove the deadbeat
483 LANGameSlot slot;
484 slot.setState(SLOT_OPEN);
485 m_currentGame->setSlot( player, slot );
486 }
487 OnPlayerLeave(UnicodeString(msg->name));
488 m_currentGame->getLANSlot(player)->setState(SLOT_OPEN);
489 m_currentGame->resetAccepted();
490 RequestGameOptions(GenerateGameOptionsString(), false, senderIP);
491 //m_currentGame->endGame();
492 }
493 break;
494 }
495 DEBUG_ASSERTCRASH(player < MAX_SLOTS, ("Didn't find player!"));
496 }
497 }
498 else if (m_inLobby)
499 {
500 // Look for dissappearing games
501 LANGameInfo *game = m_games;
502 while (game)
503 {
504 if (game->getName().compare(msg->GameToLeave.gameName) == 0)
505 {
506 removeGame(game);
507 delete game;
508 OnGameList(m_games);
509 break;
510 }
511 game = game->getNext();
512 }
513 }

642void LANAPI::handleInActive(LANMessage *msg, UnsignedInt senderIP) {
643 if (m_inLobby || (m_currentGame == NULL) || (m_currentGame->isGameInProgress())) {
644 return;
645 }
646
647 // check to see if we are the host of this game.
648 if (m_currentGame->amIHost() == FALSE) {
649 return;
650 }
651
652 UnicodeString playerName;
653 playerName = msg->name;
654
655 Int slotNum = m_currentGame->getSlotNum(playerName);
656 if (slotNum < 0)
657 return;
658 GameSlot *slot = m_currentGame->getSlot(slotNum);
659 if (slot == NULL) {
660 return;
661 }
662
663 if (senderIP != slot->getIP()) {
664 return;
665 }
666
667 // don't want to unaccept the host, that's silly. They can't hit start alt-tabbed anyways.
668 if (senderIP == TheLAN->GetLocalIP()) {
669 return;
670 }
671
672 // only unaccept if the timer hasn't started yet.
673 if (m_gameStartTime != 0) {
674 return;
675 }
676
677 slot->unAccept();
678 AsciiString options = GenerateGameOptionsString();
679 RequestGameOptions(options, FALSE);
680 lanUpdateSlotList();
