45File * Win32LocalFileSystem::openFile(const Char *filename, Int access /* = 0 */)
46{
47 //USE_PERF_TIMER(Win32LocalFileSystem_openFile)
48 Win32LocalFile *file = newInstance( Win32LocalFile );
49
50 // sanity check
51 if (strlen(filename) <= 0) {
52 return NULL;
53 }
54
55 if (access & File::WRITE) {
56 // if opening the file for writing, we need to make sure the directory is there
57 // before we try to create the file.
58 AsciiString string;
59 string = filename;
60 AsciiString token;
61 AsciiString dirName;
62 string.nextToken(&token, "\\/");
63 dirName = token;
64 while ((token.find('.') == NULL) || (string.find('.') != NULL)) {
65 createDirectory(dirName);
66 string.nextToken(&token, "\\/");
67 dirName.concat('\\');
68 dirName.concat(token);
69 }
70 }
71
72 if (file->open(filename, access) == FALSE) {
73 file->close();
74 file->deleteInstance();
75 file = NULL;
76 } else {
77 file->deleteOnClose();
78 }
79
80// this will also need to play nice with the STREAMING type that I added, if we ever enable this

82// srj sez: this speeds up INI loading, but makes BIG files unusable.
83// don't enable it without further tweaking.

124void Win32LocalFileSystem::getFileListInDirectory(const AsciiString& currentDirectory, const AsciiString& originalDirectory, const AsciiString& searchName, FilenameList & filenameList, Bool searchSubdirectories) const
125{
126 HANDLE fileHandle = NULL;
127 WIN32_FIND_DATA findData;
128
129 char search[_MAX_PATH];
130 AsciiString asciisearch;
131 asciisearch = originalDirectory;
132 asciisearch.concat(currentDirectory);
133 asciisearch.concat(searchName);
134 strcpy(search, asciisearch.str());
135
136 Bool done = FALSE;
137
138 fileHandle = FindFirstFile(search, &findData);
139 done = (fileHandle == INVALID_HANDLE_VALUE);
140
141 while (!done) {
142 if (!(findData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
143 (strcmp(findData.cFileName, ".") && strcmp(findData.cFileName, ".."))) {
144 // if we haven't already, add this filename to the list.
145 // a stl set should only allow one copy of each filename
146 AsciiString newFilename;
147 newFilename = originalDirectory;
148 newFilename.concat(currentDirectory);
149 newFilename.concat(findData.cFileName);
150 if (filenameList.find(newFilename) == filenameList.end()) {
151 filenameList.insert(newFilename);
152 }
153 }
154
155 done = (FindNextFile(fileHandle, &findData) == 0);
156 }
157 FindClose(fileHandle);
158
159 if (searchSubdirectories) {
160 AsciiString subdirsearch;
161 subdirsearch = originalDirectory;
162 subdirsearch.concat(currentDirectory);
163 subdirsearch.concat("*.");
164 fileHandle = FindFirstFile(subdirsearch.str(), &findData);
165 done = fileHandle == INVALID_HANDLE_VALUE;
166
167 while (!done) {
168 if ((findData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
169 (strcmp(findData.cFileName, ".") && strcmp(findData.cFileName, ".."))) {
170
171 AsciiString tempsearchstr;
172 tempsearchstr.concat(currentDirectory);
173 tempsearchstr.concat(findData.cFileName);
174 tempsearchstr.concat('\\');
175
176 // recursively add files in subdirectories if required.
177 getFileListInDirectory(tempsearchstr, originalDirectory, searchName, filenameList, searchSubdirectories);
178 }
179
180 done = (FindNextFile(fileHandle, &findData) == 0);
181 }
182
183 FindClose(fileHandle);
184 }
185
186}

188Bool Win32LocalFileSystem::getFileInfo(const AsciiString& filename, FileInfo *fileInfo) const
189{
190 WIN32_FIND_DATA findData;
191 HANDLE findHandle = NULL;
192 findHandle = FindFirstFile(filename.str(), &findData);
193
194 if (findHandle == INVALID_HANDLE_VALUE) {
195 return FALSE;
196 }
197
198 fileInfo->timestampHigh = findData.ftLastWriteTime.dwHighDateTime;
199 fileInfo->timestampLow = findData.ftLastWriteTime.dwLowDateTime;
200 fileInfo->sizeHigh = findData.nFileSizeHigh;
201 fileInfo->sizeLow = findData.nFileSizeLow;
202
203 FindClose(findHandle);
204
205 return TRUE;
206}
