98void PropagandaCenterBehavior::onDelete( void )
99{
100
101 // extend functionality
102 PrisonBehavior::onDelete();
103
104 //
105 // go through our list of brainwashed objects, and if they are still under our
106 // control, return them to their original owners
107 //
108 for( BrainwashedIDListContIterator it = m_brainwashedList.begin();
109 it != m_brainwashedList.end();
110 ++it )
111 {
112 Object *obj;
113
114 // get this object
115 obj = TheGameLogic->findObjectByID( *it );
116 if( obj )
117 {
118
119 // return this object under the control of the original owner
120 obj->restoreOriginalTeam();
121
122 } // end if
123
124 } // end for
125
126 // clear the list
127 m_brainwashedList.clear();
128
129} // end onDelete

133UpdateSleepTime PropagandaCenterBehavior::update( void )
134{
135 Object *us = getObject();
136 const PropagandaCenterBehaviorModuleData *modData = getPropagandaCenterBehaviorModuleData();
137
138 // extend functionality
139 PrisonBehavior::update();
140
141 // if we have a prisoner inside, continue the brainwashing on them (one at a time)
142 if( m_brainwashingSubjectID != INVALID_ID )
143 {
144 Object *brainwashingSubject = TheGameLogic->findObjectByID( m_brainwashingSubjectID );
145
146 if( brainwashingSubject )
147 {
148
149 // if we've been in here long enough, we come out brainwashed
150 if( TheGameLogic->getFrame() - m_brainwashingSubjectStartFrame >= modData->m_brainwashDuration )
151 {
152
153 // only can exit if the prison allows us to
154 ExitDoorType exitDoor = reserveDoorForExit(brainwashingSubject->getTemplate(), brainwashingSubject);
155 if(exitDoor != DOOR_NONE_AVAILABLE)
156 {
157
158 // place this object under the control of the player
159 Player *player = us->getControllingPlayer();
160 DEBUG_ASSERTCRASH( player, ("Brainwashing: No controlling player for '%s'\n", us->getTemplate()->getName().str()) );
161 if( player )
162 brainwashingSubject->setTemporaryTeam( player->getDefaultTeam() );
163
164 // remove any surrender status from this object
165 AIUpdateInterface *ai = brainwashingSubject->getAIUpdateInterface();
166 if( ai )
167 ai->setSurrendered( NULL, FALSE );
168
169 // add this object to our brainwashed list if we're not already in it
170 for( BrainwashedIDListIterator it = m_brainwashedList.begin();
171 it != m_brainwashedList.end(); ++it )
172 if( *it == brainwashingSubject->getID() )
173 break; // exit for
174
175 if( it == m_brainwashedList.end() )
176 m_brainwashedList.push_front( brainwashingSubject->getID() );
177
178 // exit the prison
179 exitObjectViaDoor( brainwashingSubject, exitDoor );
180
181 } // end if
182
183 } // end if
184
185 } // end if,
186
187 } // end if
188
189 // if we have no brainwashing subject, hook one up if we have people inside us
190 if( m_brainwashingSubjectID == INVALID_ID )
191 {
192
193 // find the first object in our containment list
194 if( getContainList().begin() != getContainList().end() )
195 {
196 Object *obj = getContainList().front();
197
198 if( obj )
199 {
200
201 // assign brainwashing subject on this frame
202 m_brainwashingSubjectID = obj->getID();
203 m_brainwashingSubjectStartFrame = TheGameLogic->getFrame();
204
205 } // end if
206
207 } // end if
208
209 } // end if
210
211 return UPDATE_SLEEP_NONE;
212} // end update
