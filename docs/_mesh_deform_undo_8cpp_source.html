87//
88// Restore
89//
91void
92VertexRestoreClass::Restore (int is_undo)
93{
94 assert (m_pMesh != NULL);
95 assert (m_pModData != NULL);
96 assert (m_pModifier != NULL);
97
98 // Is this being called as part of an undo operation?
99 if (is_undo != 0) {
100
101 //
102 // Ensure the modifier is in the state it was when
103 // the undo operation was recorded
104 //
105 m_pModData->Set_Current_Set (m_SetIndex);
106 m_pModData->Peek_Set (m_SetIndex).Set_Current_Key_Frame (m_KeyframeIndex);
107
108 //
109 // Apply the original vertex positions to the mesh
110 //
111 Apply_Vertex_Data (m_VertexList);
112
113 //
114 // Notify the mesh of geometry changes
115 //
116 m_pModifier->NotifyDependents (FOREVER, PART_GEOM | PART_VERTCOLOR, REFMSG_CHANGE);
117 m_pModifier->Update_UI (m_pModData);
118 }
119
120 return ;
121}

210//
211// Copy_Vertex_State
212//
214void
215VertexPositionRestoreClass::Copy_Vertex_State (DEFORM_LIST &list)
216{
217 //
218 // Make a copy of each vertex in the current keyframe
219 //
220 list.Delete_All ();
221 MeshDeformSetClass &set_obj = m_pModData->Peek_Set (m_SetIndex);
222 int count = set_obj.Get_Vertex_Count (m_KeyframeIndex);
223 for (int index = 0; index < count; index ++) {
224 const VERT_INFO &data = set_obj.Get_Vertex_Data (m_KeyframeIndex, index);
225 list.Add (VERT_INFO (data.index, data.value));
226 }
227
228 return ;
229}
