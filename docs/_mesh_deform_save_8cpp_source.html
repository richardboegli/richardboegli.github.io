54MeshDeformSaveClass::Initialize
55(
56 MeshBuilderClass &builder,
57 Object * object,
58 Mesh & mesh,
59 Matrix3 * transform
60)
61{
62 // Start fresh
63 Reset ();
64
65 //
66 // Attempt to gain access to the IDerivedObject this node references
67 //
68 int test = object->SuperClassID ();
69 int test2 = GEN_DERIVOB_CLASS_ID;
70 if ((object != NULL) &&
71 (object->SuperClassID () == GEN_DERIVOB_CLASS_ID)) {
72
73 //
74 // Loop through all the modifiers and see if we can find the
75 // Westwood Damage Mesh modifier.
76 //
77 IDerivedObject *derived_object = static_cast<IDerivedObject *> (object);
78 int modifier_count = derived_object->NumModifiers ();
79 bool found = false;
80 for (int index = 0; (index < modifier_count) && !found; index ++) {
81
82 //
83 // If this is the right modifier, then initialize using the
84 // data it contains.
85 //
86 Modifier *modifier = derived_object->GetModifier (index);
87 if ((modifier != NULL) && (modifier->ClassID () == _MeshDeformClassID)) {
88
89 //
90 // Attempt to get at the modifier data for this context
91 //
92 ModContext *mod_context = derived_object->GetModContext (index);
93 if ((mod_context != NULL) && (mod_context->localData != NULL)) {
94 MeshDeformModData *mod_data = static_cast<MeshDeformModData *> (mod_context->localData);
95 Initialize (builder, mesh, *mod_data, transform);
96 }
97
98 // Found it!
99 found = true;
100 }
101 }
102 }
103
104 return ;
105}

116 MeshBuilderClass & builder,
117 Mesh & mesh,
118 MeshDeformModData & mod_data,
119 Matrix3 * transform
120)
121{
122 //
123 // Loop through all the sets in the modifier
124 //
125 for (int index = 0; index < mod_data.Get_Set_Count (); index ++) {
126
127 //
128 // If this set isn't empty then add its data to our list
129 //
130 MeshDeformSetClass &deform_set = mod_data.Peek_Set (index);
131 if (deform_set.Is_Empty () == false) {
132
133 //
134 // Add this set to our list
135 //
136 MeshDeformSaveSetClass *save_set = new MeshDeformSaveSetClass;
137 deform_set.Save (builder, mesh, *save_set, transform);
138 m_DeformSets.Add (save_set);
139 }
140 }
141
142 return ;

248//
249// Export_Keyframes
250//
252bool
253MeshDeformSaveClass::Export_Keyframes
254(
255 ChunkSaveClass & chunk_save,
256 MeshDeformSaveSetClass &set_save
257)
258{
259 bool retval = true;
260
261 //
262 // Loop through all the keyframes in the set
263 //
264 int count = set_save.Get_Keyframe_Count ();
265 for (int keyframe_index = 0; (keyframe_index < count) && retval; keyframe_index ++) {
266
267 //
268 // Write a chunk of information out for this keyframe
269 //
270 retval &= chunk_save.Begin_Chunk (W3D_CHUNK_DEFORM_KEYFRAME);
271 if (retval) {
272 W3dDeformKeyframeInfo keyframe_info = { 0 };
273 keyframe_info.DeformPercent = set_save.Get_Deform_State (keyframe_index);
274 keyframe_info.DataCount = set_save.Get_Deform_Data_Count (keyframe_index);
275
276 retval &= (chunk_save.Write (&keyframe_info, sizeof (keyframe_info)) == sizeof (keyframe_info));
277 if (retval) {
278
279 //
280 // Loop through all the verticies in this keyframe
281 //
282 int data_count = set_save.Get_Deform_Data_Count (keyframe_index);
283 for (int index = 0; (index < data_count) && retval; index ++) {
284 MeshDeformSaveSetClass::DEFORM_DATA &data = set_save.Get_Deform_Data (keyframe_index, index);
285
286 //
287 // Write a chunk of information out for this vertex
288 //
289 retval &= chunk_save.Begin_Chunk (W3D_CHUNK_DEFORM_DATA);
290 if (retval) {
291 W3dDeformData data_struct = { 0 };
292 data_struct.VertexIndex = data.vert_index;
293 data_struct.Position.X = data.position.x;
294 data_struct.Position.Y = data.position.y;
295 data_struct.Position.Z = data.position.z;
296 data_struct.Color.R = data.color.x * 255;
297 data_struct.Color.G = data.color.y * 255;
298 data_struct.Color.B = data.color.z * 255;
299
300 // If we are using vertex alpha instead of vertex color, then convert
301 // the v-color into an alpha setting
302 data_struct.Color.A = 255;
303 if (m_AlphaPasses != 0) {
304 data_struct.Color.A = (data_struct.Color.R + data_struct.Color.G + data_struct.Color.B) / 3.0F;
305 }
306
307 retval &= (chunk_save.Write (&data_struct, sizeof (data_struct)) == sizeof (data_struct));
308 retval &= chunk_save.End_Chunk ();
309 }
310 }
311 }
312
313 retval &= chunk_save.End_Chunk ();
314 }
315 }
316
317 // Return the true/false result code
318 return retval;
319}
