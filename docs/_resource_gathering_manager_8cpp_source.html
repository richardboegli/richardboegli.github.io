108static Real computeRelativeCost( Object *queryObject, Object *destObject, Real *pureDistanceSquared )
109{
113
114 //A good score is a very small number.
115
116 if( queryObject == NULL || destObject == NULL )
117 return FLT_MAX;
118
119 if( !TheActionManager->canTransferSuppliesAt(queryObject, destObject) )
120 return FLT_MAX;// Handles emptyness and alliances
121
122 DockUpdateInterface *dockInterface = destObject->getDockUpdateInterface();
123 if( !dockInterface->isClearToApproach( queryObject ) )
124 return FLT_MAX;
125
126 // since we don't care about the distance as a distance per se, but rather as
127 // a goodness-factor, save some time by getting the dist-sqr (srj)
128 Real distSquared = ThePartitionManager->getDistanceSquared(queryObject, destObject, FROM_CENTER_3D);
129
130 // I need the distance, but I don't want to count on the coincidence that
131 // the abstract 'cost' this function returns happens to be just the distance, since it could
132 // become more complicated
133 if( pureDistanceSquared )
134 *pureDistanceSquared = distSquared;
135
136 return distSquared;
137}

139Object *ResourceGatheringManager::findBestSupplyWarehouse( Object *queryObject )
140{
141 Object *bestWarehouse = NULL;
142 Real maxDistanceSquared = 100000;
143
144 if( ( queryObject == NULL ) || ( queryObject->getAI() == NULL ) )
145 return NULL;
146
147 SupplyTruckAIInterface *supplyTruckAI = queryObject->getAI()->getSupplyTruckAIInterface();
148 if( supplyTruckAI )
149 {
150 // Check for a dock override being set.
151 ObjectID dockID = supplyTruckAI->getPreferredDockID();
152 Object *dock = TheGameLogic->findObjectByID(dockID);
153 if( dock )
154 {
155 static const NameKeyType key_warehouseUpdate = NAMEKEY("SupplyWarehouseDockUpdate");
156 SupplyWarehouseDockUpdate *warehouseModule = (SupplyWarehouseDockUpdate*)dock->findUpdateModule( key_warehouseUpdate );
157 //If remotely okay, let User win.
158 if( warehouseModule && computeRelativeCost( queryObject, dock, NULL ) != FLT_MAX )
159 return dock;
160 }
161 // Please note, there is not a separate Warehouse and Center memory by Design. Because
162 // we lack a UI way to click Warehouse and drag to center to set up a specific path, the
163 // practical realization has been made that you do not want separate memory.
164
165 // Design wants a harvester to give up and return to base if it is "too far" to the warehouse.
166 // Note, the "PreferedDock" will override this, and there is no distance max on Centers.
167 maxDistanceSquared = supplyTruckAI->getWarehouseScanDistance() * supplyTruckAI->getWarehouseScanDistance();
168 }
169
170 //Otherwise, search for a good one.
171 Real bestCost = FLT_MAX;
172
173 objectIDListIterator iterator = m_supplyWarehouses.begin();
174 while( iterator != m_supplyWarehouses.end() )
175 {
176 ObjectID currentID = *iterator;
177 Object *currentWarehouse =TheGameLogic->findObjectByID(currentID);
178
179 if( currentWarehouse == NULL )
180 {
181 iterator = m_supplyWarehouses.erase( iterator );
182 }
183 else
184 {
185 Real distanceSquared;
186 Real currentCost = computeRelativeCost( queryObject, currentWarehouse, &distanceSquared );
187 if( (currentCost < bestCost) && (distanceSquared < maxDistanceSquared) )
188 {
189 bestWarehouse = currentWarehouse;
190 bestCost = currentCost;
191 }
192
193 iterator++;
194 }
195 }
196
197 return bestWarehouse;
198}

200Object *ResourceGatheringManager::findBestSupplyCenter( Object *queryObject )
201{

203

218 if( centerModule && computeRelativeCost( queryObject, dock, NULL ) != FLT_MAX )
219 return dock;
220 }
221 // Please note, there is not a separate Warehouse and Center memory by Design. Because
222 // we lack a UI way to click Warehouse and drag to center to set up a specific path, the
223 // practical realization has been made that you do not want separate memory.
224 }
225
226 //Otherwise, search for a good one.
227 Real bestCost = FLT_MAX;
228
229 objectIDListIterator iterator = m_supplyCenters.begin();
230 while( iterator != m_supplyCenters.end() )
231 {
232 ObjectID currentID = *iterator;
