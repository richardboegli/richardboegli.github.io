95void WaterTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
96{
97 if (m != TRACK_L) return;
98 if (WaterOptions::getCreatingWaterAreas()) {
99 fillTheArea(m, viewPt, pView, pDoc);
100 return;
101 }
102 Coord3D docPt;
103 pView->viewToDocCoords(viewPt, &docPt);
104 m_poly_unsnappedMouseDownPt = docPt;
105 poly_pickOnMouseDown(viewPt, pView);
106 if (!poly_snapToPoly(&docPt)) {
107 pView->snapPoint(&docPt);
108 }
109 docPt.z = m_currentZ;
110 m_poly_mouseDownPt = docPt;
111 if (m_poly_curSelectedPolygon) {
112 DEBUG_ASSERTCRASH(m_poly_curSelectedPolygon->isWaterArea(), ("Should be water."));
113 m_currentZ = m_poly_curSelectedPolygon->getPoint(0)->z;
114 m_poly_mouseDownPt.z = m_currentZ;
115 }
116 startMouseDown(m, viewPt, pView, pDoc);
117 m_poly_curSelectedPolygon->setWaterArea(true);
118
119}

144void WaterTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
145{
146 Coord3D docPt;
147 pView->viewToDocCoords(viewPt, &docPt);
148 if (m == TRACK_NONE) {
149 PolygonTrigger *pCur = m_poly_curSelectedPolygon;
150 Int curPt = m_poly_dragPointNdx;
151 m_poly_unsnappedMouseDownPt = docPt;
152 poly_pickOnMouseDown(viewPt, pView);
153 m_poly_mouseUpPlus = false;
154 m_poly_mouseUpMove = false;
155 if (m_poly_curSelectedPolygon) {
156 if (poly_pickPoint(m_poly_curSelectedPolygon, viewPt, pView) >= 0) {
157 m_poly_mouseUpMove = true;
158 } else {
159 m_poly_mouseUpPlus = true;
160 }
161 }
162 m_poly_curSelectedPolygon = pCur;
163 m_poly_dragPointNdx = curPt;
164 return; // setCursor will use the value of m_mouseUpRotate. jba.
165 }
166
167 if (m != TRACK_L) return;
168 if (!poly_snapToPoly(&docPt)) {
169 pView->snapPoint(&docPt);
170 }
171 if (m_poly_moveUndoable) {
172 ICoord3D iDocPt;
173 iDocPt.x = floor(docPt.x+0.5f-m_poly_mouseDownPt.x);
174 iDocPt.y = floor(docPt.y+0.5f-m_poly_mouseDownPt.y);
175 iDocPt.z = 0;
176 m_poly_moveUndoable->SetOffset(iDocPt);
177 pView->Invalidate();
178 return;
179 }
180 if (m_poly_dragPointNdx >= 0 && m_poly_curSelectedPolygon) {
181 ICoord3D iDocPt;
182 iDocPt.x = floor(docPt.x+0.5f);
183 iDocPt.y = floor(docPt.y+0.5f);
184 iDocPt.z = m_currentZ;
185 m_poly_curSelectedPolygon->setPoint(iDocPt, m_poly_dragPointNdx);
186 pView->Invalidate();
187 }

202void WaterTool::fillTheArea(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
203{
204 Int waterHeight = m_currentZ;
205
206 Coord3D docPt;
207 pView->viewToDocCoords(viewPt, &docPt);
208
209 Int pointLimit = 3000; // In case we get lost.
210
211 PolygonTrigger *pNew = newInstance(PolygonTrigger)(100);
212 AsciiString name;
213 name.format("Water Area %d", pNew->getID());
214 pNew->setTriggerName(name);
215
216 CPoint ndx;
217 Int width = 1;
218 getCenterIndex(&docPt, width, &ndx, pDoc);
219 Int i, j;
220 i = ndx.x;
221 j = ndx.y;
222
223 WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
224 if (i<0 || i>=pMap->getXExtent()-1) return;
225 if (j<0 || j>=pMap->getYExtent()-1) return;
226
227 Int intMapHeight = pMap->getHeight(i, j);
228 if (waterHeight<mapZtoHeight(intMapHeight)) {
229 return;
230 }
231
232 // Move left till we find an edge.
233 while (i>0) {
234 i--;
235 intMapHeight = pMap->getHeight(i, j);
236 if (waterHeight<mapZtoHeight(intMapHeight)) {
237 break;
238 }
239 }
240
241 Bool bottom = true; // bottom left is dry, bottom right wet.
242 Bool right = false; // right top is wet, right bottom dry.
243 Bool left = false; // left bottom wet, left top dry.
244 Bool top = false; // top left wet, top right dry.
245
246 Int startI = -1;
247 Int startJ = -1;
248 Bool firstTime = true;
249
250 // Build area in index coords for now, convert later.
251 ICoord3D pt;
252 ICoord3D tmpPt;
253 Int curPoint = 0;
254 while (pointLimit>0) {
255 if (curPoint>2 && i==startI && j==startJ) break;
256 pointLimit--;
257 pt.x = i*MAP_XY_FACTOR;
258 pt.y = j*MAP_XY_FACTOR;
259 pt.z = waterHeight;
260 intMapHeight = pMap->getHeight(i, j);
261#ifdef INTENSE_DEBUG
262 if (bottom) {
263 DEBUG_LOG(("Bottom %d,%d\n", i, j));
264 } else if (left) {
265 DEBUG_LOG(("Left %d,%d\n", i, j));
266 } else if (right) {
267 DEBUG_LOG(("Right %d,%d\n", i, j));
268 } else if (top) {
269 DEBUG_LOG(("Top %d,%d\n", i, j));
270 }
271#endif
272 if (bottom) {
273 bottom = false;
274 if (waterHeight<mapZtoHeight(intMapHeight)) {
275 DEBUG_ASSERTCRASH(waterHeight>mapZtoHeight(pMap->getHeight(i+1, j)), ("Logic error. jba."));
276 Real dx = (mapZtoHeight(pMap->getHeight(i,j))-waterHeight) / (mapZtoHeight(pMap->getHeight(i,j)-pMap->getHeight(i+1,j)));
277 pt.x += dx*MAP_XY_FACTOR;
278 Bool topLeftDry = (waterHeight<mapZtoHeight(pMap->getHeight(i, j+1)));
279 Bool topRightDry = (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j+1)));
280 if (topRightDry) {
281 i++;
282 left = true;
283 } else if (topLeftDry) {
284 if (j<pMap->getYExtent()-2) {
285 j++;
286 bottom = true;
287 } else {
288 tmpPt = pt;
289 pt.y += MAP_XY_FACTOR;
290 tmpPt.x -= pMap->getBorderSize()*MAP_XY_FACTOR;
291 tmpPt.y -= pMap->getBorderSize()*MAP_XY_FACTOR;
292 pNew->addPoint(tmpPt);
293 left = true;
294 i++;
295 }
296 } else {
297 if (i>0) {
298 i--;
299 right = true;
300 } else {
301 j++;
302 bottom = true;
303 }
304 }
305 } else {
306 DEBUG_ASSERTCRASH(i==0, ("Logic error. jba."));
307 left = true;
308 while (j<pMap->getYExtent()-2 && waterHeight>mapZtoHeight(pMap->getHeight(i, j+1))) {
309 j++;
310 }
311 pt.y = (j)*MAP_XY_FACTOR;
312 if (waterHeight>mapZtoHeight(pMap->getHeight(i, j+1))) {
313 pt.y = (j+1)*MAP_XY_FACTOR;
314 } else {
315 continue;
316 }
317
318 }
319 } else if (left) {
320 left = false;
321 if (waterHeight<mapZtoHeight(pMap->getHeight(i, j+1))) {
322 DEBUG_ASSERTCRASH(waterHeight>mapZtoHeight(intMapHeight), ("Logic error. jba."));
323 Real dy = (waterHeight-mapZtoHeight(pMap->getHeight(i,j))) / (mapZtoHeight(pMap->getHeight(i,j+1)-pMap->getHeight(i,j)));
324 pt.y += dy*MAP_XY_FACTOR;
325 Bool bottomRightDry = (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j)));
326 Bool topRightDry = (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j+1)));
327 if (bottomRightDry) {
328 j--;
329 top = true;
330 } else if (topRightDry) {
331 if (i<pMap->getXExtent()-2) {
332 i++;
333 left = true;
334 } else {
335 tmpPt = pt;
336 pt.x += MAP_XY_FACTOR;
337 tmpPt.x -= pMap->getBorderSize()*MAP_XY_FACTOR;
338 tmpPt.y -= pMap->getBorderSize()*MAP_XY_FACTOR;
339 pNew->addPoint(tmpPt);
340 top = true;
341 j--;
342 }
343 } else {
344 if (j<pMap->getYExtent()-1) {
345 j++;
346 bottom = true;
347 } else {
348 i++;
349 left = true;
350 }
351 }
352 } else {
353 pt.y = (j+1)*MAP_XY_FACTOR;
354 DEBUG_ASSERTCRASH(j==pMap->getYExtent()-2, ("Logic error. jba."));
355 while (i<pMap->getXExtent()-2 && waterHeight>mapZtoHeight(pMap->getHeight(i+1, j+1))) {
356 i++;
357 }
358 top = true;
359 pt.x = i*MAP_XY_FACTOR;
360 if (waterHeight>mapZtoHeight(pMap->getHeight(i+1, j+1))) {
361 pt.x = (i+1)*MAP_XY_FACTOR;
362 } else {
363 continue;
364 }
365 }
366 } else if (right) {
367 right = false;
368 pt.x = (i+1)*MAP_XY_FACTOR;
369 if (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j))) {
370 DEBUG_ASSERTCRASH(waterHeight>mapZtoHeight(pMap->getHeight(i+1,j+1)), ("Logic error. jba."));
371 Real dy = (mapZtoHeight(pMap->getHeight(i+1,j))-waterHeight) / (mapZtoHeight(pMap->getHeight(i+1,j)-pMap->getHeight(i+1,j+1)));
372 pt.y += dy*MAP_XY_FACTOR;
373 Bool bottomLeftDry = (waterHeight<mapZtoHeight(pMap->getHeight(i, j)));
374 Bool topLeftDry = (waterHeight<mapZtoHeight(pMap->getHeight(i, j+1)));
375 if (topLeftDry) {
376 j++;
377 bottom = true;
378 } else if (bottomLeftDry) {
379 if (i>0) {
380 i--;
381 right = true;
382 } else {
383 tmpPt = pt;
384 pt.x -= MAP_XY_FACTOR;
385 tmpPt.x -= pMap->getBorderSize()*MAP_XY_FACTOR;
386 tmpPt.y -= pMap->getBorderSize()*MAP_XY_FACTOR;
387 pNew->addPoint(tmpPt);
388 bottom = true;
389 j++;
390 }
391 } else {
392 if (j>0) {
393 j--;
394 top = true;
395 } else {
396 i--;
397 right = true;
398 }
399 }
400 } else {
401 DEBUG_ASSERTCRASH(j==0, ("Logic error. jba."));
402 while (i>0 && waterHeight>mapZtoHeight(pMap->getHeight(i, j))) {
403 i--;
404 }
405 bottom = true;
406 pt.x = (i)*MAP_XY_FACTOR;
407 if (waterHeight<mapZtoHeight(pMap->getHeight(i, j))) {
408 continue;
409 }
410 }
411 } else if (top) {
412 top = false;
413 pt.y = (j+1)*MAP_XY_FACTOR;
414 if (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j+1))) {
415 DEBUG_ASSERTCRASH(waterHeight>mapZtoHeight(pMap->getHeight(i, j+1)), ("Logic error. jba."));
416 Real dx = (waterHeight-mapZtoHeight(pMap->getHeight(i,j+1))) / (mapZtoHeight(pMap->getHeight(i+1,j+1)-pMap->getHeight(i,j+1)));
417 pt.x += dx*MAP_XY_FACTOR;
418 Bool bottomLeftDry = (waterHeight<mapZtoHeight(pMap->getHeight(i, j)));
419 Bool bottomRightDry = (waterHeight<mapZtoHeight(pMap->getHeight(i+1, j)));
420 if (bottomLeftDry) {
421 i--;
422 right = true;
423 } else if (bottomRightDry) {
424 if (j>0) {
425 j--;
426 top = true;
427 } else {
428 tmpPt = pt;
429 pt.y -= MAP_XY_FACTOR;
430 tmpPt.x -= pMap->getBorderSize()*MAP_XY_FACTOR;
431 tmpPt.y -= pMap->getBorderSize()*MAP_XY_FACTOR;
432 pNew->addPoint(tmpPt);
433 i--;
434 right = true;
435 }
436 } else {
437 if (i<pMap->getXExtent()-2) {
438 i++;
439 left = true;
440 } else {
441 j--;
442 top = true;
443 }
444 }
445 } else {
446 DEBUG_ASSERTCRASH(i==pMap->getXExtent()-2, ("Logic error. jba."));
447 while (j>0 && waterHeight>mapZtoHeight(pMap->getHeight(i+1, j))) {
448 j--;
449 }
450 right = true;
451 pt.y = (j+1)*MAP_XY_FACTOR;
452 if (waterHeight>mapZtoHeight(pMap->getHeight(i, j))) {
453 pt.y = (j)*MAP_XY_FACTOR;
454 } else {
455 continue;
456 }
457 pt.x = (i+1)*MAP_XY_FACTOR;
458 }
459 } else {
460 DEBUG_CRASH(("Logic error. jba.")); // shouldn't get here.
461 }
462 pt.x -= pMap->getBorderSize()*MAP_XY_FACTOR;
463 pt.y -= pMap->getBorderSize()*MAP_XY_FACTOR;
464 pNew->addPoint(pt);
465 curPoint++;
466 if (firstTime) {
467 startI = i;
468 startJ = j;
469 firstTime = false;
470 }
471
472 }
473
474 if (pNew->getNumPoints()>2) {
475 PolygonTrigger *pBetter = adjustSpacing(pNew, WaterOptions::getSpacing());
476 pNew->deleteInstance();
477 pNew = pBetter;
478 pNew->setWaterArea(true);
479 AddPolygonUndoable *pUndo = new AddPolygonUndoable(pNew);
480 pDoc->AddAndDoUndoable(pUndo);
481 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
482 m_poly_curSelectedPolygon = pNew;
483 m_poly_dragPointNdx = -1;
484 WaterOptions::update();
485 } else {
486 pNew->deleteInstance();
487 }

492PolygonTrigger * WaterTool::adjustSpacing(PolygonTrigger *trigger, Real spacing)
493{
494 PolygonTrigger *pNew = newInstance(PolygonTrigger)(trigger->getNumPoints());
495// Real endLen=0;
496 Int i;
497 Real curSpacingLen = spacing;
498 ICoord3D pt = *trigger->getPoint(0);
499 pNew->addPoint(pt);
500 Bool didCurPoint = true;
501 for (i=0; i<trigger->getNumPoints()-1; i++) {
502 ICoord3D curPt = *trigger->getPoint(i);
503 ICoord3D nextPt = *trigger->getPoint(i+1);
504 Real dx = nextPt.x-curPt.x;
505 Real dy = nextPt.y-curPt.y;
506 Real curLen = sqrt(dx*dx+dy*dy);
507 if (curLen > 4*MAP_XY_FACTOR && curLen>2*spacing) {
508 if (!didCurPoint) pNew->addPoint(curPt);
509 else pNew->setPoint(curPt, pNew->getNumPoints()-1);
510 pNew->addPoint(nextPt);
511 didCurPoint = true;
512 curSpacingLen = spacing;
513 } else if (curSpacingLen>curLen) {
514 curSpacingLen -= curLen;
515 } else {
516 while (curLen >= curSpacingLen) {
517 // cur len > curSpacingLen.
518 Real factor = curSpacingLen/curLen;
519 curPt.x += dx*factor;
520 curPt.y += dy*factor;
521 pNew->addPoint(curPt);
522 didCurPoint = false;
523 dx = nextPt.x-curPt.x;
524 dy = nextPt.y-curPt.y;
525 curLen -= curSpacingLen;
526 curSpacingLen = spacing;
527 }
528 curSpacingLen -= curLen;
529 if ((curLen)<MAP_XY_FACTOR/2) {
530 didCurPoint = true;
531 }
532 }
533 }
534 pNew->setTriggerName(trigger->getTriggerName());
535 return pNew;
