76static void GadgetTabControlUpdatePaneNames( GameWindow *tabControl )
77{
78 TabControlData *tabData = (TabControlData *)tabControl->winGetUserData();
79 WinInstanceData *controlInstData = tabControl->winGetInstanceData();
80
81 for( Int paneIndex = 0; paneIndex < NUM_TAB_PANES; paneIndex++ )
82 {
83 WinInstanceData *paneInstData = tabData->subPanes[paneIndex]->winGetInstanceData();
84
85 char buffer[128];//legal limit is 64, which will be checked at save.
86 sprintf( buffer, "%s Pane %d", controlInstData->m_decoratedNameString.str() ,paneIndex );
87 paneInstData->m_decoratedNameString = buffer;
88 if( TheHierarchyView )
89 TheHierarchyView->updateWindowName( tabData->subPanes[paneIndex] );
90 }
91}
