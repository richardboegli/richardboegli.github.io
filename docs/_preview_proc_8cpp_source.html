69//=============================================================================
70LRESULT CALLBACK PreviewProc( HWND hWnd, UINT message,
71 WPARAM wParam, LPARAM lParam )
72{
73
74 switch( message )
75 {
76
77 case WM_PAINT:
78 {
79 PAINTSTRUCT ps;
80 HDC hdc;
81 HBRUSH whiteBrush = (HBRUSH)GetStockObject( WHITE_BRUSH );
82
83 hdc = BeginPaint(hWnd, &ps);
84
85 SelectObject( hdc, GetStockObject( WHITE_PEN ) );
86
87 // find the target texture page
88 TexturePage *page;
89 for( page = TheImagePacker->getFirstTexturePage();
90 page;
91 page = page->m_next )
92 {
93
94 // if this is not the target page, ignore it
95 if( page->getID() != TheImagePacker->getTargetPreviewPage() )
96 continue;
97
98 // draw image based on the generated texture or silhouette
99 if( TheImagePacker->getUseTexturePreview() )
100 {
101 Int x, y;
102 Byte r, g, b;
103 HPEN prevPen, pen;
104
105 for( y = 0; y < page->getHeight(); y++ )
106 {
107
108 for( x = 0; x < page->getWidth(); x++ )
109 {
110
111 // get the color here
112 page->getPixel( x, y, &r, &g, &b );
113
114 // create a new pen of the right color
115 pen = CreatePen( 1, 1, RGB( r, g, b, ) );
116
117 // select pen into hdc
118 prevPen = (HPEN)SelectObject( hdc, pen );
119
120 // draw ... what is the Win32 put pixel function???
121 MoveToEx( hdc, x, y, NULL );
122 LineTo( hdc, x + 1, y );
123
124 // put the old pen back
125 SelectObject( hdc, prevPen );
126
127 // delete the created pen
128 DeleteObject( pen );
129
130 } // end for x
131
132 } // end for y
133
134 } // end if
135 else
136 {
137
138 // go through all the images on this page
139 ImageInfo *image;
140 for( image = page->getFirstImage();
141 image;
142 image = image->m_nextPageImage )
143 {
144 RECT rect;
145
146 rect.left = image->m_pagePos.lo.x;
147 rect.top = image->m_pagePos.lo.y;
148 rect.right = image->m_pagePos.hi.x + 1; // FillRect not inclusive
149 rect.bottom = image->m_pagePos.hi.y + 1; // FillRect not inclusive
150 FillRect( hdc, &rect, whiteBrush );
151
152 } // end for image
153
154 } // end else
155
156 } // end for page
157
158 EndPaint( hWnd, &ps );
159 break;
160
161 } // end paint
162
163 } // end switch
164
165 return DefWindowProc( hWnd, message, wParam, lParam );
166
167} // end PreviewProc

218// UpdatePreviewWindow ========================================================
220//=============================================================================
221void UpdatePreviewWindow( void )
222{
223 HWND preview;
224
225 // sanity
226 if( TheImagePacker == NULL )
227 return;
228
229 // get preview window
230 preview = TheImagePacker->getPreviewWindow();
231
232 // if window not here don't bother
233 if( preview == NULL )
234 return;
235
236 // make the title
237 char title[ 256 ];
238
239 // construct title
240 sprintf( title, "Page #%d of %d",
241 TheImagePacker->getTargetPreviewPage(),
242 TheImagePacker->getPageCount() );
243 SetWindowText( preview, title );
244
245 // adjust the window rect so the client area is the target packed size
246 RECT clientRect;
247 POINT p;
248
249 p.x = 0;
250 p.y = 0;
251 ClientToScreen( preview, &p );
252 clientRect.left = p.x;
253 clientRect.right = clientRect.left + TheImagePacker->getTargetWidth();
254 clientRect.top = p.y;
255 clientRect.bottom = clientRect.top + TheImagePacker->getTargetHeight();
256 AdjustWindowRect( &clientRect, PREVIEW_STYLE, FALSE );
257 MoveWindow( preview,
258 clientRect.left,
259 clientRect.top,
260 clientRect.right - clientRect.left,
261 clientRect.bottom - clientRect.top,
262 TRUE );
263
264 // invalidate the client area for redraw
265 InvalidateRect( preview, NULL, TRUE );
266
267} // end UpdatePreviewWindow
