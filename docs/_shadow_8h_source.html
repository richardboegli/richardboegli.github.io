40// shadow bit flags, keep this in sync with TheShadowNames
41enum ShadowType
42{
43 SHADOW_NONE = 0x00000000,
44 SHADOW_DECAL = 0x00000001, //shadow decal applied via modulate blend
45 SHADOW_VOLUME = 0x00000002,
46 SHADOW_PROJECTION = 0x00000004,
47 SHADOW_DYNAMIC_PROJECTION = 0x00000008, //extra setting for shadows which need dynamic updates
48 SHADOW_DIRECTIONAL_PROJECTION = 0x00000010, //extra setting for shadow decals that rotate with sun direction
49 SHADOW_ALPHA_DECAL = 0x00000020, //not really for shadows but for other decal uses. Alpha blended.
50 SHADOW_ADDITIVE_DECAL = 0x00000040 //not really for shadows but for other decal uses. Additive blended.
51};

154inline void Shadow::setOpacity(Int value)
155{
156 m_opacity=value;
157
158 if (m_type & SHADOW_ALPHA_DECAL)
159 {
160 m_diffuse = (m_color & 0x00ffffff) + (value << 24);
161// m_diffuse = m_color | (value << 24);ML changed
162 }
163 else
164 {
165 if (m_type & SHADOW_ADDITIVE_DECAL)
166 {
167 Real fvalue=(Real)m_opacity/255.0f;
168 m_diffuse=REAL_TO_INT(((Real)(m_color & 0xff) * fvalue))
169 |REAL_TO_INT(((Real)((m_color >> 8) & 0xff) * fvalue))
170 |REAL_TO_INT(((Real)((m_color >> 16) & 0xff) * fvalue));
171 }
172 }
173}
174
175inline void Shadow::setColor(Color value)
176{
177 m_color = value & 0x00ffffff; //filter out alpha
178
179 if (m_type & SHADOW_ALPHA_DECAL)
180 {
181 m_diffuse=m_color | (m_opacity << 24);
182 }
183 else
184 {
185 if (m_type & SHADOW_ADDITIVE_DECAL)
186 {
187 Real fvalue=(Real)m_opacity/255.0f;
188 m_diffuse=REAL_TO_INT(((Real)(m_color & 0xff) * fvalue))
189 |REAL_TO_INT(((Real)((m_color >> 8) & 0xff) * fvalue))
190 |REAL_TO_INT(((Real)((m_color >> 16) & 0xff) * fvalue));
191 }
192 }
193}
