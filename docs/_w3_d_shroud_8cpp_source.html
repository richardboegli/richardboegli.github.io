110//-----------------------------------------------------------------------------
115void W3DShroud::init(WorldHeightMap *pMap, Real worldCellSizeX, Real worldCellSizeY)
116{
117 DEBUG_ASSERTCRASH( m_pSrcTexture == NULL, ("ReAcquire of existing shroud textures"));
118 DEBUG_ASSERTCRASH( pMap != NULL, ("Shroud init with NULL WorldHeightMap"));
119
120 Int dstTextureWidth=0;
121 Int dstTextureHeight=0;
122 m_cellWidth=worldCellSizeX;
123 m_cellHeight=worldCellSizeY;
124
125 //Precompute a bounding box for entire shroud layer
126 if (pMap)
127 {
128 m_numCellsX = REAL_TO_INT_CEIL((Real)(pMap->getXExtent() - 1 - pMap->getBorderSizeInline()*2)*MAP_XY_FACTOR/m_cellWidth);
129 m_numCellsY = REAL_TO_INT_CEIL((Real)(pMap->getYExtent() - 1 - pMap->getBorderSizeInline()*2)*MAP_XY_FACTOR/m_cellHeight);
130
131 //Maximum visible cells will depend on maximum drawable terrain size plus 1 for partial cells (since
132 //shroud cells are larger than terrain cells).
133 dstTextureWidth=m_numMaxVisibleCellsX=REAL_TO_INT_FLOOR((Real)(pMap->getDrawWidth()-1)*MAP_XY_FACTOR/m_cellWidth)+1;
134 dstTextureHeight=m_numMaxVisibleCellsY=REAL_TO_INT_FLOOR((Real)(pMap->getDrawHeight()-1)*MAP_XY_FACTOR/m_cellHeight)+1;
135
136 dstTextureWidth = m_numCellsX;
137 dstTextureHeight = m_numCellsY;
138
139 dstTextureWidth += 2; //enlarge by 2 pixels so we can have a border color all the way around.
140 unsigned int depth = 1;
141 dstTextureHeight += 2; //enlarge by 2 pixels so we can have border color all the way around.
142 TextureLoader::Validate_Texture_Size((unsigned int &)dstTextureWidth,(unsigned int &)dstTextureHeight, depth);
143 }
144
145
146 UnsignedInt srcWidth,srcHeight;
147
148 srcWidth=m_numCellsX;
149 //vertical size is larger by 1 pixel so that we have some unused pixels to use in clearing the video texture.
150 //To clear the video texture, I will copy pixels from this unused area. There is no other way to clear a video
151 //memory texture to a known value because you can't lock it - only copy into it.
152 srcHeight=m_numCellsY;
153 srcHeight += 1;
154
155#ifdef DO_FOG_INTERPOLATION
156 m_finalFogData = new W3DShroudLevel[srcWidth*srcHeight];
157 m_currentFogData = new W3DShroudLevel[srcWidth*srcHeight];
158 //Clear the fog to black
159 memset(m_currentFogData,0,srcWidth*srcHeight);
160 memset(m_finalFogData,0,srcWidth*srcHeight);
161#endif
162
163#if defined(_DEBUG) || defined(_INTERNAL)
164 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
165 m_pSrcTexture = DX8Wrapper::_Create_DX8_Surface(srcWidth,srcHeight, WW3D_FORMAT_A4R4G4B4);
166 else
167#endif
168 m_pSrcTexture = DX8Wrapper::_Create_DX8_Surface(srcWidth,srcHeight, WW3D_FORMAT_R5G6B5);
169
170 DEBUG_ASSERTCRASH( m_pSrcTexture != NULL, ("Failed to Allocate Shroud Src Surface"));
171
172 D3DLOCKED_RECT rect;
173
174 //Get a pointer to source surface pixels.
175 HRESULT res = m_pSrcTexture->LockRect(&rect,NULL,D3DLOCK_NO_DIRTY_UPDATE);
176 m_pSrcTexture->UnlockRect();
177
178 DEBUG_ASSERTCRASH( res == D3D_OK, ("Failed to lock shroud src surface"));
179 res = 0;// just to avoid compiler warnings
180
181 m_srcTextureData=rect.pBits;
182 m_srcTexturePitch=rect.Pitch;
183
184 //clear entire texture to black
185 memset(m_srcTextureData,0,m_srcTexturePitch*srcHeight);
186
187#if defined(_DEBUG) || defined(_INTERNAL)
188 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
189 fillShroudData(TheGlobalData->m_shroudAlpha); //initialize shroud to a known value
190#endif
191
192 if (dstTextureWidth != m_dstTextureWidth || dstTextureHeight != m_dstTextureHeight )
193 ReleaseResources(); //need a new sized shroud
194
195 if (!m_pDstTexture )
196 { m_dstTextureWidth = dstTextureWidth;
197 m_dstTextureHeight = dstTextureHeight;
198 ReAcquireResources(); //allocate video memory surface
199 }
200
201 //Force a refresh of shroud data since we just created a new source texture.
202 if (ThePartitionManager)
203 ThePartitionManager->refreshShroudForLocalPlayer();
204}

208void W3DShroud::reset()
209{
210 //Free old shroud data since it may no longer fit new map.
211 if (m_pSrcTexture)
212 m_pSrcTexture->Release();
213 m_pSrcTexture=NULL;
214 if (m_finalFogData)
215 delete [] m_finalFogData;
216 m_finalFogData=NULL;
217 if (m_currentFogData)
218 delete [] m_currentFogData;
219 m_currentFogData=NULL;
220 m_clearDstTexture = TRUE; //always refill the destination texture after a reset
221}

230//-----------------------------------------------------------------------------
232Bool W3DShroud::ReAcquireResources(void)
233{
234 if (!m_dstTextureWidth)
235 return TRUE; //nothing to reaquire since shroud was never initialized with valid data
236
237 DEBUG_ASSERTCRASH( m_pDstTexture == NULL, ("ReAcquire of existing shroud texture"));
238
239 // Create destination texture (stored in video memory).
240 // Since we control the video memory copy, we can do partial updates more efficiently. Or do shift blits.
241#if defined(_DEBUG) || defined(_INTERNAL)
242 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
243 m_pDstTexture = MSGNEW("TextureClass") TextureClass(m_dstTextureWidth,m_dstTextureHeight,WW3D_FORMAT_A4R4G4B4,MIP_LEVELS_1, TextureClass::POOL_DEFAULT);
244 else
245#endif
246 m_pDstTexture = MSGNEW("TextureClass") TextureClass(m_dstTextureWidth,m_dstTextureHeight,WW3D_FORMAT_R5G6B5,MIP_LEVELS_1, TextureClass::POOL_DEFAULT);
247
248 DEBUG_ASSERTCRASH( m_pDstTexture != NULL, ("Failed ReAcquire of shroud texture"));
249
250 if (!m_pDstTexture)
251 { //could not create a valid texture
252 m_dstTextureWidth = 0;
253 m_dstTextureHeight = 0;
254 return FALSE;
255 }
256 m_pDstTexture->Get_Filter().Set_U_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
257 m_pDstTexture->Get_Filter().Set_V_Addr_Mode(TextureFilterClass::TEXTURE_ADDRESS_CLAMP);
258 m_pDstTexture->Get_Filter().Set_Mip_Mapping(TextureFilterClass::FILTER_TYPE_NONE);
259 m_clearDstTexture = TRUE; //force clearing of destination texture first time it's used.
260
261 return TRUE;
262}

264//-----------------------------------------------------------------------------
265W3DShroudLevel W3DShroud::getShroudLevel(Int x, Int y)
266{
267 DEBUG_ASSERTCRASH( m_pSrcTexture != NULL, ("Reading empty shroud"));
268
269 if (x < m_numCellsX && y < m_numCellsY)
270 {
271 UnsignedShort pixel=*(UnsignedShort *)((Byte *)m_srcTextureData + x*2 + y*m_srcTexturePitch);
272
273#if defined(_DEBUG) || defined(_INTERNAL)
274 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
275 //in this mode, alpha channel holds intensity
276 return (W3DShroudLevel)((1.0f-((Real)(pixel >> 12)/15.0f))*255.0f);
277 else
278#endif
279 //in this mode, green has the best precision at 6 bits.
280 return (W3DShroudLevel)((Real)((pixel >> 5)&0x3f)/63.0f*255.0f);
281 }
282 return 0;
283}

285//-----------------------------------------------------------------------------
286void W3DShroud::setShroudLevel(Int x, Int y, W3DShroudLevel level, Bool textureOnly)
287{
288 DEBUG_ASSERTCRASH( m_pSrcTexture != NULL, ("Writing empty shroud. Usually means that map failed to load."));
289
290 if (!m_pSrcTexture)
291 return;
292
293 if (x < m_numCellsX && y < m_numCellsY)
294 {
295 if (level < TheGlobalData->m_shroudAlpha)
296 level = TheGlobalData->m_shroudAlpha;
297
298#if defined(_DEBUG) || defined(_INTERNAL)
299 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
300 {
302 Int redVal = TheGlobalData->m_shroudColor.red;
303 Int greenVal = TheGlobalData->m_shroudColor.green;
304 Int blueVal = TheGlobalData->m_shroudColor.blue;
305// Int redVal = (SHROUD_COLOR >> 16) & 0xff;
306// Int greenVal = (SHROUD_COLOR >> 8) & 0xff;
307// Int blueVal = SHROUD_COLOR & 0xff;
308 Int alphaVal = 255 - level;
309
310 UnsignedShort pixel=((blueVal>>4)&0xf) | (((greenVal>>4)&0xf)<<4) | (((redVal>>4)&0xf)<<8) | (((alphaVal>>4)&0xf)<<12);
311 *(UnsignedShort *)((Byte *)m_srcTextureData + x*2 + y*m_srcTexturePitch)=pixel;
312 }
313 else
314#endif
315 {
316#ifdef DO_FOG_INTERPOLATION
317 if (!textureOnly)
318 m_finalFogData[x+y*m_numCellsX]=level;
319#endif
320 UnsignedInt bluepixel = (UnsignedInt)((Real)level*((Real)(TheGlobalData->m_shroudColor.getAsInt()&0xff)/255.0f));
321 UnsignedInt greenpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff00)>>8)/255.0f));
322 UnsignedInt redpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff0000)>>16)/255.0f));
323// UnsignedInt bluepixel = (UnsignedInt)((Real)level*((Real)(SHROUD_COLOR&0xff)/255.0f));
324// UnsignedInt greenpixel = (UnsignedInt)((Real)level*((Real)((SHROUD_COLOR&0xff00)>>8)/255.0f));
325// UnsignedInt redpixel = (UnsignedInt)((Real)level*((Real)((SHROUD_COLOR&0xff0000)>>16)/255.0f));
326 if (level == 255)
327 { //unshrouded pixels should be fully lit
328 redpixel = 255;
329 greenpixel = 255;
330 bluepixel = 255;
331 }
332
333 UnsignedShort *texel = (UnsignedShort *)((Byte *)m_srcTextureData + x*2 + y*m_srcTexturePitch);
334
335// For those interested, MLorenzen has this bock commented out until he gets back on Mon, Sept. 30 2002
336 // If this code is still here by mid october, nuke it!
337// UnsignedInt texelRed = ((*texel >> 8 ) & 0xf8);
338// UnsignedInt texelGreen =((*texel >> 3 ) & 0xfc);
339// UnsignedInt texelBlue = ((*texel << 3 ) & 0xf8);
340// if (texelRed < texelGreen && texelGreen > texelBlue)
341// {
342// bluepixel += redpixel;
343// bluepixel -= redpixel;
344// }
345
346 *texel = ( ((bluepixel&0xf8) >> 3) | ((greenpixel&0xfc)<<3) | ((redpixel&0xf8)<<8));
347 }
348 return;
349 }

352//-----------------------------------------------------------------------------
354void W3DShroud::fillShroudData(W3DShroudLevel level)
355{
356
357 Int x,y;
358 UnsignedShort pixel;
359
360 if (level < TheGlobalData->m_shroudAlpha)
361 level = TheGlobalData->m_shroudAlpha;
362
363#if defined(_DEBUG) || defined(_INTERNAL)
364 //convert value to pixel format
365 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
366 {
367 Int redVal = TheGlobalData->m_shroudColor.red;
368 Int greenVal = TheGlobalData->m_shroudColor.green;
369 Int blueVal = TheGlobalData->m_shroudColor.blue;
370// Int redVal = (SHROUD_COLOR >> 16) & 0xff;
371// Int greenVal = (SHROUD_COLOR >> 8) & 0xff;
372// Int blueVal = SHROUD_COLOR & 0xff;
373 Int alphaVal = 255 - level;
374
375 pixel=((blueVal>>4)&0xf) | (((greenVal>>4)&0xf)<<4) | (((redVal>>4)&0xf)<<8) | (((alphaVal>>4)&0xf)<<12);
376 }
377 else
378#endif
379 {
380 UnsignedInt bluepixel = (UnsignedInt)((Real)level*((Real)(TheGlobalData->m_shroudColor.getAsInt()&0xff)/255.0f));
381 UnsignedInt greenpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff00)>>8)/255.0f));
382 UnsignedInt redpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff0000)>>16)/255.0f));
383// UnsignedInt bluepixel = (UnsignedInt)((Real)level*((Real)(SHROUD_COLOR&0xff)/255.0f));
384// UnsignedInt greenpixel = (UnsignedInt)((Real)level*((Real)((SHROUD_COLOR&0xff00)>>8)/255.0f));
385// UnsignedInt redpixel = (UnsignedInt)((Real)level*((Real)((SHROUD_COLOR&0xff0000)>>16)/255.0f));
386
387 if (level == 255)
388 { //unshrouded pixels should be fully lit
389 redpixel = 255;
390 greenpixel = 255;
391 bluepixel = 255;
392 }
393 pixel=( ((bluepixel&0xf8) >> 3) | ((greenpixel&0xfc)<<3) | ((redpixel&0xf8)<<8));
394 }
395
396 UnsignedShort *ptr=(UnsignedShort *)m_srcTextureData;
397 Int pitch = m_srcTexturePitch >> 1; //2 bytes per pointer increment
398 for (y=0; y<m_numCellsY; y++)
399 {
400 for (x=0; x<m_numCellsX; x++)
401 ptr[x]=pixel;
402 ptr += pitch;
403 }
404
405#ifdef DO_FOG_INTERPOLATION
406 //Set the final shroud state. May differe from current state because of time interpolation.
407 W3DShroudLevel *cptr=m_finalFogData;
408 pitch = m_numCellsX;
409 for (y=0; y<m_numCellsY; y++)
410 {
411 for (x=0; x<m_numCellsX; x++)
412 cptr[x]=level;
413 ptr += pitch;
414 }
415#endif
416}

418void W3DShroud::fillBorderShroudData(W3DShroudLevel level, SurfaceClass* pDestSurface)
419{
420 Int x,y;
421 UnsignedShort pixel;
422
423 if (level < TheGlobalData->m_shroudAlpha)
424 level = TheGlobalData->m_shroudAlpha;
425
426#if defined(_DEBUG) || defined(_INTERNAL)
427 //convert value to pixel format
428 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
429 {
430 Int redVal = TheGlobalData->m_shroudColor.red;
431 Int greenVal = TheGlobalData->m_shroudColor.green;
432 Int blueVal = TheGlobalData->m_shroudColor.blue;
433 Int alphaVal = 255 - level;
434
435 pixel=((blueVal>>4)&0xf) | (((greenVal>>4)&0xf)<<4) | (((redVal>>4)&0xf)<<8) | (((alphaVal>>4)&0xf)<<12);
436 }
437 else
438#endif
439 {
440 UnsignedInt bluepixel = (UnsignedInt)((Real)level*((Real)(TheGlobalData->m_shroudColor.getAsInt()&0xff)/255.0f));
441 UnsignedInt greenpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff00)>>8)/255.0f));
442 UnsignedInt redpixel = (UnsignedInt)((Real)level*((Real)((TheGlobalData->m_shroudColor.getAsInt()&0xff0000)>>16)/255.0f));
443
444 if (level == 255)
445 { //unshrouded pixels should be fully lit
446 redpixel = 255;
447 greenpixel = 255;
448 bluepixel = 255;
449 }
450 pixel=( ((bluepixel&0xf8) >> 3) | ((greenpixel&0xfc)<<3) | ((redpixel&0xf8)<<8));
451 }
452
453 //Skip to unused texels within the shroud data
454 UnsignedShort *ptr=(UnsignedShort *)m_srcTextureData + m_numCellsY*(m_srcTexturePitch >> 1);
455
456 //Fill unused texels with border color
457 for (x=0; x<m_numCellsX; x++)
458 ptr[x]=pixel;
459
460 //Fill destination texture with border color
461
462 RECT srcRect;
463
464 //create a rectangle enclosing bottom row of unused pixels long enough
465 //to cover destination width.
466 srcRect.left=0;
467 srcRect.top=m_numCellsY;
468 srcRect.right= m_numCellsX;
469 srcRect.bottom= m_numCellsY+1;
470
471 POINT dstPoint={0,0};
472
473 Int numFullCopies = m_dstTextureWidth/srcRect.right;
474 Int numExtraPixels = m_dstTextureWidth%srcRect.right;
475
476 for (y=0; y<m_dstTextureHeight; y++)
477 {
478 dstPoint.y=y;
479 dstPoint.x=0;
480
481 for (x=0; x<numFullCopies; x++)
482 {
483 dstPoint.x = x * srcRect.right; //advance to next set of pixel in row.
484
485 DX8Wrapper::_Copy_DX8_Rects(
486 m_pSrcTexture,
487 &srcRect,
488 1,
489 pDestSurface->Peek_D3D_Surface(),
490 &dstPoint);
491 }
492 if (numExtraPixels)
493 { Int oldVal=srcRect.right;
494 dstPoint.x = numFullCopies * oldVal;
495 srcRect.right = numExtraPixels;
496 DX8Wrapper::_Copy_DX8_Rects(
497 m_pSrcTexture,
498 &srcRect,
499 1,
500 pDestSurface->Peek_D3D_Surface(),
501 &dstPoint);
502 srcRect.right = oldVal;
503 }
504 }

524//-----------------------------------------------------------------------------
526void W3DShroud::render(CameraClass *cam)
527{
528 if (!m_pSrcTexture)
529 return; //nothing to update from. Must be in reset state.
530
531 if (DX8Wrapper::_Get_D3D_Device8() && (DX8Wrapper::_Get_D3D_Device8()->TestCooperativeLevel()) != D3D_OK)
532 return; //device not ready to render anything
533
534#if defined(_DEBUG) || defined(_INTERNAL)
535 if (TheGlobalData && TheGlobalData->m_fogOfWarOn != m_drawFogOfWar)
536 {
537 //fog state has changed since last time shroud system was initialized
538 reset();
539 ReleaseResources();
540 init(TheTerrainRenderObject->getMap(),m_cellWidth,m_cellHeight);
541 ThePartitionManager->refreshShroudForLocalPlayer();
542 m_drawFogOfWar=TheGlobalData->m_fogOfWarOn;
543 m_clearDstTexture=TRUE;
544 }
545#endif
546
547 DEBUG_ASSERTCRASH( m_pSrcTexture != NULL, ("Updating unallocated shroud texture"));
548
549#ifdef LOAD_DUMMY_SHROUD
550
551 static doInit=1;
552 if (doInit)
553 {
554 //some temporary code here to debug the shroud.
555
557 fillShroudData(1.0f); //force to all shrouded
558
559 Short *src=(Short *)m_srcTextureData;
560 //fill with some dummy values
561 src[0]=(char)0xff;
562 src[1]=(char)0xff;
563 src[m_numCellsX]=(char)0xff;
564 src[m_numCellsX+1]=(char)0xff;
565 src[m_numCellsX*2]=(char)0xff;
566
567 src[m_numCellsX*9+8]=(char)0xff;
568 src[m_numCellsX*8+8]=(char)0xff;
569 src[m_numCellsX*7+8]=(char)0xff;
570 src[m_numCellsX*8+9]=(char)0xff;
571 src[m_numCellsX*8+7]=(char)0xff;
572
573 DummyTexture=WW3DAssetManager::Get_Instance()->Get_Texture("shroud1024.tga");
574
575 Short *dataDest=(Short *)((char *)m_srcTextureData); //offset to correct row of full sysmem shroud
576 Int pitchDest = m_srcTexturePitch >> 1; //2 bytes per pixel so divide byte count by 2.
577
578 //Copy the dummy shroud into our game shroud.
579 SurfaceClass *pSurface=DummyTexture->Get_Surface_Level(0);
580 Int pitch;
581 Int *dataSrc=(Int *)((char *)pSurface->Lock(&pitch)); //offset to correct row of full sysmem shroud
582 pitch >>= 2; //4 bytes per pixel so divide byte count by 4.
583 SurfaceClass::SurfaceDescription desc;
584 pSurface->Get_Description(desc);
585
586 //Check if source data is larger than our current shroud
587 desc.Width = __min(desc.Width,m_numCellsX);
588 desc.Height = __min(desc.Height,m_numCellsY);
589
590 for (Int y=0; y<desc.Height; y++)
591 {
592 for (Int x=0; x<desc.Width; x++)
593 {
594#if defined(_DEBUG) || defined(_INTERNAL)
595 if (TheGlobalData && TheGlobalData->m_fogOfWarOn)
596 {

727#define FOG_INTERPOLATION_RATE (255.0f/1000.0f) //take one second to go from black to fully lit.
728//-----------------------------------------------------------------------------
729void W3DShroud::interpolateFogLevels(RECT *rect)
730{
731 static UnsignedInt prevTime = timeGetTime();
732
733 UnsignedInt timeDiff=timeGetTime()-prevTime;
734
735 if (!timeDiff)
736 return; //no time has elapsed
737
738 prevTime +=timeDiff; //update for next frame
739
740 Int maxFogChange=FOG_INTERPOLATION_RATE * (Real)timeDiff; //maximum amount of fog change allowed in frame.
741 if (maxFogChange > 255)
742 maxFogChange = 255;
743 W3DShroudLevel levelDelta = maxFogChange;
744
745 W3DShroudLevel *startLevel=m_currentFogData;
746 W3DShroudLevel *finalLevel=m_finalFogData;
747
748 for (Int j=0; j<m_numCellsY; j++)
749 {
750 for (Int i=0; i<m_numCellsX; i++,startLevel++,finalLevel++)
751 if (*startLevel != *finalLevel)
752 { //fog needs fading.
753 if (*startLevel == *finalLevel)
754 continue;
755 else
756 if (*finalLevel < *startLevel)
757 {
758 if ((*startLevel - *finalLevel) < levelDelta)
759 *startLevel = *finalLevel; //change too large so clamp to final value.
760 else
761 *startLevel -= levelDelta;
762 }
763 else
764 if (*finalLevel > *startLevel)
765 {
766 if ((*finalLevel - *startLevel) < levelDelta)
767 *startLevel = *finalLevel; //change too large so clamp to final value.
768 else
769 *startLevel += levelDelta;
770 }
771 setShroudLevel(i,j,*startLevel,TRUE);
772 }
773 }
