75 virtual SeismicSimStatusCode filterCallback( WorldHeightMapInterfaceClass *heightMap, const SeismicSimulationNode *node )
76 {
77
78
79 Int life = node->m_life;
80
81 if ( heightMap == NULL )
82 return SEISMIC_STATUS_INVALID;
83
84
85 if ( life == 0 )
86 return SEISMIC_STATUS_ACTIVE;
87 if ( life < 15 )
88 {
89 // ADD HEIGHT BECAUSE THE EXPLOSION IS PUSHING DIRT UP
90
91 Real magnitude = node->m_magnitude;
92
93 Real offsScalar = magnitude / (Real)life; // real-life, get it?
94 Int radius = node->m_radius;
95 Int border = heightMap->getBorderSize();
96 Int centerX = node->m_center.x + border ;
97 Int centerY = node->m_center.y + border ;
98
99 UnsignedInt workspaceWidth = radius*2;
100 Real *workspace = NEW( Real[ sqr(workspaceWidth) ] );
101 Real *workspaceEnd = workspace + sqr(workspaceWidth);
102
103
104 for ( Real *t = workspace; t < workspaceEnd; ++t ) *t = 0.0f;// clear the workspace
105
106 for (Int x = 0; x < radius; ++x)
107 {
108 for (Int y = 0; y < radius; ++y)
109 {
110
111 Real distance = sqrt( sqr(x) + sqr(y) );//Pythagoras
112
113 if ( distance < radius )
114 {
115 Real distScalar = cos( ( distance / radius * (PI/2) ) );
116 Real height = (offsScalar * distScalar);
117
118 workspace[ (radius + x) + workspaceWidth * (radius + y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX + x, centerY + y ) ;//kaleidoscope
119
120 if ( x != 0 ) // non-zero test prevents cross-shaped double stamp
121 {
122 workspace[ (radius - x) + workspaceWidth * (radius + y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX - x, centerY + y ) ;
123 if ( y != 0 )
124 workspace[ (radius - x) + workspaceWidth * (radius - y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX - x, centerY - y ) ;
125 }
126 if ( y != 0 )
127 workspace[ (radius + x) + workspaceWidth * (radius - y) ] = height + heightMap->getBilinearSampleSeismicZVelocity( centerX + x, centerY - y ) ;
128 }
129 }
130 }
131
132 // stuff the values from the workspace into the heightmap's velocities
133 for (x = 0; x < workspaceWidth; ++x)
134 for (Int y = 0; y < workspaceWidth; ++y)
135 heightMap->setSeismicZVelocity( centerX - radius + x, centerY - radius + y, workspace[ x + workspaceWidth * y ] );
136
137 delete [] workspace;
138
139 return SEISMIC_STATUS_ACTIVE;
140 }
141 else
142 return SEISMIC_STATUS_ZERO_ENERGY;
143 }
144
145 virtual Real applyGravityCallback( Real velocityIn )
146 {
147 Real velocityOut = velocityIn;
148 velocityOut -= 1.5f;
149 return velocityOut;
150 }

387void W3DTerrainVisual::handleSeismicSimulations( void )
388{
389 if ( ! m_clientHeightMap || ! m_logicHeightMap || ! m_terrainRenderObject )
390 return;
391
392
393 if ( ! m_seismicSimulationList.empty() )
394 {
395 SeismicSimulationListIt it = m_seismicSimulationList.begin();
396
397
398 m_clientHeightMap->clearSeismicUpdateFlags();
399
400
401 while ( it != m_seismicSimulationList.end() )
402 {
403 SeismicSimulationNode *ssn = &*it;
404
405 if ( ssn )
406 {
407 SeismicSimulationFilterBase::SeismicSimStatusCode code = ssn->handleFilterCallback( m_clientHeightMap );
408 DEBUG_ASSERTCRASH( code != SeismicSimulationFilterBase::SEISMIC_STATUS_INVALID, ("Trouble in the Seismic simulator.") );
409
410 switch ( code )
411 {
412 case SeismicSimulationFilterBase::SEISMIC_STATUS_ACTIVE:
413 {
414 break;
415 }
416 case SeismicSimulationFilterBase::SEISMIC_STATUS_ZERO_ENERGY:
417 {
418 break;
419 }
420 }
421
422 Int border = m_clientHeightMap->getBorderSizeInline();
423
424 // Now we apply some gravity to the dirt, so it falls back to its "original" height
425 UnsignedInt fallCount = 0;
426 for (Int x = border+ssn->m_region.lo.x; x < border+ssn->m_region.hi.x; ++x)
427 {
428 for (Int y = border+ssn->m_region.lo.y; y < border+ssn->m_region.hi.y; ++y)
429 {
430 if ( ! m_clientHeightMap->getSeismicUpdateFlag( x, y ) )
431 {
432 UnsignedByte heightOfOriginal = m_logicHeightMap->getHeight( x, y ); // LOGIC, YES DEFINITELY THE LOGIC
433
434 Real oldSpeed = m_clientHeightMap->getSeismicZVelocity( x, y );
435 Real newSpeed = ssn->applyGravity( oldSpeed );// - 0.5f;
436
437 m_clientHeightMap->setSeismicZVelocity( x, y, newSpeed );
438
439 Int heightToUse = m_clientHeightMap->getHeight( x, y ) + newSpeed ;
440
441
442 if (heightToUse <= heightOfOriginal)
443 {
444 heightToUse = heightOfOriginal;
445 m_clientHeightMap->setSeismicZVelocity( x, y, 0.0f ); //poof! the dirt hit ground level so stop "falling"
446 }
447 else
448 {
449 ++fallCount;
450
451 if ( heightToUse > 255 )
452 heightToUse = 255;
453
454 }
455 m_clientHeightMap->setRawHeight( x, y, heightToUse );
456 m_clientHeightMap->setSeismicUpdateFlag( x, y, TRUE );
457 }
458
459 }
460 }
461
462 if ( fallCount == 0 )
463 ssn->m_clean = TRUE;
464
465 }
466
467 ++it;
468
469 }
470 }

482void W3DTerrainVisual::updateSeismicSimulations( void )
483{
484
485 if (m_logicHeightMap==NULL)
486 return;
487
488 if (m_clientHeightMap==NULL)
489 return;
490
491 if (m_terrainRenderObject==NULL)
492 return;
493
494 if ( ! m_seismicSimulationList.empty() )
495 {
496 SeismicSimulationListIt it = m_seismicSimulationList.begin();
497
498 // First we run through the list and do our business for each region
499
500 while ( it != m_seismicSimulationList.end() )
501 {
502 SeismicSimulationNode *hur = &*it;
503 if ( hur )
504 {
505 Int border = m_clientHeightMap->getBorderSizeInline();
506
507 TheTerrainRenderObject->updateBlock(
508 hur->m_region.lo.x + border,
509 hur->m_region.lo.y + border,
510 hur->m_region.hi.x + border,
511 hur->m_region.hi.y + border,
512 m_clientHeightMap,
513 0);
514 }
515
516 ++it;
517
518 }
519 // Then we check to see if these need to get erased from the list
520 it = m_seismicSimulationList.begin();
521 while ( it != m_seismicSimulationList.end() )
522 {
523 SeismicSimulationNode *hur = &*it;
524 if ( hur->m_clean )
525 {
526 it = m_seismicSimulationList.erase( it );
527 }
528 else
529 ++it;
530
531 }
532
533 }

735 Coord3D *rayEnd,
736 Coord3D *result )
737{

740 // sanity
741 if( rayStart == NULL || rayEnd == NULL )
742 return hit;

744 if( m_terrainRenderObject )
745 {
746 CastResultStruct res;
747 LineSegClass lineSeg( Vector3( rayStart->x, rayStart->y, rayStart->z ),
748 Vector3( rayEnd->x, rayEnd->y, rayEnd->z ) );
749 RayCollisionTestClass rayTest( lineSeg, &res );
750
751 hit = m_terrainRenderObject->Cast_Ray( rayTest );
752 if( hit && result )
753 {
754 Vector3 point = rayTest.Result->ContactPoint;
755
756 result->x = point.X;
757 result->y = point.Y;
758 result->z = point.Z;
759
760 } // end if
761
762 } // end if
763
764 // return hit result
765 return hit;
