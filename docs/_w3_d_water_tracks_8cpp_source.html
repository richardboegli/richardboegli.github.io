181void WaterTracksObj::init( Real width, Real length, Vector2 &start, Vector2 &end, Char *texturename, Int waveTimeOffset)
182{
183 freeWaterTracksResources(); //free old resources used by this track
184
185 //save original settings used to create this wave
186 m_initStartPos = start;
187 m_initEndPos = end;
188 m_initTimeOffset = waveTimeOffset;
189
190 m_boundingSphere.Init(Vector3(0,0,0),400);
191 m_boundingBox.Center.Set(0.0f, 0.0f, 0.0f);
192 m_boundingBox.Extent.Set(400.0f, 400.0f, 1.0f);
193 m_x=WATER_STRIP_X;
194 m_y=WATER_STRIP_Y;
195 m_elapsedMs=m_initTimeOffset;
196 m_startPos=start;
197 m_perpDir=m_waveDir=end-start;
198 m_perpDir.Rotate(-1.57079632679f); //get vector perpendicular to wave motion.
199 m_perpDir.Normalize();
200
201 m_waveDir=m_perpDir;
202 m_waveDir.Rotate(PI/2); //get vector along wave travel direction.
203 //move back by width of wave so start point turns into maximum reach of wave.
204 //m_startPos -= m_waveDir*m_width;
205 //move back initial tip off of wave a couple units off the final position
206 //to give it some room to travel. Travel vector is stored in m_waveDir.
207 m_waveDistance = waveTypeInfo[m_type].m_waveDistance; //total distance traveled by wave front
208
209 m_waveDir *= m_waveDistance;
210 m_startPos -= m_waveDir; //move start point down away from shoreline
211
212 m_initialVelocity=waveTypeInfo[m_type].m_initialVelocity; //velocity per ms
213 m_totalMs = m_waveDistance/m_initialVelocity; //amount of time for wave to travel complete distance
214
215 m_fadeMs = waveTypeInfo[m_type].m_fadeMs; //time for wave to fade out after it stops on beach
216
217 m_waveInitialWidth=length * waveTypeInfo[m_type].m_initialWidthFraction;
218 m_waveInitialHeight=m_waveInitialWidth * waveTypeInfo[m_type].m_initialHeightWidthFraction;
219 m_waveFinalWidth=length;
220 m_waveFinalHeight=width;
221
222 //get total time for front to complete its cycle
223 m_timeToReachBeach=(m_waveDistance - m_waveFinalHeight)/m_initialVelocity;
224 m_frontSlowDownAcc= -(m_initialVelocity*m_initialVelocity)/(2*m_waveFinalHeight); //deceleration of wave after it hits land
225 m_timeToStop = -m_initialVelocity/m_frontSlowDownAcc;
226 m_timeToRetreat = sqrt(fabs(2.0f*m_waveFinalHeight/m_frontSlowDownAcc));
227 m_totalMs = m_timeToReachBeach + m_timeToStop + m_timeToRetreat; //total time that wave front is on screen
228 m_backSlowDownAcc = (2.0f*m_waveInitialHeight/(m_timeToStop*m_timeToStop));//((m_waveInitialHeight - m_velocity*m_timeToStop)*2.0f)/(m_timeToStop*m_timeToStop);
229 m_timeToCompress = waveTypeInfo[m_type].m_timeToCompress; //time for back of wave to continue moving forward after front starts retreating.
230
231
232 if (m_type == WaveTypeStationary)
233 { //this is a stationary wave slightly behind starting point
234 m_timeToRetreat = 1000; //time to fade out.
235 m_totalMs = m_timeToReachBeach + m_timeToStop+m_fadeMs+m_timeToRetreat; //trigger when other wave stops.
236 m_startPos = start;
237 m_fadeMs = 1000; //time for wave to fade out after it stops on beach
238 }
239
240 m_stageZeroTexture=WW3DAssetManager::Get_Instance()->Get_Texture(texturename);
241}

270 m_elapsedMs=0;
271 m_initialVelocity=0.001f*MAP_XY_FACTOR; //velocity per ms
272 m_totalMs=m_waveDir.Length()/m_initialVelocity;
273 m_fadeMs = 3000; //time for wave to fade out after it stops on beach
274
275 m_stageZeroTexture=WW3DAssetManager::Get_Instance()->Get_Texture(texturename);

311Int WaterTracksObj::render(DX8VertexBufferClass *vertexBuffer, Int batchStart)
312{
313 VertexFormatXYZDUV1 *vb;
314 Vector2 waveTailOrigin,waveFrontOrigin;
315 Real ooWaveDirLen=1.0f/m_waveDir.Length(); //one over length
316 Real waterHeight;
317 Real waveAlpha;
318 Real widthFrac;
319 Real heightFrac;
320
321 if (batchStart < (WATER_VB_PAGES*WATER_STRIP_X*WATER_STRIP_Y-m_x*m_y))
322 { //we have room in current VB, append new verts
323 if(vertexBuffer->Get_DX8_Vertex_Buffer()->Lock(batchStart*vertexBuffer->FVF_Info().Get_FVF_Size(),m_x*m_y*vertexBuffer->FVF_Info().Get_FVF_Size(),(unsigned char**)&vb,D3DLOCK_NOOVERWRITE) != D3D_OK)
324 return batchStart;
325 }
326 else
327 { //ran out of room in last VB, request a substitute VB.
328 if(vertexBuffer->Get_DX8_Vertex_Buffer()->Lock(0,m_x*m_y*vertexBuffer->FVF_Info().Get_FVF_Size(),(unsigned char**)&vb,D3DLOCK_DISCARD) != D3D_OK)
329 return batchStart;
330 batchStart=0; //reset start of page to first vertex
331 }
332
333 //Adjust wave position in a non-linear way so that it slows down as it hits the target. Using 1/4 sine wave
334 //seems to work okay since it maxes out at 1.0 at our final position.
335 //Real timeFrac=(Real)m_elapsedMs/(Real)m_totalMs;//sinf(0.5f*3.14159f*(Real)m_elapsedMs/(Real)m_totalMs);
336
337 //Real displacement=m_elapsedMs*waveInitialV+0.5*waveAcceleration*m_elapsed*m_elapsed;
338
339 heightFrac=1.0f;
340 widthFrac = 1.0f;
341
342 if (m_type == WaveTypeStationary)
343 { //stationary wave
344 waveFrontOrigin = m_startPos;
345 waveFrontOrigin -= m_perpDir*m_waveFinalWidth*0.5f; //offset to left edge of wave
346 waveTailOrigin = waveFrontOrigin - m_waveFinalHeight * ooWaveDirLen*m_waveDir;
347 waveAlpha = 0.0f;
348
349 if (m_elapsedMs >= m_totalMs)
350 m_elapsedMs = 0; //done with effect*/
351 if (m_elapsedMs > (m_timeToReachBeach + m_timeToStop -1000 + m_fadeMs))
352 { //fading out
353 waveAlpha = m_elapsedMs-(m_timeToReachBeach + m_timeToStop - 1000 +m_fadeMs);//(m_totalMs-m_timeToRetreat -m_fadeMs - m_elapsedMs)/m_fadeMs;
354 waveAlpha = waveAlpha / m_timeToRetreat;
355 waveAlpha = 1.0f - waveAlpha;
356 if (waveAlpha < 0.0f)
357 waveAlpha = 0.0f;
358 }
359 else
360 if (m_elapsedMs > (m_timeToReachBeach + m_timeToStop - 1000))
361 { //start fading up
362
363 waveAlpha = m_elapsedMs-(m_timeToReachBeach + m_timeToStop - 1000);//(m_totalMs-m_timeToRetreat -m_fadeMs - m_elapsedMs)/m_fadeMs;
364 waveAlpha = waveAlpha / m_fadeMs;
365 if (waveAlpha > 1.0f)
366 waveAlpha = 1.0f;
367 }
368 }
369 else
370 { //moving wave
371
372 //get coordinate of top left of wave strip
373 if (m_elapsedMs < m_timeToReachBeach)
374 { //wave has not reached beach yet so position only depends on velocity
375 waveAlpha = m_elapsedMs / m_timeToReachBeach;
376 widthFrac = waveAlpha;
377 widthFrac=(m_waveInitialWidth + widthFrac* (m_waveFinalWidth-m_waveInitialWidth))/m_waveFinalWidth;
378
379 waveFrontOrigin = m_startPos + m_initialVelocity*m_elapsedMs*ooWaveDirLen*m_waveDir;
380 waveFrontOrigin -= m_perpDir*m_waveFinalWidth*0.5f*widthFrac; //offset to left edge of wave
381 //Tail of wave will be behind the front by fixed amount.
382 waveTailOrigin = waveFrontOrigin - m_waveInitialHeight * ooWaveDirLen*m_waveDir;
383 }
384 else //wave has reached beach and is decelerating
385 if (m_elapsedMs < m_totalMs)
386 { waveAlpha = 1.0f;
387 widthFrac = 1.0f;
388 //Get position of wave when it hit the beach
389 waveFrontOrigin = m_startPos + m_initialVelocity*m_timeToReachBeach*ooWaveDirLen*m_waveDir;
390 waveTailOrigin = waveFrontOrigin; //store position for calculating tail position
391 //Add movement after it hit the beach
392 Real elapsedMs=m_elapsedMs - m_timeToReachBeach;
393 waveFrontOrigin += (m_initialVelocity*elapsedMs+0.5f*m_frontSlowDownAcc*elapsedMs*elapsedMs)*ooWaveDirLen*m_waveDir;
394 waveFrontOrigin -= m_perpDir*m_waveFinalWidth*0.5f*widthFrac; //offset to left edge of wave
395
396 Real timeSinceBacktrack = m_elapsedMs - m_timeToReachBeach - m_timeToStop;
397 if (timeSinceBacktrack < 0)
398 timeSinceBacktrack = 0;
399 waveAlpha = timeSinceBacktrack/m_fadeMs;
400 if (waveAlpha > 1.0f)
401 waveAlpha = 1.0f;
402
403 waveAlpha = 1.0f - waveAlpha;
404
405 //Get position of tail when front hits the beach.
406 waveTailOrigin -= m_waveInitialHeight * ooWaveDirLen*m_waveDir;
407
408 if (m_elapsedMs > (m_timeToReachBeach+m_timeToStop+m_timeToCompress))
409 { elapsedMs=elapsedMs;
410 waveTailOrigin += (0.5f*m_backSlowDownAcc*(m_timeToStop+m_timeToCompress)*(m_timeToStop+m_timeToCompress))*ooWaveDirLen*m_waveDir;
411 //get time since wave should have stopped moving forward
412 Real newElapsed = m_elapsedMs - (m_timeToReachBeach+m_timeToStop+m_timeToCompress);
413 waveTailOrigin += (0.5f*m_frontSlowDownAcc*newElapsed*newElapsed)*ooWaveDirLen*m_waveDir;
414 }
415 else
416 // if (m_elapsedMs < (m_totalMs-m_timeToRetreat))
417 //find position of tail including slowdown after it hit the beach
418 // waveTailOrigin += (m_initialVelocity*elapsedMs+0.5f*m_backSlowDownAcc*elapsedMs*elapsedMs)*ooWaveDirLen*m_waveDir;
419 waveTailOrigin += (0.5f*m_backSlowDownAcc*elapsedMs*elapsedMs)*ooWaveDirLen*m_waveDir;
420
421 waveTailOrigin -= m_perpDir*m_waveFinalWidth*0.5f*widthFrac; //offset to left edge of wave
422 }
423 else
424 { m_elapsedMs = 0;
425 waveAlpha = m_elapsedMs / m_timeToReachBeach;
426 widthFrac = waveAlpha;
427 widthFrac=(m_waveInitialWidth + widthFrac* (m_waveFinalWidth-m_waveInitialWidth))/m_waveFinalWidth;
428
429 waveFrontOrigin = m_startPos + m_initialVelocity*m_elapsedMs*ooWaveDirLen*m_waveDir;
430 waveFrontOrigin -= m_perpDir*m_waveFinalWidth*0.5f*widthFrac; //offset to left edge of wave
431 //Tail of wave will be behind the front by fixed amount.
432 waveTailOrigin = waveFrontOrigin - m_waveInitialHeight * ooWaveDirLen*m_waveDir;
433 }
434 }
435
436 //First insert tail of wave:
437 Vector2 testPoint(waveTailOrigin);
438 TheTerrainLogic->isUnderwater(testPoint.X,testPoint.Y,&waterHeight);
439 vb->x= testPoint.X;
440 vb->y= testPoint.Y;
441 vb->z=waterHeight+1.5f;
442 vb->diffuse=(REAL_TO_INT(waveAlpha*255.0f)<<24) |0xffffff;
443 if (m_flipU)
444 vb->u1=1;
445 else
446 vb->u1=0;
447 vb->v1=0;
448 vb++;
449 testPoint.Set(waveTailOrigin + m_perpDir*m_waveFinalWidth*widthFrac);
450 vb->x= testPoint.X;
451 vb->y= testPoint.Y;
452 vb->z=waterHeight+1.5f;
453 vb->diffuse=(REAL_TO_INT(waveAlpha*255.0f)<<24) |0xffffff;
454 if (m_flipU)
455 vb->u1=0.0f;
456 else
457 vb->u1=1.0f;
458 vb->v1=0;
459 vb++;
460 //insert front of wave
461 testPoint.Set(waveFrontOrigin);
462 vb->x= testPoint.X;
463 vb->y= testPoint.Y;
464 vb->z=waterHeight+1.5f;
465 vb->diffuse=(REAL_TO_INT(waveAlpha*255.0f)<<24) |0xffffff;
466 if (m_flipU)
467 vb->u1=1;
468 else
469 vb->u1=0;
470 vb->v1=1.0f;
471 vb++;
472 testPoint.Set(waveFrontOrigin + m_perpDir*m_waveFinalWidth*widthFrac);
473 vb->x= testPoint.X;
474 vb->y= testPoint.Y;
475 vb->z=waterHeight+1.5f;
476 vb->diffuse=(REAL_TO_INT(waveAlpha*255.0f)<<24) |0xffffff;
477 if (m_flipU)
478 vb->u1=0;
479 else
480 vb->u1=1.0f;
481 vb->v1=1.0f;
482 vb++;
483
484 vertexBuffer->Get_DX8_Vertex_Buffer()->Unlock();
485
486 Int idxCount=(m_y-1)*(m_x*2+2) - 2; //index count
487
488 DX8Wrapper::Set_Index_Buffer(TheWaterTracksRenderSystem->m_indexBuffer,batchStart);
489 DX8Wrapper::Draw_Strip(0,idxCount-2,0,m_x*m_y); //there are always n-2 primitives for n index strip.
490
491 return batchStart+m_x*m_y; //return new offset into unused area of vertex buffer
492}

502//=============================================================================
503WaterTracksObj *WaterTracksRenderSystem::bindTrack(waveType type)
504{
505 WaterTracksObj *mod,*nextmod,*prevmod;
506
507 mod = m_freeModules;
508 if( mod )
509 {
510 // take module off the free list
511 if( mod->m_nextSystem )
512 mod->m_nextSystem->m_prevSystem = mod->m_prevSystem;
513 if( mod->m_prevSystem )
514 mod->m_prevSystem->m_nextSystem = mod->m_nextSystem;
515 else
516 m_freeModules = mod->m_nextSystem;
517
518 mod->m_type=type;
519
520 // put module on the used list (sorted next to similar types)
521 nextmod=NULL,prevmod=NULL;
522 for( nextmod = m_usedModules; nextmod; prevmod=nextmod,nextmod = nextmod->m_nextSystem )
523 {
524 if (nextmod->m_type==type)
525 { //found start of other shadows using same texture, insert new shadow here.
526 mod->m_nextSystem=nextmod;
527 mod->m_prevSystem=prevmod;
528 nextmod->m_prevSystem=mod;
529 if (prevmod)
530 { prevmod->m_nextSystem=mod;
531 }
532 else
533 m_usedModules=mod;
534 break;
535 }
536 }
537
538 if (nextmod==NULL)
539 { //shadow with new texture. Add to top of list.
540 mod->m_nextSystem = m_usedModules;
541 if (m_usedModules)
542 m_usedModules->m_prevSystem=mod;
543 m_usedModules = mod;
544 }
545
546 mod->m_bound=true;
547 } // end if
548
549 #ifdef SYNC_WAVES
550 nextmod=m_usedModules;
551
552 while(nextmod)
553 {
554 nextmod->m_elapsedMs=nextmod->m_initTimeOffset;
555 nextmod=nextmod->m_nextSystem;
556 } // end while
557 #endif
558
559 return mod;
560} //end bindTrack

583void WaterTracksRenderSystem::releaseTrack( WaterTracksObj *mod )
584{
585 if (mod==NULL)
586 return;
587
588 assert(mod->m_bound == false);
589
590 // remove module from used list
591 if( mod->m_nextSystem )
592 mod->m_nextSystem->m_prevSystem = mod->m_prevSystem;
593 if( mod->m_prevSystem )
594 mod->m_prevSystem->m_nextSystem = mod->m_nextSystem;
595 else
596 m_usedModules = mod->m_nextSystem;
597
598 // add module to free list
599 mod->m_prevSystem = NULL;
600 mod->m_nextSystem = m_freeModules;
601 if( m_freeModules )
602 m_freeModules->m_prevSystem = mod;
603 m_freeModules = mod;
604 mod->freeWaterTracksResources();
605}

640//=============================================================================
641// WaterTracksRenderSystem::ReAcquireResources
642//=============================================================================
644//=============================================================================
645void WaterTracksRenderSystem::ReAcquireResources(void)
646{
647 Int i,j,k;
648// const Int numModules=16; ///@todo: Get a value out of gdf
649
650 // just for paranoia's sake.
651 REF_PTR_RELEASE(m_indexBuffer);
652 REF_PTR_RELEASE(m_vertexBuffer);
653
654 //Will need m_y-1 strips, each of length m_x*2.
655 //Will also need 2 extra indices to connect each strip to next one (except last strip)
656 //Total index buffer size = (m_y-1)*(m_x*2+2) - 2 (drop the extra 2 indices from last strip)
657
658 Int idxCount=(m_stripSizeY-1)*(m_stripSizeX*2+2) - 2;
659
660 m_indexBuffer=NEW_REF(DX8IndexBufferClass,(idxCount));
661
662 // Fill up the IB
663 {
664 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
665 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
666
667 for (i=0,j=0,k=0; i<idxCount; j++)
668 {
669 for (;k<(m_stripSizeX*(j+1)); k++,i+=2)
670 {
671 ib[i]=(UnsignedShort) k+m_stripSizeX;
672 ib[i+1]=(UnsignedShort) k;
673 }
674 //Generate 4 degenerate triangle to connect current strip to next strip/row of map
675 //To do this, we just repeat the last index of first strip and first index of new strip.
676 //Any triangles with repeated vertices will be skipped during rendering.
677 if (i<idxCount) //check if there is at least 1 more strip to go
678 {
679 ib[i]=k-1;
680 ib[i+1]=k+m_stripSizeX;
681 i+=2;
682 }
683 }
684 }
685
686 m_vertexBuffer=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,m_stripSizeX*m_stripSizeY*WATER_VB_PAGES,DX8VertexBufferClass::USAGE_DYNAMIC));
687 m_batchStart=0;
688}

864void WaterTracksRenderSystem::flush(RenderInfoClass & rinfo)
865{
870 Int diffuseLight;
871
872 if (!TheGlobalData->m_showSoftWaterEdge || TheWaterTransparency->m_transparentWaterDepth ==0 )
873 return;
874
875 if (TheGlobalData->m_usingWaterTrackEditor)
876 TestWaterUpdate();
877
878 update(); //update positions of all the tracks
879
880 rinfo.Camera.Apply();
881
882 if (!m_usedModules || ShaderClass::Is_Backface_Culling_Inverted())
883 return; //don't render track marks in reflections.
884
885 //According to Nvidia there's a D3D bug that happens if you don't start with a
886 //new dynamic VB each frame - so we force a DISCARD by overflowing the counter.
887 m_batchStart = 0xffff;
888
889 // adjust shading for time of day.
890 Real shadeR, shadeG, shadeB;
891 shadeR = TheGlobalData->m_terrainAmbient[0].red;
892 shadeG = TheGlobalData->m_terrainAmbient[0].green;
893 shadeB = TheGlobalData->m_terrainAmbient[0].blue;
894 shadeR += TheGlobalData->m_terrainDiffuse[0].red/2;
895 shadeG += TheGlobalData->m_terrainDiffuse[0].green/2;
896 shadeB += TheGlobalData->m_terrainDiffuse[0].blue/2;
897 shadeR*=255.0f;
898 shadeG*=255.0f;
899 shadeB*=255.0f;
900
901 diffuseLight=REAL_TO_INT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16);
902
903 Matrix3D tm(1);
904 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm); //position the water surface
905
906 DX8Wrapper::Set_Material(m_vertexMaterialClass);
907 DX8Wrapper::Set_Shader(m_shaderClass);
908
909 DX8Wrapper::Set_Vertex_Buffer(m_vertexBuffer);
910 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZBIAS,8);
911 //Force apply of render states so we can override them.
912 DX8Wrapper::Apply_Render_State_Changes();
913
914 if (TheTerrainRenderObject->getShroud())
915 {
916 W3DShaderManager::setTexture(0,TheTerrainRenderObject->getShroud()->getShroudTexture());
917 W3DShaderManager::setShader(W3DShaderManager::ST_SHROUD_TEXTURE, 1);
918
919 //modulate with shroud texture
920 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLORARG1, D3DTA_TEXTURE ); //stage 1 texture
921 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLORARG2, D3DTA_CURRENT ); //previous stage texture
922 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_COLOROP, D3DTOP_MODULATE );
923 DX8Wrapper::Set_DX8_Texture_Stage_State( 1, D3DTSS_ALPHAOP, D3DTOP_MODULATE );
924
925 //Shroud shader uses z-compare of EQUAL which wouldn't work on water because it doesn't
926 //write to the zbuffer. Change to LESSEQUAL.
927 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZFUNC, D3DCMP_LESSEQUAL);
928 }
929
930 Int LastTextureType=-1;
931
932 WaterTracksObj *mod=m_usedModules;
933
934 while( mod )
935 {
936 if (LastTextureType != mod->m_type)
937 DX8Wrapper::Set_Texture(0,mod->m_stageZeroTexture);
938
939 Int vertsRendered=mod->render(m_vertexBuffer,m_batchStart);
940
941 m_batchStart = vertsRendered; //advance past vertices already in buffer
942
943 mod = mod->m_nextSystem;
944 } //while (mod)
945
946 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZBIAS,0);
947
948 if (TheTerrainRenderObject->getShroud())
949 { //we used the shroud shader, so reset it.
950 DX8Wrapper::Set_DX8_Render_State(D3DRS_ZFUNC, D3DCMP_EQUAL);
951 W3DShaderManager::resetShader(W3DShaderManager::ST_SHROUD_TEXTURE);
952 }
953}
