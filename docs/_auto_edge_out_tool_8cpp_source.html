56void AutoEdgeOutTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
57{
58 if (m != TRACK_L) return;
59
60 Coord3D cpt;
61 pView->viewToDocCoords(viewPt, &cpt);
62
63 CPoint ndx;
64 if (!pDoc->getCellIndexFromCoord(cpt, &ndx)) {
65 return;
66 }
67
68// WorldHeightMapEdit *pMap = pDoc->GetHeightMap();
69 WorldHeightMapEdit *htMapEditCopy = pDoc->GetHeightMap()->duplicate();
70 htMapEditCopy->autoBlendOut(ndx.x, ndx.y, BlendMaterial::getBlendTexClass());
71 IRegion2D partialRange = {0,0,0,0};
72 pDoc->updateHeightMap(htMapEditCopy, false, partialRange);
73 WBDocUndoable *pUndo = new WBDocUndoable(pDoc, htMapEditCopy);
74 pDoc->AddAndDoUndoable(pUndo);
75 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
76 REF_PTR_RELEASE(htMapEditCopy);
77}
