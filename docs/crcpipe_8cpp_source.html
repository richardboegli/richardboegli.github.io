43/***********************************************************************************************
44 * CRCPipe::Put -- Retrieves the data bytes specified and calculates CRC on it. *
45 * *
46 * This routine will fetch the number of bytes requested from the straw. The data is *
47 * not modified by this straw segment, but it is examined by the CRC engine in order to *
48 * keep an accurate CRC of the data that passes through this routine. *
49 * *
50 * INPUT: source -- Pointer to the buffer that will hold the data requested. *
51 * *
52 * length -- The number of bytes requested. *
53 * *
54 * OUTPUT: Returns with the actual number of bytes stored into the buffer. If this number is *
55 * less than the number requested, then this indicates that the data stream has been *
56 * exhausted. *
57 * *
58 * WARNINGS: none *
59 * *
60 * HISTORY: *
61 * 07/03/1996 JLB : Created. *
62 *=============================================================================================*/
63int CRCPipe::Put(void const * source, int slen)
64{
65 CRC(source, slen);
66 return(Pipe::Put(source, slen));
67}

70/***********************************************************************************************
71 * CRCPipe::Result -- Fetches the current CRC of the data. *
72 * *
73 * This routine will return the CRC of the data that has passed through the pipe up to *
74 * this time. *
75 * *
76 * INPUT: none *
77 * *
78 * OUTPUT: Returns with the CRC value. *
79 * *
80 * WARNINGS: none *
81 * *
82 * HISTORY: *
83 * 07/03/1996 JLB : Created. *
84 *=============================================================================================*/
85long CRCPipe::Result(void) const
86{
87 return(CRC());
88}
