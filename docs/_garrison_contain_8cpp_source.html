86inline Real calcDistSqr(const Coord3D& a, const Coord3D& b)
87{
88 return sqr(a.x - b.x) + sqr(a.y - b.y) + sqr(a.z - b.z);

94Int GarrisonContain::findClosestFreeGarrisonPointIndex( Int conditionIndex,
95 const Coord3D *targetPos )
96{
97 DEBUG_ASSERTCRASH(m_garrisonPointsInitialized, ("garrisonPoints are not inited"));
98
99 // sanity
100 if( targetPos == NULL || m_garrisonPointsInUse == MAX_GARRISON_POINTS )
101 return GARRISON_INDEX_INVALID;
102
103 Int closestIndex = GARRISON_INDEX_INVALID;
104 Real closestDistSq = -1.0f;
105 Real distSq;
106 for( Int i = 0; i < MAX_GARRISON_POINTS; ++i )
107 {
108
109 // only consider free garrison points
110 if( m_garrisonPointData[ i ].object == NULL )
111 {
112
113 // compute the squared distance between these two points
114 distSq = calcDistSqr(*targetPos, m_garrisonPoint[ conditionIndex ][ i ]);
115
116 if( distSq < closestDistSq || closestDistSq == -1.0f )
117 {
118
119 closestDistSq = distSq;
120 closestIndex = i;
121
122 } // end if
123
124 } // end if
125
126 } // end for i
127
128 return closestIndex;
129
130} // end findClosestFreeGarrisonPointIndex

164

220/*

223 obj->getTemplate()->getName().str(),
224 obj->getID(),
225 pointIndex );
226TheInGameUI->message( msg );

235Int GarrisonContain::findConditionIndex( void )
236{
237 BodyModuleInterface *body = getObject()->getBodyModule();
238 BodyDamageType bodyDamage = body->getDamageState();
239 Int index = GARRISON_INDEX_INVALID;
240
241 switch( bodyDamage )
242 {
243
244 // --------------------------------------------------------------------------------------------
245 case BODY_PRISTINE:
246
247 index = GARRISON_POINT_PRISTINE;
248 break;
249
250 // --------------------------------------------------------------------------------------------
251 case BODY_DAMAGED:
252
253 index = GARRISON_POINT_DAMAGED;
254 break;
255
256 // --------------------------------------------------------------------------------------------
257 case BODY_REALLYDAMAGED:
258 case BODY_RUBBLE:
259
260 index = GARRISON_POINT_REALLY_DAMAGED;
261 break;
262
263 // --------------------------------------------------------------------------------------------
264 default:
265
266 DEBUG_CRASH(( "GarrisonContain::findConditionIndex - Unknown body damage type '%d'\n",
267 bodyDamage ));
268 break;
269
270 } // end switch
271
272 return index;
273

280Bool GarrisonContain::calcBestGarrisonPosition( Coord3D *sourcePos, const Coord3D *targetPos )
281{
282 // sanity
283 if( !sourcePos || !targetPos )
284 return FALSE;
285
286#if defined __DEBUG || defined _INTERNAL
287 const GarrisonContainModuleData *modData = getGarrisonContainModuleData();
288 DEBUG_ASSERTCRASH(modData->m_isEnclosingContainer, ("calcBestGarrisonPosition... SHOULD NOT GET HERE, since this container is non-enclosing") );
289#endif
290
291
292
293 // find which garrison point position array we will used based on body condition
294 Int conditionIndex = findConditionIndex();
295
296 // get the index of the garrison point that is closest to the target position
297 Int placeIndex = findClosestFreeGarrisonPointIndex( conditionIndex, targetPos );
298 if( placeIndex == GARRISON_INDEX_INVALID )
299 {
300 DEBUG_CRASH( ("GarrisonContain::calcBestGarrisonPosition - Unable to find suitable garrison point.\n") );
301 return FALSE;
302 }
303
304 sourcePos->set( &(m_garrisonPoint[ conditionIndex ][ placeIndex ]) );
305 return TRUE;
306}

309//The AI is entering the aim state and would like to move the unit to the best position, perform
310//a range check, and if it succeeds, leave him there -- otherwise, remove him immediately.
311//-------------------------------------------------------------------------------------------------
312Bool GarrisonContain::attemptBestFirePointPosition( Object *source, Weapon *weapon, Object *victim )
313{
314 //Sanity
315 if( !source || !victim || !weapon )
316 {
317 return FALSE;
318 }
319
320#if defined __DEBUG || defined _INTERNAL
321 const GarrisonContainModuleData *modData = getGarrisonContainModuleData();
322 DEBUG_ASSERTCRASH(modData->m_isEnclosingContainer, ("calcBestGarrisonPosition... SHOULD NOT GET HERE, since this container is non-enclosing") );
323#endif
324 //If this object is already at a garrison point, remove him.
325 Int existingIndex = getObjectGarrisonPointIndex( source );
326 if( existingIndex != GARRISON_INDEX_INVALID )
327 {
328 removeObjectFromGarrisonPoint( source, existingIndex );
329 }
330
331 putObjectAtBestGarrisonPoint( source, victim, NULL );
332
333 //Okay, now we have positioned the object in the best position for the victim.
334 //Now check if we are able to fire on our victim.
335 if( weapon->isWithinAttackRange( source, victim ) )
336 {
337 return TRUE;
338 }
339
340 //Crap, we failed... so remove the object from the garrison point.
341 existingIndex = getObjectGarrisonPointIndex( source );
342 if( existingIndex != GARRISON_INDEX_INVALID )
343 {
344 removeObjectFromGarrisonPoint( source, existingIndex );
345 }
346 return FALSE;
347}

395{
396

398 if( obj == NULL || (target == NULL && targetPos == NULL) )

430void GarrisonContain::removeObjectFromGarrisonPoint( Object *obj, Int index )
431{
432
433 if ( ! isEnclosingContainerFor(obj) )
434 return;// since I am not enclosed, I am not at a garrison point!
435
436#if defined __DEBUG || defined _INTERNAL
437 const GarrisonContainModuleData *modData = getGarrisonContainModuleData();
438 DEBUG_ASSERTCRASH(modData->m_isEnclosingContainer, ("calcBestGarrisonPosition... SHOULD NOT GET HERE, since this container is non-enclosing") );
439#endif
440
441 // sanity
442 if( obj == NULL )
443 return;
444
445 // search for the object in the garrison point data, if found, remove it
446 Int removeIndex = index;
447 if( removeIndex == SEARCH_FOR_REMOVE )
448 {
449
450 for( Int i = 0; i < MAX_GARRISON_POINTS; ++i )
451 {
452
453 if( m_garrisonPointData[ i ].object == obj )
454 {
455
456 removeIndex = i;
457 break;
458
459 } // end if
460
461 } // end for i
462
463 } // end if
464
465 // validate the index slot to remove
466 if( removeIndex < 0 || removeIndex >= MAX_GARRISON_POINTS )
467 {
468
469 //
470 // this is not an error, if a search was ordered, we may very well not find the
471 // object at a garrison point and therefore can't remove it
472 //
473 return;
474
475 } // end if
476
477 // remove from this spot
478 m_garrisonPointData[ removeIndex ].object = NULL;
479 m_garrisonPointData[ removeIndex ].targetID = INVALID_ID;
480 m_garrisonPointData[ removeIndex ].placeFrame = 0;
481 m_garrisonPointData[ removeIndex ].lastEffectFrame = 0;
482 --m_garrisonPointsInUse;
483
484 // destroy drawable for gun barrel and effects if present
485 if( m_garrisonPointData[ removeIndex ].effect )
486 TheGameClient->destroyDrawable( m_garrisonPointData[ removeIndex ].effect );
487 m_garrisonPointData[ removeIndex ].effect = NULL;
488
489 // set the position of the object to back to the center of the garrisoned building
490 obj->setPosition( getObject()->getPosition() );
491
492/*
493UnicodeString msg;
494msg.format( L"Removed object '%S'(%d) from point '%d'",
495 obj->getTemplate()->getName().str(),
496 obj->getID(),
497 removeIndex );
498TheInGameUI->message( msg );
499*/
500
501} // end removeObjectFromGarrisonPoint

553Bool GarrisonContain::isValidContainerFor(const Object* obj, Bool checkCapacity) const
554{
555
556 // extend functionality // this just tests kindof masks
557 if( OpenContain::isValidContainerFor( obj, checkCapacity ) == false )
558 return false;
559
560 // zero-health buildings are not garrisonable.
561 if (getObject()->getBodyModule()->getHealth() <= 0.0f)
562 return false;
563
564 // ReallyDamaged buildings are not garrisonable as well.
565 if( getObject()->getBodyModule()->getDamageState() == BODY_REALLYDAMAGED && !getObject()->isKindOf( KINDOF_GARRISONABLE_UNTIL_DESTROYED ) )
566 return false;
567
568 if (obj && !obj->isKindOf(KINDOF_NO_GARRISON))
569 {
570 if (checkCapacity)
571 {
572 Int garrisonMax = getContainMax();
573 Int containCount = getContainCount();
574 return ( containCount < garrisonMax );
575 }
576 else
577 {
578 return true;
579 }
580 }
581 return false;
582}

682void GarrisonContain::trackTargets( void )
683{
684
685
686 if ( ! isEnclosingContainerFor( 0 ) )
687 return; // since ina non-enclosing container, objects fire from their station points, instead of being juggled around between garrison firepoints
688
689
690
691 Int conditionIndex = findConditionIndex();
692 const ContainedItemsList& containList = getContainList();
693 AIUpdateInterface *ai;
694 Object *obj;
695
696 for( ContainedItemsList::const_iterator it = containList.begin(); it != containList.end(); ++it )
697 {
698
699 DEBUG_ASSERTCRASH(m_garrisonPointsInitialized, ("garrisonPoints are not inited"));
700
701 // get the object
702 obj = *it;
703
704 // only consider objects that are actually at garrison points for re-shuffling
705 Int ourIndex = getObjectGarrisonPointIndex( obj );
706 if( ourIndex != GARRISON_INDEX_INVALID )
707 {
708
709 // does this object have a target?
710 ai = obj->getAIUpdateInterface();
711 if( ai )
712 {
713 Object *victim = ai->getCurrentVictim();
714 // even though the target position can't change in some cases, still must do this code at least once.
715 const Coord3D *victimPos = ai->getCurrentVictimPos();
716
717 if( victim || victimPos )
718 {
719 if (victim)
720 victimPos = victim->getPosition();
721 const Coord3D *ourPos = obj->getPosition();
722
723 // find the closest free (of all remaining) garrison points to our target
724 Int newIndex = findClosestFreeGarrisonPointIndex( conditionIndex,
725 victimPos );
726
727 // if unable to find another garrison point, don't bother
728 if( newIndex != GARRISON_INDEX_INVALID )
729 {
730
731 // get the distance from our current index to the target
732 Real currentDistSq = calcDistSqr(*victimPos, *ourPos );
733
734 // get the distance from the newly chosen index
735 Real newDistSq = calcDistSqr(*victimPos, m_garrisonPoint[ conditionIndex ][ newIndex ] );
736
737 // if the newly chosen index is closer than our current index, switch
738 if( newDistSq < currentDistSq )
739 {
740
741 // remove from the old index
742 removeObjectFromGarrisonPoint( obj, ourIndex );
743
744 // place at the new index
745 putObjectAtGarrisonPoint( obj, victim ? victim->getID() : INVALID_ID, conditionIndex, newIndex );
746
747 } // end if, new index is closer
748
749 } // end if, possible closer index was found
750
751 //
752 // we are now either at a new garrison fire point, or we have remained at our
753 // existing point still tracking our target. Orient the effect drawable which
754 // shows the gun barrel and muzzle flash towards our target position
755 //
756 if( m_garrisonPointData[ ourIndex ].effect )
757 {
758 Coord2D v;
759 v.x = victimPos->x - ourPos->x;
760 v.y = victimPos->y - ourPos->y;
761// v.z = victomPos->z - ourPos.z;
762
763 // orient the effect object towards the victim position
764 m_garrisonPointData[ ourIndex ].effect->setOrientation( v.toAngle() );
765
766 } // end if
767
768 } // end if, victim present
769
770 } // end if, ai
771
772 } // end if, we're at a garrison point
773
774 } // end for it
775
776} // end trackTargets

788 Int i;

1497void GarrisonContain::exitObjectViaDoor( Object *exitObj, ExitDoorType exitDoor )
1498{
1499 DEBUG_ASSERTCRASH(exitDoor == DOOR_1, ("multiple exit doors not supported here"));
1500
1501 // We don't use the ExitPath system of the general OpenContain, we just send people out. The
1502 // direction of outing has been picked by Design to be the Screen Down at the default camera angle.
1503 removeFromContain( exitObj );
1504
1505 Coord3D startPosition;
1506 Coord3D endPosition;
1507
1508 Real exitAngle = getObject()->getOrientation();
1509
1510 // Garrison doesn't have reserveDoor or exitDelay, so if we do nothing, everyone will appear on top
1511 // of each other and get stuck inside each others' extent (except for the first guy). So we'll
1512 // scatter the start point around a little to make it better.
1513 startPosition = *getObject()->getPosition();
1514 // In the case of cliff bunkers, the units start in a cliff. So we want to adjust.
1515 AIUpdateInterface *ai = exitObj->getAI();
1516 if (ai) {
1517 Locomotor *loco = ai->getCurLocomotor();
1518 if (loco && !TheAI->pathfinder()->validMovementTerrain( LAYER_GROUND, loco, &startPosition)) {
1519 // try front & back.
1520 Real offset = getObject()->getGeometryInfo().getMajorRadius();
1521 startPosition.x -= offset*Cos(exitAngle);
