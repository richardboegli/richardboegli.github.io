88void SpyVisionSpecialPower::doSpecialPower( UnsignedInt commandOptions )
89{
90 if (getObject()->isDisabled())
91 return;
92
93 // call the base class action cause we are *EXTENDING* functionality
94 SpecialPowerModule::doSpecialPower( commandOptions );
95
96
97 // get the source of this power
98 Object *source = getObject();
99
100 // get module data
101 const SpyVisionSpecialPowerModuleData *modData = getSpyVisionSpecialPowerModuleData();
102
103 //
104 // the SpyVision special power increases in range and duration of effect the more
105 // units we have captured within us
106 //
107 UnsignedInt duration = modData->m_baseDurationInFrames;
108 ContainModuleInterface *contain = source->getContain();
109 if( contain )
110 {
111 // for every captured unit we get a bonus
112 duration += contain->getContainCount() * modData->m_bonusDurationPerCapturedInFrames;
113
114 // cap at the max
115 if( duration > modData->m_maxDurationInFrames )
116 duration = modData->m_maxDurationInFrames;
117 }
118
119 static const NameKeyType key_SpyVisionUpdate = NAMEKEY( "SpyVisionUpdate" );
120 SpyVisionUpdate *update = (SpyVisionUpdate*)source->findUpdateModule( key_SpyVisionUpdate );
121 if( !update )
122 {
123 return;
124 }
125
126 update->activateSpyVision( duration );
127}
