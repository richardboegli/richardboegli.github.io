71void QueueProductionExitUpdate::exitObjectViaDoor( Object *newObj, ExitDoorType exitDoor )
72{
73 DEBUG_ASSERTCRASH(exitDoor == DOOR_1, ("multiple exit doors not supported here"));
74
75 Object *creationObject = getObject();
76 if (creationObject)
77 {
78 const QueueProductionExitUpdateModuleData* md = getQueueProductionExitUpdateModuleData();
79
80 Real exitAngle = creationObject->getOrientation();
81 const Matrix3D *transform = creationObject->getTransformMatrix();
82 Vector3 loc;
83 Coord3D createPoint;
84
85 //
86 // calculate the position to create the object at, we take the coord specified
87 // in INI which is in model space, rotate it to match the building angle
88 // and translate for building location via a transform call
89 //
90 loc.Set( md->m_unitCreatePoint.x, md->m_unitCreatePoint.y, md->m_unitCreatePoint.z );
91 transform->Transform_Vector( *transform, loc, &loc );
92
93 Bool creationInAir = FALSE;
94 if( TheTerrainLogic )
95 {
96 if( loc.Z != TheTerrainLogic->getGroundHeight( loc.X, loc.Y ) )
97 creationInAir = TRUE;
98 }
99 // make sure the point is on the terrain
100 if( creationInAir && !md->m_allowAirborneCreationData )
101 loc.Z = TheTerrainLogic ? TheTerrainLogic->getGroundHeight( loc.X, loc.Y ) : 0.0f;
102
103 // we need it in Coord3D form
104 createPoint.x = loc.X;
105 createPoint.y = loc.Y;
106 createPoint.z = loc.Z;
107
108 newObj->setPosition( &createPoint );
109 newObj->setOrientation( exitAngle );
110
111 //
112 // Objects that are created in the air from producers that are in the air get
113 // a kick that will make their starting speed equal mine.
114 //
115 PhysicsBehavior *newObjectPhysics = newObj->getPhysics();
116 PhysicsBehavior *myPhysics = creationObject->getPhysics();
117
118 if( (myPhysics != NULL) && creationInAir && (newObjectPhysics != NULL) )
119 {
120 Coord3D startingForce = *myPhysics->getVelocity();
121 startingForce.x *= newObjectPhysics->getMass();
122 startingForce.y *= newObjectPhysics->getMass();
123 startingForce.z *= newObjectPhysics->getMass();
124 newObjectPhysics->applyMotiveForce( &startingForce );
125
126 const Real STARTING_PITCH_COEFF = 0.04f;
127 Real pitchRate = newObjectPhysics->getCenterOfMassOffset() * STARTING_PITCH_COEFF;
128 newObjectPhysics->setPitchRate( pitchRate );
129 }
130
131
134 // tell the AI about it
135 TheAI->pathfinder()->addObjectToPathfindMap( newObj );
136 Coord3D tmp;
137 getNaturalRallyPoint(tmp);
138 // Grid it.
139 TheAI->pathfinder()->snapPosition(newObj, &tmp);
140 std::vector<Coord3D> exitPath;
141 exitPath.push_back(tmp);
142
143 AIUpdateInterface *ai = newObj->getAIUpdateInterface();
144 if (m_rallyPointExists)
145 {
146 tmp = m_rallyPoint;
147 if (ai && ai->isDoingGroundMovement())
148 {
149 if (TheAI->pathfinder()->adjustDestination(newObj, ai->getLocomotorSet(), &tmp))
150 exitPath.push_back(tmp);
151
152 }
153 } else {
154 // Double the destination to keep redguards from stacking.
155 exitPath.push_back(tmp);
156 }
157 if (ai) {
158 ai->aiFollowExitProductionPath( &exitPath, creationObject, CMD_FROM_AI );
159 }
160 m_currentDelay = md->m_exitDelayData;
161
162 if (m_currentBurstCount)
163 m_currentBurstCount--; // fewer and fewer units to burst
164
165 }
166
167}

170Bool QueueProductionExitUpdate::getExitPosition( Coord3D& exitPosition ) const
171{
172 const Object *obj = getObject();
173 if (!obj)
174 return FALSE;
175
176 const Matrix3D *transform = obj->getTransformMatrix();
177
178 const QueueProductionExitUpdateModuleData *md = getQueueProductionExitUpdateModuleData();
179
180 Vector3 loc;
181 loc.Set( md->m_unitCreatePoint.x, md->m_unitCreatePoint.y, md->m_unitCreatePoint.z );
182 transform->Transform_Vector( *transform, loc, &loc );
183
184 exitPosition.x = loc.X;
185 exitPosition.y = loc.Y;
186 exitPosition.z = loc.Z;
187
188 return TRUE;
189
190}

206Bool QueueProductionExitUpdate::isFreeToExit() const
207{
208 Bool stillBursting = (m_currentBurstCount > 0);
209 Bool stillDelaying = (m_currentDelay != 0);
210
211 if ( stillBursting )
212 {
213 return TRUE;
214 }
215
216 return ( ! stillDelaying );
217}

234void QueueProductionExitUpdate::exitObjectByBudding( Object *newObj, Object *budHost )
235{
236 if (budHost) // object has provided a host for us
237 {
238 const Coord3D *newPos = budHost->getPosition();

240 PathfindLayerEnum newLayer = budHost->getLayer();
241

245
246 }
247 else // we need to just copy the host object
248 {
249 newObj->setPosition( getObject()->getPosition() );
250 newObj->setOrientation( getObject()->getOrientation() );
251 }
252
253 //This causes the formation to break like billard balls
254 AIUpdateInterface *ai = newObj->getAI();
255 if (ai)
256 {
257 ai->aiMoveToPosition( newObj->getPosition(), CMD_FROM_AI );// IOW, I cant stay here, if somebody else is here

259
260 const QueueProductionExitUpdateModuleData* data = getQueueProductionExitUpdateModuleData();
261 m_currentDelay = data->m_exitDelayData;
262
263 if (m_currentBurstCount)
264 m_currentBurstCount--; // fewer and fewer units to burst

266}

272Bool QueueProductionExitUpdate::getNaturalRallyPoint( Coord3D& rallyPoint, Bool offset ) const
273{
274 const QueueProductionExitUpdateModuleData *data = getQueueProductionExitUpdateModuleData();
275 Vector3 p;
276
277 //
278 // get the natural rally point from the INI definition, this coord is in model space relative
279 // to the model (0,0,0)
280 //
281 p.X = data->m_naturalRallyPoint.x;
282 p.Y = data->m_naturalRallyPoint.y;
283 p.Z = data->m_naturalRallyPoint.z;
284
285 if ( offset )
286 {
287 Vector3 offset = p;
288 offset.Normalize();
289 offset *= (2*PATHFIND_CELL_SIZE_F);
290 p+=offset;
291 }
292
293 // transform the point into world space
294 const Matrix3D *transform = getObject()->getTransformMatrix();
295 transform->Transform_Vector( *transform, p, &p );
296 rallyPoint.x = p.X; rallyPoint.y = p.Y; rallyPoint.z = p.Z;
297 return TRUE;
298}
