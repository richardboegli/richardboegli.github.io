67Real ArmorTemplate::adjustDamage(DamageType t, Real damage) const
68{
69 if (t == DAMAGE_UNRESISTABLE)
70 return damage;
71 if (t == DAMAGE_SUBDUAL_UNRESISTABLE)
72 return damage;
73
74 damage *= m_damageCoefficient[t];
75
76 if (damage < 0.0f)
77 damage = 0.0f;
78
79 return damage;
80}

83/*static*/ void ArmorTemplate::parseArmorCoefficients( INI* ini, void *instance, void* /* store */, const void* userData )
84{
85 ArmorTemplate* self = (ArmorTemplate*) instance;
86
87 const char* damageName = ini->getNextToken();
88 Real pct = INI::scanPercentToReal(ini->getNextToken());
89
90 if (stricmp(damageName, "Default") == 0)
91 {
92 for (Int i = 0; i < DAMAGE_NUM_TYPES; i++)
93 {
94 self->m_damageCoefficient[i] = pct;
95 }
96 return;
97 }
98
99 DamageType dt = (DamageType)DamageTypeFlags::getSingleBitFromName(damageName);
100 self->m_damageCoefficient[dt] = pct;
101}

120{
121 NameKeyType namekey = TheNameKeyGenerator->nameToKey(name);
122 ArmorTemplateMap::const_iterator it = m_armorTemplates.find(namekey);
123 if (it == m_armorTemplates.end())
124 {
125 return NULL;
126 }
127 else
128 {
