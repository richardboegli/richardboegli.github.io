139{
140 m_UIDirty = TRUE;
141
142#if defined( _INTERNAL ) || defined( _DEBUG )
143 UnsignedInt now = TheGameLogic->getFrame();
144 if( now == m_lastFrameMarkedDirty )
145 {
146 //Do nothing.
147 }
148 else if( now == m_lastFrameMarkedDirty + 1 )
149 {
150 m_consecutiveDirtyFrames++;
151 }
152 else
153 {
154 m_consecutiveDirtyFrames = 1;
155 }
156 m_lastFrameMarkedDirty = now;
157

622Bool CommandButton::isValidToUseOn(const Object *sourceObj, const Object *targetObj, const Coord3D *targetLocation, CommandSourceType commandSource) const
623{
624 if (m_upgradeTemplate) {
625 // @todo: Make a const version of pui. We're not altering the production queue, so this const-cast
626 // is okay.
627 ProductionUpdateInterface *pui = const_cast<Object*>(sourceObj)->getProductionUpdateInterface();
628 if (pui) {
629 const ProductionEntry *pe = pui->firstProduction();
630 while (pe) {
631 if (pe->getProductionUpgrade() != NULL)
632 return false;
633 pe = pui->nextProduction(pe);
634 }
635 return sourceObj->affectedByUpgrade(m_upgradeTemplate) && !sourceObj->hasUpgrade(m_upgradeTemplate);
636 }
637 // No ProductionUpdateInterface means we can't do this.
638 return false;
639 }
640
641 if( BitTest( m_options, COMMAND_OPTION_NEED_OBJECT_TARGET ) && !targetObj )
642 {
643 return false;
644 }
645
646 Coord3D pos;
647 if( targetLocation )
648 {
649 pos.set( targetLocation );
650 }
651
652 if( BitTest( m_options, NEED_TARGET_POS ) && !targetLocation )
653 {
654 if( targetObj )
655 {
656 pos.set( targetObj->getPosition() );
657 }
658 else
659 {
660 return false;
661 }
662 }
663
664 if( BitTest( m_options, COMMAND_OPTION_NEED_OBJECT_TARGET ) )
665 {
666 return TheActionManager->canDoSpecialPowerAtObject( sourceObj, targetObj, commandSource, m_specialPower, m_options, false );
667 }
668
669 if( BitTest( m_options, NEED_TARGET_POS ) )
670 {
671 return TheActionManager->canDoSpecialPowerAtLocation( sourceObj, &pos, commandSource, m_specialPower, NULL, m_options, false );
672 }
673
674 return TheActionManager->canDoSpecialPower( sourceObj, m_specialPower, commandSource, m_options, false );
675}

794void CommandSet::parseCommandButton( INI* ini, void *instance, void *store, const void *userData )
795{
796 const char *token = ini->getNextToken();
797
798 // get find the command button from this name
799 const CommandButton *commandButton = TheControlBar->findCommandButton( AsciiString( token ) );
800 if( commandButton == NULL )
801 {
802
803 DEBUG_CRASH(( "[LINE: %d - FILE: '%s'] Unknown command '%s' found in command set\n",
804 ini->getLineNum(), ini->getFilename().str(), token ));
805 throw INI_INVALID_DATA;
806
807 } // end if
808
809 // get the index to store the command at, and the command array itself
810 const CommandButton **buttonArray = (const CommandButton **)store;
811 Int buttonIndex = (Int)userData;
812
813 // sanity
814 DEBUG_ASSERTCRASH( buttonIndex < MAX_COMMANDS_PER_SET, ("parseCommandButton: button index '%d' out of range\n",
815 buttonIndex) );
816
817 // save it
818 buttonArray[ buttonIndex ] = commandButton;
819
820} // end parseCommand

1713void ControlBar::evaluateContextUI( void )
1714{
1715
1716 //
1717 // the UI has been "evaluated" and is now displaying the most current and correct
1718 // information to the player
1719 //
1720 m_UIDirty = FALSE;
1721
1722 // if our purchase science window is up, we will want to update it by repopulating it.
1723 if(!m_contextParent[ CP_PURCHASE_SCIENCE ]->winIsHidden())
1724 showPurchaseScience();
1725
1726 // erase any current state of the GUI by switching out to the empty context
1727 switchToContext( CB_CONTEXT_NONE, NULL );
1728
1729 // sanity, nothing selected
1730 if( TheInGameUI->getSelectCount() == 0 )
1731 return;
1732
1733 // get the list of drawable IDs from the in game UI
1734 const DrawableList *selectedDrawables = TheInGameUI->getAllSelectedDrawables();
1735
1736 // sanity
1737 if( selectedDrawables->empty() == TRUE )
1738 return;
1739
1740 //Make sure the selected objects are in fact, controllable! If not, then
1741 //we don't show any GUI commands for them!!!
1742 //This is used when we select enemy objects or objects on another team.
1743 //@todo we may want to show their portrait
1744 if( !TheInGameUI->areSelectedObjectsControllable() )
1745 {
1746 //Also make sure the unit isn't a garrisonable neutral civ team building!
1747 Drawable *draw = selectedDrawables->front();
1748
1749 //sanity
1750 if( !draw )
1751 {
1752 return;
1753 }
1754 Object *obj = draw->getObject();
1755 if( !obj )
1756 {
1757 return;
1758 }
1759
1760 if (obj->getControllingPlayer()
1761 && obj->getControllingPlayer()->getPlayerTemplate()
1762 && obj->getControllingPlayer()->getPlayerTemplate()->getBeaconTemplate().compare(obj->getTemplate()->getName()) == 0
1763 )
1764 {
1765 switchToContext( CB_CONTEXT_BEACON, draw );
1766 }
1767 else
1768 {
1769 switchToContext( CB_CONTEXT_NONE, draw );
1770 }
1771
1772 //Check for a contain interface and a enemy relationship and reject that!
1773 ContainModuleInterface *contain = obj->getContain();
1774 if( contain && contain->getContainMax() > 0 )
1775 {
1776
1777 const Player *otherPlayer = contain->getApparentControllingPlayer(ThePlayerList->getLocalPlayer());
1778 if (!otherPlayer)
1779 otherPlayer = obj->getControllingPlayer();
1780 Player *player = ThePlayerList->getLocalPlayer();
1781
1782 if( !player || !otherPlayer )
1783 {
1784 //Sanity.
1785 return;
1786 }
1787 Relationship relation = player->getRelationship( otherPlayer->getDefaultTeam() );
1788
1789 //Note: All following checks already account for the fact that this object
1790 //isn't ours.
1791
1792 //The only case we can actually see a non-controlled controlbar is a neutral garrisonable structure.
1793 if( !contain->isGarrisonable() || relation != NEUTRAL )
1794 {
1795 //Can't peek inside enemy/allied containers period!
1796 return;
1797 }
1798 }
1799 else
1800 {
1801 return;
1802 }
1803 }
1804
1805 //
1806 // when we have multiple things selected, we will only display the common commands
1807 // in the center command bar that can be displayed with multi-units selected
1808 //
1809
1810
1811 Drawable *drawToEvaluateFor = NULL;
1812 Bool multiSelect = FALSE;
1813
1814
1815 if( TheInGameUI->getSelectCount() > 1 )
1816 {
1817 // Attempt to isolate a Drawable here to evaluate
1818 // The need arises when selected is an AngryMob,
1819 // whose selection actually consists of varied units
1820 // but is represented in the UI as a single unit,
1821 // so we must isolate and evaluate only the Nexus
1822 drawToEvaluateFor = TheGameClient->findDrawableByID( TheInGameUI->getSoloNexusSelectedDrawableID() ) ;
1823 multiSelect = ( drawToEvaluateFor == NULL );
1824
1825 }
1826 else // get the first and only drawble in the selection list
1827 drawToEvaluateFor = selectedDrawables->front();
1828
1829
1830
1831 if( multiSelect )
1832 {
1833 switchToContext( CB_CONTEXT_MULTI_SELECT, NULL );
1834 }
1835 else if ( drawToEvaluateFor )// either we have exactly one drawable, or we have isolated one to evaluate for...
1836 {
1837
1838 // get the first and only drawble in the selection list
1839 //Drawable *draw = selectedDrawables->front();
1840
1841 // sanity
1842 //if( draw == NULL )
1843 // return;
1844
1845 // get object
1846 Object *obj = drawToEvaluateFor->getObject();
1847 if( obj == NULL )
1848 return;
1849
1850 // we show no interface for objects being sold
1851 if( obj->getStatusBits().test( OBJECT_STATUS_SOLD ) )
1852 return;
1853
1854 static const NameKeyType key_OCLUpdate = NAMEKEY( "OCLUpdate" );
1855 OCLUpdate *update = (OCLUpdate*)obj->findUpdateModule( key_OCLUpdate );
1856
1857 //
1858 // a command center is context sensitive itself, if a side has *NOT* been chosen we display
1859 // the side select interface for command centers only, but note how under construction is
1860 // more important than anything
1861 //
1862 Bool contextSelected = FALSE;
1863 if( obj->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
1864 {
1865
1866 switchToContext( CB_CONTEXT_UNDER_CONSTRUCTION, drawToEvaluateFor );
1867 contextSelected = TRUE;
1868
1869 } // end else if
1870
1871 // check for a regular switch to the appropriate context
1872 if( contextSelected == FALSE )
1873 {
1874 ContainModuleInterface *cmi = obj->getContain();
1875
1876 if( cmi && cmi->isGarrisonable() && obj->getCommandSetString().isEmpty() )
1877 {
1878 //Kris: This is a convenient section to graft an inventory commandset for
1879 //garrisoned troops. However, we only want to use this if we DON'T have
1880 //a commandset defined. If we do, then trust that the commandset will
1881 //handle it!
1882
1883 Player *localPlayer = ThePlayerList->getLocalPlayer();
1884 Relationship relationship;
1885
1886 // we cannot select objects that are controlled by our enemies
1887 relationship = localPlayer->getRelationship( obj->getTeam() );
1888 if( obj->isLocallyControlled() == TRUE || relationship == NEUTRAL )
1889 switchToContext( CB_CONTEXT_STRUCTURE_INVENTORY, drawToEvaluateFor );
1890
1891 } // end else if
1892 else if( update )
1893 {
1894 switchToContext( CB_CONTEXT_OCL_TIMER, drawToEvaluateFor );
1895 }
1896 else if( obj->getCommandSetString().isEmpty() == FALSE )
1897 {
1898
1899 switchToContext( CB_CONTEXT_COMMAND, drawToEvaluateFor );
1900
1901 } // end else if
1902 else if (obj->getControllingPlayer()->getPlayerTemplate()->getBeaconTemplate().compare(obj->getTemplate()->getName()) == 0)
1903 {
1904 switchToContext( CB_CONTEXT_BEACON, drawToEvaluateFor );
1905 }
1906 else
1907 switchToContext( CB_CONTEXT_NONE, drawToEvaluateFor );
1908 } // end else
1909
1910 } // end else
1911
1912} // end evaluateContextUI
