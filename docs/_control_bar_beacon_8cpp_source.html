44void ControlBar::populateBeacon( Object *beacon )
45{
46
47 // set the portrait for the thing being constructed
48 setPortraitByObject( beacon );
49
50 static NameKeyType textID = NAMEKEY("ControlBar.wnd:EditBeaconText");
51 static NameKeyType staticTextID = NAMEKEY("ControlBar.wnd:StaticTextBeaconLabel");
52 static NameKeyType clearButtonID = NAMEKEY("ControlBar.wnd:ButtonClearBeaconText");
53
54 GameWindow *textEntryWin = TheWindowManager->winGetWindowFromId(NULL, textID);
55 GameWindow *staticTextWin = TheWindowManager->winGetWindowFromId(NULL, staticTextID);
56 GameWindow *buttonWin = TheWindowManager->winGetWindowFromId(NULL, clearButtonID);
57
58 if (beacon->isLocallyControlled())
59 {
60 if (textEntryWin)
61 {
62 textEntryWin->winHide(FALSE);
63 GadgetTextEntrySetText( textEntryWin, beacon->getDrawable()->getCaptionText() );
64 TheWindowManager->winSetFocus( textEntryWin );
65 }
66
67 if (staticTextWin)
68 staticTextWin->winHide(FALSE);
69
70 if (buttonWin)
71 buttonWin->winHide(FALSE);
72 }
73 else
74 {
75 if (textEntryWin)
76 textEntryWin->winHide(TRUE);
77
78 if (staticTextWin)
79 staticTextWin->winHide(TRUE);
80
81 if (buttonWin)
82 buttonWin->winHide(TRUE);
83 }
84} // end populateBeacon
