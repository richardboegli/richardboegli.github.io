130void PopupSlider::GetChannelRect(CRect* rect)
131{
132 GetClientRect(rect);
133 if(SB_HORZ == m_kind) {
134 rect->InflateRect(-CHANNEL_X_INSET_AMOUNT, 0);
135 rect->top = PIXELS_ABOVE_SLIDER + CHANNEL_BELOW_THUMB_TOP;
136 rect->bottom = THUMB_ICON_HEIGHT - (CHANNEL_BELOW_THUMB_TOP + CHANNEL_ABOVE_THUMB_BOTTOM);
137 rect->bottom += rect->top;
138 } else {
139 rect->InflateRect(0, -CHANNEL_X_INSET_AMOUNT);
140 rect->left = PIXELS_ABOVE_SLIDER + CHANNEL_BELOW_THUMB_TOP + 2;
141 rect->right = THUMB_ICON_HEIGHT - (CHANNEL_BELOW_THUMB_TOP + CHANNEL_ABOVE_THUMB_BOTTOM);
142 rect->right += rect->left;
143 }
144}

146void PopupSlider::GetThumbIconRect(CRect* rect)
147{
148 CRect channelRect;
149 GetChannelRect(&channelRect);
150
151 int range = m_hi - m_lo;
152
153 int along = m_curValue - m_lo;
154
155 if(SB_HORZ == m_kind) {
156 int thumbMiddle = along * channelRect.Width();
157 thumbMiddle /= range;
158 thumbMiddle += channelRect.left;
159
160 if (thumbMiddle < channelRect.left) thumbMiddle = channelRect.left;
161 if (thumbMiddle > channelRect.right) thumbMiddle = channelRect.right;
162
163 rect->top = channelRect.top - CHANNEL_BELOW_THUMB_TOP;
164 rect->bottom = channelRect.bottom + CHANNEL_ABOVE_THUMB_BOTTOM;
165 rect->left = thumbMiddle - (THUMB_ICON_WIDTH / 2);
166 rect->right = rect->left + THUMB_ICON_WIDTH;
167 } else {
168 int thumbMiddle = along * channelRect.Height();
169 thumbMiddle /= range;
170 thumbMiddle = channelRect.bottom - thumbMiddle;
171
172 if (thumbMiddle < channelRect.top) thumbMiddle = channelRect.top;
173 if (thumbMiddle > channelRect.bottom) thumbMiddle = channelRect.bottom;
174
175 rect->top = thumbMiddle - (THUMB_ICON_WIDTH / 2);
176 rect->bottom = thumbMiddle + (THUMB_ICON_WIDTH/2);
177 rect->left = channelRect.left - CHANNEL_BELOW_THUMB_TOP;
178 rect->right = channelRect.right + CHANNEL_ABOVE_THUMB_BOTTOM;
179 }
180}

182void PopupSlider::MoveThumbUnderMouse(int xNew)
183{
184 CRect channelRect;
185 GetChannelRect(&channelRect);
186
187 int newVal = m_hi - m_lo;
188 if(SB_HORZ == m_kind) {
189 int along = xNew - channelRect.left;
190
191 newVal *= along;
192 newVal /= channelRect.Width();
193 newVal += m_lo;
194 } else {
195 int along = channelRect.bottom - xNew;
196
197 newVal *= along;
198 newVal /= channelRect.Height();
199 newVal += m_lo;
200 }
201 if (newVal < m_lo) newVal = m_lo;
202 if (newVal > m_hi) newVal = m_hi;
203
204 CRect iconRect;
205 GetThumbIconRect(&iconRect);
206 iconRect.InflateRect(3, 3);
207 InvalidateRect(&iconRect, FALSE);
208
209 m_curValue = newVal;
210
211 GetThumbIconRect(&iconRect);
212 iconRect.InflateRect(3, 3);
213 InvalidateRect(&iconRect, FALSE);
214}

220void PopupSlider::New(CWnd *pParentWnd, long kind,
221 PopupSliderOwner *pSliderOwner,
222 long sliderID)
223{
224 PopupSlider * pPopupSlider;
225
226 DEBUG_ASSERTCRASH(((SB_HORZ == kind) || (SB_VERT == kind)),
227 ("PopupSlider - unexpected kind of slider!"));
228
229 DEBUG_ASSERTCRASH(pSliderOwner, ("slider owner is NULL!"));
230
231 try {
232 CRect rect;
233
234 pParentWnd->GetWindowRect(&rect);
235
236 rect.left = rect.right;
237 pPopupSlider = new PopupSlider();
238
239 pPopupSlider->mSliderOwner = pSliderOwner;
240 pPopupSlider->mSliderID = sliderID;
241 pPopupSlider->m_kind = kind;
242
243 pSliderOwner->GetPopSliderInfo(pPopupSlider->mSliderID,
244 &(pPopupSlider->m_lo),
245 &(pPopupSlider->m_hi),
246 &(pPopupSlider->m_lineSize),
247 &(pPopupSlider->m_curValue));
248
249 DEBUG_ASSERTCRASH(pPopupSlider->m_hi != pPopupSlider->m_lo, ("PopupSlider: endpoint values are the same!"));
250 DEBUG_ASSERTCRASH(pPopupSlider->m_lineSize != 0, ("PopupSlider: line size is zero!"));
251
252 pPopupSlider->Create(rect, pParentWnd);
253
254 /* if the slider is successfully created, it will
255 be deleted automatically by its PostNcDestroy
256 member function */
257 } catch (...) {
258 // don't rethrow
259 if (pPopupSlider) {
260 delete pPopupSlider;
261 pPopupSlider = NULL;
262 }
263
264 } // catch
265
266 gPopupSlider = pPopupSlider;
267 // gPopupSlider will be deleted when its PostNcDestroy method is called
268}

314BOOL PopupSlider::Create(const RECT& rect, CWnd* pParentWnd)
315{
316 BOOL retVal;
317
318 try {
319 if (FALSE == m_brush3dFaceColor.CreateSolidBrush(GetSysColor(COLOR_3DFACE))) {
320 throw(-1);
321 }
322
323 mIcon = (HICON) LoadImage(AfxGetResourceHandle(),
324 MAKEINTRESOURCE(IDI_Thumb),
325 IMAGE_ICON, 0, 0, LR_LOADMAP3DCOLORS);
326
327
328 DWORD dwExStyle = WS_EX_TOPMOST;
329 DWORD dwStyle = WS_POPUP;
330 UINT nClassStyle = CS_HREDRAW | CS_VREDRAW | CS_BYTEALIGNCLIENT | CS_SAVEBITS;
331 HCURSOR hCursor = ::LoadCursor(NULL, IDC_ARROW);
332 CString className = AfxRegisterWndClass(nClassStyle, hCursor, (HBRUSH) m_brush3dFaceColor);
333
334 long winWidth, winHeight;
335
336 winWidth = ((SB_HORZ == m_kind) ? POPUP_SLIDER_BIGDIMENSION : POPUP_SLIDER_SMALLDIMENSION);
337 winHeight = ((SB_HORZ == m_kind) ? POPUP_SLIDER_SMALLDIMENSION : POPUP_SLIDER_BIGDIMENSION);
338
339 CRect winRect(rect.right - winWidth, rect.bottom, rect.right, rect.bottom + winHeight);
340
341 // we'll just use "this" for the child ID
342 if (FALSE == CWnd::CreateEx(dwExStyle, (LPCTSTR) className, "",
343 dwStyle, winRect.left, winRect.top,
344 winRect.Width(), winRect.Height(),
345 pParentWnd->GetSafeHwnd(),
346 NULL, NULL))
347 throw(-1);
348
349
350 // New code to center the slider's thumb under the parent window
351 if(pParentWnd) {
352 // Calculate the center of the parent window
353 CPoint parentWindowCenter;
354 GetCursorPos(&parentWindowCenter);
355
356 // Calculate the center of the thumb
357 CRect iconRect;
358 GetThumbIconRect(&iconRect);
359 CPoint iconCenter = iconRect.CenterPoint();
360 ClientToScreen(&iconCenter);
361
362 // from the centers, calculate how far to move the window
363 int hAdjustToCenter = 0;
364 int vAdjustToCenter = 0;
365 if(SB_HORZ == m_kind)
366 hAdjustToCenter = parentWindowCenter.x - iconCenter.x;
367 // This may work for vertical sliders, but has not been tested
368 if(SB_VERT == m_kind)
369 vAdjustToCenter = parentWindowCenter.y - iconCenter.y;
370
371 // Move the window
372 CRect myWindowRect;
373 GetWindowRect(&myWindowRect);
374 myWindowRect.OffsetRect(hAdjustToCenter, vAdjustToCenter);
375 SetWindowPos(NULL, myWindowRect.left, myWindowRect.top, myWindowRect.Width(), myWindowRect.Height(), SWP_NOACTIVATE | SWP_NOZORDER | SWP_NOSIZE | SWP_NOREDRAW);
376 }
377
378 // finally, make sure the window appears on screen
379 //MakeSureWindowIsVisible(GetSafeHwnd());
380 ShowWindow(SW_SHOW);
381
382 SetCapture();
383
384 // success (we'll set the capture after all the failable operations)
385 retVal = TRUE;
386 } catch (...) {
387 // don't rethrow
388 retVal = FALSE;
389 } // catch
390
391 return retVal;
392}

394void PopupSlider::PostNcDestroy()
395{
396 if (mSliderOwner && (m_valOnLastFinished != m_curValue)) {
397 mSliderOwner->PopSliderFinished(mSliderID, m_curValue);
398 }
399
400 CWnd::PostNcDestroy();
401
402 // now that the window has gone away, delete ourselves
403 if (gPopupSlider == this) {
404 delete gPopupSlider;
405 gPopupSlider = NULL;
406 }

410{

413 CBrush cbr(GetSysColor(COLOR_BTNFACE));

449void PopupSlider::OnLButtonDown(UINT nFlags, CPoint point)
450{
451nFlags;
452 CRect rc;
453 GetClientRect(&rc);
454 if (!mSetOrigPt) {
455 mOrigPt = point;
456 }
457 mSetOrigPt = true;
458
459
460 if (rc.PtInRect((POINT) point)) {
461 CRect iconRect;
462 GetThumbIconRect(&iconRect);
463
464 // 990909: inflate the rect a little to make it easier to grab the thumb
465 if(SB_HORZ == m_kind) {
466 iconRect.InflateRect(3, 6);
467 } else {
468 iconRect.InflateRect(6, 3);
469 }
470
471 if (iconRect.PtInRect((POINT) point)) {
472 mDraggingThumb = true;
473 } else {
474 CRect channelRect;
475 GetChannelRect(&channelRect);
476
477 if (channelRect.PtInRect((POINT) point)) {
478 if(SB_HORZ == m_kind) {
479 MoveThumbUnderMouse(point.x);
480 } else {
481 MoveThumbUnderMouse(point.y);
482 }
483 mDraggingThumb = true;
484 mEverMoved = true;
485 }
486 }
487 } else {
488 try {
489 if (mSliderOwner) {
490 mSliderOwner->PopSliderFinished(mSliderID, m_curValue);
491 }
492
493 m_valOnLastFinished = m_curValue;
494 } catch (...) {
495 }
496
497 // user clicked outside our area, close the windoid
498 PostMessage(WM_CLOSE, 0, 0);
499 }

502void PopupSlider::OnLButtonUp(UINT nFlags, CPoint point)
503{
504 if (mDraggingThumb) {
505 mDraggingThumb = false;
506
507 if (mSliderOwner) {
508 try {
509 mSliderOwner->PopSliderChanged(mSliderID, m_curValue);
510 if (mClickThrough && mEverMoved) {
511 mSliderOwner->PopSliderFinished(mSliderID, m_curValue);
512 }
513
514 m_valOnLastFinished = m_curValue;
515 } catch (...) {
516 }
517 }
518
519 if (mClickThrough && mEverMoved) {
520 PostMessage(WM_CLOSE, 0, 0);
521 }
522 } else {
523 CWnd::OnLButtonUp(nFlags, point);
524 }
525 mClickThrough = false;
526}

537void PopupSlider::OnMouseMove(UINT nFlags, CPoint point)
538{
539 if (!mSetOrigPt) {
540 mOrigPt = point;
541 }
542 mSetOrigPt = true;
543 if (mDraggingThumb) {
544 mEverMoved = true;
545 if(SB_HORZ == m_kind) {
546 MoveThumbUnderMouse(point.x);
547 } else {
548 MoveThumbUnderMouse(point.y);
549 }
550
551 if (mSliderOwner) {
552 mSliderOwner->PopSliderChanged(mSliderID, m_curValue);
553 }
554 } else if (nFlags & MK_LBUTTON) {
555 CRect rc;
556 GetChannelRect(&rc);
557 if(SB_HORZ == m_kind) {
558 rc.InflateRect(0, 6);
559 } else {
560 rc.InflateRect(6, 0);
561 }
562
563 if (rc.PtInRect(point)) {
564 // user just clicked thru, so mark the slider for click through
565 mDraggingThumb = true;
566
567 if(SB_HORZ == m_kind) {
568 if (mOrigPt.x != point.x) {
569 MoveThumbUnderMouse(point.x);
570 }
571 mClickThrough = true;
572 } else {
573 if (mOrigPt.y != point.y) {
574 MoveThumbUnderMouse(point.y);
575 }
576 mClickThrough = true;
577 }
578
579 if (mSliderOwner) {
580 mSliderOwner->PopSliderChanged(mSliderID, m_curValue);
581 }
582 } // if (PtInRect)
583 }
584}
