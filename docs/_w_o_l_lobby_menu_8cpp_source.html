149Bool handleLobbySlashCommands(UnicodeString uText)
150{
151 AsciiString message;
152 message.translate(uText);
153
154 if (message.getCharAt(0) != '/')
155 {
156 return FALSE; // not a slash command
157 }
158
159 AsciiString remainder = message.str() + 1;
160 AsciiString token;
161 remainder.nextToken(&token);
162 token.toLower();
163
164 if (token == "host")
165 {
166 UnicodeString s;
167 s.format(L"Hosting qr2:%d thread:%d", getQR2HostingStatus(), isThreadHosting);
168 TheGameSpyInfo->addText(s, GameSpyColor[GSCOLOR_DEFAULT], NULL);
169 return TRUE; // was a slash command
170 }
171 else if (token == "me" && uText.getLength()>4)
172 {
173 TheGameSpyInfo->sendChat(UnicodeString(uText.str()+4), TRUE, listboxLobbyPlayers);
174 return TRUE; // was a slash command
175 }
176 else if (token == "refresh")
177 {
178 // Added 2/19/03 added the game refresh
179 refreshGameList(TRUE);
180 refreshPlayerList(TRUE);
181 return TRUE; // was a slash command
182 }
183 /*
184 if (token == "togglegamelist")
185 {
186 NameKeyType buttonID = NAMEKEY("WOLCustomLobby.wnd:ButtonGameListToggle");
187 GameWindow *button = TheWindowManager->winGetWindowFromId(parent, buttonID);
188 if (button)
189 {
190 button->winHide(!button->winIsHidden());
191 }
192 return TRUE; // was a slash command
193 }
194 else if (token == "adjustchat")
195 {
196 NameKeyType sliderID = NAMEKEY("WOLCustomLobby.wnd:SliderChatAdjust");
197 GameWindow *slider = TheWindowManager->winGetWindowFromId(parent, sliderID);
198 if (slider)
199 {
200 slider->winHide(!slider->winIsHidden());
201 }
202 return TRUE; // was a slash command
203 }
204 */
205#if defined(_INTERNAL) || defined(_DEBUG)
206 else if (token == "fakecrc")
207 {
208 g_fakeCRC = !g_fakeCRC;
209 TheGameSpyInfo->addText(UnicodeString(L"Toggled CRC fakery"), GameSpyColor[GSCOLOR_DEFAULT], NULL);
210 return TRUE; // was a slash command
211 }
212 else if (token == "slots")
213 {
214 g_debugSlots = !g_debugSlots;
215 TheGameSpyInfo->addText(UnicodeString(L"Toggled SlotList debug"), GameSpyColor[GSCOLOR_DEFAULT], NULL);
216 return TRUE; // was a slash command
217 }
218#endif
219
220 return FALSE; // not a slash command
221}

231static void playerTooltip(GameWindow *window,
232 WinInstanceData *instData,

234{
235 Int x, y, row, col;
236 x = LOLONGTOSHORT(mouse);

242 {
243 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString);//TheGameText->fetch("TOOLTIP:PlayersInLobby") );

247 UnicodeString uName = GadgetListBoxGetText(window, row, COLUMN_PLAYERNAME);
248 AsciiString aName;
249 aName.translate(uName);
250
251 PlayerInfoMap::iterator it = TheGameSpyInfo->getPlayerInfoMap()->find(aName);

253 Bool isLocalPlayer = (TheGameSpyInfo->getLocalName().compareNoCase(info->m_name) == 0);
254
255 if (col == 0)
256 {
257 if (info->m_preorder)
258 {
259 TheMouse->setCursorTooltip( TheGameText->fetch("TOOLTIP:LobbyOfficersClub") );
260 }
261 else
262 {
263 TheMouse->setCursorTooltip( UnicodeString::TheEmptyString);
264 }
265 return;
266 }
267
268 AsciiString playerLocale = info->m_locale;
269 AsciiString localeIdentifier;
270 localeIdentifier.format("WOL:Locale%2.2d", atoi(playerLocale.str()));
271 Int playerWins = info->m_wins;
272 Int playerLosses = info->m_losses;
273 UnicodeString playerInfo;
274 playerInfo.format(TheGameText->fetch("TOOLTIP:PlayerInfo"), TheGameText->fetch(localeIdentifier).str(), playerWins, playerLosses);
275
276 UnicodeString tooltip = UnicodeString::TheEmptyString;//TheGameText->fetch("TOOLTIP:PlayersInLobby");
277 if (isLocalPlayer)
278 {
279 tooltip.format(TheGameText->fetch("TOOLTIP:LocalPlayer"), uName.str());
280 }
281 else
282 {
283 // not us
284 if (TheGameSpyInfo->getBuddyMap()->find(info->m_profileID) != TheGameSpyInfo->getBuddyMap()->end())
285 {
286 // buddy
287 tooltip.format(TheGameText->fetch("TOOLTIP:BuddyPlayer"), uName.str());
288 }
289 else
290 {
291 if (info->m_profileID)
292 {
293 // non-buddy profiled player
294 tooltip.format(TheGameText->fetch("TOOLTIP:ProfiledPlayer"), uName.str());
295 }
296 else
297 {
298 // non-profiled player
299 tooltip.format(TheGameText->fetch("TOOLTIP:GenericPlayer"), uName.str());
300 }
301 }
302 }
303
304 if (info->isIgnored())
305 {
306 tooltip.concat(TheGameText->fetch("TOOLTIP:IgnoredModifier"));
307 }
308
309 if (info->m_profileID)
310 {
311 tooltip.concat(playerInfo);
312 }
313
314 Int rank = 0;
315 Int i = 0;
316 while( info->m_rankPoints >= TheRankPointValues->m_ranks[i + 1])
317 ++i;
318 rank = i;
319 AsciiString sideName = "GUI:RandomSide";
320 if (info->m_side > 0)
321 {
322 const PlayerTemplate *fac = ThePlayerTemplateStore->getNthPlayerTemplate(info->m_side);
323 if (fac)
324 {
325 sideName.format("SIDE:%s", fac->getSide().str());
326 }
327 }
328 AsciiString rankName;
329 rankName.format("GUI:GSRank%d", rank);
330 UnicodeString tmp;
331 tmp.format(L"\n%ls %ls", TheGameText->fetch(sideName).str(), TheGameText->fetch(rankName).str());
332 tooltip.concat(tmp);
333
334 TheMouse->setCursorTooltip( tooltip, -1, NULL, 1.5f ); // the text and width are the only params used. the others are the default values.
335}

490void PopulateLobbyPlayerListbox(void)
491{
492
493 if (!listboxLobbyPlayers)
494 return;
495
496 // Display players
497 PlayerInfoMap *players = TheGameSpyInfo->getPlayerInfoMap();
498 PlayerInfoMap::iterator it;
499 BuddyInfoMap *buddies = TheGameSpyInfo->getBuddyMap();
500 BuddyInfoMap::iterator bIt;
501 if (listboxLobbyPlayers)
502 {
503 // save off old selection
504 Int maxSelectedItems = GadgetListBoxGetNumEntries(listboxLobbyPlayers);
505 Int *selectedIndices;
506 GadgetListBoxGetSelected(listboxLobbyPlayers, (Int *)(&selectedIndices));
507 std::set<AsciiString> selectedNames;
508 std::set<AsciiString>::const_iterator selIt;
509 std::set<Int> indicesToSelect;
510 UnicodeString uStr;
511 Int numSelected = 0;
512 for (Int i=0; i<maxSelectedItems; ++i)
513 {
514 if (selectedIndices[i] < 0)
515 {
516 break;
517 }
518 ++numSelected;
519 AsciiString selectedName;
520 uStr = GadgetListBoxGetText(listboxLobbyPlayers, selectedIndices[i], COLUMN_PLAYERNAME);
521 selectedName.translate(uStr);
522 selectedNames.insert(selectedName);
523 DEBUG_LOG(("Saving off old selection %d (%s)\n", selectedIndices[i], selectedName.str()));
524 }
525
526 // save off old top entry
527 Int previousTopIndex = GadgetListBoxGetTopVisibleEntry(listboxLobbyPlayers);
528
529 GadgetListBoxReset(listboxLobbyPlayers);
530
531 // Ops
532 for (it = players->begin(); it != players->end(); ++it)
533 {
534 PlayerInfo info = it->second;
535 if (info.m_flags & PEER_FLAG_OP || TheGameSpyConfig->isPlayerVIP(info.m_profileID))
536 {
537 Int index = insertPlayerInListbox(info, info.isIgnored()?GameSpyColor[GSCOLOR_PLAYER_IGNORED]:GameSpyColor[GSCOLOR_PLAYER_OWNER]);
538
539 selIt = selectedNames.find(info.m_name);
540 if (selIt != selectedNames.end())
541 {
542 DEBUG_LOG(("Marking index %d (%s) to re-select\n", index, info.m_name.str()));
543 indicesToSelect.insert(index);
544 }
545 }
546 }
547
548 // Buddies
549 for (it = players->begin(); it != players->end(); ++it)
550 {
551 PlayerInfo info = it->second;
552 bIt = buddies->find(info.m_profileID);
553 if ( !(info.m_flags & PEER_FLAG_OP || TheGameSpyConfig->isPlayerVIP(info.m_profileID)) && bIt != buddies->end() )
554 {
555 Int index = insertPlayerInListbox(info, info.isIgnored()?GameSpyColor[GSCOLOR_PLAYER_IGNORED]:GameSpyColor[GSCOLOR_PLAYER_BUDDY]);
556
557 selIt = selectedNames.find(info.m_name);
558 if (selIt != selectedNames.end())
559 {
560 DEBUG_LOG(("Marking index %d (%s) to re-select\n", index, info.m_name.str()));
561 indicesToSelect.insert(index);
562 }
563 }
564 }
565
566 // Everyone else
567 for (it = players->begin(); it != players->end(); ++it)
568 {
569 PlayerInfo info = it->second;
570 bIt = buddies->find(info.m_profileID);
571 if ( !(info.m_flags & PEER_FLAG_OP || TheGameSpyConfig->isPlayerVIP(info.m_profileID)) && bIt == buddies->end() )
572 {
573 Int index = insertPlayerInListbox(info, info.isIgnored()?GameSpyColor[GSCOLOR_PLAYER_IGNORED]:GameSpyColor[GSCOLOR_PLAYER_NORMAL]);
574
575 selIt = selectedNames.find(info.m_name);
576 if (selIt != selectedNames.end())
577 {
578 DEBUG_LOG(("Marking index %d (%s) to re-select\n", index, info.m_name.str()));
579 indicesToSelect.insert(index);
580 }
581 }
582 }
583
584 // restore selection
585 if (indicesToSelect.size())
586 {
587 std::set<Int>::const_iterator indexIt;
588 Int *newIndices = NEW Int[indicesToSelect.size()];
589 for (i=0, indexIt = indicesToSelect.begin(); indexIt != indicesToSelect.end(); ++i, ++indexIt)
590 {
591 newIndices[i] = *indexIt;
592 DEBUG_LOG(("Queueing up index %d to re-select\n", *indexIt));
593 }
594 GadgetListBoxSetSelected(listboxLobbyPlayers, newIndices, indicesToSelect.size());
595 delete[] newIndices;
596 }
597
598 if (indicesToSelect.size() != numSelected)
599 {
600 TheWindowManager->winSetLoneWindow(NULL);
601 }
602
603 // restore top visible entry
604 GadgetListBoxSetTopVisibleEntry(listboxLobbyPlayers, previousTopIndex);
605 }
