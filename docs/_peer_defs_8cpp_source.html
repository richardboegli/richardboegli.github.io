176void GameSpyInfo::setGameOptions( void )
177{
178 if (!m_isHosting)
179 return;
180
181 // set options for game lists, and UTM players in-game
182 PeerRequest req;
183 req.peerRequestType = PeerRequest::PEERREQUEST_SETGAMEOPTIONS;
184 req.options = GameInfoToAsciiString(&m_localStagingRoom).str();
185
186 Int i;
187 AsciiString mapName = TheGameState->realMapPathToPortableMapPath(m_localStagingRoom.getMap());
188 AsciiString newMapName;
189 for (i=0; i<mapName.getLength(); ++i)
190 {
191 char c = mapName.getCharAt(i);
192 if (c != '\\')
193 newMapName.concat(c);
194 else
195 newMapName.concat('/');
196 }
197 req.gameOptsMapName = newMapName.str();
198
199 req.gameOptions.numPlayers = 0;
200 req.gameOptions.numObservers = 0;
201 Int numOpenSlots = 0;
202 AsciiString playerInfo = "";
203 for (i=0; i<MAX_SLOTS; ++i)
204 {
205 Int wins = 0, losses = 0, profileID = 0;
206 GameSpyGameSlot *slot = TheGameSpyGame->getGameSpySlot(i);
207 req.gameOptsPlayerNames[i] = "";
208 if (!slot->isOccupied())
209 {
210 if (slot->isOpen())
211 ++numOpenSlots;
212 }
213 else
214 {
215 AsciiString playerName;
216 if (slot->isHuman())
217 {
218 playerName.translate(slot->getName());
219 req.gameOptsPlayerNames[i] = playerName.str();
220 PlayerInfoMap::iterator it = m_playerInfoMap.find(playerName);
221 if (it != m_playerInfoMap.end())
222 {
223 wins = it->second.m_wins;
224 losses = it->second.m_losses;
225 profileID = it->second.m_profileID;
226 }
227 req.gameOptions.wins[req.gameOptions.numObservers+req.gameOptions.numPlayers] = wins;
228 req.gameOptions.losses[req.gameOptions.numObservers+req.gameOptions.numPlayers] = losses;
229 req.gameOptions.profileID[req.gameOptions.numObservers+req.gameOptions.numPlayers] = profileID;
230 req.gameOptions.faction[req.gameOptions.numObservers+req.gameOptions.numPlayers] = slot->getPlayerTemplate();
231 req.gameOptions.color[req.gameOptions.numObservers+req.gameOptions.numPlayers] = slot->getColor();
232 if (slot->getPlayerTemplate() == PLAYERTEMPLATE_OBSERVER)
233 {
234 ++req.gameOptions.numObservers;
235 }
236 else
237 {
238 ++req.gameOptions.numPlayers;
239 }
240 }
241 else if (slot->isAI())
242 {
243 // add in AI players
244 switch (slot->getState())
245 {
246 case SLOT_EASY_AI:
247 playerName = "CE";
248 break;
249 case SLOT_MED_AI:
250 playerName = "CM";
251 break;
252 case SLOT_BRUTAL_AI:
253 playerName = "CH";
254 break;
255 }
256 req.gameOptsPlayerNames[i] = playerName.str(); // name is unused - we go off of the profileID
257 req.gameOptions.wins[req.gameOptions.numObservers+req.gameOptions.numPlayers] = 0;
258 req.gameOptions.losses[req.gameOptions.numObservers+req.gameOptions.numPlayers] = 0;
259 req.gameOptions.profileID[req.gameOptions.numObservers+req.gameOptions.numPlayers] = slot->getState();
260 req.gameOptions.faction[req.gameOptions.numObservers+req.gameOptions.numPlayers] = slot->getPlayerTemplate();
261 req.gameOptions.color[req.gameOptions.numObservers+req.gameOptions.numPlayers] = slot->getColor();
262 ++req.gameOptions.numPlayers;
263 }
264 }
265 }
266 req.gameOptions.maxPlayers = numOpenSlots + req.gameOptions.numPlayers + req.gameOptions.numObservers;
267 TheGameSpyPeerMessageQueue->addRequest(req);
268
269 req.peerRequestType = PeerRequest::PEERREQUEST_UTMROOM;
270 req.UTM.isStagingRoom = TRUE;
271 req.id = "Pings/";
272 AsciiString pings;
273 for (i=0; i<MAX_SLOTS; ++i)
274 {
275 if (i!=0)
276 pings.concat(",");
277
278 GameSpyGameSlot *slot = TheGameSpyGame->getGameSpySlot(i);
279 if (slot && slot->isHuman())
280 {
281 pings.concat(slot->getPingString());
282 }
283 else
284 {
285 pings.concat("0");
286 }
287 }
288 req.options = pings.str();
289 TheGameSpyPeerMessageQueue->addRequest(req);
290}

297void GameSpyInfo::addGroupRoom( GameSpyGroupRoom room )
298{
299 if (room.m_groupID == 0)
300 {
301 m_gotGroupRoomList = TRUE;
302
303 GroupRoomMap::iterator iter;
304
305 // figure out how many good strings we've got
306 std::vector<UnicodeString> names;
307 Int numRooms = 0;
308 for (iter = getGroupRoomList()->begin(); iter != getGroupRoomList()->end(); ++iter)
309 {
310 GameSpyGroupRoom room = iter->second;
311 if (room.m_groupID != TheGameSpyConfig->getQMChannel())
312 {
313 ++numRooms;
314
315 AsciiString groupLabel;
316 groupLabel.format("GUI:%s", room.m_name.str());
317
318 Bool exists = FALSE;
319 UnicodeString groupName = TheGameText->fetch(groupLabel, &exists);
320 if (exists)
321 {
322 names.push_back(groupName);
323 }
324 }
325 }
326
327 if (!names.empty() && names.size() != numRooms)
328 {
329 // didn't get all names. fix up
330 Int nameIndex = 0;
331 Int timesThrough = 1; // start with USA Lobby 1
332 for (iter = TheGameSpyInfo->getGroupRoomList()->begin(); iter != TheGameSpyInfo->getGroupRoomList()->end(); ++iter)
333 {
334 GameSpyGroupRoom room = iter->second;
335 if (room.m_groupID != TheGameSpyConfig->getQMChannel())
336 {
337 room.m_translatedName.format(L"%ls %d", names[nameIndex].str(), timesThrough);
338 nameIndex = (nameIndex+1)%names.size();
339 m_groupRooms[room.m_groupID] = room;
340 if (!nameIndex)
341 {
342 // we've looped through the name list already. increment the timesThrough counter
343 ++timesThrough;
344 }
345 }
346 }
347 }
348 }
349 else
350 {
351 DEBUG_LOG(("Adding group room %d (%s)\n", room.m_groupID, room.m_name.str()));
352 AsciiString groupLabel;
353 groupLabel.format("GUI:%s", room.m_name.str());
354 room.m_translatedName = TheGameText->fetch(groupLabel);
355 m_groupRooms[room.m_groupID] = room;
356 if ( !stricmp("quickmatch", room.m_name.str()) )
357 {
358 DEBUG_LOG(("Group room %d (%s) is the QuickMatch room\n", room.m_groupID, room.m_name.str()));
359 TheGameSpyConfig->setQMChannel(room.m_groupID);
360 }
361 }
362}

385void GameSpyInfo::joinBestGroupRoom( void )
386{
387 if (m_currentGroupRoomID)
388 {
389 DEBUG_LOG(("Bailing from GameSpyInfo::joinBestGroupRoom() - we were already in a room\n"));
390 m_currentGroupRoomID = 0;
391 return;
392 }
393
394 if (m_groupRooms.size())
395 {
396 int minID = -1;
397 int minPlayers = 1000;
398 GroupRoomMap::iterator iter = m_groupRooms.begin();
399 while (iter != m_groupRooms.end())
400 {
401 GameSpyGroupRoom room = iter->second;
402 DEBUG_LOG(("Group room %d: %s (%d, %d, %d, %d)\n", room.m_groupID, room.m_name.str(), room.m_numWaiting, room.m_maxWaiting,
403 room.m_numGames, room.m_numPlaying));
404
405 if (TheGameSpyConfig->getQMChannel() != room.m_groupID && minPlayers > 25 && room.m_numWaiting < minPlayers)
406 {
407 minID = room.m_groupID;
408 minPlayers = room.m_numWaiting;
409 }
410
411 ++iter;
412 }
413
414 if (minID > 0)
415 {
416 PeerRequest req;
417 req.peerRequestType = PeerRequest::PEERREQUEST_JOINGROUPROOM;
418 req.groupRoom.id = minID;
419 TheGameSpyPeerMessageQueue->addRequest(req);
420 m_playerInfoMap.clear();
421 }
422 else
423 {
424 GSMessageBoxOk(TheGameText->fetch("GUI:Error"), TheGameText->fetch("GUI:GSGroupRoomJoinFail"), NULL);
425 }
426 }
427 else
428 {
429 GSMessageBoxOk(TheGameText->fetch("GUI:Error"), TheGameText->fetch("GUI:GSGroupRoomJoinFail"), NULL);
430 }

528void GameSpyInfo::markAsStagingRoomHost( void )
529{
530 m_localStagingRoomID = 0;
531 m_joinedStagingRoom = FALSE; m_isHosting = TRUE;
532
533 // There are a few options we don't want to reset when we are hosting (they carry over
534 // from the the create game dialog).
535 // Interesting fact: oldFactionsOnly will be carried over correctly if I remove these
536 // lines. UseStats won't be. I have no idea why.
537 Int useStats = m_localStagingRoom.getUseStats();
538 Bool oldFactionsOnly = m_localStagingRoom.oldFactionsOnly();
539
540 m_localStagingRoom.reset();
541 m_localStagingRoom.enterGame();
542 m_localStagingRoom.setSeed(GetTickCount());
543
544 m_localStagingRoom.setUseStats( useStats );
545 m_localStagingRoom.setOldFactionsOnly( oldFactionsOnly );
546
547 GameSlot newSlot;
548 UnicodeString uName;
549 uName.translate(m_localName);
550 newSlot.setState(SLOT_PLAYER, uName);
551
552 m_localStagingRoom.setLocalIP(m_externalIP);
553 newSlot.setIP(m_externalIP);
554
555 m_localStagingRoom.setSlot(0,newSlot);
556 m_localStagingRoom.setLocalName(m_localName);
557
558 TheMapCache->updateCache();
559 m_localStagingRoom.setMap(getDefaultMap(TRUE));
560 m_localStagingRoom.adjustSlotsForMap(); // close slots that the map can't hold. BGC
561}

563void GameSpyInfo::markAsStagingRoomJoiner( Int game )
564{
565 m_localStagingRoomID = game;
566 m_joinedStagingRoom = TRUE; m_isHosting = FALSE;
567 m_localStagingRoom.reset();
568 m_localStagingRoom.enterGame();
569 StagingRoomMap::iterator it = m_stagingRooms.find(game);
570 if (it != m_stagingRooms.end())
571 {
572 GameSpyStagingRoom *info = it->second;
573 info->cleanUpSlotPointers();
574 AsciiString options = GameInfoToAsciiString(info);
575#ifdef DEBUG_CRASHING
576 Bool res =
577#endif
578 ParseAsciiStringToGameInfo(&m_localStagingRoom, options);
579 DEBUG_ASSERTCRASH(res, ("Could not parse game info \"%s\"", options.str()));
580 m_localStagingRoom.setInGame();
581 m_localStagingRoom.setLocalName(m_localName);
582 m_localStagingRoom.setExeCRC(info->getExeCRC());
583 m_localStagingRoom.setIniCRC(info->getIniCRC());
584 m_localStagingRoom.setAllowObservers(info->getAllowObservers());
585 m_localStagingRoom.setHasPassword(info->getHasPassword());
586 m_localStagingRoom.setGameName(info->getGameName());
587 DEBUG_LOG(("Joining game: host is %ls\n", m_localStagingRoom.getConstSlot(0)->getName().str()));
588 }

659void TearDownGameSpy( void )
660{
661 // save off cached stats
662 if (TheGameSpyInfo && TheGameSpyInfo->getLocalProfileID())
663 {
664// /* This was done on the score screen, so there is no need to do it now.
665// *
666 PSPlayerStats localPSStats = TheGameSpyPSMessageQueue->findPlayerStatsByID(TheGameSpyInfo->getLocalProfileID());
667 if (localPSStats.id != 0)
668 {
669 GameSpyMiscPreferences mPref;
670 mPref.setCachedStats(GameSpyPSMessageQueueInterface::formatPlayerKVPairs(localPSStats).c_str());
671 mPref.write();
672 }
673// */
674 }
675
676 // End our threads before we kill off the singletons they reference. No crashy-crash for you!
677 if (TheGameSpyPSMessageQueue)
678 TheGameSpyPSMessageQueue->endThread();
679 if (TheGameSpyBuddyMessageQueue)
680 TheGameSpyBuddyMessageQueue->endThread();
681 if (TheGameSpyPeerMessageQueue)
682 TheGameSpyPeerMessageQueue->endThread();
683 if (ThePinger)
684 ThePinger->endThreads();
685
686 if(TheRankPointValues)
687 {
688 delete TheRankPointValues;
689 TheRankPointValues = NULL;
690 }
691 if (TheGameSpyPSMessageQueue)
692 {
693 delete TheGameSpyPSMessageQueue;
694 TheGameSpyPSMessageQueue = NULL;
695 }
696
697 if (TheGameSpyBuddyMessageQueue)
698 {
699 delete TheGameSpyBuddyMessageQueue;
700 TheGameSpyBuddyMessageQueue = NULL;
701 }
702
703 if (TheGameSpyPeerMessageQueue)
704 {
705 delete TheGameSpyPeerMessageQueue;
706 TheGameSpyPeerMessageQueue = NULL;
707 }
708
709 if (TheGameSpyInfo)
710 {
711 if (TheGameSpyInfo->getInternalIP())
712 {
713 // we've logged in before. mark us as logging out.
714 SignalUIInteraction(SHELL_SCRIPT_HOOK_GENERALS_ONLINE_LOGOUT);
715 }
716 delete TheGameSpyInfo;
717 TheGameSpyInfo = NULL;

798Int GameSpyInfo::getPingValue( const AsciiString& otherPing )
799{
800 if (m_pingString.getLength() != otherPing.getLength())
801 {
802 return TheGameSpyConfig->getPingTimeoutInMs();
803 }
804
805 if (m_pingString.getLength() % 2 != 0)
806 {
807 return TheGameSpyConfig->getPingTimeoutInMs();
808 }
809
810 Int best = 255+255;
811 const char *myStr = m_pingString.str();
812 const char *otherStr = otherPing.str();
813
814 while (*myStr)
815 {
816 Int myVal = grabHexInt(myStr);
817 Int otherVal = grabHexInt(otherStr);
818 Int val = myVal + otherVal;
819 best = (val < best) ? val : best;
820 myStr += 2;
821 otherStr += 2;
822 }
823
824 return best * TheGameSpyConfig->getPingTimeoutInMs() / (255+255);
825}

874void GameSpyInfo::updateAdditionalGameSpyDisconnections(Int count)
875{
876 if (TheRecorder->isMultiplayer() && TheGameLogic->isInInternetGame() && TheGameSpyGame && TheGameSpyGame->getUseStats())
877 {
878 Int localID = TheGameSpyInfo->getLocalProfileID();
879 PSPlayerStats stats = TheGameSpyPSMessageQueue->findPlayerStatsByID(localID);
880
881 Player *player=ThePlayerList->getLocalPlayer();
882
883 Int ptIdx;
884 const PlayerTemplate *myTemplate = player->getPlayerTemplate();
885 DEBUG_LOG(("myTemplate = %X(%s)\n", myTemplate, myTemplate->getName().str()));
886 for (ptIdx = 0; ptIdx < ThePlayerTemplateStore->getPlayerTemplateCount(); ++ptIdx)
887 {
888 const PlayerTemplate *nthTemplate = ThePlayerTemplateStore->getNthPlayerTemplate(ptIdx);
889 DEBUG_LOG(("nthTemplate = %X(%s)\n", nthTemplate, nthTemplate->getName().str()));
890 if (nthTemplate == myTemplate)
891 {
892 break;
893 }
894 }
895
896 Bool anyAI = FALSE;
897 for (Int i=0; i<MAX_SLOTS; ++i)
898 {
899 const GameSlot *slot = TheGameInfo->getConstSlot(i);
900
901 if (slot->isAI())
902 {
903 anyAI = TRUE;
904 }
905 }
906
907 //Check for cases where we're not tracking stats.
908 if (anyAI || stats.id == 0 || myTemplate->isObserver() || player->getPlayerType() != PLAYER_HUMAN || player->isPlayerObserver())
909 return;
910
911 Int disCons=stats.discons[ptIdx];
912 disCons += count;
913 if (disCons < 0)
914 { DEBUG_LOG(("updateAdditionalGameSpyDisconnections() - disconnection count below zero\n"));
915 return; //something is wrong here
916 }
917 stats.discons[ptIdx] = disCons; //add an additional disconnection to their stats.
918
919 //Add an additional disconnection to player stats.
920 PSRequest req;
921
922 req.requestType = PSRequest::PSREQUEST_UPDATEPLAYERSTATS;
923 req.email = TheGameSpyInfo->getLocalEmail().str();
924 req.nick = TheGameSpyInfo->getLocalBaseName().str();
925 req.password = TheGameSpyInfo->getLocalPassword().str();
926 req.player = stats;
927 req.addDesync = FALSE;
928 req.addDiscon = FALSE;
929 req.lastHouse = ptIdx;
930
931 TheGameSpyPSMessageQueue->addRequest(req);
932 TheGameSpyPSMessageQueue->trackPlayerStats(stats);
933
934 // force an update of our shtuff
935 PSResponse newResp;
936 newResp.responseType = PSResponse::PSRESPONSE_PLAYERSTATS;
937 newResp.player = stats;
938 TheGameSpyPSMessageQueue->addResponse(newResp);
939
940 // cache our stuff for easy reading next time
941 GameSpyMiscPreferences mPref;
942 mPref.setCachedStats(GameSpyPSMessageQueueInterface::formatPlayerKVPairs(stats).c_str());
943 mPref.write();
944 }
