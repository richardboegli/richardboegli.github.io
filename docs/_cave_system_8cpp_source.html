68Bool CaveSystem::canSwitchIndexToIndex( Int oldIndex, Int newIndex )
69{
70 // When I grant permission, you need to do it. ie call Unregister and then re-register with the new number
71 TunnelTracker *oldTracker = NULL;
72 TunnelTracker *newTracker = NULL;
73 if( m_tunnelTrackerVector.size() > oldIndex )
74 {
75 oldTracker = m_tunnelTrackerVector[oldIndex];
76 if( oldTracker && oldTracker->getContainCount() > 0 )
77 return FALSE;// You can't switch a connection if one of the two is non empty
78 }
79 if( m_tunnelTrackerVector.size() > newIndex )
80 {
81 newTracker = m_tunnelTrackerVector[newIndex];
82 if( newTracker && newTracker->getContainCount() > 0 )
83 return FALSE;// You can't switch a connection if one of the two is non empty
84 }
85
86 // Both are either empty or non-existent, so go ahead.
87 // (Remember non-exist is only a valid case because you are going to do the switch now.)
88
89 return TRUE;
90}

92void CaveSystem::registerNewCave( Int theIndex )
93{
94 Bool needToCreate = FALSE;
95 if( theIndex >= m_tunnelTrackerVector.size() )
96 {
97 // You are new and off the edge, so I will fill NULLs up to you and then make a newTracker at that spot
98 while( theIndex >= m_tunnelTrackerVector.size() )
99 m_tunnelTrackerVector.push_back( NULL );
100
101 needToCreate = TRUE;
102 }
103 else
104 {
105 // else you either exist or have existed, so I will either let things be or re-create that slot
106 if( m_tunnelTrackerVector[theIndex] == NULL )
107 needToCreate = TRUE;
108 }
109
110 if( needToCreate )// if true, we new theIndex is the index of a NULL to be filled
111 m_tunnelTrackerVector[theIndex] = newInstance(TunnelTracker);
112}

139void CaveSystem::xfer( Xfer *xfer )
140{
141
142 // version
143 XferVersion currentVersion = 1;
144 XferVersion version = currentVersion;
145 xfer->xferVersion( &version, currentVersion );
146
147 // tunnel tracker size and data
148 UnsignedShort count = m_tunnelTrackerVector.size();
149 xfer->xferUnsignedShort( &count );
150 TunnelTracker *tracker;
151 if( xfer->getXferMode() == XFER_SAVE )
152 {
153 std::vector< TunnelTracker* >::iterator it;
154
155 for( it = m_tunnelTrackerVector.begin(); it != m_tunnelTrackerVector.end(); ++it )
156 {
157
158 // xfer data
159 tracker = *it;
160 xfer->xferSnapshot( tracker );
161
162 } // end
163
164 } // end if, save
165 else
166 {
167
168 // the list must be empty now
169 if( m_tunnelTrackerVector.empty() == FALSE )
170 {
171
172 DEBUG_CRASH(( "CaveSystem::xfer - m_tunnelTrackerVector should be empty but is not\n" ));
173 throw SC_INVALID_DATA;
174
175 } // end if
176
177 // read each item
178 for( UnsignedShort i = 0; i < count; ++i )
179 {
180
181 // allocate new tracker
182 tracker = newInstance( TunnelTracker );
183
184 // read data
185 xfer->xferSnapshot( tracker );
186
187 // put in vector
188 m_tunnelTrackerVector.push_back( tracker );
189
190 } // end for, i
191
192 } // end else, laod
193
194} // end xfer
