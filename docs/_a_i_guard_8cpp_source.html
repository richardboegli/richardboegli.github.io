64static Bool hasAttackedMeAndICanReturnFire( State *thisState, void* /*userData*/ )
65{
66 Object *obj = thisState->getMachineOwner();
67 BodyModuleInterface *bmi = obj ? obj->getBodyModule() : NULL;
68
69 if (!(obj && bmi)) {
70 return FALSE;
71 }
72
73 if (bmi->getClearableLastAttacker() == INVALID_ID) {
74 return FALSE;
75 }
76
77 // K. It appears we have a valid aggressor. Find it, and determine if we can attack it, etc.
78 Object *target = TheGameLogic->findObjectByID(bmi->getClearableLastAttacker());
79 bmi->clearLastAttacker();
80
81 // We use the clearable last attacker because we should continue attacking the guy. But if he
82 // stops attacking us, then we want our timer to kick us off of him and make us go attack
83 // other units instead.
84
85
86 if (!target) {
87 return FALSE;
88 }
89
90 if (obj->getRelationship(target) != ENEMIES) {
91 return FALSE;
92 }
93
94 // This is a quick test on the target. It will be duplicated in getAbleToAttackSpecificObject,
95 // but the payoff is worth the duplication.
96 if (target->isEffectivelyDead()) {
97 return FALSE;
98 }
99
100 //@todo: Get this out of here. Move it into the declaration of calling this function, or figure
101 // out some way to call it less often.
102
103 if (!obj->isAbleToAttack()) {
104 return FALSE;
105 }
106
107 CanAttackResult result = obj->getAbleToAttackSpecificObject(ATTACK_NEW_TARGET, target, CMD_FROM_AI);
108 if( result == ATTACKRESULT_POSSIBLE || result == ATTACKRESULT_POSSIBLE_AFTER_MOVING )
109 {
110 return TRUE;
111 }
112 return FALSE;
113}

115//-- ExitConditions -------------------------------------------------------------------------------
119Bool ExitConditions::shouldExit(const StateMachine* machine) const
120{
121 if (!machine->getGoalObject())
122 {
123 if (m_conditionsToConsider & ATTACK_ExitIfNoUnitFound)
124 {
125 return true;
126 }
127 else
128 {
129 return false;
130 }
131 }
132
133 if (m_conditionsToConsider & ATTACK_ExitIfExpiredDuration)
134 {
135 if (TheGameLogic->getFrame() >= m_attackGiveUpFrame)
136 {
137 return true;
138 }
139 }
140
141 if (m_conditionsToConsider & ATTACK_ExitIfOutsideRadius)
142 {
143 Coord3D deltaAggressor;
144 Coord3D objPos = *machine->getGoalObject()->getPosition();
145 deltaAggressor.x = objPos.x - m_center.x;
146 deltaAggressor.y = objPos.y - m_center.y;
147// deltaAggressor.z = objPos.z - m_center.z;
148 deltaAggressor.z = 0; // BGC - when we search for a target we don't account for Z, so why should we here?
149 // changing this fixed a crash where a GLARebelInfantry would be in GuardReturnState, find
150 // a target that is within range, then not be able to attack because its actually out of range.
151 // then it would look for a new target, get the same one, and proceed in an infinite recursive
152 // loop that eventually blew the stack.
153
154 if (deltaAggressor.lengthSqr() > m_radiusSqr)
155 {
156 return true;
157 }
158 }
159
160 return false;
161}

166AIGuardMachine::AIGuardMachine( Object *owner ) :
167 StateMachine(owner, "AIGuardMachine"),
168 m_targetToGuard(INVALID_ID),
169 m_areaToGuard(NULL),
170 m_nemesisToAttack(INVALID_ID),
171 m_guardMode(GUARDMODE_NORMAL)
172{
173 m_positionToGuard.zero();
174
175 static const StateConditionInfo attackAggressors[] =
176 {
177 StateConditionInfo(hasAttackedMeAndICanReturnFire, AI_GUARD_ATTACK_AGGRESSOR, NULL),
178 StateConditionInfo(NULL, NULL, NULL) // keep last
179 };
180
181 // order matters: first state is the default state.
182// srj sez: I made "return" the start state, so that if ordered to guard a position
183// that isn't the unit's current position, it moves to that position first.
184 //Kris: Except that guard return is more like an attack move, and will acquire targets while moving there.
185 //This breaks deployAI units because they have to completely unpack before realizing that there is a target in range.
186 //So I'm making AI_GUARD_INNER the first state.
187 defineState( AI_GUARD_INNER, newInstance(AIGuardInnerState)( this ), AI_GUARD_OUTER, AI_GUARD_OUTER, attackAggressors );
188 defineState( AI_GUARD_RETURN, newInstance(AIGuardReturnState)( this ), AI_GUARD_IDLE, AI_GUARD_INNER, attackAggressors );
189 defineState( AI_GUARD_IDLE, newInstance(AIGuardIdleState)( this ), AI_GUARD_INNER, AI_GUARD_RETURN, attackAggressors );
190 defineState( AI_GUARD_OUTER, newInstance(AIGuardOuterState)( this ), AI_GUARD_GET_CRATE, AI_GUARD_GET_CRATE );
191 defineState( AI_GUARD_GET_CRATE, newInstance(AIGuardPickUpCrateState)( this ), AI_GUARD_RETURN, AI_GUARD_RETURN );
192 defineState( AI_GUARD_ATTACK_AGGRESSOR, newInstance(AIGuardAttackAggressorState)( this ), AI_GUARD_INNER, AI_GUARD_INNER );
193}

210Bool AIGuardMachine::lookForInnerTarget(void)
211{
212 Object* owner = getOwner();
213 if (!owner->isAbleToAttack())
214 {
215 return false; // my, that was easy
216 }
217
218 // Check if team auto targets same victim.
219 Object *teamVictim = NULL;
220 if (owner->getTeam()->getPrototype()->getTemplateInfo()->m_attackCommonTarget)
221 {
222 teamVictim = owner->getTeam()->getTeamTargetObject();
223 if (teamVictim)
224 {
225 setNemesisID(teamVictim->getID());
226 return true; // Transitions to AIGuardInnerState.
227 }
228 }
229
230 Object* targetToGuard = findTargetToGuardByID();
231 Coord3D pos = targetToGuard ? *targetToGuard->getPosition() : *getPositionToGuard();
232
233 const PolygonTrigger* area = getAreaToGuard();
234 PartitionFilterRelationship f1(owner, PartitionFilterRelationship::ALLOW_ENEMIES);
235 PartitionFilterPossibleToAttack f2(ATTACK_NEW_TARGET, owner, CMD_FROM_AI);
236 PartitionFilterSameMapStatus filterMapStatus(owner);
237 PartitionFilterPolygonTrigger f3(area);
238 PartitionFilterIsFlying f4;
239 PartitionFilterRelationship f5(owner, PartitionFilterRelationship::ALLOW_NEUTRAL);
240 PartitionFilterPossibleToEnter f6(owner, CMD_FROM_AI);
241 PartitionFilterPossibleToHijack f7(owner, CMD_FROM_AI);
242
243 PartitionFilter *filters[16];
244 Int count = 0;
245
246 // Enter Guard state
247 if (owner->getTemplate()->isEnterGuard())
248 {
249 filters[count++] = &f6;
250
251 // Hijack Guard state
252 if (owner->getTemplate()->isHijackGuard())
253 {
254 filters[count++] = &f1;
255 filters[count++] = &f7;
256 }
257 else
258 {
259 filters[count++] = &f5;
260 }
261 }
262 // Attack Guard state
263 else
264 {
265 filters[count++] = &f1;
266 filters[count++] = &f2;
267 }
268
269 filters[count++] = &filterMapStatus;
270
271 Real visionRange = AIGuardMachine::getStdGuardRange(owner);
272
273 if (area)
274 {
275 UnsignedInt checkFrame = TheGameLogic->getFrameObjectsChangedTriggerAreas()+TheAI->getAiData()->m_guardEnemyScanRate;
276 if (TheGameLogic->getFrame()>checkFrame) {
277 return false;
278 }
279 filters[count++] = &f3;
280 visionRange = area->getRadius();
281 area->getCenterPoint(&pos);
282 }
283
284 if (getGuardMode() == GUARDMODE_GUARD_FLYING_UNITS_ONLY)
285 {
286 // only consider flying targets
287 filters[count++] = &f4;
288 }
289
290 filters[count++] = NULL;
291
292// SimpleObjectIterator* iter = ThePartitionManager->iterateObjectsInRange(
293// &pos, visionRange, FROM_CENTER_2D, filters, ITER_SORTED_NEAR_TO_FAR);
294// MemoryPoolObjectHolder hold(iter);
295// Object* target = iter->first();
296//
297// srj sez: the above code is stupid and slow. since we only want the closest object,
298// just ask for that; the above has to find ALL objects in range, but we ignore all
299// but the first (closest).
300//
301 Object* target = ThePartitionManager->getClosestObject(&pos, visionRange, FROM_CENTER_2D, filters);
302 if (target)
303 {
304 setNemesisID(target->getID());
305 return true; // Transitions to AIGuardInnerState.
306 }
307 else
308 {
309 return false;
310 }
311}

385//--------------------------------------------------------------------------------------
386StateReturnType AIGuardInnerState::onEnter( void )
387{
388 // See if we try to enter the target
389 if (getMachineOwner()->getTemplate()->isEnterGuard())
390 {
391 Object* nemesis = TheGameLogic->findObjectByID(getGuardMachine()->getNemesisID()) ;
392 if (nemesis == NULL)
393 {
394 DEBUG_LOG(("Unexpected NULL nemesis in AIGuardInnerState.\n"));
395 return STATE_SUCCESS;
396 }
397 m_enterState = newInstance(AIEnterState)(getMachine());
398
399 m_enterState->getMachine()->setGoalObject(nemesis);
400
401 StateReturnType returnVal = m_enterState->onEnter();
402 if (returnVal == STATE_CONTINUE) {
403 return STATE_CONTINUE;
404 }
405 }
406 // Or try to destroy the target
407 else
408 {
409 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
410 Coord3D pos = targetToGuard ? *targetToGuard->getPosition() : *getGuardMachine()->getPositionToGuard();
411 Object* nemesis = TheGameLogic->findObjectByID(getGuardMachine()->getNemesisID()) ;
412 if (nemesis == NULL)
413 {
414 DEBUG_LOG(("Unexpected NULL nemesis in AIGuardInnerState.\n"));
415 return STATE_SUCCESS;
416 }
417 m_exitConditions.m_center = pos;
418 m_exitConditions.m_radiusSqr = sqr(AIGuardMachine::getStdGuardRange(getMachineOwner()));
419 m_exitConditions.m_conditionsToConsider = (ExitConditions::ATTACK_ExitIfOutsideRadius |
420 ExitConditions::ATTACK_ExitIfNoUnitFound);
421
422 m_attackState = newInstance(AIAttackState)(getMachine(), false, true, false, &m_exitConditions);
423
424 m_attackState->getMachine()->setGoalObject(nemesis);
425
426 StateReturnType returnVal = m_attackState->onEnter();
427 if (returnVal == STATE_CONTINUE) {
428 return STATE_CONTINUE;
429 }
430 }
431
432 // if we had no one to attack, we were successful, so go to the next state.
433 return STATE_SUCCESS;
434}

509//--------------------------------------------------------------------------------------
510StateReturnType AIGuardOuterState::onEnter( void )
511{
512 if (getGuardMachine()->getGuardMode() == GUARDMODE_GUARD_WITHOUT_PURSUIT)
513 {
514 // "patrol" mode does not follow targets outside the guard area.
515 return STATE_SUCCESS;
516 }
517
518 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
519 Coord3D pos = targetToGuard ? *targetToGuard->getPosition() : *getGuardMachine()->getPositionToGuard();
520
521 Object* nemesis = TheGameLogic->findObjectByID(getGuardMachine()->getNemesisID()) ;
522 if (nemesis == NULL)
523 {
524 DEBUG_LOG(("Unexpected NULL nemesis in AIGuardInnerState.\n"));
525 return STATE_SUCCESS;
526 }
527 Object *obj = getMachineOwner();
528
529 Real range = TheAI->getAdjustedVisionRangeForObject(obj, AI_VISIONFACTOR_OWNERTYPE | AI_VISIONFACTOR_MOOD);
530
531 const PolygonTrigger *area = getGuardMachine()->getAreaToGuard();
532 if (area)
533 {
534 if (range < area->getRadius())
535 range = area->getRadius();
536 area->getCenterPoint(&pos);
537 }
538 m_exitConditions.m_center = pos;
539 m_exitConditions.m_radiusSqr = sqr(range);
540 m_exitConditions.m_attackGiveUpFrame = TheGameLogic->getFrame() + TheAI->getAiData()->m_guardChaseUnitFrames;
541 m_exitConditions.m_conditionsToConsider = (ExitConditions::ATTACK_ExitIfExpiredDuration |
542 ExitConditions::ATTACK_ExitIfOutsideRadius |
543 ExitConditions::ATTACK_ExitIfNoUnitFound);
544
545 m_attackState = newInstance(AIAttackState)(getMachine(), false, true, false, &m_exitConditions);
546 m_attackState->getMachine()->setGoalObject(nemesis);
547
548 StateReturnType returnVal = m_attackState->onEnter();
549 if (returnVal == STATE_CONTINUE) {
550 return STATE_CONTINUE;
551 }
552
553 // if we had no one to attack, we were successful, so go to the next state.
554 return STATE_SUCCESS;
555}
556
557//--------------------------------------------------------------------------------------
558StateReturnType AIGuardOuterState::update( void )
559{
560 if (m_attackState==NULL) return STATE_SUCCESS;
561
562 // if the position has moved (IE we're guarding an object), move with it.
563 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
564 if (targetToGuard)
565 {
566 m_exitConditions.m_center = *targetToGuard->getPosition();
567 }
568
569 Object* goalObj = m_attackState->getMachineGoalObject();
570 if (goalObj)
571 {
572 Coord3D deltaAggr;
573 deltaAggr.x = m_exitConditions.m_center.x - goalObj->getPosition()->x;
574 deltaAggr.y = m_exitConditions.m_center.y - goalObj->getPosition()->y;
575 deltaAggr.z = m_exitConditions.m_center.z - goalObj->getPosition()->z;
576 Real visionSqr = sqr(AIGuardMachine::getStdGuardRange(getMachineOwner()));
577 if (deltaAggr.lengthSqr() <= visionSqr)
578 {
579 // reset the counter
580 m_exitConditions.m_attackGiveUpFrame = TheGameLogic->getFrame() + TheAI->getAiData()->m_guardChaseUnitFrames;
581 }
582 }
583
584 return m_attackState->update();
585}

626//--------------------------------------------------------------------------------------
627StateReturnType AIGuardReturnState::onEnter( void )
628{
629 UnsignedInt now = TheGameLogic->getFrame();
630 m_nextReturnScanTime = now + GameLogicRandomValue(0, TheAI->getAiData()->m_guardEnemyReturnScanRate);
631
632// no, no, no, don't do this in onEnter, unless you like really slow maps. (srj)
633// if (getGuardMachine()->lookForInnerTarget())
634// return STATE_FAILURE; // early termination because we found a target.
635
636 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
637 m_goalPosition = targetToGuard ? *targetToGuard->getPosition() : *getGuardMachine()->getPositionToGuard();
638
639 const PolygonTrigger *area = getGuardMachine()->getAreaToGuard();
640 if (area)
641 {
642 area->getCenterPoint(&m_goalPosition);
643 }
644 AIUpdateInterface *ai = getMachineOwner()->getAIUpdateInterface();
645 if (ai && ai->isDoingGroundMovement())
646 {
647 TheAI->pathfinder()->adjustDestination(getMachineOwner(), ai->getLocomotorSet(), &m_goalPosition);
648 }
649 setAdjustsDestination(true);
650 return AIInternalMoveToState::onEnter();
651}

653//--------------------------------------------------------------------------------------
654StateReturnType AIGuardReturnState::update( void )
655{
656 UnsignedInt now = TheGameLogic->getFrame();
657 if (now >= m_nextReturnScanTime)
658 {
659 m_nextReturnScanTime = now + TheAI->getAiData()->m_guardEnemyReturnScanRate;
660 if (getGuardMachine()->lookForInnerTarget())
661 return STATE_FAILURE; // early termination because we found a target.
662 }
663
664 // Just let the return movement finish.
665 return AIInternalMoveToState::update();
666}

702//--------------------------------------------------------------------------------------
703StateReturnType AIGuardIdleState::onEnter( void )
704{
705 // first time thru, use a random amount so that everyone doesn't scan on the same frame,
706 // to avoid "spikes".
707 UnsignedInt now = TheGameLogic->getFrame();
708 m_nextEnemyScanTime = now + GameLogicRandomValue(0, TheAI->getAiData()->m_guardEnemyScanRate);
709
710 return STATE_CONTINUE;
711}
712
713//--------------------------------------------------------------------------------------
714StateReturnType AIGuardIdleState::update( void )
715{
716 //DEBUG_LOG(("AIGuardIdleState frame %d: %08lx\n",TheGameLogic->getFrame(),getMachineOwner()));
717
718 UnsignedInt now = TheGameLogic->getFrame();
719 if (now < m_nextEnemyScanTime)
720 return STATE_SLEEP(m_nextEnemyScanTime - now);
721
722 m_nextEnemyScanTime = now + TheAI->getAiData()->m_guardEnemyScanRate;
723
724#ifdef STATE_MACHINE_DEBUG
725 //getMachine()->setDebugOutput(true);
726#endif
727 Object *owner = getMachineOwner();
728 AIUpdateInterface *ai = owner->getAIUpdateInterface();
729 // Check to see if we have created a crate we need to pick up.
730 if (ai->getCrateID() != INVALID_ID)
731 {
732 getMachine()->setState(AI_GUARD_GET_CRATE);
733 return STATE_SLEEP(m_nextEnemyScanTime - now);
734 }
735
736 // if anyone is in the inner area, return success.
737 if (getGuardMachine()->lookForInnerTarget())
738 {
739 return STATE_SUCCESS; // Transitions to AIGuardInnerState.
740 }
741
742 // See if the object we are guarding moved.
743 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
744 if (targetToGuard)
745 {
746 Coord3D pos = *targetToGuard->getPosition();
747 Real delta = m_guardeePos.x-pos.x;

812//-------------------------------------------------------------------------------------------------
813StateReturnType AIGuardAttackAggressorState::onEnter( void )
814{
815 Object *obj = getMachineOwner();
816 ObjectID nemID = INVALID_ID;
817
818 if (obj->getBodyModule() && obj->getBodyModule()->getLastDamageInfo()->in.m_sourceID) {
819 nemID = obj->getBodyModule()->getLastDamageInfo()->in.m_sourceID;
820 getGuardMachine()->setNemesisID(nemID);
821 }
822
823 Object *nemesis = TheGameLogic->findObjectByID(getGuardMachine()->getNemesisID());
824 if (nemesis == NULL)
825 {
826 DEBUG_LOG(("Unexpected NULL nemesis in AIGuardAttackAggressorState.\n"));
827 return STATE_SUCCESS;
828 }
829
830 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
831 Coord3D pos = targetToGuard ? *targetToGuard->getPosition() : *getGuardMachine()->getPositionToGuard();
832 //Don't allow guarding units to leave their guard radius!
833 m_exitConditions.m_center = pos;
834 m_exitConditions.m_radiusSqr = sqr(AIGuardMachine::getStdGuardRange(getMachineOwner()));
835 m_exitConditions.m_attackGiveUpFrame = TheGameLogic->getFrame() + TheAI->getAiData()->m_guardChaseUnitFrames;
836 m_exitConditions.m_conditionsToConsider = (ExitConditions::ATTACK_ExitIfExpiredDuration |
837 ExitConditions::ATTACK_ExitIfNoUnitFound |
838 ExitConditions::ATTACK_ExitIfOutsideRadius );
839
840 m_attackState = newInstance(AIAttackState)(getMachine(), true, true, false, &m_exitConditions);
841 m_attackState->getMachine()->setGoalObject(nemesis);
842
843 StateReturnType returnVal = m_attackState->onEnter();
844 if (returnVal == STATE_CONTINUE) {
845 return STATE_CONTINUE;
846 }
847
848 // if we had no one to attack, we were successful, so go to the next state.
849 return STATE_SUCCESS;
850}

854{

857 Object* targetToGuard = getGuardMachine()->findTargetToGuardByID();
858 if (targetToGuard)
859 {
860 m_exitConditions.m_center = *targetToGuard->getPosition();
861 }
862
863 return m_attackState->update();
864}
