81static UnsignedByte * saveSurface(IDirect3DSurface8 *surface)
82{
83 D3DSURFACE_DESC desc;
84 IDirect3DSurface8 *tempSurface;
85
86 surface->GetDesc(&desc);
87
88 LPDIRECT3DDEVICE8 m_pDev=DX8Wrapper::_Get_D3D_Device8();
89
90 HRESULT hr=m_pDev->CreateImageSurface( desc.Width,desc.Height,desc.Format, &tempSurface);
91
92 hr=m_pDev->CopyRects(surface,NULL,0,tempSurface,NULL);
93
94 D3DLOCKED_RECT lrect;
95
96 DX8_ErrorCode(tempSurface->LockRect(&lrect,NULL,D3DLOCK_READONLY));
97
98 unsigned int x,y,index,index2,width,height;
99
100 width=desc.Width;
101 height=desc.Height;
102
103#ifdef CAPTURE_TO_TARGA
104 char image[3*PREVIEW_WIDTH*PREVIEW_HEIGHT];
105 //bytes are mixed in targa files, not rgb order.
106 for (y=0; y<height; y++)
107 {
108 for (x=0; x<width; x++)
109 {
110 // index for image
111 index=3*(x+y*width);
112 // index for fb
113 index2=y*lrect.Pitch+4*x;
114
115 image[index]=*((char *) lrect.pBits + index2+2);
116 image[index+1]=*((char *) lrect.pBits + index2+1);
117 image[index+2]=*((char *) lrect.pBits + index2+0);
118 }
119 }
120
121 Targa targ;
122 memset(&targ.Header,0,sizeof(targ.Header));
123 targ.Header.Width=width;
124 targ.Header.Height=height;
125 targ.Header.PixelDepth=24;
126 targ.Header.ImageType=TGA_TRUECOLOR;
127 targ.SetImage(image);
128 targ.YFlip();
129
130 targ.Save("ObjectPreview.tga",TGAF_IMAGE,false);
131
132 return NULL;
133
134#else
135
136 static UnsignedByte bgraImage[3*PREVIEW_WIDTH*PREVIEW_HEIGHT];
137 //bmp is same byte order
138 for (y=0; y<height; y++)
139 {
140 for (x=0; x<width; x++)
141 {
142 // index for image
143 index=3*(x+y*width);
144 // index for fb
145 index2=y*lrect.Pitch+4*x;
146
147 bgraImage[index]=*((UnsignedByte *) lrect.pBits + index2+0);
148 bgraImage[index+1]=*((UnsignedByte *) lrect.pBits + index2+1);
149 bgraImage[index+2]=*((UnsignedByte *) lrect.pBits + index2+2);
150 //bgraImage[index+3]=0;
151 }
152 }
153
154 //Flip the image
155 UnsignedByte *ptr,*ptr1;
156 UnsignedByte v,v1;
157
158 for (y = 0; y < (height >> 1); y++)
159 {
160 /* Compute address of lines to exchange. */
161 ptr = (bgraImage + ((width * y) * 3));
162 ptr1 = (bgraImage + ((width * (height - 1)) * 3));
163 ptr1 -= ((width * y) * 3);
164
165 /* Exchange all the pixels on this scan line. */
166 for (x = 0; x < (width * 3); x++)
167 {
168 v = *ptr;
169 v1 = *ptr1;
170 *ptr = v1;
171 *ptr1 = v;
172 ptr++;
173 ptr1++;
174 }
175 }
176
177 tempSurface->Release();
178
179 return bgraImage;
180#endif
181}

183// return an array of BGRA pixels
184static UnsignedByte * generatePreview( const ThingTemplate *tt )
185{
186 // find the default model to preview
187 RenderObjClass *model = NULL;
188 Real scale = 1.0f;
189 AsciiString modelName = "No Model Name";
190 if (tt)
191 {
192 ModelConditionFlags state;
193 state.clear();
194 WbView3d *p3View = CWorldBuilderDoc::GetActiveDoc()->GetActive3DView();
195 modelName = p3View->getBestModelName(tt, state);
196 scale = tt->getAssetScale();
197 }
198 // set render object, or create if we need to
199 if( modelName.isEmpty() == FALSE &&
200 strncmp( modelName.str(), "No ", 3 ) )
201 {
202 WW3DAssetManager *pMgr = W3DAssetManager::Get_Instance();
203 model = pMgr->Create_Render_Obj(modelName.str());
204 if (model)
205 {
206 const AABoxClass bbox = model->Get_Bounding_Box();
207// Real height = bbox.Extent.Z;
208 const SphereClass sphere = model->Get_Bounding_Sphere();
209 Real dist = sphere.Radius*0.5;
210 model->Set_Position(Vector3(-sphere.Center.X, -sphere.Center.Y, -sphere.Center.Z));
211
212 // Create reflection texture
213 TextureClass *objectTexture = DX8Wrapper::Create_Render_Target (PREVIEW_WIDTH, PREVIEW_HEIGHT);
214 if (!objectTexture)
215 {
216 model->Release_Ref();
217 return NULL;
218 }
219
220 // Set the render target
221 DX8Wrapper::Set_Render_Target_With_Z(objectTexture);
222
223 // create the camera
224 Bool orthoCamera = false;
225 CameraClass *camera = NEW_REF( CameraClass, () );
226 Matrix3D camTran;
227 camTran.Look_At(Vector3(dist*2,dist*2,dist),Vector3(0.0f, 0.0f, 0.0f),0);
228 camera->Set_Transform( camTran);
229
230 Vector2 minVec = Vector2( -1, -1 );
231 Vector2 maxVec = Vector2( +1, +1 );
232 camera->Set_View_Plane( minVec, maxVec );
233 camera->Set_Clip_Planes( 0.995f, 600.0f );
234 if (orthoCamera)
235 camera->Set_Projection_Type( CameraClass::ORTHO );
236
237 // Clear the backbuffer
238 WW3D::Begin_Render(true,true,Vector3(0.5f,0.5f,0.5f));
239 //WW3D::Begin_Render(true,true,Vector3(1.0f,1.0f,1.0f));
240
241 RenderInfoClass rinfo(*camera);
242 LightEnvironmentClass lightEnv;
243 rinfo.light_environment = &lightEnv;
244 lightEnv.Reset(Vector3(0.0f, 0.0f, 0.0f), Vector3(1.0f, 1.0f, 1.0f));
245
246 WW3D::Render(*model, rinfo);
247
248 WW3D::End_Render(false);
249
250 // Change the rendertarget back to the main backbuffer
251 DX8Wrapper::Set_Render_Target((IDirect3DSurface8 *)NULL);
252
253 SurfaceClass *surface = objectTexture->Get_Surface_Level();
254 UnsignedByte *data = saveSurface(surface->Peek_D3D_Surface());
255
256 REF_PTR_RELEASE(surface);
257
258 REF_PTR_RELEASE(objectTexture);
259 REF_PTR_RELEASE(camera);
260 return data;
261 }
262 }
263
264 return NULL;
265}

270void ObjectPreview::OnPaint()
271{
272 CPaintDC dc(this); // device context for painting
273
274 CRect clientRect;
275 GetClientRect(&clientRect);
276
277 CRect previewRect;
278 previewRect = clientRect;
279
280 CBrush brush;
281 brush.CreateSolidBrush(RGB(0,0,0));
282
283 UnsignedByte *pData;
284 pData = generatePreview(m_tTempl);
285 if (pData) {
286 DrawMyTexture(&dc, previewRect.top, previewRect.left, previewRect.Width(), previewRect.Height(), pData);
287 } else {
288 dc.FillSolidRect(&previewRect, RGB(128,128,128));
289 }
290 dc.FrameRect(&previewRect, &brush);
291}
