140 inline UnsignedByte getClipHeight(Int x, Int y) const
141 {
142 Int xextent = m_map->getXExtent() - 1;
143 Int yextent = m_map->getYExtent() - 1;
144
145 if (x < 0)
146 x = 0;
147 else if (x > xextent)
148 x = xextent;
149
150 if (y < 0)
151 y = 0;
152 else if (y > yextent)
153 y = yextent;
154
155 return m_map->getDataPtr()[x + y*m_map->getXExtent()];
156 }
