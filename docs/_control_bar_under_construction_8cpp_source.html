48void ControlBar::updateConstructionTextDisplay( Object *obj )
49{
50 UnicodeString text;
51 static UnsignedInt descID = TheNameKeyGenerator->nameToKey( "ControlBar.wnd:UnderConstructionDesc" );
52 GameWindow *descWindow = TheWindowManager->winGetWindowFromId( NULL, descID );
53
54 // santiy
55 DEBUG_ASSERTCRASH( descWindow, ("Under construction window not found\n") );
56
57 // format the message
58 text.format( TheGameText->fetch( "CONTROLBAR:UnderConstructionDesc" ),
59 obj->getConstructionPercent() );
60 GadgetStaticTextSetText( descWindow, text );
61
62 // record this as the last percentage displayed
63 m_displayedConstructPercent = obj->getConstructionPercent();
64
65} // end updateConstructionTextDisplay

70void ControlBar::populateUnderConstruction( Object *objectUnderConstruction )
71{
72
73 // sanity
74 if( objectUnderConstruction == NULL )
75 return;
76
77 // get our parent window
78 GameWindow *parent = m_contextParent[ CP_UNDER_CONSTRUCTION ];
79
80 // set the cancel construction button
82 const CommandButton *commandButton = findCommandButton( "Command_CancelConstruction" );
83 NameKeyType id;
84 id = TheNameKeyGenerator->nameToKey( "ControlBar.wnd:ButtonCancelConstruction" );
85 GameWindow *win = TheWindowManager->winGetWindowFromId( parent, id );
86
87 setControlCommand( win, commandButton );
88 win->winSetStatus( WIN_STATUS_USE_OVERLAY_STATES );
89
90 // set the text description of what is building
91 updateConstructionTextDisplay( objectUnderConstruction );
92
93 // set the portrait for the thing being constructed
94 setPortraitByObject( objectUnderConstruction );
95
96 // and show the rally point, if it should have one,
97 ExitInterface *exit = objectUnderConstruction->getObjectExitInterface();
98 if( exit )
99 showRallyPoint( exit->getRallyPoint() );
100
101
102} // end populateUnderConstruction

106void ControlBar::updateContextUnderConstruction( void )
107{
108 Object *obj = m_currentSelectedDrawable->getObject();
109
110 // if the object is no longer under construction switch to a new appropriate context
111 if( !obj->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
112 {
113
114 evaluateContextUI();
115 return;
116
117 } // end if
118
119 // if the construction percent has changed since what was last shown to the user update the text
120 if( m_displayedConstructPercent != obj->getConstructionPercent() )
121 updateConstructionTextDisplay( obj );
122
123} // end updatecontextUnderConstruction
