38GameMessageParser::GameMessageParser(GameMessage *msg)
39{
40 m_first = NULL;
41 m_argTypeCount = 0;
42
43 UnsignedByte argCount = msg->getArgumentCount();
44 GameMessageArgumentDataType lasttype = ARGUMENTDATATYPE_UNKNOWN;
45 Int thisTypeCount = 0;
46
47 for (UnsignedByte i = 0; i < argCount; ++i) {
48 GameMessageArgumentDataType type = msg->getArgumentDataType(i);
49 if (type != lasttype) {
50 if (thisTypeCount > 0) {
51 addArgType(lasttype, thisTypeCount);
52 ++m_argTypeCount;
53 }
54 lasttype = type;
55 thisTypeCount = 0;
56 }
57 ++thisTypeCount;
58 }
59 if (thisTypeCount > 0) {
60 addArgType(lasttype, thisTypeCount);
61 ++m_argTypeCount;
62 }
63}
