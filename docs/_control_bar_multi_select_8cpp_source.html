70void ControlBar::addCommonCommands( Drawable *draw, Bool firstDrawable )
71{
72 Int i;
73 const CommandButton *command;
74
75 // sanity
76 if( draw == NULL )
77 return;
78
79 Object* obj = draw->getObject();
80 if (!obj)
81 return;
82
83 if (obj->isKindOf(KINDOF_IGNORED_IN_GUI)) // ignore these guys
84 return;
85
86 // get the command set of this drawable
87 const CommandSet *commandSet = findCommandSet( obj->getCommandSetString() );
88 if( commandSet == NULL )
89 {
90
91 //
92 // if there is no command set for this drawable, none of the selected drawables
93 // can possibly have matching commands so we'll get rid of them all
94 //
95 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )
96 {
97
98 m_commonCommands[ i ] = NULL;
99 if (m_commandWindows[ i ])
100 {
101 m_commandWindows[ i ]->winHide( TRUE );
102 }
103 // After Every change to the m_commandWIndows, we need to show fill in the missing blanks with the images
104 // removed from multiplayer branch
105 //showCommandMarkers();
106
107 } // end for i
108
109 return;
110
111 } // end if
112
113
114 //
115 // easy case, if we're adding the first drawable we simply just add any of the commands
116 // in its set that can be multi-select commands to the common command set
117 //
118 if( firstDrawable == TRUE )
119 {
120
121 // just add each command that is classified as a common command
122 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )
123 {
124 // our implementation doesn't necessarily make use of the max possible command buttons
125 if (! m_commandWindows[ i ]) continue;
126
127 // get command
128 command = commandSet->getCommandButton(i);
129
130 // add if present and can be used in a multi select
131 if( command && BitTest( command->getOptions(), OK_FOR_MULTI_SELECT ) == TRUE )
132 {
133
134 // put it in the common command set
135 m_commonCommands[ i ] = command;
136
137 // show and enable this control
138 m_commandWindows[ i ]->winHide( FALSE );
139 m_commandWindows[ i ]->winEnable( TRUE );
140
141 // set the command into the control
142 setControlCommand( m_commandWindows[ i ], command );
143
144 } // end if
145
146 } // end for i
147
148 } // end if
149 else
150 {
151
152 // go through each command one by one
153 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )
154 {
155
156 // our implementation doesn't necessarily make use of the max possible command buttons
157 if (! m_commandWindows[ i ]) continue;
158
159 // get the command
160 command = commandSet->getCommandButton(i);
161
162 Bool attackMove = (command && command->getCommandType() == GUI_COMMAND_ATTACK_MOVE) ||
163 (m_commonCommands[ i ] && m_commonCommands[ i ]->getCommandType() == GUI_COMMAND_ATTACK_MOVE);
164
165 // Kris: When any units have attack move, they all get it. This is to allow
166 // combat units to be selected with the odd dozer or pilot and still retain that ability.
167 if( attackMove && !m_commonCommands[ i ] )
168 {
169 // put it in the common command set
170 m_commonCommands[ i ] = command;
171
172 // show and enable this control
173 m_commandWindows[ i ]->winHide( FALSE );
174 m_commandWindows[ i ]->winEnable( TRUE );
175
176 // set the command into the control
177 setControlCommand( m_commandWindows[ i ], command );
178 }
179 else if( command != m_commonCommands[ i ] && !attackMove )
180 {
181 //
182 // if this command does not match the command that is in the common command set then
183 // *neither* this command OR the command in the common command set are really common
184 // commands, so we will remove the one that has been stored in the common set
185 //
186
187 // remove the common command
188 m_commonCommands[ i ] = NULL;
189
190 //
191 // hide the window control cause it should have been made visible from a command
192 // that was placed in this common 'slot' earlier
193 //
194 m_commandWindows[ i ]->winHide( TRUE );
195 }
196
197 } // end if
198
199 } // end else
200
201 // After Every change to the m_commandWIndows, we need to show fill in the missing blanks with the images
202 // removed from multiplayer branch
203 //showCommandMarkers();
204
205} // end addCommonCommands

211void ControlBar::populateMultiSelect( void )
212{
213 Drawable *draw;
214 Bool firstDrawable = TRUE;
215 Bool portraitSet = FALSE;
216 const Image *portrait = NULL;
217 Object *portraitObj = NULL;
218
219 // first reset the common command data
220 resetCommonCommandData();
221
222 // by default, hide all the controls in the command section
223 for( Int i = 0; i < MAX_COMMANDS_PER_SET; i++ )
224 {
225 if (m_commandWindows[ i ])
226 {
227 m_commandWindows[ i ]->winHide( TRUE );
228 }
229 }
230
231 // sanity
232 DEBUG_ASSERTCRASH( TheInGameUI->getSelectCount() > 1,
233 ("populateMultiSelect: Can't populate multiselect context cause there are only '%d' things selected\n",
234 TheInGameUI->getSelectCount()) );
235
236 // get the list of drawable IDs from the in game UI
237 const DrawableList *selectedDrawables = TheInGameUI->getAllSelectedDrawables();
238
239 // sanity
240 DEBUG_ASSERTCRASH( selectedDrawables->empty() == FALSE, ("populateMultiSelect: Drawable list is empty\n") );
241
242 // loop through all the selected drawables
243 for( DrawableListCIt it = selectedDrawables->begin();
244 it != selectedDrawables->end(); ++it )
245 {
246
247 // get the drawable
248 draw = *it;
249
250
251 if (draw->getObject()->isKindOf(KINDOF_IGNORED_IN_GUI)) // ignore these guys
252 continue;
253
254
255 //
256 // add command for this drawable, note that we also sanity check to make sure the
257 // drawable has an object as all interesting drawables that we can select should
258 // actually have an object underneath it so that we can do interesting things with
259 // it ... otherwise we should have never selected it.
260 // NOTE that we're not considering objects that are currently in the process of
261 // being sold as those objects can't be issued anymore commands
262 //
263 if( draw && draw->getObject() &&
264 !draw->getObject()->getStatusBits().test( OBJECT_STATUS_SOLD ) )
265 {
266
267 // add the common commands of this drawable to the common command set
268 addCommonCommands( draw, firstDrawable );
269
270 // not adding the first drawble anymore
271 firstDrawable = FALSE;
272
273 //
274 // keep track of the portrait images, if all units selected have the same portrait
275 // we will display it in the right HUD, otherwise we won't
276 //
277 if( portraitSet == FALSE )
278 {
279
280 portrait = draw->getTemplate()->getSelectedPortraitImage();
281 portraitObj = draw->getObject();
282 portraitSet = TRUE;
283
284 } // end if
285 else if( draw->getTemplate()->getSelectedPortraitImage() != portrait )
286 portrait = NULL;
287
288 } // end if
289
290 } // end for, drawble id iterator
291
292 // set the portrait image
293 setPortraitByObject( portraitObj );
294
295} // end populateMultiSelect

300void ControlBar::updateContextMultiSelect( void )
301{
302 Drawable *draw;
303 Object *obj;
304 const CommandButton *command;
305 GameWindow *win;
306 Int objectsThatCanDoCommand[ MAX_COMMANDS_PER_SET ];
307 Int i;
308
309 // zero the array that counts how many objects can do each command
310 memset( objectsThatCanDoCommand, 0, sizeof( objectsThatCanDoCommand ) );
311
312 // santiy
313 DEBUG_ASSERTCRASH( TheInGameUI->getSelectCount() > 1,
314 ("updateContextMultiSelect: TheInGameUI only has '%d' things selected\n",
315 TheInGameUI->getSelectCount()) );
316
317 // get the list of drawable IDs from the in game UI
318 const DrawableList *selectedDrawables = TheInGameUI->getAllSelectedDrawables();
319
320 // sanity
321 DEBUG_ASSERTCRASH( selectedDrawables->empty() == FALSE, ("populateMultiSelect: Drawable list is empty\n") );
322
323 // loop through all the selected drawable IDs
324 for( DrawableListCIt it = selectedDrawables->begin();
325 it != selectedDrawables->end(); ++it )
326 {
327
328 // get the drawable from the ID
329 draw = *it;
330
331 if (draw->getObject()->isKindOf(KINDOF_IGNORED_IN_GUI)) // ignore these guys
332 continue;
333
334
335 // get the object
336 obj = draw->getObject();
337
338 // sanity
339 if( obj == NULL )
340 continue;
341
342 // for each of the visible command windows make sure the object can execute the command
343 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )
344 {
345
346 // get the control window
347 win = m_commandWindows[ i ];
348
349 // our implementation doesn't necessarily make use of the max possible command buttons
350 if (!win) continue;
351
352 // don't consider hidden windows
353 if( win->winIsHidden() == TRUE )
354 continue;
355
356 // get the command
357 command = (const CommandButton *)GadgetButtonGetData(win);
358 if( command == NULL )
359 continue;
360
361 // can we do the command
362 CommandAvailability availability = getCommandAvailability( command, obj, win );
363
364 win->winClearStatus( WIN_STATUS_NOT_READY );
365 win->winClearStatus( WIN_STATUS_ALWAYS_COLOR );
366
367 // enable/disable the window control
368 switch( availability )
369 {
370 case COMMAND_HIDDEN:
371 win->winHide( TRUE );
372 break;
373 case COMMAND_RESTRICTED:
374 win->winEnable( FALSE );
375 break;
376 case COMMAND_NOT_READY:
377 win->winEnable( FALSE );
378 win->winSetStatus( WIN_STATUS_NOT_READY );
379 break;
380 case COMMAND_CANT_AFFORD:
381 win->winEnable( FALSE );
382 win->winSetStatus( WIN_STATUS_ALWAYS_COLOR );
383 break;
384 default:
385 win->winEnable( TRUE );
386 break;
387 }
388
389 //If button is a CHECK_LIKE, then update it's status now.
390 if( BitTest( command->getOptions(), CHECK_LIKE ) )
391 {
392 GadgetCheckLikeButtonSetVisualCheck( win, availability == COMMAND_ACTIVE );
393 }
394
395 if( availability == COMMAND_AVAILABLE || availability == COMMAND_ACTIVE )
396 objectsThatCanDoCommand[ i ]++;
397
398 } // end for i
399
400 } // end for, selected drawables
401
402 //
403 // for each command, if any objects can do the command we enable the window, otherwise
404 // we disable it
405 //
406 for( i = 0; i < MAX_COMMANDS_PER_SET; i++ )
407 {
408 // our implementation doesn't necessarily make use of the max possible command buttons
409 if (! m_commandWindows[ i ]) continue;
410
411 // don't consider hidden commands
412 if( m_commandWindows[ i ]->winIsHidden() == TRUE )
413 continue;
414
415 // don't consider slots that don't have commands
416 if( m_commonCommands[ i ] == NULL )
417 continue;
418
419 // check the count of objects that can do the command and enable/disable the control,
420 if( objectsThatCanDoCommand[ i ] > 0 )
421 m_commandWindows[ i ]->winEnable( TRUE );
422 else
423 m_commandWindows[ i ]->winEnable( FALSE );
424
425 } // end for i
426
427 // After Every change to the m_commandWIndows, we need to show fill in the missing blanks with the images
428 // removed from multiplayer branch
429 //showCommandMarkers();
430
431
432} // end updateContextMultiSelect
