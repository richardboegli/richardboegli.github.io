106void W3dUtilityClass::Convert_Shader(const W3dShaderStruct & shader,ShaderClass * set)
107{
108 set->Set_Depth_Compare ((ShaderClass::DepthCompareType)W3d_Shader_Get_Depth_Compare(&shader));
109 set->Set_Depth_Mask ((ShaderClass::DepthMaskType)W3d_Shader_Get_Depth_Mask(&shader));
110 set->Set_Color_Mask (ShaderClass::COLOR_WRITE_ENABLE);
111 set->Set_Dst_Blend_Func ((ShaderClass::DstBlendFuncType)W3d_Shader_Get_Dest_Blend_Func(&shader));
112 set->Set_Fog_Func (ShaderClass::FOG_DISABLE);
113 set->Set_Primary_Gradient ((ShaderClass::PriGradientType)W3d_Shader_Get_Pri_Gradient(&shader));
114 set->Set_Secondary_Gradient((ShaderClass::SecGradientType)W3d_Shader_Get_Sec_Gradient(&shader));
115 set->Set_Src_Blend_Func ((ShaderClass::SrcBlendFuncType)W3d_Shader_Get_Src_Blend_Func(&shader));
116 set->Set_Texturing ((ShaderClass::TexturingType)W3d_Shader_Get_Texturing(&shader));
117 set->Set_Alpha_Test ((ShaderClass::AlphaTestType)W3d_Shader_Get_Alpha_Test(&shader));
118// Jani: No cull mode settings for w3d shader - problem?
119// set->Set_Dither_Mask (ShaderClass::DITHER_ENABLE);
120 // The tools do not enable setting post-detail stuff, and in any case we probably wouldn't
121 // want to confuse the artists with the distinction between detail and post-detail settings at
122 // this point, so we just copy the detail settings into the post-detail settings.
123// set->Set_Post_Detail_Color_Func ((ShaderClass::DetailColorFuncType) W3d_Shader_Get_Post_Detail_Color_Func (&shader));
124// set->Set_Post_Detail_Alpha_Func ((ShaderClass::DetailAlphaFuncType) W3d_Shader_Get_Post_Detail_Alpha_Func (&shader));
125 set->Set_Post_Detail_Color_Func ((ShaderClass::DetailColorFuncType) W3d_Shader_Get_Detail_Color_Func (&shader));
126 set->Set_Post_Detail_Alpha_Func ((ShaderClass::DetailAlphaFuncType) W3d_Shader_Get_Detail_Alpha_Func (&shader));
127}
