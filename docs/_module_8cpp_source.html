250void UpgradeMuxData::muxDataProcessUpgradeRemoval(Object* obj) const
251{
252 if( !m_removalUpgradeNames.empty() )
253 {
254 std::vector<AsciiString>::const_iterator it;
255 for( it = m_removalUpgradeNames.begin();
256 it != m_removalUpgradeNames.end();
257 it++)
258 {
259 const UpgradeTemplate* theTemplate = TheUpgradeCenter->findUpgrade( *it );
260 if( !theTemplate && !it->isEmpty() && !it->isNone())
261 {
262 DEBUG_CRASH(("An upgrade module references %s, which is not an Upgrade", it->str()));
263 throw INI_INVALID_DATA;
264 }
265
266 obj->removeUpgrade(theTemplate);
267 }
268 }
269}

272Bool UpgradeMuxData::isTriggeredBy(const std::string &upgrade) const
273{

276 {
277 AsciiString trigger = *it;

279 {
280 return TRUE;
281 }
282 }
283
284 return FALSE;

288void UpgradeMuxData::getUpgradeActivationMasks(UpgradeMaskType& activation, UpgradeMaskType& conflicting) const
289{
290 // already computed.
291 if (!m_activationUpgradeNames.empty() || !m_conflictingUpgradeNames.empty())
292 {
293 m_activationMask.clear();
294 m_conflictingMask.clear();
295
296 std::vector<AsciiString>::const_iterator it;
297 for( it = m_activationUpgradeNames.begin();
298 it != m_activationUpgradeNames.end();
299 it++)
300 {
301 const UpgradeTemplate* theTemplate = TheUpgradeCenter->findUpgrade( *it );
302 if( !theTemplate && !it->isEmpty() && !it->isNone())
303 {
304 DEBUG_CRASH(("An upgrade module references %s, which is not an Upgrade", it->str()));
305 throw INI_INVALID_DATA;
306 }
307
308 m_activationMask.set( theTemplate->getUpgradeMask() );
309 }
310
311 for( it = m_conflictingUpgradeNames.begin();
312 it != m_conflictingUpgradeNames.end();
313 it++)
314 {
315 const UpgradeTemplate* theTemplate = TheUpgradeCenter->findUpgrade( *it );
316 if( !theTemplate && !it->isEmpty() && !it->isNone())
317 {
318 DEBUG_CRASH(("An upgrade module references %s, which is not an Upgrade", it->str()));
319 throw INI_INVALID_DATA;
320 }
321 m_conflictingMask.set( theTemplate->getUpgradeMask() );
322 }
323
324 // We set the trigger upgrade names with the activationUpgradeNames entries to be used later.
325 // We have to do this because the activationUpgradeNames are toasted just below.
326 m_triggerUpgradeNames = m_activationUpgradeNames;
327
328 //Clear the names now that we've cached the values!
329 m_activationUpgradeNames.clear();
330 m_conflictingUpgradeNames.clear();
331 }
332 activation = m_activationMask;
333 conflicting = m_conflictingMask;
334}
