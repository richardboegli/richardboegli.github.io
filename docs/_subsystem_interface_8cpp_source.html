70static const Real MIN_TIME_THRESHOLD = 0.0002f; // .2 msec. [8/13/2003]
71void SubsystemInterface::UPDATE(void)
72{
73 __int64 startTime64;
74 __int64 endTime64,freq64;
75 GetPrecisionTimerTicksPerSec(&freq64);
76 GetPrecisionTimer(&startTime64);
77 m_startTimeConsumed = s_msConsumed;
78 update();
79 GetPrecisionTimer(&endTime64);
80 m_curUpdateTime = ((double)(endTime64-startTime64))/((double)(freq64));
81 Real subTime = s_msConsumed - m_startTimeConsumed;
82 if (m_name.isEmpty()) return;
83 if (m_curUpdateTime>MIN_TIME_THRESHOLD) {
84 m_dumpUpdate = true;
85 }
86 if (m_curUpdateTime > MIN_TIME_THRESHOLD/10.0f) {
87 //DLOG(Debug::Format("Subsys %s total time %.2f, subTime %.2f, net time %.2f\n",
88 // m_name.str(), m_curUpdateTime*1000, subTime*1000, (m_curUpdateTime-subTime)*1000 ));
89
90 m_curUpdateTime -= subTime;
91 s_msConsumed += m_curUpdateTime;
92 } else {
93 m_curUpdateTime = 0;
94 }
95
96}
