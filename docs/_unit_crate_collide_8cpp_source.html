56Bool UnitCrateCollide::executeCrateBehavior( Object *other )
57{
58 UnsignedInt unitCount = getUnitCrateCollideModuleData()->m_unitCount;
59 ThingTemplate const *unitType = TheThingFactory->findTemplate( getUnitCrateCollideModuleData()->m_unitType );
60
61 if( unitType == NULL )
62 {
63 return FALSE;
64 }
65
66 for( Int unitIndex = 0; unitIndex < unitCount; unitIndex++ )
67 {
68 Team *creationTeam = other->getControllingPlayer()->getDefaultTeam();
69 Object *newObj = TheThingFactory->newObject( unitType, creationTeam );
70 if( newObj )
71 {
72 Coord3D creationPoint = *other->getPosition();
74 FindPositionOptions fpOptions;
75 fpOptions.minRadius = 0.0f;
76 fpOptions.maxRadius = 20.0f;
77 ThePartitionManager->findPositionAround( &creationPoint,
78 &fpOptions,
79 &creationPoint );
80
81 newObj->setOrientation( other->getOrientation() );
82 newObj->setPosition( &creationPoint );
83 }
84 }
85
86 //Play a crate pickup sound.
87 AudioEventRTS soundToPlay = TheAudio->getMiscAudio()->m_crateFreeUnit;
88 soundToPlay.setObjectID( other->getID() );
89 TheAudio->addAudioEvent(&soundToPlay);
90
91 return TRUE;
92}
