58BOOL MeshMoldOptions::OnInitDialog()
59{
60 COptionsPanel::OnInitDialog();
61
62 m_updating = true;
63 m_anglePopup.SetupPopSliderButton(this, IDC_ANGLE_POPUP, this);
64 m_scalePopup.SetupPopSliderButton(this, IDC_SCALE_POPUP, this);
65 m_HeightPopup.SetupPopSliderButton(this, IDC_HEIGHT_POPUP, this);
66
67 CWnd *pWnd = GetDlgItem(IDC_MESHMOLD_TREEVIEW);
68 CRect rect(0,0,100,100);
69 if (pWnd) pWnd->GetWindowRect(&rect);
70
71 CButton *pRaise = (CButton *)GetDlgItem(IDC_RAISE_LOWER);
72 if (pRaise) {
73 pRaise->SetCheck(1);
74 }
75
76 ScreenToClient(&rect);
77 rect.DeflateRect(2,2,2,2);
78 m_moldTreeView.Create(TVS_HASLINES|TVS_LINESATROOT|TVS_HASBUTTONS|
79 TVS_SHOWSELALWAYS|TVS_DISABLEDRAGDROP, rect, this, IDC_TERRAIN_TREEVIEW);
80 m_moldTreeView.ShowWindow(SW_SHOW);
81
82 {
83 char dirBuf[_MAX_PATH];
84 char findBuf[_MAX_PATH];
85 char fileBuf[_MAX_PATH];
86 Int i;
87
88 strcpy(dirBuf, ".\\data\\Editor\\Molds");
89 int len = strlen(dirBuf);
90
91 if (len > 0 && dirBuf[len - 1] != '\\') {
92 dirBuf[len++] = '\\';
93 dirBuf[len] = 0;
94 }
95 strcpy(findBuf, dirBuf);
96 strcat(findBuf, "*.w3d");
97
98 FilenameList filenameList;
99 TheFileSystem->getFileListInDirectory(AsciiString(dirBuf), AsciiString("*.w3d"), filenameList, FALSE);
100
101 if (filenameList.size() > 0) {
102 HTREEITEM child = NULL;
103 FilenameList::iterator it = filenameList.begin();
104 do {
105 AsciiString filename = *it;
106
107 len = filename.getLength();
108 if (len<5) continue;
109 strcpy(fileBuf, filename.str());
110 for (i=strlen(fileBuf)-1; i>0; i--) {
111 if (fileBuf[i] == '.') {
112 // strip off .w3d file extension.
113 fileBuf[i] = 0;
114 }
115 }
116 char *nameStart = fileBuf;
117 for (i=0; i<strlen(fileBuf)-1; i++) {
118 if (fileBuf[i] == '\\') {
119 nameStart = fileBuf+i+1;
120 }
121 }
122
123 TVINSERTSTRUCT ins;
124 // not found, so add it.
125 ::memset(&ins, 0, sizeof(ins));
126 ins.hParent = TVI_ROOT;
127 ins.hInsertAfter = TVI_SORT;
128 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
129 ins.item.lParam = -1;
130 ins.item.pszText = nameStart;
131 ins.item.cchTextMax = strlen(nameStart);
132 child = m_moldTreeView.InsertItem(&ins);
133
134 ++it;
135 } while (it != filenameList.end());
136 if (child) m_moldTreeView.SelectItem(child);
137 }
138 }
139
140 m_staticThis = this;
141 m_updating = false;
142 setAngle(m_currentAngle);
143 setScale(m_currentScale);
144 setHeight(m_currentHeight);
145 return TRUE; // return TRUE unless you set the focus to a control
146 // EXCEPTION: OCX Property Pages should return FALSE
147}

185void MeshMoldOptions::GetPopSliderInfo(const long sliderID, long *pMin, long *pMax, long *pLineSize, long *pInitial)
186{
187 switch (sliderID) {
188 case IDC_ANGLE_POPUP:
189 *pMin = MIN_ANGLE;
190 *pMax = MAX_ANGLE;
191 *pInitial = m_currentAngle;
192 *pLineSize = 1;
193 break;
194
195 case IDC_HEIGHT_POPUP:
196 *pMin = MIN_HEIGHT;
197 *pMax = MAX_HEIGHT;
198 *pInitial = floor(m_currentHeight/MAP_HEIGHT_SCALE+0.5f);
199 *pLineSize = 1;
200 break;
201
202 case IDC_SCALE_POPUP:
203 *pMin = MIN_SCALE;
204 *pMax = MAX_SCALE;
205 *pInitial = floor(m_currentScale*100 + 0.5f);
206 if (*pInitial > *pMax/2) *pMax = *pInitial*2;
207 *pLineSize = 1;
208 break;
209
210 default:
211 // uh-oh!
212 DEBUG_CRASH(("Missing ID."));
213 break;
214 } // switch
215}

217void MeshMoldOptions::PopSliderChanged(const long sliderID, long theVal)
218{
219 CString str;
220 CWnd *pEdit;
221 switch (sliderID) {
222
223 case IDC_ANGLE_POPUP:
224 m_currentAngle = theVal;
225 str.Format("%d",m_currentAngle);
226 pEdit = m_staticThis->GetDlgItem(IDC_ANGLE_EDIT);
227 if (pEdit) pEdit->SetWindowText(str);
228 break;
229
230 case IDC_HEIGHT_POPUP:
231 m_currentHeight = theVal*MAP_HEIGHT_SCALE;
232 str.Format("%.2f",m_currentHeight);
233 pEdit = m_staticThis->GetDlgItem(IDC_HEIGHT_EDIT);
234 if (pEdit) pEdit->SetWindowText(str);
235 MeshMoldTool::updateMeshLocation(false);
236 break;
237
238 case IDC_SCALE_POPUP:
239 m_currentScale = theVal/100.0f; //scale is in %
240 str.Format("%d",theVal);
241 pEdit = m_staticThis->GetDlgItem(IDC_SCALE_EDIT);
242 if (pEdit) pEdit->SetWindowText(str);
243 MeshMoldTool::updateMeshLocation(false);
244 break;
245
246 default:
247 DEBUG_CRASH(("Missing ID."));
248 break;
249 } // switch
250}

349BOOL MeshMoldOptions::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
350{
351 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
352 if (pHdr->hdr.hwndFrom == m_moldTreeView.m_hWnd) {
353
354 if (pHdr->hdr.code == TVN_SELCHANGED) {
355 char buffer[_MAX_PATH];
356 HTREEITEM hItem = m_moldTreeView.GetSelectedItem();
357 TVITEM item;
358 ::memset(&item, 0, sizeof(item));
359 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT|TVIF_STATE;
360 item.hItem = hItem;
361 item.pszText = buffer;
362 item.cchTextMax = sizeof(buffer)-2;
363 m_moldTreeView.GetItem(&item);
364 m_meshModelName = AsciiString(buffer);
365 if (m_doingPreview) {
366 MeshMoldTool::updateMeshLocation(false);
367 }
368 }
369 }
370 return COptionsPanel::OnNotify(wParam, lParam, pResult);
371}
