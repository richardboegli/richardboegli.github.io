86static Real calcSlowDownDist(Real curSpeed, Real desiredSpeed, Real maxBraking)
87{
88 Real delta = curSpeed - desiredSpeed;
89 if (delta <= 0)
90 return 0.0f;
91
92 Real dist = (sqr(delta) / fabs(maxBraking)) * 0.5f;
93
94 // use a little fudge so that things can stop "on a dime" more easily...
95 const Real FUDGE = 1.05f;
96 return dist * FUDGE;
97}

114// return the angle delta (in 3-space) we turned.
115static Real tryToRotateVector3D(
116 Real maxAngle, // if negative, it's a percent (0...1) of the dist to rotate 'em
117 const Vector3& inCurDir,
118 const Vector3& inGoalDir,
119 Vector3& actualDir
120)
121{
122 if (isNearlyZero(maxAngle))
123 {
124 actualDir = inCurDir;
125 return 0.0f;
126 }
127
128 Vector3 curDir = inCurDir;
129 curDir.Normalize();
130
131 Vector3 goalDir = inGoalDir;
132 goalDir.Normalize();
133
134 // dot of two unit vectors is cos of angle between them.
135 Real cosine = Vector3::Dot_Product(curDir, goalDir);
136 // bound it in case of numerical error
137 Real angleBetween = (Real)ACos(clamp(-1.0f, cosine, 1.0f));
138
139 if (maxAngle < 0)
140 {
141 maxAngle = -maxAngle * angleBetween;
142 if (isNearlyZero(maxAngle))
143 {
144 actualDir = inCurDir;
145 return 0.0f;
146 }
147 }
148
149 if (fabs(angleBetween) <= maxAngle)
150 {
151 // close enough
152 actualDir = goalDir;
153 }
154 else
155 {
156 // nah, try as much as we can in the right dir.
157 // we need to rotate around the axis perpendicular to these two vecs.
158 // but: cross of two vectors is the perpendicular axis!
159#ifdef ALLOW_TEMPORARIES
160 Vector3 objCrossGoal = Vector3::Cross_Product(curDir, goalDir);
161 objCrossGoal.Normalize();
162#else
163 Vector3 objCrossGoal;
164 Vector3::Normalized_Cross_Product(curDir, goalDir, &objCrossGoal);
165#endif
166
167 angleBetween = maxAngle;
168 Matrix3D rotMtx(objCrossGoal, angleBetween);
169 actualDir = rotMtx.Rotate_Vector(curDir);
170 }
171
172 return angleBetween;
173}

176static Real tryToOrientInThisDirection3D(Object* obj, Real maxTurnRate, const Vector3& desiredDir)
177{
178 Vector3 actualDir;
179 Real relAngle = tryToRotateVector3D(maxTurnRate, obj->getTransformMatrix()->Get_X_Vector(), desiredDir, actualDir);
180 if (relAngle != 0.0f)
181 {
182 Vector3 objPos(obj->getPosition()->x, obj->getPosition()->y, obj->getPosition()->z);
183
184 Matrix3D newXform;
185 newXform.buildTransformMatrix( objPos, actualDir );
186
187 obj->setTransformMatrix( &newXform );
188 }
189 return relAngle;
190}

198//-----------------------------------------------------------------------------
199static void calcDirectionToApplyThrust(
200 const Object* obj,
201 const PhysicsBehavior* physics,
202 const Coord3D& ingoalPos,
203 Real maxAccel,
204 Vector3& goalDir
205)
206{
207 /*
208 our meta-goal here is to calculate the direction we should apply our motive force
209 in order to minimize the angle between (our velocity) and (direction towards goalpos).
210
211 this is complicated by the fact that we generally have an intrinsic velocity already,
212 that must be accounted for, and by the fact that we can only apply force in our
213 forward-x-direction (with a thrust-angle-range), and (due to limited range) might not
214 be able to apply the force in the optimal direction!
215 */
216
217 // convert to Vector3, to use all its handy stuff
218 Vector3 objPos(obj->getPosition()->x, obj->getPosition()->y, obj->getPosition()->z);
219 Vector3 goalPos(ingoalPos.x, ingoalPos.y, ingoalPos.z);
220
221 Vector3 vecToGoal = goalPos - objPos;
222 if (isNearlyZero(vecToGoal.Length2()))
223 {
224 // goal pos is essentially same as current pos, so just stay the same & return
225 goalDir = obj->getTransformMatrix()->Get_X_Vector();
226 return;
227 }
228
229 /*
230 get our cur vel into a useful Vector3 form
231 */
232 Vector3 curVel(physics->getVelocity()->x, physics->getVelocity()->y, physics->getVelocity()->z);
233
234 // add gravity to our vel so that we account for it in our calcs
235 curVel.Z += TheGlobalData->m_gravity;
236
237 Bool foundSolution = false;
238 Real distToGoalSqr = vecToGoal.Length2();
239 Real distToGoal = sqrt(distToGoalSqr);
240 Real curVelMagSqr = curVel.Length2();
241 Real curVelMag = sqrt(curVelMagSqr);
242 Real maxAccelSqr = sqr(maxAccel);
243
244 Real denom = curVelMagSqr - maxAccelSqr;
245 if (!isNearlyZero(denom))
246 {
247 // solve the (greatly simplified) quadratic...
248 Real t = (distToGoal * (curVelMag + maxAccel)) / denom;
249 Real t2 = (distToGoal * (curVelMag - maxAccel)) / denom;
250 if (t >= 0 || t2 >= 0)
251 {
252 // choose the smallest positive t.
253 if (t < 0 || (t2 >= 0 && t2 < t))
254 t = t2;
255
256 // plug it in.
257 if (!isNearlyZero(t))
258 {
259 goalDir.X = (vecToGoal.X / t) - curVel.X;
260 goalDir.Y = (vecToGoal.Y / t) - curVel.Y;
261 goalDir.Z = (vecToGoal.Z / t) - curVel.Z;
262 goalDir.Normalize();
263 foundSolution = true;
264 }
265 }
266 }
267 if (!foundSolution)
268 {
269 // Doh... no (useful) solution. revert to dumb.
270 goalDir = vecToGoal;
271 goalDir.Normalize();
272 }

274}

952//-------------------------------------------------------------------------------------------------
953void Locomotor::locoUpdate_moveTowardsPosition(Object* obj, const Coord3D& goalPos,
954 Real onPathDistToGoal, Real desiredSpeed, Bool *blocked)
955{
956 setFlag(MAINTAIN_POS_IS_VALID, false);
957
958 BodyDamageType bdt = obj->getBodyModule()->getDamageState();
959 Real maxSpeed = getMaxSpeedForCondition(bdt);
960
961 // sanity, we cannot use desired speed that is greater than our max speed we are capable of moving at
962 if( desiredSpeed > maxSpeed )
963 desiredSpeed = maxSpeed;
964
965 Real distToStopAtMaxSpeed = (maxSpeed/getBraking()) * (maxSpeed)/2.0f;
966 if (onPathDistToGoal>PATHFIND_CELL_SIZE_F && onPathDistToGoal > distToStopAtMaxSpeed)
967 {
968 setFlag(IS_BRAKING, false);
969 m_brakingFactor = 1.0f;
970 }
971 PhysicsBehavior *physics = obj->getPhysics();
972 if (physics == NULL)
973 {
974 DEBUG_CRASH(("you can only apply Locomotors to objects with Physics"));
975 return;
976 }
977
978 // Skip moveTowardsPosition if physics say you're stunned
979 if(physics->getIsStunned())
980 {
981 return;
982 }
983
984#ifdef DEBUG_OBJECT_ID_EXISTS
985// DEBUG_ASSERTLOG(obj->getID() != TheObjectIDToDebug, ("locoUpdate_moveTowardsPosition %f %f %f (dtg %f, spd %f), speed %f (%f)\n",goalPos.x,goalPos.y,goalPos.z,onPathDistToGoal,desiredSpeed,physics->getSpeed(),physics->getForwardSpeed2D()));
986#endif
987
988 //
989 // do not allow for invalid positions that the pathfinder cannot handle ... for airborne
990 // objects we don't need the pathfinder so we'll ignore this
991 //
992 if( BitTest( m_template->m_surfaces, LOCOMOTORSURFACE_AIR ) == false &&
993 !TheAI->pathfinder()->validMovementTerrain(obj->getLayer(), this, obj->getPosition()) &&
994 !getFlag(ALLOW_INVALID_POSITION))
995 {
996 // Somehow, we have gotten to an invalid location.
997 if (fixInvalidPosition(obj, physics))
998 {
999 // the we adjusted us toward a legal position, so just return.
1000 return;
1001 }
1002 }
1003
1004 // If the actual distance is farther, then use the actual distance so we get there.
1005 Real dx = goalPos.x - obj->getPosition()->x;
1006 Real dy = goalPos.y - obj->getPosition()->y;
1007 Real dz = goalPos.z - obj->getPosition()->z;
1008 Real dist = sqrt(dx*dx+dy*dy);
1009 if (dist>onPathDistToGoal)
1010 {
1011 if (!obj->isKindOf(KINDOF_PROJECTILE) && dist>2*onPathDistToGoal)
1012 {
1013 setFlag(IS_BRAKING, true);
1014 }
1015 onPathDistToGoal = dist;
1016 }
1017
1018 Coord3D nullAccel;
1019
1020 Bool treatAsAirborne = false;
1021 Coord3D pos = *obj->getPosition();
1022 Real heightAboveSurface = pos.z - TheTerrainLogic->getLayerHeight(pos.x, pos.y, obj->getLayer());
1023
1024 if( obj->getStatusBits().test( OBJECT_STATUS_DECK_HEIGHT_OFFSET ) )
1025 {
1026 heightAboveSurface -= obj->getCarrierDeckHeight();
1027 }
1028
1029 if (heightAboveSurface > -(3*3)*TheGlobalData->m_gravity)
1030 {
1031 // If we get high enough to stay up for 3 frames, then we left the ground.
1032 treatAsAirborne = true;
1033 }
1034 // We apply a zero acceleration to all units, as the call to
1035 // applyMotiveForce flags an object as being "driven" by a locomotor, rather
1036 // than being pushed around by objects bumping it.
1037 nullAccel.x = nullAccel.y = nullAccel.z = 0;
1038 physics->applyMotiveForce(&nullAccel);
1039
1040 if (*blocked)
1041 {
1042 if (desiredSpeed > physics->getVelocityMagnitude())
1043 {
1044 *blocked = false;
1045 }
1046 if (treatAsAirborne && BitTest( m_template->m_surfaces, LOCOMOTORSURFACE_AIR ) )
1047 {
1048 // Airborne flying objects don't collide for now. jba.
1049 *blocked = false;
1050 }
1051 }
1052
1053 if (*blocked)
1054 {
1055 physics->scrubVelocity2D(desiredSpeed); // stop if we are about to run into the blocking object.
1056 Real turnRate = getMaxTurnRate(obj->getBodyModule()->getDamageState());
1057 if (m_template->m_wanderWidthFactor == 0.0f)
1058 {
1059 *blocked = (TURN_NONE != rotateObjAroundLocoPivot(obj, goalPos, turnRate));
1060 }
1061
1062 // it is very important to be sure to call this in all situations, even if not moving in 2d space.
1063 handleBehaviorZ(obj, physics, goalPos);
1064 return;
1065 }
1066
1067 if (
1068// srj sez: I don't know why we didn't want HOVERs to allow to "brake".
1069// we actually really want them to, because it allows much more precise destination positioning.
1070// m_template->m_appearance == LOCO_HOVER ||
1071 m_template->m_appearance == LOCO_WINGS)
1072 {
1073 setFlag(IS_BRAKING, false);
1074 }
1075
1076 Bool wasBraking = obj->getStatusBits().test( OBJECT_STATUS_BRAKING );
1077
1078 physics->setTurning(TURN_NONE);
1079 if (getAllowMotiveForceWhileAirborne() || !treatAsAirborne)
1080 {
1081 switch (m_template->m_appearance)
1082 {
1083 case LOCO_LEGS_TWO:
1084 moveTowardsPositionLegs(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1085 break;
1086 case LOCO_CLIMBER:
1087 moveTowardsPositionClimb(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1088 break;
1089 case LOCO_WHEELS_FOUR:
1090 case LOCO_MOTORCYCLE:
1091 moveTowardsPositionWheels( obj, physics, goalPos, onPathDistToGoal, desiredSpeed );
1092 break;
1093 case LOCO_TREADS:
1094 moveTowardsPositionTreads(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1095 break;
1096 case LOCO_HOVER:
1097 moveTowardsPositionHover(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1098 break;
1099 case LOCO_WINGS:
1100 moveTowardsPositionWings(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1101 break;
1102 case LOCO_THRUST:
1103 moveTowardsPositionThrust(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1104 break;
1105 case LOCO_OTHER:
1106 default:
1107 moveTowardsPositionOther(obj, physics, goalPos, onPathDistToGoal, desiredSpeed);
1108 break;
1109 }
1110 }
1111
1112 handleBehaviorZ(obj, physics, goalPos);
1113 // Objects that are braking don't follow the normal physics, so they end up at their destination exactly.
1114 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_BRAKING ), getFlag(IS_BRAKING) );
1115
1116 if (wasBraking)
1117 {
1118 #define MIN_VEL (PATHFIND_CELL_SIZE_F/(LOGICFRAMES_PER_SECOND))
1119
1120 Coord3D pos = *obj->getPosition();
1121 if (obj->isKindOf(KINDOF_PROJECTILE))
1122 {
1123 // Projectiles never stop braking once they start. jba.
1124 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_BRAKING ) );
1125 // Projectiles cheat in 3 dimensions.
1126 dist = sqrt(dx*dx+dy*dy+dz*dz);
1127 Real vel = physics->getVelocityMagnitude();
1128 if (vel < MIN_VEL)
1129 vel = MIN_VEL;
1130 if (vel > dist)
1131 vel = dist; // do not overcompensate!
1132 // Normalize.
1133 if (dist > 0.001f)
1134 {
1135 dist = 1.0f / dist;
1136 dx *= dist;
1137 dy *= dist;
1138 dz *= dist;
1139 pos.x += dx * vel;
1140 pos.y += dy * vel;
1141 pos.z += dz * vel;
1142 }
1143 }
1144 else
1145 {
1146 // not projectiles only cheat in x & y.
1147 // Normalize.
1148 if (dist > 0.001f)
1149 {
1150 Real vel = fabs(physics->getForwardSpeed2D());
1151 if (vel < MIN_VEL)
1152 vel = MIN_VEL;
1153 if (vel > dist)
1154 vel = dist; // do not overcompensate!
1155 dist = 1.0f / dist;
1156 dx *= dist;
1157 dy *= dist;
1158 pos.x += dx * vel;
1159 pos.y += dy * vel;
1160 }
1161 }
1162 obj->setPosition(&pos);
1163 }
1164
1165}

1167//-------------------------------------------------------------------------------------------------
1168void Locomotor::moveTowardsPositionTreads(Object* obj, PhysicsBehavior *physics, const Coord3D& goalPos, Real onPathDistToGoal, Real desiredSpeed)
1169{
1170
1171 // sanity, we cannot use desired speed that is greater than our max speed we are capable of moving at
1172 BodyDamageType bdt = obj->getBodyModule()->getDamageState();
1173 Real maxSpeed = getMaxSpeedForCondition(bdt);
1174 if( desiredSpeed > maxSpeed )
1175 desiredSpeed = maxSpeed;
1176
1177 Real maxAcceleration = getMaxAcceleration(bdt);
1178
1179 // Locomotion for treaded vehicles, ie tanks.
1180
1181 //
1182 // Orient toward goal position
1183 //
1184// Real angle = obj->getOrientation();
1185// Real relAngle = ThePartitionManager->getRelativeAngle2D( obj, &goalPos );
1186// Real desiredAngle = angle + relAngle;
1187 Real relAngle ;
1188 PhysicsTurningType rotating = rotateTowardsPosition(obj, goalPos, &relAngle);
1189 physics->setTurning(rotating);
1190
1191 //
1192 // Modulate speed according to turning. The more we have to turn, the slower we go
1193 //
1194 const Real QUAETERPI = PI / 4.0f;
1195 Real angleCoeff = (Real)fabs( relAngle ) / QUAETERPI;
1196 if (angleCoeff > 1.0f)
1197 angleCoeff = 1.0;
1198
1199
1200 Real dx = obj->getPosition()->x - goalPos.x;
1201 Real dy = obj->getPosition()->y - goalPos.y;
1202
1203
1204 Real goalSpeed = (1.0f - angleCoeff) * desiredSpeed;
1205
1206
1207// if (speed < m_minTurnSpeed)
1208// speed = m_minTurnSpeed;
1209
1210 Real actualSpeed = physics->getForwardSpeed2D();
1211 Real slowDownTime = actualSpeed / getBraking();
1212 Real slowDownDist = (actualSpeed/1.50f) * slowDownTime;
1213
1214 if (sqr(dx)+sqr(dy)<sqr(2*PATHFIND_CELL_SIZE_F) && angleCoeff > 0.05) {
1215 goalSpeed = actualSpeed*0.6f;
1216 }
1217
1218 if (onPathDistToGoal < slowDownDist && !getFlag(IS_BRAKING) && !getFlag(NO_SLOW_DOWN_AS_APPROACHING_DEST))
1219 {
1220 setFlag(IS_BRAKING, true);
1221 m_brakingFactor = 1.1f;
1222 }
1223
1224 if (onPathDistToGoal>PATHFIND_CELL_SIZE_F && onPathDistToGoal > 2.0*slowDownDist)
1225 {
1226 setFlag(IS_BRAKING, false);
1227 }
1228
1229 if (getFlag(IS_BRAKING))
1230 {
1231 m_brakingFactor = slowDownDist/onPathDistToGoal;
1232 m_brakingFactor *= m_brakingFactor;
1233 if (m_brakingFactor>MAX_BRAKING_FACTOR) {
1234 m_brakingFactor = MAX_BRAKING_FACTOR;
1235 }
1236 if (slowDownDist>onPathDistToGoal) {
1237 goalSpeed = actualSpeed-getBraking();
1238 if (goalSpeed<0.0f) goalSpeed= 0.0f;
1239 } else if (slowDownDist>onPathDistToGoal*0.75f) {
1240 goalSpeed = actualSpeed-getBraking()/2.0f;
1241 if (goalSpeed<0.0f) goalSpeed = 0.0f;
1242 } else {
1243 goalSpeed = actualSpeed;
1244 }
1245 }
1246
1247
1248 //DEBUG_LOG(("Actual speed %f, Braking factor %f, slowDownDist %f, Pathdist %f, goalSpeed %f\n",
1249 // actualSpeed, m_brakingFactor, slowDownDist, onPathDistToGoal, goalSpeed));
1250
1251 //
1252 // Maintain goal speed
1253 //
1254 Real speedDelta = goalSpeed - actualSpeed;
1255 if (speedDelta != 0.0f)
1256 {
1257 Real mass = physics->getMass();
1258 Real acceleration = (speedDelta > 0.0f) ? maxAcceleration : -m_brakingFactor*getBraking();
1259 Real accelForce = mass * acceleration;
1260
1261 /*
1262 don't accelerate/brake more than necessary. do a quick calc to
1263 see how much force we really need to achieve our goal speed...
1264 */
1265 Real maxForceNeeded = mass * speedDelta;
1266 if (fabs(accelForce) > fabs(maxForceNeeded))
1267 accelForce = maxForceNeeded;
1268
1269 const Coord3D *dir = obj->getUnitDirectionVector2D();
1270
1271 Coord3D force;
1272 force.x = accelForce * dir->x;
1273 force.y = accelForce * dir->y;
1274 force.z = 0.0f;
1275

1594 Real maxTurnRate = getMaxTurnRate(condition); // in rads/frame

1598
1599 Real minTurnCircum = maxSpeed * (2*PI / maxTurnRate);
1600
1601 so therefore our minimum turn radius is:

1603 Real minTurnRadius = minTurnCircum / 2*PI;
1604
1605 so we just eliminate the middleman:

1612

1929 }

1932#ifdef USE_ZDIR_DAMPING
1933 Real zDirDamping = 0.0f;
1934#endif
1935
1936 //out of the handleBehaviorZ() function
1937 Coord3D pos = *obj->getPosition();
1938 if( m_preferredHeight != 0.0f && !getFlag(PRECISE_Z_POS) )
1939 {

1941 Real surfaceHt = getSurfaceHtAtPt(pos.x, pos.y);

1943// localGoalPos.z = goalPos.z;

1945 delta *= getPreferredHeightDamping();

1947
1948#ifdef USE_ZDIR_DAMPING

1950 // so we tend to "level out" at that height. we don't use this till
1951 // below, but go ahead and calc it now...

1957#endif

1966
1967 // what direction do we need to thrust in, in order to reach the goalpos?
1968 Vector3 desiredThrustDir;
1969 calcDirectionToApplyThrust(obj, physics, localGoalPos, maxAccel, desiredThrustDir);

1972 Real maxThrustAngle = (maxTurnRate > 0) ? (m_template->m_maxThrustAngle) : 0;
1973 Vector3 thrustDir;
1974 Real thrustAngle = tryToRotateVector3D(maxThrustAngle, forwardDir, desiredThrustDir, thrustDir);
1975
1976 // note that we are trying to orient in the direction of our vel, not the dir of our thrust.
1977 if (!isNearlyZero(physics->getVelocityMagnitude()))
1978 {
1979 const Coord3D* veltmp = physics->getVelocity();

1981 Bool adjust = true;
1982 if( obj->getStatusBits().test( OBJECT_STATUS_BRAKING ) )
1983 {
1984 // align to target, cause that's where we're going anyway.
1985
1986 vel.Set(goalPos.x - pos.x, goalPos.y-pos.y, goalPos.z-pos.z);
1987 if (isNearlyZero(sqr(vel.X)+sqr(vel.Y)+sqr(vel.Z))) {
1988 // we are at target.
1989 adjust = false;
1990 }
1991 maxTurnRate = 3*maxTurnRate;
1992 }
1993#ifdef USE_ZDIR_DAMPING
1994 if (zDirDamping != 0.0f)
1995 {
1996 Vector3 vel2D(veltmp->x, veltmp->y, 0);

1998 tryToRotateVector3D(-zDirDamping, vel, vel2D, vel);

2000#endif

2002 /*Real orient =*/ tryToOrientInThisDirection3D(obj, maxTurnRate, vel);
2003 }
2004 }
2005
2006 if (forwardSpeedDelta != 0.0f || thrustAngle != 0.0f)
2007 {
2008 if (maxForwardSpeed <= 0.0f)
2009 {
2010 maxForwardSpeed = 0.01f; // In some cases, this is 0, hack for now. jba.
2011 }
2012 Real damping = clamp(0.0f, maxAccel / maxForwardSpeed, 1.0f);
2013 Vector3 curVel(physics->getVelocity()->x, physics->getVelocity()->y, physics->getVelocity()->z);

2015 Vector3 accelVec = thrustDir * maxAccel - curVel * damping;
2016 //DEBUG_LOG(("accel %f (max %f) vel %f (max %f) damping %f\n",accelVec.Length(),maxAccel,curVel.Length(),maxForwardSpeed,damping));

2018 Real mass = physics->getMass();
2019
2020 Coord3D force;
2021 force.x = mass * accelVec.X;
2022 force.y = mass * accelVec.Y;
2023 force.z = mass * accelVec.Z;
2024
2025 // apply forces to object
2026 physics->applyMotiveForce( &force );
2027 }

2046Real Locomotor::calcLiftToUseAtPt(Object* obj, PhysicsBehavior *physics, Real curZ, Real surfaceAtPt, Real preferredHeight)
2047{
2048 /*
2049 take the classic equation:
2050
2051 x = x0 + v*t + 0.5*a*t^2
2052
2053 and solve for acceleration.
2054 */
2055 BodyDamageType bdt = obj->getBodyModule()->getDamageState();
2056 Real maxGrossLift = getMaxLift(bdt);
2057 Real maxNetLift = maxGrossLift + TheGlobalData->m_gravity; // note that gravity is always negative.
2058 if (maxNetLift < 0)
2059 maxNetLift = 0;
2060 Real curVelZ = physics->getVelocity()->z;
2061 // going down, braking is limited by net lift; going up, braking is limited by gravity
2062 Real maxAccel;
2063 if (getFlag(ULTRA_ACCURATE))
2064 maxAccel = (curVelZ < 0) ? 2*maxNetLift : -2*maxNetLift;
2065 else
2066 maxAccel = (curVelZ < 0) ? maxNetLift : TheGlobalData->m_gravity;
2067 // see how far we need to slow to dead stop, given max braking
2068 Real desiredAccel;
2069 const Real TINY_ACCEL = 0.001f;
2070 if (fabs(maxAccel) > TINY_ACCEL)
2071 {
2072 Real deltaZ = preferredHeight - curZ;
2073 // calc how far it will take for us to go from cur speed to zero speed, at max accel.
2074 // Real brakeDist = calcSlowDownDist(curVelZ, 0, maxAccel);
2075 // in theory, the above is the correct calculation, but in practice,
2076 // doesn't work in some situations (eg, opening of USA01 map). Why, I dunno.
2077 // But for now I have gone back to the old, looks-incorrect-to-me-but-works calc. (srj)
2078 Real brakeDist = (sqr(curVelZ) / fabs(maxAccel));
2079 if (fabs(brakeDist) > fabs(deltaZ))
2080 {
2081 // if the dist-to-accel (or dist-to-brake) is further than the dist-to-go,
2082 // use the max accel.
2083 desiredAccel = maxAccel;
2084 }
2085 else if (fabs(curVelZ) > m_template->m_speedLimitZ)
2086 {
2087 // or, if we're going too fast, limit it here.
2088 desiredAccel = m_template->m_speedLimitZ - curVelZ;
2089 }
2090 else
2091 {
2092 // ok, figure out the correct accel to use to get us there at zero.
2093 //
2094 // dz = v t + 0.5 a t^2
2095 // thus
2096 // a = 2(dz - v t)/t^2
2097 // and
2098 // t = (-v +- sqrt(v*v + 2*a*dz))/a
2099 //
2100 // but if we assume t=1, then
2101 // a=2(dz-v)
2102 // then, plug it back in and see if t is really 1...
2103 desiredAccel = 2.0f * (deltaZ - curVelZ);
2104 }
2105 }
2106 else
2107 {
2108 desiredAccel = 0.0f;
2109 }
2110 Real liftToUse = desiredAccel - TheGlobalData->m_gravity;
2111 if (getFlag(ULTRA_ACCURATE))
2112 {
2113 // in ultra-accurate mode, we allow cheating.
2114 const Real UP_FACTOR = 3.0f;
2115 if (liftToUse > UP_FACTOR*maxGrossLift)
2116 liftToUse = UP_FACTOR*maxGrossLift;
2117 // srj sez: we used to clip lift to zero here (not allowing neg lift).
2118 // however, I now think that allowing neg lift in ultra-accurate mode is
2119 // a good and desirable thing; in particular, it enables jets to complete
2120 // "short" landings more accurately (previously they sometimes would "float"
2121 // down, which sucked.) if you need to bump this back to zero, check it carefully...
2122 else if (liftToUse < -maxGrossLift)
2123 liftToUse = -maxGrossLift;
2124 }
2125 else
2126 {
2127 if (liftToUse > maxGrossLift)
2128 liftToUse = maxGrossLift;
2129 else if (liftToUse < 0.0f)
2130 liftToUse = 0.0f;
2131 }
2132
2133 return liftToUse;
2134}
