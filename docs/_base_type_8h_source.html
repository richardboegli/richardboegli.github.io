179// note, this function depends on the cpu rounding mode, which we set to CHOP every frame,
180// but apparently tends to be left in unpredictable modes by various system bits of
181// code, so use this function with caution -- it might not round in the way you want.
182__forceinline long fast_float2long_round(float f)
183{
184 long i;
185
186 __asm {
187 fld [f]
188 fistp [i]
189 }
190
191 return i;
192}
193
194// super fast float trunc routine, works always (independent of any FPU modes)
195// code courtesy of Martin Hoffesommer (grin)
196__forceinline float fast_float_trunc(float f)
197{
198 _asm
199 {
200 mov ecx,[f]
201 shr ecx,23
202 mov eax,0xff800000
203 xor ebx,ebx
204 sub cl,127
205 cmovc eax,ebx
206 sar eax,cl
207 and [f],eax
208 }
209 return f;
210}
211
212// same here, fast floor function
213__forceinline float fast_float_floor(float f)
214{
215 static unsigned almost1=(126<<23)|0x7fffff;
216 if (*(unsigned *)&f &0x80000000)
217 f-=*(float *)&almost1;
218 return fast_float_trunc(f);
219}
220
221// same here, fast ceil function
222__forceinline float fast_float_ceil(float f)
223{
224 static unsigned almost1=(126<<23)|0x7fffff;
225 if ( (*(unsigned *)&f &0x80000000)==0)
226 f+=*(float *)&almost1;
227 return fast_float_trunc(f);
228}

291inline Real Coord2D::toAngle( void ) const
292{
293 const Real len = length();
294 if (len == 0.0f)
295 return 0.0f;
296
297 Real c = x/len;
298 // bound it in case of numerical error
299 if (c < -1.0f)
300 c = -1.0f;
301 else if (c > 1.0f)
302 c = 1.0f;
303
304 return y < 0.0f ? -ACos(c) : ACos(c);
305} // end toAngle
