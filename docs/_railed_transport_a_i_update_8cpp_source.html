95void RailedTransportAIUpdate::loadWaypointData( void )
96{
97 const RailedTransportAIUpdateModuleData *modData = getRailedTransportAIUpdateModuleData();
98
99 // find all the possible waypoint paths we can use
100 Waypoint *start, *end;
101 AsciiString name;
102 for( Int i = 0; i < MAX_WAYPOINT_PATHS; ++i )
103 {
104
105 // find start waypoint
106 name.format( "%sStart%02d", modData->m_pathPrefixName.str(), i + 1 );
107 start = TheTerrainLogic->getWaypointByName( name );
108
109 // find end waypoint
110 name.format( "%sEnd%02d", modData->m_pathPrefixName.str(), i + 1 );
111 end = TheTerrainLogic->getWaypointByName( name );
112
113 // if we have a start and an end, we have a valid path
114 if( start && end )
115 {
116
117 m_path[ i ].startWaypointID = start->getID();
118 m_path[ i ].endWaypointID = end->getID();
119 m_numPaths++;
120
121 } // end if
122
123 } // end for i
124
125 // waypoint data is loaded
126 m_waypointDataLoaded = TRUE;
127
128} // end loadWaypointData

132void RailedTransportAIUpdate::pickAndMoveToInitialLocation( void )
133{
134 Object *us = getObject();
135 const Coord3D *ourPos = us->getPosition();
136
137 // select the path with the closest ending waypoint to our location
138 Waypoint *waypoint, *closestEndWaypoint = NULL;
139 Int closestPath = INVALID_PATH;
140 Real closestDist = 99999999.9f;
141 for( Int i = 0; i < m_numPaths; ++i )
142 {
143
144 // get this waypoint
145 waypoint = TheTerrainLogic->getWaypointByID( m_path[ i ].endWaypointID );
146 if( waypoint )
147 {
148 Coord3D v;
149
150 // vector from us to waypoint
151 v.x = waypoint->getLocation()->x - ourPos->x;
152 v.y = waypoint->getLocation()->y - ourPos->y;
153 v.z = waypoint->getLocation()->z - ourPos->z;
154
155 // what is the distance
156 Real dist = v.length();
157
158 // if this distance is smaller, use this one
159 if( dist < closestDist )
160 {
161
162 closestDist = dist;
163 closestPath = i;
164 closestEndWaypoint = waypoint;
165
166 } // end if
167
168 } // end if
169
170 } // end for i
171
172 // a path must have been found
173 DEBUG_ASSERTCRASH( closestPath != INVALID_PATH,
174 ("No suitable starting waypoint path could be found for '%s'\n",
175 us->getTemplate()->getName().str()) );
176
177 // follow the waypoint path to its destination end point
178 aiFollowWaypointPath( closestEndWaypoint, CMD_FROM_AI );
179
180 // this is now our current path
181 m_currentPath = closestPath;
182
183 // we are now "in transit"
184 setInTransit( TRUE );
185
186} // end pickAndMoveToInitialLocation

190UpdateSleepTime RailedTransportAIUpdate::update( void )
191{
192 Object *us = getObject();
193
194 // load the waypoint data if not loaded
195 if( m_waypointDataLoaded == FALSE )
196 loadWaypointData();
197
198 // extend base class
199 UpdateSleepTime result;
200 result = AIUpdateInterface::update();
201
202 // railed transports move ultra accurate like
203 Locomotor *currentLocomotor = getCurLocomotor();
204 if( currentLocomotor )
205 currentLocomotor->setUltraAccurate( TRUE );
206
207 //
208 // if we have no current path selected pick one and move to the end waypoint of that
209 // path. this will set us up in an initial position at the end of the closest path
210 // so that stuff can be loaded into us
211 //
212 if( m_currentPath == INVALID_PATH && m_numPaths > 0 )
213 pickAndMoveToInitialLocation();
214
215 //
216 // if we're in transit, see if we're close enough to the destination waypoint to be
217 // considered as "there" and open up the dock
218 //
219 if( m_inTransit )
220 {
221
222 // sanity
223 DEBUG_ASSERTCRASH( m_currentPath != INVALID_PATH,
224 ("RailedTransportAIUpdate: Invalid current path '%s'\n", m_currentPath) );
225
226 // get our target waypoint
227 Waypoint *waypoint = TheTerrainLogic->getWaypointByID( m_path[ m_currentPath ].endWaypointID );
228
229 // sanity
230 DEBUG_ASSERTCRASH( waypoint, ("RailedTransportAIUpdate: Invalid target waypoint\n") );
231
232 // how far away are we from the target waypoint
233 const Coord3D *start = us->getPosition();
234 const Coord3D *end = waypoint->getLocation();
235 Coord3D v;
236 v.x = end->x - start->x;
237 v.y = end->y - start->y;
238 v.z = end->z - start->z;
239 Real dist = v.length();
240 if( dist <= 5.0f || isIdle() )
241 {
242
243 // we are no longer in transit
244 setInTransit( FALSE );
245
246 } // end if
247
248 } // end if
249
250 return UPDATE_SLEEP_NONE;
251
252} // end update

296void RailedTransportAIUpdate::privateExecuteRailedTransport( CommandSourceType cmdSource )
297{
298 Object *us = getObject();
299
300 //
301 // find us our railed dock interface, note that we MUST go through the modules here because
302 // if we just call the method getReailedTransportDockUpdateInterface, it will execute
303 // the method for *THIS AI UPDATE MODULE* which of course is not our dock update
304 //
305 RailedTransportDockUpdateInterface *rtdui = NULL;
306 for( BehaviorModule **u = us->getBehaviorModules(); *u; ++u )
307 if( (rtdui = (*u)->getRailedTransportDockUpdateInterface()) != NULL )
308 break;
309
310 // if we've in the process of loading or unloading anything we can't do a transport sequence
311 if( rtdui == NULL || rtdui->isLoadingOrUnloading() )
312 return;
313
314 // pick the next path
315 if( ++m_currentPath >= m_numPaths )
316 m_currentPath = 0;
317
318 // find the start waypoint for our current path
319 Waypoint *startWaypoint = TheTerrainLogic->getWaypointByID( m_path[ m_currentPath ].startWaypointID );
320 DEBUG_ASSERTCRASH( startWaypoint, ("RailedTransportAIUpdate: Start waypoint not found\n") );
321
322 // follow this waypoint path
323 aiFollowWaypointPath( startWaypoint, CMD_FROM_AI );
324
325 // we are now in transit
326 setInTransit( TRUE );
327
328} // end privateExecuteRailedTransport

332void RailedTransportAIUpdate::privateEvacuate( Int exposeStealthUnits, CommandSourceType cmdSource )
333{
334 Object *us = getObject();
335
336 //
337 // find us our railed dock interface, note that we MUST go through the modules here because
338 // if we just call the method getReailedTransportDockUpdateInterface, it will execute
339 // the method for *THIS AI UPDATE MODULE* which of course is not our dock update
340 //
341 RailedTransportDockUpdateInterface *rtdui = NULL;
342 for( BehaviorModule **u = us->getBehaviorModules(); *u; ++u )
343 if( (rtdui = (*u)->getRailedTransportDockUpdateInterface()) != NULL )
344 break;
345
346 // sanity
347 if( rtdui == NULL )
348 return;
349
350 // can't unload when in transit
351 if( m_inTransit == TRUE )
352 return;
353
354 // cannot evacuate when we're loading or unloading anything
355 if( rtdui->isLoadingOrUnloading() )
356 return;
357
358 // start the manual undocking process
359 rtdui->unloadAll();
360
361} // end privateEvacuate

377void RailedTransportAIUpdate::xfer( Xfer *xfer )
378{
379 XferVersion currentVersion = 1;
380 XferVersion version = currentVersion;
381 xfer->xferVersion( &version, currentVersion );
382
383 // extend base class
384 AIUpdateInterface::xfer(xfer);
385
386 xfer->xferBool(&m_inTransit);
387 xfer->xferInt(&m_numPaths);
388 if (m_numPaths > MAX_WAYPOINT_PATHS) {
389 DEBUG_CRASH(("m_numPaths %d exceeds limit %d.", m_numPaths, MAX_WAYPOINT_PATHS));
390 throw SC_INVALID_DATA;
391 }
392 Int i;
393 for (i=0; i<m_numPaths; i++) {
394 xfer->xferUnsignedInt(&m_path[i].startWaypointID);
395 xfer->xferUnsignedInt(&m_path[i].endWaypointID);
396 }
397 xfer->xferInt(&m_currentPath);
398 xfer->xferBool(&m_waypointDataLoaded);
399
400} // end xfer
