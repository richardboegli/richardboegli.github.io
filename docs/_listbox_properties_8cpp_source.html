221static void removeScrollbar( GameWindow *listbox )
222{
223 ListboxData *listData = (ListboxData *)listbox->winGetUserData();
224
225 // delete the up button
226 TheWindowManager->winDestroy( listData->upButton );
227 listData->upButton = NULL;
228
229 // delete down button
230 TheWindowManager->winDestroy( listData->downButton );
231 listData->downButton = NULL;
232
233 // delete the slider
234 TheWindowManager->winDestroy( listData->slider );
235 listData->slider = NULL;
236
237 // remove the scrollbar flag from the listbox data
238 listData->scrollBar = FALSE;
239
240} // end removeScrollbar
