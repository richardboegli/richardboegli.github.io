173static Bool isAnythingTooClose2D(const std::vector<Object*>& v, const Coord3D& pos, Real minDistSqr)
174{
175 for (std::vector<Object*>::const_iterator it = v.begin(); it != v.end(); ++it)
176 {
177 const Coord3D* p = (*it)->getPosition();
178 Real distSqr = sqr(p->x - pos.x) + sqr(p->y - pos.y);
179 if (distSqr < minDistSqr)
180 return true;
181 }
182 return false;
183}

195Object* GenerateMinefieldBehavior::placeMineAt(const Coord3D& pt, const ThingTemplate* mineTemplate, Team* team, const Object* producer)
196{
197 Coord3D tmp = pt;
198 tmp.z = 99999.0f;
199 PathfindLayerEnum layer = TheTerrainLogic->getHighestLayerForDestination(&tmp);
200
201 if (layer == LAYER_GROUND && TheTerrainLogic->isUnderwater(pt.x, pt.y))
202 return NULL;
203
204 if (layer == LAYER_GROUND && TheTerrainLogic->isCliffCell(pt.x, pt.y))
205 return NULL;
206
207 Real orient = GameLogicRandomValueReal(-PI, PI);
208
209 // if the mine will be "mostly" under a structure, don't place it.
210 // for now, "mostly" means "central third of radius would overlap"
211 const GenerateMinefieldBehaviorModuleData* d = getGenerateMinefieldBehaviorModuleData();
212 GeometryInfo geom = mineTemplate->getTemplateGeometryInfo();
213 Real mineRadius = mineTemplate->getTemplateGeometryInfo().getBoundingCircleRadius();
214 geom.expandFootprint(mineRadius * -(1.0f - d->m_skipIfThisMuchUnderStructure));
215 ObjectIterator *iter = ThePartitionManager->iteratePotentialCollisions( &pt, geom, orient );
216 MemoryPoolObjectHolder hold(iter);
217 for (Object* them = iter->first(); them; them = iter->next())
218 {
219 if (them->isKindOf(KINDOF_STRUCTURE))
220 return NULL;
221 }
222
223 Object* mine = TheThingFactory->newObject(mineTemplate, team);
224 mine->setPosition(&pt);
225 mine->setOrientation(orient);
226 mine->setProducer(producer);
227
228 for (BehaviorModule** bmi = mine->getBehaviorModules(); *bmi; ++bmi)
229 {
230 LandMineInterface* lmi = (*bmi)->getLandMineInterface();
231 if (lmi)
232 {
233 lmi->setScootParms(*producer->getPosition(), pt);
234 break;
235 }
236 }
237
238 // Keep track of the mines
239 if (mine && d->m_upgradable)
240 {
241 m_mineList.push_back(mine->getID());
242 }
243
244 return mine;
245}

249void GenerateMinefieldBehavior::placeMinesAlongLine(const Coord3D& posStart, const Coord3D& posEnd, const ThingTemplate* mineTemplate, Bool skipOneAtStart)
250{
251 const Object* obj = getObject();
252 const GenerateMinefieldBehaviorModuleData* d = getGenerateMinefieldBehaviorModuleData();
253 Team* team = obj->getControllingPlayer()->getDefaultTeam();
254
255 Real dx = posEnd.x - posStart.x;
256 Real dy = posEnd.y - posStart.y;
257 Real len = sqrt(sqr(dx) + sqr(dy));
258 Real mineRadius = mineTemplate->getTemplateGeometryInfo().getBoundingCircleRadius();
259 Real mineDiameter = mineRadius * 2.0f;
260 Real mineJitter = mineRadius*d->m_randomJitter;
261 Int numMines = REAL_TO_INT_CEIL(len / mineDiameter);
262 if (numMines < 1)
263 numMines = 1;
264 Real inc = len/numMines;
265 for (Real place = skipOneAtStart ? inc : 0; place <= len; place += inc)
266 {
267 Coord3D pt;
268 pt.x = posStart.x + place * dx / len;
269 pt.y = posStart.y + place * dy / len;
270 pt.z = TheTerrainLogic->getGroundHeight( pt.x, pt.y );
271 offsetBySmallRandomAmount(pt, mineJitter);
272 placeMineAt(pt, mineTemplate, team, obj);
273 }
274}

278static void makeCorner(const Coord3D& pos, Real majorRadius, Real minorRadius, const Matrix3D& mtx, Coord3D& corner)
279{
280 Vector3 tmp;
281 tmp.X = majorRadius;
282 tmp.Y = minorRadius;
283 tmp.Z = 0;
284 Matrix3D::Transform_Vector(mtx, tmp, &tmp);
285 corner.x = tmp.X;
286 corner.y = tmp.Y;
287 corner.z = tmp.Z;

292void GenerateMinefieldBehavior::placeMinesAroundRect(const Coord3D& pos, Real majorRadius, Real minorRadius, const ThingTemplate* mineTemplate)
293{
294 const Object* obj = getObject();
295 const Matrix3D* mtx = obj->getTransformMatrix();
296
297 Coord3D pt[4];
298 makeCorner(pos, majorRadius, minorRadius, *mtx, pt[0]);
299 makeCorner(pos, -majorRadius, minorRadius, *mtx, pt[1]);
300 makeCorner(pos, -majorRadius, -minorRadius, *mtx, pt[2]);
301 makeCorner(pos, majorRadius, -minorRadius, *mtx, pt[3]);
302
303 placeMinesAlongLine(pt[0], pt[1], mineTemplate, true);
304 placeMinesAlongLine(pt[1], pt[2], mineTemplate, true);
305 placeMinesAlongLine(pt[2], pt[3], mineTemplate, true);
306 placeMinesAlongLine(pt[3], pt[0], mineTemplate, true);
307}

311void GenerateMinefieldBehavior::placeMinesAroundCircle(const Coord3D& pos, Real radius, const ThingTemplate* mineTemplate)
312{
313 const Object* obj = getObject();
314 const GenerateMinefieldBehaviorModuleData* d = getGenerateMinefieldBehaviorModuleData();
315 Team* team = obj->getControllingPlayer()->getDefaultTeam();
316
317 Real circum = 2.0f * PI * radius;
318 Real mineRadius = mineTemplate->getTemplateGeometryInfo().getBoundingCircleRadius();
319 Real mineDiameter = mineRadius * 2.0f;
320 Real mineJitter = mineRadius*d->m_randomJitter;
321 Int numMines = REAL_TO_INT_CEIL(circum / mineDiameter);
322 if (numMines < 1)
323 numMines = 1;
324 Real angleInc = (2*PI)/numMines;
325 Real angleLim = (2*PI) - angleInc*0.5f;
326 for (Real angle = 0; angle < angleLim; angle += angleInc)
327 {
328 Coord3D pt;
329 pt.x = pos.x + radius * Cos(angle);
330 pt.y = pos.y + radius * Sin(angle);
331 pt.z = TheTerrainLogic->getGroundHeight( pt.x, pt.y );
332 offsetBySmallRandomAmount(pt, mineJitter);
333 placeMineAt(pt, mineTemplate, team, obj);
334 }

339void GenerateMinefieldBehavior::placeMinesInFootprint(const GeometryInfo& geom, const ThingTemplate* mineTemplate)
340{
341 const Object* obj = getObject();
342 const GenerateMinefieldBehaviorModuleData* d = getGenerateMinefieldBehaviorModuleData();
343 Team* team = obj->getControllingPlayer()->getDefaultTeam();
344
345 Real area = geom.getFootprintArea();
346 Int numMines = REAL_TO_INT_CEIL(d->m_minesPerSquareFoot * area);
347 if (numMines < 1)
348 numMines = 1;
349
350 const Coord3D* target = getMinefieldTarget();
351 std::vector<Object*> minesCreatedSoFar;
352 Real minDistSqr = sqr(mineTemplate->getTemplateGeometryInfo().getBoundingCircleRadius() * 2.0f);
353 for (int i = 0; i < numMines; ++i)
354 {
355 Coord3D pt;
356 Int maxRetry = 100;
357 do
358 {
359 geom.makeRandomOffsetWithinFootprint(pt);
360 pt.x += target->x;
361 pt.y += target->y;
362 pt.z += target->z;
363 --maxRetry;
364 } while (isAnythingTooClose2D(minesCreatedSoFar, pt, minDistSqr) && maxRetry > 0);
365 DEBUG_ASSERTCRASH(maxRetry>0,("ran out of retries %f",minDistSqr));
366
367 if (getObject()->getGeometryInfo().isPointInFootprint(*target, pt))
368 continue;
369
370 Object* mine = placeMineAt(pt, mineTemplate, team, obj); // can return null.
371 if (mine)
372 minesCreatedSoFar.push_back(mine);
373 }
374}

378void GenerateMinefieldBehavior::placeMines()
379{
380 if (m_generated)
381 return;
382
383 m_generated = true;
384
385 const Object* obj = getObject();
386 const GenerateMinefieldBehaviorModuleData* d = getGenerateMinefieldBehaviorModuleData();
387 const ThingTemplate* mineTemplate = 0;
388
389 if (m_upgraded)
390 mineTemplate = TheThingFactory->findTemplate(d->m_mineNameUpgraded);
391 else
392 mineTemplate = TheThingFactory->findTemplate(d->m_mineName);
393
394 if (!mineTemplate)
395 {
396 DEBUG_CRASH(("mine %s not found\n",d->m_mineName.str()));
397 return;
398 }
399
400 const Coord3D* target = getMinefieldTarget();
401 if (d->m_smartBorder)
402 {
403 GeometryInfo geom = obj->getGeometryInfo();
404
405 if (!d->m_smartBorderSkipInterior)
406 {
407 geom = mineTemplate->getTemplateGeometryInfo();
408 placeMineAt(*target, mineTemplate, obj->getControllingPlayer()->getDefaultTeam(), obj);
409 }
410
411 if (d->m_alwaysCircular)
412 geom.set(GEOMETRY_CYLINDER, false, 1, geom.getBoundingCircleRadius(), geom.getBoundingCircleRadius());
413
414 Real mineRadius = mineTemplate->getTemplateGeometryInfo().getBoundingCircleRadius();
415 Real mineDiameter = mineRadius * 2.0f;
416 geom.expandFootprint(mineRadius);
417 do
418 {
419 if (geom.getGeomType() == GEOMETRY_BOX && !d->m_alwaysCircular)
420 {
421 placeMinesAroundRect(*target, geom.getMajorRadius(), geom.getMinorRadius(), mineTemplate);
422 }
423 else
424 {
425 placeMinesAroundCircle(*target, geom.getMajorRadius(), mineTemplate);
426 }
427 geom.expandFootprint(mineDiameter);
428 } while (geom.getBoundingCircleRadius() < d->m_distanceAroundObject);
429 }
430 else if (d->m_borderOnly)
431 {
432 GeometryInfo geom = obj->getGeometryInfo();
433 geom.expandFootprint(d->m_distanceAroundObject);
434
435 if (geom.getGeomType() == GEOMETRY_BOX && !d->m_alwaysCircular)
436 {
437 placeMinesAroundRect(*target, geom.getMajorRadius(), geom.getMinorRadius(), mineTemplate);
438 }
439 else
440 {
441 placeMinesAroundCircle(*target, geom.getMajorRadius(), mineTemplate);
442 }
443 }
444 else
445 {
446 GeometryInfo geom = obj->getGeometryInfo();
447 geom.expandFootprint(d->m_distanceAroundObject);
448
449 if (d->m_alwaysCircular)
450 geom.set(GEOMETRY_CYLINDER, false, 1, geom.getBoundingCircleRadius(), geom.getBoundingCircleRadius());
451
452 placeMinesInFootprint(geom, mineTemplate);
453 }
454
455 FXList::doFXObj(d->m_genFX, obj);
456
457}

461UpdateSleepTime GenerateMinefieldBehavior::update()
462{
463 // Test to see if we need to replace the current mines with upgraded ones
464 if (!m_upgraded && getGenerateMinefieldBehaviorModuleData()->m_upgradable)
465 {
466 if (m_generated)
467 {
468 // Upgraded minefield to next level for China Player
469 const UpgradeTemplate *upgradeTemplate = TheUpgradeCenter->findUpgrade( "Upgrade_ChinaEMPMines" );
470
471 if (upgradeTemplate)
472 {
473 UpgradeMaskType upgradeMask = upgradeTemplate->getUpgradeMask();
474 UpgradeMaskType objMask = getObject()->getObjectCompletedUpgradeMask();
475 if (objMask.testForAny(upgradeMask))
476 {
477 m_upgraded = TRUE;
478
479 // Remove all old mine objects if present
480 for (std::list<ObjectID>::iterator it = m_mineList.begin(); it != m_mineList.end(); ++it)
481 {
482 ObjectID objID = *it;
483 Object *obj = TheGameLogic->findObjectByID(objID);
484 if (obj)
485 {
486 TheGameLogic->destroyObject(obj);
487 }
488 }
489 m_mineList.clear();
490
491 // Place new mines down (Replace old ones that we removed
492 m_generated = false;
493 placeMines();
494 }
495 }
496 }
497
498 return UPDATE_SLEEP_NONE;
499 }
500
501 return UPDATE_SLEEP_FOREVER;
502}

523void GenerateMinefieldBehavior::xfer( Xfer *xfer )
524{
525
526 // version
527 XferVersion currentVersion = 1;
528 XferVersion version = currentVersion;
529 xfer->xferVersion( &version, currentVersion );
530
531 // base class
532 BehaviorModule::xfer( xfer );
533
534 // mux "base class"
535 UpgradeMux::upgradeMuxXfer( xfer );
536
537 // generated
538 xfer->xferBool( &m_generated );
539 xfer->xferBool( &m_hasTarget );
540 xfer->xferBool( &m_upgraded );
541
542 xfer->xferCoord3D( &m_target );
543
544 // spaces info count and objectID data
545 UnsignedByte spacesCount = m_mineList.size();
546 xfer->xferUnsignedByte( &spacesCount );
547 if( xfer->getXferMode() == XFER_SAVE )
548 {
549 // save all elements
550 std::list<ObjectID>::iterator it;
551 for( it = m_mineList.begin(); it != m_mineList.end(); ++it )
552 {
553 // object in this space
554 xfer->xferObjectID( &(*it) );
555 } // end for, it
556
557 } // end if, save
558 else if( xfer->getXferMode() == XFER_LOAD )
559 {
560 ObjectID objectID;
561 m_mineList.clear();
562
563 // read all elements
564 std::list<ObjectID>::iterator it;
565 it = m_mineList.begin();
566 for(int i = 0; i < spacesCount; ++i )
567 {
568 // read object id
569 xfer->xferObjectID( &objectID );
570
571 m_mineList.push_back(objectID);
572 } // end for, i
573 } // end else, load
574
575} // end xfer
