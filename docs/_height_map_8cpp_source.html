175UnsignedInt HeightMapRenderObjClass::doTheDynamicLight(VERTEX_FORMAT *vb, VERTEX_FORMAT *vbMirror, Vector3*light, Vector3*normal, W3DDynamicLight *pLights[], Int numLights)
176{
177#ifdef USE_NORMALS
178 return;
179#else
180 Real shadeR, shadeG, shadeB;
181 Int diffuse = vbMirror->diffuse;
182#ifdef _DEBUG
183 //vbMirror->diffuse += 30; // Shows which vertexes are geting touched by dynamic light. debug only.
184#endif
185
186 // (gth) avoiding the extra divides (compiler unfortunately didn't do this automatically...)
187 const float oo255 = (1.0f/255.0f);
188 shadeR = ((diffuse>>16)&0x00FF) * oo255;
189 shadeG = ((diffuse>>8)&0x00FF) * oo255;
190 shadeB = (diffuse&0x00FF) * oo255;
191
192 Int alpha = (diffuse>>24)&0x00FF;
193 Int k;
194 for (k=0; k<numLights; k++) {
195 W3DDynamicLight *pLight = pLights[k];
196 if (!pLight->isEnabled()) {
197 continue; // he is turned off.
198 }
199 Vector3 lightDirection(vbMirror->x, vbMirror->y, vbMirror->z);
200 Real factor = 1.0f;
201 switch(pLight->Get_Type()) {
202 case LightClass::POINT:
203 case LightClass::SPOT: {
204 Vector3 lightLoc = pLight->Get_Position();
205 lightDirection -= lightLoc;
206 double range, midRange;
207 pLight->Get_Far_Attenuation_Range(midRange, range);
208 Real dist = lightDirection.Length();
209 if (dist >= range) continue;
210 if (midRange < 0.1) continue;
211 factor = 1.0f - (dist - midRange) / (range - midRange);
212 factor = WWMath::Clamp(factor,0.0f,1.0f);
213
214 // (gth) normalize here since we have the length
215 lightDirection /= dist;
216 }
217 break;
218 case LightClass::DIRECTIONAL:
219 pLight->Get_Spot_Direction(lightDirection);
220 factor = 1.0;
221 break;
222 };
223
224 // (gth) unneeded due to above normalization
225 //lightDirection.Normalize();
226
227 Vector3 lightRay(-lightDirection.X, -lightDirection.Y, -lightDirection.Z);
228 Real shade = Vector3::Dot_Product(lightRay, *normal);
229 shade *= factor;
230 Vector3 diffuse;
231 pLight->Get_Diffuse(&diffuse);
232 Vector3 ambient;
233 pLight->Get_Ambient(&ambient);
234 if (shade > 1.0) shade = 1.0;
235 if(shade < 0.0f) shade = 0.0f;
236 shadeR += shade*diffuse.X;
237 shadeG += shade*diffuse.Y;
238 shadeB += shade*diffuse.Z;
239 shadeR += factor*ambient.X;
240 shadeG += factor*ambient.Y;
241 shadeB += factor*ambient.Z;
242
243 }
244 if (shadeR > 1.0) shadeR = 1.0;
245 if(shadeR < 0.0f) shadeR = 0.0f;
246 if (shadeG > 1.0) shadeG = 1.0;
247 if(shadeG < 0.0f) shadeG = 0.0f;
248 if (shadeB > 1.0) shadeB = 1.0;
249 if(shadeB < 0.0f) shadeB = 0.0f;
250 shadeR*=255.0f;
251 shadeG*=255.0f;
252 shadeB*=255.0f;
253
254// (gth) faster float to int conversion, return the result so we can re-use it.
255// vb->diffuse=REAL_TO_INT(shadeB) | (REAL_TO_INT(shadeG) << 8) | (REAL_TO_INT(shadeR) << 16) | ((int)alpha << 24);
256 UnsignedInt light_val = WWMath::Float_To_Int_Chop(shadeB) | (WWMath::Float_To_Int_Chop(shadeG) << 8) | (WWMath::Float_To_Int_Chop(shadeR) << 16) | ((int)alpha << 24);
257 vb->diffuse = light_val;
258 return light_val;
259
260#endif
261}

313Int HeightMapRenderObjClass::updateVB(DX8VertexBufferClass *pVB, char *data, Int x0, Int y0, Int x1, Int y1, Int originX, Int originY, WorldHeightMap *pMap, RefRenderObjListIterator *pLightsIterator)
314{
315 Int i,j;
316 Vector3 lightRay[MAX_GLOBAL_LIGHTS];
317 const Coord3D *lightPos;
318 Int xCoord, yCoord;
319 Int vn0,un0,vp1,up1;
320 Vector3 l2r,n2f,normalAtTexel;
321 Int vertsPerRow=(VERTEX_BUFFER_TILE_LENGTH)*4; //vertices per row of VB
322
323 Int cellOffset = 1;
324 if (HALF_RES_MESH) {
325 cellOffset = 2;
326 }
327
328 REF_PTR_SET(m_map, pMap); //update our heightmap pointer in case it changed since last call.
329 if (m_vertexBufferTiles && pMap)
330 {
331#ifdef _DEBUG
332 assert(x0 >= originX && y0 >= originY && x1>x0 && y1>y0 && x1<=originX+VERTEX_BUFFER_TILE_LENGTH && y1<=originY+VERTEX_BUFFER_TILE_LENGTH);
333#endif
334
335 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(pVB);
336 VERTEX_FORMAT *vbHardware = (VERTEX_FORMAT*)lockVtxBuffer.Get_Vertex_Array();
337 VERTEX_FORMAT *vBase = (VERTEX_FORMAT*)data;
338 // Note that we are building the vertex buffer data in the memory buffer, data.
339 // At the bottom, we will copy the final vertex data for one cell into the
340 // hardware vertex buffer.
341
342 for (j=y0; j<y1; j++)
343 {
344 VERTEX_FORMAT *vb = vBase;
345 if (HALF_RES_MESH) {
346 if (j&1) continue;
347 vb += ((j-originY)/2)*vertsPerRow/2; //skip to correct row in vertex buffer
348 vb += ((x0-originX)/2)*4; //skip to correct vertex in row.
349 } else {
350 vb += (j-originY)*vertsPerRow; //skip to correct row in vertex buffer
351 vb += (x0-originX)*4; //skip to correct vertex in row.
352 }
353 vn0 = getYWithOrigin(j)-cellOffset;
354 if (vn0 < -pMap->getDrawOrgY())
355 vn0=-pMap->getDrawOrgY();
356 vp1 = getYWithOrigin(j+cellOffset)+cellOffset;
357 if (vp1 >= pMap->getYExtent()-pMap->getDrawOrgY())
358 vp1=pMap->getYExtent()-pMap->getDrawOrgY()-1;
359
360 yCoord = getYWithOrigin(j)+pMap->getDrawOrgY();
361 for (i=x0; i<x1; i++)
362 {
363 if (HALF_RES_MESH) {
364 if (i&1) continue;
365 }
366 un0 = getXWithOrigin(i)-cellOffset;
367 if (un0 < -pMap->getDrawOrgX())
368 un0=-pMap->getDrawOrgX();
369 up1 = getXWithOrigin(i+cellOffset)+cellOffset;
370 if (up1 >= pMap->getXExtent()-pMap->getDrawOrgX())
371 up1=pMap->getXExtent()-pMap->getDrawOrgX()-1;
372 xCoord = getXWithOrigin(i)+pMap->getDrawOrgX();
373
374 //update the 4 vertices in this block
375 float U[4], V[4];
376 UnsignedByte alpha[4];
377 float UA[4], VA[4];
378 Bool flipForBlend = false; // True if the blend needs the triangles flipped.
379
380 if (pMap) {
381 pMap->getUVData(getXWithOrigin(i),getYWithOrigin(j),U, V, HALF_RES_MESH);
382 pMap->getAlphaUVData(getXWithOrigin(i),getYWithOrigin(j), UA, VA, alpha, &flipForBlend, HALF_RES_MESH);
383 }
384
385
386 for (Int lightIndex=0; lightIndex < TheGlobalData->m_numGlobalLights; lightIndex++)
387 {
388 lightPos=&TheGlobalData->m_terrainLightPos[lightIndex];
389 lightRay[lightIndex].Set(-lightPos->x,-lightPos->y, -lightPos->z);
390 }
391
392 //top-left sample
393 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset, getYWithOrigin(j)) - pMap->getDisplayHeight(un0, getYWithOrigin(j))));
394 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i), (getYWithOrigin(j)+cellOffset)) - pMap->getDisplayHeight(getXWithOrigin(i), vn0)));
395
396#ifdef ALLOW_TEMPORARIES
397 normalAtTexel= Normalize(Vector3::Cross_Product(l2r,n2f));
398#else
399 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
400#endif
401
402 vb->x=xCoord;
403 vb->y=yCoord;
404 vb->z= ((float)pMap->getDisplayHeight(getXWithOrigin(i), getYWithOrigin(j)))*MAP_HEIGHT_SCALE;
405 vb->x = ADJUST_FROM_INDEX_TO_REAL(vb->x);
406 vb->y = ADJUST_FROM_INDEX_TO_REAL(vb->y);
407 vb->u1=U[0];
408 vb->v1=V[0];
409 vb->u2=UA[0];
410 vb->v2=VA[0];
411 doTheLight(vb, lightRay, &normalAtTexel, pLightsIterator, alpha[0]);
412 vb++;
413
414 //top-right sample
415 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(up1 , getYWithOrigin(j) ) - pMap->getDisplayHeight(getXWithOrigin(i) , getYWithOrigin(j) )));
416 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset , (getYWithOrigin(j)+cellOffset) ) - pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset , vn0 )));
417
418#ifdef ALLOW_TEMPORARIES
419 normalAtTexel= Normalize(Vector3::Cross_Product(l2r,n2f));
420#else
421 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
422#endif
423
424 vb->x=xCoord+cellOffset;
425 vb->y=yCoord;
426 vb->z= ((float)pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset, getYWithOrigin(j)))*MAP_HEIGHT_SCALE;
427 vb->x = ADJUST_FROM_INDEX_TO_REAL(vb->x);
428 vb->y = ADJUST_FROM_INDEX_TO_REAL(vb->y);
429 vb->u1=U[1];
430 vb->v1=V[1];
431 vb->u2=UA[1];
432 vb->v2=VA[1];
433 doTheLight(vb, lightRay, &normalAtTexel, pLightsIterator, alpha[1]);
434 vb++;
435
436 //bottom-right sample
437 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(up1 , (getYWithOrigin(j)+cellOffset) ) - pMap->getDisplayHeight(getXWithOrigin(i) , (getYWithOrigin(j)+cellOffset) )));
438 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset , vp1 ) - pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset , getYWithOrigin(j) )));
439
440#ifdef ALLOW_TEMPORARIES
441 normalAtTexel= Normalize(Vector3::Cross_Product(l2r,n2f));
442#else
443 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
444#endif
445
446 vb->x=xCoord+cellOffset;
447 if (yCoord + 1 == pMap->getDrawOrgY() + m_y - 1) {
448 vb->y=yCoord+1;
449 } else {
450 vb->y=yCoord+cellOffset;
451 }
452 vb->z= ((float)pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset, getYWithOrigin(j)+cellOffset))*MAP_HEIGHT_SCALE;
453 vb->x = ADJUST_FROM_INDEX_TO_REAL(vb->x);
454 vb->y = ADJUST_FROM_INDEX_TO_REAL(vb->y);
455 vb->u1=U[2];
456 vb->v1=V[2];
457 vb->u2=UA[2];
458 vb->v2=VA[2];
459 doTheLight(vb, lightRay, &normalAtTexel, pLightsIterator, alpha[2]);
460 vb++;
461
462 //bottom-left sample
463 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i)+cellOffset , (getYWithOrigin(j)+cellOffset) ) - pMap->getDisplayHeight(un0 , (getYWithOrigin(j)+cellOffset) )));
464 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(pMap->getDisplayHeight(getXWithOrigin(i) , vp1 ) - pMap->getDisplayHeight(getXWithOrigin(i) , getYWithOrigin(j) )));
465
466#ifdef ALLOW_TEMPORARIES
467 normalAtTexel= Normalize(Vector3::Cross_Product(l2r,n2f));
468#else
469 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
470#endif
471
472 if (xCoord == pMap->getDrawOrgX()) {
473 vb->x=xCoord;
474 //if (vb->x < 0) vb->x = 0;
475 } else {
476 vb->x=xCoord;
477 }
478 if (yCoord + 1 == pMap->getDrawOrgY() + m_y - 1) {
479 vb->y=yCoord+1;
480 } else {
481 vb->y=yCoord+cellOffset;
482 }
483 vb->z= ((float)pMap->getDisplayHeight(getXWithOrigin(i), getYWithOrigin(j)+cellOffset))*MAP_HEIGHT_SCALE;
484 vb->x = ADJUST_FROM_INDEX_TO_REAL(vb->x);
485 vb->y = ADJUST_FROM_INDEX_TO_REAL(vb->y);
486 vb->u1=U[3];
487 vb->v1=V[3];
488 vb->u2=UA[3];
489 vb->v2=VA[3];
490 doTheLight(vb, lightRay, &normalAtTexel, pLightsIterator, alpha[3]);
491 vb++;
492
493 VERTEX_FORMAT *pCurVertices = vb-4;
494#ifdef FLIP_TRIANGLES // jba - reduces "diamonding" in some cases, not others. Better cliffs, though.
495 VERTEX_FORMAT tmpVertex;
496 if (flipForBlend) {
497 tmpVertex = pCurVertices[0];
498 pCurVertices[0] = pCurVertices[1];
499 pCurVertices[1] = pCurVertices[2];
500 pCurVertices[2] = pCurVertices[3];
501 pCurVertices[3] = tmpVertex;
502 }
503#endif
504
505 if (m_showImpassableAreas) {
506 // Color impassable cells "red"
507 DEBUG_ASSERTCRASH(PATHFIND_CELL_SIZE_F == MAP_XY_FACTOR, ("Pathfind must be terrain cell size, or this code needs reworking. John A."));
508 Real borderHiX = (pMap->getXExtent()-2*pMap->getBorderSizeInline())*MAP_XY_FACTOR;
509 Real borderHiY = (pMap->getYExtent()-2*pMap->getBorderSizeInline())*MAP_XY_FACTOR;
510 Bool border = pCurVertices[0].x == -MAP_XY_FACTOR || pCurVertices[0].y == -MAP_XY_FACTOR;
511 Bool cliffMapped = pMap->isCliffMappedTexture(getXWithOrigin(i), getYWithOrigin(j));
512 if (pCurVertices[0].x == borderHiX) {
513 border = true;
514 }
515 if (pCurVertices[0].y == borderHiY) {
516 border = true;
517 }
518 Bool isCliff = pMap->getCliffState(getXWithOrigin(i)+pMap->getDrawOrgX(), getYWithOrigin(j)+pMap->getDrawOrgY())
519 || showAsVisibleCliff(getXWithOrigin(i) + pMap->getDrawOrgX(), getYWithOrigin(j)+pMap->getDrawOrgY());
520
521 if ( isCliff || border || cliffMapped) {
522 Int cellX, cellY;
523 for (cellX=0; cellX<2; cellX++) {
524 for (cellY=0; cellY<2; cellY++) {
525 Int vertex = cellX+2*cellY;
526 if (border) {
527 Bool doBorder = false;
528 if (pCurVertices[vertex].y >= 0 && pCurVertices[vertex].y <= borderHiY) {
529 if (pCurVertices[vertex].x == 0 || pCurVertices[vertex].x == borderHiX) {
530 doBorder = true;
531 }
532 }
533 if (pCurVertices[vertex].x >= 0 && pCurVertices[vertex].x <= borderHiX) {
534 if (pCurVertices[vertex].y == 0 || pCurVertices[vertex].y == borderHiY) {
535 doBorder = true;
536 }
537 }
538 if (doBorder) {
539 pCurVertices[vertex].diffuse &= 0xFF0000ff; // blue with alpha.
540 }
541 } else if (isCliff) {
542 pCurVertices[vertex].diffuse &= 0xFFFF0000; // red with alpha.
543 }
544 if (cliffMapped && vertex==0) {
545 pCurVertices[vertex].diffuse &= 0xFF000000; // Black.
546 pCurVertices[vertex].diffuse |= 0xff00; // Add green.
547 }
548 }
549 }
550 }
551 }
552
553 // Note - We have been building the vertex buffer in the memory location.
554 // Now copy the set of vertices into the hardware buffer.
555 // We don't copy the whole vertex buffer because we often update only
556 // a couple of rows and its a lot faster to just copy the ones that change.
557 Int offset = pCurVertices - vBase;
558 memcpy(vbHardware+offset, pCurVertices, 4*sizeof(VERTEX_FORMAT));
559 }
560 }
561 return 0; //success.
562 }
563 return -1;
564}

724Int HeightMapRenderObjClass::updateVBForLightOptimized(DX8VertexBufferClass *pVB, char *data, Int x0, Int y0, Int x1, Int y1, Int originX, Int originY, W3DDynamicLight *pLights[], Int numLights)
725{
726 Int i,j,k;
727 Int vn0,un0,vp1,up1;
728 Vector3 l2r,n2f,normalAtTexel;
729 Int vertsPerRow=(VERTEX_BUFFER_TILE_LENGTH)*4; //vertices per row of VB
730
731 if (m_vertexBufferTiles && m_map)
732 {
733#ifdef _DEBUG
734 assert(x0 >= originX && y0 >= originY && x1>x0 && y1>y0 && x1<=originX+VERTEX_BUFFER_TILE_LENGTH && y1<=originY+VERTEX_BUFFER_TILE_LENGTH);
735#endif
736
737 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(pVB);
738 VERTEX_FORMAT *vBase = (VERTEX_FORMAT*)lockVtxBuffer.Get_Vertex_Array();
739 VERTEX_FORMAT *vb;
740
741 //
742 // (gth) the optimization in this function is to take advantage of verts in the same
743 // x,y position who have already computed their lighting. To do this, we need to set up
744 // some offsets in the vertex buffer. I've computed these offsets to be consistent with
745 // the formula's that Generals is using but in the case of the "half-res-mesh" I'm not
746 // sure things are correct...
747 //
748 Int quad_right_offset;
749 Int quad_below_offset;
750 Int quad_below_right_offset;
751
752 if (HALF_RES_MESH == false) {
753 // offset = (j-originY)*vertsPerRow+4*(i-originX);
754 quad_right_offset = 4;
755 quad_below_offset = vertsPerRow;
756 quad_below_right_offset = vertsPerRow + 4;
757
758 } else {
759 // offset = (j-originY)*vertsPerRow/4+2*(i-originX);
760 quad_right_offset = 2;
761 quad_below_offset = vertsPerRow/4;
762 quad_below_right_offset = vertsPerRow/4 + 2;
763 }
764
765 //
766 // i,j loop over the quads affected by the light. Each quad has its *own* 4 vertices. This
767 // means that for any vertex position on the map, there are actually 4 copies of the vertex.
768 //
769 for (j=y0; j<y1; j++)
770 {
771 if (HALF_RES_MESH) {
772 if (j&1) continue;
773 }
774 Int yCoord = getYWithOrigin(j)+m_map->getDrawOrgY()-m_map->getBorderSizeInline();
775 Bool intersect = false;
776 for (k=0; k<numLights; k++) {
777 if (pLights[k]->m_minY <= yCoord+1 &&
778 pLights[k]->m_maxY >= yCoord) {
779 intersect = true;
780 }
781 if (pLights[k]->m_prevMinY <= yCoord+1 &&
782 pLights[k]->m_prevMaxY >= yCoord) {
783 intersect = true;
784 }
785 }
786 if (!intersect) {
787 continue;
788 }
789 vn0 = getYWithOrigin(j)-1;
790 if (vn0 < -m_map->getDrawOrgY())
791 vn0=-m_map->getDrawOrgY();
792 vp1 = getYWithOrigin(j+1)+1;
793 if (vp1 >= m_map->getYExtent()-m_map->getDrawOrgY())
794 vp1=m_map->getYExtent()-m_map->getDrawOrgY()-1;
795
796 for (i=x0; i<x1; i++)
797 {
798 if (HALF_RES_MESH) {
799 if (i&1) continue;
800 }
801 Int xCoord = getXWithOrigin(i)+m_map->getDrawOrgX()-m_map->getBorderSizeInline();
802 Bool intersect = false;
803 for (k=0; k<numLights; k++) {
804 if (pLights[k]->m_minX <= xCoord+1 &&
805 pLights[k]->m_maxX >= xCoord &&
806 pLights[k]->m_minY <= yCoord+1 &&
807 pLights[k]->m_maxY >= yCoord) {
808 intersect = true;
809 }
810 if (pLights[k]->m_prevMinX <= xCoord+1 &&
811 pLights[k]->m_prevMaxX >= xCoord &&
812 pLights[k]->m_prevMinY <= yCoord+1 &&
813 pLights[k]->m_prevMaxY >= yCoord) {
814 intersect = true;
815 }
816 }
817 if (!intersect) {
818 continue;
819 }
820 // vb is the pointer to the vertex in the hardware dx8 vertex buffer.
821 Int offset = (j-originY)*vertsPerRow+4*(i-originX);
822 if (HALF_RES_MESH) {
823 offset = (j-originY)*vertsPerRow/4+2*(i-originX);
824 }
825 vb = vBase + offset; //skip to correct row in vertex buffer
826 // vbMirror is the pointer to the vertex in our memory based copy.
827 // The important point is that we can read out of our copy to get the original
828 // diffuse color, and xyz location. It is VERY SLOW to read out of the
829 // hardware vertex buffer, possibly worse... jba.
830 VERTEX_FORMAT *vbMirror = ((VERTEX_FORMAT*)data) + offset;
831 VERTEX_FORMAT *vbaseMirror = ((VERTEX_FORMAT*)data);
832 un0 = getXWithOrigin(i)-1;
833 if (un0 < -m_map->getDrawOrgX())
834 un0=-m_map->getDrawOrgX();
835 up1 = getXWithOrigin(i+1)+1;
836 if (up1 >= m_map->getXExtent()-m_map->getDrawOrgX())
837 up1=m_map->getXExtent()-m_map->getDrawOrgX()-1;
838
839 Vector3 lightRay(0,0,0);
840
841 //
842 // (gth) Following the set of rules below lets us take advantage of lighting values that have
843 // been previously computed. The idea is to copy them ahead to future quads that will need them
844 // and then not compute them when we get to those quads. This also avoids having to read-back
845 // from the vertex buffer but we do jump around in memory... probably bad anyway, maybe we should
846 // compute into a temporary buffer and copy all at once...
847 //
848 unsigned long light_copy;
849
850 // top-left sample -> only compute when i==0 and j==0
851 if ((i==x0) && (j==y0)) {
852 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i)+1, getYWithOrigin(j)) - m_map->getDisplayHeight(un0, getYWithOrigin(j))));
853 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i), (getYWithOrigin(j)+1)) - m_map->getDisplayHeight(getXWithOrigin(i), vn0)));
854 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
855 doTheDynamicLight(vb, vbMirror, &lightRay, &normalAtTexel, pLights, numLights);
856 }
857 vb++; vbMirror++;
858
859 //top-right sample -> compute when j==0, then copy to (right,0)
860 if (j==y0) {
861 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(up1 , getYWithOrigin(j) ) - m_map->getDisplayHeight(getXWithOrigin(i) , getYWithOrigin(j) )));
862 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i)+1 , (getYWithOrigin(j)+1) ) - m_map->getDisplayHeight(getXWithOrigin(i)+1 , vn0 )));
863 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
864 light_copy = doTheDynamicLight(vb, vbMirror, &lightRay, &normalAtTexel, pLights, numLights);
865
866 if (i < x1-1) {
867 // copy light to (right,0)
868 (vBase + offset + quad_right_offset)->diffuse = (light_copy&0x00FFFFFF) | ((vbaseMirror + offset + quad_right_offset)->diffuse&0xff000000) ;
869 }
870 }
871 vb++; vbMirror++;
872
873 //bottom-right sample -> always compute, then copy to (right,3), (down,1), (down+right,0)
874 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(up1 , (getYWithOrigin(j)+1) ) - m_map->getDisplayHeight(getXWithOrigin(i) , (getYWithOrigin(j)+1) )));
875 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i)+1 , vp1 ) - m_map->getDisplayHeight(getXWithOrigin(i)+1 , getYWithOrigin(j) )));
876 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
877 light_copy = doTheDynamicLight(vb, vbMirror, &lightRay, &normalAtTexel, pLights, numLights);
878
879 if (i < x1-1) {
880 // copy light to (right,3)
881 //(vBase + offset + quad_right_offset + 3)->diffuse = light_copy;
882 (vBase + offset + quad_right_offset + 3)->diffuse = (light_copy&0x00FFFFFF) | ((vbaseMirror + offset + quad_right_offset + 3)->diffuse&0xff000000) ;
883 }
884 if (j < y1-1) {
885 // copy light to (down,1)
886 //(vBase + offset + quad_below_offset + 1)->diffuse = light_copy;
887 (vBase + offset + quad_right_offset + 1)->diffuse = (light_copy&0x00FFFFFF) | ((vbaseMirror + offset + quad_right_offset + 1)->diffuse&0xff000000) ;
888 }
889 if ((i < x1-1) && (j < y1-1)) {
890 // copy light to (right+down,0)
891 //(vBase + offset + quad_below_right_offset)->diffuse = light_copy;
892 (vBase + offset + quad_right_offset)->diffuse = (light_copy&0x00FFFFFF) | ((vbaseMirror + offset + quad_right_offset)->diffuse&0xff000000) ;
893 }
894 vb++; vbMirror++;
895
896 //bottom-left sample -> compute when i==0, otherwise copy from (left,2)
897 if (i==x0) {
898 l2r.Set(2*MAP_XY_FACTOR,0,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i)+1 , (getYWithOrigin(j)+1) ) - m_map->getDisplayHeight(un0 , (getYWithOrigin(j)+1) )));
899 n2f.Set(0,2*MAP_XY_FACTOR,MAP_HEIGHT_SCALE*(m_map->getDisplayHeight(getXWithOrigin(i) , vp1 ) - m_map->getDisplayHeight(getXWithOrigin(i) , getYWithOrigin(j) )));
900 Vector3::Normalized_Cross_Product(l2r, n2f, &normalAtTexel);
901 light_copy = doTheDynamicLight(vb, vbMirror, &lightRay, &normalAtTexel, pLights, numLights);
902
903 if (j < y1-1) {
904 // copy light to (down,0)
905 //(vBase + offset + quad_below_offset)->diffuse = light_copy;
906 (vBase + offset + quad_below_offset)->diffuse = (light_copy&0x00FFFFFF) | ((vbaseMirror + offset + quad_below_offset)->diffuse&0xff000000) ;
907 }
908 }
909 vb++; vbMirror++;
910 }
911 }
912 return 0; //success.
913 }
914 return -1;
915}

926void HeightMapRenderObjClass::doPartialUpdate(const IRegion2D &partialRange, WorldHeightMap *htMap, RefRenderObjListIterator *pLightsIterator)
927{
928 // Adjust range into the current drawn map range.
929 Int minX = partialRange.lo.x - htMap->getDrawOrgX();
930 Int maxX = partialRange.hi.x - htMap->getDrawOrgX();
931 Int minY = partialRange.lo.y - htMap->getDrawOrgY();
932 Int maxY = partialRange.hi.y - htMap->getDrawOrgY();
933 if (minX<0) minX = 0;
934 if (minY<0) minY = 0;
935 if (maxX > m_x-1) maxX = m_x-1;
936 if (maxY > m_y-1) maxY = m_y-1;
937 if (maxX < minX) return;
938 if (maxY < minY) return;
939 if (m_originX == 0 && m_originY == 0) {
940 // simple case.
941 updateBlock(minX, minY, maxX, maxY,
942 htMap, pLightsIterator);
943 }
944 else
945 {
946 minY = minY+m_originY;
947 maxY = maxY+m_originY;
948
949 if (minY> m_y-1) {
950 minY -= m_y-1;
951 maxY -= m_y-1;
952 }
953 if (maxY > m_y-1) {
954 maxY -= m_y-1;
955 updateBlock(0, minY, m_x-1, m_y-1, htMap, pLightsIterator);
956 updateBlock(0, 0, m_x-1, maxY, htMap, pLightsIterator);
957 } else {
958 updateBlock(0, minY, m_x-1, maxY, htMap, pLightsIterator);
959 }
960 }
961
962 if (!m_extraBlendTilePositions)
963 { //Need to allocate memory
964 m_extraBlendTilePositions = NEW Int[DEFAULT_MAX_MAP_EXTRABLEND_TILES];
965 m_extraBlendTilePositionsSize = DEFAULT_MAX_MAP_EXTRABLEND_TILES;
966 }
967
968 //Find list of all extra blend tiles used on map. These are tiles with 3 materials/textures
969 //over the same tile and require an extra render pass.
970
971 Int i, j;
972 //First remove any existing extra blend tiles within this partial region
973 for (j=0; j<m_numExtraBlendTiles; j++)
974 { Int x = m_extraBlendTilePositions[j] & 0xffff;
975 Int y = m_extraBlendTilePositions[j] >> 16;
976 if (x >= partialRange.lo.x && x < partialRange.hi.x &&
977 y >= partialRange.lo.y && y < partialRange.hi.y)
978 { //this tile is inside region being updated so remove it by shifting tile array
979 memcpy(m_extraBlendTilePositions+j,m_extraBlendTilePositions+j+1,(m_numExtraBlendTiles-1-j)*sizeof(Int));
980 m_numExtraBlendTiles--;
981 j--; //need to look at index j again because this tile was removed
982 }
983 }
984
985 for (j=partialRange.lo.y; j<partialRange.hi.y; j++)
986 for (i=partialRange.lo.x; i<partialRange.hi.x; i++)
987 {
988 if (j<0 || i<0) continue;
989 Real U[4],V[4];
990 UnsignedByte alpha[4];
991 Bool flipState,cliffState;
992 if (htMap->getExtraAlphaUVData(i,j,U,V,alpha,&flipState, &cliffState))
993 { if (m_numExtraBlendTiles >= m_extraBlendTilePositionsSize)
994 { //no more room to store extra blend tiles so enlarge the buffer.
995 Int *tempPositions=NEW Int[m_extraBlendTilePositionsSize+512];
996 memcpy(tempPositions, m_extraBlendTilePositions, m_extraBlendTilePositionsSize*sizeof(Int));
997 delete [] m_extraBlendTilePositions;
998 //enlarge by more tiles to reduce memory trashing
999 m_extraBlendTilePositions = tempPositions;
1000 m_extraBlendTilePositionsSize += 512;
1001 }
1002 //Pack x and y position into single integer since maps are limited in size
1003 m_extraBlendTilePositions[m_numExtraBlendTiles]=i | (j <<16);
1004 m_numExtraBlendTiles++;
1005 }
1006 }
1007 updateShorelineTiles(partialRange.lo.x,partialRange.lo.y,partialRange.hi.x,partialRange.hi.y,htMap);
1008
1009 updateViewImpassableAreas(TRUE, minX, maxX, minY, maxY);
1010}

1014//=============================================================================
1018Int HeightMapRenderObjClass::updateBlock(Int x0, Int y0, Int x1, Int y1, WorldHeightMap *pMap, RefRenderObjListIterator *pLightsIterator)
1019{
1020#ifdef _DEBUG
1021 DEBUG_ASSERTCRASH(x0>=0, ("HeightMapRenderObjClass::UpdateBlock parameters extend beyond left edge."));
1022 DEBUG_ASSERTCRASH(y0>=0, ("HeightMapRenderObjClass::UpdateBlock parameters extend beyond bottom edge."));
1023 DEBUG_ASSERTCRASH(x1<m_x, ("HeightMapRenderObjClass::UpdateBlock parameters extend beyond right edge."));
1024 DEBUG_ASSERTCRASH(y1<m_y, ("HeightMapRenderObjClass::UpdateBlock parameters extend beyond top edge."));
1025 DEBUG_ASSERTCRASH(x0<=x1, ("HeightMapRenderObjClass::UpdateBlock parameters have inside-out rectangle (on X)."));
1026 DEBUG_ASSERTCRASH(y0<=y1, ("HeightMapRenderObjClass::UpdateBlock parameters have inside-out rectangle (on Y)."));
1027#endif
1028 Invalidate_Cached_Bounding_Volumes();
1029 if (pMap) {
1030 REF_PTR_SET(m_stageZeroTexture, pMap->getTerrainTexture());
1031 REF_PTR_SET(m_stageOneTexture, pMap->getAlphaTerrainTexture());
1032 }
1033
1034 Int i,j;
1035 DX8VertexBufferClass **pVB;
1036 Int originX,originY;
1037 //step through each vertex buffer that needs updating
1038 for (j=0; j<m_numVBTilesY; j++)
1039 {
1040 originY=j*VERTEX_BUFFER_TILE_LENGTH; //location of this VB on the large full-size heightmap
1041 Int yMin, yMax;
1042 yMin = originY;
1043 if (y0>yMin) yMin = y0;
1044 yMax = originY+VERTEX_BUFFER_TILE_LENGTH;
1045 if (y1<yMax) yMax = y1;
1046 if (yMin >= yMax) {
1047 continue;
1048 }

1703 m_needFullUpdate = false;
1704 updateBlock(0, 0, m_x-1, m_y-1, m_map, pLightsIterator);
1705 m_updating = false;
1706 return;
1707 }
1708
1709 if (m_x >= m_map->getXExtent() && m_y >= m_map->getYExtent())
1710 {
1711 m_updating = false;
1712 return; // no need to center.
1713 }
1714
1715 Int cellOffset = 1;
1716 if (HALF_RES_MESH) {
1717 cellOffset = 2;
1718 }
1719 // determine the ray corresponding to the camera and distance to projection plane
1720 Matrix3D camera_matrix = camera->Get_Transform();
1721
1722 Vector3 camera_location = camera->Get_Position();
1723
1724 Vector3 rayLocation;
1725 Vector3 rayDirection;
1726 Vector3 rayDirectionPt;
1727 // the projected ray has the same origin as the camera
1728 rayLocation = camera_location;
1729 // determine the location of the screen coordinate in camera-model space
1730 const ViewportClass &viewport = camera->Get_Viewport();
1731 Int i, j, minHt;
1732
1733 Real intersectionZ;
1734 minHt = m_map->getMaxHeightValue();
1735 for (i=0; i<m_x; i++) {
1736 for (j=0; j<m_y; j++) {
1737 Short cur = m_map->getDisplayHeight(i,j);
1738 if (cur<minHt) minHt = cur;
1739 }
1740 }
1741 intersectionZ = (float)minHt;
1742// float aspect = camera->Get_Aspect_Ratio();
1743
1744 Vector2 min,max;
1745 camera->Get_View_Plane(min,max);
1746 float xscale = (max.X - min.X);
1747 float yscale = (max.Y - min.Y);
1748
1749 float zmod = -1.0; // Scene->vpd; // Note: view plane distance is now always 1.0 from the camera
1750 float minX = 200000;
1751 float maxX = -minX;
1752 float minY = 200000;
1753 float maxY = -minY;
1754 for (i=0; i<2; i++) {
1755 for (j=0; j<2; j++) {
1756 float xmod = (-i + 0.5 + viewport.Min.X) * zmod * xscale;// / aspect;
1757 float ymod = (j - 0.5 - viewport.Min.Y) * zmod * yscale;// * aspect;
1758
1759 // transform the screen coordinates by the camera's matrix into world coordinates.
1760 float x = zmod * camera_matrix[0][2] + xmod * camera_matrix[0][0] + ymod * camera_matrix[0][1];
1761 float y = zmod * camera_matrix[1][2] + xmod * camera_matrix[1][0] + ymod * camera_matrix[1][1];
1762 float z = zmod * camera_matrix[2][2] + xmod * camera_matrix[2][0] + ymod * camera_matrix[2][1];
1763
1764 rayDirection.Set(x,y,z);
1765 rayDirection.Normalize();
1766 rayDirectionPt = rayLocation+rayDirection;
1767
1768 x = Vector3::Find_X_At_Z(intersectionZ, rayLocation, rayDirectionPt);
1769 y = Vector3::Find_Y_At_Z(intersectionZ, rayLocation, rayDirectionPt);
1770 if (x<minX) minX = x;
1771 if (x>maxX) maxX = x;
1772 if (y<minY) minY = y;
1773 if (y>maxY) maxY = y;
1774 }
1775 }
1776
1777 // convert back to cell indexes.
1778 minX /= MAP_XY_FACTOR;
1779 maxX /= MAP_XY_FACTOR;
1780 minY /= MAP_XY_FACTOR;
1781 maxY /= MAP_XY_FACTOR;
1782

1784 maxX += m_map->getBorderSizeInline();
1785 minY += m_map->getBorderSizeInline();
1786 maxY += m_map->getBorderSizeInline();
1787

1789 visMinY = m_map->getYExtent();
1790 visMaxX = 0;
1791 visMaxY = 0;
1792
1794 if (minX<0) minX=0;
1795 if (minY<0) minY=0;
1796 if (maxX > visMinX) maxX = visMinX;
1797 if (maxY > visMinY) maxY = visMinY;
1798

1801 if (limit > WIDE_STEP/2) {
1802 limit=WIDE_STEP/2;
1803 }

1805
1806 if (m_map) {
1807 Int newOrgX;
1808 if (visMaxX-visMinX > m_x) {
1809 newOrgX = (maxX+minX)/2-m_x/2.0;
1810 } else {
1811 newOrgX = (visMaxX+visMinX)/2-m_x/2.0;
1812 }
1813
1814 Int newOrgY;
1815 if (visMaxY - visMinY > m_y) {
1816 newOrgY = visMinY+1;
1817 } else {

1819 }

1822 newOrgY = (visMaxY+visMinY)/2-m_y/2.0;
1823 }
1824 if (HALF_RES_MESH) {
1825 newOrgX &= 0xFFFFFFFE;
1826 newOrgY &= 0xFFFFFFFE;
1827 }
1828 Int deltaX = newOrgX - m_map->getDrawOrgX();
1829 Int deltaY = newOrgY - m_map->getDrawOrgY();
1830 if (IABS(deltaX) > m_x/2 || IABS(deltaY)>m_x/2) {
1831 m_map->setDrawOrg(newOrgX, newOrgY);
1832 m_originY = 0;
1833 m_originX = 0;
1834 updateBlock(0, 0, m_x-1, m_y-1, m_map, pLightsIterator);
1835 m_updating = false;
1836 return;
1837 }
1838
1839 if (abs(deltaX)>CENTER_LIMIT || abs(deltaY)>CENTER_LIMIT) {
1840 if (abs(deltaY) >= CENTER_LIMIT) {
1841 if (m_map->setDrawOrg(m_map->getDrawOrgX(), newOrgY)) {
1842 Int minY = 0;
1843 Int maxY = 0;
1844 deltaY -= newOrgY - m_map->getDrawOrgY();
1845 m_originY += deltaY;
1846 if (m_originY >= m_y-1) m_originY -= m_y-1;
1847 if (deltaY<0) {
1848 minY = m_originY;
1849 maxY = m_originY-deltaY;
1850 } else {
1851 minY = m_originY - deltaY;
1852 maxY = m_originY;
1853 }
1854 minY-=cellOffset;
1855 if (m_originY < 0) m_originY += m_y-1;
1856 if (minY<0) {
1857 minY += m_y-1;
1858 if (minY<0) minY = 0;
1859 updateBlock(0, minY, m_x-1, m_y-1, m_map, pLightsIterator);
1860 updateBlock(0, 0, m_x-1, maxY, m_map, pLightsIterator);
1861 } else {
1862 updateBlock(0, minY, m_x-1, maxY, m_map, pLightsIterator);
1863 }
1864 }
1865 // It is much more efficient to update a cople of columns one frame, and then
1866 // a couple of rows. So if we aren't "jumping" to a new view, and have done X
1867 // recently, return.
1868 if (abs(deltaX) < BIG_JUMP && !m_doXNextTime) {
1869 m_updating = false;
1870 m_doXNextTime = true;
1871 return; // Only do the y this frame. Do x next frame. jba.
1872 }
1873 }
1874 if (abs(deltaX) > CENTER_LIMIT) {
1875 m_doXNextTime = false;
1876 newOrgX = m_map->getDrawOrgX() + deltaX;
1877 if (m_map->setDrawOrg(newOrgX, m_map->getDrawOrgY())) {
1878 Int minX = 0;
1879 Int maxX = 0;
1880 deltaX -= newOrgX - m_map->getDrawOrgX();
1881 m_originX += deltaX;
1882 if (m_originX >= m_x-1) m_originX -= m_x-1;
1883 if (deltaX<0) {
1884 minX = m_originX;
1885 maxX = m_originX-deltaX;
1886 } else {
1887 minX = m_originX - deltaX;
1888 maxX = m_originX;
1889 }
1890 minX-=cellOffset;
1891 maxX+=cellOffset;
1892 if (m_originX < 0) m_originX += m_x-1;
1893 if (minX<0) {
1894 minX += m_x-1;
1895 if (minX<0) minX = 0;
1896 updateBlock(minX,0,m_x-1, m_y-1, m_map, pLightsIterator);
1897 updateBlock(0,0,maxX, m_y-1, m_map, pLightsIterator);
1898 } else {
1899 updateBlock(minX,0,maxX, m_y-1, m_map, pLightsIterator);
1900 }
1901 }
1902 }
1903 }
1904 }
1905 m_updating = false;
1906}
