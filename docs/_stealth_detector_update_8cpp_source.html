88StealthDetectorUpdate::StealthDetectorUpdate( Thing *thing, const ModuleData* moduleData ) : UpdateModule( thing, moduleData )
89{
90 const StealthDetectorUpdateModuleData *data = getStealthDetectorUpdateModuleData();
91 m_enabled = !data->m_initiallyDisabled;
92 // start these guys with random phasings so that we don't
93 // have all of 'em check on the same frame.
94 setWakeFrame(getObject(), m_enabled ? UPDATE_SLEEP(GameLogicRandomValue(1, data->m_updateRate)) : UPDATE_SLEEP_FOREVER);
95}

117class PartitionFilterStealthedOrStealthGarrisoned : public PartitionFilter
118{
119public:
120 PartitionFilterStealthedOrStealthGarrisoned() { }
121
122 virtual Bool allow(Object *objOther);
123
124#if defined(_DEBUG) || defined(_INTERNAL)
125 virtual const char* debugGetName() { return "PartitionFilterStealthedOrStealthGarrisoned"; }
126#endif
127};
128
129Bool PartitionFilterStealthedOrStealthGarrisoned::allow( Object *objOther)
130{
131 if( ! objOther )
132 return FALSE;
133
134 if( objOther->getStatusBits().test( OBJECT_STATUS_STEALTHED ) )
135 return TRUE;
136
137 ContainModuleInterface *contain = objOther->getContain();
138 if( contain && contain->isGarrisonable() && contain->getStealthUnitsContained() )
139 return TRUE;
140
141 return FALSE;
142}

147UpdateSleepTime StealthDetectorUpdate::update( void )
148{
149 const StealthDetectorUpdateModuleData *data = getStealthDetectorUpdateModuleData();
150 Object* self = getObject();
151
152 if (self->isEffectivelyDead())
153 return UPDATE_SLEEP_FOREVER;
154
155 // We have to wait until we are fully constructed, but we will detect the moment we finish
156 if( self->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) )
157 return UPDATE_SLEEP_NONE;
158
159 // We turn off forever the moment we are sold.
160 if( self->testStatus(OBJECT_STATUS_SOLD) )
161 return UPDATE_SLEEP_FOREVER;
162
163 //Are we contained by anything?
164 Object *containedBy = self->getContainedBy();
165 if( containedBy )
166 {
167 ContainModuleInterface *contain = containedBy->getContain();
168 if( contain )
169 {
170 //Are we eligible to detect stealth while in a container?
171 if( contain->isGarrisonable() )
172 {
173 //We are in a garrisonable structure.
174 if( !data->m_canDetectWhileGarrisoned )
175 {
176 //But we can't detect stuff while inside.
177 return UPDATE_SLEEP(data->m_updateRate);
178 }
179 }
180 else if( !data->m_canDetectWhileTransported )
181 {
182 //We are in a normal container and can't detect!
183 return UPDATE_SLEEP(data->m_updateRate);
184 }
185 }
186 }
187
188 // only consider items that are currently stealthed.
189 PartitionFilterStealthedOrStealthGarrisoned filterStealthOrStealthGarrisoned;
190 //PartitionFilterAcceptByObjectStatus filterStatus(OBJECT_STATUS_STEALTHED, 0);
191 PartitionFilterRelationship filterTeam(self, PartitionFilterRelationship::ALLOW_ENEMIES | PartitionFilterRelationship::ALLOW_NEUTRAL );
192 PartitionFilterAcceptByKindOf filterKindof(data->m_extraDetectKindof, data->m_extraDetectKindofNot);
193 PartitionFilterSameMapStatus filterMapStatus(getObject());
194 PartitionFilter* filters[] = { &filterStealthOrStealthGarrisoned, &filterTeam, &filterKindof, &filterMapStatus, NULL };
195
196 Real visionRange = self->getVisionRange();
197 if( data->m_detectionRange > 0.0f )
198 {
199 visionRange = data->m_detectionRange;
200 }
201 Bool foundSomeone = FALSE;
202
203 SimpleObjectIterator *iter = ThePartitionManager->iterateObjectsInRange(
204 self, visionRange, FROM_CENTER_2D, filters);
205 MemoryPoolObjectHolder hold(iter);
206 for (Object *them = iter->first(); them; them = iter->next())
207 {
208 if ( them->isEffectivelyDead() )
209 continue;
210
211 StealthUpdate* stealth = them->getStealth();
212 if ( stealth )
213 {
214
215 // we have found someone
216 foundSomeone = TRUE;
217
218 //
219 // if this object was not previously detected it is now being revealed and we
220 // want to do some UI feedback
221 //
222 if( them->testStatus( OBJECT_STATUS_DETECTED ) == FALSE )
223 {
224
225 // for the player revealing the stealth unit do some UI feedback
226 if( ThePlayerList->getLocalPlayer() == self->getControllingPlayer() &&
227 self->getRelationship( them ) != ALLIES )
228 {
229 Bool doFeedback = TRUE;
230
231 //Kris: Aug 18, 2003 -- This whole system is no longer viable with the inclusion of so many
232 //stealth units. It's possible to get spammed hundreds of times in a row in MD_GLA02 because
233 //we're blindly creating radar events for each detection (unless they are mines). The best thing
234 //to do is tryEvent for everything...
235 doFeedback = TheRadar->tryEvent( RADAR_EVENT_STEALTH_DISCOVERED, them->getPosition() );
236 //OLD CODE:
237 /*
238 // do a radar event, for mines we only make events if there weren't other
239 // mine events within close proximity and time to other mines
240 //
241 if( them->isKindOf( KINDOF_MINE ) || them->isKindOf( KINDOF_BOOBY_TRAP ) || them->isKindOf( KINDOF_DEMOTRAP ) )
242 doFeedback = TheRadar->tryEvent( RADAR_EVENT_STEALTH_DISCOVERED, them->getPosition() );
243 else
244 TheRadar->createEvent( them->getPosition(), RADAR_EVENT_STEALTH_DISCOVERED );
245 */
246
247 // do audio and UI message if we need to do feedback
248 if( doFeedback )
249 {
250 // audio msg
251 static AudioEventRTS discoveredSound = TheAudio->getMiscAudio()->m_stealthDiscoveredSound;
252 discoveredSound.setPlayerIndex( self->getControllingPlayer()->getPlayerIndex() );
253 TheAudio->addAudioEvent( &discoveredSound );
254 // ui msg
255 TheInGameUI->message( TheGameText->fetch( "MESSAGE:StealthDiscovered" ) );
256
257 // If revealing this unit is suppose to cause an Eva event, do it
258 EvaMessage message = stealth->getEnemyDetectionEvaEvent();
259 if ( message != EVA_Invalid && TheEva != NULL )
260 {
261 TheEva->setShouldPlay( message );
262 }
263 } // end if
264
265 } // end if
266
267 // for the unit being revealed, do some UI feedback
268 if( ThePlayerList->getLocalPlayer() == them->getControllingPlayer() &&
269 self->getRelationship( them ) != ALLIES )
270 {
271 Bool doFeedback = TRUE;
272
273 //
274 // do a radar event, for mines we only make events if there weren't other
275 // mine events within close proximity and time to other mines
276 //
277 if( them->isKindOf( KINDOF_MINE ) || them->isKindOf( KINDOF_BOOBY_TRAP ) || them->isKindOf( KINDOF_DEMOTRAP ) )
278 doFeedback = TheRadar->tryEvent( RADAR_EVENT_STEALTH_NEUTRALIZED, them->getPosition() );
279 else
280 TheRadar->createEvent( them->getPosition(), RADAR_EVENT_STEALTH_NEUTRALIZED );
281
282 // do audio and UI message if we need to do feedback
283 if( doFeedback )
284 {
285
286 // audio msg
287 static AudioEventRTS neutralizedSound = TheAudio->getMiscAudio()->m_stealthNeutralizedSound;
288 neutralizedSound.setPlayerIndex( them->getControllingPlayer()->getPlayerIndex() );
289 TheAudio->addAudioEvent( &neutralizedSound );
290 // ui msg
291 TheInGameUI->message( TheGameText->fetch( "MESSAGE:StealthNeutralized" ) );
292
293 // If revealing this unit is suppose to cause an Eva event, do it
294 EvaMessage message = stealth->getOwnDetectionEvaEvent();
295 if ( message != EVA_Invalid && TheEva != NULL )
296 {
297 TheEva->setShouldPlay( message );
298 }
299 } // end if
300
301 } // end if
302
303 } // end if, them was not previously detected
304
305 // updateRate PLUS 1 is necessary to ensure it stays detected 'till we are called again...
306 stealth->markAsDetected(data->m_updateRate + 1);
307
310 Drawable *theirDraw = them->getDrawable();
311 if ( theirDraw && !them->isKindOf(KINDOF_MINE))
312 {
313 theirDraw->setSecondMaterialPassOpacity( 1.0f );
314 }
315
316 if (data->m_IRGridParticleSysTmpl)
317 {
318 const ParticleSystemTemplate *gridTemplate = data->m_IRGridParticleSysTmpl;
319 if (gridTemplate)
320 {
321 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem( gridTemplate );//GRID
322 if (sys)
323 {
324 Coord3D gridPosition = *them->getPosition();
325 gridPosition.z = self->getPosition()->z + 17;
326 gridPosition.x -= ((Int)gridPosition.x)%12;
327 gridPosition.y -= ((Int)gridPosition.y)%12;
328
329 sys->setPosition( &gridPosition );
330 }
331 }
332 }
333
334 }//end if them has stealthupdate
335 else // perhaps they are garrisoning something stealthy, eh?
336 {
337 ContainModuleInterface *contain = them->getContain();
338 if( contain && contain->isGarrisonable() && contain->getStealthUnitsContained() )
339 {
340 Object* rider = NULL;
341 for(ContainedItemsList::const_iterator it = contain->getContainedItemsList()->begin(); it != contain->getContainedItemsList()->end(); ++it)
342 {
343 rider = *it;
344
345 StealthUpdate *stealth = rider->getStealth();
346 if ( stealth )
347 {
348 // we have found someone
349 foundSomeone = TRUE;
350 if( self->getControllingPlayer() != rider->getControllingPlayer() && self->getRelationship( rider ) != ALLIES )
351 {
352 stealth->markAsDetected(data->m_updateRate + 2);
353 }
354
355 }
356 }
357 }
358 }
359 }
360
361
362 const Player *localPlayer = ThePlayerList->getLocalPlayer();
363
364 //Make sure the detector is visible to the local player before we add effects or sounds.
365 if ( self->getShroudedStatus( localPlayer->getPlayerIndex() ) <= OBJECTSHROUD_PARTIAL_CLEAR )
366 {
367 if ( self->testStatus( OBJECT_STATUS_STEALTHED ) == FALSE || self->getControllingPlayer() == localPlayer )
368 {
369 Drawable *myDraw = self->getDrawable();
370 Coord3D bonePosition = {-1.66f,5.5f,15};
371 if (myDraw)
372 myDraw->getPristineBonePositions( data->m_IRParticleSysBone.str(), 0, &bonePosition, NULL, 1);
373
374 const ParticleSystemTemplate *pingTemplate;
375 if ( foundSomeone )
376 pingTemplate = data->m_IRBrightParticleSysTmpl;
377 else
378 pingTemplate = data->m_IRParticleSysTmpl;
379
380 if (pingTemplate)
381 {
382 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem( pingTemplate );
383 if (sys)
384 {
385 if (myDraw)
386 sys->attachToDrawable( myDraw );
387 else
388 sys->attachToObject( self );
389
390 sys->setPosition( &bonePosition );
391 }
392 }
393
394 const ParticleSystemTemplate *beaconTemplate = data->m_IRBeaconParticleSysTmpl;
395 if (beaconTemplate)
396 {
397 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem( beaconTemplate );//BEACON
398 if (sys)
399 {
400 if (myDraw)
401 sys->attachToDrawable( myDraw );
402 else
403 sys->attachToObject( self );
404
405 sys->setPosition( &bonePosition );
406
407 }
408 }
409
410 AudioEventRTS IRPingSound;
411 if (foundSomeone)
412 IRPingSound = data->m_loudPingSound;
413 else
414 IRPingSound = data->m_pingSound;
415
416 IRPingSound.setObjectID( self->getID() );
417 TheAudio->addAudioEvent(&IRPingSound);
418
419 }
420
421 } // end if doIRFX
422
423
424 return UPDATE_SLEEP(data->m_updateRate);
425
426}
