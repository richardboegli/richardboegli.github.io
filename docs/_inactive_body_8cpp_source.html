61Real InactiveBody::estimateDamage( DamageInfoInput& damageInfo ) const

63 // Inactive bodies have no health so no damage can really be done
64 Real amount = 0.0f;
65
66 // exception!

77void InactiveBody::attemptDamage( DamageInfo *damageInfo )
78{
79 if( damageInfo == NULL )
80 return;
81
82 if( damageInfo->in.m_damageType == DAMAGE_HEALING )
83 {
84 // Healing and Damage are separate, so this shouldn't happen
85 attemptHealing( damageInfo );
86 return;
87 }
88
89 // Inactive bodies have no health so no damage can really be done
90 damageInfo->out.m_actualDamageDealt = 0.0f;
91 damageInfo->out.m_actualDamageClipped = 0.0f;
92 damageInfo->out.m_noEffect = true;
93
94 // exception: damage type KILL always wipes us out
95 if (damageInfo->in.m_damageType == DAMAGE_UNRESISTABLE)
96 {
97
98 DEBUG_ASSERTCRASH(!getObject()->getTemplate()->isPrerequisite(), ("Prerequisites should not have InactiveBody"));
99
100 damageInfo->out.m_noEffect = false;
101
102 // since we have no Health, we do not call DamageModules, nor do DamageFX.
103 // however, we DO process DieModules.
104 if (!m_dieCalled)
105 {
106 getObject()->onDie( damageInfo );
107 m_dieCalled = true;
108 }
109 }
110}
