66static void adjustVector(Coord3D *vec, const Matrix3D* mtx)
67{
68 if (mtx)
69 {
70 Vector3 vectmp;
71 vectmp.X = vec->x;
72 vectmp.Y = vec->y;
73 vectmp.Z = vec->z;
74 vectmp = mtx->Rotate_Vector(vectmp);
75 vec->x = vectmp.X;
76 vec->y = vectmp.Y;
77 vec->z = vectmp.Z;
78 }
79}

86void FXNugget::doFXObj(const Object* primary, const Object* secondary) const
87{
88 const Coord3D* p = primary ? primary->getPosition() : NULL;
89 const Matrix3D* mtx = primary ? primary->getTransformMatrix() : NULL;
90 const Real speed = 0.0f; // yes, that's right -- NOT the object's speed.
91 const Coord3D* s = secondary ? secondary->getPosition() : NULL;
92 doFXPos(p, mtx, speed, s);
93}

146static Real calcDist(const Coord3D& src, const Coord3D& dst)
147{
148 Real dx = dst.x - src.x;
149 Real dy = dst.y - src.y;
150 Real dz = dst.z - src.z;
151 return sqrt(dx*dx + dy*dy + dz*dz);
152}

155class TracerFXNugget : public FXNugget
156{
157 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(TracerFXNugget, "TracerFXNugget")
158public:
159
160 TracerFXNugget()
161 {
162 m_tracerName.set("GenericTracer");
163 m_boneName.clear();
164 m_speed = 0.0f; // means "use passed-in speed"
165 m_decayAt = 1.0f;
166 m_length = 10.0f;
167 m_width = 1.0f;
168 m_color.red = m_color.green = m_color.blue = 1.0f;
169 m_probability = 1.0f;
170 }
171
172 virtual void doFXPos(const Coord3D *primary, const Matrix3D* primaryMtx, const Real primarySpeed, const Coord3D *secondary, const Real /*overrideRadius*/ ) const
173 {
174 if (m_probability <= GameClientRandomValueReal(0, 1))
175 return;
176
177 if (primary && secondary)
178 {
179 Drawable *tracer = TheThingFactory->newDrawable(TheThingFactory->findTemplate(m_tracerName));
180 if(!tracer)
181 return;
182
183 //Kris -- Redid this section Sept 18, 2002
184 //Calculate tracer orientations to face from primary to secondary position. This
185 //should be the direction that the projectile is being fired towards. It doesn't make
186 //sense that the old stuff made use of the muzzle fx bone orientation (because it's a
187 //subobject). It had other problems because of elevation variations the tracers would
188 //stay on the ground.
189 //tracer->setTransformMatrix(primaryMtx);
190 Matrix3D tracerMtx;
191 Vector3 pos( primary->x, primary->y, primary->z );
192 Vector3 dir( secondary->x - primary->x, secondary->y - primary->y, secondary->z - primary->z );
193 dir.Normalize(); //This is fantastically crucial for calling buildTransformMatrix!!!!!
194 tracerMtx.buildTransformMatrix( pos, dir );
195 tracer->setTransformMatrix( &tracerMtx );
196 tracer->setPosition(primary);
197
198 Real speed = m_speed;
199 if (speed == 0.0f)
200 {
201 speed = primarySpeed;
202 }
203
204 TracerDrawInterface* tdi = NULL;
205 for (DrawModule** d = tracer->getDrawModules(); *d; ++d)
206 {
207 if ((tdi = (*d)->getTracerDrawInterface()) != NULL)
208 {
209 tdi->setTracerParms(speed, m_length, m_width, m_color, 1.0f);
210 }
211 }
212
213 // estimate how long it will take us to get to the destination
214 Real dist = calcDist(*primary, *secondary) - m_length;
215 Real frames = (dist >= 0.0f && speed >= 0.0f) ? (dist / speed) : 1;
216 Int framesAdjusted = REAL_TO_INT_CEIL(frames * m_decayAt);
217 tracer->setExpirationDate(TheGameLogic->getFrame() + framesAdjusted);
218 }
219 else
220 {
221 DEBUG_CRASH(("You must have a primary and secondary source for this effect"));
222 }
223 }

226 {

230 { "BoneName", INI::parseAsciiString, NULL, offsetof( TracerFXNugget, m_boneName ) },
231 { "Speed", INI::parseVelocityReal, NULL, offsetof( TracerFXNugget, m_speed ) },

505class ParticleSystemFXNugget : public FXNugget
506{
507 MEMORY_POOL_GLUE_WITH_USERLOOKUP_CREATE(ParticleSystemFXNugget, "ParticleSystemFXNugget")
508public:
509
510 ParticleSystemFXNugget()
511 {
512 m_name.clear();
513 m_count = 1;
514 m_radius.setRange(0, 0, GameClientRandomVariable::CONSTANT);
515 m_height.setRange(0, 0, GameClientRandomVariable::CONSTANT);
516 // -1 means "don't mess with it, just accept the particle-system's defaults"
517 m_delay.setRange(-1, -1, GameClientRandomVariable::CONSTANT);
518 m_offset.x = m_offset.y = m_offset.z = 0;
519 m_orientToObject = false;
520 m_attachToObject = false;
521 m_createAtGroundHeight = FALSE;
522 m_useCallersRadius = FALSE;

526 virtual void doFXPos(const Coord3D *primary, const Matrix3D* primaryMtx, const Real /*primarySpeed*/, const Coord3D * /*secondary*/, const Real overrideRadius ) const
527 {
528 if (primary)
529 {
530 reallyDoFX(primary, primaryMtx, NULL, overrideRadius);
531 }
532 else
533 {
534 DEBUG_CRASH(("You must have a primary source for this effect"));
535 }

538 virtual void doFXObj(const Object* primary, const Object* secondary) const
539 {
540 if (primary)
541 {
542
543 if (m_ricochet && secondary)
544 {
545 // HERE WE MUST BUILD A MATRIX WHICH WILL ORIENT THE NEW PARTICLE SYSTEM TO FACE AWAY FROM THE SECONDARY OBJECT
546 // THE RESULT SHOULD LOOK LIKE THE DIRECTION OF THE "ATTACK" IS CARRIED THROUGH LIKE A RICOCHET
547 Real deltaX = primary->getPosition()->x - secondary->getPosition()->x;
548 Real deltaY = primary->getPosition()->y - secondary->getPosition()->y;
549 Real aimingAngle = atan2(deltaY, deltaX);
550 Matrix3D aimingMatrix(1);
551 aimingMatrix.Rotate_Z( aimingAngle );
552
553 reallyDoFX(primary->getPosition(), &aimingMatrix, primary, 0.0f);
554 }
555 else
556 // if we have an object, then adjust the offset and direction by the object's transformation
557 // matrix, so that (say) an offset of +10 in the z axis "follows" the orientation of the object.
558 reallyDoFX(primary->getPosition(), primary->getTransformMatrix(), primary, 0.0f);
559 }
560 else
561 {
562 DEBUG_CRASH(("You must have a primary source for this effect"));
563 }
564 }

592protected:
593
594 void reallyDoFX(const Coord3D *primary, const Matrix3D* mtx, const Object* thingToAttachTo, Real overrideRadius ) const
595 {
596 Coord3D offset = m_offset;
597 if (mtx)
598 {
599 adjustVector(&offset, mtx);
600 }
601
602 const ParticleSystemTemplate *tmp = TheParticleSystemManager->findTemplate(m_name);
603 DEBUG_ASSERTCRASH(tmp, ("ParticleSystem %s not found\n",m_name.str()));
604 if (tmp)
605 {
606 for (Int i = 0; i < m_count; i++ )
607 {
608 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem(tmp);
609 if (sys)
610 {
611 Coord3D newPos;
612 Real radius = m_radius.getValue();
613 Real angle = GameClientRandomValueReal( 0.0f, 2.0f * PI );
614
615 newPos.x = primary->x + offset.x + radius * cos(angle);
616 newPos.y = primary->y + offset.y + radius * sin(angle);
617 if( m_createAtGroundHeight && TheTerrainLogic )
618 {
619 //old way:
620 //newPos.z = TheTerrainLogic->getGrsoundHeight( newPos.x, newPos.y ) + 1;// The plus one prevents scissoring with terrain
621
622 //new way: now we allow bridges in the GroundHeight.
623 PathfindLayerEnum layer = TheTerrainLogic->getLayerForDestination(&newPos);
624 newPos.z = TheTerrainLogic->getLayerHeight( newPos.x, newPos.y, layer );
625 }
626 else
627 newPos.z = primary->z + offset.z + m_height.getValue();
628
629
630 if (m_orientToObject && mtx)
631 {
632 sys->setLocalTransform(mtx);
633 }
634 if (m_rotateX != 0.0f)
635 sys->rotateLocalTransformX(m_rotateX);
636 if (m_rotateY != 0.0f)
637 sys->rotateLocalTransformY(m_rotateY);
638 if (m_rotateZ != 0.0f)
639 sys->rotateLocalTransformZ(m_rotateZ);
640
641 if (m_attachToObject && thingToAttachTo)
