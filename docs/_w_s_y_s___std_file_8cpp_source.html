133Bool StdFile::open( const Char *filename, Int access )
134{
135 if( !File::open( filename, access) )
136 {
137 return FALSE;
138 }
139
140 /* here we translate WSYS file access to the std C equivalent */
141
142 int flags = 0;
143
144 if(m_access & CREATE) flags |= _O_CREAT;
145 if(m_access & TRUNCATE) flags |= _O_TRUNC;
146 if(m_access & APPEND) flags |= _O_APPEND;
147 if(m_access & TEXT) flags |= _O_TEXT;
148 if(m_access & BINARY) flags |= _O_BINARY;
149
150 if((m_access & READWRITE )== READWRITE )
151 {
152 flags |= _O_RDWR;
153 }
154 else if(m_access & WRITE)
155 {
156 flags |= _O_WRONLY;
157 }
158 else
159 flags |= _O_RDONLY;
160
161 m_handle = _open( filename, flags , _S_IREAD | _S_IWRITE);
162
163 if( m_handle == -1 )
164 {
165 goto error;
166 }
167
168 if ( m_access & APPEND )
169 {
170 if ( seek ( 0, END ) < 0 )
171 {
172 goto error;
173 }
174 }
175
176 return TRUE;
177
178error:
179
180 close();
181
182 return FALSE;
183}

233Int StdFile::seek( Int pos, seekMode mode)
234{
235 int lmode;
236
237 switch( mode )
238 {
239 case START:
240 lmode = SEEK_SET;
241 break;
242 case CURRENT:
243 lmode = SEEK_CUR;
244 break;
245 case END:
246 lmode = SEEK_END;
247 break;
248 default:
249 // bad seek mode
250 return -1;
251 }
252
253 return _lseek( m_handle, pos, lmode );
