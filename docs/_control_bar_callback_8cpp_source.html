64WindowMsgHandledType LeftHUDInput( GameWindow *window, UnsignedInt msg,
65 WindowMsgData mData1, WindowMsgData mData2 )
66{
67
68 // get player
69 Player *player = ThePlayerList->getLocalPlayer();
70
71 //
72 // if the player doesn't have a radar, or the radar is hidden, and the radar is not being
73 // forced to on, we just eat input over the radar window
74 //
75 if( !TheRadar->isRadarForced() && (TheRadar->isRadarHidden() || !player->hasRadar()) )
76 return MSG_HANDLED;
77
78 // If the middle mouse button is depressed, then just let the message fall all the
79 // way back to the usual middle mouse button processing.
80 // jkmcd
81 if( TheMouse->getMouseStatus()->middleState == MBS_Down )
82 return MSG_IGNORED;
83
84 switch( msg )
85 {
86
93
94 // ------------------------------------------------------------------------
95 case GWM_NONE:
96 case GWM_MOUSE_ENTERING:
97 case GWM_MOUSE_LEAVING:
98 {
99
100 //
101 // consider changing the mouse cursor if we are not in the process of firing
102 // targeted "superweapons" which we can use the radar itself to fire
103 //
104 Bool targeting = FALSE;
105 const CommandButton *command = TheInGameUI->getGUICommand();
106 if( command
107 && (command->getCommandType() == GUI_COMMAND_SPECIAL_POWER || command->getCommandType() == GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT)
108 && BitTest( command->getOptions(), NEED_TARGET_POS ) )
109 targeting = TRUE;
110
111 if( targeting == FALSE )
112 {
113 const DrawableList *drawableList = TheInGameUI->getAllSelectedLocalDrawables();
114 Mouse::MouseCursor cur = Mouse::ARROW;
115
116 if (!(drawableList->empty() || msg == GWM_MOUSE_LEAVING))
117 {
118 if (command && command->getCommandType() == GUI_COMMAND_ATTACK_MOVE)
119 {
120 cur = Mouse::ATTACKMOVETO;
121 }
122 else
123 {
124 cur = Mouse::MOVETO;
125 }
126 }
127
128 // Groovy
129 TheMouse->setCursor(cur);
130
131 } // end if
132
133 return MSG_HANDLED;
134 }
135
136 // --------------------------------------------------------------------------------------------
137 case GWM_MOUSE_POS:
138 {
139
140 // get mouse position
141 ICoord2D mouse;
142 mouse.x = mData1 & 0xFFFF;
143 mouse.y = mData1 >> 16;
144
145 // get window screen position
146 ICoord2D screenPos;
147 window->winGetScreenPosition( &screenPos.x, &screenPos.y );
148
149 // set mouse position to be relative to this window
150 mouse.x -= screenPos.x;
151 mouse.y -= screenPos.y;
152
153 // is the mouse in the radar window
154 ICoord2D radar;
155 if( (TheRadar->isRadarHidden() == FALSE || TheRadar->isRadarForced()) &&
156 TheRadar->localPixelToRadar( &mouse, &radar ) )
157 {
158
159/*
160//
161// this is an example piece of code to find the object under the pixel position
162// of the radar ... should we in the future wish to allow commands to be executed
163// on objects throught he radar. note tho that this is extremely hard to do because
164// the pixels on the radar are very small and it's hard to do accurate targeting
165//
166
167 Object *obj = TheRadar->objectUnderRadarPixel( &mouse );
168 UnicodeString msg;
169 if( obj )
170 msg.format( L"Object under mouse on radar '%S'(%d)",
171 obj->getTemplate()->getName().str(), obj->getID() );
172 else
173 msg.format( L"Mouse (%d,%d) in Radar window L(%d,%d)", mouse.x, mouse.y, radar.x, radar.y );
174 TheInGameUI->message( msg );
175*/
176
177 // keep the cursor for any context commands
178 const CommandButton *command = TheInGameUI->getGUICommand();
179 if( command
180 && (command->getCommandType() == GUI_COMMAND_SPECIAL_POWER || command->getCommandType() == GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT)
181 && BitTest( command->getOptions(), NEED_TARGET_POS ) )
182 {
183 Int index = TheMouse->getCursorIndex( command->getCursorName() );
184
185 if( index != Mouse::INVALID_MOUSE_CURSOR )
186 TheMouse->setCursor( (Mouse::MouseCursor)index );
187 else
188 TheMouse->setCursor( Mouse::CROSS );
189
190 } // end if
191 else
192 {
193 // Else we are not super targeting, so we have to try to refresh the move cursor.
194 // We can't just do this on Enter and Exit, because hotkeys allow state to change
195 // while we are in the radar.
196 const DrawableList *drawableList = TheInGameUI->getAllSelectedLocalDrawables();
197 Mouse::MouseCursor cur = Mouse::ARROW;
198
199 if (!(drawableList->empty() || msg == GWM_MOUSE_LEAVING))
200 {
201 if (command && command->getCommandType() == GUI_COMMAND_ATTACK_MOVE)
202 {
203 cur = Mouse::ATTACKMOVETO;
204 }
205 else
206 {
207 cur = Mouse::MOVETO;
208 }
209 }
210
211 // Groovy
212 TheMouse->setCursor(cur);
213 }
214
215 } // end if
216
217 break;
218
219 } // end case mouse position
220
221 // ------------------------------------------------------------------------
222 case GWM_RIGHT_UP:// Here to eat
223 case GWM_LEFT_UP:// Here to eat
224 break;
225
226 case GWM_RIGHT_DOWN:
227 case GWM_LEFT_DOWN:
228 {
229 ICoord2D mouse;
230 ICoord2D radar;
231 ICoord2D size;
232 ICoord2D screenPos;
233 Coord3D world;
234
235 // get window size
236 window->winGetSize( &size.x, &size.y );
237
238 // get mouse position
239 mouse.x = mData1 & 0xFFFF;
240 mouse.y = mData1 >> 16;
241
242 // get window screen position
243 window->winGetScreenPosition( &screenPos.x, &screenPos.y );
244
245 // set mouse position to be relative to this window
246 mouse.x -= screenPos.x;
247 mouse.y -= screenPos.y;
248
249 //
250 // translate mouse position to radar position ... we know that the mouse
251 // location given to us here is relative to the HUD window, which is
252 // completely drawn with the radar ... so it's just a translation from
253 // our window size we're drawing into to the radar cell size
254 //
255 if( (TheRadar->isRadarHidden() == FALSE || TheRadar->isRadarForced()) &&
256 TheRadar->localPixelToRadar( &mouse, &radar ) &&
257 TheRadar->radarToWorld( &radar, &world ) )
258 {
259
260 // No drawables, or a right click automatically means its a look at.
261 // Having drawables and being in attack move mode means that we should attack move.
262 // Having drawables and not being in attack move mode means that we should move.
263
264 const DrawableList *drawableList = TheInGameUI->getAllSelectedLocalDrawables(); // locally-owned only
265
266 // see if the user wants to move the tactical view
267 if ( drawableList->empty()
268 || (! TheGlobalData->m_useAlternateMouse && msg == GWM_RIGHT_DOWN)
269 || (TheGlobalData->m_useAlternateMouse && msg == GWM_LEFT_DOWN) )
270 {
271 TheTacticalView->lookAt( &world );
272 break;
273 }
274
275 // evaluate any special powers that can be executed from the radar
276 const CommandButton *command = TheInGameUI->getGUICommand();
277 if( command
278 && (command->getCommandType() == GUI_COMMAND_SPECIAL_POWER || command->getCommandType() == GUI_COMMAND_SPECIAL_POWER_FROM_SHORTCUT)
279 && BitTest( command->getOptions(), NEED_TARGET_POS )
280 )
281 {
282
283 // do the command
284 TheGameClient->evaluateContextCommand( NULL, &world, CommandTranslator::DO_COMMAND );
285
286 } // end if
287 else if( command && command->getCommandType() == GUI_COMMAND_ATTACK_MOVE)
288 {
289 // Attack move has changed from a modifier to a command, so it moves up here.
290
291 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_DO_ATTACKMOVETO );
292 msg->appendLocationArgument( world );
293
294 // Play the unit voice response
295 pickAndPlayUnitVoiceResponse(TheInGameUI->getAllSelectedDrawables(), GameMessage::MSG_DO_ATTACKMOVETO);
296 }
297 else
298 {
299 GameMessage *newMsg = NULL;
300
301 // Do the superweapon stuff here, before issuing these other messages
302
303 // GS Leaving commented out to show that isInAttackMoveToMode is NEVER SET. It's a command now, not a modifier.
304// if (TheInGameUI->isInAttackMoveToMode())
305// {
306// newMsg = TheMessageStream->appendMessage(GameMessage::MSG_DO_ATTACKMOVETO);
307// newMsg->appendLocationArgument(world);
308// // Play the unit voice response
309// pickAndPlayUnitVoiceResponse(drawableList, GameMessage::MSG_DO_ATTACKMOVETO);
310// break;
311// }
312
313 newMsg = TheMessageStream->appendMessage(GameMessage::MSG_DO_MOVETO);
314 newMsg->appendLocationArgument(world);
315 // Play the unit voice response
316 pickAndPlayUnitVoiceResponse(drawableList, GameMessage::MSG_DO_MOVETO);
317
318 } // end else
319
320 }
321
322
323 break;
324
325 } // end left down
326
327 // ------------------------------------------------------------------------
328 default:
329 return MSG_IGNORED;
330
331 } // end switch( msg )
332
333 TheInGameUI->clearAttackMoveToMode();
334 return MSG_HANDLED;
335
336} // end LeftHUDInput

352WindowMsgHandledType ControlBarSystem( GameWindow *window, UnsignedInt msg,
353 WindowMsgData mData1, WindowMsgData mData2 )
354{
355 static NameKeyType buttonCommunicator = NAMEKEY_INVALID;
356 if(TheScriptEngine && TheScriptEngine->isGameEnding())
357 return MSG_IGNORED;
358 switch( msg )
359 {
360 // --------------------------------------------------------------------------------------------
361 case GWM_CREATE:
362 {
363
364 // get ids for our children controls
365 buttonCommunicator = TheNameKeyGenerator->nameToKey( AsciiString("ControlBar.wnd:PopupCommunicator") );
366
367 break;
368
369 } // end create
370
371 //---------------------------------------------------------------------------------------------
372 case GBM_MOUSE_ENTERING:
373 case GBM_MOUSE_LEAVING:
374 {
375 GameWindow *control = (GameWindow *)mData1;
376
377 TheControlBar->processContextSensitiveButtonTransition( control, (GadgetGameMessage)msg);
378 break;
379 }
380
381 //---------------------------------------------------------------------------------------------
382 case GBM_SELECTED:
383 case GBM_SELECTED_RIGHT:
384 {
385 GameWindow *control = (GameWindow *)mData1;
386 static NameKeyType beaconPlacementButtonID = NAMEKEY("ControlBar.wnd:ButtonPlaceBeacon");
387 static NameKeyType beaconDeleteButtonID = NAMEKEY("ControlBar.wnd:ButtonDeleteBeacon");
388 static NameKeyType beaconClearTextButtonID = NAMEKEY("ControlBar.wnd:ButtonClearBeaconText");
389 static NameKeyType beaconGeneralButtonID = NAMEKEY("ControlBar.wnd:ButtonGeneral");
390// static NameKeyType buttonSmallID = NAMEKEY("ControlBar.wnd:ButtonSmall");
391// static NameKeyType buttonMediumID = NAMEKEY("ControlBar.wnd:ButtonMedium");
392 static NameKeyType buttonLargeID = NAMEKEY("ControlBar.wnd:ButtonLarge");
393 static NameKeyType buttonOptions = NAMEKEY("ControlBar.wnd:ButtonOptions");
394 static NameKeyType buttonIdleWorker = NAMEKEY("ControlBar.wnd:ButtonIdleWorker");
395
396 Int controlID = control->winGetWindowId();
397 if( controlID == buttonCommunicator )
398 {
399 ToggleDiplomacy(FALSE);
400 }
401 else if( controlID == beaconPlacementButtonID && TheGameLogic->isInMultiplayerGame() &&
402 ThePlayerList->getLocalPlayer()->isPlayerActive())
403 {
404 const CommandButton *commandButton = TheControlBar->findCommandButton( "Command_PlaceBeacon" );
405 TheInGameUI->setGUICommand( commandButton );
406 }
407 else if( controlID == beaconDeleteButtonID && TheGameLogic->isInMultiplayerGame() )
408 {
409 TheMessageStream->appendMessage( GameMessage::MSG_REMOVE_BEACON );
410 }
411 else if( controlID == beaconClearTextButtonID && TheGameLogic->isInMultiplayerGame() )
412 {
413 static NameKeyType textID = NAMEKEY("ControlBar.wnd:EditBeaconText");
414 GameWindow *win = TheWindowManager->winGetWindowFromId(NULL, textID);
415 if (win)
416 {
417 GadgetTextEntrySetText( win, UnicodeString::TheEmptyString );
418 }
419 }
420 else if( controlID == beaconGeneralButtonID)
421 {
422 HideQuitMenu( );
423 TheControlBar->togglePurchaseScience();
424 }
425 //else if( controlID == buttonSmallID)
426 // {
427 // TheControlBar->switchControlBarStage( CONTROL_BAR_STAGE_LOW );
428 // }
429 // else if( controlID == buttonMediumID)
430 // {
431 // TheControlBar->switchControlBarStage( CONTROL_BAR_STAGE_SQUISHED );
432 // }
433
434 else if( controlID == buttonLargeID)
435 {
436 TheControlBar->toggleControlBarStage();
437 }
438 else if( controlID == buttonOptions)
439 {
440 ToggleQuitMenu();
441 }
442 else if( controlID == buttonIdleWorker)
443 {
444 HideQuitMenu( );
445 TheInGameUI->selectNextIdleWorker();
446 }
447 else
448 {
449 //
450 // all buttons from all the context sensitive user interface windows are part of the
451 // control bar, send the button processing that way
452 //
453 TheControlBar->processContextSensitiveButtonClick( control, (GadgetGameMessage)msg );
454 }
455 break;
456
457 } // end button selected
458
459 //---------------------------------------------------------------------------------------------
460 case GEM_EDIT_DONE:
461 {
462 GameWindow *control = (GameWindow *)mData1;
463 Int controlID = control->winGetWindowId();
464 static NameKeyType textID = NAMEKEY("ControlBar.wnd:EditBeaconText");
465 if (controlID == textID)
466 {
467 // set beacon text
468 if (TheInGameUI->getSelectCount() == 1)
469 {
470 GameMessage *msg = TheMessageStream->appendMessage( GameMessage::MSG_SET_BEACON_TEXT );
471 UnicodeString newText = GadgetTextEntryGetText( control );
472 TheLanguageFilter->filterLine(newText);
473 const WideChar * c = newText.str();
474 while ( c && *c )
475 {
476 msg->appendWideCharArgument( *c++ );
477 }
478 msg->appendWideCharArgument( L'\0' ); // trailing NULL
479 }
480 }
481 break;
482 } // end edit done
483
484 //---------------------------------------------------------------------------------------------
485 default:
486 return MSG_IGNORED;
487
488 } // end switch( msg )
489
490 return MSG_HANDLED;
491
492} // end ControlBarSystem

501void ShowControlBar( Bool immediate )
502{
503 showReplayControls();
504 if(TheControlBar)
505 TheControlBar->showSpecialPowerShortcut();
506 if (TheWindowManager)
507 {
508 Int id = (Int)TheNameKeyGenerator->nameToKey(AsciiString("ControlBar.wnd:ControlBarParent"));
509 GameWindow *window = TheWindowManager->winGetWindowFromId(NULL, id);
510
511 if (window)
512 {
513 TheControlBar->switchControlBarStage(CONTROL_BAR_STAGE_DEFAULT);
514 TheTacticalView->setHeight((Int)(TheDisplay->getHeight() * 0.80f));
515 if (TheControlBar->m_animateWindowManager && !immediate)
516 {
517 TheControlBar->m_animateWindowManager->reset();
518 //TheControlBar->m_animateWindowManager->registerGameWindow(window, WIN_ANIMATION_SLIDE_BOTTOM_TIMED, TRUE, 1000, 0);
519 TheControlBar->m_animateWindowManager->registerGameWindow(window, WIN_ANIMATION_SLIDE_BOTTOM, TRUE, 500, 0);
520 TheControlBar->animateSpecialPowerShortcut(TRUE);
521 }
522 window->winHide(FALSE);
523 }
524
525 }
526
527 // We want to get everything recalced since this is a major state change.
528 if(TheControlBar)
529 TheControlBar->markUIDirty();
530
531}// void ShowControlBar(void)

536void HideControlBar( Bool immediate )
537{
538 hideReplayControls();
539 if(TheControlBar)

542 {
543 Int id = (Int)TheNameKeyGenerator->nameToKey(AsciiString("ControlBar.wnd:ControlBarParent"));

560 }

562 {
563 TheControlBar->m_animateWindowManager->reverseAnimateWindow();
564 TheControlBar->animateSpecialPowerShortcut(FALSE);
565 }
566
567 //Always get rid of the purchase science screen!

570 TheControlBar->hidePurchaseScience();
571 }
572 }

578void ToggleControlBar( Bool immediate )
579{
580 toggleReplayControls();
581
582 if (TheWindowManager)
583 {
584 Int id = (Int)TheNameKeyGenerator->nameToKey(AsciiString("ControlBar.wnd:ControlBarParent"));
585 GameWindow *window = TheWindowManager->winGetWindowFromId(NULL, id);
586
587 if (window)
588 {
589 if (window->winIsHidden())
590 {
591 if(TheControlBar)
592 TheControlBar->showSpecialPowerShortcut();
593
594 //now hidden, we're making it visible again so shrink viewport under the window
595 TheTacticalView->setHeight((Int)(TheDisplay->getHeight() * 0.80f));
596 window->winHide(!window->winIsHidden());
597 TheControlBar->switchControlBarStage(CONTROL_BAR_STAGE_DEFAULT);
598 if (TheControlBar->m_animateWindowManager && !immediate)
599 {
600 TheControlBar->m_animateWindowManager->reset();
601 //TheControlBar->m_animateWindowManager->registerGameWindow(window, WIN_ANIMATION_SLIDE_BOTTOM_TIMED, FALSE, 500, 0);
602 TheControlBar->m_animateWindowManager->registerGameWindow(window, WIN_ANIMATION_SLIDE_BOTTOM, TRUE, 500, 0);
603 TheControlBar->animateSpecialPowerShortcut(TRUE);
604 }
605 }
606 else
607 {
608 if(TheControlBar)
609 TheControlBar->hideSpecialPowerShortcut();
610 TheTacticalView->setHeight(TheDisplay->getHeight());
611 window->winHide(!window->winIsHidden());
612 }
613
614 }
615
616 }
