70//=============================================================================
71static LRESULT CALLBACK checkBoxPropertiesCallback( HWND hWndDialog,
72 UINT message,
73 WPARAM wParam,
74 LPARAM lParam )
75{
76 Int returnCode;
77
78 //
79 // handle any common messages between all property dialogs cause they
80 // are designed to have controls doing the same functionality
81 // and names
82 //
83 if( HandleCommonDialogMessages( hWndDialog, message,
84 wParam, lParam, &returnCode ) == TRUE )
85 return returnCode;
86
87 switch( message )
88 {
89
90 // ------------------------------------------------------------------------
91 case WM_COMMAND:
92 {
93// Int notifyCode = HIWORD( wParam ); // notification code
94 Int controlID = LOWORD( wParam ); // control ID
95// HWND hWndControl = (HWND)lParam; // control window handle
96
97 switch( controlID )
98 {
99
100 // --------------------------------------------------------------------
101 case IDOK:
102 {
103 GameWindow *window = TheEditor->getPropertyTarget();
104
105 // sanity
106 if( window )
107 {
108 ImageAndColorInfo *info;
109
110 // save the common properties
111 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
112 break;
113
114 // save the image and color data
115 // ----------------------------------------------------------------
116 info = GetStateInfo( CHECK_BOX_ENABLED );
117 GadgetCheckBoxSetEnabledImage( window, info->image );
118 GadgetCheckBoxSetEnabledColor( window, info->color );
119 GadgetCheckBoxSetEnabledBorderColor( window, info->borderColor );
120
121 info = GetStateInfo( CHECK_BOX_ENABLED_UNCHECKED_BOX );
122 GadgetCheckBoxSetEnabledUncheckedBoxImage( window, info->image );
123 GadgetCheckBoxSetEnabledUncheckedBoxColor( window, info->color );
124 GadgetCheckBoxSetEnabledUncheckedBoxBorderColor( window, info->borderColor );
125
126 info = GetStateInfo( CHECK_BOX_ENABLED_CHECKED_BOX );
127 GadgetCheckBoxSetEnabledCheckedBoxImage( window, info->image );
128 GadgetCheckBoxSetEnabledCheckedBoxColor( window, info->color );
129 GadgetCheckBoxSetEnabledCheckedBoxBorderColor( window, info->borderColor );
130
131 // ----------------------------------------------------------------
132 info = GetStateInfo( CHECK_BOX_DISABLED );
133 GadgetCheckBoxSetDisabledImage( window, info->image );
134 GadgetCheckBoxSetDisabledColor( window, info->color );
135 GadgetCheckBoxSetDisabledBorderColor( window, info->borderColor );
136
137 info = GetStateInfo( CHECK_BOX_DISABLED_UNCHECKED_BOX );
138 GadgetCheckBoxSetDisabledUncheckedBoxImage( window, info->image );
139 GadgetCheckBoxSetDisabledUncheckedBoxColor( window, info->color );
140 GadgetCheckBoxSetDisabledUncheckedBoxBorderColor( window, info->borderColor );
141
142 info = GetStateInfo( CHECK_BOX_DISABLED_CHECKED_BOX );
143 GadgetCheckBoxSetDisabledCheckedBoxImage( window, info->image );
144 GadgetCheckBoxSetDisabledCheckedBoxColor( window, info->color );
145 GadgetCheckBoxSetDisabledCheckedBoxBorderColor( window, info->borderColor );
146
147 // ----------------------------------------------------------------
148 info = GetStateInfo( CHECK_BOX_HILITE );
149 GadgetCheckBoxSetHiliteImage( window, info->image );
150 GadgetCheckBoxSetHiliteColor( window, info->color );
151 GadgetCheckBoxSetHiliteBorderColor( window, info->borderColor );
152
153 info = GetStateInfo( CHECK_BOX_HILITE_UNCHECKED_BOX );
154 GadgetCheckBoxSetHiliteUncheckedBoxImage( window, info->image );

157

159 GadgetCheckBoxSetHiliteCheckedBoxImage( window, info->image );
160 GadgetCheckBoxSetHiliteCheckedBoxColor( window, info->color );
161 GadgetCheckBoxSetHiliteCheckedBoxBorderColor( window, info->borderColor );
162
163 } // end if
164
165 DestroyWindow( hWndDialog );
166 break;
167
168 } // end OK
169
170 // --------------------------------------------------------------------
171 case IDCANCEL:
172 {
173
174 DestroyWindow( hWndDialog );
175 break;
176
177 } // end cancel
178
179 } // end switch( LOWORD( wParam ) )
180
181 return 0;
182
183 } // end of WM_COMMAND
184
185 // ------------------------------------------------------------------------
186 case WM_CLOSE:
187 {
188
189 DestroyWindow( hWndDialog );
190 return 0;
191
192 } // end close
193
194 // ------------------------------------------------------------------------
195 default:
196 return 0;
197
198 } // end of switch
199
200} // end checkBoxPropertiesCallback

206// InitCheckBoxPropertiesDialog ===============================================
208//=============================================================================
209HWND InitCheckBoxPropertiesDialog( GameWindow *window )
210{
211 HWND dialog;
212
213 // create the dialog box
214 dialog = CreateDialog( TheEditor->getInstance(),
215 (LPCTSTR)CHECK_BOX_PROPERTIES_DIALOG,
216 TheEditor->getWindowHandle(),
217 (DLGPROC)checkBoxPropertiesCallback );
218 if( dialog == NULL )
219 return NULL;
220
221 // do the common initialization
222 CommonDialogInitialize( window, dialog );
223
224 //
225 // store in the image and color table the values for this putton
226 //
227 const Image *image;
228 Color color, borderColor;
229
230 // --------------------------------------------------------------------------
231 image = GadgetCheckBoxGetEnabledImage( window );
232 color = GadgetCheckBoxGetEnabledColor( window );
233 borderColor = GadgetCheckBoxGetEnabledBorderColor( window );
234 StoreImageAndColor( CHECK_BOX_ENABLED, image, color, borderColor );
235
236 image = GadgetCheckBoxGetEnabledUncheckedBoxImage( window );
237 color = GadgetCheckBoxGetEnabledUncheckedBoxColor( window );
238 borderColor = GadgetCheckBoxGetEnabledUncheckedBoxBorderColor( window );
239 StoreImageAndColor( CHECK_BOX_ENABLED_UNCHECKED_BOX, image, color, borderColor );
240
241 image = GadgetCheckBoxGetEnabledCheckedBoxImage( window );
242 color = GadgetCheckBoxGetEnabledCheckedBoxColor( window );
243 borderColor = GadgetCheckBoxGetEnabledCheckedBoxBorderColor( window );
244 StoreImageAndColor( CHECK_BOX_ENABLED_CHECKED_BOX, image, color, borderColor );
245
246 // --------------------------------------------------------------------------
247 image = GadgetCheckBoxGetDisabledImage( window );
248 color = GadgetCheckBoxGetDisabledColor( window );
249 borderColor = GadgetCheckBoxGetDisabledBorderColor( window );
250 StoreImageAndColor( CHECK_BOX_DISABLED, image, color, borderColor );
251
252 image = GadgetCheckBoxGetDisabledUncheckedBoxImage( window );
253 color = GadgetCheckBoxGetDisabledUncheckedBoxColor( window );
254 borderColor = GadgetCheckBoxGetDisabledUncheckedBoxBorderColor( window );
255 StoreImageAndColor( CHECK_BOX_DISABLED_UNCHECKED_BOX, image, color, borderColor );
256
257 image = GadgetCheckBoxGetDisabledCheckedBoxImage( window );
258 color = GadgetCheckBoxGetDisabledCheckedBoxColor( window );
259 borderColor = GadgetCheckBoxGetDisabledCheckedBoxBorderColor( window );
260 StoreImageAndColor( CHECK_BOX_DISABLED_CHECKED_BOX, image, color, borderColor );
261
262 // --------------------------------------------------------------------------
263 image = GadgetCheckBoxGetHiliteImage( window );
264 color = GadgetCheckBoxGetHiliteColor( window );
265 borderColor = GadgetCheckBoxGetHiliteBorderColor( window );
266 StoreImageAndColor( CHECK_BOX_HILITE, image, color, borderColor );

275 borderColor = GadgetCheckBoxGetHiliteCheckedBoxBorderColor( window );
276 StoreImageAndColor( CHECK_BOX_HILITE_CHECKED_BOX, image, color, borderColor );
277
278 // select the button enabled state for display
279 SwitchToState( CHECK_BOX_ENABLED, dialog );
280
281 //
282 // initialize the dialog with values from the window
283 //
284
285 return dialog;
286
287} // end InitCheckBoxPropertiesDialog
