71void parseUpgradePair( INI *ini, void *instance, void *store, const void *userData )
72{
73 upgradePair info;
74 info.type = "";
75 info.amount = 0;
76
77 const char *token = ini->getNextToken( ini->getSepsColon() );
78
79 if ( stricmp(token, "UpgradeType") == 0 )
80 {
81 token = ini->getNextTokenOrNull( ini->getSepsColon() );
82 if (!token) throw INI_INVALID_DATA;
83
84 info.type = token;
85 }
86 else
87 throw INI_INVALID_DATA;
88
89
90 token = ini->getNextTokenOrNull( ini->getSepsColon() );
91 if ( stricmp(token, "Boost") == 0 )
92 info.amount = INI::scanInt(ini->getNextToken( ini->getSepsColon() ));
93 else
94 throw INI_INVALID_DATA;
95
96 // Insert the info into the upgrade list
97 std::list<upgradePair> * theList = (std::list<upgradePair>*)store;
98 theList->push_back(info);
99
100} // end parseFactionObjectCreationList

120void AutoDepositUpdate::awardInitialCaptureBonus( Player *player )
121{
122 m_depositOnFrame = TheGameLogic->getFrame() + getAutoDepositUpdateModuleData()->m_depositFrame;
123 if(!player || !m_awardInitialCaptureBonus || getAutoDepositUpdateModuleData()->m_initialCaptureBonus <= 0)
124 return;
125
126 player->getMoney()->deposit( getAutoDepositUpdateModuleData()->m_initialCaptureBonus );
127 player->getScoreKeeper()->addMoneyEarned( getAutoDepositUpdateModuleData()->m_initialCaptureBonus );
128
129 //Display cash income floating over the blacklotus
130 if(getAutoDepositUpdateModuleData()->m_initialCaptureBonus > 0)
131 {
132 UnicodeString moneyString;
133 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), getAutoDepositUpdateModuleData()->m_initialCaptureBonus );
134 Coord3D pos;
135 pos.set( getObject()->getPosition() );
136 pos.z += 10.0f; //add a little z to make it show up above the unit.
137 Color color = player->getPlayerColor() | GameMakeColor( 0, 0, 0, 230 );
138 TheInGameUI->addFloatingText( moneyString, &pos, color );
139 }
140
141 m_awardInitialCaptureBonus = FALSE;
142}

146UpdateSleepTime AutoDepositUpdate::update( void )
147{
148 const AutoDepositUpdateModuleData *modData = getAutoDepositUpdateModuleData();
150 if( TheGameLogic->getFrame() >= m_depositOnFrame)
151 {
152 if (!m_initialized) {
153 // Note - we have to set these in update, because during load the team is set,
154 // and we don't want to award initial bonus on load. jba :)
155 m_awardInitialCaptureBonus = TRUE;
156 m_initialized = TRUE;
157 }
158 m_depositOnFrame = TheGameLogic->getFrame() + modData->m_depositFrame;
159
160 if(getObject()->isNeutralControlled() || modData->m_depositAmount <= 0 )
161 return UPDATE_SLEEP_NONE;
162
163 // makes sure that buildings under construction do not get a bonus CCB
164 if( getObject()->getConstructionPercent() != CONSTRUCTION_COMPLETE )
165 return UPDATE_SLEEP_NONE;
166
167 int moneyAmount = modData->m_depositAmount + getUpgradedSupplyBoost();
168
169 if( modData->m_isActualMoney )
170 {
171 getObject()->getControllingPlayer()->getMoney()->deposit( moneyAmount );
172 getObject()->getControllingPlayer()->getScoreKeeper()->addMoneyEarned( modData->m_depositAmount);
173 }
174
175 Bool displayMoney = moneyAmount > 0 ? TRUE : FALSE;
176 if( getObject()->testStatus(OBJECT_STATUS_STEALTHED) )
177 {
178 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
179 if( !getObject()->isLocallyControlled() && !getObject()->testStatus(OBJECT_STATUS_DETECTED) )
180 {
181 displayMoney = FALSE;
182 }
183
184 }
185
186 if( displayMoney )
187 {
188
189 const Object *owner = getObject();
190 if ( owner )
191 {
192
193 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
194 UnicodeString moneyString;
195 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), moneyAmount );
196 Coord3D pos;
197 pos.set( getObject()->getPosition() );
198 pos.z += 10.0f; //add a little z to make it show up above the unit.
199
200 if ( owner->isKindOf( KINDOF_STRUCTURE ) )
201 {
202 Real width = owner->getGeometryInfo().getMajorRadius() * 0.3f;
203 Real depth = owner->getGeometryInfo().getMinorRadius() * 0.3f;
204 pos.x += GameClientRandomValue(-width,width);
205 pos.y += GameClientRandomValue(-depth,depth);
206 }
207
208
209 Color color = getObject()->getControllingPlayer()->getPlayerColor() | GameMakeColor( 0, 0, 0, 230 );
210 TheInGameUI->addFloatingText( moneyString, &pos, color );
211 }
212 }
213 }
214
215 return UPDATE_SLEEP_NONE;
216}

219Int AutoDepositUpdate::getUpgradedSupplyBoost() const
220{
221 Player *player = getObject()->getControllingPlayer();
222 if (!player) return 0;
223
224 // Loop through the upgrade pairs and see if an upgrade is present that adds supply boost
225 std::list<upgradePair>::const_iterator it = getAutoDepositUpdateModuleData()->m_upgradeBoost.begin();
226 while (it != getAutoDepositUpdateModuleData()->m_upgradeBoost.end())
227 {
228 upgradePair info = *it;
229
230 // Check if the player has the desired upgrade. If so return the boost
231 static const UpgradeTemplate *upgradeTemplate = TheUpgradeCenter->findUpgrade( info.type.c_str() );
232 if (player && upgradeTemplate && player->hasUpgradeComplete(upgradeTemplate))
233 {
234 return info.amount;
235 }
236
237 // check next
238 ++it;
239 }
240
241 return 0;
242}
