106/*******************************************************************************
107
108 Matrix3D
109
110 Three important notes:
111 - I use *column-vectors*
112 - I use a *right-handed* coordinate system
113 - These matrices are *orthogonal*
114
115 3D Transformation matrices. This class is really a 4x4 homogeneous
116 matrix where the last row is assumed to always be 0 0 0 1. However,
117 since I don't store the last row, you cant do some things that you can
118 do with a real 4x4 homogeneous matrix.
119
120 I use column-vectors so normally transformations are post-multiplied
121 and camera transformations should be pre-multiplied. The methods of
122 this class called Translate, Rotate_X, etc. all perform post-multiplication
123 with the current matix. These methods (Translate, Rotate_X, etc) also
124 have been hand-coded to only perform the necessary arithmetic. The
125 * operator can be used for general purpose matrix multiplication or to
126 transform a vector by a matrix.
127
128 Some operations in this class assume that the matrix is orthogonal.
129
130*********************************************************************************/

579/***********************************************************************************************
580 * Matrix3D::Set -- init a matrix to be a rotation about the given axis *
581 * *
582 * INPUT: *
583 * *
584 * OUTPUT: *
585 * *
586 * WARNINGS: *
587 * *
588 * HISTORY: *
589 * 2/24/98 GTH : Created. *
590 *=============================================================================================*/
591WWINLINE void Matrix3D::Set(const Vector3 & axis,float s,float c)
592{
593 assert(WWMath::Fabs(axis.Length2() - 1.0f) < 0.001f);
594
595 Row[0].Set(
596 (float)(axis[0]*axis[0] + c*(1.0f - axis[0]*axis[0])),
597 (float)(axis[0]*axis[1]*(1.0f - c) - axis[2]*s),
598 (float)(axis[2]*axis[0]*(1.0f - c) + axis[1]*s),
599 0.0f
600 );
601
602 Row[1].Set(
603 (float)(axis[0]*axis[1]*(1.0f - c) + axis[2]*s),
604 (float)(axis[1]*axis[1] + c*(1.0f - axis[1]*axis[1])),
605 (float)(axis[1]*axis[2]*(1.0f - c) - axis[0]*s),
606 0.0f
607 );
608
609 Row[2].Set(
610 (float)(axis[2]*axis[0]*(1.0f - c) - axis[1]*s),
611 (float)(axis[1]*axis[2]*(1.0f - c) + axis[0]*s),
612 (float)(axis[2]*axis[2] + c*(1 - axis[2]*axis[2])),
613 0.0f
614 );
615}
