105/*------------------------------------------------------------------------------------------------
106void SpecialAbilityUpdate::update( void )
107
108This is the brains of the entire special ability update. There are several optional steps and
109variations that can be processed for any particular type of special ability. A special ability
110that has every option will do the following in order:
111
112 1 -- APPROACH: If I'm not close enough to the target, then approach it
113 2 -- UNPACK: If I need to unpack before I can prepare, then do so now (this uses the model
114 condition unpack).
115 3 -- PREPARE: If I need to perform a task for a period of time before I can trigger my special
116 ability, then do so now. A good example is aiming with a targetting laser for a few
117 seconds before firing your special weapon.
118 4 -- TRIGGER: Once preparation is complete, fire your special ability now.
119 5 -- PACK: If I need to pack after finishing my attack, do so now.
120 6 -- FINISH: Clean up the states, and turn off the update.
121
122Variations:
123
124 Persistent Specials -- A persistent special will continually trigger it's effect every so often
125 and never end. A good example of this is the disable building hack. The hacker will run up
126 to the target building, unpack, prepare (firing hack stream), then after a period of time,
127 the building becomes disabled. But because it's persistent, we reset the preparation and
128 trigger the building disabled code over and over again -- which is on a timer.
129 No Target Specials -- You can link two different main specials together. Colonel Burton has the
130 ability to lay C4 charges on multiple targets. Activating these specials require a target.
131 The non-target version is actually the detonator, which goes through all the special objects
132 and detonates each one of them.
133
134Options:
135 SpecialPowerTemplate -- Defines the special power template that links to a command.
136 StartAbilityRange -- Specify how far you want your unit to be from the target before
137 start your special ability.
138 AbilityAbortRange -- After starting an attack, it'll allow preparation unless the target goes
139 beyond this range. If this happens, the ability is aborted outright.
140 PreparationTime -- How long it takes to prepare your special once in position and unpacked.
141 PersistentPrepTime -- This value defines whether or not you are using a persistent special.
142 Once the special ability is triggered, it'll wait until this specified
143 delay occurs and it'll trigger it again, for ever until the unit dies,
144 the target dies, or the unit decides to do something else.
145 PackTime -- How long it takes to pack up the unit after triggering a non persistent
146 special ability or after ordering the unit to do something else, or the
147 target dies (if applicable).
148 PackUnpackVariationFactor -- Randomizes the pack and unpack time by specified range. The closer
149 to zero, the smaller the variation. 0.2 would have a pack or unpack range
150 of xTime +/- 20%. This is important because it represents averages.
151 UnpackTime -- Same as PackTime, except used once entering range of target.
152 SkipPackingWithNoTarget -- This option is used by the No Target Special variation, and only uses
153 packing/unpacking when you have a specific target. In the case of Colonel
154 Burton, this value is set to true. When he plants a C4 charge, he has a
155 target, therefore he requires unpacking (laying the charge). When he runs
156 away and decides to detonate the charge, he calls the same special
157 ability, but without a target -- therefore the packing is skipped and
158 detonates it right away.
159 SpecialObject -- Defines the special object the unit will create via the special ability.
160 In one case, it creates and maintains the laser or binary stream during
161 preparation. In Colonel Burton's case, it keeps track of the C4 charges
162 that he has placed.
163 MaxSpecialObjects -- Defines the max number of special objects that can exist at any given
164 time. The laser example only has one, but the C4 charges can have more.
165 SpecialObjectsPersistent -- If this flag is set, then the objects will remain should the owner
166 decides to do something else... C4 charges are a good example.
167 EffectDuration -- Defines the duration of the special ability. In the case of disabling
168 the building (hacker), this value will dictate how long the building
169 will be disabled should the hacker die or stop the attack.
170 UniqueSpecialObjectTargets -- Prevents the owner from placing multiple special objects on the
171 same target. C4 charges once again.
172 SpecialObjectsPersistWhenOwnerDies -- If the owner dies, the special objects will remain in
173 the world. Timed C4 charges is a good example, but remote C4 charges
174 is a bad example -- because it requires the owner to detonate them.
175 FlipObjectAfterPacking -- Simply rotates the object 180 degrees after packing (due to special
176 animation case).
177 FlipObjectAfterUnPacking -- Simply rotates the object 180 degrees after unpacking (due to
178 special animation case). Used by colonel burton after planting charge.
179
180
181 Lets use the Age of Kings Trebuchet as a simple example:
182 Variation -- PersistentPrepTime (set to true in ini file)
183 1 -- APPROACH: Get within range before unpacking
184 2 -- UNPACK: Unpack the treb at this location.
185 3 -- PREPARE: Aim at the target
186 4 -- TRIGGER: Fire the treb.
187 5 -- RESET PREPARATION: Go to step 3 until target dead.
188 6 -- PACK: Assuming we are done... pack up
189 7 -- FINISH: Stop the special ability
190
191-------------------------------------------------------------------------------------------------*/
192UpdateSleepTime SpecialAbilityUpdate::update( void )
193{
194
196 // Note: This will be complicated because one difficult example is handling Burton when he dies while
197 // bombs are in the world.
198
199 //Validation of special objects makes sure that they still exist. Sometimes special
200 //objects can be destroyed or removed, or detonate, etc! When they go missing,
201 //they are removed from the special object list and free up slots for those units
202 //that can have multiple.
203 //We need to do this now because it's possible the special objects need to be checked while
204 //the the special ability is over (thus inactive).
205
206 const SpecialAbilityUpdateModuleData* data = getSpecialAbilityUpdateModuleData();
207
208 validateSpecialObjects();
209
210 //Important! This check will see if there has been any commands issued by either the player
211 //or script. When told to do something else, we need to immediately cleanup our special ability.
212 //This also means some things might be left around like timed charges to detonate.
213 if( getObject()->isEffectivelyDead() )
214 {
215 onExit( TRUE );
216 return calcSleepTime();
217 }
218
219 if( !m_active ) // Not active.
220 return calcSleepTime();
221
222 AIUpdateInterface *ai = getObject()->getAIUpdateInterface();
223 if( !ai )
224 {
225 onExit( false );
226 return calcSleepTime();
227 }
228 if( ai->getLastCommandSource() != CMD_FROM_AI )
229 {
230 onExit( false );
231 return calcSleepTime();
232 }
233 if( ai->isMoving() && isPowerCurrentlyInUse() && !m_facingInitiated )
234 {
235 // Capture is broken by movement just as if we had been given a direct command (above check).
236 // However, the time of Facing the target is considered isPowerCurrentlyInUse, but isMoving. So let that slide.
237 switch(data->m_specialPowerTemplate->getSpecialPowerType() )
238 {
239 case SPECIAL_INFANTRY_CAPTURE_BUILDING:
240 case SPECIAL_BLACKLOTUS_CAPTURE_BUILDING:
241 {
242 onExit( false );
243 return calcSleepTime();
244 }
245 break;
246
247 default:
248 break;
249 }
250 }
251
252
253 //STEP 2 & 5(6) -- Handles packing and unpacking in progress. If packing
254 //then ends the special ability once complete. Things that don't pack
255 //will never be handled, nor do things that aren't in a packing state.
256 if( handlePackingProcessing() )
257 {
258 return calcSleepTime();
259 }
260
261 Bool shouldAbort = false;
262
263 // A dead target will end our special (if we are using a target).
264 if (m_targetID != INVALID_ID)
265 {
266 Object* target = TheGameLogic->findObjectByID(m_targetID);
267
268 if (target != NULL)
269 {
270 if (target->isEffectivelyDead())
271 shouldAbort = TRUE;
272 else switch (data->m_specialPowerTemplate->getSpecialPowerType())
273 {
274 case SPECIAL_INFANTRY_CAPTURE_BUILDING:
275 case SPECIAL_BLACKLOTUS_CAPTURE_BUILDING:
276 case SPECIAL_HACKER_DISABLE_BUILDING:
277 {
278 if (target->getTeam() == getObject()->getTeam())
279 {
280 // it's been captured by a colleague! we should stop.
281 shouldAbort = TRUE;
282 }
283 //deliberately falling through...
284 }
285 case SPECIAL_BLACKLOTUS_STEAL_CASH_HACK:
286 case SPECIAL_BOOBY_TRAP:
287 {
288 if ( target->testStatus( OBJECT_STATUS_STEALTHED ) && (target->testStatus( OBJECT_STATUS_DETECTED ) == FALSE ) )
289 {
290 if ( !isPreparationComplete() )
291 shouldAbort = TRUE;
292 }
293 break;
294 }
295 case SPECIAL_REMOTE_CHARGES:
296 case SPECIAL_TIMED_CHARGES:
297 {
298 if ( ! needToUnpack() )
299 {
300 if ( target->testStatus( OBJECT_STATUS_STEALTHED ) && (target->testStatus( OBJECT_STATUS_DETECTED ) == FALSE ) )
301 {
302 if ( !isPreparationComplete() )
303 shouldAbort = TRUE;
304 }
305 }
306 break;
307 }
308 case SPECIAL_MISSILE_DEFENDER_LASER_GUIDED_MISSILES:
309 {
310 if ( target->isKindOf( KINDOF_STRUCTURE ) )
311 shouldAbort = TRUE;
312 //deliberately falling through
313 }
314 case SPECIAL_BLACKLOTUS_DISABLE_VEHICLE_HACK:
315 {
316 if ( target->testStatus( OBJECT_STATUS_STEALTHED ) && (target->testStatus( OBJECT_STATUS_DETECTED ) == FALSE ) )
317 {
318 // where'd my target go? 'Twas here just a second ago.
319 shouldAbort = TRUE;
320 }
321 break;
322 }
323
324 }
325
326 }
327
328 }
329
330 SpecialPowerModuleInterface *spm = getMySPM();
331
332 if ( shouldAbort || spm == NULL )
333 {
334 // doh, a colleague has already captured it. just stop.
335 ai->aiIdle( CMD_FROM_AI );
336 onExit( false );
337 return calcSleepTime();
338 }
339
340
341
342 //DETERMINE OUR PHASE! BRAIN LOGIC
343 if( !isPreparationComplete() )
344 {
345 //The special ability has fired, now continue to process the special ability
346 //until it expires
347
348 Bool SPMReady = TRUE;// normally considered ready since this ability has just been initiated
349 // Lorenzen added this additional flag to support the NapalmBombDrop
350 // It causes this update to force a recharge of the SPM between drops
351 if( isPersistentAbility() && getDoesPersistenceRequireRecharge() )//unless I intend to persist in this ability's effect, whereupon I must verify that power is recharged
352 SPMReady = ( spm->isReady() && spm->getReadyFrame() < TheGameLogic->getFrame() );
353
354 if ( SPMReady )// if power requires recharging, lets freeze prep countdown until power is ready
355 m_prepFrames--;
356
357 if( isPreparationComplete() )
358 {
359 //STEP 4 -- TRIGGER (with preparation)
360 triggerAbilityEffect();
361 if( isPersistentAbility() )
362 {
363 //VARIATION -- PERSISTENCE (repeats preparation)
364 resetPreparation();
365
366 //tell the special power module to restart the recharge timer
367 if ( getDoesPersistenceRequireRecharge() )
368 spm->startPowerRecharge();
369
370
371 }
372 else
373 {
374 endPreparation();
375 if( needToPack() )
376 {
377 //STEP 5 -- PACK
378 //Note: If we actually do pack, then cleanup will be handled in
379 //handlePackingProcess(), near the top of this function.
380 startPacking(true);
381 }
382 else
383 {
384 //STEP 6 -- FINISH
385 finishAbility();
386 }
387 }
388 }
389 else
390 {
391 //Process the preparation if it's still not complete.
392 Bool continuePrep = continuePreparation();
393 if( !continuePrep )
394 {
395 //We failed so abort!
396 endPreparation();
397 if( needToPack() )
398 {
399 //STEP 5 -- PACK
400 //Note: If we actually do pack, then cleanup will be handled in
401 //handlePackingProcess(), near the top of this function.
402 startPacking(false);
403 }
404 else
405 {
406 //STEP 6 -- FINISH
407 finishAbility();
408 }
409 }
410 }
411 }
412 else if( isWithinStartAbilityRange() )
413 {
414 m_withinStartAbilityRange = true;
415 if( !isFacing() && needToFace() )
416 {
417 startFacing();
418 return calcSleepTime();
419 }
420
421 if( needToUnpack() )
422 {
423 //STEP 2 -- UNPACK
424 startUnpacking();
425 return calcSleepTime();
426 }
427
428 if( m_packingState == STATE_UNPACKED )
429 {
430 //STEP 3 -- PREPARE
431 startPreparation();
432
433 if( isPreparationComplete() )
434 {
435 //STEP 4 -- TRIGGER (skipping preparation)
436 triggerAbilityEffect();
437
438
439 // Lorenzen added this additional flag to support the NapalmBombDrop
440 // It causes this update to force a recharge of the SPM between drops
441 if( isPersistentAbility() && getDoesPersistenceRequireRecharge())
442 {
443 //VARIATION -- PERSISTENCE (repeats preparation)
444 resetPreparation();
445
446 //tell the special power module to restart the recharge timer
447 spm->startPowerRecharge();
448
449 return calcSleepTime();
450
451 }
452 else
453 endPreparation();
454
455 if( needToPack() )
456 {
457 //STEP 5 -- PACK
458 //Note: If we actually do pack, then cleanup will be handled in
459 //handlePackingProcess(), near the top of this function.
460 startPacking(true);
461 }
462 else
463 {
464 //STEP 6 -- FINISH
465 finishAbility();
466 }
467 }
468 }
469 }
470 else if( ai->isIdle() )
471 {
472 //STEP 1 -- APPROACH
473 approachTarget();
474 }
475 return calcSleepTime();
476}

479Bool SpecialAbilityUpdate::initiateIntentToDoSpecialPower( const SpecialPowerTemplate *specialPowerTemplate,
480 const Object *targetObj,
481 const Coord3D *targetPos,
482 const Waypoint *way,
483 UnsignedInt commandOptions )
484{
485 const SpecialAbilityUpdateModuleData* data = getSpecialAbilityUpdateModuleData();
486 const SpecialPowerTemplate *spTemplate = data->m_specialPowerTemplate;
487
488 if( spTemplate != specialPowerTemplate )
489 {
490 //Check to make sure our modules are connected.
491 return FALSE;
492 }
493
494 //Clear target values
495 m_targetID = INVALID_ID;
496 m_targetPos.zero();
497 m_locationCount = 0;
498 m_prepFrames = 0;
499 m_animFrames = 0;
500 m_packingState = STATE_PACKED;
501 m_facingInitiated = false;
502 m_facingComplete = false;
503 m_withinStartAbilityRange = false;
504
505// getObject()->getControllingPlayer()->getAcademyStats()->recordSpecialPowerUsed( specialPowerTemplate );
506
507 getObject()->clearModelConditionFlags(
508 MAKE_MODELCONDITION_MASK4( MODELCONDITION_UNPACKING, MODELCONDITION_PACKING, MODELCONDITION_FIRING_A, MODELCONDITION_RAISING_FLAG ) );
509
510 if( targetObj )
511 {
512 //Get the target!
513 m_targetID = targetObj ? targetObj->getID() : INVALID_ID;
514 }
515 else if( targetPos )
516 {
517 //Get the position!
518 m_targetPos = *targetPos;
519 }
520
521 //Clear any old AI before starting this special ability.
522 if( !getObject()->getAIUpdateInterface() )
523 {
524 return FALSE;
525 }
526 getObject()->getAIUpdateInterface()->aiIdle( CMD_FROM_AI );
527
528 //Determine whether we are triggering a command (rather than executing special at location or target)
529 m_noTargetCommand = !targetObj && !targetPos;
530
531 if( data->m_unpackTime == 0 || m_noTargetCommand && data->m_skipPackingWithNoTarget )
532 {
533 //Only unpack if we need to -- setting it to unpacked will skip step 2 in the update
534 m_packingState = STATE_UNPACKED;
535 }
536
537 m_active = true;
538
539 //Prevent other mutually exclusive specials from running (kill them now if we're starting something else)
540 SpecialAbilityUpdate *disableSA;
541 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_BLACKLOTUS_DISABLE_VEHICLE_HACK );
542 if( disableSA && disableSA != this )
543 disableSA->onExit( FALSE );
544 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_BLACKLOTUS_STEAL_CASH_HACK );
545 if( disableSA && disableSA != this )
546 disableSA->onExit( FALSE );
547 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_BLACKLOTUS_CAPTURE_BUILDING );
548 if( disableSA && disableSA != this )
549 disableSA->onExit( FALSE );
550 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_REMOTE_CHARGES );
551 if( disableSA && disableSA != this )
552 disableSA->onExit( FALSE );
553 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_TIMED_CHARGES );
554 if( disableSA && disableSA != this )
555 disableSA->onExit( FALSE );
556 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_INFANTRY_CAPTURE_BUILDING );
557 if( disableSA && disableSA != this )
558 disableSA->onExit( FALSE );
559 disableSA = getObject()->findSpecialAbilityUpdate( SPECIAL_BOOBY_TRAP );
560 if( disableSA && disableSA != this )
561 disableSA->onExit( FALSE );
562
563
564 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
565
566 return TRUE;
567}

570Bool SpecialAbilityUpdate::isPowerCurrentlyInUse( const CommandButton *command ) const
571{
572 if( command )
573 {
574 if( command->getSpecialPowerTemplate() && command->getSpecialPowerTemplate()->getSpecialPowerType() == SPECIAL_REMOTE_CHARGES )

577 {

581 }
582 }
583 }
584
585
586 if( m_packingState != STATE_NONE )
587 {
588 //exception for powers with zero reload time... they are ready to use immediately!
589 if ( (m_packingState == STATE_PACKING || m_packingState == STATE_PACKED) &&
590 command && command->getSpecialPowerTemplate()->getReloadTime() == 0 )
591 return false;
592
593 if ( m_withinStartAbilityRange )
594 {
595 return true;
596 }
597 }
598
599 return false;
600}

637//-------------------------------------------------------------------------------------------------
638//Some special abilities requires packing or unpacking. When setup, all this function does is
639//decrement the counter and clear the model state when finished.
640//Returns TRUE if we are still packing or unpacking.
641//-------------------------------------------------------------------------------------------------
642Bool SpecialAbilityUpdate::handlePackingProcessing()
643{
644 const SpecialAbilityUpdateModuleData* data = getSpecialAbilityUpdateModuleData();
645 if( m_animFrames > 0 )
646 {
647 m_animFrames--;
648 if( m_animFrames == 0 )
649 {
650 // We're done, so clear the states.
651 getObject()->clearModelConditionFlags( MAKE_MODELCONDITION_MASK2( MODELCONDITION_UNPACKING, MODELCONDITION_PACKING ) );
652
653 if( m_packingState == STATE_UNPACKING )
654 {
655 if( data->m_flipObjectAfterUnpacking )
656 {
657 getObject()->setOrientation( getObject()->getOrientation() + PI );
658 }
659 m_packingState = STATE_UNPACKED;
660 }
661 else if( m_packingState == STATE_PACKING )
662 {
663 if( data->m_flipObjectAfterPacking )
664 {
665 getObject()->setOrientation( getObject()->getOrientation() + PI );
666 }
667 //We just finished packing up, therefore
668 //we have completed our special ability.
669 m_packingState = STATE_PACKED;
670
671 //Do exit preparation.
672 finishAbility();
673
674 //Complete the special ability now
675 return true;
676 }
677
678 //We're finished processing
679 return false;
680 }
681
682
683
684 //This is new... the ability to disable stealth before triggering
685 if( getSpecialAbilityUpdateModuleData()->m_loseStealthOnTrigger &&
686 m_animFrames < getSpecialAbilityUpdateModuleData()->m_preTriggerUnstealthFrames)
687 {
688 StealthUpdate* stealth = getObject()->getStealth();
689 if( stealth )
690 {
691 stealth->markAsDetected();
692 }
693 }
694
695
696
697 //We're still processing
698 return true;
699 }
700 //We're not processing
701 return false;
702}

723Bool SpecialAbilityUpdate::needToUnpack() const
724{
725 const SpecialAbilityUpdateModuleData* data = getSpecialAbilityUpdateModuleData();
726 if( m_packingState == STATE_PACKED )

728 if( data->m_skipPackingWithNoTarget && m_noTargetCommand )

821Bool SpecialAbilityUpdate::isWithinStartAbilityRange() const
822{
823 const SpecialAbilityUpdateModuleData* data = getSpecialAbilityUpdateModuleData();
824 const Object *self = getObject();
825
826 //Quickly convert very short range approachs to "contact" class requiring collision before
827 //stopping.
828 Real range = data->m_startAbilityRange;
829 const Real UNDERSIZE = PATHFIND_CELL_SIZE_F * 0.25f;
830 range = __max( 0.0f, range - UNDERSIZE );
831
832 if( m_withinStartAbilityRange )
833 {
834 //Only get within range once.
835 return true;
836 }
837
838 Real fDistSquared = 0.0f;
839 Object *target = NULL;
840 if( m_targetID != INVALID_ID )
841 {
842 target = TheGameLogic->findObjectByID( m_targetID );
843 if( target )
844 {
845 fDistSquared = ThePartitionManager->getDistanceSquared( self, target, FROM_BOUNDINGSPHERE_2D );
846 }
847 }
848 else if( m_targetPos.x || m_targetPos.y || m_targetPos.z ) //It's zero if not used...
849 {
850 fDistSquared = ThePartitionManager->getDistanceSquared( self, &m_targetPos, FROM_BOUNDINGSPHERE_2D );
851 }
852 else
853 {
854 //No position, so this step is useless
855 return true;
856 }
857
858 //Check to see how far we are from the target!
859 Real fStartRangeSquared = data->m_startAbilityRange * data->m_startAbilityRange;
860 if( fDistSquared <= fStartRangeSquared )
861 {
862 if( range == 0.0f && m_targetID != INVALID_ID )
863 {
864 //We want to ensure we collided with our target first!
865 ObjectIterator *iter = ThePartitionManager->iteratePotentialCollisions( self->getPosition(), self->getGeometryInfo(), 0.0f );
866 MemoryPoolObjectHolder hold(iter);
867 for( Object *them = iter->first(); them; them = iter->next() )
868 {
869 if( target == them )
870 {
871 return true;
872 }
873 }
874 return false;
875 }
876
877 if( data->m_approachRequiresLOS )
878 {
879 //Make sure we can see the target!
880 PartitionFilterLineOfSight filterLOS( self );
881 PartitionFilter *filters[] = { &filterLOS, NULL };
882 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( self, range, FROM_BOUNDINGSPHERE_2D, filters, ITER_SORTED_NEAR_TO_FAR );
883 for( Object *theTarget = iter->first(); theTarget; theTarget = iter->next() )
884 {
885 //LOS check succeeded.
886 if( target == theTarget )
887 {
888 return true;
889 }
890 }
891 }
892 else
893 {
894 return true;
895 }
896 }
897 return false;
898}

1834 ai->aiIdle(CMD_FROM_AI);

1840 //TheAudio->addAudioEvent(&event);
