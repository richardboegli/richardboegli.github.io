148Int W3DStatusCircle::initData(void)
149{
150 Int i;
151
152 m_needUpdate = true;
153 freeMapResources(); //free old data and ib/vb
154
155 m_numTriangles = NUM_TRI;
156 m_indexBuffer=NEW_REF(DX8IndexBufferClass,(m_numTriangles*3));
157
158 // Fill up the IB
159 DX8IndexBufferClass::WriteLockClass lockIdxBuffer(m_indexBuffer);
160 UnsignedShort *ib=lockIdxBuffer.Get_Index_Array();
161
162 for (i=0; i<3*m_numTriangles; i+=3)
163 {
164 ib[0]=i;
165 ib[1]=i+1;
166 ib[2]=i+2;
167
168 ib+=3; //skip the 3 indices we just filled
169 }
170
171 m_vertexBufferCircle=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,m_numTriangles*3,DX8VertexBufferClass::USAGE_DEFAULT));
172 m_vertexBufferScreen=NEW_REF(DX8VertexBufferClass,(DX8_FVF_XYZDUV1,2*3,DX8VertexBufferClass::USAGE_DEFAULT));
173
174 //go with a preset material for now.
175 m_vertexMaterialClass=VertexMaterialClass::Get_Preset(VertexMaterialClass::PRELIT_DIFFUSE);
176
177 m_shaderClass = ShaderClass::ShaderClass(SC_ALPHA);// _PresetOpaque2DShader;//; //_PresetOpaqueShader;
178
179
180 return 0;
181}

186Int W3DStatusCircle::updateCircleVB(void)
187{
188 Int i, k;
189 Real shade;
190 DX8VertexBufferClass *pVB = m_vertexBufferCircle;
191 if (m_vertexBufferCircle )
192 {
193 m_needUpdate = false;
194 DX8VertexBufferClass::WriteLockClass lockVtxBuffer(pVB);
195 VertexFormatXYZDUV1 *vb = (VertexFormatXYZDUV1*)lockVtxBuffer.Get_Vertex_Array();
196
197 const Real theZ = 0.0f;
198 const Real theRadius = 0.02f;
199 const Int theAlpha = 127;
200 Int diffuse = m_diffuse + (theAlpha<<24); // b g<<8 r<<16 a<<24.
201 Int limit = m_numTriangles;
202 float curAngle = 0;
203 float deltaAngle = 2*PI/limit;
204 for (i=0; i<limit; i++)
205 {
206
207 shade=0.7f*255.0f;
208 for (k=0; k<3; k++) {
209 vb->z= theZ;
210 if (k==0) {
211 vb->x= 0;
212 vb->y= 0;
213 } else if (k==1) {
214 Vector3 vec(theRadius,0,theZ);
215 vec.Rotate_Z(curAngle);
216 vb->x= vec.X;
217 vb->y= vec.Y;
218 } else if (k==2) {
219 Real angle = curAngle+deltaAngle;
220 if (i==limit-1) {
221 angle = 0;
222 }
223 Vector3 vec(theRadius,0,theZ);
224 vec.Rotate_Z(angle);
225 vb->x= vec.X;
226 vb->y= vec.Y;
227 }
228 vb->diffuse = diffuse;
229 vb->u1=0;
230 vb->v1=0;
231 vb++;
232 }
233 curAngle += deltaAngle;
234
235 }
236 return 0; //success.
237 }
238 return -1;
239}

304void W3DStatusCircle::Render(RenderInfoClass & rinfo)
305{
306 if (!TheGameLogic->isInGame() || TheGameLogic->getGameMode() == GAME_SHELL)
307 return;
308
309 if (m_indexBuffer == NULL) {
310 initData();
311 }
312 if (m_indexBuffer == NULL) {
313 return;
314 }
315 Bool setIndex = false;
316 Matrix3D tm(true);
317 if( TheGlobalData->m_showTeamDot )
318 {
319 if (m_needUpdate) {
320 updateCircleVB();
321 }
322 //Apply the shader and material
323 DX8Wrapper::Set_Material(m_vertexMaterialClass);
324 DX8Wrapper::Set_Shader(m_shaderClass);
325 DX8Wrapper::Set_Texture(0, NULL);
326 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
327 DX8Wrapper::Set_Vertex_Buffer(m_vertexBufferCircle);
328 setIndex = true;
329
330 Vector3 vec(0.95f, 0.67f, 0);
331 Matrix3x3 rot(true);
332
333 tm.Set_Translation(vec);
334
335 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
336 DX8Wrapper::Draw_Triangles( 0,NUM_TRI, 0, (m_numTriangles*3));
337 }
338
339
340 ScriptEngine::TFade fade = TheScriptEngine->getFade();
341 if (fade == ScriptEngine::FADE_NONE) {
342 return;
343 }
344
345 if (!setIndex) {
346 DX8Wrapper::Set_Material(m_vertexMaterialClass);
347 DX8Wrapper::Set_Index_Buffer(m_indexBuffer,0);
348 DX8Wrapper::Set_Texture(0, NULL);
349 }
350
351 tm.Make_Identity();
352 Real intensity = TheScriptEngine->getFadeValue();
353 Int clr = 255*intensity;
354 Int diffuse = (0xff<<24)|(clr<<16)|(clr<<8)|clr; // b g<<8 r<<16 a<<24.
355 updateScreenVB(diffuse);
356 DX8Wrapper::Set_Transform(D3DTS_WORLD,tm);
357 DX8Wrapper::Set_Shader(ShaderClass(SC_ADD));
358 DX8Wrapper::Set_Vertex_Buffer(m_vertexBufferScreen);
359 DX8Wrapper::Apply_Render_State_Changes();
360 switch (fade) {
361 default:
362 case ScriptEngine::FADE_ADD:
363 DX8Wrapper::Draw_Triangles( 0,2, 0, (2*3));
364 break;
365 case ScriptEngine::FADE_SUBTRACT:
366 DX8Wrapper::Set_DX8_Render_State(D3DRS_BLENDOP, D3DBLENDOP_REVSUBTRACT );
367 DX8Wrapper::Draw_Triangles( 0,2, 0, (2*3));
368 DX8Wrapper::Set_DX8_Render_State(D3DRS_BLENDOP, D3DBLENDOP_ADD );
369 break;
370 case ScriptEngine::FADE_SATURATE:
371 // 4x multiply
372 DX8Wrapper::Set_DX8_Render_State(D3DRS_SRCBLEND,D3DBLEND_DESTCOLOR);
373 DX8Wrapper::Set_DX8_Render_State(D3DRS_DESTBLEND,D3DBLEND_SRCCOLOR);
374 DX8Wrapper::Draw_Triangles( 0,2, 0, (2*3));
375 DX8Wrapper::Draw_Triangles( 0,2, 0, (2*3));
376 break;
377 case ScriptEngine::FADE_MULTIPLY:
378 // Straight multiply
379 DX8Wrapper::Set_DX8_Render_State(D3DRS_SRCBLEND,D3DBLEND_ZERO);
380 DX8Wrapper::Set_DX8_Render_State(D3DRS_DESTBLEND,D3DBLEND_SRCCOLOR);
381 DX8Wrapper::Draw_Triangles( 0,2, 0, (2*3));
382 break;
383 }
384 ShaderClass::Invalidate();
385}
