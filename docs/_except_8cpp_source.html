157/***********************************************************************************************
158 * _purecall -- This function overrides the C library Pure Virtual Function Call error *
159 * *
160 * *
161 * *
162 * INPUT: Nothing *
163 * *
164 * OUTPUT: 0 = no error *
165 * *
166 * WARNINGS: None *
167 * *
168 * HISTORY: *
169 * 8/22/00 11:42AM ST : Created *
170 *=============================================================================================*/
171int __cdecl _purecall(void)
172{
173 int return_code = 0;
174
175#ifdef WWDEBUG
176 /*
177 ** Use int3 to cause an exception.
178 */
179 WWDEBUG_SAY(("Pure Virtual Function call. Oh No!\n"));
180 _asm int 0x03;
181#endif //_DEBUG_ASSERT
182
183 return(return_code);
184}

355
356 if (imagehelp != NULL) {
357 DebugString ("Exception Handler: Found IMAGEHLP.DLL - linking to required functions\n");
358 char const *function_name = NULL;
359 unsigned long *fptr = (unsigned long*) &_SymCleanup;
360 int count = 0;

376 ** Retrieve the programs symbols if they are available

394 _SymSetOptions(SYMOPT_DEFERRED_LOADS | SYMOPT_UNDNAME);

424 unsigned long access_address = 0;

426 if (e_info->ExceptionRecord->ExceptionCode == EXCEPTION_ACCESS_VIOLATION) {

617 for (int fp=0 ; fp<SIZE_OF_80387_REGISTERS / 10 ; fp++) {
618 sprintf(scrap, "ST%d : ", fp);

621 sprintf(scrap, "%02X", context->FloatSave.RegisterArea[(fp*10) + b]);
622 Add_Txt(scrap);
623 }

625 void *fp_data_ptr = (void*)(&context->FloatSave.RegisterArea[fp*10]);
626 double fp_value;
627
628 /*
629 ** Convert FP dump from temporary real value (10 bytes) to double (8 bytes).
630 */
631 _asm {
632 push eax
633 mov eax,fp_data_ptr
634 fld tbyte ptr [eax]
635 fstp qword ptr [fp_value]
636 pop eax
637 }
638 sprintf(scrap, " %+#.17e\r\n", fp_value);
639 Add_Txt(scrap);
640 }

642 /*
643 ** Dump the bytes at EIP. This will make it easier to match the crash address with later versions of the game.
644 */
645 DebugString("EIP bytes dump...\n");
646 sprintf(scrap, "\r\nBytes at CS:EIP (%08X) : ", context->Eip);
647
648 unsigned char *eip_ptr = (unsigned char *) (context->Eip);
649 char bytestr[32];
650
651 for (int c = 0 ; c < 32 ; c++) {
652 if (IsBadReadPtr(eip_ptr, 1)) {
653 strcat(scrap, "?? ");
654 } else {
655 sprintf(bytestr, "%02X ", *eip_ptr);
656 strcat(scrap, bytestr);
657 }
658 eip_ptr++;
659 }
660
661 strcat(scrap, "\r\n\r\n");
662 Add_Txt(scrap);
663
664 /*
665 ** Dump out the values on the stack.
666 */
667 DebugString("Stack dump...\n");
668 Add_Txt("Stack dump (* indicates possible code address) :\r\n");
669 unsigned long *stackptr = (unsigned long*) context->Esp;

671 for (int j=0 ; j<2048 ; j++) {

684 strcat(scrap, "DATA_PTR\r\n");
685 } else {
686 sprintf(scrap, "%08X: %08X", stackptr, *stackptr);
687
688 if (symbols_available) {
689 symptr->SizeOfStruct = sizeof(symbol);
690 symptr->MaxNameLength = 128;
691 symptr->Size = 0;
692 symptr->Address = *stackptr;
693
694 if (_SymGetSymFromAddr != NULL && _SymGetSymFromAddr (GetCurrentProcess(), *stackptr, &displacement, symptr)) {
695 char symbuf[256];
696 sprintf(symbuf, " - %s + %08X", symptr->Name, displacement);
697 strcat(scrap, symbuf);
698 }
699 } else {
700 strcat (scrap, " *");
701 }
702 strcat (scrap, "\r\n");
703 }
704 }
705 Add_Txt(scrap);
706 stackptr++;
707 }
708
709 /*
710 ** Unload the symbols.
711 */
712 if (symbols_available) {
713 if (_SymCleanup != NULL) {
714 _SymCleanup (GetCurrentProcess());
715 }
716
717 if (symload) {
718 if (_SymUnloadModule != NULL) {
719 _SymUnloadModule(GetCurrentProcess(), NULL);
720 }
721 }
722
723 }
724
725 if (imagehelp) {
726 FreeLibrary(imagehelp);
727 }
728
729 Add_Txt ("\r\n\r\n");
730}

1190/***********************************************************************************************
1191 * Stack_Walk -- Walk the stack and get the last n return addresses *
1192 * *
1193 * *
1194 * *
1195 * INPUT: Ptr to return address list *
1196 * Number of return addresses to fetch *
1197 * Ptr to optional context. NULL means use current *
1198 * *
1199 * OUTPUT: Number of return addresses found *
1200 * *
1201 * WARNINGS: None *
1202 * *
1203 * HISTORY: *
1204 * 6/12/2001 11:57AM ST : Created *
1205 *=============================================================================================*/
1206int Stack_Walk(unsigned long *return_addresses, int num_addresses, CONTEXT *context)
1207{
1208 static HINSTANCE _imagehelp = (HINSTANCE) -1;
1209
1210 /*
1211 ** If this is the first time through then fix up the imagehelp function pointers since imagehlp.dll
1212 ** can't be statically linked.
1213 */
1214 if (ImageHelp == (HINSTANCE)-1) {
1215 Load_Image_Helper();
1216 }
1217
1218 /*
1219 ** If there is no debug support .dll available then we can't walk the stack.
1220 */
1221 if (ImageHelp == NULL) {
1222 return(0);
1223 }
1224
1225 /*
1226 ** Set up the stack frame structure for the start point of the stack walk (i.e. here).
1227 */
1228 STACKFRAME stack_frame;
1229 memset(&stack_frame, 0, sizeof(stack_frame));
1230
1231 unsigned long reg_eip, reg_ebp, reg_esp;
1232
1233 __asm {
1234here:
1235 lea eax,here
1236 mov reg_eip,eax
1237 mov reg_ebp,ebp
1238 mov reg_esp,esp
1239 }
1240
1241 stack_frame.AddrPC.Mode = AddrModeFlat;
1242 stack_frame.AddrPC.Offset = reg_eip;
1243 stack_frame.AddrStack.Mode = AddrModeFlat;
1244 stack_frame.AddrStack.Offset = reg_esp;
1245 stack_frame.AddrFrame.Mode = AddrModeFlat;
1246 stack_frame.AddrFrame.Offset = reg_ebp;
1247
1248 /*
1249 ** Use the context struct if it was provided.
1250 */
1251 if (context) {
1252 stack_frame.AddrPC.Offset = context->Eip;
1253 stack_frame.AddrStack.Offset = context->Esp;
1254 stack_frame.AddrFrame.Offset = context->Ebp;
1255 }
1256
1257 int pointer_index = 0;
1258
1259 /*
1260 ** Walk the stack by the requested number of return address iterations.
1261 */
1262 for (int i = 0; i < num_addresses + 1; i++) {
1263 if (_StackWalk(IMAGE_FILE_MACHINE_I386, GetCurrentProcess(), GetCurrentThread(), &stack_frame, NULL, NULL, _SymFunctionTableAccess, _SymGetModuleBase, NULL)) {
1264
1265 /*
1266 ** First result will always be the return address we were called from.
1267 */
1268 if (i==0 && context == NULL) {
1269 continue;
1270 }
1271 unsigned long return_address = stack_frame.AddrReturn.Offset;
1272 return_addresses[pointer_index++] = return_address;
1273 } else {
1274 break;
1275 }
1276 }
1277
1278 return(pointer_index);
1279}
