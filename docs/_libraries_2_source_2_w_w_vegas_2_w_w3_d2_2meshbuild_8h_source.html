72/*
73** MeshBuilderClass
74** This class will process meshes for you, splitting all vertices which do not
75** share all parameters (such as texture coordinates), sort the polygons by
76** material, and put them into strip order.
77**
78** To "build" a mesh:
79** 1. Reset the builder with the number of polys you're going to sumbit
80** 2. Enable the vertex channels that you want
81** 3. Submit each face in the form of a FaceClass
82** 4. Call Build_Mesh
83**
84** To use the results:
85** 1. Call Get_Vertex_Count and Get_Face_Count to get the counts
86** 2. Loop through the verts, looking at each one using Get_Vertex
87** 3. Loop through the faces, looking at each one using Get_Face
88**
89** *NOTE* This class is meant to be relatively self-sufficient. It is used in a
90** variety of different applications which are built on completely different
91** code-bases. Do not introduce dependencies into this module lightly! :-)
92*/
93class MeshBuilderClass
94{
95
96public:
97
98 enum {
99 STATE_ACCEPTING_INPUT = 0, // mesh builder is accepting input triangles
100 STATE_MESH_PROCESSED, // mesh builder has processed the mesh
101
102 MAX_PASSES = 4, // maximum number of material passes supported
103 MAX_STAGES = 8, // maximum number of texture stages supported in a single pass
104 };
