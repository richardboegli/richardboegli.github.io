115{
116 ProductionUpdateModuleData* data = (ProductionUpdateModuleData*)instance;

118 const char* countStr = ini->getNextTokenOrNull();

263Bool ProductionUpdate::queueUpgrade( const UpgradeTemplate *upgrade )
264{
265
266 // sanity
267 if( upgrade == NULL )
268 return FALSE;
269
270 // get the player
271 Player *player = getObject()->getControllingPlayer();
272
273 // sanity check to make sure we can build this upgrade
274 if( upgrade->getUpgradeType() == UPGRADE_TYPE_PLAYER &&
275 TheUpgradeCenter->canAffordUpgrade( player, upgrade ) == FALSE )
276 return FALSE;
277 else if( upgrade->getUpgradeType() == UPGRADE_TYPE_OBJECT &&
278 (getObject()->hasUpgrade( upgrade ) == TRUE ||
279 getObject()->affectedByUpgrade( upgrade ) == FALSE) )
280 return FALSE;
281
282 // you cannot queue the production of an upgrade twice in this queue
283 if( isUpgradeInQueue( upgrade ) == TRUE )
284 return FALSE;
285
286 // STOP cheaters by making sure they can actually build this
287 if( !getObject()->canProduceUpgrade(upgrade) )
288 return FALSE;
289
290 //
291 // you cannot queue a player upgrade production if you are producing one already somewhere else
292 // (or that somewhere else could even possibly be here)
293 //
294 if( upgrade->getUpgradeType() == UPGRADE_TYPE_PLAYER &&
295 (player->hasUpgradeComplete( upgrade ) || player->hasUpgradeInProduction( upgrade )) )
296 return FALSE;
297
298 if (m_productionCount >= getProductionUpdateModuleData()->m_maxQueueEntries)
299 {
300 DEBUG_CRASH(("Production Queue is full... how did we get here?"));
301 return FALSE;
302 }
303
304 // take the cost for the build away from the player
305 Money *money = player->getMoney();
306 money->withdraw( upgrade->calcCostToBuild( player ) );
307
308 // allocate a new production entry
309 ProductionEntry *production = newInstance(ProductionEntry);
310
311 // assing production entry data
312 production->m_type = PRODUCTION_UPGRADE;
313 production->m_upgradeToResearch = upgrade;
314 production->m_productionID = PRODUCTIONID_INVALID; // not needed for upgrades, you can only have one of
315 // this type in the queue
316
317 // tie to the end of the production queue
318 addToProductionQueue( production );
319
320 // add this upgrade as in progress in the player
321 player->addUpgrade( upgrade, UPGRADE_STATUS_IN_PRODUCTION );
322
323
324
325 return TRUE; // queued
326
327} // end queueUpgrade

332void ProductionUpdate::cancelUpgrade( const UpgradeTemplate *upgrade )
333{
334
335 // sanity
336 if( upgrade == NULL )
337 return;
338
339 // get the player
340 Player *player = getObject()->getControllingPlayer();
341
342 // sanity, you can't cancel it if the player isn't actually building one
343 if( upgrade->getUpgradeType() == UPGRADE_TYPE_PLAYER && player->hasUpgradeInProduction( upgrade ) == FALSE )
344 return;
345
346 //
347 // find the production entry for this upgrade in the queue here, there can only be one
348 // of this type in the queue
349 //
350 ProductionEntry *production;
351 for( production = m_productionQueue; production; production = production->m_next )
352 {
353
354 if( production->m_type == PRODUCTION_UPGRADE &&
355 production->m_upgradeToResearch == upgrade )
356 break;
357
358 } // end for
359
360 // sanity, entry not found
361 if( production == NULL )
362 return;
363
364 // refund money back to the player
365 Money *money = player->getMoney();
366 money->deposit( production->m_upgradeToResearch->calcCostToBuild( player ) );
367
368 // remove this production from the queue
369 removeFromProductionQueue( production );
370
371 // delete production instance
372 production->deleteInstance();
373
374 //
375 // remove the IN_PRODUCTION status of this upgrade from the player, object upgrades don't
376 // have any other IN_PRODUCTION status other than their existence in the build queue
377 //
378 if( upgrade->getUpgradeType() == UPGRADE_TYPE_PLAYER )
379 player->removeUpgrade( upgrade );
380
381} // end cancelUpgrade

387Bool ProductionUpdate::queueCreateUnit( const ThingTemplate *unitType, ProductionID productionID )
388{
389 const ProductionUpdateModuleData *data = getProductionUpdateModuleData();
390
391 // if we can't create the unit do nothing
392 if( TheBuildAssistant->canMakeUnit( getObject(), unitType ) != CANMAKE_OK )
393 return FALSE;
394
395 ExitDoorType exitDoor = DOOR_NONE_AVAILABLE;
396
398 // ProductionUpdate to be part of ParkingPlaceBehavior, which I don't currently
399 // have time for...
400 ParkingPlaceBehaviorInterface* pp = NULL;
401 for (BehaviorModule** i = getObject()->getBehaviorModules(); *i; ++i)
402 {
403 if ((pp = (*i)->getParkingPlaceBehaviorInterface()) != NULL)
404 {
405 if (pp->shouldReserveDoorWhenQueued(unitType))
406 {
407 ExitInterface* exitInterface = getObject()->getObjectExitInterface();
408 if (exitInterface)
409 {
410 exitDoor = exitInterface->reserveDoorForExit(unitType, NULL);
411 }
412 if (exitDoor == DOOR_NONE_AVAILABLE)
413 {
414 return FALSE;
415 }
416 break;
417 }
418 }
419 }
420
421 if (m_productionCount >= getProductionUpdateModuleData()->m_maxQueueEntries)
422 {
423 DEBUG_CRASH(("Production Queue is full... how did we get here?"));
424 return FALSE;
425 }
426
427 // take the cost for the build away from the player
428 Player *player = getObject()->getControllingPlayer();
429 Money *money = player->getMoney();
430 money->withdraw( unitType->calcCostToBuild( player ) );
431
432 // allocate a new production entry
433 ProductionEntry *production = newInstance(ProductionEntry);
434
435 // Check to see if there is a quantity modifier entry. What this means is if we are building a particular unit
436 // and it has a modifier, we build that number of objects instead of just one. A good example is the Chinese
437 // barracks building redguards -- they are built 4 at a time, but the user or script only pays for one and
438 // builds four for that price!
439 production->m_productionQuantityTotal = 1;
440 production->m_productionQuantityProduced = 0;
441 for( std::vector<QuantityModifier>::const_iterator it = data->m_quantityModifiers.begin(); it != data->m_quantityModifiers.end(); ++it )
442 {
443 const ThingTemplate* productionTemplate = TheThingFactory->findTemplate( it->m_templateName );
444 if( productionTemplate && productionTemplate->isEquivalentTo( unitType ) )
445 {
446 production->m_productionQuantityTotal = it->m_quantity;
447 break;
448 }
449 }
450
451 // assing production entry data
452 production->m_type = PRODUCTION_UNIT;
453 production->m_objectToProduce = unitType;
454 production->m_productionID = productionID;
455 production->m_exitDoor = exitDoor;
456
457 // tie to the end of the production queue
458 addToProductionQueue( production );
459
460 return TRUE; // unit queued
461
462} // end queueMakeUnit

537void ProductionUpdate::updateDoors()
538{
539 const ProductionUpdateModuleData *d = getProductionUpdateModuleData();
540
541 UnsignedInt now = TheGameLogic->getFrame();
542 for (Int i = 0; i < DOOR_COUNT_MAX; ++i)
543 {
544 //
545 // if we have an open door, see if enough time has passed for us to close it ... likewise
546 // if we were closing a door we will clear the closing condition after an amount of time
547 // has passed
548 //
549 if( m_doors[i].m_doorOpenedFrame )
550 {
551
552 if( now - m_doors[i].m_doorOpenedFrame > d->m_doorOpeningTime )
553 {
554
555 // set our frame markers for door states
556 m_doors[i].m_doorOpenedFrame = 0;
557 m_doors[i].m_doorWaitOpenFrame = now;
558
559 // set the flags that will show the difference in the model
560 m_clearFlags.set( theOpeningFlags[i], true );
561 m_setFlags.set( theOpeningFlags[i], false );
562
563 m_setFlags.set( theWaitingOpenFlags[i] );
564 m_flagsDirty = TRUE;
565
566 } // end if
567
568 } // end if
569 else if( m_doors[i].m_doorWaitOpenFrame )
570 {
571
572 if( now - m_doors[i].m_doorWaitOpenFrame > d->m_doorWaitOpenTime && !m_doors[i].m_holdOpen )
573 {
574
575 // set our frame marker for closing
576 m_doors[i].m_doorWaitOpenFrame = 0;
577 m_doors[i].m_doorClosedFrame = now;
578
579 // set the flags that show the difference in the art
580 m_clearFlags.set( theWaitingOpenFlags[i], true );
581 m_setFlags.set( theWaitingOpenFlags[i], false );
582
583 m_setFlags.set( theClosingFlags[i] );
584 m_flagsDirty = TRUE;
585
586 } // end if
587
588 } // end if
589 else if( m_doors[i].m_doorClosedFrame && !m_doors[i].m_holdOpen )
590 {
591
592 if( now - m_doors[i].m_doorClosedFrame > d->m_doorClosingTime )
593 {
594
595 // set our frame marker for the closed door frame
596 m_doors[i].m_doorClosedFrame = 0;
597
598 // set the flags that will show the difference in the model
599 m_clearFlags.set( theClosingFlags[i], true );
600 m_setFlags.set( theClosingFlags[i], false );
601 m_flagsDirty = TRUE;
602
603 } // end if
604
605 } // end else if
606 }
607}

611UpdateSleepTime ProductionUpdate::update( void )
612{
614 ProductionEntry *production = m_productionQueue;
615 const ProductionUpdateModuleData *d = getProductionUpdateModuleData();
616 UnsignedInt now = TheGameLogic->getFrame();
617 Object *us = getObject();
618
619 // update the door behaviors
620 if( d->m_numDoorAnimations > 0 )
621 updateDoors();
622
623 //
624 // if we're in the construction complete state ... we need to check frame
625 // we're in it and get out of it when the time is up
626 //
627 if( m_constructionCompleteFrame )
628 {
629
630 if( now - m_constructionCompleteFrame > d->m_constructionCompleteDuration )
631 {
632
633 // set our frame marker to be out of construction complete state
634 m_constructionCompleteFrame = 0;
635
636 // set the flags that show the difference in the model
637 m_clearFlags.set( MODELCONDITION_CONSTRUCTION_COMPLETE, true );
638 m_setFlags.set( MODELCONDITION_CONSTRUCTION_COMPLETE, false );
639 m_flagsDirty = TRUE;
640
641 } // end if
642
643 } // end if
644
645 // if we have dirty bits that need to be set and cleared, do it here all at once
646 if( m_flagsDirty == TRUE )
647 {
648 Drawable *draw = us->getDrawable();
649
650 if( draw )
651 draw->clearAndSetModelConditionFlags( m_clearFlags, m_setFlags );
652
653 // clear the things we just set and turn off our dirty flag
654 m_clearFlags.clear();
655 m_setFlags.clear();
656 m_flagsDirty = FALSE;
657
658 } // end if
659
660 // if nothing in the queue get outta here
661 if( production == NULL )
662 return UPDATE_SLEEP_NONE;
663
664 //
665 // if we've become OBJECT_STATUS_SOLD, halt all production ... leave things in the
666 // queue because if the sell process completes we get money back for them, for now we
667 // will be just frozen in time
668 // Actually, there will be nothing in the queue since everything gets cancel/refunded
669 // at the start of sell, but we still don't want to do anything here.
670 //
671 if( us->getStatusBits().test( OBJECT_STATUS_SOLD ) )
672 return UPDATE_SLEEP_NONE;
673
674 // get the player that is building this thing
675 Player *player = us->getControllingPlayer();
676
677 // sanity
678 if( player == NULL )
679 {
680
681 // remove from queue list
682 removeFromProductionQueue( production );
683
684 // delete the production entry
685 production->deleteInstance();
686
687 return UPDATE_SLEEP_NONE;
688
689 } // end if
690
691 //
692 // you can disallow types of units on the fly in scripts, so if there is something
693 // in the queue that suddenly can't be build then get rid of it
694 //
695 if( production->getProductionType() == PRODUCTION_UNIT &&
696 player->allowedToBuild( production->getProductionObject() ) == FALSE )
697 {
698
699 // Don't cancel dozers in the queue. jba.
700 if (!production->getProductionObject()->isKindOf(KINDOF_DOZER))
701 {
702
703 cancelUnitCreate(production->getProductionID());
704 return UPDATE_SLEEP_NONE;
705
706 } // end if
707
708 } // end if
709
710 // increase the frames we've been under production for
711 production->m_framesUnderConstruction++;
712
713 // how many total logic frames does it take to produce this unit
714 Int totalProductionFrames;
715 if( production->m_type == PRODUCTION_UNIT )
716 totalProductionFrames = production->m_objectToProduce->calcTimeToBuild( player );
717 else
718 totalProductionFrames = production->m_upgradeToResearch->calcTimeToBuild( player );
719
720 // figure out our percent complete
721 production->m_percentComplete = INT_TO_REAL( production->m_framesUnderConstruction ) /
722 INT_TO_REAL( totalProductionFrames ) *
723 100.0f;
724
725 // if we've reached 100% or more we're done, tada!
726 if( production->m_percentComplete >= 100.0f )
727 {
728
729 // handle unit production items
730 if( production->m_type == PRODUCTION_UNIT )
731 {
732 Object *creationBuilding = us;
733
734 // create this unit
735
736 //if I've got a ExitModule
737 ExitInterface *exitInterface = creationBuilding->getObjectExitInterface();
738 if( exitInterface )
739 {
740
741 //
742 // only exit if the exit interface will let us ... if we can't we'll just keep this
743 // unit in the production and keep "overbuilding" it, every frame we will check to see
744 // if we can exit and then move onto production of the next one
745 //
746 Int numberToTry = production->getProductionQuantityRemaining();// this will change midloop, so it can't be the For clause
747 for( int i = 0; i < numberToTry; i++ )
748 {
749 ExitDoorType exitDoor = production->getExitDoor();
750 if (exitDoor == DOOR_NONE_AVAILABLE)
751 {
752 exitDoor = exitInterface->reserveDoorForExit(production->m_objectToProduce, NULL);
753 production->setExitDoor(exitDoor);
754 }
755
756 if (exitDoor != DOOR_NONE_AVAILABLE)
757 {
758 // note, could be DOOR_NONE_NEEDED! so door could be null. (srj)
759 DoorInfo* door = (exitDoor >= 0 && exitDoor < DOOR_COUNT_MAX) ? &m_doors[exitDoor] : NULL;
760
761 //
762 // if the producing structure has a door opening animation we will set the condition
763 // bits to open the door (note that we must also take care to remove any condition
764 // that had us previously closing a door)
765 //
766 const ProductionUpdateModuleData *d = getProductionUpdateModuleData();
767 if( d->m_numDoorAnimations > 0 && door != NULL )
768 {
769
770 // if the door is closed, open it
771 if( door->m_doorOpenedFrame == 0 && door->m_doorWaitOpenFrame == 0 && door->m_doorClosedFrame == 0 )
772 {
773
774 door->m_doorOpenedFrame = now;
775 m_setFlags.set( theOpeningFlags[exitDoor] );
776 m_flagsDirty = TRUE;
777
778 } // end if
779 // if the door is waiting-open, keep it there
780 else if( door->m_doorWaitOpenFrame != 0 )
781 {
782
783 door->m_doorWaitOpenFrame = now;
784
785 } // end else if
