94void CreateObjectDie::onDie( const DamageInfo * damageInfo )
95{
96 const CreateObjectDieModuleData *data = getCreateObjectDieModuleData();
97 if (!isDieApplicable(damageInfo))
98 return;
99
100 Object *damageDealer = TheGameLogic->findObjectByID( damageInfo->in.m_sourceID );
101
102 Object *newObject = ObjectCreationList::create( data->m_ocl, getObject(), damageDealer );
103
104 //If we're transferring previous health, we're transfering the last known
105 //health before we died. In the case of the sneak attack tunnel network, it
106 //is killed after the lifetime update expires.
107 if( newObject && data->m_transferPreviousHealth )
108 {
109 //Convert old health to new health.
110 Object *oldObject = getObject();
111 BodyModuleInterface *oldBody = oldObject->getBodyModule();
112 BodyModuleInterface *newBody = newObject->getBodyModule();
113 if( oldBody && newBody )
114 {
115 //First transfer subdual damage
116 DamageInfo damInfo;
117 Real subdualDamageAmount = oldBody->getCurrentSubdualDamageAmount();
118 if( subdualDamageAmount > 0.0f )
119 {
120 damInfo.in.m_amount = subdualDamageAmount;
121 damInfo.in.m_damageType = DAMAGE_SUBDUAL_UNRESISTABLE;
122 damInfo.in.m_sourceID = INVALID_ID;
123 newBody->attemptDamage( &damInfo );
124 }
125
126 //Now transfer the previous health from the old object to the new.
127 damInfo.in.m_amount = oldBody->getMaxHealth() - oldBody->getPreviousHealth();
128 damInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
129 damInfo.in.m_sourceID = oldBody->getLastDamageInfo()->in.m_sourceID;
130 if( damInfo.in.m_amount > 0.0f )
131 {
132 newBody->attemptDamage( &damInfo );
133 }
134
135 }
136
137 //Transfer attackers.
138 for( Object *obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject() )
139 {
140 AIUpdateInterface* ai = obj->getAI();
141 if (!ai)
142 continue;
143
144 ai->transferAttack( oldObject->getID(), newObject->getID() );
145 }
146 }
147
148
149} // end onDie
