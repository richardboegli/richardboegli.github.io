87void ThingFactory::addTemplate( ThingTemplate *tmplate )
88{
89 ThingTemplateHashMapIt tIt = m_templateHashMap.find(tmplate->getName());
90
91 if (tIt != m_templateHashMap.end()) {
92 DEBUG_CRASH(("Duplicate Thing Template name found: %s\n", tmplate->getName().str()));

94

130ThingTemplate *ThingFactory::newTemplate( const AsciiString& name )
131{
132 ThingTemplate *newTemplate;
133
134 // allocate template
135 newTemplate = newInstance(ThingTemplate);
136
137 // if the default template is present, get it and copy over any data to the new template
138 const ThingTemplate *defaultT = findTemplate( AsciiString( "DefaultThingTemplate" ), FALSE );
139 if( defaultT )
140 {
141
142 // copy over static data
143 *newTemplate = *defaultT;
144 newTemplate->setCopiedFromDefault();
145
146 } // end if
147
148 // give template a unique identifier
149 newTemplate->friend_setTemplateID( m_nextTemplateID++ );
150 DEBUG_ASSERTCRASH( m_nextTemplateID != 0, ("m_nextTemplateID wrapped to zero") );
151
152 // assign name
153 newTemplate->friend_setTemplateName( name );
154
155 // add to list
156 addTemplate( newTemplate );
157
158 // return the newly created template
159 return newTemplate;
160
161}

169ThingTemplate* ThingFactory::newOverride( ThingTemplate *thingTemplate )
170{
171
172 // sanity
173 DEBUG_ASSERTCRASH( thingTemplate, ("newOverride(): NULL 'parent' thing template\n") );
174
175 // sanity just for debuging, the weapon must be in the master list to do overrides
176 DEBUG_ASSERTCRASH( findTemplate( thingTemplate->getName() ) != NULL,
177 ("newOverride(): Thing template '%s' not in master list\n",
178 thingTemplate->getName().str()) );
179
180 // find final override of the 'parent' template
181 ThingTemplate *child = (ThingTemplate*) thingTemplate->friend_getFinalOverride();
182
183 // allocate new template
184 ThingTemplate *newTemplate = newInstance(ThingTemplate);
185
186 // copy data from final override to 'newTemplate' as a set of initial default values
187 *newTemplate = *child;
188 newTemplate->setCopiedFromDefault();
189
190 newTemplate->markAsOverride();
191 child->setNextOverride(newTemplate);
192
193 // return the newly created override for us to set values with etc
194 return newTemplate;
195
196} // end newOverride

209void ThingFactory::reset( void )
210{
211 ThingTemplate *t;
212 // go through all templates and delete any overrides
213 for( t = m_firstTemplate; t; /* empty */ )
214 {
215 Bool possibleAdjustment = FALSE;
216 // t itself can be deleted if it is something created for this map only. Therefore,
217 // we need to store what the next item is so that we don't orphan a bunch of templates.
218 ThingTemplate *nextT = t->friend_getNextTemplate();
219 if (t == m_firstTemplate) {
220 possibleAdjustment = TRUE;
221 }
222
223 // if stillValid is NULL after we delete the overrides, then this template was created for
224 // this map only. If it also happens to be m_firstTemplate, then we need to update m_firstTemplate
225 // as well. Finally, if it was only created for this map, we need to remove the name from the
226 // hash map, to prevent any crashes.
227
228 AsciiString templateName = t->getName();
229
230 Overridable *stillValid = t->deleteOverrides();
231 if (stillValid == NULL && possibleAdjustment) {
232 m_firstTemplate = nextT;
233 }
234
235 if (stillValid == NULL) {
236 // Also needs to be removed from the Hash map.
237 m_templateHashMap.erase(templateName);
238 }
239
240 t = nextT;
241 }
242} // end reset

269ThingTemplate *ThingFactory::findTemplateInternal( const AsciiString& name, Bool check )
270{
271 ThingTemplateHashMapIt tIt = m_templateHashMap.find(name);
272
273 if (tIt != m_templateHashMap.end()) {
274 return tIt->second;
275 }
276
277#ifdef LOAD_TEST_ASSETS
278 if (!strncmp(name.str(), TEST_STRING, strlen(TEST_STRING)))
279 {
280 ThingTemplate *tmplate = newTemplate( AsciiString( "Un-namedTemplate" ) );
281
282 // load the values
283 tmplate->initForLTA( name );
284
285 // Kinda lame, but necessary.
286 m_templateHashMap.erase("Un-namedTemplate");
287 m_templateHashMap[name] = tmplate;
288
289 // add tmplate template to the database
290 return findTemplateInternal( name );
291
292 }
293
294#endif
295
296 if( check && name.isNotEmpty() )
297 {
298 DEBUG_CRASH( ("Failed to find thing template %s (case sensitive) This issue has a chance of crashing after you ignore it!", name.str() ) );
299 }
300 return NULL;
301
302} // end getTemplate

305Object *ThingFactory::newObject( const ThingTemplate *tmplate, Team *team, ObjectStatusMaskType statusBits )
306{
307 if (tmplate == NULL)
308 throw ERROR_BAD_ARG;
309
310 const std::vector<AsciiString>& asv = tmplate->getBuildVariations();
311 if (!asv.empty())
312 {
313 Int which = GameLogicRandomValue(0, asv.size()-1);
314 const ThingTemplate* tmp = findTemplate( asv[which] );
315 if (tmp != NULL)
316 tmplate = tmp;
317 }
318
319 DEBUG_ASSERTCRASH(!tmplate->isKindOf(KINDOF_DRAWABLE_ONLY), ("You may not create Objects with the template %s, only Drawables\n",tmplate->getName().str()));
320
321 // have the game logic create an object of the correct type.
322 // (this will throw an exception on failure.)
323 //Added ability to pass in optional statusBits. This is needed to be set prior to
324 //the onCreate() calls... in the case of constructing.
325 Object *obj = TheGameLogic->friend_createObject( tmplate, statusBits, team );
326
327 // run the create function for the thing
328 for (BehaviorModule** m = obj->getBehaviorModules(); *m; ++m)
329 {
330 CreateModuleInterface* create = (*m)->getCreate();
331 if (!create)
332 continue;
333
334 create->onCreate();
335 }
336
337 //
338 // all objects are part of the partition manager system, add it to that
339 // system now
340 //
341 ThePartitionManager->registerObject( obj );
342
343 obj->initObject();
344
345 return obj;
346
347}

372/*static*/ void ThingFactory::parseObjectDefinition( INI* ini, const AsciiString& name, const AsciiString& reskinFrom )
373{
374#if defined(_DEBUG) || defined(_INTERNAL)
375 TheThingTemplateBeingParsedName = name;
376#endif
377
378 // find existing item if present
379 ThingTemplate *thingTemplate = TheThingFactory->findTemplateInternal( name, FALSE );
380 if( !thingTemplate )
381 {
382 // no item is present, create a new one
383 thingTemplate = TheThingFactory->newTemplate( name );
384 if ( ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES )
385 {
386 // This ThingTemplate is actually an override, so we will mark it as such so that it properly
387 // gets deleted on ::reset().
388 thingTemplate->markAsOverride();
389 }
390 }
391 else if( ini->getLoadType() != INI_LOAD_CREATE_OVERRIDES )
392 {
393 //Holy crap, this sucks to debug!!!
394 //If you have two different objects, the previous code would simply
395 //allow you to define multiple objects with the same name, and just
396 //nuke the old one with the new one. So, I (KM) have added this
397 //assert to notify in case of two same-name objects.
398 DEBUG_CRASH(( "[LINE: %d in '%s'] Duplicate factionunit %s found!", ini->getLineNum(), ini->getFilename().str(), name.str() ));
399 }
400 else
401 {
402 thingTemplate = TheThingFactory->newOverride( thingTemplate );
403 }
404
405 if (reskinFrom.isNotEmpty())
406 {
407 const ThingTemplate* reskinTmpl = TheThingFactory->findTemplate(reskinFrom);
408 if (reskinTmpl)
409 {
410 thingTemplate->copyFrom(reskinTmpl);
411 thingTemplate->setCopiedFromDefault();
412 thingTemplate->setReskinnedFrom(reskinTmpl);
413 ini->initFromINI( thingTemplate, thingTemplate->getReskinFieldParse() );
414 }
415 else
416 {
417 DEBUG_CRASH(("ObjectReskin must come after the original Object (%s, %s).\n",reskinFrom.str(),name.str()));
418 throw INI_INVALID_DATA;
419 }
420 }
421 else
422 {
423 ini->initFromINI( thingTemplate, thingTemplate->getFieldParse() );
424 }
425
426 thingTemplate->validate();
427
428 if( ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES )
429 {
430 thingTemplate->resolveNames();
431 }
432
433#if defined(_DEBUG) || defined(_INTERNAL)
434 TheThingTemplateBeingParsedName.clear();
435#endif
436}

542#endif

544 } // end for

546#ifdef CHECK_THING_NAMES
547 dumpMissingStringNames();
