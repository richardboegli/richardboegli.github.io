68void ControlBar::populateButtonProc( Object *obj, void *userData )
69{
70 PopulateButtonInfo* info = (PopulateButtonInfo*)userData;
71
72 // sanity
73 DEBUG_ASSERTCRASH( info->buttonIndex < MAX_STRUCTURE_INVENTORY_BUTTONS,
74 ("Too many objects inside '%s' for the inventory buttons to hold",
75 info->source->getTemplate()->getName().str()) );
76
77 // put object in inventory data
78 info->self->m_containData[ info->buttonIndex ].control = info->inventoryButtons[ info->buttonIndex ];
79 info->self->m_containData[ info->buttonIndex ].objectID = obj->getID();
80
81 // set the UI button that will allow us to press it and cause the object to exit the container
82 const Image *image;
83 image = obj->getTemplate()->getButtonImage();
84 GadgetButtonSetEnabledImage( info->inventoryButtons[ info->buttonIndex ], image );
85
86 //Show the auto-contained object's veterancy symbol!
87 image = calculateVeterancyOverlayForObject( obj );
88 GadgetButtonDrawOverlayImage( info->inventoryButtons[ info->buttonIndex ], image );
89
90 // Enable the button
91 info->inventoryButtons[ info->buttonIndex ]->winEnable( TRUE );
92
93 // move to the next button index
94 info->buttonIndex++;
95
96}

100void ControlBar::populateStructureInventory( Object *building )
101{
102 Int i;
103
104 // reset the inventory data
105 resetContainData();
106
107 // reset hotkeys -- seeing it only is reset in switching contexts. This is a special case
108 // because we are building the hotkeys on the fly sort of... and it changes as guys enter
109 // and leave. Taking this call out will cause multiple hotkeys to be added.
110 if(TheHotKeyManager)
111 TheHotKeyManager->reset();
112
113 // Start by hiding all the buttons. Otherwise buttons we don't use will have the buttons
114 // the last thing selected left behind.
115 for( Int commandIndex = 0; commandIndex < MAX_COMMANDS_PER_SET; commandIndex++ )
116 {
117 if( m_commandWindows[commandIndex] )
118 {
119 m_commandWindows[commandIndex]->winHide(TRUE);
120 }
121 }
122
123 // get the contain module of the object
124 ContainModuleInterface *contain = building->getContain();
125 DEBUG_ASSERTCRASH( contain, ("Object in structure inventory does not contain a Contain Module\n") );
126 if (!contain)
127 return;
128
130 const CommandButton *evacuateCommand = findCommandButton( "Command_Evacuate" );
131 setControlCommand( m_commandWindows[ EVACUATE_ID ], evacuateCommand );
132 m_commandWindows[ EVACUATE_ID ]->winEnable( FALSE );
133
135 const CommandButton *stopCommand = findCommandButton( "Command_Stop" );
136 setControlCommand( m_commandWindows[ STOP_ID ], stopCommand );
137 m_commandWindows[ STOP_ID ]->winEnable( FALSE );
138
139 // get the inventory exit command to assign into the button
141 const CommandButton *exitCommand = findCommandButton( "Command_StructureExit" );
142
143 // get window handles for each of the inventory buttons
144 AsciiString windowName;
145 for( i = 0; i < MAX_STRUCTURE_INVENTORY_BUTTONS; i++ )
146 {
147 // show the window
148 m_commandWindows[ i ]->winHide( FALSE );
149
150 //
151 // disable the button for now, it will be enabled if there is something there
152 // for its contents
153 //
154 m_commandWindows[ i ]->winEnable( FALSE );
155 m_commandWindows[ i ]->winSetStatus( WIN_STATUS_ALWAYS_COLOR );
156 m_commandWindows[ i ]->winClearStatus( WIN_STATUS_NOT_READY );
157
158 // set an inventory command into the game window UI element
159 setControlCommand( m_commandWindows[ i ], exitCommand );
160
161 // Clear any veterancy icon incase the unit leaves!
162 GadgetButtonDrawOverlayImage( m_commandWindows[ i ], NULL );
163 //
164 // if the structure can hold a lesser amount inside it than what the GUI displays
165 // we will completely hide the buttons that can't contain anything
166 //
167 if( i + 1 > contain->getContainMax() )
168 m_commandWindows[ i ]->winHide( TRUE );
169
170
171 } // end for i
172
173 // show the window
174 m_commandWindows[ EVACUATE_ID ]->winHide( FALSE );
175 m_commandWindows[ STOP_ID ]->winHide( FALSE );
176
177 // if there is at least one item in there enable the evacuate and stop buttons
178 if( contain->getContainCount() != 0 )
179 {
180 m_commandWindows[ EVACUATE_ID ]->winEnable( TRUE );
181 m_commandWindows[ STOP_ID ]->winEnable( TRUE );
182 }
183
184 //
185 // iterate each of the objects inside the container and put them in a button, note
186 // we're iterating in reverse order here
187 //
188 PopulateButtonInfo info;
189 info.source = building;
190 info.buttonIndex = 0;
191 info.self = this;
192 info.inventoryButtons = m_commandWindows;
193 contain->iterateContained(populateButtonProc, &info, FALSE );
194
195 //
196 // save how many items were contained by the object at this time so that we can update
197 // it if they change in the future while selected
198 //
199 m_lastRecordedInventoryCount = contain->getContainCount();
200
201} // end populateStructureInventory

205void ControlBar::updateContextStructureInventory( void )
206{
207 Object *source = m_currentSelectedDrawable->getObject();
208
209 //
210 // we're visible, so there is something selected. It is possible that we had a building
211 // selected that can be garrisoned, and while it was selected the enemy occupied it.
212 // in that case we want to unselect the building so that we can't see the contents
213 //
214 Player *localPlayer = ThePlayerList->getLocalPlayer();
215 if( source->isLocallyControlled() == FALSE &&
216 localPlayer->getRelationship( source->getTeam() ) != NEUTRAL )
217 {
218 Drawable *draw = source->getDrawable();
219
220 if( draw )
221 TheInGameUI->deselectDrawable( draw );
222 return;
223
224 } // end if
225
226 //
227 // if the object being displayed in the interface has a different count than we last knew
228 // about we need to repopulate the buttons of the interface
229 //
230 ContainModuleInterface *contain = source->getContain();
231 DEBUG_ASSERTCRASH( contain, ("No contain module defined for object in the iventory bar\n") );
232 if (!contain)
233 return;
234
235 if( m_lastRecordedInventoryCount != contain->getContainCount() )
236 populateStructureInventory( source );
237
238} // end updateContextStructureInventory
