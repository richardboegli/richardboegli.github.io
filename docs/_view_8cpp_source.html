225void View::getScreenCornerWorldPointsAtZ( Coord3D *topLeft, Coord3D *topRight,
226 Coord3D *bottomLeft, Coord3D *bottomRight,
227 Real z )
228{
229 ICoord2D screenTopLeft, screenTopRight, screenBottomLeft, screenBottomRight;
230 ICoord2D origin;
231 Int viewWidth = getWidth();
232 Int viewHeight = getHeight();
233
234 // sanity
235 if( topLeft == NULL || topRight == NULL || bottomLeft == NULL || bottomRight == NULL )
236 return;
237
238 // setup the screen coords for the 4 corners of the viewable display
239 getOrigin( &origin.x, &origin.y );
240 screenTopLeft.x = origin.x; // upper left
241 screenTopLeft.y = origin.y; // upper left
242 screenTopRight.x = origin.x + viewWidth; // upper right
243 screenTopRight.y = origin.y; // upper right
244 screenBottomLeft.x = origin.x + viewWidth; // lower right
245 screenBottomLeft.y = origin.y + viewHeight; // lower right
246 screenBottomRight.x = origin.x; // lower left
247 screenBottomRight.y = origin.y + viewHeight; // lower left
248
249 // project
250 screenToWorldAtZ( &screenTopLeft, topLeft, z );
251 screenToWorldAtZ( &screenTopRight, topRight, z );
252 screenToWorldAtZ( &screenBottomLeft, bottomLeft, z );
253 screenToWorldAtZ( &screenBottomRight, bottomRight, z );
254
255} // end getScreenCornerWorldPointsAtZ
