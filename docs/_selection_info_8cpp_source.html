91extern Bool contextCommandForNewSelection(const DrawableList *currentlySelectedDrawables,
92 const DrawableList *newlySelectedDrawables,
93 SelectionInfo *outSelectionInfo,
94 Bool selectionIsPoint)
95{
96 if (!(currentlySelectedDrawables && newlySelectedDrawables && outSelectionInfo))
97 return FALSE;
98
99 Bool forceFire = TheInGameUI->isInForceAttackMode();
100 Bool forceMove = TheInGameUI->isInForceMoveToMode();
101
102 if (forceFire || forceMove) {
103 return FALSE;
104 }
105
106
107 Player *localPlayer = ThePlayerList->getLocalPlayer();
108 DrawableListCIt it;
109 for (it = currentlySelectedDrawables->begin(); it != currentlySelectedDrawables->end(); ++it) {
110 if (!(*it)) {
111 continue;
112 }
113
114 Object *obj = (*it)->getObject();
115 if (!obj) {
116 continue;
117 }
118
119 if (obj->isLocallyControlled()) {
120 ++outSelectionInfo->currentCountMine;
121 if (obj->isKindOf(KINDOF_INFANTRY)) {
122 ++outSelectionInfo->currentCountMineInfantry;
123 } else if (obj->isKindOf(KINDOF_STRUCTURE)) {
124 ++outSelectionInfo->currentCountMineBuildings;
125 }
126 } else {
127 Relationship rel = localPlayer->getRelationship(obj->getTeam());
128 if (rel == ALLIES) {
129 ++outSelectionInfo->currentCountFriends;
130 } else if (rel == ENEMIES) {
131 ++outSelectionInfo->currentCountEnemies;
132 } else if (rel == NEUTRAL) {
133 ++outSelectionInfo->currentCountCivilians;
134 }
135 }
136 }
137
138 Drawable *newMine = NULL;
139 Drawable *newFriendly = NULL;
140 Drawable *newEnemy = NULL;
141 Drawable *newCivilian = NULL;
142
143 for (it = newlySelectedDrawables->begin(); it != newlySelectedDrawables->end(); ++it) {
144 if (!(*it)) {
145 continue;
146 }
147
148 Object *obj = (*it)->getObject();
149 if (!obj) {
150 continue;
151 }
152
153 if (TheActionManager->canPlayerGarrison(localPlayer, obj, CMD_FROM_PLAYER)) {
154 ++outSelectionInfo->newCountGarrisonableBuildings;
155 }
156 if (obj->isKindOf(KINDOF_CRATE)) {
157 ++outSelectionInfo->newCountCrates;
158 }
159
160 if (obj->isLocallyControlled()) {
161 ++outSelectionInfo->newCountMine;
162 newMine = *it;
163 if (obj->isKindOf(KINDOF_STRUCTURE)) {
164 ++outSelectionInfo->newCountMineBuildings;
165 }
166 } else {
167 Relationship rel = localPlayer->getRelationship(obj->getTeam());
168 if (rel == ALLIES) {
169 newFriendly = *it;
170 ++outSelectionInfo->newCountFriends;
171 } else if (rel == ENEMIES) {
172 newEnemy = *it;
173 ++outSelectionInfo->newCountEnemies;
174 } else if (rel == NEUTRAL) {
175 newCivilian = *it;
176 ++outSelectionInfo->newCountCivilians;
177 }
178 }
179 }
180
181 DEBUG_ASSERTCRASH(outSelectionInfo->currentCountEnemies <= 1, ("Selection bug. jkmcd"));
182 DEBUG_ASSERTCRASH(outSelectionInfo->currentCountFriends <= 1, ("Selection bug. jkmcd"));
183 DEBUG_ASSERTCRASH(outSelectionInfo->currentCountCivilians <= 1, ("Selection bug. jkmcd"));
184
185 if (outSelectionInfo->currentCountEnemies > 0) {
186 // If we have an enemy selected, there are no context sensitive commands
187 return FALSE;
188 }
189
190 if (outSelectionInfo->currentCountFriends > 0) {
191 return FALSE;
192 }
193
194 if (outSelectionInfo->currentCountCivilians > 0) {
195 return FALSE;
196 }
197
198 if (TheGlobalData->m_useAlternateMouse) {
199 // context sensitive commands never apply when selecting in alternate mouse mode
200 return FALSE;
201 }
202
203 if (outSelectionInfo->currentCountMine > 0) {
204 if (outSelectionInfo->newCountEnemies > 0) {
205 if (outSelectionInfo->newCountEnemies == 1 && selectionIsPoint) {
206 return TheGameClient->evaluateContextCommand(newEnemy, newEnemy->getPosition(), CommandTranslator::EVALUATE_ONLY) != GameMessage::MSG_INVALID;
207 }
208
209 return selectionIsPoint;
210 }
211
212 if (outSelectionInfo->newCountMine > 0) {
213 if (outSelectionInfo->newCountMine == 1 && selectionIsPoint && !TheInGameUI->isInPreferSelectionMode()) {
214 return TheGameClient->evaluateContextCommand(newMine, newMine->getPosition(), CommandTranslator::EVALUATE_ONLY) != GameMessage::MSG_INVALID;
215 }
216
217 return FALSE;
218 }
219
220 if (outSelectionInfo->newCountFriends > 0) {
221 if (outSelectionInfo->newCountFriends == 1 && selectionIsPoint) {
222 return TheGameClient->evaluateContextCommand(newFriendly, newFriendly->getPosition(), CommandTranslator::EVALUATE_ONLY) != GameMessage::MSG_INVALID;
223 }
224 return FALSE;
225 }
226
227 if (outSelectionInfo->currentCountMineInfantry > 0 && outSelectionInfo->newCountGarrisonableBuildings == 1) {
228 return TRUE;
229 }
230
231 if (outSelectionInfo->newCountCivilians > 0) {
232 if (outSelectionInfo->newCountCivilians == 1 && selectionIsPoint) {
233 return TheGameClient->evaluateContextCommand(newCivilian, newCivilian->getPosition(), CommandTranslator::EVALUATE_ONLY) != GameMessage::MSG_INVALID;
234 }
235 return FALSE;
236 }
237
238 if (outSelectionInfo->newCountCrates > 0) {
239 return (outSelectionInfo->newCountCrates == 1 && selectionIsPoint);
240 }
241 }
242
243 if (outSelectionInfo->currentCountMine == 0) {
244 return FALSE;
245 }
246
247 return selectionIsPoint;
248}

251UnsignedInt getPickTypesForContext( Bool forceAttackMode )
252{
253 UnsignedInt types = PICK_TYPE_SELECTABLE;
254
255 if (forceAttackMode)
256 types |= PICK_TYPE_FORCEATTACKABLE;
257
258 //
259 // if we have a gui context command that allows for a shrubbery target then we want to
260 // pick that type too (generally shrubbery aren't pickable cause it would get in
261 // the way with movement and general selection)
262 //
263 const CommandButton *command = TheInGameUI->getGUICommand();
264
265 if (command != NULL) {
266 if (BitTest( command->getOptions(), ALLOW_MINE_TARGET)) {
267 types |= PICK_TYPE_MINES;
268 }
269
270 if (BitTest( command->getOptions(), ALLOW_SHRUBBERY_TARGET ) ) {
271 types |= PICK_TYPE_SHRUBBERY;
272 }
273 } else {
274 types |= getPickTypesForCurrentSelection(forceAttackMode);
275 }
276
277 return types;
278
279} // end getPickTypesForContext

282UnsignedInt getPickTypesForCurrentSelection( Bool forceAttackMode )
283{
284 UnsignedInt retVal = 0;
285 if (!TheInGameUI->areSelectedObjectsControllable()) {
286 return retVal;
287 }
288
289 const DrawableList *allSelectedDrawables = TheInGameUI->getAllSelectedDrawables();
290
291 for (DrawableListCIt cit = allSelectedDrawables->begin(); cit != allSelectedDrawables->end(); ++cit) {
292 Drawable *draw = *cit;
293 if (!draw) {
294 continue;
295 }
296
297 Object *obj = draw->getObject();
298 if (!obj) {
299 continue;
300 }
301
302// srj sez: thanks to new, area-effect disarming, we NO LONGER want to do this...
303// if (obj->hasWeaponToDealDamageType(DAMAGE_DISARM)) {
304// retVal |= PICK_TYPE_MINES;
305// }
306
307 if (obj->hasWeaponToDealDamageType(DAMAGE_FLAME) && forceAttackMode ) {
308 retVal |= PICK_TYPE_SHRUBBERY;
309 }
310
311 // For efficiency.
312 if (BitTest(retVal, PICK_TYPE_MINES | PICK_TYPE_SHRUBBERY)) {
313 break;
314 }
315 }
316
317 return retVal;
318
319}

342// Given a drawable, add it to an stl list specified by userData.
343// Useful for iterateDrawablesInRegion.
344Bool addDrawableToList( Drawable *draw, void *userData )
345{
346 PickDrawableStruct *pds = (PickDrawableStruct *) userData;
347#if defined(_DEBUG) || defined(_INTERNAL)
348 if (TheGlobalData->m_allowUnselectableSelection) {
349 pds->drawableListToFill->push_back(draw);
350 return TRUE;
351 }
352#endif
353
354 if (!pds->drawableListToFill)
355 return FALSE;
356
357 if (!draw->getTemplate()->isAnyKindOf(pds->kindofsToMatch))
358 return FALSE;
359
360 if (!draw->isSelectable())
361 {
362 const Object *obj = draw->getObject();
363 if ( obj && obj->getContainedBy() )//hmm, interesting... he is not selectable but he is contained
364 {// What we are after here is to propagate the selection the selection ti the container
365 // if the cobtainer is non-enclosing... see also selectionxlat, in the left_click case
366
367 ContainModuleInterface *contain = obj->getContainedBy()->getContain();
368 Drawable *containDraw = obj->getContainedBy()->getDrawable();
369 if (contain && ! contain->isEnclosingContainerFor( obj ) && containDraw )
370 return addDrawableToList( containDraw, userData );
371 }
372 else
373 return FALSE;
374 }
375
376 //Kris: Aug 9, 2003!!! Wow, this bug has been around a LONG time!!
377 //Basically, it was possible to drag select a single enemy/neutral unit even if you couldn't see it
378 //including stealthed units.
379 const Object *obj = draw->getObject();
380 if( obj )
381 {
382 const Player *player = ThePlayerList->getLocalPlayer();
383 Relationship rel = player->getRelationship( obj->getTeam() );
384 if( rel == NEUTRAL || rel == ENEMIES )
385 {
386 if( obj->getShroudedStatus( player->getPlayerIndex() ) >= OBJECTSHROUD_FOGGED )
387 {
388 return FALSE;
389 }
390
391 //If stealthed, no way!
392 if( obj->testStatus( OBJECT_STATUS_STEALTHED ) && !obj->testStatus( OBJECT_STATUS_DETECTED ) )
393 {
394 return FALSE;
395 }
396 }
397 }
398
399 pds->drawableListToFill->push_back(draw);
400 return TRUE;
