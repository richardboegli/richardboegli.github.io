28
29 Launcher application for games/apps using the chat API. This should be
30 run by the user and it will start the actual game executable. If a patch
31 file has been downloaded the patch will be applied before starting the game.
32
33 This does not download patches or do version checks, the game/app is responsible
34 for that. This just applies patches that are in the correct location for the
35 game. All patches should be in the "Patches" folder of the app.
36
37 The launcher should have a config file (launcher.cfg) so it knows which apps
38 should be checked for patches. The file should look like this:
39
40 # comment
41 # RUN = the game to launch
42 RUN = . notepad.exe # directory and app name
43 # RUN2 = the 2nd launcher if using one
44 RUN2 = . wordpad.exe
45 # FLAG = time in seconds of when to stop using 2nd launcher
46 FLAG = 996778113
47 #
48 # Sku's to check for patches
49 #
50 SKU1 = 1100 SOFTWARE\Westwood\WOnline # skus and registry keys
51 SKU2 = 1234 SOFTWARE\Westwood\FakeGame
52
53\*****************************************************************************/

75void RunGame(char *thePath, ConfigFile &config, Process &proc)
76{
77 char patchFile[MAX_PATH];
78 bool launchgame = true;
79
80 // MDC 8/23/2001 Wait 3 seconds for the installer to release the mutex
82
83 while (true)
84 {
85 int skuIndex;
86
87 while ((skuIndex = Find_Patch(patchFile, MAX_PATH, config)) != 0)
88 {
89 Apply_Patch(patchFile, config, skuIndex);
90 launchgame = true;
91 }
92
93 // launch the game if first pass, or found a patch
94 if (launchgame)
95 {
96 // don't relaunch unless we find a patch (or checking for patches)
97 launchgame = false;
98 myChdir(thePath);
99
100#ifndef COPY_PROTECT
101 Create_Process(proc);
102#else // COPY_PROTECT
103
104 InitializeProtect();
105 Create_Process(proc);
106 SendProtectMessage(proc.hProcess, proc.dwThreadID);
107
108#endif // COPY_PROTECT
109
110 DWORD exit_code;
111 Wait_Process(proc, &exit_code);
112
113 if (exit_code == UPDATE_RETVAL)
114 {
115 // They just want to check for patches
116 launchgame = true;
117
118 // Start patchgrabber
119 Process patchgrab;
120 strcpy(patchgrab.directory,proc.directory); // same dir as game
121 strcpy(patchgrab.command,"patchget.dat"); // the program that grabs game patches
122 strcpy(patchgrab.args,"");
123 Create_Process(patchgrab);
124 Wait_Process(patchgrab); // wait for completion
125 }
126
127#ifdef COPY_PROTECT
128 ShutdownProtect();
129#endif
130 }
131 else
132 {
133 Delete_Patches(config); // delete all patches
134 break;
135 }
136 }
137}

146//
147// Called by WinMain
148//
149int main(int argc, char *argv[])
150{
151 char patchFile[MAX_PATH];
152
153 char cwd[MAX_PATH]; // save current directory before game start
154 _getcwd(cwd, MAX_PATH);
155
156 // Goto the folder where launcher is installed
157 myChdir(argv[0]);
158
159 // extract the program name from argv[0]. Change the extension to
160 // .lcf (Launcher ConFig). This is the name of our config file.
161 char configName[MAX_PATH + 3];
162 strcpy(configName, argv[0]);
163 char* extension = configName;
164 char* tempptr;
165
166 while ((tempptr = strchr(extension + 1, '.')))
167 {
168 extension = tempptr;
169 }
170
171 if (*extension == '.')
172 {
173 *extension = 0;
174 }
175
176#ifdef DEBUG
178 char debugFile[MAX_PATH + 3];
179 strcpy(debugFile, configName);
180 strcat(debugFile, ".txt");
181 strcpy(debugLogName, strrchr(configName, '\\'));
182 strcat(debugLogName, "Log");
183 FileD outputDevice(debugFile, true);
184 MsgManager::setAllStreams(&outputDevice);
185 DBGMSG("Launcher initialized");
186#endif
187
188 CreatePrimaryWin("generals");
189
190 InitCommonControls();
191
192 strcat(configName, ".lcf");
193
194 DBGMSG("Config Name: "<<configName);
195
196 ConfigFile config;
197 FILE* in = fopen(configName, "r");
198
199 if (in == NULL)
200 {
201 MessageBox(NULL,"You must run the game from its install directory.",
202 "Launcher config file missing",MB_OK);
203 exit(-1);
204 }
205
206 bit8 ok = config.readFile(in);
207 fclose(in);
208
209 if (ok == FALSE)
210 {
211 MessageBox(NULL,"File 'launcher.cfg' is corrupt","Error",MB_OK);
212 exit(-1);
213 }
214
215 // Load process info
216 Process proc;
217 Read_Process_Info(config,proc);
218
219 // Possible 2nd EXE
220 Process proc2;
221 int hasSecondEXE = Read_Process_Info(config,proc2,"RUN2");
222
223 DBGMSG("Read process info");
224
225 for (int i = 1; i < argc; i++)
226 {
227 strcat(proc.args, " ");
228 strcat(proc.args, argv[i]);
229
230 if (hasSecondEXE)
231 {
232 strcat(proc2.args, " ");
233 strcat(proc2.args, argv[i]);
234 }
235 }
236
237 DBGMSG("ARGS: "<<proc.args);
238
239 // Just spawn the patchgrabber & apply any patches it downloads
240 if ((argc >= 2) && (strcmp(argv[1], "GrabPatches") == 0))
241 {
242 // Start patchgrabber
243 Process patchgrab;
244
245 // same dir as game
246 strcpy(patchgrab.directory, proc.directory);
247
248 // the program that grabs game patches
249 strcpy(patchgrab.command, "patchget.dat");
250 strcpy(patchgrab.args, "");
251
252 Create_Process(patchgrab);
253 Wait_Process(patchgrab);
254
255 // Apply any patches I find
256 int skuIndex;
257
258 while ((skuIndex = Find_Patch(patchFile, MAX_PATH, config)) != 0)
259 {
260 Apply_Patch(patchFile, config, skuIndex);
261 }
262
263 myChdir(cwd);
264 return 0;
265 }
266
267 // Look for patch file(s) to apply
268 bool launchgame = true;
269
270 time_t cutoffTime = 0;
271 if (hasSecondEXE)
272 {
273 Wstring timeStr;
274 if (config.getString("FLAG", timeStr)!=FALSE)
275 {
276 cutoffTime = atoi(timeStr.get());
277 }
278
279 if (cutoffTime == 0)
280 {
281 // We didn't have the FLAG parameter; somebody's been hacking. No game for you! Bad hacker!
282 DBGMSG("Saw cutoffTime of 0; real time is " << time(NULL));
283 MessageBox(NULL,"File 'launcher.cfg' is corrupt","Error",MB_OK);
284 exit(-1);
285 }
286
287 if (time(NULL) > cutoffTime)
288 {
289 // The future is now! Just run the game.
290 RunGame(argv[0], config, proc);
291 }
292 else
293 {
294 // Its still early in the product's lifetime, so run the 2nd (SafeDisk'd) launcher.
295 // We don't have to wait around since it'll do the entire talk to game, look for patches,
296 // etc. deal.
297 RunLauncher(argv[0], proc2);
298 }
299 }
300 else
301 {
302 // We're the second (or only) launcher, so act normally
303 RunGame(argv[0], config, proc);
304 }
305
306 myChdir(cwd);
307
308 // Exit normally
309 return 0;
310}

316void CreatePrimaryWin(char *prefix)
317{
318 char name[256];
319 sprintf(name, "launcher_%s", prefix);
320
321 DBGMSG("CreatePrimary: "<<name);
322
323 /*
324 ** set up and register window class
325 */
326 WNDCLASS wc;
327 wc.style = CS_HREDRAW | CS_VREDRAW;
328 wc.lpfnWndProc = DefWindowProc;
329 wc.cbClsExtra = 0; // Don't need any extra class data
330 wc.cbWndExtra = 0; // No extra win data
331 wc.hInstance = Global_instance;
332 wc.hIcon=LoadIcon(Global_instance, MAKEINTRESOURCE(IDI_GENERALS));
333 wc.hCursor = NULL;
334 wc.hbrBackground = NULL;
335 wc.lpszMenuName = name;
336 wc.lpszClassName = name;
337 RegisterClass(&wc);
338
339 /*
340 ** create a window
341 */
342 HWND hwnd = CreateWindowEx(WS_EX_TOPMOST, name, name, WS_POPUP, 0, 0,
343 GetSystemMetrics(SM_CXSCREEN), GetSystemMetrics(SM_CYSCREEN),
344 NULL, NULL, Global_instance, NULL);
345
346 if(!hwnd)
347 {
348 DBGMSG("Couldn't make window!");
349 }
350 else
351 {
352 DBGMSG("Window created!");
353 }
354}

369void myChdir(char *path)
370{
371 char drive[10];
372 char dir[255];
373 char file[255];
374 char ext[64];
375 char filepath[513];
376 int abc;
377
378 _splitpath( path, drive, dir, file, ext );
379 _makepath ( filepath, drive, dir, NULL, NULL );
380
381 if ( filepath[ strlen( filepath ) - 1 ] == '\\' )
382 {
383 filepath[ strlen( filepath ) - 1 ] = '\0';
384 }
385 abc = (unsigned)( toupper( filepath[0] ) - 'A' + 1 );
386 if ( !_chdrive( abc ))
387 {
388 abc = chdir( filepath ); // Will fail with ending '\\'
389 }
390 // should be in proper folder now....
391}
