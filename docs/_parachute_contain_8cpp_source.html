150void ParachuteContain::calcSwayMtx(const Coord3D* offset, Matrix3D* mtx)
151{
152 mtx->Make_Identity();
153 mtx->Translate(offset->x, offset->y, offset->z);
154 mtx->In_Place_Pre_Rotate_X(m_roll);
155 mtx->In_Place_Pre_Rotate_Y(m_pitch);
156 mtx->Translate(-offset->x, -offset->y, -offset->z);
157}

160void ParachuteContain::updateBonePositions()
161{
162 if (m_needToUpdateParaBones)
163 {
164 m_needToUpdateParaBones = false; // yeah, even if not found.
165
166 Drawable* parachuteDraw = getObject()->getDrawable();
167 if (parachuteDraw)
168 {
169 if (parachuteDraw->getPristineBonePositions( "PARA_COG", 0, &m_paraSwayBone, NULL, 1) != 1)
170 {
171 DEBUG_CRASH(("PARA_COG not found\n"));
172 m_paraSwayBone.zero();
173 }
174
175 if (parachuteDraw->getPristineBonePositions( "PARA_ATTCH", 0, &m_paraAttachBone, NULL, 1 ) != 1)
176 {
177 DEBUG_CRASH(("PARA_ATTCH not found\n"));
178 m_paraAttachBone.zero();
179 }
180 }
181 //DEBUG_LOG(("updating para bone positions %d...\n",TheGameLogic->getFrame()));
182 }
183
184 if (m_needToUpdateRiderBones)
185 {
186
187 m_needToUpdateRiderBones = false; // yeah, even if not found.
188
189 Object* rider = (getContainCount() > 0) ? getContainList().front() : NULL;
190 Drawable* riderDraw = rider ? rider->getDrawable() : NULL;
191 if (riderDraw)
192 {
193 if (riderDraw->getPristineBonePositions( "PARA_MAN", 0, &m_riderAttachBone, NULL, 1) != 1)
194 {
195 //DEBUG_LOG(("*** No parachute-attach bone... using object height!\n"));
196 m_riderAttachBone.zero();
197 m_riderAttachBone.z += riderDraw->getDrawableGeometryInfo().getMaxHeightAbovePosition();
198 }
199 }
200
201 //DEBUG_LOG(("updating rider bone positions %d...\n",TheGameLogic->getFrame()));
202 }

206void ParachuteContain::updateOffsetsFromBones()
207{
208 const Coord3D* objPos = getObject()->getPosition();
209
210 getObject()->convertBonePosToWorldPos(&m_paraSwayBone, NULL, &m_paraSwayOffset, NULL);
211 m_paraSwayOffset.x -= objPos->x;
212 m_paraSwayOffset.y -= objPos->y;
213 m_paraSwayOffset.z -= objPos->z;
214
215 getObject()->convertBonePosToWorldPos(&m_paraAttachBone, NULL, &m_paraAttachOffset, NULL);
216 m_paraAttachOffset.x -= objPos->x;
217 m_paraAttachOffset.y -= objPos->y;
218 m_paraAttachOffset.z -= objPos->z;
219
220 Object* rider = (getContainCount() > 0) ? getContainList().front() : NULL;
221 if (rider)
222 {
223 const Coord3D* riderPos = rider->getPosition();
224
225 rider->convertBonePosToWorldPos(&m_riderAttachBone, NULL, &m_riderAttachOffset, NULL);
226 m_riderAttachOffset.x -= riderPos->x;
227 m_riderAttachOffset.y -= riderPos->y;
228 m_riderAttachOffset.z -= riderPos->z;
229
230 m_riderAttachOffset.x = m_paraAttachOffset.x - m_riderAttachOffset.x;
231 m_riderAttachOffset.y = m_paraAttachOffset.y - m_riderAttachOffset.y;
232 m_riderAttachOffset.z = m_paraAttachOffset.z - m_riderAttachOffset.z;
233
234 m_riderSwayOffset.x = m_paraSwayOffset.x - m_riderAttachOffset.x;
235 m_riderSwayOffset.y = m_paraSwayOffset.y - m_riderAttachOffset.y;
236 m_riderSwayOffset.z = m_paraSwayOffset.z - m_riderAttachOffset.z;
237 }
238}

246Bool ParachuteContain::isValidContainerFor(const Object* rider, Bool checkCapacity) const
247{
248 if (!rider)
249 return false;
250
251 // extend functionality
252 if( OpenContain::isValidContainerFor( rider, checkCapacity ) == false )
253 return false;
254
255 Int transportSlotCount = rider->getTransportSlotCount();
256
257 // if 0, this object isn't transportable.
258 // (exception: infantry are always transportable by parachutes, regardless
259 // of this.... this allows us to paradrop pilots, but not transport them
260 // by other means)
261 if (transportSlotCount == 0 && !rider->isKindOf(KINDOF_INFANTRY) && !rider->isKindOf(KINDOF_PARACHUTABLE))
262 return false;
263
264 // we can only "hold" one item at a time.
265 if (getContainCount() > 0)
266 return false;
267
268 return true;
269}

285UpdateSleepTime ParachuteContain::update( void )
286{
287 OpenContain::update();
288
289 Object* parachute = getObject();
290 if( parachute->isDisabledByType( DISABLED_HELD ) )
291 {
292 return UPDATE_SLEEP_NONE; // my, that was easy
293 }
294
295 AIUpdateInterface *parachuteAI = parachute->getAI();
296
297 Drawable* draw = parachute->getDrawable();
298 const ParachuteContainModuleData* d = getParachuteContainModuleData();
299 Object* rider = (getContainCount() > 0) ? getContainList().front() : NULL;
300
301 if (m_startZ == NO_START_Z) {
302 m_startZ = parachute->getPosition()->z;
303 Real groundHeight = TheTerrainLogic->getGroundHeight(parachute->getPosition()->x, parachute->getPosition()->y);
304 if (m_startZ-groundHeight < 2*d->m_paraOpenDist) {
305 // Oh dear - we ejected too close to the ground, and there isn't enough
306 // room to open the chute. Well, since it's only a game, we'll fudge
307 // a little so that the pilot doesn't slam into the ground & stick.
308 m_startZ = groundHeight+2*d->m_paraOpenDist;
309 }
310 }
311
312 if (!m_opened)
313 {
314 // see if we need to open.
315 if (fabs(m_startZ - parachute->getPosition()->z) >= d->m_paraOpenDist)
316 {
317 m_opened = true;
318 parachute->clearAndSetModelConditionState(MODELCONDITION_FREEFALL, MODELCONDITION_PARACHUTING);
319 m_needToUpdateParaBones = true;
320 if (rider)
321 {
322 rider->clearAndSetModelConditionState(MODELCONDITION_FREEFALL, MODELCONDITION_PARACHUTING);
323 m_needToUpdateRiderBones = true;
324
325 AudioEventRTS soundToPlay = d->m_parachuteOpenSound;
326 soundToPlay.setObjectID( rider->getID() );
327 TheAudio->addAudioEvent( &soundToPlay );
328 }
329
330
331 // When a parachute opens, it should look for a good place to land. This could be explicitly set
332 // by a DeliverPayload, otherwise any place clear is good.
333 if( parachuteAI )
334 {
335 Coord3D target = *parachute->getPosition();
336 if( m_isLandingOverrideSet )
337 {
338 target = m_landingOverride;
339 if( parachuteAI->getCurLocomotor() )
340 parachuteAI->getCurLocomotor()->setUltraAccurate( TRUE );
341 }
342 else
343 {
344 FindPositionOptions fpOptions;
345 fpOptions.minRadius = 0.0f;
346 fpOptions.maxRadius = 100.0f;
347 fpOptions.relationshipObject = NULL;
348 fpOptions.flags = FPF_NONE;
349 ThePartitionManager->findPositionAround( &target, &fpOptions, &target );
350 }
351 parachuteAI->aiMoveToPosition( &target, CMD_FROM_AI );
352 }
353 }
354 else if ( rider )
355 rider->clearAndSetModelConditionState( MODELCONDITION_PARACHUTING, MODELCONDITION_FREEFALL );
356
357 }
358 draw->setDrawableHidden(!m_opened);
359
360 if (!m_opened || getContainCount() == 0)
361 {
362 // unopened, or empty, chutes, don't collide with anything, to simplify
363 // ejections, paradrops, landings, etc...
364 parachute->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
365 if (rider)
366 rider->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
367 }
368 else
369 {
370 // opened/nonempty chutes DO collide...
371 parachute->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
372 if (rider)
373 rider->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
374 }
375
376 AIUpdateInterface* ai = parachute->getAIUpdateInterface();
377 if (ai && !parachute->isEffectivelyDead())
378 {
379 ai->chooseLocomotorSet(m_opened ? LOCOMOTORSET_NORMAL : LOCOMOTORSET_FREEFALL);
380
381 Locomotor* locomotor = ai->getCurLocomotor();
382 if (locomotor)
383 {
384 // damp the swaying a bunch when we get close, so that things land vertically (or nearly so)
385 Real altitudeDamping = 0;
386 if (getContainCount() > 0)
387 {
388 Object* rider = getContainList().front();
389 const Real ALTITUDE_DAMP_START = 20.0f;
390 if (rider->getHeightAboveTerrain() <= ALTITUDE_DAMP_START)
391 altitudeDamping = d->m_lowAltitudeDamping;
392 }
393
394 if (m_opened)
395 {
396 const Real PITCH_STIFFNESS = locomotor->getPitchStiffness();
397 const Real ROLL_STIFFNESS = locomotor->getRollStiffness();
398 const Real PITCH_DAMPING = locomotor->getPitchDamping() + altitudeDamping;
399 const Real ROLL_DAMPING = locomotor->getRollDamping() + altitudeDamping;
400 m_pitchRate += ((-PITCH_STIFFNESS * m_pitch) + (-PITCH_DAMPING * m_pitchRate)); // spring/damper
401 m_rollRate += ((-ROLL_STIFFNESS * m_roll) + (-ROLL_DAMPING * m_rollRate)); // spring/damper
402
403 m_pitch += m_pitchRate;
404 m_roll += m_rollRate;
405
406 if( m_isLandingOverrideSet )
407 {
408 // Need to wait until after opening to do this, or else we are sending the message to the Freefall locomotor.
409 locomotor->setCloseEnoughDist( 10.0 );
410 locomotor->setCloseEnoughDist3D( FALSE );
411 }
412 }
413
414 if (draw)
415 {
416 updateBonePositions();
417 updateOffsetsFromBones();
418
419 Matrix3D tmp;
420 calcSwayMtx(&m_paraSwayOffset, &tmp);
421 draw->setInstanceMatrix(&tmp);
422 }
423
424 positionContainedObjectsRelativeToContainer();
425 }
426 }
427
428 // allow us to land on bridges!
429 const Coord3D* paraPos = getObject()->getPosition();
430 PathfindLayerEnum newLayer = TheTerrainLogic->getHighestLayerForDestination(paraPos);
431 getObject()->setLayer(newLayer);
432 if (rider)
433 rider->setLayer(newLayer);
434
435 // If we have lost our passenger for whatever reason, die early. Otherwise we just sit around forever.
436 if( getContainCount() == 0 )
437 getObject()->kill();
438
439 // the collide system doesn't always collide us with the ground if we fall into water.
440 // so force the issue.
441 Real waterZ;
442 if (!getObject()->isEffectivelyDead()
443 && getObject()->getLayer() == LAYER_GROUND
444 && TheTerrainLogic->isUnderwater(paraPos->x, paraPos->y, &waterZ)
445 && (paraPos->z - waterZ) < d->m_killWhenLandingInWaterSlop)
446 {
447 getObject()->kill();
448 }
449
450
451 return UPDATE_SLEEP_NONE;
452}

472//-------------------------------------------------------------------------------------------------
473void ParachuteContain::onRemoving( Object *rider )
474{
475 OpenContain::onRemoving(rider);
476
477 const ParachuteContainModuleData* d = getParachuteContainModuleData();
478
479 // object is no longer held inside a transport
480 rider->clearDisabled( DISABLED_HELD );
481 rider->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_PARACHUTING ) );
482
483 // mark parachute as "no-collisions"... it is just ephemeral at this point,
484 // and having the chute collide with the soldier (and both bounce apart) is
485 // just dumb-lookin'...
486 getObject()->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
487
488 // position him correctly.
489 positionRider(rider);
490
491 rider->clearModelConditionFlags(MAKE_MODELCONDITION_MASK2(MODELCONDITION_FREEFALL, MODELCONDITION_PARACHUTING));
492 m_needToUpdateRiderBones = true;
493
494 // temporarily mark the guy as being allowed to fall
495 // (overriding his locomotor's stick-to-ground attribute).
496 // this will be reset (by PhysicsBehavior) when he touches the ground.
497 PhysicsBehavior* physics = rider->getPhysics();
498 if (physics)
499 {
500 physics->setAllowToFall(true);
501
502 Coord3D force;
503 force.zero();
504 physics->applyForce(&force); // force its physics to wake up... should be done when DISABLED_HELD is cleared, but it not, and scared to do it now.
505 }
506
507
508 AIUpdateInterface* riderAI = rider->getAIUpdateInterface();
509 if (riderAI)
510 {
511 Player* controller = rider->getControllingPlayer();
512 if (controller && controller->isSkirmishAIPlayer())
513 {
514 riderAI->aiHunt(CMD_FROM_AI); // hunt, as per Dustin's request.
515 }
516 else
517 {
518 bool hasRallyPoint = false;
519 // Get the transport of the rider
520 Object *transport = TheGameLogic->findObjectByID(rider->getProducerID());
521 if (transport)
522 {
523 // Get the building that produced the transport
524 Object *transportProducer = TheGameLogic->findObjectByID(transport->getProducerID());
525 if (transportProducer)
526 {
527 // See if we need to set a rally point for the object being parachuted
528 ExitInterface* exitInterface = transportProducer->getObjectExitInterface();
529 if (exitInterface && exitInterface->useSpawnRallyPoint())
530 {
531 exitInterface->exitObjectViaDoor(rider, DOOR_1);
532 hasRallyPoint = true;
533 }
534 }
535 }
536
537 if (!hasRallyPoint)
538 riderAI->aiIdle(CMD_FROM_AI); // become idle.
539 }
540 }
541
542 // if we land in the water, we die. alas.
543 const Coord3D* riderPos = rider->getPosition();
544 Real waterZ, terrainZ;
545 if (TheTerrainLogic->isUnderwater(riderPos->x, riderPos->y, &waterZ, &terrainZ)
546 && riderPos->z <= waterZ + d->m_killWhenLandingInWaterSlop
547 && rider->getLayer() == LAYER_GROUND)
548 {
549 // don't call kill(); do it manually, so we can specify DEATH_FLOODED
550 DamageInfo damageInfo;
551 damageInfo.in.m_damageType = DAMAGE_WATER; // use this instead of UNRESISTABLE so we don't get a dusty damage effect
552 damageInfo.in.m_deathType = DEATH_FLOODED;
553 damageInfo.in.m_sourceID = INVALID_ID;
554 damageInfo.in.m_amount = HUGE_DAMAGE_AMOUNT;
555 rider->attemptDamage( &damageInfo );
556 }
557
558 // Kill if we landed on impassable ground
559 Int cellX = REAL_TO_INT( rider->getPosition()->x / PATHFIND_CELL_SIZE );
560 Int cellY = REAL_TO_INT( rider->getPosition()->y / PATHFIND_CELL_SIZE );
561
562 PathfindCell* cell = TheAI->pathfinder()->getCell( rider->getLayer(), cellX, cellY );
563 PathfindCell::CellType cellType = cell ? cell->getType() : PathfindCell::CELL_IMPASSABLE;
564
565 // If we land outside the map from a faulty parachute, we die too.
566 // Otherwise we exist outside the PartitionManger like a cheater.
567 if( rider->isOffMap()
568 || (cellType == PathfindCell::CELL_CLIFF)
569 || (cellType == PathfindCell::CELL_WATER)
570 || (cellType == PathfindCell::CELL_IMPASSABLE) )
571 {
572 // The Paradrop command was legal, the parachute destination was legal, but the parachute
573 // can still fail to adjust back on the map. SO this is the place to cap the cheater.
574 rider->kill();
575 }
576
577 // Note: for future enhancement of this feature, we should test the object against the cell type he is on,
578 // using obj->getAI()->hasLocomotorForSurface( __ ). We cshould not assume here that the parachutist can not
579 // find happiness on cliffs or water or whatever.
580
581
582}

585void ParachuteContain::positionRider(Object* rider)
586{
587 updateBonePositions();
588 updateOffsetsFromBones();
589
590 Coord3D pos = *getObject()->getPosition();
592 pos.x += m_riderAttachOffset.x;
593 pos.y += m_riderAttachOffset.y;
594 pos.z += m_riderAttachOffset.z;
595 //DUMPCOORD3D(&pos);
596 rider->setPosition(&pos);
597
598 Real alt = rider->getHeightAboveTerrain();
599 if (alt < 0.0f)
600 {
601 // don't let him go below ground.
602 pos.z -= alt;
603 rider->setPosition(&pos);
604 }
605
606 rider->setOrientation(getObject()->getOrientation());
607
608 Drawable* draw = rider->getDrawable();
609 if (draw)
610 {
611 if( rider->isDisabledByType( DISABLED_HELD ) )
612 {
613 Matrix3D tmp;
614 calcSwayMtx(&m_riderSwayOffset, &tmp);
615 draw->setInstanceMatrix(&tmp);
616 }
617 else
618 {
619 draw->setInstanceMatrix(NULL);
620 }
621 }
622}

641//-------------------------------------------------------------------------------------------------
642void ParachuteContain::onDie( const DamageInfo * damageInfo )
643{
644 // if we are airborne when killed, the guy falls screaming to his death...
645 if (getObject()->isSignificantlyAboveTerrain())
646 {
647 Object* rider = (getContainCount() > 0) ? getContainList().front() : NULL;
648 if (rider)
649 {
650 removeAllContained();
651 const ParachuteContainModuleData* d = getParachuteContainModuleData();
652 if (d->m_freeFallDamagePercent > 0.0f)
653 {
654 // do some damage just for losing your parachute.
655 // not very realistic, but practical to help ensure that
656 // you really do die from going "splat" on the ground.
657 DamageInfo extraDamageInfo;
658 extraDamageInfo.in.m_damageType = DAMAGE_FALLING;
659 extraDamageInfo.in.m_deathType = DEATH_SPLATTED;
660 extraDamageInfo.in.m_sourceID = damageInfo->in.m_sourceID;
661 extraDamageInfo.in.m_amount = rider->getBodyModule()->getMaxHealth() * d->m_freeFallDamagePercent;
662 rider->attemptDamage(&extraDamageInfo);
663 }
664 PhysicsBehavior* physics = rider->getPhysics();
665 if (physics)
666 {
667 physics->setAllowToFall(true);
