196Bool FileSystem::doesFileExist(const Char *filename) const
197{
198 USE_PERF_TIMER(FileSystem)
199
200 unsigned key=TheNameKeyGenerator->nameToLowercaseKey(filename);
201 std::map<unsigned,bool>::iterator i=m_fileExist.find(key);
202 if (i!=m_fileExist.end())
203 return i->second;
204
205 if (TheLocalFileSystem->doesFileExist(filename))
206 {
207 m_fileExist[key]=true;
208 return TRUE;
209 }
210 if (TheArchiveFileSystem->doesFileExist(filename))
211 {
212 m_fileExist[key]=true;
213 return TRUE;
214 }
215
216 m_fileExist[key]=false;
217 return FALSE;
218}

264//============================================================================
265// FileSystem::areMusicFilesOnCD
266//============================================================================
267Bool FileSystem::areMusicFilesOnCD()
268{
269 if (!TheCDManager) {
270 DEBUG_LOG(("FileSystem::areMusicFilesOnCD() - No CD Manager; returning false\n"));
271 return FALSE;
272 }
273
274 AsciiString cdRoot;
275 Int dc = TheCDManager->driveCount();
276 for (Int i = 0; i < dc; ++i) {
277 DEBUG_LOG(("FileSystem::areMusicFilesOnCD() - checking drive %d\n", i));
278 CDDriveInterface *cdi = TheCDManager->getDrive(i);
279 if (!cdi) {
280 continue;
281 }
282
283 cdRoot = cdi->getPath();
284 if (!cdRoot.endsWith("\\"))
285 cdRoot.concat("\\");
286 cdRoot.concat("genseczh.big");
287 DEBUG_LOG(("FileSystem::areMusicFilesOnCD() - checking for %s\n", cdRoot.str()));
288 File *musicBig = TheLocalFileSystem->openFile(cdRoot.str());
289 if (musicBig)
290 {
291 DEBUG_LOG(("FileSystem::areMusicFilesOnCD() - found it!\n"));
292 musicBig->close();
293 return TRUE;
294 }
295 }
296 return FALSE;
297}

301void FileSystem::loadMusicFilesFromCD()
302{
303 if (!TheCDManager) {
304 return;
305 }
306
307 AsciiString cdRoot;
308 Int dc = TheCDManager->driveCount();
309 for (Int i = 0; i < dc; ++i) {
310 CDDriveInterface *cdi = TheCDManager->getDrive(i);
311 if (!cdi) {
312 continue;
313 }
314
315 cdRoot = cdi->getPath();
316 if (TheArchiveFileSystem->loadBigFilesFromDirectory(cdRoot, MUSIC_BIG)) {
317 break;
318 }
319 }
