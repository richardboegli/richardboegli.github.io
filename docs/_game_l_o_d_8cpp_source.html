184{
185 const char *c;
186 AsciiString name;
187
188 // read the name
189 c = ini->getNextToken();
190 name.set( c ); //name of detail level - low, medium, high
191
192 if( TheGameLODManager )
193 {
194 StaticGameLODLevel index = (StaticGameLODLevel)TheGameLODManager->getStaticGameLODIndex(name);

203 INI::parseIndexList(ini,NULL,&preset->m_videoType,VideoNames);
204 INI::parseInt(ini,NULL,&preset->m_memory,NULL);
205 }
206 }

268void GameLODManager::init(void)
269{
270 INI ini;
271 //Get Presets for each LOD level.
272 ini.load( AsciiString( "Data\\INI\\GameLOD.ini" ), INI_LOAD_OVERWRITE, NULL );
273
274 //Get presets for each known hardware configuration
275 ini.load( AsciiString( "Data\\INI\\GameLODPresets.ini"), INI_LOAD_OVERWRITE, NULL);
276
277 //Get Presets for custom LOD level by pulling them out of initial globaldata (which should
278 //have all settings already applied).
279 refreshCustomStaticLODLevel();
280
281 //Override with user preferences
282 OptionPreferences optionPref;
283
284 StaticGameLODLevel userSetDetail=(StaticGameLODLevel)optionPref.getStaticGameDetail();
285
286 m_idealDetailLevel=(StaticGameLODLevel)optionPref.getIdealStaticGameDetail();
287
288 //always get this data in case we need it later.
289 testMinimumRequirements(NULL,&m_cpuType,&m_cpuFreq,&m_numRAM,NULL,NULL,NULL);
290
291 if ((Real)(m_numRAM)/(Real)(256*1024*1024) >= PROFILE_ERROR_LIMIT)
292 m_memPassed=TRUE; //check if they have at least 256 MB
293
294 if (m_idealDetailLevel == STATIC_GAME_LOD_UNKNOWN || TheGlobalData->m_forceBenchmark)
295 {
296 if (m_cpuType == XX || TheGlobalData->m_forceBenchmark)
297 {
298 //need to run the benchmark
299 testMinimumRequirements(NULL,NULL,NULL,NULL,&m_intBenchIndex,&m_floatBenchIndex,&m_memBenchIndex);
300
301 if (TheGlobalData->m_forceBenchmark)
302 { //we want to see the numbers. So dump them to a logfile.
303 FILE *fp=fopen("Benchmark.txt","w");
304 if (fp)
305 {
306 fprintf(fp,"BenchProfile = %s %d %f %f %f", CPUNames[m_cpuType], m_cpuFreq, m_intBenchIndex, m_floatBenchIndex, m_memBenchIndex);
307 fclose(fp);
308 }
309 }
310
311 m_compositeBenchIndex = m_intBenchIndex + m_floatBenchIndex;
312
313 StaticGameLODLevel currentLevel=STATIC_GAME_LOD_LOW;
314 BenchProfile *prof=m_benchProfiles;
315 m_cpuType = P3; //assume lowest spec.
316 m_cpuFreq = 1000; //assume lowest spec.
317 for (Int k=0; k<m_numBenchProfiles; k++)
318 {
319 //Check if we're within 5% of the performance of this cpu profile.
320 if (m_intBenchIndex/prof->m_intBenchIndex >= PROFILE_ERROR_LIMIT && m_floatBenchIndex/prof->m_floatBenchIndex >= PROFILE_ERROR_LIMIT && m_memBenchIndex/prof->m_memBenchIndex >= PROFILE_ERROR_LIMIT)
321 {
322 for (Int i=STATIC_GAME_LOD_HIGH; i >= STATIC_GAME_LOD_LOW; i--)
323 {
324 LODPresetInfo *preset=&m_lodPresets[i][0]; //pointer to first preset at this LOD level.
325 for (Int j=0; j<m_numLevelPresets[i]; j++)
326 {
327 if( prof->m_cpuType == preset->m_cpuType && ((Real)prof->m_mhz/(Real)preset->m_mhz >= PROFILE_ERROR_LIMIT))
328 { currentLevel = (StaticGameLODLevel)i;
329 m_cpuType = prof->m_cpuType;
330 m_cpuFreq = prof->m_mhz;
331 break;
332 }
333 preset++; //skip to next preset
334 }
335 if (currentLevel >= i)
336 break; //we already found a higher level than the remaining presets so no need to keep searching.
337 }
338 }
339 prof++;
340 }
341 } //finding equivalent CPU to unkown cpu.
342 } //find data needed to determine m_idealDetailLevel
343
344 if (userSetDetail == STATIC_GAME_LOD_CUSTOM)
345 {
346 TheWritableGlobalData->m_textureReductionFactor = optionPref.getTextureReduction();
347 TheWritableGlobalData->m_useShadowVolumes = optionPref.get3DShadowsEnabled();
348 TheWritableGlobalData->m_useShadowDecals = optionPref.get2DShadowsEnabled();
349 TheWritableGlobalData->m_enableBehindBuildingMarkers = optionPref.getBuildingOcclusionEnabled();
350 TheWritableGlobalData->m_maxParticleCount = optionPref.getParticleCap();
351 TheWritableGlobalData->m_enableDynamicLOD = optionPref.getDynamicLODEnabled();
352 TheWritableGlobalData->m_useFpsLimit = optionPref.getFPSLimitEnabled();
353 TheWritableGlobalData->m_useLightMap = optionPref.getLightmapEnabled();
354 TheWritableGlobalData->m_useCloudMap = optionPref.getCloudShadowsEnabled();
355 TheWritableGlobalData->m_showSoftWaterEdge = optionPref.getSmoothWaterEnabled();
356 TheWritableGlobalData->m_useHeatEffects = optionPref.getUseHeatEffects();
357 TheWritableGlobalData->m_useDrawModuleLOD = optionPref.getExtraAnimationsDisabled();
358 TheWritableGlobalData->m_useTreeSway = !TheWritableGlobalData->m_useDrawModuleLOD; //borrow same setting.
359 TheWritableGlobalData->m_useTrees = optionPref.getTreesEnabled();
360 }
361
362 setStaticLODLevel(userSetDetail);
363}

445StaticGameLODLevel GameLODManager::findStaticLODLevel(void)
446{
447 //Check if we have never done the test on current system
448 if (m_idealDetailLevel == STATIC_GAME_LOD_UNKNOWN)
449 {
450 //search all our presets for matching hardware
451 m_idealDetailLevel = STATIC_GAME_LOD_LOW;
452
453 //get system configuration - only need vide chip type, got rest in ::init().
454 testMinimumRequirements(&m_videoChipType,NULL,NULL,NULL,NULL,NULL,NULL);
455 if (m_videoChipType == DC_UNKNOWN)
456 m_videoChipType = DC_TNT2; //presume it's at least TNT2 level
457
458 Int numMBRam=m_numRAM/(1024*1024);
459
460 for (Int i=STATIC_GAME_LOD_HIGH; i >= STATIC_GAME_LOD_LOW; i--)
461 {
462 LODPresetInfo *preset=&m_lodPresets[i][0]; //pointer to first preset at this LOD level.
463 for (Int j=0; j<m_numLevelPresets[i]; j++)
464 {
465
466 if( m_cpuType == preset->m_cpuType &&
467 ((Real)m_cpuFreq/(Real)preset->m_mhz >= PROFILE_ERROR_LIMIT) &&//make sure we're within 5% or higher
468 m_videoChipType >= preset->m_videoType &&
469 ((Real)numMBRam/(Real)preset->m_memory >= PROFILE_ERROR_LIMIT)
470 )
471 { m_idealDetailLevel = (StaticGameLODLevel)i;
472 break;
473 }
474
475 preset++; //skip to next preset
476
477 }
478 if (m_idealDetailLevel >= i)
479 break; //we already found a higher level than the remaining presets so no need to keep searching.
480 }
481 //Save ideal detail level for future usage
482 OptionPreferences optionPref;
483 optionPref["IdealStaticGameLOD"] = getStaticGameLODLevelName(m_idealDetailLevel);
484 if (getStaticLODLevel() == STATIC_GAME_LOD_UNKNOWN) //save for future usage.
485 optionPref["StaticGameLOD"] = getStaticGameLODLevelName(m_idealDetailLevel);
486 optionPref.write();
487 }
488
489 return m_idealDetailLevel;
490}

493Bool GameLODManager::setStaticLODLevel(StaticGameLODLevel level)
494{
495 if (!TheGlobalData->m_enableStaticLOD)
496 { m_currentStaticLOD = STATIC_GAME_LOD_CUSTOM;
497 return FALSE;
498 }
499
500 if (level == STATIC_GAME_LOD_UNKNOWN || (level != STATIC_GAME_LOD_CUSTOM && m_currentStaticLOD == level))
501 return FALSE; //level is already applied. Custom levels are always applied since random options could change.
502
503 applyStaticLODLevel(level);
504 m_currentStaticLOD = level;
505
506 return TRUE;
507}

509void GameLODManager::applyStaticLODLevel(StaticGameLODLevel level)
510{
512// m_sampleCount2D=6;
513// m_sampleCount3D=24;
514// m_streamCount=2;
515// m_useEmissiveNightMaterials=TRUE;
516
517 //save previous info for this level since it may be overwritten by refreshCustomStaticLODLevel().
518 StaticGameLODInfo prevLodBackup;
519 if (m_currentStaticLOD != STATIC_GAME_LOD_UNKNOWN)
520 prevLodBackup=m_staticGameLODInfo[m_currentStaticLOD];
521
522 if (level == STATIC_GAME_LOD_CUSTOM)
523 refreshCustomStaticLODLevel(); //store current settings into custom preset
524
525 StaticGameLODInfo *lodInfo=&m_staticGameLODInfo[level];
526 StaticGameLODInfo *prevLodInfo=&prevLodBackup;
527
528 Int requestedTextureReduction = 0;
529 Bool requestedTrees = m_memPassed; //only use trees if memory requirement passed.
530 if (level == STATIC_GAME_LOD_CUSTOM)
531 { requestedTextureReduction = lodInfo->m_textureReduction;
532 requestedTrees = lodInfo->m_useTrees;
533 }
534 else
535 if (level >= STATIC_GAME_LOD_LOW)
536 { //normal non-custom level gets texture reduction based on recommendation
537 requestedTextureReduction = getRecommendedTextureReduction();
538 }
539
540 if (TheGlobalData)
541 {
542 TheWritableGlobalData->m_maxParticleCount=lodInfo->m_maxParticleCount;
543 TheWritableGlobalData->m_useShadowVolumes=lodInfo->m_useShadowVolumes;
544 TheWritableGlobalData->m_useShadowDecals=lodInfo->m_useShadowDecals;
545
546 //Check if texture resolution changed. No need to apply when current is unknown because display will do it
547 if (requestedTextureReduction != m_currentTextureReduction)
548 {
549 TheWritableGlobalData->m_textureReductionFactor = requestedTextureReduction;
550 if (TheGameClient)
551 TheGameClient->adjustLOD(0); //apply the new setting stored in globaldata
552 }
553
554 //Check if shadow state changed
555 if (m_currentStaticLOD == STATIC_GAME_LOD_UNKNOWN ||
556 lodInfo->m_useShadowVolumes != prevLodInfo->m_useShadowVolumes ||
557 lodInfo->m_useShadowDecals != prevLodInfo->m_useShadowDecals)
558 {
559 if (TheGameClient)
560 {
561 TheGameClient->releaseShadows(); //free all shadows
562 TheGameClient->allocateShadows(); //allocate those shadows that are enabled.
563 }
564 }
565
566 TheWritableGlobalData->m_useCloudMap=lodInfo->m_useCloudMap;
567 TheWritableGlobalData->m_useLightMap=lodInfo->m_useLightMap;
568 TheWritableGlobalData->m_showSoftWaterEdge=lodInfo->m_showSoftWaterEdge;
569 //Check if shoreline blending mode has changed
570 if (m_currentStaticLOD == STATIC_GAME_LOD_UNKNOWN || lodInfo->m_showSoftWaterEdge != prevLodInfo->m_showSoftWaterEdge)
571 {
572 if (TheTerrainVisual)
573 TheTerrainVisual->setShoreLineDetail();
574 }
575
576 TheWritableGlobalData->m_maxTankTrackEdges=lodInfo->m_maxTankTrackEdges;
577 TheWritableGlobalData->m_maxTankTrackOpaqueEdges=lodInfo->m_maxTankTrackOpaqueEdges;
578 TheWritableGlobalData->m_maxTankTrackFadeDelay=lodInfo->m_maxTankTrackFadeDelay;
579 TheWritableGlobalData->m_useTreeSway=lodInfo->m_useTreeSway;
580 TheWritableGlobalData->m_useDrawModuleLOD=!lodInfo->m_useBuildupScaffolds;
581 TheWritableGlobalData->m_useHeatEffects=lodInfo->m_useHeatEffects;
582 TheWritableGlobalData->m_enableDynamicLOD = lodInfo->m_enableDynamicLOD;
