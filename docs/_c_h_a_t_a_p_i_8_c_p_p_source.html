229static std::string trim(std::string s, const std::string& delim)
230{
231 unsigned int i;
232 i = s.find_first_not_of(delim);
233 if (i != s.npos)
234 {
235 s = s.substr(i);
236 }
237
238 i = s.find_last_not_of(delim);
239 if (i>=0 && i<s.npos)
240 {
241 s = s.substr(0, i+1);
242 }
243
244 return s;
245}

249static std::string getNextLine(std::string in, std::string& remainder)
250{
251 int lineEnd;
252 lineEnd = in.find_first_of("\n\r", 0);
253 if (lineEnd < 1)
254 {
255 remainder = "";
256 return in;
257 }
258
259 std::string out = in.substr(0, lineEnd);
260 remainder = in.substr(lineEnd+1);
261
262 remainder = trim(remainder, "\r\n\t ");
263 out = trim(out, "\r\t\n ");
264}

268//-----------------------------------------------------------------------------
269inline const char* skipSeps(const char* p, const char* seps)
270{
271 while (*p && strchr(seps, *p) != NULL)
272 ++p;
273 return p;
274}
275
276//-----------------------------------------------------------------------------
277inline const char* skipNonSeps(const char* p, const char* seps)
278{
279 while (*p && strchr(seps, *p) == NULL)
280 ++p;
281 return p;
282}
283
284//-----------------------------------------------------------------------------
285bool nextToken(std::string& base, std::string& tok, const char* seps = NULL)
286{
287 if (base.empty())
288 return false;
289
290 if (seps == NULL)
291 seps = " \n\r\t";
292
293 const char* start = skipSeps(base.c_str(), seps);
294 const char* end = skipNonSeps(start, seps);
295
296 if (end > start)
297 {
298 int len = end - start;
299 char* tmp = new char[len+1];
300 memcpy(tmp, start, len);
301 tmp[len] = 0;
302
303 tok = tmp;
304 delete[] tmp;
305
306 base = end;
307
308 return true;
309 }
310 else
311 {
312 base = tok = "";
313 return false;
314 }
315}

319static void queuePatch(bool mandatory, std::string downloadURL)
320{
321 // downloadURL is of the form "ftp://ftp.ea.com:user@pass/pub/munkee/bananna.rtp"
322 QueuedDownload q;
323 bool success = true;
324
325 std::string connectionType;
326 success = success && nextToken(downloadURL, connectionType, ":");
327
328 std::string server;
329 success = success && nextToken(downloadURL, server, ":/");
330
331 std::string user;
332 success = success && nextToken(downloadURL, user, ":@");
333
334 std::string pass;
335 success = success && nextToken(downloadURL, pass, "@/");
336
337 std::string filePath;
338 success = success && nextToken(downloadURL, filePath, "");
339
340 if (!success && !user.empty())
341 {
342 // no user/pass combo - move the file into it's proper place
343 filePath = user;
344 user = ""; // LFeenanEA - Credentials removed as per Security requirements
345 pass = "";
346 success = true;
347 }
348
349 std::string fileStr = filePath;
350 unsigned int slashPos = filePath.find_last_of('/');
351 std::string fileDir = "patches\\";
352 std::string fileName = "";
353 if (slashPos == filePath.npos)
354 {
355 fileName = filePath;
356 }
357 else
358 {
359 fileName = filePath.substr(slashPos+1);
360 }
361 fileDir.append(fileName);
362
363 DEBUG_LOG(("download URL split: %d [%s] [%s] [%s] [%s] [%s] [%s] [%s]\n",
364 success, connectionType.c_str(), server.c_str(), user.c_str(), pass.c_str(),
365 filePath.c_str(), fileName.c_str(), fileDir.c_str()));
366
367 if (!success)
368 return;
369
370 q.file = filePath;
371 q.localFile = fileDir;
372 q.password = pass;
373 q.regKey = "";
374 q.server = server;
375 q.tryResume = true;
376 q.userName = user;
377
378 std::list<QueuedDownload>::iterator it = queuedDownloads.begin();
379 while (it != queuedDownloads.end())
380 {
381 if (it->localFile == q.localFile)
382 return; // don't add it if it exists already (because we can check multiple times)
383 ++it;
384 }
385
386 queuedDownloads.push_back(q);
387}

391static GHTTPBool patchCheckCallback( GHTTPRequest request, GHTTPResult result, char * buffer, int bufferLen, void * param )
392{
393 --checksLeft;
394 DEBUG_ASSERTCRASH(checksLeft>=0, ("Too many callbacks"));
395
396 DEBUG_LOG(("Result=%d, buffer=[%s], len=%d\n", result, buffer, bufferLen));
397 if (result != GHTTPSuccess)
398 {
399 cantConnect = true;
400 if (!checksLeft)
401 {
402 startOnline();
403 }
404 return GHTTPTrue;
405 }
406
407 std::string message = buffer;
408 std::string line;
409 while (nextToken(message, line, "\r\n"))
410 {
411 std::string type, req, url;
412 bool ok = true;
413 ok = ok && nextToken(line, type, " ");
414 ok = ok && nextToken(line, req, " ");
415 ok = ok && nextToken(line, url, " ");
416 if (ok && type == "patch")
417 {
418 DEBUG_LOG(("Saw a patch: %d/[%s]\n", atoi(req.c_str()), url.c_str()));
419 queuePatch( atoi(req.c_str()), url );
420 }
421 else if (ok && type == "server")
422 {
423 }
424 }
425
426 if (!checksLeft)
427 {
428 startOnline();
429 }
430
431 return GHTTPTrue;
432}

778typedef struct SRecord {
779 int ID; // ID number of the string resource.
780 int TimeStamp; // 'Time' that this string was last requested.
781 char String[2048]; // Copy of string resource.
782
783 SRecord(void) : ID(-1), TimeStamp(-1) {}
784} SRecord;

786
787/***********************************************************************************************
788 * Fetch_String -- Fetches a string resource. *
789 * *
790 * Fetches a string resource and returns a pointer to its text. *
791 * *
792 * INPUT: id -- The ID number of the string resource to fetch. *
793 * *
794 * OUTPUT: Returns with a pointer to the actual text of the string resource. *
795 * *
796 * WARNINGS: none *
797 * *
798 * HISTORY: *
799 * 12/25/1996 JLB : Created. *
800 *=============================================================================================*/
801char const * Fetch_String(int id)
802{
803 static SRecord _buffers[64];
804 static int _time = 0;
805
806 /*
807 ** Determine if the string ID requested is valid. If not then return an empty string pointer.
808 */
809 if (id == -1 || id == TXT_NONE) return("");
810
811 /*
812 ** Adjust the 'time stamp' tracking value. This is an artificial value used merely to track
813 ** the relative age of the strings requested.
814 */
815 _time = _time+1;
816
817 /*
818 ** Check to see if the requested string has already been fetched into a buffer. If so, then
819 ** return a pointer to that string (update the time stamp as well).
820 */
821 for (int index = 0; index < ARRAY_SIZE(_buffers); index++) {
822 if (_buffers[index].ID == id) {
