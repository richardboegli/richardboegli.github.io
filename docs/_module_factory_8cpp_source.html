569//-------------------------------------------------------------------------------------------------
570Int ModuleFactory::findModuleInterfaceMask(const AsciiString& name, ModuleType type)
571{
572 if (name.isEmpty())
573 return 0;
574
575 const ModuleTemplate* moduleTemplate = findModuleTemplate(name, type);
576 if (moduleTemplate)
577 {
578 return moduleTemplate->m_whichInterfaces;
579 }
580
581 return 0;
582}
583
584//-------------------------------------------------------------------------------------------------
585ModuleData* ModuleFactory::newModuleDataFromINI(INI* ini, const AsciiString& name, ModuleType type,
586 const AsciiString& moduleTag)
587{
588 if (name.isEmpty())
589 return NULL;
590
591 const ModuleTemplate* moduleTemplate = findModuleTemplate(name, type);
592 if (moduleTemplate)
593 {
594 ModuleData* md = (*moduleTemplate->m_createDataProc)(ini);
595 md->setModuleTagNameKey( NAMEKEY( moduleTag ) );
596 m_moduleDataList.push_back(md);
597 return md;
598 }
599
600 return NULL;
601}

606/*static*/ NameKeyType ModuleFactory::makeDecoratedNameKey(const AsciiString& name, ModuleType type)

608 char tmp[256];
609 tmp[0] = '0' + (int)type;
610 strcpy(&tmp[1], name.str());

612}

615const ModuleFactory::ModuleTemplate* ModuleFactory::findModuleTemplate(const AsciiString& name, ModuleType type)
616{
