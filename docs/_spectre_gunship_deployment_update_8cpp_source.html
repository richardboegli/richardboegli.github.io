145Bool SpectreGunshipDeploymentUpdate::initiateIntentToDoSpecialPower(const SpecialPowerTemplate *specialPowerTemplate, const Object *targetObj, const Coord3D *targetPos, const Waypoint *way, UnsignedInt commandOptions )
146{
147 const SpectreGunshipDeploymentUpdateModuleData *data = getSpectreGunshipDeploymentUpdateModuleData();
148
149 if( m_specialPowerModule->getSpecialPowerTemplate() != specialPowerTemplate )
150 {
151 //Check to make sure our modules are connected.
152 return FALSE;
153 }
154
155// getObject()->getControllingPlayer()->getAcademyStats()->recordSpecialPowerUsed( specialPowerTemplate );
156
157 if( !BitTest( commandOptions, COMMAND_FIRED_BY_SCRIPT ) )
158 {
159/******CHANGE*******/ m_initialTargetPosition.set( targetPos );
160 }
161 else
162 {
163 UnsignedInt now = TheGameLogic->getFrame();
164 m_specialPowerModule->setReadyFrame( now );
165/******CHANGE*******/ m_initialTargetPosition.set( targetPos );
166// setLogicalStatus( GUNSHIPDEPLOY_STATUS_INSERTING );
167 }
168
169 Object *newGunship = TheGameLogic->findObjectByID( m_gunshipID );
170 const ThingTemplate *gunshipTemplate = TheThingFactory->findTemplate( data->m_gunshipTemplateName );
171 if( newGunship != NULL )
172 {
173// disengageAndDepartAO( newGunship );
174 m_gunshipID = INVALID_ID;
175 newGunship = NULL;
176 }
177
178
179 // Lets make a gunship, since we have none.
180 {
181 newGunship = TheThingFactory->newObject( gunshipTemplate, getObject()->getTeam() );
182 }
183
184 DEBUG_ASSERTCRASH( newGunship, ("SpecterGunshipUpdate failed to find or create a gunship object"));
185 if ( newGunship )
186 {
187 //PRODUCER
188 newGunship->setProducer( getObject() );
189
190 //POSITION
191 Coord3D creationCoord;
192 switch (data->m_createLoc)
193 {
194 case CREATE_GUNSHIP_AT_EDGE_NEAR_SOURCE:
195 creationCoord = TheTerrainLogic->findClosestEdgePoint( getObject()->getPosition() );
196 break;
197 case CREATE_GUNSHIP_AT_EDGE_FARTHEST_FROM_SOURCE:
198 creationCoord = TheTerrainLogic->findFarthestEdgePoint( getObject()->getPosition() );
199 break;
200 case CREATE_GUNSHIP_AT_EDGE_NEAR_TARGET:
201 creationCoord = TheTerrainLogic->findClosestEdgePoint(targetPos);
202 break;
203 case CREATE_GUNSHIP_AT_EDGE_FARTHEST_FROM_TARGET:
204 default:
205 creationCoord = TheTerrainLogic->findFarthestEdgePoint(targetPos);
206 break;
207 }
208
209
210
211
212 // HERE WE NEED TO CREATE THE POINT FURTHER OFF THE MAP SO WE CANT SEE THE LAME HOVER AND ACCELLERATE BEHAVIOR
213 Coord3D deltaToCreationPoint = m_initialTargetPosition;
214 deltaToCreationPoint.sub( &creationCoord );
215 Real distanceFromTarget = deltaToCreationPoint.length();
216 deltaToCreationPoint.normalize();
217 deltaToCreationPoint.x *= ( distanceFromTarget + data->m_gunshipOrbitRadius );
218 deltaToCreationPoint.y *= ( distanceFromTarget + data->m_gunshipOrbitRadius );
219 creationCoord.x = m_initialTargetPosition.x - deltaToCreationPoint.x;
220 creationCoord.y = m_initialTargetPosition.y - deltaToCreationPoint.y;
221
222 Real preferredElevation = newGunship->getAI()->getCurLocomotor()->getPreferredHeight();
223 creationCoord.z = preferredElevation;
224 newGunship->setPosition( &creationCoord );
225
226 //ORIENTATION
227 Real orient = atan2( m_initialTargetPosition.y - creationCoord.y, m_initialTargetPosition.x - creationCoord.x);
228 newGunship->setOrientation( orient );
229
230 // ID
231 m_gunshipID = newGunship->getID();
232
233 // FIRE THE SPECIAL POWER OF THE GUNSHIP
234 SpecialPowerModuleInterface *shipSPMInterface = newGunship->getSpecialPowerModule( specialPowerTemplate );
235 if( shipSPMInterface )
236 {
237 SpecialPowerModule *spModule = (SpecialPowerModule*)shipSPMInterface;
238 spModule->markSpecialPowerTriggered( &m_initialTargetPosition );
239 spModule->doSpecialPowerAtLocation( &m_initialTargetPosition, INVALID_ANGLE, commandOptions );
240
241 }
242
243 // MAKE THE GUNSHIP SELECTED
244
245 TheGameLogic->selectObject( newGunship, TRUE, getObject()->getControllingPlayer()->getPlayerMask(), TRUE );
246
247
248 }
249
250
251 SpecialPowerModuleInterface *spmInterface = getObject()->getSpecialPowerModule( specialPowerTemplate );
252 if( spmInterface )
253 {
254 SpecialPowerModule *spModule = (SpecialPowerModule*)spmInterface;
255 spModule->markSpecialPowerTriggered( &m_initialTargetPosition );
256 }
257
258 return TRUE;
259}

267UpdateSleepTime SpectreGunshipDeploymentUpdate::update()
268{
269// const SpectreGunshipDeploymentUpdateModuleData *data = getSpectreGunshipDeploymentUpdateModuleData();
270
271
272 Object *me = getObject();
273 // Abort conditions.
274 if( me->testStatus(OBJECT_STATUS_SOLD)
275 || me->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION)
276 || me->isEffectivelyDead() )
277 {
278
279 return UPDATE_SLEEP_FOREVER;
280 }
281
282 return UPDATE_SLEEP_NONE;
283
284}
