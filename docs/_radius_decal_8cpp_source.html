58void RadiusDecalTemplate::createRadiusDecal(const Coord3D& pos, Real radius, const Player* owningPlayer, RadiusDecal& result) const
59{
60 result.clear();
61
62 if (owningPlayer == NULL)
63 {
64 DEBUG_CRASH(("You MUST specify a non-NULL owningPlayer to createRadiusDecal. (srj)\n"));
65 return;
66 }
67
68 if (m_name.isEmpty() || radius <= 0.0f)
69 return;
70
71 // it is now considered nonEmpty, regardless of the state of m_decal, etc
72 result.m_empty = false;
73
74 if (!m_onlyVisibleToOwningPlayer ||
75 owningPlayer->getPlayerIndex() == ThePlayerList->getLocalPlayer()->getPlayerIndex())
76 {
77 Shadow::ShadowTypeInfo decalInfo;
78 decalInfo.allowUpdates = FALSE; // shadow texture will never update
79 decalInfo.allowWorldAlign = TRUE; // shadow image will wrap around world objects
80 decalInfo.m_type = m_shadowType;
81 strcpy(decalInfo.m_ShadowName, m_name.str()); // name of your texture
82 decalInfo.m_sizeX = radius*2; // world space dimensions
83 decalInfo.m_sizeY = radius*2; // world space dimensions
84
85 result.m_decal = TheProjectedShadowManager->addDecal(&decalInfo);
86 if (result.m_decal)
87 {
88 result.m_decal->setAngle(0.0f);
89 result.m_decal->setColor(m_color == 0 ? owningPlayer->getPlayerColor() : m_color);
90 result.m_decal->setPosition(pos.x, pos.y, pos.z);
91 result.m_template = this;
92 }
93 else
94 {
95 DEBUG_CRASH(("Unable to add decal %s\n",decalInfo.m_ShadowName));
96 }
97 }
98}

195// ------------------------------------------------------------------------------------------------
196void RadiusDecal::update()
197{
198 if (m_decal && m_template)
199 {
200 UnsignedInt now = TheGameLogic->getFrame();
201 Real theta = (2*PI) * (Real)(now % m_template->m_opacityThrobTime) / (Real)m_template->m_opacityThrobTime;
202 Real percent = 0.5f * (Sin(theta) + 1.0f);
203 Int opac;
204 if( TheGameLogic->getDrawIconUI() )
205 {
206 opac = REAL_TO_INT((m_template->m_minOpacity + percent * (m_template->m_maxOpacity - m_template->m_minOpacity)) * 255.0f);
207 }
208 else
209 {
210 //Scripts turned this off, so don't show them!
211 opac = 0;
212 }
213 m_decal->setOpacity(opac);
214 }
215}
