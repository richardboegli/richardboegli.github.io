39// Packet-level encryption is an XOR operation, for speed reasons. To get
40// the max throughput, we only XOR whole 4-byte words, so the last bytes
41// can be non-XOR'd.
42
43// This assumes the buf is a multiple of 4 bytes. Extra is not encrypted.
44static inline void encryptBuf( unsigned char *buf, Int len )
45{
46 UnsignedInt mask = 0x0000Fade;
47
48 UnsignedInt *uintPtr = (UnsignedInt *) (buf);
49
50 for (int i=0 ; i<len/4 ; i++) {
51 *uintPtr = (*uintPtr) ^ mask;
52 *uintPtr = htonl(*uintPtr);
53 uintPtr++;
54 mask += 0x00000321; // just for fun
55 }
56}
57
58// This assumes the buf is a multiple of 4 bytes. Extra is not encrypted.
59static inline void decryptBuf( unsigned char *buf, Int len )
60{
61 UnsignedInt mask = 0x0000Fade;
62
63 UnsignedInt *uintPtr = (UnsignedInt *) (buf);
64
65 for (int i=0 ; i<len/4 ; i++) {
66 *uintPtr = htonl(*uintPtr);
67 *uintPtr = (*uintPtr) ^ mask;
68 uintPtr++;
69 mask += 0x00000321; // just for fun
70 }
71}

91Bool Transport::init( UnsignedInt ip, UnsignedShort port )
92{
93 // ----- Initialize Winsock -----
94 if (!m_winsockInit)
95 {
96 WORD verReq = MAKEWORD(2, 2);
97 WSADATA wsadata;
98
99 int err = WSAStartup(verReq, &wsadata);
100 if (err != 0) {
101 return false;
102 }
103
104 if ((LOBYTE(wsadata.wVersion) != 2) || (HIBYTE(wsadata.wVersion) !=2)) {
105 WSACleanup();
106 return false;
107 }
108 m_winsockInit = true;
109 }
110
111 // ------- Bind our port --------
112 if (m_udpsock)
113 delete m_udpsock;
114 m_udpsock = NEW UDP();
115
116 if (!m_udpsock)
117 return false;
118
119 int retval = -1;
120 time_t now = timeGetTime();
121 while ((retval != 0) && ((timeGetTime() - now) < 1000)) {
122 retval = m_udpsock->Bind(ip, port);
123 }
124
125 if (retval != 0) {
126 DEBUG_CRASH(("Could not bind to 0x%8.8X:%d\n", ip, port));
127 DEBUG_LOG(("Transport::init - Failure to bind socket with error code %x\n", retval));
128 delete m_udpsock;
129 m_udpsock = NULL;
130 return false;
131 }
132
133 // ------- Clear buffers --------
134 for (int i=0; i<MAX_MESSAGES; ++i)
135 {
136 m_outBuffer[i].length = 0;
137 m_inBuffer[i].length = 0;
138#if defined(_DEBUG) || defined(_INTERNAL)
139 m_delayedInBuffer[i].message.length = 0;
140#endif
141 }
142 for (i=0; i<MAX_TRANSPORT_STATISTICS_SECONDS; ++i)
143 {
144 m_incomingBytes[i] = 0;
145 m_outgoingBytes[i] = 0;
146 m_unknownBytes[i] = 0;
147 m_incomingPackets[i] = 0;
148 m_outgoingPackets[i] = 0;
149 m_unknownPackets[i] = 0;
150 }
151 m_statisticsSlot = 0;
152 m_lastSecond = timeGetTime();
153
154 m_port = port;
155
156#if defined(_DEBUG) || defined(_INTERNAL)
157 if (TheGlobalData->m_latencyAverage > 0 || TheGlobalData->m_latencyNoise)
158 m_useLatency = true;
159
160 if (TheGlobalData->m_packetLoss)
161 m_usePacketLoss = true;
162#endif
163
164 return true;
165}

186 {
187 retval = FALSE;

198Bool Transport::doSend() {
199 if (!m_udpsock)
200 {
201 DEBUG_LOG(("Transport::doSend() - m_udpSock is NULL!\n"));
202 return FALSE;
203 }
204
205 Bool retval = TRUE;
206
207 // Statistics gathering
208 UnsignedInt now = timeGetTime();
209 if (m_lastSecond + 1000 < now)
210 {
211 m_lastSecond = now;
212 m_statisticsSlot = (m_statisticsSlot + 1) % MAX_TRANSPORT_STATISTICS_SECONDS;
213 m_outgoingPackets[m_statisticsSlot] = 0;
214 m_outgoingBytes[m_statisticsSlot] = 0;
215 m_incomingPackets[m_statisticsSlot] = 0;
216 m_incomingBytes[m_statisticsSlot] = 0;
217 m_unknownPackets[m_statisticsSlot] = 0;
218 m_unknownBytes[m_statisticsSlot] = 0;
219 }
220
221 // Send all messages
222 int i;
223 for (i=0; i<MAX_MESSAGES; ++i)
224 {
225 if (m_outBuffer[i].length != 0)
226 {
227 int bytesSent = 0;
228 // Send this message
229 if ((bytesSent = m_udpsock->Write((unsigned char *)(&m_outBuffer[i]), m_outBuffer[i].length + sizeof(TransportMessageHeader), m_outBuffer[i].addr, m_outBuffer[i].port)) > 0)
230 {
231 //DEBUG_LOG(("Sending %d bytes to %d:%d\n", m_outBuffer[i].length + sizeof(TransportMessageHeader), m_outBuffer[i].addr, m_outBuffer[i].port));
232 m_outgoingPackets[m_statisticsSlot]++;
233 m_outgoingBytes[m_statisticsSlot] += m_outBuffer[i].length + sizeof(TransportMessageHeader);
234 m_outBuffer[i].length = 0; // Remove from queue
235// DEBUG_LOG(("Transport::doSend - sent %d butes to %d.%d.%d.%d:%d\n", bytesSent,
236// (m_outBuffer[i].addr >> 24) & 0xff,
237// (m_outBuffer[i].addr >> 16) & 0xff,
238// (m_outBuffer[i].addr >> 8) & 0xff,
239// m_outBuffer[i].addr & 0xff,
240// m_outBuffer[i].port));
241 }
242 else
243 {
244 //DEBUG_LOG(("Could not write to socket!!! Not discarding message!\n"));
245 retval = FALSE;
246 //DEBUG_LOG(("Transport::doSend returning FALSE\n"));
247 }
248 }
249 } // for (i=0; i<MAX_MESSAGES; ++i)
250
251#if defined(_DEBUG) || defined(_INTERNAL)
252 // Latency simulation - deliver anything we're holding on to that is ready
253 if (m_useLatency)
254 {
255 for (i=0; i<MAX_MESSAGES; ++i)
256 {
257 if (m_delayedInBuffer[i].message.length != 0 && m_delayedInBuffer[i].deliveryTime <= now)
258 {
259 for (int j=0; j<MAX_MESSAGES; ++j)
260 {
261 if (m_inBuffer[j].length == 0)
262 {
263 // Empty slot; use it
264 memcpy(&m_inBuffer[j], &m_delayedInBuffer[i].message, sizeof(TransportMessage));
265 m_delayedInBuffer[i].message.length = 0;
266 break;
267 }
268 }
269 }
270 }
271 }
272#endif
273 return retval;
274}

276Bool Transport::doRecv()
277{
278 if (!m_udpsock)
279 {
280 DEBUG_LOG(("Transport::doRecv() - m_udpSock is NULL!\n"));
281 return FALSE;
282 }
283
284 Bool retval = TRUE;
285
286 // Read in anything on our socket
287 sockaddr_in from;
288#if defined(_DEBUG) || defined(_INTERNAL)
289 UnsignedInt now = timeGetTime();
290#endif
291
292 TransportMessage incomingMessage;
293 unsigned char *buf = (unsigned char *)&incomingMessage;
294 int len = MAX_MESSAGE_LEN;
295// DEBUG_LOG(("Transport::doRecv - checking\n"));
296 while ( (len=m_udpsock->Read(buf, MAX_MESSAGE_LEN, &from)) > 0 )
297 {
298#if defined(_DEBUG) || defined(_INTERNAL)
299 // Packet loss simulation
300 if (m_usePacketLoss)
301 {
302 if ( TheGlobalData->m_packetLoss >= GameClientRandomValue(0, 100) )
303 {
304 continue;
305 }
306 }
307#endif
308
309// DEBUG_LOG(("Transport::doRecv - Got something! len = %d\n", len));
310 // Decrypt the packet
311// DEBUG_LOG(("buffer = "));
312// for (Int munkee = 0; munkee < len; ++munkee) {
313// DEBUG_LOG(("%02x", *(buf + munkee)));
314// }
315// DEBUG_LOG(("\n"));
316 decryptBuf(buf, len);
317
318 incomingMessage.length = len - sizeof(TransportMessageHeader);
319
320 if (len <= sizeof(TransportMessageHeader) || !isGeneralsPacket( &incomingMessage ))
321 {
322 m_unknownPackets[m_statisticsSlot]++;
323 m_unknownBytes[m_statisticsSlot] += len;
324 continue;
325 }
326
327 // Something there; stick it somewhere
328// DEBUG_LOG(("Saw %d bytes from %d:%d\n", len, ntohl(from.sin_addr.S_un.S_addr), ntohs(from.sin_port)));
329 m_incomingPackets[m_statisticsSlot]++;
330 m_incomingBytes[m_statisticsSlot] += len;
331
332 for (int i=0; i<MAX_MESSAGES; ++i)
333 {
334#if defined(_DEBUG) || defined(_INTERNAL)
335 // Latency simulation
336 if (m_useLatency)
337 {
338 if (m_delayedInBuffer[i].message.length == 0)
339 {
340 // Empty slot; use it
341 m_delayedInBuffer[i].deliveryTime =
342 now + TheGlobalData->m_latencyAverage +
343 (Int)(TheGlobalData->m_latencyAmplitude * sin(now * TheGlobalData->m_latencyPeriod)) +
344 GameClientRandomValue(-TheGlobalData->m_latencyNoise, TheGlobalData->m_latencyNoise);
345 m_delayedInBuffer[i].message.length = incomingMessage.length;
346 m_delayedInBuffer[i].message.addr = ntohl(from.sin_addr.S_un.S_addr);
347 m_delayedInBuffer[i].message.port = ntohs(from.sin_port);
348 memcpy(&m_delayedInBuffer[i].message, buf, len);
349 break;
350 }
351 }
352 else
353 {
354#endif
355 if (m_inBuffer[i].length == 0)
356 {
357 // Empty slot; use it
358 m_inBuffer[i].length = incomingMessage.length;
359 m_inBuffer[i].addr = ntohl(from.sin_addr.S_un.S_addr);
360 m_inBuffer[i].port = ntohs(from.sin_port);
361 memcpy(&m_inBuffer[i], buf, len);
362 break;
363 }
364#if defined(_DEBUG) || defined(_INTERNAL)
365 }
366#endif
367 }
368 //DEBUG_ASSERTCRASH(i<MAX_MESSAGES, ("Message lost!"));
369 }
370
371 if (len == -1) {
372 // there was a socket error trying to perform a read.
373 //DEBUG_LOG(("Transport::doRecv returning FALSE\n"));
374 retval = FALSE;
375 }
376
377 return retval;
378}

380Bool Transport::queueSend(UnsignedInt addr, UnsignedShort port, const UnsignedByte *buf, Int len /*,
381 NetMessageFlags flags, Int id */)
382{
383 int i;
384
385 if (len < 1 || len > MAX_PACKET_SIZE)
386 {
387 return false;
388 }
389
390 for (i=0; i<MAX_MESSAGES; ++i)
391 {
392 if (m_outBuffer[i].length == 0)
393 {
394 // Insert data here
395 m_outBuffer[i].length = len;
396 memcpy(m_outBuffer[i].data, buf, len);
397 m_outBuffer[i].addr = addr;
398 m_outBuffer[i].port = port;
399// m_outBuffer[i].header.flags = flags;
400// m_outBuffer[i].header.id = id;
401 m_outBuffer[i].header.magic = GENERALS_MAGIC_NUMBER;
402
403 CRC crc;
404 crc.computeCRC( (unsigned char *)(&(m_outBuffer[i].header.magic)), m_outBuffer[i].length + sizeof(TransportMessageHeader) - sizeof(UnsignedInt) );
405// DEBUG_LOG(("About to assign the CRC for the packet\n"));
406 m_outBuffer[i].header.crc = crc.get();
407
408 // Encrypt packet
409// DEBUG_LOG(("buffer: "));
410 encryptBuf((unsigned char *)&m_outBuffer[i], len + sizeof(TransportMessageHeader));
411// DEBUG_LOG(("\n"));
412
413 return true;
414 }
415 }
416 return false;
417}

419Bool Transport::isGeneralsPacket( TransportMessage *msg )
420{
421 if (!msg)
422 return false;
423
424 if (msg->length < 0 || msg->length > MAX_MESSAGE_LEN)
425 return false;
426
427 CRC crc;
428// crc.computeCRC( (unsigned char *)msg->data, msg->length );
429 crc.computeCRC( (unsigned char *)(&(msg->header.magic)), msg->length + sizeof(TransportMessageHeader) - sizeof(UnsignedInt) );
430
431 if (crc.get() != msg->header.crc)
432 return false;
433
434 if (msg->header.magic != GENERALS_MAGIC_NUMBER)
435 return false;
436
437 return true;
438}
