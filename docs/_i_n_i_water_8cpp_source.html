49void INI::parseWaterSettingDefinition( INI* ini )
50{
51 AsciiString name;
52 WaterSetting *waterSetting = NULL;
53
54 // read the name
55 const char* token = ini->getNextToken();
56
57 name.set( token );
58
59 // get the water setting we want to load based on name
60 char **timeOfDayName = TimeOfDayNames;
61 Int timeOfDayIndex = 0; // TIME_OF_DAY_INVALID
62 while( timeOfDayName && *timeOfDayName )
63 {
64
65 if( stricmp( *timeOfDayName, name.str() ) == 0 )
66 {
67
68 waterSetting = &WaterSettings[ timeOfDayIndex ];
69 break;
70
71 } // end if
72
73 // next name
74 timeOfDayName++;
75 timeOfDayIndex++;
76
77 } // end while
78
79 // check for no time of day match
80 if( waterSetting == NULL )
81 throw INI_INVALID_DATA;
82
83 // parse the data
84 ini->initFromINI( waterSetting, waterSetting->getFieldParse() );
85
86} // end parseWaterSetting

89void INI::parseWaterTransparencyDefinition( INI *ini )
90{
91 if (TheWaterTransparency == NULL) {
92 TheWaterTransparency = newInstance(WaterTransparencySetting);
93 } else if (ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES) {
94 WaterTransparencySetting* wt = (WaterTransparencySetting*) (TheWaterTransparency.getNonOverloadedPointer());
95 WaterTransparencySetting* wtOverride = newInstance(WaterTransparencySetting);
96 *wtOverride = *wt;
97
98 // Mark that it is an override.
99 wtOverride->markAsOverride();
100
101 wt->friend_getFinalOverride()->setNextOverride(wtOverride);
102 } else {
103 throw INI_INVALID_DATA;
104 }
105
106 WaterTransparencySetting* waterTrans = (WaterTransparencySetting*) (TheWaterTransparency.getNonOverloadedPointer());
107 waterTrans = (WaterTransparencySetting*) (waterTrans->friend_getFinalOverride());
108 // parse the data
109 ini->initFromINI( waterTrans, TheWaterTransparency->getFieldParse() );
110
111 // If we overrode any skybox textures, then call the W3D Water stuff.
112 if (ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES) {
113 // Check to see if we overrode any skybox textures.
114 // If we did, then we need to replace them in the model.
115 // Copy/Paste monkeys PLEASE TAKE NOTE. This technique only works for the skybox because we
116 // know that there will never be more than one sky box. If you were to use this technique for
117 // technicals, for instance, it would make all technicals in the level have the same new
118 // texture.
119
120 const WaterTransparencySetting* wtOriginal = TheWaterTransparency.getNonOverloadedPointer();
121 OVERRIDE<WaterTransparencySetting> wtOverride = TheWaterTransparency;
122
123 if (wtOriginal == wtOverride)
124 return;
125
126 const AsciiString *oldTextures[5],*newTextures[5];
127
128 //Copy current texture names into arrays
129 oldTextures[0]=&wtOriginal->m_skyboxTextureN;
130 newTextures[0]=&wtOverride->m_skyboxTextureN;
131 oldTextures[1]=&wtOriginal->m_skyboxTextureE;
132 newTextures[1]=&wtOverride->m_skyboxTextureE;
133 oldTextures[2]=&wtOriginal->m_skyboxTextureS;
134 newTextures[2]=&wtOverride->m_skyboxTextureS;
135 oldTextures[3]=&wtOriginal->m_skyboxTextureW;
136 newTextures[3]=&wtOverride->m_skyboxTextureW;
137 oldTextures[4]=&wtOriginal->m_skyboxTextureT;
138 newTextures[4]=&wtOverride->m_skyboxTextureT;
139
140 TheTerrainVisual->replaceSkyboxTextures(oldTextures, newTextures);
141 }
142}
