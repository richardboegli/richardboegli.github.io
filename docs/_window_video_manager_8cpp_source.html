37// Filename: WindowVideoManager.cpp
38//
39// author: Chris Huybregts
40//
41// purpose: Every window is setup to be able to draw a movie. The
42// WindowVideoManager will take care of setting up the window,
43// creating/destroying the buffers, and the ability to pause/stop
44// movies.
45//
46// To Use: Create a manager and initialize it. Make sure the manager's Update
47// is called every frame or how ever often it needs to be updated.
48// Call reset if the manager needs to be cleared.
49// Play a movie in a window by passing a window pointer, a movie name,
50// and the playtype.
51// If a user trys to play a two different movies on the same window,
52// only the last one added will play.
53// It's important when destroying a window, that window is removed
54// from the manager (Or call Reset if you know no other windows are
55// playing a movie).
56//

193void WindowVideoManager::update( void )
194{
195 WindowVideoMap::iterator it = m_playingVideos.begin();
196
197 if(m_pauseAllMovies || m_stopAllMovies)
198 return;
199
200 //Iterate through the maps
201 while(it != m_playingVideos.end())
202 {
203 WindowVideo *winVid = it->second;
204
205 if(!winVid)
206 {
207 DEBUG_CRASH(("There's No WindowVideo in the m_playignVideos list"));
208 return;
209 }
210 GameWindow *win = winVid->getWin();
211
212 if(winVid->getState() == WINDOW_VIDEO_STATE_HIDDEN && (win->winIsHidden() == FALSE))
213 {
214 resumeMovie(win);
215 }
216
217 if(winVid->getState() == WINDOW_VIDEO_STATE_PLAY && win->winIsHidden())
218 {
219 hideMovie(win);
220 }
221
222 // Only advance the frame if we're playing
223 if(winVid->getState() != WINDOW_VIDEO_STATE_PLAY)
224 {
225 it++;
226 continue;
227 }
228
229 // Get the Stream and the buffer to update for each animation
230 VideoStreamInterface *videoStream = winVid->getVideoStream();
231 VideoBuffer *videoBuffer = winVid->getVideoBuffer();
232
233 if ( videoStream && videoBuffer )
234 {
235 if ( videoStream->isFrameReady())
236 {
237 videoStream->frameDecompress();
238 videoStream->frameRender( videoBuffer );
239 videoStream->frameNext();
240
241 // If we reach frame Index of 0, we might have to pause, or loop.
242 if ( videoStream->frameIndex() == 0 )
243 {
244 if(winVid->getPlayType() == WINDOW_PLAY_MOVIE_ONCE)
245 stopMovie(win);
246 else if (winVid->getPlayType() == WINDOW_PLAY_MOVIE_SHOW_LAST_FRAME)
247 pauseMovie(win);
248 }
249 }
250 }
251
252 it++;
253 }
254}

256void WindowVideoManager::playMovie( GameWindow *win, AsciiString movieName, WindowVideoPlayType playType )
257{
258 // if we already have a movie playing for that window, kill it.
259 stopAndRemoveMovie( win );
260
261 // create the new stream
262 VideoStreamInterface *videoStream = TheVideoPlayer->open( movieName );
263 if ( videoStream == NULL )
264 {
265 return;
266 }
267
268 // Create the new buffer
269 VideoBuffer *videoBuffer = TheDisplay->createVideoBuffer();
270 if ( videoBuffer == NULL ||
271 !videoBuffer->allocate( videoStream->width(),
272 videoStream->height())
273 )
274 {
275 // If we failed to create the buffer...
276 delete videoBuffer;
277 videoBuffer = NULL;
278
279 if ( videoStream )
280 videoStream->close();
281 videoStream = NULL;
282
283 return;
284 }
285
286 // now that we have everything, create the new WindowVideo Structure
287 WindowVideo *winVid = NEW WindowVideo;
288
289 // init it.
290 winVid->init( win, movieName,playType,videoBuffer,videoStream);
291
292 // add it to our map.
293 m_playingVideos[win] = winVid;
294
295 m_pauseAllMovies = FALSE;
296 m_stopAllMovies = FALSE;
297}
