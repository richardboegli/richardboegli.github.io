159void Matrix3::Set(const Quaternion & q)
160{
161 Row[0][0] = (float)(1.0 - 2.0 * (q[1] * q[1] + q[2] * q[2]));
162 Row[0][1] = (float)(2.0 * (q[0] * q[1] - q[2] * q[3]));
163 Row[0][2] = (float)(2.0 * (q[2] * q[0] + q[1] * q[3]));
164
165 Row[1][0] = (float)(2.0 * (q[0] * q[1] + q[2] * q[3]));
166 Row[1][1] = (float)(1.0 - 2.0f * (q[2] * q[2] + q[0] * q[0]));
167 Row[1][2] = (float)(2.0 * (q[1] * q[2] - q[0] * q[3]));
168
169 Row[2][0] = (float)(2.0 * (q[2] * q[0] - q[1] * q[3]));
170 Row[2][1] = (float)(2.0 * (q[1] * q[2] + q[0] * q[3]));
171 Row[2][2] =(float)(1.0 - 2.0 * (q[1] * q[1] + q[0] * q[0]));
172}

288{
289 Matrix3 tmp;
290 Matrix3 * Aptr;
291 float tmp1,tmp2,tmp3;
292
293 // Check for aliased parameters, copy the 'A' matrix into a temporary if the
294 // result is going into 'A'. (in this case, this function is no better than
295 // the overloaded C++ operator...)
296 if (set_res == &A) {
297 tmp = A;
298 Aptr = &tmp;
299 } else {
300 Aptr = (Matrix3 *)&A;
301 }
302
303 tmp1 = B[0][0];
304 tmp2 = B[1][0];
305 tmp3 = B[2][0];
306
307 (*set_res)[0][0] = (float)((*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3);
308 (*set_res)[1][0] = (float)((*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3);
309 (*set_res)[2][0] = (float)((*Aptr)[2][0]*tmp1 + (*Aptr)[2][1]*tmp2 + (*Aptr)[2][2]*tmp3);
310
311 tmp1 = B[0][1];
312 tmp2 = B[1][1];
313 tmp3 = B[2][1];
314
315 (*set_res)[0][1] = (float)((*Aptr)[0][0]*tmp1 + (*Aptr)[0][1]*tmp2 + (*Aptr)[0][2]*tmp3);
316 (*set_res)[1][1] = (float)((*Aptr)[1][0]*tmp1 + (*Aptr)[1][1]*tmp2 + (*Aptr)[1][2]*tmp3);

328int Matrix3::Is_Orthogonal(void) const
329{
330 Vector3 x(Row[0].X,Row[0].Y,Row[0].Z);
331 Vector3 y(Row[1].X,Row[1].Y,Row[1].Z);
332 Vector3 z(Row[2].X,Row[2].Y,Row[2].Z);
333
334 if (Vector3::Dot_Product(x,y) > WWMATH_EPSILON) return 0;
335 if (Vector3::Dot_Product(y,z) > WWMATH_EPSILON) return 0;
336 if (Vector3::Dot_Product(z,x) > WWMATH_EPSILON) return 0;
337
338 if (WWMath::Fabs(x.Length() - 1.0f) > WWMATH_EPSILON) return 0;
339 if (WWMath::Fabs(y.Length() - 1.0f) > WWMATH_EPSILON) return 0;
340 if (WWMath::Fabs(z.Length() - 1.0f) > WWMATH_EPSILON) return 0;
341
342 return 1;
343}

345void Matrix3::Re_Orthogonalize(void)
346{
347 Vector3 x(Row[0][0],Row[0][1],Row[0][2]);
348 Vector3 y(Row[1][0],Row[1][1],Row[1][2]);
349 Vector3 z;
350
351 Vector3::Cross_Product(x,y,&z);
352 Vector3::Cross_Product(z,x,&y);
353
354 float len = x.Length();
355 if (len < WWMATH_EPSILON) {
356 Make_Identity();
357 return;
358 } else {
359 x /= len;
360 }
361
362 len = y.Length();
363 if (len < WWMATH_EPSILON) {
364 Make_Identity();
365 return;
366 } else {
367 y /= len;
368 }
369
370 len = z.Length();
371 if (len < WWMATH_EPSILON) {
372 Make_Identity();
373 return;
374 } else {
375 z /= len;
376 }
377
378 Row[0][0] = x.X;
379 Row[0][1] = x.Y;
380 Row[0][2] = x.Z;
381
382 Row[1][0] = y.X;
383 Row[1][1] = y.Y;
384 Row[1][2] = y.Z;
385
386 Row[2][0] = z.X;
387 Row[2][1] = z.Y;
388 Row[2][2] = z.Z;
389}
