49** that needs to store the data into a memory buffer. This can only serve as the final
50** link in the chain of pipe segments.
51*/

53{

58
59 private:
60 Buffer BufferPtr;
61 int Index;

63 bool Is_Valid(void) {return(BufferPtr.Is_Valid());}
