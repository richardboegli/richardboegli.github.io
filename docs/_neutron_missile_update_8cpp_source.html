168{

179 m_intermedPos = m_targetPos;

184 // Otherwise, we are just a Coord shot.
185 m_targetPos = *victimPos;
186 m_intermedPos = m_targetPos;

200void NeutronMissileUpdate::doLaunch( void )
201{
202 if (!m_isLaunched)
203 {
204 Object *launcher = TheGameLogic->findObjectByID( m_launcherID );
205
206 // if our launch vehicle is gone, destroy ourselves
207 if (launcher == NULL)
208 {
209 m_launcherID = INVALID_ID;
210 TheGameLogic->destroyObject( getObject() );
211 return;
212 }
213
214 Matrix3D attachTransform;
215 if (!launcher->getDrawable() ||
216 !launcher->getDrawable()->getProjectileLaunchOffset(m_attach_wslot, m_attach_specificBarrelToUse, &attachTransform, TURRET_INVALID, NULL))
217 {
218 DEBUG_CRASH(("ProjectileLaunchPos %d %d not found!\n",m_attach_wslot, m_attach_specificBarrelToUse));
219 attachTransform.Make_Identity();
220 }
221
222 Matrix3D worldTransform;
223 launcher->convertBonePosToWorldPos(NULL, &attachTransform, NULL, &worldTransform);
224
225 Vector3 tmp = worldTransform.Get_Translation();
226 Coord3D worldPos;
227 worldPos.x = tmp.X;
228 worldPos.y = tmp.Y;
229 worldPos.z = tmp.Z;
230
231 //
232 // the missile on the raising up launch platform is actually 45 degrees from the missile
233 // that is flying around the world ... we want to rotate it "on end and in place" so
234 // that we don't see any decals on the side of the missile 'pop' to the new angle
236 //
237 worldTransform.Rotate_X( (PI / 2.0f) );
238
239 getObject()->getDrawable()->setDrawableHidden(false);
240 getObject()->setTransformMatrix(&worldTransform);
241 getObject()->setPosition(&worldPos);
242
243 getObject()->getExperienceTracker()->setExperienceSink( m_launcherID );
244
245 m_isLaunched = true;
246
247 if (getNeutronMissileUpdateModuleData()->m_targetFromDirectlyAbove)
248 m_reachedIntermediatePos = false;
249
250 FXList::doFXObj(getNeutronMissileUpdateModuleData()->m_launchFX, getObject());
251 m_heightAtLaunch = getObject()->getPosition()->z;
252 m_frameAtLaunch = TheGameLogic->getFrame();
253
254 }
255
256 // fall
257 Coord3D pos = *getObject()->getPosition();
258
259 pos.x += m_vel.x;
260 pos.y += m_vel.y;
261 pos.z += m_vel.z;
262
263 getObject()->setPosition( &pos );
264
265 FXList::doFXObj(getNeutronMissileUpdateModuleData()->m_ignitionFX, getObject());
266
267 if (m_exhaustSysTmpl != NULL)
268 TheParticleSystemManager->createAttachedParticleSystemID(m_exhaustSysTmpl, getObject());
269
270 m_state = ATTACK;
271 m_stateTimestamp = TheGameLogic->getFrame();
272
273 // arm the missile's "warhead"
274 m_isArmed = true;
275}

277//-----------------------------------------------------------------------------
278// return the angle (in 3-space) we actually turned.
279static Real calcTransform(const Object* obj, const Coord3D *pos, Real maxTurnRate, Matrix3D* newTransform )
280{
281 // convert to Vector3, to use all its handy stuff
282 Vector3 objPos(obj->getPosition()->x, obj->getPosition()->y, obj->getPosition()->z);
283 Vector3 otherPos(pos->x, pos->y, pos->z);
284
285 Vector3 objDir = obj->getTransformMatrix()->Rotate_Vector(Vector3(1.0f, 0.0f, 0.0f));
286 Vector3 otherDir = otherPos - objPos;
287 otherDir.Normalize();
288
289 // dot of two unit vectors is cos of angle between them
290 Real c = Vector3::Dot_Product(objDir, otherDir);
291 // bound it in case of numerical error
292 if (c < -1.0)
293 c = -1.0;
294 else if (c > 1.0)
295 c = 1.0;
296
297 Real angle = (Real)ACos( c );
298
299 Vector3 newDir;
300 if (fabs(angle) < maxTurnRate)
301 {
302 // close enough -- point exactly in the right dir
303 newDir = otherDir;
304 }
305 else
306 {
307 // nah, turn as much as we can... this is icky, 'cuz
308 // we need to rotate around the axis perpendicular to these two vecs
309 angle = maxTurnRate;
310
311 // cross of two vectors is the perpendicular axis
312#ifdef ALLOW_TEMPORARIES
313 Vector3 objCrossOther = Vector3::Cross_Product(objDir, otherDir);
314 objCrossOther.Normalize();
315#else
316 Vector3 objCrossOther;
317 Vector3::Normalized_Cross_Product(objDir, otherDir, &objCrossOther);
318#endif
319
320 Matrix3D rotMtx(objCrossOther, angle);
321 newDir = rotMtx.Rotate_Vector(objDir);
322
323 }
324
325 newTransform->buildTransformMatrix( objPos, newDir );
326
327 return angle;
328}

334void NeutronMissileUpdate::doAttack( void )
335{
336 Matrix3D mx;
337 Real speed = getNeutronMissileUpdateModuleData()->m_relativeSpeed;
338
339 if (getNeutronMissileUpdateModuleData()->m_targetFromDirectlyAbove && m_reachedIntermediatePos)
340 speed *= STRAIGHT_DOWN_SLOW_FACTOR;
341
342 // if we're still in the no-turning-time, OR if we're out of fuel
343 if (m_noTurnDistLeft > 0.0f)
344 {
345 mx = *getObject()->getTransformMatrix();
346 }
347 else
348 {
349 {
350 //
351 // Orient toward destination
352 //
353 Real relAngle = calcTransform(getObject(), m_reachedIntermediatePos ? &m_targetPos : &m_intermedPos, getNeutronMissileUpdateModuleData()->m_maxTurnRate, &mx);
354
355 //
356 // Modulate speed according to turning. The more we have to turn, the slower we go
357 //
358 Real angleCoeff = (Real)fabs( relAngle ) / (PI / 2.0f);
359 if (angleCoeff > 1.0f)
360 angleCoeff = 1.0;
361 }
362 }
363
364 // get true forward direction of missile
365 Vector3 trueDir = mx.Get_X_Vector();
366 trueDir.Normalize();
367
368 //
369 // Move forward along forward direction
370 //
371 Real damping = getNeutronMissileUpdateModuleData()->m_forwardDamping;
372 m_accel.x = speed * trueDir.X - damping * m_vel.x;
373 m_accel.y = speed * trueDir.Y - damping * m_vel.y;
374 m_accel.z = speed * trueDir.Z - damping * m_vel.z;
375
376 m_vel.x += m_accel.x;
377 m_vel.y += m_accel.y;
378 m_vel.z += m_accel.z;
379
380 Coord3D pos = *getObject()->getPosition();
381
382 const NeutronMissileUpdateModuleData* d = getNeutronMissileUpdateModuleData();
383 UnsignedInt now = TheGameLogic->getFrame();
384 if (d->m_specialSpeedTime > 0 && now <= m_frameAtLaunch + d->m_specialSpeedTime)
385 {
386 getObject()->getDrawable()->setInstanceMatrix(NULL);
387 UnsignedInt elapsed = now - m_frameAtLaunch;
388 if (elapsed < d->m_specialSpeedTime)
389 {
390 Real timeFrac = (Real)elapsed / (Real)d->m_specialSpeedTime;
391 Real accelFactor = d->m_specialAccelFactor;
392 if (accelFactor < 0.01f) accelFactor = 0.01f;
393 Coord3D newPos = pos;
394 newPos.z = m_heightAtLaunch + (sqr(accelFactor * timeFrac) / accelFactor) * d->m_specialSpeedHeight;
395
396 m_vel.x = newPos.x - pos.x;
397 m_vel.y = newPos.y - pos.y;
398 m_vel.z = newPos.z - pos.z;
399
400 if (d->m_specialJitterDistance > 0.0f)
401 {
402 Real amplitude = (1.0f - timeFrac) * d->m_specialJitterDistance;
403 Vector3 vectmp;
404 vectmp.X = 0;
405 // MDC: moving to GameLogicRandomValue. This does not need to be synced, but having it so makes searches *so* much nicer.
406 vectmp.Y = GameLogicRandomValueReal(-1.0, 1.0) * amplitude;
407 vectmp.Z = GameLogicRandomValueReal(-1.0, 1.0) * amplitude;
408 vectmp = mx.Rotate_Vector(vectmp);
409 Matrix3D mtxtmp(1);
410 mtxtmp.Translate(vectmp);
411 getObject()->getDrawable()->setInstanceMatrix( &mtxtmp );
412 }
413 }
414 }
415
416 pos.x += m_vel.x;
417 pos.y += m_vel.y;
418 pos.z += m_vel.z;
419
420//DEBUG_LOG(("vel %f accel %f z %f\n",m_vel.length(),m_accel.length(), pos.z));
421//Real vm = sqrt(m_vel.x*m_vel.x+m_vel.y*m_vel.y+m_vel.z*m_vel.z);
422//DEBUG_LOG(("vel is %f %f %f (%f)\n",m_vel.x,m_vel.y,m_vel.z,vm));
423 getObject()->setTransformMatrix( &mx );
424 getObject()->setPosition( &pos );
425
426}

430Bool NeutronMissileUpdate::projectileHandleCollision( Object *other )
431{
432 Object* obj = getObject();
433
434 // check if our warhead is "armed" - if not, we are inert
435 if (projectileIsArmed() == false)
436 return true;
437
438 // Don't hit your own launcher, ever.
439 if (other != NULL && projectileGetLauncherID() == other->getID())
440 return true;
441
442 // collided with something... blow'd up!
443 detonate();
444
445 // mark ourself as "no collisions" (since we might still exist in slow death mode)
446 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
447 return true;
448}

472UpdateSleepTime NeutronMissileUpdate::update( void )
473{
474 m_deliveryDecal.update();
475
476 if (!m_reachedIntermediatePos)
477 {
478 Real distSqr = ThePartitionManager->getDistanceSquared(getObject(), &m_intermedPos, FROM_CENTER_3D);
479 Real boundSqr = sqr(getObject()->getGeometryInfo().getBoundingSphereRadius());
480 if (distSqr <= boundSqr)
481 {
482 m_reachedIntermediatePos = true;
483 getObject()->setPosition(&m_intermedPos);
484 Real vel = m_vel.length();
485 m_vel.x = 0;
486 m_vel.y = 0;
487 m_vel.z = -vel * STRAIGHT_DOWN_SLOW_FACTOR;
488
489 }
490 }
491
492 Coord3D oldPos = *getObject()->getPosition();
493 Bool oldPosValid = (m_state == ATTACK); // not valid till *after* we've launched
494 switch( m_state )
495 {
496 case PRELAUNCH:
497 // nothing... just ignore it.
498 break;
499
500 case LAUNCH:
501 doLaunch();
502 break;
503
504 case ATTACK:
505 doAttack();
506 break;
507
508 case DEAD:
509 // do nothing
510 break;
511 }
512 if (m_noTurnDistLeft > 0.0f && oldPosValid)
513 {
514 Coord3D newPos = *getObject()->getPosition();
515 Real distThisTurn = sqrt(sqr(newPos.x-oldPos.x) + sqr(newPos.y-oldPos.y) + sqr(newPos.z-oldPos.z));
516 //DEBUG_LOG(("noTurnDist goes from %f to %f\n",m_noTurnDistLeft,m_noTurnDistLeft-distThisTurn));
517 m_noTurnDistLeft -= distThisTurn;
518 }
519
520 if (m_state != PRELAUNCH && m_state != DEAD && !getObject()->isAboveTerrain())
521 {
522 // the normal always points straight down, though we could
523 // get the alignWithTerrain() normal if it proves interesting
524 Coord3D normal;
525 normal.x = normal.y = 0.0f;
526 normal.z = -1.0f;
527 getObject()->onCollide(NULL, getObject()->getPosition(), &normal);
528 }
529 return UPDATE_SLEEP_NONE;
530}
