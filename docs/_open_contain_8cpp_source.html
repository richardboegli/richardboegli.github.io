223void OpenContain::addOrRemoveObjFromWorld(Object* obj, Bool add)
224{
225
226 //
227 // it's expensive to add and remove structures from the pathfinder, but we really shouldn't
228 // be doing it anyway as it doesn't make much sense to contain a structure, but we'll
229 // check for it here and print a warning
230 //
231 if( obj->isKindOf( KINDOF_STRUCTURE ) )
232 DEBUG_LOG(( "WARNING: Containing/Removing structures like '%s' is potentially a very expensive and slow operation\n",
233 obj->getTemplate()->getName().str() ));
234
235
236 if (add)
237 {
238 ThePartitionManager->registerObject( obj );
239
240 if( obj->getDrawable() )
241 {
242 obj->getDrawable()->setDrawableHidden( false );
243 }
244
245 // add object to pathfind map
246 TheAI->pathfinder()->addObjectToPathfindMap( obj );
247
248 }
249 else
250 {
251 // remove object from its group (if any)
252 obj->leaveGroup();
253
254 // remove rider from partition manager
255 ThePartitionManager->unRegisterObject( obj );
256
257 // hide the drawable associated with rider
258 if( obj->getDrawable() )
259 obj->getDrawable()->setDrawableHidden( true );
260
261 // remove object from pathfind map
262 TheAI->pathfinder()->removeObjectFromPathfindMap( obj );
263
264 }
265
266 // if we're a non-enclosing container (eg, parachute), and we're inside an
267 // enclosing container (eg, b52), we need to be able to show/hide the contained riders...
268 if( obj->getContain() )
269 {
270 const ContainedItemsList* items = obj->getContain()->getContainedItemsList();
271 if (items)
272 {
273 for(ContainedItemsList::const_iterator it = items->begin(); it != items->end(); ++it)
274 {
275 if( !obj->getContain()->isEnclosingContainerFor(*it) )
276 addOrRemoveObjFromWorld(*it, add);
277 }
278 }
279 }
280}

287void OpenContain::addToContain( Object *rider )
288{
289 if( getObject()->checkAndDetonateBoobyTrap(rider) )
290 {
291 // Whoops, I was mined. Cancel if I (or they) am now dead.
292 if( getObject()->isEffectivelyDead() || rider->isEffectivelyDead() )
293 {
294 return;
295 }
296 }
297
298 // sanity
299 if( rider == NULL )
300 return;
301
302 Drawable *riderDraw = rider->getDrawable();
303 Bool wasSelected = FALSE;
304 if( riderDraw && riderDraw->isSelected() )
305 {
306 wasSelected = TRUE;
307 }
308
309#if defined(_DEBUG) || defined(_INTERNAL)
310 if( !isValidContainerFor( rider, false ) )
311 {
312 Object *reportObject = rider;
313 if( rider->getContain() )
314 {
315 //Report the first thing inside it!
316 const ContainedItemsList *items = rider->getContain()->getContainedItemsList();
317 if( items )
318 {
319 reportObject = *items->begin();
320 }
321 }
322 DEBUG_CRASH( ("OpenContain::addToContain() - Object %s not valid for container %s!", reportObject?reportObject->getTemplate()->getName().str():"NULL", getObject()->getTemplate()->getName().str() ) );
323 }
324#endif
325
326 // this object cannot be contained by this module if it is already contained in something
327 if( rider->getContainedBy() != NULL )
328 {
329
330 DEBUG_LOG(( "'%s' is trying to contain '%s', but '%s' is already contained by '%s'\n",
331 getObject()->getTemplate()->getName().str(),
332 rider->getTemplate()->getName().str(),
333 rider->getTemplate()->getName().str(),
334 rider->getContainedBy()->getTemplate()->getName().str() ));
335 return;
336
337 }
338
339 // Which list to physically add to needs to be overridable
340 addToContainList(rider);
341
342 m_playerEnteredMask = rider->getControllingPlayer()->getPlayerMask();
343
344 if (isEnclosingContainerFor( rider ))
345 {
346 addOrRemoveObjFromWorld(rider, false);
347 }
348
349 // ensure our contents are positions correctly.
350 redeployOccupants();
351
352 // trigger an onContaining event for the object that just "ate" something
353 if( getObject()->getContain() )
354 {
355 getObject()->getContain()->onContaining( rider, wasSelected );
356 }
357
358 // trigger an onContainedBy event for the object that just got "eaten" by us
359 rider->onContainedBy( getObject() );
360
361 doLoadSound();
362
363}

382void OpenContain::removeFromContain( Object *rider, Bool exposeStealthUnits )
383{
384
385 // sanity
386 if( rider == NULL )
387 return;
388
389 //
390 // we can only remove this object from the contains list of this module if
391 // it is actually contained by this module
392 //
393 Object *containedBy = rider->getContainedBy();
394 if( containedBy != getObject() )
395 {
396
397 DEBUG_LOG(( "'%s' is trying to un-contain '%s', but '%s' is really contained by '%s'\n",
398 getObject()->getTemplate()->getName().str(),
399 rider->getTemplate()->getName().str(),
400 rider->getTemplate()->getName().str(),
401 containedBy ? containedBy->getTemplate()->getName().str() : "Nothing" ));
402 return;
403
404 }
405
406 ContainedItemsList::iterator it = std::find(m_containList.begin(), m_containList.end(), rider);
407 if (it != m_containList.end())
408 {
409 // note that this invalidates the iterator!
410 removeFromContainViaIterator( it, exposeStealthUnits );
411 }
412
413}

435void OpenContain::killAllContained( void )
436{
437 ContainedItemsList::iterator it = m_containList.begin();
438
439 while ( it != m_containList.end() )
440 {
441 Object *rider = *it;
442
443
444 if ( rider )
445 {
446 it = m_containList.erase(it);
447 m_containListSize--;
448
449 onRemoving( rider );
450 rider->onRemovedFrom( getObject() );
451 rider->kill();
452
453 }
454 else
455 ++it;
456
457 } // end while
458
459
460 DEBUG_ASSERTCRASH( m_containListSize == 0, ("killallcontain just made a booboo, list size != zero.") );

630void OpenContain::removeFromContainViaIterator( ContainedItemsList::iterator it, Bool exposeStealthUnits )
631{
632
633/*
634 #ifdef _DEBUG
635 TheInGameUI->message( UnicodeString( L"'%S(%d)' no longer contains '%S(%d)'" ),
636 getObject()->getTemplate()->getName().str(),
637 getObject()->getID(),
638 itemToRemove->m_object->getTemplate()->getName().str(),
639 itemToRemove->m_object->getID() );
640 #endif
641*/
642 Object *rider = *it;
643
644 // remove item from list
645 m_containList.erase(it);
646 m_containListSize--;
647 if( rider->isKindOf( KINDOF_STEALTH_GARRISON ) )
648 {
649 m_stealthUnitsContained--;
650 if( exposeStealthUnits )
651 {
652 StealthUpdate* stealth = rider->getStealth();
653 if( stealth )
654 {
655 stealth->markAsDetected();
656 }
657 }
658 }
659
660
661 if (isEnclosingContainerFor( rider ))
662 {
663 addOrRemoveObjFromWorld(rider, true);
664 rider->setPosition( getObject()->getPosition() );
665 // if we are not enclosed, then just walk away from where we "are."
666
667 }
668
670 rider->setLayer( getObject()->getLayer() );
671
672 doUnloadSound();
673
674 // trigger an onRemoving event for 'm_object' no longer containing 'itemToRemove->m_object'
675 DEBUG_ASSERTCRASH(getObject()->getContain() == this, ("hmm, wrong container"));
676 if( getObject()->getContain() )
677 {
678 getObject()->getContain()->onRemoving( rider );
679 }
680
681 // trigger an onRemovedFrom event for 'remove'
682 DEBUG_ASSERTCRASH(getObject()->getContain() == this, ("hmm, wrong container 2"));
683 rider->onRemovedFrom( getObject() );
684
685}

687//-------------------------------------------------------------------------------------------------
688void OpenContain::scatterToNearbyPosition(Object* rider)
689{
690 Object *theContainer = getObject();
691
692 //
693 // for now we will just set the position of the object that is being removed from us
694 // at a random angle away from our center out some distance
695 //
696
697 //
698 // pick an angle that is in the view of the current camera position so that
699 // the thing will come out "toward" the player and they can see it
700 // NOPE, can't do that ... all players screen angles will be different, unless
701 // we maintain the angle of each players screen in the player structure or something
702 //
703 Real angle = GameLogicRandomValueReal( 0.0f, 2.0f * PI );
704// angle = TheTacticalView->getAngle();
705// angle -= GameLogicRandomValueReal( PI / 3.0f, 2.0f * (PI / 3.0F) );
706
707 Real minRadius = theContainer->getGeometryInfo().getBoundingCircleRadius();
708 Real maxRadius = minRadius + minRadius / 2.0f;
709 const Coord3D *containerPos = theContainer->getPosition();
710 Real dist = GameLogicRandomValueReal( minRadius, maxRadius );
711
712 Coord3D pos;
713 pos.x = dist * Cos( angle ) + containerPos->x;
714 pos.y = dist * Sin( angle ) + containerPos->y;
715 pos.z = TheTerrainLogic->getLayerHeight( pos.x, pos.y, theContainer->getLayer() );
716
717 // set orientation
718 rider->setOrientation( angle );
719
720 AIUpdateInterface *ai = rider->getAI();
721 if( ai )
722 {
723 // set position of the object at center of building and move them toward pos
724 rider->setPosition( theContainer->getPosition() );
725 ai->ignoreObstacle(theContainer);
726 ai->aiMoveToPosition( &pos, CMD_FROM_AI );
727
728 } // end if
729 else
730 {
731
732 // no ai, just set position at the target pos
733 rider->setPosition( &pos );
734
735 } // end else
736}

782void OpenContain::onCollide( Object *other, const Coord3D *loc, const Coord3D *normal )
783{
784 // colliding with nothing? we don't care.
785 if( other == NULL )
786 return;
787
788 // ok, step two: only contain stuff that wants us to contain it.
789 // (it would suck to have accidental collisions contain things...)
790
791 // must be an AI object....
792 AIUpdateInterface *ai = other->getAI();
793 if (ai == NULL)
794 return;
795
796 // ...and must be trying to enter our object.

918 obj->getTemplate()->getName().str() ));
