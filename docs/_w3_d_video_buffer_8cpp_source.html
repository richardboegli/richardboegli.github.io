117Bool W3DVideoBuffer::allocate( UnsignedInt width, UnsignedInt height )
118{
119 free();
120
121 m_width = width;
122 m_height = height;
123 m_textureWidth = width;;
124 m_textureHeight = height;;
125 unsigned int temp_depth=1;
126 TextureLoader::Validate_Texture_Size( m_textureWidth, m_textureHeight, temp_depth);
127
128 WW3DFormat w3dFormat = TypeToW3DFormat( m_format );
129
130 if ( w3dFormat == WW3D_FORMAT_UNKNOWN )
131 {
132 return NULL;
133 }
134
135 m_texture = MSGNEW("TextureClass") TextureClass ( m_textureWidth, m_textureHeight, w3dFormat, MIP_LEVELS_1 );
136
137 if ( m_texture == NULL )
138 {
139 return FALSE;
140 }
141
142 if ( lock() == NULL )
143 {
144 free();
145 return FALSE;
146 }
147
148 unlock();
149
150
151 return TRUE;
152}

167void* W3DVideoBuffer::lock( void )
168{
169 void *mem = NULL;
170
171 if ( m_surface != NULL )
172 {
173 unlock();
174 }
175
176 m_surface = m_texture->Get_Surface_Level();
177
178 if ( m_surface )
179 {
180 mem = m_surface->Lock( (Int*) &m_pitch );
181 }
182
183 return mem;
184}
