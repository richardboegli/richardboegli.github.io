47/*********************************************************************
48
49 There are 24 possible conventions for Euler angles. They can
50 be designated by:
51
52 EulerAxis = Axis used initially
53 EulerParity = parity of axis permutation (even = x,y,z)
54 EulerRepeat = is last axis a repeat of the initial axis?
55 EulerFrame = frame from which axes are taken (rotating or static)
56
57 For example, in this system an euler angle convention of:
58
59 Rotate_X(a0);
60 Rotate_Y(a1);
61 Rotate_Z(a2);
62
63 would be described as EulerOrderXYZr.
64
65*********************************************************************/

163/***********************************************************************************************
164 * EulerAnglesClass::From_Matrix -- computes the equivalent euler angles for the given matrix *
165 * *
166 * INPUT: *
167 * *
168 * OUTPUT: *
169 * *
170 * WARNINGS: *
171 * *
172 * HISTORY: *
173 * 08/11/1997 GH : Created. *
174 *=============================================================================================*/
175void EulerAnglesClass::From_Matrix(const Matrix3D & M, int order)
176{
177 int i,j,k,h,n,s,f;
178
179 Order = order;
180 _euler_unpack_order(order,i,j,k,h,n,s,f);
181
182 if (s == EULER_REPEAT_YES) {
183 double sy = sqrt(M[i][j]*M[i][j] + M[i][k]*M[i][k]);
184
185 if (sy > 16*FLT_EPSILON) {
186
187 Angle[0] = WWMath::Atan2(M[i][j],M[i][k]);
188 Angle[1] = WWMath::Atan2(sy,M[i][i]);
189 Angle[2] = WWMath::Atan2(M[j][i],-M[k][i]);
190
191 } else {
192
193 Angle[0] = WWMath::Atan2(-M[j][k],M[j][j]);
194 Angle[1] = WWMath::Atan2(sy,M[i][i]);
195 Angle[2] = 0.0;
196 }
197
198 } else {
199
200 double cy = sqrt(M[i][i]*M[i][i] + M[j][i]*M[j][i]);
201
202 if (cy > 16*FLT_EPSILON) {
203
204 Angle[0] = WWMath::Atan2(M[k][j],M[k][k]);
205 Angle[1] = WWMath::Atan2(-M[k][i],cy);
206 Angle[2] = WWMath::Atan2(M[j][i],M[i][i]);
207
208 } else {
209
210 Angle[0] = WWMath::Atan2(-M[j][k],M[j][j]);
211 Angle[1] = WWMath::Atan2(-M[k][i],cy);
212 Angle[2] = 0;
213 }
214 }
215
216 if (n==EULER_PARITY_ODD) { Angle[0] = -Angle[0]; Angle[1] = -Angle[1]; Angle[2] = -Angle[2]; }
217 if (f==EULER_FRAME_ROTATING) { double t = Angle[0]; Angle[0] = Angle[2]; Angle[2] = t; }
218
219 // Trying to "clean" up the eulers, special cased for XYZr
220 if (order == EulerOrderXYZr) {
221
222 const double PI = 3.141592654;
223
224 double x2 = PI + Angle[0];
225 double y2 = PI - Angle[1];
226 double z2 = PI + Angle[2];
227
228 if (x2 > PI) {
229 x2 = x2 - 2*PI;
230 }
231
232 if (y2 > PI) {
233 y2 = y2 - 2*PI;
234 }
235
236 if (z2 > PI) {
237 z2 = z2 - 2*PI;
238 }
239
240 double mag0 = Angle[0]*Angle[0] + Angle[1]*Angle[1] + Angle[2]*Angle[2];
241 double mag1 = x2*x2 + y2*y2 + z2*z2;
242
243 if (mag1 < mag0) {
244 Angle[0] = x2;
245 Angle[1] = y2;
246 Angle[2] = z2;
247 }
248 }
249}

253/***********************************************************************************************
254 * EulerAnglesClass::To_Matrix -- Builds a matrix from the given euler angles *
255 * *
256 * INPUT: *
257 * *
258 * OUTPUT: *
259 * *
260 * WARNINGS: *
261 * *
262 * HISTORY: *
263 * 08/11/1997 GH : Created. *
264 *=============================================================================================*/
265void EulerAnglesClass::To_Matrix(Matrix3D & M)
266{
267 M.Make_Identity();
268
269 double a0,a1,a2;
270 double ti,tj,th,ci,cj,ch,si,sj,sh,cc,cs,sc,ss;
271 int i,j,k,h,n,s,f;
272
273 a0 = Angle[0];
274 a1 = Angle[1];
275 a2 = Angle[2];
276
277 _euler_unpack_order(Order,i,j,k,h,n,s,f);
278 if (f == EULER_FRAME_ROTATING) {
279 double t = a0; a0 = a2; a2 = t;
280 }
281
282 if (n == EULER_PARITY_ODD) {
283 a0 = -a0; a1 = -a1; a2 = -a2;
284 }
285
286 ti = a0; tj = a1; th = a2;
287 ci = WWMath::Cos(ti); cj = WWMath::Cos(tj); ch = WWMath::Cos(th);
288 si = WWMath::Sin(ti); sj = WWMath::Sin(tj); sh = WWMath::Sin(th);
289
290 cc = ci*ch;
291 cs = ci*sh;
292 sc = si*ch;
293 ss = si*sh;
294
295 if (s == EULER_REPEAT_YES) {
296
297 M[i][i] = (float)(cj); M[i][j] = (float)(sj*si); M[i][k] = (float)(sj*ci);
298 M[j][i] = (float)(sj*sh); M[j][j] = (float)(-cj*ss+cc); M[j][k] = (float)(-cj*cs-sc);
299 M[k][i] = (float)(-sj*ch); M[k][j] = (float)(cj*sc+cs); M[k][k] = (float)(cj*cc-ss);
300
301 } else {
302
303 M[i][i] = (float)(cj*ch); M[i][j] = (float)(sj*sc-cs); M[i][k] = (float)(sj*cc+ss);
304 M[j][i] = (float)(cj*sh); M[j][j] = (float)(sj*ss+cc); M[j][k] = (float)(sj*cs-sc);
305 M[k][i] = (float)(-sj); M[k][j] = (float)(cj*si); M[k][k] = (float)(cj*ci);
306
307 }
308}

323int _euler_axis_j(int order)

341}

343int _euler_axis_h(int order)
344{
345 if (EULER_REPEAT(order) == 1) {
346 return _euler_axis_k(order);
347 } else {
348 return _euler_axis_i(order);
349 }

352void _euler_unpack_order(int order,int &i,int &j,int &k,int &h,int &n,int &s,int &f)
353{
354
355 f = order & 1;
356 order >>= 1;
357
358 s = order & 1;
359 order >>= 1;
360
361 n = order & 1;
362 order >>= 1;
363
364 i = _euler_safe[order & 3];
365 j = _euler_next[i+n];
366 k = _euler_next[i+1-n];
367 h = (s ? k : i);
368}
