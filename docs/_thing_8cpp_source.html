100const Coord3D* Thing::getUnitDirectionVector2D() const
101{
102 //USE_PERF_TIMER(ThingMatrixStuff)
103 if (!(m_cacheFlags & VALID_DIRVECTOR))
104 {
105 Real angle = getOrientation();
106 m_cachedDirVector.x = Cos( angle );
107 m_cachedDirVector.y = Sin( angle );
108 m_cachedDirVector.z = 0;
109 m_cacheFlags |= VALID_DIRVECTOR;
110 }
111
112 return &m_cachedDirVector;
113}

132// the nice thing about this is that we don't have to recalc out cached terrain stuff.
133void Thing::setPositionZ( Real z )
134{
135 //USE_PERF_TIMER(ThingMatrixStuff)
136 if( !m_template->isKindOf( KINDOF_STICK_TO_TERRAIN_SLOPE) )
137 {
138 Real oldAngle = m_cachedAngle;
139 Coord3D oldPos = m_cachedPos;
140 Matrix3D oldMtx = m_transform;
141
142 m_transform.Set_Z_Translation( z );
143 m_cachedPos.z = z;
144
145 if (m_cacheFlags & VALID_ALTITUDE_TERRAIN)
146 {
147 m_cachedAltitudeAboveTerrain += (z - oldPos.z);
148 }
149 if (m_cacheFlags & VALID_ALTITUDE_SEALEVEL)
150 {
151 m_cachedAltitudeAboveTerrainOrWater += (z - oldPos.z);
152 }
153
154 reactToTransformChange(&oldMtx, &oldPos, oldAngle);
155 }
156 else
157 {
158 Matrix3D mtx;
159 const Bool stickToGround = true; // yes, set the "z" pos
160 Coord3D pos = m_cachedPos;
161 pos.z = z;
162 TheTerrainLogic->alignOnTerrain(getOrientation(), pos, stickToGround, mtx );
163 setTransformMatrix(&mtx);
164 }
165 DEBUG_ASSERTCRASH(!(_isnan(getPosition()->x) || _isnan(getPosition()->y) || _isnan(getPosition()->z)), ("Drawable/Object position NAN! '%s'\n", m_template->getName().str() ));
166}

198void Thing::setOrientation( Real angle )
199{
200 //USE_PERF_TIMER(ThingMatrixStuff)
201 Coord3D u, x, y, z, pos;
202
203 // setOrientation always forces us straight up in the Z axis,
204 // or aligned with the terrain if we have the magic flag set.
205 // don't want this? call setTransformMatrix instead.
206
207 Real oldAngle = m_cachedAngle;
208 Coord3D oldPos = m_cachedPos;
209 Matrix3D oldMtx = m_transform;
210
211 pos.x = m_transform.Get_X_Translation();
212 pos.y = m_transform.Get_Y_Translation();
213 pos.z = m_transform.Get_Z_Translation();
214 if( m_template->isKindOf( KINDOF_STICK_TO_TERRAIN_SLOPE) )
215 {
216 Matrix3D mtx;
217 const Bool stickToGround = true; // yes, set the "z" pos
218 TheTerrainLogic->alignOnTerrain(angle, pos, stickToGround, m_transform );
219 }
220 else
221 {
222 z.x = 0.0f;
223 z.y = 0.0f;
224 z.z = 1.0f;
225
226 u.x = Cos(angle);
227 u.y = Sin(angle);
228 u.z = 0.0f;
229
230 y.crossProduct( &z, &u, &y );
231 x.crossProduct( &y, &z, &x );
232
233 m_transform.Set( x.x, y.x, z.x, pos.x,
234 x.y, y.y, z.y, pos.y,
235 x.z, y.z, z.z, pos.z );
236 }
237
238 //DEBUG_ASSERTCRASH(-PI <= angle && angle <= PI, ("Please pass only normalized (-PI..PI) angles to setOrientation (%f).\n", angle));
239 m_cachedAngle = normalizeAngle(angle);
240 m_cachedPos = pos;
241 m_cacheFlags &= ~VALID_DIRVECTOR; // but don't clear the altitude flags.
242
243 reactToTransformChange(&oldMtx, &oldPos, oldAngle);
244 DEBUG_ASSERTCRASH(!(_isnan(getPosition()->x) || _isnan(getPosition()->y) || _isnan(getPosition()->z)), ("Drawable/Object position NAN! '%s'\n", m_template->getName().str() ));
245}

250void Thing::setTransformMatrix( const Matrix3D *mx )
251{
252 //USE_PERF_TIMER(ThingMatrixStuff)
253 Real oldAngle = m_cachedAngle;
254 Coord3D oldPos = m_cachedPos;
255 Matrix3D oldMtx = m_transform;
256
257 m_transform = *mx;
258 m_cachedPos.x = m_transform.Get_X_Translation();
259 m_cachedPos.y = m_transform.Get_Y_Translation();
260 m_cachedPos.z = m_transform.Get_Z_Translation();
261 m_cachedAngle = m_transform.Get_Z_Rotation();
262 m_cacheFlags = 0;
263
264 reactToTransformChange(&oldMtx, &oldPos, oldAngle);
265 DEBUG_ASSERTCRASH(!(_isnan(getPosition()->x) || _isnan(getPosition()->y) || _isnan(getPosition()->z)), ("Drawable/Object position NAN! '%s'\n", m_template->getName().str() ));
266}

287Real Thing::calculateHeightAboveTerrain() const
288{
289 //USE_PERF_TIMER(ThingMatrixStuff)
290 const Coord3D* pos = getPosition();
291 Real terrainZ = TheTerrainLogic->getGroundHeight( pos->x, pos->y );
292 Real myZ = pos->z;
293 return myZ - terrainZ;
294}
295
296//-------------------------------------------------------------------------------------------------
297Real Thing::getHeightAboveTerrain() const
298{
299 if (!(m_cacheFlags & VALID_ALTITUDE_TERRAIN))
300 {
301 m_cachedAltitudeAboveTerrain = calculateHeightAboveTerrain();
302 m_cacheFlags |= VALID_ALTITUDE_TERRAIN;
303 }
304 return m_cachedAltitudeAboveTerrain;
305}
306
307//-------------------------------------------------------------------------------------------------
308Real Thing::getHeightAboveTerrainOrWater() const
309{
310 //USE_PERF_TIMER(ThingMatrixStuff)
311 if (!(m_cacheFlags & VALID_ALTITUDE_SEALEVEL))
312 {
313 const Coord3D* pos = getPosition();
314 Real waterZ;
315 if (TheTerrainLogic->isUnderwater(pos->x, pos->y, &waterZ))
316 {
317 m_cachedAltitudeAboveTerrainOrWater = pos->z - waterZ;
318 }
319 else
320 {
321 m_cachedAltitudeAboveTerrainOrWater = getHeightAboveTerrain();
322 }
323 m_cacheFlags |= VALID_ALTITUDE_SEALEVEL;
324 }
325 return m_cachedAltitudeAboveTerrainOrWater;
326}

332Bool Thing::isSignificantlyAboveTerrain() const
333{
334 // If it's high enough that it will take more than 3 frames to return to the ground,
335 // then it's significantly airborne. jba
336 return (getHeightAboveTerrain() > -(3*3)*TheGlobalData->m_gravity);
337}

341void Thing::convertBonePosToWorldPos(const Coord3D* bonePos, const Matrix3D* boneTransform, Coord3D* worldPos, Matrix3D* worldTransform) const
342{
343 if (worldTransform)
344 {
345#ifdef ALLOW_TEMPORARIES
346 *worldTransform = m_transform * (*boneTransform);
347#else
348 worldTransform->mul(m_transform, *boneTransform);
349#endif
350 }
351 if (worldPos)
352 {
353 Vector3 vector;
354 vector.X = bonePos->x;
355 vector.Y = bonePos->y;
356 vector.Z = bonePos->z;
357 m_transform.Transform_Vector(m_transform, vector, &vector);
358 worldPos->x = vector.X;
359 worldPos->y = vector.Y;
360 worldPos->z = vector.Z;
361 }
362}

367void Thing::transformPoint( const Coord3D *in, Coord3D *out )
368{
369
370 // santiy

372 return;

377

383
384 // do the transform

386
387 // store converted vector in 'out'
388 out->x = vectorOut.X;
389 out->y = vectorOut.Y;
390 out->z = vectorOut.Z;
391
392} // end transformPoint
