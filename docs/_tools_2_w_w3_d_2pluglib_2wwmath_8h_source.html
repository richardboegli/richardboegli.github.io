122inline float WWMath::Sign(float val)
123{

125 return +1.0f;

157inline float WWMath::Wrap(float val, float min /*= 0.0f*/, float max /*= 1.0f*/)
158{
159 // Implemented as an if rather than a while, to long loops
160 if ( val >= max ) val -= (max-min);
161 if ( val < min ) val += (max-min);
162
163 if ( val < min ) {
164 val = min;
165 }
166 if ( val > max ) {
167 val = max;
168 }
169 return val;
170}

233inline bool WWMath::Is_Valid_Float(float x)
234{
235 unsigned long * plong = (unsigned long *)(&x);
236 unsigned long exponent = ((*plong) & 0x7F800000) >> (32-9);
237
238 // if exponent is 0xFF, this is a NAN
239 if (exponent == 0xFF) {
240 return false;
241 }
242 return true;
243}

245inline bool WWMath::Is_Valid_Double(double x)
246{
247 unsigned long * plong = (unsigned long *)(&x) + 1;
248 unsigned long exponent = ((*plong) & 0x7FF00000) >> (32-12);
249
250 // if exponent is 0x7FF, this is a NAN
251 if (exponent == 0x7FF) {
252 return false;
253 }
254 return true;
255}
