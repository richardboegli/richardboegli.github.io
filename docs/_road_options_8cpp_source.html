119void RoadOptions::updateSelection(void)
120{
121// MapObject *theMapObj = NULL;
122 Int angled = 0;
123 Int tight = 0;
124 Int broad = 0;
125 Int join = 0;
126 AsciiString roadName;
127 Bool multipleNames = false;
128
129 if (!m_staticThis) return;
130 MapObject *pMapObj;
131
132 for (pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext()) {
133 if (pMapObj->isSelected() && pMapObj->getFlag(FLAG_ROAD_FLAGS)) {
134 if (roadName.isEmpty()) {
135 roadName = pMapObj->getName();
136 } else {
137 if (!(roadName==pMapObj->getName())) {
138 multipleNames = true;
139 }
140 }
141 if (pMapObj->getFlag(FLAG_ROAD_CORNER_ANGLED)) {
142 angled = 1;
143 } else if (pMapObj->getFlag(FLAG_ROAD_CORNER_TIGHT)) {
144 tight = 1;
145 } else {
146 broad = 1;
147 }
148 if (pMapObj->getFlag(FLAG_ROAD_JOIN)) {
149 join = 1;
150 }
151 }
152 }
153 if (!roadName.isEmpty() && !multipleNames) {
154 CWnd *pLabel = m_staticThis->GetDlgItem(IDC_ROAD_NAME);
155 if (pLabel) {
156 pLabel->SetWindowText(roadName.str());
157 m_staticThis->findAndSelect(TVI_ROOT, roadName);
158 }
159 } else {
160 m_staticThis->setRoadTreeViewSelection(TVI_ROOT, m_currentRoadIndex);
161 }
162 if (angled+broad+tight==0) {
163 // nothing selected
164 if (m_angleCorners) {
165 angled = 1;
166 } else if (m_tightCurve) {
167 tight = 1;
168 } else {
169 broad = 1;
170 }
171 if (m_doJoin) {
172 join = 1;
173 }
174 } else if (angled+broad+tight==1) {
175 // One type selected
176 } else {
177 angled = tight = broad = join = 0;
178 }
179 if (m_staticThis) {
180 CButton *pButton = (CButton *)m_staticThis->GetDlgItem(IDC_TIGHT_CURVE);
181 pButton->SetCheck(tight);
182 pButton = (CButton *)m_staticThis->GetDlgItem(IDC_BROAD_CURVE);
183 pButton->SetCheck(broad);
184 pButton = (CButton *)m_staticThis->GetDlgItem(IDC_ANGLED);
185 pButton->SetCheck(angled);
186 pButton = (CButton *)m_staticThis->GetDlgItem(IDC_JOIN);
187 pButton->SetCheck(join);
188 }
189}

212BOOL RoadOptions::OnInitDialog()
213{
214 CDialog::OnInitDialog();
215
216 m_updating = true;
217
218 CWnd *pWnd = GetDlgItem(IDC_ROAD_TREEVIEW);
219 CRect rect;
220 pWnd->GetWindowRect(&rect);
221
222 ScreenToClient(&rect);
223 rect.DeflateRect(2,2,2,2);
224 m_roadTreeView.Create(TVS_HASLINES|TVS_LINESATROOT|TVS_HASBUTTONS|
225 TVS_SHOWSELALWAYS|TVS_DISABLEDRAGDROP, rect, this, IDC_ROAD_TREEVIEW);
226 m_roadTreeView.ShowWindow(SW_SHOW);
227
228 Int index = 0;
229 m_numberOfRoads = 0;
230
231 // load roads from INI
232 TerrainRoadType *road;
233 for( road = TheTerrainRoads->firstRoad(); road; road = TheTerrainRoads->nextRoad( road ) )
234 {
235
236 addRoad( (char *)road->getName().str(), index, TVI_ROOT );
237 index++;
238 m_numberOfRoads++;
239
240 } // end for raod
241
242 // load roads from test assets
243#ifdef LOAD_TEST_ASSETS
244 {
245 char dirBuf[_MAX_PATH];
246 char findBuf[_MAX_PATH];
247 char fileBuf[_MAX_PATH];
248
249 strcpy(dirBuf, ROAD_DIRECTORY);
250 int len = strlen(dirBuf);
251
252 strcpy(findBuf, dirBuf);
253
254 FilenameList filenameList;
255 TheFileSystem->getFileListInDirectory(AsciiString(findBuf), AsciiString("*.tga"), filenameList, FALSE);
256
257 if (filenameList.size() > 0) {
258 FilenameList::iterator it = filenameList.begin();
259 do {
260 AsciiString filename = *it;
261
262 if ((filename.compare(".") == 0) || (filename.compare("..") == 0)) {
263 ++it;
264 continue;
265 }
266 len = filename.getLength();
267 if (len<5) {
268 ++it;
269 continue;
270 }
271 // only do .tga files
272 if (!(filename.endsWith(".tga"))) {
273 ++it;
274 continue;
275 }
276 strcpy(fileBuf, TEST_STRING);
277 strcat(fileBuf, "\\");
278 strcat(fileBuf, filename.str());
279 addRoad(fileBuf, index, TVI_ROOT);
280 index++;
281 m_numberOfRoads++;
282 ++it;
283 } while (it != filenameList.end());
284
285 }
286 }
287#endif
288
289 m_numberOfBridges = 0;
290 // add bridge defs from INI
291 TerrainRoadType *bridge;
292 for( bridge = TheTerrainRoads->firstBridge();
293 bridge;
294 bridge = TheTerrainRoads->nextBridge( bridge ) )
295 {
296
297 addRoad( (char *)bridge->getName().str(), index, TVI_ROOT );
298 index++;
299 m_numberOfBridges++;
300
301 } // end for bridge
302
303 m_currentRoadIndex = 1;
304 setRoadTreeViewSelection(TVI_ROOT, m_currentRoadIndex);
305 updateLabel();
306 m_staticThis = this;
307 updateSelection();
308 m_updating = false;
309 return TRUE; // return TRUE unless you set the focus to a control
310 // EXCEPTION: OCX Property Pages should return FALSE
311}

315HTREEITEM RoadOptions::findOrAdd(HTREEITEM parent, char *pLabel)
316{
317 TVINSERTSTRUCT ins;
318 char buffer[_MAX_PATH];
319 ::memset(&ins, 0, sizeof(ins));
320 HTREEITEM child = m_roadTreeView.GetChildItem(parent);
321 while (child != NULL) {
322 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
323 ins.item.hItem = child;
324 ins.item.pszText = buffer;
325 ins.item.cchTextMax = sizeof(buffer)-2;
326 m_roadTreeView.GetItem(&ins.item);
327 if (strcmp(buffer, pLabel) == 0) {
328 return(child);
329 }
330 child = m_roadTreeView.GetNextSiblingItem(child);
331 }
332
333 // not found, so add it.
334 ::memset(&ins, 0, sizeof(ins));
335 ins.hParent = parent;
336 ins.hInsertAfter = TVI_LAST;
337 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
338 ins.item.lParam = -1;
339 ins.item.pszText = pLabel;
340 ins.item.cchTextMax = strlen(pLabel);
341 child = m_roadTreeView.InsertItem(&ins);
342 return(child);
343}

372void RoadOptions::addRoad(char *pPath, Int terrainNdx, HTREEITEM parent)
373{
374 TerrainRoadType *road;
375 char buffer[_MAX_PATH];
376 Bool doAdd = FALSE;
377
378 // try to find the road in our INI definition
379 road = TheTerrainRoads->findRoadOrBridge( AsciiString( pPath ) );
380 if( road )
381 {
382
383 // roads go in a road tree, bridges in their own tree
384 if( road->isBridge() == TRUE )
385 parent = findOrAdd( parent, "Bridges" );
386 else
387 parent = findOrAdd( parent, "Roads" );
388
389 // set the name to place as the name of the road entry in INI
390 strcpy( buffer, road->getName().str() );
391
392 // do the add
393 doAdd = TRUE;
394
395 } // end if
396
397#ifdef LOAD_TEST_ASSETS
398 if (!doAdd && !strncmp(TEST_STRING, pPath, strlen(TEST_STRING))) {
399 parent = findOrAdd(parent, TEST_STRING);
400 strcpy(buffer, pPath + strlen(TEST_STRING) + 1);
401 doAdd = true;
402 }
403#endif
404
405 if( doAdd )
406 {
407 TVINSERTSTRUCT ins;
408
409 ::memset(&ins, 0, sizeof(ins));
410 ins.hParent = parent;
411 ins.hInsertAfter = TVI_LAST;
412 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
413 ins.item.lParam = terrainNdx;
414 ins.item.pszText = buffer;
415 ins.item.cchTextMax = strlen(buffer)+2;
416 m_roadTreeView.InsertItem(&ins);
417 }

447void RoadOptions::SelectConnected(void)
448{
449 std::list<MapObject*> roadSegs;
450 std::list<MapObject*> connectedSegs;
451 for (MapObject* pMapObj = MapObject::getFirstMapObject(); pMapObj; pMapObj = pMapObj->getNext())
452 {
453 if (pMapObj->getFlag(FLAG_ROAD_POINT1))
454 {
455 if (pMapObj->isSelected() || pMapObj->getNext() && pMapObj->getNext()->isSelected())
456 {
457 connectedSegs.push_back(pMapObj);
458 }
459 else
460 {
461 roadSegs.push_back(pMapObj);
462 }
463 }
464 }
465 Bool changed = true;
466 while (changed)
467 {
468 changed = false;
469 for (std::list<MapObject*>::iterator it = roadSegs.begin(); it != roadSegs.end(); ++it)
470 {
471 MapObject* o = *it;
472 const Coord3D *oLoc = o->getLocation();
473 const Coord3D *onLoc = o->getNext()->getLocation();
474 for (std::list<MapObject*>::iterator connected = connectedSegs.begin(); connected != connectedSegs.end(); ++connected)
475 {
476 MapObject* p = *connected;
477 const Coord3D *pLoc = p->getLocation();
478 const Coord3D *pnLoc = p->getNext()->getLocation();
479
480 Real dx1 = oLoc->x - pLoc->x;
481 Real dy1 = oLoc->y - pLoc->y;
482 dx1 = abs(dx1);
483 dy1 = abs(dy1);
484 Real qd1 = max(dx1, dy1);
485 //Real dist1 = sqrt(dx1*dx1+dy1*dy1);
486
487 Real dx2 = oLoc->x - pnLoc->x;
488 Real dy2 = oLoc->y - pnLoc->y;
489 dx2 = abs(dx2);
490 dy2 = abs(dy2);
491 Real qd2 = max(dx2, dy2);
492 //Real dist2 = sqrt(dx2*dx2+dy2*dy2);
493
494 Real dx3 = onLoc->x - pLoc->x;
495 Real dy3 = onLoc->y - pLoc->y;
496 dx3 = abs(dx3);
497 dy3 = abs(dy3);
498 Real qd3 = max(dx3, dy3);
499 //Real dist3 = sqrt(dx3*dx3+dy3*dy3);
500
501 Real dx4 = onLoc->x - pnLoc->x;
502 Real dy4 = onLoc->y - pnLoc->y;
503 dx4 = abs(dx4);
504 dy4 = abs(dy4);
505 Real qd4 = max(dx4, dy4);
506 //Real dist4 = sqrt(dx4*dx4+dy4*dy4);
507
508 if (qd1 < MAP_XY_FACTOR/100 || qd2 < MAP_XY_FACTOR/100 || qd3 < MAP_XY_FACTOR/100 || qd4 < MAP_XY_FACTOR/100) {
509 connectedSegs.push_back(o);
510 roadSegs.erase(it);
511 changed = true;
512 break;
513 }
514 }
515 }
516 }
517
518 for (std::list<MapObject*>::iterator connected = connectedSegs.begin(); connected != connectedSegs.end(); ++connected)
519 {
520 MapObject* p = *connected;
521 if (p) {
522 p->setSelected(true);
523 p->getNext()->setSelected(true);
524 }
525 }
526}

538BOOL RoadOptions::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
539{
540 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
541 if (pHdr->hdr.hwndFrom == m_roadTreeView.m_hWnd) {
542 if (pHdr->hdr.code == TVN_SELCHANGED) {
543 char buffer[NAME_MAX_LEN];
544 HTREEITEM hItem = m_roadTreeView.GetSelectedItem();
545 TVITEM item;
546 ::memset(&item, 0, sizeof(item));
547 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT;
548 item.hItem = hItem;
549 item.pszText = buffer;
550 item.cchTextMax = sizeof(buffer)-2;
551 m_roadTreeView.GetItem(&item);
552 if (item.lParam >= 0) {
553 m_currentRoadIndex = item.lParam;
554 m_currentRoadName = buffer;
555 updateLabel();
556 } else if (!(item.state & TVIS_EXPANDEDONCE) ) {
557 HTREEITEM child = m_roadTreeView.GetChildItem(hItem);
558 while (child != NULL) {
559 hItem = child;
560 child = m_roadTreeView.GetChildItem(hItem);
561 }
562 if (hItem != m_roadTreeView.GetSelectedItem()) {
563 m_roadTreeView.SelectItem(hItem);
564 updateSelection();
565 }
566 }
567 }
568 }
569
570 return CDialog::OnNotify(wParam, lParam, pResult);
571}
