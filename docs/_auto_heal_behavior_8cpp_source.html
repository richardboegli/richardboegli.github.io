64static void checkForAutoHeal( Object *testObj, void *userData )
65{
66 AutoHealPlayerScanHelper *helper = (AutoHealPlayerScanHelper*)userData;
67 ObjectPointerList *listToAddTo = helper->m_objectList;
68
69 if( testObj->isEffectivelyDead() )
70 return;
71
72 if( testObj->getControllingPlayer() != helper->m_theHealer->getControllingPlayer() )
73 return;
74
75 if( testObj->isOffMap() )
76 return;
77
78 if( helper->m_skipSelfForHealing && testObj == helper->m_theHealer )
79 return;
80
81 if( !testObj->isAnyKindOf(helper->m_kindOfToTest) )
82 return;
83
84 if( testObj->isAnyKindOf( helper->m_forbiddenKindOf ) )
85 return;
86
87 if( testObj->getBodyModule()->getHealth() >= testObj->getBodyModule()->getMaxHealth() )
88 return;
89
90 listToAddTo->push_back(testObj);
91}

95AutoHealBehavior::AutoHealBehavior( Thing *thing, const ModuleData* moduleData ) : UpdateModule( thing, moduleData )
96{
97 const AutoHealBehaviorModuleData *d = getAutoHealBehaviorModuleData();
98
99 m_radiusParticleSystemID = INVALID_PARTICLE_SYSTEM_ID;
100 m_soonestHealFrame = 0;
101 m_stopped = false;
102 Object *obj = getObject();
103
104 {
105 if( d->m_radiusParticleSystemTmpl )
106 {
107 ParticleSystem *particleSystem;
108
109 particleSystem = TheParticleSystemManager->createParticleSystem( d->m_radiusParticleSystemTmpl );
110 if( particleSystem )
111 {
112 particleSystem->setPosition( obj->getPosition() );
113 m_radiusParticleSystemID = particleSystem->getSystemID();
114 }
115 }
116 }
117
118 if (d->m_initiallyActive)
119 {
120 giveSelfUpgrade();
121 // start these guys with random phasings so that we don't
122 // have all of 'em check on the same frame.
123 UnsignedInt delay = getAutoHealBehaviorModuleData()->m_healingDelay;
124 setWakeFrame(getObject(), UPDATE_SLEEP(GameLogicRandomValue(1, delay)));
125 }
126 else
127 {
128 setWakeFrame(getObject(), UPDATE_SLEEP_FOREVER);
129 }
130}

160void AutoHealBehavior::onDamage( DamageInfo *damageInfo )
161{
162 if (m_stopped)
163 return;
164
165 const AutoHealBehaviorModuleData *d = getAutoHealBehaviorModuleData();
166 if (isUpgradeActive() && d->m_radius == 0.0f)
167 {
168 // if this is nonzero, getting damaged resets our healing process. so go to
169 // sleep for this long.
170 if (d->m_startHealingDelay > 0)
171 {
172 setWakeFrame(getObject(), UPDATE_SLEEP(d->m_startHealingDelay));
173 }
174 else if( TheGameLogic->getFrame() > m_soonestHealFrame )
175 {
176 // We can only force an immediate wake if we are ready to heal. Otherwise we will
177 // heal on a timer AND at every damage input.
178 setWakeFrame(getObject(), UPDATE_SLEEP_NONE);
179 }
180 }
181}

186UpdateSleepTime AutoHealBehavior::update( void )
187{
188 if (m_stopped)
189 return UPDATE_SLEEP_FOREVER;
190
191 Object *obj = getObject();
192 const AutoHealBehaviorModuleData *d = getAutoHealBehaviorModuleData();
193
194 // do not heal if our status bit is not on.
195 // do not heal if our status is effectively dead. There ain't no coming back, man!
196 if (!isUpgradeActive() || obj->isEffectivelyDead())
197 {
198 DEBUG_ASSERTCRASH(isUpgradeActive(), ("hmm, this should not be possible"));
199 return UPDATE_SLEEP_FOREVER;
200 }
201
202//DEBUG_LOG(("doing auto heal %d\n",TheGameLogic->getFrame()));
203
204 if( d->m_affectsWholePlayer )
205 {
206 // Even newer system, I can ignore radius and iterate objects on the owning player. Faster than scanning range 10,000,000
207 ObjectPointerList objectsToHeal;
208 Player *owningPlayer = getObject()->getControllingPlayer();
209 if( owningPlayer )
210 {
211 AutoHealPlayerScanHelper helper;
212 helper.m_kindOfToTest = d->m_kindOf;
213 helper.m_forbiddenKindOf = d->m_forbiddenKindOf;
214 helper.m_objectList = &objectsToHeal;
215 helper.m_theHealer = getObject();
216 helper.m_skipSelfForHealing = d->m_skipSelfForHealing;
217
218 // Smack all objects with this function, and we will end up with a list of Objects deserving of pulseHealObject
219 owningPlayer->iterateObjects( checkForAutoHeal, &helper );
220
221 for( ObjectPointerListIterator iter = objectsToHeal.begin(); iter != objectsToHeal.end(); ++iter )
222 {
223 pulseHealObject(*iter);
224 }
225 objectsToHeal.clear();
226 }
227 return UPDATE_SLEEP(d->m_healingDelay);
228 }
229 else if( d->m_radius == 0.0f )
230 {
231 //ORIGINAL SYSTEM -- JUST HEAL SELF!
232
233 // do not heal if we are at max health already
234 BodyModuleInterface *body = obj->getBodyModule();
235 if( body->getHealth() < body->getMaxHealth() )
236 {
237 pulseHealObject( obj );
238 return UPDATE_SLEEP(d->m_healingDelay);
239 }
240 else
241 {
242 // go to sleep forever -- we'll wake back up when we are damaged again
243 return UPDATE_SLEEP_FOREVER;
244 }
245 }
246 else
247 {
248 //EXPANDED SYSTEM -- HEAL FRIENDLIES IN RADIUS
249 // setup scan filters
250 PartitionFilterRelationship relationship( obj, PartitionFilterRelationship::ALLOW_ALLIES );
251 PartitionFilterSameMapStatus filterMapStatus(obj);
252 PartitionFilterAlive filterAlive;
253 PartitionFilter *filters[] = { &relationship, &filterAlive, &filterMapStatus, NULL };
254
255 // scan objects in our region
256 ObjectIterator *iter = ThePartitionManager->iterateObjectsInRange( obj->getPosition(), d->m_radius, FROM_CENTER_2D, filters );
257 MemoryPoolObjectHolder hold( iter );
258 for( obj = iter->first(); obj; obj = iter->next() )
259 {
260 // do not heal if we are at max health already
261 BodyModuleInterface *body = obj->getBodyModule();
262 if( body->getHealth() < body->getMaxHealth() )
263 {
264 if( obj->isAnyKindOf( d->m_kindOf ) && !obj->isAnyKindOf( d->m_forbiddenKindOf ) )
265 {
266 if( !d->m_skipSelfForHealing || obj != getObject() )
267 {
268 pulseHealObject( obj );
269
270 if( d->m_singleBurst && TheGameLogic->getDrawIconUI() )
271 {
272 if( TheAnim2DCollection && TheGlobalData->m_getHealedAnimationName.isEmpty() == FALSE )
273 {
274 Anim2DTemplate *animTemplate = TheAnim2DCollection->findTemplate( TheGlobalData->m_getHealedAnimationName );
275
276 if ( animTemplate )
277 {
278 Coord3D iconPosition;
279 iconPosition.set(obj->getPosition()->x,
280 obj->getPosition()->y,
281 obj->getPosition()->z + obj->getGeometryInfo().getMaxHeightAbovePosition() );
282 TheInGameUI->addWorldAnimation( animTemplate, &iconPosition, WORLD_ANIM_FADE_ON_EXPIRE,
283 TheGlobalData->m_getHealedAnimationDisplayTimeInSeconds,
284 TheGlobalData->m_getHealedAnimationZRisePerSecond);
285 }
286 }
287 }
288 }
289 }
290 }
291 } // end for obj
292
293 return UPDATE_SLEEP( d->m_singleBurst ? UPDATE_SLEEP_FOREVER : d->m_healingDelay );
294 }

299void AutoHealBehavior::pulseHealObject( Object *obj )
300{
301 if (m_stopped)
302 return;
303
304 const AutoHealBehaviorModuleData *data = getAutoHealBehaviorModuleData();
305
306
307 if ( data->m_radius == 0.0f )
308 obj->attemptHealing(data->m_healingAmount, getObject());
309 else
310 obj->attemptHealingFromSoleBenefactor( data->m_healingAmount, getObject(), data->m_healingDelay );
311
312
313 if( data->m_unitHealPulseParticleSystemTmpl )
314 {
315 ParticleSystem *system = TheParticleSystemManager->createParticleSystem( data->m_unitHealPulseParticleSystemTmpl );
316 if( system )
317 {
318 system->setPosition( obj->getPosition() );
319 }
320 }
321
322 m_soonestHealFrame = TheGameLogic->getFrame() + data->m_healingDelay;// In case onDamage tries to wake us up early
323}
