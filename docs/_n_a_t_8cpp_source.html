57 * time it would normally take to make 7 connections. Since each connection
58 * could potentially take several seconds, this can be a HUGE time savings.

182// if we are negotiating now, check to see if this round is done
183// if it is, check to see if we're completely done with all rounds.

188enum { MS_TO_WAIT_FOR_STATS = 5000 };
189NATStateType NAT::update() {
190 static UnsignedInt s_startStatWaitTime = 0;
191 if (m_NATState == NATSTATE_DONE) {
192 m_NATState = NATSTATE_IDLE;
193 } else if (m_NATState == NATSTATE_WAITFORSTATS) {
194 // check for all stats
195 Bool gotAllStats = TRUE;
196 Bool timedOut = FALSE;
197 for (Int i=0; i<MAX_SLOTS; ++i)
198 {
199 const GameSpyGameSlot *slot = TheGameSpyGame->getGameSpySlot(i);
200 if (slot && slot->isHuman())
201 {
202 PSPlayerStats stats = TheGameSpyPSMessageQueue->findPlayerStatsByID(slot->getProfileID());
203 if (stats.id == 0)
204 {
205 gotAllStats = FALSE;
206 //DEBUG_LOG(("Failed to find stats for %ls(%d)\n", slot->getName().str(), slot->getProfileID()));
207 }
208 }
209 }
210 // check for timeout. Timing out is not a fatal error - it just means we didn't get the other
211 // player's stats. We'll see 0/0 as his record, but we can still play him just fine.
212 UnsignedInt now = timeGetTime();
213 if (now > s_startStatWaitTime + MS_TO_WAIT_FOR_STATS)
214 {
215 DEBUG_LOG(("Timed out waiting for stats. Let's just start the dang game.\n"));
216 timedOut = TRUE;
217 }
218 if (gotAllStats || timedOut)
219 {
220 m_NATState = NATSTATE_DONE;
221 TheEstablishConnectionsMenu->endMenu();
222 if (TheFirewallHelper != NULL) {
223 delete TheFirewallHelper;
224 TheFirewallHelper = NULL;
225 }
226 }
227 } else if (m_NATState == NATSTATE_DOCONNECTIONPATHS) {
228 if (allConnectionsDoneThisRound() == TRUE) {
229 // we finished this round, move on to the next one.
230 ++m_connectionRound;
231// m_roundTimeout = timeGetTime() + TheGameSpyConfig->getRoundTimeout();
232 m_roundTimeout = timeGetTime() + m_timeForRoundTimeout;
233 DEBUG_LOG(("NAT::update - done with connection round, moving on to round %d\n", m_connectionRound));
234
235 // we finished that round, now check to see if we're done, or if there are more rounds to go.
236 if (allConnectionsDone() == TRUE) {
237 // we're all done, time to go back home.
238 m_NATState = NATSTATE_WAITFORSTATS;
239
240 // 2/19/03 BGC - we have successfully negotaited a NAT thingy, so our behavior must be correct
241 // so therefore we don't need to refresh our NAT even if we previously thought we had to.
242 TheFirewallHelper->flagNeedToRefresh(FALSE);
243
244 s_startStatWaitTime = timeGetTime();
245 DEBUG_LOG(("NAT::update - done with all connections, woohoo!!\n"));
246 /*
247 m_NATState = NATSTATE_DONE;
248 TheEstablishConnectionsMenu->endMenu();
249 if (TheFirewallHelper != NULL) {
250 delete TheFirewallHelper;
251 TheFirewallHelper = NULL;
252 }
253 */
254 } else {
255 doThisConnectionRound();
256 }
257 }
258 NATConnectionState state = connectionUpdate();
259
260 if (timeGetTime() > m_roundTimeout) {
261 DEBUG_LOG(("NAT::update - round timeout expired\n"));
262 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
263 notifyUsersOfConnectionFailed(m_localNodeNumber);
264 }
265
266 if (state == NATCONNECTIONSTATE_FAILED) {
267 // if we fail
268 m_NATState = NATSTATE_FAILED;
269 TheEstablishConnectionsMenu->endMenu();
270 if (TheFirewallHelper != NULL) {
271 // we failed NAT negotiation, perhaps we need to redetect our firewall settings.
272 // We don't trust the user to do it for themselves so we force them to do it next time
273 // the log in.
274 // 2/19/03 - ok, we don't want to do this right away, if the user tries to play in another game
275 // before they log out and log back in the game won't have a chance at working.
276 // so we need to simply flag it so that when they log out the firewall behavior gets blown away.
277 TheFirewallHelper->flagNeedToRefresh(TRUE);
278// TheWritableGlobalData->m_firewallBehavior = FirewallHelperClass::FIREWALL_TYPE_UNKNOWN;
279// TheFirewallHelper->writeFirewallBehavior();
280
281 delete TheFirewallHelper;
282 TheFirewallHelper = NULL;
283 }
284 // we failed to connect, so we don't have to pass on the transport to the network.
285 if (m_transport != NULL) {
286 delete m_transport;
287 m_transport = NULL;
288 }
289 }
290 }
291 return m_NATState;
292}

295// update transport, check for PROBE packets from our target.
296// check to see if its time to PROBE our target
297// MANGLER:
298// if we are talking to the mangler, check to see if we got a response
299// if we didn't get a response, check to see if its time to send another packet to it
300NATConnectionState NAT::connectionUpdate() {
301
302 GameSlot *targetSlot = NULL;
303 if (m_targetNodeNumber >= 0) {
304 targetSlot = m_slotList[m_connectionNodes[m_targetNodeNumber].m_slotIndex];
305 } else {
306 return m_connectionStates[m_localNodeNumber];
307 }
308
309 if (m_beenProbed == FALSE) {
310 if (timeGetTime() >= m_nextPortSendTime) {
311// sendMangledPortNumberToTarget(m_previousSourcePort, targetSlot);
312 sendMangledPortNumberToTarget(m_sourcePorts[m_targetNodeNumber], targetSlot);
313// m_nextPortSendTime = timeGetTime() + TheGameSpyConfig->getRetryInterval();
314 m_nextPortSendTime = timeGetTime() + m_timeBetweenRetries;
315 }
316 }
317
318 // check to see if its time to send out our keepalives.
319 if (timeGetTime() >= m_nextKeepaliveTime) {
320 for (Int node = 0; node < m_numNodes; ++node) {
321 if (m_myConnections[node] == TRUE) {
322 // we've made this connection, send a keepalive.
323 Int slotIndex = m_connectionNodes[node].m_slotIndex;
324 GameSlot *slot = m_slotList[slotIndex];
325 DEBUG_ASSERTCRASH(slot != NULL, ("Trying to send keepalive to a NULL slot"));
326 if (slot != NULL) {
327 UnsignedInt ip = slot->getIP();
328 DEBUG_LOG(("NAT::connectionUpdate - sending keep alive to node %d at %d.%d.%d.%d:%d\n", node,
329 ip >> 24, (ip >> 16) & 0xff, (ip >> 8) & 0xff, ip & 0xff, slot->getPort()));
330 m_transport->queueSend(ip, slot->getPort(), (const unsigned char *)"KEEPALIVE", strlen("KEEPALIVE") + 1);
331 }
332 }
333 }
334// m_nextKeepaliveTime = timeGetTime() + TheGameSpyConfig->getKeepaliveInterval();
335 m_nextKeepaliveTime = timeGetTime() + m_keepaliveInterval;
336 }
337
338 m_transport->update();
339
340 // check to see if we've been probed.
341 for (Int i = 0; i < MAX_MESSAGES; ++i) {
342 if (m_transport->m_inBuffer[i].length > 0) {
343#ifdef DEBUG_LOGGING
344 UnsignedInt ip = m_transport->m_inBuffer[i].addr;
345#endif
346 DEBUG_LOG(("NAT::connectionUpdate - got a packet from %d.%d.%d.%d:%d, length = %d\n",
347 ip >> 24, (ip >> 16) & 0xff, (ip >> 8) & 0xff, ip & 0xff, m_transport->m_inBuffer[i].port, m_transport->m_inBuffer[i].length));
348 UnsignedByte *data = m_transport->m_inBuffer[i].data;
349 if (memcmp(data, "PROBE", strlen("PROBE")) == 0) {
350 Int fromNode = atoi((char *)data + strlen("PROBE"));
351 DEBUG_LOG(("NAT::connectionUpdate - we've been probed by node %d.\n", fromNode));
352
353 if (fromNode == m_targetNodeNumber) {
354 DEBUG_LOG(("NAT::connectionUpdate - probe was sent by our target, setting connection state %d to done.\n", m_targetNodeNumber));
355 setConnectionState(m_targetNodeNumber, NATCONNECTIONSTATE_DONE);
356
357 if (m_transport->m_inBuffer[i].addr != targetSlot->getIP()) {
358 UnsignedInt fromIP = m_transport->m_inBuffer[i].addr;
359#ifdef DEBUG_LOGGING
360 UnsignedInt slotIP = targetSlot->getIP();
361#endif
362 DEBUG_LOG(("NAT::connectionUpdate - incomming packet has different from address than we expected, incoming: %d.%d.%d.%d expected: %d.%d.%d.%d\n",
363 fromIP >> 24, (fromIP >> 16) & 0xff, (fromIP >> 8) & 0xff, fromIP & 0xff,
364 slotIP >> 24, (slotIP >> 16) & 0xff, (slotIP >> 8) & 0xff, slotIP & 0xff));
365 targetSlot->setIP(fromIP);
366 }
367 if (m_transport->m_inBuffer[i].port != targetSlot->getPort()) {
368 DEBUG_LOG(("NAT::connectionUpdate - incoming packet came from a different port than we expected, incoming: %d expected: %d\n",
369 m_transport->m_inBuffer[i].port, targetSlot->getPort()));
370 targetSlot->setPort(m_transport->m_inBuffer[i].port);
371 m_sourcePorts[m_targetNodeNumber] = m_transport->m_inBuffer[i].port;
372 }
373 notifyUsersOfConnectionDone(m_targetNodeNumber);
374 }
375
376 m_transport->m_inBuffer[i].length = 0;
377 }
378 if (memcmp(data, "KEEPALIVE", strlen("KEEPALIVE")) == 0) {
379 // keep alive packet, just toss it.
380 DEBUG_LOG(("NAT::connectionUpdate - got keepalive from %d.%d.%d.%d:%d\n",
381 ip >> 24, (ip >> 16) & 0xff, (ip >> 8) && 0xff, ip & 0xff, m_transport->m_inBuffer[i].port));
382 m_transport->m_inBuffer[i].length = 0;
383 }
384 }
385 }
386
387 // we are waiting for our target to tell us that they have received our probe.
388 if (m_connectionStates[m_localNodeNumber] == NATCONNECTIONSTATE_WAITINGFORRESPONSE) {
389 // check to see if it's time to probe our target.
390 if ((m_timeTillNextSend != -1) && (m_timeTillNextSend <= timeGetTime())) {
391 if (m_numRetries > m_maxNumRetriesAllowed) {
392 DEBUG_LOG(("NAT::connectionUpdate - too many retries, connection failed.\n"));
393 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
394
395 notifyUsersOfConnectionFailed(m_localNodeNumber);
396 } else {
397 DEBUG_LOG(("NAT::connectionUpdate - trying to send another probe (#%d) to our target\n", m_numRetries+1));
398 // Send a probe.
399 sendAProbe(targetSlot->getIP(), targetSlot->getPort(), m_localNodeNumber);
400// m_timeTillNextSend = timeGetTime() + TheGameSpyConfig->getRetryInterval();
401 m_timeTillNextSend = timeGetTime() + m_timeBetweenRetries;
402
403 // tell the target they've been probed. In other words, our port is open.
404 notifyTargetOfProbe(targetSlot);
405
406 ++m_numRetries;
407 }
408 }
409 }
410
411 // we are waiting for a response from the mangler to tell us what port we're using.
412 if (m_connectionStates[m_localNodeNumber] == NATCONNECTIONSTATE_WAITINGFORMANGLERRESPONSE) {
413 UnsignedShort mangledPort = 0;
414 if (TheFirewallHelper != NULL) {
415 mangledPort = TheFirewallHelper->getManglerResponse(m_packetID);
416 }
417 if (mangledPort != 0) {
418 // we got a response. now we need to start probing (unless of course we have a netgear)
419 processManglerResponse(mangledPort);
420
421 // we know there is a firewall helper if we got here.
422 TheFirewallHelper->closeSpareSocket(m_spareSocketPort);
423 m_spareSocketPort = 0;
424 } else {
425 if (timeGetTime() >= m_manglerRetryTime) {
426 ++m_manglerRetries;
427// if (m_manglerRetries > TheGameSpyConfig->getMaxManglerRetries()) {
428 if (m_manglerRetries > m_maxAllowedManglerRetries) {
429 // we couldn't communicate with the mangler, just use our non-mangled
430 // port number and hope that works.
431 DEBUG_LOG(("NAT::connectionUpdate - couldn't talk with the mangler using default port number\n"));
432 sendMangledPortNumberToTarget(getSlotPort(m_connectionNodes[m_localNodeNumber].m_slotIndex), targetSlot);
433 m_sourcePorts[m_targetNodeNumber] = getSlotPort(m_connectionNodes[m_localNodeNumber].m_slotIndex);
434 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORRESPONSE);
435 } else {
436 if (TheFirewallHelper != NULL) {
437 DEBUG_LOG(("NAT::connectionUpdate - trying to send to the mangler again. mangler address: %d.%d.%d.%d, from port: %d, packet ID:%d\n",
438 m_manglerAddress >> 24, (m_manglerAddress >> 16) & 0xff, (m_manglerAddress >> 8) & 0xff, m_manglerAddress & 0xff, m_spareSocketPort, m_packetID));
439 TheFirewallHelper->sendToManglerFromPort(m_manglerAddress, m_spareSocketPort, m_packetID);
440 }
441// m_manglerRetryTime = TheGameSpyConfig->getRetryInterval() + timeGetTime();
442 m_manglerRetryTime = m_manglerRetryTimeInterval + timeGetTime();
443 }
444 }
445 }
446 }
447
448 if (m_connectionStates[m_localNodeNumber] == NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT) {
449 if (timeGetTime() > m_timeoutTime) {
450 DEBUG_LOG(("NAT::connectionUpdate - waiting too long to get the other player's port number, failed.\n"));
451 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
452
453 notifyUsersOfConnectionFailed(m_localNodeNumber);
454 }
455 }
456
457 return m_connectionStates[m_localNodeNumber];
458}

463void NAT::establishConnectionPaths() {
464 DEBUG_LOG(("NAT::establishConnectionPaths - entering\n"));
465 m_NATState = NATSTATE_DOCONNECTIONPATHS;
466 DEBUG_LOG(("NAT::establishConnectionPaths - using %d as our starting port number\n", m_startingPortNumber));
467 if (TheEstablishConnectionsMenu == NULL) {
468 TheEstablishConnectionsMenu = NEW EstablishConnectionsMenu;
469 }
470 TheEstablishConnectionsMenu->initMenu();
471
472 if (TheFirewallHelper == NULL) {
473 TheFirewallHelper = createFirewallHelper();
474 }
475
476 DEBUG_ASSERTCRASH(m_slotList != NULL, ("NAT::establishConnectionPaths - don't have a slot list"));
477 if (m_slotList == NULL) {
478 return;
479 }
480
481 // determine how many nodes we have.
482 m_numNodes = 0;
483 for (Int i = 0; i < MAX_SLOTS; ++i) {
484 if (m_slotList[i] != NULL) {
485 if (m_slotList[i]->isHuman()) {
486 DEBUG_LOG(("NAT::establishConnectionPaths - slot %d is %ls\n", i, m_slotList[i]->getName().str()));
487 ++m_numNodes;
488 }
489 }
490 }
491 DEBUG_LOG(("NAT::establishConnectionPaths - number of nodes: %d\n", m_numNodes));
492
493 if (m_numNodes < 2)
494 {
495 // just start the game - there isn't anybody to which to connect. :P
496 m_NATState = NATSTATE_DONE;
497 return;
498 }
499
500 m_connectionRound = 0;
501 m_connectionPairIndex = m_numNodes - 2;
502 Bool connectionAssigned[MAX_SLOTS];
503
504 for (i = 0; i < MAX_SLOTS; ++i) {
505 m_connectionNodes[i].m_slotIndex = -1;
506 connectionAssigned[i] = FALSE;
507 m_sourcePorts[i] = 0;
508 }
509
510 m_previousSourcePort = 0;
511
512// check for netgear bug behavior.
513// as an aside, if there are more than 2 netgear bug firewall's in the game,
514// it probably isn't going to work so well. stupid netgear.
515
516// nodes with a netgear bug behavior need to be matched up first. This prevents
517// the NAT table from being reset for connections to other nodes. This also happens
518// to be the reason why I call them "nodes" rather than "slots" or "players" as the
519// ordering has to be messed with to get the netgears to make love, not war.
520 DEBUG_LOG(("NAT::establishConnectionPaths - about to set up the node list\n"));
521 DEBUG_LOG(("NAT::establishConnectionPaths - doing the netgear stuff\n"));
522 UnsignedInt otherNetgearNum = -1;
523 for (i = 0; i < MAX_SLOTS; ++i) {
524 if ((m_slotList != NULL) && (m_slotList[i] != NULL)) {
525 if ((m_slotList[i]->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) != 0) {
526 if (otherNetgearNum == -1) {
527 // this is the start of a new pair, put it in as the first non -1 node connection pair thing.
528 Int nodeindex = 0;
529 while ((m_connectionPairs[m_connectionPairIndex][0][nodeindex] == -1) || (m_connectionNodes[nodeindex].m_slotIndex != -1)) {
530 ++nodeindex;
531 }
532 m_connectionNodes[nodeindex].m_slotIndex = i;
533 m_connectionNodes[nodeindex].m_behavior = m_slotList[i]->getNATBehavior();
534 connectionAssigned[i] = TRUE;
535 otherNetgearNum = nodeindex;
536 DEBUG_LOG(("NAT::establishConnectionPaths - first netgear in pair. assigning node %d to slot %d (%ls)\n", nodeindex, i, m_slotList[i]->getName().str()));
537 } else {
538 // this is the second in the pair of netgears, pair this up with the other one
539 // for the first round.
540 Int nodeindex = 0;
541 while (m_connectionPairs[m_connectionPairIndex][0][nodeindex] != otherNetgearNum) {
542 ++nodeindex;
543 }
544 m_connectionNodes[nodeindex].m_slotIndex = i;
545 m_connectionNodes[nodeindex].m_behavior = m_slotList[i]->getNATBehavior();
546 connectionAssigned[i] = TRUE;
547 otherNetgearNum = -1;
548 DEBUG_LOG(("NAT::establishConnectionPaths - second netgear in pair. assigning node %d to slot %d (%ls)\n", nodeindex, i, m_slotList[i]->getName().str()));
549 }
550 }
551 }
552 }
553
554 // fill in the rest of the nodes with the remaining slots.
555 DEBUG_LOG(("NAT::establishConnectionPaths - doing the non-Netgear nodes\n"));
556 for (i = 0; i < MAX_SLOTS; ++i) {
557 if (connectionAssigned[i] == TRUE) {
558 continue;
559 }
560 if (m_slotList[i] == NULL) {
561 continue;
562 }
563 if (!(m_slotList[i]->isHuman())) {
564 continue;
565 }
566 // find the first available connection node for this slot.
567 Int nodeindex = 0;
568 while (m_connectionNodes[nodeindex].m_slotIndex != -1) {
569 ++nodeindex;
570 }
571 DEBUG_LOG(("NAT::establishConnectionPaths - assigning node %d to slot %d (%ls)\n", nodeindex, i, m_slotList[i]->getName().str()));
572 m_connectionNodes[nodeindex].m_slotIndex = i;
573 m_connectionNodes[nodeindex].m_behavior = m_slotList[i]->getNATBehavior();
574 connectionAssigned[i] = TRUE;
575 }
576
577// sanity check
578#if defined(_DEBUG) || defined(_INTERNAL)
579 for (i = 0; i < m_numNodes; ++i) {
580 DEBUG_ASSERTCRASH(connectionAssigned[i] == TRUE, ("connection number %d not assigned", i));
581 }
582#endif
583
584 // find the local node number.
585 for (i = 0; i < m_numNodes; ++i) {
586 if (m_connectionNodes[i].m_slotIndex == TheGameSpyGame->getLocalSlotNum()) {
587 m_localNodeNumber = i;
588 DEBUG_LOG(("NAT::establishConnectionPaths - local node is %d\n", m_localNodeNumber));
589 break;
590 }
591 }
592
593 // set up the names in the connection window.
594 Int playerNum = 0;
595 for (i = 0; i < MAX_SLOTS; ++i) {
596 while ((i < MAX_SLOTS) && (m_slotList[i] != NULL) && !(m_slotList[i]->isHuman())) {
597 ++i;
598 }
599 if (i >= MAX_SLOTS) {
600 break;
601 }
602 if (i != TheGameSpyGame->getLocalSlotNum()) {
603 TheEstablishConnectionsMenu->setPlayerName(playerNum, m_slotList[i]->getName());
604 TheEstablishConnectionsMenu->setPlayerStatus(playerNum, NATCONNECTIONSTATE_WAITINGTOBEGIN);
605 ++playerNum;
606 }
607 }
608
609// m_roundTimeout = timeGetTime() + TheGameSpyConfig->getRoundTimeout();
610 m_roundTimeout = timeGetTime() + m_timeForRoundTimeout;
611
612 // make the connections for this round.
613 // this song is cool.
614 doThisConnectionRound();
615}

654// figure out which port I'll be using.
655// send the port number to our target for this round.
656// init the m_connectionStates for all players.
657void NAT::doThisConnectionRound() {
658 DEBUG_LOG(("NAT::doThisConnectionRound - starting process for connection round %d\n", m_connectionRound));
659 // clear out the states from the last round.
660 m_targetNodeNumber = -1;
661
662 for (Int i = 0; i < MAX_SLOTS; ++i) {
663 setConnectionState(i, NATCONNECTIONSTATE_NOSTATE);
664 }
665
666 m_beenProbed = FALSE;
667 m_numRetries = 0;
668
669 for (i = 0; i < m_numNodes; ++i) {
670 Int targetNodeNumber = m_connectionPairs[m_connectionPairIndex][m_connectionRound][i];
671 DEBUG_LOG(("NAT::doThisConnectionRound - node %d needs to connect to node %d\n", i, targetNodeNumber));
672 if (targetNodeNumber != -1) {
673 if (i == m_localNodeNumber) {
674 m_targetNodeNumber = targetNodeNumber;
675 DEBUG_LOG(("NAT::doThisConnectionRound - Local node is connecting to node %d\n", m_targetNodeNumber));
676 UnsignedInt targetSlotIndex = m_connectionNodes[(m_connectionPairs[m_connectionPairIndex][m_connectionRound][i])].m_slotIndex;
677 GameSlot *targetSlot = m_slotList[targetSlotIndex];
678 GameSlot *localSlot = m_slotList[m_connectionNodes[m_localNodeNumber].m_slotIndex];
679
680 DEBUG_ASSERTCRASH(localSlot != NULL, ("local slot is NULL"));
681 DEBUG_ASSERTCRASH(targetSlot != NULL, ("trying to negotiate with a NULL target slot, slot is %d", m_connectionPairs[m_connectionPairIndex][m_connectionRound][i]));
682 DEBUG_LOG(("NAT::doThisConnectionRound - Target slot index = %d (%ls)\n", targetSlotIndex, m_slotList[targetSlotIndex]->getName().str()));
683 DEBUG_LOG(("NAT::doThisConnectionRound - Target slot has NAT behavior 0x%8X, local slot has NAT behavior 0x%8X\n", targetSlot->getNATBehavior(), localSlot->getNATBehavior()));
684
685#if defined(DEBUG_LOGGING)
686 UnsignedInt targetIP = targetSlot->getIP();
687 UnsignedInt localIP = localSlot->getIP();
688#endif
689
690 DEBUG_LOG(("NAT::doThisConnectionRound - Target slot has IP %d.%d.%d.%d Local slot has IP %d.%d.%d.%d\n",
691 targetIP >> 24, (targetIP >> 16) & 0xff, (targetIP >> 8) & 0xff, targetIP & 0xff,
692 localIP >> 24, (localIP >> 16) & 0xff, (localIP >> 8) & 0xff, localIP & 0xff));
693
694 if (((targetSlot->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) == 0) &&
695 ((localSlot->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) != 0)) {
696
697 // we have a netgear bug type behavior and the target does not, so we need them to send to us
698 // first to avoid having our NAT table reset.
699
700 DEBUG_LOG(("NAT::doThisConnectionRound - Local node has a netgear and the target node does not, need to delay our probe.\n"));
701 m_timeTillNextSend = -1;
702 }
703
704 // figure out which port number I'm using for this connection
705 // this merely starts to talk to the mangler server, we have to keep calling
706 // the update function till we get a response.
707 DEBUG_LOG(("NAT::doThisConnectionRound - About to attempt to get the next mangled source port\n"));
708 sendMangledSourcePort();
709// m_nextPortSendTime = timeGetTime() + TheGameSpyConfig->getRetryInterval();
710 m_nextPortSendTime = timeGetTime() + m_timeBetweenRetries;
711// m_timeoutTime = timeGetTime() + TheGameSpyConfig->getPortTimeout();
712 m_timeoutTime = timeGetTime() + m_timeToWaitForPort;
713 } else {
714 // this is someone else that needs to connect to someone, so wait till they tell us
715 // that they're done.
716 setConnectionState(i, NATCONNECTIONSTATE_WAITINGFORRESPONSE);
717 }
718 } else {
719 // no one to connect to, so this one is done.
720 DEBUG_LOG(("NAT::doThisConnectionRound - node %d has no one to connect to, so they're done\n", i));
721 setConnectionState(i, NATCONNECTIONSTATE_DONE);
722 }
723 }

735// find the next mangled source port, and then send it to the other player.
736// if this requires talking to the mangler, we'll have to wait till a later update
737// to send our port out.
738void NAT::sendMangledSourcePort() {
739 UnsignedShort sourcePort = getSlotPort(m_connectionNodes[m_localNodeNumber].m_slotIndex);
740
741 FirewallHelperClass::tFirewallBehaviorType fwType = m_slotList[m_connectionNodes[m_localNodeNumber].m_slotIndex]->getNATBehavior();
742 GameSlot *targetSlot = m_slotList[m_connectionNodes[m_targetNodeNumber].m_slotIndex];
743 DEBUG_ASSERTCRASH(targetSlot != NULL, ("NAT::sendMangledSourcePort - targetSlot is NULL"));
744 if (targetSlot == NULL) {
745 DEBUG_LOG(("NAT::sendMangledSourcePort - targetSlot is NULL, failed this connection\n"));
746 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
747 return;
748 }
749
750 GameSlot *localSlot = m_slotList[m_connectionNodes[m_localNodeNumber].m_slotIndex];
751 DEBUG_ASSERTCRASH(localSlot != NULL, ("NAT::sendMangledSourcePort - localSlot is NULL, WTF?"));
752 if (localSlot == NULL) {
753 DEBUG_LOG(("NAT::sendMangledSourcePort - localSlot is NULL, failed this connection\n"));
754 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
755 return;
756 }
757
758 // check to see if the target and I are behind the same NAT
759 if (targetSlot->getIP() == localSlot->getIP()) {
760#if defined(DEBUG_LOGGING)
761 UnsignedInt localip = localSlot->getIP();
762 UnsignedInt targetip = targetSlot->getIP();
763#endif
764 DEBUG_LOG(("NAT::sendMangledSourcePort - target and I are behind the same NAT, no mangling\n"));
765 DEBUG_LOG(("NAT::sendMangledSourcePort - I am %ls, target is %ls, my IP is %d.%d.%d.%d, target IP is %d.%d.%d.%d\n", localSlot->getName().str(), targetSlot->getName().str(),
766 localip >> 24, (localip >> 16) & 0xff, (localip >> 8) & 0xff, localip & 0xff,
767 targetip >> 24, (targetip >> 16) & 0xff, (targetip >> 8) & 0xff, targetip & 0xff));
768
769 sendMangledPortNumberToTarget(sourcePort, targetSlot);
770 m_sourcePorts[m_targetNodeNumber] = sourcePort;
771 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT);
772 // In case you're wondering, we don't set the m_previousSourcePort here because this will be a different source
773 // address than what other nodes will likely see (unless of course there are more than
774 // two of us behind the same NAT, but we won't worry about that cause theres no mangling
775 // in that case anyways)
776 return;
777 }
778
779 // check to see if we are NAT'd at all.
780 if ((fwType == 0) || (fwType == FirewallHelperClass::FIREWALL_TYPE_SIMPLE)) {
781 // no mangling, just return the source port
782 DEBUG_LOG(("NAT::sendMangledSourcePort - no mangling, just using the source port\n"));
783 sendMangledPortNumberToTarget(sourcePort, targetSlot);
784 m_previousSourcePort = sourcePort;
785 m_sourcePorts[m_targetNodeNumber] = sourcePort;
786 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT);
787 return;
788 }
789
790 // check to see if our NAT keeps the same source port for different destinations.
791 // if this is the case, and we've already worked out what our mangled port number is
792 // then we don't have to figure it out again.
793 if (((fwType & FirewallHelperClass::FIREWALL_TYPE_DESTINATION_PORT_DELTA) == 0) &&
794 ((fwType & FirewallHelperClass::FIREWALL_TYPE_SMART_MANGLING) == 0)) {
795 DEBUG_LOG(("NAT::sendMangledSourcePort - our firewall doesn't NAT based on destination address, checking for old connections from this address\n"));
796 if (m_previousSourcePort != 0) {
797 DEBUG_LOG(("NAT::sendMangledSourcePort - Previous source port was %d, using that one\n", m_previousSourcePort));
798 sendMangledPortNumberToTarget(m_previousSourcePort, targetSlot);
799 m_sourcePorts[m_targetNodeNumber] = m_previousSourcePort;
800 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT);
801 return;
802 } else {
803 DEBUG_LOG(("NAT::sendMangledSourcePort - Previous source port not found\n"));
804 }
805 }
806
807 // At this point we know that our NAT uses some kind of relative port mapping scheme, so we
808 // need to talk to the mangler to find out where we are now so we can find out where we'll be on the
809 // next port allocation.
810
811 // get the address of the mangler we need to talk to.
812 Char manglerName[256];
813 FirewallHelperClass::getManglerName(1, manglerName);
814 DEBUG_LOG(("NAT::sendMangledSourcePort - about to call gethostbyname for mangler at %s\n", manglerName));
815 struct hostent *hostInfo = gethostbyname(manglerName);
816
817 if (hostInfo == NULL) {
818 DEBUG_LOG(("NAT::sendMangledSourcePort - gethostbyname failed for mangler address %s\n", manglerName));
819 // can't find the mangler, we're screwed so just send the source port.
820 sendMangledPortNumberToTarget(sourcePort, targetSlot);
821 m_sourcePorts[m_targetNodeNumber] = sourcePort;
822 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT);
823 return;
824 }
825
826 memcpy(&m_manglerAddress, &(hostInfo->h_addr_list[0][0]), 4);
827 m_manglerAddress = ntohl(m_manglerAddress);
828 DEBUG_LOG(("NAT::sendMangledSourcePort - mangler %s address is %d.%d.%d.%d\n", manglerName,
829 m_manglerAddress >> 24, (m_manglerAddress >> 16) & 0xff, (m_manglerAddress >> 8) & 0xff, m_manglerAddress & 0xff));
830
831 DEBUG_LOG(("NAT::sendMangledSourcePort - NAT behavior = 0x%08x\n", fwType));
832
833// m_manglerRetryTime = TheGameSpyConfig->getRetryInterval() + timeGetTime();
834 m_manglerRetryTime = m_manglerRetryTimeInterval + timeGetTime();
835 m_manglerRetries = 0;
836
837 if (TheFirewallHelper != NULL) {
838 m_spareSocketPort = TheFirewallHelper->getNextTemporarySourcePort(0);
839 TheFirewallHelper->openSpareSocket(m_spareSocketPort);
840 TheFirewallHelper->sendToManglerFromPort(m_manglerAddress, m_spareSocketPort, m_packetID);
841// m_manglerRetryTime = TheGameSpyConfig->getRetryInterval() + timeGetTime();
842 m_manglerRetryTime = m_manglerRetryTimeInterval + timeGetTime();
843 }
844
845 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLERRESPONSE);
846}

848void NAT::processManglerResponse(UnsignedShort mangledPort) {
849 DEBUG_LOG(("NAT::processManglerResponse - Work out what my NAT'd port will be\n"));
850
851 GameSlot *targetSlot = m_slotList[m_connectionNodes[m_targetNodeNumber].m_slotIndex];
852 DEBUG_ASSERTCRASH(targetSlot != NULL, ("NAT::processManglerResponse - targetSlot is NULL"));
853 if (targetSlot == NULL) {
854 DEBUG_LOG(("NAT::processManglerResponse - targetSlot is NULL, failed this connection\n"));
855 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
856 return;
857 }
858
859 Short delta = TheGlobalData->m_firewallPortAllocationDelta;
860 UnsignedShort sourcePort = getSlotPort(m_connectionNodes[m_localNodeNumber].m_slotIndex);
861 UnsignedShort returnPort = 0;
862
863 FirewallHelperClass::tFirewallBehaviorType fwType = m_slotList[m_connectionNodes[m_localNodeNumber].m_slotIndex]->getNATBehavior();
864
865 if ((fwType & FirewallHelperClass::FIREWALL_TYPE_SIMPLE_PORT_ALLOCATION) != 0) {
866 returnPort = mangledPort + delta;
867 } else {
868 // to steal a line from Steve Tall...
869 // Rats. It's a relative mangler. This is much harder. Damn NAT32 guy.
870 if (delta == 100) {
871 // Special NAT32 section.
872 // NAT32 mangles source UDP port by ading 1700 + 100*NAT table index.
873 returnPort = mangledPort - m_spareSocketPort;
874 returnPort -= 1700;
875
876 returnPort += delta;
877 returnPort += sourcePort;
878 returnPort += 1700;
879
880 } else if (delta == 0) {
881 returnPort = sourcePort;
882 } else {
883 returnPort = mangledPort / delta;
884 returnPort = returnPort * delta;
885
886 returnPort += (sourcePort % delta);
887 returnPort += delta;
888 }
889 }
890
891 // This bit is probably doomed.
892 if (returnPort > 65535) {
893 returnPort -= 65535;
894 }
895 if (returnPort < 1024) {
896 returnPort += 1024;
897 }
898
899 DEBUG_LOG(("NAT::processManglerResponse - mangled port is %d\n", returnPort));
900 m_previousSourcePort = returnPort;
901
902 sendMangledPortNumberToTarget(returnPort, targetSlot);
903 m_sourcePorts[m_targetNodeNumber] = returnPort;
904 if (targetSlot->getPort() == 0) {
905 // we haven't got the target's mangled port number yet, wait for it.
906 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORMANGLEDPORT);
907 } else {
908 // in this case we should have already sent a PROBE, so we'll just change the state
909 // and leave it at that.
910 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_WAITINGFORRESPONSE);
911 }

1003// got the mangled port for our target for this round.
1004void NAT::gotMangledPort(Int nodeNumber, UnsignedShort mangledPort) {
1005
1006 // if we've already finished the connection, then we don't need to process this.
1007 if (m_connectionStates[m_localNodeNumber] == NATCONNECTIONSTATE_DONE) {
1008 DEBUG_LOG(("NAT::gotMangledPort - got a mangled port, but we've already finished this connection, ignoring.\n"));
1009 return;
1010 }
1011
1012 GameSlot *targetSlot = m_slotList[m_connectionNodes[m_targetNodeNumber].m_slotIndex];
1013 DEBUG_ASSERTCRASH(targetSlot != NULL, ("NAT::gotMangledPort - targetSlot is NULL"));
1014 if (targetSlot == NULL) {
1015 DEBUG_LOG(("NAT::gotMangledPort - targetSlot is NULL, failed this connection\n"));
1016 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
1017 return;
1018 }
1019
1020 GameSlot *localSlot = m_slotList[m_connectionNodes[m_localNodeNumber].m_slotIndex];
1021 DEBUG_ASSERTCRASH(localSlot != NULL, ("NAT::gotMangledPort - localSlot is NULL, WTF?"));
1022 if (localSlot == NULL) {
1023 DEBUG_LOG(("NAT::gotMangledPort - localSlot is NULL, failed this connection\n"));
1024 setConnectionState(m_localNodeNumber, NATCONNECTIONSTATE_FAILED);
1025 return;
1026 }
1027
1028 if (nodeNumber != m_targetNodeNumber) {
1029 DEBUG_LOG(("NAT::gotMangledPort - got a mangled port number for someone that isn't my target. node = %d, target node = %d\n", nodeNumber, m_targetNodeNumber));
1030 return;
1031 }
1032
1033 targetSlot->setPort(mangledPort);
1034 DEBUG_LOG(("NAT::gotMangledPort - got mangled port number %d from our target node (%ls)\n", mangledPort, targetSlot->getName().str()));
1035 if (((localSlot->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) == 0) || (m_beenProbed == TRUE) ||
1036 (((localSlot->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) != 0) && ((targetSlot->getNATBehavior() & FirewallHelperClass::FIREWALL_TYPE_NETGEAR_BUG) != 0))) {
1037#ifdef DEBUG_LOGGING
1038 UnsignedInt ip = targetSlot->getIP();
1039#endif
1040 DEBUG_LOG(("NAT::gotMangledPort - don't have a netgear or we have already been probed, or both my target and I have a netgear, send a PROBE. Sending to %d.%d.%d.%d:%d\n",
1041 ip >> 24, (ip >> 16) & 0xff, (ip >> 8) & 0xff, ip & 0xff, targetSlot->getPort()));
1042
1043 sendAProbe(targetSlot->getIP(), targetSlot->getPort(), m_localNodeNumber);

1235// Int node = atoi(ptr + strlen("CONNDONE"));
