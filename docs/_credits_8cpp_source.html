186void CreditsManager::update( void )
187{
188 if(m_isFinished)
189 return;
190 m_framesSinceStarted++;
191
192 if(m_framesSinceStarted%m_scrollRatePerFrames != 0)
193 return;
194
195
196 Int y = 0;
197 Int yTest = 0;
198 Int lastHeight = 0;
199 Int start = m_scrollDown? 0:TheDisplay->getHeight();
200 Int end =m_scrollDown? TheDisplay->getHeight():0;
201 Int offsetStartMultiplyer = m_scrollDown? -1:0; // if we're scrolling from the top, we need to subtract the height
202 Int offsetEndMultiplyer = m_scrollDown? 0:1;
203 Int directionMultiplyer = m_scrollDown? 1:-1;
204 CreditsLineList::iterator drawIt = m_displayedCreditLineList.begin();
205 while (drawIt != m_displayedCreditLineList.end())
206 {
207 CreditsLine *cLine = *drawIt;
208 y = cLine->m_pos.y = cLine->m_pos.y + (m_scrollRate * directionMultiplyer);
209 lastHeight = cLine->m_height;
210 yTest = y + ((lastHeight + CREDIT_SPACE_OFFSET) * offsetEndMultiplyer);
211 if(((m_scrollDown && (yTest > end)) || (!m_scrollDown && (yTest < end))))
212 {
213 TheDisplayStringManager->freeDisplayString(cLine->m_displayString);
214 TheDisplayStringManager->freeDisplayString(cLine->m_secondDisplayString);
215 cLine->m_displayString = NULL;
216 cLine->m_secondDisplayString = NULL;
217 drawIt = m_displayedCreditLineList.erase(drawIt);
218 }
219 else
220 drawIt++;
221 }
222
223 y= y + ((lastHeight + CREDIT_SPACE_OFFSET) * offsetStartMultiplyer);
224
225 // is it time to add a new string?
226 if(!((m_scrollDown && (yTest >= start)) || (!m_scrollDown && (yTest <= start))))
227 return;
228
229 if(m_displayedCreditLineList.size() == 0 && m_creditLineListIt == m_creditLineList.end())
230 m_isFinished = TRUE;
231
232 if(m_creditLineListIt == m_creditLineList.end())
233 return;
234
235 CreditsLine *cLine = *m_creditLineListIt;
236 ICoord2D pos;
237 switch (cLine->m_style)
238 {
239 case CREDIT_STYLE_TITLE:
240 {
241 cLine->m_color = m_titleColor;
242
243 if(TheGlobalLanguageData&& !cLine->m_text.isEmpty())
244 {
245 DisplayString *ds = TheDisplayStringManager->newDisplayString();
246 if(!ds)
247 return;
248 ds->setFont(TheFontLibrary->getFont(TheGlobalLanguageData->m_creditsTitleFont.name,
249 TheGlobalLanguageData->adjustFontSize(TheGlobalLanguageData->m_creditsTitleFont.size),
250 TheGlobalLanguageData->m_creditsTitleFont.bold));
251 ds->setText(cLine->m_text);
252 ds->getSize(&pos.x,&pos.y);
253 cLine->m_height = pos.y;
254 cLine->m_pos.x = TheDisplay->getWidth()/2 - pos.x/2 ;
255 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
256 cLine->m_displayString = ds;
257 }
258 }
259 break;
260 case CREDIT_STYLE_POSITION:
261 {
262 cLine->m_color = m_positionColor;
263
264 if(TheGlobalLanguageData && !cLine->m_text.isEmpty())
265 {
266 DisplayString *ds = TheDisplayStringManager->newDisplayString();
267 if(!ds)
268 return;
269 ds->setFont(TheFontLibrary->getFont(TheGlobalLanguageData->m_creditsPositionFont.name,
270 TheGlobalLanguageData->adjustFontSize(TheGlobalLanguageData->m_creditsPositionFont.size),
271 TheGlobalLanguageData->m_creditsPositionFont.bold));
272 ds->setText(cLine->m_text);
273 ds->getSize(&pos.x,&pos.y);
274 cLine->m_height = pos.y;
275 cLine->m_pos.x = TheDisplay->getWidth()/2 - pos.x/2 ;
276 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
277 cLine->m_displayString = ds;
278 }
279 }
280 break;
281 case CREDIT_STYLE_NORMAL:
282 {
283 cLine->m_color = m_normalColor;
284
285 if(TheGlobalLanguageData && !cLine->m_text.isEmpty())
286 {
287 DisplayString *ds = TheDisplayStringManager->newDisplayString();
288 if(!ds)
289 return;
290 ds->setFont(TheFontLibrary->getFont(TheGlobalLanguageData->m_creditsNormalFont.name,
291 TheGlobalLanguageData->adjustFontSize(TheGlobalLanguageData->m_creditsNormalFont.size),
292 TheGlobalLanguageData->m_creditsNormalFont.bold));
293 ds->setText(cLine->m_text);
294 ds->getSize(&pos.x,&pos.y);
295 cLine->m_height = pos.y;
296 cLine->m_pos.x = TheDisplay->getWidth()/2 - pos.x/2 ;
297 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
298 cLine->m_displayString = ds;
299 }
300 }
301 break;
302 case CREDIT_STYLE_COLUMN:
303 {
304 cLine->m_color = m_normalColor;
305
306 if(TheGlobalLanguageData && !cLine->m_text.isEmpty())
307 {
308 DisplayString *ds = TheDisplayStringManager->newDisplayString();
309 if(!ds)
310 return;
311 ds->setFont(TheFontLibrary->getFont(TheGlobalLanguageData->m_creditsNormalFont.name,
312 TheGlobalLanguageData->adjustFontSize(TheGlobalLanguageData->m_creditsNormalFont.size),
313 TheGlobalLanguageData->m_creditsNormalFont.bold));
314 ds->setText(cLine->m_text);
315 ds->getSize(&pos.x,&pos.y);
316 cLine->m_height = pos.y;
317 cLine->m_pos.x = TheDisplay->getWidth()/2 - pos.x/2 ;
318 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
319 cLine->m_displayString = ds;
320 }
321 if(TheGlobalLanguageData && !cLine->m_secondText.isEmpty())
322 {
323 DisplayString *ds = TheDisplayStringManager->newDisplayString();
324 if(!ds)
325 return;
326 ds->setFont(TheFontLibrary->getFont(TheGlobalLanguageData->m_creditsNormalFont.name,
327 TheGlobalLanguageData->adjustFontSize(TheGlobalLanguageData->m_creditsNormalFont.size),
328 TheGlobalLanguageData->m_creditsNormalFont.bold));
329 ds->setText(cLine->m_secondText);
330 ds->getSize(&pos.x,&pos.y);
331 cLine->m_height = pos.y;
332 cLine->m_pos.x = TheDisplay->getWidth()/2 - pos.x/2 ;
333 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
334 cLine->m_secondDisplayString = ds;
335
336 }
337 }
338 break;
339 case CREDIT_STYLE_BLANK:
340 {
341 cLine->m_height = m_normalFontHeight;
342 cLine->m_pos.y = start + (cLine->m_height * offsetStartMultiplyer);
343 }
344 break;
345 }
346
347 m_displayedCreditLineList.push_back(cLine);
348
349 if(m_creditLineListIt != m_creditLineList.end())
350 m_creditLineListIt++;
351

354void CreditsManager::draw( void )
355{
356 CreditsLineList::iterator drawIt = m_displayedCreditLineList.begin();
357 while (drawIt != m_displayedCreditLineList.end())
358 {
359 CreditsLine *cLine = *drawIt;
360 Int heightChunk = TheDisplay->getHeight()/3;
361 Real perc = 0.0f;
362 if(cLine->m_pos.y < heightChunk || cLine->m_pos.y > heightChunk * 2)
363 {
364 // adjust the color
365 if( cLine->m_pos.y < 0 || cLine->m_pos.y > TheDisplay->getHeight())
366 perc = 0.0f;
367 else if( cLine->m_pos.y < heightChunk)
368 perc = INT_TO_REAL(cLine->m_pos.y) / heightChunk;
369 else
370 perc = 1 - INT_TO_REAL(cLine->m_pos.y - 2*heightChunk) / heightChunk;
371 }
372 else
373 perc = 1.0f;
374 UnsignedByte r,g,b,a;
375 GameGetColorComponents(cLine->m_color, &r, &g, &b, &a);
376 Int color = GameMakeColor( r,g,b, a * perc);
377 Int bColor= GameMakeColor( 0,0,0, a * perc);
378
379 switch (cLine->m_style) {
380 case CREDIT_STYLE_TITLE:
381 case CREDIT_STYLE_POSITION:
382 case CREDIT_STYLE_NORMAL:
383 {
384 if(cLine->m_displayString)
385 cLine->m_displayString->draw(cLine->m_pos.x,cLine->m_pos.y,color, bColor, 1,1 );
386 }
387 break;
388 case CREDIT_STYLE_COLUMN:
389 {
390 Int chunk = TheDisplay->getWidth()/3;
391 ICoord2D pos;
392 if(cLine->m_displayString)
393 {
394 cLine->m_displayString->getSize(&pos.x, &pos.y);
395 cLine->m_displayString->draw(chunk - (pos.x/2),cLine->m_pos.y,color, bColor, 1,1 );
396 }
397 if(cLine->m_secondDisplayString)
398 {
399 cLine->m_secondDisplayString->getSize(&pos.x, &pos.y);
400 cLine->m_secondDisplayString->draw(2*chunk - (pos.x/2),cLine->m_pos.y,color, bColor, 1,1 );
401 }
402 }
403 break;
404 }
405
406 drawIt++;
407 }

431void CreditsManager::addText( AsciiString text )
432{
433 CreditsLine *cLine = new CreditsLine;
434
435 switch (m_currentStyle)
436 {
437 case CREDIT_STYLE_TITLE:
438 case CREDIT_STYLE_POSITION:
439 case CREDIT_STYLE_NORMAL:
440 {
441 cLine->m_text = getUnicodeString(text);
442 cLine->m_style = m_currentStyle;
443 m_creditLineList.push_back(cLine);
444 }
445 break;
446 case CREDIT_STYLE_COLUMN:
447 {
448 CreditsLineList::reverse_iterator rIt = m_creditLineList.rbegin();
449 CreditsLine *rcLine = *rIt;
450 if(rIt == m_creditLineList.rend() || rcLine->m_style != CREDIT_STYLE_COLUMN
451 || (rcLine->m_style == CREDIT_STYLE_COLUMN && rcLine->m_done == TRUE))
452 {
453 cLine->m_text = getUnicodeString(text);
454 cLine->m_style = CREDIT_STYLE_COLUMN;
455 cLine->m_useSecond = TRUE;
456 m_creditLineList.push_back(cLine);
457 }
458 else
459 {
460 rcLine->m_secondText = getUnicodeString(text);
461 rcLine->m_done = TRUE;
462 delete cLine;
463 }
464
465 }
466 break;
467 default:
468 DEBUG_ASSERTCRASH( FALSE, ("CreditsManager::addText we tried to add a credit text with the wrong style before it. Style is %d\n", m_currentStyle) );
469 delete cLine;
470 }
