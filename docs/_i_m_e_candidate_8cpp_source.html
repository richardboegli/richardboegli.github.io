93void IMECandidateTextAreaDraw( GameWindow *window, WinInstanceData *instData )
94{
95 // set up for rendering
96 ICoord2D origin, size, start, end;
97 Color textColor,
98 textBorder,
99 textSelectColor,
100 textSelectBorder;
101 IRegion2D textRegion;
102 Color black = GameMakeColor( 0, 0, 0, 255);
103
104 // get window position and size
105 window->winGetScreenPosition( &origin.x, &origin.y );
106 window->winGetSize( &size.x, &size.y );
107
108 // get a nice region from the positions
109 textRegion.lo.x = origin.x;
110 textRegion.lo.y = origin.y;
111 textRegion.hi.x = origin.x + size.x;
112 textRegion.hi.y = origin.y + size.y;
113
114 // get the right colors for drawing
115 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
116 {
117
118 textSelectColor = window->winGetDisabledTextColor();
119 textSelectBorder = window->winGetDisabledTextBorderColor();
120 textColor = window->winGetDisabledTextColor();
121 textBorder = window->winGetDisabledTextBorderColor();
122
123 } // end if, disabled
124 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
125 {
126
127 textColor = window->winGetEnabledTextColor();
128 textBorder = window->winGetEnabledTextBorderColor();
129 textSelectColor = window->winGetHiliteTextColor();
130 textSelectBorder = window->winGetHiliteTextBorderColor();
131
132 } // end else if, hilited
133 else
134 {
135
136 textSelectColor = window->winGetHiliteTextColor();
137 textSelectBorder = window->winGetHiliteTextBorderColor();
138 textColor = window->winGetEnabledTextColor();
139 textBorder = window->winGetEnabledTextBorderColor();
140
141 } // end else, just enabled
142
143
144 {
145 Real borderWidth = 1.0f;
146
147 start.x = origin.x;
148 start.y = origin.y;
149 end.x = start.x + size.x;
150 end.y = start.y + size.y;
151 TheWindowManager->winOpenRect( black, borderWidth,
152 start.x, start.y, end.x, end.y );
153 }
154
155 if ( Dstring == NULL )
156 {
157 return;
158 }
159
160 IMEManagerInterface *ime = (IMEManagerInterface*)window->winGetUserData();
161
162 if ( ime == NULL )
163 {
164 return;
165 }
166
167 GameFont *font = window->winGetFont() ;
168 Int height;
169
170 // set the font
171 Dstring->setFont( font );
172
173 // cacl line height
174 height = font->height + IMECandidateWindowLineSpacing;
175
176 // set the clip region
177 Dstring->setClipRegion( &textRegion );
178
179 Int first = ime->getCandidatePageStart();
180 Int total = ime->getCandidateCount();
181 Int pageSize = ime->getCandidatePageSize();
182 Int selected = ime->getSelectedCandidateIndex();
183
184 Int count = pageSize;
185
186 if ( count + first > total )
187 {
188 count = total - first;
189 }
190
191 selected = selected - first;
192 UnicodeString number;
193
194 // calulate the widest number text
195 Int width;
196 Dstring->setText(UnicodeString(L"00:"));
197 width = Dstring->getWidth();
198
199 // calc y start pos
200 Int y = origin.y;
201 Int leftEdge = origin.x + 10 + width;
202
203 for ( Int i = 0; i < count; i++, y+= height )
204 {
205 UnicodeString *candidate = ime->getCandidate( first + i );
206 Int tcolor, bcolor;
207
208 if ( i == selected )
209 {
210 tcolor = textSelectColor;
211 bcolor = textSelectBorder;
212 }
213 else
214 {
215 tcolor = textColor;
216 bcolor = textBorder;
217 }
218
219 // draw number tab first
220 number.format( L"%d:", i + ime->getIndexBase());
221 Dstring->setText( number );
222 width = Dstring->getWidth();
223 Dstring->draw( leftEdge - width, y, tcolor, black);
224
225 // draw candidate
226 Dstring->setText( *candidate );
227 Dstring->draw( leftEdge, y, tcolor, black );
228 }
229}

235{
236 // set up for rendering
237 ICoord2D origin, size, start, end;
238 Color backColor,
239 backBorder;
240 Real borderWidth = 1.0f;
241
242 // get window position and size
243 window->winGetScreenPosition( &origin.x, &origin.y );
244 window->winGetSize( &size.x, &size.y );
245
246 // get the right colors for drawing
247 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
248 {
249
250 backColor = window->winGetDisabledColor( 0 );
251 backBorder = window->winGetDisabledBorderColor( 0 );
252
253 } // end if, disabled
254 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
255 {
256
257 backColor = window->winGetHiliteColor( 0 );
258 backBorder = window->winGetHiliteBorderColor( 0 );
259
260 } // end else if, hilited
261 else
262 {
263
264 backColor = window->winGetEnabledColor( 0 );
265 backBorder = window->winGetEnabledBorderColor( 0 );
266
267 } // end else, just enabled
268
269 // draw the back border
270 if( backBorder != WIN_COLOR_UNDEFINED )
271 {
272
273 start.x = origin.x;
274 start.y = origin.y;
275 end.x = start.x + size.x;
276 end.y = start.y + size.y;
277 TheWindowManager->winOpenRect( backBorder, borderWidth,
278 start.x, start.y, end.x, end.y );
279 } // end if
280
281 // draw the filled back
282 if( backColor != WIN_COLOR_UNDEFINED )
283 {
284
285 start.x = origin.x + 1;
286 start.y = origin.y + 1;
287 end.x = start.x + size.x - 2;
288 end.y = start.y + size.y - 2;
289 TheWindowManager->winFillRect( backColor, 0, start.x, start.y, end.x, end.y );
290
291 } // end if
292}
