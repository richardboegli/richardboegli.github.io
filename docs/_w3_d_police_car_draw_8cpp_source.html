47W3DDynamicLight *W3DPoliceCarDraw::createDynamicLight( void )
48{
49 W3DDynamicLight *light = NULL;
50
51 // get me a dynamic light from the scene
52 light = W3DDisplay::m_3DScene->getADynamicLight();
53 if( light )
54 {
55
56 light->setEnabled( TRUE );
57 light->Set_Ambient( Vector3( 0.0f, 0.0f, 0.0f ) );
58 // Use all ambient, and no diffuse. This produces a circle of light on
59 // even and uneven ground. Diffuse lighting shows up ground unevenness, which looks
60 // funny on a searchlight. So no diffuse. jba.
61 light->Set_Diffuse( Vector3( 0.0f, 0.0f, 0.0f ) );
62 light->Set_Position( Vector3( 0.0f, 0.0f, 0.0f ) );
63 light->Set_Far_Attenuation_Range( 5, 15 );
64
65 } // end if
66
67 return light;
68
69} // end createDynamicSearchLight

101void W3DPoliceCarDraw::doDrawModule(const Matrix3D* transformMtx)
102{
103 const Real floatAmt = 8.0f;
104 const Real animAmt = 0.25;
105
106 // get pointers to our render objects that we'll need
107 RenderObjClass* policeCarRenderObj = getRenderObject();
108 if( policeCarRenderObj == NULL )
109 return;
110
111 HAnimClass *anim = policeCarRenderObj->Peek_Animation();
112 if (anim)
113 {
114 Real frames = anim->Get_Num_Frames();
115 m_curFrame += animAmt;
116 if (m_curFrame > frames-1) {
117 m_curFrame = 0;
118 }
119 policeCarRenderObj->Set_Animation(anim, m_curFrame);
120 }
121 Real red = 0;
122 Real green = 0;
123 Real blue = 0;
124 if (m_curFrame < 3) {
125 red = 1; green = 0.5;
126 } else if (m_curFrame < 6) {
127 red = 1;
128 } else if (m_curFrame < 7) {
129 red = 1; green = 0.5;
130 } else if (m_curFrame < 9) {
131 red = 0.5+(9-m_curFrame)/4;
132 blue = (m_curFrame-5)/6;
133 } else if (m_curFrame < 12) {
134 blue=1;
135 } else if (m_curFrame <= 14) {
136 green = (m_curFrame-11)/3;
137 blue = (14-m_curFrame)/2;
138 red = (m_curFrame-11)/3;
139 }
140
141 // make us a light if we don't already have one
142 if( m_light == NULL )
143 m_light = createDynamicLight();
144
145
146 // if we have a search light, position it
147 if( m_light )
148 {
149 Coord3D pos = *getDrawable()->getPosition();
150 m_light->Set_Diffuse( Vector3( red, green, blue) );
151 m_light->Set_Ambient( Vector3( red/2, green/2, blue/2) );
152 m_light->Set_Far_Attenuation_Range( 3, 20 );
153 m_light->Set_Position( Vector3( pos.x,pos.y,pos.z+floatAmt ) );
154 }
155 W3DTruckDraw::doDrawModule(transformMtx);
156}
