64static LadderInfo *parseLadder(AsciiString raw)
65{
66 DEBUG_LOG(("Looking at ladder:\n%s\n", raw.str()));
67 LadderInfo *lad = NULL;
68 AsciiString line;
69 while (raw.nextToken(&line, "\n"))
70 {
71 if (line.getCharAt(line.getLength()-1) == '\r')
72 line.removeLastChar(); // there is a trailing '\r'
73
74 line.trim();
75
76 if (line.isEmpty())
77 continue;
78
79 // woohoo! got a line!
80 line.trim();
81 if ( !lad && line.startsWith("<Ladder ") )
82 {
83 // start of a ladder def
84 lad = NEW LadderInfo;
85
86 // fill in some info
87 AsciiString tokenName, tokenAddr, tokenPort, tokenHomepage;
88 line.removeLastChar(); // the '>'
89 line = line.str() + 7; // the "<Ladder "
90 line.nextToken(&tokenAddr, "\" ");
91 line.nextToken(&tokenPort, " ");
92 line.nextToken(&tokenHomepage, " ");
93
94 lad->name = MultiByteToWideCharSingleLine(tokenName.str()).c_str();
95 while (lad->name.getLength() > 20)
96 lad->name.removeLastChar(); // Per Harvard's request, ladder names are limited to 20 chars
97 lad->address = tokenAddr;
98 lad->port = atoi(tokenPort.str());
99 lad->homepageURL = tokenHomepage;
100 }
101 else if ( lad && line.startsWith("Name ") )
102 {
103 lad->name = MultiByteToWideCharSingleLine(line.str() + 5).c_str();
104 }
105 else if ( lad && line.startsWith("Desc ") )
106 {
107 lad->description = MultiByteToWideCharSingleLine(line.str() + 5).c_str();
108 }
109 else if ( lad && line.startsWith("Loc ") )
110 {
111 lad->location = MultiByteToWideCharSingleLine(line.str() + 4).c_str();
112 }
113 else if ( lad && line.startsWith("TeamSize ") )
114 {
115 lad->playersPerTeam = atoi(line.str() + 9);
116 }
117 else if ( lad && line.startsWith("RandomMaps ") )
118 {
119 lad->randomMaps = atoi(line.str() + 11);
120 }
121 else if ( lad && line.startsWith("RandomFactions ") )
122 {
123 lad->randomFactions = atoi(line.str() + 15);
124 }
125 else if ( lad && line.startsWith("Faction ") )
126 {
127 AsciiString faction = line.str() + 8;
128 AsciiStringList outStringList;
129 ThePlayerTemplateStore->getAllSideStrings(&outStringList);
130
131 AsciiStringList::iterator aIt = std::find(outStringList.begin(), outStringList.end(), faction);
132 if (aIt != outStringList.end())
133 {
134 // valid faction - now check for dupes
135 aIt = std::find(lad->validFactions.begin(), lad->validFactions.end(), faction);
136 if (aIt == lad->validFactions.end())
137 {
138 lad->validFactions.push_back(faction);
139 }
140 }
141 }
142 /*
143 else if ( lad && line.startsWith("QM ") )
144 {
145 lad->validQM = atoi(line.str() + 3);
146 }
147 else if ( lad && line.startsWith("Custom ") )
148 {
149 lad->validCustom = atoi(line.str() + 7);
150 }
151 */
152 else if ( lad && line.startsWith("MinWins ") )
153 {
154 lad->minWins = atoi(line.str() + 8);
155 }
156 else if ( lad && line.startsWith("MaxWins ") )
157 {
158 lad->maxWins = atoi(line.str() + 8);
159 }
160 else if ( lad && line.startsWith("CryptedPass ") )
161 {
162 lad->cryptedPassword = line.str() + 12;
163 }
164 else if ( lad && line.compare("</Ladder>") == 0 )
165 {
166 DEBUG_LOG(("Saw a ladder: name=%ls, addr=%s:%d, players=%dv%d, pass=%s, replay=%d, homepage=%s\n",
167 lad->name.str(), lad->address.str(), lad->port, lad->playersPerTeam, lad->playersPerTeam, lad->cryptedPassword.str(),
168 lad->submitReplay, lad->homepageURL.str()));
169 // end of a ladder
170 if (lad->playersPerTeam >= 1 && lad->playersPerTeam <= MAX_SLOTS/2)
171 {
172 if (lad->validFactions.size() == 0)
173 {
174 DEBUG_LOG(("No factions specified. Using all.\n"));
175 lad->validFactions.clear();
176 Int numTemplates = ThePlayerTemplateStore->getPlayerTemplateCount();
177 for ( Int i = 0; i < numTemplates; ++i )
178 {
179 const PlayerTemplate *pt = ThePlayerTemplateStore->getNthPlayerTemplate(i);
180 if (!pt)
181 continue;
182
183 if (pt->isPlayableSide() && pt->getSide().compare("Boss") != 0 )
184 lad->validFactions.push_back(pt->getSide());
185 }
186 }
187 else
188 {
189 AsciiStringList validFactions = lad->validFactions;
190 for (AsciiStringListIterator it = validFactions.begin(); it != validFactions.end(); ++it)
191 {
192 AsciiString faction = *it;
193 AsciiString marker;
194 marker.format("INI:Faction%s", faction.str());
195 DEBUG_LOG(("Faction %s has marker %s corresponding to str %ls\n", faction.str(), marker.str(), TheGameText->fetch(marker).str()));
196 }
197 }
198
199 if (lad->validMaps.size() == 0)
200 {
201 DEBUG_LOG(("No maps specified. Using all.\n"));
202 std::list<AsciiString> qmMaps = TheGameSpyConfig->getQMMaps();
203 for (std::list<AsciiString>::const_iterator it = qmMaps.begin(); it != qmMaps.end(); ++it)
204 {
205 AsciiString mapName = *it;
206
207 // check sizes on the maps before allowing them
208 const MapMetaData *md = TheMapCache->findMap(mapName);
209 if (md && md->m_numPlayers >= lad->playersPerTeam*2)
210 {
211 lad->validMaps.push_back(mapName);
212 }
213 }
214 }
215 return lad;
216 }
217 else
218 {
219 // no maps? don't play on it!
220 delete lad;
221 lad = NULL;
222 return NULL;
223 }
224 }
225 else if ( lad && line.startsWith("Map ") )
226 {
227 // valid map
228 AsciiString mapName = line.str() + 4;
229 mapName.trim();
230 if (mapName.isNotEmpty())
231 {
232 mapName.format("%s\\%s\\%s.map", TheMapCache->getMapDir().str(), mapName.str(), mapName.str());
233 mapName = TheGameState->portableMapPathToRealMapPath(TheGameState->realMapPathToPortableMapPath(mapName));
234 mapName.toLower();
235 std::list<AsciiString> qmMaps = TheGameSpyConfig->getQMMaps();
236 if (std::find(qmMaps.begin(), qmMaps.end(), mapName) != qmMaps.end())
237 {
238 // check sizes on the maps before allowing them
239 const MapMetaData *md = TheMapCache->findMap(mapName);
240 if (md && md->m_numPlayers >= lad->playersPerTeam*2)
241 lad->validMaps.push_back(mapName);
242 }
243 }
244 }
245 else
246 {
247 // bad ladder - kill it
248 delete lad;
249 lad = NULL;
250 }
251 }
252
253 if (lad)
254 {
255 delete lad;
256 lad = NULL;
257 }
258 return NULL;
259}

261LadderList::LadderList()
262{
263 //Int profile = TheGameSpyInfo->getLocalProfileID();
264
265 AsciiString rawMotd = TheGameSpyConfig->getLeftoverConfig();
266 AsciiString line;
267 Bool inLadders = FALSE;
268 Bool inSpecialLadders = FALSE;
269 Bool inLadder = FALSE;
270 LadderInfo *lad = NULL;
271 Int index = 1;
272 AsciiString rawLadder;
273
274 while (rawMotd.nextToken(&line, "\n"))
275 {
276 if (line.getCharAt(line.getLength()-1) == '\r')
277 line.removeLastChar(); // there is a trailing '\r'
278
279 line.trim();
280
281 if (line.isEmpty())
282 continue;
283
284 if (!inLadders && line.compare("<Ladders>") == 0)
285 {
286 inLadders = TRUE;
287 rawLadder.clear();
288 }
289 else if (inLadders && line.compare("</Ladders>") == 0)
290 {
291 inLadders = FALSE;
292 }
293 else if (!inSpecialLadders && line.compare("<SpecialLadders>") == 0)
294 {
295 inSpecialLadders = TRUE;
296 rawLadder.clear();
297 }
298 else if (inSpecialLadders && line.compare("</SpecialLadders>") == 0)
299 {
300 inSpecialLadders = FALSE;
301 }
302 else if (inLadders || inSpecialLadders)
303 {
304 if (line.startsWith("<Ladder ") && !inLadder)
305 {
306 inLadder = TRUE;
307 rawLadder.clear();
308 rawLadder.concat(line);
309 rawLadder.concat('\n');
310 }
311 else if (line.compare("</Ladder>") == 0 && inLadder)
312 {
313 inLadder = FALSE;
314 rawLadder.concat(line);
315 rawLadder.concat('\n');
316 if ((lad = parseLadder(rawLadder)) != NULL)
317 {
318 lad->index = index++;
319 if (inLadders)
320 {
321 DEBUG_LOG(("Adding to standard ladders\n"));
322 m_standardLadders.push_back(lad);
323 }
324 else
325 {
326 DEBUG_LOG(("Adding to special ladders\n"));
327 m_specialLadders.push_back(lad);
328 }
329 }
330 rawLadder.clear();
331 }
332 else if (inLadder)
333 {
334 rawLadder.concat(line);
335 rawLadder.concat('\n');
336 }
337 }
338 }
339
340 // look for local ladders
341 loadLocalLadders();
342
343 DEBUG_LOG(("After looking for ladders, we have %d local, %d special && %d normal\n", m_localLadders.size(), m_specialLadders.size(), m_standardLadders.size()));
344}

452void LadderList::loadLocalLadders( void )
453{
454 AsciiString dirname;
455 dirname.format("%sGeneralsOnline\\Ladders\\", TheGlobalData->getPath_UserData().str());
456 FilenameList filenameList;
457 TheFileSystem->getFileListInDirectory(dirname, AsciiString("*.ini"), filenameList, TRUE);
458
459 Int index = -1;
460
461 FilenameList::iterator it = filenameList.begin();
462 while (it != filenameList.end())
463 {

466 filename.toLower();

472void LadderList::checkLadder( AsciiString fname, Int index )
473{
474 File *fp = TheFileSystem->openFile(fname.str(), File::READ | File::TEXT);
475 char buf[1024];
476 AsciiString rawData;
477 if (fp)
478 {
479 Int len;
480 while (!fp->eof())
481 {
482 len = fp->read(buf, 1023);
483 buf[len] = 0;
484 buf[1023] = 0;
485 rawData.concat(buf);
486 }
487 fp->close();
488 fp = NULL;
489 }
490
491 DEBUG_LOG(("Read %d bytes from '%s'\n", rawData.getLength(), fname.str()));
492 if (rawData.isEmpty())
493 return;
494
495 LadderInfo *li = parseLadder(rawData);
496 if (!li)
497 {
498 return;
499 }
500
501 // sanity check
502 if (li->address.isEmpty())
503 {
504 DEBUG_LOG(("Bailing because of li->address.isEmpty()\n"));
505 delete li;
506 return;
507 }
508
509 if (!li->port)
510 {
511 DEBUG_LOG(("Bailing because of !li->port\n"));
512 delete li;
513 return;
514 }
515
516 if (li->validMaps.size() == 0)
517 {
518 DEBUG_LOG(("Bailing because of li->validMaps.size() == 0\n"));
519 delete li;
520 return;
521 }
522
523 li->index = index;
524
525 // ladders are QM-only at this point, which kinda invalidates the whole concept of local ladders. Oh well.
526 li->validQM = FALSE; // no local ladders in QM
527 li->validCustom = FALSE;
528
529 //for (Int i=0; i<4; ++i)
530 // fname.removeLastChar(); // remove .lad
531 //li->name = UnicodeString(MultiByteToWideCharSingleLine(fname.reverseFind('\\')+1).c_str());
532
533 DEBUG_LOG(("Adding local ladder %ls\n", li->name.str()));
534 m_localLadders.push_back(li);
535}
