44UpdateSleepTime UpdateModule::frameToSleepTime(
45 UnsignedInt frame1,
46 UnsignedInt frame2,
47 UnsignedInt frame3,
48 UnsignedInt frame4
49)
50{
51 DEBUG_ASSERTCRASH(frame1 != 0 && frame2 != 0 && frame3 != 0 && frame4 != 0, ("probably should not pass zero to frameToSleepTime."));
52 if (frame1 > frame2) frame1 = frame2;
53 if (frame1 > frame3) frame1 = frame3;
54 if (frame1 > frame4) frame1 = frame4;
55 UnsignedInt now = TheGameLogic->getFrame();
56 if (frame1 > now)
57 {
58 return UPDATE_SLEEP(frame1 - now);
59 }
60 else if (frame1 == now)
61 {
62 return UPDATE_SLEEP_NONE;
63 }
64 else
65 {
66 DEBUG_CRASH(("frameToSleepTime: frame is in the past. oops."));
67 return UPDATE_SLEEP_NONE;
68 }
69}

73UpdateSleepTime UpdateModule::getWakeFrame() const
74{
75 UnsignedInt now = TheGameLogic->getFrame();
76 UnsignedInt nextCallFrame = m_nextCallFrameAndPhase >> 2;
77 if (nextCallFrame > now)
78 return UPDATE_SLEEP(nextCallFrame - now);
79 else
80 return UPDATE_SLEEP_NONE;
81}

108void UpdateModule::xfer( Xfer *xfer )
109{
110
111 // version
112 const XferVersion currentVersion = 1;
113 XferVersion version = currentVersion;
114 xfer->xferVersion( &version, currentVersion );
115
116 // extend base class
117 BehaviorModule::xfer( xfer );
118
119#if defined(_DEBUG) || defined(_INTERNAL)
120 /*
121 this is a fix for the following scenario:
122
123 save a game that has an object that uses module "FOO"
124 now change the code for module "FOO" from being nonsleepy to being sleepy
125 now reload that saved game
126 as soon as "FOO" attempts to wake itself up, you're dead.
127
128 this fix simply looks to see if the module in question is now sleepy in code,
129 but was saved in sleepy form, and if so, quietly nudges a reasonable
130 value into m_nextCallFrameAndPhase.
131 */
132 #define FIX_OLD_SAVES
133#endif
134
135#ifdef FIX_OLD_SAVES
136 #ifdef ALLOW_NONSLEEPY_UPDATES
137 Bool thisModuleIsNowSleepy = (m_nextCallFrameAndPhase != 0);
138 #else
139 const Bool thisModuleIsNowSleepy = true;
140 #endif
141#endif
142
143 // next call frame and phase
144 xfer->xferUnsignedInt( &m_nextCallFrameAndPhase );
145
146 if (xfer->getXferMode() == XFER_LOAD)
147 {
148 /*
149 ensure the phase matches.
150
151 pre-Jan 2, 2003:
152 PHASE_INITIAL = 0,
153 PHASE_NORMAL = 1,
154 PHASE_FINAL = 2
155
156 post-Jan 2, 2003:
157 PHASE_INITIAL = 0,
158 PHASE_PHYSICS = 1,
159 PHASE_NORMAL = 2,
160 PHASE_FINAL = 3
161 */
162 m_nextCallFrameAndPhase &= ~3;
163 m_nextCallFrameAndPhase |= getUpdatePhase();
164 }
165
166#ifdef FIX_OLD_SAVES
167 if (xfer->getXferMode() == XFER_LOAD
168 && thisModuleIsNowSleepy
169 && m_nextCallFrameAndPhase == 0)
170 {
171 #ifdef ALLOW_NONSLEEPY_UPDATES
172 // when the file was saved, this module was nonsleepy, but now it is sleepy.
173 #else
174 // note that 'm_nextCallFrameAndPhase' should only be zero for legacy save files.
175 #endif
176 friend_setNextCallFrame(TheGameLogic->getFrame());
177 }
178#endif
179
180 // m_indexInLogic is not saved -- it's restored in gamelogic::postprocess.
181 if( xfer->getXferMode() == XFER_LOAD )
182 {
183 m_indexInLogic = -1;
184 }
185
186} // end xfer
