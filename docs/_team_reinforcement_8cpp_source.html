65BOOL TeamReinforcement::OnInitDialog()
66{
67 CPropertyPage::OnInitDialog();
68 Bool exists;
69
70 CComboBox *pCombo = (CComboBox *)GetDlgItem(IDC_TRANSPORT_COMBO);
71 EditParameter::loadTransports(pCombo);
72 AsciiString transport = m_teamDict->getAsciiString(TheKey_teamTransport, &exists);
73 Int stringNdx = pCombo->AddString(NONE_STRING);
74 Int noneNdx = stringNdx;
75 if (exists && !transport.isEmpty()) {
76 Int ndx = pCombo->FindStringExact(-1, transport.str());
77 if (ndx != CB_ERR) {
78 stringNdx = ndx;
79 }
80 }
81 pCombo->SetCurSel(stringNdx);
82
83 CButton *pCheck = (CButton *) GetDlgItem(IDC_TEAM_STARTS_FULL);
84 Bool teamStartsFull = m_teamDict->getBool(TheKey_teamStartsFull, &exists);
85 pCheck->SetCheck(teamStartsFull?1:0);
86
87 pCheck = (CButton *) GetDlgItem(IDC_TRANSPORTS_EXIT);
88 Bool transportsExit = m_teamDict->getBool(TheKey_teamTransportsExit, &exists);
89 pCheck->SetCheck(transportsExit?1:0);
90 Bool enable = !(stringNdx==noneNdx);
91 pCheck->EnableWindow(enable);
92 pCombo->EnableWindow(enable);
93
94 pCheck = (CButton *) GetDlgItem(IDC_DEPLOY_BY);
95 pCheck->SetCheck( (stringNdx==noneNdx)?0:1);
96
97 pCombo = (CComboBox *)GetDlgItem(IDC_VETERANCY);
98 Int value = m_teamDict->getInt(TheKey_teamVeterancy, &exists);
99 pCombo->SetCurSel(value);
100
101 pCombo = (CComboBox *)GetDlgItem(IDC_WAYPOINT_COMBO);
102 EditParameter::loadWaypoints(pCombo);
103 AsciiString homeWaypoint = m_teamDict->getAsciiString(TheKey_teamReinforcementOrigin, &exists);
104 stringNdx = pCombo->AddString(NONE_STRING);
105 if (exists) {
106 Int ndx = pCombo->FindStringExact(-1, homeWaypoint.str());
107 if (ndx != CB_ERR) {
108 stringNdx = ndx;
109 }
110 }
111 pCombo->SetCurSel(stringNdx);
112
113
114 return TRUE; // return TRUE unless you set the focus to a control
115 // EXCEPTION: OCX Property Pages should return FALSE
116}

118void TeamReinforcement::OnDeployBy()
119{
120 CButton *pCheck = (CButton *) GetDlgItem(IDC_DEPLOY_BY);
121 if (pCheck->GetCheck()) {
122 CComboBox *pCombo = (CComboBox *)GetDlgItem(IDC_TRANSPORT_COMBO);
123 pCombo->SetCurSel(0);
124 pCombo->EnableWindow();
125 pCheck = (CButton *) GetDlgItem(IDC_TRANSPORTS_EXIT);
126 pCheck->EnableWindow();
127 } else {
128 CComboBox *pCombo = (CComboBox *)GetDlgItem(IDC_TRANSPORT_COMBO);
129 pCombo->SetCurSel(-1);
130 pCombo->EnableWindow(false);
131 pCheck = (CButton *) GetDlgItem(IDC_TRANSPORTS_EXIT);
132 pCheck->EnableWindow(false);
133 m_teamDict->setAsciiString(TheKey_teamTransport, AsciiString::TheEmptyString);
134 }
135}
