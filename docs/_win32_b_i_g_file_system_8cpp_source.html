85ArchiveFile * Win32BIGFileSystem::openArchiveFile(const Char *filename) {
86 File *fp = TheLocalFileSystem->openFile(filename, File::READ | File::BINARY);
87 AsciiString archiveFileName;
88 archiveFileName = filename;
89 archiveFileName.toLower();
90 Int archiveFileSize = 0;
91 Int numLittleFiles = 0;
92
93 ArchiveFile *archiveFile = NEW Win32BIGFile;
94
95 DEBUG_LOG(("Win32BIGFileSystem::openArchiveFile - opening BIG file %s\n", filename));
96
97 if (fp == NULL) {
98 DEBUG_CRASH(("Could not open archive file %s for parsing", filename));
99 return NULL;
100 }
101
102 AsciiString asciibuf;
103 char buffer[_MAX_PATH];
104 fp->read(buffer, 4); // read the "BIG" at the beginning of the file.
105 buffer[4] = 0;
106 if (strcmp(buffer, BIGFileIdentifier) != 0) {
107 DEBUG_CRASH(("Error reading BIG file identifier in file %s", filename));
108 fp->close();
109 fp = NULL;
110 return NULL;
111 }
112
113 // read in the file size.
114 fp->read(&archiveFileSize, 4);
115
116 DEBUG_LOG(("Win32BIGFileSystem::openArchiveFile - size of archive file is %d bytes\n", archiveFileSize));
117
118// char t;
119
120 // read in the number of files contained in this BIG file.
121 // change the order of the bytes cause the file size is in reverse byte order for some reason.
122 fp->read(&numLittleFiles, 4);
123 numLittleFiles = ntohl(numLittleFiles);
124
125 DEBUG_LOG(("Win32BIGFileSystem::openArchiveFile - %d are contained in archive\n", numLittleFiles));
126// for (Int i = 0; i < 2; ++i) {
127// t = buffer[i];
128// buffer[i] = buffer[(4-i)-1];
129// buffer[(4-i)-1] = t;
130// }
131
132 // seek to the beginning of the directory listing.
133 fp->seek(0x10, File::START);
134 // read in each directory listing.
135 ArchivedFileInfo *fileInfo = NEW ArchivedFileInfo;
136
137 for (Int i = 0; i < numLittleFiles; ++i) {
138 Int filesize = 0;
139 Int fileOffset = 0;
140 fp->read(&fileOffset, 4);
141 fp->read(&filesize, 4);
142
143 filesize = ntohl(filesize);
144 fileOffset = ntohl(fileOffset);
145
146 fileInfo->m_archiveFilename = archiveFileName;
147 fileInfo->m_offset = fileOffset;
148 fileInfo->m_size = filesize;
149
150 // read in the path name of the file.
151 Int pathIndex = -1;
152 do {
153 ++pathIndex;
154 fp->read(buffer + pathIndex, 1);
155 } while (buffer[pathIndex] != 0);
156
157 Int filenameIndex = pathIndex;
158 while ((buffer[filenameIndex] != '\\') && (buffer[filenameIndex] != '/') && (filenameIndex >= 0)) {
159 --filenameIndex;
160 }
161
162 fileInfo->m_filename = (char *)(buffer + filenameIndex + 1);
163 fileInfo->m_filename.toLower();
164 buffer[filenameIndex + 1] = 0;
165
166 AsciiString path;
167 path = buffer;
168
169 AsciiString debugpath;
170 debugpath = path;
171 debugpath.concat(fileInfo->m_filename);
172// DEBUG_LOG(("Win32BIGFileSystem::openArchiveFile - adding file %s to archive file %s, file number %d\n", debugpath.str(), fileInfo->m_archiveFilename.str(), i));
173
174 archiveFile->addFile(path, fileInfo);
175 }
176
177 archiveFile->attachFile(fp);
178
179 delete fileInfo;
180 fileInfo = NULL;
181
182 // leave fp open as the archive file will be using it.
183
184 return archiveFile;
185}

187void Win32BIGFileSystem::closeArchiveFile(const Char *filename) {
188 // Need to close the specified big file
189 ArchiveFileMap::iterator it = m_archiveFileMap.find(filename);
190 if (it == m_archiveFileMap.end()) {
191 return;
192 }
193
194 if (stricmp(filename, MUSIC_BIG) == 0) {
195 // Stop the current audio
196 TheAudio->stopAudio(AudioAffect_Music);
197
198 // No need to turn off other audio, as the lookups will just fail.
199 }
200 DEBUG_ASSERTCRASH(stricmp(filename, MUSIC_BIG) == 0, ("Attempting to close Archive file '%s', need to add code to handle its shutdown correctly.", filename));
201
202 // may need to do some other processing here first.
203
204 delete (it->second);
205 m_archiveFileMap.erase(it);
206}

214Bool Win32BIGFileSystem::loadBigFilesFromDirectory(AsciiString dir, AsciiString fileMask, Bool overwrite) {
215
216 FilenameList filenameList;
217 TheLocalFileSystem->getFileListInDirectory(dir, AsciiString(""), fileMask, filenameList, TRUE);
218
219 Bool actuallyAdded = FALSE;
220 FilenameListIter it = filenameList.begin();
221 while (it != filenameList.end()) {
222 ArchiveFile *archiveFile = openArchiveFile((*it).str());
223
224 if (archiveFile != NULL) {
225 DEBUG_LOG(("Win32BIGFileSystem::loadBigFilesFromDirectory - loading %s into the directory tree.\n", (*it).str()));
226 loadIntoDirectoryTree(archiveFile, *it, overwrite);
227 m_archiveFileMap[(*it)] = archiveFile;
228 DEBUG_LOG(("Win32BIGFileSystem::loadBigFilesFromDirectory - %s inserted into the archive file map.\n", (*it).str()));
229 actuallyAdded = TRUE;
230 }
231
232 it++;
233 }
234
235 return actuallyAdded;
236}
