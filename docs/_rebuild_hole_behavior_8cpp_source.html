120void RebuildHoleBehavior::newWorkerRespawnProcess( Object *existingWorker )
121{
122 const RebuildHoleBehaviorModuleData *modData = getRebuildHoleBehaviorModuleData();
123
124 // if we have an existing worker, get rid of it
125 if( existingWorker )
126 {
127 DEBUG_ASSERTCRASH(existingWorker->getID() == m_workerID, ("m_workerID mismatch in RebuildHole"));
128 TheGameLogic->destroyObject( existingWorker );
129 }
130 m_workerID = INVALID_ID;
131
132 // set the timer for the next worker respawn
133 m_workerWaitCounter = modData->m_workerRespawnDelay;
134
135 //
136 // this method is called when a worker needs to be respawned from the hole. One of those
137 // situations is where the building was killed. Since during building reconstruction
138 // we made the hole "effectively not here" we will always want to make the hole
139 // "here again" (able to be selected, targeted by the AI etc) because it's the
140 // "focus" of this small area again
141 //
142 getObject()->maskObject( FALSE );
143
144} // end newWorkerRespawnProcess

148void RebuildHoleBehavior::startRebuildProcess( const ThingTemplate *rebuild, ObjectID spawnerID )
149{
150
151 // save what we're gonna do
152 m_rebuildTemplate = rebuild;
153
154 // store the object that spawned this hole (even though it's likely being destroyed)
155 m_spawnerObjectID = spawnerID;
156
157 // start the spawning process for a worker
158 Object *worker = TheGameLogic->findObjectByID(m_workerID);
159 newWorkerRespawnProcess( worker ); //Kill the worker if we have one.
160
161}

165void RebuildHoleBehavior::transferBombs( Object *reconstruction )
166{
167
168 Object *self = getObject();
169
170 Object *obj = TheGameLogic->getFirstObject();
171 while( obj )
172 {
173 if( obj->isKindOf( KINDOF_MINE ) )
174 {
175 static NameKeyType key_StickyBombUpdate = NAMEKEY( "StickyBombUpdate" );
176 StickyBombUpdate *update = (StickyBombUpdate*)obj->findUpdateModule( key_StickyBombUpdate );
177 if( update && update->getTargetObject() == self )
178 {
179 update->setTargetObject( reconstruction );
180 }
181 }
182 obj = obj->getNextObject();
183 }
184}

188UpdateSleepTime RebuildHoleBehavior::update( void )
189{
190 const RebuildHoleBehaviorModuleData *modData = getRebuildHoleBehaviorModuleData();
191 Object *hole = getObject();
192 Object *reconstructing = NULL;
193 Object *worker = NULL;
194
195 // get the worker object if we have one
196 if( m_workerID != 0 )
197 {
198
199 // get the worker
200 worker = TheGameLogic->findObjectByID( m_workerID );
201
202 // if the worker is no longer there, start the respawning process for a worker again
203 if( worker == NULL )
204 newWorkerRespawnProcess( NULL );
205
206 } // end if
207
208 // if we have a reconstructing object built, get the actual object pointer
209 if( m_reconstructingID != 0 )
210 {
211
212 // get object pointer
213 reconstructing = TheGameLogic->findObjectByID( m_reconstructingID );
214
215 //
216 // if that object does not exist anymore, we need to kill a worker if we have one
217 // and start the spawning process over again
218 //
219 if( reconstructing == NULL )
220 {
221 newWorkerRespawnProcess( worker );
222 m_reconstructingID = INVALID_ID;
223 } // end if
224
225 } // end if
226
227 // see if it's time for us to spawn a worker
228 if( worker == NULL && m_workerWaitCounter > 0 )
229 {
230
231 // decrement counter and respawn if it's time
232 if( --m_workerWaitCounter == 0 )
233 {
234
235 // resolve the worker template pointer if necessary
236 if( m_workerTemplate == NULL )
237 m_workerTemplate = TheThingFactory->findTemplate( modData->m_workerTemplateName );
238
239 // create a worker
240 worker = TheThingFactory->newObject( m_workerTemplate, hole->getTeam() );
241 if( worker )
242 {
243
244 // set the position of the worker to that of the hole
245 worker->setPosition( hole->getPosition() );
246
247 // save the ID of the worker spawned
248 m_workerID = worker->getID();
249
250 //
251 // tell the worker to begin construction of a building if one does not
252 // exist yet. If one does, have construction resume
253 //
254 AIUpdateInterface *ai = worker->getAIUpdateInterface();
255 if( ai )
256 {
257
258 if( reconstructing == NULL )
259 reconstructing = ai->construct( m_rebuildTemplate,
260 hole->getPosition(),
261 hole->getOrientation(),
262 hole->getControllingPlayer(),
263 TRUE );
264 else
265 ai->aiResumeConstruction( reconstructing, CMD_FROM_AI );
266
267 for ( Object *obj = TheGameLogic->getFirstObject(); obj; obj = obj->getNextObject() )
268 {
269 // Just like the building transfers attackers to the hole when it creates us, we need to transfer
270 // attackers to our replacement building before we mask ourselves.
271 AIUpdateInterface* ai = obj->getAI();
272 if (!ai)
273 continue;
274
275 ai->transferAttack(hole->getID(), reconstructing->getID());
276 }
277
278 // save the id of what we are reconstructing
279 m_reconstructingID = reconstructing->getID();
280
281 //Kris: Hacking the building to set the hole as the producer... so if the site dies, we
282 //can transfer the attack back to the hole. The object has OBJECT_STATUS_RECONSTRUCTING
283 //which we check when the object dies.
284 reconstructing->setProducer( hole );
285
286 // we want to prevent the player from selecting and doing things with this worker
287 worker->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNSELECTABLE ) );
288
289 //
290 // we want to prevent the player and the AI from selecting or targeting the hole
291 // cause the focus in this area (while reconstruction is happening) is the
292 // actual reconstructing building
293 //
294 hole->maskObject( TRUE );
295
296 transferBombs( reconstructing );
297
298 } // end if
299
300 } // end if, worker
301
302 } // end if, time to spawn a worker
303
304 } // end if, check for working respawn
305
306 // holes get auto-healed when they're sittin around
307 BodyModuleInterface *body = hole->getBodyModule();
308 if( body->getHealth() < body->getMaxHealth() )
309 {
310 DamageInfo healingInfo;
311
312 // do some healing
313 healingInfo.in.m_amount = (modData->m_holeHealthRegenPercentPerSecond / LOGICFRAMES_PER_SECOND) *
314 body->getMaxHealth();
315 healingInfo.in.m_sourceID = hole->getID();
316 healingInfo.in.m_damageType = DAMAGE_HEALING;
317 healingInfo.in.m_deathType = DEATH_NONE;
318 body->attemptHealing( &healingInfo );
319
320 } // end if
321
322 // when re-construction is complete, we remove this hole and worker
323 if( reconstructing && !reconstructing->getStatusBits().test( OBJECT_STATUS_UNDER_CONSTRUCTION ) )
324 {
325 // Transfer hole name to new building
326 TheScriptEngine->transferObjectName( hole->getName(), reconstructing );
327
328 // make the worker go away
329 if( worker )
330 TheGameLogic->destroyObject( worker );
331
332 // make the hole go away
333 TheGameLogic->destroyObject( hole );
334
335 } // end if
336
337 return UPDATE_SLEEP_NONE;
338
339} // end update

342//-------------------------------------------------------------------------------------------------
343void RebuildHoleBehavior::onDie( const DamageInfo *damageInfo )
344{
345 if( m_workerID != INVALID_ID )
346 {
347 // Our rebuilding building and us the hole can be killed in the same frame, which means we may not have
348 // deleted our generated worker since we do that in our update.
349 Object *worker = TheGameLogic->findObjectByID(m_workerID);
350 if( worker )
351 {
352 TheGameLogic->destroyObject(worker);
353 m_workerID = INVALID_ID;
354 }
355 }
356
357 Object *obj = getObject();
358
359 // destroy us
360 TheGameLogic->destroyObject( obj );
361
362} // end onDie

406void RebuildHoleBehavior::xfer( Xfer *xfer )
407{
408
409 // version
410 XferVersion currentVersion = 2;
411 XferVersion version = currentVersion;
412 xfer->xferVersion( &version, currentVersion );
413
414 // extend base class
415 UpdateModule::xfer( xfer );
416
417 // worker ID
418 xfer->xferObjectID( &m_workerID );
419
420 // reconstructing id
421 xfer->xferObjectID( &m_reconstructingID );
422
423 // spawner ID
424 if( version >= 2 )
425 xfer->xferObjectID( &m_spawnerObjectID );
426
427 // worker wait counter
428 xfer->xferUnsignedInt( &m_workerWaitCounter );
429
430 // worker template
431 AsciiString workerName = m_workerTemplate ? m_workerTemplate->getName() : AsciiString::TheEmptyString;
432 xfer->xferAsciiString( &workerName );
433 if( xfer->getXferMode() == XFER_LOAD )
434 {
435
436 if( workerName != AsciiString::TheEmptyString )
437 {
438
439 m_workerTemplate = TheThingFactory->findTemplate( workerName );
440 if( m_workerTemplate == NULL )
441 {
442
443 DEBUG_CRASH(( "RebuildHoleBehavior::xfer - Unable to find template '%s'\n",
444 workerName.str() ));
445 throw SC_INVALID_DATA;
446
447 } // end if
448
449 } // end if
450 else
451 m_workerTemplate = NULL;
452
453 } // end if
454
455 // rebuild template
456 AsciiString rebuildName = m_rebuildTemplate ? m_rebuildTemplate->getName() : AsciiString::TheEmptyString;
457 xfer->xferAsciiString( &rebuildName );
458 if( xfer->getXferMode() == XFER_LOAD )
459 {
460
461 if( rebuildName != AsciiString::TheEmptyString )
462 {
463
464 m_rebuildTemplate = TheThingFactory->findTemplate( rebuildName );
465 if( m_rebuildTemplate == NULL )
466 {
467
468 DEBUG_CRASH(( "RebuildHoleBehavior::xfer - Unable to find template '%s'\n",
469 rebuildName.str() ));
470 throw SC_INVALID_DATA;
471
472 } // end if
473
474 } // end if
475 else
476 m_rebuildTemplate = NULL;
477
478 } // end if
479
480} // end xfer
