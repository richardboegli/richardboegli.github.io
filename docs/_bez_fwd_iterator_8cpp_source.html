53void BezFwdIterator::start(void)
54{
55 mStep = 0;
56
57 if (mStepsDesired <= 1)
58 return;
59
60 float d = 1.0f / (mStepsDesired - 1);
61 float d2 = d * d;
62 float d3 = d * d2;
63
64 D3DXVECTOR4 px(mBezSeg.m_controlPoints[0].x, mBezSeg.m_controlPoints[1].x, mBezSeg.m_controlPoints[2].x, mBezSeg.m_controlPoints[3].x);
65 D3DXVECTOR4 py(mBezSeg.m_controlPoints[0].y, mBezSeg.m_controlPoints[1].y, mBezSeg.m_controlPoints[2].y, mBezSeg.m_controlPoints[3].y);
66 D3DXVECTOR4 pz(mBezSeg.m_controlPoints[0].z, mBezSeg.m_controlPoints[1].z, mBezSeg.m_controlPoints[2].z, mBezSeg.m_controlPoints[3].z);
67
68 D3DXVECTOR4 cVec[3];
69 D3DXVec4Transform(&cVec[0], &px, &BezierSegment::s_bezBasisMatrix);
70 D3DXVec4Transform(&cVec[1], &py, &BezierSegment::s_bezBasisMatrix);
71 D3DXVec4Transform(&cVec[2], &pz, &BezierSegment::s_bezBasisMatrix);
72
73 mCurrPoint = mBezSeg.m_controlPoints[0];
74
75 int i = 3;
76 while (i--) {
77 float a = cVec[i].x;
78 float b = cVec[i].y;
79 float c = cVec[i].z;
80
81 float *pD, *pDD, *pDDD;
82
83 if (i == 2) {
84 pD = &mDq.z;
85 pDD = &mDDq.z;
86 pDDD = &mDDDq.z;
87 } else if (i == 1) {
88 pD = &mDq.y;
89 pDD = &mDDq.y;
90 pDDD = &mDDDq.y;
91 } else if (i == 0) {
92 pD = &mDq.x;
93 pDD = &mDDq.x;
94 pDDD = &mDDDq.x;
95 }
96
97 (*pD) = a * d3 + b * d2 + c * d;
98 (*pDD) = 6 * a * d3 + 2 * b * d2;
99 (*pDDD) = 6 * a * d3;
100 }
101}
