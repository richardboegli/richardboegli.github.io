222void W3DTankDraw::updateTreadPositions(Real uvDelta)
223{
224 Real offset_u;
225 TreadObjectInfo *pTread=m_treads;
226
227 for (Int i=0; i<m_treadCount; i++)
228 {
229 if (pTread->m_type == TREAD_LEFT) //this tread needs to scroll forwards
230 offset_u = pTread->m_materialSettings.customUVOffset.X + uvDelta;
231 else
232 if (pTread->m_type == TREAD_RIGHT) //this tread needs to scroll backwards
233 offset_u = pTread->m_materialSettings.customUVOffset.X - uvDelta;
234
235 // ensure coordinates of offset are in [0, 1] range:
236 offset_u = offset_u - WWMath::Floor(offset_u);
237 pTread->m_materialSettings.customUVOffset.Set(offset_u,0);
238 pTread++;
239 }
240}

243void W3DTankDraw::updateTreadObjects(void)
244{
245 RenderObjClass *robj=getRenderObject();
246
247 //clear all previous tread pointers
248 for (Int i=0; i<m_treadCount; i++)
249 REF_PTR_RELEASE(m_treads[i].m_robj);
250 m_treadCount = 0;
251
252 //Make sure this object has defined a speed for tread scrolling.
253 if (getW3DTankDrawModuleData() && getW3DTankDrawModuleData()->m_treadAnimationRate && robj)
254 {
255 for (Int i=0; i < robj->Get_Num_Sub_Objects() && m_treadCount < MAX_TREADS_PER_TANK; i++)
256 {
257 RenderObjClass *subObj=robj->Get_Sub_Object(i);
258 const char *meshName;
259 //Check if subobject name starts with "TREADS".
260 if (subObj && subObj->Class_ID() == RenderObjClass::CLASSID_MESH && subObj->Get_Name()
261 && ( (meshName=strchr(subObj->Get_Name(),'.') ) != 0 && *(meshName++))
262 &&_strnicmp(meshName,"TREADS", 6) == 0)
263 { //check if sub-object has the correct material to do texture scrolling.
264 MaterialInfoClass *mat=subObj->Get_Material_Info();
265 if (mat)
266 { for (Int j=0; j<mat->Vertex_Material_Count(); j++)
267 {
268 VertexMaterialClass *vmaterial=mat->Peek_Vertex_Material(j);
269 LinearOffsetTextureMapperClass *mapper=(LinearOffsetTextureMapperClass *)vmaterial->Peek_Mapper();
270 if (mapper && mapper->Mapper_ID() == TextureMapperClass::MAPPER_ID_LINEAR_OFFSET)
271 { mapper->Set_UV_Offset_Delta(Vector2(0,0)); //disable automatic scrolling
272 subObj->Add_Ref(); //increase reference since we're storing the pointer
273 m_treads[m_treadCount].m_robj=subObj;
274 m_treads[m_treadCount].m_type = TREAD_MIDDLE; //default type
275 subObj->Set_User_Data(&m_treads[m_treadCount].m_materialSettings); //tell W3D about custom material settings
276 m_treads[m_treadCount].m_materialSettings.customUVOffset=Vector2(0,0);
277 switch (meshName[6]) //check next character after 'TREADS'
278 {
279 case 'L':
280 case 'l': m_treads[m_treadCount].m_type = TREAD_LEFT;
281 break;
282 case 'R':
283 case 'r': m_treads[m_treadCount].m_type = TREAD_RIGHT;
284 break;
285 }
286 m_treadCount++;
287 }
288 }
289 REF_PTR_RELEASE(mat);
290 }
291 }
292 REF_PTR_RELEASE(subObj);
293 }
294 }
295
296 m_prevRenderObj = robj;
297}

308void W3DTankDraw::doDrawModule(const Matrix3D* transformMtx)
309{
310 const Real DEBRIS_THRESHOLD = 0.00001f;
311
312 Bool frozen = TheTacticalView->isTimeFrozen() && !TheTacticalView->isCameraMovementFinished();
313 frozen = frozen || TheScriptEngine->isTimeFrozenDebug() || TheScriptEngine->isTimeFrozenScript();
314 if (frozen)
315 return;
316 if (getRenderObject()==NULL) return;
317 if (getRenderObject() != m_prevRenderObj) {
318 updateTreadObjects();
319 }
320
321 // get object from logic
322 Object *obj = getDrawable()->getObject();
323 if (obj == NULL)
324 return;
325
326 // get object physics state
327 PhysicsBehavior *physics = obj->getPhysics();
328 if (physics == NULL)
329 return;
330
331 const Coord3D *vel = physics->getVelocity();
332
333 // if tank is moving, kick up dust and debris
334 Real velMag = vel->x*vel->x + vel->y*vel->y; // only care about moving on the ground
335
336 if (velMag > DEBRIS_THRESHOLD && !getDrawable()->isDrawableEffectivelyHidden() && !getFullyObscuredByShroud())
337 startMoveDebris();
338 else
339 stopMoveDebris();
340
341 // kick debris higher the faster we move
342 Coord3D velMult;
343 velMag = (Real)sqrt( velMag );
344
345 velMult.x = 0.5f * velMag + 0.1f;
346 if (velMult.x > 1.0f)
347 velMult.x = 1.0f;
348
349 velMult.y = velMult.x;
350
351 velMult.z = velMag + 0.1f;
352 if (velMult.z > 1.0f)
353 velMult.z = 1.0f;
354
355 m_treadDebrisLeft->setVelocityMultiplier( &velMult );
356 m_treadDebrisRight->setVelocityMultiplier( &velMult );
357
358 m_treadDebrisLeft->setBurstCountMultiplier( velMult.z );
359 m_treadDebrisRight->setBurstCountMultiplier( velMult.z );
360
361 //Update movement of treads
362 if (m_treadCount)
363 {
364 PhysicsTurningType turn=physics->getTurning();
365 Real offset_u;
366 Real treadScrollSpeed=getW3DTankDrawModuleData()->m_treadAnimationRate;
367 TreadObjectInfo *pTread=m_treads;
368 Real maxSpeed=obj->getAIUpdateInterface()->getCurLocomotorSpeed();
369
370 //For optimization sake, we only do complex tread scrolling when tank
371 //is mostly stationary and turning
372 if (turn != TURN_NONE && physics->getVelocityMagnitude()/maxSpeed < getW3DTankDrawModuleData()->m_treadPivotSpeedFraction)
373 {
374 //Check if we have turned enough since last draw to require animation
375 Coord3D dir;
376 obj->getUnitDirectionVector2D(dir);
377 Real angleToGoal = dir.x * m_lastDirection.x + dir.y * m_lastDirection.y;
378
379 if (fabs(1.0f-angleToGoal) > 0.00001f) //check if difference in angle cosines is greater than some cutoff.
380 {
381 if (turn == TURN_NEGATIVE) //turning right
382 updateTreadPositions(-treadScrollSpeed);
383 else //turning left
384 updateTreadPositions(treadScrollSpeed);
385 }
386 m_lastDirection=dir; //update for next frame
387 }
388 else
389 if (physics->isMotive() && physics->getVelocityMagnitude()/maxSpeed >= getW3DTankDrawModuleData()->m_treadDriveSpeedFraction)
390 { //do simple scrolling based only on speed when tank is moving straight at high speed.
391 //we stop scrolling when tank slows down to reduce the appearance of sliding
392 //tread scrolling speed was not directly tied into tank velocity because it looked odd
393 //under certain situations when tank moved sideways.
394 for (Int i=0; i<m_treadCount; i++)
395 {
396 offset_u = pTread->m_materialSettings.customUVOffset.X - treadScrollSpeed;
397 // ensure coordinates of offset are in [0, 1] range:
398 offset_u = offset_u - WWMath::Floor(offset_u);
399 pTread->m_materialSettings.customUVOffset.Set(offset_u,0);
400 pTread++;
401 }
402 }
403 }
404
405 W3DModelDraw::doDrawModule(transformMtx);
406}
