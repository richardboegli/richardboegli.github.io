283//*****************************************************************************
284// DrawButton::Is_Mouse_In_Region -- Check if mouse values are in button area.
285//
286// INPUT: int mouse_x -- mouse x position
287// int mouse_y -- mouse y position
288//
289// OUTPUT: bool -- true of false.
290//
291// WARNINGS: No keyboard/mouse/paint handling built in. Do manually.
292// Note: width is shortened below to accomodate actual bitmap area.
293//
294// HISTORY:
295// 07/15/1996 MML : Created.
296//=============================================================================
297
298bool DrawButton::Is_Mouse_In_Region ( int mouse_x, int mouse_y )
299{
300 if ( mouse_x < 0 || mouse_y < 0 ) {
301 return( false );
302 }
303
304 if (( mouse_x >= rect.left ) &&
305 ( mouse_y >= rect.top ) &&
306 ( mouse_x <= rect.left + StretchedWidth ) &&
307 ( mouse_y <= rect.top + StretchedHeight )) {
308 return ( TRUE );
309 }
310 return ( FALSE );
311}
