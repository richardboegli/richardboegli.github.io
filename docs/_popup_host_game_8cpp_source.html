127void HandleCustomLadderSelection(Int ladderID)
128{
129 if (!parentPopup)
130 return;
131
132 CustomMatchPreferences pref;
133
134 if (ladderID == 0)
135 {
136 pref.setLastLadder(AsciiString::TheEmptyString, 0);
137 pref.write();
138 return;
139 }
140
141 const LadderInfo *info = TheLadderList->findLadderByIndex(ladderID);
142 if (!info)
143 {
144 pref.setLastLadder(AsciiString::TheEmptyString, 0);
145 }
146 else
147 {
148 pref.setLastLadder(info->address, info->port);
149 }
150
151 pref.write();
152}

154void PopulateCustomLadderListBox( GameWindow *win )
155{
156 if (!parentPopup || !win)
157 return;
158
159 isPopulatingLadderBox = true;
160
161 CustomMatchPreferences pref;
162
163 Color specialColor = GameSpyColor[GSCOLOR_MAP_SELECTED];
164 Color normalColor = GameSpyColor[GSCOLOR_MAP_UNSELECTED];
165 Color favoriteColor = GameSpyColor[GSCOLOR_MAP_UNSELECTED];
166 Color localColor = GameSpyColor[GSCOLOR_MAP_UNSELECTED];
167 Int index;
168 GadgetListBoxReset( win );
169
170 std::set<const LadderInfo *> usedLadders;
171
172 // start with "No Ladder"
173 index = GadgetListBoxAddEntryText( win, TheGameText->fetch("GUI:NoLadder"), normalColor, -1 );
174 GadgetListBoxSetItemData( win, 0, index );
175
176 // add the last ladder
177 Int selectedPos = 0;
178 AsciiString lastLadderAddr = pref.getLastLadderAddr();
179 UnsignedShort lastLadderPort = pref.getLastLadderPort();
180 const LadderInfo *info = TheLadderList->findLadder( lastLadderAddr, lastLadderPort );
181 if (info && info->index > 0 && info->validCustom)
182 {
183 usedLadders.insert(info);
184 index = GadgetListBoxAddEntryText( win, info->name, favoriteColor, -1 );
185 GadgetListBoxSetItemData( win, (void *)(info->index), index );
186 selectedPos = index;
187 }
188
189 // our recent ladders
190 LadderPreferences ladPref;
191 ladPref.loadProfile( TheGameSpyInfo->getLocalProfileID() );
192 const LadderPrefMap recentLadders = ladPref.getRecentLadders();
193 for (LadderPrefMap::const_iterator cit = recentLadders.begin(); cit != recentLadders.end(); ++cit)
194 {
195 AsciiString addr = cit->second.address;
196 UnsignedShort port = cit->second.port;
197 if (addr == lastLadderAddr && port == lastLadderPort)
198 continue;
199 const LadderInfo *info = TheLadderList->findLadder( addr, port );
200 if (info && info->index > 0 && info->validCustom && usedLadders.find(info) == usedLadders.end())
201 {
202 usedLadders.insert(info);
203 index = GadgetListBoxAddEntryText( win, info->name, favoriteColor, -1 );
204 GadgetListBoxSetItemData( win, (void *)(info->index), index );
205 }
206 }
207
208 // local ladders
209 const LadderInfoList *lil = TheLadderList->getLocalLadders();
210 LadderInfoList::const_iterator lit;
211 for (lit = lil->begin(); lit != lil->end(); ++lit)
212 {
213 const LadderInfo *info = *lit;
214 if (info && info->index < 0 && info->validCustom && usedLadders.find(info) == usedLadders.end())
215 {
216 usedLadders.insert(info);
217 index = GadgetListBoxAddEntryText( win, info->name, localColor, -1 );
218 GadgetListBoxSetItemData( win, (void *)(info->index), index );
219 }
220 }
221
222 // special ladders
223 lil = TheLadderList->getSpecialLadders();
224 for (lit = lil->begin(); lit != lil->end(); ++lit)
225 {
226 const LadderInfo *info = *lit;
227 if (info && info->index > 0 && info->validCustom && usedLadders.find(info) == usedLadders.end())
228 {
229 usedLadders.insert(info);
230 index = GadgetListBoxAddEntryText( win, info->name, specialColor, -1 );
231 GadgetListBoxSetItemData( win, (void *)(info->index), index );
232 }
233 }
234
235 // standard ladders
236 lil = TheLadderList->getStandardLadders();
237 for (lit = lil->begin(); lit != lil->end(); ++lit)
238 {
239 const LadderInfo *info = *lit;
240 if (info && info->index > 0 && info->validCustom && usedLadders.find(info) == usedLadders.end())
241 {
242 usedLadders.insert(info);
243 index = GadgetListBoxAddEntryText( win, info->name, normalColor, -1 );
244 GadgetListBoxSetItemData( win, (void *)(info->index), index );
245 }
246 }
247
248 GadgetListBoxSetSelected( win, selectedPos );
249 isPopulatingLadderBox = false;
250}

427 } // end escape

429 } // end switch( key )

431 } // end char

433 } // end switch( msg )

442WindowMsgHandledType PopupHostGameSystem( GameWindow *window, UnsignedInt msg, WindowMsgData mData1, WindowMsgData mData2 )
443{
444 switch( msg )
445 {
446
447 // --------------------------------------------------------------------------------------------
448 case GWM_CREATE:
449 {
450
451 break;
452
453 } // end create
454 //---------------------------------------------------------------------------------------------
455 case GWM_DESTROY:
456 {
457 parentPopup = NULL;
458
459 break;
460
461 } // end case
462
463 //----------------------------------------------------------------------------------------------
464 case GWM_INPUT_FOCUS:
465 {
466
467 // if we're givin the opportunity to take the keyboard focus we must say we want it
468 if( mData1 == TRUE )
469 *(Bool *)mData2 = TRUE;
470
471 break;
472
473 } // end input
474
475 //----------------------------------------------------------------------------------------------
476 case GEM_UPDATE_TEXT:
477 {
478 GameWindow *control = (GameWindow *)mData1;
479 Int controlID = control->winGetWindowId();
480
481 if ( controlID == textEntryGameNameID )
482 {
483 UnicodeString txtInput;
484
485 // grab the game's name
486 txtInput.set(GadgetTextEntryGetText( textEntryGameName ));
487
488 // Clean up the text (remove leading/trailing chars, etc)
489 const WideChar *c = txtInput.str();
490 while (c && (iswspace(*c)))
491 c++;
492
493 if (c)
494 txtInput = UnicodeString(c);
495 else
496 txtInput = UnicodeString::TheEmptyString;
497
498 // Put the whitespace-free version in the box
499 GadgetTextEntrySetText( textEntryGameName, txtInput );
500
501 }// if ( controlID == textEntryPlayerNameID )
502 break;
503 }//case GEM_UPDATE_TEXT:
504 //---------------------------------------------------------------------------------------------
505 case GCM_SELECTED:
506 {
507 GameWindow *control = (GameWindow *)mData1;
508 Int controlID = control->winGetWindowId();
509 Int pos = -1;
510 GadgetComboBoxGetSelectedPos(control, &pos);
511
512 if (controlID == comboBoxLadderNameID && !isPopulatingLadderBox)
513 {
514 if (pos >= 0)
515 {
516 Int ladderID = (Int)GadgetComboBoxGetItemData(control, pos);
517 if (ladderID < 0)
518 {
519 // "Choose a ladder" selected - open overlay
520 PopulateCustomLadderComboBox(); // this restores the non-"Choose a ladder" selection
521 GameSpyOpenOverlay( GSOVERLAY_LADDERSELECT );
522 }
523 }
524 }
525 break;
526 } // case GCM_SELECTED
527
528 //---------------------------------------------------------------------------------------------
529 case GBM_SELECTED:
530 {
531 GameWindow *control = (GameWindow *)mData1;
532 Int controlID = control->winGetWindowId();
533
534 if( controlID == buttonCancelID )
535 {
536 parentPopup = NULL;
537 GameSpyCloseOverlay(GSOVERLAY_GAMEOPTIONS);
538 SetLobbyAttemptHostJoin( FALSE );
539 }
540 else if( controlID == buttonCreateGameID)
541 {
542 UnicodeString name;
543 name = GadgetTextEntryGetText(textEntryGameName);
544 name.trim();
545 if(name.getLength() <= 0)
546 {
547 name.translate(TheGameSpyInfo->getLocalName());
548 GadgetTextEntrySetText(textEntryGameName, name);
549 }
550 createGame();
551 parentPopup = NULL;
552 GameSpyCloseOverlay(GSOVERLAY_GAMEOPTIONS);
553 }
554 break;
555 }
556 default:
557 return MSG_IGNORED;
558
559 } // end switch
560
561 return MSG_HANDLED;
562

570void createGame( void )
571{
572 TheGameSpyInfo->setCurrentGroupRoom(0);
573 PeerRequest req;
574 UnicodeString gameName = GadgetTextEntryGetText(textEntryGameName);
575 req.peerRequestType = PeerRequest::PEERREQUEST_CREATESTAGINGROOM;
576 req.text = gameName.str();
577 TheGameSpyGame->setGameName(gameName);
578 AsciiString passwd;
579 passwd.translate(GadgetTextEntryGetText(textEntryGamePassword));
580 req.password = passwd.str();
581 CustomMatchPreferences customPref;
582 Bool aO = GadgetCheckBoxIsChecked(checkBoxAllowObservers);
583 Bool limitArmies = GadgetCheckBoxIsChecked( checkBoxLimitArmies );
584 Bool useStats = GadgetCheckBoxIsChecked( checkBoxUseStats );
585 customPref.setAllowsObserver(aO);
586 customPref.setFactionsLimited( limitArmies );
587 customPref.setUseStats( useStats );
588 customPref.write();
589 req.stagingRoomCreation.allowObservers = aO;
590 req.stagingRoomCreation.useStats = useStats;
591 TheGameSpyGame->setAllowObservers(aO);
592 TheGameSpyGame->setOldFactionsOnly( limitArmies );
593 TheGameSpyGame->setUseStats( useStats );
594 req.stagingRoomCreation.exeCRC = TheGlobalData->m_exeCRC;
595 req.stagingRoomCreation.iniCRC = TheGlobalData->m_iniCRC;
596 req.stagingRoomCreation.gameVersion = TheGameSpyInfo->getInternalIP();
597 req.stagingRoomCreation.restrictGameList = TheGameSpyConfig->restrictGamesToLobby();
598
599 Int ladderSelectPos = -1, ladderID = -1;
600 GadgetComboBoxGetSelectedPos(comboBoxLadderName, &ladderSelectPos);
601 req.ladderIP = "localhost";
602 req.stagingRoomCreation.ladPort = 0;
603 if (ladderSelectPos >= 0)
604 {
605 ladderID = (Int)GadgetComboBoxGetItemData(comboBoxLadderName, ladderSelectPos);
606 if (ladderID != 0)
607 {
608 // actual ladder
609 const LadderInfo *info = TheLadderList->findLadderByIndex(ladderID);
610 if (info)
611 {
612 req.ladderIP = info->address.str();
613 req.stagingRoomCreation.ladPort = info->port;
614 }
615 }
616 }
617 TheGameSpyGame->setLadderIP(req.ladderIP.c_str());
618 TheGameSpyGame->setLadderPort(req.stagingRoomCreation.ladPort);
619 req.hostPingStr = TheGameSpyInfo->getPingString().str();
620
621 TheGameSpyPeerMessageQueue->addRequest(req);
622}
