166void BoneFXUpdateModuleData::parseFXList( INI *ini, void *instance,
167 void *store, const void *userData )
168{
169 const char *token;
170 BoneFXListInfo *info = (BoneFXListInfo *)store;
171
172 // parse the location bone or location
173 parseFXLocInfo( ini, instance, &info->locInfo );
174
175 // make sure we have an "OnlyOnce:" token
176 token = ini->getNextToken( ini->getSepsColon() );
177 if (stricmp( token, "onlyonce" ) != 0)
178 {
179
180 // error
181 throw INI_INVALID_DATA;
182
183 } // end if
184
185 ini->parseBool( ini, instance, &info->onlyOnce, NULL);
186
187 parseGameLogicRandomDelay( ini, instance, &info->gameLogicDelay);
188
189 // make sure we have an "FXList:" token
190 token = ini->getNextToken( ini->getSepsColon() );
191 if( stricmp( token, "fxlist" ) != 0 )
192 {
193
194 // error
195 throw INI_INVALID_DATA;
196
197 } // end if
198
199 // parse the fx list name
200 ini->parseFXList( ini, instance, &info->fx, NULL );
201
202} // end parseFXList

290UpdateSleepTime BoneFXUpdate::update( void )
291{
293 const BoneFXUpdateModuleData *d = getBoneFXUpdateModuleData();
294 Int now = TheGameLogic->getFrame();
295
296 if (m_active == FALSE) {
297 initTimes();
298 m_active = TRUE;
299 }
300
301 for (Int i = 0; i < BONE_FX_MAX_BONES; ++i) {
302 //Check to see if its time to fire off any cool stuff.
303 if ((m_nextFXFrame[m_curBodyState][i] != -1) && (m_nextFXFrame[m_curBodyState][i] <= now)) {
304 doFXListAtBone(d->m_fxList[m_curBodyState][i].fx, &(m_FXBonePositions[m_curBodyState][i]));
305 computeNextLogicFXTime(&(d->m_fxList[m_curBodyState][i]), m_nextFXFrame[m_curBodyState][i]);
306 }
307 if ((m_nextOCLFrame[m_curBodyState][i] != -1) && (m_nextOCLFrame[m_curBodyState][i] <= now)) {
308 doOCLAtBone(d->m_OCL[m_curBodyState][i].ocl, &(m_OCLBonePositions[m_curBodyState][i]));
309 computeNextLogicFXTime(&(d->m_OCL[m_curBodyState][i]), m_nextOCLFrame[m_curBodyState][i]);
310 }
311 if ((m_nextParticleSystemFrame[m_curBodyState][i] != -1) && (m_nextParticleSystemFrame[m_curBodyState][i] <= now)) {
312 doParticleSystemAtBone(d->m_particleSystem[m_curBodyState][i].particleSysTemplate, &(m_PSBonePositions[m_curBodyState][i]));
313 computeNextClientFXTime(&(d->m_particleSystem[m_curBodyState][i]), m_nextParticleSystemFrame[m_curBodyState][i]);
314 }
315 }
316 return UPDATE_SLEEP_NONE;
317}

321void BoneFXUpdate::initTimes() {
322 Int i;
323 const BoneFXUpdateModuleData *d = getBoneFXUpdateModuleData();
324 Int now = TheGameLogic->getFrame();
325
326 for (i = 0; i < BONE_FX_MAX_BONES; ++i) {
327 if (d->m_fxList[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0) {
328 m_nextFXFrame[m_curBodyState][i] = now + REAL_TO_INT(d->m_fxList[m_curBodyState][i].gameLogicDelay.getValue());
329 } else {
330 m_nextFXFrame[m_curBodyState][i] = -1;
331 }
332 if (d->m_OCL[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0) {
333 m_nextOCLFrame[m_curBodyState][i] = now + REAL_TO_INT(d->m_OCL[m_curBodyState][i].gameLogicDelay.getValue());
334 } else {
335 m_nextOCLFrame[m_curBodyState][i] = -1;
336 }
337 if (d->m_particleSystem[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0) {
338 m_nextParticleSystemFrame[m_curBodyState][i] = now + REAL_TO_INT(d->m_particleSystem[m_curBodyState][i].gameClientDelay.getValue());
339 } else {
340 m_nextParticleSystemFrame[m_curBodyState][i] = -1;
341 }
342 }
343}

384void BoneFXUpdate::doFXListAtBone(const FXList *fxList, const Coord3D *bonePosition)
385{
386 if (m_bonesResolved[m_curBodyState] == FALSE) {
387 resolveBoneLocations();
388 }
389
390 // if we are restricted by the damage type executing effect, bail out of here
391 const BoneFXUpdateModuleData *d = getBoneFXUpdateModuleData();
392 const DamageInfo *lastDamageInfo = getObject()->getBodyModule()->getLastDamageInfo();
393 if( lastDamageInfo && getDamageTypeFlag( d->m_damageFXTypes, lastDamageInfo->in.m_damageType ) == FALSE )
394 return;
395
396 // the bonePosition variable will have been made right by the call to
397 // resolveBoneLocations. Either that or it was correct to begin with.
398 Object *building = getObject();
399
400 // Convert the bone's position relative to the origin of the building to the current
401 // bone position in the world.
402 Coord3D newPos;
403 building->convertBonePosToWorldPos(bonePosition, NULL, &newPos, NULL);
404
405 // execute the fx list at the calculated bone position.
406 FXList::doFXPos(fxList, &newPos, NULL);
407}

501// This function is going to suck lots of time, should only be called once.
502void BoneFXUpdate::resolveBoneLocations() {
503 Int i;
504 const BoneFXUpdateModuleData *d = getBoneFXUpdateModuleData();
505 Object *building = getObject();
506 if (building == NULL) {
507 DEBUG_ASSERTCRASH(building != NULL, ("There is no object?"));
508 return;
509 }
510
511 Drawable *drawable = building->getDrawable();
512 if (drawable == NULL) {
513 DEBUG_ASSERTCRASH(drawable != NULL, ("There is no drawable?"));
514 }
515
516 if (d == NULL) {
517 return;
518 }
519
520 for (i = 0; i < BONE_FX_MAX_BONES; ++i) {
521 if (d->m_fxList[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0)
522 {
523 const BoneFXListInfo *info = &(d->m_fxList[m_curBodyState][i]);
524 drawable->getPristineBonePositions(info->locInfo.boneName.str(), 0, &m_FXBonePositions[m_curBodyState][i], NULL, 1);
525 }
526
527 if (d->m_OCL[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0)
528 {
529 const BoneOCLInfo *info = &(d->m_OCL[m_curBodyState][i]);
530 drawable->getPristineBonePositions(info->locInfo.boneName.str(), 0, &m_OCLBonePositions[m_curBodyState][i], NULL, 1);
531 }
532
533 if (d->m_particleSystem[m_curBodyState][i].locInfo.boneName.compare(AsciiString::TheEmptyString) != 0)
534 {
535 const BoneParticleSystemInfo *info = &(d->m_particleSystem[m_curBodyState][i]);
536 drawable->getPristineBonePositions(info->locInfo.boneName.str(), 0, &m_PSBonePositions[m_curBodyState][i], NULL, 1);
537 }
538 }
539 m_bonesResolved[m_curBodyState] = TRUE;
540}

572void BoneFXUpdate::xfer( Xfer *xfer )
573{
574
575 // version
576 XferVersion currentVersion = 1;
577 XferVersion version = currentVersion;
578 xfer->xferVersion( &version, currentVersion );
579
580 // extend base class
581 UpdateModule::xfer( xfer );
582
583 // particle system vector count and data
584 UnsignedShort particleSystemCount = m_particleSystemIDs.size();
585 xfer->xferUnsignedShort( &particleSystemCount );
586 ParticleSystemID systemID;
587 if( xfer->getXferMode() == XFER_SAVE )
588 {
589 std::vector<ParticleSystemID>::const_iterator it;
590
591 for( it = m_particleSystemIDs.begin(); it != m_particleSystemIDs.end(); ++it )
592 {
593
594 systemID = *it;
595 xfer->xferUser( &systemID, sizeof( ParticleSystemID ) );
596
597 } // end for
598
599 } // end if, save
600 else
601 {
602
603 // the list should be emtpy right now
604 if( m_particleSystemIDs.empty() == FALSE )
605 {
606
607 DEBUG_CRASH(( "BoneFXUpdate::xfer - m_particleSystemIDs should be empty but is not\n" ));
608 throw SC_INVALID_DATA;
609
610 } // end if
611
612 // read all data
613 for( UnsignedShort i = 0; i < particleSystemCount; ++i )
614 {
615
616 // read id
617 xfer->xferUser( &systemID, sizeof( ParticleSystemID ) );
618
619 // put at end of vector
620 m_particleSystemIDs.push_back( systemID );
621
622 } // end for, i
623
624 } // end else
625
626 // next fx frame
627 xfer->xferUser( m_nextFXFrame, sizeof( Int ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
628
629 // next OCL farme
630 xfer->xferUser( m_nextOCLFrame, sizeof( Int ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
631
632 // next particle system frame
633 xfer->xferUser( m_nextParticleSystemFrame, sizeof( Int ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
634
635 // fx bone positions
636 xfer->xferUser( m_FXBonePositions, sizeof( Coord3D ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
637
638 // ocl bone positions
639 xfer->xferUser( m_OCLBonePositions, sizeof( Coord3D ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
640
641 // particle system bone positions
642 xfer->xferUser( m_PSBonePositions, sizeof( Coord3D ) * BODYDAMAGETYPE_COUNT * BONE_FX_MAX_BONES );
643
644 // current body state
645 xfer->xferUser( &m_curBodyState, sizeof( BodyDamageType ) );
646
647 // bones resolved
648 xfer->xferUser( m_bonesResolved, sizeof( Bool ) * BODYDAMAGETYPE_COUNT );
649
650 // active
651 xfer->xferBool( &m_active );
652
653} // end xfer
