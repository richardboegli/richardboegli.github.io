141void SpecialPowerStore::parseSpecialPowerDefinition( INI *ini )
142{
143 // read the name
144 AsciiString name = ini->getNextToken();
145
146 SpecialPowerTemplate* specialPower = TheSpecialPowerStore->findSpecialPowerTemplatePrivate( name );
147
148 if ( ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES )
149 {
150 if (specialPower)
151 {
152 SpecialPowerTemplate* child = (SpecialPowerTemplate*)specialPower->friend_getFinalOverride();
153 specialPower = newInstance(SpecialPowerTemplate);
154 *specialPower = *child;
155 child->setNextOverride(specialPower);
156 specialPower->markAsOverride();
157 //TheSpecialPowerStore->m_specialPowerTemplates.push_back(specialPower); // nope, do NOT do this
158 }
159 else
160 {
161 specialPower = newInstance(SpecialPowerTemplate);
162 const SpecialPowerTemplate *defaultTemplate = TheSpecialPowerStore->findSpecialPowerTemplate( "DefaultSpecialPower" );
163 if( defaultTemplate )
164 *specialPower = *defaultTemplate;
165 specialPower->friend_setNameAndID(name, ++TheSpecialPowerStore->m_nextSpecialPowerID);
166 specialPower->markAsOverride();
167 TheSpecialPowerStore->m_specialPowerTemplates.push_back(specialPower);
168 }
169 }
170 else
171 {
172 if (specialPower)
173 {
174 throw INI_INVALID_DATA;
175 }
176 else
177 {
178 specialPower = newInstance(SpecialPowerTemplate);
179 const SpecialPowerTemplate *defaultTemplate = TheSpecialPowerStore->findSpecialPowerTemplate( "DefaultSpecialPower" );
180 if( defaultTemplate )
181 *specialPower = *defaultTemplate;
182 specialPower->friend_setNameAndID(name, ++TheSpecialPowerStore->m_nextSpecialPowerID);
183 TheSpecialPowerStore->m_specialPowerTemplates.push_back(specialPower);
184 }
185 }
186
187 // parse the ini definition
188 if (specialPower)
189 ini->initFromINI( specialPower, specialPower->getFieldParse() );
190}

324Bool SpecialPowerStore::canUseSpecialPower( Object *obj, const SpecialPowerTemplate *specialPowerTemplate )
325{
326
327 // sanity
328 if( obj == NULL || specialPowerTemplate == NULL )
329 return FALSE;
330
331 // as a first sanity check, the object must have a module capable of executing the power
332 if( obj->getSpecialPowerModule( specialPowerTemplate ) == NULL )
333 return FALSE;
334
335 //
336 // in order to execute the special powers we have attached special power modules to the objects
337 // that can use them. However, just because an object has a module that is capable of
338 // doing the power, does not mean the object and the player can actually execute the
339 // power because some powers require a specialized science that the player must select and
340 // they cannot have all of them.
341 //
342
343 // check for requried science
344 ScienceType requiredScience = specialPowerTemplate->getRequiredScience();
345 if( requiredScience != SCIENCE_INVALID )
346 {
347 Player *player = obj->getControllingPlayer();
348
349 if( player->hasScience( requiredScience ) == FALSE )
350 return FALSE;
351
352 } // end if
353
354
355 // I THINK THIS IS WHERE WE BAIL OUT IF A DIFFERENT CONYARD IS ALREADY CHARGIN THIS SPECIAL RIGHT NOW //LORENZEN
356
357
358 // all is well
359 return TRUE;
360
361} // end canUseSpecialPower
