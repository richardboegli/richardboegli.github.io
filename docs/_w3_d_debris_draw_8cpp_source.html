103void W3DDebrisDraw::setModelName(AsciiString name, Color color, ShadowType t)
104{
105 if (m_renderObject == NULL && !name.isEmpty())
106 {
107 Int hexColor = 0;
108 if (color != 0)
109 hexColor = color | 0xFF000000;
110 m_renderObject = W3DDisplay::m_assetManager->Create_Render_Obj(name.str(), getDrawable()->getScale(), hexColor);
111 DEBUG_ASSERTCRASH(m_renderObject, ("Debris model %s not found!\n",name.str()));
112 if (m_renderObject)
113 {
114 W3DDisplay::m_3DScene->Add_Render_Object(m_renderObject);
115
116 m_renderObject->Set_User_Data(getDrawable()->getDrawableInfo());
117
118 Matrix3D transform;
121// transform.Set(Vector3(0,0,9999));
122 transform.Set(Vector3(0,0,0));
123 m_renderObject->Set_Transform(transform);
124 }
125
126 if (t != SHADOW_NONE)
127 {
128 Shadow::ShadowTypeInfo shadowInfo;
129 shadowInfo.m_type = t;
130 shadowInfo.m_sizeX=0;
131 shadowInfo.m_sizeY=0;
132 m_shadow = TheW3DShadowManager->addShadow(m_renderObject, &shadowInfo);
133 }
134 else
135 {
136 if (TheW3DShadowManager && m_shadow)
137 {
138 TheW3DShadowManager->removeShadow(m_shadow);
139 m_shadow = NULL;
140 }
141 }
142
143 // save the model name and color
144 m_modelName = name;
145 m_modelColor = color;
146
147 }
148}

151void W3DDebrisDraw::setAnimNames(AsciiString initial, AsciiString flying, AsciiString final, const FXList* finalFX)
152{
153 int i;
154 for (i = 0; i < STATECOUNT; ++i)
155 {
156 REF_PTR_RELEASE(m_anims[i]);
157 m_anims[i] = NULL;
158 }
159
160 m_anims[INITIAL] = initial.isEmpty() ? NULL : W3DDisplay::m_assetManager->Get_HAnim(initial.str());
161 m_anims[FLYING] = flying.isEmpty() ? NULL : W3DDisplay::m_assetManager->Get_HAnim(flying.str());
162 if (stricmp(final.str(), "STOP") == 0)
163 {
164 m_finalStop = true;
165 final = flying;
166 }
167 else
168 {
169 m_finalStop = false;
170 }
171 m_anims[FINAL] = final.isEmpty() ? NULL : W3DDisplay::m_assetManager->Get_HAnim(final.str());
172 m_state = 0;
173 m_frames = 0;
174 m_fxFinal = finalFX;
175
176 m_animInitial = initial;
177 m_animFlying = flying;
178 m_animFinal = final;
179
180}

213void W3DDebrisDraw::doDrawModule(const Matrix3D* transformMtx)
214{
215 if (m_renderObject)
216 {
217
218 Matrix3D scaledTransform;
219 if (getDrawable()->getInstanceScale() != 1.0f)
220 { //do custom scaling of the W3D model.
221 scaledTransform=*transformMtx;
222 scaledTransform.Scale(getDrawable()->getInstanceScale());
223 transformMtx = &scaledTransform;
224 m_renderObject->Set_ObjectScale(getDrawable()->getInstanceScale());
225 }
226 m_renderObject->Set_Transform(*transformMtx);
227
228 static const RenderObjClass::AnimMode TheAnimModes[STATECOUNT] =
229 {
230 RenderObjClass::ANIM_MODE_ONCE,
231 RenderObjClass::ANIM_MODE_LOOP,
232 RenderObjClass::ANIM_MODE_ONCE
233 };
234
235 Int oldState = m_state;
236 Object* obj = getDrawable()->getObject();
237 const Int MIN_FINAL_FRAMES = 3;
238 if (m_state != FINAL && obj != NULL && !obj->isAboveTerrain() && m_frames > MIN_FINAL_FRAMES)
239 {
240 m_state = FINAL;
241 }
242 else if (m_state < FINAL && (isAnimationComplete(m_renderObject)))
243 {
244 ++m_state;
245 }
246 HAnimClass* hanim = m_anims[m_state];
247 if (hanim != NULL && (hanim != m_renderObject->Peek_Animation() || oldState != m_state))
248 {
249 RenderObjClass::AnimMode m = TheAnimModes[m_state];
250 if (m_state == FINAL)
251 {
252 FXList::doFXPos(m_fxFinal, getDrawable()->getPosition(), getDrawable()->getTransformMatrix(), 0, NULL, 0.0f);
253 if (m_finalStop)
254 m = RenderObjClass::ANIM_MODE_MANUAL;
255 }
256 m_renderObject->Set_Animation(hanim, 0, m);
257 }
258 ++m_frames;
259 }
260}

278void W3DDebrisDraw::xfer( Xfer *xfer )
279{
280
281 // version
282 XferVersion currentVersion = 1;
283 XferVersion version = currentVersion;
284 xfer->xferVersion( &version, currentVersion );
285
286 // extend base class
287 DrawModule::xfer( xfer );
288
289 // model name
290 xfer->xferAsciiString( &m_modelName );
291
292 // model color
293 xfer->xferColor( &m_modelColor );
294
295 // set the model and color
296 if( xfer->getXferMode() == XFER_LOAD )
297 setModelName( m_modelName, m_modelColor, SHADOW_NONE );
298
299 // animation initial
300 xfer->xferAsciiString( &m_animInitial );
301
302 // anim flying
303 xfer->xferAsciiString( &m_animFlying );
304
305 // anim final
306 xfer->xferAsciiString( &m_animFinal );
307
308 // when loading, set the animations
309 if( xfer->getXferMode() == XFER_LOAD )
310 setAnimNames( m_animInitial, m_animFlying, m_animFinal, NULL );
311
312 // state
313 xfer->xferInt( &m_state );
314
315 // frames
316 xfer->xferInt( &m_frames );
317
318 // final stop
319 xfer->xferBool( &m_finalStop );
320
321} // end xfer
