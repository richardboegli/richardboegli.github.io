180// Get_Callback
182template <class T> T
183AudioCallbackListClass<T>::Get_Callback (int index, uint32 *user_data)
184{
185 if (user_data != NULL) {

190}

193// Remove_Callback
195template <class T> void
196AudioCallbackListClass<T>::Remove_Callback (T pointer)
197{
198 for (int index = 0; index < ActiveCount; index ++) {
199 if (Vector[index].callback_ptr == pointer) {
200 Delete (index);
201 break;
202 }
203 }
204
205 return ;
