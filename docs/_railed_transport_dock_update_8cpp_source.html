125Bool RailedTransportDockUpdate::action( Object *docker, Object *drone )
126{
127 Object *us = getObject();
128
129 // sanity
130 if( docker == NULL )
131 return FALSE;
132
133 // set this object as docking with us if not already done so
134 if( m_dockingObjectID != docker->getID() )
135 {
136 //
137 // given the amount of time we want it to take to "pull" the object inside, figure out
138 // how much distance it should traverse towards our center every frame
139 //
140 const Coord3D *dockerPos = docker->getPosition();
141 const Coord3D *dockPos = us->getPosition();
142 Coord3D v;
143 v.x = dockPos->x - dockerPos->x;
144 v.y = dockPos->y - dockerPos->y;
145 v.z = dockPos->z - dockerPos->z;
146
147 // how far do we have to go
148 Real mag = v.length();
149 const RailedTransportDockUpdateModuleData *modData = getRailedTransportDockUpdateModuleData();
150
151 //Are we close enough to even be able to get sucked in?
152 if( mag <= modData->m_toleranceDistance )
153 {
154 m_dockingObjectID = docker->getID();
155
156 // don't let the user interact with this object anymore
157 TheGameLogic->deselectObject(docker, PLAYERMASK_ALL, TRUE);
158 docker->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNSELECTABLE ) );
159
160 // hold the object so physics doesn't mess with it anymore
161 docker->setDisabled( DISABLED_HELD );
162
163
164 // now that we know how far we must go, now much distance should we travel every frame
165 m_pullInsideDistancePerFrame = mag / modData->m_pullInsideDurationInFrames;
166
167 // orient docker so its facing toward the transport
168 Coord2D angleVector;
169 angleVector.x = dockPos->x - dockerPos->x;
170 angleVector.y = dockPos->y - dockerPos->y;
171 docker->setOrientation( angleVector.toAngle() );
172 }
173
174 } // end if
175
176 return TRUE;
177
178} // end action

250void RailedTransportDockUpdate::doPullInDocking( void )
251{
252
253 //
254 // if we're pulling an object inside of us, do that pull now. we need this so that the
255 // railed transport can "pull" objects inside it because typically those objects can only drive
256 // on land and have a hard time driving "inside" the railed transport ... so we fake it!
257 //
258 if( m_dockingObjectID != INVALID_ID )
259 {
260 Object *us = getObject();
261 Object *docker = TheGameLogic->findObjectByID( m_dockingObjectID );
262
263 // check for docker gone
264 if( docker == NULL )
265 m_dockingObjectID = INVALID_ID;
266
267 // pull it
268 if( docker )
269 {
270 const Coord3D *dockerPos = docker->getPosition();
271 const Coord3D *dockPos = us->getPosition();
272
273 // get the vector from the docker to the dock pos
274 Coord3D v;
275 v.x = dockPos->x - dockerPos->x;
276 v.y = dockPos->y - dockerPos->y;
277 v.z = dockPos->z - dockerPos->z;
278 v.normalize();
279
280 // apply "movement" to the vector
281 v.x *= m_pullInsideDistancePerFrame;
282 v.y *= m_pullInsideDistancePerFrame;
283
284 // apply current position of the docker to the vector
285 v.x += dockerPos->x;
286 v.y += dockerPos->y;
287 v.z = dockerPos->z; // keep Z height the same and just scoot along the ground
288
289 // set the new position
290 docker->setPosition( &v );
291
292 //
293 // set the model condition for the object as "moving" even though it really
294 // isn't in the traditional sense, but we don't want them to scoot slide into
295 // the transport and look wierd
296 //
297 docker->setModelConditionState( MODELCONDITION_MOVING );
298
299 // if we're at the destination then stop and put is inside the dock object
300 Real distSq = ThePartitionManager->getDistanceSquared( docker, us, FROM_CENTER_2D );
301 Real closeEnoughDistance = 6.0f;
302 if( distSq <= (closeEnoughDistance * closeEnoughDistance) )
303 {
304
305 // the object is now no longer "moving"
306 docker->clearModelConditionState( MODELCONDITION_MOVING );
307
308 // stop the dock action
309 cancelDock( docker );
310
311 // stop the docker from doing anything by going idle
312 AIUpdateInterface *dockerAI = docker->getAIUpdateInterface();
313 if( dockerAI )
314 dockerAI->aiIdle( CMD_FROM_AI );
315
316 // put object inside us
317 ContainModuleInterface *contain = us->getContain();
318 if( contain )
319 {
320 contain->addToContain( docker );
321 }
322
323 // no object is docking now
324 m_dockingObjectID = INVALID_ID;
325
326 } // end if
327
328 } // end if
329
330 } // end if
331
332} // end doPullInDocking

337void RailedTransportDockUpdate::doPushOutDocking( void )
338{
339
340 if( m_unloadingObjectID )
341 {
342 Object *unloader = TheGameLogic->findObjectByID( m_unloadingObjectID );
343
344 // if unloader is not found (like they got destroyed) unload the next object inside
345 if( unloader == NULL )
346 {
347
348 unloadNext();
349 return;
350
351 } // end if
352
353 // pull it
354 if( unloader )
355 {
356 const Coord3D *unloaderPos = unloader->getPosition();
357
358 // get the destination point as the DOCKEND
359 Coord3D destPos;
360 getExitPosition( unloader, &destPos );
361 destPos.z = TheTerrainLogic->getGroundHeight( destPos.x, destPos.y );
362
363 // get the vector from the unloader to the destination point
364 Coord3D v;
365 v.x = destPos.x - unloaderPos->x;
366 v.y = destPos.y - unloaderPos->y;
367 v.z = destPos.z - unloaderPos->z;
368 v.normalize();
369
370 // apply "movement" to that vector
371 v.x *= m_pushOutsideDistancePerFrame;
372 v.y *= m_pushOutsideDistancePerFrame;
373
374 // apply current position of the unloader to the vector
375 v.x += unloaderPos->x;
376 v.y += unloaderPos->y;
377 v.z = destPos.z; // keep Z height the same and just scoot along the ground
378
379 // set the new position
380 unloader->setPosition( &v );
381
382 //
383 // set the model condition for the object as "moving" even though it really
384 // isn't in the traditional sense, but we don't want them to scoot slide into
385 // the transport and look wierd
386 //
387 unloader->setModelConditionState( MODELCONDITION_MOVING );
388
389 // if we're at the destination then stop and unload the next object if present
390 Real distSq = sqr( destPos.x - v.x ) + sqr( destPos.y - v.y ) + sqr( destPos.z - v.z );
391 Real closeEnoughDistance = 3.0f;
392 if( distSq <= sqr( closeEnoughDistance ) )
393 {
394 Object *us = getObject();
395
396 // the object is now no longer "moving"
397 unloader->clearModelConditionState( MODELCONDITION_MOVING );
398
399 // set the unloaded object as idle
400 AIUpdateInterface *unloaderAI = unloader->getAIUpdateInterface();
401 if( unloaderAI )
402 unloaderAI->aiIdle( CMD_FROM_AI );
403
404 // clear the held status from this unloading object
405 unloader->clearDisabled( DISABLED_HELD );
406
407 // we can now be selected by the player again
408 unloader->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_UNSELECTABLE ) );
409
410 // tell the unloader to move to one of the dock positions and out of the way
411 Drawable *draw = us->getDrawable();
412 if( unloaderAI && draw )
413 {
414 Coord3D finalPos;
415
416 draw->getPristineBonePositions( "DOCKWAITING07", 0, &finalPos, NULL, 1 );
417 us->convertBonePosToWorldPos( &finalPos, NULL, &finalPos, NULL );
418 unloaderAI->aiMoveToPosition( &finalPos, CMD_FROM_AI );
419
420 } // end if
421
422 // unload the next object
423 unloadNext();
424
425 } // end if
426
427 } // end if
428
429 } // end if, m_unloadingID
430
431} // end doPushOutDocking

452void RailedTransportDockUpdate::unloadNext( void )
453{
454 Object *us = getObject();
455
456 // by default, setup our unloading process to be done with no objects being considered
457 m_unloadingObjectID = INVALID_ID;
458
459 //
460 // if our unload count is zero we can't unload any more until we receive a command to
461 // unload another one or everything we've got
462 //
463 if( m_unloadCount == 0 )
464 return;
465
466 // better be an open container
467 ContainModuleInterface *contain = us->getContain();
468 OpenContain *openContain = contain ? contain->asOpenContain() : NULL;
469 DEBUG_ASSERTCRASH( openContain, ("Unloading next from railed transport, but '%s' has no open container\n",
470 us->getTemplate()->getName().str()) );
471
472 // get the first contained object
473 Object *unloader = NULL;
474 openContain->iterateContained( getFirstContain, &unloader, FALSE );
475 if( unloader )
476 {
477
478 // remove us from the container
479 openContain->removeFromContain( unloader );
480
481 // set position of the loader to our position
482 unloader->setPosition( us->getPosition() );
483
484 // orient unloader to the same angle as us so we can drive out the front
485 unloader->setOrientation( us->getOrientation() );
486
487 // mark us as HELD so physics or anything else can't mess with our position
488 unloader->setDisabled( DISABLED_HELD );
489
490 //
491 // get the dock point that we're going to go to ... that is where we came in
492 // at the DOCKEND point
493 //
494 Coord3D dockPosition;
495 getExitPosition( unloader, &dockPosition );
496
497 // get unloader position
498 const Coord3D *unloaderPos = unloader->getPosition();
499
500 // how far is it from our current position to the dock position
501 Coord3D v;
502 v.x = dockPosition.x - unloaderPos->x;
503 v.y = dockPosition.y - unloaderPos->y;
504 v.z = dockPosition.z - unloaderPos->z;
505 Real mag = v.length();
506
507 // now that we know how far we must go, now much distance should we travel every frame
508 const RailedTransportDockUpdateModuleData *modData = getRailedTransportDockUpdateModuleData();
509 m_pushOutsideDistancePerFrame = mag / modData->m_pushOutsideDurationInFrames;
510
511 // set this as our current unloader
512 m_unloadingObjectID = unloader->getID();
513
514 // we've now used an unload (if we're keeping count for single exits)
515 if( m_unloadCount != UNLOAD_ALL )
516 --m_unloadCount;
517
518 } // end if
519
520} // end unloadNext
