188void BattlePlanUpdate::onDelete()
189{
190
191 // extend base class
192 UpdateModule::onDelete();
193
194 // delete our vision object, if it exists
195 Object *obj;
196 if( m_visionObjectID != INVALID_ID )
197 {
198 obj = TheGameLogic->findObjectByID( m_visionObjectID );
199 if( obj )
200 TheGameLogic->destroyObject( obj );
201 } // end if
202
203 // If we get destroyed, then make sure we remove our bonus!
204 // srj sez: we can't do this in the dtor because our team
205 // (and thus controlling player) has already been nulled by then...
206 Player* player = getObject()->getControllingPlayer();
207 // however, player CAN legitimately be null during game reset cycles
208 // (and which point it doesn't really matter if we can remove the bonus or not)
209 //DEBUG_ASSERTCRASH(player != NULL, ("Hmm, controller is null"));
210 if( player && m_planAffectingArmy != PLANSTATUS_NONE )
211 {
212 player->changeBattlePlan( m_planAffectingArmy, -1, m_bonuses );
213 }

240 m_searchAndDestroyPack.setEventName( data->m_searchAndDestroyPackName );

242 m_holdTheLineUnpack.setEventName( data->m_holdTheLineUnpackName );

267Bool BattlePlanUpdate::initiateIntentToDoSpecialPower(const SpecialPowerTemplate *specialPowerTemplate, const Object *targetObj, const Coord3D *targetPos, const Waypoint *way, UnsignedInt commandOptions )
268{
269 if( m_specialPowerModule->getSpecialPowerTemplate() != specialPowerTemplate )
270 {
271 //Check to make sure our modules are connected.
272 return FALSE;
273 }
274
275 //Set the desired status based on the command button option!
276 if( BitTest( commandOptions, OPTION_ONE ) )
277 {
278 m_desiredPlan = PLANSTATUS_BOMBARDMENT;
279 }
280 else if( BitTest( commandOptions, OPTION_TWO ) )
281 {
282 m_desiredPlan = PLANSTATUS_HOLDTHELINE;
283 }
284 else if( BitTest( commandOptions, OPTION_THREE ) )
285 {
286 m_desiredPlan = PLANSTATUS_SEARCHANDDESTROY;
287 }
288 else
289 {
290 DEBUG_CRASH( ("Selected an unsupported strategy for strategy center.") );
291 return FALSE;
292 }
293
294 getObject()->getControllingPlayer()->getAcademyStats()->recordBattlePlanSelected();
295 return TRUE;
296}

322UpdateSleepTime BattlePlanUpdate::update()
323{
324
325 if( m_invalidSettings )
326 {
327 // can't return UPDATE_SLEEP_FOREVER unless we are sleepy...
328 return UPDATE_SLEEP_NONE;
330 }
331
332 //const BattlePlanUpdateModuleData *data = getBattlePlanUpdateModuleData();
333 //Object *obj = getObject();
334
335 UnsignedInt now = TheGameLogic->getFrame();
336
337 if( m_nextReadyFrame <= now )
338 {
339 switch( m_status )
340 {
341 case TRANSITIONSTATUS_IDLE:
342 //There's only two cases where we are in an idle status -- upon initialization
343 //when no plan has yet been selected. The other case is after we've finished
344 //packing the previous plan up and are waiting to unpack the new state.
345 if( m_desiredPlan != PLANSTATUS_NONE )
346 {
347 m_currentPlan = m_desiredPlan;
348 setStatus( TRANSITIONSTATUS_UNPACKING );
349 }
350 break;
351 case TRANSITIONSTATUS_UNPACKING:
352 //If we're unpacking, we are forcing the user to wait until the plan is unpacked
353 //before allowing him to select a new plan. The plan doesn't become active until
354 //we're finished unpacking.
355 setStatus( TRANSITIONSTATUS_ACTIVE );
356 if( m_currentPlan == PLANSTATUS_BOMBARDMENT )
357 {
358 enableTurret( true );
359 }
360 break;
361 case TRANSITIONSTATUS_ACTIVE:
362 //If we're active and the user has selected a different plan, then we need to
363 //pack up.
364 if( m_currentPlan != m_desiredPlan )
365 {
366 if( m_currentPlan == PLANSTATUS_BOMBARDMENT )
367 {
368 //Special case situation -- in bombardment status, we need to center
369 //the turret prior to packing up, so handle it here.
370 AIUpdateInterface *ai = getObject()->getAI();
371 if( ai )
372 {
373 if( isTurretInNaturalPosition() )
374 {
375 //It's centered, so pack
376 setStatus( TRANSITIONSTATUS_PACKING );
377 m_centeringTurret = false;
378 enableTurret( false );
379 }
380 else if( !m_centeringTurret )
381 {
382 //It's not centered, and not trying to center, so order it to center.
383 ai->aiIdle( CMD_FROM_AI );
384 recenterTurret();
385 m_centeringTurret = true;
386 }
387 }
388 }
389 else
390 {
391 setStatus( TRANSITIONSTATUS_PACKING );
392 }
393 }
394 break;
395 case TRANSITIONSTATUS_PACKING:
396 //If we finished packing, then go idle until we can switch to our new plan.
397 setStatus( TRANSITIONSTATUS_IDLE );
398 break;
399 }
400 }
401
402 return UPDATE_SLEEP_NONE;
403}

408void BattlePlanUpdate::createVisionObject()
409{
410 if (m_visionObjectID != INVALID_ID) // don't want two.
411 return;
412
413 const BattlePlanUpdateModuleData *data = getBattlePlanUpdateModuleData();
414 Object *obj = getObject();
415
416 // get template of object to create
417 const ThingTemplate *tt = TheThingFactory->findTemplate( data->m_visionObjectName );
418 DEBUG_ASSERTCRASH( tt, ("BattlePlanUpdate::setStatus - Invalid vision object name '%s'\n",
419 data->m_visionObjectName.str()) );
420
421 if (!tt)
422 return;
423
424 Player *pPlayer = ThePlayerList->getNeutralPlayer();
425 // sanity
426 if(!pPlayer)
427 return;
428
429 Object *visionObject;
430
431 // create object for this player
432 visionObject = TheThingFactory->newObject( tt, pPlayer->getDefaultTeam() );
433 if( visionObject )
434 {
435
436 // record we have an object
437 m_visionObjectID = visionObject->getID();
438
439 // set position
440 visionObject->setPosition( obj->getPosition() );
441
442 // set the shroud clearing range
443 visionObject->setShroudClearingRange( obj->getGeometryInfo().getBoundingSphereRadius() );
444
445 } // end if
446
447} // end createVisionObject

450void BattlePlanUpdate::setStatus( TransitionStatus newStatus )
451{
452 const BattlePlanUpdateModuleData *data = getBattlePlanUpdateModuleData();
453 Object *obj = getObject();
454
455 if( m_status == newStatus )
456 {
457 return;
458 }
459
460 TransitionStatus oldStatus = m_status;
461
462 //Turn off old defining states and sounds
463 switch( oldStatus )
464 {
465 case TRANSITIONSTATUS_IDLE:
466 break;
467
468 case TRANSITIONSTATUS_UNPACKING:
469 switch( m_currentPlan )
470 {
471 case PLANSTATUS_BOMBARDMENT:
472 obj->clearModelConditionState( MODELCONDITION_DOOR_1_OPENING );
473 TheAudio->removeAudioEvent( m_bombardmentUnpack.getPlayingHandle() );
474 break;
475 case PLANSTATUS_HOLDTHELINE:
476 obj->clearModelConditionState( MODELCONDITION_DOOR_2_OPENING );
477 TheAudio->removeAudioEvent( m_holdTheLineUnpack.getPlayingHandle() );
478 break;
479 case PLANSTATUS_SEARCHANDDESTROY:
480 obj->clearModelConditionState( MODELCONDITION_DOOR_3_OPENING );
481 TheAudio->removeAudioEvent( m_searchAndDestroyUnpack.getPlayingHandle() );
482 break;
483 }
484 break;
485
486 case TRANSITIONSTATUS_ACTIVE:
487 switch( m_currentPlan )
488 {
489 case PLANSTATUS_BOMBARDMENT:
490 obj->clearModelConditionState( MODELCONDITION_DOOR_1_WAITING_TO_CLOSE );
491 break;
492 case PLANSTATUS_HOLDTHELINE:
493 obj->clearModelConditionState( MODELCONDITION_DOOR_2_WAITING_TO_CLOSE );
494 break;
495 case PLANSTATUS_SEARCHANDDESTROY:
496 obj->clearModelConditionState( MODELCONDITION_DOOR_3_WAITING_TO_CLOSE );
497 TheAudio->removeAudioEvent( m_searchAndDestroyIdle.getPlayingHandle() );
498 break;
499 }
500 break;
501
502 case TRANSITIONSTATUS_PACKING:
503 switch( m_currentPlan )
504 {
505 case PLANSTATUS_BOMBARDMENT:
506 obj->clearModelConditionState( MODELCONDITION_DOOR_1_CLOSING );
507 TheAudio->removeAudioEvent( m_bombardmentPack.getPlayingHandle() );
508 break;
509 case PLANSTATUS_HOLDTHELINE:
510 obj->clearModelConditionState( MODELCONDITION_DOOR_2_CLOSING );
511 TheAudio->removeAudioEvent( m_holdTheLinePack.getPlayingHandle() );
512 break;
513 case PLANSTATUS_SEARCHANDDESTROY:
514 obj->clearModelConditionState( MODELCONDITION_DOOR_3_CLOSING );
515 TheAudio->removeAudioEvent( m_searchAndDestroyPack.getPlayingHandle() );
516 break;
517 }
518 break;
519 }
520
521 UnsignedInt now = TheGameLogic->getFrame();
522
523 //Handle entering new status
524 switch( newStatus )
525 {
526 case TRANSITIONSTATUS_IDLE:
527 {
528 m_currentPlan = PLANSTATUS_NONE;
529 m_nextReadyFrame = now + data->m_transitionIdleFrames;
530 break;
531 }
532
533 case TRANSITIONSTATUS_UNPACKING:
534 {
535
536 // play a radar blip showing the battle plan change in the color of the
537 TheRadar->createPlayerEvent( obj->getControllingPlayer(),
538 obj->getPosition(),
539 RADAR_EVENT_BATTLE_PLAN );
540
541 //this is now handled with ShroudRevealToAllRange in thingTemplate
542 //createVisionObject();
543
544 switch( m_currentPlan )
545 {
546 case PLANSTATUS_BOMBARDMENT:
547 obj->setModelConditionState( MODELCONDITION_DOOR_1_OPENING );
548 obj->getDrawable()->setAnimationLoopDuration( data->m_bombardmentPlanAnimationFrames );
549 m_nextReadyFrame = now + data->m_bombardmentPlanAnimationFrames;
550 if( m_bombardmentUnpack.getEventName().isNotEmpty() )
551 {
552 m_bombardmentUnpack.setObjectID( obj->getID() );
553 m_bombardmentUnpack.setPlayingHandle( TheAudio->addAudioEvent( &m_bombardmentUnpack ) );
554 }
555
556 // display a message to *all* users
557 TheInGameUI->message( TheGameText->fetch( data->m_bombardmentMessageLabel ) );
558 if( m_bombardmentAnnouncement.getEventName().isEmpty() == FALSE )
559 {
560 m_bombardmentAnnouncement.setPosition( obj->getPosition() );
561 TheAudio->addAudioEvent( &m_bombardmentAnnouncement );
562 }
563 break;
564
565 case PLANSTATUS_HOLDTHELINE:
566 obj->setModelConditionState( MODELCONDITION_DOOR_2_OPENING );
567 obj->getDrawable()->setAnimationLoopDuration( data->m_holdTheLinePlanAnimationFrames );
568 m_nextReadyFrame = now + data->m_holdTheLinePlanAnimationFrames;
569 if( m_holdTheLineUnpack.getEventName().isNotEmpty() )
570 {
571 m_holdTheLineUnpack.setObjectID( obj->getID() );
572 m_holdTheLineUnpack.setPlayingHandle( TheAudio->addAudioEvent( &m_holdTheLineUnpack ) );
573 }
574
575 // display a message to *all* users
576 TheInGameUI->message( TheGameText->fetch( data->m_holdTheLineMessageLabel ) );
577 if( m_holdTheLineAnnouncement.getEventName().isEmpty() == FALSE )
578 {
579 m_holdTheLineAnnouncement.setPosition( obj->getPosition() );
580 TheAudio->addAudioEvent( &m_holdTheLineAnnouncement );
581 }
582 break;
583
584 case PLANSTATUS_SEARCHANDDESTROY:
585 obj->setModelConditionState( MODELCONDITION_DOOR_3_OPENING );
586 obj->getDrawable()->setAnimationLoopDuration( data->m_searchAndDestroyPlanAnimationFrames );
587 m_nextReadyFrame = now + data->m_searchAndDestroyPlanAnimationFrames;
588 if( m_searchAndDestroyUnpack.getEventName().isNotEmpty() )
589 {
590 m_searchAndDestroyUnpack.setObjectID( obj->getID() );
591 m_searchAndDestroyUnpack.setPlayingHandle( TheAudio->addAudioEvent( &m_searchAndDestroyUnpack ) );
592 }
593
594 // display a message to *all* users
595 TheInGameUI->message( TheGameText->fetch( data->m_searchAndDestroyMessageLabel ) );
596 if( m_searchAndDestroyAnnouncement.getEventName().isEmpty() == FALSE )
597 {
598 m_searchAndDestroyAnnouncement.setPosition( obj->getPosition() );
599 TheAudio->addAudioEvent( &m_searchAndDestroyAnnouncement );
600 }
601 break;
602
603 }
604
605 break;
606
607 }
608
609 case TRANSITIONSTATUS_ACTIVE:
610
611 setBattlePlan( m_currentPlan );
612
613 switch( m_currentPlan )
614 {
615 case PLANSTATUS_BOMBARDMENT:
616 obj->setModelConditionState( MODELCONDITION_DOOR_1_WAITING_TO_CLOSE );
617 break;
618 case PLANSTATUS_HOLDTHELINE:
619 obj->setModelConditionState( MODELCONDITION_DOOR_2_WAITING_TO_CLOSE );
620 break;
621 case PLANSTATUS_SEARCHANDDESTROY:
622 obj->setModelConditionState( MODELCONDITION_DOOR_3_WAITING_TO_CLOSE );
623 if( m_searchAndDestroyIdle.getEventName().isNotEmpty() )
624 {
625 m_searchAndDestroyIdle.setObjectID( obj->getID() );
626 m_searchAndDestroyIdle.setPlayingHandle( TheAudio->addAudioEvent( &m_searchAndDestroyIdle ) );
627 }
628 break;
629 }
630 break;
631
632 case TRANSITIONSTATUS_PACKING:
633

732//------------------------------------------------------------------------------------------------
733void BattlePlanUpdate::setBattlePlan( BattlePlanStatus plan )
734{
735 const BattlePlanUpdateModuleData *data = getBattlePlanUpdateModuleData();
736 Object *obj = getObject();
737 Player *player = obj->getControllingPlayer();
738 if( player )
739 {
740 switch( m_planAffectingArmy )
741 {
742 case PLANSTATUS_BOMBARDMENT:
743 {
744 //Remove the previous plan!
745 player->changeBattlePlan( PLANSTATUS_BOMBARDMENT, -1, m_bonuses );
746
747 //The only building bonus is actually the turret, and that's already handled!
748 break;
749 }
750 case PLANSTATUS_HOLDTHELINE:
751 {
752 //Remove the previous plan!
753 player->changeBattlePlan( PLANSTATUS_HOLDTHELINE, -1, m_bonuses );
754
755 //Remove building health bonuses
756 if( data->m_strategyCenterHoldTheLineMaxHealthScalar != 1.0f )
757 {
758 BodyModuleInterface *body = obj->getBodyModule();
759 body->setMaxHealth( body->getMaxHealth() * 1.0f / data->m_strategyCenterHoldTheLineMaxHealthScalar, data->m_strategyCenterHoldTheLineMaxHealthChangeType );
760 }
761 break;
762 }
763 case PLANSTATUS_SEARCHANDDESTROY:
764 {
765 //Remove the previous plan!
766 player->changeBattlePlan( PLANSTATUS_SEARCHANDDESTROY, -1, m_bonuses );
767
768 //Remove sight range bonus
769 if( data->m_strategyCenterSearchAndDestroySightRangeScalar != 1.0f )
770 {
771 obj->setVisionRange( obj->getVisionRange() * 1.0f / data->m_strategyCenterSearchAndDestroySightRangeScalar );
772 obj->setShroudClearingRange( obj->getShroudClearingRange() * 1.0f / data->m_strategyCenterSearchAndDestroySightRangeScalar );
773 }
774
775 //Remove stealth detection
776 if( data->m_strategyCenterSearchAndDestroyDetectsStealth )
777 {
778 static NameKeyType key_StealthDetectorUpdate = NAMEKEY( "StealthDetectorUpdate" );
779 StealthDetectorUpdate *update = (StealthDetectorUpdate*)obj->findUpdateModule( key_StealthDetectorUpdate );
780 if( update )
781 {
782 update->setSDEnabled( false );
783 }
784 }
785
786 break;
787 }
788 }
789
790 //Revert to default no-bonuses!
791 m_bonuses->m_armorScalar = 1.0f;
792 m_bonuses->m_sightRangeScalar = 1.0f;
793 m_bonuses->m_bombardment = 0;
794 m_bonuses->m_searchAndDestroy = 0;
795 m_bonuses->m_holdTheLine = 0;
796
797 //Add new bonuses!
798 switch( plan )
799 {
800 case PLANSTATUS_NONE:
801 //Paralyze troops!
802 player->iterateObjects( paralyzeTroop, (void*)data );
803 break;
804
805 case PLANSTATUS_BOMBARDMENT:
806 //Set the bombardment bonuses
807 m_bonuses->m_bombardment = 1; //for weapon bonuses
808
809 //Add the new plan!
810 player->changeBattlePlan( PLANSTATUS_BOMBARDMENT, 1, m_bonuses );
811 break;
812
813 case PLANSTATUS_HOLDTHELINE:
814 //Add building health bonuses
815 if( data->m_strategyCenterHoldTheLineMaxHealthScalar )
816 {
817 BodyModuleInterface *body = obj->getBodyModule();
818 body->setMaxHealth( body->getMaxHealth() * data->m_strategyCenterHoldTheLineMaxHealthScalar, data->m_strategyCenterHoldTheLineMaxHealthChangeType );
819 }
820
821 //Set the hold-the-line bonuses
822 m_bonuses->m_armorScalar = data->m_holdTheLineArmorDamageScalar;
823 m_bonuses->m_holdTheLine = 1; //for weapon bonuses
824
825 //Add the new plan!
826 player->changeBattlePlan( PLANSTATUS_HOLDTHELINE, 1, m_bonuses );
827 break;
828 case PLANSTATUS_SEARCHANDDESTROY:
829 //Add sight range bonus
830 if( data->m_strategyCenterSearchAndDestroySightRangeScalar != 1.0f )
831 {
832 obj->setVisionRange( obj->getVisionRange() * data->m_strategyCenterSearchAndDestroySightRangeScalar );
833 obj->setShroudClearingRange( obj->getShroudClearingRange() * data->m_strategyCenterSearchAndDestroySightRangeScalar );
834 }
835
836 //Enable stealth detection
837 if( data->m_strategyCenterSearchAndDestroyDetectsStealth )
838 {
839 static NameKeyType key_StealthDetectorUpdate = NAMEKEY( "StealthDetectorUpdate" );
840 StealthDetectorUpdate *update = (StealthDetectorUpdate*)obj->findUpdateModule( key_StealthDetectorUpdate );
841 if( update )
842 {
843 update->setSDEnabled( true );
844 }
845 }
846
847 //Set the search-and-destroy bonuses
848 m_bonuses->m_searchAndDestroy = 1; //for weapon bonuses
849 m_bonuses->m_sightRangeScalar = data->m_searchAndDestroySightRangeScalar;
850
851 //Add the new plan!
852 player->changeBattlePlan( PLANSTATUS_SEARCHANDDESTROY, 1, m_bonuses );
853 break;
854 }
855 m_planAffectingArmy = plan;
856 }
857}
