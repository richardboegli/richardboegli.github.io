56// TagBlockFile: Enables a file to have named (tagged) variable size blocks. User can
57// then open a block for reading. User may also create new blocks at the end of the
58// file. There may only be one block open for writting, unlimited blocks can be open for
59// reading. It can be thought of as a .MIX file that can have files added to it at any time.
60// One problem is it is a Write Once/Read Many solution.
61//
62// Usage: All user access to a TagBlockFile is done through TagBlockHandle.
63// A TagBlockHandle is created by TagBlockFile::Create_Tag() or Open_Tag(). It is destroyed
64// by either TagBlockFile::Close_Tag() or you can just destroy the handle with delete().
65//
66//
67

117 // This is the header that is in both the IndexFile and the DataFile.
118 // They should be match except for the difference in the Version number as defined by enum.
119 struct FileHeader

135 // Each block in the file has a header before it.
136 struct BlockHeader
137 {
138 BlockHeader() {memset(this, 0, sizeof(*this));}
139 BlockHeader(int index, int tagsize, int datasize):Index(index),TagSize(tagsize),DataSize(datasize) {}
140 BlockHeader(BlockHeader& bh) {memcpy(this, &bh, sizeof(BlockHeader));}
141
142 // Used to verify file integrity.
143 int Index;
144
145 // Size of tagname (including NULL) that follows this block.
146 int TagSize;
147
148 // Size of block not including header.
149 int DataSize;
150
151 // A variable length name (NULL terminated) follows this structure.
152 // The name is then followed by the Data.
153
154 // The entire length of the block is sizeof(BlockHeader) + TagSize + DataSize.
155 };
