62Bool SalvageCrateCollide::isValidToExecute( const Object *other ) const
63{
64 if( ! CrateCollide::isValidToExecute( other ) )
65 return FALSE;
66
67 // Only salvage units can pick up a Salvage crate
68 if( ! other->getTemplate()->isKindOf( KINDOF_SALVAGER ) )
69 return FALSE;
70
71 return TRUE;
72}

75Bool SalvageCrateCollide::executeCrateBehavior( Object *other )
76{
77 if( eligibleForArmorSet(other) )// No percent chance on this one, if you can get it, you get it.

79 doArmorSet(other);
80

93 TheAudio->addAudioEvent( &soundToPlay );
94
95 //Play the unit voice acknowledgement for upgrading weapons.

97 //const AudioEventRTS *soundToPlayPtr = other->getTemplate()->getPerUnitSound( "VoiceSalvage" );
98 //soundToPlay = *soundToPlayPtr;
99 //soundToPlay.setObjectID( other->getID() );
100 //TheAudio->addAudioEvent( &soundToPlay );
101 }
102 else if( eligibleForLevel( other ) && testLevelChance() )
103 {
104 doLevelGain( other );
105
106 //Sound will play in
107 //soundToPlay = TheAudio->getMiscAudio()->m_unitPromoted;
108 }
109 else // just assume the testMoneyChance
110 {
111 doMoney( other );
112 AudioEventRTS soundToPlay = TheAudio->getMiscAudio()->m_crateMoney;
113 soundToPlay.setObjectID( other->getID() );
114 TheAudio->addAudioEvent(&soundToPlay);
115 }
116
117 other->getControllingPlayer()->getAcademyStats()->recordSalvageCollected();
118
119 return TRUE;
120}

123Bool SalvageCrateCollide::eligibleForWeaponSet( Object *other )
124{
125 if( other == NULL )
126 return FALSE;
127
128 // A kindof marks eligibility, and you must not be fully upgraded
129 if( !other->isKindOf(KINDOF_WEAPON_SALVAGER) )
130 return FALSE;
131 if( other->testWeaponSetFlag(WEAPONSET_CRATEUPGRADE_TWO) )
132 return FALSE;
133
134 return TRUE;

153Bool SalvageCrateCollide::eligibleForLevel( Object *other )
154{
155 if( other == NULL )
156 return FALSE;
157
158 // Sorry, you are max level
159 if( other->getExperienceTracker()->getVeterancyLevel() == LEVEL_HEROIC )
160 return FALSE;
161
162 // Sorry, you can't gain levels
163 if( !other->getExperienceTracker()->isTrainable() )
164 return FALSE;
165

170Bool SalvageCrateCollide::testWeaponChance()
171{
172 const SalvageCrateCollideModuleData *md = getSalvageCrateCollideModuleData();
173 if( md->m_weaponChance == 1.0f )
174 return TRUE; // don't waste a random number for a 100%
175
176 Real randomNumber = GameLogicRandomValueReal( 0, 1 );
177 if( randomNumber < md->m_weaponChance )
178 return TRUE;
179
180 return FALSE;

198void SalvageCrateCollide::doWeaponSet( Object *other )
199{
200 if( other->testWeaponSetFlag( WEAPONSET_CRATEUPGRADE_ONE ) )
201 {
202 other->clearWeaponSetFlag( WEAPONSET_CRATEUPGRADE_ONE );
203 other->setWeaponSetFlag( WEAPONSET_CRATEUPGRADE_TWO );
204 }
205 else
206 {
207 other->setWeaponSetFlag( WEAPONSET_CRATEUPGRADE_ONE );
208 }
209}

211// ------------------------------------------------------------------------------------------------
212void SalvageCrateCollide::doArmorSet( Object *other )
213{
214 if( other->testArmorSetFlag( ARMORSET_CRATE_UPGRADE_ONE ) )
215 {
216 other->clearArmorSetFlag( ARMORSET_CRATE_UPGRADE_ONE );
217 other->setArmorSetFlag( ARMORSET_CRATE_UPGRADE_TWO );
218
219 other->clearAndSetModelConditionState(MODELCONDITION_ARMORSET_CRATEUPGRADE_ONE, MODELCONDITION_ARMORSET_CRATEUPGRADE_TWO);
220 }
221 else
222 {
223 other->setArmorSetFlag( ARMORSET_CRATE_UPGRADE_ONE );
224
225 other->setModelConditionState(MODELCONDITION_ARMORSET_CRATEUPGRADE_ONE);
226 }
227}

236void SalvageCrateCollide::doMoney( Object *other )
237{
238 const SalvageCrateCollideModuleData *md = getSalvageCrateCollideModuleData();
239
240 Int money;
241 if( md->m_minimumMoney != md->m_maximumMoney )// Random value doesn't like to get a constant range
242 money = GameLogicRandomValue( md->m_minimumMoney, md->m_maximumMoney );
243 else
244 money = md->m_minimumMoney;
245
246 if( money > 0 )
247 {
248 other->getControllingPlayer()->getMoney()->deposit( money );
249 other->getControllingPlayer()->getScoreKeeper()->addMoneyEarned( money );
250
251 //Display cash income floating over the crate. Position is me, everything else is them.
252 UnicodeString moneyString;
253 moneyString.format( TheGameText->fetch( "GUI:AddCash" ), money );
254 Coord3D pos;
255 pos.set( getObject()->getPosition() );
256 pos.z += 10.0f; //add a little z to make it show up above the unit.
257 Color color = other->getControllingPlayer()->getPlayerColor() | GameMakeColor( 0, 0, 0, 230 );
258 TheInGameUI->addFloatingText( moneyString, &pos, color );
259 }
260}
