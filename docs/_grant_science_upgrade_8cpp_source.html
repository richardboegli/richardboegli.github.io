90void GrantScienceUpgrade::upgradeImplementation( )
91{
92 if( m_scienceType == SCIENCE_INVALID )
93 {
94 const GrantScienceUpgradeModuleData *data = getGrantScienceUpgradeModuleData();
95 m_scienceType = TheScienceStore->getScienceFromInternalName( data->m_grantScienceName );
96 }
97
98 Object *obj = getObject();
99 Player *player = obj->getControllingPlayer();
100 if( player )
101 {
102 player->grantScience( m_scienceType );
103 }
104}
