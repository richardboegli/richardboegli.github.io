174void W3DTruckDraw::createEmitters( void )
175{
176 if (getDrawable()->isDrawableEffectivelyHidden())
177 return;
178 if (getW3DTruckDrawModuleData())
179 {
180 const ParticleSystemTemplate *sysTemplate;
181
182 if (!m_dustEffect) {
183
184 sysTemplate = TheParticleSystemManager->findTemplate(getW3DTruckDrawModuleData()->m_dustEffectName);
185 if (sysTemplate)
186 {
187 m_dustEffect = TheParticleSystemManager->createParticleSystem( sysTemplate );
188 m_dustEffect->attachToObject(getDrawable()->getObject());
189 // important: mark it as do-not-save, since we'll just re-create it when we reload.
190 m_dustEffect->setSaveable(FALSE);
191 } else {
192 if (!getW3DTruckDrawModuleData()->m_dustEffectName.isEmpty()) {
193 DEBUG_LOG(("*** ERROR - Missing particle system '%s' in thing '%s'\n",
194 getW3DTruckDrawModuleData()->m_dustEffectName.str(), getDrawable()->getObject()->getTemplate()->getName().str()));
195 }
196 }
197
198 }
199 if (!m_dirtEffect) {
200 sysTemplate = TheParticleSystemManager->findTemplate(getW3DTruckDrawModuleData()->m_dirtEffectName);
201 if (sysTemplate)
202 {
203 m_dirtEffect = TheParticleSystemManager->createParticleSystem( sysTemplate );
204 m_dirtEffect->attachToObject(getDrawable()->getObject());
205 // important: mark it as do-not-save, since we'll just re-create it when we reload.
206 m_dirtEffect->setSaveable(FALSE);
207 } else {
208 if (!getW3DTruckDrawModuleData()->m_dirtEffectName.isEmpty()) {
209 DEBUG_LOG(("*** ERROR - Missing particle system '%s' in thing '%s'\n",
210 getW3DTruckDrawModuleData()->m_dirtEffectName.str(), getDrawable()->getObject()->getTemplate()->getName().str()));
211 }
212 }
213 }
214 if (!m_powerslideEffect) {

389void W3DTruckDraw::doDrawModule(const Matrix3D* transformMtx)
390{
391
392 W3DModelDraw::doDrawModule(transformMtx);
393
394 if (!TheGlobalData->m_showClientPhysics)
395 return;
396 const W3DTruckDrawModuleData *moduleData = getW3DTruckDrawModuleData();
397 if (moduleData==NULL) return; // shouldn't ever happen.
398
399 Bool frozen = TheTacticalView->isTimeFrozen() && !TheTacticalView->isCameraMovementFinished();
400 frozen = frozen || TheScriptEngine->isTimeFrozenDebug() || TheScriptEngine->isTimeFrozenScript();
401 if (frozen)
402 return;
403
404 const Real ACCEL_THRESHOLD = 0.01f;
405 const Real SIZE_CAP = 2.0f;
406 // get object from logic
407 Object *obj = getDrawable()->getObject();
408 if (obj == NULL)
409 return;
410
411 if (getRenderObject()==NULL) return;
412 if (getRenderObject() != m_prevRenderObj) {
413 DEBUG_LOG(("W3DTruckDraw::doDrawModule - shouldn't update bones. jba\n"));
414 updateBones();
415 }
416
417 // get object physics state
418 PhysicsBehavior *physics = obj->getPhysics();
419 if (physics == NULL)
420 return;
421
422 const Coord3D *vel = physics->getVelocity();
423 Real speed = physics->getVelocityMagnitude();
424
425 const TWheelInfo *wheelInfo = getDrawable()->getWheelInfo(); // note, can return null!
426 AIUpdateInterface *ai = obj->getAI();
427 if (m_cabBone && wheelInfo) {
428 Matrix3D cabXfrm(1);
429 cabXfrm.Make_Identity();
430 Real desiredAngle = wheelInfo->m_wheelAngle*moduleData->m_cabRotationFactor;
431
432 // Check goal angle.
433 if (ai && ai->getPath())
434 {
435 Coord3D pointOnPath;
436 ai->getPath()->peekCachedPointOnPath(pointOnPath);
437 Real angleToGoal = ThePartitionManager->getRelativeAngle2D( obj, &pointOnPath );
438 //DEBUG_LOG(("To goal %f, desired %f ", 180*angleToGoal/PI, 180*desiredAngle/PI));
439 if (angleToGoal<0) {
440 if (desiredAngle<angleToGoal) desiredAngle=angleToGoal;
441 if (desiredAngle>0) desiredAngle = 0;
442 } else {
443 if (desiredAngle>angleToGoal) desiredAngle = angleToGoal;
444 if (desiredAngle<0) desiredAngle = 0;
445 }
446 //DEBUG_LOG(("final desired %f ", 180*desiredAngle/PI));
447 }
448
449 Real deltaAngle = desiredAngle - m_curCabRotation;
450 deltaAngle *= moduleData->m_rotationDampingFactor;
451 m_curCabRotation += deltaAngle;
452 cabXfrm.Rotate_Z(m_curCabRotation);
453 getRenderObject()->Capture_Bone( m_cabBone );
454 getRenderObject()->Control_Bone( m_cabBone, cabXfrm );
455 if (m_trailerBone && wheelInfo) {
456 desiredAngle = -wheelInfo->m_wheelAngle*moduleData->m_trailerRotationFactor;
457 Real deltaAngle = desiredAngle - m_curTrailerRotation;
458 deltaAngle *= moduleData->m_rotationDampingFactor;
459 m_curTrailerRotation += deltaAngle;
460 cabXfrm.Make_Identity();
461 cabXfrm.Rotate_Z(m_curTrailerRotation);
462 getRenderObject()->Capture_Bone( m_trailerBone );
463 getRenderObject()->Control_Bone( m_trailerBone, cabXfrm );
464 }
465 }
466
467 if (m_frontLeftTireBone || m_rearLeftTireBone)
468 {
469 Real powerslideRotationAddition = moduleData->m_powerslideRotationAddition;
470 if (ai) {
471 Locomotor *loco = ai->getCurLocomotor();
472 if (loco) {
473 if (loco->isMovingBackwards()) {
474 speed = -speed; // rotate wheels backwards. jba.
475 powerslideRotationAddition = -powerslideRotationAddition;
476 }
477 }
478 }
479 const Real rotationFactor = moduleData->m_rotationSpeedMultiplier;
480 m_frontWheelRotation += rotationFactor*speed;
481 if (m_isPowersliding)
482 {
483 m_rearWheelRotation += rotationFactor*(speed + powerslideRotationAddition);
484 }
485 else
486 {
487 m_rearWheelRotation += rotationFactor*speed;
488 }
489
490 // For now, just use the same values for mid wheels -- may want to do independent calcs later...
491 m_midFrontWheelRotation = m_frontWheelRotation;
492 m_midRearWheelRotation = m_rearWheelRotation;
493
494 Matrix3D wheelXfrm(1);
495
496
497
498 if (m_frontLeftTireBone && wheelInfo)
499 {
500 wheelXfrm.Make_Identity();
501 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_frontLeftHeightOffset);
502 wheelXfrm.Rotate_Z(wheelInfo->m_wheelAngle);
503 wheelXfrm.Rotate_Y(m_frontWheelRotation);
504 getRenderObject()->Capture_Bone( m_frontLeftTireBone );
505 getRenderObject()->Control_Bone( m_frontLeftTireBone, wheelXfrm );
506
507
508 wheelXfrm.Make_Identity();
509 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_frontRightHeightOffset);
510 wheelXfrm.Rotate_Z(wheelInfo->m_wheelAngle);
511 wheelXfrm.Rotate_Y(m_frontWheelRotation);
512 getRenderObject()->Capture_Bone( m_frontRightTireBone );
513 getRenderObject()->Control_Bone( m_frontRightTireBone, wheelXfrm );
514 }
515 if (m_rearLeftTireBone && wheelInfo)
516 {
517 wheelXfrm.Make_Identity();
518 wheelXfrm.Rotate_Y(m_rearWheelRotation);
519 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearLeftHeightOffset);
520 getRenderObject()->Capture_Bone( m_rearLeftTireBone );
521 getRenderObject()->Control_Bone( m_rearLeftTireBone, wheelXfrm );
522
523 wheelXfrm.Make_Identity();
524 wheelXfrm.Rotate_Y(m_rearWheelRotation);
525 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearRightHeightOffset);
526
527 //@todo TROUBLE HERE, THE BONE INDICES DO NOT MATCH THE RENDEROBJECTS BONES, SOMETIMES
528
529 getRenderObject()->Capture_Bone( m_rearRightTireBone );
530 getRenderObject()->Control_Bone( m_rearRightTireBone, wheelXfrm );
531 }
532 if (m_midFrontLeftTireBone && wheelInfo)
533 {
534 wheelXfrm.Make_Identity();
535 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_frontLeftHeightOffset);
536 wheelXfrm.Rotate_Z(wheelInfo->m_wheelAngle);
537 wheelXfrm.Rotate_Y(m_midFrontWheelRotation);
538 getRenderObject()->Capture_Bone( m_midFrontLeftTireBone );
539 getRenderObject()->Control_Bone( m_midFrontLeftTireBone, wheelXfrm );
540
541 wheelXfrm.Make_Identity();
542 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_frontRightHeightOffset);
543 wheelXfrm.Rotate_Z(wheelInfo->m_wheelAngle);
544 wheelXfrm.Rotate_Y(m_midFrontWheelRotation);
545 getRenderObject()->Capture_Bone( m_midFrontRightTireBone );
546 getRenderObject()->Control_Bone( m_midFrontRightTireBone, wheelXfrm );
547 }
548 if (m_midRearLeftTireBone && wheelInfo)
549 {
550 wheelXfrm.Make_Identity();
551 wheelXfrm.Rotate_Y(m_midRearWheelRotation);
552 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearLeftHeightOffset);
553 getRenderObject()->Capture_Bone( m_midRearLeftTireBone );
554 getRenderObject()->Control_Bone( m_midRearLeftTireBone, wheelXfrm );
555
556 wheelXfrm.Make_Identity();
557 wheelXfrm.Rotate_Y(m_midRearWheelRotation);
558 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearRightHeightOffset);
559 getRenderObject()->Capture_Bone( m_midRearRightTireBone );
560 getRenderObject()->Control_Bone( m_midRearRightTireBone, wheelXfrm );
561 }
562 if (m_midMidLeftTireBone && wheelInfo)
563 {
564 wheelXfrm.Make_Identity();
565 wheelXfrm.Rotate_Y(m_midRearWheelRotation);
566 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearLeftHeightOffset);
567 getRenderObject()->Capture_Bone( m_midMidLeftTireBone );
568 getRenderObject()->Control_Bone( m_midMidLeftTireBone, wheelXfrm );
569
570 wheelXfrm.Make_Identity();
571 wheelXfrm.Rotate_Y(m_midRearWheelRotation);
572 wheelXfrm.Adjust_Z_Translation(wheelInfo->m_rearRightHeightOffset);
573 getRenderObject()->Capture_Bone( m_midMidRightTireBone );
574 getRenderObject()->Control_Bone( m_midMidRightTireBone, wheelXfrm );
575 }
576 }
577
578 Bool wasPowersliding = m_isPowersliding;
579 m_isPowersliding = false;
580 if (physics->isMotive() && !obj->isSignificantlyAboveTerrain()) {
581 enableEmitters(true);
582 Coord3D accel = *physics->getAcceleration();
583 accel.z = 0; // ignore gravitational force.
584 Bool accelerating = accel.length()>ACCEL_THRESHOLD;
585 //DEBUG_LOG(("Accel %f, speed %f\n", accel.length(), speed));
586 if (accelerating) {
587 Real dot = accel.x*vel->x + accel.y*vel->y;
588 if (dot<0) {
589 accelerating = false; // decelerating, actually.
590 }
591 }
592 if (m_dustEffect) {
593 // Need more dust the faster we go.
594 if (speed>SIZE_CAP) {
595 speed = SIZE_CAP;
596 }
597 m_dustEffect->setSizeMultiplier(speed);
598 }
599 if (m_dirtEffect) {
600 if (wheelInfo && wheelInfo->m_framesAirborne>3) {
601 Real factor = 1 + wheelInfo->m_framesAirborne/16;
602 if (factor>2.0) factor = 2.0;
603 m_dustEffect->setSizeMultiplier(factor*SIZE_CAP);
604 m_dustEffect->trigger();
605 m_landingSound.setObjectID(obj->getID());
606 TheAudio->addAudioEvent(&m_landingSound);
607 } else {
608 if (!accelerating || speed>2.0f) {
609 m_dirtEffect->stop();
610 }
611 }
612 }
613 if (m_powerslideEffect) {
614 if (physics->getTurning() == TURN_NONE) {
615 m_powerslideEffect->stop();
616 } else {
617 m_isPowersliding = true;
618 m_powerslideEffect->start();
619 }
620 }
621 if (m_dirtEffect) {
622 if (!accelerating || speed>2.0f) {
623 m_dirtEffect->stop();
624 }
625 }
626 }
627 else
628 enableEmitters(false);
629
630 m_wasAirborne = obj->isSignificantlyAboveTerrain();
631
632 if(!wasPowersliding && m_isPowersliding) {
633 // start sound
634 m_powerslideSound.setObjectID(obj->getID());
635 m_powerslideSound.setPlayingHandle(TheAudio->addAudioEvent(&m_powerslideSound));
636 } else if (wasPowersliding && !m_isPowersliding) {
637 TheAudio->removeAudioEvent(m_powerslideSound.getPlayingHandle());
638 }
639
640}
