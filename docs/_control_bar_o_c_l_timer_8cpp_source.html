47void ControlBar::updateOCLTimerTextDisplay( UnsignedInt totalSeconds, Real percent )
48{
49 UnicodeString text;
50 static UnsignedInt descID = TheNameKeyGenerator->nameToKey( "ControlBar.wnd:OCLTimerStaticText" );
51 GameWindow *descWindow = TheWindowManager->winGetWindowFromId( NULL, descID );
52
53 static UnsignedInt barID = TheNameKeyGenerator->nameToKey( "ControlBar.wnd:OCLTimerProgressBar" );
54 GameWindow *barWindow = TheWindowManager->winGetWindowFromId( NULL, barID );
55
56 // santiy
57 DEBUG_ASSERTCRASH( descWindow, ("Under construction window not found\n") );
58
59 Int minutes = totalSeconds / 60;
60 Int seconds = totalSeconds - (minutes * 60);
61
62 // format the message
63 if( seconds < 10 )
64 text.format( TheGameText->fetch( "CONTROLBAR:OCLTimerDescWithPadding" ), minutes, seconds );
65 else
66 text.format( TheGameText->fetch( "CONTROLBAR:OCLTimerDesc" ), minutes, seconds );
67
68 GadgetStaticTextSetText( descWindow, text );
69 GadgetProgressBarSetProgress(barWindow, (percent * 100));
70
71 // record this as the last time displayed
72 m_displayedOCLTimerSeconds = totalSeconds;
73
74} // end updateOCLTimerTextDisplay

79void ControlBar::populateOCLTimer( Object *creatorObject )
80{
81
82 // sanity
83 if( creatorObject == NULL )
84 return;
85
86 // get our parent window
87 GameWindow *parent = m_contextParent[ CP_OCL_TIMER ];
88
89 // set the sell button
91 NameKeyType id;
92 id = TheNameKeyGenerator->nameToKey( "ControlBar.wnd:OCLTimerSellButton" );
93 GameWindow *win = TheWindowManager->winGetWindowFromId( parent, id );
94
95 if( !creatorObject->isKindOf(KINDOF_TECH_BUILDING) )
96 {
97 // Surgical bug fix. srj is right, this is stupid.
98 const CommandButton *commandButton = findCommandButton( "Command_Sell" );
99 setControlCommand( win, commandButton );
100 win->winSetStatus( WIN_STATUS_USE_OVERLAY_STATES );
101 }
102 // Another last minute hack due to time constraint and feature creep
103 else if( creatorObject->isKindOf(KINDOF_TECH_BUILDING) && creatorObject->isKindOf(KINDOF_AUTO_RALLYPOINT) )
104 {
105 // This time we want a rally point button to show up instead of a sell button
106 const CommandButton *commandButton = findCommandButton( "Command_SetRallyPoint" );
107 setControlCommand( win, commandButton );
108 win->winSetStatus( WIN_STATUS_USE_OVERLAY_STATES );
109
110 //
111 // for objects that have a production exit interface, we may have a rally point set.
112 // if we do, we want to show that rally point in the world
113 //
114 ExitInterface *exit = creatorObject->getObjectExitInterface();
115 if( exit )
116 {
117 //
118 // if a rally point is set, show the rally point, if we don't have it set hide any rally
119 // point we might have visible
120 //
121 showRallyPoint( exit->getRallyPoint() );
122
123 } // end if
124 }
125 else
126 {
127 win->winHide(TRUE);
128 }
129
130 // set the text percent and bar of our timer we are displaying
131 updateContextOCLTimer( );
132
133 // set the portrait for the thing being constructed
134 setPortraitByObject( creatorObject );
135
136} // end populateUnderConstruction

140void ControlBar::updateContextOCLTimer( void )
141{
142 Object *obj = m_currentSelectedDrawable->getObject();
143
144 static const NameKeyType key_OCLUpdate = NAMEKEY( "OCLUpdate" );
145 OCLUpdate *update = (OCLUpdate*)obj->findUpdateModule( key_OCLUpdate );
146
147 UnsignedInt frames = update->getRemainingFrames();
148 UnsignedInt seconds = frames / LOGICFRAMES_PER_SECOND;
149
150 Real percent = update->getCountdownPercent();
151
152 // if the time has changed since what was last shown to the user update the text
153 if( m_displayedOCLTimerSeconds != seconds )
154 updateOCLTimerTextDisplay( seconds, percent );
155
156} // end updatecontextUnderConstruction
