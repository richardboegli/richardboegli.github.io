122void DamageInfoInput::xfer( Xfer *xfer )
123{
124
125 // version
126 XferVersion currentVersion = 3;
127 XferVersion version = currentVersion;
128 xfer->xferVersion( &version, currentVersion );
129
130 // source id
131 xfer->xferObjectID( &m_sourceID );
132
133 // source player mask
134 xfer->xferUser( &m_sourcePlayerMask, sizeof( PlayerMaskType ) );
135
136 // damage type
137 xfer->xferUser( &m_damageType, sizeof( DamageType ) );
138
139 // damage FX Override
140 if( version >= 2 )
141 xfer->xferUser( &m_damageFXOverride, sizeof( DamageType ) );
142
143 // death type
144 xfer->xferUser( &m_deathType, sizeof( DeathType ) );
145
146 // amount
147 xfer->xferReal( &m_amount );
148
149 // kill no matter what (old versions default to FALSE).
150 if( currentVersion >= 2 )
151 {
152 xfer->xferBool( &m_kill );
153 }
154
155 xfer->xferUser( &m_damageStatusType, sizeof(ObjectStatusTypes) );//It's an enum
156
157 xfer->xferCoord3D(&m_shockWaveVector);
158 xfer->xferReal( &m_shockWaveAmount );
159 xfer->xferReal( &m_shockWaveRadius );
160 xfer->xferReal( &m_shockWaveTaperOff );
161
162 if( version >= 3 )
163 {
164 AsciiString thingString = m_sourceTemplate ? m_sourceTemplate->getName() : AsciiString::TheEmptyString;
165 xfer->xferAsciiString( &thingString );
166 if( xfer->getXferMode() == XFER_LOAD )
167 {
168 m_sourceTemplate = TheThingFactory->findTemplate( thingString );
169 }
170 }
171
172} // end xfer
