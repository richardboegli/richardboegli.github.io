91bool DebugWindowDialog::CanProceed(void)
92{
93 if (mNumberOfStepsAllowed < 0) {
94 return true;
95 } else if (mNumberOfStepsAllowed == 0) {
96 if (mStepping) {
97 mStepping = false;
98 _UpdatePauseButton();
99 }
100 return false;
101 }
102
103 --mNumberOfStepsAllowed;
104 return true;
105}

118void DebugWindowDialog::AdjustVariable(const std::string& varName, const std::string& varValue)
119{
120 for (VecPairStringIt it = mVariables.begin(); it != mVariables.end(); it++) {
121 if (it->first == varName) {
122 it->second = varValue;
123 _RebuildVarsString();
124 return;
125 }
126 }
127
128 PairString newPair;
129 newPair.first = varName;
130 newPair.second = varValue;
131 mVariables.push_back(newPair);
132
133 _RebuildVarsString();
134}

150void DebugWindowDialog::_RebuildVarsString(void)
151{
152 int cursorPosBeg, cursorPosEnd;
153 ((CEdit*)GetDlgItem(IDC_Variables))->GetSel(cursorPosBeg, cursorPosEnd);
154 mVariablesString = "";
155
156 for (VecPairStringIt it = mVariables.begin(); it != mVariables.end(); it++) {
157 mVariablesString += it->first;
158 mVariablesString += " = ";
159 mVariablesString += it->second;
160 mVariablesString += "\r\n";
161 }
162
163
164
165 // Push the new string
166 mVariablesDisplayString = mVariablesString.c_str();
167 GetDlgItem(IDC_Variables)->SetWindowText(mVariablesDisplayString);
168 ((CEdit*)GetDlgItem(IDC_Variables))->GetSel(cursorPosBeg, cursorPosEnd);
169}

171void DebugWindowDialog::_RebuildMesgString(void)
172{
173 mMessagesString = "";
174
175 for (VecStringIt it = mMessages.begin(); it != mMessages.end(); it++) {
176 mMessagesString += (*it);
177 mMessagesString += "\r\n";
178 }
179
180 // Push the new string
181 mMessagesDisplayString = mMessagesString.c_str();
182 GetDlgItem(IDC_Messages)->SetWindowText(mMessagesDisplayString);
183 ((CEdit*)GetDlgItem(IDC_Messages))->SetSel(mMessagesString.length(), mMessagesString.length(), false);
184}
