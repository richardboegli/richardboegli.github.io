123//-----------------------------------------------------------------------------
128Bool CanSelectDrawable( const Drawable *draw, Bool dragSelecting )
129{
130
131 if(!draw || !draw->getObject())
132 {
133 return FALSE; // can't select
134 }
135 const Object *obj = draw->getObject();
136
137 if( obj->isEffectivelyDead() && !obj->isKindOf(KINDOF_ALWAYS_SELECTABLE))
138 {
139 //Don't select dead/dying units.
140 return FALSE;
141 }
142
143 //Added this to support attacking cargo planes without being able to select them.
144 //I added the KINDOF_FORCEATTACKABLE to them, but unsure if it's possible to select
145 //something without the KINDOF_SELECTABLE -- so doing a LATE code change. My gut
146 //says we should simply have the KINDOF_SELECTABLE check only... but best to be safe.
147 if( !obj->isKindOf( KINDOF_SELECTABLE ) && obj->isKindOf( KINDOF_FORCEATTACKABLE ) )
148 {
149 return FALSE;
150 }
151
152 // hidden objects cannot be selected
153 if( draw->isDrawableEffectivelyHidden() )
154 {
155 return FALSE; // can't select
156 }
157
158 // ignore objects obscured by the GUI
159 GameWindow *window = NULL;
160 if (TheWindowManager)
161 {
162 const Coord3D *c = draw->getPosition();
163 ICoord2D c2;
164 TheTacticalView->worldToScreen(c, &c2);
165 window = TheWindowManager->getWindowUnderCursor(c2.x, c2.y);
166 }
167
168 while (window)
169 {
170 // check to see if it or any of its parents are opaque. If so, we can't select anything.
171 if (!BitTest( window->winGetStatus(), WIN_STATUS_SEE_THRU ))
172 {
173 return FALSE;
174 }
175
176 window = window->winGetParent();
177 }
178
179 //
180 // structures cannot be selected by a drag select, you must individually pick them
181 // NOTE that this is really a convenience for the multi select context sensitive UI,
182 // later we might want to allow you to drag select buildings if only one building is
183 // actually in the selection area, but don't forget complications like holding down
184 // a key to "add" to an already existing selection list
185 //
186 // not allowing you to have multiple buildings selected drastically simplifies the
187 // user interface ... including all those context sensitive commands that we
188 // can just assume are for a single building selected.
189 //
190 if( dragSelecting && draw->isKindOf( KINDOF_STRUCTURE ) )
191 {
192 return FALSE;
193 }
194
195 // You cannot select something that has a logic override of unselectability or masked
196 if( obj->getStatusBits().testForAny( MAKE_OBJECT_STATUS_MASK2( OBJECT_STATUS_UNSELECTABLE, OBJECT_STATUS_MASKED ) ) )
197 {
198 return FALSE;
199 }
200
201 if (!obj->isSelectable())
202 {
203 return false;
204 }
205 //Now allowing the selection of everything including enemies... but only if not drag selecting.
206 //In fact the only way you can drag select is if the unit is on your team.
207 if( dragSelecting && !obj->isLocallyControlled() )
208 {
209 return FALSE;
210 }
211
212 //Now we can select anything that is selectable.
213 return TRUE;
214
215} // end canSelect

303{
304 if (CanSelectDrawable( draw, dragSelecting ))
305 {

310 {

315 }

317 }
318

320

322

325 createTeamMsg->appendObjectIDArgument( draw->getObject()->getID() );
326
327 return true; // selected
328
329 } // end if
330
331 return false; // not selected
332

369//-----------------------------------------------------------------------------
377GameMessageDisposition SelectionTranslator::translateGameMessage(const GameMessage *msg)
378{
379 GameMessageDisposition disp = KEEP_MESSAGE;
380
381 if( !TheInGameUI->getInputEnabled() )
382 {
383 //Keep the message so the other translaters (WindowXlat) can handle.
384 if( m_dragSelecting )
385 {
386 //Turn off drag select
387 m_dragSelecting = FALSE;
388 TheInGameUI->setSelecting( FALSE );
389 TheInGameUI->endAreaSelectHint(NULL);
390 TheTacticalView->setMouseLock( FALSE );
391 }
392 return KEEP_MESSAGE;
393 }
394
395 GameMessage::Type t = msg->getType();
396 switch (t)
397 {
398 case GameMessage::MSG_META_BEGIN_FORCEATTACK:
399 TheInGameUI->setForceAttackMode( true );
400 break;
401
402 case GameMessage::MSG_META_END_FORCEATTACK:
403 TheInGameUI->setForceAttackMode( false );
404 break;
405
406 //-----------------------------------------------------------------------------
407 case GameMessage::MSG_RAW_MOUSE_POSITION:
408 {
409 ICoord2D pixel;
410 pixel = msg->getArgument( 0 )->pixel;
411
412
413 // modifier appears to be unused, and the argument doesn't exist. jba.
414 //Int modifier = msg->getArgument( 1 )->integer;
415
416 if (m_leftMouseButtonIsDown)
417 {
418 ICoord2D delta;
419
420 delta.x = abs(pixel.x - m_selectFeedbackAnchor.x);
421 delta.y = abs(pixel.y - m_selectFeedbackAnchor.y);
422
423 // if mouse has moved while left button is down, begin drag selection
424 if (delta.x > TheMouse->m_dragTolerance || delta.y > TheMouse->m_dragTolerance)
425 {
426 if (m_dragSelecting == false)
427 {
428 m_dragSelecting = true;
429 TheTacticalView->setMouseLock( TRUE );
430 TheInGameUI->setSelecting( TRUE );
431 }
432 }
433
434 // create "hint" messages defining selection region under construction
435 if (m_dragSelecting)
436 {
437 // insert area selection "hint" message into stream
438 GameMessage *hintMsg = TheMessageStream->appendMessage( GameMessage::MSG_AREA_SELECTION_HINT );
439
440 // build rectangular region defined by the drag selection
441 IRegion2D pixelRegion;
442 buildRegion( &m_selectFeedbackAnchor, &pixel, &pixelRegion );
443 hintMsg->appendPixelRegionArgument( pixelRegion );
444 }
445 }
446 else //left button is not down (not drag select)
447 {
448 // insert Mouseover hint into stream for CommandTranslator and HintSpy to see.
449 GameMessage *mouseoverMessage;
450
451 //Kris: We want to show information such as the popup text on objects that are forceattackable even
452 // when we're not in force attackable mode!
453 UnsignedInt pickType = getPickTypesForContext( true /*TheInGameUI->isInForceAttackMode()*/ );
454
455 Drawable *underCursor = TheTacticalView->pickDrawable( &pixel, TheInGameUI->isInForceAttackMode(), (PickType) pickType );
456 Object *objUnderCursor = underCursor ? underCursor->getObject() : NULL;
457
458 if( objUnderCursor && (!objUnderCursor->isEffectivelyDead() || objUnderCursor->isKindOf( KINDOF_ALWAYS_SELECTABLE )) )
459 {
460 mouseoverMessage = TheMessageStream->appendMessage( GameMessage::MSG_MOUSEOVER_DRAWABLE_HINT );
461 mouseoverMessage->appendDrawableIDArgument( underCursor->getID() );
462 }
463 else// else this is a mouseover terrain
464 {
465 Coord3D position;
466
467 TheTacticalView->screenToTerrain( &pixel, &position );
468 mouseoverMessage = TheMessageStream->appendMessage( GameMessage::MSG_MOUSEOVER_LOCATION_HINT );
469 mouseoverMessage->appendLocationArgument( position );
470 }
471 }
472
473 break;
474 }
475
476 //-----------------------------------------------------------------------------
477 case GameMessage::MSG_MOUSE_LEFT_DOUBLE_CLICK:
478 {
479 Int modifiers = msg->getArgument(1)->integer;
480
481 // Pressing ctrl is disallowed for double clicking
482 if (TheInGameUI->isInForceAttackMode())
483 break;
484
485 const IRegion2D& region = msg->getArgument(0)->pixelRegion;
486
487 // Single point. If there's a unit in there, double click will select all of them.
488 if (region.height() == 0 && region.width() == 0)
489 {
490 Bool selectAcrossMap = (BitTest(modifiers, KEY_STATE_ALT) ? TRUE : FALSE);
491
492 // only allow things that are selectable. Also, we aren't allowed to
493 Drawable *picked = TheTacticalView->pickDrawable( &region.lo, FALSE, PICK_TYPE_SELECTABLE);
494
495 // If there wasn't anyone to pick, then we want to propagate this double click.
496 if (picked == NULL)
497 break;
498
499 if (!picked->isMassSelectable())
500 break;
501
502 Object *pickedObj = picked->getObject();
503
504 // We have to have an object in order to be able to do interesting double click stuff on
505 // him. Also, if it is a structure, it is already selected, so don't select all the units
506 // like him.
507 if (pickedObj == NULL || !pickedObj->isLocallyControlled())
508 break;
509
510 // Ok. The logic is a little bit weird here. What we need to do is deselect everything
511 // except for this one picked thing. Store off the old selection, pick the single clicked thing.
512 // Then if
513 DrawableList listOfSelectedDrawables;
514 if (TheInGameUI->isInPreferSelectionMode()) {
515 listOfSelectedDrawables = *TheInGameUI->getAllSelectedDrawables();
516 }
517
518 // Pick just that one guy.
519 selectSingleDrawableWithoutSound(picked);
520
521 // Yay. Either select across the screen or the world depending on selectAcrossMap
522 if (selectAcrossMap)
523 TheInGameUI->selectMatchingAcrossMap();
524 else
525 TheInGameUI->selectMatchingAcrossScreen();
526
527 // emit "picked" message
528 GameMessage *pickMsg = TheMessageStream->appendMessage( GameMessage::MSG_AREA_SELECTION );
529 pickMsg->appendDrawableIDArgument( picked->getID() );
530
531 if (TheInGameUI->isInPreferSelectionMode() && !listOfSelectedDrawables.empty()) {
532 GameMessage *selectMore = TheMessageStream->appendMessage( GameMessage::MSG_CREATE_SELECTED_GROUP_NO_SOUND );
533 selectMore->appendBooleanArgument(FALSE);
534 for (DrawableListIt it = listOfSelectedDrawables.begin(); it != listOfSelectedDrawables.end(); ++it) {
535 Drawable *draw = *it;
536 if (draw && draw->isSelectable()) {
537 TheInGameUI->selectDrawable(draw);
538 selectMore->appendObjectIDArgument(draw->getObject()->getID());
539 }
540 }
541 }
542
543 disp = DESTROY_MESSAGE;
544 }
545 break;
546 }
547
548 //-----------------------------------------------------------------------------
549 case GameMessage::MSG_MOUSEOVER_DRAWABLE_HINT:
550 {
551 if (TheInGameUI->isScrolling()) {
552 // dont show this now.
553 break;
554 }
555
556 DrawableID id = msg->getArgument(0)->drawableID;
557 Drawable *draw = TheGameClient->findDrawableByID(id);
558 if (!draw) {
559 break;
560 }
561
562 GameMessage::Type msgType = TheGameClient->evaluateContextCommand(draw, draw->getPosition(), CommandTranslator::EVALUATE_ONLY);
563 if( msgType == GameMessage::MSG_INVALID )
564 {
565 TheInGameUI->createMouseoverHint(msg); // this sets the cursor
566 disp = DESTROY_MESSAGE;
567 const CommandButton *command = TheInGameUI->getGUICommand();
568
569 Bool ignoreCommand = FALSE;
570 if( command )
571 {
572 if( command->getCommandType() == GUI_COMMAND_ATTACK_MOVE ||
573 command->getCommandType() == GUI_COMMAND_GUARD ||
574 command->getCommandType() == GUI_COMMAND_GUARD_WITHOUT_PURSUIT ||
575 command->getCommandType() == GUI_COMMAND_GUARD_FLYING_UNITS_ONLY )
576 {
577 //These GUI commands can take care of themselves -- don't let
578 //the selection translator meddle.
579 ignoreCommand = TRUE;
580 }
581 }
582 if( !ignoreCommand && !draw->getTemplate()->isKindOf( KINDOF_SHRUBBERY ) )
583 {
584 if( CanSelectDrawable( draw, FALSE ) )
585 {
586 TheMouse->setCursor(Mouse::SELECTING);
587 }
588 else
589 {
590 TheMouse->setCursor( Mouse::ARROW );
591 }
592 }
593 }
594
595 break;
596 }
597
598 //-----------------------------------------------------------------------------
599 case GameMessage::MSG_MOUSE_LEFT_CLICK:
600 {
601 // If the quit menu is visible, we need to not process left clicks through the selection translator.
602 if (TheInGameUI->isQuitMenuVisible())
603 {
604 disp = DESTROY_MESSAGE;
605 break;
606 }
607
608 // Basically, we need to first determine if there are any drawables in the region of interest.
609 // If there aren't then this click should move forward.
610 IRegion2D selectionRegion = msg->getArgument(0)->pixelRegion;
611 Bool isPoint = (selectionRegion.height() == 0 && selectionRegion.width() == 0);
612
613 DrawableList drawablesThatWillSelect;
614 PickDrawableStruct pds;
615 pds.drawableListToFill = &drawablesThatWillSelect;
616 TheTacticalView->iterateDrawablesInRegion(&selectionRegion, addDrawableToList, &pds);
617
618 if (drawablesThatWillSelect.empty())
619 {
620 break;
621 }
622
623 // if there were drawables in the region, then we should determine if there is a context
624 // sensitive command that should take place. If there is, then this isn't a selection thing
625 const DrawableList *currentList = TheInGameUI->getAllSelectedDrawables();
626 if (!currentlyLookingForSelection())
627 {
628 break;
629 }
630
631 SelectionInfo si;
632 if (contextCommandForNewSelection(currentList, &drawablesThatWillSelect, &si, isPoint))
633 {
634 break;
635 }
636
637 // There isn't a context command, so this is a selection thing. Now, based on the keys,
638 // determine whether or not we should create a new group, or append these guys to our existing
639 // group.
640
641 Bool addToGroup = TheInGameUI->isInPreferSelectionMode();
642
643 if (si.currentCountEnemies > 0 ||
644 si.currentCountCivilians > 0 ||
645 si.currentCountFriends > 0 ||
646 si.currentCountMineBuildings > 0)
647 {
648 // force a new group creation
649 addToGroup = FALSE;
650 }
651
652 // If there are any of my units, then select those.
653 if (si.newCountMine > 0)
654 {
655 si.selectMine = TRUE;
656
657 // EXACTLY ONE CLICKED OR DRAGGED BUILDING
658 if ( si.newCountMineBuildings == 1 && si.newCountMine == 1 )
659 {
660 addToGroup = FALSE;
661 si.selectMineBuildings = TRUE;
662 }
663 else if ( si.newCountMineBuildings > 0 )
664 { // What we are after here is to allow the drag select to get the building,
665 // if the other things in the list are going to be ignored anyway
666 // so we find out whether the other things are not selectible
667 // this came up with the new AmericaBuildingFireBase, which shows its contained
668 // but does not let you select them. The selection is propagated to the container
669 // in new code in SelectionInfo.cpp, in the static addDrawableToList();
670 // -Mark Lorenzen, 6/12/03
671 Bool onlyTheOneBuildingIsSelectableAnyway = TRUE;
672 DrawableID buildingID = INVALID_DRAWABLE_ID;
673 for (DrawableListIt it = drawablesThatWillSelect.begin(); it != drawablesThatWillSelect.end(); ++it)
674 {
675 const Drawable *d = *it;
676 if ( d->isKindOf( KINDOF_STRUCTURE ) )
677 {// make sure there is really only the one building in the list, as it may be multiply listed
678
679 if ( buildingID == INVALID_DRAWABLE_ID ) // this is the first building
680 buildingID = d->getID();
681 else if ( buildingID != d->getID() )//oops, more than one building!
682 onlyTheOneBuildingIsSelectableAnyway = FALSE;
683 }
684 else if ( d->isSelectable() )
685 onlyTheOneBuildingIsSelectableAnyway = FALSE;
686
687 if ( ! onlyTheOneBuildingIsSelectableAnyway )
688 break;
689 }
690 if ( onlyTheOneBuildingIsSelectableAnyway )
691 {
692 addToGroup = FALSE;
693 si.selectMineBuildings = TRUE;
694 }
695 }
696
697 }
698 else if (si.newCountEnemies > 0 && si.newCountCivilians > 0 && si.newCountFriends > 0)
699 {
700 // No go here
701 break;
702 }
703 else if (si.newCountEnemies == 1)
704 {
705 addToGroup = FALSE;
706 si.selectEnemies = TRUE;
707 }
708 else if (si.newCountCivilians == 1)
709 {
710 addToGroup = FALSE;
711 si.selectCivilians = TRUE;
712 }
713 else if (si.newCountFriends == 1)
714 {
715 addToGroup = FALSE;
716 si.selectFriends = TRUE;
717 }
718
719 // If we're not going to select anything, just bail now.
720 if (!(si.selectMine || si.selectEnemies || si.selectCivilians || si.selectFriends))
721 {
722 break;
723 }
724
725 // If we've made it here, its time to do some selecting.
726 disp = DESTROY_MESSAGE;
727
728 // Whenever we manually select something, reset the last selected group.
729 m_lastGroupSelGroup = -1;
730
731 if (TheInGameUI->isInPreferSelectionMode() && isPoint && areAllSelected(drawablesThatWillSelect))
732 {
733 // If this was a point, shift was pressed and we already have that unit selected, then we
734 // need to deselect those units.
735 GameMessage *newMsg = TheMessageStream->appendMessage(GameMessage::MSG_REMOVE_FROM_SELECTED_GROUP);
736 Drawable *draw = NULL;
737 DrawableListIt it;
738 for (it = drawablesThatWillSelect.begin(); it != drawablesThatWillSelect.end(); ++it)
739 {
740 draw = *it;
741 if (!draw)
742 {
743 continue;
744 }
745
746 Object *objToDeselect = draw->getObject();
747 if (!objToDeselect)
748 {
749 continue;
750 }
751
752 newMsg->appendObjectIDArgument(objToDeselect->getID());
753 TheInGameUI->deselectDrawable(draw);
754 }
755 }
756 else
757 {
758 if (!addToGroup)
759 {
760 deselectAll();
761 }
762
763 GameMessage *newMsg = TheMessageStream->appendMessage(GameMessage::MSG_CREATE_SELECTED_GROUP);
764 newMsg->appendBooleanArgument(!addToGroup);
765
766 Player *localPlayer = ThePlayerList->getLocalPlayer();
767
768 Int newDrawablesSelected = 0;
769 Drawable *draw = NULL;
770 DrawableListIt it;
771 for (it = drawablesThatWillSelect.begin(); it != drawablesThatWillSelect.end(); ++it)
772 {
773 draw = *it;
774 if (!draw)
775 {
776 continue;
777 }
778
779 Object *obj = draw->getObject();
780 if (!obj)
781 {
782 continue;
783 }
784
785 if (obj && obj->getContainedBy() != NULL)
786 {
787 // we're contained, and so we shouldn't be selectable.
788 continue;
