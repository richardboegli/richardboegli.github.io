57void main(int argc, char **argv)
58{
59 std::string inFile = "";
60 std::string outFile = "";
61 CompressionType compressType = CompressionManager::getPreferredCompression();
62
63 for (int i=1; i<argc; ++i)
64 {
65 if ( !stricmp(argv[i], "-help") )
66 {
67 dumpHelp(argv[0]);
68 exit(0);
69 }
70
71 if ( !strcmp(argv[i], "-in") )
72 {
73 ++i;
74 if (i<argc)
75 {
76 inFile = argv[i];
77 }
78 }
79
80 if ( !strcmp(argv[i], "-out") )
81 {
82 ++i;
83 if (i<argc)
84 {
85 outFile = argv[i];
86 }
87 }
88
89 if ( !strcmp(argv[i], "-type") )
90 {
91 ++i;
92 if (i<argc)
93 {
94 for (int j=COMPRESSION_MIN; j<=COMPRESSION_MAX; ++j)
95 {
96 if ( !stricmp(CompressionManager::getCompressionNameByType((CompressionType)j), argv[i]) )
97 {
98 compressType = (CompressionType)j;
99 break;
100 }
101 }
102 }
103 }
104 }
105
106 if (inFile.empty())
107 {
108 dumpHelp(argv[0]);
109 exit(0);
110 }
111
112 DEBUG_LOG(("IN:'%s' OUT:'%s' Compression:'%s'\n",
113 inFile.c_str(), outFile.c_str(), CompressionManager::getCompressionNameByType(compressType)));
114
115 if (outFile.empty())
116 {
117 // just check compression
118 FILE *fp = fopen(inFile.c_str(), "rb");
119 if (!fp)
120 {
121 DEBUG_LOG(("Cannot open '%s'\n", inFile.c_str()));
122 return;
123 }
124 fseek(fp, 0, SEEK_END);
125 int size = ftell(fp);
126 fseek(fp, 0, SEEK_SET);
127
128 char data[8];
129 int numRead = fread(data, 1, 8, fp);
130
131 if (numRead != 8)
132 {
133 DEBUG_LOG(("Cannot read header from '%s'\n", inFile.c_str()));
134 return;
135 }
136
137 CompressionType usedType = CompressionManager::getCompressionType(data, 8);
138 if (usedType == COMPRESSION_NONE)
139 {
140 DEBUG_LOG(("No compression on '%s'\n", inFile.c_str()));
141 return;
142 }
143
144 int uncompressedSize = CompressionManager::getUncompressedSize(data, 8);
145
146 DEBUG_LOG(("'%s' is compressed using %s, from %d to %d bytes, %g%% of its original size\n",
147 inFile.c_str(), CompressionManager::getCompressionNameByType(usedType),
148 uncompressedSize, size, size/(double)(uncompressedSize+0.1)*100.0));
149
150 return;
151 }
152
153 // compress file
154}
