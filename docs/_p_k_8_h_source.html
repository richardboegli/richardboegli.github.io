49/*
50** This class holds a public or private key used in Public Key Cryptography. It also serves
51** as the conduit for encrypting/decrypting data using that key. Cryptography, using this
52** method, has a couple of characteristics that affect how it is used. One, the process of
53** encrypting/decrypting is very slow. This limits the effective quantity of data that can
54** be processed. Two, the ciphertext is larger than the plaintext. This property generally
55** limits its use to streaming data as opposed to random access data. The data is processed
56** in blocks. The size of the ciphertext and plaintext blocks can be determined only from
57** the key itself.
58**
59** A reasonable use of this technology would be to encrypt only critical data such as the
60** password for a fast general purpose cryptographic algorithm.
61*/
62class PKey
63{
64 public:
65 PKey(void) : Modulus(0), Exponent(0), BitPrecision(0) {}
66 PKey(void const * exponent, void const * modulus); // DER initialization.
67
68 int Encrypt(void const * source, int slen, void * dest) const;
69 int Decrypt(void const * source, int slen, void * dest) const;
70
71 static void Generate(Straw & random, int bits, PKey & fastkey, PKey & slowkey);
72
73 int Plain_Block_Size(void) const {return((BitPrecision-1)/8);}
74 int Crypt_Block_Size(void) const {return(Plain_Block_Size()+1);}
75 int Block_Count(int plaintext_length) const {return((((plaintext_length-1)/Plain_Block_Size())+1));}
76
77 int Encode_Modulus(void * buffer) const;
78 int Encode_Exponent(void * buffer) const;
79
80 void Decode_Modulus(void * buffer);
81 void Decode_Exponent(void * buffer);
82
83 static long Fast_Exponent(void) {return(65537L);}

96};
