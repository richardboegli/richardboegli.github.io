63void CrateSystem::reset( void )
64{
65 // clean up overrides
66 std::vector<CrateTemplate *>::iterator it;
67 for( it = m_crateTemplateVector.begin(); it != m_crateTemplateVector.end(); )
68 {
69 CrateTemplate *currentTemplate = *it;
70 if( currentTemplate )
71 {
72 Overridable *tempCrateTemplate = currentTemplate->deleteOverrides();
73 if (!tempCrateTemplate)
74 {
75 // base dude was an override - kill it from the vector
76 it = m_crateTemplateVector.erase(it);
77 }
78 else
79 {
80 ++it;
81 }
82 }
83 else
84 {
85 it = m_crateTemplateVector.erase(it);
86 }
87 }
88}

90void CrateSystem::parseCrateTemplateDefinition(INI* ini)
91{
92 AsciiString name;
93
94 // read the crateTemplate name
95 const char* c = ini->getNextToken();
96 name.set(c);
97
98 CrateTemplate *crateTemplate = TheCrateSystem->friend_findCrateTemplate(name);
99 if (crateTemplate == NULL) {
100 crateTemplate = TheCrateSystem->newCrateTemplate(name);
101
102 if (ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES) {
103 crateTemplate->markAsOverride();
104 }
105 } else if( ini->getLoadType() != INI_LOAD_CREATE_OVERRIDES ) {
106 DEBUG_CRASH(( "[LINE: %d in '%s'] Duplicate crate %s found!", ini->getLineNum(), ini->getFilename().str(), name.str() ));
107 } else {
108 crateTemplate = TheCrateSystem->newCrateTemplateOverride(crateTemplate);
109 }
110
111 // parse the ini weapon definition
112 ini->initFromINI(crateTemplate, crateTemplate->getFieldParse());
113}

115CrateTemplate *CrateSystem::newCrateTemplate( AsciiString name )
116{
117 // sanity
118 if(name.isEmpty())
119 return NULL;
120
121 // allocate a new weapon
122 CrateTemplate *ct = newInstance(CrateTemplate);
123
124 // if the default template is present, get it and copy over any data to the new template
125 const CrateTemplate *defaultCT = findCrateTemplate(AsciiString("DefaultCrate"));
126 if(defaultCT)
127 {
128 *ct = *defaultCT;
129 }
130
131 ct->setName( name );
132 m_crateTemplateVector.push_back(ct);
133
134 return ct;
135}

137CrateTemplate *CrateSystem::newCrateTemplateOverride( CrateTemplate *crateToOverride )
138{
139 if (!crateToOverride) {
140 return NULL;
141 }
142
143 CrateTemplate *newOverride = newInstance(CrateTemplate);
144 *newOverride = *crateToOverride;
145
146 newOverride->markAsOverride();
147
148 crateToOverride->setNextOverride(newOverride);
149 return newOverride;
150}

152const CrateTemplate *CrateSystem::findCrateTemplate(AsciiString name) const
153{
154 // search weapon list for name
155 for (Int i = 0; i < m_crateTemplateVector.size(); i++)
156 if(m_crateTemplateVector[i]->getName() == name) {
157 CrateTemplateOverride overridable(m_crateTemplateVector[i]);
158 return overridable;
159 }
160
161
162 return NULL;
163}

169 if(m_crateTemplateVector[i]->getName() == name) {
170 CrateTemplateOverride overridable(m_crateTemplateVector[i]);
171 return const_cast<CrateTemplate*>((const CrateTemplate *)overridable);
172 }
173 return NULL;
174}

211void CrateTemplate::parseCrateCreationEntry( INI* ini, void *instance, void *, const void* )
212{
213 CrateTemplate *self = (CrateTemplate *)instance;
214
215 const char *token = ini->getNextToken();
216 AsciiString crateName = token;
217
218 token = ini->getNextToken();
219 Real crateValue;
220 if (sscanf( token, "%f", &crateValue ) != 1)
221 throw INI_INVALID_DATA;
222
223 crateCreationEntry newEntry;
224 newEntry.crateName = crateName;
225 newEntry.crateChance = crateValue;
226
227 self->m_possibleCrates.push_back( newEntry );
228}
