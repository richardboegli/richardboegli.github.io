85void ProjectileStreamUpdate::addProjectile( ObjectID sourceID, ObjectID newID, ObjectID victimID, const Coord3D *victimPos )
86{
87 DEBUG_ASSERTCRASH( m_owningObject == INVALID_ID || m_owningObject == sourceID, ("Two objects are trying to use the same Projectile stream.") );//Don't cross the streams!
88 if( m_owningObject == INVALID_ID )
89 m_owningObject = sourceID;
90
91 // Keep track of target, and insert a hole if the target has changed
92 if( victimID != INVALID_ID )
93 {
94 // Object shot
95 if( victimID != m_targetObject )
96 {
97 // Changed targets, insert a hole to break the stream
98 m_projectileIDs[ m_nextFreeIndex ] = INVALID_ID;
99 m_nextFreeIndex = (m_nextFreeIndex + 1) % MAX_PROJECTILE_STREAM;
100
101 // And mark this as our new target
102 m_targetObject = victimID;
103 }
104
105 // Clear position so we know we are an object shot
106 m_targetPosition.zero();
107 }
108 else if( victimPos != NULL )
109 {
110 if( ! (m_targetPosition == (*victimPos)) )
111 {
112 // New position, so insert hole
113 m_projectileIDs[ m_nextFreeIndex ] = INVALID_ID;
114 m_nextFreeIndex = (m_nextFreeIndex + 1) % MAX_PROJECTILE_STREAM;
115
116 // And mark this as our new target
117 m_targetPosition = (*victimPos);
118 }
119
120 // Clear object so we know we are a position shot
121 m_targetObject = INVALID_ID;
122 }
123 else
124 {
125 DEBUG_CRASH(("A projectile stream was fired at neither an object nor a position. Probably bad."));
126 }
127
128 // Keep track of the id in a circular array
129 m_projectileIDs[ m_nextFreeIndex ] = newID;
130 m_nextFreeIndex = (m_nextFreeIndex + 1) % MAX_PROJECTILE_STREAM;
131 DEBUG_ASSERTCRASH( m_nextFreeIndex != m_firstValidIndex, ("Need to increase the allowed number of simultaneous particles in ProjectileStreamUpdate.") );
132}

134void ProjectileStreamUpdate::cullFrontOfList()
135{
136 while( (m_firstValidIndex != m_nextFreeIndex) && (TheGameLogic->findObjectByID( m_projectileIDs[m_firstValidIndex] ) == NULL) )
137 {
138 // Chew off the front if they are gone. Don't chew on the middle, as bad ones there are just a break in the chain
139 m_firstValidIndex = (m_firstValidIndex + 1) % MAX_PROJECTILE_STREAM;
140 }
141}

143Bool ProjectileStreamUpdate::considerDying()
144{
145 if( m_firstValidIndex == m_nextFreeIndex && m_owningObject != INVALID_ID )
146 {
147 //If I have no projectiles to watch, and my master is dead, then yes, I want to die
148 if( TheGameLogic->findObjectByID(m_owningObject) == NULL )
149 return TRUE;
150 }
151
152 return FALSE;
153}

155void ProjectileStreamUpdate::getAllPoints( Vector3 *points, Int *count )
156{
157 Int pointCount = 0;
158 Int pointIndex = m_firstValidIndex;
159
160
161 Object *obj = TheGameLogic->findObjectByID(m_owningObject);
162
163
164 while( pointIndex != m_nextFreeIndex )
165 {
166 // Go through the array I think of as good. Holes in the middle get 0,0,0. I write
167 // to pointCount because I am unrolling a circular array into the same sized flat one
168 // since I am writing anyway.
169 Object *projectile = TheGameLogic->findObjectByID( m_projectileIDs[pointIndex] );
170
171 if( projectile )
172 {
173 Coord3D thisPoint = *projectile->getPosition();
174 points[pointCount].X = thisPoint.x;
175 points[pointCount].Y = thisPoint.y;
176 points[pointCount].Z = thisPoint.z;
177
178
179 if ( obj && obj->isKindOf( KINDOF_VEHICLE ) ) // this makes the stream skim along my roof, if I have a roof
180 {
181 const Coord3D *pos = obj->getPosition();
182 Real myTop = obj->getGeometryInfo().getMaxHeightAbovePosition() + pos->z + 0.5f;
183 Coord3D delta;
184 delta.x = pos->x - points[pointCount].X;
185 delta.y = pos->y - points[pointCount].Y;
186 delta.z = 0.0f;
187 if( delta.length() <= obj->getGeometryInfo().getMajorRadius() * 1.5f )
188 points[pointCount].Z = MAX( points[pointCount].Z, myTop );
189 }
190
191
192
193
194 }
195 else
196 {
197 points[pointCount].X = 0;
198 points[pointCount].Y = 0;
199 points[pointCount].Z = 0;
200 }
201
202 pointIndex = (pointIndex + 1) % MAX_PROJECTILE_STREAM;
203 pointCount++;
204 }
205
206 *count = pointCount;
207}

233void ProjectileStreamUpdate::xfer( Xfer *xfer )
234{
235
236 // version
237 XferVersion currentVersion = 2;
238 XferVersion version = currentVersion;
239 xfer->xferVersion( &version, currentVersion );
240
241 // extend base class
242 UpdateModule::xfer( xfer );
243
244 // projectile ids
245 xfer->xferUser( m_projectileIDs, sizeof( ObjectID ) * MAX_PROJECTILE_STREAM );
246
247 // next free index
248 xfer->xferInt( &m_nextFreeIndex );
249
250 // first valid index
251 xfer->xferInt( &m_firstValidIndex );
252
253 // owning object
254 xfer->xferObjectID( &m_owningObject );
255
256 if( version >= 2 )
257 {
258 xfer->xferObjectID( &m_targetObject );
259 xfer->xferCoord3D( &m_targetPosition );
260 }
261
262} // end xfer
