265Script *ScriptDialog::getCurScript(void)
266{
267 if (m_curSelection.m_objType == ListType::SCRIPT_IN_PLAYER_TYPE || m_curSelection.m_objType == ListType::SCRIPT_IN_GROUP_TYPE) {
268 ScriptList *pSL = m_sides.getSideInfo(m_curSelection.m_playerIndex)->getScriptList();
269 if (pSL) {
270 Script *pScr=NULL;
271 if (m_curSelection.m_objType == ListType::SCRIPT_IN_PLAYER_TYPE) {
272 pScr = pSL->getScript();
273 } else {
274 Int groupNdx;
275 ScriptGroup *pGroup = pSL->getScriptGroup();
276 for (groupNdx = 0; pGroup; groupNdx++,pGroup=pGroup->getNext()) {
277 if (groupNdx == m_curSelection.m_groupIndex) {
278 pScr = pGroup->getScript();
279 break;
280 }
281 }
282 }
283 Int scriptNdx;
284 for (scriptNdx = 0; pScr; scriptNdx++,pScr=pScr->getNext()) {
285 if (scriptNdx == m_curSelection.m_scriptIndex) {
286 return pScr;
287 }
288 }
289 }
290 }
291 return NULL;
292}

316void ScriptDialog::updateScriptWarning(Script *pScript)
317{
318 pScript->setWarnings(false);
319 OrCondition *pOr;
320 for (pOr= pScript->getOrCondition(); pOr; pOr = pOr->getNextOrCondition()) {
321 Condition *pCondition;
322 for (pCondition = pOr->getFirstAndCondition(); pCondition; pCondition = pCondition->getNext()) {
323 pCondition->setWarnings(false);
324 Int i;
325 for (i=0; i<pCondition->getNumParameters(); i++) {
326 AsciiString warning;
327 warning = EditParameter::getWarningText(pCondition->getParameter(i), FALSE);
328 if (!warning.isEmpty()) {
329 pScript->setWarnings(true);
330 pCondition->setWarnings(true);
331 }
332 }
333 }
334 }
335 ScriptAction *pAction;
336 for (pAction = pScript->getAction(); pAction; pAction = pAction->getNext()) {
337 pAction->setWarnings(false);
338 Int i;
339 for (i=0; i<pAction->getNumParameters(); i++) {
340 AsciiString warning;
341 warning = EditParameter::getWarningText(pAction->getParameter(i), TRUE);
342 if (!warning.isEmpty()) {
343 pScript->setWarnings(true);
344 pAction->setWarnings(true);
345 }
346 }
347 }

381void ScriptDialog::updateWarnings(Bool forceUpdate)
382{
383 if (m_staticThis && m_staticThis->m_autoUpdateWarnings == false && forceUpdate == false)
384 return; //user has disabled warnings to speed up the script editor
385
386 SidesList *sidesListP = TheSidesList;
387 Int i;
388 if (m_staticThis) sidesListP = &m_staticThis->m_sides;
389 for (i=0; i<sidesListP->getNumSides(); i++) {
390 ScriptList *pSL = sidesListP->getSideInfo(i)->getScriptList();
391 Script *pScr;
392 for (pScr = pSL->getScript(); pScr; pScr=pScr->getNext()) {
393 updateScriptWarning(pScr);
394 }
395 ScriptGroup *pGroup;
396 for (pGroup = pSL->getScriptGroup(); pGroup; pGroup=pGroup->getNext()) {
397 pGroup->setWarnings(false);
398 for (pScr = pGroup->getScript(); pScr; pScr=pScr->getNext()) {
399 updateScriptWarning(pScr);
400 if (pScr->hasWarnings()) {
401 pGroup->setWarnings(true);
402 }
403 }

410void ScriptDialog::patchScriptParametersForGC(Script *pScript)
411{

416 Condition *pCondition;

418 pCondition->setWarnings(false);

420 for (i=0; i<pCondition->getNumParameters(); i++) {
421 AsciiString warning;
422 Parameter *pParm = pCondition->getParameter(i);
423 warning = EditParameter::getWarningText(pParm, FALSE);
424 if (!warning.isEmpty()) {
425 if (pParm->getParameterType() == Parameter::OBJECT_TYPE)
426 { //see if removing the GC prefix fixes this warning:
427 AsciiString uiString = pParm->getString();
428 if (uiString.isEmpty())
429 uiString = "???";

481/*Checks all script parameters for obsolete values (example: mission disk using GC_ templates)*/
482void ScriptDialog::checkParametersForGC(void)
483{
484 SidesList *sidesListP = TheSidesList;
485 Int i;
486 if (m_staticThis) sidesListP = &m_staticThis->m_sides;
487 for (i=0; i<sidesListP->getNumSides(); i++) {
488 ScriptList *pSL = sidesListP->getSideInfo(i)->getScriptList();
489 Script *pScr;
490 for (pScr = pSL->getScript(); pScr; pScr=pScr->getNext()) {
491 updateScriptWarning(pScr);
492 if (pScr->hasWarnings())
493 { //check if this is using invalid GC parameters
494 patchScriptParametersForGC(pScr);
495 }
496 }
497 ScriptGroup *pGroup;
498 for (pGroup = pSL->getScriptGroup(); pGroup; pGroup=pGroup->getNext()) {
499 pGroup->setWarnings(false);
500 for (pScr = pGroup->getScript(); pScr; pScr=pScr->getNext()) {
501 updateScriptWarning(pScr);
502 if (pScr->hasWarnings()) {
503 //check if this is using invalid GC parameters.
504 patchScriptParametersForGC(pScr);
505 if (pScr->hasWarnings()) //patching may have removed warning
506 pGroup->setWarnings(true);
507 }
508 }
509 }
510 }

638Bool ScriptDialog::updateIcons(HTREEITEM hItem)
639{
640 const ListType saveList = m_curSelection;
641 Bool warnings = false;
642 CTreeCtrl *pTree = (CTreeCtrl*)GetDlgItem(IDC_SCRIPT_TREE);
643 HTREEITEM child = pTree->GetChildItem(hItem);
644
645 while (child != NULL) {
646 ListType lt;
647 lt.IntToList(pTree->GetItemData(child));
648
650 if (lt.m_objType == ListType::PLAYER_TYPE)
651 {
652 if (updateIcons(child)) {
653 pTree->SetItemState(child, INDEXTOSTATEIMAGEMASK(3), TVIS_STATEIMAGEMASK);
654 } else {
655 pTree->SetItemState(child, INDEXTOSTATEIMAGEMASK(1), TVIS_STATEIMAGEMASK);
656 }
657 }
658
660 else if (lt.m_objType == ListType::GROUP_TYPE)
661 {
662 m_curSelection = lt;
663 if (updateIcons(child)) {
664 pTree->SetItemState(child, INDEXTOSTATEIMAGEMASK(3), TVIS_STATEIMAGEMASK);
665 warnings = true;
666 } else {
667 setIconGroup(child);
668 }
669 }
670
672 else
673 {
674 m_curSelection = lt;
675 Script *pScr = getCurScript();
676 DEBUG_ASSERTCRASH(pScr, ("Unexpected."));
677 if (pScr) {
678 if (pScr->hasWarnings()) {
679 pTree->SetItemState(child, INDEXTOSTATEIMAGEMASK(4), TVIS_STATEIMAGEMASK);
680 warnings = true;
681 } else {
682 setIconScript(child);
683 }
684 }
685 }
686
687 child = pTree->GetNextSiblingItem(child);
688 }
689 m_curSelection = saveList;
690 return warnings;
691}

838HTREEITEM ScriptDialog::findItem(ListType sel, Bool failSafe)
839{
840 CTreeCtrl *pTree = (CTreeCtrl*)GetDlgItem(IDC_SCRIPT_TREE);
841 HTREEITEM player = pTree->GetChildItem(TVI_ROOT);
842 TVITEM item;
843 while (player != NULL) {
844 ::memset(&item, 0, sizeof(item));
845 item.mask = TVIF_HANDLE|TVIF_PARAM;
846 item.hItem = player;
847 pTree->GetItem(&item);
848 ListType lt;
849 lt.IntToList(item.lParam);
850 if (lt.m_playerIndex==sel.m_playerIndex) {
851 break;
852 }
853 player = pTree->GetNextSiblingItem(player);
854 }
855 DEBUG_ASSERTCRASH(player, ("Couldn't find player."));
856 if (!player) return NULL;
857 if (sel.m_objType == ListType::PLAYER_TYPE) {
858 return player;
859 }
860
861 HTREEITEM group;
862 if (sel.m_objType == ListType::SCRIPT_IN_PLAYER_TYPE) {
863 group = player; // top level scripts are grouped under player.
864 } else {
865 group = pTree->GetChildItem(player);
866 while (group != NULL) {
867 ::memset(&item, 0, sizeof(item));
868 item.mask = TVIF_HANDLE|TVIF_PARAM;
869 item.hItem = group;
870 pTree->GetItem(&item);
871 ListType lt;
872 lt.IntToList(item.lParam);
873 if (lt.m_groupIndex==sel.m_groupIndex) {
874 break;
875 }
876 DEBUG_ASSERTCRASH(lt.m_objType == ListType::GROUP_TYPE, ("Not group"));
877 group = pTree->GetNextSiblingItem(group);
878 }
879 }
880 DEBUG_ASSERTCRASH(group, ("Couldn't find group."));
881 if (!group) return NULL;
882 if (sel.m_objType == ListType::GROUP_TYPE) {
883 return group;
884 }
885
886 HTREEITEM script;
887 for (script = pTree->GetChildItem(group); script != NULL; script = pTree->GetNextSiblingItem(script)) {
888 ::memset(&item, 0, sizeof(item));
889 item.mask = TVIF_HANDLE|TVIF_PARAM;
890 item.hItem = script;
891 pTree->GetItem(&item);
892 ListType lt;
893 lt.IntToList(item.lParam);
894 if (sel.m_objType == ListType::SCRIPT_IN_PLAYER_TYPE && lt.m_objType == ListType::GROUP_TYPE) {
895 continue;
896 }
897 DEBUG_ASSERTCRASH(lt.m_objType == ListType::SCRIPT_IN_PLAYER_TYPE || lt.m_objType == ListType::SCRIPT_IN_GROUP_TYPE, ("Not script"));
898 if (lt.m_scriptIndex==sel.m_scriptIndex) {
899 break;
900 }
901 }
902
903 if (script || !failSafe) {
904 DEBUG_ASSERTCRASH(script, ("Couldn't find script."));
905 return script;
906 }
907
908 // at least select the group if possible.
909 return group;
910}
