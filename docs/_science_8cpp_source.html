125//-----------------------------------------------------------------------------
126void ScienceInfo::addRootSciences(ScienceVec& v) const
127{
128 if (m_prereqSciences.empty())
129 {
130 // we're a root. add ourselves.
131 if (std::find(v.begin(), v.end(), m_science) == v.end())
132 v.push_back(m_science);
133 }
134 else
135 {
136 // we're not a root. add the roots of all our prereqs.
137 for (ScienceVec::const_iterator it = m_prereqSciences.begin(); it != m_prereqSciences.end(); ++it)
138 {
139 const ScienceInfo* si = TheScienceStore->findScienceInfo(*it);
140 if (si)
141 si->addRootSciences(v);
142 }
143 }
144}

161//-----------------------------------------------------------------------------
162/*static*/ void ScienceStore::friend_parseScienceDefinition( INI* ini )
163{
164 const char* c = ini->getNextToken();
165 NameKeyType nkt = NAMEKEY(c);
166 ScienceType st = (ScienceType)nkt;
167
168 if (TheScienceStore)
169 {
170
171 static const FieldParse myFieldParse[] =
172 {
173 { "PrerequisiteSciences", INI::parseScienceVector, NULL, offsetof( ScienceInfo, m_prereqSciences ) },
174 { "SciencePurchasePointCost", INI::parseInt, NULL, offsetof( ScienceInfo, m_sciencePurchasePointCost ) },
175 { "IsGrantable", INI::parseBool, NULL, offsetof( ScienceInfo, m_grantable ) },
176 { "DisplayName", INI::parseAndTranslateLabel, NULL, offsetof( ScienceInfo, m_name) },
177 { "Description", INI::parseAndTranslateLabel, NULL, offsetof( ScienceInfo, m_description) },
178 { 0, 0, 0, 0 }
179 };
180
181 ScienceInfo* info = NULL;
182
183 // see if the science already exists. (can't use findScienceInfo() since it is const and should remain so.)
184 for (ScienceInfoVec::iterator it = TheScienceStore->m_sciences.begin(); it != TheScienceStore->m_sciences.end(); ++it)
185 {
186 // note that we don't use getFinalOverride here. this is correct and as-desired.
187 if ((*it)->m_science == st)
188 {
189 info = *it;
190 break;
191 }
192 }
193
194 if (ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES)
195 {
196 ScienceInfo* newInfo = newInstance(ScienceInfo);
197
198 if (info == NULL)
199 {
200 // only add if it's not overriding an existing one.
201 info = newInfo;
202 info->markAsOverride(); // yep, so we will get cleared on reset()
203 TheScienceStore->m_sciences.push_back(info);
204 }
205 else
206 {
207 // copy data from final override to 'newInfo' as a set of initial default values
208 info = (ScienceInfo*)(info->friend_getFinalOverride());
209
210 *newInfo = *info;
211 info->setNextOverride(newInfo);
212 newInfo->markAsOverride(); // must do AFTER the copy
213
214 // use the newly created override for us to set values with etc
215 info = newInfo;
216 //TheScienceStore->m_sciences.push_back(info); // NO, BAD, WRONG -- don't add in this case.
217 }
218 }
219 else
220 {
221 if (info != NULL)
222 {
223 DEBUG_CRASH(("duplicate science %s!\n",c));
224 throw INI_INVALID_DATA;
225 }
226 info = newInstance(ScienceInfo);
227 TheScienceStore->m_sciences.push_back(info);
228 }
229
230 ini->initFromINI(info, myFieldParse);
231 info->m_science = st;
232 info->addRootSciences(info->m_rootSciences);
233 }
234}

322//-----------------------------------------------------------------------------
325void ScienceStore::getPurchasableSciences(const Player* player, ScienceVec& purchasable, ScienceVec& potentiallyPurchasable) const
326{
327 purchasable.clear();
328 potentiallyPurchasable.clear();
329 for (ScienceInfoVec::const_iterator it = m_sciences.begin(); it != m_sciences.end(); ++it)
330 {
331 const ScienceInfo* si = (const ScienceInfo*)(*it)->getFinalOverride();
332
333 if (si->m_sciencePurchasePointCost == 0)
334 {
335 // 0 means "cannot be purchased"
336 continue;
337 }
338
339 if (player->hasScience(si->m_science))
340 {
341 continue;
342 }
343
344 if (playerHasPrereqsForScience(player, si->m_science))
345 {
346 purchasable.push_back(si->m_science);
347 }
348 else if (playerHasRootPrereqsForScience(player, si->m_science))
349 {
350 potentiallyPurchasable.push_back(si->m_science);
351 }
352 }
353}
