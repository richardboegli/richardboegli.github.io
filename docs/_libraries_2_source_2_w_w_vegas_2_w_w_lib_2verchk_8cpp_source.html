62bool GetVersionInfo(char* filename, VS_FIXEDFILEINFO* fileInfo)
63 {
64 if (filename == NULL || fileInfo == NULL)
65 {
66 return false;
67 }
68
69 // Get version information from the application
70 DWORD verHandle;
71 DWORD verInfoSize = GetFileVersionInfoSize(filename, &verHandle);
72
73 if (verInfoSize)
74 {
75 // If we were able to get the information, process it:
76 HANDLE memHandle = GlobalAlloc(GMEM_MOVEABLE, verInfoSize);
77
78 if (memHandle)
79 {
80 LPVOID buffer = GlobalLock(memHandle);
81
82 if (buffer)
83 {
84 BOOL success = GetFileVersionInfo(filename, verHandle, verInfoSize, buffer);
85
86 if (success)
87 {
88 VS_FIXEDFILEINFO* data;
89 UINT dataSize = 0;
90 success = VerQueryValue(buffer, "\\", (LPVOID*)&data, &dataSize);
91
92 if (success && (dataSize == sizeof(VS_FIXEDFILEINFO)))
93 {
94 memcpy(fileInfo, data, sizeof(VS_FIXEDFILEINFO));
95 return true;
96 }
97 }
98
99 GlobalUnlock(memHandle);
100 }
101
102 GlobalFree(memHandle);
103 }
104 }
105
106 return false;
107 }

141bool
142Get_Image_File_Header (const char *filename, IMAGE_FILE_HEADER *file_header)
143{
144 bool retval = false;
145
146 //
147 // Attempt to open the file
148 //
149 FileClass *file=_TheFileFactory->Get_File(filename);
150
151 if (file && file->Open ()) {
152
153 //
154 // Read the dos header (all PE exectuable files begin with this)
155 //
156 IMAGE_DOS_HEADER dos_header;
157 if (file->Read (&dos_header, sizeof (dos_header)) == sizeof (dos_header)) {
158
159 //
160 // Determine the index where the image header resides
161 //
162 int file_header_offset = dos_header.e_lfanew + sizeof (DWORD);
163 file->Seek (file_header_offset, SEEK_SET);
164
165 //
166 // Read the image header from the file
167 //
168 int size = sizeof (IMAGE_FILE_HEADER);
169 if (file->Read (file_header, size) == size) {
170 retval = true;
171 }
172 }
173 }
174
175 _TheFileFactory->Return_File(file);
176 file=NULL;
177
178 return retval;
179}

188Get_Image_File_Header (HINSTANCE app_instance, IMAGE_FILE_HEADER *file_header)
189{
190 bool retval = false;
191
192 //
193 // Read the dos header (all PE exectuable files begin with this)
194 //
195 IMAGE_DOS_HEADER *dos_header = (IMAGE_DOS_HEADER *)app_instance;
196 if (dos_header != NULL) {
197
198 //
199 // Determine the offset where the image header resides
200 //
201 int image_header_offset = dos_header->e_lfanew + sizeof (DWORD);
202
203 //
204 // Copy the file header into the provided structure
205 //
206 ::memcpy ( file_header,
207 (((char *)dos_header) + image_header_offset),
208 sizeof (IMAGE_FILE_HEADER));
209 retval = true;
210 }
211
212
213 return retval;
214}

218//
219// Compare_EXE_Version
220//
221// Used to compare 2 versions of an executable, the currently executing
222// exe and a version saved to disk. These exe's do not need to have
223// a version resource.
224//
225// The return value is the same as strcmp, < 0 if the current process is
226// older, 0 if they are the same, and > 0 if the current process is newer.
227//
229int
230Compare_EXE_Version (int app_instance, const char *filename)
231{
232 int retval = 0;
233
234 //
235 // Get the image header for both executables
236 //
237 IMAGE_FILE_HEADER header1 = { 0 };
238 IMAGE_FILE_HEADER header2 = { 0 };
239 if ( ::Get_Image_File_Header ((HINSTANCE)app_instance, &header1) &&
240 ::Get_Image_File_Header (filename, &header2))
241 {
242 retval = int(header1.TimeDateStamp - header2.TimeDateStamp);
243 }
244
245 return retval;
246}
