52static GameWindow *gogoExMessageBox(Int x, Int y, Int width, Int height, UnsignedShort buttonFlags,
53 UnicodeString titleString, UnicodeString bodyString, void *userData,
54 MessageBoxFunc yesCallback,
55 MessageBoxFunc noCallback,
56 MessageBoxFunc okCallback,
57 MessageBoxFunc cancelCallback )
58{
59 // first check to make sure we have some buttons to display
60 if(buttonFlags == 0 )
61 {
62 return NULL;
63 }
64
65 GameWindow *parent = TheWindowManager->winCreateFromScript( AsciiString("Menus/MessageBox.wnd") );
66 TheWindowManager->winSetModal( parent );
67 TheWindowManager->winSetFocus( NULL ); // make sure we lose focus from other windows even if we refuse focus ourselves
68 TheWindowManager->winSetFocus( parent );
69
70 // If the user wants the size to be different then the default
71 float ratioX, ratioY = 1;
72
73 if( width > 0 && height > 0 )
74 {
75 ICoord2D temp;
76 //First grab the percent increase/decrease compaired to the default size
77 parent->winGetSize( &temp.x, &temp.y);
78 ratioX = (float)width / (float)temp.x;
79 ratioY = (float)height / (float)temp.y;
80 //Set the window's new size
81 parent->winSetSize( width, height);
82
83 //Resize/reposition all the children windows based off the ratio
84 GameWindow *child;
85 for( child = parent->winGetChild(); child; child = child->winGetNext() )
86 {
87 child->winGetSize(&temp.x, &temp.y);
88 temp.x =Int(temp.x * ratioX);
89 temp.y =Int(temp.y * ratioY);
90 child->winSetSize(temp.x, temp.y);
91
92 child->winGetPosition(&temp.x, &temp.y);
93 temp.x =Int(temp.x * ratioX);
94 temp.y =Int(temp.y * ratioY);
95 child->winSetPosition(temp.x, temp.y);
96 }
97 }
98
99 // If the user wants to position the message box somewhere other then default
100 if( x >= 0 && y >= 0)
101 parent->winSetPosition(x, y);
102
103 // Reposition the buttons
104 Int buttonX[3], buttonY[3];
105
106 //In the layout, buttonOk will be in the first button position
107 NameKeyType buttonOkID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonOk" ) );
108 GameWindow *buttonOk = TheWindowManager->winGetWindowFromId(parent, buttonOkID);
109 buttonOk->winGetPosition(&buttonX[0], &buttonY[0]);
110
111 NameKeyType buttonYesID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonYes" ) );
112 GameWindow *buttonYes = TheWindowManager->winGetWindowFromId(parent, buttonYesID);
113 //buttonNo in the second position
114 NameKeyType buttonNoID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonNo" ) );
115 GameWindow *buttonNo = TheWindowManager->winGetWindowFromId(parent, buttonNoID);
116 buttonNo->winGetPosition(&buttonX[1], &buttonY[1]);
117
118 //and buttonCancel in the third
119 NameKeyType buttonCancelID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonCancel" ) );
120 GameWindow *buttonCancel = TheWindowManager->winGetWindowFromId(parent, buttonCancelID);
121 buttonCancel->winGetPosition(&buttonX[2], &buttonY[2]);
122
123 //we shouldn't have button OK and Yes on the same dialog
124 if((buttonFlags & (MSG_BOX_OK | MSG_BOX_YES)) == (MSG_BOX_OK | MSG_BOX_YES) )
125 {
126 DEBUG_ASSERTCRASH(false, ("Passed in MSG_BOX_OK and MSG_BOX_YES. Big No No."));
127 }
128
129 //Position the OK button if we have one
130 if( (buttonFlags & MSG_BOX_OK) == MSG_BOX_OK)
131 {
132 buttonOk->winSetPosition(buttonX[0], buttonY[0]);
133 buttonOk->winHide(FALSE);
134 }
135 else if( (buttonFlags & MSG_BOX_YES) == MSG_BOX_YES)
136 {
137 //Position the Yes if we have one
138 buttonYes->winSetPosition(buttonX[0], buttonY[0]);
139 buttonYes->winHide(FALSE);
140 }
141
142 if((buttonFlags & (MSG_BOX_NO | MSG_BOX_CANCEL)) == (MSG_BOX_NO | MSG_BOX_CANCEL) )
143 {
144 //If we have both the No and Cancel button, then the no should go in the middle position
145 buttonNo->winSetPosition(buttonX[1], buttonY[1]);
146 buttonCancel->winSetPosition(buttonX[2], buttonY[2]);
147 buttonNo->winHide(FALSE);
148 buttonCancel->winHide(FALSE);
149 }
150 else if( (buttonFlags & MSG_BOX_NO) == MSG_BOX_NO)
151 {
152 //if we just have the no button, then position it in the right most spot
153 buttonNo->winSetPosition(buttonX[2], buttonY[2]);
154 buttonNo->winHide(FALSE);
155 }
156 else if( (buttonFlags & MSG_BOX_CANCEL) == MSG_BOX_CANCEL)
157 {
158 //else if we just have the Cancel button, well, it should always go in the right spot
159 buttonCancel->winSetPosition(buttonX[2], buttonY[2]);
160 buttonCancel->winHide(FALSE);
161 }
162
163 // Fill the text into the text boxes
164 NameKeyType staticTextTitleID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:StaticTextTitle" ) );
165 GameWindow *staticTextTitle = TheWindowManager->winGetWindowFromId(parent, staticTextTitleID);
166 GadgetStaticTextSetText(staticTextTitle,titleString);
167
168 NameKeyType staticTextMessageID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:StaticTextMessage" ) );
169 GameWindow *staticTextMessage = TheWindowManager->winGetWindowFromId(parent, staticTextMessageID);
170 GadgetStaticTextSetText(staticTextMessage,bodyString);
171
172 // create a structure that will pass the functions to
173 WindowExMessageBoxData *MsgBoxCallbacks = NEW WindowExMessageBoxData;
174 MsgBoxCallbacks->cancelCallback = cancelCallback;
175 MsgBoxCallbacks->noCallback = noCallback;
176 MsgBoxCallbacks->okCallback = okCallback;
177 MsgBoxCallbacks->yesCallback = yesCallback;
178 MsgBoxCallbacks->userData = userData;
179 //pass the structure to the dialog
180 parent->winSetUserData( MsgBoxCallbacks );
181
182 parent->winSetSystemFunc(ExtendedMessageBoxSystem);
183
184 //make sure the dialog is showing and bring it to the top
185 parent->winHide(FALSE);
186 parent->winBringToTop();
187
188 return parent;
189}// gogoExMessageBox

230WindowMsgHandledType ExtendedMessageBoxSystem( GameWindow *window, UnsignedInt msg,
231 WindowMsgData mData1, WindowMsgData mData2 )
232{
233
234
235 switch( msg )
236 {
237
238 //---------------------------------------------------------------------------------------------
239 case GWM_DESTROY:
240 {
241 delete (WindowExMessageBoxData *)window->winGetUserData();
242 window->winSetUserData( NULL );
243 break;
244
245 } // end case
246
247 // --------------------------------------------------------------------------------------------
248 case GWM_INPUT_FOCUS:
249 {
250
251 // if we're givin the opportunity to take the keyboard focus we must say we want it
252 if( mData1 == TRUE )
253 *(Bool *)mData2 = TRUE;
254
255 break;
256
257 } // end input
258
259 //---------------------------------------------------------------------------------------------
260 case GBM_SELECTED:
261 {
262 GameWindow *control = (GameWindow *)mData1;
263 Int controlID = control->winGetWindowId();
264 static NameKeyType buttonOkID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonOk" ) );
265 static NameKeyType buttonYesID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonYes" ) );
266 static NameKeyType buttonNoID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonNo" ) );
267 static NameKeyType buttonCancelID = TheNameKeyGenerator->nameToKey( AsciiString( "MessageBox.wnd:ButtonCancel" ) );
268 WindowExMessageBoxData *MsgBoxCallbacks = (WindowExMessageBoxData *)window->winGetUserData();
269
270 MessageBoxReturnType ret = MB_RETURN_CLOSE;
271
272 if( controlID == buttonOkID )
273 {
274 if (MsgBoxCallbacks->okCallback)
275 ret = MsgBoxCallbacks->okCallback(MsgBoxCallbacks->userData);
276 } // end if
277 else if( controlID == buttonYesID )
278 {
279 if (MsgBoxCallbacks->yesCallback)
280 ret = MsgBoxCallbacks->yesCallback(MsgBoxCallbacks->userData);
281 } // end else if
282 else if( controlID == buttonNoID )
283 {
284 if (MsgBoxCallbacks->noCallback)
285 ret = MsgBoxCallbacks->noCallback(MsgBoxCallbacks->userData);
286 } // end else if
287 else if( controlID == buttonCancelID )
288 {
289 if (MsgBoxCallbacks->cancelCallback)
290 ret = MsgBoxCallbacks->cancelCallback(MsgBoxCallbacks->userData);
291 } // end else if
292
293 if (ret == MB_RETURN_CLOSE)
294 TheWindowManager->winDestroy(window);
295
296 break;
297
298 } // end selected
299
300 //---------------------------------------------------------------------------------------------
301 default:
302 return MSG_IGNORED;
303
304 } // end switch
305
306 return MSG_HANDLED;
307
308} // end ExtendedMessageBoxSystem
