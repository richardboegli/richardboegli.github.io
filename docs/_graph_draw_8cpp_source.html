80void GraphDraw::render()
81{
82 Int width = TheDisplay->getWidth();
83
84 // divide the width by two because we're going to use the left half of the screen for labels.
85 //width /= 2;
86
87 // give more to bars than labels. (srj)
88 Int start = width * 0.33f;
89 width -= start;
90
91 Int height = TheDisplay->getHeight();
92
93 Int totalCount = m_graphEntries.size();
94 DEBUG_ASSERTCRASH(totalCount < MAX_GRAPH_VALUES, ("MAX_GRAPH_VALUES must be increased, not all labels will appear (max %d, cur %d).\n",MAX_GRAPH_VALUES,totalCount));
95 DEBUG_ASSERTCRASH(BAR_HEIGHT * totalCount < height, ("BAR_HEIGHT must be reduced, as bars are being drawn off-screen.\n"));
96 VecGraphEntriesIt it;
97
98 Int count = 0;
99 for (it = m_graphEntries.begin(); it != m_graphEntries.end(); ++it) {
100
101 if (count < MAX_GRAPH_VALUES) {
102 // draw the label.
103 UnicodeString uniStr;
104 uniStr.translate(it->first);
105
106 m_displayStrings[count]->setText(uniStr);
107 m_displayStrings[count]->draw(5, count * BAR_HEIGHT, 0xFFFFFFFF, 0x00000000, 1, 1);
108 }
109
110 TheDisplay->drawFillRect(start, count * BAR_HEIGHT - (BAR_SPACE / 2), it->second / 100000.0f * width, BAR_HEIGHT - BAR_SPACE, 0x7FFFFFFF);
111
112 ++count;
113 }
114}
