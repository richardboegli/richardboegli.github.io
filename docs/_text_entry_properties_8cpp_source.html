70//=============================================================================
71static LRESULT CALLBACK textEntryPropertiesCallback( HWND hWndDialog,
72 UINT message,
73 WPARAM wParam,
74 LPARAM lParam )
75{
76 Int returnCode;
77
78 //
79 // handle any common messages between all property dialogs cause they
80 // are designed to have controls doing the same functionality
81 // and names
82 //
83 if( HandleCommonDialogMessages( hWndDialog, message,
84 wParam, lParam, &returnCode ) == TRUE )
85 return returnCode;
86
87 switch( message )
88 {
89
90 // ------------------------------------------------------------------------
91 case WM_COMMAND:
92 {
93// Int notifyCode = HIWORD( wParam ); // notification code
94 Int controlID = LOWORD( wParam ); // control ID
95// HWND hWndControl = (HWND)lParam; // control window handle
96
97 switch( controlID )
98 {
99
100 // --------------------------------------------------------------------
101 case IDOK:
102 {
103 GameWindow *window = TheEditor->getPropertyTarget();
104
105 // sanity
106 if( window )
107 {
108 ImageAndColorInfo *info;
109
110 // save the common properties
111 if( SaveCommonDialogProperties( hWndDialog, window ) == FALSE )
112 break;
113
114 // save the image and color data
115 // ----------------------------------------------------------------
116 info = GetStateInfo( TEXT_ENTRY_ENABLED_LEFT );
117 GadgetTextEntrySetEnabledImageLeft( window, info->image );
118 GadgetTextEntrySetEnabledColor( window, info->color );
119 GadgetTextEntrySetEnabledBorderColor( window, info->borderColor );
120 info = GetStateInfo( TEXT_ENTRY_ENABLED_RIGHT );
121 GadgetTextEntrySetEnabledImageRight( window, info->image );
122 info = GetStateInfo( TEXT_ENTRY_ENABLED_CENTER );
123 GadgetTextEntrySetEnabledImageCenter( window, info->image );
124 info = GetStateInfo( TEXT_ENTRY_ENABLED_SMALL_CENTER );
125 GadgetTextEntrySetEnabledImageSmallCenter( window, info->image );
126
127 // ----------------------------------------------------------------
128 info = GetStateInfo( TEXT_ENTRY_DISABLED_LEFT );
129 GadgetTextEntrySetDisabledImageLeft( window, info->image );
130 GadgetTextEntrySetDisabledColor( window, info->color );
131 GadgetTextEntrySetDisabledBorderColor( window, info->borderColor );
132 info = GetStateInfo( TEXT_ENTRY_DISABLED_RIGHT );
133 GadgetTextEntrySetDisabledImageRight( window, info->image );
134 info = GetStateInfo( TEXT_ENTRY_DISABLED_CENTER );
135 GadgetTextEntrySetDisabledImageCenter( window, info->image );
136 info = GetStateInfo( TEXT_ENTRY_DISABLED_SMALL_CENTER );
137 GadgetTextEntrySetDisabledImageSmallCenter( window, info->image );
138
139 // ----------------------------------------------------------------
140 info = GetStateInfo( TEXT_ENTRY_HILITE_LEFT );
141 GadgetTextEntrySetHiliteImageLeft( window, info->image );
142 GadgetTextEntrySetHiliteColor( window, info->color );
143 GadgetTextEntrySetHiliteBorderColor( window, info->borderColor );
144 info = GetStateInfo( TEXT_ENTRY_HILITE_RIGHT );
145 GadgetTextEntrySetHiliteImageRight( window, info->image );
146 info = GetStateInfo( TEXT_ENTRY_HILITE_CENTER );
147 GadgetTextEntrySetHiliteImageCenter( window, info->image );
148 info = GetStateInfo( TEXT_ENTRY_HILITE_SMALL_CENTER );
149 GadgetTextEntrySetHiliteImageSmallCenter( window, info->image );
150
151 // text entry props
152 EntryData *entryData = (EntryData *)window->winGetUserData();
153
154 entryData->maxTextLen = GetDlgItemInt( hWndDialog, EDIT_MAX_CHARS, NULL, TRUE );
155 entryData->secretText = IsDlgButtonChecked( hWndDialog, CHECK_SECRET_TEXT );
156 entryData->aSCIIOnly = IsDlgButtonChecked( hWndDialog, CHECK_ASCII_TEXT );
157 if( IsDlgButtonChecked( hWndDialog, RADIO_LETTERS_AND_NUMBERS ) )
158 {
159
160 entryData->alphaNumericalOnly = TRUE;
161 entryData->numericalOnly = FALSE;
162
163 } // end if
164 else if( IsDlgButtonChecked( hWndDialog, RADIO_NUMBERS ) )
165 {
166
167 entryData->alphaNumericalOnly = FALSE;
168 entryData->numericalOnly = TRUE;
169
170 } // end else if
171 else
172 {
173
174 entryData->alphaNumericalOnly = FALSE;
175 entryData->numericalOnly = FALSE;
176
177 } // end else
178
179 } // end if
180
181 DestroyWindow( hWndDialog );
182 break;
183
184 } // end OK
185
186 // --------------------------------------------------------------------
187 case IDCANCEL:
188 {
189
190 DestroyWindow( hWndDialog );
191 break;
192
193 } // end cancel
194
195 } // end switch( LOWORD( wParam ) )
196
197 return 0;
198
199 } // end of WM_COMMAND
200
201 // ------------------------------------------------------------------------
202 case WM_CLOSE:
203 {
204
205 DestroyWindow( hWndDialog );
206 return 0;
207
208 } // end close
209
210 // ------------------------------------------------------------------------
211 default:
212 return 0;
213
214 } // end of switch
215
216} // end textEntryPropertiesCallback

223// InitTextEntryPropertiesDialog ==============================================
225//=============================================================================
226HWND InitTextEntryPropertiesDialog( GameWindow *window )
227{
228 HWND dialog;
229
230 // create the dialog box
231 dialog = CreateDialog( TheEditor->getInstance(),
232 (LPCTSTR)TEXT_ENTRY_PROPERTIES_DIALOG,
233 TheEditor->getWindowHandle(),
234 (DLGPROC)textEntryPropertiesCallback );
235 if( dialog == NULL )
236 return NULL;
237
238 // do the common initialization
239 CommonDialogInitialize( window, dialog );
240
241 //
242 // store in the image and color table the values for this putton
243 //
244 const Image *image;
245 Color color, borderColor;
246
247 // --------------------------------------------------------------------------
248 image = GadgetTextEntryGetEnabledImageLeft( window );
249 color = GadgetTextEntryGetEnabledColor( window );
250 borderColor = GadgetTextEntryGetEnabledBorderColor( window );
251 StoreImageAndColor( TEXT_ENTRY_ENABLED_LEFT, image, color, borderColor );
252 image = GadgetTextEntryGetEnabledImageRight( window );
253 StoreImageAndColor( TEXT_ENTRY_ENABLED_RIGHT, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
254 image = GadgetTextEntryGetEnabledImageCenter( window );
255 StoreImageAndColor( TEXT_ENTRY_ENABLED_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
256 image = GadgetTextEntryGetEnabledImageSmallCenter( window );
257 StoreImageAndColor( TEXT_ENTRY_ENABLED_SMALL_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
258
259 // --------------------------------------------------------------------------
260 image = GadgetTextEntryGetDisabledImageLeft( window );
261 color = GadgetTextEntryGetDisabledColor( window );
262 borderColor = GadgetTextEntryGetDisabledBorderColor( window );
263 StoreImageAndColor( TEXT_ENTRY_DISABLED_LEFT, image, color, borderColor );
264 image = GadgetTextEntryGetDisabledImageRight( window );
265 StoreImageAndColor( TEXT_ENTRY_DISABLED_RIGHT, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
266 image = GadgetTextEntryGetDisabledImageCenter( window );
267 StoreImageAndColor( TEXT_ENTRY_DISABLED_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
268 image = GadgetTextEntryGetDisabledImageSmallCenter( window );
269 StoreImageAndColor( TEXT_ENTRY_DISABLED_SMALL_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
270
271 // --------------------------------------------------------------------------
272 image = GadgetTextEntryGetHiliteImageLeft( window );
273 color = GadgetTextEntryGetHiliteColor( window );
274 borderColor = GadgetTextEntryGetHiliteBorderColor( window );
275 StoreImageAndColor( TEXT_ENTRY_HILITE_LEFT, image, color, borderColor );
276 image = GadgetTextEntryGetHiliteImageRight( window );
277 StoreImageAndColor( TEXT_ENTRY_HILITE_RIGHT, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
278 image = GadgetTextEntryGetHiliteImageCenter( window );
279 StoreImageAndColor( TEXT_ENTRY_HILITE_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
280 image = GadgetTextEntryGetHiliteImageSmallCenter( window );
281 StoreImageAndColor( TEXT_ENTRY_HILITE_SMALL_CENTER, image, WIN_COLOR_UNDEFINED, WIN_COLOR_UNDEFINED );
282
283 // fill out text entry properties
284 EntryData *entryData = (EntryData *)window->winGetUserData();
285
286 SetDlgItemInt( dialog, EDIT_MAX_CHARS, entryData->maxTextLen, TRUE );
287 if( entryData->secretText )
288 CheckDlgButton( dialog, CHECK_SECRET_TEXT, BST_CHECKED );
289 if( entryData->aSCIIOnly )
290 CheckDlgButton( dialog, CHECK_ASCII_TEXT, BST_CHECKED );
291 if( entryData->numericalOnly )
292 CheckDlgButton( dialog, RADIO_NUMBERS, BST_CHECKED );
293 else if( entryData->alphaNumericalOnly )
294 CheckDlgButton( dialog, RADIO_LETTERS_AND_NUMBERS, BST_CHECKED );
295 else
296 CheckDlgButton( dialog, RADIO_ANY_TEXT, BST_CHECKED );
297
298 // select the button enabled state for display
299 SwitchToState( TEXT_ENTRY_ENABLED_LEFT, dialog );
300
301 //
302 // initialize the dialog with values from the window
303 //
304
305 return dialog;
306
307} // end InitTextEntryPropertiesDialog
