101void EditObjectParameter::addObject( const ThingTemplate *thingTemplate )
102{
103 char buffer[ _MAX_PATH ];
104 HTREEITEM parent = TVI_ROOT;
105 const char *leafName;
106 //
107 // if we have an thing template in mapObject, we've read it from the new INI database,
108 // we will sort those items into the tree based on properties of the template that
109 // make it easier for us to browse when building levels
110 //
111 // Feel free to reorganize how this tree is constructed from the template
112 // data at will, whatever makes it easier for design
113 //
114 if( thingTemplate )
115 {
116
117 // first check for test sorted objects
118 if( thingTemplate->getEditorSorting() == ES_TEST )
119 parent = findOrAdd( parent, "TEST" );
120
121 // first sort by Side, either create or find the tree item with matching side name
122 AsciiString side = thingTemplate->getDefaultOwningSide();
123 DEBUG_ASSERTCRASH(!side.isEmpty(), ("NULL default side in template\n") );
124 strcpy( buffer, side.str() );
125 parent = findOrAdd( parent, buffer );
126
127 // next tier uses the editor sorting that design can specify in the INI
128 for( EditorSortingType i = ES_FIRST;
129 i < ES_NUM_SORTING_TYPES;
130 i = (EditorSortingType)(i + 1) )
131 {
132
133 if( thingTemplate->getEditorSorting() == i )
134 {
135
136 parent = findOrAdd( parent, EditorSortingNames[ i ] );
137 break; // exit for
138
139 } // end if
140
141 } // end for i
142
143 if( i == ES_NUM_SORTING_TYPES )
144 parent = findOrAdd( parent, "UNSORTED" );
145
146 // the leaf name is the name of the template
147 leafName = thingTemplate->getName().str();
148
149 } // end if
150
151 // add to the tree view
152 if( leafName )
153 {
154 TVINSERTSTRUCT ins;
155
156 ::memset(&ins, 0, sizeof(ins));
157 ins.hParent = parent;
158 ins.hInsertAfter = TVI_SORT;
159 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
160 ins.item.lParam = 0;
161 ins.item.pszText = (char*)leafName;
162 ins.item.cchTextMax = strlen(leafName)+2;
163 m_objectTreeView.InsertItem(&ins);
164
165 }
166
167}

207HTREEITEM EditObjectParameter::findOrAdd(HTREEITEM parent, const char *pLabel)
208{
209 TVINSERTSTRUCT ins;
210 char buffer[_MAX_PATH];
211 ::memset(&ins, 0, sizeof(ins));
212 HTREEITEM child = m_objectTreeView.GetChildItem(parent);
213 while (child != NULL) {
214 ins.item.mask = TVIF_HANDLE|TVIF_TEXT;
215 ins.item.hItem = child;
216 ins.item.pszText = buffer;
217 ins.item.cchTextMax = sizeof(buffer)-2;
218 m_objectTreeView.GetItem(&ins.item);
219 if (strcmp(buffer, pLabel) == 0) {
220 return(child);
221 }
222 child = m_objectTreeView.GetNextSiblingItem(child);
223 }
224
225 // not found, so add it.
226 ::memset(&ins, 0, sizeof(ins));
227 ins.hParent = parent;
228 ins.hInsertAfter = TVI_SORT;
229 ins.item.mask = TVIF_PARAM|TVIF_TEXT;
230 ins.item.lParam = -1;
231 ins.item.pszText = (char*)pLabel;
232 ins.item.cchTextMax = strlen(pLabel);
233 child = m_objectTreeView.InsertItem(&ins);
234 return(child);
235}

237BOOL EditObjectParameter::OnNotify(WPARAM wParam, LPARAM lParam, LRESULT* pResult)
238{
239 NMTREEVIEW *pHdr = (NMTREEVIEW *)lParam;
240
241 // Handle events from the tree control.
242 if (pHdr->hdr.idFrom == IDC_TERRAIN_TREEVIEW) {
243 if (pHdr->hdr.code == TVN_KEYDOWN) {
244 NMTVKEYDOWN *pKey = (NMTVKEYDOWN*)lParam;
245 Int key = pKey->wVKey;
246 if (key==VK_SHIFT || key==VK_SPACE) {
247 HTREEITEM hItem = m_objectTreeView.GetSelectedItem();
248 if (!m_objectTreeView.ItemHasChildren(hItem)) {
249 hItem = m_objectTreeView.GetParentItem(hItem);
250 }
251 m_objectTreeView.Expand(hItem, TVE_TOGGLE);
252 return 0;
253 }
254 return 0;
255 }
256 }
257 return CDialog::OnNotify(wParam, lParam, pResult);
258}

260void EditObjectParameter::OnOK()
261{
262 char buffer[_MAX_PATH];
263 HTREEITEM hItem = m_objectTreeView.GetSelectedItem();
264 if (!hItem) {
265 ::MessageBeep(MB_ICONEXCLAMATION);
266 return;
267 }
268
269 TVITEM item;
270 ::memset(&item, 0, sizeof(item));
271 item.mask = TVIF_HANDLE|TVIF_PARAM|TVIF_TEXT|TVIF_STATE;
272 item.hItem = hItem;
273 item.pszText = buffer;
274 item.cchTextMax = sizeof(buffer)-2;
275 m_objectTreeView.GetItem(&item);
276 AsciiString objName = buffer;
277 // We used to try to find the TT here, but now we don't because we
278 // use object lists as well.
279 m_parameter->friend_setString(objName);
280 CDialog::OnOK();
281}
