171//-------------------------------------------------------------------------------------------------
172UpdateSleepTime DeliverPayloadAIUpdate::update( void )
173{
174 m_deliveryDecal.update();
175
176 if(!(isAiInDeadState()) && m_deliverPayloadStateMachine)
177 m_deliverPayloadStateMachine->updateStateMachine();
178
179 //Handling diving logic (regardless of state)
180 if( m_diveState != DIVESTATE_POSTDIVE )
181 {
182 if( m_diveState == DIVESTATE_PREDIVE )
183 {
184 //Check to see if we are close enough to start diving.
185 Real startDiveDistanceSquared = sqr( getData()->m_diveStartDistance );
186 Real currentDistanceSquared = ThePartitionManager->getDistanceSquared( getObject(), getTargetPos(), FROM_CENTER_2D );
187 if( currentDistanceSquared <= startDiveDistanceSquared )
188 {
189 m_diveState = DIVESTATE_DIVING;
190 getObject()->getAIUpdateInterface()->getCurLocomotor()->setUsePreciseZPos( true );
191
192 AudioEventRTS soundDive = *(getObject()->getTemplate()->getPerUnitSound("StartDive"));
193 if( soundDive.getEventName().isNotEmpty() )
194 {
195 soundDive.setPosition( getObject()->getPosition() );
196 TheAudio->addAudioEvent( &soundDive );
197 }
198 }
199 }
200 else
201 {
202 //Check to see when we shall end diving
203 Real endDiveDistanceSquared = sqr( getData()->m_diveEndDistance );
204 Real currentDistanceSquared = ThePartitionManager->getDistanceSquared( getObject(), getTargetPos(), FROM_CENTER_3D );
205 if( currentDistanceSquared <= endDiveDistanceSquared )
206 {
207 m_diveState = DIVESTATE_POSTDIVE;
208 getObject()->getAIUpdateInterface()->getCurLocomotor()->setUsePreciseZPos( false );
209 }
210
211 if( m_data.m_strafingWeaponSlot != -1 )
212 {
213 //SHOOT!!
214 const Coord3D *velocity = getObject()->getPhysics()->getVelocity();
215 if( velocity->z < 5.0f )
216 {
217 //Calc strafe ratio
218 Real startDiveDistance = getData()->m_diveStartDistance;
219 Real endDiveDistance = sqrt( endDiveDistanceSquared );
220 Real currentDistance = sqrt( currentDistanceSquared );
221
222 Real diveRatio = (startDiveDistance - currentDistance) / (startDiveDistance - endDiveDistance);
223
224 Coord3D velocity = *getObject()->getPhysics()->getVelocity();
225 velocity.z = 0.0f;
226 velocity.normalize();
227 velocity.scale( diveRatio * 100.0f );
228
229 Coord3D backwards = velocity;
230 backwards.scale( 0.33f );
231
232 Coord3D strafePoint = *getTargetPos();
233 strafePoint.sub( &backwards );
234
235 strafePoint.add( &velocity );
236 strafePoint.z = TheTerrainLogic->getGroundHeight( strafePoint.x, strafePoint.y );
237
238 // lock it just till the weapon is empty or the attack is "done"
239 getObject()->setWeaponLock( m_data.m_strafingWeaponSlot, LOCKED_TEMPORARILY );
240 getObject()->fireCurrentWeapon( &strafePoint );
241 FXList::doFXPos( m_data.m_strafeFX, &strafePoint );
242 }
243 }
244 }
245 }
246
247 AIUpdateInterface::update();
248
249 return UPDATE_SLEEP_NONE; // ignore our parent, and never sleep
250}

253void DeliverPayloadAIUpdate::deliverPayload(
254 const Coord3D *moveToPos,
255 const Coord3D *targetPos,
256 const DeliverPayloadData *data
257)
258{
259
260 //****************************************************
261 //THIS GETS CALLED VIA OBJECT CREATION LISTS ONLY!!!
262 //****************************************************
263
264 if (m_deliverPayloadStateMachine)
265 m_deliverPayloadStateMachine->deleteInstance();
266 m_deliverPayloadStateMachine = NULL;
267
268 m_moveToPos = *moveToPos;
269 m_targetPos = *targetPos;
270 m_data = *data;
271
272 m_deliveryDecal.clear();
273 m_data.m_deliveryDecalTemplate.createRadiusDecal(*targetPos,
274 m_data.m_deliveryDecalRadius, getObject()->getControllingPlayer(), m_deliveryDecal);
275
276 if( m_data.m_diveStartDistance <= 0.0f )
277 {
278 //We can't dive, so setting it to post-dive will prevent
279 //dive logic from being processed.
280 m_diveState = DIVESTATE_POSTDIVE;
281 }
282 else
283 {
284 m_diveState = DIVESTATE_PREDIVE;
285 }
286
287 m_visibleItemsDelivered = 0;
288
289 //If we have visible bones, then show each of those subobjects.
290 Drawable *draw = getObject()->getDrawable();
291 if( draw )
292 {
293 for( int i = 1; i <= m_data.m_visibleNumBones; i++ )
294 {
295 if( m_data.m_visibleSubObjectName.isNotEmpty() )
296 {
297 AsciiString name;
298 name.format( "%s%02d", m_data.m_visibleSubObjectName.str(), i );
299 draw->showSubObject( name, true );
300 }
301 }
302 }
303
304
305 // must make the state machine AFTER initing the other stuff, since it may inquire of its values...
306 m_deliverPayloadStateMachine = newInstance(DeliverPayloadStateMachine)( getObject() );
307 m_deliverPayloadStateMachine->initDefaultState();
308#ifdef _DEBUG
309 m_deliverPayloadStateMachine->setName("DeliverPayloadSpecificAI");
310#endif
311}

343Real DeliverPayloadAIUpdate::calcMinTurnRadius(Real* timeToTravelThatDist) const
344{
345 const Locomotor* loco = getCurLocomotor();
346 BodyDamageType bdt = getObject()->getBodyModule()->getDamageState();
348 Real maxSpeed = loco->getMaxSpeedForCondition(bdt); // in dist/frame
349 Real maxTurnRate = loco->getMaxTurnRate(bdt); // in rads/frame
350
351 /*
352 our minimum circumference will be like so:
353
354 Real minTurnCircum = maxSpeed * (2*PI / maxTurnRate);
355
356 so therefore our minimum turn radius is:
357
358 Real minTurnRadius = minTurnCircum / 2*PI;
359
360 so we just eliminate the middleman:
361 */
362 Real minTurnRadius = (maxTurnRate > 0.0f) ? (maxSpeed / maxTurnRate) : 999999.0f;
363
364 if (timeToTravelThatDist)
365 *timeToTravelThatDist = minTurnRadius / maxSpeed;
366
367 return minTurnRadius;
368}

372Bool DeliverPayloadAIUpdate::isCloseEnoughToTarget()
373{
374 // In addition to testing distance, it is also sensitive to being in/outward bound
377
378 Real allowedDistanceSqr = sqr( getAllowedDistanceToTarget() );
379 Real currentDistanceSqr = ThePartitionManager->getDistanceSquared( getObject(), getTargetPos(), FROM_CENTER_2D );
380 Bool inBound = m_previousDistanceSqr > currentDistanceSqr;
381 m_previousDistanceSqr = currentDistanceSqr;// for the next test
382
383 if ( inBound )
384 allowedDistanceSqr = sqr(getAllowedDistanceToTarget() + getPreOpenDistance());
385
386 //DEBUG_LOG(("Dist to target is %f (allowed %f)\n",sqrt(currentDistanceSqr),sqrt(allowedDistanceSqr)));
387
388
389 if ( allowedDistanceSqr > currentDistanceSqr )
390 return TRUE;
391
392 return FALSE;
393

690//-------------------------------------------------------------------------------------------------
691StateReturnType DeliveringState::update() // Kick a dude out every so often
692{
693 if (m_dropDelayLeft > 0)
694 {
695 --m_dropDelayLeft;
696 return STATE_CONTINUE;
697 }
698
699 Object *owner = getMachineOwner();
700 DeliverPayloadAIUpdate *ai = (DeliverPayloadAIUpdate*)owner->getAIUpdateInterface();
701 if( !ai )
702 {
703 return STATE_FAILURE;
704 }
705
706 ai->friend_setFreeToExit(true);
707
708 m_didOpen = true;
709 m_dropDelayLeft = ai->getDropDelay();
710
711 if (!ai->isCloseEnoughToTarget())

964{

981 // ok, play some games here... the pathfinder is pretty dumb about things with big turning radii,
982 // so we have to force the issue here... basically, estimate our necessary turning radius
983 // based on loco values, move off far enough so we can turn, then head back. (if we just say

998 reApproachPoint.y = owner->getPosition()->y + dir->y * minReApproachDist;
