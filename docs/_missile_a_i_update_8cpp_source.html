212//-------------------------------------------------------------------------------------------------
213// The actual firing of the missile once setup.
214//-------------------------------------------------------------------------------------------------
215void MissileAIUpdate::projectileFireAtObjectOrPosition( const Object *victim, const Coord3D *victimPos, const WeaponTemplate *detWeap, const ParticleSystemTemplate* exhaustSysOverride )
216{
217 const MissileAIUpdateModuleData* d = getMissileAIUpdateModuleData();
218 Object *obj = getObject();
219
220 m_exhaustSysTmpl = exhaustSysOverride;
221 m_detonationWeaponTmpl = detWeap;
222 Real initialVelToUse = d->m_initialVel;
223 if (d->m_useWeaponSpeed)
224 {
225 Real weaponSpeed = detWeap->getWeaponSpeed();
226 initialVelToUse = weaponSpeed;
227
228 Locomotor* curLoco = getCurLocomotor();
229 if (curLoco)
230 {
231 m_maxAccel = weaponSpeed;
232 curLoco->setMaxSpeed(weaponSpeed);
233 curLoco->setMaxAcceleration(m_maxAccel);
234 }
235 }
236
237 Real deltaZ = victimPos->z - obj->getPosition()->z;
238 Real dx = victimPos->x - obj->getPosition()->x;
239 Real dy = victimPos->y - obj->getPosition()->y;
240 Real xyDist = sqrt(sqr(dx)+sqr(dy));
241 if (xyDist<1) xyDist = 1;
242 Real zFactor = 0;
243 if (deltaZ>0) {
244 zFactor = deltaZ/xyDist;
245 }
246
247
248 Vector3 dir = getObject()->getTransformMatrix()->Get_X_Vector();
249 dir.Normalize();
250 dir.Z += 2*zFactor;
251 dir.Normalize();
252 PhysicsBehavior* physics = getObject()->getPhysics();
253 if (physics && initialVelToUse > 0)
254 {
255 Real forceMag = physics->getMass() * initialVelToUse;
256
257 Coord3D force;
258 force.x = forceMag * dir.X;
259 force.y = forceMag * dir.Y;
260 force.z = forceMag * dir.Z;
261
262 physics->applyMotiveForce( &force );
263 }
264
265 Vector3 objPos(obj->getPosition()->x, obj->getPosition()->y, obj->getPosition()->z);
266
267 Matrix3D newXform;
268 newXform.buildTransformMatrix( objPos, dir );
269 obj->setTransformMatrix( &newXform );
270
271 switchToState(LAUNCH);
272 m_isTrackingTarget = false;
273 // Missiles do their thing by colliding with something and exploding. So they Move to the target
274 // instead of Attacking the target.
275 if (victim && d->m_tryToFollowTarget)
276 {
277 getStateMachine()->setGoalPosition(victim->getPosition());
278 // ick. const-cast is evil. fix. (srj)
279 aiMoveToObject(const_cast<Object*>(victim), CMD_FROM_AI );
280 m_originalTargetPos = *victim->getPosition();
281 m_isTrackingTarget = TRUE;// Remember that I was originally shot at a moving object, so if the
282 // target dies I can do something cool.
283 m_victimID = victim->getID();
284 }
285 else
286 {
287 // Otherwise, we are just a Coord shot.
288 Coord3D initialPos = *victimPos;
289 m_originalTargetPos = *victimPos;
290 if (d->m_lockDistance>0.0f) {
291 initialPos.z += APPROACH_HEIGHT;
292 }
293 aiMoveToPosition(&initialPos, CMD_FROM_AI );
294 m_victimID = INVALID_ID;
295 }
296
297 setCurrentVictim( victim );
298 m_prevPos = *getObject()->getPosition();
299}

302//-------------------------------------------------------------------------------------------------
303Bool MissileAIUpdate::projectileHandleCollision( Object *other )
304{
305 Object* obj = getObject();
306 const MissileAIUpdateModuleData* d = getMissileAIUpdateModuleData();
307
308 // check if our warhead is "armed" - if not, we are inert
309 if (projectileIsArmed() == false)
310 return true;
311
312 if (other==NULL) {
313 // we hit the ground. Check to see if we hit something unexpected.
314 Coord3D goal = *getGoalPosition();
315 Coord3D pos = *obj->getPosition();
316 Coord3D delta;
317 delta.x = pos.x-goal.x;
318 delta.y = pos.y-goal.y;
319 delta.z = pos.z-goal.z;
320 if (delta.z > PATHFIND_CELL_SIZE_F) {
321 // we're above our target goal.
322 if (delta.length() > 3*PATHFIND_CELL_SIZE_F) {
323 // we're somewhere else.
324 return true;
325 }
326 }
327 }
328
329 if (other != NULL)
330 {
331 Object *projectileLauncher = TheGameLogic->findObjectByID( projectileGetLauncherID() );
332
333 // if it's not the specific thing we were targeting, see if we should incidentally collide...
334 if (!m_detonationWeaponTmpl->shouldProjectileCollideWith(projectileLauncher, obj, other, m_victimID))
335 {
336 //DEBUG_LOG(("ignoring projectile collision with %s at frame %d\n",other->getTemplate()->getName().str(),TheGameLogic->getFrame()));
337 return true;
338 }
339
340 if (d->m_garrisonHitKillCount > 0)
341 {
342 ContainModuleInterface* contain = other->getContain();
343 if( contain && contain->getContainCount() > 0 && contain->isGarrisonable() && !contain->isImmuneToClearBuildingAttacks() )
344 {
345 Int numKilled = 0;
346
347 // garrisonable buildings subvert the normal process here.
348 const ContainedItemsList* items = contain->getContainedItemsList();
349 if (items)
350 {
351 for (ContainedItemsList::const_iterator it = items->begin(); *it != NULL && numKilled < d->m_garrisonHitKillCount; )
352 {
353 Object* thingToKill = *it++;
354 if (!thingToKill->isEffectivelyDead() && thingToKill->isKindOfMulti(d->m_garrisonHitKillKindof, d->m_garrisonHitKillKindofNot))
355 {
356 //DEBUG_LOG(("Killed a garrisoned unit (%08lx %s) via Flash-Bang!\n",thingToKill,thingToKill->getTemplate()->getName().str()));
357 if (projectileLauncher)
358 projectileLauncher->scoreTheKill( thingToKill );
359 thingToKill->kill();
360 ++numKilled;
361 }
362 } // next contained item
363 } // if items
364
365 if (numKilled > 0)
366 {
367 // note, fx is played at center of building, not at grenade's location
368 FXList::doFXObj(d->m_garrisonHitKillFX, other, NULL);
369
370 // don't do the normal explosion; just destroy ourselves & return
371 TheGameLogic->destroyObject(obj);
372
373 return true;
374 }
375 } // if a garrisonable thing
376 }
377 }
378
379 // collided with something... blow'd up!
380 detonate();
381
382 // mark ourself as "no collisions" (since we might still exist in slow death mode)
383 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
384 return true;
385}

388void MissileAIUpdate::detonate()
389{
390 Object* obj = getObject();
391
392 if (m_detonationWeaponTmpl)
393 {
394
395 TheWeaponStore->handleProjectileDetonation(m_detonationWeaponTmpl, obj, obj->getPosition(), m_extraBonusFlags, !m_noDamage );
396
397 if( m_detonationWeaponTmpl->getDieOnDetonate() )
398 {
399 DamageInfo damageInfo;
400 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
401 damageInfo.in.m_deathType = DEATH_DETONATED;
402 damageInfo.in.m_sourceID = INVALID_ID;
403 damageInfo.in.m_amount = obj->getBodyModule()->getMaxHealth();
404 obj->attemptDamage( &damageInfo );
405 }
406 }
407 else if( !m_noDamage )
408 {
409 DamageInfo damageInfo;
410 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
411 damageInfo.in.m_deathType = DEATH_DETONATED;
412 damageInfo.in.m_sourceID = INVALID_ID;
413 damageInfo.in.m_amount = obj->getBodyModule()->getMaxHealth();
414 obj->attemptDamage( &damageInfo );
415 }
416 if (obj->getDrawable())
417 obj->getDrawable()->setDrawableHidden(true);
418 // Delay destroying the object two frames to let the contrail catch up. jba.
419
420 switchToState(KILL_SELF);
421
422 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_MISSILE_KILLING_SELF ) );
423
424}

437void MissileAIUpdate::doKillSelfState()
438{
439 const MissileAIUpdateModuleData *modData = getMissileAIUpdateModuleData();
440
441 if (m_stateTimestamp > TheGameLogic->getFrame() - modData->m_killSelfDelay )
442 {
443 // Hold in this state [modData->m_killSelfDelay] frames to let the contrail catch up. jba.
444 return;
445 }
446 Object* obj = getObject();
447 if (m_detonationWeaponTmpl)
448 {
449 if ( modData->m_detonateCallsKill )
450 obj->kill(); // kill it (vs destroying it) so that its Die modules are called
451 else
452 TheGameLogic->destroyObject( obj );
453 }
454 switchToState(DEAD);
455}

500//-------------------------------------------------------------------------------------------------
501void MissileAIUpdate::doAttackState(Bool turnOK)
502{
503 Locomotor* curLoco = getCurLocomotor();
504
505 const MissileAIUpdateModuleData* d = getMissileAIUpdateModuleData();
506 if (TheGameLogic->getFrame() >= m_fuelExpirationDate)
507 {
508 if( d->m_detonateOnNoFuel )
509 {
510 detonate();
511 return;
512 }
513
514 if (curLoco)
515 {
516 curLoco->setMaxAcceleration(0);
517 curLoco->setMaxTurnRate(0);
518 }
519 tossExhaust();
520 }
521 else
522 {
523 if (curLoco)
524 {
525 curLoco->setMaxAcceleration(m_maxAccel);
526 curLoco->setMaxTurnRate(turnOK ? BIGNUM : 0);
527 }
528 }
529
530 if (d->m_lockDistance > 0)
531 {
532 Real lockDistanceSquared = d->m_lockDistance;
533 Real distanceToTargetSquared;
534 if (m_isTrackingTarget && (getGoalObject() != NULL)) {
535 distanceToTargetSquared = ThePartitionManager->getDistanceSquared( getObject(), getGoalObject(), FROM_CENTER_2D);
536 } else {
537 distanceToTargetSquared = ThePartitionManager->getDistanceSquared( getObject(), getGoalPosition(), FROM_CENTER_2D );
538 }
539 if (lockDistanceSquared>0) {
540 if (!m_isTrackingTarget) {
541 // Immobile or ground target. Halve the lock distance.
542 lockDistanceSquared *= 0.5f;
543 }
544 lockDistanceSquared *= lockDistanceSquared;
545 if (distanceToTargetSquared < lockDistanceSquared) {
546 if (!m_isTrackingTarget) {
547 // Ground pos. Change to original goal.
548 aiMoveToPosition(&m_originalTargetPos, CMD_FROM_AI );
549 }
550 switchToState(KILL);
551 return;
552 }
553 }
554 }
555
556 if(curLoco && curLoco->getPreferredHeight() > 0)
557 {
558 // Am I close enough to the target to ignore my preferred height setting?
559 Real distanceToTargetSquared = ThePartitionManager->getDistanceSquared( getObject(), getGoalPosition(), FROM_CENTER_2D );
560 Real diveDistanceSquared = d->m_diveDistance;
561 if (curLoco && curLoco->getPreferredHeight()) {
562 diveDistanceSquared *= diveDistanceSquared;
563 if( distanceToTargetSquared < diveDistanceSquared )
564 curLoco->setUsePreciseZPos( true );
565 }
566
567 }
568
569 if (m_noTurnDistLeft <= 0.0f)
570 {
571 switchToState(ATTACK);
572 }
573
574 // If I was fired at a flyer and have lost target (most likely they died), then I need to do something better
575 // than cloverleaf around their last spot.
576 if( m_isTrackingTarget && (getGoalObject() == NULL) )
577 airborneTargetGone();
578}

580//-------------------------------------------------------------------------------------------------
581void MissileAIUpdate::doKillState(void)
582{
583 if (TheGameLogic->getFrame() >= m_fuelExpirationDate)
584 {
585 if( getMissileAIUpdateModuleData()->m_detonateOnNoFuel )
586 {
587 detonate();
588 return;
589 }
590
591 // srj sez: killstate missiles should not be allowed to live forever.
592 airborneTargetGone();
593 return;
594 }
595
596 /* Kill state uses the OBJECT_STATUS_BRAKING, which is a flag to the locomotor to
597 "Cheat" and move directly towards the target. This ensures that the missile will hit.
598 jba. */
599 Locomotor* curLoco = getCurLocomotor();
600 Object *obj = getObject();
601 // Objects that are braking don't follow the normal physics, so they end up at their destination exactly.
602 obj->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_BRAKING ) );
603
604 if (curLoco)
605 {
606 curLoco->setMaxAcceleration(m_maxAccel);
607 curLoco->setMaxTurnRate(BIGNUM);
608 }
609 if (isIdle()) {
610 // we finished the move
611 if (getGoalObject()!=NULL) {
612 Locomotor* curLoco = getCurLocomotor();
613 Real closeEnough = 1.0f;
614 if (curLoco)
615 {
616 closeEnough = curLoco->getMaxSpeedForCondition(BODY_PRISTINE);
617 }
618 Real distanceToTargetSq = ThePartitionManager->getDistanceSquared( getObject(), getGoalObject(), FROM_BOUNDINGSPHERE_3D);
619 //DEBUG_LOG(("Distance to target %f, closeEnough %f\n", sqrt(distanceToTargetSq), closeEnough));
620 if (distanceToTargetSq < closeEnough*closeEnough) {
621 Coord3D pos = *getGoalObject()->getPosition();
622 getObject()->setPosition(&pos);
623 detonate();
624 } else{
625 aiMoveToObject(getGoalObject(), CMD_FROM_AI );
626 }
627 } else {
628 detonate();
629 }
630 }
631 // If I was fired at a flyer and have lost target (most likely they died), then I need to do something better
632 // than cloverleaf around their last spot.
633 if( m_isTrackingTarget && (getGoalObject() == NULL) )
634 airborneTargetGone();
635}

648//-------------------------------------------------------------------------------------------------
652UpdateSleepTime MissileAIUpdate::update()
653{
654 Coord3D newPos = *getObject()->getPosition();
655 if (m_noTurnDistLeft > 0.0f && m_state >= IGNITION)
656 {
657 Real distThisTurn = sqrtf(sqr(newPos.x-m_prevPos.x) + sqr(newPos.y-m_prevPos.y) + sqr(newPos.z-m_prevPos.z));
658 m_noTurnDistLeft -= distThisTurn;
659 m_prevPos = newPos;
660 }
661
662 //If this missile has been marked to divert to countermeasures, check when
663 //that will occur, then do it when the timer expires.
664 if( m_framesTillDecoyed && m_framesTillDecoyed <= TheGameLogic->getFrame() )
665 {
666 Object *missile = getObject();
667 m_framesTillDecoyed = 0;
668 m_noDamage = TRUE;
669 Object *victim = TheGameLogic->findObjectByID( m_victimID );
670 if( victim )
671 {
672 ObjectID targetID = missile->calculateCountermeasureToDivertTo( *victim );
673 if( targetID != INVALID_ID )
674 {
675 victim = TheGameLogic->findObjectByID( targetID );
676 getStateMachine()->setGoalPosition(victim->getPosition());
677 // ick. const-cast is evil. fix. (srj)
