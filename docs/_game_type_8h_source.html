182//-------------------------------------------------------------------------------------------------
183// Pathfind layers - ground is the first layer, each bridge is another. jba.
184// Layer 1 is the ground.
185// Layer 2 is the top layer - bridge if one is present, ground otherwise.
186// Layer 2 - LAYER_LAST -1 are bridges.
187// Layer_WALL is a special "wall" layer for letting units run aroound on top of a wall
188// made of structures.
189// Note that the bridges just index in the pathfinder, so you don't actually
190// have a LAYER_BRIDGE_1 enum value.
191enum PathfindLayerEnum {LAYER_INVALID = 0, LAYER_GROUND = 1, LAYER_WALL = 15, LAYER_LAST=15};
