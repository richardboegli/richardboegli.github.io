52void LanguageFilter::init() {
53 m_wordList.clear();
54
55 // read in the file already.
56 File *file1 = TheFileSystem->openFile(BadWordFileName, File::READ | File::BINARY);
57 if (file1 == NULL) {
58 return;
59 }
60
61 wchar_t word[128];
62 while (readWord(file1, word)) {
63 Int wordLen = wcslen(word);
64 if (wordLen == 0) {
65 continue;
66 }
67 for (Int i = 0; i < wordLen; ++i) {
68 word[i] = word[i] ^ LANGUAGE_XOR_KEY;
69 }
70 UnicodeString uniword(word);
71 unHaxor(uniword);
72 //DEBUG_LOG(("Just read %ls from the bad word file. Entered as %ls\n", word, uniword.str()));
73 m_wordList[uniword] = true;
74 }
75
76 file1->close();
77 file1 = NULL;
78}

89void LanguageFilter::filterLine(UnicodeString &line)
90{
91 WideChar *buf = NEW WideChar[line.getLength()+1];
92 wcscpy(buf, line.str());
93
94 UnicodeString newLine(line);
95 UnicodeString token(L"");
96
97 while (newLine.nextToken(&token, UnicodeString(L" ;,.!?:=\\/><`~()&^%#\n\t"))) {
98 wchar_t *pos = wcsstr(buf, token.str());
99 if (pos == NULL) {
100 DEBUG_CRASH(("Couldn't find the token in its own string."));
101 continue;
102 }
103
104 Int len = token.getLength(); // need to get the length of the original word, not the unhaxor'd word.
105
106 unHaxor(token);
107 LangMapIter iter = m_wordList.find(token);
108 if (iter != m_wordList.end()) {
109 DEBUG_LOG(("Found word %ls in bad word list. Token was %ls\n", (*iter).first.str(), token.str()));
110 for (Int i = 0; i < len; ++i) {
111 *pos = L'*';
112 ++pos;
113 }
114 }
115 }
116
117 line.set(buf);
118 delete[] buf;
119}

121void LanguageFilter::unHaxor(UnicodeString &word) {
122 Int len = word.getLength();
123 UnicodeString newWord(L"");
124 for (Int i = 0; i < len; ++i) {
125 wchar_t c = word.getCharAt(i);
126 if ((c == L'p') || (c == L'P')) {
127 if (((i + 1) < len) && ((word.getCharAt(i+1) == L'h') || (word.getCharAt(i+1) == L'H'))) {
128 newWord.concat(L'f');
129 ++i; // skip the h
130 } else {
131 // not a problem at all.
132 newWord.concat(c);
133 }
134 } else if (c == L'1') {
135 newWord.concat(L'l');
136 } else if (c == L'3') {
137 newWord.concat(L'e');
138 } else if (c == L'4') {
139 newWord.concat(L'a');
140 } else if (c == L'5') {
141 newWord.concat(L's');
142 } else if (c == L'6') {
143 newWord.concat(L'b');
144 } else if (c == L'7') {
145 newWord.concat(L't');
146 } else if (c == L'0') {
147 newWord.concat(L'o');
148 } else if (c == L'@') {
149 newWord.concat(L'a');
150 } else if (c == L'$') {
151 newWord.concat(L's');
152 } else if (c == L'+') {
153 newWord.concat(L't');
154 } else if (wcsrchr(ignoredChars, c) == NULL) {
155 newWord.concat(c);
156 }
157 }
158 word.set(newWord);
159}

162Bool LanguageFilter::readWord(File *file1, UnsignedShort *buf) {
163 Int index = 0;
164 Bool retval = TRUE;
165 Int val = 0;
166
167 UnsignedShort c;
168
169 val = file1->read(&c, sizeof(UnsignedShort));
170 if ((val == -1) || (val == 0)) {
171 buf[index] = 0;
172 return FALSE;
173 }
174 buf[index] = c;
175
176 while (buf[index] != L' ') {
177 ++index;
178 val = file1->read(&c, sizeof(UnsignedShort));
179 if ((val == -1) || (val == 0)) {
180 c = WEOF;
181 }
182
183 if ((c == WEOF) || (c == L' ')) {
184 buf[index] = 0;
185 if (c == WEOF) {
186 retval = FALSE;
187 }
188 break;
189 }
190 buf[index] = c;
191 }
192 return retval;
193}
