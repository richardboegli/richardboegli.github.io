369void Xfer::xferSTLObjectIDVector( std::vector<ObjectID> *objectIDVectorData )
370{
371 //
372 // the fact that this is a list and a little higher level than a simple data type
373 // is reason enough to have every one of these versioned
374 //
375 XferVersion currentVersion = 1;
376 XferVersion version = currentVersion;
377 xferVersion( &version, currentVersion );
378
379 // xfer the count of the vector
380 UnsignedShort listCount = objectIDVectorData->size();
381 xferUnsignedShort( &listCount );
382
383 // xfer vector data
384 ObjectID objectID;
385 if( getXferMode() == XFER_SAVE || getXferMode() == XFER_CRC )
386 {
387
388 // save all ids
389 std::vector< ObjectID >::const_iterator it;
390 for( it = objectIDVectorData->begin(); it != objectIDVectorData->end(); ++it )
391 {
392
393 objectID = *it;
394 xferObjectID( &objectID );
395
396 } // end for
397
398 } // end if, save
399 else if( getXferMode() == XFER_LOAD )
400 {
401
402 // sanity, the list should be empty before we transfer more data into it
403 if( objectIDVectorData->size() != 0 )
404 {
405
406 DEBUG_CRASH(( "Xfer::xferSTLObjectIDList - object vector should be empty before loading\n" ));
407 throw XFER_LIST_NOT_EMPTY;
408
409 } // end if
410
411 // read all ids
412 for( UnsignedShort i = 0; i < listCount; ++i )
413 {
414
415 xferObjectID( &objectID );
416 objectIDVectorData->push_back( objectID );
417
418 } // end for, i
419
420 } // end else if
421 else
422 {
423
424 DEBUG_CRASH(( "xferSTLObjectIDList - Unknown xfer mode '%d'\n", getXferMode() ));
425 throw XFER_MODE_UNKNOWN;
426
427 } // end else

732void Xfer::xferUpgradeMask( UpgradeMaskType *upgradeMaskData )
733{
734
735 // this deserves a version number
736 XferVersion currentVersion = 1;
737 XferVersion version = currentVersion;
738 xferVersion( &version, currentVersion );
739
740 //Kris: The Upgrade system has been converted from Int64 to BitFlags. However because the
741 //names of upgrades are saved to preserve order reassignments (inserting a new upgrade in
742 //the INI file will skew the bit values), we must continue saving the names of the upgrades
743 //in order to recalculate the actual bit value of said upgrade.
744 //---------------------------------------------------------------------------------------------
745 //NOTE: The xfer code didn't have to change with the bitset upgrades, because either way, we're
746 //converting data <-> Ascii, so the minor syntax works with the before and after code!
747
748 // check which type of xfer we're doing
749 if( getXferMode() == XFER_SAVE )
750 {
751 AsciiString upgradeName;
752
753 // count how many bits are set in the mask
754 UnsignedShort count = 0;
755 UpgradeTemplate *upgradeTemplate;
756 for( upgradeTemplate = TheUpgradeCenter->firstUpgradeTemplate(); upgradeTemplate; upgradeTemplate = upgradeTemplate->friend_getNext() )
757 {
758 // if the mask of this upgrade is set, it counts
759 if( upgradeMaskData->testForAll( upgradeTemplate->getUpgradeMask() ) )
760 {
761 count++;
762 }
763 } // end for, upgradeTemplate
764
765 // write the count
766 xferUnsignedShort( &count );
767
768 // write out the upgrades as strings
769 for( upgradeTemplate = TheUpgradeCenter->firstUpgradeTemplate(); upgradeTemplate; upgradeTemplate = upgradeTemplate->friend_getNext() )
770 {
771 // if the mask of this upgrade is set, it counts
772 if( upgradeMaskData->testForAll( upgradeTemplate->getUpgradeMask() ) )
773 {
774 upgradeName = upgradeTemplate->getUpgradeName();
775 xferAsciiString( &upgradeName );
776 } // end if
777 } // end for, upgradeTemplate
778 } // end if, save
779 else if( getXferMode() == XFER_LOAD )
780 {
781 AsciiString upgradeName;
782 const UpgradeTemplate *upgradeTemplate;
783
784 // how many strings are we going to read from the file
785 UnsignedShort count;
786 xferUnsignedShort( &count );
787
788 // zero the mask data
789 upgradeMaskData->clear();
790
791 // read all the strings and set the mask vaules
792 for( UnsignedShort i = 0; i < count; ++i )
793 {
794
795 // read the string
796 xferAsciiString( &upgradeName );
797
798 // find this upgrade template
799 upgradeTemplate = TheUpgradeCenter->findUpgrade( upgradeName );
800 if( upgradeTemplate == NULL )
801 {
802
803 DEBUG_CRASH(( "Xfer::xferUpgradeMask - Unknown upgrade '%s'\n", upgradeName.str() ));
804 throw XFER_UNKNOWN_STRING;
805
806 } // end if
807
808 // set the mask data
809 upgradeMaskData->set( upgradeTemplate->getUpgradeMask() );
810
811 } // end for i
812
813 } // end else if, load
814 else if( getXferMode() == XFER_CRC )
815 {
816
817 // just xfer implementation the data itself
818 xferImplementation( upgradeMaskData, sizeof( UpgradeMaskType ) );
819
820 } // end else if, crc
821 else
822 {
823
824 DEBUG_CRASH(( "xferUpgradeMask - Unknown xfer mode '%d'\n", getXferMode() ));
825 throw XFER_MODE_UNKNOWN;
826
827 } // end else
828
829} // end xferUpgradeMask
