88Bool SupplyCenterDockUpdate::action( Object* docker, Object *drone )
89{
90 const SupplyCenterDockUpdateModuleData *data = getSupplyCenterDockUpdateModuleData();
91 SupplyTruckAIInterface* supplyTruckAI = NULL;
92 if( docker->getAIUpdateInterface() == NULL )
93 return FALSE;
94
95 supplyTruckAI = docker->getAIUpdateInterface()->getSupplyTruckAIInterface();
96
97 DEBUG_ASSERTCRASH( supplyTruckAI != NULL, ("Something Docking with a Supply Center must have a Supply-truck like AIUpdate") );
98 if( supplyTruckAI == NULL )
99 return FALSE;
100
101 UnsignedInt value = 0;
102 Player *ownerPlayer = getObject()->getControllingPlayer();
103 while( supplyTruckAI->loseOneBox() )
104 value += ownerPlayer->getSupplyBoxValue();
105
106 // Add money boost from upgrades that give extra money
107 value += supplyTruckAI->getUpgradedSupplyBoost();
108
109 if( value > 0)
110 {
111 Money *ownerPlayerMoney = ownerPlayer->getMoney();
112 ownerPlayerMoney->deposit(value);
113 ownerPlayer->getScoreKeeper()->addMoneyEarned(value);
114
115
116 if( data->m_grantTemporaryStealthFrames > 0 )
117 {
118 StealthUpdate *stealth = docker->getStealth();
119 //Only grant temporary stealth to the default stealth update. It's
120 //possible that another type of stealth was granted... like the
121 //GPS scrambler. We want that to take precendence.
122 if( getObject()->testStatus( OBJECT_STATUS_STEALTHED ) )
123 {
124 if( !stealth )
125 {
126 DEBUG_CRASH( ("SupplyCenterDockUpdate::action() -- It shouldn't be possible for a unit to be OBJECT_STATUS_STEALTHED without a StealthUpdate module!") );
127 }
128 else if( stealth->isTemporaryGrant() || !docker->testStatus( OBJECT_STATUS_CAN_STEALTH ) )
129 {
130 stealth->receiveGrant( TRUE, data->m_grantTemporaryStealthFrames );
131 }
132 }
133 }
134 }
135
136 Bool displayMoney = value > 0 ? TRUE : FALSE;
137 if( getObject()->testStatus(OBJECT_STATUS_STEALTHED) )
138 {
139 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
140 if( !getObject()->isLocallyControlled() && !getObject()->testStatus(OBJECT_STATUS_DETECTED) )
141 {
142 displayMoney = FALSE;
143 }
144 }
145
146 if( displayMoney )
147 {
148 // OY LOOK! I AM USING LOCAL PLAYER. Do not put anything other than TheInGameUI->addFloatingText in the block this controls!!!
149 // Setup info for adding a floating text
150 Coord3D pos;
151 const Coord3D *dockerPos;
152 UnicodeString moneys;
153 moneys.format( TheGameText->fetch( "GUI:AddCash" ), value );
154 dockerPos = docker->getPosition();
155 pos.x = dockerPos->x;
156 pos.y = dockerPos->y;
157 pos.z = TheTerrainLogic->getGroundHeight(pos.x, pos.y);//dockerPos->z + docker->getGeometryInfo().getHeight();
158 Color color = ownerPlayer->getPlayerColor() | GameMakeColor( 0, 0, 0, 230 );
159
160 TheInGameUI->addFloatingText(moneys, &pos, color);
161 }
162
163
164 return FALSE;
165}
