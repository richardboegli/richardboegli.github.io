95void TeamRelationMap::xfer( Xfer *xfer )
96{
97
98 // version
99 XferVersion currentVersion = 1;
100 XferVersion version = currentVersion;
101 xfer->xferVersion( &version, currentVersion );
102
103 // team relation count
104 TeamRelationMapType::iterator teamRelationIt;
105 UnsignedShort teamRelationCount = m_map.size();
106 xfer->xferUnsignedShort( &teamRelationCount );
107
108 // team relations
109 TeamID teamID;
110 Relationship r;
111 if( xfer->getXferMode() == XFER_SAVE )
112 {
113
114 // go through all team relations
115 for( teamRelationIt = m_map.begin(); teamRelationIt != m_map.end(); ++teamRelationIt )
116 {
117
118 // write team ID
119 teamID = (*teamRelationIt).first;
120 xfer->xferUser( &teamID, sizeof( TeamID ) );
121
122 // write relationship
123 r = (*teamRelationIt).second;
124 xfer->xferUser( &r, sizeof( Relationship ) );
125
126 } // end for
127
128 } // end if, save
129 else
130 {
131
132 for( UnsignedShort i = 0; i < teamRelationCount; ++i )
133 {
134
135 // read team ID
136 xfer->xferUser( &teamID, sizeof( TeamID ) );
137
138 // read relationship
139 xfer->xferUser( &r, sizeof( Relationship ) );
140
141 // assign relationship
142 m_map[teamID] = r;
143
144 } // end for, i
145
146 } // end else load
147
148} // end xfer

434{
435
436 // version
437 XferVersion currentVersion = 1;
438 XferVersion version = currentVersion;
439 xfer->xferVersion( &version, currentVersion );
440
441 // unique team ID counter
442 xfer->xferUser( &m_uniqueTeamID, sizeof( TeamID ) );
443
444 // how many team prototypes of data do we have to write
445 UnsignedShort prototypeCount = m_prototypes.size();
446 xfer->xferUnsignedShort( &prototypeCount );
447
448 //
449 // prototypes cannot change in number during run time so the count should be the
450 // same as that already loaded into us from a map load
451 //
452 if( prototypeCount != m_prototypes.size() )
453 {
454
455 DEBUG_CRASH(( "TeamFactory::xfer - Prototype count mismatch '%d should be '%d'\n",
456 prototypeCount, m_prototypes.size() ));
457 throw SC_INVALID_DATA;
458
459 } // end if
460
461 // xfer each of the prototype information
462 TeamPrototypeMap::iterator it;
463 TeamPrototypeID teamPrototypeID;
464 TeamPrototype *teamPrototype;
465 AsciiString prototypeName;
466 if( xfer->getXferMode() == XFER_SAVE )
467 {
468
469 // iterate each prototype and xfer if it needs to be in the save file
470 for( it = m_prototypes.begin(); it != m_prototypes.end(); ++it )
471 {
472
473 // get prototype
474 teamPrototype = it->second;
475
476 // xfer prototype id
477 teamPrototypeID = teamPrototype->getID();
478 xfer->xferUser( &teamPrototypeID, sizeof( TeamPrototypeID ) );
479
480 // xfer prototype data
481 xfer->xferSnapshot( teamPrototype );
482
483 } //end for, it
484
485 } // end if, saving
486 else
487 {
488
489 // read all the team prototype info

492
493 // read prototype ID

496 // find the prototype

498
499 // sanity

503 DEBUG_CRASH(( "TeamFactory::xfer - Unable to find team prototype by id\n" ));

506 } // end if

1127// ------------------------------------------------------------------------
1128Bool TeamPrototype::evaluateProductionCondition(void)
1129{
1130 if (m_productionConditionAlwaysFalse) {
1131 // Set if we don't have a script.
1132 return false;
1133 }
1134 if (m_productionConditionScript) {
1135 // If we are doing peridic evaluation, check the frame.
1136 if (TheGameLogic->getFrame()<m_productionConditionScript->getFrameToEvaluate()) {
1137 return false;
1138 }
1139 Int delaySeconds = m_productionConditionScript->getDelayEvalSeconds();
1140
1141 if (delaySeconds>0) {
1142 m_productionConditionScript->setFrameToEvaluate(TheGameLogic->getFrame()+delaySeconds*LOGICFRAMES_PER_SECOND);
1143 }
1144 return TheScriptEngine->evaluateConditions(m_productionConditionScript, NULL, getControllingPlayer());
1145 }
1146 // We don't have a script yet, so check for one.
1147 if (m_teamTemplate.m_productionCondition.isEmpty()) {
1148 // No script defined.
1149 m_productionConditionAlwaysFalse = true;
1150 return false;
1151 }
1152 const Script *pScript = TheScriptEngine->findScriptByName(m_teamTemplate.m_productionCondition);
1153 if (pScript) {
1154 // Check difficulty.
1155 switch (getControllingPlayer()->getPlayerDifficulty() ) {
1156 case DIFFICULTY_EASY:
1157 if (!pScript->isEasy()) {
1158 m_productionConditionAlwaysFalse = true;
1159 return false;
1160 }
1161 break;
1162 case DIFFICULTY_NORMAL:
1163 if (!pScript->isNormal()) {
1164 m_productionConditionAlwaysFalse = true;
1165 return false;
1166 }
1167 break;
1168 case DIFFICULTY_HARD:
1169 if (!pScript->isHard()) {
1170 m_productionConditionAlwaysFalse = true;
1171 return false;
1172 }
1173 break;
1174 }
1175
1176 // Make a copy of the script locally, just for paranoia's sake. We can't be sure
1177 // exactly what order the teams & scripts will get reset, so be safe.
1178 m_productionConditionScript = pScript->duplicate();
1179 return TheScriptEngine->evaluateConditions(m_productionConditionScript, NULL, getControllingPlayer());
1180 }
1181 // Couldn't find a script.
1182 m_productionConditionAlwaysFalse = true;
1183 return false;
1184}

1255

1260

1807void Team::updateState(void)
1808{
1809 m_enteredOrExited = false;
1810 if (!m_active) {
1811 return;
1812 }
1813 const TeamTemplateInfo *pInfo = m_proto->getTemplateInfo();
1814 if (m_created)
1815 {
1816 m_created = false;
1817 // Run the on create script, if any.
1818 if (!pInfo->m_scriptOnCreate.isEmpty())
1819 {
1820 TheScriptEngine->runScript(pInfo->m_scriptOnCreate, this);
1821 }
1822
1823 // Set up info for the onDestroyed script, if needed.
1824 if (!pInfo->m_scriptOnDestroyed.isEmpty() )
1825 {
1826 for (DLINK_ITERATOR<Object> iter = iterate_TeamMemberList(); !iter.done(); iter.advance())
1827 {
1828 m_curUnits++;
1829 }
1830 m_destroyThreshold = m_curUnits - (m_curUnits * pInfo->m_destroyedThreshold);
1831 if (m_destroyThreshold>m_curUnits-1) m_destroyThreshold = m_curUnits-1;
1832 if (m_destroyThreshold<0) m_destroyThreshold = 0;
1833 }
1834 }
1835
1836 // Do enemy sighted/on clear checks.
1837 if (m_checkEnemySighted) {
1838 m_prevSeeEnemy = m_seeEnemy;
1839 m_seeEnemy = false;
1840 Bool anyAliveInTeam = false; // If we're all dead, don't do all clear.
1841 // only consider enemies.
1842 for (DLINK_ITERATOR<Object> iter = iterate_TeamMemberList(); !iter.done(); iter.advance())
1843 {
1844 if (iter.cur()->isEffectivelyDead())
1845 continue;
1846
1847 PartitionFilterRelationship filterTeam(iter.cur(), PartitionFilterRelationship::ALLOW_ENEMIES);
1848
1849 // and only stuff that is not dead
1850 PartitionFilterAlive filterAlive;
1851 PartitionFilterSameMapStatus filterMapStatus(iter.cur());
1852
1853 PartitionFilter *filters[] = { &filterTeam, &filterAlive, &filterMapStatus, NULL };
1854 Real visionRange = iter.cur()->getVisionRange();
1855 anyAliveInTeam = true;
1856 Object *pObj = ThePartitionManager->getClosestObject( iter.cur(), visionRange,
1857 FROM_CENTER_2D, filters );
1858 if (pObj) {
1859 m_seeEnemy = true;
1860 break;
1861 }
1862 }
1863 if (anyAliveInTeam) {
1864 if (m_prevSeeEnemy != m_seeEnemy)
1865 {
1866 if (m_seeEnemy)
1867 {
1868 // fire onEnemySighted
1869 TheScriptEngine->runScript(pInfo->m_scriptOnEnemySighted, this);
1870 } else {
1871 // fire on all clear.
1872 TheScriptEngine->runScript(pInfo->m_scriptOnAllClear, this);
1873 }
1874 }
1875 }
1876 }
1877
1878 // Do onDestroyed checks.
1879 if (!pInfo->m_scriptOnDestroyed.isEmpty())
1880 {
1881 Int prevUnits = m_curUnits;
1882 m_curUnits = 0;
1883 for (DLINK_ITERATOR<Object> iter = iterate_TeamMemberList(); !iter.done(); iter.advance())
1884 {
1885 if (iter.cur()->isEffectivelyDead())
1886 continue;
1887
1888 m_curUnits++;
1889 }
1890 if (m_curUnits != prevUnits && m_curUnits <= m_destroyThreshold)
1891 {
1892 TheScriptEngine->runScript(pInfo->m_scriptOnDestroyed, this);
1893 m_destroyThreshold = -1; // don't trigger again.
1894 }
1895 }
1896
1897 // Do onIdle checks.
1898 if (!pInfo->m_scriptOnIdle.isEmpty())
1899 {
1900 Bool isIdle = true;
1901 Bool anyAliveInTeam = false; // If we're all dead, don't do all clear.
1902 for (DLINK_ITERATOR<Object> iter = iterate_TeamMemberList(); !iter.done(); iter.advance())
1903 {
1904 if (iter.cur()->isEffectivelyDead()) {
1905 continue;
1906 }
1907 AIUpdateInterface *ai = iter.cur()->getAIUpdateInterface();
1908 if (!ai) continue;
1909 anyAliveInTeam = true;
1910 if (!ai->isIdle()) {
1911 isIdle = false;
1912 }
1913 }
1914 if (anyAliveInTeam && isIdle && m_wasIdle)
1915 {
1916 TheScriptEngine->runScript(pInfo->m_scriptOnIdle, this);
1917 }
1918 m_wasIdle = isIdle;
1919 }
