69static void saturateRGB(RGBColor& color, Real factor)
70{
71 color.red *= factor;
72 color.green *= factor;
73 color.blue *= factor;
74
75 Real halfFactor = factor * 0.5f;
76
77 color.red -= halfFactor;
78 color.green -= halfFactor;
79 color.blue -= halfFactor;
80
81}

134UpdateSleepTime EMPUpdate::update( void )
135{
137
138 Object *obj = getObject();
139
140 const EMPUpdateModuleData *data = getEMPUpdateModuleData();
141 Drawable *dr = obj->getDrawable();
142 UnsignedInt now = TheGameLogic->getFrame();
143
144 m_currentScale += ( m_targetScale - m_currentScale ) * 0.05f;
145 dr->setInstanceScale( m_currentScale );
146
147 if ( now < m_tintEnvPlayFrame)
148 {
149 RGBColor start = data->m_startColor;
150 saturateRGB( start, 2 );
151 dr->colorTint( &start );
152 }
153 if ( now == m_tintEnvPlayFrame)
154 {
155 RGBColor end = data->m_endColor;
156 saturateRGB( end, 5 );
157 dr->colorFlash( &end, 9999, m_tintEnvFadeFrames, TRUE );
158 doDisableAttack();
159 }
160
161 //Real curSpin = obj->getOrientation();
162 //curSpin += m_spinRate;
163 //curSpin = normalizeAngle(curSpin);
164 //obj->setOrientation( curSpin );
165
166 if( now >= m_dieFrame )
167 obj->kill();
168
169 return UPDATE_SLEEP_NONE;
170}

174void EMPUpdate::doDisableAttack( void )
175{
176 Object *object = getObject();
177 const EMPUpdateModuleData *data = getEMPUpdateModuleData();
178 if( !object || !data )
179 return; //sanity
180
181 Real radius = data->m_effectRadius;
182 Real curVictimDistSqr;
183 const Coord3D *pos = object->getPosition();
184
185 //Kris -- October 28, 2003 -- Patch 1.01
186 //If the EMP hits an airborne target, then don't allow the EMP
187 //blast to effect anything on the ground.
188 Object *producer = TheGameLogic->findObjectByID( object->getProducerID() );
189 Object *intendedVictim = NULL;
190 Bool onlyEffectAirborne = FALSE;
191 Bool intendedVictimProcessed = FALSE;
192 if( producer && producer->getAI() )
193 {
194 intendedVictim = producer->getAI()->getCurrentVictim();
195 if( intendedVictim && intendedVictim->isAirborneTarget() )
196 {
197 onlyEffectAirborne = TRUE;
198 }
199 }
200
201 SimpleObjectIterator *iter;
202 Object *curVictim;
203
204 if (radius > 0.0f)
205 {
206 iter = ThePartitionManager->iterateObjectsInRange(pos,
207 radius, FROM_BOUNDINGSPHERE_3D);
208
209 curVictim = iter->firstWithNumeric(&curVictimDistSqr);
210 }
211
212 MemoryPoolObjectHolder hold(iter);
213
214 for ( ; curVictim != NULL; curVictim = iter ? iter->nextWithNumeric(&curVictimDistSqr) : NULL)
215 {
216 if ( curVictim != object)
217 {
218
219 //Kris -- October 28, 2003 -- Patch 1.01
220 //If the EMP hits an airborne target, then don't allow the EMP
221 //blast to effect anything on the ground.
222 if( onlyEffectAirborne && !curVictim->isAirborneTarget() )
223 {
224 continue;
225 }
226
227 //Some EMP attacks don't affect our own buildings.
228 if( data->m_doesNotAffectMyOwnBuildings && curVictim->isKindOf( KINDOF_STRUCTURE ) )
229 {
230 if( curVictim->getControllingPlayer() == object->getControllingPlayer() )
231 {
232 continue;
233 }
234 }
235
236
237
238
242
243
244
245
246
247
248
249
250 if ( !curVictim->isKindOf( KINDOF_VEHICLE ) && !curVictim->isKindOf(KINDOF_STRUCTURE) && !curVictim->isKindOf(KINDOF_SPAWNS_ARE_THE_WEAPONS) )
251 {
252 //DONT DISABLE PEOPLE, EXCEPT FOR STINGER SOLDIERS
253 continue;
254 }
255 else if ( curVictim->isKindOf( KINDOF_AIRCRAFT ) && curVictim->isAirborneTarget() )// is in the sky
256 {
257 // WITHIN THE SET OF ALL FLYING THINGS, WE WANT TO EXEMPT SUPERWEAPON TRANSPORTS
258// if ( curVictim->isKindOf( KINDOF_TRANSPORT ) ) // is transport kindof
259// if ( curVictim->getContain() ) // does carry stuff
260// if ( curVictim->getContain()->getContainCount() > 0 ) // is carrying something
261// if ( ! curVictim->isKindOf( KINDOF_PRODUCED_AT_HELIPAD ) ) // but not a helicopter
262// continue;
263
264 if ( curVictim->isKindOf( KINDOF_EMP_HARDENED ) ) // self-explanitory
265 continue;
266
267 curVictim->kill();// @todo this should use some sort of DEADSTICK DIE or something...
268 Drawable *drw = curVictim->getDrawable();
269 if ( drw )
270 {
271 drw->setTintStatus( TINT_STATUS_DISABLED );// paint it black
272 }
273 continue;
274 }
275 else if ( curVictim->isKindOf( KINDOF_STRUCTURE ) )
276 {
277 if ( ! curVictim->isFactionStructure() )
278 continue;
279 }
280 // handle cases where we do not want allies to be hit by it's own EMP weapons
281 else if ( (data->m_rejectMask & WEAPON_AFFECTS_ALLIES) && curVictim->getRelationship( object ) == ALLIES)
282 {
283 continue;
284 }
285
286 //Disable the target for a specified amount of time.
287 curVictim->setDisabledUntil( DISABLED_EMP, TheGameLogic->getFrame() + data->m_disabledDuration );
288
289 //Kris -- October 28, 2003 -- Patch 1.01
290 if( intendedVictim == curVictim )
291 {
292 //We know the intended victim was in range. This will catch an edge case
293 //where the intended target is hit, but the range is off enough to not effect it.
294 intendedVictimProcessed = TRUE;
295 }
296
297 Drawable *drw = curVictim->getDrawable();
298 if ( drw )
299 {
300
301 const ParticleSystemTemplate *tmp = data->m_disableFXParticleSystem;
302 if (tmp)
303 {
304 Real victimHeight = curVictim->getGeometryInfo().getMaxHeightAbovePosition();
305 Real victimFootprintArea = curVictim->getGeometryInfo().getFootprintArea();
306 Real victimVolume = victimFootprintArea * MIN(victimHeight, 10.0f);
307
308 UnsignedInt emitterCount = MAX(15, REAL_TO_INT_CEIL(data->m_sparksPerCubicFoot * victimVolume));
309
310 for (Int e = 0 ; e < emitterCount; ++e)
311 {
312
313 ParticleSystem *sys = TheParticleSystemManager->createParticleSystem(tmp);
314
315 if (sys)
316 {
317 Coord3D offs = {0,0,0};
318 curVictim->getGeometryInfo().makeRandomOffsetWithinFootprint( offs );
319 offs.z = GameLogicRandomValue(3, victimHeight);
320
321 //This puts all the sparks within a quadrahemicycloid (rectangular dome) volume
322 //The same shape as a four cornered camping dome tent, for those with less Greek
323 if (offs.length() > victimHeight)
324 {
325 Real resoreX = offs.x;
326 Real resoreY = offs.y;
327 offs.normalize();
328 offs.z *= victimHeight;
329 offs.x = resoreX;
330 offs.y = resoreY;
331 }
332
333 sys->attachToObject(curVictim);
334 sys->setPosition( &offs );
335 sys->setSystemLifetime(MAX(0, data->m_disabledDuration - 30));
336 sys->setInitialDelay(GameLogicRandomValue(1,100));
337 }
338 }
339 }
340 }
341
342 }
343 }
344
345 //Kris -- October 28, 2003 -- Patch 1.01
346 //Handle edge case when the EMP explodes, but "misses" the intended target.
347 if( intendedVictim && !intendedVictimProcessed && intendedVictim->isKindOf( KINDOF_AIRCRAFT ) )
348 {
349 if( !intendedVictim->isKindOf( KINDOF_EMP_HARDENED ) )
350 {
351 //Victim position
352 Coord3D coord;
353 coord.set( intendedVictim->getPosition() );
354 //Subtract this object (distance from missile to victim's previous position)
355 coord.sub( pos );
356
357 Real lengthSqr = coord.lengthSqr();
358 if( lengthSqr <= radius * 2.0f || lengthSqr <= 40.0f * 40.0f )
359 {
360 //Disable the target for a specified amount of time.
361 intendedVictim->setDisabledUntil( DISABLED_EMP, TheGameLogic->getFrame() + data->m_disabledDuration );
362 }
363 }
364 }
365}

449UpdateSleepTime LeafletDropBehavior::update( void )
450{
451
452 if ( ! m_fxFired )
453 {
454 // start shoveling out those leaflets, boys.
