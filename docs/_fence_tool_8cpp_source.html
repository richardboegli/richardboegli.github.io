58void FenceTool::updateMapObjectList(Coord3D downPt, Coord3D curPt, WbView* pView, CWorldBuilderDoc *pDoc, Bool checkPlayers)
59{
60 Bool shiftKey = (0x8000 & ::GetAsyncKeyState(VK_SHIFT))!=0;
61 Real angle = ObjectTool::calcAngle(downPt, curPt, pView);
62 Coord3D delta;
63 Coord3D normalDelta;
64 delta.x = curPt.x-downPt.x;
65 delta.y = curPt.y-downPt.y;
66 delta.z = 0;
67 Real length = delta.length();
68 Int newCount = floor(length/m_curObjectWidth);
69 if (newCount<1) newCount = 1;
70 if (newCount > MAX_OBJECTS) newCount = MAX_OBJECTS;
71 if (shiftKey) {
72 // stretch the fence spacing. Keep the same count.
73 delta.x /= (m_objectCount*m_curObjectWidth);
74 delta.y /= (m_objectCount*m_curObjectWidth);
75 } else {
76 // Use exactly the spacing.
77 m_objectCount = newCount;
78 delta.normalize();
79 }
80 normalDelta = delta;
81 if (angle==0) {
82 normalDelta.x = 1;
83 normalDelta.y = 0;
84 }
85 normalDelta.normalize();
86
87 Int i;
88 if (m_mapObjectList == NULL) return;
89 MapObject *pCurObj = m_mapObjectList;
90 for (i=1; i<m_objectCount; i++) {
91 if (pCurObj->getNext() == NULL) {
92 pCurObj->setNextMap(ObjectOptions::duplicateCurMapObjectForPlace(&downPt, angle, checkPlayers));
93 }
94 pCurObj=pCurObj->getNext();
95 if (pCurObj == NULL) return;
96 }
97 WbView3d *p3View = pDoc->GetActive3DView();
98 MapObject *pXtraObjects = pCurObj->getNext();
99 pCurObj->setNextMap(NULL);
100 if (pXtraObjects) {
101 p3View->removeFenceListObjects(pXtraObjects);
102 pXtraObjects->deleteInstance();
103 pXtraObjects = NULL;
104 }
105
106 pCurObj = m_mapObjectList;
107 for (i=0; i<m_objectCount; i++) {
108 Real factor = m_curObjectWidth*(i);
109 Coord3D curLoc = downPt;
110 curLoc.x += factor*delta.x;
111 curLoc.y += factor*delta.y;
112 curLoc.x += m_curObjectOffset*normalDelta.x;
113 curLoc.y += m_curObjectOffset*normalDelta.y;
114 curLoc.z = MAGIC_GROUND_Z;
115
116 pCurObj->setAngle(angle);
117 pCurObj->setLocation(&curLoc);
118 pCurObj=pCurObj->getNext();
119 }
120
121 p3View->updateFenceListObjects(m_mapObjectList);
122
123}

148void FenceTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
149{
150 if (m != TRACK_L) return;
151
152 Coord3D cpt;
153 pView->viewToDocCoords(viewPt, &cpt);
154 pView->snapPoint(&cpt);
155 m_downPt2d = viewPt;
156 m_downPt3d = cpt;
157 if (m_mapObjectList) {
158 m_mapObjectList->deleteInstance();
159 m_mapObjectList = NULL;
160 }
161 if (FenceOptions::hasSelectedObject()) {
162 FenceOptions::update();
163 m_curObjectWidth = FenceOptions::getFenceSpacing();
164 m_curObjectOffset = FenceOptions::getFenceOffset();
165 m_mapObjectList = ObjectOptions::duplicateCurMapObjectForPlace(&m_downPt3d, 0, false);
166 m_objectCount = 1;
167 mouseMoved(m, viewPt, pView, pDoc);
168 }

172void FenceTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
173{
174// Bool justAClick = true;
175 Coord3D cpt;
176 pView->viewToDocCoords(viewPt, &cpt); // Do constrain.
177 WbView3d *p3View = pDoc->GetActive3DView();
178 Coord3D loc = cpt;
179 pView->snapPoint(&loc);
180 Real angle = 0 ;
181 if (m == TRACK_L) { // Mouse is down, so fence.
182 p3View->setObjTracking(NULL, loc, angle, false);
183 updateMapObjectList(m_downPt3d,loc, pView, pDoc, false);
184 return;
185 }
186 MapObject *pCur = ObjectOptions::getObjectNamed(AsciiString(ObjectOptions::getCurObjectName()));
187 p3View->setObjTracking(NULL, m_downPt3d, 0, false);
188 loc.z = ObjectOptions::getCurObjectHeight();
189 if (pCur && FenceOptions::hasSelectedObject()) {
190 // Display the transparent version of this object.
191 m_curObjectOffset = FenceOptions::getFenceOffset();
192 loc.x += m_curObjectOffset;
193 p3View->setObjTracking(pCur, loc, angle, true);
194 } else {
195 // Don't display anything.
196 p3View->setObjTracking(NULL, loc, angle, false);
197 }
198}

201void FenceTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
202{
203 if (m != TRACK_L) return;
204 Coord3D cpt;
205 pView->viewToDocCoords(viewPt, &cpt); // Do constrain.
206 Coord3D loc = cpt;
207 pView->snapPoint(&loc);
208 updateMapObjectList(m_downPt3d, loc, pView, pDoc, true);
209 WbView3d *p3View = pDoc->GetActive3DView();
210 if (m_mapObjectList) {
211 p3View->removeFenceListObjects(m_mapObjectList);
212 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, m_mapObjectList);
213 pDoc->AddAndDoUndoable(pUndo);
214 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
215 m_mapObjectList = NULL; // undoable owns it now.
216 }
