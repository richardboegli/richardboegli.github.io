46/*
47** This is an abstract interface class for a graphic surface. Graphic operations will use this
48** interface to perform their function. The philosphy behind this interface is that it represents
49** a small but useful set of functions. Emphasis is placed on supporting those functions which are
50** likely to have hardware assist.
51*/
52class Surface
53{
54 public:
55 Surface(int width, int height) : Width(width), Height(height) {}
56 virtual ~Surface(void) {};
57
58 /*
59 ** Copies regions from one surface to another.
60 */
61 virtual bool Blit_From(Rect const & dcliprect, Rect const & destrect, Surface const & source, Rect const & scliprect, Rect const & sourcerect, bool trans=false) = 0;
62 virtual bool Blit_From(Rect const & destrect, Surface const & source, Rect const & sourcerect, bool trans=false) = 0;
63 virtual bool Blit_From(Surface const & source, bool trans=false) = 0;
64
65 /*
66 ** Fills a region with a constant color.
67 */
68 virtual bool Fill_Rect(Rect const & rect, int color) = 0;
69 virtual bool Fill_Rect(Rect const & cliprect, Rect const & fillrect, int color) = 0;
70 virtual bool Fill(int color) = 0;
71
72 /*
73 ** Fetches and stores a pixel to the display (pixel is in surface format).
74 */
75 virtual bool Put_Pixel(Point2D const & point, int color) = 0;
76 virtual int Get_Pixel(Point2D const & point) const = 0;
77
78 /*
79 ** Draws lines onto the surface.
80 */
81 virtual bool Draw_Line(Point2D const & startpoint, Point2D const & endpoint, int color) = 0;
82 virtual bool Draw_Line(Rect const & cliprect, Point2D const & startpoint, Point2D const & endpoint, int color) = 0;
83
84 /*
85 ** Draws rectangle onto the surface.
86 */
87 virtual bool Draw_Rect(Rect const & rect, int color) = 0;
88 virtual bool Draw_Rect(Rect const & cliprect, Rect const & rect, int color) = 0;
89
90 /*
91 ** Gets and frees a direct pointer to the video memory.
92 */
93 virtual void * Lock(Point2D point = Point2D(0, 0)) const = 0;
94 virtual bool Unlock(void) const = 0;
95 virtual bool Is_Locked(void) const = 0;
96
97 /*
98 ** Queries information about the surface.
99 */
100 virtual int Bytes_Per_Pixel(void) const = 0;
101 virtual int Stride(void) const = 0;
102 virtual Rect Get_Rect(void) const {return(Rect(0, 0, Width, Height));}
103 virtual int Get_Width(void) const {return(Width);}
104 virtual int Get_Height(void) const {return(Height);}
105
106 /*
107 ** Hack function to serve the purpose that RTTI was invented for, but since
108 ** the Watcom compiler doesn't support RTTI, we must resort to using this
109 ** alternative.
110 */
111 virtual bool Is_Direct_Draw(void) const {return(false);}
112
113 protected:
114
115 /*
116 ** Records logical pixel dimensions of the surface.
117 */
118 int Width;
119 int Height;
120};
