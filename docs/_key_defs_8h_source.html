233// state for keyboard IO ------------------------------------------------------
234enum
235{
236 KEY_STATE_NONE = 0x0000, // No modifier state
237 KEY_STATE_UP = 0x0001, // Key is up (default state)
238 KEY_STATE_DOWN = 0x0002, // Key is down
239 KEY_STATE_LCONTROL = 0x0004, // Left control is pressed
240 KEY_STATE_RCONTROL = 0x0008, // Right control is pressed
241 KEY_STATE_LSHIFT = 0x0010, // left shift is pressed
242 KEY_STATE_RSHIFT = 0x0020, // right shift is pressed
243 KEY_STATE_LALT = 0x0040, // left alt is pressed
244 KEY_STATE_RALT = 0x0080, // right alt is pressed
245 KEY_STATE_AUTOREPEAT = 0x0100, // Key is down due to autorepeat (only seen in conjunction with KEY_STATE_DOWN)
246 KEY_STATE_CAPSLOCK = 0x0200, // Caps Lock key is on.
247 KEY_STATE_SHIFT2 = 0x0400, // Alternate shift key is pressed (I think this is for foreign keyboards..)
248
249 // modifier combinations when left/right isn't a factor
250 KEY_STATE_CONTROL = (KEY_STATE_LCONTROL | KEY_STATE_RCONTROL),
251 KEY_STATE_SHIFT = (KEY_STATE_LSHIFT | KEY_STATE_RSHIFT | KEY_STATE_SHIFT2 ),
252 KEY_STATE_ALT = (KEY_STATE_LALT | KEY_STATE_RALT)
253
254}; // end KeyStateType
