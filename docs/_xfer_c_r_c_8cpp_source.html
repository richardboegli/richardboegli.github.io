98void XferCRC::addCRC( UnsignedInt val )
99{
100 int hibit;
101
102 val = htonl(val);
103
104 if (m_crc & 0x80000000)
105 {
106 hibit = 1;
107 }
108 else
109 {
110 hibit = 0;
111 }
112
113 m_crc <<= 1;
114 m_crc += val;
115 m_crc += hibit;
116
117} // end addCRC

140void XferCRC::xferImplementation( void *data, Int dataSize )
141{
142
143 if (!data || dataSize < 1)
144 {
145 return;
146 }
147
148 const UnsignedInt *uintPtr = (const UnsignedInt *) (data);
149
150 for (Int i=0 ; i<dataSize/4 ; i++)
151 {
152 addCRC (*uintPtr++);
153 }
154
155 int leftover = dataSize & 3;
156 if (leftover)
157 {
158 UnsignedInt val = 0;
159 const unsigned char *c = (const unsigned char *)uintPtr;
160 for (Int i=0; i<leftover; i++)
161 {
162 val += (c[i] << (i*8));
163 }
164 val = htonl(val);
165 addCRC (val);
166 }
167
168} // end xferImplementation

216void XferDeepCRC::open( AsciiString identifier )
217{
218
219 m_xferMode = XFER_SAVE;
220
221 // sanity, check to see if we're already open
222 if( m_fileFP != NULL )
223 {
224
225 DEBUG_CRASH(( "Cannot open file '%s' cause we've already got '%s' open\n",
226 identifier.str(), m_identifier.str() ));
227 throw XFER_FILE_ALREADY_OPEN;
228
229 } // end if
230
231 // call base class
232 Xfer::open( identifier );
233
234 // open the file
235 m_fileFP = fopen( identifier.str(), "w+b" );
236 if( m_fileFP == NULL )
237 {
238
239 DEBUG_CRASH(( "File '%s' not found\n", identifier.str() ));
240 throw XFER_FILE_NOT_FOUND;
241
242 } // end if
243
244 // initialize CRC to brand new one at zero
245 m_crc = 0;
246
247} // end open

276void XferDeepCRC::xferImplementation( void *data, Int dataSize )
277{
278
279 if (!data || dataSize < 1)
280 {
281 return;
282 }
283
284 // sanity
285 DEBUG_ASSERTCRASH( m_fileFP != NULL, ("XferSave - file pointer for '%s' is NULL\n",

287
288 // write data to file
289 if( fwrite( data, dataSize, 1, m_fileFP ) != 1 )
290 {
291
292 DEBUG_CRASH(( "XferSave - Error writing to file '%s'\n", m_identifier.str() ));
293 throw XFER_WRITE_ERROR;
294
295 } // end if
296
297 XferCRC::xferImplementation( data, dataSize );
298
299} // end xferImplementation

312void XferDeepCRC::xferAsciiString( AsciiString *asciiStringData )
313{
314
315 // sanity
316 if( asciiStringData->getLength() > 16385 )
317 {
318
319 DEBUG_CRASH(( "XferSave cannot save this ascii string because it's too long. Change the size of the length header (but be sure to preserve save file compatability\n" ));
320 throw XFER_STRING_ERROR;
321
322 } // end if
323
324 // save length of string to follow
325 UnsignedShort len = asciiStringData->getLength();
326 xferUnsignedShort( &len );
327
328 // save string data
329 if( len > 0 )
330 xferUser( (void *)asciiStringData->str(), sizeof( Byte ) * len );

332} // end xferAsciiString
