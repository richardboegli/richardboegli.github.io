51Real ObjectTool::calcAngle(Coord3D downPt, Coord3D curPt, WbView* pView)
52{
53 enum {HYSTERESIS = 3};
54 double dx = curPt.x - downPt.x;
55 double dy = curPt.y - downPt.y;
56 double dist = sqrt(dx*dx+dy*dy);
57 double angle = 0;
58 if (dist < 0.1) // check for div-by-zero.
59 {
60 angle = 0;
61 }
62 else if (abs(dx) > abs(dy))
63 {
64 angle = acos( (double)dx / dist);
65 if (dy<0) angle = -angle;
66 }
67 else
68 {
69 angle = asin( ((double)dy) / dist);
70 if (dx<0) angle = PI-angle;
71 }
72 if (angle > PI) angle -= 2*PI;
73#ifdef _DEBUG
74 CString buf;
75 buf.Format("Angle %f rad, %d degrees\n", angle, (int)(angle*180/PI));
76 ::OutputDebugString(buf);
77#endif
78 return((Real)angle);
79}

114void ObjectTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
115{
116 Bool justAClick = true;
117 Coord3D cpt;
118 pView->viewToDocCoords(viewPt, &cpt, false); // Don't constrain.
119 Coord3D loc = cpt;
120 pView->snapPoint(&loc);
121 if (m == TRACK_L) { // Mouse is down, so preview the angle if > hysteresis.
122 // always check hysteresis in view coords.
123 justAClick = (abs(viewPt.x - m_downPt2d.x)<HYSTERESIS || abs(viewPt.x - m_downPt2d.x)<HYSTERESIS);
124 loc = m_downPt3d;
125 }
126 MapObject *pCur = ObjectOptions::getObjectNamed(AsciiString(ObjectOptions::getCurObjectName()));
127 Real angle = justAClick ? 0 : calcAngle(loc, cpt, pView);
128 if (justAClick && pCur && pCur->getThingTemplate()) {
129 angle = pCur->getThingTemplate()->getPlacementViewAngle();
130 }
131 WbView3d *p3View = pDoc->GetActive3DView();
132 p3View->setObjTracking(NULL, m_downPt3d, 0, false);
133 loc.z = ObjectOptions::getCurObjectHeight();
134 if (pCur) {
135 // Display the transparent version of this object.
136 p3View->setObjTracking(pCur, loc, angle, true);
137 } else {
138 // Don't display anything.
139 p3View->setObjTracking(NULL, loc, angle, false);
140 }
141}

144void ObjectTool::mouseUp(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
145{
146 if (m != TRACK_L) return;
147
148 // always check hysteresis in view coords.
149 Bool justAClick = (abs(viewPt.x - m_downPt2d.x)<HYSTERESIS || abs(viewPt.x - m_downPt2d.x)<HYSTERESIS);
150
151 Coord3D cpt;
152 pView->viewToDocCoords(viewPt, &cpt, false); // Don't constrain.
153
154 Coord3D loc = m_downPt3d;
155 pView->snapPoint(&loc);
156 loc.z = ObjectOptions::getCurObjectHeight();
157 Real angle = justAClick ? 0 : calcAngle(loc, cpt, pView);
158 MapObject *pNew = ObjectOptions::duplicateCurMapObjectForPlace(&loc, angle, true);
159 if (pNew) {
160 if (justAClick && pNew->getThingTemplate()) {
161 angle = pNew->getThingTemplate()->getPlacementViewAngle();
162 pNew->setAngle(angle);
163 }
164 AddObjectUndoable *pUndo = new AddObjectUndoable(pDoc, pNew);
165 pDoc->AddAndDoUndoable(pUndo);
166 REF_PTR_RELEASE(pUndo); // belongs to pDoc now.
167 pNew = NULL; // undoable owns it now.
168 }
