152//-------------------------------------------------------------------------------------------------
153static Bool calcTrajectory(
154 const Coord3D& start, // in: where the projectile starts
155 const Coord3D& end, // in: where the projectile wants to end up
156 Real velocity, // in: the initial speed of the projectile
157 Real minPitch, // in: min pitch (-PI/2)
158 Real maxPitch, // in: max pitch (-PI/2)
159 Bool preferShortPitch, // in: prefer the shorter or longer path?
160 Real& angle, // out: the angle to aim
161 Real& pitch // out: the pitch to aim for
162)
163{
164 Bool exactTarget = false;
165
166 angle = 0.0f;
167 pitch = 0.0f;
168
169 if (velocity <= 0.0f)
170 {
171 DEBUG_CRASH(("cant get there from here (1)"));
172 return false;
173 }
174
175 Real dx = end.x - start.x;
176 Real dy = end.y - start.y;
177 Real dz = end.z - start.z;
178
179 // calculating the angle is trivial.
180 angle = atan2(dy, dx);
181
182 // calculating the pitch requires a bit more effort.
183 Real horizDistSqr = sqr(dx) + sqr(dy);
184 Real horizDist = sqrt(horizDistSqr);
185
186 // calc the two possible pitches that will cover the given horizontal range.
187 // (this is actually only true if dz==0, but is a good first guess)
188 Real gravity = fabs(TheGlobalData->m_gravity);
189 Real gravityTwoDZ = gravity * 2.0f * dz;
190
191 // let's start by aiming directly for it. we know this isn't right (unless gravity
192 // is zero, which it's not) but is a good starting point...
193 Real theta = atan2(dz, horizDist);
194 // if the angle isn't pretty shallow, we can get a better initial guess by using
195 // the code below...
196 const Real SHALLOW_ANGLE = 0.5f * PI / 180.0f;
197 if (fabs(theta) > SHALLOW_ANGLE)
198 {
199 Real t = horizDist / velocity;
200 Real vz = (dz/t + 0.5f*gravity*t);
201 Real sineOfAngle = clamp(-1.0f, vz / velocity, 1.0f);
202 theta = ASin(sineOfAngle)*0.5f;
203 }
204
205/*
206 this is, in theory, the "right" formula for dz==0, but I
207 get better results with the stuff above:
208
209 Real sineOfAngle = (gravity * horizDist) / sqr(velocity);
210 if (sineOfAngle > 1.0f)
211 {
212 return false;
213 }
214 Real theta = ASin(sineOfAngle)*0.5f;
215*/
216
217 Real pitches[2];
218 Real cosPitches[2];
219 Real sinPitches[2];
220 Real theta_min = max(minPitch, -PI/2);
221 Real theta_max = min(maxPitch, PI/2);
222 const Real MIN_ANGLE_DIFF = (PI/(180.0f*16.0f)); // 1/16th of a degree. yes, we need that accuracy.
223//Int numLoops = 0;
224 while (theta_max > theta_min + MIN_ANGLE_DIFF)
225 {
226//++numLoops;
227 pitches[0] = theta; // shallower angle
228 pitches[1] = (theta >= 0.0) ? (PI/2 - theta) : (-PI/2 - theta); // steeper angle
229
230 DEBUG_ASSERTCRASH(pitches[0]<=PI/2&&pitches[0]>=-PI/2,("bad pitches[0] %f\n",rad2deg(pitches[0])));
231 DEBUG_ASSERTCRASH(pitches[1]<=PI/2&&pitches[1]>=-PI/2,("bad pitches[1] %f\n",rad2deg(pitches[1])));
232
233 // calc the horiz-speed & time for each.
234 // note that time can only be negative for 90<angle<270, and since we
235 // ruled those out above, we're gold.
236 sinPitches[0] = Sin(pitches[0]);
237 sinPitches[1] = Sin(pitches[1]);
238 cosPitches[0] = Cos(pitches[0]);
239 cosPitches[1] = Cos(pitches[1]);
240 Real t0 = (horizDist / (velocity * cosPitches[0]));
241 Real t1 = (horizDist / (velocity * cosPitches[1]));
242
243 t0 = MAX(0,t0);
244 t1 = MAX(0,t1);
245
246
247 DEBUG_ASSERTCRASH(t0>=0&&t1>=0,("neg time"));
248
249 Int preferred = ((t0 < t1) == (preferShortPitch)) ? 0 : 1;
250
251 // ok, NOW... since dz is virtually NEVER zero, do a little approximation
252 // to get a better guess. (solving the equations directly are hard and
253 // it's simpler to approximate)
254 Bool tooClose = false;
255 Real vx, vz, root;
256
257 vz = velocity*sinPitches[preferred];
258
259 root = sqr(vz) - gravityTwoDZ;
260 if (root < 0.0f)
261 {
262 // oops, no solution for our preferred pitch. try the other one.
263 if (preferred == 0)
264 tooClose = true; // if this fails for the shallow case, it's 'cuz the result is too close
265 preferred = 1 - preferred;
266 vz = velocity*sinPitches[preferred];
267 root = sqr(vz) - gravityTwoDZ;
268 if (root < 0.0f)
269 {
270 DEBUG_CRASH(("cant get there from here (2)"));
271 return false;
272 }
273 }
274
275#ifdef ONLY_RETURN_CLIPPED_PITCHES
276 // if we get this far, we have some plausible solution...
277 // it actually may be off, but let's go ahead and set the result
278 // (along with the "exact" flag) so that the caller will want to
279 // try to use the result. (we'll probably iterate and overwrite
280 // this result with a more precise one.)
281 if (within(minPitch, pitches[preferred], maxPitch))
282 {
283 pitch = pitches[preferred];
284 exactTarget = true;
285 }
286#else
287 // if we get this far, we have some plausible solution...
288 // it actually may be off, but let's go ahead and set the result
289 // (along with the "exact" flag) so that the caller will want to
290 // try to use the result. (we'll probably iterate and overwrite
291 // this result with a more precise one.) NOTE however that we
292 // don't validate it's within the proper range... caller must do that!
293 pitch = pitches[preferred];
294 exactTarget = true;
295#endif
296
297 vx = velocity*cosPitches[preferred];
298 Real actualRange = (vx*(vz + sqrt(root)))/gravity;
299 const Real CLOSE_ENOUGH_RANGE = 5.0f;
300 if (tooClose || (actualRange < horizDist - CLOSE_ENOUGH_RANGE))
301 {
302 theta_min = theta;
303 theta = (theta + theta_max) * 0.5f;
304 }
305 else if (actualRange > horizDist + CLOSE_ENOUGH_RANGE)
306 {
307 theta_max = theta;
308 theta = (theta + theta_min) * 0.5f;
309 }
310 else
311 {
312 break;
313 }
314 }
315
316//DEBUG_LOG(("took %d loops to find a match\n",numLoops));
317 if (exactTarget)
318 return true;
319
320 return false;
321}

355//-------------------------------------------------------------------------------------------------
356// The actual firing of the missile once setup. Uses a Bezier curve with points parameterized in ini
357//-------------------------------------------------------------------------------------------------
358void DumbProjectileBehavior::projectileFireAtObjectOrPosition( const Object *victim, const Coord3D *victimPos, const WeaponTemplate *detWeap, const ParticleSystemTemplate* exhaustSysOverride )
359{
360 const DumbProjectileBehaviorModuleData* d = getDumbProjectileBehaviorModuleData();
361 Object* projectile = getObject();
362 Real weaponSpeed = detWeap ? detWeap->getWeaponSpeed() : 0.0f;
363 Real minWeaponSpeed = detWeap ? detWeap->getMinWeaponSpeed() : 0.0f;
364
365 // if an object, aim at the center, not the ground part
366 Coord3D victimPosToUse;
367 if (victim)
368 victim->getGeometryInfo().getCenterPosition(*victim->getPosition(), victimPosToUse);
369 else
370 victimPosToUse = *victimPos;
371
372 if( detWeap && detWeap->isScaleWeaponSpeed() )
373 {
374 // Some weapons want to scale their start speed to the range
375 Real minRange = detWeap->getMinimumAttackRange();
376 Real maxRange = detWeap->getUnmodifiedAttackRange();
377 Real range = sqrt(ThePartitionManager->getDistanceSquared( projectile, &victimPosToUse, FROM_CENTER_2D ) );
378 Real rangeRatio = (range - minRange) / (maxRange - minRange);
379 m_flightPathSpeed = (rangeRatio * (weaponSpeed - minWeaponSpeed)) + minWeaponSpeed;
380 }
381 else
382 {
383 m_flightPathSpeed = weaponSpeed;
384 }
385
386 PhysicsBehavior* physics = projectile->getPhysics();
387 if ( d->m_tumbleRandomly && physics)
388 {
389 physics->setPitchRate( GameLogicRandomValueReal( -1.0f/PI, 1.0f/PI ) );
390 physics->setYawRate( GameLogicRandomValueReal( -1.0f/PI, 1.0f/PI ) );
391 physics->setRollRate( GameLogicRandomValueReal( -1.0f/PI, 1.0f/PI ) );
392 }
393
394 m_flightPathStart = *getObject()->getPosition();
395 m_flightPathEnd = victimPosToUse;
396 if (!calcFlightPath(true))
397 {
398 //Can only fail if wildly incorrect points
399 TheGameLogic->destroyObject( projectile );
400 return;
401 }
402 m_currentFlightPathStep = 0;// We are at the first point, because the launching put us there
403}

406//-------------------------------------------------------------------------------------------------
407Bool DumbProjectileBehavior::calcFlightPath(Bool recalcNumSegments)
408{
409 const DumbProjectileBehaviorModuleData* d = getDumbProjectileBehaviorModuleData();
410
411 Coord3D controlPoints[4];
412 //First point is us, last point is them
413 controlPoints[0] = m_flightPathStart;
414 controlPoints[3] = m_flightPathEnd;
415
416 Real highestInterveningTerrain;
417 Bool onMap = ThePartitionManager->estimateTerrainExtremesAlongLine( controlPoints[0], controlPoints[3], NULL, &highestInterveningTerrain, NULL, NULL );
418 if( !onMap )
419 {
420 return false;
421 }
422
423 // X and Y for inner points are along the line between us, so normalize and scale a vector between us, but
424 // only use the x and y of the result
425 Vector3 targetVector;// 0 origin vector between me and him
426 targetVector.X = controlPoints[3].x - controlPoints[0].x;
427 targetVector.Y = controlPoints[3].y - controlPoints[0].y;
428 targetVector.Z = controlPoints[3].z - controlPoints[0].z;
429
430 Real targetDistance = targetVector.Length();
431 targetVector.Normalize();
432 Vector3 firstPointAlongLine = targetVector * (targetDistance * d->m_firstPercentIndent );
433 Vector3 secondPointAlongLine = targetVector * (targetDistance * d->m_secondPercentIndent );
434
435 controlPoints[1].x = firstPointAlongLine.X + controlPoints[0].x;// add world start to offset along the origin based vector
436 controlPoints[1].y = firstPointAlongLine.Y + controlPoints[0].y;
437 controlPoints[2].x = secondPointAlongLine.X + controlPoints[0].x;
438 controlPoints[2].y = secondPointAlongLine.Y + controlPoints[0].y;
439
440 // Z's are determined using the highest intervening height so they won't hit hills, low end bounded by current Zs
441 highestInterveningTerrain = max( highestInterveningTerrain, controlPoints[0].z );
442 highestInterveningTerrain = max( highestInterveningTerrain, controlPoints[3].z );
443 controlPoints[1].z = highestInterveningTerrain + d->m_firstHeight;
444 controlPoints[2].z = highestInterveningTerrain + d->m_secondHeight;
445
446 // With four control points, we have a curve. We will decide how many frames we want to take to get to the target,
447 // and fill our vector with those curve points.
448 BezierSegment flightCurve( controlPoints );
449 if (recalcNumSegments)
450 {
451 Real flightDistance = flightCurve.getApproximateLength();
452 m_flightPathSegments = ceil( flightDistance / m_flightPathSpeed );
453 }
454 flightCurve.getSegmentPoints( m_flightPathSegments, &m_flightPath );
455 DEBUG_ASSERTCRASH(m_flightPathSegments == m_flightPath.size(), ("m_flightPathSegments mismatch"));
456
457#if defined(_DEBUG) || defined(_INTERNAL)
458 if( TheGlobalData->m_debugProjectilePath )
459 displayFlightPath();
460#endif
461
462 return true;
463}

467Bool DumbProjectileBehavior::projectileHandleCollision( Object *other )
468{
469 const DumbProjectileBehaviorModuleData* d = getDumbProjectileBehaviorModuleData();
470
471 if (other != NULL)
472 {
473 Object *projectileLauncher = TheGameLogic->findObjectByID( projectileGetLauncherID() );
474
475 // if it's not the specific thing we were targeting, see if we should incidentally collide...
476 if (!m_detonationWeaponTmpl->shouldProjectileCollideWith(projectileLauncher, getObject(), other, m_victimID))
477 {
478 //DEBUG_LOG(("ignoring projectile collision with %s at frame %d\n",other->getTemplate()->getName().str(),TheGameLogic->getFrame()));
479 return true;
480 }
481
482 if (d->m_garrisonHitKillCount > 0)
483 {
484 ContainModuleInterface* contain = other->getContain();
485 if( contain && contain->getContainCount() > 0 && contain->isGarrisonable() && !contain->isImmuneToClearBuildingAttacks() )
486 {
487 Int numKilled = 0;
488
489 // garrisonable buildings subvert the normal process here.
490 const ContainedItemsList* items = contain->getContainedItemsList();
491 if (items)
492 {
493 for (ContainedItemsList::const_iterator it = items->begin(); *it != NULL && numKilled < d->m_garrisonHitKillCount; )
494 {
495 Object* thingToKill = *it++;
496 if (!thingToKill->isEffectivelyDead() && thingToKill->isKindOfMulti(d->m_garrisonHitKillKindof, d->m_garrisonHitKillKindofNot))
497 {
498 //DEBUG_LOG(("Killed a garrisoned unit (%08lx %s) via Flash-Bang!\n",thingToKill,thingToKill->getTemplate()->getName().str()));
499 if (projectileLauncher)
500 projectileLauncher->scoreTheKill( thingToKill );
501 thingToKill->kill();
502 ++numKilled;
503 }
504 } // next contained item
505 } // if items
506
507 if (numKilled > 0)
508 {
509 // note, fx is played at center of building, not at grenade's location
510 FXList::doFXObj(d->m_garrisonHitKillFX, other, NULL);
511
512 getObject()->getControllingPlayer()->getAcademyStats()->recordClearedGarrisonedBuilding();
513
514 // don't do the normal explosion; just destroy ourselves & return
515 TheGameLogic->destroyObject(getObject());
516
517 return true;
518 }
519 } // if a garrisonable thing
520 }
521
522 }
523
524 // collided with something... blow'd up!
525 detonate();
526
527 // mark ourself as "no collisions" (since we might still exist in slow death mode)
528 getObject()->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_NO_COLLISIONS ) );
529 return true;
530}

533void DumbProjectileBehavior::detonate()
534{
535 if ( m_hasDetonated )
536 return;
537
538 Object* obj = getObject();
539 if (m_detonationWeaponTmpl)
540 {
541 TheWeaponStore->handleProjectileDetonation(m_detonationWeaponTmpl, obj, obj->getPosition(), m_extraBonusFlags);
542
543 if ( getDumbProjectileBehaviorModuleData()->m_detonateCallsKill )
544 {
545 // don't call kill(); do it manually, so we can specify DEATH_DETONATED
546 DamageInfo damageInfo;
547 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
548 damageInfo.in.m_deathType = DEATH_DETONATED;
549 damageInfo.in.m_sourceID = INVALID_ID;
550 damageInfo.in.m_amount = obj->getBodyModule()->getMaxHealth();
551 obj->attemptDamage( &damageInfo );
552 }
553 else
554 {
555 TheGameLogic->destroyObject( obj );
556 }
557
558 }
559 else
560 {
561 // don't call kill(); do it manually, so we can specify DEATH_DETONATED
562 DamageInfo damageInfo;
563 damageInfo.in.m_damageType = DAMAGE_UNRESISTABLE;
564 damageInfo.in.m_deathType = DEATH_DETONATED;
565 damageInfo.in.m_sourceID = INVALID_ID;
566 damageInfo.in.m_amount = obj->getBodyModule()->getMaxHealth();
567 obj->attemptDamage( &damageInfo );
568 }
569
570 if (obj->getDrawable())
571 obj->getDrawable()->setDrawableHidden(true);
572
573 m_hasDetonated = TRUE;
574
575}

577//-------------------------------------------------------------------------------------------------
581UpdateSleepTime DumbProjectileBehavior::update()
582{
583 const DumbProjectileBehaviorModuleData* d = getDumbProjectileBehaviorModuleData();
584
585 if (m_lifespanFrame != 0 && TheGameLogic->getFrame() >= m_lifespanFrame)
586 {
587 // lifetime demands detonation
588 detonate();
589 return UPDATE_SLEEP_NONE;
590 }
591
592 if( m_currentFlightPathStep >= m_flightPath.size() )
593 {
594 // No more steps to use. Would go out of bounds on vector, so have to do something.
595 // We could allow physics to take over and make us fall, but the point of this whole task
596 // is to guarentee where the shell explodes. This way, it _will_ explode at the target point.
597 detonate();
598 return UPDATE_SLEEP_NONE;
599 }
600
601 if (m_victimID != INVALID_ID && d->m_flightPathAdjustDistPerFrame > 0.0f)
602 {
603 Object* victim = TheGameLogic->findObjectByID(m_victimID);
604 if (victim)
605 {
606 Coord3D newVictimPos;
607 victim->getGeometryInfo().getCenterPosition(*victim->getPosition(), newVictimPos);
608 Coord3D delta;
609 delta.x = newVictimPos.x - m_flightPathEnd.x;
610 delta.y = newVictimPos.y - m_flightPathEnd.y;
611 delta.z = newVictimPos.z - m_flightPathEnd.z;
612 Real distVictimMovedSqr = sqr(delta.x) + sqr(delta.y) + sqr(delta.z);
613 if (distVictimMovedSqr > 0.1f)
614 {
615 Real distVictimMoved = sqrtf(distVictimMovedSqr);
616 if (distVictimMoved > d->m_flightPathAdjustDistPerFrame)
617 distVictimMoved = d->m_flightPathAdjustDistPerFrame;
618 delta.normalize();
619 m_flightPathEnd.x += distVictimMoved * delta.x;
620 m_flightPathEnd.y += distVictimMoved * delta.y;
621 m_flightPathEnd.z += distVictimMoved * delta.z;
622 if (!calcFlightPath(false))
623 {
624 DEBUG_CRASH(("Hmm, recalc of flight path returned false... should this happen?"));
625 detonate();
626 return UPDATE_SLEEP_NONE;
627 }
628 }
629 }
630 }
631
632 //Otherwise, continue to force the flight path
633 Coord3D flightStep = m_flightPath[m_currentFlightPathStep];
634
635 if (d->m_orientToFlightPath && (!d->m_tumbleRandomly) )
636 {
637 if ( m_currentFlightPathStep > 0)
638 {
639 // this seems reasonable; however, if this object has a PhysicsBehavior on it, this calc will be wrong,
640 // since Physics is applying gravity, which we duly ignore, but the prevPos won't be what we expect.
641 // get it from the flight path instead. (srj)
642 //Coord3D prevPos = *getObject()->getPosition();
643
644 Coord3D prevPos = m_flightPath[m_currentFlightPathStep - 1];
645
646 Vector3 curDir(flightStep.x - prevPos.x, flightStep.y - prevPos.y, flightStep.z - prevPos.z);
647 curDir.Normalize(); // buildTransformMatrix wants it this way
648 Matrix3D orientMtx;
649 orientMtx.buildTransformMatrix(Vector3(flightStep.x, flightStep.y, flightStep.z), curDir);
650 getObject()->setTransformMatrix(&orientMtx);
651 }
652 else // oops! how do we orient the projectile on the zeroeth frame? This didn't matter until we started using the
653 //long, blurry projectile graphics which look badly oriented on step 0 of the flight path
654 // so lets orient it the same as if it were on frame 1!
655 {
656 Coord3D prevPos = m_flightPath[0];
657 Coord3D curPos = m_flightPath[1];
658
659 Vector3 curDir(curPos.x - prevPos.x, curPos.y - prevPos.y, curPos.z - prevPos.z);
660 curDir.Normalize(); // buildTransformMatrix wants it this way
661 Matrix3D orientMtx;
662 orientMtx.buildTransformMatrix(Vector3(flightStep.x, flightStep.y, flightStep.z), curDir);
663 getObject()->setTransformMatrix(&orientMtx);
664 }
665
666 }
667 else
668 {
669 getObject()->setPosition(&flightStep);
670 }
671
672 // note that we want to use getHighestLayerForDestination() here, so that anything even slightly
673 // below the bridge translates into GROUND. (getLayerForDestination just does a "closest" check)
674 PathfindLayerEnum oldLayer = getObject()->getLayer();
675 PathfindLayerEnum newLayer = TheTerrainLogic->getHighestLayerForDestination(getObject()->getPosition());
676 getObject()->setLayer(newLayer);
677
678 if (oldLayer != LAYER_GROUND && newLayer == LAYER_GROUND)
679 {
680 // see if we' still in the bridge's xy area
681 Coord3D tmp = *getObject()->getPosition();
682 tmp.z = 9999.0f;
683 PathfindLayerEnum testLayer = TheTerrainLogic->getHighestLayerForDestination(&tmp);
684 if (testLayer == oldLayer)
685 {
686 // ensure we are slightly above the bridge, to account for fudge & sloppy art
687 const Real FUDGE = 2.0f;
688 tmp.z = TheTerrainLogic->getLayerHeight(tmp.x, tmp.y, testLayer) + FUDGE;
689 getObject()->setPosition(&tmp);
690 // blow'd up!
691 detonate();
692 return UPDATE_SLEEP_NONE;
693 }
694 }
695
696 ++m_currentFlightPathStep;
697
698 return UPDATE_SLEEP_NONE;//This no longer flys with physics, so it needs to not sleep
699}
