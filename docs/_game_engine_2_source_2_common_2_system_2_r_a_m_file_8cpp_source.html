165//============================================================================
166// RAMFile::open
167//============================================================================
168
169Bool RAMFile::open( File *file )
170{
171 //USE_PERF_TIMER(RAMFile)
172 if ( file == NULL )
173 {
174 return NULL;
175 }
176
177 Int access = file->getAccess();
178
179 if ( !File::open( file->getName(), access ))
180 {
181 return FALSE;
182 }
183
184 // read whole file in to memory
185 m_size = file->size();
186 m_data = MSGNEW("RAMFILE") char [ m_size ]; // pool[]ify
187
188 if ( m_data == NULL )
189 {
190 return FALSE;
191 }
192
193 m_size = file->read( m_data, m_size );
194
195 if ( m_size < 0 )
196 {
197 delete [] m_data;
198 m_data = NULL;
199 return FALSE;
200 }
201
202 m_pos = 0;
203
204 return TRUE;
205}

259//=================================================================
260// RAMFile::read
261//=================================================================
262// if buffer is null, just advance the current position by 'bytes'
263Int RAMFile::read( void *buffer, Int bytes )
264{
265 if( m_data == NULL )
266 {
267 return -1;
268 }
269
270 Int bytesLeft = m_size - m_pos ;
271
272 if ( bytes > bytesLeft )
273 {
274 bytes = bytesLeft;
275 }
276
277 if (( bytes > 0 ) && ( buffer != NULL ))
278 {
279 memcpy ( buffer, &m_data[m_pos], bytes );
280 }
281
282 m_pos += bytes;
283
284 return bytes;
285}

296//=================================================================
297// RAMFile::seek
298//=================================================================
299
300Int RAMFile::seek( Int pos, seekMode mode)
301{
302 Int newPos;
303
304 switch( mode )
305 {
306 case START:
307 newPos = pos;
308 break;
309 case CURRENT:
310 newPos = m_pos + pos;
311 break;
312 case END:
313 DEBUG_ASSERTCRASH(pos <= 0, ("RAMFile::seek - position should be <= 0 for a seek starting from the end."));
314 newPos = m_size + pos;
315 break;
316 default:
317 // bad seek mode
318 return -1;
319 }
320
321 if ( newPos < 0 )
322 {
323 newPos = 0;
324 }
325 else if ( newPos > m_size )
326 {
327 newPos = m_size;
328 }
329
330 m_pos = newPos;
331
332 return m_pos;
333
334}

406

424//=================================================================
425// RAMFile::nextLine
426//=================================================================
427void RAMFile::nextLine(Char *buf, Int bufSize)
428{
429 Int i = 0;
430 // seek to the next new-line character
431 while ((m_pos < m_size) && (m_data[m_pos] != '\n')) {
432 if ((buf != NULL) && (i < (bufSize-1))) {
433 buf[i] = m_data[m_pos];
434 ++i;
435 }
436 ++m_pos;
437 }
438
439 // we got to the new-line character, now go one past it.
440 if (m_pos < m_size) {
441 if ((buf != NULL) && (i < bufSize)) {
442 buf[i] = m_data[m_pos];
443 ++i;
444 }
445 ++m_pos;
446 }
447 if (buf != NULL) {
448 if (i < bufSize) {
449 buf[i] = 0;
450 } else {
451 buf[bufSize] = 0;
452 }
453 }
454 if (m_pos >= m_size) {
455 m_pos = m_size;
456 }
457}
