81void RulerTool::mouseDown(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
82{
83 if (m != TRACK_L) return;
84
85 if (m_View == NULL) {
86 // Save so that when we are done the view can stop drawing the rulers.
87 m_View = pView;
88 }
89
90 Coord3D cpt;
91 pView->viewToDocCoords(viewPt, &cpt);
92
93 m_downPt3d = cpt;
94 pView->snapPoint(&m_downPt3d);

98void RulerTool::mouseMoved(TTrackingMode m, CPoint viewPt, WbView* pView, CWorldBuilderDoc *pDoc)
99{
100 if (m != TRACK_L) return;
101
102 if (m_View == NULL) {
103 // Save so that when we are done the view can stop drawing the rulers.
104 m_View = pView;
105 }
106
107 Coord3D cpt;
108 pView->viewToDocCoords(viewPt, &cpt, false);
109
110 if (m_rulerType == RULER_CIRCLE) {
111 Coord3D pt;
112 pt.x = cpt.x + m_savedLength;
113 pt.y = cpt.y + m_savedLength;
114 pt.z = cpt.z;
115 RulerOptions::setWidth(m_savedLength);
116 pView->doRulerFeedback(RULER_CIRCLE);
117 pView->rulerFeedbackInfo(cpt, pt, m_savedLength);
118 pView->Invalidate();
119 } else { //m_rulerType == RULER_LINE
120 Coord3D diff;
121 diff.set(&cpt);
122 diff.sub(&m_downPt3d);
123 m_savedLength = diff.length();
124 RulerOptions::setWidth(m_savedLength);
125 pView->doRulerFeedback(RULER_LINE);
126 pView->rulerFeedbackInfo(cpt, m_downPt3d, m_savedLength);
127 pView->Invalidate();
128 }
129
130 pDoc->updateAllViews();
131}

133void RulerTool::setLength(Real length)
134{
135 if (!m_staticThis || (m_staticThis->m_rulerType != RULER_CIRCLE)) {
136 // This should only be called when the ruler type is a circle.
137 // Line rulers are always extended by the mouse.
138 return;
139 }
140
141 m_staticThis->m_savedLength = length;
142 if (m_staticThis->m_View) {
143 m_staticThis->m_View->rulerFeedbackInfo(m_staticThis->m_downPt3d, m_staticThis->m_downPt3d, length);
144 m_staticThis->m_View->Invalidate();
145 }
146
147 CString str;
148 str.Format("Diameter (in feet): %f", length * 2.0f);
149 CMainFrame::GetMainFrame()->SetMessageText(str);
150}

152Bool RulerTool::switchType()
153{
154 if (!m_staticThis) {
155 return (FALSE);
156 }
157
158 if (m_staticThis->m_rulerType == RULER_LINE) {
159 m_staticThis->m_rulerType = RULER_CIRCLE;
160 } else {
161 m_staticThis->m_rulerType = RULER_LINE;
162 }
163 if (m_staticThis->m_View != NULL) {
164 m_staticThis->m_View->doRulerFeedback(m_staticThis->m_rulerType);
165 }
166
167 return (TRUE);
168}
