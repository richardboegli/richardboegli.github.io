134//
135// Collect_Logical_Sounds
136//
138void
139SoundSceneClass::Collect_Logical_Sounds (int listener_count)
140{
141 WWPROFILE ("Collect_Logical_Sounds");
142
143 uint32 timestamp = ::GetTickCount ();
144
145 //
146 // Determine how many listeners to process
147 //
148 int count = listener_count;
149 int max = MAX_LOGICAL_LISTENER_UPDATES_PER_FRAME;
150 if ((count < 0) || (count > max)) {
151 count = max;
152 }
153
154 PriorityMultiListIterator<LogicalListenerClass> priority_queue (&m_LogicalListeners);
155 LogicalListenerClass *listener = NULL;
156
157 //
158 // Loop over as many of the listeners as we want to process this
159 // frame.
160 //
161 for ( int total = 0;
162 total < count && priority_queue.Process_Head (&listener);
163 total ++)
164 {
165 LogicalListenerClass::Set_Oldest_Timestamp (listener->Get_Timestamp ());
166 listener->Set_Timestamp (LogicalListenerClass::Get_New_Timestamp ());
167 listener->On_Frame_Update ();
168
169 //
170 // Collect a list of the sounds this listener can hear.
171 //
172 Vector3 position = listener->Get_Position ();
173 m_LogicalCullingSystem.Reset_Collection ();
174 m_LogicalCullingSystem.Collect_Objects (position);
175
176 //
177 // Now loop through the list of sounds this listener can hear
178 // and notify their callback.
179 //
180 SoundCullObjClass * cull_obj;
181 for ( cull_obj = m_LogicalCullingSystem.Get_First_Collected_Object();
182 cull_obj != NULL;
183 cull_obj = m_LogicalCullingSystem.Get_Next_Collected_Object (cull_obj))
184 {
185 //
186 // Get a pointer to the current 'cull-sound' object.
187 //
188 LogicalSoundClass *sound_obj = (LogicalSoundClass *)cull_obj->Peek_Sound_Obj ();
189
190 //
191 // Test this sound against the scale associated with the current listener to
192 // see if the listener can really "hear" the sound.
193 //
194 const Vector3 &sound_pos = cull_obj->Get_Bounding_Box ().Center;
195 Vector3 listener_pos = listener->Get_Position ();
196 float dropoff_radius = sound_obj->Get_DropOff_Radius ();
197 float scale = listener->Get_Effective_Scale ();
198 float test_radius2 = (dropoff_radius * scale) * (dropoff_radius * scale);
199 if ((listener_pos - sound_pos).Length2 () <= test_radius2) {
200
201 //
202 // Is the sound ready to notify?
203 //
204 if (sound_obj->Allow_Notify (timestamp)) {
205 listener->On_Event (AudioCallbackClass::EVENT_LOGICAL_HEARD, (uint32)listener, (uint32)sound_obj);
206 }
207 }
208 }
209 }
210
211 //
212 // Loop through and remove any single shot sounds that have
213 // been completely processed
214 //
215 MultiListIterator<LogicalSoundClass> single_shot_it (&m_SingleShotLogicalSounds);
216 for (single_shot_it.First (); !single_shot_it.Is_Done (); single_shot_it.Next ()) {
217 LogicalSoundClass *sound_obj = single_shot_it.Peek_Obj ();
218
219 //
220 // Remove this sound if its been completely processed
221 //
222 if (sound_obj->Get_Listener_Timestamp () <= LogicalListenerClass::Get_Oldest_Timestamp ()) {
223 sound_obj->Remove_From_Scene ();
224 single_shot_it.Remove_Current_Object ();
225 single_shot_it.Prev ();
226 }
227 }
228
229 return ;
230}

234//
235// Collect_Audible_Sounds
236//
238void
239SoundSceneClass::Collect_Audible_Sounds
240(
241 Listener3DClass * listener,
242 COLLECTED_SOUNDS &list
243)
244{
245 WWPROFILE ("Collect_Audible_Sounds");
246
247 //
248 // Collect a list of the audible dynamic sounds
249 //
250 Vector3 listener_pos = listener->Get_Position ();
251 m_DynamicCullingSystem.Reset_Collection ();
252 m_DynamicCullingSystem.Collect_Objects (listener_pos);
253
254 //
255 // Collect a list of the audible static sounds
256 //
257 m_StaticCullingSystem.Reset_Collection ();
258 m_StaticCullingSystem.Collect_Objects (listener_pos);
259
260 //
261 // Loop through all the dynamic sounds that are currently audible and make sure
262 // they are 'really' audible. The culling systems just check bounding boxes
263 // but we need to be able to check attenuation spheres.
264 //
265 SoundCullObjClass * cull_obj = NULL;
266 for ( cull_obj = m_DynamicCullingSystem.Get_First_Collected_Object();
267 cull_obj != NULL;
268 cull_obj = m_DynamicCullingSystem.Get_Next_Collected_Object(cull_obj))
269 {
270 // Get a pointer to the current 'cull-sound' object
271 AudibleSoundClass *sound_obj = (AudibleSoundClass *)cull_obj->Peek_Sound_Obj ();
272
273 // Perform a quick sphere-cull check to make sure this
274 // sound should really be audible
275 Vector3 pos = sound_obj->Get_Position ();
276 float radius = sound_obj->Get_DropOff_Radius ();
277 float radius2 = radius * radius;
278 float length2 = (pos - listener_pos).Length2 ();
279 if (length2 <= radius2) {
280
281 AudibleInfoClass *audible_info = W3DNEW AudibleInfoClass (sound_obj, length2);
282 list.Add (audible_info);
283
284 //
285 // Update this sound's runtime priority based on its distance
286 // from the sound emitter.
287 //
288 float length = (pos - listener_pos).Quick_Length ();
289 float priority = (length > 0) ? 1 - (length / radius) : 1.0F;
290 sound_obj->Set_Runtime_Priority (priority);
291 }
292 }
293
294 //
295 // Loop through all the static sounds that are currently audible and make sure
296 // they are 'really' audible. The culling systems just check bounding boxes
297 // but we need to be able to check attenuation spheres.
298 //
299 for ( cull_obj = m_StaticCullingSystem.Get_First_Collected_Object();
300 cull_obj != NULL;
301 cull_obj = m_StaticCullingSystem.Get_Next_Collected_Object(cull_obj))
302 {
303 AudibleSoundClass *sound_obj = (AudibleSoundClass *)cull_obj->Peek_Sound_Obj ();
304
305 // Perform a quick sphere-cull check to make sure this
306 // sound should really be audible
307 Vector3 pos = sound_obj->Get_Position ();
308 float radius = sound_obj->Get_DropOff_Radius ();
309 float radius2 = radius * radius;
310 float length2 = (pos - listener_pos).Length2 ();
311 if (length2 <= radius2) {
312
313 AudibleInfoClass *audible_info = W3DNEW AudibleInfoClass (sound_obj, length2);
314 list.Add (audible_info);
315
316 //
317 // Update this sound's runtime priority based on its distance
318 // from the sound emitter.
319 //
320 float length = (pos - listener_pos).Quick_Length ();
321 float priority = (length > 0) ? 1 - (length / radius) : 1.0F;
322 sound_obj->Set_Runtime_Priority (priority);
323 }
324 }
325
326 return ;
327}

331//
332// On_Frame_Update
333//
334// Note: This method could be made more efficient by using another data structure besides
335// linked lists. However the differece may be negligable due to the low density of 3D sounds
336// that are audible at once.
337//
339void
340SoundSceneClass::On_Frame_Update (unsigned int milliseconds)
341{
342 WWPROFILE ("On_Frame_Update");
343
344 COLLECTED_SOUNDS auxiliary_sounds;
345 COLLECTED_SOUNDS primary_sounds;
346
347 //
348 // First, collect any auxiliary sounds that are audible
349 //
350 if (m_2ndListener != NULL) {
351 m_2ndListener->On_Frame_Update (milliseconds);
352 Collect_Audible_Sounds (m_2ndListener, auxiliary_sounds);
353 }
354
355 //
356 // Update the listener's position/velocity, etc
357 //
358 m_Listener->On_Frame_Update (milliseconds);
359
360 //
361 // Collect the primary sounds that are audible
362 //
363 Collect_Audible_Sounds (m_Listener, primary_sounds);
364
365 //
366 // Loop through the auxiliary sounds and make sure
367 // sounds are only played once, either primary or auxiliary.
368 //
369 MultiListIterator<AudibleInfoClass> aux_iterator (&auxiliary_sounds);
370 MultiListIterator<AudibleInfoClass> pri_iterator (&primary_sounds);
371 AUDIBLE_SOUND_LIST audible_sounds;
372
373 for (aux_iterator.First (); !aux_iterator.Is_Done (); aux_iterator.Next ()) {
374 AudibleInfoClass *aux_info = aux_iterator.Peek_Obj ();
375
376 //
377 // Loop through all the primary sounds and remove any
378 // that are 'overpowered' by the same sound in the
379 // other listener.
380 //
381 bool found = false;
382 for (pri_iterator.First (); !pri_iterator.Is_Done () && !found; pri_iterator.Next ()) {
383 AudibleInfoClass *pri_info = pri_iterator.Peek_Obj ();
384
385 //
386 // Is this sound in both lists?
387 //
388 found = (aux_info->sound_obj == pri_info->sound_obj);
389 if (found) {
390 if (aux_info->distance2 < pri_info->distance2) {
391 delete pri_info;
392 primary_sounds.Remove (pri_info);
393 } else {
394 delete aux_info;
395 auxiliary_sounds.Remove (aux_info);
396 }
397 }
398 }
399 }
400
401 //
402 // Add the primary audible sounds into the master list
403 //
404 for (pri_iterator.First (); !pri_iterator.Is_Done (); pri_iterator.Next ()) {
405 AudibleInfoClass *pri_info = pri_iterator.Peek_Obj ();
406 audible_sounds.Add (pri_info->sound_obj);
407
408 //
409 // Let the sound know what it's listener's position is
410 //
411 pri_info->sound_obj->Set_Listener_Transform (m_Listener->Get_Transform ());
412
413 //
414 // Free the audible info object
415 //
416 pri_iterator.Remove_Current_Object ();
417 pri_iterator.Prev ();
418 delete pri_info;
419 }
420
421 //
422 // Add the auxiliary audible sounds into the master list
423 //
424 for (aux_iterator.First (); !aux_iterator.Is_Done (); aux_iterator.Next ()) {
425 AudibleInfoClass *aux_info = aux_iterator.Peek_Obj ();
426 audible_sounds.Add (aux_info->sound_obj);
427
428 //
429 // Let the sound know what it's listener's position is
430 //
431 aux_info->sound_obj->Set_Listener_Transform (m_2ndListener->Get_Transform ());
432
433 //
434 // Convert the sound to a Pseudo-3D sound that has
435 // a 'tinny' filter applied to it.
436 //
437 /*aux_info.sound_obj->Convert_To_Filtered ();
438 AudibleSoundClass *tinny_sound = aux_info.sound_obj->As_Converted_Format ();
439 if (tinny_sound != NULL) {
440 audible_sounds.Add (tinny_sound);
441 }*/
442
443 //
444 // Free the audible info object
445 //
446 aux_iterator.Remove_Current_Object ();
447 aux_iterator.Prev ();
448 delete aux_info;
449 }
450
451 //
452 // Loop through all the sounds that were audible last frame
453 // and see if they are still audible this frame.
454 //
455 MultiListIterator<AudibleSoundClass> audible_iterator (&m_LastSoundsAudible);
456 for (audible_iterator.First (); !audible_iterator.Is_Done (); audible_iterator.Next ()) {
457 AudibleSoundClass *sound_obj = audible_iterator.Peek_Obj ();
458
459 //
460 // Is this sound still audible?
461 //
462 if (audible_sounds.Is_In_List (sound_obj)) {
463
464 //
465 // Make sure the sound is playing, then remove it from
466 // the newly-audible list so we don't process it again
467 //
468 sound_obj->Cull_Sound (false);
469 audible_sounds.Remove (sound_obj);
470
471 } else {
472
473 //
474 // If the sound isn't audible any more then remove
475 // it from the list
476 //
477 audible_iterator.Remove_Current_Object ();
478 audible_iterator.Prev ();
479
480 //
481 // Make sure we cull the sound
482 //
483 WWASSERT(sound_obj != NULL);
484 sound_obj->Cull_Sound (true);
485 sound_obj->Set_Runtime_Priority (0);
486 }
487 }
488
489 //
490 // Loop through all the newly-audible sounds and
491 // make sure they are playing.
492 //
493 MultiListIterator<AudibleSoundClass> newly_audible_it (&audible_sounds);
494 for (newly_audible_it.First (); !newly_audible_it.Is_Done (); newly_audible_it.Next ()) {
495 AudibleSoundClass *sound_obj = newly_audible_it.Peek_Obj ();
496
497 //
498 // Make sure the sound has a valid Miles handle (so it can make noise)
499 //
500 sound_obj->Cull_Sound (false);
501
502 //
503 // If this sound is still in the scene (it may have 'stopped'
504 // while it was culled) then start playing it...
505 //
506 if (sound_obj->Is_In_Scene ()) {
507 m_LastSoundsAudible.Add (sound_obj);
508 }
509 }
510
511 return ;
512}

516//
517// Add_Sound
518//
520void
521SoundSceneClass::Add_Sound
522(
523 AudibleSoundClass * sound_obj,
524 bool start_playing
525)
526{
527 WWPROFILE ("Add_Sound");
528 WWMEMLOG(MEM_SOUND);
529
530 WWASSERT (sound_obj != NULL);
531 if (sound_obj != NULL && sound_obj->Is_In_Scene () == false) {
532 bool cull_sound = true;
533
534
535 // Create a wrapper object for the sound that we can use
536 // with the different culling systems.
537 SoundCullObjClass *cullable_sound = W3DNEW SoundCullObjClass;
538 cullable_sound->Set_Sound_Obj (sound_obj);
539 sound_obj->Set_Cullable_Wrapper (cullable_sound);
540
541 //
542 // Add this object to the dynamic culling system
543 //
544 m_DynamicCullingSystem.Add_Object (cullable_sound);
545 m_DynamicSounds.Add (cullable_sound);
546 Update_Sound (cullable_sound);
547
548 //
549 // If the listener can hear this sound, then make sure
550 // we start it off non-culled
551 //
552 if (m_IsBatchMode == false) {
553
554 Vector3 listener_pos = m_Listener->Get_Position ();
555 Vector3 sound_pos = sound_obj->Get_Position ();
556 float radius = sound_obj->Get_DropOff_Radius ();
557 float radius2 = radius * radius;
558
559 if (((listener_pos - sound_pos).Length2 ()) < radius2) {
560 cull_sound = false;
561 m_LastSoundsAudible.Add (sound_obj);
562 start_playing = true;
563 sound_obj->Set_Listener_Transform (m_Listener->Get_Transform ());
564 }
565 }
566
567 //
568 // Make sure the sound is appropriately culled
569 //
570 sound_obj->Cull_Sound (cull_sound);
571
572 //
573 // Start the sound playing if requested
574 //
575 if (start_playing) {
576 sound_obj->Play ();
577 }
578 }
579
580 return ;
581}

585//
586// Remove_Sound
587//
588// Note: This method should really be rewritten to use a different list type. Linked lists
589// are probably too inefficient (especially if we have 100s or 1000s of sounds)
590//
592void
593SoundSceneClass::Remove_Sound
594(
595 AudibleSoundClass *sound_obj,
596 bool stop_playing
597)
598{
599 WWPROFILE ("Remove_Sound");
600
601 if (sound_obj == NULL) {
602 return ;
603 }
604
605 //
606 // Make sure we remove this sound from the list of last audible sounds.
607 //
608 if (m_LastSoundsAudible.Is_In_List (sound_obj)) {
609 m_LastSoundsAudible.Remove (sound_obj);
610 }
611
612 //
613 // Is this sound really in the scene?
614 //
615 SoundCullObjClass *cull_obj = sound_obj->Peek_Cullable_Wrapper ();
616 if (cull_obj != NULL && m_DynamicSounds.Is_In_List (cull_obj)) {
617
618 //
619 // Stop playing the sound if necessary
620 //
621 if (stop_playing) {
622 sound_obj->Stop ();
623 }
624
625 //
626 // Flush the sound's cull-wrapper since we are removing it from the scene
627 //
628 sound_obj->Set_Cullable_Wrapper (NULL);
629
630 //
631 // Remove this sound from the dynamic culling system
632 //
633 m_DynamicCullingSystem.Remove_Object (cull_obj);
634 m_DynamicSounds.Remove (cull_obj);
635
636 //
637 // Register the sound for deletion at an appropriate time
638 //
639 WWAudioThreadsClass::Add_Delayed_Release_Object (cull_obj);
640 }
641
642 return ;
643}

787//
788// Add_Logical_Sound
789//
791void
792SoundSceneClass::Add_Logical_Sound
793(
794 LogicalSoundClass * sound_obj,
795 bool single_shot
796)
797{
798 WWPROFILE ("Add_Logical_Sound");
799
800 WWASSERT (sound_obj != NULL);
801 if (sound_obj != NULL) {
802
803 //
804 // Check to make sure we don't add this sound twice
805 //
806 if (Is_Logical_Sound_In_Scene (sound_obj, single_shot) == false) {
807 sound_obj->Set_Listener_Timestamp (LogicalListenerClass::Get_Newest_Timestamp ());
808
809 //
810 // Create a wrapper object for the sound that we can use
811 // with the different culling systems.
812 //
813 SoundCullObjClass *cullable_sound = W3DNEW SoundCullObjClass;
814 cullable_sound->Set_Sound_Obj (sound_obj);
815 sound_obj->Set_Cullable_Wrapper (cullable_sound);
816
817 //
818 // Add this object to the logical culling system
819 //
820 m_LogicalCullingSystem.Add_Object (cullable_sound);
821
822 //
823 // Add this sound to our current sounds list
824 //
825 if (single_shot) {
826 m_SingleShotLogicalSounds.Add (sound_obj);
827 } else {
828 m_LogicalSounds.Add (sound_obj);
829 }
830
831 //
832 // Keep a reference on this sound object
833 //
834 sound_obj->Add_Ref ();
835
836 //
837 // Make sure the cull-object has the most up-to-date information
