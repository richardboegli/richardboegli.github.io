113UpdateSleepTime BaseRegenerateUpdate::update( void )
114{
115 // this is us!
116 Object *me = getObject();
117
118 if( me->testStatus(OBJECT_STATUS_UNDER_CONSTRUCTION) )
119 {
120 return UPDATE_SLEEP_NONE;
121 }
122
123 if( me->testStatus(OBJECT_STATUS_SOLD) )
124 {
125 return UPDATE_SLEEP_FOREVER;
126 }
127
128 // do not heal if we are at max health already
129 BodyModuleInterface *body = me->getBodyModule();
130 if( body->getMaxHealth() == body->getHealth() )
131 {
132 return UPDATE_SLEEP_FOREVER; // sleep till we get damaged again
133 }
134 else
135 {
136 // we don't really need to heal every frame. do it every 3 frames or so.
137 const Int HEAL_RATE = 3;
138
139 // do some healing
140 Real amount = HEAL_RATE * (body->getMaxHealth() * TheGlobalData->m_baseRegenHealthPercentPerSecond) /
141 LOGICFRAMES_PER_SECOND;
142 me->attemptHealing(amount, me);
143
144 return UPDATE_SLEEP(HEAL_RATE);
145 }
146} // end update
