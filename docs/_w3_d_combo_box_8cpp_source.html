79void W3DGadgetComboBoxDraw( GameWindow *window, WinInstanceData *instData )
80{
81 Int width, height, fontHeight, x, y;
82 Color background, border, titleColor, titleBorder;
83// ComboBoxData *combo = (ComboBoxData *)window->winGetUserData();
84 ICoord2D size;
85 DisplayString *title = instData->getTextDisplayString();
86
87 // get window position and size
88 window->winGetScreenPosition( &x, &y );
89 window->winGetSize( &size.x, &size.y );
90
91 // get font height
92 fontHeight = TheWindowManager->winFontHeight( instData->getFont() );
93
94 // alias width and height from size
95 width = size.x;
96 height = size.y;
97
98 // get the right colors
99 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
100 {
101 background = GadgetComboBoxGetDisabledColor( window );
102 border = GadgetComboBoxGetDisabledBorderColor( window );
103 titleColor = window->winGetDisabledTextColor();
104 titleBorder = window->winGetDisabledTextBorderColor();
105 } // end if, disabled
106 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
107 {
108 background = GadgetComboBoxGetHiliteColor( window );
109 border = GadgetComboBoxGetHiliteBorderColor( window );
110 titleColor = window->winGetHiliteTextColor();
111 titleBorder = window->winGetHiliteTextBorderColor();
112 } // end else if, hilited
113 else
114 {
115 background = GadgetComboBoxGetEnabledColor( window );
116 border = GadgetComboBoxGetEnabledBorderColor( window );
117 titleColor = window->winGetEnabledTextColor();
118 titleBorder = window->winGetEnabledTextBorderColor();
119 } // end else, enabled
120
121 // Draw the title
122 if( title && title->getTextLength() )
123 {
124
125 // set the font of this text to that of the window if not already
126 if( title->getFont() != window->winGetFont() )
127 title->setFont( window->winGetFont() );
128
129 // draw the text
130 title->draw( x + 1, y, titleColor, titleBorder );
131
132 y += fontHeight + 1;
133 height -= fontHeight + 1;
134
135 } // end if
136
137 // draw the back border
138 if( border != WIN_COLOR_UNDEFINED )
139 TheWindowManager->winOpenRect( border, WIN_DRAW_LINE_WIDTH,
140 x, y, x + width, y + height );
141
142 // draw background
143 if( background != WIN_COLOR_UNDEFINED )
144 TheWindowManager->winFillRect( background, WIN_DRAW_LINE_WIDTH,
145 x + 1, y + 1,
146 x + width - 1, y + height - 1 );
147
148} // end W3DGadgetComboBoxDraw

153void W3DGadgetComboBoxImageDraw( GameWindow *window, WinInstanceData *instData )
154{
155 Int width, height, x, y;
156 const Image *image;
157// ComboBoxData *combo = (ComboBoxData *)window->winGetUserData();
158 ICoord2D size;
159 Color titleColor, titleBorder;
160 DisplayString *title = instData->getTextDisplayString();
161
162 // get window position and size
163 window->winGetScreenPosition( &x, &y );
164 window->winGetSize( &size.x, &size.y );
165
166 // save off width and height so we can change them
167 width = size.x;
168 height = size.y;
169
170 // get the image
171 if( BitTest( window->winGetStatus(), WIN_STATUS_ENABLED ) == FALSE )
172 {
173 image = GadgetComboBoxGetDisabledImage( window );
174 titleColor = window->winGetDisabledTextColor();
175 titleBorder = window->winGetDisabledTextBorderColor();
176 }
177 else if( BitTest( instData->getState(), WIN_STATE_HILITED ) )
178 {
179 image = GadgetComboBoxGetHiliteImage( window );
180 titleColor = window->winGetHiliteTextColor();
181 titleBorder = window->winGetHiliteTextBorderColor();
182 }
183 else
184 {
185 image = GadgetComboBoxGetEnabledImage( window );
186 titleColor = window->winGetEnabledTextColor();
187 titleBorder = window->winGetEnabledTextBorderColor();
188 }
189
190 // draw the back image
191 if( image )
192 {
193 ICoord2D start, end;
194
195 start.x = x + instData->m_imageOffset.x;
196 start.y = y + instData->m_imageOffset.y;
197 end.x = start.x + width;
198 end.y = start.y + height;
199 TheWindowManager->winDrawImage( image,
200 start.x, start.y,
201 end.x, end.y );
202
203 } // end if
204
205 // Draw the title
206 if( title && title->getTextLength() )
207 {
208
209 // set font to font of the window if not already
210 if( title->getFont() != window->winGetFont() )
211 title->setFont( window->winGetFont() );
212
213 // draw the text
214 title->draw( x + 1, y, titleColor, titleBorder );
215
216 y += TheWindowManager->winFontHeight( instData->getFont() );
217 height -= TheWindowManager->winFontHeight( instData->getFont() ) + 1;
218
219 } // end if
220
221
222} // end W3DGadgetComboBoxImageDraw
