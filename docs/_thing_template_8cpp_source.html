92/*
93 NOTE NOTE NOTE -- s_objectFieldParseTable and s_objectReskinFieldParseTable must be updated in tandem!
94
95 "s_objectReskinFieldParseTable" is intended to be used for parsing the "ObjectReskin" keyword, and
96 should only allow you to specify things that affect the appearance of a template.
97
98 The idea is that some units are functionally the same, but different visually; rather than replicate
99 all the settings, you can specify it as "this one is like that one, but looks different".
100
101 Thus, currently, the only things in the reskin table are:
102
103 -- DrawModules
104 -- DrawModuleData
105 -- Geometry
106
107 So: if you add/remove/modify any settings that deal with visual appearance, you *may* want to
108 add 'em to the reskin table... but do so VERY CAUTIOUSLY and with careful deliberation (and
109 after checking around for other opinions).
110
111*/

377{
378 Bool cleared = false;
379
380 // do NOT clear... we only want to modify this if we return true.

426 Bool disallowed = fullTemplate->isAnyKindOf( ImmuneToGPSScramblerMask );
427
428 static KindOfMaskType CandidateForGPSScramblerMask;
429 CandidateForGPSScramblerMask.set(KINDOF_SCORE);
430 CandidateForGPSScramblerMask.set(KINDOF_VEHICLE);
431 CandidateForGPSScramblerMask.set(KINDOF_INFANTRY);
432 CandidateForGPSScramblerMask.set(KINDOF_PORTABLE_STRUCTURE);
433 Bool candidate = fullTemplate->isAnyKindOf( CandidateForGPSScramblerMask );
434
435 Bool ret = false;
436
437 std::vector<Nugget>::iterator it = m_info.begin();
438 while( it != m_info.end() )
439 {
440 if( (it->interfaceMask & interfaceMask) != 0 && it->copiedFromDefault )
441 {
442 if ( it->inheritable )
443 {
444 if( it->m_moduleTag.compare("ModuleTag_DefaultAutoHealBehavior") == 0 && !fullTemplate->isTrainable() )
445 {
446 // Don't inherit this module if it is entirely useless to us.
447 it = m_info.erase( it );
448 ret = true;
449 }
450 else
451 {
452 ++it;//skip to the next nugget, 'cause we inherit this one
453 }
454 }
455 else if ( it->overrideableByLikeKind)
456 {
457
458 AsciiString oldName = it->first;
459 if ( oldName == newName //we will dump this instance, since the INI author requested a specific one of the same class
460 || disallowed // or, we just do not Add these special overrideables to these kinds of templates, so just dump it
461 || candidate == FALSE )
462 {
463 it = m_info.erase( it );
464 ret = true;
465 }
466 else
467 ++it;//no match, preserve the default instnace of this Module for now
468 }
469 else // just dump this instance of this Module, since one of the same interface mask has been added by caller
470 {
471 it = m_info.erase( it );
472 ret = true;
473 }
474 }
475 else
476 ++it;
477 }
478 return ret;
479}

510void ThingTemplate::parseModuleName(INI* ini, void *instance, void* store, const void* userData)
511{
512 ThingTemplate* self = (ThingTemplate*)instance;
513 ModuleInfo* mi = (ModuleInfo*)store;
514 ModuleType type = (ModuleType)(UnsignedInt)userData;
515 const char* token = ini->getNextToken();
516 AsciiString tokenStr = token;
517
518 // get the tag string (it is now required)
519 AsciiString moduleTagStr;
520 try
521 {
522 moduleTagStr = ini->getNextToken();
523 }
524 catch( ... )
525 {
526
527 DEBUG_CRASH(( "[LINE: %d - FILE: '%s'] Module tag not found for module '%s' on thing template '%s'. Module tags are required and must be unique for all modules within an object definition\n",
528 ini->getLineNum(), ini->getFilename().str(),
529 tokenStr.str(), self->getName().str() ));
530 throw;
531
532 }
533
534 Int interfaceMask;
535
536 // ugh -- special case for "Body".
537 if (type == 999)
538 {
539 type = MODULETYPE_BEHAVIOR;
540 // what interface(s) does this module support?
541 interfaceMask = TheModuleFactory->findModuleInterfaceMask(tokenStr, type);
542 if ((interfaceMask & (MODULEINTERFACE_BODY)) == 0)
543 {
544 DEBUG_CRASH(("Only Body allowed here"));
545 throw INI_INVALID_DATA;
546 }
547 }
548 else
549 {
550 interfaceMask = TheModuleFactory->findModuleInterfaceMask(tokenStr, type);
551 if ((interfaceMask & (MODULEINTERFACE_BODY)) != 0)
552 {
553 DEBUG_CRASH(("No Body allowed here"));
554 throw INI_INVALID_DATA;
555 }
556 }
557
558 // if we're overriding, we can totally skip over this block
559 if (ini->getLoadType() == INI_LOAD_CREATE_OVERRIDES)
560 {
561 if (self->m_moduleParsingMode == MODULEPARSE_ADD_REMOVE_REPLACE)
562 {
563 // do nothing, just fall thru
564 }
565 else
566 {
567 DEBUG_CRASH(("[LINE: %d - FILE: '%s'] You must use AddModule to add modules in override INI files.\n",
568 ini->getLineNum(), ini->getFilename().str(), self->getName().str()));
569 throw INI_INVALID_DATA;
570 }
571 }
572 else
573 {
574
575// if (self->getName().compare("GLAVehicleQuadCannon"))
576// DEBUG_ASSERTCRASH( FALSE, ("WE ARE CLEARING DEFAULT MODULES FROM A QUAD CANNON.") );
577
578 self->m_behaviorModuleInfo.clearCopiedFromDefaultEntries(interfaceMask, tokenStr, self );
579 self->m_drawModuleInfo.clearCopiedFromDefaultEntries(interfaceMask, tokenStr, self );
580 self->m_clientUpdateModuleInfo.clearCopiedFromDefaultEntries(interfaceMask, tokenStr, self );
581 }
582
583 if (self->m_moduleParsingMode == MODULEPARSE_ADD_REMOVE_REPLACE
584 && self->m_moduleBeingReplacedName.isNotEmpty()
585 && self->m_moduleBeingReplacedName != tokenStr)
586 {
587 DEBUG_CRASH(("[LINE: %d - FILE: '%s'] ReplaceModule must replace modules with another module of the same type, but you are attempting to replace a %s with a %s for Object %s.\n",
588 ini->getLineNum(), ini->getFilename().str(), self->m_moduleBeingReplacedName.str(), tokenStr.str(), self->getName().str()));
589 throw INI_INVALID_DATA;
590 }
591
592 if (self->m_moduleParsingMode == MODULEPARSE_ADD_REMOVE_REPLACE
593 && self->m_moduleBeingReplacedTag.isNotEmpty()
594 && self->m_moduleBeingReplacedTag == moduleTagStr)
595 {
596 DEBUG_CRASH(("[LINE: %d - FILE: '%s'] ReplaceModule must specify a new, unique tag for the replaced module, but you are not doing so for %s (%s) for Object %s.\n",
597 ini->getLineNum(), ini->getFilename().str(), moduleTagStr.str(), self->m_moduleBeingReplacedName.str(), self->getName().str()));
598 throw INI_INVALID_DATA;
599 }
600
601 ModuleData* data = TheModuleFactory->newModuleDataFromINI(ini, tokenStr, type, moduleTagStr);
602
603 if (data->isAiModuleData())
604 {
605 Bool replaced = mi->clearAiModuleInfo();
606 if (replaced)
607 {
608 //Kris: Commented this out for SPAM reasons. Do we really need this?
609 //DEBUG_LOG(("replaced an AI for %s!\n",self->getName().str()));
610 }
611 }
612
613 Bool inheritable = (self->m_moduleParsingMode == MODULEPARSE_INHERITABLE);
614 Bool overrideableByLikeKind = (self->m_moduleParsingMode == MODULEPARSE_OVERRIDEABLE_BY_LIKE_KIND);
615 mi->addModuleInfo(self, tokenStr, moduleTagStr, data, interfaceMask, inheritable, overrideableByLikeKind);
616}

1273 {
1274 m_prereqInfo[i].resolveNames();
