100UpdateSleepTime DeployStyleAIUpdate::update( void )
101{
102 // have to call our parent's isIdle, because we override it to never return true
103 // when we have a pending command...
104 Object *self = getObject();
105 Weapon *weapon = self->getCurrentWeapon();
106 UnsignedInt now = TheGameLogic->getFrame();
107 const DeployStyleAIUpdateModuleData *data = getDeployStyleAIUpdateModuleData();
108
109 //Are we attempting to move? If so we can't do it unless we are undeployed.
110 Bool isTryingToMove = isWaitingForPath() || getPath();
111
112 //Are we trying to attack something. If so, we need to be in range before we can do so.
113 Bool isTryingToAttack = getStateMachine()->isInAttackState();
114 Bool isInRange = FALSE;
115
116 //Are we in guard mode? If so, are we idle... idle guarders deploy for fastest response against attackers.
117 Bool isInGuardIdleState = getStateMachine()->isInGuardIdleState();
118
119 AIUpdateInterface *ai = self->getAI();
120
121 if( isTryingToAttack && weapon )
122 {
123 Object *victim = ai->getCurrentVictim();
124 if( victim )
125 {
126 isInRange = weapon->isWithinAttackRange( self, victim );
127 }
128 else
129 {
130 const Coord3D *pos = ai->getCurrentVictimPos();
131 if( pos )
132 {
133 isInRange = weapon->isWithinAttackRange( self, pos );
134 }
135 }
136 }
137
138 if( m_frameToWaitForDeploy != 0 && now >= m_frameToWaitForDeploy)
139 {
140 switch( m_state )
141 {
142 case DEPLOY:
143 setMyState( READY_TO_ATTACK );
144 break;
145 case UNDEPLOY:
146 setMyState( READY_TO_MOVE );
147 break;
148 }
149 }
150
151 if( isInRange || isInGuardIdleState )
152 {
153 switch( m_state )
154 {
155 case READY_TO_MOVE:
156 //We're need to deploy before we can attack.
157 setMyState( DEPLOY );
158 break;
159 case READY_TO_ATTACK:
160 //Let the AI handle attacking.
161 break;
162 case DEPLOY:
163 //We can't start attacking until we are fully deployed.
164 break;
165 case UNDEPLOY:
166 if( m_frameToWaitForDeploy != 0 )
167 {
168 //Reverse the undeploy at it's current frame!
169 setMyState( DEPLOY, TRUE );
170 }
171 break;
172 case ALIGNING_TURRETS:
173 //If turrets are aligning, we are able to attack right now.
174 setMyState( READY_TO_ATTACK );
175 break;
176 }
177 }
178 else if( isTryingToMove )
179 {
180 switch( m_state )
181 {
182 case READY_TO_MOVE:
183 //We're ready... ai will handle moving.
184 break;
185 case READY_TO_ATTACK:
186 {
187 WhichTurretType tur = getWhichTurretForCurWeapon();
188 if( tur != TURRET_INVALID )
189 {
190 if( doTurretsHaveToCenterBeforePacking() )
191 {
192 //Turrets need to center before we can undeploy.
193 setMyState( ALIGNING_TURRETS );
194 break;
195 }
196 }
197 //Start undeploying.
198 setMyState( UNDEPLOY );
199 break;
200 }
201 case DEPLOY:
202 if( m_frameToWaitForDeploy != 0 )
203 {
204 //Reverse the deploy at it's current frame!
205 setMyState( UNDEPLOY, TRUE );
206 }
207 break;
208 case UNDEPLOY:
209 break;
210 case ALIGNING_TURRETS:
211 {
212 WhichTurretType tur = getWhichTurretForCurWeapon();
213 if( tur != TURRET_INVALID )
214 {
215 if( isTurretInNaturalPosition( tur ) )
216 {
217 //Turrets are centers, so now we can undeploy.
218 setMyState( UNDEPLOY );
219 }
220 }
221 break;
222 }
223 }
224 }
225
226 switch( m_state )
227 {
228 case READY_TO_MOVE:
229 if( isTryingToMove )
230 {
231 self->setModelConditionState( MODELCONDITION_MOVING );
232 }
233 break;
234
235 case READY_TO_ATTACK:
236 break;
237
238 case DEPLOY:
239 if( data->m_manualDeployAnimations )
240 {
241 UnsignedInt totalFrames = getPackTime();
242 UnsignedInt framesLeft = m_frameToWaitForDeploy - now;
243 Drawable *draw = self->getDrawable();
244 if( draw )
245 {
246 draw->setAnimationFrame( totalFrames - framesLeft );
247 }
248 }
249 getStateMachine()->setTemporaryState( AI_BUSY, UPDATE_SLEEP_NONE );
250 setLocomotorGoalNone();
251 break;
252 case UNDEPLOY:
253 if( data->m_manualDeployAnimations )
254 {
255 UnsignedInt framesLeft = m_frameToWaitForDeploy - now;
256 Drawable *draw = self->getDrawable();
257 if( draw )
258 {
259 draw->setAnimationFrame( framesLeft );
260 }
261 }
262 getStateMachine()->setTemporaryState( AI_BUSY, UPDATE_SLEEP_NONE );
263 setLocomotorGoalNone();
264 break;
265
266 case ALIGNING_TURRETS:
267 getStateMachine()->setTemporaryState( AI_BUSY, UPDATE_SLEEP_NONE );
268 setLocomotorGoalNone();
269 break;
270 }
271
272 AIUpdateInterface::update();
273 //We can't sleep the deploy AI because any new commands need to be caught and sent
274 //into busy state during the update.
275 return UPDATE_SLEEP_NONE;

280void DeployStyleAIUpdate::setMyState( DeployStateTypes stateID, Bool reverseDeploy )
281{
282 m_state = stateID;
283 Object *self = getObject();
284 UnsignedInt now = TheGameLogic->getFrame();
285 const DeployStyleAIUpdateModuleData *data = getDeployStyleAIUpdateModuleData();
286
287 switch( stateID )
288 {
289 case DEPLOY:
290 {
291 //Tell our object to deploy (so it can continue the same attack later).
292 self->clearAndSetModelConditionFlags( MAKE_MODELCONDITION_MASK( MODELCONDITION_PACKING ),
293 MAKE_MODELCONDITION_MASK( MODELCONDITION_UNPACKING ) );
294
295 if( reverseDeploy )
296 {
297 //This is a zero to max animation.
298 UnsignedInt totalFrames = getUnpackTime();
299 UnsignedInt framesLeft = m_frameToWaitForDeploy - now;
300 m_frameToWaitForDeploy = now + totalFrames - framesLeft;
301 if( data->m_manualDeployAnimations )
302 {
303 Drawable *draw = self->getDrawable();
304 if( draw )
305 {
306 draw->setAnimationFrame( totalFrames - framesLeft );
307 }
308 }
309 }
310 else
311 {
312 m_frameToWaitForDeploy = getUnpackTime() + now;
313 }
314
315 //Play deploy sound
316 const ThingTemplate *thing = self->getTemplate();
317 const AudioEventRTS* soundToPlayPtr = thing->getPerUnitSound( "Deploy" );
318 if( soundToPlayPtr )
319 {
320 AudioEventRTS soundToPlay = *soundToPlayPtr;
321 soundToPlay.setObjectID( self->getID() );
322 TheAudio->addAudioEvent( &soundToPlay );
323 }
324
325 break;
326 }
327 case UNDEPLOY:
328 {
329 //This status will tell the approach AI state to succeed automatically. This prevents
330 //twitching on deploy. Make sure we clear it now so when it goes into that state, it'll
331 //actually move.
332 self->clearStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DEPLOYED ) );
333
334 //Tell our object to pack up (so it can continue the same move later).
335 self->clearAndSetModelConditionFlags( MAKE_MODELCONDITION_MASK2( MODELCONDITION_UNPACKING, MODELCONDITION_DEPLOYED ),
336 MAKE_MODELCONDITION_MASK( MODELCONDITION_PACKING ) );
337
338 if( reverseDeploy )
339 {
340 //This is a max to zero animation.
341 UnsignedInt totalFrames = getUnpackTime();
342 UnsignedInt framesLeft = m_frameToWaitForDeploy - now;
343 m_frameToWaitForDeploy = now + totalFrames - framesLeft;
344 if( data->m_manualDeployAnimations )
345 {
346 Drawable *draw = self->getDrawable();
347 if( draw )
348 {
349 draw->setAnimationFrame( framesLeft );
350 }
351 }
352 }
353 else
354 {
355 m_frameToWaitForDeploy = getPackTime() + now;
356 }
357
358 if( doTurretsFunctionOnlyWhenDeployed() )
359 {
360 WhichTurretType tur = getWhichTurretForCurWeapon();
361 if( tur != TURRET_INVALID )
362 {
363 setTurretEnabled( tur, false );
364 }
365 }
366
367 //Play undeploy sound
368 const ThingTemplate *thing = self->getTemplate();
369 const AudioEventRTS* soundToPlayPtr = thing->getPerUnitSound( "Undeploy" );
370 if( soundToPlayPtr )
371 {
372 AudioEventRTS soundToPlay = *soundToPlayPtr;
373 soundToPlay.setObjectID( self->getID() );
374 TheAudio->addAudioEvent( &soundToPlay );
375 }
376
377 break;
378 }
379 case READY_TO_MOVE:
380 {
381 m_frameToWaitForDeploy = 0;
382
383 self->clearModelConditionFlags( MAKE_MODELCONDITION_MASK( MODELCONDITION_PACKING ) );
384 break;
385 }
386 case READY_TO_ATTACK:
387 {
388 //This status will tell the approach AI state to succeed automatically. This prevents
389 //twitching on deploy.
390 self->setStatus( MAKE_OBJECT_STATUS_MASK( OBJECT_STATUS_DEPLOYED ) );
391
392 m_frameToWaitForDeploy = 0;
393
394 self->clearAndSetModelConditionFlags( MAKE_MODELCONDITION_MASK( MODELCONDITION_UNPACKING ),
395 MAKE_MODELCONDITION_MASK( MODELCONDITION_DEPLOYED) );
396
397 if( doTurretsFunctionOnlyWhenDeployed() )
398 {
399 WhichTurretType tur = getWhichTurretForCurWeapon();
400 if( tur != TURRET_INVALID )
401 {
402 setTurretEnabled( tur, true );
403 }
404 }
405
406 break;
407 }
408 case ALIGNING_TURRETS:
409 {
410 m_frameToWaitForDeploy = 0;
411 WhichTurretType tur = getWhichTurretForCurWeapon();
412 if( tur != TURRET_INVALID )
413 {
414 recenterTurret( tur );
415 }
416 break;
417 }
418 }

439void DeployStyleAIUpdate::xfer( Xfer *xfer )
440{
441 // version
442 XferVersion currentVersion = 4;
443 XferVersion version = currentVersion;
444 xfer->xferVersion( &version, currentVersion );
445
446 // extend base class
447 AIUpdateInterface::xfer(xfer);
448
449 if( version >= 4 )
450 {
451 xfer->xferUser(&m_state, sizeof(m_state));
452 xfer->xferUnsignedInt(&m_frameToWaitForDeploy);
453 }
454 else if( xfer->getXferMode() == XFER_LOAD )
455 {
456 //Good riddance!!!
457 Bool obsoleteBool;
458 UnsignedInt obsoleteUnsignedInt;
459 ObjectID obsoleteObjectID;
460 Coord3D obsoleteCoord3D;
461 AICommandParmsStorage obsoleteAICommandParmsStorage;
462
463 xfer->xferBool( &obsoleteBool );
464 xfer->xferUnsignedInt( &obsoleteUnsignedInt );
465 xfer->xferUser(&m_state, sizeof(m_state));
466
467 if( version >= 2 )
468 {
469 xfer->xferObjectID( &obsoleteObjectID );
470 xfer->xferObjectID( &obsoleteObjectID );
471 xfer->xferCoord3D( &obsoleteCoord3D );
472 xfer->xferBool( &obsoleteBool );
473 xfer->xferBool( &obsoleteBool );
474 xfer->xferBool( &obsoleteBool );
475 }
476 if( version >= 3 )
477 {
478 xfer->xferBool( &obsoleteBool );
479 xfer->xferBool( &obsoleteBool );
480 }
481
482 obsoleteAICommandParmsStorage.doXfer( xfer );
483
484 if( version < 2 )
485 {
486 obsoleteAICommandParmsStorage.doXfer(xfer);
487 }
488
489 //Initialize unit to able to move.
490 m_state = READY_TO_MOVE;
491 }
492
493} // end xfer
