37void dumpBufferToLog(const void *vBuf, Int len, const char *fname, Int line)
38{
39 DEBUG_LOG(("======= dumpBufferToLog() %d bytes =======\n", len));
40 DEBUG_LOG(("Source: %s:%d\n", fname, line));
41 const char *buf = (const char *)vBuf;
42 Int numLines = len / 8;
43 if ((len % 8) != 0)
44 {
45 ++numLines;
46 }
47 for (Int dumpindex = 0; dumpindex < numLines; ++dumpindex)
48 {
49 Int offset = dumpindex*8;
50 DEBUG_LOG(("\t%5.5d\t", offset));
51 Int dumpindex2;
52 Int numBytesThisLine = min(8, len - offset);
53 for (dumpindex2 = 0; dumpindex2 < numBytesThisLine; ++dumpindex2)
54 {
55 Int c = (buf[offset + dumpindex2] & 0xff);
56 DEBUG_LOG(("%02X ", c));
57 }
58 for (; dumpindex2 < 8; ++dumpindex2)
59 {
60 DEBUG_LOG((" "));
61 }
62 DEBUG_LOG((" | "));
63 for (dumpindex2 = 0; dumpindex2 < numBytesThisLine; ++dumpindex2)
64 {
65 char c = buf[offset + dumpindex2];
66 DEBUG_LOG(("%c", (isprint(c)?c:'.')));
67 }
68 DEBUG_LOG(("\n"));
69 }
70 DEBUG_LOG(("End of packet dump\n"));
71}

79UnsignedInt ResolveIP(AsciiString host)
80{
81 struct hostent *hostStruct;
82 struct in_addr *hostNode;
83
84 if (host.getLength() == 0)
85 {
86 DEBUG_LOG(("ResolveIP(): Can't resolve NULL\n"));
87 return 0;
88 }
89
90 // String such as "127.0.0.1"
91 if (isdigit(host.getCharAt(0)))
92 {
93 return ( ntohl(inet_addr(host.str())) );
94 }
95
96 // String such as "localhost"
97 hostStruct = gethostbyname(host.str());
98 if (hostStruct == NULL)
99 {
100 DEBUG_LOG(("ResolveIP(): Can't resolve %s\n", host.str()));
101 return 0;
102 }
103 hostNode = (struct in_addr *) hostStruct->h_addr;
104 return ( ntohl(hostNode->s_addr) );
105}
